CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        INITIALIZE tree AS dictionary with default empty list for each key  
        FOR each child AND parent IN zipped pid AND ppid  
            APPEND child TO tree[parent]  
        END FOR  
        
        SET queue TO list containing kill  
        SET killed_processes TO empty list  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND SET TO current  
            APPEND current TO killed_processes  
            APPEND all children of current FROM tree TO queue  
        END WHILE  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
