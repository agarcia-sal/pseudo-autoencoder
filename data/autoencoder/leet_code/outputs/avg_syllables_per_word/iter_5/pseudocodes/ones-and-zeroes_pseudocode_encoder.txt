CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        CREATE two_dimensional_list dp WITH dimensions (m plus 1) BY (n plus 1) FILLED WITH zeroes  
        
        FOR each string s IN strs  
            SET count_0 TO number of '0' characters in s  
            SET count_1 TO number of '1' characters in s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp[i][j] TO greater of dp[i][j] AND dp[i minus count_0][j minus count_1] plus 1  
        RETURN dp[m][n]  
    END FUNCTION  
END CLASS
