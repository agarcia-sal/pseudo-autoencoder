CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO mapping with default empty list  
        FOR each u, v IN edges  
            APPEND pair of scores at index v and v TO graph at key u  
            APPEND pair of scores at index u and u TO graph at key v  
        FOR each node IN graph  
            SET graph at node TO list of largest three elements from graph at node based on scores  
        SET max_score TO negative one  
        FOR each u, v IN edges  
            FOR each score1, x IN graph at u  
                FOR each score2, y IN graph at v  
                    IF size of set containing x, y, u, v EQUALS four  
                        SET max_score TO greater of max_score and sum of score1, score2, scores at u, and scores at v  
        RETURN max_score  
    END FUNCTION  
END CLASS
