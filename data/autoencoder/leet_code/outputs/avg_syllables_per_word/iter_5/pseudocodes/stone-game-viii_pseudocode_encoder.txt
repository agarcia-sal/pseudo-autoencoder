CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO a list of zeros WITH LENGTH EQUAL TO LENGTH OF stones  
        SET prefix_sum AT POSITION 0 TO stones AT POSITION 0  
        FOR i FROM 1 TO LENGTH OF stones MINUS 1  
            SET prefix_sum AT POSITION i TO prefix_sum AT POSITION i MINUS 1 PLUS stones AT POSITION i  
        SET dp TO prefix_sum AT LAST POSITION  
        FOR i FROM LENGTH OF stones MINUS 2 DOWN TO 1  
            IF dp GREATER THAN prefix_sum AT POSITION i MINUS dp THEN  
                KEEP dp AS IS  
            ELSE  
                SET dp TO prefix_sum AT POSITION i MINUS dp  
        RETURN dp  
    END FUNCTION  
END CLASS
