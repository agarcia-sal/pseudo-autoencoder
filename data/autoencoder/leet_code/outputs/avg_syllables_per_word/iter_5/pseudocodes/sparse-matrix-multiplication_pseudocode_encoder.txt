CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO LENGTH OF mat1  
        SET k TO LENGTH OF FIRST ELEMENT IN mat1  
        SET k TO LENGTH OF mat2  
        SET n TO LENGTH OF FIRST ELEMENT IN mat2  
  
        INITIALIZE result AS MATRIX OF SIZE m BY n WITH ALL ELEMENTS SET TO 0  
  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR p FROM 0 TO k MINUS 1  
                    INCREMENT result AT POSITION [i][j] BY mat1 AT POSITION [i][p] MULTIPLIED BY mat2 AT POSITION [p][j]  
                END FOR  
            END FOR  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
