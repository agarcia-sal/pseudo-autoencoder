CLASS Solution
    FUNCTION primePalindrome(n)
        FUNCTION is_prime(num)
            IF num LESS THAN 2
                RETURN False
            IF num EQUAL TO 2 OR num EQUAL TO 3
                RETURN True
            IF num MODULO 2 EQUAL TO 0 OR num MODULO 3 EQUAL TO 0
                RETURN False
            SET i TO 5
            WHILE i TIMES i LESS THAN OR EQUAL TO num
                IF num MODULO i EQUAL TO 0 OR num MODULO (i PLUS 2) EQUAL TO 0
                    RETURN False
                INCREMENT i BY 6
            RETURN True
        END FUNCTION

        FUNCTION is_palindrome(num)
            RETURN string form of num EQUAL TO reverse string form of num
        END FUNCTION

        IF n LESS THAN OR EQUAL TO 2
            RETURN 2

        IF n GREATER THAN OR EQUAL TO 10 AND n LESS THAN OR EQUAL TO 11
            RETURN 11

        WHILE True
            IF n MODULO 2 EQUAL TO 0
                INCREMENT n BY 1
                CONTINUE

            IF is_palindrome(n) AND is_prime(n)
                RETURN n

            IF length of string form of n MODULO 2 EQUAL TO 0
                SET n TO 10 RAISED TO power LENGTH OF string form of n PLUS 1
            ELSE
                INCREMENT n BY 2
END CLASS
