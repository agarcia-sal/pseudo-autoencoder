CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS "0" OR num2 EQUALS "0"  
            RETURN "0"  
        END IF  
        
        SET result TO list of zeros WITH LENGTH equal to LENGTH OF num1 PLUS LENGTH OF num2  
        
        SET num1 TO reverse of num1  
        SET num2 TO reverse of num2  
        
        FOR each index i FROM 0 TO LENGTH OF num1 MINUS 1  
            FOR each index j FROM 0 TO LENGTH OF num2 MINUS 1  
                SET mul TO numeric value of character at position i in num1 MULTIPLIED BY numeric value of character at position j in num2  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS 1  
                SET sum_ TO mul PLUS value at position p1 in result  
                
                SET value at position p1 in result TO sum_ MODULO 10  
                INCREMENT value at position p2 in result BY sum_ DIVIDED BY 10  
            END FOR  
        END FOR  
        
        SET result_str TO string formed by concatenating reversed result list as characters  
        RETURN result_str WITHOUT leading zeros  
    END FUNCTION  
END CLASS
