CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO LENGTH OF temperatures  
        SET answer TO list of zeroes with size n  
        SET stack TO empty list  
  
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT empty AND temperature at i IS GREATER THAN temperature at index at top of stack  
                SET prev_index TO last element removed from stack  
                SET answer at prev_index TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
  
        RETURN answer  
    END FUNCTION  
END CLASS
