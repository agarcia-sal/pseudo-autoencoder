CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values is empty  
        RETURN None  
    SET root TO new TreeNode with value at values position zero  
    SET i TO 1  
    SET queue TO new empty collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN length of values AND value at values position i is not None  
            SET node.left TO new TreeNode with value at values position i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND value at values position i is not None  
            SET node.right TO new TreeNode with value at values position i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF p.val is not EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node is None  
                RETURN 0  
            INCREMENT current_sum BY node.val  
            SET paths_to_here TO prefix_sums value at key (current_sum MINUS targetSum) OR zero if absent  
            SET prefix_sums value at key current_sum TO prefix_sums value at key current_sum OR zero plus 1  
            SET paths_in_left TO helper(node.left, current_sum, prefix_sums)  
            SET paths_in_right TO helper(node.right, current_sum, prefix_sums)  
            DECREMENT prefix_sums value at key current_sum BY 1  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        
        SET prefix_sums TO dictionary with key zero and value one  
        RETURN helper(root, 0, prefix_sums)  
    END FUNCTION  
END CLASS
