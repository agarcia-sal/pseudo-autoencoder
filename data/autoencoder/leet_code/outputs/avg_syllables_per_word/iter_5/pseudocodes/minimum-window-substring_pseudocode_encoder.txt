CLASS Solution  
    FUNCTION minWindow(s, t)  
        SET t_count TO Counter of characters in t  
        SET required TO number of unique keys in t_count  
        SET left TO 0  
        SET right TO 0  
        SET formed TO 0  
        SET window_counts TO empty Counter  
        SET min_length TO infinite  
        SET min_start TO 0  

        WHILE right LESS THAN length of s  
            INCREMENT count of s[right] in window_counts BY 1  

            IF s[right] is in t_count AND count of s[right] in window_counts EQUALS count in t_count  
                INCREMENT formed BY 1  
            END IF  

            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO s[left]  

                IF right MINUS left PLUS 1 LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS 1  
                    SET min_start TO left  
                END IF  

                DECREMENT count of char in window_counts BY 1  
                IF char in t_count AND count of char in window_counts LESS THAN count in t_count  
                    DECREMENT formed BY 1  
                END IF  

                INCREMENT left BY 1  
            END WHILE  

            INCREMENT right BY 1  
        END WHILE  

        IF min_length EQUALS infinite  
            RETURN empty string  
        ELSE  
            RETURN substring of s FROM min_start TO min_start PLUS min_length  
        END IF  
    END FUNCTION  
END CLASS
