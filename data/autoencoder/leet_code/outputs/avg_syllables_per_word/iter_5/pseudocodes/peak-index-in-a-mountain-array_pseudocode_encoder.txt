CLASS Solution  
    FUNCTION peakIndexInMountainArray(arr)  
        SET left TO 0  
        SET right TO LENGTH OF arr MINUS 1  
        WHILE left LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            IF arr AT mid LESS THAN arr AT mid PLUS 1  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
        RETURN left  
    END FUNCTION  
END CLASS
