CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF LENGTH OF nums EQUALS 1 THEN  
                RETURN absolute value of nums at position 0 MINUS 24 LESS THAN 0.000001  
            END IF  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1 DO  
                FOR j FROM 0 TO LENGTH OF nums MINUS 1 DO  
                    IF i NOT EQUAL TO j THEN  
                        SET new_nums TO list of nums at positions k FOR EACH k FROM 0 TO LENGTH OF nums MINUS 1 WHERE k NOT EQUAL TO i AND k NOT EQUAL TO j  
                        
                        FOR EACH op IN list of operations truediv, mul, add, sub DO  
                            IF (op IS add OR op IS mul) AND j GREATER THAN i THEN  
                                CONTINUE TO NEXT op  
                            END IF  
                            
                            IF op IS NOT truediv OR nums at position j NOT EQUAL TO 0 THEN  
                                APPEND result of op applied to nums at position i AND nums at position j TO new_nums  
                                IF solve(new_nums) IS TRUE THEN  
                                    RETURN TRUE  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN FALSE  
        END FUNCTION  
        
        IMPORT operations truediv, mul, add, sub  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
