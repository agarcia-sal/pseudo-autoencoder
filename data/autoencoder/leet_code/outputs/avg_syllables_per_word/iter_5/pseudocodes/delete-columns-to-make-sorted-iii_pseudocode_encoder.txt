CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO LENGTH OF strs  
        SET n TO LENGTH OF strs AT INDEX 0  
        
        SET dp TO LIST OF LENGTH n WITH ALL VALUES 1  
        
        FOR i FROM 1 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF FOR ALL k FROM 0 TO m MINUS 1, VALUE AT strs[k] AT j IS LESS THAN OR EQUAL TO VALUE AT strs[k] AT i THEN  
                    SET dp AT i TO MAXIMUM OF dp AT i AND dp AT j PLUS 1  
                END IF  
            END FOR  
        END FOR  
        
        RETURN n MINUS MAXIMUM VALUE IN dp  
    END FUNCTION  
END CLASS
