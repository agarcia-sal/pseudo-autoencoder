CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO binary representation of n as string without the leading indicator  
        SET length TO length of bin_rep  
        
        INITIALIZE dp as list of zeros with length of length PLUS one  
        SET dp at index zero TO one  
        SET dp at index one TO two  
        
        FOR i FROM two TO length  
            SET dp at index i TO dp at index i minus one PLUS dp at index i minus two  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR i FROM zero TO length minus one  
            IF character at index i in bin_rep IS '1'  
                INCREMENT result BY dp at index length minus i minus one  
                IF prev_bit IS one  
                    DECREMENT result BY one  
                    EXIT loop  
            END IF  
            
            SET prev_bit TO one IF character at index i in bin_rep IS '1' ELSE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
