CLASS Solution  
    FUNCTION minDays(n)  
        DEFINE FUNCTION dp(remaining) WITH MEMOIZATION  
            IF remaining LESS THAN OR EQUAL TO 1  
                RETURN remaining  
            END IF  
            SET option_one TO remainder of remaining DIVIDED BY 2 PLUS dp(remaining DIVIDED BY 2)  
            SET option_two TO remainder of remaining DIVIDED BY 3 PLUS dp(remaining DIVIDED BY 3)  
            RETURN 1 PLUS minimum of option_one AND option_two  
        END FUNCTION  
        
        RETURN dp(n)  
    END FUNCTION  
END CLASS
