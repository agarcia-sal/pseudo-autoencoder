CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO LENGTH OF s  
        SET f TO 2D list of size n by n with all values TRUE  
        SET k TO 0  
        SET mx TO 1  
        FOR i FROM n MINUS 2 DOWNTO 0  
            FOR j FROM i PLUS 1 TO n MINUS 1  
                SET f[i][j] TO FALSE  
                IF s AT i EQUALS s AT j THEN  
                    SET f[i][j] TO f AT i PLUS 1 AND j MINUS 1  
                    IF f[i][j] IS TRUE AND mx LESS THAN j MINUS i PLUS 1 THEN  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS 1  
        RETURN SUBSTRING OF s FROM k TO k PLUS mx  
    END FUNCTION  
END CLASS
