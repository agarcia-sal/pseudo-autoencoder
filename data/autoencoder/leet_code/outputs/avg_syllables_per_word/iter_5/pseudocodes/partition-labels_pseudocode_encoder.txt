CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO a dictionary mapping each character in s to its last position in s  
        SET start TO 0  
        SET end TO 0  
        SET partition_sizes TO an empty list  
        FOR each index i and character char IN s  
            SET end TO the greater value between end and last_occurrence of char  
            IF i EQUALS end THEN  
                APPEND end MINUS start PLUS 1 TO partition_sizes  
                SET start TO i PLUS 1  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
