CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  
        
        CREATE dp AS a 2D list of size (m plus 1) BY (n plus 1) FILLED WITH zeros  
        
        FOR i FROM 0 TO m  
            SET dp at position i and 0 TO i  
        FOR j FROM 0 TO n  
            SET dp at position 0 and j TO j  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF character at position i minus 1 in word1 EQUALS character at position j minus 1 in word2  
                    SET dp at position i and j TO dp at position i minus 1 and j minus 1  
                ELSE  
                    SET dp at position i and j TO 1 PLUS minimum OF dp at position i minus 1 and j, dp at position i and j minus 1, dp at position i minus 1 and j minus 1  
        
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
