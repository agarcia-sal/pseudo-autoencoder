CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        SORT nums IN ASCENDING ORDER  
        SET n TO LENGTH OF nums  
        INITIALIZE f AS LIST OF 1 REPEATED n TIMES  
        SET k TO 0  
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF nums AT i MODULO nums AT j EQUALS 0 THEN  
                    SET f AT i TO MAXIMUM OF f AT i AND f AT j PLUS 1  
            IF f AT k LESS THAN f AT i THEN  
                SET k TO i  
        SET m TO f AT k  
        SET i TO k  
        INITIALIZE ans AS EMPTY LIST  
        WHILE m GREATER THAN 0  
            IF nums AT k MODULO nums AT i EQUALS 0 AND f AT i EQUALS m THEN  
                APPEND nums AT i TO ans  
                SET k TO i  
                DECREMENT m BY 1  
            DECREMENT i BY 1  
        RETURN ans  
    END FUNCTION  
END CLASS
