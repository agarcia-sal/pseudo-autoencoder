CLASS TrieNode  
    FUNCTION __init__()  
        SET children TO new default dictionary that returns TrieNode  
        SET deleted TO False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION deleteDuplicateFolder(paths)  
        SET ans TO empty list  
        SET root TO new TrieNode  
        SET subtreeToNodes TO new default dictionary with list as default value  

        FOR each path IN sorted paths  
            SET node TO root  
            FOR each s IN path  
                SET node TO node.children[s]  
        
        FUNCTION buildSubtreeToRoots(node)  
            SET subtree TO concatenate '(', concatenation FOR each s IN node.children OF s plus buildSubtreeToRoots(node.children[s]), and ')'  
            IF subtree is not equal to '()' THEN  
                APPEND node TO subtreeToNodes[subtree]  
            RETURN subtree  
        END FUNCTION  

        CALL buildSubtreeToRoots(root)  

        FOR each nodes IN values of subtreeToNodes  
            IF length of nodes is greater than 1 THEN  
                FOR each node IN nodes  
                    SET node.deleted TO True  
        
        FUNCTION constructPath(node, path)  
            FOR each s, child IN node.children  
                IF child.deleted is False THEN  
                    CALL constructPath(child, path concatenated with s)  
            IF path is not empty THEN  
                APPEND path TO ans  
        END FUNCTION  

        CALL constructPath(root, empty list)  
        RETURN ans  
    END FUNCTION  
END CLASS
