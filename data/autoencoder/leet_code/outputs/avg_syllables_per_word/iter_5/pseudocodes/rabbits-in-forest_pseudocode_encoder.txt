CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO count occurrences of each element in answers  
        SET total_rabbits TO 0  
        FOR each answer, count IN answer_counts  
            SET group_size TO answer PLUS 1  
            SET groups TO (count PLUS answer) DIVIDED BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
