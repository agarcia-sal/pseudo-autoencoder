CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        DEFINE dp(i) WITH MEMOIZATION  
            SET max_jumps TO 1  
            FOR x FROM 1 TO d  
                IF i PLUS x IS LESS THAN LENGTH OF arr THEN  
                    IF arr AT POSITION i PLUS x IS GREATER THAN OR EQUAL TO arr AT POSITION i THEN  
                        BREAK LOOP  
                    SET candidate TO 1 PLUS dp(i PLUS x)  
                    IF candidate IS GREATER THAN max_jumps THEN  
                        SET max_jumps TO candidate  
            FOR x FROM 1 TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO 0 THEN  
                    IF arr AT POSITION i MINUS x IS GREATER THAN OR EQUAL TO arr AT POSITION i THEN  
                        BREAK LOOP  
                    SET candidate TO 1 PLUS dp(i MINUS x)  
                    IF candidate IS GREATER THAN max_jumps THEN  
                        SET max_jumps TO candidate  
            RETURN max_jumps  
        END FUNCTION  
          
        SET result TO NEGATIVE INFINITY  
        FOR i FROM 0 TO LENGTH OF arr MINUS 1  
            SET candidate TO dp(i)  
            IF candidate IS GREATER THAN result THEN  
                SET result TO candidate  
        RETURN result  
    END FUNCTION  
END CLASS
