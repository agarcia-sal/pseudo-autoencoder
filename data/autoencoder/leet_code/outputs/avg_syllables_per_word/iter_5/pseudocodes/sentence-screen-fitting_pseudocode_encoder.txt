CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO concatenate sentence words separated by space plus a trailing space  
        SET total_length TO length of s  
        SET position TO 0  
        
        FOR each row IN rows  
            INCREMENT position BY cols  
            
            IF character at (position MODULO total_length) in s EQUALS space  
                INCREMENT position BY 1  
            ELSE  
                WHILE position GREATER THAN 0 AND character at ((position MINUS 1) MODULO total_length) in s NOT EQUAL space  
                    DECREMENT position BY 1  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
