CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        IF element at index left in nums IS LESS THAN element at index right in nums  
            RETURN element at index left in nums  
        
        WHILE left IS LESS THAN right  
            SET mid TO integer division of (left PLUS right) BY 2  
            
            IF element at index mid in nums IS GREATER THAN element at index right in nums  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
        
        RETURN element at index left in nums  
    END FUNCTION  
END CLASS
