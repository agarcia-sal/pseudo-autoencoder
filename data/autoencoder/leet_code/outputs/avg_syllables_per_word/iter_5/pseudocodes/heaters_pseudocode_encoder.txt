CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT houses  
        SORT heaters  
        
        SET min_radius TO 0  
        SET j TO 0  
        
        FOR each house IN houses  
            WHILE j LESS THAN LENGTH OF heaters MINUS 1 AND absolute value OF heaters AT j MINUS house GREATER THAN OR EQUAL TO absolute value OF heaters AT j PLUS 1 MINUS house  
                INCREMENT j BY 1  
            SET distance TO absolute value OF heaters AT j MINUS house  
            SET min_radius TO the greater value BETWEEN min_radius AND distance  
        RETURN min_radius  
    END FUNCTION  
END CLASS
