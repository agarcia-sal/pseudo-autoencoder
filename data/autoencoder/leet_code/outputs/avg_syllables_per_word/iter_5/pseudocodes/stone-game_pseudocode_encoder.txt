CLASS Solution  
    FUNCTION stoneGame(piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN 0  
            END IF  
            SET left_choice TO piles AT i MINUS dfs(i INCREMENTED BY 1, j)  
            SET right_choice TO piles AT j MINUS dfs(i, j DECREMENTED BY 1)  
            RETURN the maximum of left_choice and right_choice  
        END FUNCTION  
        RETURN dfs(0, LENGTH OF piles DECREMENTED BY 1) GREATER THAN 0  
    END FUNCTION  
END CLASS
