CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO dictionary mapping each integer from 0 TO n MINUS 1 TO empty list  
        FOR each u, v, w IN flights  
            APPEND tuple of (v, w) TO graph[u]  
        
        SET min_heap TO list containing tuple (0, MINUS 1, src)  
        SET min_cost TO dictionary mapping src TO MINUS 1  
        
        WHILE min_heap is not empty  
            EXTRACT tuple (cost, stops, city) WITH minimum cost FROM min_heap  
            
            IF city EQUALS dst  
                RETURN cost  
            
            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE to next iteration  
            
            FOR each neighbor, price IN graph[city]  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS 1  
                
                IF neighbor NOT IN min_cost OR next_stops LESS THAN min_cost[neighbor] OR next_cost LESS THAN minimum cost value in min_heap  
                    SET min_cost[neighbor] TO next_stops  
                    INSERT tuple (next_cost, next_stops, neighbor) INTO min_heap  
        
        RETURN MINUS 1  
    END FUNCTION  
END CLASS
