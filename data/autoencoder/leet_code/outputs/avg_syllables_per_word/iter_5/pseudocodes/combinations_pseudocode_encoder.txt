CLASS Solution  
    FUNCTION combine(n, k)  
        FUNCTION backtrack(start, path)  
            IF LENGTH OF path IS EQUAL TO k  
                APPEND COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n INCLUSIVE  
                APPEND i TO path  
                CALL backtrack WITH i INCREMENTED BY 1 AND path  
                REMOVE LAST ELEMENT FROM path  
            END FOR  
        END FUNCTION  
        
        SET result TO EMPTY LIST  
        CALL backtrack WITH 1 AND EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
