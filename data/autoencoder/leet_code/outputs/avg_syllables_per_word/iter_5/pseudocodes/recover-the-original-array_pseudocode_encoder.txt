CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums  
        SET n TO LENGTH OF nums  
        SET first TO nums AT INDEX 0  
        
        FOR i FROM 1 TO n MINUS 1  
            SET k TO (nums AT INDEX i MINUS first) DIVIDED BY 2  
            IF k LESS THAN OR EQUAL TO 0 OR first PLUS 2 TIMES k IS NOT EQUAL TO nums AT INDEX i  
                CONTINUE TO NEXT i  
            END IF  
            
            SET count TO COUNTER OF nums  
            SET arr TO EMPTY LIST  
            
            FOR each x IN nums  
                IF count[x] EQUALS 0  
                    CONTINUE TO NEXT x  
                END IF  
                IF count[x PLUS 2 TIMES k] EQUALS 0  
                    BREAK OUT OF LOOP  
                END IF  
                
                APPEND x PLUS k TO arr  
                DECREMENT count[x] BY 1  
                DECREMENT count[x PLUS 2 TIMES k] BY 1  
            END FOR  
            
            IF LENGTH OF arr EQUALS n DIVIDED BY 2  
                RETURN arr  
            END IF  
        END FOR  
        
        RETURN EMPTY LIST  
    END FUNCTION  
END CLASS
