CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO LENGTH OF s  
            SET dp TO 2D ARRAY OF SIZE n BY n FILLED WITH 0  
            
            FOR i FROM 0 TO n MINUS 1  
                SET dp[i][i] TO 1  
            
            FOR cl FROM 2 TO n  
                FOR i FROM 0 TO n MINUS cl  
                    SET j TO i PLUS cl MINUS 1  
                    IF CHARACTER AT s[i] EQUALS CHARACTER AT s[j] AND cl EQUALS 2 THEN  
                        SET dp[i][j] TO 2  
                    ELSE IF CHARACTER AT s[i] EQUALS CHARACTER AT s[j] THEN  
                        SET dp[i][j] TO dp[i PLUS 1][j MINUS 1] PLUS 2  
                    ELSE  
                        SET dp[i][j] TO MAXIMUM OF dp[i][j MINUS 1] AND dp[i PLUS 1][j]  
            
            RETURN dp[0][n MINUS 1]  
        END FUNCTION  
        
        RETURN LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
