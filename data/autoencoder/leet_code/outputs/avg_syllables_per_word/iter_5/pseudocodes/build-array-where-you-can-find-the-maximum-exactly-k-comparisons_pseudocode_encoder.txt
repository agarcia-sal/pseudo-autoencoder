CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        
        INITIALIZE dp AS three-dimensional list with dimensions (n + 1) BY (m + 1) BY (k + 1), FILLED WITH 0  
        
        FOR j FROM 1 TO m  
            SET dp AT [1][j][1] TO 1  
        
        FOR i FROM 2 TO n  
            FOR j FROM 1 TO m  
                FOR p FROM 1 TO k  
                    FOR x FROM 1 TO j MINUS 1  
                        SET dp AT [i][j][p] TO dp AT [i][j][p] PLUS dp AT [i MINUS 1][x][p MINUS 1] MODULO MOD  
                    
                    SET dp AT [i][j][p] TO dp AT [i][j][p] PLUS dp AT [i MINUS 1][j][p] MULTIPLIED BY j MODULO MOD  
        
        SET result TO 0  
        FOR j FROM 1 TO m  
            SET result TO result PLUS dp AT [n][j][k] MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
