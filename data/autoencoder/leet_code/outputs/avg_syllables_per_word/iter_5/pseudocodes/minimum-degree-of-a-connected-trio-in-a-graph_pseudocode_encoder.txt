CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO a mapping from each integer from 1 to n TO an empty set  
        
        FOR each pair u, v IN edges  
            ADD v TO graph at key u  
            ADD u TO graph at key v  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR i FROM 1 TO n  
            FOR j FROM i PLUS 1 TO n  
                FOR k FROM j PLUS 1 TO n  
                    IF j IS IN graph at key i AND k IS IN graph at key i AND k IS IN graph at key j THEN  
                        SET trio_found TO true  
                        SET trio_degree TO size of graph at i PLUS size of graph at j PLUS size of graph at k MINUS 6  
                        SET min_degree TO the smaller value BETWEEN min_degree AND trio_degree  
        
        IF trio_found IS true THEN  
            RETURN min_degree  
        ELSE  
            RETURN -1  
    END FUNCTION  
END CLASS
