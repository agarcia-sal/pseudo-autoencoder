CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF LENGTH OF nums EQUALS 1  
            RETURN STRING REPRESENTATION OF nums AT INDEX 0  
        ELSE IF LENGTH OF nums EQUALS 2  
            RETURN CONCATENATION OF STRING REPRESENTATION OF nums AT INDEX 0, "DIVIDED BY", STRING REPRESENTATION OF nums AT INDEX 1  
        ELSE  
            RETURN CONCATENATION OF STRING REPRESENTATION OF nums AT INDEX 0, "DIVIDED BY", "OPEN PARENTHESIS", JOIN STRING REPRESENTATIONS OF ELEMENTS FROM INDEX 1 TO END OF nums WITH "DIVIDED BY" SEPARATOR, "CLOSE PARENTHESIS"  
    END FUNCTION  
END CLASS
