CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO 2 raised to the power 31 MINUS 1  
        SET INT_MIN TO NEGATIVE 2 raised to the power 31  
        
        SET result TO 0  
        SET index TO 0  
        SET n TO LENGTH OF s  
        SET sign TO 1  
        
        WHILE index LESS THAN n AND CHARACTER AT s AT index EQUALS whitespace  
            INCREMENT index BY 1  
        
        IF index LESS THAN n AND (CHARACTER AT s AT index EQUALS plus SIGN OR CHARACTER AT s AT index EQUALS minus SIGN)  
            SET sign TO NEGATIVE 1 IF CHARACTER AT s AT index EQUALS minus SIGN OTHERWISE 1  
            INCREMENT index BY 1  
        
        WHILE index LESS THAN n AND CHARACTER AT s AT index IS digit  
            SET digit TO numeric VALUE OF CHARACTER AT s AT index  
            IF result GREATER THAN ((INT_MAX MINUS digit) DIVIDED BY 10)  
                IF sign EQUALS 1  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
            SET result TO result MULTIPLIED BY 10 PLUS digit  
            INCREMENT index BY 1  
        
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
