CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MOD TO 10 raised to the power 9 plus 7  
        SET max_index TO minimum of steps DIVIDED BY 2 and arrLen minus 1  
        SET dp TO list of zeroes with length max_index plus 1  
        SET dp at position 0 TO 1  
        
        FOR each step FROM 1 TO steps  
            SET prev_dp TO a copy of dp  
            FOR each i FROM 0 TO max_index  
                SET dp at position i TO prev_dp at position i  
                IF i GREATER THAN 0 THEN  
                    SET dp at position i TO dp at position i PLUS prev_dp at position i MINUS 1 MODULO MOD  
                IF i LESS THAN max_index THEN  
                    SET dp at position i TO dp at position i PLUS prev_dp at position i PLUS 1 MODULO MOD  
        
        RETURN dp at position 0  
    END FUNCTION  
END CLASS
