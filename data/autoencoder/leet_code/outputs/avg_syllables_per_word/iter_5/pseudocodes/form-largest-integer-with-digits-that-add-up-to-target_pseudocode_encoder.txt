CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO empty mapping  
        FOR each index i AND value c IN cost  
            SET cost_to_digit[c] TO string representation of i incremented BY 1  
        SET dp TO list with first element empty string and target elements set TO string "-1"  
        FOR t FROM 1 TO target INCLUSIVE  
            FOR each cost c AND digit d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND dp AT position t MINUS c IS NOT "-1"  
                    SET candidate TO concatenation of d AND dp AT position t MINUS c  
                    IF dp AT position t IS "-1" OR length OF candidate GREATER THAN length OF dp AT position t  
                        OR length OF candidate EQUALS length OF dp AT position t AND candidate IS GREATER THAN dp AT position t  
                        SET dp AT position t TO candidate  
        IF dp AT position target IS NOT "-1"  
            RETURN dp AT position target  
        ELSE  
            RETURN "0"  
    END FUNCTION  
END CLASS
