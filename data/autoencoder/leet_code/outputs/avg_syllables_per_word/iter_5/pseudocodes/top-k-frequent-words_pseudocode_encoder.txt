CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO frequency count of each word in words  
        SET heap TO list of pairs of negative frequency and word for each word and its frequency in count  
        TRANSFORM heap into a heap structure  
        SET result TO list obtained by extracting the word component from heap by removing the top element k times  
        RETURN result  
    END FUNCTION  
END CLASS
