CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted list of cuts appended with 0 and n  
        
        FUNCTION dp(left, right) WITH memoization  
            IF right minus left LESS THAN OR EQUAL TO 1 THEN  
                RETURN 0  
            END IF  
            
            SET min_cost TO infinity  
            FOR i FROM left plus 1 TO right minus 1 DO  
                SET cost TO cuts at right MINUS cuts at left PLUS dp(left, i) PLUS dp(i, right)  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(0, LENGTH OF cuts MINUS 1)  
    END FUNCTION  
END CLASS
