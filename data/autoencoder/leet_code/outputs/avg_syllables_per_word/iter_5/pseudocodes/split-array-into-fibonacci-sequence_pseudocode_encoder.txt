CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS LENGTH OF num AND LENGTH OF path GREATER THAN OR EQUAL TO 3  
                RETURN path  
            END IF  
            FOR end FROM start PLUS 1 TO LENGTH OF num PLUS 1  
                IF CHARACTER AT num INDEX start EQUALS '0' AND end GREATER THAN start PLUS 1  
                    CONTINUE TO NEXT ITERATION  
                END IF  
                SET next_num TO INTEGER REPRESENTATION OF SUBSTRING OF num FROM start TO end  
                IF next_num GREATER THAN OR EQUAL TO 2 TO THE POWER OF 31  
                    BREAK LOOP  
                END IF  
                IF LENGTH OF path LESS THAN 2 OR next_num EQUALS SUM OF LAST TWO ELEMENTS IN path  
                    SET result TO backtrack(end, path WITH next_num APPENDED)  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN EMPTY LIST  
        END FUNCTION  
        RETURN backtrack(0, EMPTY LIST)  
    END FUNCTION  
END CLASS
