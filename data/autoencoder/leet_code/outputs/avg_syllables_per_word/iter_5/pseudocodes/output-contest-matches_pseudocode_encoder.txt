CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings from "1" to string of n  
        WHILE n GREATER THAN 1  
            FOR i FROM 0 TO n DIVIDED BY 2 MINUS 1  
                SET teams at position i TO concatenation of "(" plus teams at position i plus "," plus teams at position (n MINUS 1 MINUS i) plus ")"  
            SET n TO n DIVIDED BY 2  
        RETURN teams at position 0  
    END FUNCTION  
END CLASS
