CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO LENGTH OF floor  
        SET dp TO two-dimensional list of size (n PLUS 1) BY (numCarpets PLUS 1) FILLED WITH 0  
        
        FOR i FROM 1 TO n INCLUSIVE  
            SET dp AT i AND 0 TO dp AT (i MINUS 1) AND 0 PLUS 1 IF floor AT (i MINUS 1) IS '1' OTHERWISE 0  
        
        FOR j FROM 1 TO numCarpets INCLUSIVE  
            FOR i FROM 1 TO n INCLUSIVE  
                SET dp AT i AND j TO dp AT (i MINUS 1) AND j PLUS 1 IF floor AT (i MINUS 1) IS '1' OTHERWISE 0  
                SET start_index TO i MINUS carpetLen  
                IF start_index IS LESS THAN 0  
                    SET start_index TO 0  
                SET dp AT i AND j TO MINIMUM OF dp AT i AND j AND dp AT start_index AND (j MINUS 1)  
        
        RETURN dp AT n AND numCarpets  
    END FUNCTION  
END CLASS
