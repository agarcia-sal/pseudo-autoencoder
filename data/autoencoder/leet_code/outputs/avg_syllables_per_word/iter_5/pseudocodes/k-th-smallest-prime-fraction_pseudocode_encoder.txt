CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO LENGTH OF arr  
        SET min_heap TO empty list  
        
        FOR i FROM 0 TO n MINUS 2  
            INSERT fraction with numerator arr[i] and denominator arr[n MINUS 1] and indices i, n MINUS 1 INTO min_heap  
        END FOR  
        
        FOR counter FROM 1 TO k MINUS 1  
            REMOVE smallest fraction _, i, j FROM min_heap  
            IF j MINUS 1 GREATER THAN i THEN  
                INSERT fraction with numerator arr[i] and denominator arr[j MINUS 1] and indices i, j MINUS 1 INTO min_heap  
            END IF  
        END FOR  
        
        REMOVE smallest fraction _, i, j FROM min_heap  
        RETURN list containing arr[i] AND arr[j]  
    END FUNCTION  
END CLASS
