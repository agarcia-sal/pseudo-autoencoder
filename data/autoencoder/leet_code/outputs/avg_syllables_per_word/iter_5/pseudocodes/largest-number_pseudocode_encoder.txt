CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of each element in nums  
  
        FUNCTION compare(x, y)  
            IF concatenation of x and y IS GREATER THAN concatenation of y and x  
                RETURN negative one  
            ELSE IF concatenation of x and y IS LESS THAN concatenation of y and x  
                RETURN one  
            ELSE  
                RETURN zero  
        END FUNCTION  
  
        SORT nums_str USING compare function as custom comparator  
  
        IF first element of nums_str IS EQUAL TO string "0"  
            RETURN string "0"  
  
        RETURN concatenation of all elements in nums_str  
    END FUNCTION  
END CLASS
