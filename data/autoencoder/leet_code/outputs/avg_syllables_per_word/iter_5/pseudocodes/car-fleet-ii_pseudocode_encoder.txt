CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET n TO LENGTH OF cars  
        SET result TO list of -1.0 repeated n times  
        SET stack TO empty list  

        FOR i FROM n MINUS 1 DOWNTO 0  
            SET pos, speed TO cars AT index i  
            
            WHILE stack IS NOT empty  
                SET next_index TO last element in stack  
                SET next_pos, next_speed TO cars AT next_index  
                
                IF speed LESS THAN OR EQUAL TO next_speed THEN  
                    REMOVE last element from stack  
                    CONTINUE  
                END IF  
                
                SET collision_time TO (next_pos MINUS pos) DIVIDED BY (speed MINUS next_speed)  
                
                IF result AT next_index GREATER THAN 0 AND collision_time GREATER THAN OR EQUAL TO result AT next_index THEN  
                    REMOVE last element from stack  
                    CONTINUE  
                END IF  
                
                SET result AT i TO collision_time  
                BREAK  
            END WHILE  
            
            APPEND i TO stack  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
