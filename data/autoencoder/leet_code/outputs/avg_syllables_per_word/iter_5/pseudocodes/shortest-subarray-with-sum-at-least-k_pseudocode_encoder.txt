CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing 0  
        FOR each num IN nums  
            APPEND last element in prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        INITIALIZE dq AS empty deque  
        SET min_length TO infinity  
        
        FOR each index i AND value current_sum IN prefix_sum  
            WHILE dq is not empty AND current_sum MINUS prefix_sum at front index of dq IS GREATER THAN OR EQUAL TO k  
                SET min_length TO minimum of min_length AND i MINUS front index popped from dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO prefix_sum at last index of dq  
                REMOVE last index from dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS NOT infinity  
            RETURN min_length  
        ELSE  
            RETURN -1  
        END IF  
    END FUNCTION  
END CLASS
