CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF LENGTH OF t EQUALS LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR EACH character c AND count v IN cnt  
                IF v GREATER THAN 1  
                    DECREMENT cnt[c] BY 2  
                    CALL dfs WITH ARGUMENT c CONCATENATED WITH t CONCATENATED WITH c  
                    INCREMENT cnt[c] BY 2  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO COUNTER OF CHARACTERS IN s  
        SET mid TO EMPTY STRING  
        FOR EACH character c AND count v IN cnt  
            IF v MODULO 2 EQUALS 1  
                IF mid IS NOT EMPTY  
                    RETURN EMPTY LIST  
                END IF  
                SET mid TO c  
                DECREMENT cnt[c] BY 1  
            END IF  
        END FOR  
        SET ans TO EMPTY LIST  
        CALL dfs WITH ARGUMENT mid  
        RETURN ans  
    END FUNCTION  
END CLASS
