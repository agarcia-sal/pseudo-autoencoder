CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO infinity  
        
        FOR each num IN arr  
            SET new_seen TO empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num and x TO new_seen  
            SET seen TO union of new_seen and set containing num  
            
            FOR each val IN seen  
                SET current_diff TO absolute value of val MINUS target  
                IF current_diff LESS THAN min_diff  
                    SET min_diff TO current_diff  
        RETURN min_diff  
    END FUNCTION  
END CLASS
