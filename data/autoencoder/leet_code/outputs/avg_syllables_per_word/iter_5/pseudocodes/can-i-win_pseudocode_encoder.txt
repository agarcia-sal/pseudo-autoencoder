CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS 1) DIVIDED BY 2 IS LESS THAN desiredTotal  
            RETURN False  
        END IF  

        IF desiredTotal IS LESS THAN OR EQUAL TO 0  
            RETURN True  
        END IF  

        SET memo TO empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF current_total AND used_numbers EXIST IN memo  
                RETURN memo value FOR current_total AND used_numbers  
            END IF  

            FOR number FROM 1 TO maxChoosableInteger INCLUSIVE  
                IF number IS NOT USED IN used_numbers  
                    IF current_total PLUS number IS GREATER THAN OR EQUAL TO desiredTotal OR can_win WITH current_total PLUS number AND used_numbers INCLUDING number RETURNS False  
                        SET memo value FOR current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET memo value FOR current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_win WITH 0 AND 0  
    END FUNCTION  
END CLASS
