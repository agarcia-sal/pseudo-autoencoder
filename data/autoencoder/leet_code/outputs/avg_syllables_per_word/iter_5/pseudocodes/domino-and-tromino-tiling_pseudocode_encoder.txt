CLASS Solution  
    FUNCTION numTilings(n)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS LIST OF ZEROS WITH LENGTH n PLUS 1  
        INITIALIZE dp1 AS LIST OF ZEROS WITH LENGTH n PLUS 1  

        SET dp AT INDEX 0 TO 1  
        SET dp AT INDEX 1 TO 1  
        IF n IS GREATER THAN OR EQUAL TO 2  
            SET dp AT INDEX 2 TO 2  
            SET dp1 AT INDEX 2 TO 1  
        END IF  

        FOR i FROM 3 TO n INCLUSIVE  
            SET dp AT INDEX i TO (dp AT INDEX i MINUS 1 PLUS dp AT INDEX i MINUS 2 PLUS 2 TIMES dp1 AT INDEX i MINUS 1) MODULO MOD  
            SET dp1 AT INDEX i TO (dp AT INDEX i MINUS 2 PLUS dp1 AT INDEX i MINUS 1) MODULO MOD  
        END FOR  

        RETURN dp AT INDEX n  
    END FUNCTION  
END CLASS
