CLASS Solution  
    FUNCTION judgeSquareSum(c)  
        SET left TO 0  
        SET right TO integer square root of c  
        WHILE left LESS THAN OR EQUAL TO right  
            SET current_sum TO left multiplied BY left PLUS right multiplied BY right  
            IF current_sum EQUALS c  
                RETURN True  
            ELSE IF current_sum LESS THAN c  
                INCREMENT left BY 1  
            ELSE  
                DECREMENT right BY 1  
        RETURN False  
    END FUNCTION  
END CLASS
