CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at index 0 of row EQUALS 0  
                FOR index FROM 0 TO LENGTH OF row MINUS 1  
                    SET element at index of row TO 1 MINUS element at index of row  
        FOR col FROM 1 TO LENGTH OF first row OF grid MINUS 1  
            SET count_ones TO 0  
            FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                INCREMENT count_ones BY element at col of grid at row_index  
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY 2  
                FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                    SET element at col of grid at row_index TO 1 MINUS element at col of grid at row_index  
        SET score TO 0  
        FOR each row IN grid  
            SET binary_string TO concatenation of string representations of each element in row  
            INCREMENT score BY integer value of binary_string interpreted as base 2  
        RETURN score  
    END FUNCTION  
END CLASS
