CLASS Solution
    FUNCTION findPermutation(s)
        SET result TO list of integers from 1 to LENGTH OF s PLUS 1
        SET start TO 0
        FOR end FROM 0 TO LENGTH OF s
            IF end EQUALS LENGTH OF s OR CHARACTER AT s[end] EQUALS 'I' THEN
                REVERSE SUBLIST OF result FROM start TO end INCLUSIVE
                SET start TO end PLUS 1
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
