CLASS Solution  
    FUNCTION getModifiedArray(length, updates)  
        SET diff TO list of zeros of size length plus one  
        FOR each startIdx, endIdx, inc IN updates  
            INCREMENT diff at startIdx BY inc  
            IF endIdx plus one LESS THAN length  
                DECREMENT diff at endIdx plus one BY inc  
        SET arr TO list of zeros of size length  
        SET arr at zero TO diff at zero  
        FOR i FROM one TO length MINUS one  
            SET arr at i TO arr at i minus one PLUS diff at i  
        RETURN arr  
    END FUNCTION  
END CLASS
