CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values at index 0  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO NEW TreeNode WITH values at index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO NEW TreeNode WITH values at index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS None  
            RETURN None  
        IF root.val IS LESS THAN low  
            RETURN self.trimBST(root.right, low, high)  
        IF root.val IS GREATER THAN high  
            RETURN self.trimBST(root.left, low, high)  
        SET root.left TO self.trimBST(root.left, low, high)  
        SET root.right TO self.trimBST(root.right, low, high)  
        RETURN root  
    END FUNCTION  
END CLASS
