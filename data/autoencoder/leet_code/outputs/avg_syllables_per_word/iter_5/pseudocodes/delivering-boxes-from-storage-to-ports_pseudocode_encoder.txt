CLASS Solution  
    FUNCTION boxDelivering(boxes, portsCount, maxBoxes, maxWeight)  
        SET n TO LENGTH OF boxes  
        SET dp TO LIST OF size n PLUS 1 WITH ALL VALUES AS infinity  
        SET dp AT INDEX 0 TO 0  
        
        SET left TO 0  
        SET current_weight TO 0  
        SET extra_trips TO 0  
        
        FOR right FROM 0 TO n MINUS 1  
            INCREMENT current_weight BY weight OF boxes AT right  
            IF right GREATER THAN 0 AND port OF boxes AT right NOT EQUAL TO port OF boxes AT right MINUS 1  
                INCREMENT extra_trips BY 1  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND (number OF boxes FROM left TO right GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight)  
                DECREMENT current_weight BY weight OF boxes AT left  
                IF left GREATER THAN 0 AND port OF boxes AT left NOT EQUAL TO port OF boxes AT left MINUS 1  
                    DECREMENT extra_trips BY 1  
                END IF  
                INCREMENT left BY 1  
            END WHILE  
            
            SET dp AT INDEX right PLUS 1 TO dp AT INDEX left PLUS 2 PLUS extra_trips  
        END FOR  
        
        RETURN dp AT INDEX n  
    END FUNCTION  
END CLASS
