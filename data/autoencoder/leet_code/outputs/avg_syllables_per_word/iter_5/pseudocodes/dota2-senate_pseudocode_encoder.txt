CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant queue  
        INITIALIZE dire queue  
        
        FOR each index i AND character s IN senate  
            IF s EQUALS 'R'  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
        
        WHILE radiant IS NOT EMPTY AND dire IS NOT EMPTY  
            SET r_index TO REMOVE FRONT FROM radiant  
            SET d_index TO REMOVE FRONT FROM dire  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS LENGTH OF senate TO dire  
        
        IF radiant IS NOT EMPTY  
            RETURN "Radiant"  
        ELSE  
            RETURN "Dire"  
    END FUNCTION  
END CLASS
