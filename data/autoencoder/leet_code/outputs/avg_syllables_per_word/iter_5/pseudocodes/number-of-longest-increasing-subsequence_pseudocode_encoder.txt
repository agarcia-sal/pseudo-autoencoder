CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF nums IS empty  
            RETURN 0  
        SET n TO LENGTH OF nums  
        SET lengths TO LIST OF 1 REPEATED n TIMES  
        SET counts TO LIST OF 1 REPEATED n TIMES  
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF nums AT i GREATER THAN nums AT j  
                    IF lengths AT j PLUS 1 GREATER THAN lengths AT i  
                        SET lengths AT i TO lengths AT j PLUS 1  
                        SET counts AT i TO counts AT j  
                    ELSE IF lengths AT j PLUS 1 EQUALS lengths AT i  
                        INCREMENT counts AT i BY counts AT j  
        SET longest TO MAXIMUM VALUE IN lengths  
        SET result TO 0  
        FOR i FROM 0 TO n MINUS 1  
            IF lengths AT i EQUALS longest  
                INCREMENT result BY counts AT i  
        RETURN result  
    END FUNCTION  
END CLASS
