CLASS Solution  
    FUNCTION findNumOfValidWords(words, puzzles)  
        FUNCTION to_mask(s)  
            SET mask TO 0  
            FOR each char IN s  
                SET mask TO mask OR 1 LEFT SHIFTED BY (ordinal value of char MINUS ordinal value of 'a')  
            RETURN mask  
        END FUNCTION  
        
        SET word_count TO new mapping with default integer zero  
        FOR each word IN words  
            SET mask TO to_mask(word)  
            IF bit count of mask IS LESS THAN OR EQUAL TO 7  
                INCREMENT word_count at key mask BY 1  
        
        SET result TO empty list  
        FOR each puzzle IN puzzles  
            SET first_char_mask TO 1 LEFT SHIFTED BY (ordinal value of first character of puzzle MINUS ordinal value of 'a')  
            SET puzzle_mask TO to_mask(puzzle from second character to end)  
            SET count TO word_count at key first_char_mask  
            
            SET subset TO puzzle_mask  
            WHILE subset IS NOT ZERO  
                INCREMENT count BY word_count at key (subset OR first_char_mask)  
                SET subset TO (subset MINUS 1) AND puzzle_mask  
            
            APPEND count TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
