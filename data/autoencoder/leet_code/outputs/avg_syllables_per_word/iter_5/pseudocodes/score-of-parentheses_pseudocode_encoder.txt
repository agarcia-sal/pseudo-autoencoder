CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO 0  
        
        FOR each char IN s  
            IF char EQUALS '('  
                APPEND score TO stack  
                SET score TO 0  
            ELSE  
                SET score TO popped value FROM stack PLUS (1 IF score EQUALS 0 ELSE 2 TIMES score)  
        RETURN score  
    END FUNCTION  
END CLASS
