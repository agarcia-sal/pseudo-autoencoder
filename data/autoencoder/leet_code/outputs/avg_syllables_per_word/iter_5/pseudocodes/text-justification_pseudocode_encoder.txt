CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF LENGTH OF line EQUALS 1 OR is_last_line IS TRUE THEN  
                RETURN join words in line separated by a space PLUS append spaces TO fill maxWidth minus num_of_letters minus LENGTH OF line plus one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO LENGTH OF line MINUS 1  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
              
                FOR i FROM 0 TO num_of_gaps MINUS 1 DO  
                    IF i LESS THAN extra_spaces THEN  
                        APPEND spaces of length spaces_between_words PLUS one TO line at index i  
                    ELSE  
                        APPEND spaces of length spaces_between_words TO line at index i  
                END FOR  
              
                RETURN concatenate all elements in line into a single string  
            END IF  
        END FUNCTION  
      
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO 0  
      
        FOR each word IN words DO  
            IF num_of_letters PLUS LENGTH OF word PLUS LENGTH OF line GREATER THAN maxWidth THEN  
                APPEND justify_line(line, num_of_letters, FALSE) TO result  
                SET line TO empty list  
                SET num_of_letters TO 0  
            END IF  
          
            APPEND word TO line  
            INCREMENT num_of_letters BY LENGTH OF word  
        END FOR  
      
        APPEND justify_line(line, num_of_letters, TRUE) TO result  
      
        RETURN result  
    END FUNCTION  
END CLASS
