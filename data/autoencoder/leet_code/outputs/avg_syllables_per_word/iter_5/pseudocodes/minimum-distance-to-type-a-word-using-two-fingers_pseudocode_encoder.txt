CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 IS None  
                RETURN 0  
            SET x1, y1 TO divmod(ORDINAL VALUE OF c1 MINUS ORDINAL VALUE OF 'A', 6)  
            SET x2, y2 TO divmod(ORDINAL VALUE OF c2 MINUS ORDINAL VALUE OF 'A', 6)  
            RETURN absolute value of x1 MINUS x2 PLUS absolute value of y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2) WITH MEMOIZATION  
            IF i EQUALS LENGTH OF word  
                RETURN 0  
            SET c TO character AT POSITION i IN word  
            SET d1 TO distance BETWEEN f1 AND c PLUS dp FOR i PLUS 1 WITH f1 SET TO c AND f2 UNCHANGED  
            SET d2 TO distance BETWEEN f2 AND c PLUS dp FOR i PLUS 1 WITH f1 UNCHANGED AND f2 SET TO c  
            RETURN MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN dp WITH i SET TO 0, f1 SET TO None, AND f2 SET TO None  
    END FUNCTION  
END CLASS
