CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO 10 RAISED TO 9 PLUS 7  
        SORT nums  
        SET n TO LENGTH OF nums  
        SET total_sum TO 0  
        SET power TO LIST OF 1 REPEATED n TIMES  
        
        FOR i FROM 1 TO n MINUS 1  
            SET power[i] TO power[i MINUS 1] MULTIPLIED BY 2 MODULO MOD  
        
        FOR i FROM 0 TO n MINUS 1  
            SET total_sum TO total_sum PLUS nums[i] MULTIPLIED BY (power[i] MINUS power[n MINUS i MINUS 1]) MODULO MOD  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
