CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO the set of all x WHERE x and y are elements zipped from fronts and backs AND x EQUALS y  
        SET all_numbers TO the set of all elements from fronts and backs combined  
        SET good_numbers TO all_numbers WITHOUT any elements in same  
        IF good_numbers IS NOT empty  
            RETURN the minimum element in good_numbers  
        ELSE  
            RETURN 0  
        END IF  
    END FUNCTION  
END CLASS
