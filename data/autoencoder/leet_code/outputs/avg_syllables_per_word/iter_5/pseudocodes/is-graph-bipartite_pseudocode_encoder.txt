CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO LENGTH OF graph  
        SET color TO list of size n filled with 0  
        
        FUNCTION dfs(node, c)  
            SET color at index node TO c  
            FOR each neighbor IN graph at index node  
                IF color at index neighbor IS EQUAL TO c  
                    RETURN False  
                IF color at index neighbor IS EQUAL TO 0  
                    IF dfs(neighbor, NEGATION OF c) IS False  
                        RETURN False  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM 0 TO n MINUS 1  
            IF color at index i IS EQUAL TO 0  
                IF dfs(i, 1) IS False  
                    RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
