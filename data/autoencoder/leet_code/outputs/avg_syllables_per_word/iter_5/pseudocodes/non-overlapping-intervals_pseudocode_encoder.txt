CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals IS EMPTY  
            RETURN 0  
        END IF  

        SORT intervals BY end time of each interval  

        SET end TO end time of first interval in intervals  
        SET count TO 0  

        FOR i FROM 1 TO LENGTH OF intervals MINUS 1  
            IF start time of intervals at i IS GREATER THAN OR EQUAL TO end  
                SET end TO end time of intervals at i  
            ELSE  
                INCREMENT count BY 1  
            END IF  
        END FOR  

        RETURN count  
    END FUNCTION  
END CLASS
