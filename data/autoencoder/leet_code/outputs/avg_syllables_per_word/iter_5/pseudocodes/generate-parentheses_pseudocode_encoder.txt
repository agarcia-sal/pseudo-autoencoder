CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s, left, right)  
            IF LENGTH OF s EQUALS 2 MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s APPENDED BY '(' , left INCREMENTED BY 1, right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s APPENDED BY ')' , left, right INCREMENTED BY 1  
            END IF  
        END FUNCTION  
      
        SET result TO EMPTY LIST  
        CALL backtrack WITH EMPTY STRING, 0, 0  
        RETURN result  
    END FUNCTION  
END CLASS
