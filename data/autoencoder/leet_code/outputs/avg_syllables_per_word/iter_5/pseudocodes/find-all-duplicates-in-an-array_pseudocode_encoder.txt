CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            WHILE value AT nums AT i IS NOT EQUAL TO value AT nums AT (value AT nums AT i MINUS 1)  
                SWAP value AT nums AT (value AT nums AT i MINUS 1) WITH value AT nums AT i  
        SET result TO EMPTY LIST  
        FOR each index i AND value v IN nums  
            IF v IS NOT EQUAL TO i PLUS 1  
                APPEND v TO result  
        RETURN result  
    END FUNCTION  
END CLASS
