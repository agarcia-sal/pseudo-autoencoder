CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO 0  
        APPEND list containing target and 0 TO stations  

        FOR each position, fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is less than position  
                SET current_fuel TO current_fuel MINUS the largest fuel amount extracted from max_heap  
                INCREMENT stops BY 1  
            END WHILE  
            IF current_fuel is less than position THEN  
                RETURN -1  
            END IF  
            ADD negative of fuel TO max_heap to maintain max-heap behavior  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
