CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO version1 SPLIT BY '.'  
        SET v2_revisions TO version2 SPLIT BY '.'  
        SET max_length TO GREATER OF LENGTH OF v1_revisions AND LENGTH OF v2_revisions  
        FOR i FROM 0 TO max_length MINUS 1  
            IF i IS LESS THAN LENGTH OF v1_revisions  
                SET v1 TO INTEGER VALUE OF v1_revisions AT i  
            ELSE  
                SET v1 TO 0  
            IF i IS LESS THAN LENGTH OF v2_revisions  
                SET v2 TO INTEGER VALUE OF v2_revisions AT i  
            ELSE  
                SET v2 TO 0  
            IF v1 IS LESS THAN v2  
                RETURN -1  
            ELSE IF v1 IS GREATER THAN v2  
                RETURN 1  
        RETURN 0  
    END FUNCTION  
END CLASS
