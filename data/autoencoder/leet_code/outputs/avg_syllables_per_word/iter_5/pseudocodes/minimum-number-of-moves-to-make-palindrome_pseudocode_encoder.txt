CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO list of characters in s  
        SET n TO length of s  
        SET moves TO 0  
        
        WHILE length of s GREATER THAN 1  
            SET found_match TO false  
            FOR i FROM length of s MINUS 1 DOWNTO 1  
                IF character at s index i EQUALS character at s index 0 THEN  
                    SET found_match TO true  
                    FOR j FROM i TO length of s MINUS 2  
                        SWAP characters at s index j AND s index j PLUS 1  
                        INCREMENT moves BY 1  
                    END FOR  
                    REMOVE first character FROM s  
                    REMOVE last character FROM s  
                    BREAK FOR LOOP  
                END IF  
            END FOR  
            
            IF found_match IS false THEN  
                REMOVE first character FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
