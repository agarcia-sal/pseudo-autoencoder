CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS 1 THEN  
            RETURN s  
        END IF  
        
        SET rows TO list of empty strings with length numRows  
        SET current_row TO 0  
        SET going_down TO false  
        
        FOR each char IN s  
            APPEND char TO rows at current_row  
            IF current_row EQUALS 0 OR current_row EQUALS numRows MINUS 1 THEN  
                TOGGLE going_down  
            END IF  
            IF going_down IS true THEN  
                INCREMENT current_row BY 1  
            ELSE  
                DECREMENT current_row BY 1  
            END IF  
        END FOR  
        
        RETURN concatenation of all strings in rows  
    END FUNCTION  
END CLASS
