CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  

        FOR each num IN nums  
            IF num MODULO 2 IS NOT 0  
                SET num TO num MULTIPLIED BY 2  
            END IF  
            IF num IS LESS THAN min_value  
                SET min_value TO num  
            END IF  
            PUSH negative of num INTO max_heap  
        END FOR  

        SET min_deviation TO infinity  

        WHILE negative of max_heap AT INDEX 0 MODULO 2 IS 0  
            SET max_value TO negative of POP from max_heap  
            IF max_value MINUS min_value IS LESS THAN min_deviation  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY 2  
            IF max_value IS LESS THAN min_value  
                SET min_value TO max_value  
            END IF  
            PUSH negative of max_value INTO max_heap  
        END WHILE  

        IF negative of max_heap AT INDEX 0 MINUS min_value IS LESS THAN min_deviation  
            SET min_deviation TO negative of max_heap AT INDEX 0 MINUS min_value  
        END IF  

        RETURN min_deviation  
    END FUNCTION  
END CLASS
