CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new empty mapping from content to list of paths  
        
        FOR each path IN paths  
            SET parts TO the list of substrings obtained by splitting path by spaces  
            SET directory TO the first element of parts  
            
            FOR each file_info IN all elements of parts starting from the second  
                SET file_name, content_part TO the two substrings obtained by splitting file_info at the first '('  
                REMOVE the last character from content_part to obtain content  
                
                APPEND the concatenation of directory, "/", and file_name TO content_to_paths at key content  
        
        SET duplicate_groups TO an empty list  
        FOR each group IN all values of content_to_paths  
            IF the number of elements in group is greater than one  
                APPEND group TO duplicate_groups  
        
        RETURN duplicate_groups  
    END FUNCTION  
END CLASS
