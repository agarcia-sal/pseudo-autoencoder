CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO 3 THEN  
            RETURN n  
        END IF  
  
        SET dp TO list of zeros with length n PLUS 1  
  
        FOR i FROM 1 TO 3 INCLUSIVE DO  
            SET dp at index i TO i  
        END FOR  
  
        FOR i FROM 4 TO n INCLUSIVE DO  
            FOR j FROM i MINUS 3 DOWN TO 1 DO  
                SET dp at index i TO maximum of dp at index i AND dp at index j MULTIPLIED BY (i MINUS j MINUS 1)  
            END FOR  
        END FOR  
  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
