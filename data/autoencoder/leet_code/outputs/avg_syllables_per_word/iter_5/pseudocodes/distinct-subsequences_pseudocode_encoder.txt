CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO LENGTH OF s  
        SET n TO LENGTH OF t  
        
        CREATE dp AS 2D array WITH DIMENSIONS (m PLUS 1) BY (n PLUS 1), FILLED WITH 0  
        
        FOR i FROM 0 TO m  
            SET dp[i][0] TO 1  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT s AT (i MINUS 1) EQUALS CHARACTER AT t AT (j MINUS 1) THEN  
                    SET dp[i][j] TO dp[i MINUS 1][j MINUS 1] PLUS dp[i MINUS 1][j]  
                ELSE  
                    SET dp[i][j] TO dp[i MINUS 1][j]  
        
        RETURN dp[m][n]  
    END FUNCTION  
END CLASS
