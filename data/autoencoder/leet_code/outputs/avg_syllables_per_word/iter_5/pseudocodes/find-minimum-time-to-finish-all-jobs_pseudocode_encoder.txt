CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO LENGTH OF jobs  
        
        SET subset_sums TO LIST OF SIZE 2 TO THE POWER OF n WITH ALL VALUES 0  
        FOR i FROM 0 TO n MINUS 1  
            FOR mask FROM 0 TO 2 TO THE POWER OF i MINUS 1  
                SET subset_sums AT (mask OR 2 TO THE POWER OF i) TO subset_sums AT mask PLUS jobs AT i  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS 1  
                RETURN subset_sums AT mask  
            SET res TO INFINITY  
            SET submask TO mask  
            WHILE submask IS NOT 0  
                SET res TO MINIMUM OF res AND MAXIMUM OF dp(k MINUS 1, mask XOR submask) AND subset_sums AT submask  
                SET submask TO (submask MINUS 1) BITWISE AND mask  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, 2 TO THE POWER OF n MINUS 1)  
    END FUNCTION  
END CLASS
