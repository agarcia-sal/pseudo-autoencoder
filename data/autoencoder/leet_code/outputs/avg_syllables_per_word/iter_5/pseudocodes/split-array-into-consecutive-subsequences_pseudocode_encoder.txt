CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO count frequencies of elements in nums  
        SET end_with TO empty count dictionary  
        
        FOR each num IN nums  
            IF num_count at num EQUALS 0  
                CONTINUE  
            
            IF end_with at num MINUS 1 GREATER THAN 0  
                DECREMENT end_with at num MINUS 1 BY 1  
                INCREMENT end_with at num BY 1  
            ELSE IF num_count at num PLUS 1 GREATER THAN 0 AND num_count at num PLUS 2 GREATER THAN 0  
                DECREMENT num_count at num PLUS 1 BY 1  
                DECREMENT num_count at num PLUS 2 BY 1  
                INCREMENT end_with at num PLUS 2 BY 1  
            ELSE  
                RETURN False  
            
            DECREMENT num_count at num BY 1  
        
        RETURN True  
    END FUNCTION  
END CLASS
