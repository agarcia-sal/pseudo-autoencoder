CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO TreeNode WITH values AT FIRST POSITION  
    SET i TO 1  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO TreeNode WITH values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO TreeNode WITH values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth IS EQUAL TO 1  
            RETURN TreeNode WITH val SET TO val AND left SET TO root  
        SET queue TO new deque CONTAINING (root, 1)  
        WHILE queue IS NOT EMPTY  
            SET node, level TO REMOVE FIRST ELEMENT FROM queue  
            IF level IS EQUAL TO depth MINUS 1  
                SET node.left TO TreeNode WITH val SET TO val AND left SET TO node.left  
                SET node.right TO TreeNode WITH val SET TO val AND right SET TO node.right  
            ELSE IF level IS LESS THAN depth MINUS 1  
                IF node.left IS NOT None  
                    APPEND (node.left, level PLUS 1) TO queue  
                IF node.right IS NOT None  
                    APPEND (node.right, level PLUS 1) TO queue  
        RETURN root  
    END FUNCTION  
END CLASS
