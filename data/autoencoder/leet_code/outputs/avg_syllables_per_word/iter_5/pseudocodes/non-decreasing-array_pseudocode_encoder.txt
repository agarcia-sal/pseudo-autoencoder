CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO False  
        FOR i FROM 1 TO LENGTH OF nums MINUS 1  
            IF nums AT i IS LESS THAN nums AT i MINUS 1  
                IF modified IS True  
                    RETURN False  
                SET modified TO True  
                IF i IS LESS THAN 2 OR nums AT i IS GREATER THAN OR EQUAL TO nums AT i MINUS 2  
                    SET nums AT i MINUS 1 TO nums AT i  
                ELSE  
                    SET nums AT i TO nums AT i MINUS 1  
        RETURN True  
    END FUNCTION  
END CLASS
