CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS A TWO-DIMENSIONAL LIST OF SIZE (n PLUS 1) BY (k PLUS 1) WITH ALL VALUES SET TO 0  
        SET dp AT POSITION 0, 0 TO 1  
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 1 TO k INCLUSIVE  
                SET dp AT POSITION i, j TO THE REMAINDER OF (j MULTIPLIED BY dp AT POSITION i MINUS 1, j PLUS dp AT POSITION i MINUS 1, j MINUS 1) DIVIDED BY MOD  
        RETURN dp AT POSITION n, k  
    END FUNCTION  
END CLASS
