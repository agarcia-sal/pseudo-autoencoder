CLASS Solution  
    FUNCTION minimumWeight(n, edges, src1, src2, dest)  
        FUNCTION dijkstra(graph, start)  
            SET distances TO list of size n with all values as infinity  
            SET distances at index start TO 0  
            SET heap TO list containing tuple of 0 and start  
            
            WHILE heap is not empty  
                SET current_distance, current_node TO REMOVE smallest element from heap  
                
                IF current_distance GREATER THAN distances at index current_node  
                    CONTINUE to next iteration  
                
                FOR each neighbor, weight in graph at index current_node  
                    SET distance TO current_distance PLUS weight  
                    
                    IF distance LESS THAN distances at index neighbor  
                        SET distances at index neighbor TO distance  
                        ADD tuple of distance and neighbor TO heap  
            RETURN distances  
        END FUNCTION  
        
        SET graph TO list of n empty lists  
        SET reverse_graph TO list of n empty lists  
        
        FOR each u, v, w IN edges  
            APPEND tuple of v and w TO graph at index u  
            APPEND tuple of u and w TO reverse_graph at index v  
        
        SET dist_from_src1 TO dijkstra(graph, src1)  
        SET dist_from_src2 TO dijkstra(graph, src2)  
        SET dist_to_dest TO dijkstra(reverse_graph, dest)  
        
        SET min_weight TO infinity  
        
        FOR i FROM 0 TO n MINUS 1  
            IF dist_from_src1 at index i NOT EQUAL TO infinity AND dist_from_src2 at index i NOT EQUAL TO infinity AND dist_to_dest at index i NOT EQUAL TO infinity  
                SET current_sum TO dist_from_src1 at index i PLUS dist_from_src2 at index i PLUS dist_to_dest at index i  
                IF current_sum LESS THAN min_weight  
                    SET min_weight TO current_sum  
        
        IF min_weight EQUAL TO infinity  
            RETURN negative 1  
        ELSE  
            RETURN min_weight  
    END FUNCTION  
END CLASS
