CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS EMPTY OR LENGTH OF stones IS LESS THAN 2  
            RETURN False  
        
        SET stone_set TO CONVERT stones TO SET  
        SET memo TO EMPTY MAP  
        
        FUNCTION can_jump_to(position, jump)  
            IF position IS EQUAL TO LAST ELEMENT OF stones  
                RETURN True  
            
            IF (position, jump) EXISTS IN memo  
                RETURN VALUE OF memo AT (position, jump)  
            
            FOR next_jump FROM jump MINUS 1 TO jump PLUS 1  
                IF next_jump IS GREATER THAN 0 AND position PLUS next_jump EXISTS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump) IS True  
                        SET memo AT (position, jump) TO True  
                        RETURN True  
            
            SET memo AT (position, jump) TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(FIRST ELEMENT OF stones, 0)  
    END FUNCTION  
END CLASS
