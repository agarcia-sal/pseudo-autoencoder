CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO LENGTH OF dominoes  
        SET forces TO list of zeros with length n  

        SET force TO 0  
        FOR i FROM 0 TO n MINUS 1  
            IF dominoes AT i EQUALS 'R'  
                SET force TO n  
            ELSE IF dominoes AT i EQUALS 'L'  
                SET force TO 0  
            ELSE  
                SET force TO MAXIMUM OF force MINUS 1 AND 0  
            END IF  
            INCREMENT forces AT i BY force  
        END FOR  

        SET force TO 0  
        FOR i FROM n MINUS 1 DOWNTO 0  
            IF dominoes AT i EQUALS 'L'  
                SET force TO n  
            ELSE IF dominoes AT i EQUALS 'R'  
                SET force TO 0  
            ELSE  
                SET force TO MAXIMUM OF force MINUS 1 AND 0  
            END IF  
            DECREMENT forces AT i BY force  
        END FOR  

        SET result TO empty list  
        FOR each f IN forces  
            IF f GREATER THAN 0  
                APPEND 'R' TO result  
            ELSE IF f LESS THAN 0  
                APPEND 'L' TO result  
            ELSE  
                APPEND '.' TO result  
            END IF  
        END FOR  

        RETURN JOIN elements of result into a string  
    END FUNCTION  
END CLASS
