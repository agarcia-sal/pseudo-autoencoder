CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO list of pairs (character, index) in start where character is not 'X'  
        SET end_filtered TO list of pairs (character, index) in end where character is not 'X'  
        
        IF LENGTH OF start_filtered NOT EQUAL TO LENGTH OF end_filtered THEN  
            RETURN False  
        END IF  
        
        FOR each pair (c1, i1) IN start_filtered AND each pair (c2, i2) IN end_filtered SIMULTANEOUSLY  
            IF c1 NOT EQUAL TO c2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUAL TO 'L' AND i1 LESS THAN i2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUAL TO 'R' AND i1 GREATER THAN i2 THEN  
                RETURN False  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
