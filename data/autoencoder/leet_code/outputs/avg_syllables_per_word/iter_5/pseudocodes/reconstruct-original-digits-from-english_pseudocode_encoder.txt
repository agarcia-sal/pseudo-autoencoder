CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO frequency count of characters in s  
        SET digits TO list of ten zeros  
        SET digits at position 0 TO count of character 'z'  
        SET digits at position 2 TO count of character 'w'  
        SET digits at position 4 TO count of character 'u'  
        SET digits at position 6 TO count of character 'x'  
        SET digits at position 8 TO count of character 'g'  
        SET digits at position 3 TO count of character 'h' MINUS digits at position 8  
        SET digits at position 5 TO count of character 'f' MINUS digits at position 4  
        SET digits at position 7 TO count of character 's' MINUS digits at position 6  
        SET digits at position 9 TO count of character 'i' MINUS digits at position 5 MINUS digits at position 6 MINUS digits at position 8  
        SET digits at position 1 TO count of character 'o' MINUS digits at position 0 MINUS digits at position 2 MINUS digits at position 4  
        SET result TO empty list  
        FOR i FROM 0 TO 9  
            APPEND string representation of i REPEATED digits at position i times TO result  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
