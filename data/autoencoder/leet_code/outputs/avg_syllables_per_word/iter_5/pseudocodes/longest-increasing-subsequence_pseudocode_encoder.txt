CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums IS empty  
            RETURN 0  
        END IF  
        
        SET dp TO empty list  
        
        FOR each num IN nums  
            SET idx TO position to insert num in dp using binary search to maintain sorted order  
            
            IF idx IS equal to LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET element at position idx in dp TO num  
            END IF  
        END FOR  
        
        RETURN LENGTH OF dp  
    END FUNCTION  
END CLASS
