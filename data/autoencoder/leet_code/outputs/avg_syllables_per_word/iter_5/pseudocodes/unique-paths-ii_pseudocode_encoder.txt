CLASS Solution
    FUNCTION uniquePathsWithObstacles(obstacleGrid)
        IF obstacleGrid IS empty OR obstacleGrid at position 0 0 EQUALS 1 THEN
            RETURN 0
        END IF

        SET m TO LENGTH OF obstacleGrid
        SET n TO LENGTH OF obstacleGrid at position 0

        SET dp TO 2D array of size m by n FILLED WITH 0

        SET dp at position 0 0 TO 1

        FOR j FROM 1 TO n MINUS 1 DO
            IF obstacleGrid at position 0 j EQUALS 0 THEN
                SET dp at position 0 j TO dp at position 0 j MINUS 1
            END IF
        END FOR

        FOR i FROM 1 TO m MINUS 1 DO
            IF obstacleGrid at position i 0 EQUALS 0 THEN
                SET dp at position i 0 TO dp at position i MINUS 1 0
            END IF
        END FOR

        FOR i FROM 1 TO m MINUS 1 DO
            FOR j FROM 1 TO n MINUS 1 DO
                IF obstacleGrid at position i j EQUALS 0 THEN
                    SET dp at position i j TO dp at position i MINUS 1 j PLUS dp at position i j MINUS 1
                END IF
            END FOR
        END FOR

        RETURN dp at position m MINUS 1 n MINUS 1
    END FUNCTION
END CLASS
