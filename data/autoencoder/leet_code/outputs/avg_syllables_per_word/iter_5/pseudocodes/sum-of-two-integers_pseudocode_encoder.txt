CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO maximum 32-bit integer value  
        SET mask TO bitmask for 32 bits  

        WHILE b is not equal to zero  
            SET carry TO bitwise AND of a and b shifted left by one  
            SET a TO bitwise XOR of a and b AND mask  
            SET b TO carry AND mask  
        END WHILE  

        IF a is less than or equal to MAX  
            RETURN a  
        ELSE  
            RETURN bitwise complement of (a XOR mask)  
        END IF  
    END FUNCTION  
END CLASS
