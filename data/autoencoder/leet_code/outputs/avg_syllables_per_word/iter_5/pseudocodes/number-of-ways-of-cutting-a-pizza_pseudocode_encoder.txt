CLASS Solution  
    FUNCTION ways(pizza, k)  
        SET mod TO 1000000007  
        SET m TO LENGTH OF pizza  
        SET n TO LENGTH OF pizza AT 0  
        SET s TO 2D list of zeros with dimensions (m + 1) BY (n + 1)  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                SET s[i][j] TO s[i - 1][j] PLUS s[i][j - 1] MINUS s[i - 1][j - 1] PLUS 1 IF pizza AT (i - 1) AT (j - 1) EQUALS 'A' ELSE 0  
        
        DEFINE FUNCTION dfs(i, j, k)  
            IF k EQUALS 0 THEN  
                RETURN 1 IF s[m][n] MINUS s[i][n] MINUS s[m][j] PLUS s[i][j] GREATER THAN 0 ELSE 0  
            SET ans TO 0  
            FOR x FROM i PLUS 1 TO m MINUS 1  
                IF s[x][n] MINUS s[i][n] MINUS s[x][j] PLUS s[i][j] GREATER THAN 0 THEN  
                    INCREMENT ans BY dfs(x, j, k MINUS 1)  
            FOR y FROM j PLUS 1 TO n MINUS 1  
                IF s[m][y] MINUS s[i][y] MINUS s[m][j] PLUS s[i][j] GREATER THAN 0 THEN  
                    INCREMENT ans BY dfs(i, y, k MINUS 1)  
            RETURN ans MODULO mod  
        
        RETURN dfs(0, 0, k MINUS 1)  
    END FUNCTION  
END CLASS
