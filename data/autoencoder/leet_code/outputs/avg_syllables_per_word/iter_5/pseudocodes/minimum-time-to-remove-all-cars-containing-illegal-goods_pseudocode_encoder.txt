CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO LENGTH OF s  
        SET left_costs TO list of zeroes with length n  
        SET right_costs TO list of zeroes with length n  

        IF s at index 0 EQUALS '1'  
            SET left_costs at index 0 TO 1  
        ELSE  
            SET left_costs at index 0 TO 0  

        FOR i FROM 1 TO n MINUS 1  
            IF s at index i EQUALS '0'  
                SET left_costs at index i TO left_costs at index i MINUS 1  
            ELSE  
                SET left_costs at index i TO MINIMUM OF (i PLUS 1) AND (left_costs at index i MINUS 1 PLUS 2)  

        IF s at index n MINUS 1 EQUALS '1'  
            SET right_costs at index n MINUS 1 TO 1  
        ELSE  
            SET right_costs at index n MINUS 1 TO 0  

        FOR i FROM n MINUS 2 DOWNTO 0  
            IF s at index i EQUALS '0'  
                SET right_costs at index i TO right_costs at index i PLUS 1  
            ELSE  
                SET right_costs at index i TO MINIMUM OF (n MINUS i) AND (right_costs at index i PLUS 1 PLUS 2)  

        SET min_time TO MINIMUM OF left_costs at index n MINUS 1 AND right_costs at index 0  

        FOR i FROM 0 TO n MINUS 2  
            SET min_time TO MINIMUM OF min_time AND (left_costs at index i PLUS right_costs at index i PLUS 1)  

        RETURN min_time  
    END FUNCTION  
END CLASS
