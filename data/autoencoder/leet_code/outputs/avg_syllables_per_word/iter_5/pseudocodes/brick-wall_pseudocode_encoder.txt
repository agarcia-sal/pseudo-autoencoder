CLASS Solution  
    FUNCTION leastBricks(wall)  
        SET gap_counts TO a default dictionary with integer default value  
        
        FOR each row IN wall  
            SET total TO 0  
            FOR each brick IN all bricks in row except the last  
                INCREMENT total BY brick  
                INCREMENT gap_counts at key total BY 1  
        
        SET max_gaps TO the maximum value in gap_counts or 0 if empty  
        
        RETURN length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
