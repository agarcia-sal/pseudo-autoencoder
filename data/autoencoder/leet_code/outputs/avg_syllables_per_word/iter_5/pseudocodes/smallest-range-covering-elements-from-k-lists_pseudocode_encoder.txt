CLASS Solution  
    FUNCTION smallestRange(nums)  
        INITIALIZE min_heap AS empty list  
        SET max_value TO negative infinity  
        
        FOR each index i IN range LENGTH OF nums  
            PUSH tuple (nums at i at 0, i, 0) INTO min_heap  
            SET max_value TO maximum OF max_value AND nums at i at 0  
        
        SET smallest_range TO infinity  
        SET result TO list of -100000 AND 100000  
        
        WHILE min_heap IS NOT empty  
            POP tuple (min_value, list_index, element_index) FROM min_heap  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list of min_value AND max_value  
            
            IF element_index PLUS 1 IS LESS THAN LENGTH OF nums at list_index THEN  
                SET next_value TO nums at list_index at element_index PLUS 1  
                PUSH tuple (next_value, list_index, element_index PLUS 1) INTO min_heap  
                SET max_value TO maximum OF max_value AND next_value  
            ELSE  
                BREAK  
        
        RETURN result  
    END FUNCTION  
END CLASS
