CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF LENGTH OF edges IS NOT EQUAL TO n MINUS 1  
            RETURN False  
        END IF  

        SET adj_list TO A MAPPING FROM EACH INTEGER i FROM 0 TO n MINUS 1 TO AN EMPTY LIST  
        FOR EACH pair a, b IN edges  
            APPEND b TO adj_list AT a  
            APPEND a TO adj_list AT b  
        END FOR  

        SET visited TO AN EMPTY SET  

        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR EACH neighbor IN adj_list AT node  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  

        CALL dfs WITH 0  

        RETURN WHETHER LENGTH OF visited IS EQUAL TO n  
    END FUNCTION  
END CLASS
