CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO dictionary with key 0 mapped to 1  
        SET current_sum TO 0  
        SET count TO 0  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF cumulative_sum_freq CONTAINS key current_sum MINUS k THEN  
                INCREMENT count BY value from cumulative_sum_freq at key current_sum MINUS k  
            END IF  
            IF cumulative_sum_freq CONTAINS key current_sum THEN  
                INCREMENT value from cumulative_sum_freq at key current_sum BY 1  
            ELSE  
                SET value from cumulative_sum_freq at key current_sum TO 1  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
