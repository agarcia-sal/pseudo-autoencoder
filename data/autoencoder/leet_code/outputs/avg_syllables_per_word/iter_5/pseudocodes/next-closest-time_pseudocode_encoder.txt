CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO unique characters in time excluding colon  
        SET hour, minute TO integer values obtained by splitting time at colon  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour TO integer formed by concatenating h1 and h2  
                            SET minute TO integer formed by concatenating m1 and m2  
                            IF hour LESS THAN 24 AND minute LESS THAN 60  
                                APPEND tuple (hour, minute) TO times  
            RETURN sorted times  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        SET current_time_index TO position of tuple (hour, minute) IN all_times  
        
        IF current_time_index PLUS 1 LESS THAN length of all_times  
            SET next_hour, next_minute TO element at current_time_index PLUS 1 IN all_times  
        ELSE  
            SET next_hour, next_minute TO first element IN all_times  
        
        RETURN formatted string of next_hour and next_minute as two-digit numbers separated by colon  
    END FUNCTION  
END CLASS
