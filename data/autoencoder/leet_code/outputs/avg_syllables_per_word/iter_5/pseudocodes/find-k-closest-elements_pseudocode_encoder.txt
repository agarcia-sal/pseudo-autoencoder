CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO bisect_left_position_of x IN arr MINUS 1  
        SET right TO left PLUS 1  
        
        WHILE right MINUS left MINUS 1 LESS THAN k  
            IF left EQUALS -1 OR (right LESS THAN LENGTH OF arr AND absolute_value_of arr AT right MINUS x LESS THAN absolute_value_of arr AT left MINUS x)  
                INCREMENT right BY 1  
            ELSE  
                DECREMENT left BY 1  
        END WHILE  
        
        RETURN SUBLIST OF arr FROM left PLUS 1 UP TO right  
    END FUNCTION  
END CLASS
