CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a mapping with default empty list  
        FOR each pair (frm, to) IN tickets sorted in descending order  
            APPEND to TO graph at key frm  
        
        SET itinerary TO empty list  
        
        FUNCTION visit(airport)  
            WHILE graph at airport is not empty  
                CALL visit with last element removed from graph at airport  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with "JFK"  
        RETURN itinerary reversed  
    END FUNCTION  
END CLASS
