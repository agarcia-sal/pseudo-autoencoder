CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN 0 AND stack IS NOT empty AND last element in stack GREATER THAN digit  
                REMOVE last element from stack  
                DECREMENT k BY 1  
            APPEND digit TO stack  
        
        IF k GREATER THAN 0  
            SET final_stack TO elements of stack UP TO LENGTH OF stack MINUS k  
        ELSE  
            SET final_stack TO stack  
        
        SET result TO concatenation of elements in final_stack WITH leading zeros REMOVED  
        
        IF result IS empty  
            RETURN string "0"  
        ELSE  
            RETURN result  
    END FUNCTION  
END CLASS
