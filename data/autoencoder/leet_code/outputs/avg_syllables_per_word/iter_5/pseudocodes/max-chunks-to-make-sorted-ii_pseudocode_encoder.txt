CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO first element of arr  
        FOR each num IN arr  
            SET current_max TO maximum of current_max and num  
            APPEND current_max TO max_so_far  
        SET sorted_arr TO arr sorted in ascending order  
        SET chunks TO 0  
        SET max_chunk TO 0  
        FOR index FROM 0 TO length of arr MINUS 1  
            SET max_chunk TO maximum of max_chunk and element at index in arr  
            IF max_chunk EQUALS element at index in sorted_arr  
                INCREMENT chunks BY 1  
        RETURN chunks  
    END FUNCTION  
END CLASS
