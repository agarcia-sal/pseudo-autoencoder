CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF nums IS empty  
            RETURN 0  
        
        SET num_set TO set of elements in nums  
        SET longest_streak TO 0  
        
        FOR each num IN num_set  
            IF num minus 1 is NOT in num_set  
                SET current_num TO num  
                SET current_streak TO 1  
                
                WHILE current_num plus 1 IS in num_set  
                    INCREMENT current_num BY 1  
                    INCREMENT current_streak BY 1  
                
                SET longest_streak TO the greater value between longest_streak and current_streak  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
