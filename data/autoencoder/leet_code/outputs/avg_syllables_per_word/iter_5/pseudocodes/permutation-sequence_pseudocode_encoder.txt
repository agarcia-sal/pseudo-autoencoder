CLASS Solution  
    FUNCTION getPermutation(n, k)  
        IMPORT math  
        
        SET numbers TO list of integers from 1 TO n  
        DECREMENT k BY 1  
        SET result TO empty list  
        
        FOR i FROM n DOWNTO 1  
            SET fact TO factorial of (i MINUS 1)  
            SET index TO k DIVIDED BY fact  
            APPEND string representation of numbers at position index TO result  
            REMOVE element at position index FROM numbers  
            SET k TO remainder of k DIVIDED BY fact  
        END FOR  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
