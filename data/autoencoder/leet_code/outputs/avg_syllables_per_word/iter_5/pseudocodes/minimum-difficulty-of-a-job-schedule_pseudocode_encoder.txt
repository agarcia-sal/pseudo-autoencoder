CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN MAXIMUM VALUE IN jobDifficulty FROM INDEX i TO END  
            END IF  
            
            SET min_difficulty TO infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO GREATER OF current_max AND jobDifficulty AT INDEX j  
                SET min_difficulty TO LESSER OF min_difficulty AND current_max PLUS dp(j PLUS one, d MINUS one)  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
END CLASS
