CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        CREATE adj_list AS map FROM each course IN range numCourses TO empty list  
        CREATE in_degrees AS list OF zeroes WITH length numCourses  
        
        FOR each pair dest, src IN prerequisites  
            APPEND dest TO adj_list[src]  
            INCREMENT in_degrees[dest] BY 1  
        
        INITIALIZE queue WITH all courses IN range numCourses WHERE in_degrees[course] EQUALS zero  
        INITIALIZE topological_order AS empty list  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET TO current  
            APPEND current TO topological_order  
            
            FOR each neighbor IN adj_list[current]  
                DECREMENT in_degrees[neighbor] BY 1  
                IF in_degrees[neighbor] EQUALS zero THEN  
                    APPEND neighbor TO queue  
        
        IF LENGTH OF topological_order EQUALS numCourses THEN  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
    END FUNCTION  
END CLASS
