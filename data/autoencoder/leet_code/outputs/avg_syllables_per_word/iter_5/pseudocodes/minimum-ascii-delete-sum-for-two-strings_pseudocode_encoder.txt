CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO LENGTH OF s1  
        SET n TO LENGTH OF s2  
          
        INITIALIZE dp AS 2D ARRAY WITH DIMENSIONS (m PLUS 1) BY (n PLUS 1) FILLED WITH 0  
          
        FOR j FROM 1 TO n  
            SET dp AT 0 AND j TO dp AT 0 AND (j MINUS 1) PLUS ASCII VALUE OF CHARACTER AT POSITION (j MINUS 1) IN s2  
          
        FOR i FROM 1 TO m  
            SET dp AT i AND 0 TO dp AT (i MINUS 1) AND 0 PLUS ASCII VALUE OF CHARACTER AT POSITION (i MINUS 1) IN s1  
          
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT POSITION (i MINUS 1) IN s1 EQUALS CHARACTER AT POSITION (j MINUS 1) IN s2  
                    SET dp AT i AND j TO dp AT (i MINUS 1) AND (j MINUS 1)  
                ELSE  
                    SET dp AT i AND j TO MINIMUM OF (dp AT (i MINUS 1) AND j PLUS ASCII VALUE OF CHARACTER AT POSITION (i MINUS 1) IN s1) AND (dp AT i AND (j MINUS 1) PLUS ASCII VALUE OF CHARACTER AT POSITION (j MINUS 1) IN s2)  
          
        RETURN dp AT m AND n  
    END FUNCTION  
END CLASS
