CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO LENGTH OF arr  
        SET dp TO TWO-DIMENSIONAL LIST OF SIZE n BY n FILLED WITH 0  

        FOR i FROM 0 TO n MINUS 1  
            SET dp[i][i] TO 1  

        FOR i FROM 0 TO n MINUS 2  
            IF arr[i] EQUALS arr[i PLUS 1]  
                SET dp[i][i PLUS 1] TO 1  
            ELSE  
                SET dp[i][i PLUS 1] TO 2  

        FOR length FROM 3 TO n  
            FOR i FROM 0 TO n MINUS length  
                SET j TO i PLUS length MINUS 1  
                IF arr[i] EQUALS arr[j]  
                    SET dp[i][j] TO dp[i PLUS 1][j MINUS 1]  
                ELSE  
                    SET dp[i][j] TO INFINITY  
                    FOR k FROM i TO j MINUS 1  
                        SET dp[i][j] TO MINIMUM OF dp[i][j] AND dp[i][k] PLUS dp[k PLUS 1][j]  

        RETURN dp[0][n MINUS 1]  
    END FUNCTION  
END CLASS
