CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    SET head TO NEW ListNode WITH val values at position zero AND next None  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO NEW ListNode WITH val val AND next None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head IS None OR head.next IS None  
            RETURN head  
        SET dummy TO NEW ListNode WITH val -5001 AND next None  
        SET dummy.next TO head  
        SET current TO head.next  
        SET head.next TO None  
        WHILE current IS NOT None  
            SET next_node TO current.next  
            SET prev TO dummy  
            WHILE prev.next IS NOT None AND prev.next.val IS LESS THAN current.val  
                SET prev TO prev.next  
            END WHILE  
            SET current.next TO prev.next  
            SET prev.next TO current  
            SET current TO next_node  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
