CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of twenty-six elements all equal to minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each index i and character char IN s  
            SET index TO numerical position of char in alphabet starting at zero  
            INCREMENT current_appeal BY i MINUS last element at position index  
            INCREMENT total_appeal BY current_appeal  
            SET last element at position index TO i  
        END FOR  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
