CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS 0 THEN  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS 1 THEN  
                RETURN list containing strings "0", "1", "8"  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS 2, final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUAL TO final_length THEN  
                    APPEND string "0" concatenated with middle concatenated with "0" TO result  
                END IF  
                APPEND string "1" concatenated with middle concatenated with "1" TO result  
                APPEND string "6" concatenated with middle concatenated with "9" TO result  
                APPEND string "8" concatenated with middle concatenated with "8" TO result  
                APPEND string "9" concatenated with middle concatenated with "6" TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO 0  
            FOR length FROM LENGTH OF low TO LENGTH OF high INCLUSIVE  
                FOR each num IN generate_strobogrammatic(length, length)  
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high) THEN  
                        CONTINUE TO next iteration  
                    END IF  
                    INCREMENT count BY 1  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
