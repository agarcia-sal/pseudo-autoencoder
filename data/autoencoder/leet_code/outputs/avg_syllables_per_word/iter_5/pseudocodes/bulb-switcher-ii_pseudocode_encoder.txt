CLASS Solution  
    FUNCTION flipLights(n, presses)  
        SET n TO minimum of n and 6  
        
        IF presses EQUALS 0 THEN  
            RETURN 1  
        END IF  
        
        IF n EQUALS 1 THEN  
            RETURN 2  
        END IF  
        
        IF n EQUALS 2 THEN  
            IF presses EQUALS 1 THEN  
                RETURN 3  
            ELSE  
                RETURN 4  
            END IF  
        END IF  
        
        IF presses EQUALS 1 THEN  
            RETURN 4  
        END IF  
        
        IF presses EQUALS 2 THEN  
            RETURN 7  
        END IF  
        
        RETURN 8  
    END FUNCTION  
END CLASS
