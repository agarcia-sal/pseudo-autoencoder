CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s IS EMPTY  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO numeric value of c MINUS numeric value of first character of s  
                SET wrapped_difference TO difference MODULO 26  
                APPEND string representation of wrapped_difference TO canonical_form  
            END FOR  
            RETURN concatenation of canonical_form elements SEPARATED BY comma  
        END FUNCTION  
        
        SET groups TO empty dictionary with default value as empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form(s)  
            APPEND s TO groups at key canonical_form  
        END FOR  
        
        RETURN list of all values in groups  
    END FUNCTION  
END CLASS
