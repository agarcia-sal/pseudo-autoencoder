CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET d TO EMPTY DICTIONARY  
        FOR i FROM 0 TO LENGTH OF req_skills MINUS 1  
            SET d AT req_skills AT i TO i  
        SET m TO LENGTH OF req_skills  
        SET n TO LENGTH OF people  
        SET p TO LIST OF ZEROS OF SIZE n  
        FOR i FROM 0 TO n MINUS 1  
            FOR EACH skill IN people AT i  
                SET p AT i TO p AT i BITWISE OR WITH 1 LEFT SHIFTED BY d AT skill  
        SET f TO LIST FILLED WITH infinity OF SIZE 1 LEFT SHIFTED BY m  
        SET g TO LIST FILLED WITH ZEROS OF SIZE 1 LEFT SHIFTED BY m  
        SET h TO LIST FILLED WITH ZEROS OF SIZE 1 LEFT SHIFTED BY m  
        SET f AT 0 TO 0  
        FOR i FROM 0 TO 1 LEFT SHIFTED BY m MINUS 1  
            IF f AT i EQUALS infinity  
                CONTINUE TO NEXT ITERATION  
            FOR j FROM 0 TO n MINUS 1  
                IF f AT i PLUS 1 LESS THAN f AT (i BITWISE OR p AT j)  
                    SET f AT (i BITWISE OR p AT j) TO f AT i PLUS 1  
                    SET g AT (i BITWISE OR p AT j) TO j  
                    SET h AT (i BITWISE OR p AT j) TO i  
        SET i TO (1 LEFT SHIFTED BY m) MINUS 1  
        SET ans TO EMPTY LIST  
        WHILE i NOT EQUAL TO 0  
            APPEND g AT i TO ans  
            SET i TO h AT i  
        RETURN ans  
    END FUNCTION  
END CLASS
