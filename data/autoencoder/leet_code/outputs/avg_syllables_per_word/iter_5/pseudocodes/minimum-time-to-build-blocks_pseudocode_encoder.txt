CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        TRANSFORM blocks INTO a min-heap  
        WHILE LENGTH OF blocks GREATER THAN 1  
            REMOVE and SET first TO the smallest element FROM blocks  
            REMOVE and SET second TO the next smallest element FROM blocks  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks  
        RETURN the only remaining element IN blocks  
    END FUNCTION  
END CLASS
