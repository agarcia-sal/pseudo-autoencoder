CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO list of zeros with length n PLUS 1  
        SET dp at index 1 TO 1  
        
        FOR i FROM 2 TO n  
            SET max_product TO 0  
            FOR j FROM 1 TO i MINUS 1  
                SET product_option_one TO j MULTIPLIED BY dp at index i MINUS j  
                SET product_option_two TO j MULTIPLIED BY i MINUS j  
                SET max_product TO maximum of max_product, product_option_one, and product_option_two  
            SET dp at index i TO max_product  
        
        RETURN dp at index n  
    END FUNCTION  
END CLASS
