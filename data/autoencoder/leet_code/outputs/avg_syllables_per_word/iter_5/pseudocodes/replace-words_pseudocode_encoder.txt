CLASS Solution  
    FUNCTION replaceWords(dictionary, sentence)  
        SORT dictionary BY length OF elements  
        SET words TO list OF words IN sentence  
        
        FUNCTION replace(word)  
            FOR each root IN dictionary  
                IF word STARTS WITH root THEN  
                    RETURN root  
            RETURN word  
        END FUNCTION  
        
        SET replaced_words TO empty list  
        FOR each word IN words  
            APPEND replace(word) TO replaced_words  
        SET result TO joined replaced_words WITH spaces  
        RETURN result  
    END FUNCTION  
END CLASS
