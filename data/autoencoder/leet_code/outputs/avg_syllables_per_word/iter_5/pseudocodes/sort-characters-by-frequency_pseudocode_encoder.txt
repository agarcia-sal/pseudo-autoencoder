CLASS Solution  
    FUNCTION frequencySort(s)  
        IMPORT Counter FROM collections  
        SET frequency TO Counter OF s  
        SET sorted_characters TO LIST OF keys IN frequency SORTED BY frequency VALUE IN DESCENDING ORDER  
        SET result TO EMPTY STRING  
        FOR each char IN sorted_characters  
            APPEND char REPEATED frequency OF char TIMES TO result  
        RETURN result  
    END FUNCTION  
END CLASS
