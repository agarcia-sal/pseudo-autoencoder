CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO LENGTH OF s  
        SET dp TO 2D ARRAY OF SIZE n BY n FILLED WITH 0  
        
        FOR i FROM 0 TO n MINUS 1  
            SET dp AT i AT i TO 1  
        
        FOR length FROM 2 TO n  
            FOR i FROM 0 TO n MINUS length  
                SET j TO i PLUS length MINUS 1  
                IF s AT i EQUALS s AT j  
                    SET dp AT i AT j TO dp AT i PLUS 1 AT j MINUS 1 PLUS 2  
                ELSE  
                    SET dp AT i AT j TO MAXIMUM OF dp AT i PLUS 1 AT j AND dp AT i AT j MINUS 1  
        
        RETURN dp AT 0 AT n MINUS 1  
    END FUNCTION  
END CLASS
