CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s_list)  
            SET stack TO empty list  
            SET sign TO plus symbol  
            SET num TO 0  
            
            WHILE LENGTH OF s_list GREATER THAN 0  
                SET char TO REMOVE FIRST ELEMENT FROM s_list  
                
                IF char IS a digit THEN  
                    SET num TO num MULTIPLIED BY 10 PLUS numerical value of char  
                
                IF char EQUALS open parenthesis THEN  
                    SET num TO CALL helper WITH s_list  
                
                IF char IS NOT a digit AND char IS NOT whitespace OR LENGTH OF s_list EQUALS 0 THEN  
                    IF sign EQUALS plus symbol THEN  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol THEN  
                        APPEND negative num TO stack  
                    ELSE IF sign EQUALS multiplication symbol THEN  
                        SET last element of stack TO last element MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol THEN  
                        SET last element of stack TO integer part of last element DIVIDED BY num  
                    
                    SET sign TO char  
                    SET num TO 0  
                
                IF char EQUALS close parenthesis THEN  
                    BREAK  
            
            RETURN sum of all elements in stack  
        END FUNCTION  
        
        RETURN CALL helper WITH list created from s  
    END FUNCTION  
END CLASS
