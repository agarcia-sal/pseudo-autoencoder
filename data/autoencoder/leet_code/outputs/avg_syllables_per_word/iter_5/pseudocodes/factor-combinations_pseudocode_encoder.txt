CLASS Solution  
    FUNCTION getFactors(n)  
        FUNCTION backtrack(start, target)  
            FOR i FROM start TO integer square root of target INCLUSIVE  
                IF target MODULO i EQUALS 0 THEN  
                    APPEND path CONCATENATED WITH i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        SET path TO empty list  
        CALL backtrack WITH 2 AND n  
        RETURN result  
    END FUNCTION  
END CLASS
