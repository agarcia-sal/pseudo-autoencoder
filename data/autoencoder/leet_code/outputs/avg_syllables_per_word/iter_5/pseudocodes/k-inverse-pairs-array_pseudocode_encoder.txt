CLASS Solution  
    FUNCTION kInversePairs(n, k)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS LIST OF LISTS WITH DIMENSIONS (n PLUS 1) BY (k PLUS 1) FILLED WITH 0  
        SET dp AT 0 AT 0 TO 1  
        
        FOR i FROM 1 TO n INCLUSIVE  
            SET prefix_sum TO 0  
            FOR j FROM 0 TO k INCLUSIVE  
                SET prefix_sum TO prefix_sum PLUS dp AT (i MINUS 1) AT j MODULO MOD  
                IF j IS GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS dp AT (i MINUS 1) AT (j MINUS i) PLUS MOD MODULO MOD  
                END IF  
                SET dp AT i AT j TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN dp AT n AT k  
    END FUNCTION  
END CLASS
