CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO LENGTH OF dist  
        INITIALIZE dp AS two dimensional list with n rows and n plus one columns FILLED WITH infinity  
        SET total_dist TO SUM OF dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN minus one  
        
        SET dp AT row zero column zero TO dist AT index zero  
        FOR each j FROM one UP TO n minus one  
            SET dp AT row zero column j TO dist AT index zero  
        
        FOR each i FROM one UP TO n minus one  
            FOR each j FROM zero UP TO i  
                IF j LESS THAN i  
                    SET dp AT row i column j TO CEILING OF (dp AT row i minus one column j DIVIDED BY speed) MULTIPLIED BY speed PLUS dist AT index i  
                IF j GREATER THAN zero  
                    SET dp AT row i column j TO MINIMUM OF dp AT row i column j AND dp AT row i minus one column j minus one PLUS dist AT index i  
        
        FOR each j FROM zero UP TO n minus one  
            IF dp AT row n minus one column j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
