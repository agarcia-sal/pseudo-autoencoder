CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of (0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')  
        SET m TO length of maze  
        SET n TO length of first element in maze  
        SET start TO tuple from ball  
        SET hole TO tuple from hole  
        SET pq TO list containing tuple (0, empty string, start)  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            EXTRACT dist, path, (x, y) FROM removing the smallest element of pq  
            
            IF (x, y) EQUALS hole  
                RETURN path  
            
            IF (x, y) IS IN visited  
                CONTINUE to next iteration  
            
            ADD (x, y) TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO 0  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO 0 AND LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO 0 AND LESS THAN n AND maze at position (new_x plus dx, new_y plus dy) IS equal to 0  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY 1  
                    
                    IF (new_x, new_y) EQUALS hole  
                        BREAK out of WHILE  
                
                IF (new_x, new_y) IS NOT IN visited  
                    INSERT tuple (dist plus count, path concatenated with direction, (new_x, new_y)) INTO pq maintaining smallest element order  
        
        RETURN string "impossible"  
    END FUNCTION  
END CLASS
