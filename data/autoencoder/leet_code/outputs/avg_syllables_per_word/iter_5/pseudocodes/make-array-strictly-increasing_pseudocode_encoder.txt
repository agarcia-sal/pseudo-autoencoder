CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2  
        SET dp TO dictionary with key -1 and value 0  
        
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            FOR each last_num, ops IN dp  
                IF num GREATER THAN last_num  
                    SET new_dp[num] TO minimum of new_dp at num or ops  
                END IF  
                
                SET idx TO position in arr2 where last_num would be inserted on the right  
                IF idx LESS THAN length of arr2  
                    SET new_dp at arr2 at idx TO minimum of new_dp at arr2 at idx or ops INCREMENTED BY 1  
                END IF  
            END FOR  
            
            IF new_dp is empty  
                RETURN negative 1  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN minimum value among all values in dp  
    END FUNCTION  
END CLASS
