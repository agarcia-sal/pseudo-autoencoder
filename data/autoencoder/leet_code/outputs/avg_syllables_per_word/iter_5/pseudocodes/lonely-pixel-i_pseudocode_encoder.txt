CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture IS EMPTY OR picture FIRST ELEMENT IS EMPTY THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF picture FIRST ELEMENT  
        SET row_count TO LIST OF SIZE m FILLED WITH 0  
        SET col_count TO LIST OF SIZE n FILLED WITH 0  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF picture ELEMENT AT i AND j EQUALS 'B' THEN  
                    INCREMENT row_count ELEMENT AT i BY 1  
                    INCREMENT col_count ELEMENT AT j BY 1  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixel_count TO 0  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF picture ELEMENT AT i AND j EQUALS 'B' AND row_count ELEMENT AT i EQUALS 1 AND col_count ELEMENT AT j EQUALS 1 THEN  
                    INCREMENT lonely_pixel_count BY 1  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
