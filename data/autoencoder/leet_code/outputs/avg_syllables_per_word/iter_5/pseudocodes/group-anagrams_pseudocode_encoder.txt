CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a default dictionary of lists  
        FOR each s IN strs  
            SET sorted_str TO the characters of s arranged in sorted order combined into a string  
            APPEND s TO the list in anagrams corresponding to sorted_str  
        RETURN the list of all values in anagrams  
    END FUNCTION  
END CLASS
