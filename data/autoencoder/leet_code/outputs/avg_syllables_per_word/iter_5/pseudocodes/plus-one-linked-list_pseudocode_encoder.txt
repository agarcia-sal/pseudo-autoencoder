CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO ListNode WITH values AT FIRST POSITION AND next AS None  
    SET p TO head  
    FOR each val IN values STARTING FROM SECOND ELEMENT  
        SET node TO ListNode WITH val AND next AS None  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne(head)  
        FUNCTION reverseList(node)  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO node.next  
                SET node.next TO prev  
                SET prev TO node  
                SET node TO next_node  
            RETURN prev  
        END FUNCTION  

        SET reversed_head TO reverseList(head)  
        SET carry TO 1  
        SET current TO reversed_head  

        WHILE current IS NOT None AND carry EQUALS 1  
            INCREMENT current.val BY carry  
            SET carry TO current.val DIVIDED BY 10  
            SET current.val TO current.val MODULO 10  

            IF current.next IS None AND carry EQUALS 1  
                SET current.next TO ListNode WITH val AS carry AND next AS None  
                SET carry TO 0  

            SET prev TO current  
            SET current TO current.next  

        RETURN reverseList(reversed_head)  
    END FUNCTION  
END CLASS
