CLASS Solution  
    FUNCTION maximalPathQuality(values, edges, maxTime)  
        SET graph TO empty mapping with default empty list  
        FOR each u, v, time IN edges  
            APPEND pair of v and time TO graph AT key u  
            APPEND pair of u and time TO graph AT key v  
        
        SET visited TO empty set  
        SET max_quality TO 0  
        
        FUNCTION dfs(node, current_time, current_quality)  
            IF node EQUALS 0 THEN  
                SET max_quality TO maximum of max_quality and current_quality  
            END IF  
            
            FOR each neighbor, time IN graph AT key node  
                IF time IS LESS THAN OR EQUAL TO current_time THEN  
                    IF neighbor NOT IN visited THEN  
                        ADD neighbor TO visited  
                        CALL dfs WITH neighbor, current_time MINUS time, current_quality PLUS values AT index neighbor  
                        REMOVE neighbor FROM visited  
                    ELSE  
                        CALL dfs WITH neighbor, current_time MINUS time, current_quality  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        ADD 0 TO visited  
        CALL dfs WITH 0, maxTime, values AT index 0  
        
        RETURN max_quality  
    END FUNCTION  
END CLASS
