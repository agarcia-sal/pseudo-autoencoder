CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS '+'  
                        APPEND l plus r TO results  
                    ELSE IF operator EQUALS '-'  
                        APPEND l minus r TO results  
                    ELSE IF operator EQUALS '*'  
                        APPEND l times r TO results  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr consists only of digits  
                RETURN list containing integer value of sub_expr  
            
            SET results TO empty list  
            FOR each index i FROM zero TO length of sub_expr MINUS one  
                IF character at sub_expr position i is one of plus, minus, or times operators  
                    SET left_results TO helper(sub_expr from start TO i minus one)  
                    SET right_results TO helper(sub_expr from i plus one TO end)  
                    APPEND all elements from compute(left_results, right_results, character at position i) TO results  
            RETURN results  
        END FUNCTION  
        
        RETURN helper(expression)  
    END FUNCTION  
END CLASS
