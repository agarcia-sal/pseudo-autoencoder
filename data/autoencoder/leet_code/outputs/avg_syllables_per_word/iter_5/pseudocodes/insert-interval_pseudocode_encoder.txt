CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals IS EMPTY  
            RETURN list containing newInterval  
        SET merged TO empty list  
        SET i TO 0  
        SET n TO length of intervals  
        WHILE i LESS THAN n AND end of intervals at i LESS THAN start of newInterval  
            APPEND intervals at i TO merged  
            INCREMENT i BY 1  
        WHILE i LESS THAN n AND start of intervals at i LESS THAN OR EQUAL TO end of newInterval  
            SET newInterval start TO minimum of newInterval start AND intervals at i start  
            SET newInterval end TO maximum of newInterval end AND intervals at i end  
            INCREMENT i BY 1  
        APPEND newInterval TO merged  
        WHILE i LESS THAN n  
            APPEND intervals at i TO merged  
            INCREMENT i BY 1  
        RETURN merged  
    END FUNCTION  
END CLASS
