CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF LENGTH OF s LESS THAN k THEN  
            RETURN 0  
        END IF  

        SET count TO COUNT FREQUENCY OF EACH CHARACTER IN s  

        FOR each char IN count  
            IF count OF char LESS THAN k THEN  
                RETURN MAXIMUM OF longestSubstring FOR EACH substring IN s SPLIT BY char  
            END IF  
        END FOR  

        RETURN LENGTH OF s  
    END FUNCTION  
END CLASS
