CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO LENGTH OF t THEN  
                APPEND concatenation OF t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS 1  
            IF t AT INDEX i IS ALPHABETIC THEN  
                TOGGLE CASE OF t AT INDEX i  
                CALL dfs WITH i PLUS 1  
            END IF  
        END FUNCTION  

        SET t TO LIST OF CHARACTERS IN s  
        SET ans TO EMPTY LIST  
        CALL dfs WITH 0  
        RETURN ans  
    END FUNCTION  
END CLASS
