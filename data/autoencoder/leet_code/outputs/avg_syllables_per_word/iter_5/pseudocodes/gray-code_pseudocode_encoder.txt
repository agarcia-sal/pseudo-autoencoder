CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS 0 THEN  
            RETURN list containing 0  
        END IF  
  
        SET prev_gray TO RESULT OF CALLING grayCode WITH n MINUS 1  
  
        SET mask TO 1 LEFT SHIFTED BY n MINUS 1  
  
        SET current_gray TO prev_gray CONCATENATED WITH list of mask OR num FOR each num IN REVERSED prev_gray  
  
        RETURN current_gray  
    END FUNCTION  
END CLASS
