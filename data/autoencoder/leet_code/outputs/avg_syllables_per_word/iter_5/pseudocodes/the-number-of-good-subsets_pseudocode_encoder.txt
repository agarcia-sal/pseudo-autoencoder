CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO frequency count of elements in nums  
        INITIALIZE dp AS list of zeros with length equal to two raised to the power of ten  
        SET dp at index zero TO 1  

        FOR each num IN keys of count  
            IF num EQUALS 1 THEN  
                SET dp at index zero TO dp at index zero MULTIPLIED BY two raised to the power of count at 1  
                SET dp at index zero TO dp at index zero MODULO MOD  
                CONTINUE to next iteration  
            END IF  

            IF num NOT IN prime_factors THEN  
                CONTINUE to next iteration  
            END IF  

            SET factors TO prime_factors at num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR with one shifted left by prime MINUS one  
            END FOR  

            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero THEN  
                    SET dp at index bitwise OR of m and mask TO dp at that index PLUS dp at m MULTIPLIED BY count at num  
                    SET dp at index bitwise OR of m and mask TO dp at that index MODULO MOD  
                END IF  
            END FOR  
        END FOR  

        SET result TO sum of dp from index one to end  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
