CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO dictionary with key 0 and value negative 1  
        SET current_sum TO 0  
        
        FOR each index i AND element num IN nums  
            INCREMENT current_sum BY num  
            
            IF k is not equal to 0 THEN  
                SET current_sum TO current_sum MODULO k  
            END IF  
            
            IF current_sum EXISTS IN remainder_map THEN  
                IF i MINUS remainder_map AT current_sum IS GREATER THAN 1 THEN  
                    RETURN True  
                END IF  
            ELSE  
                SET remainder_map AT current_sum TO i  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
