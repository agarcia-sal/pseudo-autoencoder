CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            RETURN k MULTIPLIED BY (k PLUS 1) MULTIPLIED BY (k PLUS 2) DIVIDED BY 6  
        END FUNCTION  

        SET left TO 0  
        SET right TO n  

        WHILE left LESS THAN right  
            SET mid TO (left PLUS right PLUS 1) DIVIDED BY 2  
            IF triangular_pyramid(mid) LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS 1  
            END IF  
        END WHILE  

        SET boxes_used TO triangular_pyramid(left)  
        SET remaining TO n MINUS boxes_used  
        SET floor_boxes TO left MULTIPLIED BY (left PLUS 1) DIVIDED BY 2  

        SET additional_boxes TO 0  
        WHILE remaining GREATER THAN 0  
            INCREMENT additional_boxes BY 1  
            DECREMENT remaining BY additional_boxes  
        END WHILE  

        RETURN floor_boxes PLUS additional_boxes  
    END FUNCTION  
END CLASS
