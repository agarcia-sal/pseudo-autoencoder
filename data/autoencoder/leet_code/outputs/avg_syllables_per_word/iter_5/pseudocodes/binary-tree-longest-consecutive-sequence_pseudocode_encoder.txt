CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    INITIALIZE queue AS EMPTY QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO FIRST ELEMENT FROM queue REMOVED  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root IS None  
            RETURN 0  
        
        FUNCTION dfs(node, parent, current_length)  
            IF node IS None  
                RETURN current_length  
            
            IF parent IS NOT None AND node.val IS EQUAL TO parent PLUS 1  
                INCREMENT current_length BY 1  
            ELSE  
                SET current_length TO 1  
            
            SET left_length TO dfs(node.left, node.val, current_length)  
            SET right_length TO dfs(node.right, node.val, current_length)  
            
            RETURN THE MAXIMUM OF current_length, left_length, right_length  
        END FUNCTION  
        
        RETURN dfs(root, None, 0)  
    END FUNCTION  
END CLASS
