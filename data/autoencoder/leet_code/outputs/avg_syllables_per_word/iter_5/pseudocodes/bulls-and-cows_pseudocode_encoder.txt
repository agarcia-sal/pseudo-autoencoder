CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO 0  
        SET cows TO 0  

        SET secret_count TO new mapping with default integer zero  
        SET guess_count TO new mapping with default integer zero  

        FOR each pair (s, g) IN secret and guess zipped  
            IF s EQUALS g THEN  
                INCREMENT bulls BY 1  
            ELSE  
                INCREMENT secret_count AT s BY 1  
                INCREMENT guess_count AT g BY 1  
        
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY minimum of secret_count AT digit AND guess_count AT digit  

        RETURN string composed of bulls FOLLOWED BY "A" FOLLOWED BY cows FOLLOWED BY "B"  
    END FUNCTION  
END CLASS
