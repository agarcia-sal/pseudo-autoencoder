CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO new TreeNode with values at index 0  
    SET i TO 1  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO new TreeNode with values at index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO new TreeNode with values at index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(root, k)  
        IF root.left IS None AND root.right IS None  
            RETURN root.val  
        
        SET graph TO a default dictionary with lists as default values  
        
        FUNCTION dfs(node, parent)  
            IF parent IS NOT None  
                APPEND parent.val TO graph at node.val  
                APPEND node.val TO graph at parent.val  
            IF node.left IS NOT None  
                CALL dfs(node.left, node)  
            IF node.right IS NOT None  
                CALL dfs(node.right, node)  
        END FUNCTION  
        
        CALL dfs(root, None)  
        
        SET queue TO double-ended queue initialized with k  
        SET visited TO empty set  
        
        WHILE queue IS NOT EMPTY  
            SET current TO REMOVE first element FROM queue  
            ADD current TO visited  
            
            IF LENGTH OF graph at current IS EQUAL TO 1 AND current IS NOT EQUAL TO root.val  
                RETURN current  
            
            FOR each neighbor IN graph at current  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
        RETURN -1  
    END FUNCTION  
END CLASS
