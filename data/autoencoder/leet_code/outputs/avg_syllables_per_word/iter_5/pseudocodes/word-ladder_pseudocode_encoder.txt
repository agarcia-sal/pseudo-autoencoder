CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN 0  
        
        SET word_dict TO dictionary with default empty list  
        SET word_length TO length of beginWord  
        
        FOR each word IN wordList DO  
            FOR i FROM 0 TO word_length MINUS 1 DO  
                SET intermediate_key TO concatenation of substring of word from start to i, "*" and substring of word from i+1 to end  
                APPEND word TO word_dict at intermediate_key  
        
        SET queue TO a deque initialized with tuple of beginWord and 1  
        SET visited TO set containing beginWord  
        
        WHILE queue is not empty DO  
            SET current_word, current_length TO front element removed from queue  
            
            FOR i FROM 0 TO word_length MINUS 1 DO  
                SET intermediate_word TO concatenation of substring of current_word from start to i, "*" and substring of current_word from i+1 to end  
                
                FOR each word IN word_dict at intermediate_word DO  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS 1  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND tuple of word and current_length PLUS 1 TO queue  
        
        RETURN 0  
    END FUNCTION  
END CLASS
