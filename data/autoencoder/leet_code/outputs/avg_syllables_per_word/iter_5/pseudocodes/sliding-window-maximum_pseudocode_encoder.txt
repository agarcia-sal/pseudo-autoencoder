CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums IS EMPTY OR k EQUALS 0 THEN  
            RETURN EMPTY LIST  
        END IF  
        
        INITIALIZE deq AS EMPTY DEQUE  
        INITIALIZE result AS EMPTY LIST  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1 DO  
            IF deq IS NOT EMPTY AND deq FRONT IS LESS THAN i MINUS k PLUS 1 THEN  
                REMOVE FRONT ELEMENT FROM deq  
            END IF  
            
            WHILE deq IS NOT EMPTY AND VALUE AT nums AT deq BACK IS LESS THAN nums AT i DO  
                REMOVE BACK ELEMENT FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS 1 THEN  
                APPEND nums AT deq FRONT TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
