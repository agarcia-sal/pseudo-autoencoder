CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the frequency mapping of elements in nums  
        SET unique_nums TO the sorted list of keys from count  
        
        SET dp TO a list of zeros with length equal to the length of unique_nums plus one  
        
        SET dp at position 1 TO unique_nums at position 0 multiplied by count value for unique_nums at position 0  
        
        FOR i FROM 2 TO length of unique_nums plus 1  
            SET num TO unique_nums at position i minus 1  
            IF num IS EQUAL TO unique_nums at position i minus 2 plus 1 THEN  
                SET dp at position i TO the maximum value between dp at position i minus 1 AND dp at position i minus 2 plus num multiplied by count value for num  
            ELSE  
                SET dp at position i TO dp at position i minus 1 plus num multiplied by count value for num  
        RETURN dp at last position  
    END FUNCTION  
END CLASS
