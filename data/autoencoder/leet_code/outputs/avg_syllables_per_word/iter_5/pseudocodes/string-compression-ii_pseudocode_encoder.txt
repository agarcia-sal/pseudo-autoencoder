CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(s, k)  
        FUNCTION dp(i, k, prev_char, prev_count)  
            IF k LESS THAN 0 THEN  
                RETURN infinity  
            END IF  
            IF i EQUALS LENGTH OF s THEN  
                RETURN 0  
            END IF  
            IF CHARACTER AT s INDEX i EQUALS prev_char THEN  
                SET increment TO 1 IF prev_count EQUALS 1 OR 9 OR 99 OTHERWISE 0  
                RETURN increment PLUS dp(i PLUS 1, k, prev_char, prev_count PLUS 1)  
            ELSE  
                SET keep TO 1 PLUS dp(i PLUS 1, k, CHARACTER AT s INDEX i, 1)  
                SET delete TO dp(i PLUS 1, k MINUS 1, prev_char, prev_count)  
                RETURN MINIMUM OF keep AND delete  
            END IF  
        END FUNCTION  
        RETURN dp(0, k, EMPTY STRING, 0)  
    END FUNCTION  
END CLASS
