CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS empty  
            RETURN False  
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF board AT index 0  
        INITIALIZE visited AS 2D list of False with dimensions m BY n  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS LENGTH OF word  
                RETURN True  
            IF x LESS THAN 0 OR x GREATER THAN OR EQUAL TO m OR y LESS THAN 0 OR y GREATER THAN OR EQUAL TO n OR visited AT x AND y IS True OR board AT x AND y NOT EQUAL TO word AT index  
                RETURN False  
            SET visited AT x AND y TO True  
            FOR each pair (dx, dy) IN list OF (-1, 0), (1, 0), (0, -1), (0, 1)  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS 1) IS True  
                    RETURN True  
            SET visited AT x AND y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF dfs(i, j, 0) IS True  
                    RETURN True  
        RETURN False  
    END FUNCTION  
END CLASS
