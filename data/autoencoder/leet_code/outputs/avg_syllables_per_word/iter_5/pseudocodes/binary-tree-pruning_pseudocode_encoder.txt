CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT 0, left WITH DEFAULT None, right WITH DEFAULT None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode WITH val AS values AT position 0  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from front of queue  
        IF i IS LESS THAN length of values AND values AT position i IS NOT None  
            SET node.left TO new TreeNode WITH val AS values AT position i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values AT position i IS NOT None  
            SET node.right TO new TreeNode WITH val AS values AT position i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self, root)  
        IF root IS None  
            RETURN None  
        SET root.left TO self.pruneTree(root.left)  
        SET root.right TO self.pruneTree(root.right)  
        IF root.val IS EQUAL TO 0 AND root.left IS None AND root.right IS None  
            RETURN None  
        RETURN root  
    END FUNCTION  
END CLASS
