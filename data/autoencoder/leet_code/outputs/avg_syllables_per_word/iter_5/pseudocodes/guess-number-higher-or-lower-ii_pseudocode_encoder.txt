CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right) WITH CACHE  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN 0  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right INCLUSIVE  
                SET cost TO pivot PLUS MAXIMUM OF dp(left, pivot MINUS 1) AND dp(pivot PLUS 1, right)  
                SET min_cost TO MINIMUM OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp(1, n)  
    END FUNCTION  
END CLASS
