CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF LENGTH OF s1 PLUS LENGTH OF s2 NOT EQUAL TO LENGTH OF s3 THEN  
            RETURN False  
        END IF  

        INITIALIZE dp AS LIST OF False WITH LENGTH OF s2 PLUS 1  
        SET dp AT 0 TO True  

        FOR i FROM 0 TO LENGTH OF s1 INCLUSIVE  
            FOR j FROM 0 TO LENGTH OF s2 INCLUSIVE  
                IF i GREATER THAN 0 THEN  
                    SET dp AT j TO dp AT j AND CHARACTER AT s1 AT i MINUS 1 EQUALS CHARACTER AT s3 AT i PLUS j MINUS 1  
                END IF  
                IF j GREATER THAN 0 THEN  
                    SET dp AT j TO dp AT j OR (dp AT j MINUS 1 AND CHARACTER AT s2 AT j MINUS 1 EQUALS CHARACTER AT s3 AT i PLUS j MINUS 1)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp AT LENGTH OF s2  
    END FUNCTION  
END CLASS
