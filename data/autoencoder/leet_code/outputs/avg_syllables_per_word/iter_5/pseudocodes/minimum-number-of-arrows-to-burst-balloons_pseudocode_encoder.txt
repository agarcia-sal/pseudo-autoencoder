CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BASED ON each point's second element  
        SET arrows TO 0  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF balloon's first element GREATER THAN last_arrow_position  
                INCREMENT arrows BY 1  
                SET last_arrow_position TO balloon's second element  
        RETURN arrows  
    END FUNCTION  
END CLASS
