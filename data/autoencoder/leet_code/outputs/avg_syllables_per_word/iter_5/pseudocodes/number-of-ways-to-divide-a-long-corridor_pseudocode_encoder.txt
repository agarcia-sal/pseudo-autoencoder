CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO ten to the power of nine plus seven  
        SET seats TO list of indices where character in corridor is equal to seat symbol  
        
        IF length of seats modulo two is not equal to zero OR length of seats is less than two THEN  
            RETURN zero  
        END IF  
        
        SET ways TO one  
        FOR index FROM two TO length of seats STEP two  
            SET ways TO ways multiplied by difference between seats at current index and seats at previous index  
            SET ways TO ways modulo MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
