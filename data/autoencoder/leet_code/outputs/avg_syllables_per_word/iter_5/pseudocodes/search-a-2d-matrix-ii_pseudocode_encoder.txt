CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EMPTY OR matrix AT INDEX 0 IS EMPTY  
            RETURN False  
        END IF  
  
        SET row TO 0  
        SET col TO LENGTH OF matrix AT INDEX 0 MINUS 1  
  
        WHILE row IS LESS THAN LENGTH OF matrix AND col IS GREATER THAN OR EQUAL TO 0  
            IF matrix AT INDEX row AT INDEX col IS EQUAL TO target  
                RETURN True  
            ELSE IF matrix AT INDEX row AT INDEX col IS GREATER THAN target  
                DECREMENT col BY 1  
            ELSE  
                INCREMENT row BY 1  
            END IF  
        END WHILE  
  
        RETURN False  
    END FUNCTION  
END CLASS
