CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two-dimensional list of size (LENGTH OF s plus 1) by (LENGTH OF p plus 1) filled with False  
        SET dp at position 0,0 TO True  

        FOR j FROM 1 TO LENGTH OF p DO  
            IF character at position j minus 1 in p EQUALS '*' THEN  
                SET dp at position 0,j TO dp at position 0,j minus 2  
            END IF  
        END FOR  

        FOR i FROM 1 TO LENGTH OF s DO  
            FOR j FROM 1 TO LENGTH OF p DO  
                IF character at position j minus 1 in p EQUALS character at position i minus 1 in s OR character at position j minus 1 in p EQUALS '.' THEN  
                    SET dp at position i,j TO dp at position i minus 1,j minus 1  
                ELSE IF character at position j minus 1 in p EQUALS '*' THEN  
                    SET dp at position i,j TO dp at position i,j minus 2  
                    IF character at position j minus 2 in p EQUALS character at position i minus 1 in s OR character at position j minus 2 in p EQUALS '.' THEN  
                        SET dp at position i,j TO dp at position i,j OR dp at position i minus 1,j  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at position LENGTH OF s,LENGTH OF p  
    END FUNCTION  
END CLASS
