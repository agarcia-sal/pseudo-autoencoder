CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing 0 and firstPerson  
        SORT meetings BY their time attribute  
        SET i TO 0  
        WHILE i LESS THAN length of meetings  
            SET current_time TO the time of meetings at index i  
            SET graph TO an empty mapping from person to list of connected people  
            WHILE i LESS THAN length of meetings AND the time of meetings at index i EQUALS current_time  
                SET x, y, _ TO the meeting at index i  
                APPEND y TO graph at key x  
                APPEND x TO graph at key y  
                INCREMENT i BY 1  
            END WHILE  
            SET queue TO a double-ended queue with people from know_secret who are in graph keys  
            SET visited TO an empty set  
            WHILE queue is not empty  
                REMOVE the first person FROM queue and SET TO person  
                IF person is in visited  
                    CONTINUE to next iteration  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN graph at person  
                    IF neighbor NOT IN visited  
                        APPEND neighbor TO queue  
            END WHILE  
        END WHILE  
        RETURN list form of know_secret  
    END FUNCTION  
END CLASS
