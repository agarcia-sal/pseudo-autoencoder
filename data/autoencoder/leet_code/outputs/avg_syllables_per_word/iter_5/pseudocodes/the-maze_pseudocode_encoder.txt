CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO list of (0, 1), (1, 0), (0, -1), (-1, 0)  
        SET m TO length of maze  
        SET n TO length of maze at index 0  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF (x, y) is in visited THEN  
                RETURN False  
            IF list consisting of x and y equals destination THEN  
                RETURN True  
            ADD (x, y) to visited  
            
            FOR each (dx, dy) IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx plus dx GREATER THAN OR EQUAL TO 0 AND nx plus dx LESS THAN m AND ny plus dy GREATER THAN OR EQUAL TO 0 AND ny plus dy LESS THAN n AND maze at position (nx plus dx, ny plus dy) EQUALS 0  
                    SET nx TO nx plus dx  
                    SET ny TO ny plus dy  
                IF dfs(nx, ny) THEN  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs(value at start index 0, value at start index 1)  
    END FUNCTION  
END CLASS
