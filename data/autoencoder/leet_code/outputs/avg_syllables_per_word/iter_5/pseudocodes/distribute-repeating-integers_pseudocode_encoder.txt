CLASS Solution  
    FUNCTION canDistribute(nums, quantity)  
        SET count TO list of frequencies of elements in nums  
        SORT quantity in descending order  
        
        FUNCTION can_satisfy(index)  
            IF index EQUALS LENGTH OF quantity THEN  
                RETURN True  
            END IF  
            FOR i FROM 0 TO LENGTH OF count MINUS 1  
                IF count AT i IS GREATER THAN OR EQUAL TO quantity AT index THEN  
                    DECREMENT count AT i BY quantity AT index  
                    IF can_satisfy(index PLUS 1) THEN  
                        RETURN True  
                    END IF  
                    INCREMENT count AT i BY quantity AT index  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN can_satisfy(0)  
    END FUNCTION  
END CLASS
