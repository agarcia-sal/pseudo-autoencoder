CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary mapping each character from 'a' to 'z' TO 0  
        SET current_length TO 0  
        FOR index FROM 0 TO length of s MINUS 1  
            IF index GREATER THAN 0 AND (ASCII value of s at index MINUS ASCII value of s at index MINUS 1 EQUALS 1 OR (s at index EQUALS 'a' AND s at index MINUS 1 EQUALS 'z'))  
                INCREMENT current_length BY 1  
            ELSE  
                SET current_length TO 1  
            SET max_length at s at index TO the greater value BETWEEN max_length at s at index AND current_length  
        SET total TO the sum of all values in max_length  
        RETURN total  
    END FUNCTION  
END CLASS
