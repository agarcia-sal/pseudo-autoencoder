CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of False values with length LENGTH OF nums  
        SET max_length TO 0  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            IF visited AT i IS False THEN  
                SET current_length TO 0  
                SET k TO i  
                WHILE visited AT k IS False  
                    SET visited AT k TO True  
                    SET k TO nums AT k  
                    INCREMENT current_length BY 1  
                END WHILE  
                IF current_length GREATER THAN max_length THEN  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
