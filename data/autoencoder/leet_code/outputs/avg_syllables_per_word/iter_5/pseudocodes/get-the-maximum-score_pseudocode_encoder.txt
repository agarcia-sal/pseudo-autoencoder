CLASS Solution  
    FUNCTION maxSum(nums1, nums2)  
        SET i TO 0  
        SET j TO 0  
        SET sum1 TO 0  
        SET sum2 TO 0  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        
        WHILE i LESS THAN LENGTH OF nums1 AND j LESS THAN LENGTH OF nums2  
            IF nums1 AT i LESS THAN nums2 AT j  
                INCREMENT sum1 BY nums1 AT i  
                INCREMENT i BY 1  
            ELSE IF nums1 AT i GREATER THAN nums2 AT j  
                INCREMENT sum2 BY nums2 AT j  
                INCREMENT j BY 1  
            ELSE  
                SET sum1 TO MAXIMUM OF sum1 AND sum2 PLUS nums1 AT i  
                SET sum2 TO sum1  
                INCREMENT i BY 1  
                INCREMENT j BY 1  
        
        WHILE i LESS THAN LENGTH OF nums1  
            INCREMENT sum1 BY nums1 AT i  
            INCREMENT i BY 1  
        
        WHILE j LESS THAN LENGTH OF nums2  
            INCREMENT sum2 BY nums2 AT j  
            INCREMENT j BY 1  
        
        RETURN MAXIMUM OF sum1 AND sum2 MODULO MOD  
    END FUNCTION  
END CLASS
