CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY starting point OF each interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR ending point OF last item IN merged IS LESS THAN starting point OF interval  
                APPEND interval TO merged  
            ELSE  
                SET ending point OF last item IN merged TO maximum OF ending point OF last item IN merged AND ending point OF interval  
        RETURN merged  
    END FUNCTION  
END CLASS
