CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        CREATE graph as a mapping from nodes to list of nodes  
        CREATE indegree as a list of zeros with length numCourses  
        
        FOR each pair course, prereq IN prerequisites  
            APPEND course TO graph at key prereq  
            INCREMENT indegree at index course BY 1  
        
        CREATE queue as a collection with all courses WHERE indegree equals 0  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND ASSIGN TO current_course  
            DECREMENT numCourses BY 1  
            
            FOR each next_course IN graph at key current_course  
                DECREMENT indegree at index next_course BY 1  
                IF indegree at index next_course EQUALS 0  
                    ADD next_course TO queue  
        
        RETURN whether numCourses EQUALS 0  
    END FUNCTION  
END CLASS
