CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        SET dp TO 2D array of size m by n FILLED WITH 1  
        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                SET dp at position i, j TO dp at position i MINUS 1, j PLUS dp at position i, j MINUS 1  
        RETURN value at dp position m MINUS 1, n MINUS 1  
    END FUNCTION  
END CLASS
