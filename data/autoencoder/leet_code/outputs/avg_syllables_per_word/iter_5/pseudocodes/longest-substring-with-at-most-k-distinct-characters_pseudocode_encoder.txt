CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS 0  
            RETURN 0  
        END IF  

        SET char_map TO empty map  
        SET left TO 0  
        SET max_length TO 0  

        FOR right FROM 0 TO LENGTH OF s MINUS 1  
            SET char_map AT s[right] TO right  

            IF SIZE OF char_map GREATER THAN k  
                SET leftmost_char TO character IN char_map WITH MINIMUM associated value  
                SET left TO char_map POP leftmost_char PLUS 1  
            END IF  

            SET max_length TO GREATER OF max_length AND (right MINUS left PLUS 1)  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
