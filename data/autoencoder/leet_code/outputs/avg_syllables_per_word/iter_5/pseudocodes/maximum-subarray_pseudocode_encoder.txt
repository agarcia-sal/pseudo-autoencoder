CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO first element of nums  
        SET current_sum TO first element of nums  
        FOR each num IN nums starting from the second element  
            SET current_sum TO the greater value BETWEEN num AND current_sum plus num  
            SET max_sum TO the greater value BETWEEN max_sum AND current_sum  
        RETURN max_sum  
    END FUNCTION  
END CLASS
