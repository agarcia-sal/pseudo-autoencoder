CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET list_of_pairs TO list of pairs from position and speed sorted in descending order by position  
        SET time_to_target TO empty list  
        FOR each pair p, s IN list_of_pairs  
            SET time TO (target MINUS p) DIVIDED BY s  
            APPEND time TO time_to_target  
        SET fleets TO 0  
        SET current_max_time TO 0  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY 1  
                SET current_max_time TO time  
        RETURN fleets  
    END FUNCTION  
END CLASS
