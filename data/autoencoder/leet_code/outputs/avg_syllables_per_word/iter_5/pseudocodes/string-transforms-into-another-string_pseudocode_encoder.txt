CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2 THEN  
            RETURN True  
        END IF  
        
        SET conversion_map TO empty mapping  
        
        FOR each char1, char2 IN pairs from str1 and str2  
            IF char1 EXISTS IN conversion_map THEN  
                IF conversion_map AT char1 DOES NOT EQUAL char2 THEN  
                    RETURN False  
                END IF  
            ELSE  
                SET conversion_map AT char1 TO char2  
            END IF  
        END FOR  
        
        IF SIZE OF unique characters IN str2 EQUALS 26 THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
