CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s is empty OR first character of s EQUALS '0'  
            RETURN 0  
        END IF  
        
        SET dp TO a list of zeros with length LENGTH OF s PLUS 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 1  
        
        FOR i FROM 2 TO LENGTH OF s PLUS 1  
            IF character at position i MINUS 1 in s NOT EQUAL TO '0'  
                INCREMENT dp at index i BY dp at index i MINUS 1  
            END IF  
            
            SET two_digit TO integer value of substring of s from position i MINUS 2 TO i  
            IF two_digit GREATER THAN OR EQUAL TO 10 AND two_digit LESS THAN OR EQUAL TO 26  
                INCREMENT dp at index i BY dp at index i MINUS 2  
            END IF  
        END FOR  
        
        RETURN last element in dp  
    END FUNCTION  
END CLASS
