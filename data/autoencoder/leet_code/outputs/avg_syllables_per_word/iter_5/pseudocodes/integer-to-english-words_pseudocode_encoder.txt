CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS 0  
            RETURN "Zero"  
        END IF  

        SET below_twenty TO LIST OF STRINGS ["", "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten",  
                                           "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"]  
        SET tens TO LIST OF STRINGS ["", "", "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"]  
        SET thousands TO LIST OF STRINGS ["", "Thousand", "Million", "Billion"]  

        FUNCTION helper(n)  
            IF n EQUALS 0  
                RETURN EMPTY STRING  
            ELSE IF n LESS THAN 20  
                RETURN below_twenty AT INDEX n CONCATENATED WITH A SPACE  
            ELSE IF n LESS THAN 100  
                RETURN tens AT INDEX (n DIVIDED BY 10) CONCATENATED WITH A SPACE CONCATENATED WITH helper(n MODULO 10)  
            ELSE  
                RETURN below_twenty AT INDEX (n DIVIDED BY 100) CONCATENATED WITH " Hundred " CONCATENATED WITH helper(n MODULO 100)  
            END IF  
        END FUNCTION  

        SET result TO EMPTY STRING  
        SET i TO 0  

        WHILE num GREATER THAN 0  
            IF num MODULO 1000 NOT EQUAL TO 0  
                SET result TO helper(num MODULO 1000) CONCATENATED WITH thousands AT INDEX i CONCATENATED WITH A SPACE CONCATENATED WITH result  
            END IF  
            SET num TO num DIVIDED BY 1000  
            INCREMENT i BY 1  
        END WHILE  

        RETURN result WITH LEADING AND TRAILING WHITESPACE REMOVED  
    END FUNCTION  
END CLASS
