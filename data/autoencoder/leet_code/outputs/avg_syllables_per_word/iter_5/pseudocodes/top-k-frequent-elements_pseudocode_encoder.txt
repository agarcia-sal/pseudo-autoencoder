CLASS Solution  
    FUNCTION topKFrequent(nums, k)  
        SET frequency_map TO map with default integer zero  
        FOR each num IN nums  
            INCREMENT frequency_map[num] BY 1  
        SET max_frequency TO maximum value IN frequency_map  
        SET buckets TO list of empty lists with length max_frequency PLUS 1  
        FOR each num, freq IN frequency_map items  
            APPEND num TO buckets at position freq  
        SET result TO empty list  
        FOR freq FROM max_frequency DOWN TO 1  
            IF length of buckets at position freq IS GREATER THAN 0 THEN  
                EXTEND result BY buckets at position freq  
                IF length of result IS GREATER THAN OR EQUAL TO k THEN  
                    RETURN first k elements of result  
        RETURN result  
    END FUNCTION  
END CLASS
