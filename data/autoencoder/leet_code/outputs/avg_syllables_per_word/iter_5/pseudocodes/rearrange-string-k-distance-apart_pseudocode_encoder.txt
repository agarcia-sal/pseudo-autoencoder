CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO 1 THEN  
            RETURN s  
        END IF  
          
        SET freq TO frequency count of characters in s  
          
        SET max_heap TO list of pairs (negative count, character) for each character and count in freq  
        TRANSFORM max_heap INTO a heap structure  
          
        SET wait_queue TO empty list  
          
        SET result TO empty list  
          
        WHILE max_heap IS NOT empty DO  
            EXTRACT the element with highest priority FROM max_heap INTO count, char  
            APPEND char TO result  
              
            APPEND pair (count incremented BY 1, char) TO wait_queue  
              
            IF length of wait_queue EQUALS k THEN  
                REMOVE the first element FROM wait_queue INTO count, char  
                IF count LESS THAN 0 THEN  
                    ADD pair (count, char) BACK INTO max_heap  
                END IF  
            END IF  
        END WHILE  
          
        IF length of result NOT EQUAL TO length of s THEN  
            RETURN empty string  
        END IF  
          
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
