CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO dictionary WITH key 0 MAPPED TO value -1  
        SET cumulative_sum TO 0  
        SET max_length TO 0  
        
        FOR each index i AND element num IN nums  
            INCREMENT cumulative_sum BY num  
            
            IF cumulative_sum MINUS k EXISTS AS key IN cumulative_sum_index  
                SET max_length TO the GREATER OF max_length AND i MINUS cumulative_sum_index VALUE FOR key cumulative_sum MINUS k  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST AS key IN cumulative_sum_index  
                SET cumulative_sum_index VALUE FOR key cumulative_sum TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
