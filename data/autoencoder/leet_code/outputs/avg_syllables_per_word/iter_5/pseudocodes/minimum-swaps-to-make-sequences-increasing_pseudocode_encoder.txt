CLASS Solution  
    FUNCTION minSwap(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        INITIALIZE dp AS LIST OF n ELEMENTS EACH CONTAINING TWO ZEROS  
        
        SET dp AT INDEX 0 AND POSITION 0 TO 0  
        SET dp AT INDEX 0 AND POSITION 1 TO 1  
        
        FOR i FROM 1 TO n MINUS 1  
            IF nums1 AT i IS GREATER THAN nums1 AT i MINUS 1 AND nums2 AT i IS GREATER THAN nums2 AT i MINUS 1 THEN  
                IF nums1 AT i IS GREATER THAN nums2 AT i MINUS 1 AND nums2 AT i IS GREATER THAN nums1 AT i MINUS 1 THEN  
                    SET dp AT i AND 0 TO MINIMUM OF dp AT i MINUS 1 AND 0 AND dp AT i MINUS 1 AND 1  
                    SET dp AT i AND 1 TO dp AT i AND 0 PLUS 1  
                ELSE  
                    SET dp AT i AND 0 TO dp AT i MINUS 1 AND 0  
                    SET dp AT i AND 1 TO dp AT i MINUS 1 AND 1 PLUS 1  
            ELSE  
                IF nums1 AT i IS GREATER THAN nums2 AT i MINUS 1 AND nums2 AT i IS GREATER THAN nums1 AT i MINUS 1 THEN  
                    SET dp AT i AND 0 TO dp AT i MINUS 1 AND 1  
                    SET dp AT i AND 1 TO dp AT i MINUS 1 AND 0 PLUS 1  
                ELSE  
                    SET dp AT i AND 0 TO dp AT i MINUS 1 AND 1  
                    SET dp AT i AND 1 TO dp AT i MINUS 1 AND 0 PLUS 1  
        
        RETURN MINIMUM OF dp AT n MINUS 1 AND 0 AND dp AT n MINUS 1 AND 1  
    END FUNCTION  
END CLASS
