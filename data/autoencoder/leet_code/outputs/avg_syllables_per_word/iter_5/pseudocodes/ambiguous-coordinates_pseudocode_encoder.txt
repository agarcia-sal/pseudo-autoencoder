CLASS Solution
    FUNCTION ambiguousCoordinates(s)
        FUNCTION make(frag)
            SET n TO LENGTH OF frag
            FOR d FROM 1 TO n
                SET left TO SUBSTRING OF frag FROM START TO d
                SET right TO SUBSTRING OF frag FROM d TO END
                IF (left DOES NOT START WITH '0' OR left IS EQUAL TO '0') AND (right DOES NOT END WITH '0')
                    YIELD left CONCATENATED WITH ('.' IF d IS NOT EQUAL TO n ELSE EMPTY STRING) CONCATENATED WITH right
                END IF
            END FOR
        END FUNCTION

        SET s TO SUBSTRING OF s FROM SECOND CHARACTER TO SECOND LAST CHARACTER
        RETURN LIST OF STRING FOR EACH i FROM 1 TO LENGTH OF s MINUS 1 WHERE 
            FOR EACH a IN make(SUBSTRING OF s FROM START TO i)
                FOR EACH b IN make(SUBSTRING OF s FROM i TO END)
                    YIELD STRING CONCATENATED FROM '(', a, ', ', b, ')'
                END FOR
            END FOR
    END FUNCTION
END CLASS
