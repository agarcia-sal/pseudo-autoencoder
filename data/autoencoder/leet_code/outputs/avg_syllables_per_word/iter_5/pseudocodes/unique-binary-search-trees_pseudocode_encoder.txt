CLASS Solution  
    FUNCTION numTrees(n)  
        SET dp TO list of zeroes with length n PLUS 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 1  
        FOR i FROM 2 TO n  
            FOR j FROM 1 TO i  
                INCREMENT dp at index i BY dp at index j MINUS 1 MULTIPLIED BY dp at index i MINUS j  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
