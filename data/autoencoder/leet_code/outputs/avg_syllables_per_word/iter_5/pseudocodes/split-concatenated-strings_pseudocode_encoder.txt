CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            SET reversed_str TO the reverse of strs AT i  
            IF reversed_str GREATER THAN strs AT i  
                SET strs AT i TO reversed_str  
        
        SET doubled_strs TO concatenation of strs AND strs  
        
        SET max_string TO empty string  
        
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            FOR each is_reversed IN [False, True]  
                IF is_reversed IS True  
                    SET current TO the reverse of strs AT i  
                ELSE  
                    SET current TO strs AT i  
                
                SET remaining TO concatenation of elements in doubled_strs FROM i PLUS 1 TO i PLUS LENGTH OF strs MINUS 1  
                
                FOR j FROM 0 TO LENGTH OF current  
                    SET candidate TO concatenation of substring of current FROM j TO end, remaining, and substring of current FROM start TO j  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
