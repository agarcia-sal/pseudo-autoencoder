CLASS Solution  
    FUNCTION splitArray(nums)  
        SET n TO LENGTH OF nums  
        IF n LESS THAN 7  
            RETURN False  
        
        SET prefix_sum TO list of zeros with length n  
        SET prefix_sum AT 0 TO nums AT 0  
        FOR i FROM 1 TO n MINUS 1  
            SET prefix_sum AT i TO prefix_sum AT (i MINUS 1) PLUS nums AT i  
        
        FOR j FROM 3 TO n MINUS 3  
            SET seen TO empty set  
            FOR i FROM 1 TO j MINUS 1  
                IF prefix_sum AT (i MINUS 1) EQUALS prefix_sum AT (j MINUS 1) MINUS prefix_sum AT i  
                    ADD prefix_sum AT (i MINUS 1) TO seen  
            FOR k FROM j PLUS 2 TO n MINUS 1  
                IF prefix_sum AT (n MINUS 1) MINUS prefix_sum AT k EQUALS prefix_sum AT (k MINUS 1) MINUS prefix_sum AT j  
                   AND prefix_sum AT (k MINUS 1) MINUS prefix_sum AT j IS IN seen  
                    RETURN True  
        
        RETURN False  
    END FUNCTION  
END CLASS
