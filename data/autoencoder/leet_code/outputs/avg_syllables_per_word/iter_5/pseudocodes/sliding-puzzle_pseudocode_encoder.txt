CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples created from each row in board  
        SET target TO tuple containing (1, 2, 3) and (4, 5, 0)  
        
        SET neighbors TO mapping of positions TO their possible zero tile moves:  
            0 maps TO set containing 1 and 3  
            1 maps TO set containing 0, 2, and 4  
            2 maps TO set containing 1 and 5  
            3 maps TO set containing 0 and 4  
            4 maps TO set containing 1, 3, and 5  
            5 maps TO set containing 2 and 4  
        
        FOR i FROM 0 TO 1  
            FOR j FROM 0 TO 2  
                IF start at position i, j EQUALS 0 THEN  
                    SET zero_pos TO i MULTIPLIED BY 3 PLUS j  
        
        SET queue TO deque containing tuple of start, zero_pos, and 0  
        SET visited TO set containing start  
        
        WHILE queue IS NOT empty DO  
            SET state, zero_pos, moves TO dequeue from queue  
            
            IF state EQUALS target THEN  
                RETURN moves  
            
            FOR each neighbor IN neighbors at zero_pos DO  
                SET new_state TO list of lists copied from state  
                SET i TO zero_pos DIVIDED BY 3  
                SET j TO zero_pos MODULO 3  
                SET ni TO neighbor DIVIDED BY 3  
                SET nj TO neighbor MODULO 3  
                
                SWAP new_state at i,j WITH new_state at ni,nj  
                SET new_tuple_state TO tuple of tuples created from each row in new_state  
                
                IF new_tuple_state NOT IN visited THEN  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state, neighbor, moves PLUS 1 TO queue  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
