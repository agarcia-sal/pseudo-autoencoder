CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO SPLIT input BY newline  
        SET stack TO empty list  
        SET max_length TO 0  
        
        FOR each line IN lines  
            SET depth TO number of tab characters in line  
            SET name TO substring of line starting from index depth  
            
            WHILE LENGTH OF stack GREATER THAN depth  
                REMOVE last element FROM stack  
            
            IF stack is not empty  
                SET current_length TO last element in stack PLUS LENGTH OF name PLUS 1  
            ELSE  
                SET current_length TO LENGTH OF name  
            
            IF name contains '.'  
                SET max_length TO maximum of max_length and current_length  
            ELSE  
                ADD current_length TO stack  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
