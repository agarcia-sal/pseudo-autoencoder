CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO LENGTH OF passingFees  
        SET graph TO list of empty lists with size n  
        FOR each x, y, time IN edges  
            APPEND tuple of (y, time) TO graph at index x  
            APPEND tuple of (x, time) TO graph at index y  
        SET pq TO list containing tuple of (passingFees at index 0, 0, 0)  
        SET visited_time TO list of infinity values with size n  
        SET visited_time at index 0 TO 0  
        WHILE pq is not empty  
            EXTRACT smallest element from pq INTO current_cost, current_city, current_time  
            IF current_city EQUALS n MINUS 1  
                RETURN current_cost  
            FOR each neighbor, travel_time IN graph at index current_city  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS passingFees at index neighbor  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN visited_time at index neighbor OR new_cost LESS THAN current_cost)  
                    SET visited_time at index neighbor TO new_time  
                    ADD tuple of (new_cost, neighbor, new_time) TO pq  
        RETURN negative one  
    END FUNCTION  
END CLASS
