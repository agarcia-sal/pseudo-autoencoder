CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO 0  
        FOR each p1 IN points  
            SET cnt TO a new empty Counter  
            FOR each p2 IN points  
                SET dx TO difference of p1 first element minus p2 first element  
                SET dy TO difference of p1 second element minus p2 second element  
                SET d TO dx multiplied by dx plus dy multiplied by dy  
                INCREMENT ans BY cnt[d]  
                INCREMENT cnt[d] BY 1  
        RETURN ans multiplied BY 2  
    END FUNCTION  
END CLASS
