CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SPLIT time INTO hours AND minutes BY colon  
            SET total_minutes TO hours MULTIPLIED BY 60 PLUS minutes  
            RETURN total_minutes  
        END FUNCTION  
        
        SET minutes_list TO list OF to_minutes APPLIED TO each item IN timePoints  
        SORT minutes_list  
        
        SET min_diff TO infinity  
        FOR index FROM 1 TO LENGTH OF minutes_list MINUS 1  
            SET current_diff TO minutes_list AT index MINUS minutes_list AT index MINUS 1  
            SET min_diff TO smaller OF min_diff AND current_diff  
        END FOR  
        
        SET circular_diff TO (1440 MINUS last item IN minutes_list) PLUS first item IN minutes_list  
        SET min_diff TO smaller OF min_diff AND circular_diff  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
