CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM 0 TO 8  
                IF board AT row AND i EQUALS num THEN  
                    RETURN False  
            FOR i FROM 0 TO 8  
                IF board AT i AND col EQUALS num THEN  
                    RETURN False  
            SET start_row TO 3 MULTIPLIED BY (row DIVIDED BY 3)  
            SET start_col TO 3 MULTIPLIED BY (col DIVIDED BY 3)  
            FOR i FROM 0 TO 2  
                FOR j FROM 0 TO 2  
                    IF board AT (start_row PLUS i) AND (start_col PLUS j) EQUALS num THEN  
                        RETURN False  
            RETURN True  
        END FUNCTION  

        FUNCTION solve(board)  
            FOR row FROM 0 TO 8  
                FOR col FROM 0 TO 8  
                    IF board AT row AND col EQUALS dot THEN  
                        FOR num FROM 1 TO 9 AS STRING  
                            IF is_valid(board, row, col, num) THEN  
                                SET board AT row AND col TO num  
                                IF solve(board) THEN  
                                    RETURN True  
                                SET board AT row AND col TO dot  
                        RETURN False  
            RETURN True  
        END FUNCTION  

        CALL solve(board)  
    END FUNCTION  
END CLASS
