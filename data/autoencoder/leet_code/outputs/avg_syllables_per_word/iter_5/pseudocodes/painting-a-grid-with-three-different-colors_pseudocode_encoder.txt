CLASS Solution  
    FUNCTION colorTheGrid(m, n)  
        SET MOD TO 10 raised to 9 plus 7  

        FUNCTION generate_valid_columns(prev_col)  
            SET colors TO list containing 1, 2, 3  
            SET valid_cols TO empty list  

            FUNCTION backtrack(col, row)  
                IF row EQUALS m THEN  
                    APPEND tuple of col TO valid_cols  
                    RETURN  
                END IF  
                FOR each color IN colors DO  
                    IF col IS empty OR last element of col NOT EQUAL TO color THEN  
                        IF prev_col EXISTS AND prev_col at row EQUALS color THEN  
                            CONTINUE with next color  
                        END IF  
                        APPEND color TO col  
                        CALL backtrack(col, row plus 1)  
                        REMOVE last element from col  
                    END IF  
                END FOR  
            END FUNCTION  

            CALL backtrack(empty list, 0)  
            RETURN valid_cols  
        END FUNCTION  

        SET all_valid_cols TO generate_valid_columns(None)  
        SET num_valid_cols TO LENGTH OF all_valid_cols  

        SET compatible_cols TO mapping from each index in range num_valid_cols TO empty list  
        FOR i FROM 0 TO num_valid_cols MINUS 1 DO  
            FOR j FROM 0 TO num_valid_cols MINUS 1 DO  
                IF FOR ALL k IN 0 TO m MINUS 1, all_valid_cols at i at k NOT EQUAL TO all_valid_cols at j at k THEN  
                    APPEND j TO compatible_cols at i  
                END IF  
            END FOR  
        END FOR  

        SET dp TO list containing 1 repeated num_valid_cols times  

        FOR each _ IN range from 1 TO n MINUS 1 DO  
            SET next_dp TO list containing 0 repeated num_valid_cols times  
            FOR i FROM 0 TO num_valid_cols MINUS 1 DO  
                FOR each j IN compatible_cols at i DO  
                    SET next_dp at j TO (next_dp at j PLUS dp at i) MODULO MOD  
                END FOR  
            END FOR  
            SET dp TO next_dp  
        END FOR  

        RETURN sum of dp MODULO MOD  
    END FUNCTION  
END CLASS
