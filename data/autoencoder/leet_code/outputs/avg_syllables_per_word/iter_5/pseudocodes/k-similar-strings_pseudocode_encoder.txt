CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO 0  
            WHILE i LESS THAN LENGTH OF s AND CHARACTER AT s AT i EQUALS CHARACTER AT s2 AT i  
                INCREMENT i BY 1  
            FOR j FROM i PLUS 1 TO LENGTH OF s MINUS 1  
                IF CHARACTER AT s AT j EQUALS CHARACTER AT s2 AT i AND CHARACTER AT s AT j NOT EQUAL CHARACTER AT s2 AT j  
                    YIELD CONCATENATION OF s FROM START TO i, CHARACTER AT s AT j, s FROM i PLUS 1 TO j MINUS 1, CHARACTER AT s AT i, s FROM j PLUS 1 TO END  
        END FUNCTION  
        
        CREATE queue AS EMPTY DEQUE  
        ADD TUPLE OF s1 AND 0 TO queue  
        CREATE visited AS SET CONTAINING s1  
        
        WHILE queue IS NOT EMPTY  
            REMOVE LEFTMOST ELEMENT FROM queue INTO s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND TUPLE OF neighbor AND swaps PLUS 1 TO queue  
        END WHILE  
    END FUNCTION  
END CLASS
