CLASS Solution  
    FUNCTION isTransformable(s, t)  
        INITIALIZE pos AS a default dictionary of empty lists  
        FOR i, c IN enumeration of s  
            ADD i TO the heap at pos[int(c)]  
        FOR i, c IN enumeration of t  
            SET digit TO int(c)  
            IF the heap at pos[digit] IS empty  
                RETURN False  
            FOR each smaller FROM 0 UP TO digit MINUS 1  
                IF the heap at pos[smaller] IS NOT empty AND its smallest element IS LESS THAN the smallest element of pos[digit]  
                    RETURN False  
            REMOVE the smallest element from pos[digit]  
        RETURN True  
    END FUNCTION  
END CLASS
