CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM MINIMUM OF LENGTH OF words AT i AND LENGTH OF words AT j DOWN TO 1  
                IF SUBSTRING OF words AT i FROM END OF LENGTH k EQUALS SUBSTRING OF words AT j FROM START OF LENGTH k  
                    RETURN k  
            RETURN 0  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS (1 LEFT SHIFTED BY n) MINUS 1  
                RETURN (0, empty string)  
            
            SET min_len TO infinity  
            SET best_path TO empty string  
            FOR j FROM 0 TO n MINUS 1  
                IF bit j NOT SET IN mask  
                    SET length, path TO dp(mask WITH bit j SET, j)  
                    INCREMENT length BY LENGTH OF words AT j MINUS overlap(i, j)  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO SUBSTRING OF words AT j FROM overlap(i, j) TO END CONCATENATED WITH path  
            RETURN (min_len, best_path)  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO empty string  
        FOR i FROM 0 TO n MINUS 1  
            SET length, path TO dp(bit i SET, i)  
            INCREMENT length BY LENGTH OF words AT i  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO words AT i CONCATENATED WITH path  
        RETURN shortest_path  
    END FUNCTION  
END CLASS
