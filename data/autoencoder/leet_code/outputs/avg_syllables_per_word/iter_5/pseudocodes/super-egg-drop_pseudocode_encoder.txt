CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        SET dp TO list of zeros with length k plus one  
        SET moves TO 0  
        WHILE dp at index k LESS THAN n  
            INCREMENT moves BY 1  
            FOR eggs FROM k DOWN TO 1  
                SET dp at index eggs TO 1 PLUS dp at index eggs MINUS 1 PLUS dp at index eggs  
        RETURN moves  
    END FUNCTION  
END CLASS
