CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF sum of gas IS LESS THAN sum of cost THEN  
            RETURN negative one  
        END IF  

        SET start TO zero  
        SET total_tank TO zero  

        FOR each index i FROM zero TO length of gas MINUS one DO  
            SET total_tank TO total_tank PLUS gas at i MINUS cost at i  

            IF total_tank IS LESS THAN zero THEN  
                SET start TO i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  

        RETURN start  
    END FUNCTION  
END CLASS
