CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS REVERSE OF sub  
        END FUNCTION  
  
        SET n TO LENGTH OF s  
  
        SET is_pal TO LIST OF LISTS OF BOOLEAN False WITH SIZE n BY n  
  
        FOR i FROM n MINUS 1 DOWNTO 0  
            FOR j FROM i TO n MINUS 1  
                IF CHARACTER AT s AT i EQUALS CHARACTER AT s AT j AND (j MINUS i LESS THAN OR EQUAL TO 1 OR is_pal AT (i PLUS 1) AT (j MINUS 1) IS True)  
                    SET is_pal AT i AT j TO True  
                END IF  
            END FOR  
        END FOR  
  
        FOR i FROM 1 TO n MINUS 2  
            IF is_pal AT 0 AT (i MINUS 1) IS True  
                FOR j FROM i TO n MINUS 2  
                    IF is_pal AT i AT j IS True AND is_pal AT (j PLUS 1) AT (n MINUS 1) IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  
  
        RETURN False  
    END FUNCTION  
END CLASS
