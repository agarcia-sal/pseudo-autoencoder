CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as first element of values  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element from queue  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO new TreeNode with val as values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO new TreeNode with val as values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(root)  
        FUNCTION rob_from_node(node)  
            IF node IS None  
                RETURN 0, 0  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node(node.left)  
            SET right_robbed, right_not_robbed TO rob_from_node(node.right)  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO MAXIMUM OF left_robbed AND left_not_robbed PLUS MAXIMUM OF right_robbed AND right_not_robbed  
            RETURN rob_this_node, not_rob_this_node  
        END FUNCTION  
        RETURN MAXIMUM OF rob_from_node(root)  
    END FUNCTION  
END CLASS
