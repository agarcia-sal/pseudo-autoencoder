CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY length AND lexicographically  
        SET valid_words TO empty set  
        SET longest_word TO empty string  
        FOR each word IN words  
            IF length of word EQUALS 1 OR substring of word EXCLUDING last character IN valid_words  
                ADD word TO valid_words  
                IF length of word GREATER THAN length of longest_word  
                    SET longest_word TO word  
        RETURN longest_word  
    END FUNCTION  
END CLASS
