CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO LENGTH OF statements  
        SET max_good TO 0  
        
        FOR i FROM 0 TO 2 RAISED TO n MINUS 1  
            SET good TO LIST OF n ELEMENTS WITH VALUE False  
            SET bad TO LIST OF n ELEMENTS WITH VALUE False  
            
            FOR j FROM 0 TO n MINUS 1  
                IF BIT j IN i IS SET  
                    SET good[j] TO True  
                ELSE  
                    SET bad[j] TO True  
            
            SET valid TO True  
            FOR j FROM 0 TO n MINUS 1  
                IF good[j] IS True  
                    FOR k FROM 0 TO n MINUS 1  
                        IF (statements[j][k] EQUALS 0 AND good[k] IS True) OR (statements[j][k] EQUALS 1 AND bad[k] IS True)  
                            SET valid TO False  
                            BREAK  
                    IF valid IS False  
                        BREAK  
            
            IF valid IS True  
                SET count_good TO NUMBER OF True ELEMENTS IN good  
                SET max_good TO THE GREATER OF max_good AND count_good  
        
        RETURN max_good  
    END FUNCTION  
END CLASS
