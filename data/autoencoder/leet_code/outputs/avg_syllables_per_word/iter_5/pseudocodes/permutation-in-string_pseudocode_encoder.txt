CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET len_s1 TO LENGTH OF s1  
        SET len_s2 TO LENGTH OF s2  
        
        IF len_s1 GREATER THAN len_s2 THEN  
            RETURN False  
        END IF  
        
        SET count_s1 TO FREQUENCY COUNT OF CHARACTERS IN s1  
        SET count_s2 TO FREQUENCY COUNT OF CHARACTERS IN FIRST len_s1 CHARACTERS OF s2  
        
        IF count_s1 EQUALS count_s2 THEN  
            RETURN True  
        END IF  
        
        FOR i FROM len_s1 TO len_s2 MINUS 1 DO  
            INCREMENT count_s2 FOR CHARACTER s2 AT POSITION i BY 1  
            DECREMENT count_s2 FOR CHARACTER s2 AT POSITION (i MINUS len_s1) BY 1  
            
            IF count_s2 FOR CHARACTER s2 AT POSITION (i MINUS len_s1) EQUALS 0 THEN  
                REMOVE CHARACTER s2 AT POSITION (i MINUS len_s1) FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2 THEN  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
