CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        SET position TO dictionary mapping each person to their index in row  
        SET swaps TO 0  
        
        FOR i FROM 0 TO LENGTH OF row MINUS 1 STEP 2  
            SET correct_pair TO person that pairs with row at index i  
            
            IF person at index i plus 1 in row IS NOT correct_pair  
                SET swap_idx TO position of correct_pair  
                
                SWAP persons at indices i plus 1 and swap_idx in row  
                
                UPDATE position of person now at swap_idx TO swap_idx  
                UPDATE position of person now at i plus 1 TO i plus 1  
                
                INCREMENT swaps BY 1  
        RETURN swaps  
    END FUNCTION  
END CLASS
