CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty map  
        SET max_length TO 0  
        SET start TO 0  
        
        FOR end, char IN enumeration of s  
            IF char exists in char_index_map AND char_index_map value for char GREATER THAN OR EQUAL TO start  
                SET start TO value from char_index_map for char INCREMENTED BY 1  
            END IF  
            SET char_index_map value for char TO end  
            SET max_length TO greater value BETWEEN max_length AND end MINUS start INCREMENTED BY 1  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
