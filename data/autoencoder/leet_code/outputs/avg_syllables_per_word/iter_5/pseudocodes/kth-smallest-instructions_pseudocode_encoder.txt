CLASS Solution  
    FUNCTION kthSmallestPath(destination, k)  
        SET row TO first element of destination  
        SET col TO second element of destination  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        
        FOR i FROM 0 TO total_moves MINUS 1  
            IF col GREATER THAN 0  
                SET paths_with_H TO number of combinations of row PLUS col MINUS 1 CHOOSE col MINUS 1  
                IF k LESS THAN OR EQUAL TO paths_with_H  
                    APPEND 'H' TO path  
                    DECREMENT col BY 1  
                ELSE  
                    APPEND 'V' TO path  
                    DECREMENT row BY 1  
                    DECREMENT k BY paths_with_H  
            ELSE  
                APPEND 'V' TO path  
                DECREMENT row BY 1  
        
        RETURN concatenation of all characters in path  
    END FUNCTION  
END CLASS
