CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS empty OR first row of picture IS empty THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF first row of picture  
        SET row_count TO list of m zeros  
        SET col_count TO list of n zeros  
        SET row_patterns TO map with default 0  
        
        FOR r FROM 0 TO m MINUS 1  
            SET pattern TO concatenation of elements in picture at row r  
            INCREMENT row_patterns at pattern BY 1  
            FOR c FROM 0 TO n MINUS 1  
                IF picture at row r and column c IS 'B' THEN  
                    INCREMENT row_count at r BY 1  
                    INCREMENT col_count at c BY 1  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO 0  
        
        FOR r FROM 0 TO m MINUS 1  
            FOR c FROM 0 TO n MINUS 1  
                IF picture at row r and column c IS 'B' THEN  
                    SET pattern TO concatenation of elements in picture at row r  
                    IF row_count at r EQUALS target AND col_count at c EQUALS target AND row_patterns at pattern EQUALS target THEN  
                        INCREMENT lonely_pixels BY 1  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
