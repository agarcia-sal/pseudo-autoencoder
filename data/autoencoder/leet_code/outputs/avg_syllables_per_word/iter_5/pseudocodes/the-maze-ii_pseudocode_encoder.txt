CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF first element IN maze  
        SET directions TO list of (1, 0), (-1, 0), (0, 1), (0, -1)  
        CONVERT start TO tuple  
        CONVERT destination TO tuple  
        INITIALIZE visited AS empty set  
        INITIALIZE priority queue pq WITH element (0, start)  

        WHILE pq IS NOT empty  
            EXTRACT element with smallest distance FROM pq INTO dist, (x, y)  

            IF (x, y) EQUALS destination  
                RETURN dist  
            END IF  

            IF (x, y) IS IN visited  
                CONTINUE to next iteration  
            END IF  

            ADD (x, y) TO visited  

            FOR each (dx, dy) IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO 0  

                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO 0 AND LESS THAN m  
                  AND new_y plus dy IS GREATER THAN OR EQUAL TO 0 AND LESS THAN n  
                  AND maze at position (new_x plus dx, new_y plus dy) EQUALS 0  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY 1  
                END WHILE  

                IF (new_x, new_y) IS NOT IN visited  
                    ADD (dist plus count, (new_x, new_y)) TO pq  
                END IF  
            END FOR  
        END WHILE  

        RETURN -1  
    END FUNCTION  
END CLASS
