CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum value in nums  
        SET num_set TO the set of elements in nums  
        SET count TO 0  
        
        FOR g FROM 1 TO max_num  
            SET gcd_val TO 0  
            FOR m FROM g TO max_num IN STEPS OF g  
                IF m is in num_set THEN  
                    SET gcd_val TO the greatest common divisor of gcd_val and m  
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY 1  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
