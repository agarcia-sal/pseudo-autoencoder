CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO set constructed from bank  
        SET q TO a deque containing the pair (start, 0)  
        SET mp TO dictionary mapping 'A' TO 'TCG', 'T' TO 'ACG', 'C' TO 'ATG', 'G' TO 'ATC'  
        WHILE q is not empty  
            SET t, step TO first element removed from q  
            IF t equals end  
                RETURN step  
            FOR each index i and character v in t  
                FOR each character j in mp at key v  
                    SET next TO string formed by replacing character at i in t with j  
                    IF next is in s  
                        APPEND the pair (next, step plus 1) TO q  
                        REMOVE next FROM s  
        RETURN negative 1  
    END FUNCTION  
END CLASS
