CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            RETURN (val1 GREATER THAN 0 AND val2 GREATER THAN 0) OR (val1 LESS THAN 0 AND val2 LESS THAN 0)  
        END FUNCTION  
        
        SET n TO LENGTH OF nums  
        
        FOR i FROM 0 TO n MINUS 1  
            SET slow TO i  
            SET fast TO i  
            
            IF nums AT i EQUALS 0  
                CONTINUE TO NEXT i  
            
            WHILE TRUE  
                SET slow TO slow PLUS nums AT slow MODULO n  
                IF slow LESS THAN 0  
                    INCREMENT slow BY n  
                
                SET fast TO fast PLUS nums AT fast MODULO n  
                IF fast LESS THAN 0  
                    INCREMENT fast BY n  
                SET fast TO fast PLUS nums AT fast MODULO n  
                IF fast LESS THAN 0  
                    INCREMENT fast BY n  
                
                IF NOT sameDirection(nums AT slow, nums AT i) OR NOT sameDirection(nums AT fast, nums AT i)  
                    BREAK FROM WHILE  
                
                IF slow EQUALS fast  
                    IF slow EQUALS (slow PLUS nums AT slow) MODULO n  
                        BREAK FROM WHILE  
                    RETURN True  
        
        RETURN False  
    END FUNCTION  
END CLASS
