CLASS Solution  
    FUNCTION minmaxGasDist(stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO 0  
            FOR i FROM 1 TO LENGTH OF stations MINUS 1  
                INCREMENT additional_stations BY INTEGER PART OF (stations AT i MINUS stations AT i MINUS 1) DIVIDED BY max_dist  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO 0  
        SET right TO 100000000  
        WHILE right MINUS left GREATER THAN 0.000001  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            IF possible(mid) IS TRUE  
                SET right TO mid  
            ELSE  
                SET left TO mid  
        RETURN left  
    END FUNCTION  
END CLASS
