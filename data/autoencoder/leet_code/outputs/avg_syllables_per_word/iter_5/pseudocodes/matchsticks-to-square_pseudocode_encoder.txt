CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO sum of matchsticks  
        IF total_length MODULO 4 NOT EQUAL TO 0 THEN  
            RETURN False  
        END IF  
        
        SET side_length TO total_length DIVIDED BY 4  
        SORT matchsticks IN descending order  
        
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS length of matchsticks THEN  
                RETURN whether all sides are EQUAL TO side_length  
            END IF  
            
            FOR i FROM 0 TO 3  
                IF sides at i PLUS matchsticks at index LESS THAN OR EQUAL TO side_length THEN  
                    INCREMENT sides at i BY matchsticks at index  
                    IF can_form_sides(sides, index PLUS 1) IS True THEN  
                        RETURN True  
                    END IF  
                    DECREMENT sides at i BY matchsticks at index  
                END IF  
                
                IF sides at i EQUALS 0 THEN  
                    BREAK  
                END IF  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN can_form_sides(list of four zeros, 0)  
    END FUNCTION  
END CLASS
