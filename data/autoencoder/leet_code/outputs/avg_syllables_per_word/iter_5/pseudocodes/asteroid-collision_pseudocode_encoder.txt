CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE stack is not empty AND asteroid is less than 0 AND top element of stack is greater than 0  
                IF top element of stack is less than negative asteroid  
                    REMOVE top element from stack  
                    CONTINUE to next iteration of WHILE loop  
                ELSE IF top element of stack is equal to negative asteroid  
                    REMOVE top element from stack  
                END IF  
                BREAK from WHILE loop  
            END WHILE  
            IF WHILE loop did not break  
                ADD asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
