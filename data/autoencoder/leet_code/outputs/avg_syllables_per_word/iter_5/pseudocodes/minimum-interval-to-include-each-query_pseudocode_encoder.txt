CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        SORT intervals BY start time  
        SET sorted_queries TO list of pairs (query, original index) sorted BY query  
          
        INITIALIZE min_heap AS empty heap  
        SET result TO list of length of queries FILLED WITH -1  
          
        SET interval_index TO 0  
        FOR each query, original_index IN sorted_queries  
            WHILE interval_index LESS THAN length of intervals AND start of intervals at interval_index LESS THAN OR EQUAL TO query  
                SET start, end TO intervals at interval_index  
                SET interval_size TO end MINUS start PLUS 1  
                ADD (interval_size, end) TO min_heap  
                INCREMENT interval_index BY 1  
              
            WHILE min_heap IS NOT empty AND the smallest end value in min_heap LESS THAN query  
                REMOVE smallest element FROM min_heap  
              
            IF min_heap IS NOT empty  
                SET result at original_index TO smallest interval_size in min_heap  
          
        RETURN result  
    END FUNCTION  
END CLASS
