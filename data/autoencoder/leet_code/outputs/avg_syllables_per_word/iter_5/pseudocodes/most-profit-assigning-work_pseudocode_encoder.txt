CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list of pairs combining difficulty and profit  
        SORT jobs by difficulty  
        SORT worker by ability  
        
        SET max_profit TO 0  
        SET best_profit TO 0  
        SET job_index TO 0  
        SET n TO length of jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND difficulty of jobs at job_index LESS THAN OR EQUAL TO ability  
                SET best_profit TO greater value between best_profit AND profit of jobs at job_index  
                INCREMENT job_index BY 1  
            INCREMENT max_profit BY best_profit  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
