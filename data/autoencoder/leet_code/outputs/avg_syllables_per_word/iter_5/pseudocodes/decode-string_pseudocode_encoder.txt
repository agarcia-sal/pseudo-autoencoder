CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS digit  
                SET d TO d concatenated with char  
            ELSE IF char IS alphabetic  
                SET a TO a concatenated with char  
            ELSE IF char IS equal to '['  
                APPEND tuple of (a, integer value of d) TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char IS equal to ']'  
                SET p, n TO element popped from stack  
                SET a TO p concatenated with a repeated n times  
        RETURN a  
    END FUNCTION  
END CLASS
