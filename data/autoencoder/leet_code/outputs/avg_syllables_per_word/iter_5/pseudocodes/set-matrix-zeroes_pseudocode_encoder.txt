CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix IS empty OR matrix AT 0 IS empty THEN  
            RETURN  
        END IF  

        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix AT 0  

        SET first_row_has_zero TO false  
        FOR j FROM 0 TO n MINUS 1  
            IF matrix AT 0 AT j EQUALS 0 THEN  
                SET first_row_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        SET first_col_has_zero TO false  
        FOR i FROM 0 TO m MINUS 1  
            IF matrix AT i AT 0 EQUALS 0 THEN  
                SET first_col_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix AT i AT j EQUALS 0 THEN  
                    SET matrix AT i AT 0 TO 0  
                    SET matrix AT 0 AT j TO 0  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix AT i AT 0 EQUALS 0 OR matrix AT 0 AT j EQUALS 0 THEN  
                    SET matrix AT i AT j TO 0  
                END IF  
            END FOR  
        END FOR  

        IF first_row_has_zero THEN  
            FOR j FROM 0 TO n MINUS 1  
                SET matrix AT 0 AT j TO 0  
            END FOR  
        END IF  

        IF first_col_has_zero THEN  
            FOR i FROM 0 TO m MINUS 1  
                SET matrix AT i AT 0 TO 0  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
