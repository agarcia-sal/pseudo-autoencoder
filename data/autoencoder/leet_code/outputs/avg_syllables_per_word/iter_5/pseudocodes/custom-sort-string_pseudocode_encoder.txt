CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO EMPTY DICTIONARY  
        FOR each index i AND character c IN order  
            SET d AT c TO i  
        SET sorted_characters TO s SORTED BY d VALUE OF character OR 0 IF character NOT IN d  
        SET result TO JOIN sorted_characters INTO STRING  
        RETURN result  
    END FUNCTION  
END CLASS
