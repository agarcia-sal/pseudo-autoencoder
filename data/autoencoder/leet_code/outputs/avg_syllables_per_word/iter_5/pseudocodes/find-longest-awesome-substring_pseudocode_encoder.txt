CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO dictionary WITH key 0 MAPPED TO value -1  
        SET prefix_xor TO 0  
        SET max_length TO 0  
        
        FOR each index i AND character char IN s  
            SET digit TO integer value of char  
            TOGGLE bit corresponding TO digit IN prefix_xor  
            
            IF prefix_xor EXISTS IN prefix_map  
                SET max_length TO maximum OF max_length AND i MINUS prefix_map AT prefix_xor  
            END IF  
            
            FOR each j FROM 0 TO 9  
                SET flipped TO prefix_xor TOGGLED AT bit j  
                IF flipped EXISTS IN prefix_map  
                    SET max_length TO maximum OF max_length AND i MINUS prefix_map AT flipped  
                END IF  
            END FOR  
            
            IF prefix_xor NOT EXISTS IN prefix_map  
                STORE i IN prefix_map AT prefix_xor  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
