CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO 0 THEN  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        
        FOR i FROM 0 TO MINIMUM OF k AND LENGTH OF nums1 MINUS 1  
            PUSH tuple of (nums1 at i PLUS nums2 at 0, i, 0) INTO min_heap  
        END FOR  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT EMPTY AND LENGTH OF result IS LESS THAN k  
            POP smallest tuple FROM min_heap INTO variables _, i, j  
            APPEND list of nums1 at i AND nums2 at j TO result  
            
            IF j PLUS 1 IS LESS THAN LENGTH OF nums2 THEN  
                PUSH tuple of (nums1 at i PLUS nums2 at j PLUS 1, i, j PLUS 1) INTO min_heap  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
