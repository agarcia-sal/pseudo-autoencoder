CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a mapping with default integer zero  
        
        FOR each i IN nums  
            FOR each j IN nums  
                INCREMENT pair_and_count value for key bitwise AND of i and j BY one  
        
        SET triplet_count TO zero  
        FOR each k IN nums  
            FOR each pair_and, count IN pair_and_count  
                IF bitwise AND of pair_and and k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
        
        RETURN triplet_count  
    END FUNCTION  
END CLASS
