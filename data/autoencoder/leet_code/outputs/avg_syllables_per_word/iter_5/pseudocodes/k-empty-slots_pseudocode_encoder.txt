CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO empty list  
        
        FOR each day, position IN bulbs starting from day 1  
            SET i TO position where position would be inserted in bloom_days to keep it sorted  
            
            IF i GREATER THAN 0 AND position MINUS bloom_days at i MINUS 1 EQUALS k THEN  
                RETURN day  
            END IF  
            
            IF i LESS THAN LENGTH OF bloom_days AND bloom_days at i MINUS position MINUS 1 EQUALS k THEN  
                RETURN day  
            END IF  
            
            INSERT position INTO bloom_days AT index i  
        END FOR  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
