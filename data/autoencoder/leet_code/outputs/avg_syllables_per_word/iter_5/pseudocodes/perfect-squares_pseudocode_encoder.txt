CLASS Solution  
    FUNCTION numSquares(n)  
        SET dp TO list of length n plus 1 with all elements set to infinity  
        SET dp at index 0 TO 0  
        SET squares TO empty list  
        SET i TO 1  
        WHILE i multiplied BY i is less than or equal to n  
            APPEND i multiplied BY i TO squares  
            INCREMENT i BY 1  
        FOR i FROM 1 TO n inclusive  
            FOR each square IN squares  
                IF i is less than square  
                    BREAK loop  
                SET dp at index i TO minimum of dp at i and dp at i minus square plus 1  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
