CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        
        FOR each num IN nums traversed from right to left  
            IF num LESS THAN second THEN  
                RETURN True  
            END IF  
            WHILE stack is not empty AND top of stack LESS THAN num  
                SET second TO REMOVE top element from stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
