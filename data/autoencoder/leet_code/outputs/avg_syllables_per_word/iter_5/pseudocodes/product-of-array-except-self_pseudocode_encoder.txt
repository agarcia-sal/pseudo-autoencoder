CLASS Solution  
    FUNCTION productExceptSelf(nums)  
        SET n TO LENGTH OF nums  
        SET answer TO LIST OF LENGTH n FILLED WITH 1  
          
        SET left_product TO 1  
        FOR i FROM 0 TO n MINUS 1  
            SET answer AT i TO left_product  
            SET left_product TO left_product MULTIPLIED BY nums AT i  
          
        SET right_product TO 1  
        FOR i FROM n MINUS 1 DOWN TO 0  
            SET answer AT i TO answer AT i MULTIPLIED BY right_product  
            SET right_product TO right_product MULTIPLIED BY nums AT i  
          
        RETURN answer  
    END FUNCTION  
END CLASS
