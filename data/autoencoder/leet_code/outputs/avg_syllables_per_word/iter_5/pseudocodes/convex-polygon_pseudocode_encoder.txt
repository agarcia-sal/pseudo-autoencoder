CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN difference of horizontal distance from o to a MULTIPLIED BY difference of vertical distance from o to b  
            MINUS difference of vertical distance from o to a MULTIPLIED BY difference of horizontal distance from o to b  
        END FUNCTION  
        
        SET n TO LENGTH OF points  
        IF n LESS THAN OR EQUAL TO 3  
            RETURN True  
          
        SET orientation TO 0  
        
        FOR i FROM 0 TO n MINUS 1  
            SET o TO points AT index i  
            SET a TO points AT index (i PLUS 1) MODULO n  
            SET b TO points AT index (i PLUS 2) MODULO n  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUAL TO 0  
                IF orientation EQUALS 0  
                    SET orientation TO 1 IF cross GREATER THAN 0 ELSE -1  
                ELSE IF (orientation EQUALS 1 AND cross LESS THAN 0) OR (orientation EQUALS -1 AND cross GREATER THAN 0)  
                    RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
