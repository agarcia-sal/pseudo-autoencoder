CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS 1 THEN  
            RETURN "1"  
        END IF  
        
        SET current_sequence TO "1"  
        
        FOR each number FROM 1 TO n MINUS 1 DO  
            SET next_sequence TO empty string  
            SET count TO 1  
            SET previous_char TO first character OF current_sequence  
            
            FOR each index FROM 1 TO LENGTH OF current_sequence MINUS 1 DO  
                IF character AT current_sequence AT index EQUALS previous_char THEN  
                    INCREMENT count BY 1  
                ELSE  
                    APPEND string representation OF count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO character AT current_sequence AT index  
                    SET count TO 1  
                END IF  
            END FOR  
            
            APPEND string representation OF count TO next_sequence  
            APPEND previous_char TO next_sequence  
            
            SET current_sequence TO next_sequence  
        END FOR  
        
        RETURN current_sequence  
    END FUNCTION  
END CLASS
