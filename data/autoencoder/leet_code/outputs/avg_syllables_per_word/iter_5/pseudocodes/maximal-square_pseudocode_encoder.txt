CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix IS EMPTY OR matrix FIRST ELEMENT IS EMPTY  
            RETURN 0  
        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix FIRST ELEMENT  
        SET dp TO LIST OF LISTS FILLED WITH 0 OF SIZE (m PLUS 1) BY (n PLUS 1)  
        SET max_side TO 0  
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF matrix ELEMENT AT (i MINUS 1, j MINUS 1) EQUALS '1'  
                    SET dp ELEMENT AT (i, j) TO MINIMUM OF dp ELEMENTS AT (i MINUS 1, j), (i, j MINUS 1), AND (i MINUS 1, j MINUS 1) PLUS 1  
                    SET max_side TO MAXIMUM OF max_side AND dp ELEMENT AT (i, j)  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
