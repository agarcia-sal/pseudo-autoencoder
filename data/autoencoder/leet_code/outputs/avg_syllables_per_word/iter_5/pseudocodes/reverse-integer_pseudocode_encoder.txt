CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN 0  
            SET sign TO -1  
        ELSE  
            SET sign TO 1  
        END IF  
        SET reversed_x TO integer value of string representation of absolute value of x reversed  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative 2 TO THE POWER OF 31 OR result GREATER THAN positive 2 TO THE POWER OF 31 MINUS 1  
            RETURN 0  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
