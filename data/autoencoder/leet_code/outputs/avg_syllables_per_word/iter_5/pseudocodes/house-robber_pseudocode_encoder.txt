CLASS Solution  
    FUNCTION rob(nums)  
        IF nums IS empty  
            RETURN 0  
        IF LENGTH OF nums IS 1  
            RETURN FIRST ELEMENT OF nums  
        
        SET dp TO list of zeros with LENGTH OF nums  
        SET dp AT 0 TO FIRST ELEMENT OF nums  
        SET dp AT 1 TO GREATER OF FIRST ELEMENT OF nums AND SECOND ELEMENT OF nums  
        
        FOR i FROM 2 TO LENGTH OF nums MINUS 1  
            SET dp AT i TO GREATER OF dp AT i MINUS 1 AND dp AT i MINUS 2 PLUS nums AT i  
        
        RETURN LAST ELEMENT OF dp  
    END FUNCTION  
END CLASS
