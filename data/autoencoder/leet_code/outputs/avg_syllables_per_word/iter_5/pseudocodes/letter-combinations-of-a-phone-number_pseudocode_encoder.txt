CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits IS empty  
            RETURN empty list  
        END IF  
        
        SET phone_map TO dictionary mapping digits '2' TO '9' with corresponding letters as values  
        
        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits IS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map value for first character of next_digits  
                    CALL backtrack WITH combination concatenated WITH letter, and next_digits excluding first character  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string, digits  
        RETURN output  
    END FUNCTION  
END CLASS
