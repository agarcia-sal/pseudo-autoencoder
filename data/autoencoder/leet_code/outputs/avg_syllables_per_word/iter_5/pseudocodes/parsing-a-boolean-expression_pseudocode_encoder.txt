CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character c IN expression  
            IF c IS IN 'tf!&|'  
                APPEND c TO stk  
            ELSE IF c IS ')'  
                SET t TO 0  
                SET f TO 0  
                WHILE last element in stk IS IN 'tf'  
                    INCREMENT t BY 1 IF last element in stk IS 't'  
                    INCREMENT f BY 1 IF last element in stk IS 'f'  
                    REMOVE last element FROM stk  
                END WHILE  
                SET op TO last element popped FROM stk  
                IF op IS '!'  
                    SET c TO 't' IF f IS 0 ELSE 'f'  
                ELSE IF op IS '&'  
                    SET c TO 'f' IF f IS GREATER THAN 0 ELSE 't'  
                ELSE IF op IS '|'  
                    SET c TO 't' IF t IS GREATER THAN 0 ELSE 'f'  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN True IF first element of stk IS 't' ELSE False  
    END FUNCTION  
END CLASS
