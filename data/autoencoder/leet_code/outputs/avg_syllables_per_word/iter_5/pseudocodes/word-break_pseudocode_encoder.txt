CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set of elements in wordDict  
        SET dp TO a list of Boolean False with length LENGTH OF s PLUS 1  
        SET dp at index 0 TO True  
        
        FOR i FROM 1 TO LENGTH OF s DO  
            FOR j FROM 0 TO i MINUS 1 DO  
                IF dp at index j IS True AND substring of s from j TO i IS IN wordSet THEN  
                    SET dp at index i TO True  
                    BREAK inner loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at index LENGTH OF s  
    END FUNCTION  
END CLASS
