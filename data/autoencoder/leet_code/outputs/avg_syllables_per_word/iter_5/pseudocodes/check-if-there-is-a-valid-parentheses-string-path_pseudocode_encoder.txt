CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT INDEX 0  
        
        IF (m PLUS n MINUS 1) MODULO 2 IS NOT EQUAL TO 0 THEN  
            RETURN False  
        
        SET dp TO THREE-DIMENSIONAL LIST OF False WITH DIMENSIONS (m PLUS 1) BY (n PLUS 1) BY (m PLUS n PLUS 1)  
        
        SET dp AT 0, 1, 0 TO True  
        
        FOR i FROM 0 TO m MINUS 1 DO  
            FOR j FROM 0 TO n MINUS 1 DO  
                FOR k FROM 0 TO m PLUS n DO  
                    IF dp AT i, j PLUS 1, k IS True OR dp AT i PLUS 1, j, k IS True THEN  
                        IF grid AT i, j IS '(' AND k PLUS 1 IS LESS THAN m PLUS n PLUS 1 THEN  
                            SET dp AT i PLUS 1, j PLUS 1, k PLUS 1 TO True  
                        ELSE IF grid AT i, j IS ')' AND k MINUS 1 IS GREATER THAN OR EQUAL TO 0 THEN  
                            SET dp AT i PLUS 1, j PLUS 1, k MINUS 1 TO True  
        
        RETURN dp AT m, n, 0  
    END FUNCTION  
END CLASS
