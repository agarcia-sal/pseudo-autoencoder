CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS 1 THEN  
            RETURN 1  
        END IF  
        
        SET factors TO empty list  
        FOR digit FROM 9 DOWN TO 2 DO  
            WHILE num MODULO digit EQUALS 0 DO  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS 1 THEN  
            RETURN 0  
        END IF  
        
        SORT factors IN ascending order  
        SET result TO integer formed by concatenating elements of factors  
        
        IF result GREATER THAN 2 TO THE POWER OF 31 MINUS 1 THEN  
            RETURN 0  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
