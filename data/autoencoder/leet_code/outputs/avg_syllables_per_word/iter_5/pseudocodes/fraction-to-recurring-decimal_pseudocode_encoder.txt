CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF (numerator LESS THAN 0) XOR (denominator LESS THAN 0) THEN  
            SET sign TO "-"  
        ELSE  
            SET sign TO ""  
        END IF  
        
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
        
        SET quotient TO numerator DIVIDED BY denominator  
        SET remainder TO numerator MODULO denominator  
        SET integer_part TO string representation of quotient  
        
        IF remainder EQUALS 0 THEN  
            RETURN concatenation of sign and integer_part  
        END IF  
        
        INITIALIZE seen_remainders AS empty dictionary  
        INITIALIZE fractional_part AS empty list  
        
        WHILE remainder NOT EQUAL TO 0  
            IF remainder IS IN seen_remainders THEN  
                SET start TO value in seen_remainders at key remainder  
                SET non_repeating TO concatenation of elements in fractional_part up to index start  
                SET repeating TO concatenation of elements in fractional_part from index start onward  
                RETURN concatenation of sign, integer_part, ".", non_repeating, "(", repeating, ")"  
            END IF  
            
            SET seen_remainders at key remainder TO length of fractional_part  
            
            SET remainder TO remainder MULTIPLIED BY 10  
            SET quotient TO remainder DIVIDED BY denominator  
            SET remainder TO remainder MODULO denominator  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN concatenation of sign, integer_part, ".", concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
