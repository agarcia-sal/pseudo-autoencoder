CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO new TreeNode with values at index zero  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO queue REMOVE first element  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(root)  
        IF root IS None  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new deque containing root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO LENGTH OF queue  
            SET current_level TO empty list  
            FOR counter FROM zero TO level_size MINUS one  
                SET node TO queue REMOVE first element  
                APPEND node.val TO current_level  
                IF node.left EXISTS  
                    APPEND node.left TO queue  
                END IF  
                IF node.right EXISTS  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
