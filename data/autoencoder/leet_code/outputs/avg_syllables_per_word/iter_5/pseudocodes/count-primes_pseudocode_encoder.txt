CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO 2  
            RETURN 0  
        END IF  
  
        INITIALIZE is_prime AS list of True values with length n  
        SET is_prime at index 0 TO False  
        SET is_prime at index 1 TO False  
  
        FOR start FROM 2 TO integer part of square root of n INCREMENTED BY 1  
            IF is_prime at index start IS True  
                FOR multiple FROM start times start TO less than n IN STEPS OF start  
                    SET is_prime at index multiple TO False  
                END FOR  
            END IF  
        END FOR  
  
        SET count TO sum of all True values in is_prime  
        RETURN count  
    END FUNCTION  
END CLASS
