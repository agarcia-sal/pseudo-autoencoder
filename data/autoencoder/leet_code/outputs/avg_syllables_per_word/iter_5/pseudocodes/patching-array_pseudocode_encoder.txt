CLASS Solution  
    FUNCTION minPatches(nums, n)  
        SET missing TO 1  
        SET patches TO 0  
        SET i TO 0  
        
        WHILE missing IS LESS THAN OR EQUAL TO n  
            IF i IS LESS THAN LENGTH OF nums AND nums AT INDEX i IS LESS THAN OR EQUAL TO missing  
                SET missing TO missing PLUS nums AT INDEX i  
                INCREMENT i BY 1  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY 1  
        RETURN patches  
    END FUNCTION  
END CLASS
