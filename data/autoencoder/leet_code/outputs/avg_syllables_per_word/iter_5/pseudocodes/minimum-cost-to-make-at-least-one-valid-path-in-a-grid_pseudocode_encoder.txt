CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT INDEX 0  
        SET directions TO LIST OF (0, 1), (0, -1), (1, 0), (-1, 0)  
        SET costs TO A TWO-DIMENSIONAL LIST OF size m BY n FILLED WITH infinity  
        SET costs AT INDEX 0 AT INDEX 0 TO 0  
        
        SET pq TO LIST CONTAINING TUPLE OF 0, 0, 0  
        SET visited TO EMPTY SET  
        
        WHILE pq IS NOT EMPTY  
            EXTRACT cost, i, j FROM REMOVING THE SMALLEST ELEMENT IN pq  
            
            IF (i, j) IS IN visited  
                CONTINUE TO NEXT ITERATION  
            ADD (i, j) TO visited  
            
            IF i EQUALS m MINUS 1 AND j EQUALS n MINUS 1  
                RETURN cost  
            
            FOR EACH k AND (di, dj) IN ENUMERATION OF directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO 0 AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO 0 AND nj IS LESS THAN n  
                    SET new_cost TO cost PLUS (1 IF k PLUS 1 IS NOT EQUAL TO grid AT i AT j ELSE 0)  
                    IF new_cost IS LESS THAN costs AT ni AT nj  
                        SET costs AT ni AT nj TO new_cost  
                        ADD (new_cost, ni, nj) TO pq  
        RETURN -1  
    END FUNCTION  
END CLASS
