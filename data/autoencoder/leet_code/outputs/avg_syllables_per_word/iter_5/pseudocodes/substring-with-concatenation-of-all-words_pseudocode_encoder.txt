CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s IS empty OR words IS empty OR words AT index 0 IS empty  
            RETURN empty list  
        SET word_length TO LENGTH OF words AT index 0  
        SET num_words TO LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO COUNTER OF words  
        SET result TO empty list  
        FOR i FROM 0 TO LENGTH OF s MINUS total_length PLUS 1  
            SET seen TO EMPTY COUNTER  
            FOR j FROM i TO i PLUS total_length STEP word_length  
                SET word TO SUBSTRING OF s FROM j TO j PLUS word_length  
                INCREMENT seen[word] BY 1  
                IF seen[word] IS GREATER THAN word_count[word]  
                    BREAK INNER LOOP  
            ELSE  
                APPEND i TO result  
        RETURN result  
    END FUNCTION  
END CLASS
