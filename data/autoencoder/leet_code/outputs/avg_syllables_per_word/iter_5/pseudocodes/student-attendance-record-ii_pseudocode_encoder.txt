CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS A THREE-DIMENSIONAL LIST OF SIZE (n PLUS 1) BY 2 BY 3 FILLED WITH ZEROES  
        SET dp AT INDEX 0, 0, 0 TO 1  
        
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 0 TO 1 INCLUSIVE  
                FOR k FROM 0 TO 2 INCLUSIVE  
                    SET dp AT INDEX i, j, 0 TO dp AT i, j, 0 PLUS dp AT i MINUS 1, j, k MODULO MOD  
                    IF j EQUALS 0 THEN  
                        SET dp AT INDEX i, 1, 0 TO dp AT i, 1, 0 PLUS dp AT i MINUS 1, 0, k MODULO MOD  
                    END IF  
                    IF k LESS THAN 2 THEN  
                        SET dp AT INDEX i, j, k PLUS 1 TO dp AT i, j, k PLUS 1 PLUS dp AT i MINUS 1, j, k MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO 0  
        FOR j FROM 0 TO 1 INCLUSIVE  
            FOR k FROM 0 TO 2 INCLUSIVE  
                SET result TO result PLUS dp AT n, j, k MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
