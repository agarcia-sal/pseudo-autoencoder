CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO 0  
        SET right TO LENGTH OF numbers MINUS 1  
  
        WHILE left LESS THAN right  
            SET current_sum TO numbers AT left PLUS numbers AT right  
            IF current_sum EQUALS target  
                RETURN LIST OF left PLUS 1 AND right PLUS 1  
            ELSE IF current_sum LESS THAN target  
                INCREMENT left BY 1  
            ELSE  
                DECREMENT right BY 1  
    END FUNCTION  
END CLASS
