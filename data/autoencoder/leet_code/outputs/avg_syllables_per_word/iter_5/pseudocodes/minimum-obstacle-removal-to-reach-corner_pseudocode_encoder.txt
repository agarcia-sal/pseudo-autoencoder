CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT index zero  
        SET directions TO list containing (0, 1), (1, 0), (0, -1), (-1, 0)  
        SET min_heap TO list containing tuple (0, 0, 0) representing cost zero at position (0, 0)  
        SET visited TO set containing tuple (0, 0)  
        
        WHILE min_heap is not empty  
            REMOVE the element with smallest cost FROM min_heap and unpack INTO cost, x, y  
            
            IF x EQUALS m MINUS 1 AND y EQUALS n MINUS 1 THEN  
                RETURN cost  
            END IF  
            
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND tuple (nx, ny) NOT IN visited THEN  
                    ADD tuple (nx, ny) TO visited  
                    SET new_cost TO cost PLUS grid AT nx AND ny  
                    ADD tuple (new_cost, nx, ny) TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
