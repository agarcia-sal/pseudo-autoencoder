CLASS Solution  
    FUNCTION minDistance(houses, k)  
        SORT houses IN ASCENDING ORDER  
        SET n TO LENGTH OF houses  
        
        FUNCTION compute_cost_matrix()  
            CREATE cost AS TWO-DIMENSIONAL LIST OF SIZE n BY n INITIALIZED WITH ZEROES  
            FOR i FROM n MINUS 1 DOWNTO 0  
                FOR j FROM i PLUS 1 TO n MINUS 1  
                    SET cost[i][j] TO cost[i PLUS 1][j MINUS 1] PLUS houses[j] MINUS houses[i]  
            RETURN cost  
        END FUNCTION  
        
        SET cost TO compute_cost_matrix()  
        
        CREATE dp AS TWO-DIMENSIONAL LIST OF SIZE (n PLUS 1) BY (k PLUS 1) WITH ALL VALUES SET TO INFINITY  
        SET dp[0][0] TO 0  
        
        FOR i FROM 1 TO n  
            FOR j FROM 1 TO k  
                FOR p FROM 0 TO i MINUS 1  
                    SET dp[i][j] TO THE MINIMUM OF dp[i][j] AND dp[p][j MINUS 1] PLUS cost[p][i MINUS 1]  
        RETURN dp[n][k]  
    END FUNCTION  
END CLASS
