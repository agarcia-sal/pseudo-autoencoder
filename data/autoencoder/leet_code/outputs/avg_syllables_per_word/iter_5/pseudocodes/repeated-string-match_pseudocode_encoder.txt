CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET min_repeats TO smallest integer GREATER THAN OR EQUAL TO (length of b DIVIDED BY length of a)  
        IF b is substring of a repeated min_repeats times THEN  
            RETURN min_repeats  
        ELSE IF b is substring of a repeated (min_repeats PLUS 1) times THEN  
            RETURN min_repeats PLUS 1  
        ELSE  
            RETURN -1  
    END FUNCTION  
END CLASS
