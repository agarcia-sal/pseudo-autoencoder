CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO 0  
        SET current_sum TO 0  
        SET min_length TO infinity  
        
        FOR right FROM 0 TO LENGTH OF nums MINUS 1  
            INCREMENT current_sum BY nums AT right  
            
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO SMALLER OF min_length AND right MINUS left PLUS 1  
                DECREMENT current_sum BY nums AT left  
                INCREMENT left BY 1  
        
        IF min_length IS NOT infinity  
            RETURN min_length  
        ELSE  
            RETURN 0  
        END IF  
    END FUNCTION  
END CLASS
