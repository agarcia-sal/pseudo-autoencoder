CLASS Solution  
    FUNCTION kthGrammar(n, k)  
        IF n EQUALS 1 THEN  
            RETURN 0  
        END IF  
        
        SET length_of_previous_row TO 2 RAISED TO THE POWER OF (n MINUS 2)  
        
        IF k LESS THAN OR EQUAL TO length_of_previous_row THEN  
            RETURN CALL kthGrammar WITH (n MINUS 1, k)  
        ELSE  
            SET previous_symbol TO CALL kthGrammar WITH (n MINUS 1, k MINUS length_of_previous_row)  
            RETURN 1 MINUS previous_symbol  
        END IF  
    END FUNCTION  
END CLASS
