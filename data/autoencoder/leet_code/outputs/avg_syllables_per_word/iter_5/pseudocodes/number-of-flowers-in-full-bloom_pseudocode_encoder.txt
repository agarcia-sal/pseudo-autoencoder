CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of start times extracted from flowers  
        SET end_times TO sorted list of end times extracted from flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO position in start_times where person would be inserted to the right  
            SET ended TO position in end_times where person would be inserted to the left  
            APPEND started minus ended TO result  
        RETURN result  
    END FUNCTION  
END CLASS
