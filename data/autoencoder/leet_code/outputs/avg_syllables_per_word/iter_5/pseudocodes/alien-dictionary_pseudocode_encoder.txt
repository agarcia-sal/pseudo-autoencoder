CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO empty mapping from character to set of characters  
        SET indegree TO empty mapping from character to integer  
        SET all_chars TO set of all characters in concatenation of words  
        
        FOR i FROM 0 TO LENGTH OF words MINUS 2  
            SET word1 TO words at index i  
            SET word2 TO words at index i plus 1  
            SET min_length TO minimum of LENGTH OF word1 and LENGTH OF word2  
            SET found TO false  
            FOR j FROM 0 TO min_length MINUS 1  
                IF character at index j of word1 IS NOT EQUAL TO character at index j of word2 THEN  
                    IF character at index j of word2 IS NOT IN graph at character at index j of word1 THEN  
                        ADD character at index j of word2 TO graph at character at index j of word1  
                        INCREMENT indegree at character at index j of word2 BY 1  
                    SET found TO true  
                    BREAK out of inner loop  
            IF found IS false AND LENGTH OF word1 GREATER THAN LENGTH OF word2 THEN  
                RETURN empty string  
        
        SET queue TO list of characters in all_chars WHERE indegree at character IS EQUAL TO 0  
        SET result TO empty list  
        
        WHILE queue IS NOT empty  
            REMOVE first element from queue AND SET TO char  
            APPEND char TO result  
            FOR each neighbor IN graph at char  
                DECREMENT indegree at neighbor BY 1  
                IF indegree at neighbor IS EQUAL TO 0 THEN  
                    APPEND neighbor TO queue  
        
        IF LENGTH OF result IS NOT EQUAL TO LENGTH OF all_chars THEN  
            RETURN empty string  
        
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
