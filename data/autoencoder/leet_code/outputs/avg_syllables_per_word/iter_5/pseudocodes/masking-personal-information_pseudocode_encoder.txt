CLASS Solution  
    FUNCTION maskPII(s)  
        IF s CONTAINS '@' THEN  
            SPLIT s INTO local AND domain AT '@'  
            SET masked_local TO CONCATENATE first character of local, five asterisks, and last character of local  
            RETURN concatenation of masked_local converted to lowercase, '@', and domain converted to lowercase  
        ELSE  
            EXTRACT all digit characters from s INTO digits  
            SET local_number TO concatenation of three asterisks, hyphen, three asterisks, hyphen, and last four characters of digits  
            IF LENGTH OF digits EQUALS ten THEN  
                RETURN local_number  
            ELSE  
                SET country_code_length TO LENGTH OF digits MINUS ten  
                SET country_code TO concatenation of plus sign, country_code_length asterisks, and hyphen  
                RETURN concatenation of country_code and local_number  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
