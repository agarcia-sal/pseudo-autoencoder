CLASS Solution  
    FUNCTION integerReplacement(n)  
        SET count TO 0  
        WHILE n IS NOT EQUAL TO 1  
            IF n MODULO 2 IS EQUAL TO 0  
                SET n TO n DIVIDED BY 2  
            ELSE  
                IF n IS EQUAL TO 3  
                    DECREMENT n BY 1  
                ELSE IF (n PLUS 1) MODULO 4 IS EQUAL TO 0  
                    INCREMENT n BY 1  
                ELSE  
                    DECREMENT n BY 1  
            INCREMENT count BY 1  
        RETURN count  
    END FUNCTION  
END CLASS
