CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_of_t TO iterator OVER t  
            RETURN whether every character c IN s EXISTS IN iterator_of_t IN SEQUENCE  
        END FUNCTION  

        SORT strs BY descending LENGTH AND lexicographical ORDER  

        FOR each index i AND word IN strs  
            SET unique TO true  
            FOR each index j AND other IN strs  
                IF i IS NOT EQUAL TO j AND is_subsequence(word, other) IS true  
                    SET unique TO false  
                    BREAK inner loop  
            IF unique IS true  
                RETURN LENGTH OF word  
        RETURN negative one  
    END FUNCTION  
END CLASS
