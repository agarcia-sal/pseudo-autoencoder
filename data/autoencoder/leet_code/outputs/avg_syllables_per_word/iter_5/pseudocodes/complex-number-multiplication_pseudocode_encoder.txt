CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO SPLIT num BY '+'  
            SET real TO CONVERT parts AT INDEX 0 TO INTEGER  
            SET imaginary TO CONVERT SUBSTRING OF parts AT INDEX 1 WITHOUT LAST CHARACTER TO INTEGER  
            RETURN real, imaginary  
        END FUNCTION  
        
        SET real1, imag1 TO CALL parse_complex WITH num1  
        SET real2, imag2 TO CALL parse_complex WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN STRING REPRESENTATION OF real_part CONCATENATED WITH "+" CONCATENATED WITH STRING REPRESENTATION OF imaginary_part CONCATENATED WITH "i"  
    END FUNCTION  
END CLASS
