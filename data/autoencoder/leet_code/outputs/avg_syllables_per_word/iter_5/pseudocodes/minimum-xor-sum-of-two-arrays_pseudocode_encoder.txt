CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        INITIALIZE dp AS LIST OF SIZE 2 TO THE POWER n WITH ALL VALUES SET TO INFINITY  
        SET dp AT INDEX 0 TO 0  
        
        FOR mask FROM 0 TO 2 TO THE POWER n MINUS 1  
            SET bit_count TO NUMBER OF ONES IN BINARY REPRESENTATION OF mask  
            FOR j FROM 0 TO n MINUS 1  
                IF mask HAS THE j-th BIT SET  
                    SET dp AT INDEX mask TO MINIMUM OF dp AT INDEX mask AND dp AT INDEX (mask WITH j-th BIT UNSET) PLUS XOR OF nums1 AT INDEX (bit_count MINUS 1) AND nums2 AT INDEX j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp AT INDEX (2 TO THE POWER n MINUS 1)  
    END FUNCTION  
END CLASS
