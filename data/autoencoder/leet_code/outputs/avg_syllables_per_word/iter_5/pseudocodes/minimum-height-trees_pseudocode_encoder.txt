CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS 1 THEN  
            RETURN list containing 0  
        END IF  
        
        SET graph TO a new mapping with default empty list  
        SET degree TO a list of zeros with length n  
        
        FOR each pair u, v IN edges DO  
            APPEND v TO graph at key u  
            APPEND u TO graph at key v  
            INCREMENT degree at index u BY 1  
            INCREMENT degree at index v BY 1  
        END FOR  
        
        SET leaves TO a new queue containing all indices i IN range from 0 to n MINUS 1 WHERE degree at i EQUALS 1  
        
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN 2 DO  
            SET leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR leaves_count times DO  
                REMOVE the front leaf FROM leaves  
                FOR each neighbor IN graph at key leaf DO  
                    DECREMENT degree at neighbor BY 1  
                    IF degree at neighbor EQUALS 1 THEN  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN the list form of leaves  
    END FUNCTION  
END CLASS
