CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing n  
        SET length TO number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS 1 DOWN TO 1 STEP MINUS 1  
            IF digit at position i is less than digit at position i MINUS 1 THEN  
                SET digit at position i MINUS 1 TO string representation of integer value of digit at position i MINUS 1 MINUS 1  
                SET marker TO i  
        FOR i FROM marker TO length MINUS 1  
            SET digit at position i TO character '9'  
        SET result TO integer formed by joining all characters in digits  
        RETURN result  
    END FUNCTION  
END CLASS
