CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value that is the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode with value that is the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode with value that is the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue containing root  
        SET left_to_right TO True  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty double ended queue  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                IF left_to_right EQUALS True  
                    APPEND node.val TO current_level  
                ELSE  
                    APPEND node.val TO the front of current_level  
                END IF  
                IF node.left NOT EQUALS none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND a list converted from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
