CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode where val EQUALS element at position one of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode where val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode where val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree where first parameter IS p.left AND second parameter IS q.left AND is_same_tree where first parameter IS p.right AND second parameter IS q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  

        FUNCTION is_leaf(node)  
            RETURN node.left IS none AND node.right IS none  
        END FUNCTION  

        SET left_boundary TO empty list  
        SET current TO root.left  
        WHILE current IS NOT none  
            IF NOT is_leaf where parameter IS current  
                APPEND current.val TO left_boundary  
            END IF  
            IF current.left IS NOT none  
                SET current TO current.left  
            ELSE  
                SET current TO current.right  
            END IF  
        END WHILE  

        SET right_boundary TO empty list  
        SET current TO root.right  
        WHILE current IS NOT none  
            IF NOT is_leaf where parameter IS current  
                APPEND current.val TO right_boundary  
            END IF  
            IF current.right IS NOT none  
                SET current TO current.right  
            ELSE  
                SET current TO current.left  
            END IF  
        END WHILE  

        SET leaves TO empty list  
        FUNCTION collect_leaves(node)  
            IF node IS none  
                RETURN  
            END IF  
            IF is_leaf where parameter IS node  
                APPEND node.val TO leaves  
            END IF  
            CALL collect_leaves where parameter IS node.left  
            CALL collect_leaves where parameter IS node.right  
        END FUNCTION  

        CALL collect_leaves where parameter IS root  

        IF is_leaf where parameter IS root  
            SET leaves TO empty list  
        END IF  

        RETURN list consisting of root.val followed by elements of left_boundary followed by elements of leaves followed by elements of right_boundary in reverse order  
    END FUNCTION  
END CLASS
