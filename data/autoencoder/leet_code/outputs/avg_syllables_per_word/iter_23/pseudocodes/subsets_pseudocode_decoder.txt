CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN a list containing an empty list  
        END IF  
        
        SET subsets_without_first TO the RESULT OF calling subsets WITH the ELEMENTS OF nums FROM position two TO the END  
        
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list consisting OF the ELEMENT at position one of nums FOLLOWED BY all ELEMENTS OF subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        
        RETURN the concatenation OF subsets_without_first FOLLOWED BY subsets_with_first  
    END FUNCTION  
END CLASS
