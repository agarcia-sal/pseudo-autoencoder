CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position one of nums MINUS twenty four LESS THAN one multiplied BY ten to the POWER of negative six  
            END IF  
            FOR index i FROM one TO the LENGTH OF nums  
                FOR index j FROM one TO the LENGTH OF nums  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY list  
                        FOR index k FROM one TO the LENGTH OF nums  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        FOR each operation op IN the list containing true division multiplication addition subtraction  
                            IF (op EQUALS addition OR op EQUALS multiplication) AND j GREATER THAN i  
                                CONTINUE to the next iteration  
                            END IF  
                            IF op NOT EQUALS true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying operation op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN false  
        END FUNCTION  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
