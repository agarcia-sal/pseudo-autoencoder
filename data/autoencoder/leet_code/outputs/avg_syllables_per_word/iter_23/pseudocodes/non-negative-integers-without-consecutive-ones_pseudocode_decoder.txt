CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET bin_representation TO the binary form of n starting from the second character to the end  
        SET the_length TO the number of characters in bin_representation  
        CREATE a list dp with the_length plus one elements all set to zero  
        SET the first element of dp to one  
        SET the second element of dp to two  
        FOR each index i FROM two TO the_length inclusive  
            SET the element at position i of dp TO the element at position i MINUS one of dp PLUS the element at position i MINUS two of dp  
        END FOR  
        SET result TO zero  
        SET prev_bit TO zero  
        FOR each index i FROM zero TO the_length MINUS one  
            IF the element at position i of bin_representation EQUALS one  
                INCREMENT result BY the element at position the_length MINUS i MINUS one of dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    EXIT the loop  
                END IF  
            END IF  
            IF the element at position i of bin_representation EQUALS one  
                SET prev_bit TO one  
            ELSE  
                SET prev_bit TO zero  
            END IF  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
