CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO an empty mapping with default zero for each key  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring of s from position index TO position index PLUS ten MINUS one  
            INCREMENT the value for sequence IN sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence AND each count IN the items of sequence_count  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
