CLASS Solution  
    FUNCTION getFactors(n)  
        FUNCTION backtrack(start, target)  
            FOR i FROM start TO the integer part of the square root of target PLUS one  
                IF the remainder of target DIVIDED BY i EQUALS zero  
                    APPEND the concatenation of path FOLLOWED BY i FOLLOWED BY target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION
