CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            SET difference_a_zero TO element at position zero of a MINUS element at position zero of o  
            SET difference_b_one TO element at position one of b MINUS element at position one of o  
            SET term_one TO difference_a_zero MULTIPLIED BY difference_b_one  
            SET difference_a_one TO element at position one of a MINUS element at position one of o  
            SET difference_b_zero TO element at position zero of b MINUS element at position zero of o  
            SET term_two TO difference_a_one MULTIPLIED BY difference_b_zero  
            RETURN term_one MINUS term_two  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position position i PLUS one MODULO n of points  
            SET b TO element at position position i PLUS two MODULO n of points  
            SET cross TO cross_product with parameters o, a, b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
