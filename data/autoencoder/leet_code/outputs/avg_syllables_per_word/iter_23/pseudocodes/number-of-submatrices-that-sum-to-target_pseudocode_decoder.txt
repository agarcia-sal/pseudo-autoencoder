CLASS Solution  
    FUNCTION numSubmatrixSumTarget(matrix, target)  
        FUNCTION subarraySum(nums, target)  
            SET count TO zero  
            SET seen_sums TO a mapping from number to number with default zero  
            SET cumulative_sum TO zero  
            FOR each num IN nums  
                SET cumulative_sum TO cumulative_sum PLUS num  
                IF cumulative_sum EQUALS target THEN  
                    INCREMENT count BY one  
                END IF  
                IF cumulative_sum MINUS target EXISTS IN seen_sums THEN  
                    INCREMENT count BY the value at cumulative_sum MINUS target key in seen_sums  
                END IF  
                INCREMENT the value at cumulative_sum key in seen_sums BY one  
            END FOR  
            RETURN count  
        END FUNCTION  
      
        SET rows TO the length of matrix  
        SET cols TO the length of the element at position zero of matrix  
        SET total_count TO zero  
      
        FOR top FROM zero TO rows MINUS one  
            SET current_cols TO a list of zero repeated cols times  
            FOR bottom FROM top TO rows MINUS one  
                FOR col FROM zero TO cols MINUS one  
                    SET element at position col of current_cols TO the element at position col of current_cols PLUS the element at position col of the element at position bottom of matrix  
                END FOR  
                SET total_count TO total_count PLUS the result of calling subarraySum with current_cols and target  
            END FOR  
        END FOR  
      
        RETURN total_count  
    END FUNCTION  
END CLASS
