CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO list of elements converted to integer of splitting ip by dot delimiter  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two raised to the power of eight PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  
        
        FUNCTION int_to_ip(num)  
            SET segments TO empty list  
            FOR i FROM three TO zero DECREMENT BY one  
                SET segment TO remainder of num DIVIDED BY two hundred fifty-six after integer division of num divided by two raised to the power of eight MULTIPLIED BY i  
                CONVERT segment TO string  
                APPEND this string TO segments  
            END FOR  
            JOIN all elements in segments with dot delimiter INTO a single string  
            RETURN this string  
        END FUNCTION  
        
        FUNCTION largest_cidr_size(start, n)  
            IF remainder of start DIVIDED BY two EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x IS LESS THAN OR EQUAL TO n AND remainder of start DIVIDED BY x EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  
        
        SET result TO empty list  
        SET start_ip TO ip_to_int(ip)  
        
        WHILE n IS GREATER THAN zero  
            SET size TO largest_cidr_size(start_ip, n)  
            SET prefix_length TO thirty-two MINUS the quantity of bit length of size MINUS one  
            SET cidr_notation TO concatenation of int_to_ip(start_ip), forward slash character, and string representation of prefix_length  
            APPEND cidr_notation TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
