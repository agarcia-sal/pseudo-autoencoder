CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line EQUALS true  
                RETURN the concatenation of all elements in line SEPARATED BY a single space PLUS the repetition of a single space character TIMES the expression maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO the integer division of total_spaces BY num_of_gaps  
                SET extra_spaces TO the modulo of total_spaces BY num_of_gaps  
                FOR index i FROM zero TO num_of_gaps MINUS one  
                    IF index i LESS THAN extra_spaces  
                        APPEND the repetition of a single space character TIMES the sum of spaces_between_words PLUS one TO the element at position index i of line  
                    ELSE  
                        APPEND the repetition of a single space character TIMES spaces_between_words TO the element at position index i of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements of line WITH NO SEPARATOR  
            END IF  
        END FUNCTION  
  
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
  
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of calling justify_line WITH arguments line num_of_letters AND false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
  
        APPEND the result of calling justify_line WITH arguments line num_of_letters AND true TO result  
  
        RETURN result  
    END FUNCTION  
END CLASS
