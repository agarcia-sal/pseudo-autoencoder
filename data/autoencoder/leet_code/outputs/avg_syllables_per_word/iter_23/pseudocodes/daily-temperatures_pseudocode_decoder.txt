CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a LIST OF zero VALUES WITH LENGTH n  
        SET stack TO an EMPTY LIST  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the ELEMENT AT position i OF temperatures IS GREATER THAN the ELEMENT AT the ELEMENT AT the LAST POSITION OF stack OF temperatures  
                SET prev_index TO the ELEMENT REMOVED FROM the LAST POSITION OF stack  
                SET the ELEMENT AT position prev_index OF answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
