CLASS Solution  
    FUNCTION groupStrings WITH parameter strings  
        FUNCTION get_canonical_form WITH parameter s  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the position of character c IN the alphabet MINUS the position of the first character of s IN the alphabet  
                IF difference IS LESS THAN zero  
                    SET difference TO difference PLUS twenty six  
                END IF  
                APPEND the string representation of difference TO canonical_form  
            END FOR  
            RETURN the concatenation of elements of canonical_form joined by the string representing a comma  
        END FUNCTION  
        
        SET groups TO a new dictionary with default value empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form WITH argument s  
            APPEND s TO the list associated with canonical_form in groups  
        END FOR  
        
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
