CLASS Solution  
    FUNCTION numMatchingSubseq(s parameter s AS string words parameter words AS list OF string) RETURNS integer  
        SET waiting TO an empty dictionary mapping each unique character in s TO an empty double ended queue  
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element from it  
            IF first_letter is a key in waiting  
                APPEND it TO the queue corresponding to first_letter in waiting  
            END IF  
        END FOR  
        SET match_count TO zero  
        FOR each character c IN s  
            IF c is a key in waiting  
                SET current_iterators TO the queue corresponding to c in waiting  
                SET waiting at c TO an empty double ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next element from it if it exists otherwise to no value  
                    IF next_char has no value  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is a key in waiting  
                            APPEND it TO the queue corresponding to next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN match_count  
    END FUNCTION  
END CLASS
