CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode element at position one of values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO NEW TreeNode element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO NEW TreeNode element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left q.left) AND is_same_tree(p.right q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self root targetSum)  
        FUNCTION helper(node current_sum prefix_sums)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS node.val  
            SET paths_to_here TO value FROM prefix_sums WHERE key EQUALS current_sum MINUS targetSum OR zero IF NOT FOUND  
            SET prefix_sums AT KEY current_sum TO value FROM prefix_sums WHERE key EQUALS current_sum OR zero PLUS one  
            SET paths_in_left TO helper(node.left current_sum prefix_sums)  
            SET paths_in_right TO helper(node.right current_sum prefix_sums)  
            DECREMENT prefix_sums AT KEY current_sum BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO dictionary with key zero AND value one  
        RETURN helper(root zero prefix_sums)  
    END FUNCTION  
END CLASS
