CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            ASSIGN to node.left a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            ASSIGN to node.right a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS None THEN  
                RETURN zero, zero, positive infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO dfs(the left attribute of node)  
            SET right_no_camera, right_monitored, right_camera TO dfs(the right attribute of node)  
            SET no_camera TO the sum of left_monitored PLUS right_monitored  
            SET monitored TO the minimum of the following three values:  
                left_camera PLUS the minimum of right_monitored AND right_camera  
                right_camera PLUS the minimum of left_monitored AND left_camera  
                left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum of left_no_camera, left_monitored, and left_camera PLUS the minimum of right_no_camera, right_monitored, and right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO dfs(root)  
        RETURN the minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
