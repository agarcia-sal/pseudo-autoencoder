CLASS Solution  
    FUNCTION searchMatrix(matrix parameter LIST OF LIST OF INTEGER target parameter INTEGER) RETURNS BOOLEAN  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO integer division of left PLUS right BY two  
            SET mid_value TO element at position integer division of mid BY n of matrix at position mid MODULO n  
            IF mid_value EQUALS target  
                RETURN true  
            ELSE IF mid_value LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
