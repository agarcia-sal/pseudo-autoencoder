CLASS Solution  
    FUNCTION reachNumber(target)  
        SET target TO the absolute value of target  
        SET k TO zero  
        SET sum_steps TO zero  
        WHILE sum_steps LESS THAN target  
            INCREMENT k BY one  
            INCREMENT sum_steps BY k  
        END WHILE  
        IF the remainder of the difference between sum_steps and target when divided by two EQUALS zero  
            RETURN k  
        ELSE  
            IF the remainder of k when divided by two EQUALS zero  
                RETURN k PLUS one  
            ELSE  
                RETURN k PLUS two  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
