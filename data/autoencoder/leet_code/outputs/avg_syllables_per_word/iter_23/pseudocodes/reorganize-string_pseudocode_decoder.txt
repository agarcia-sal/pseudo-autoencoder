CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of each character in s  
        SET max_heap TO list of pairs of negative frequency and corresponding character for each character and frequency in count  
        TRANSFORM max_heap INTO a heap data structure based on the first element of each pair  
        
        SET previous character TO None  
        SET previous frequency TO zero  
        SET result to an empty list  
        
        WHILE max_heap is not empty OR previous frequency is less than zero DO  
            IF max_heap is empty AND previous frequency is less than zero THEN  
                RETURN an empty string  
            END IF  
            
            SET frequency and character TO remove the element with the smallest first value from max_heap  
            APPEND character TO result  
            
            INCREMENT frequency BY one  
            
            IF previous frequency is less than zero THEN  
                ADD the pair of previous frequency and previous character back into max_heap  
            END IF  
            
            SET previous character TO character  
            SET previous frequency TO frequency  
        END WHILE  
        
        RETURN concatenation of all elements of result  
    END FUNCTION  
END CLASS
