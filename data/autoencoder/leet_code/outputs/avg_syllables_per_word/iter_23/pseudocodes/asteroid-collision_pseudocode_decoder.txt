CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO an empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid is LESS THAN zero AND the element at the last position of stack is GREATER THAN zero  
                IF the element at the last position of stack is LESS THAN the negation of asteroid  
                    REMOVE the element at the last position FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF the element at the last position of stack EQUALS the negation of asteroid  
                    REMOVE the element at the last position FROM stack  
                END IF  
                BREAK out of the WHILE loop  
            END WHILE  
            IF the WHILE loop was not broken by a break statement  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
