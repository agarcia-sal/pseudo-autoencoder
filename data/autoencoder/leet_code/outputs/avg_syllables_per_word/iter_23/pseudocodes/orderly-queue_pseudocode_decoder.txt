CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the concatenation of the characters of s arranged in ascending lexicographical order  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the substring from position i TO the last position of s PLUS the substring from position zero TO position i MINUS one of s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
