CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY the LENGTH OF word AND THEN BY word IN LEXICOGRAPHICAL ORDER  
        SET valid_words TO an empty SET  
        SET longest_word TO an empty STRING  
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the SUBSTRING of word from the FIRST CHARACTER TO the SECOND TO LAST CHARACTER IS IN valid_words THEN  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word THEN  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
