CLASS Solution  
    FUNCTION minDistance(height param_height width param_width tree param_tree squirrel param_squirrel nuts param_nuts)  
        FUNCTION distance(p1 param_p1 p2 param_p2)  
            SET horizontal_difference TO the absolute value of the value at position zero of p1 MINUS the value at position zero of p2  
            SET vertical_difference TO the absolute value of the value at position one of p1 MINUS the value at position one of p2  
            SET total_difference TO horizontal_difference PLUS vertical_difference  
            RETURN total_difference  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance called with nut and tree  
            SET first_nut_distance TO distance called with squirrel and nut PLUS distance called with nut and tree  
            SET later_nut_distance TO two MULTIPLIED BY distance called with nut and tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            SET max_save TO the greater of max_save AND save  
        END FOR  
        
        SET result_distance TO total_distance MINUS max_save  
        RETURN result_distance  
    END FUNCTION  
END CLASS
