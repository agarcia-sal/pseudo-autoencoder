CLASS Solution  
    FUNCTION minKBitFlips(nums k)  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a list of zeros with LENGTH n PLUS one  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET flip_state TO the EXCLUSIVE OR OF flip_state AND element at position i of flipped  
            
            IF element at position i of nums EQUALS flip_state  
                IF i PLUS k GREATER THAN n  
                    RETURN minus one  
                END IF  
                SET flip_state TO the EXCLUSIVE OR OF flip_state AND one  
                SET element at position i PLUS k of flipped TO the EXCLUSIVE OR OF element at position i PLUS k of flipped AND one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
