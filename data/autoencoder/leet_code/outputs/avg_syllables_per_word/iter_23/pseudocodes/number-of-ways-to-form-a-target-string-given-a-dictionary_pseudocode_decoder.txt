CLASS Solution  
    FUNCTION numWays(words PARAMETER, target PARAMETER)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF the element at position zero of words  
        SET m TO the LENGTH OF target  
        
        SET freq TO an empty list  
        FOR position FROM zero TO n MINUS one  
            APPEND an empty mapping from character to integer zero TO freq  
        END FOR  
        
        FOR each word IN words  
            SET j TO zero  
            WHILE j LESS THAN n  
                INCREMENT the integer value associated with the element at position j of word IN the mapping at position j of freq BY one  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        
        SET dp TO an empty list  
        FOR i FROM zero TO m  
            SET current_row TO an empty list  
            FOR j FROM zero TO n  
                APPEND zero TO current_row  
            END FOR  
            APPEND current_row TO dp  
        END FOR  
        
        FOR j FROM zero TO n  
            SET the element at position j of the element at position zero of dp TO one  
        END FOR  
        
        SET i TO one  
        WHILE i LESS THAN OR EQUAL TO m  
            SET j TO one  
            WHILE j LESS THAN OR EQUAL TO n  
                SET the element at position j of the element at position i of dp TO the element at position j MINUS one of the element at position i of dp  
                
                SET current_char TO the element at position i MINUS one of target  
                SET current_freq TO the integer value associated with current_char IN the mapping at position j MINUS one of freq  
                
                IF current_freq GREATER THAN zero THEN  
                    INCREMENT the element at position j of the element at position i of dp BY the product of the element at position j MINUS one of the element at position i MINUS one of dp AND current_freq  
                    SET the element at position j of the element at position i of dp TO the element at position j of the element at position i of dp MODULO MOD  
                END IF  
                
                INCREMENT j BY one  
            END WHILE  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN the element at position n of the element at position m of dp  
    END FUNCTION  
END CLASS
