CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        WHILE left LESS THAN right  
            SET minimum_height TO height AT POSITION left  
            IF height AT POSITION right LESS THAN minimum_height  
                SET minimum_height TO height AT POSITION right  
            END IF  
            SET width TO right MINUS left  
            SET current_area TO minimum_height MULTIPLIED BY width  
            IF current_area GREATER THAN max_area  
                SET max_area TO current_area  
            END IF  
            IF height AT POSITION left LESS THAN height AT POSITION right  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN max_area  
    END FUNCTION  
END CLASS
