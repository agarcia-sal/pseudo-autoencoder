CLASS ListNode  
    FUNCTION __init__ (val EQUALS zero, next EQUALS none)  
        SET the member val TO val  
        SET the member next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node (values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode with val EQUALS element at position one of values  
    SET p TO head  
    FOR each val IN values starting from position two to the last element  
        SET node TO new instance of ListNode with val EQUALS val  
        SET member next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list (p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN logical condition that member val of p1 EQUALS member val of p2 AND the result of calling is_same_list with parameters member next of p1 and member next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates (head)  
        SET dummy TO new instance of ListNode with val EQUALS zero AND next EQUALS head  
        SET prev TO dummy  
        WHILE head holds a non none value  
            IF member next of head holds a non none value AND member val of head EQUALS member val of member next of head  
                WHILE member next of head holds a non none value AND member val of head EQUALS member val of member next of head  
                    SET head TO member next of head  
                END WHILE  
                SET member next of prev TO member next of head  
            ELSE  
                SET prev TO member next of prev  
            END IF  
            SET head TO member next of head  
        END WHILE  
        RETURN member next of dummy  
    END FUNCTION  
END CLASS
