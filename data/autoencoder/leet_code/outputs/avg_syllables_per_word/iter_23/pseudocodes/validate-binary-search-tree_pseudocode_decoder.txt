CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the collection values is empty  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the collection queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is no value AND q is no value  
        RETURN true  
    ELSE IF p is no value OR q is no value  
        RETURN false  
    ELSE IF the value of p.val NOT EQUALS the value of q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST with parameters self root  
        FUNCTION validate with parameters node low high  
            IF node is no value  
                RETURN true  
            END IF  
            IF the value of node.val is LESS THAN OR EQUAL TO low OR the value of node.val is GREATER THAN OR EQUAL TO high  
                RETURN false  
            END IF  
            RETURN the result of validate with arguments node.left low node.val AND the result of validate with arguments node.right node.val high  
        END FUNCTION  
        RETURN the result of validate with argument root and the lowest possible value and the highest possible value  
    END FUNCTION  
END CLASS
