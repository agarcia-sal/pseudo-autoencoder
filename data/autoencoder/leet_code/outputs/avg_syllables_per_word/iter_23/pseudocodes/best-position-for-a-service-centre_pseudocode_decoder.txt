CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET n TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each element IN positions  
            LET x1 BE the element at position one of the element  
            LET y1 BE the element at position two of the element  
            INCREMENT x BY x1  
            INCREMENT y BY y1  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO zero POINT nine nine nine  
        SET eps TO zero POINT zero zero zero zero zero one  
        SET alpha TO zero POINT five  
        WHILE true holds  
            SET grad_x TO zero  
            SET grad_y TO zero  
            SET dist TO zero  
            FOR each element IN positions  
                LET x1 BE the element at position one of the element  
                LET y1 BE the element at position two of the element  
                SET a TO x MINUS x1  
                SET b TO y MINUS y1  
                SET c TO the square root of a MULTIPLIED BY a PLUS b MULTIPLIED BY b  
                INCREMENT grad_x BY a DIVIDED BY the sum of c PLUS zero POINT zero zero zero zero zero zero zero one  
                INCREMENT grad_y BY b DIVIDED BY the sum of c PLUS zero POINT zero zero zero zero zero zero zero one  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            DECREMENT x BY dx  
            DECREMENT y BY dy  
            SET alpha TO alpha MULTIPLIED BY decay  
            IF the absolute value of dx LESS THAN OR EQUAL TO eps AND the absolute value of dy LESS THAN OR EQUAL TO eps  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
