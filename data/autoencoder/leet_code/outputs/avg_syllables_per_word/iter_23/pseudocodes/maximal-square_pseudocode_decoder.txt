CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        INITIALIZE dp AS a two dimensional list of zero with the number of rows equal to m PLUS one and the number of columns equal to n PLUS one  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix THEN element at position j MINUS one EQUALS the character one  
                    SET dp element at position i and position j TO the MINIMUM OF dp element at position i MINUS one and position j AND dp element at position i and position j MINUS one AND dp element at position i MINUS one and position j MINUS one PLUS one  
                    SET max_side TO the MAXIMUM OF max_side AND dp element at position i and position j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
