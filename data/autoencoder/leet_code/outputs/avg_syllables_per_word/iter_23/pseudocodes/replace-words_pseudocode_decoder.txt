CLASS Solution  
    FUNCTION replaceWords WITH PARAMETERS dictionary sentence  
        SORT dictionary BY LENGTH OF element FROM shortest TO longest  
        SET words TO the SPLIT OF sentence BY spaces  
        FUNCTION replace WITH PARAMETER word  
            FOR each root IN dictionary  
                IF word STARTS WITH root  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        SET replaced_words TO an empty list  
        FOR each word IN words  
            APPEND the RESULT OF replace WITH ARGUMENT word TO replaced_words  
        END FOR  
        SET result_sentence TO the JOIN OF replaced_words WITH spaces  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
