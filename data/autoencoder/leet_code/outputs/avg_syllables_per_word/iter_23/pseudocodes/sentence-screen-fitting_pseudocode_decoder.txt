CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements of sentence separated by single space PLUS a single space at the end  
        SET total_length TO the length of s  
        SET position TO zero  
        FOR each number from one TO rows  
            INCREMENT position BY cols  
            IF the element at position MODULO total_length of s EQUALS a space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at position MINUS one MODULO total_length of s NOT EQUALS a space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN the integer division of position BY total_length  
    END FUNCTION  
END CLASS
