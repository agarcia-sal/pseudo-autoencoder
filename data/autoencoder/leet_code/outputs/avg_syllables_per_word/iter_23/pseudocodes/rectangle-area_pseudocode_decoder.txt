CLASS Solution  
    FUNCTION computeArea(ax1 ay1 ax2 ay2 bx1 by1 bx2 by2)  
        SET area1 TO the difference of the element ax2 MINUS the element ax1 MULTIPLIED BY the difference of the element ay2 MINUS the element ay1  
        SET area2 TO the difference of the element bx2 MINUS the element bx1 MULTIPLIED BY the difference of the element by2 MINUS the element by1  
        SET overlap_width TO the greater value BETWEEN zero AND the difference of the smaller value BETWEEN ax2 AND bx2 MINUS the larger value BETWEEN ax1 AND bx1  
        SET overlap_height TO the greater value BETWEEN zero AND the difference of the smaller value BETWEEN ay2 AND by2 MINUS the larger value BETWEEN ay1 AND by1  
        SET overlap_area TO the product of overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
