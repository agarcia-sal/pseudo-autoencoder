CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each number in nums  
        SET dp TO a list of zeros with the length of two raised to the power of ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two raised to the power of the frequency of one in count  
                CONTINUE to the next iteration  
            END IF  
            
            IF num NOT IN the keys of prime_factors  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO the value at key num in prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM the length of dp MINUS one DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero  
                    SET element at position m OR mask of dp TO the sum of element at position m OR mask of dp and element at position m of dp MULTIPLIED BY the frequency of num in count MODULO the constant MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements from position one TO the end of dp MODULO the constant MOD  
        RETURN result  
    END FUNCTION  
END CLASS
