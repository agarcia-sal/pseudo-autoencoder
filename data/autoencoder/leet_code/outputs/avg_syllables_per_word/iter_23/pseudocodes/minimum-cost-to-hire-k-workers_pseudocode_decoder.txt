CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO sorted list of pairs where each pair is the ratio of the wage divided by the quality and the quality for each worker taken from corresponding positions of wage and quality lists  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair of ratio and quality_value IN workers  
            PUSH negative quality_value INTO max_heap  
            INCREMENT total_quality BY quality_value  
            IF the LENGTH of max_heap GREATER THAN k THEN  
                INCREMENT total_quality BY the value obtained by POPPING from max_heap  
            END IF  
            IF the LENGTH of max_heap EQUALS k THEN  
                SET candidate_cost TO total_quality MULTIPLIED BY ratio  
                IF candidate_cost LESS THAN min_cost THEN  
                    SET min_cost TO candidate_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
