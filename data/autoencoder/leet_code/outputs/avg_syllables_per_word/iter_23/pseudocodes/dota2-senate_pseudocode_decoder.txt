CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        IMPORT a double-ended queue data structure named deque from collections  
        
        SET radiant TO an empty deque  
        SET dire TO an empty deque  
        
        FOR each index i and element s IN the enumeration of senate  
            IF s EQUALS the character representing Radiant party  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant is not empty AND dire is not empty  
            REMOVE the element at the front of radiant and ASSIGN it TO r_index  
            REMOVE the element at the front of dire and ASSIGN it TO d_index  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant is not empty  
            RETURN the string representing Radiant party  
        ELSE  
            RETURN the string representing Dire party  
        END IF  
    END FUNCTION  
END CLASS
