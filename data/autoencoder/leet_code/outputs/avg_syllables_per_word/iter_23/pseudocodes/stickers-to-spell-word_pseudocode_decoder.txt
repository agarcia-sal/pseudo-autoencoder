CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET frequency_counters TO a new list  
        FOR each sticker IN stickers  
            APPEND frequency count of characters in sticker TO frequency_counters  
        END FOR  
        SORT frequency_counters by the negative of the maximum frequency count value in each counter  
        SET queue TO a new double ended queue containing the pair of target and zero  
        SET memo to a new map with key target and value zero  
        WHILE the queue holds any elements  
            SET current_target and used_stickers TO the elements dequeued from the front of the queue  
            FOR each sticker_frequency IN frequency_counters  
                IF the first character of current_target IS NOT IN the keys of sticker_frequency THEN  
                    CONTINUE to next iteration  
                END IF  
                SET remaining_frequency TO frequency count of characters in current_target  
                DECREMENT remaining_frequency by the values in sticker_frequency  
                SET new_target_string TO empty string  
                FOR each character IN characters sorted in ascending order of elements in remaining_frequency including repetitions  
                    APPEND character TO new_target_string  
                END FOR  
                IF new_target_string IS an empty string THEN  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target_string IS NOT present AS a key in memo THEN  
                    SET memo[new_target_string] TO used_stickers PLUS one  
                    APPEND the pair of new_target_string and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
