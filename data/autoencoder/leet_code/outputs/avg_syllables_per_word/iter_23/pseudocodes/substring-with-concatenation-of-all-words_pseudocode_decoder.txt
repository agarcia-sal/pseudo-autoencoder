CLASS Solution  
    FUNCTION findSubstring(s parameter words parameter)  
        IF s EQUALS nothing OR words EQUALS nothing OR the element at position zero of words EQUALS nothing THEN  
            RETURN an empty list  
        END IF  
        SET word_length TO the LENGTH OF the element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the FREQUENCY COUNTER OF words  
        SET result TO an empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an empty FREQUENCY COUNTER  
            FOR j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO the SUBSTRING OF s from position j TO position j PLUS word_length MINUS one  
                INCREMENT the count of word in seen BY one  
                IF the count of word in seen GREATER THAN the count of word in word_count THEN  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop did not break THEN  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
