CLASS Solution  
    FUNCTION minCost(n number, cuts list of numbers)  
        SET cuts TO sorted list of cuts with zero and n appended  
        
        FUNCTION dp(left index, right index)  
            IF right index MINUS left index LESS THAN OR EQUAL TO one  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i index FROM left index PLUS one TO right index MINUS one  
                SET cost TO element at position right index of cuts MINUS element at position left index of cuts PLUS dp called with left index AND i index PLUS dp called with i index AND right index  
                IF cost LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp called with zero AND length of cuts MINUS one  
    END FUNCTION  
END CLASS
