CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN no value  
    END IF  
    SET head TO new instance of ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from the second element to the last element  
        SET node TO new instance of ListNode with val  
        SET the element next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS no value AND p2 EQUALS no value THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value THEN  
        RETURN false  
    END IF  
    RETURN the result of the expression that the element val of p1 EQUALS the element val of p2 AND the result of the call to is_same_list with arguments the element next of p1 and the element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(head n)  
        SET dummy TO new instance of ListNode with zero  
        SET the element next of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the element next of first  
        END FOR  
        WHILE first NOT EQUAL TO no value  
            SET first TO the element next of first  
            SET second TO the element next of second  
        END WHILE  
        SET the element next of second TO the element next of the element next of second  
        RETURN the element next of dummy  
    END FUNCTION  
END CLASS
