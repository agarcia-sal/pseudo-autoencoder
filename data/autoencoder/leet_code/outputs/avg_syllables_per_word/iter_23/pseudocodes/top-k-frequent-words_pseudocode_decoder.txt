CLASS Solution  
    FUNCTION topKFrequent(words AS List of strings, k AS integer) RETURNS List of strings  
        SET count TO a new mapping from word to frequency counting each word in words  
        SET heap TO an empty list  
        FOR each word AND frequency IN the entries of count  
            APPEND a tuple consisting of the negative value of frequency AND the word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap data structure to maintain heap property  
        SET result TO an empty list  
        FOR iteration FROM one TO k  
            SET top_element TO the element extracted from the top of heap  
            APPEND the second element of top_element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
