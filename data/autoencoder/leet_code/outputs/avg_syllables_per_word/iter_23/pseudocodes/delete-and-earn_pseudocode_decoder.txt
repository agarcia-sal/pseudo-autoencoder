CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a mapping from each number in nums to the total quantity of that number  
        SET unique_nums TO the sequence of distinct numbers in count arranged in order from smallest to largest  
        SET dp TO a list of zeros with length equal to the length of unique_nums incremented by one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the quantity of that element in count  
        FOR i FROM two TO the length of unique_nums incremented by one  
            SET num TO the element at position i MINUS one of unique_nums  
            IF num EQUALS the element at position i MINUS two of unique_nums PLUS one  
                SET the element at position i of dp TO the greater value between the element at position i MINUS one of dp AND the element at position i MINUS two of dp PLUS num MULTIPLIED BY the quantity of num in count  
            ELSE  
                SET the element at position i of dp TO the element at position i MINUS one of dp PLUS num MULTIPLIED BY the quantity of num in count  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
