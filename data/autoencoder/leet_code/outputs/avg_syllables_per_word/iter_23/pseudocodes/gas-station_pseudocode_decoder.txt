CLASS Solution  
    FUNCTION canCompleteCircuit(gas gas_list, cost cost_list)  
        IF the SUM OF all elements IN gas_list LESS THAN the SUM OF all elements IN cost_list  
            RETURN the NEGATIVE of one  
        END IF  
        SET start TO zero  
        SET total_tank TO zero  
        FOR index i FROM zero TO the LENGTH OF gas_list MINUS one  
            SET total_tank TO total_tank PLUS the element at position i of gas_list MINUS the element at position i of cost_list  
            IF total_tank LESS THAN zero  
                SET start TO i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        RETURN start  
    END FUNCTION  
END CLASS
