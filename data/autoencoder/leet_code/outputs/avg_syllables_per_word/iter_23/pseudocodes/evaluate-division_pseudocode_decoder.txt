CLASS Solution  
    FUNCTION calcEquation(equations, values, queries)  
        SET graph TO a default dictionary where every new key maps to an empty dictionary  
        FOR each pair of elements a and b IN equations AND each corresponding value IN values  
            SET the element at key b of the dictionary at key a of graph TO value  
            SET the element at key a of the dictionary at key b of graph TO one DIVIDED BY value  
        END FOR  

        FUNCTION dfs(start, end, visited)  
            IF start NOT IN graph OR end NOT IN graph THEN  
                RETURN negative one  
            END IF  
            IF start EQUALS end THEN  
                RETURN one  
            END IF  
            ADD start TO visited  
            FOR each neighbor and weight IN the dictionary at key start of graph  
                IF neighbor NOT IN visited THEN  
                    SET result TO dfs(neighbor, end, visited)  
                    IF result NOT EQUALS negative one THEN  
                        RETURN weight MULTIPLIED BY result  
                    END IF  
                END IF  
            END FOR  
            REMOVE start FROM visited  
            RETURN negative one  
        END FUNCTION  

        SET results TO an empty list  
        FOR each query IN queries  
            SET start and end TO the elements OF query  
            SET visited TO an empty set  
            SET result TO dfs(start, end, visited)  
            APPEND result TO results  
        END FOR  
        RETURN results  
    END FUNCTION  
END CLASS
