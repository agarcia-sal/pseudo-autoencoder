CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing every occurrence of a plus sign in side with a hash sign followed by a plus sign and then replacing every occurrence of a minus sign in side with a hash sign followed by a minus sign and then splitting the resulting string by hash sign  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token EQUALS empty string THEN  
                    CONTINUE to next iteration  
                END IF  
                IF the character x is contained in token THEN  
                    IF token EQUALS the character x OR token EQUALS the plus sign followed by the character x THEN  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS the minus sign followed by the character x THEN  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET x_count TO x_count PLUS the integer value of the string formed by replacing the character x in token with empty string  
                    END IF  
                ELSE  
                    SET num_sum TO num_sum PLUS the integer value of token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  

        SET left_side AND right_side TO the two parts of equation obtained by splitting equation at the equals sign  
        SET left_x AND left_num TO the result of calling parse_side with left_side  
        SET right_x AND right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero THEN  
            IF total_num EQUALS zero THEN  
                RETURN Infinite solutions  
            ELSE  
                RETURN No solution  
            END IF  
        ELSE  
            RETURN the string formed by concatenating the character x the equals sign and the result of total_num integer divided by total_x  
        END IF  
    END FUNCTION  
END CLASS
