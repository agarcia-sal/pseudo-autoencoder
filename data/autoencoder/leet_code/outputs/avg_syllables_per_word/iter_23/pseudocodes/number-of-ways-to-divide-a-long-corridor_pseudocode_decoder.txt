CLASS Solution  
    FUNCTION numberOfWays WITH corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO an empty list  
        FOR each index AND character IN corridor  
            IF character EQUALS the uppercase letter corresponding to seat  
                APPEND index TO seats  
            END IF  
        END FOR  
        
        IF the length of seats MODULO two NOT EQUALS zero OR the length of seats LESS THAN two  
            RETURN zero  
        END IF  
        
        SET ways TO one  
        FOR index FROM two TO the length of seats MINUS one STEP two  
            SET difference TO the element at position index of seats MINUS the element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
