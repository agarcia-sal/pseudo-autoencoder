CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new empty mapping with the element zero associated with position negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each index i AND character char IN s by enumerating positions and elements  
            SET digit TO the integer value represented by the character char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR with the number one shifted left by digit positions  
            IF prefix_xor EXISTS AS A KEY IN prefix_map  
                SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the value associated with prefix_xor in prefix_map  
            END IF  
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR with the number one shifted left by j positions  
                IF flipped EXISTS AS A KEY IN prefix_map  
                    SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the value associated with flipped in prefix_map  
                END IF  
            END FOR  
            IF prefix_xor DOES NOT EXIST AS A KEY IN prefix_map  
                ASSIGN the value i TO the key prefix_xor IN prefix_map  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
