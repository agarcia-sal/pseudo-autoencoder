CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of size two RAISED TO the power of n where each element is positive infinity  
        SET element at position zero of dp TO zero  
        FOR mask FROM zero TO two RAISED TO the power of n MINUS one  
            SET bit_count TO the NUMBER OF ONES in the binary representation of mask  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j in the binary representation of mask EQUALS one  
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND dp at position mask MINUS two RAISED TO the power of j PLUS the bitwise exclusive OR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position two RAISED TO the power of n MINUS one  
    END FUNCTION  
END CLASS
