CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val LEFT right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the collection values is empty THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the collection queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none THEN  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none THEN  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p is none AND q is none THEN  
        RETURN true  
    ELSE IF p is none OR q is none THEN  
        RETURN false  
    ELSE IF the value property of p is NOT EQUALS the value property of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree with parameters the left property of p AND the left property of q AND is_same_tree with parameters the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST WITH PARAMETER root  
        SET self.max_sum TO zero  

        FUNCTION helper WITH PARAMETER node  
            IF node is none THEN  
                RETURN true, zero, positive infinity, negative infinity  
            END IF  
            SET left_is_bst, left_sum, left_min, left_max TO helper called with the left property of node  
            SET right_is_bst, right_sum, right_min, right_max TO helper called with the right property of node  

            IF left_is_bst IS true AND right_is_bst IS true AND left_max IS LESS THAN the value property of node AND the value property of node IS LESS THAN right_min THEN  
                SET current_sum TO left_sum PLUS right_sum PLUS the value property of node  
                SET current_min TO the smaller BETWEEN the value property of node AND left_min  
                SET current_max TO the larger BETWEEN the value property of node AND right_max  
                SET self.max_sum TO the larger BETWEEN self.max_sum AND current_sum  
                RETURN true, current_sum, current_min, current_max  
            END IF  

            RETURN false, zero, positive infinity, negative infinity  
        END FUNCTION  

        CALL helper WITH PARAMETER root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
