CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list containing n elements each assigned a new empty mapping from values to counts  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                IF the mapping at position j of dp CONTAINS the key d  
                    INCREMENT the value at key d of the mapping at position i of dp BY the value at key d of the mapping at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key d of the mapping at position j of dp  
                ELSE  
                    INCREMENT the value at key d of the mapping at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
