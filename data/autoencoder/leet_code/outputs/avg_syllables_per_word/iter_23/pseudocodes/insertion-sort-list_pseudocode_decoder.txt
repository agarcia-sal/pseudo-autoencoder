CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last element of values  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list(the element next of p1 the element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self head)  
        IF head EQUALS None OR the element next of head EQUALS None  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with value negative five thousand zero hundred and one  
        SET the element next of dummy TO head  
        SET current TO the element next of head  
        SET the element next of head TO None  
        WHILE current IS NOT None  
            SET next_node TO the element next of current  
            SET prev TO dummy  
            WHILE the element next of prev IS NOT None AND the element val of the element next of prev LESS THAN the element val of current  
                SET prev TO the element next of prev  
            END WHILE  
            SET the element next of current TO the element next of prev  
            SET the element next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the element next of dummy  
    END FUNCTION  
END CLASS
