CLASS Solution  
    FUNCTION maximizeSweetness(sweetness list OF integers, k integer) RETURNS integer  
        FUNCTION canDivide(min_sweet integer) RETURNS boolean  
            SET pieces TO zero  
            SET current_sweet TO zero  
            FOR each sweet IN sweetness  
                INCREMENT current_sweet BY sweet  
                IF current_sweet GREATER THAN OR EQUAL TO min_sweet THEN  
                    INCREMENT pieces BY one  
                    SET current_sweet TO zero  
                    IF pieces GREATER THAN k THEN  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            RETURN pieces GREATER THAN k  
        END FUNCTION  

        SET left TO the minimum value IN sweetness  
        SET right TO the sum OF sweetness DIVIDED BY the sum OF k PLUS one  
        SET result TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the sum OF left PLUS right DIVIDED BY two  
            IF canDivide(mid) IS true THEN  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
