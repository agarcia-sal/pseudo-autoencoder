CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the tuple of x AND y IS IN visited OR element at position x of grid and position y EQUALS zero THEN  
                RETURN  
            END IF  
            ADD the tuple of x AND y TO visited  
            FOR each pair dx AND dy IN the list of pairs negative one zero PLUS one zero zero negative one zero PLUS one  
                CALL dfs WITH x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO a list of tuples of i AND j FOR i FROM zero TO the LENGTH OF grid MINUS one FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one WHERE element at position i of grid and position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero THEN  
                RETURN True  
            END IF  
            CALL dfs WITH the first element at position zero of land_cells AND the second element at position zero of land_cells AND visited  
            RETURN whether the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT is_connected() THEN  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid and position j EQUALS one THEN  
                    SET element at position i of grid and position j TO zero  
                    IF NOT is_connected() THEN  
                        RETURN one  
                    END IF  
                    SET element at position i of grid and position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
