CLASS Solution  
    FUNCTION minCost(costs)  
        IF the LENGTH OF costs EQUALS zero  
            RETURN zero  
        END IF  
        SET prev_red TO element at position zero at element at position zero of costs  
        SET prev_blue TO element at position one at element at position zero of costs  
        SET prev_green TO element at position two at element at position zero of costs  
        FOR i FROM one TO the LENGTH OF costs MINUS one  
            SET current_red TO element at position zero at element at position i of costs PLUS the SMALLER OF prev_blue AND prev_green  
            SET current_blue TO element at position one at element at position i of costs PLUS the SMALLER OF prev_red AND prev_green  
            SET current_green TO element at position two at element at position i of costs PLUS the SMALLER OF prev_red AND prev_blue  
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        RETURN the SMALLER OF prev_red AND prev_blue AND prev_green  
    END FUNCTION  
END CLASS
