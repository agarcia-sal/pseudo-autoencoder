CLASS Solution  
    FUNCTION trapRainWater heightMap  
        IF heightMap EQUALS nothing OR element at position zero of heightMap EQUALS nothing  
            RETURN zero  
        END IF  
  
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a two dimensional list of m elements each containing a list of n elements set to False  
        SET min_heap TO an empty list  
  
        FOR index i FROM zero TO m MINUS one  
            PUSH a tuple containing element at position i of heightMap at position zero AND i AND zero INTO min_heap  
            PUSH a tuple containing element at position i of heightMap at position n MINUS one AND i AND n MINUS one INTO min_heap  
            SET element at position i at position zero of visited TO True  
            SET element at position i at position n MINUS one of visited TO True  
        END FOR  
  
        FOR index j FROM zero TO n MINUS one  
            PUSH a tuple containing element at position zero of heightMap at position j AND zero AND j INTO min_heap  
            PUSH a tuple containing element at position m MINUS one of heightMap at position j AND m MINUS one AND j INTO min_heap  
            SET element at position zero at position j of visited TO True  
            SET element at position m MINUS one at position j of visited TO True  
        END FOR  
  
        SET directions TO a list containing a tuple of minus one AND zero AND a tuple of one AND zero AND a tuple of zero AND minus one AND a tuple of zero AND one  
        SET water_trapped TO zero  
  
        WHILE min_heap is not empty  
            REMOVE AND RETURN the smallest element from min_heap AND DESTRUCTURE it INTO height AND x AND y  
  
            FOR each tuple dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx at position ny of visited EQUALS False  
                    SET element at position nx at position ny of visited TO True  
                    IF height GREATER THAN element at position nx at position ny of heightMap  
                        INCREMENT water_trapped BY height MINUS element at position nx at position ny of heightMap  
                    END IF  
                    PUSH a tuple containing the GREATER OF height AND element at position nx at position ny of heightMap AND nx AND ny INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
  
        RETURN water_trapped  
    END FUNCTION  
END CLASS
