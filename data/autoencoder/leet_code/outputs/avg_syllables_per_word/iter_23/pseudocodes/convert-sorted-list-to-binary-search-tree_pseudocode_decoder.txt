CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with parameter element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO the last position  
        SET node TO new ListNode with parameter val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list with parameters the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with parameter element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the length of values AND element at position i PLUS one of values NOT EQUALS none  
            SET the left attribute of node TO new TreeNode with parameter element at position i PLUS one of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i PLUS one of values NOT EQUALS none  
            SET the right attribute of node TO new TreeNode with parameter element at position i PLUS one of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with parameters the left attribute of p and the left attribute of q AND is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head EQUALS none  
            RETURN none  
        END IF  

        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next attribute of fast NOT EQUALS right  
                SET slow TO the next attribute of slow  
                SET fast TO the next attribute of the next attribute of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right  
                RETURN none  
            END IF  

            SET mid TO findMiddle with parameters left and right  

            SET node TO new TreeNode with parameter the value attribute of mid  
            SET the left attribute of node TO convertListToBST with parameters left and mid  
            SET the right attribute of node TO convertListToBST with parameters the next attribute of mid and right  
            RETURN node  
        END FUNCTION  

        RETURN convertListToBST with parameters head and none  
    END FUNCTION  
END CLASS
