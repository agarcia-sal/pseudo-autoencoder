CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of strings converted from each element in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x AND concatenation of y IS GREATER THAN concatenation of y AND concatenation of x  
                RETURN negative one  
            ELSE IF concatenation of x AND concatenation of y IS LESS THAN concatenation of y AND concatenation of x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str BY comparison defined by compare  
        
        IF element at position zero of nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        RETURN concatenation of all elements in nums_str in order  
    END FUNCTION  
END CLASS
