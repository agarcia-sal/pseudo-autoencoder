CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        SET left_sums TO a new list CONTAINING zero  
        FOR each num IN the SUBLIST of nums FROM position zero TO position mid MINUS one  
            SET extended_sums TO a new empty list  
            FOR each s IN left_sums  
                APPEND the SUM of s PLUS num TO extended_sums  
            END FOR  
            APPEND all elements of extended_sums TO left_sums  
        END FOR  
        SET right_sums TO a new list CONTAINING zero  
        FOR each num IN the SUBLIST of nums FROM position mid TO the END  
            SET extended_sums TO a new list  
            FOR each s IN right_sums  
                APPEND the SUM of s PLUS num TO extended_sums  
            END FOR  
            APPEND all elements of extended_sums TO right_sums  
        END FOR  
        SORT right_sums IN ascending order  
        SET min_diff TO positive infinity  
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the POSITION found BY performing BINARY SEARCH FOR target IN right_sums at the LEFTMOST suitable index  
            IF pos IS LESS THAN the LENGTH OF right_sums  
                SET candidate_diff TO the ABSOLUTE VALUE OF the SUM of left_sum PLUS the element at position pos of right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            IF pos IS GREATER THAN zero  
                SET candidate_diff TO the ABSOLUTE VALUE OF the SUM of left_sum PLUS the element at position pos MINUS one of right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
