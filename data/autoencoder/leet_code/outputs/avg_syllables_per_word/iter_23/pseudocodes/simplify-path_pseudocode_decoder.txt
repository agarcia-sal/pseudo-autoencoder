CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of splitting path by the forward slash character  
        SET stack TO an empty collection  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS a single dot character  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two consecutive dots  
                IF stack is not empty  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                ADD component TO the end of stack  
            END IF  
        END FOR  
        SET simplified_path TO the forward slash character CONCATENATED WITH all elements of stack JOINED by the forward slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
