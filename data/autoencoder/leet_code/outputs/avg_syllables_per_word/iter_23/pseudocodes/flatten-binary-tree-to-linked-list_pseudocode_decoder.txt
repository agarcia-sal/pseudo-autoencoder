CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE and ASSIGN the leftmost element of queue TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node dot left TO new TreeNode with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node dot right TO new TreeNode with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p dot left AND q dot left AND is_same_tree with p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS none  
            RETURN  
        END IF  
        INITIALIZE stack as a list containing root  
        WHILE the LENGTH OF stack GREATER THAN zero  
            REMOVE and ASSIGN the last element of stack TO node  
            IF node dot right NOT EQUALS none  
                APPEND node dot right TO stack  
            END IF  
            IF node dot left NOT EQUALS none  
                APPEND node dot left TO stack  
            END IF  
            SET node dot left TO none  
            IF the LENGTH OF stack GREATER THAN zero  
                SET node dot right TO the last element of stack  
            ELSE  
                SET node dot right TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
