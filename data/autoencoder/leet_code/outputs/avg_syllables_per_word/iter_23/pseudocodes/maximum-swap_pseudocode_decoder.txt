CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO the list of characters representing num converted to string  
        SET last TO an empty dictionary  
        FOR index FROM zero TO the length of digits MINUS one  
            SET element_at_index_of_digits TO digit_character  
            CONVERT digit_character TO integer digit_value  
            STORE index INTO last FOR key digit_value  
        END FOR  
        FOR index FROM zero TO the length of digits MINUS one  
            SET digit_character TO element at position index of digits  
            CONVERT digit_character TO integer digit_value  
            FOR candidate_digit FROM nine DOWN TO digit_value PLUS one  
                IF last HAS key candidate_digit AND element at key candidate_digit of last GREATER THAN index THEN  
                    SET position_to_swap TO element at key candidate_digit of last  
                    SET temporary TO element at position index of digits  
                    SET element at position index of digits TO element at position position_to_swap of digits  
                    SET element at position position_to_swap of digits TO temporary  
                    SET joined_string TO the concatenation of all elements in digits in sequence  
                    CONVERT joined_string TO integer swapped_number  
                    RETURN swapped_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
