CLASS Solution  
    FUNCTION grayCode WITH parameter n  
        IF n EQUALS zero THEN  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode WITH argument n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one positions  
        SET current_gray TO prev_gray CONCATENATED WITH a list formed by FOR each number IN the reverse order of prev_gray APPLYING bitwise OR BETWEEN mask AND number  
        RETURN current_gray  
    END FUNCTION  
END CLASS
