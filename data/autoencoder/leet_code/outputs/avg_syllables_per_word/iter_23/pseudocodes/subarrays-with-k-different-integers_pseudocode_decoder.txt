CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO empty mapping from element to integer  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the element at position right of nums IN count BY one  
                WHILE the NUMBER OF keys IN count GREATER THAN t  
                    DECREMENT the element at position left of nums IN count BY one  
                    IF the element at position left of nums IN count EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY the value of right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the RESULT OF atMostKDistinct WITH argument k MINUS the RESULT OF atMostKDistinct WITH argument k MINUS one  
    END FUNCTION  
END CLASS
