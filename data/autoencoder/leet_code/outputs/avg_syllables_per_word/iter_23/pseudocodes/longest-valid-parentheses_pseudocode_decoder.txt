CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a list containing NEGATIVE one  
        SET max_length TO zero  
        FOR each position i AND each character char IN the string s  
            IF the character char EQUALS an opening parenthesis character  
                APPEND the position i TO the stack  
            ELSE  
                REMOVE the last element FROM the stack  
                IF the stack HAS NO elements  
                    APPEND the position i TO the stack  
                ELSE  
                    SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN the position i AND the last element of the stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
