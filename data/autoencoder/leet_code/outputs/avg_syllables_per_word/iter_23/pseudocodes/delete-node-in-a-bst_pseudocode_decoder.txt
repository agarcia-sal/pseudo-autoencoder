CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of is_same_tree called with p.left and q.left AND is_same_tree called with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root IS None THEN  
            RETURN None  
        END IF  
        IF the val attribute of root IS GREATER THAN key THEN  
            SET root.left TO the result of calling deleteNode on self with root.left and key  
            RETURN root  
        END IF  
        IF the val attribute of root IS LESS THAN key THEN  
            SET root.right TO the result of calling deleteNode on self with root.right and key  
            RETURN root  
        END IF  
        IF root.left IS None THEN  
            RETURN root.right  
        END IF  
        IF root.right IS None THEN  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left IS NOT None  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
