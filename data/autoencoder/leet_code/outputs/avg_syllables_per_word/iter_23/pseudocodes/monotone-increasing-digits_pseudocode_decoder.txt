CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters forming string representation of n  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one IN STEPS OF minus one  
            IF element at position i of digits LESS THAN element at position i MINUS one of digits  
                SET element at position i MINUS one of digits TO string representation of integer value of element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET element at position i of digits TO string representation of digit nine  
        END FOR  
        SET result TO integer value of concatenation of all elements of digits in order  
        RETURN result  
    END FUNCTION  
END CLASS
