CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO none)  
        SET self-val TO val  
        SET self-next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO A new ListNode WITH val EQUAL TO the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the END  
        SET node TO A new ListNode WITH val EQUAL TO val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN False  
    END IF  
    RETURN the condition that the val of p1 EQUALS the val of p2 AND the result of is_same_list WITH p1 next AND p2 next  
END FUNCTION

CLASS Solution  
    FUNCTION plusOne(head)  
        FUNCTION reverseList(node)  
            SET prev TO none  
            WHILE node NOT EQUALS none  
                SET next_node TO the next of node  
                SET the next of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION

        SET reversed_head TO the result of reverseList WITH head  
        SET carry TO one  
        SET current TO reversed_head  
        WHILE current NOT EQUALS none AND carry EQUALS one  
            SET current val TO current val PLUS carry  
            SET carry TO current val DIVIDED BY ten  
            SET current val TO current val MODULO ten  
            IF the next of current EQUALS none AND carry EQUALS one THEN  
                SET the next of current TO A new ListNode WITH val EQUAL TO carry  
                SET carry TO zero  
            END IF  
            SET prev TO current  
            SET current TO the next of current  
        END WHILE  
        RETURN the result of reverseList WITH reversed_head  
    END FUNCTION  
END CLASS
