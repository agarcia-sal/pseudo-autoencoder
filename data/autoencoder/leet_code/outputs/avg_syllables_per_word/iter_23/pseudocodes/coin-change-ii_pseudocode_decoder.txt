CLASS Solution  
    FUNCTION change(amount parameter AS integer, coins parameter AS list of integers) RETURNS integer  
        SET dp TO a list of zeros with the LENGTH EQUAL TO the amount PLUS one  
        SET element at position zero of dp TO one  
        FOR each coin IN coins  
            FOR x FROM coin TO amount  
                INCREMENT element at position x of dp BY element at position x MINUS coin of dp  
            END FOR  
        END FOR  
        RETURN element at position amount of dp  
    END FUNCTION  
END CLASS
