CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR the element at position index PLUS one of expression EQUALS the character colon  
                RETURN the element at position index of expression AND index PLUS two  
            END IF  
            SET true_part AND next_index TO the RESULT OF evaluate with argument index PLUS two  
            SET false_part AND final_index TO the RESULT OF evaluate with argument next_index  
            IF the element at position index of expression EQUALS character T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        SET result AND unused_variable TO the RESULT OF evaluate with argument zero  
        RETURN result  
    END FUNCTION  
END CLASS
