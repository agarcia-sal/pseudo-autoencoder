CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary containing key zero mapped to value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN the enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the value at key count in prefix_sum_index  
            ELSE  
                SET the value at key count in prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
