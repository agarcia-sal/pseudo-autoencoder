CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new default dictionary mapping values to zero  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO two parts count_string AND domain_string based on the first whitespace  
            SET count TO the integer value of count_string  
            SPLIT domain_string INTO subdomains by the delimiter of a period  
            FOR index FROM zero TO the length of subdomains MINUS one  
                JOIN the elements of subdomains from position index TO the end separated by a period INTO subdomain  
                INCREMENT the value mapped by subdomain in domain_count BY count  
            END FOR  
        END FOR  
        SET result_list TO an empty list  
        FOR each domain-key AND count-value IN the items of domain_count  
            CREATE a string by concatenating the string form of count-value PLUS one space PLUS domain-key  
            APPEND this string TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
