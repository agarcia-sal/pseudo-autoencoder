CLASS Solution  
    FUNCTION numBusesToDestination(routes parameter, source parameter, target parameter)  
        IF source parameter EQUALS target parameter  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a new mapping from elements to list of elements with default empty list  
        FOR each bus and route IN the enumeration of routes parameter  
            FOR each stop IN route  
                APPEND bus TO the list at key stop IN stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO a new empty double ended queue  
        SET visited_buses TO a new empty set  
        SET visited_stops TO a new empty set  

        FOR each bus IN the list at key source parameter IN stop_to_buses  
            APPEND the tuple of bus and one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source parameter TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the first element FROM bus_queue AND ASSIGN TO current_bus and bus_count  

            FOR each stop IN the element at position current_bus of routes parameter  
                IF stop EQUALS target parameter  
                    RETURN bus_count  
                END IF  

                IF stop NOT IN visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list at key stop IN stop_to_buses  
                        IF next_bus NOT IN visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND the tuple of next_bus and bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN the minus one value  
    END FUNCTION  
END CLASS
