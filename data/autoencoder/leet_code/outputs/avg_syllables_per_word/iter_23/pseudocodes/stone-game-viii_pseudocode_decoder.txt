CLASS Solution  
    FUNCTION stoneGameVIII WITH parameter stones  
        SET prefix_sum TO list of zeros with length equal to the LENGTH OF stones  
        SET element at position one of prefix_sum TO element at position one of stones  
        FOR index i FROM two TO LENGTH OF stones  
            SET element at position i of prefix_sum TO element at position i MINUS one of prefix_sum PLUS element at position i of stones  
        END FOR  
        SET dp TO element at the last position of prefix_sum  
        FOR index i FROM the LENGTH OF stones MINUS two DOWN TO two STEP minus one  
            IF dp GREATER THAN OR EQUAL TO element at position i of prefix_sum MINUS dp  
                SET dp TO dp  
            ELSE  
                SET dp TO element at position i of prefix_sum MINUS dp  
            END IF  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
