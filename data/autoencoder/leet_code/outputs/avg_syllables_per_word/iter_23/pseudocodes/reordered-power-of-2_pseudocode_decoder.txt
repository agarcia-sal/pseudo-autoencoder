CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO ten raised to the power of nine  
            ADD the tuple of the digits of power_of_2 sorted in ascending order TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the digits of n sorted in ascending order  
        RETURN whether sorted_n_digits is an element of power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
