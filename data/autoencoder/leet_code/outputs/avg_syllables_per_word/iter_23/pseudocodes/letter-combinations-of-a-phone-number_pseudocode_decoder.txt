CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        SET phone_map TO mapping from each digit character two through nine TO corresponding string of letters representing the telephone keypad letters  
        FUNCTION backtrack(combination_string, next_digits_string)  
            IF the LENGTH OF next_digits_string EQUALS zero THEN  
                APPEND combination_string TO output list  
            ELSE  
                FOR each letter_character IN the string of letters from phone_map mapped by the element at position zero of next_digits_string  
                    CALL backtrack WITH combination_string concatenated with letter_character AND the substring of next_digits_string from position one TO the end  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
