CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred THEN  
            RETURN one point zero  
        END IF  
        
        SET n TO the result of n PLUS twenty four DIVIDED BY twenty five rounded down to the nearest integer  
        
        SET memo TO an empty collection mapping pairs to probabilities  
        
        FUNCTION dp(a, b)  
            IF the pair of a and b EXISTS in memo THEN  
                RETURN the value associated with the pair of a and b in memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero THEN  
                RETURN point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero THEN  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero THEN  
                RETURN zero point zero  
            END IF  
            
            SET prob TO point two five MULTIPLIED BY the sum of dp of a MINUS four and b PLUS dp of a MINUS three and b MINUS one PLUS dp of a MINUS two and b MINUS two PLUS dp of a MINUS one and b MINUS three  
            
            ASSIGN prob TO memo with the key pair of a and b  
            RETURN prob  
        END FUNCTION  
        
        RETURN the result of dp with parameters n and n  
    END FUNCTION  
END CLASS
