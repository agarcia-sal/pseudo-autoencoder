CLASS Solution  
    FUNCTION generateParenthesis WITH parameter n  
        FUNCTION backtrack WITH parameters s assigned to empty string left assigned to zero right assigned to zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH argument OF s CONCATENATED WITH open parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH argument OF s CONCATENATED WITH close parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
