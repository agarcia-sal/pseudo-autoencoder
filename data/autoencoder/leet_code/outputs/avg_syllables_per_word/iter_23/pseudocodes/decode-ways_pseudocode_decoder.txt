CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS EMPTY OR element at position one of s EQUALS character zero THEN  
            RETURN zero  
        END IF  
        SET dp TO list of zeros with LENGTH of s PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        FOR i FROM two TO LENGTH of s PLUS one  
            IF element at position i MINUS one of s NOT EQUALS character zero THEN  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of dp  
            END IF  
            SET two_digit TO integer value represented by substring of s from position i MINUS two TO position i  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six THEN  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS two of dp  
            END IF  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
