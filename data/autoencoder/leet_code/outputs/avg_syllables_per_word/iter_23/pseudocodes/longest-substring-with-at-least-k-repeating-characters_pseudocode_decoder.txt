CLASS Solution  
    FUNCTION longestSubstring(s parameter s AS string k parameter k AS integer) RETURNS integer  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        
        ASSIGN reference TO the frequency count OF characters IN s  
        
        FOR each character IN the frequency count  
            IF the frequency OF character LESS THAN k  
                RETURN the MAXIMUM OF the following values FOR each substring IN the list OF substrings OBTAINED BY SPLITTING s BY character  
                    CALL longestSubstring WITH substring AND k  
                END OF MAXIMUM  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
