CLASS Solution  
    FUNCTION minimumOneBitOperations WITH parameter n  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one SHIFTED LEFT BY the value of k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one SHIFTED LEFT BY the value of k  
        RETURN one SHIFTED LEFT BY the value of k PLUS one MINUS one MINUS the RESULT OF calling minimumOneBitOperations WITH argument the EXCLUSIVE OR of n AND most_significant_bit  
    END FUNCTION  
END CLASS
