CLASS ListNode  
    FUNCTION __init__ with parameters self val EQUALS zero next EQUALS None  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode INITIALIZED WITH element at position one of values  
    SET p TO head  
    FOR each val IN elements of values FROM position two TO END  
        SET node TO a new ListNode INITIALIZED WITH val  
        SET p next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that element val at p1 EQUALS element val at p2 AND the return value of is_same_list called WITH parameters element next at p1 AND element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETERS self head  
        IF head EQUALS None OR element next at head EQUALS None  
            RETURN  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUALS None AND element next at fast NOT EQUALS None  
            SET slow TO element next at slow  
            SET fast TO the element next at the element next at fast  
        END WHILE  
        SET prev TO None  
        SET curr TO slow  
        WHILE curr NOT EQUALS None  
            SET next_temp TO element next at curr  
            SET element next at curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE element next at second NOT EQUALS None  
            SET temp1 TO element next at first  
            SET temp2 TO element next at second  
            SET element next at first TO second  
            SET element next at second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
