CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of the value infinity repeated fifteen times  
        FOR each pair of values f and r IN tires  
            SET total_time TO zero  
            FOR i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK the inner loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position i TO the smaller value BETWEEN min_times at position i AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of the value infinity repeated the value numLaps PLUS one times  
        SET dp at position zero TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET candidate_time TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                SET dp at position i TO the smaller value BETWEEN dp at position i AND candidate_time  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
