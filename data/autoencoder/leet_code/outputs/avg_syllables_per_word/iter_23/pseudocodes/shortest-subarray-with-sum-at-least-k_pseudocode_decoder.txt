CLASS Solution  
    FUNCTION shortestSubarray WITH PARAMETERS nums AND k  
        SET prefix_sum TO a list containing the zero value  
        FOR each number IN nums  
            APPEND the element at the last position of prefix_sum PLUS the current number TO prefix_sum  
        END FOR  
        ASSIGN reference TO an empty double ended queue dq  
        SET min_length TO positive infinity  
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE the double ended queue dq is not empty AND the current_sum MINUS the element at the position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the lesser value BETWEEN min_length AND the difference BETWEEN i AND the element removed from the front of dq  
            END WHILE  
            WHILE the double ended queue dq is not empty AND the current_sum IS LESS THAN OR EQUAL TO the element at the position at the back of dq in prefix_sum  
                REMOVE the element at the back of dq  
            END WHILE  
            ADD the current index i TO the back of dq  
        END FOR  
        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
