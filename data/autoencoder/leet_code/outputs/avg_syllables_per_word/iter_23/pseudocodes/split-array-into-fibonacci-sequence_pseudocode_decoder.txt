CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS THE LENGTH OF num AND THE LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO THE LENGTH OF num PLUS one  
                IF THE element at position start of num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE TO THE NEXT ITERATION  
                END IF  
                SET next_num TO THE INTEGER VALUE OF THE substring of num FROM position start TO position end MINUS one  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK THE LOOP  
                END IF  
                IF THE LENGTH OF path LESS THAN two OR next_num EQUALS THE element at THE last position OF path PLUS THE element at THE second LAST position OF path  
                    SET result TO THE VALUE RETURNED BY backtrack WITH PARAMETERS end AND path WITH next_num APPENDED  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN AN EMPTY LIST  
        END FUNCTION  
        RETURN THE VALUE RETURNED BY backtrack WITH PARAMETERS zero AND AN EMPTY LIST  
    END FUNCTION  
END CLASS
