CLASS Solution  
    FUNCTION minOperations(target AS List OF integers arr AS List OF integers) RETURNS integer  
        SET index_map TO an empty mapping FROM integer TO integer  
        SET current_index TO zero  
        FOR each value IN target  
            ASSIGN index_map value TO current_index  
            INCREMENT current_index BY one  
        END FOR  

        SET transformed_arr TO an empty list of integers  
        FOR each value IN arr  
            IF value IS IN index_map  
                APPEND element at position value OF index_map TO transformed_arr  
            END IF  
        END FOR  

        SET lis TO an empty list of integers  
        FOR each num IN transformed_arr  
            SET position TO the first position where num CAN BE inserted TO lis TO KEEP lis sorted in ascending order  
            IF position EQUALS the LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET element at position position OF lis TO num  
            END IF  
        END FOR  

        SET lcs_length TO the LENGTH OF lis  
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
