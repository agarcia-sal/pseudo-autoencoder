CLASS Solution  
    FUNCTION minSteps(n)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET operations TO zero  
        SET factor TO two  
        WHILE n GREATER THAN one HOLDING  
            WHILE n MODULO factor EQUALS zero HOLDING  
                INCREMENT operations BY factor  
                SET n TO n DIVIDED BY factor  
            END WHILE  
            INCREMENT factor BY one  
        END WHILE  
        RETURN operations  
    END FUNCTION  
END CLASS
