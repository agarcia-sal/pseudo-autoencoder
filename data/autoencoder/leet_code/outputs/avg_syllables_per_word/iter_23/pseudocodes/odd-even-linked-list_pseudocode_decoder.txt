CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN elements of values starting from position two to the end  
        SET node TO new ListNode with value val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN self val of p1 EQUALS self val of p2 AND is_same_list(next of p1, next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self, head)  
        IF head EQUALS None OR next of head EQUALS None OR next of next of head EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO next of head  
        SET even_head TO even  
        WHILE even IS NOT None AND next of even IS NOT None  
            SET next of odd TO next of even  
            SET odd TO next of odd  
            SET next of even TO next of odd  
            SET even TO next of even  
        END WHILE  
        SET next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
