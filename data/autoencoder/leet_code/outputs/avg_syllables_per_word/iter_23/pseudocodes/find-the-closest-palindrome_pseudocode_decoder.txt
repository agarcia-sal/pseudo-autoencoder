CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER VALUE OF n MINUS one  
        END IF  
        
        SET candidates TO an EMPTY SET  
        
        ADD the NUMBER ten RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD the NUMBER ten RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER VALUE OF the SUBSTRING of n FROM position zero TO the INTEGER VALUE OF the WHOLE NUMBER DIVISION of length PLUS one BY two  
        
        FOR i FROM negative one TO one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF length MODULO two EQUALS zero  
                SET candidate TO the CONCATENATION OF new_prefix WITH the REVERSED STRING OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION OF new_prefix WITH the REVERSED STRING OF the SUBSTRING OF new_prefix FROM position LENGTH OF new_prefix MINUS two TO position zero EXCLUSIVE  
            END IF  
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF PRESENT  
        
        SET closest TO the ELEMENT OF candidates WITH the MINIMUM VALUE ACCORDING TO THE TUPLE OF the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN the ELEMENT AND original_number AND the ELEMENT ITSELF  
        
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
