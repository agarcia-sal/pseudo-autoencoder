CLASS Solution  
    FUNCTION minSwapsCouples WITH PARAMETER row  
        SET position TO an empty dictionary mapping each person TO their index IN row  
        SET swaps TO zero  
        FOR i FROM zero TO the LENGTH OF row MINUS one STEP two  
            SET correct_pair TO element at position i of row BITWISE EXCLUSIVE OR WITH one  
            IF element at position i PLUS one of row NOT EQUALS correct_pair  
                SET swap_idx TO the index mapped from correct_pair IN position  
                SWAP element at position i PLUS one of row WITH element at position swap_idx of row  
                SET the index mapped from element at position swap_idx of row IN position TO swap_idx  
                SET the index mapped from element at position i PLUS one of row IN position TO i PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
