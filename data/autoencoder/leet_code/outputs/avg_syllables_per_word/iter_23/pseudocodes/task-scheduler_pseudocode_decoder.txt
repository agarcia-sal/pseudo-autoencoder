CLASS Solution  
    FUNCTION leastInterval(tasks parameter OF TYPE list OF strings n parameter OF TYPE integer) RETURNS integer  
        SET task_counts TO the counts of each unique element IN tasks  
        SET max_heap TO an empty list  
        FOR each count IN the values of task_counts  
            APPEND the negative of count TO max_heap  
        END FOR  
        PERFORM heapify operation ON max_heap  
        
        SET time TO zero  
        SET queue TO an empty list  
        
        WHILE the LENGTH OF max_heap IS GREATER THAN zero OR the LENGTH OF queue IS GREATER THAN zero  
            WHILE the LENGTH OF queue IS GREATER THAN zero AND the first element at position zero AND position zero of queue EQUALS time  
                REMOVE the first element FROM queue AND ASSIGN to tuple with variables unused_value and count  
                IF count IS LESS THAN zero  
                    PERFORM heap push operation TO max_heap WITH count  
                END IF  
            END WHILE  
            
            IF the LENGTH OF max_heap IS GREATER THAN zero  
                PERFORM heap pop operation FROM max_heap AND ASSIGN TO variable count  
                SET count TO count PLUS one  
                IF count IS LESS THAN zero  
                    APPEND a tuple with first element the sum of time PLUS n PLUS one AND second element count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
