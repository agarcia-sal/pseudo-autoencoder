CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of each character in time with all colons removed  
        SET hour AND minute TO integer values parsed by splitting time at the colon delimiter  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each first_digit_in_hour IN digits  
                FOR each second_digit_in_hour IN digits  
                    FOR each first_digit_in_minute IN digits  
                        FOR each second_digit_in_minute IN digits  
                            SET hour_value TO the integer value of concatenating first_digit_in_hour and second_digit_in_hour as strings  
                            SET minute_value TO the integer value of concatenating first_digit_in_minute and second_digit_in_minute as strings  
                            IF hour_value LESS THAN twenty_four AND minute_value LESS THAN sixty  
                                APPEND the pair of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the position of the pair of hour and minute in the list all_times  

        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the pair at position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the first pair in all_times  
        END IF  

        RETURN a string representing next_hour and next_minute formatted as two digit numbers separated by a colon  
    END FUNCTION  
END CLASS
