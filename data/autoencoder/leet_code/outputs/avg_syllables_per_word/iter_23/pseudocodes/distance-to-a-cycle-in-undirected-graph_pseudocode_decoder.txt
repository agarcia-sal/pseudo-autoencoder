CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a new mapping with default value as a list  
        FOR each pair u and v IN edges  
            APPEND v TO the list at graph referenced by u  
            APPEND u TO the list at graph referenced by v  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node IN path  
                RETURN the sublist of path from position cycle_start TO the end of path  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list at graph referenced by node  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO the result of dfs called with neighbor, node, path, and visited  
                    IF cycle EXISTS THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN nothing  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO nothing  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO the result of dfs called with node, negative one, an empty list, and visited  
                IF cycle EXISTS THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zeros with length n  
        SET queue TO a new double ended queue initialized with the elements of cycle  
        SET visited TO a new set initialized with the elements of cycle  
        
        SET level TO zero  
        WHILE queue IS NOT empty DO  
            SET count TO the number of elements currently IN queue  
            FOR index FROM one TO count  
                SET node TO the element removed from the front of queue  
                SET the element at position node of distance TO level  
                FOR each neighbor IN the list at graph referenced by node  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
