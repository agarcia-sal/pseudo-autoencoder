CLASS Solution  
    FUNCTION shortestPathLength(graph)  
        IF graph EQUALS empty THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF graph  
        
        SET queue TO a new double-ended queue containing for each node from zero TO n MINUS one a pair of node and one shifted left by node positions  
        
        SET visited TO a new set containing for each node from zero TO n MINUS one a pair of node and one shifted left by node positions  
        
        SET steps TO zero  
        SET all_nodes_visited TO one shifted left by n positions MINUS one  
        
        WHILE the LENGTH OF queue is greater than zero  
            SET current_length TO the LENGTH OF queue  
            FOR loop_index FROM zero TO current_length MINUS one  
                REMOVE from the left of queue a pair current_node and visited_nodes  
                
                IF visited_nodes EQUALS all_nodes_visited THEN  
                    RETURN steps  
                END IF  
                
                FOR each neighbor IN the collection at position current_node of graph  
                    SET next_visited_nodes TO visited_nodes OR one shifted left by neighbor positions  
                    SET next_state TO a pair of neighbor and next_visited_nodes  
                    
                    IF next_state NOT IN visited THEN  
                        ADD next_state TO visited  
                        ADD next_state TO the right end of queue  
                    END IF  
                END FOR  
            END FOR  
            
            INCREMENT steps BY one  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
