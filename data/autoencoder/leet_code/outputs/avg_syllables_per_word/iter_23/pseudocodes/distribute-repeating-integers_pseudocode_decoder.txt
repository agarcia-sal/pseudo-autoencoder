CLASS Solution  
    FUNCTION canDistribute(nums parameter, quantity parameter)  
        SET count TO the list of the frequencies of each unique element in nums parameter  
        SORT quantity parameter INTO descending order  
        FUNCTION can_satisfy(index parameter)  
            IF index parameter EQUALS the length of quantity parameter THEN  
                RETURN True  
            END IF  
            FOR i variable FROM zero TO the length of count MINUS one  
                IF element at position i of count GREATER THAN OR EQUAL TO element at position index parameter of quantity parameter THEN  
                    DECREMENT element at position i of count BY element at position index parameter of quantity parameter  
                    IF the result of calling can_satisfy with argument index parameter PLUS one THEN  
                        RETURN True  
                    END IF  
                    INCREMENT element at position i of count BY element at position index parameter of quantity parameter  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN the result of calling can_satisfy with argument zero  
    END FUNCTION  
END CLASS
