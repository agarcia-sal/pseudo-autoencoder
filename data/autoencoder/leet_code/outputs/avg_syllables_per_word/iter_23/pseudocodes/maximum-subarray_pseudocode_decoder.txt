CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position one of nums  
        SET current_sum TO element at position one of nums  
        FOR each num IN the elements of nums starting from position two to the end  
            IF the value of num IS GREATER THAN the value of current_sum PLUS num  
                SET current_sum TO num  
            ELSE  
                SET current_sum TO current_sum PLUS num  
            END IF  
            IF the value of current_sum IS GREATER THAN the value of max_sum  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
