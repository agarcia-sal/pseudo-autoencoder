CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of the queue  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a new double ended queue initialized with root as its only element  
        SET leftmost_value TO none  
        WHILE the queue is not empty  
            SET level_size TO the length of queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO the element removed from the left of queue  
                IF i EQUALS zero  
                    SET leftmost_value TO the val attribute of node  
                END IF  
                IF the left attribute of node IS NOT none  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node IS NOT none  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
