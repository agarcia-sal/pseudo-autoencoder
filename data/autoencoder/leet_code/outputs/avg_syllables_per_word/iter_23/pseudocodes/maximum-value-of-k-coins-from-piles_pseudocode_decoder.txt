CLASS Solution  
    FUNCTION maxValueOfCoins WITH PARAMETERS piles AND k  
        SET n TO the LENGTH OF piles  
        SET dp TO a new list WITH n PLUS one elements EACH BEING a list WITH k PLUS one zeros  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one OF piles  
                SET current_value TO zero  
                FOR l FROM zero TO the LESSER OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero THEN  
                        INCREMENT current_value BY element at position l MINUS one OF current_pile  
                    END IF  
                    SET dp at position i and j TO the GREATER OF dp at position i and j AND the SUM of dp at position i MINUS one and j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
