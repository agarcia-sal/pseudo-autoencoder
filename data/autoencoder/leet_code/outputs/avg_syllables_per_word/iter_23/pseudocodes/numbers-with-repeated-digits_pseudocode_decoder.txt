CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the number of characters in s  
            SET count TO zero  
            FOR each i FROM one TO length MINUS one  
                SET count TO count PLUS nine MULTIPLIED BY perm of nine AND i MINUS one  
            END FOR  
            SET seen TO an empty set  
            FOR each position i AND character digit IN s  
                FOR each digit_option FROM zero IF i EQUALS zero OTHERWISE one TO the integer value of digit MINUS one  
                    IF digit_option NOT IN seen  
                        SET count TO count PLUS perm of nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF the integer value of digit IN seen  
                    EXIT the FOR loop  
                END IF  
                ADD the integer value of digit TO seen  
            END FOR  
            IF the FOR loop completed without break  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION

        RETURN n MINUS count_unique_digits of n  
    END FUNCTION

    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR each i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
