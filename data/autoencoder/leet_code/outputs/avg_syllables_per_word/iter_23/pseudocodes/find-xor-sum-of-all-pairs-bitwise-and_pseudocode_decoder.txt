CLASS Solution  
    FUNCTION getXORSum(arr1 arr1_list_of_integers arr2 arr2_list_of_integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1_list_of_integers  
            SET xor_sum_arr1 TO xor_sum_arr1 XOR num  
        END FOR  
        FOR each num IN arr2_list_of_integers  
            SET xor_sum_arr2 TO xor_sum_arr2 XOR num  
        END FOR  
        RETURN bitwise AND of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
