CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO set of two five six nine  
        SET same_digits TO set of zero one eight  

        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF all digits IN num_str ARE IN the union of change_digits AND same_digits THEN  
                IF any digit IN num_str IS IN change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  

        SET good_count TO zero  
        FOR each i FROM one TO n  
            IF is_good_number(i) EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
