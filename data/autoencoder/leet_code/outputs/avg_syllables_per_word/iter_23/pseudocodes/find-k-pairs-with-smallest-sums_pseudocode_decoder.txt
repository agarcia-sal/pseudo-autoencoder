CLASS Solution  
    FUNCTION kSmallestPairs(nums1 LIST OF INTEGER nums2 LIST OF INTEGER k INTEGER) RETURNS LIST OF LIST OF INTEGER  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR i FROM zero TO minimum of k AND the LENGTH OF nums1 MINUS one  
            PUSH onto min_heap the tuple of the sum of element at position i of nums1 PLUS element at position zero of nums2 AND the index i AND zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap is not empty AND the LENGTH OF result LESS THAN k  
            POP the smallest element from min_heap AND destructure into discard_sum AND i AND j  
            APPEND list of element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                PUSH onto min_heap the tuple of the sum of element at position i of nums1 PLUS element at position j PLUS one of nums2 AND i AND j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
