CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all ELEMENTS OF path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH the STRING REPRESENTATION OF count AND the ELEMENT at position index OF word AND zero AS count  
                ELSE  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH the ELEMENT at position index OF word AND zero AS count  
                END IF  
                CALL backtrack WITH index INCREMENTED BY one AND path AND count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an EMPTY list  
        CALL backtrack WITH zero, an EMPTY list, AND zero  
        RETURN result  
    END FUNCTION  
END CLASS
