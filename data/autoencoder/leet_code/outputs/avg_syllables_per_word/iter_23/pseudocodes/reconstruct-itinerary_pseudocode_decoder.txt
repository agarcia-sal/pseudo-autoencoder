CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a mapping from keys to empty lists such that missing keys produce empty lists  
        FOR each pair of departure and arrival IN the tickets sorted in ascending lexicographical order and then reversed  
            APPEND arrival TO the list associated with departure in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph is not empty  
                CALL visit with the last element removed from the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit with the string consisting of the letters J F K in that order  
        RETURN the elements of itinerary in reverse order  
    END FUNCTION  
END CLASS
