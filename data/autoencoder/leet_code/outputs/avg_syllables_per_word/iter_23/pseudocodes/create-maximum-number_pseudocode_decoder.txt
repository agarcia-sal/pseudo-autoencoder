CLASS Solution  
    FUNCTION maxNumber(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        FUNCTION maxSingleArray(nums PARAMETER, k PARAMETER)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND the LENGTH OF stack IS GREATER THAN zero AND the element at the last position of stack IS LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack as a list  
        END FUNCTION  

        FUNCTION merge(arr1 PARAMETER, arr2 PARAMETER)  
            SET result TO an empty list  
            WHILE the LENGTH OF arr1 PLUS the LENGTH OF arr2 IS GREATER THAN zero  
                IF arr1 IS GREATER THAN arr2 in lexicographical comparison THEN  
                    APPEND the element at the first position of arr1 TO result  
                    REMOVE the element at the first position FROM arr1  
                ELSE  
                    APPEND the element at the first position of arr2 TO result  
                    REMOVE the element at the first position FROM arr2  
                END IF  
            END WHILE  
            RETURN result  
        END FUNCTION  

        SET max_number TO an empty list  
        FOR i FROM zero TO k INCLUSIVE  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2 THEN  
                SET candidate TO merge the result of maxSingleArray with nums1 AND i AND maxSingleArray with nums2 AND k MINUS i  
                IF max_number IS LESS THAN candidate in lexicographical comparison THEN  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
