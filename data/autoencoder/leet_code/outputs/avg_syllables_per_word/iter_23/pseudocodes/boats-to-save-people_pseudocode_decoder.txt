CLASS Solution  
    FUNCTION numRescueBoats(people, limit)  
        SORT the list of people by their weights in ascending order  
        SET left TO zero  
        SET right TO the length of the list of people MINUS one  
        SET boats TO zero  
        WHILE the variable left is less than or equal to the variable right  
            IF the element at position left of the list of people PLUS the element at position right of the list of people IS LESS THAN OR EQUAL TO the variable limit  
                INCREMENT the variable left BY one  
            END IF  
            DECREMENT the variable right BY one  
            INCREMENT the variable boats BY one  
        END WHILE  
        RETURN the variable boats  
    END FUNCTION  
END CLASS
