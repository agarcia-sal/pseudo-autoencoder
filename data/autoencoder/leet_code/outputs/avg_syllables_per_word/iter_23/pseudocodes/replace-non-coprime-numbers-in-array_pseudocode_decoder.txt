CLASS Solution  
    FUNCTION replaceNonCoprimes WITH parameter nums  
        SET stack TO an empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack is greater than zero  
                SET g TO the greatest common divisor of the element at the last position of stack AND num  
                IF g EQUALS one  
                    BREAK the WHILE loop  
                END IF  
                SET num TO the product of the element at the last position of stack AND num DIVIDED BY g  
                REMOVE the element at the last position FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
