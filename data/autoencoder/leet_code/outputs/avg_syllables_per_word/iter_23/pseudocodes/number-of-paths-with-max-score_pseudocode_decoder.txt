CLASS Solution  
    FUNCTION pathsWithMaxScore(board)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF board  
        SET dp TO a two dimensional list of size n BY n WHERE each element IS a list containing NEGATIVE infinity AND zero  
        SET element at position n MINUS one OF row at position n MINUS one OF dp TO a list containing zero AND one  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the element at position j OF the element at position i OF board EQUALS to the character representing an obstacle OR EQUALS to the character representing the start THEN  
                    CONTINUE to the next iteration of the inner loop  
                END IF  
                IF the element at position j OF the element at position i OF board EQUALS to the character representing the end THEN  
                    SET value TO zero  
                ELSE  
                    SET value TO the integer parsed from the element at position j OF the element at position i OF board  
                END IF  
                FOR each pair x AND y IN the list containing the pairs one and zero, zero and one, one and one  
                    SET ni TO i PLUS x  
                    SET nj TO j PLUS y  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n THEN  
                        IF the first element of the element at position nj OF row at position ni OF dp IS GREATER THAN the first element of the element at position j OF row at position i OF dp THEN  
                            SET the element at position j OF row at position i OF dp TO a list containing the first element of the element at position nj OF row at position ni OF dp PLUS value AND the second element of the element at position nj OF row at position ni OF dp  
                        ELSE IF the first element of the element at position nj OF row at position ni OF dp EQUALS the first element of the element at position j OF row at position i OF dp THEN  
                            SET the second element of the element at position j OF row at position i OF dp TO the sum of the second element of the element at position j OF row at position i OF dp AND the second element of the element at position nj OF row at position ni OF dp MODULO MOD  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF the first element of the element at position zero OF row at position zero OF dp EQUALS NEGATIVE infinity THEN  
            RETURN a list containing zero AND zero  
        END IF  
        RETURN a list containing the first element of the element at position zero OF row at position zero OF dp AND the second element of the element at position zero OF row at position zero OF dp  
    END FUNCTION  
END CLASS
