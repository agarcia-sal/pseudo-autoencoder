CLASS Solution  
    FUNCTION uniqueLetterString WITH PARAMETER s  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO a list of twenty six elements EACH equal to negative one  
        SET last2 TO a list of twenty six elements EACH equal to negative one  
        SET result TO zero  
        
        FOR each index i AND character char IN s  
            SET idx TO the Unicode code point of char MINUS the Unicode code point of uppercase A  
            
            SET result TO the result PLUS the product of the difference of i MINUS the element at position idx of last AND the difference of the element at position idx of last MINUS the element at position idx of last2  
            SET result TO the remainder of result DIVIDED BY MOD  
            
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        
        FOR each integer i FROM zero TO twenty five  
            SET result TO the result PLUS the product of the difference of n MINUS the element at position i of last AND the difference of the element at position i of last MINUS the element at position i of last2  
            SET result TO the remainder of result DIVIDED BY MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
