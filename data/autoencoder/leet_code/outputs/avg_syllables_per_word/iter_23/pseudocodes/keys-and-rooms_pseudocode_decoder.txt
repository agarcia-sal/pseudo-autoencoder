CLASS Solution  
    FUNCTION canVisitAllRooms WITH parameter rooms  
        SET visited TO empty collection of unique elements  
        SET stack TO collection containing element zero  
        WHILE the LENGTH OF stack GREATER THAN zero holds  
            SET current_room TO the element at the last position of stack  
            REMOVE the element at the last position FROM stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room of rooms  
                IF key NOT PRESENT IN visited THEN  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
