CLASS Solution  
    FUNCTION getSum(a b)  
        SET MAX TO the maximum value of a thirty two bit integer  
        SET mask TO the mask value for thirty two bits  
        WHILE b NOT EQUALS zero holds  
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a AND b AND MASK applied  
            SET b TO the bitwise AND of carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX holds  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
