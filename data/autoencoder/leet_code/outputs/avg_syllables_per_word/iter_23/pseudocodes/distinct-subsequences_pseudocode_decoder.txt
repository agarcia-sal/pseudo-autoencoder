CLASS Solution  
    FUNCTION numDistinct(s string_parameter t string_parameter) RETURNS integer  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        SET dp TO a two dimensional list of zeros with m PLUS one rows and n PLUS one columns  
        FOR i FROM zero TO m  
            SET the element at row i and column zero of dp TO one  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of s EQUALS the element at position j MINUS one of t THEN  
                    SET the element at row i and column j of dp TO the element at row i MINUS one and column j MINUS one of dp PLUS the element at row i MINUS one and column j of dp  
                ELSE  
                    SET the element at row i and column j of dp TO the element at row i MINUS one and column j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row m and column n of dp  
    END FUNCTION  
END CLASS
