CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the sum of all elements in matchsticks  
        IF total_length MODULO four NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN descending order  
        
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS the LENGTH OF matchsticks THEN  
                FOR each side IN sides  
                    IF side NOT EQUALS side_length THEN  
                        RETURN False  
                    END IF  
                END FOR  
                RETURN True  
            END IF  
            
            FOR i FROM zero TO three  
                IF element at position i of sides PLUS element at position index of matchsticks LESS THAN OR EQUAL TO side_length THEN  
                    SET element at position i of sides TO element at position i of sides PLUS element at position index of matchsticks  
                    IF can_form_sides(sides, index PLUS one) THEN  
                        RETURN True  
                    END IF  
                    SET element at position i of sides TO element at position i of sides MINUS element at position index of matchsticks  
                END IF  
                
                IF element at position i of sides EQUALS zero THEN  
                    BREAK the loop  
                END IF  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN can_form_sides(a list of four zeros, zero)  
    END FUNCTION  
END CLASS
