CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of Boolean true values with the length of n  
        SET the element at position zero of is_prime TO Boolean false  
        SET the element at position one of is_prime TO Boolean false  
        FOR each start FROM two TO the integer part of the square root of n PLUS one  
            IF the element at position start of is_prime EQUALS Boolean true  
                FOR each multiple FROM start MULTIPLIED BY start TO n MINUS one WITH STEP start  
                    SET the element at position multiple of is_prime TO Boolean false  
                END FOR  
            END IF  
        END FOR  
        SET total_primes TO the count of elements in is_prime that EQUAL Boolean true  
        RETURN total_primes  
    END FUNCTION  
END CLASS
