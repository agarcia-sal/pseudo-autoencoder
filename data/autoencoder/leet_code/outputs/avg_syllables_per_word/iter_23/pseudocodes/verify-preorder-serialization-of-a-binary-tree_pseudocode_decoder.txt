CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the list of elements obtained by splitting preorder at each occurrence of a comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the single character hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN whether slots EQUALS zero  
    END FUNCTION  
END CLASS
