CLASS Solution  
    FUNCTION numOfArrays WITH PARAMETERS n m k  
        SET MOD TO the number one followed by nine zeros PLUS seven  
        
        INITIALIZE dp AS a three dimensional list where the first dimension is from zero TO n  
        each containing a two dimensional list where the first dimension is from zero TO m  
        each containing a list of length k PLUS one all initialized TO zero  
        
        FOR each j FROM one TO m  
            SET element at position one AT position j AT position one OF dp TO one  
        END FOR  
        
        FOR each i FROM two TO n  
            FOR each j FROM one TO m  
                FOR each p FROM one TO k  
                    
                    FOR each x FROM one TO j MINUS one  
                        SET element at position i AT position j AT position p OF dp TO element at position i AT position j AT position p OF dp PLUS element at position i MINUS one AT position x AT position p MINUS one OF dp  
                        SET element at position i AT position j AT position p OF dp TO element at position i AT position j AT position p OF dp MODULO MOD  
                    END FOR  
                    
                    SET element at position i AT position j AT position p OF dp TO element at position i AT position j AT position p OF dp PLUS element at position i MINUS one AT position j AT position p OF dp MULTIPLIED BY j  
                    SET element at position i AT position j AT position p OF dp TO element at position i AT position j AT position p OF dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each j FROM one TO m  
            SET result TO result PLUS element at position n AT position j AT position k OF dp MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
