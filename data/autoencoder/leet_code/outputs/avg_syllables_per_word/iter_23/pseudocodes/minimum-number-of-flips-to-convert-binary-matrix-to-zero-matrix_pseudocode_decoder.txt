CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list containing elements of tuple negative one zero tuple one zero tuple zero negative one tuple zero one tuple zero zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  
        
        FUNCTION matrix_to_tuple(matrix)  
            SET result TO empty list  
            FOR each row IN matrix  
                APPEND tuple constructed from row TO result  
            END FOR  
            RETURN tuple constructed from result  
        END FUNCTION  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple containing m elements each of which is a tuple containing n zeroes  
        SET initial_state TO result of calling matrix_to_tuple with mat  
        SET queue TO list containing one element which is a pair of initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  
        
        WHILE the LENGTH OF queue GREATER THAN zero  
            REMOVE first element FROM queue AND ASSIGN to current_state steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  
            
            SET current_matrix TO empty list  
            FOR each row IN current_state  
                CONVERT row to list AND APPEND TO current_matrix  
            END FOR  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deep copy OF current_matrix  
                    CALL flip WITH next_matrix i j  
                    SET next_state TO result of calling matrix_to_tuple with next_matrix  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND pair of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
