CLASS Solution  
    FUNCTION findMaxValueOfEquation(points WITH TYPE List OF List OF Integer, k WITH TYPE Integer) RETURNS Integer  
        SET max_value TO the smallest possible number  
        ASSIGN reference TO a new empty double ended queue called deque_  
        
        FOR each pair of xj and yj IN points  
            WHILE deque_ is not empty AND xj MINUS element at position zero of deque_ at position one is GREATER THAN k  
                REMOVE the element at the front FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND yj PLUS xj PLUS element at position zero of deque_ at position zero  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj IS GREATER THAN OR EQUAL TO element at the last position of deque_ at position zero  
                REMOVE the element at the back FROM deque_  
            END WHILE  
            
            APPEND the pair of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
