CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty parameter, d parameter)  
        SET n TO the LENGTH OF jobDifficulty parameter  
        IF n LESS THAN d parameter  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i parameter, d parameter)  
            IF d parameter EQUALS one  
                SET maximum difficulty TO the maximum value among the elements from position i parameter TO the END OF jobDifficulty parameter  
                RETURN maximum difficulty  
            END IF  
            
            SET minimum difficulty TO positive infinity  
            SET current maximum difficulty TO zero  
            
            FOR j parameter FROM i parameter TO n MINUS d parameter PLUS one MINUS one  
                SET current maximum difficulty TO the maximum value BETWEEN current maximum difficulty AND element at position j parameter OF jobDifficulty parameter  
                SET candidate difficulty TO current maximum difficulty PLUS the result of dp FUNCTION called with j parameter PLUS one AND d parameter MINUS one  
                IF candidate difficulty LESS THAN minimum difficulty  
                    SET minimum difficulty TO candidate difficulty  
                END IF  
            END FOR  
            
            RETURN minimum difficulty  
        END FUNCTION  
        
        RETURN the result of dp FUNCTION called with zero AND d parameter  
    END FUNCTION  
END CLASS
