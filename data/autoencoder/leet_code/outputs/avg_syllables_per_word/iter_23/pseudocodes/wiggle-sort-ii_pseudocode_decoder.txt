CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sortedNums TO A COPY OF nums SORTED IN ASCENDING ORDER  
        SET n TO THE LENGTH OF nums  
        SET left TO THE INTEGER DIVISION OF n MINUS one BY two  
        SET right TO n MINUS one  
        FOR i FROM zero TO n MINUS one  
            IF i MODULO two EQUALS zero  
                SET element at position i of nums TO element at position left of sortedNums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i of nums TO element at position right of sortedNums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
