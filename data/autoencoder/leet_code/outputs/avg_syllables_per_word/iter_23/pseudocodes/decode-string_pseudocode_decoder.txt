CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char is a digit character  
                SET d TO d PLUS char  
            ELSE IF char is an alphabetic character  
                SET a TO a PLUS char  
            ELSE IF char EQUALS the character representing an opening square bracket  
                APPEND the tuple of a and the integer conversion of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS the character representing a closing square bracket  
                SET p and n TO the last tuple popped FROM stack  
                SET a TO p PLUS a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
