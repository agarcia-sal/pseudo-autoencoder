CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeros with the LENGTH EQUAL TO two RAISED TO THE POWER OF n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET the element at position mask PLUS two RAISED TO THE POWER OF i of subset_sums TO the element at position mask of subset_sums PLUS the element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN the element at position mask of subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS GREATER THAN zero  
                SET candidate TO the GREATER THAN OR EQUAL TO MAXIMUM VALUE BETWEEN the CALL OF dp WITH k MINUS one AND MASK XOR submask AND the element at position submask of subset_sums  
                IF candidate IS LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, two RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
