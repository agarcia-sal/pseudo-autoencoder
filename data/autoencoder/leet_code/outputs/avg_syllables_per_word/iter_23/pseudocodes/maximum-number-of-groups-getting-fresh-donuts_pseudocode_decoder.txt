CLASS Solution  
    FUNCTION maxHappyGroups(batchSize groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of batchSize elements each set TO zero  
        
        FOR each group IN groups  
            INCREMENT the element at position the remainder of group DIVIDED BY batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder [variable number of remainder_count elements])  
            SET remainder_count TO the LIST OF remainder_count arguments  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF the element at position index of remainder_count EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                
                DECREMENT the element at position index of remainder_count BY one  
                
                SET new_remainder TO the remainder MINUS index, MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder followed by unpacked remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder followed by unpacked remainder_count)  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT the element at position index of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp zero followed by unpacked remainder_count  
    END FUNCTION  
END CLASS
