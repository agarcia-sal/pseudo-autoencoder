CLASS Solution  
    FUNCTION minimumWhiteTiles WITH PARAMETERS floor numCarpets carpetLen  
        SET n TO the LENGTH OF floor  
        CREATE a two dimensional list dp WITH dimensions n PLUS one BY numCarpets PLUS one FILLED WITH zero  
        
        FOR i FROM one TO n  
            SET dp element at position i zero TO dp element at position i MINUS one zero PLUS the RESULT OF checking whether element at position i MINUS one of floor EQUALS the character one AS numeric value  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at position i j TO dp element at position i MINUS one j PLUS the RESULT OF checking whether element at position i MINUS one of floor EQUALS the character one AS numeric value  
                SET dp element at position i j TO THE MINIMUM OF dp element at position i j AND dp element at position which is THE GREATER OF zero AND i MINUS carpetLen j MINUS one  
            END FOR  
        END FOR  
        
        RETURN dp element at position n numCarpets  
    END FUNCTION  
END CLASS
