CLASS Solution  
    FUNCTION stoneGameV(stoneValue)  
        SET prefix_sum TO a list of zeros with the length of the LENGTH OF stoneValue PLUS one  
        FOR i FROM zero TO the LENGTH OF stoneValue MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of stoneValue  
        END FOR  

        FUNCTION get_range_sum(left, right)  
            RETURN element at position right PLUS one of prefix_sum MINUS element at position left of prefix_sum  
        END FUNCTION  

        FUNCTION dp(left, right)  
            IF left EQUALS right THEN  
                RETURN zero  
            END IF  
            SET max_score TO zero  
            FOR i FROM left TO right MINUS one  
                SET left_sum TO get_range_sum with left and i  
                SET right_sum TO get_range_sum with i PLUS one and right  
                IF left_sum LESS THAN right_sum THEN  
                    IF max_score LESS THAN left_sum PLUS dp with left and i THEN  
                        SET max_score TO left_sum PLUS dp with left and i  
                    END IF  
                ELSE IF left_sum GREATER THAN right_sum THEN  
                    IF max_score LESS THAN right_sum PLUS dp with i PLUS one and right THEN  
                        SET max_score TO right_sum PLUS dp with i PLUS one and right  
                    END IF  
                ELSE  
                    IF max_score LESS THAN left_sum PLUS dp with left and i THEN  
                        SET max_score TO left_sum PLUS dp with left and i  
                    END IF  
                    IF max_score LESS THAN right_sum PLUS dp with i PLUS one and right THEN  
                        SET max_score TO right_sum PLUS dp with i PLUS one and right  
                    END IF  
                END IF  
            END FOR  
            RETURN max_score  
        END FUNCTION  

        RETURN dp with zero and the LENGTH OF stoneValue MINUS one  
    END FUNCTION  
END CLASS
