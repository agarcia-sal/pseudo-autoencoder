CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY the second element of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO the smallest possible value representing negative infinity  
        FOR each balloon IN points  
            IF the first element of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the second element of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
