CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze ball hole  
        SET directions TO list of tuples of number zero and number one and letter r number one and number zero and letter d number zero and negative number one and letter l negative number one and number zero and letter u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position number zero of maze  
        SET start TO tuple of ball  
        SET hole TO tuple of hole  
        SET pq TO list containing a tuple of number zero an empty string and start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            SET tuple dist path position TO remove and return smallest element from pq according to distance  
            SET x TO element at position number zero of position  
            SET y TO element at position number one of position  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF position IS IN visited  
                CONTINUE to next iteration of WHILE loop  
            END IF  
            
            ADD position TO visited  
            
            FOR each tuple dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO number zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO number zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO number zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AND element at position new_y PLUS dy of that element EQUALS number zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY number one  
                    IF new_x EQUALS the first element of hole AND new_y EQUALS the second element of hole  
                        BREAK the WHILE loop  
                    END IF  
                END WHILE  
                
                IF tuple of new_x and new_y IS NOT IN visited  
                    ADD tuple of dist PLUS count path concatenated with direction and tuple of new_x and new_y TO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
