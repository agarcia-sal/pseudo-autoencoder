CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER) RETURNS string  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO minus sign string  
        ELSE  
            SET sign TO empty string  
        END IF  
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
        SET quotient AND remainder TO the results of dividing numerator BY denominator with quotient and remainder  
        SET integer_part TO the string representation of quotient  
        IF remainder EQUALS zero  
            RETURN concatenation of sign AND integer_part  
        END IF  
        SET seen_remainders TO empty dictionary mapping remainder integers TO positions  
        SET fractional_part TO empty list of strings  
        WHILE remainder NOT EQUALS zero  
            IF remainder EXISTS AS KEY IN seen_remainders  
                SET start TO the value in seen_remainders AT remainder  
                SET non_repeating TO concatenation of all elements in fractional_part from position zero UP TO position start  
                SET repeating TO concatenation of all elements in fractional_part from position start TO the end  
                RETURN concatenation of sign AND integer_part AND period string AND non_repeating AND open parenthesis string AND repeating AND close parenthesis string  
            END IF  
            ASSIGN position in fractional_part as the current length of fractional_part TO seen_remainders AT remainder  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient AND remainder TO the results of dividing remainder BY denominator with quotient and remainder  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
        RETURN concatenation of sign AND integer_part AND period string AND concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
