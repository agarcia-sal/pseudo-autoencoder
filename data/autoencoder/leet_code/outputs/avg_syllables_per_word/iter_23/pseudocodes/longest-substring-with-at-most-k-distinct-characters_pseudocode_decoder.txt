CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s STRING_PARAMETER k INTEGER_PARAMETER) RETURNS INTEGER  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO length of s MINUS one  
            ASSIGN to char_map at element at position right of s the value right  
            
            IF length of char_map GREATER THAN k THEN  
                SET leftmost_char TO element in char_map with minimum value in char_map  
                SET left TO value in char_map at leftmost_char PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO greater value between max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
