CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero THEN  
            RETURN zero  
        END IF  
        
        SET num_counts TO an empty collection of key mapped to count  
        FOR each num IN nums  
            IF num IS IN num_counts THEN  
                INCREMENT the count mapped by num_counts at key num BY one  
            ELSE  
                SET the count mapped by num_counts at key num TO one  
            END IF  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN the keys of num_counts  
            IF k EQUALS zero THEN  
                IF the count mapped by num_counts at key num GREATER THAN one THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF the collection num_counts has a key that is the sum of num PLUS k THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
