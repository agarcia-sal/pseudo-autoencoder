CLASS Solution  
    FUNCTION smallestSufficientTeam(self req_skills people)  
        SET mapping d TO an empty dictionary  
        SET index i TO zero  
        FOR each skill s IN req_skills  
            SET mapping d at skill s TO the current index i  
            INCREMENT index i BY one  
        END FOR  
        SET total_skills m TO the LENGTH OF req_skills  
        SET total_people n TO the LENGTH OF people  
        SET list p TO a list of size n FILLED WITH zero  
        SET index i TO zero  
        FOR each skill_list ss IN people  
            FOR each skill s IN skill_list ss  
                SET variable p at position i TO variable p at position i PLUS BITWISE OR WITH ONE LEFT SHIFTED BY value mapping d at skill s  
            END FOR  
            INCREMENT index i BY one  
        END FOR  
        SET list f TO a list of size two RAISED TO total_skills FILLED WITH positive infinity  
        SET list g TO a list of size two RAISED TO total_skills FILLED WITH zero  
        SET list h TO a list of size two RAISED TO total_skills FILLED WITH zero  
        SET value f at position zero TO zero  
        SET variable i TO zero  
        WHILE variable i LESS THAN two RAISED TO total_skills  
            IF value f at position i EQUALS positive infinity  
                INCREMENT variable i BY one  
                CONTINUE to next iteration  
            END IF  
            SET variable j TO zero  
            WHILE variable j LESS THAN total_people  
                SET new_state TO variable i BITWISE OR variable p at position j  
                IF value f at position i PLUS one LESS THAN value f at position new_state  
                    SET value f at position new_state TO value f at position i PLUS one  
                    SET value g at position new_state TO variable j  
                    SET value h at position new_state TO variable i  
                END IF  
                INCREMENT variable j BY one  
            END WHILE  
            INCREMENT variable i BY one  
        END WHILE  
        SET variable i TO two RAISED TO total_skills MINUS one  
        SET list ans TO an empty list  
        WHILE variable i NOT EQUAL TO zero  
            APPEND value g at position i TO list ans  
            SET variable i TO value h at position i  
        END WHILE  
        RETURN list ans  
    END FUNCTION  
END CLASS
