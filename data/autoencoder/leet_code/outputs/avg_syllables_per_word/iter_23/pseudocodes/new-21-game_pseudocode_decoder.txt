CLASS Solution  
    FUNCTION new21Game(n k maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference GREATER THAN maxPts  
                    SET difference TO maxPts  
                END IF  
                RETURN difference DIVIDED BY maxPts  
            END IF  
            RETURN dfs(i PLUS one) PLUS dfs(i PLUS one) MINUS dfs(i PLUS maxPts PLUS one) DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
