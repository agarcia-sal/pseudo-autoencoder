CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR i FROM thirty-one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            INITIALIZE prefixes AS an empty set  
            FOR each num IN nums  
                SET prefix TO the result of shifting num right by i positions  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the result of current_candidate EXCLUSIVE OR prefix IS IN prefixes THEN  
                    SET max_xor TO current_candidate  
                    EXIT FROM the current loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
