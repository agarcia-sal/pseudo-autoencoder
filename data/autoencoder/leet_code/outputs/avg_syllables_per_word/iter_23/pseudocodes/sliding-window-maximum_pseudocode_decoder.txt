CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  
        
        SET deq TO a new empty double ended queue  
        SET result TO a new empty list  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq is not empty AND the element at position zero of deq LESS THAN i MINUS k PLUS one  
                REMOVE element from the front of deq  
            END IF  
            
            WHILE deq is not empty AND the element at position element at the last position of deq of nums LESS THAN the element at position i of nums  
                REMOVE element from the back of deq  
            END WHILE  
            
            APPEND i TO deq at the back  
            
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
