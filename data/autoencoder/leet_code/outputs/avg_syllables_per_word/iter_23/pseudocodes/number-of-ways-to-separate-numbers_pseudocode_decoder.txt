CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an empty collection for storing computed results  

        FUNCTION dp(index, prev_len)  
            IF index EQUALS n  
                RETURN one  
            END IF  
            IF the pair of index AND prev_len EXISTS IN memo  
                RETURN the element at the pair of index AND prev_len IN memo  
            END IF  

            SET count TO zero  
            FOR length FROM one TO n MINUS index PLUS one  
                IF the element at position index of num EQUALS the character zero  
                    BREAK the loop  
                END IF  

                SET current_num TO the substring from position index TO position index PLUS length MINUS one of num  

                IF prev_len EQUALS zero OR current_num IS GREATER THAN OR EQUAL TO the substring from position index MINUS prev_len TO position index MINUS one of num  
                    SET count TO count PLUS the result of dp WITH parameters index PLUS length AND length  
                    SET count TO count MODULO MOD  
                END IF  
            END FOR  

            SET the element at the pair of index AND prev_len IN memo TO count  
            RETURN count  
        END FUNCTION  

        RETURN the result of dp WITH parameters zero AND zero  
    END FUNCTION  
END CLASS
