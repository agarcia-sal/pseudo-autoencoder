CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF every character in sub_expr IS a digit  
                RETURN a list containing the integer value of sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR position i FROM zero TO the length of sub_expr MINUS one  
                IF the character at position i of sub_expr IS the plus symbol OR the minus symbol OR the multiplication symbol  
                    SET left_results TO the result of calling helper WITH the substring of sub_expr FROM the start TO position i MINUS one  
                    SET right_results TO the result of calling helper WITH the substring of sub_expr FROM position i PLUS one TO the end  
                    APPEND every element of compute USING left_results, right_results, and the character at position i of sub_expr TO results  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN the result of calling helper WITH expression  
    END FUNCTION  
END CLASS
