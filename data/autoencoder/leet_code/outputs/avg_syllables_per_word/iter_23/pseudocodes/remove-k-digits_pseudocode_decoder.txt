CLASS Solution  
    FUNCTION removeKdigits(num STRING, k INTEGER) RETURNS STRING  
        SET stack TO an empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at the last position of stack GREATER THAN digit  
                REMOVE the element at the last position FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO the elements from the first position TO the position at the length of stack MINUS k of stack  
        END IF  
        
        SET result TO the concatenation of all elements in final_stack WITH leading zeros REMOVED  
        
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN the string representing zero  
        END IF  
    END FUNCTION  
END CLASS
