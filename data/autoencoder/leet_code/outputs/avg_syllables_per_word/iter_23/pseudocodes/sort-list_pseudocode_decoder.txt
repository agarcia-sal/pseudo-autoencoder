CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values: list)  
    IF values EQUALS no elements  
        RETURN None  
    END IF  
    SET head TO new ListNode with value as the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list with the element next of p1 AND the element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head: Optional ListNode) RETURNS Optional ListNode  
        IF head EQUALS None OR the element next of head EQUALS None  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the element next of head  
            WHILE fast NOT EQUALS None AND the element next of fast NOT EQUALS None  
                SET slow TO the element next of slow  
                SET fast TO the element next of the element next of fast  
            END WHILE  
            SET mid TO the element next of slow  
            SET the element next of slow TO None  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO new ListNode with default values  
            SET current TO dummy  
            WHILE l1 NOT EQUALS None AND l2 NOT EQUALS None  
                IF the element val of l1 LESS THAN the element val of l2  
                    SET the element next of current TO l1  
                    SET l1 TO the element next of l1  
                ELSE  
                    SET the element next of current TO l2  
                    SET l2 TO the element next of l2  
                END IF  
                SET current TO the element next of current  
            END WHILE  
            IF l1 NOT EQUALS None  
                SET the element next of current TO l1  
            ELSE  
                SET the element next of current TO l2  
            END IF  
            RETURN the element next of dummy  
        END FUNCTION  

        SET left AND right TO split with head  
        SET left TO the result of self.sortList called with left  
        SET right TO the result of self.sortList called with right  

        RETURN merge with left AND right  
    END FUNCTION  
END CLASS
