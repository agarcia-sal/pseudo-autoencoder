CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode object with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode object with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode object with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left child of p and the left child of q AND the result of is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth IS EQUALS one THEN  
            RETURN a new TreeNode object with val and left child root  
        END IF  
        SET queue TO an empty double ended queue holding a tuple of root and one  
        WHILE the LENGTH OF queue IS GREATER THAN zero DO  
            SET node and level TO the tuple removed from the left end of queue  
            IF level IS EQUALS depth MINUS one THEN  
                SET node.left TO a new TreeNode object with val and left child node.left  
                SET node.right TO a new TreeNode object with val and right child node.right  
            ELSE IF level IS LESS THAN depth MINUS one THEN  
                IF node.left IS NOT None THEN  
                    APPEND a tuple of node.left and level PLUS one TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND a tuple of node.right and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
