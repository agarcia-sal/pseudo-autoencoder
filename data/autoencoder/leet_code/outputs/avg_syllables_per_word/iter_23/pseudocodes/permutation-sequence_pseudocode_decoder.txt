CLASS Solution  
    FUNCTION getPermutation(n k)  
        SET numbers TO an empty list  
        FOR each number FROM one TO n  
            APPEND number TO numbers  
        END FOR  
        DECREMENT k BY one  
        SET result TO an empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO the factorial of i MINUS one  
            SET index TO the integer quotient of k DIVIDED BY fact  
            APPEND the element at position index of numbers converted to string TO result  
            REMOVE the element at position index FROM numbers  
            SET k TO the remainder when k DIVIDED BY fact  
        END FOR  
        SET permutation_string TO the concatenation of all elements in result in sequence  
        RETURN permutation_string  
    END FUNCTION  
END CLASS
