CLASS Solution  
    FUNCTION largestOverlap(img1 WITH TYPE List OF List OF Integer, img2 WITH TYPE List OF List OF Integer) RETURNS Integer  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img1 AT position j EQUALS one  
                    APPEND the pair consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img2 AT position j EQUALS one  
                    APPEND the pair consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a mapping from pairs to integers with default value zero  
        FOR each pair consisting of x1 AND y1 IN points1  
            FOR each pair consisting of x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key the pair of dx AND dy in overlap_count BY one  
            END FOR  
        END FOR  
        RETURN the maximum value among all values in overlap_count OR zero if overlap_count is empty  
    END FUNCTION  
END CLASS
