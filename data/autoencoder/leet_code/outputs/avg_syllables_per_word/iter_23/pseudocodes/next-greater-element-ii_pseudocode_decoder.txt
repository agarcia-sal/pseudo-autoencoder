CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of n elements each set TO minus one  
        SET stack TO an empty list  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder WHEN i IS DIVIDED BY n  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position the last element of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_index TO the last element popped from stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
