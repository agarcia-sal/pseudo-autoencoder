CLASS Solution  
    FUNCTION visiblePoints(points parameter LIST OF LIST OF INTEGERS angle parameter INTEGER location parameter LIST OF INTEGERS) RETURNS INTEGER  
        SET angles TO empty list  
        SET extra_points TO zero  
        FOR each element WITH variables x AND y IN points  
            IF x EQUALS element at position zero of location AND y EQUALS element at position one of location THEN  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET difference in y TO y MINUS element at position one of location  
            SET difference in x TO x MINUS element at position zero of location  
            SET angle_rad TO the arctangent of difference in y over difference in x  
            SET angle_deg TO the angle_rad converted into degrees  
            APPEND angle_deg TO angles  
        END FOR  
        SORT angles in ascending order  
        SET doubled_angles TO a new list consisting of each element in angles PLUS three hundred sixty  
        SET angles TO concatenation of angles FOLLOWED BY doubled_angles  
        SET max_visible TO zero  
        SET left TO zero  
        FOR each element right FROM zero TO the length of angles MINUS one  
            WHILE element at position right of angles MINUS element at position left of angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET current_count TO right MINUS left PLUS one  
            IF current_count GREATER THAN max_visible THEN  
                SET max_visible TO current_count  
            END IF  
        END FOR  
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
