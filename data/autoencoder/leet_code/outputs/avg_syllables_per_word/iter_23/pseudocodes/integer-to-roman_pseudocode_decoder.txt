CLASS Solution  
    FUNCTION intToRoman(num)  
        SET list_of_values TO list of one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one  
        SET list_of_symbols TO list of the character M PLUS the character C followed by the character M PLUS the character D PLUS the character C followed by the character D PLUS the character C PLUS the character X followed by the character C PLUS the character L PLUS the character X followed by the character L PLUS the character X PLUS the character I followed by the character X PLUS the character V PLUS the character I followed by the character V PLUS the character I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR each iteration FROM zero TO the result of integer division of num BY the element at position i of list_of_values MINUS one  
                APPEND the element at position i of list_of_symbols TO roman_numeral  
                DECREMENT num BY the element at position i of list_of_values  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
