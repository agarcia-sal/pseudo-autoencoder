CLASS Solution  
    FUNCTION closestToTarget(arr parameter TAKES a list of integers target parameter TAKES an integer) RETURNS an integer  
        SET seen TO an empty collection of unique values  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET seen TO the set formed by applying bitwise AND operation between num AND each element x IN seen combined WITH the set containing num  
            FOR each val IN seen  
                SET current_difference TO the absolute value of val MINUS target  
                IF current_difference LESS THAN min_diff  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
