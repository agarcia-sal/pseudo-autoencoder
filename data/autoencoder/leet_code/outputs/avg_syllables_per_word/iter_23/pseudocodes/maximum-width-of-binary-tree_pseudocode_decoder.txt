CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is NOT empty  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS nothing AND q IS nothing THEN  
        RETURN True  
    ELSE IF p IS nothing OR q IS nothing THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left and q.left AND is_same_tree of p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS nothing THEN  
            RETURN zero  
        END IF  
        SET queue TO a double-ended queue initialized with the tuple of root and zero  
        SET max_width TO zero  
        WHILE queue is NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET _, first_index TO the first element of queue  
            FOR each iteration index FROM zero TO level_length MINUS one  
                REMOVE the first element of queue AND ASSIGN TO node, index  
                IF iteration index EQUALS level_length MINUS one THEN  
                    IF max_width LESS THAN index MINUS first_index PLUS one THEN  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left IS NOT nothing THEN  
                    APPEND the tuple of node.left and index MULTIPLIED BY two TO queue  
                END IF  
                IF node.right IS NOT nothing THEN  
                    APPEND the tuple of node.right and index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
