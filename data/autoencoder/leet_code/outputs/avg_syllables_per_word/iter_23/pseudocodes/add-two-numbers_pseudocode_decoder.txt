CLASS ListNode  
    FUNCTION __init__(self,val SET TO zero,next SET TO None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN elements of values FROM position two TO the end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1,p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND the result of is_same_list with parameters p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1,l2)  
        SET dummy_head TO new ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry NOT EQUALS zero  
            IF l1 IS NOT None  
                SET val1 TO the value of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the value of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode with total MODULO ten  
            SET current TO current.next  
            
            IF l1 IS NOT None  
                SET l1 TO l1.next  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
