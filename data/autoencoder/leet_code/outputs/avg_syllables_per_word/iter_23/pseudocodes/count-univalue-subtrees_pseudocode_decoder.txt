CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value being the element at position one of values  
    SET i TO one  
    SET queue TO a new empty deque collection  
    APPEND root TO queue  
    WHILE the length of queue is greater than zero  
        REMOVE the element from the front of queue and ASSIGN it TO node  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new TreeNode with value being the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new TreeNode with value being the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with first argument being the left child of p AND second argument being the left child of q  
            AND the result of is_same_tree with first argument being the right child of p AND second argument being the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS none THEN  
                RETURN true AND zero  
            END IF  
            SET is_left_unival AND left_count TO the result of calling helper with the left child of node  
            SET is_right_unival AND right_count TO the result of calling helper with the right child of node  
            IF is_left_unival AND is_right_unival THEN  
                IF (node.left EQUALS none OR the value of the left child of node EQUALS the value of node) AND (node.right EQUALS none OR the value of the right child of node EQUALS the value of node) THEN  
                    RETURN true AND left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN false AND left_count PLUS right_count  
        END FUNCTION  
        SET unused TO the first value returned by calling helper with root  
        SET count TO the second value returned by calling helper with root  
        RETURN count  
    END FUNCTION  
END CLASS
