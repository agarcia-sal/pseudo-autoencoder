CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count holds  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        SET start TO start PLUS the result of integer division of n MINUS one BY length  
        SET digit_string TO the string representation of start  
        SET digit_position TO the remainder of n MINUS one DIVIDED BY length  
        RETURN the element at digit_position of digit_string converted to integer  
    END FUNCTION  
END CLASS
