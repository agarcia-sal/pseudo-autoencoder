CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping digits zero to zero one to one six to nine eight to eight nine to six  
        SET valid_digits TO list of digits zero one six eight nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero  
                SET digit TO the remainder of num MODULO ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value from rotate_map at digit  
                SET num TO the integer division of num BY ten  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit current)  
            IF current GREATER THAN limit  
                RETURN zero  
            END IF  
            SET count TO one IF current NOT EQUALS zero AND is_confusing(current) OTHERWISE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                INCREMENT count BY the result of count_confusing_numbers WITH arguments limit and new_number  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN the result of count_confusing_numbers WITH argument n  
    END FUNCTION  
END CLASS
