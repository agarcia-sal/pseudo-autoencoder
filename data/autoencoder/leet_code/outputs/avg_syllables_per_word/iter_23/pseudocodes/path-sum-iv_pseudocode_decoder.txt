CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MINUS depth MULTIPLIED BY ten  
            SET value TO the remainder of num DIVIDED BY ten  
            SET the element with key depth and pos IN tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value in tree with key depth and pos OR nothing if not present  
            IF current_value EQUALS nothing THEN  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child_depth TO depth PLUS one  
            SET left_child_pos TO pos MULTIPLIED BY two MINUS one  
            SET right_child_depth TO depth PLUS one  
            SET right_child_pos TO pos MULTIPLIED BY two  
            SET left_child_key TO left_child_depth and left_child_pos  
            SET right_child_key TO right_child_depth and right_child_pos  
            IF left_child_key NOT IN tree AND right_child_key NOT IN tree THEN  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs with parameters left_child_depth, left_child_pos, new_path_sum  
            SET right_sum TO dfs with parameters right_child_depth, right_child_pos, new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs with parameters one, one, zero  
    END FUNCTION  
END CLASS
