CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two TO THE POWER OF thirty ONE MINUS one  
        SET INT_MIN TO NEGATIVE two TO THE POWER OF thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index LESS THAN n AND element at position index of s EQUALS the space character  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (element at position index of s EQUALS the plus character OR element at position index of s EQUALS the minus character)  
            IF element at position index of s EQUALS the minus character  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND element at position index of s IS A DIGIT  
            SET digit TO the integer VALUE OF element at position index of s  
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
