CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO an empty mapping with default value as empty list  
        SET indegree TO a list of zeros with length equal to numCourses  
        FOR each pair course AND prereq IN prerequisites  
            APPEND course TO the list accessed by prereq in graph  
            INCREMENT indegree at position course BY one  
        END FOR  
        SET queue TO a double-ended queue containing each course IN the range from zero TO numCourses MINUS one where indegree at position course EQUALS zero  
        WHILE the queue is not empty  
            REMOVE the element at the front of queue and ASSIGN it TO current_course  
            DECREMENT numCourses BY one  
            FOR each next_course IN the list accessed by current_course in graph  
                DECREMENT indegree at position next_course BY one  
                IF indegree at position next_course EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the predicate that numCourses EQUALS zero  
    END FUNCTION  
END CLASS
