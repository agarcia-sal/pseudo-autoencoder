CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY the element at position zero of each envelope IN ascending order AND by the element at position one of each envelope IN descending order  
        SET dp TO empty list  
        FOR each pair OF width IGNORE height IN envelopes  
            SET idx TO the position found by binary searching the leftmost place to insert height into dp to keep dp sorted  
            IF idx EQUALS the LENGTH OF dp  
                APPEND height TO dp  
            ELSE  
                SET the element at position idx of dp TO height  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
