CLASS Solution  
    FUNCTION findReplaceString(s string indices List of integer sources List of string targets List of string) RETURNS string  
        SET replacements TO the sorted list of tuples each containing element at position from indices element at position from sources element at position from targets  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each tuple of index source target IN replacements  
            APPEND the substring of s from position prev_end TO position index MINUS one TO result  
            IF the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the substring of s from position prev_end TO the last position OF s TO result  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
