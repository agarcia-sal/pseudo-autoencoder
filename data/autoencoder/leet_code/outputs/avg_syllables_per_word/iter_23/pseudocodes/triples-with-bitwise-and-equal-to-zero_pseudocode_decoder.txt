CLASS Solution  
    FUNCTION countTriplets(nums)  
        ASSIGN reference TO an empty dictionary with default integer zero named pair_and_count  
        FOR each element named i IN nums  
            FOR each element named j IN nums  
                INCREMENT the value associated with the result of the bitwise AND between i AND j in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element named k IN nums  
            FOR each key named pair_and and corresponding value named count IN the collection of items in pair_and_count  
                IF the bitwise AND between pair_and AND k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
