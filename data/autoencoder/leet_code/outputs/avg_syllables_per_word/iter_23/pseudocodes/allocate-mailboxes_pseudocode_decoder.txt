CLASS Solution  
    FUNCTION minDistance(houses AS List of integer, k AS integer) RETURNS integer  
        SORT houses IN ascending order  
        SET n TO the LENGTH OF houses  
        
        SET cost TO a two dimensional list of integers with n rows and n columns each initialized to zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at row i and column j TO cost element at row i PLUS one and column j MINUS one PLUS the element at position j of houses MINUS the element at position i of houses  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list of real numbers with n PLUS one rows and k PLUS one columns each initialized to infinity  
        SET dp element at row zero and column zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp element at row i and column j TO the minimum value of the current dp element at row i and column j AND the sum of dp element at row p and column j MINUS one AND cost element at row p and column i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp element at row n and column k  
    END FUNCTION  
END CLASS
