CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        SET MOD to one billion plus seven  
        SET a to one  
        SET e to one  
        SET i to one  
        SET o to one  
        SET u to one  
        FOR each integer from two TO n  
            SET a_next to the sum of e and i and u  
            SET e_next to the sum of a and i  
            SET i_next to the sum of e and o  
            SET o_next to i  
            SET u_next to the sum of i and o  
            SET a to the remainder of a_next DIVIDED BY MOD  
            SET e to the remainder of e_next DIVIDED BY MOD  
            SET i to the remainder of i_next DIVIDED BY MOD  
            SET o to the remainder of o_next DIVIDED BY MOD  
            SET u to the remainder of u_next DIVIDED BY MOD  
        END FOR  
        SET total to the sum of a and e and i and o and u  
        SET result to the remainder of total DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
