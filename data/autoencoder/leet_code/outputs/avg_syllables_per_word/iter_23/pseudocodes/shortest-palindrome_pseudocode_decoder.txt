CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reversed order of characters in check_str  
        END FUNCTION  

        FOR i FROM the LENGTH OF s TO zero DECREMENT BY one  
            IF is_palindrome(substring from position zero TO position i MINUS one of s)  
                SET to_add TO reversed order of characters in substring from position i TO the LENGTH OF s MINUS one of s  
                RETURN concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
