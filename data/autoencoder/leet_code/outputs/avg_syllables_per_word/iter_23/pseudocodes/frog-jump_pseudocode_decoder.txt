CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN false  
        END IF  
        
        SET stone_set TO a set containing all elements of stones  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS element at the last position of stones  
                RETURN true  
            END IF  
            
            IF the tuple of position and jump EXISTS AS a key IN memo  
                RETURN the value associated with the tuple of position and jump IN memo  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND the sum of position PLUS next_jump EXISTS IN stone_set  
                    IF can_jump_to(the sum of position PLUS next_jump, next_jump)  
                        SET memo at key tuple of position and jump TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET memo at key tuple of position and jump TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_jump_to(element at the first position of stones, zero)  
    END FUNCTION  
END CLASS
