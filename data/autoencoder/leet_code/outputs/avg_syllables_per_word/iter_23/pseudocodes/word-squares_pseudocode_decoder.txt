CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO an empty mapping from prefix to list of words  
        FOR each word IN words  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO the substring from position zero TO position index MINUS one of word  
                APPEND word TO the list at prefix_to_words for prefix  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the first element of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO the concatenation of elements at position the LENGTH OF square of each word in square  
            FOR each candidate IN the list at prefix_to_words for prefix  
                CALL backtrack with the concatenation of square AND a list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack with a list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
