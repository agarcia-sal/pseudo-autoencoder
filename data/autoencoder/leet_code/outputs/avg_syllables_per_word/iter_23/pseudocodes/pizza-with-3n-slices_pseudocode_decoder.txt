CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            SET pick TO the element at position start of slices PLUS the CALL TO dp with start INCREMENTED BY two and end and k DECREMENTED BY one  
            SET skip TO the CALL TO dp with start INCREMENTED BY one and end and k  
            RETURN the GREATER THAN of pick AND skip  
        END FUNCTION  
        
        SET case1 TO the CALL TO dp with one and the LENGTH OF slices MINUS one and n  
        SET case2 TO the CALL TO dp with zero and the LENGTH OF slices MINUS two and n  
        
        RETURN the GREATER THAN of case1 AND case2  
    END FUNCTION  
END CLASS
