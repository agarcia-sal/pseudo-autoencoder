CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_of_values_less_than_or_equal_to_x TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count_of_values_less_than_or_equal_to_x BY one  
                END IF  
            END FOR  
            RETURN count_of_values_less_than_or_equal_to_x GREATER THAN x  
        END FUNCTION  
        RETURN the smallest index in the range from zero TO the LENGTH OF nums MINUS one such that FUNCTION f returns True at that index  
    END FUNCTION  
END CLASS
