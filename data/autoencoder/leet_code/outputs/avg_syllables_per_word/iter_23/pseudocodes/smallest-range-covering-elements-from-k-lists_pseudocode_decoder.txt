CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            ADD tuple of element at position zero of element at position index of nums AND index AND zero TO min_heap USING heap push  
            SET max_value TO max_value GREATER THAN element at position zero of element at position index of nums RETURN max_value ELSE element at position zero of element at position index of nums  
        END FOR  
        SET smallest_range TO infinity  
        SET result TO list containing negative one hundred thousand AND one hundred thousand  
        WHILE min_heap IS NOT empty  
            REMOVE smallest element FROM min_heap USING heap pop AND ASSIGN TO min_value AND list_index AND element_index  
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF element at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of element at position list_index of nums  
                ADD tuple of next_value AND list_index AND element_index PLUS one TO min_heap USING heap push  
                SET max_value TO max_value GREATER THAN next_value RETURN max_value ELSE next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
