CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty mapping  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN the value ASSOCIATED WITH state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring FROM position index TO position index PLUS one OF state EQUALS two plus signs THEN  
                    SET new_state TO the concatenation OF the substring FROM the start TO position index OF state PLUS two minus signs PLUS the substring FROM position index PLUS two TO the end OF state  
                    IF the result OF can_win_helper WITH new_state IS EQUAL TO false THEN  
                        ASSIGN the value true TO the mapping memo AT key state  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            ASSIGN the value false TO the mapping memo AT key state  
            RETURN false  
        END FUNCTION  
        
        RETURN the result OF calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
