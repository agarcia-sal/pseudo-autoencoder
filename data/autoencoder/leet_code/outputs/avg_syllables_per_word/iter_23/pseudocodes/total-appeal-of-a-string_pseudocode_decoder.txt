CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list of twenty six elements each set to negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each index i AND character char IN s WITH enumeration  
            SET character_index TO the numerical position of character char IN the alphabet starting from zero  
            INCREMENT current_appeal BY the difference of index i MINUS the element at position character_index of list last  
            INCREMENT total_appeal BY current_appeal  
            SET the element at position character_index of list last TO index i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
