CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN sublist of values from position two TO end  
        SET node TO new ListNode with val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list with next of p1 and next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween(self head left right)  
        IF head EQUALS None OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with zero  
        SET next of dummy TO head  
        SET prev TO dummy  
        FOR counter FROM one TO left MINUS one  
            SET prev TO next of prev  
        END FOR  
        SET current TO next of prev  
        FOR counter FROM one TO right MINUS left  
            SET next_node TO next of current  
            SET next of current TO next of next_node  
            SET next of next_node TO next of prev  
            SET next of prev TO next_node  
        END FOR  
        RETURN next of dummy  
    END FUNCTION  
END CLASS
