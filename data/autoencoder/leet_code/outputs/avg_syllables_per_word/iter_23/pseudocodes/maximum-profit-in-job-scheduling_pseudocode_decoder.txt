CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the list of tuples combining element at position of startTime WITH element at position of endTime WITH element at position of profit SORTED BY the second element of each tuple in ascending order  
        SET end_times TO the list of second elements of each tuple in jobs  
        SET dp TO a list of zero values with the length EQUAL TO the length of jobs PLUS one  
        FOR i FROM one TO the length of jobs  
            SET j TO the result of binary search for the position to insert the first element of the tuple at position i MINUS one of jobs in end_times so that the order is maintained  
            IF the element at position i MINUS one of jobs EQUALS the start time of the current job  
                DO NOTHING  
            END IF  
            SET dp at position i TO the maximum value BETWEEN the element at position i MINUS one of dp AND the sum of the element at position j of dp PLUS the third element of the tuple at position i MINUS one of jobs  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
