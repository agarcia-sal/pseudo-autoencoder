CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode OBJECT with value at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode OBJECT with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode OBJECT with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder WITH PARAMETERS self root  
        IF root IS None  
            RETURN an empty list  
        END IF  
        SET column_table TO a dictionary with default value as an empty list  
        SET queue TO a double-ended queue containing a tuple of root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            REMOVE the element from the left end of queue AND SPLIT into node AND column_index  
            IF node IS NOT None  
                APPEND the value of node TO the list at key column_index in column_table  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple of node.left AND column_index MINUS one TO queue  
                APPEND a tuple of node.right AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result_list TO an empty list  
        FOR each x FROM min_column TO max_column  
            APPEND the list at key x in column_table TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
