CLASS Solution  
    FUNCTION rangeBitwiseAnd(left parameter, right parameter)  
        SET shift TO zero  
        WHILE left parameter LESS THAN right parameter  
            SET left parameter TO left parameter DIVIDED BY two discarding remainder  
            SET right parameter TO right parameter DIVIDED BY two discarding remainder  
            INCREMENT shift BY one  
        END WHILE  
        RETURN left parameter MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
