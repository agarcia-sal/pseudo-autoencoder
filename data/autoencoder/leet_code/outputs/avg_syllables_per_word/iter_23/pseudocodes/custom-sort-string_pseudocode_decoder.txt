CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        FOR each i FROM zero TO the LENGTH OF order MINUS one  
            SET character TO element at position i of order  
            SET value TO i  
            SET d at key character TO value  
        END FOR  
        SET sorted_characters TO a list of characters from s sorted according to the following rule:  
            FOR each character x IN s  
                GET the value from dictionary d with key x if it exists OTHERWISE zero  
            END FOR  
        SET result_string TO the concatenation of all characters in sorted_characters in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
