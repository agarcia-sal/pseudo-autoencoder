CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start first second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the string representation of first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring of num from position start TO position start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN the result of is_valid_sequence with start PLUS length second the integer value of expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the integer value of n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the substring of num from the beginning TO position i  
                SET second TO the substring of num from position i TO position j  
                
                IF the LENGTH OF first GREATER THAN one AND the element at position zero of first EQUALS the character zero  
                    CONTINUE  
                END IF  
                IF the LENGTH OF second GREATER THAN one AND the element at position zero of second EQUALS the character zero  
                    CONTINUE  
                END IF  
                
                IF the result of is_valid_sequence with start j first converted to an integer second converted to an integer EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
