CLASS Solution  
    FUNCTION carFleet(target parameter position parameter speed parameter)  
        SET paired_sorted_position_speed TO the list of pairs formed by matching each element of position and speed sorted in order from the largest first based on the first element of each pair  
        SET time_to_target TO an empty list  
        FOR each pair IN paired_sorted_position_speed  
            SET p TO the first element of pair  
            SET s TO the second element of pair  
            SET time_for_car TO the result of target MINUS p DIVIDED BY s  
            APPEND time_for_car TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
