CLASS Solution  
    FUNCTION solve(board)  
        IF board EQUALS empty OR element at position zero of board EQUALS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        FUNCTION dfs(x y)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of board at position y NOT EQUALS the character spoken as OH  
                RETURN  
            END IF  
            SET element at position x of board at position y TO the character spoken as ESS  
            CALL dfs WITH x PLUS one y  
            CALL dfs WITH x MINUS one y  
            CALL dfs WITH x y PLUS one  
            CALL dfs WITH x y MINUS one  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            CALL dfs WITH i zero  
            CALL dfs WITH i n MINUS one  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            CALL dfs WITH zero j  
            CALL dfs WITH m MINUS one j  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS the character spoken as OH  
                    SET element at position i of board at position j TO the character spoken as EX  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS the character spoken as ESS  
                    SET element at position i of board at position j TO the character spoken as OH  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
