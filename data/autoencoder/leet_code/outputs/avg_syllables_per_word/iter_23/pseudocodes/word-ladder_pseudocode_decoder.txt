CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        
        SET word_dict TO a new default dictionary that produces empty lists  
        SET word_length TO the LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring from start TO position i MINUS one of word PLUS an asterisk PLUS substring from position i PLUS one TO end of word  
                APPEND word TO the list at element intermediate_key of word_dict  
            END FOR  
        END FOR  
        
        SET queue TO a new double ended queue containing the tuple of beginWord and one  
        SET visited TO a new set containing beginWord  
        
        WHILE queue is not empty  
            SET current_word TO the first element of the first tuple in queue  
            SET current_length TO the second element of the first tuple in queue  
            REMOVE the first element FROM queue  
            
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring from start TO position i MINUS one of current_word PLUS an asterisk PLUS substring from position i PLUS one TO end of current_word  
                
                FOR each word IN the list at element intermediate_word of word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
