CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two TO zero DECREMENT BY one  
            FOR col FROM zero TO the LENGTH OF the element at position row of triangle MINUS one  
                SET value_below_left TO the element at position col of the element at position row PLUS one of triangle  
                SET value_below_right TO the element at position col PLUS one of the element at position row PLUS one of triangle  
                SET minimum_adjacent_value TO the smaller OF value_below_left AND value_below_right  
                SET element at position col of the element at position row of triangle TO the element at position col of the element at position row of triangle PLUS minimum_adjacent_value  
            END FOR  
        END FOR  
        RETURN the element at position zero of the element at position zero of triangle  
    END FUNCTION  
END CLASS
