CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the values collection EQUALS empty  
        RETURN null  
    END IF  
    SET head TO a new ListNode constructed with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the end  
        SET node TO a new ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS null AND p2 EQUALS null  
        RETURN True  
    END IF  
    IF p1 EQUALS null OR p2 EQUALS null  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight with parameters head AND k  
        IF head EQUALS null OR the next attribute of head EQUALS null OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next attribute of tail IS NOT null  
            SET tail TO the next attribute of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next attribute of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR an iteration variable ignored FROM one TO steps_to_new_tail  
            SET new_tail TO the next attribute of new_tail  
        END FOR  
        SET new_head TO the next attribute of new_tail  
        SET the next attribute of new_tail TO null  
        RETURN new_head  
    END FUNCTION  
END CLASS
