CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO combining difficulty and profit into pairs such that each pair contains the element at position index of difficulty and the element at position index of profit for each index  
        SORT jobs in ascending order by the first element of each pair representing difficulty  
        SORT worker in ascending order by their ability  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the difficulty component of the job at position job_index is LESS THAN OR EQUAL TO ability  
                SET best_profit TO the greater value between best_profit and the profit component of the job at position job_index  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
