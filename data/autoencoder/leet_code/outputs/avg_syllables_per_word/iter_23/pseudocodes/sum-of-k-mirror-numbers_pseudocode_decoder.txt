CLASS Solution  
    FUNCTION kMirror(k parameter INTEGER, n parameter INTEGER) RETURNS INTEGER  
        FUNCTION is_palindrome(s parameter STRING) RETURNS BOOLEAN  
            RETURN the condition that s EQUALS the string formed by taking elements of s from last TO first in reverse order  
        END FUNCTION  
        
        FUNCTION to_base_k(num parameter INTEGER, k parameter INTEGER) RETURNS STRING  
            IF the condition that num EQUALS zero THEN  
                RETURN the string zero  
            END IF  
            SET digits TO an empty list  
            WHILE the condition that num DOES NOT EQUAL zero HOLDS  
                APPEND the integer value of num MODULO k TO digits  
                SET num TO the integer division of num BY k  
            END WHILE  
            RETURN the concatenation of string representations of elements in digits taken from last TO first in reverse order  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length parameter INTEGER) RETURNS A GENERATOR OF STRING  
            IF the condition that length EQUALS one THEN  
                FOR i FROM one TO nine  
                    YIELD the string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE power of the integer division result of (length MINUS one) BY two  
                SET end TO ten RAISED TO THE power of the integer division result of (length PLUS one) BY two  
                FOR half FROM start TO one LESS THAN end  
                    SET half_str TO string representation of half  
                    IF the condition that length MODULO two EQUALS zero THEN  
                        SET full TO half_str CONCATENATED WITH the reverse of half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH the reverse of half_str EXCLUDING the last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the condition that the LENGTH OF k_mirror_numbers IS LESS THAN n HOLDS  
            FOR each pal IN generate_palindromes(length)  
                SET num TO the integer representation of pal  
                IF the condition that is_palindrome(the result of to_base_k(num, k)) HOLDS THEN  
                    APPEND num TO k_mirror_numbers  
                    IF the condition that the LENGTH OF k_mirror_numbers EQUALS n THEN  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the sum of all elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
