CLASS Solution  
    FUNCTION isTransformable(s parameter, t parameter) RETURNS boolean  
        DEFINE pos AS mapping FROM digit TO list of positions initially empty  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            CONVERT character TO digit  
            ADD index TO the list associated WITH digit in pos preserving ascending order  
        END FOR  
        
        FOR index FROM zero TO the LENGTH OF t MINUS one  
            SET character TO element at position index of t  
            CONVERT character TO digit  
            IF the list associated WITH digit in pos IS empty THEN  
                RETURN boolean false  
            END IF  
            FOR smaller_digit FROM zero TO digit MINUS one  
                IF the list associated WITH smaller_digit in pos IS NOT empty AND the first element of the list associated WITH smaller_digit in pos IS LESS THAN the first element of the list associated WITH digit in pos THEN  
                    RETURN boolean false  
                END IF  
            END FOR  
            REMOVE the first element FROM the list associated WITH digit in pos  
        END FOR  
        
        RETURN boolean true  
    END FUNCTION  
END CLASS
