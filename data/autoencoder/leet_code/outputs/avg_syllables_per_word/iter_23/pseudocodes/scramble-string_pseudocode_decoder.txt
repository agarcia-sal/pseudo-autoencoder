CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 s2  
        IF the value of s1 EQUALS the value of s2  
            RETURN True  
        END IF  
        IF the sorted characters of s1 NOT EQUALS the sorted characters of s2  
            RETURN False  
        END IF  
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF the result of calling isScramble on the substring of s1 from position one TO position i AND the substring of s2 from position one TO position i IS True AND the result of calling isScramble on the substring of s1 from position i PLUS one TO the end AND the substring of s2 from position i PLUS one TO the end IS True  
                RETURN True  
            END IF  
            IF the result of calling isScramble on the substring of s1 from position one TO position i AND the substring of s2 from the end MINUS i PLUS one TO the end IS True AND the result of calling isScramble on the substring of s1 from position i PLUS one TO the end AND the substring of s2 from position one TO the end MINUS i IS True  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
