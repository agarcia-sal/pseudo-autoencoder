CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours AND minutes TO the results of splitting time into components divided by the word indicating colon converted to integers  
            RETURN the result of hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  

        SET minutes_list TO the list formed by applying to_minutes to each element of timePoints  
        SORT the elements of minutes_list in ascending order  
        SET min_diff TO a value representing positive infinity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO the result of one thousand four hundred forty MINUS the element at the last position of minutes_list PLUS the element at the first position of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
