CLASS Solution  
    FUNCTION rearrangeSticks(n integer, k integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list with n PLUS one elements each being a list with k PLUS one zeros  
        SET element at position zero of element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET value_of_first_term TO element at position j MINUS one of element at position i MINUS one of dp  
                SET value_of_second_term TO i MINUS one MULTIPLIED BY element at position j of element at position i MINUS one of dp  
                SET element at position j of element at position i of dp TO the remainder after division of the sum of value_of_first_term PLUS value_of_second_term by MOD  
            END FOR  
        END FOR  
        RETURN element at position k of element at position n of dp  
    END FUNCTION  
END CLASS
