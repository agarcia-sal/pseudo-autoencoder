CLASS Solution  
    FUNCTION isMatch(s parameter s, p parameter p) RETURNS boolean  
        SET dp TO a list of lists where each inner list contains False repeated the LENGTH OF p PLUS one and the outer list repeats this inner list LENGTH OF s PLUS one times  
        SET element at position zero at position zero of dp TO True  
        FOR j FROM one TO LENGTH OF p PLUS one MINUS one  
            IF element at position j MINUS one of p EQUALS the character asterisk  
                SET element at position zero at position j of dp TO element at position zero at position j MINUS one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO LENGTH OF s PLUS one MINUS one  
            FOR j FROM one TO LENGTH OF p PLUS one MINUS one  
                IF element at position j MINUS one of p EQUALS the character asterisk  
                    SET element at position i at position j of dp TO the logical OR of element at position i MINUS one at position j of dp AND element at position i at position j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS the question mark character OR element at position i MINUS one of s EQUALS element at position j MINUS one of p  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s at position LENGTH OF p of dp  
    END FUNCTION  
END CLASS
