CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i of board MINUS one  
                IF element at position i of board AT position j EQUALS capital letter X THEN  
                    IF (i EQUALS zero OR element at position i MINUS one of board AT position j EQUALS period) AND (j EQUALS zero OR element at position i of board AT position j MINUS one EQUALS period) THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
