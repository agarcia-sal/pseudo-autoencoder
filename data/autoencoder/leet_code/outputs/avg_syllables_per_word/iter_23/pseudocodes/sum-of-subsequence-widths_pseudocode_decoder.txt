CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO ten EXPONENTIATED TO the power of nine PLUS seven  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        CREATE power AS a list WITH n elements ALL SET TO one  
        
        FOR i FROM one TO n MINUS one  
            SET element at position i of power TO the element at position i MINUS one of power MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS the element at position i of nums MULTIPLIED BY the difference BETWEEN the element at position i of power AND the element at position n MINUS i MINUS one of power ALL MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
