CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the LIST OF the FIRST element OF each pair IN flowers SORTED IN ascending order  
        SET end_times TO the LIST OF the SECOND element OF each pair IN flowers SORTED IN ascending order  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the POSITION WHERE person WOULD BE INSERTED TO THE RIGHT IN start_times TO MAINTAIN ORDER  
            SET ended TO the POSITION WHERE person WOULD BE INSERTED TO THE LEFT IN end_times TO MAINTAIN ORDER  
            APPEND the VALUE OF started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
