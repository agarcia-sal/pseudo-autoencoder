CLASS TreeNode  
    FUNCTION __init__ with parameters val MINUS default zero left MINUS default None right MINUS default None  
        SET the attribute val OF this instance TO val  
        SET the attribute left OF this instance TO left  
        SET the attribute right OF this instance TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values as list  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element obtained by removing the first element of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the attribute left OF node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the attribute left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the attribute right OF node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the attribute right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p AND q  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the attribute val OF p NOT EQUALS the attribute val OF q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters the attribute left OF p AND the attribute left OF q AND the result of is_same_tree with parameters the attribute right OF p AND the attribute right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree with parameters preorder as list of integers AND inorder as list of integers returning an optional TreeNode  
        IF preorder EQUALS an empty list OR inorder EQUALS an empty list THEN  
            RETURN None  
        END IF  
        SET root_val TO the element at position zero of preorder  
        SET root TO a new instance of TreeNode initialized with root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET the attribute left OF root TO the result of calling this function buildTree with the parameters being the sublist of preorder from position one TO position root_index_inorder PLUS one AND the sublist of inorder from position zero TO position root_index_inorder  
        SET the attribute right OF root TO the result of calling this function buildTree with the parameters being the sublist of preorder from position root_index_inorder PLUS one TO the end AND the sublist of inorder from position root_index_inorder PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
