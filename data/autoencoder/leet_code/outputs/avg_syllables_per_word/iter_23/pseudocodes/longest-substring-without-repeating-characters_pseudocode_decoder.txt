CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO an empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end AND character char IN the enumeration of s  
            IF char EXISTS AS A KEY IN char_index_map AND the value at char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            SET max_length TO the greater value BETWEEN max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
