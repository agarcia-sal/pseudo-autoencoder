CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET number_of_rows TO the LENGTH OF strs  
        SET number_of_columns TO the LENGTH OF the element at position zero of strs  
        SET dp TO a list of number_of_columns elements each set TO one  
        FOR index_i FROM one TO number_of_columns MINUS one  
            FOR index_j FROM zero TO index_i MINUS one  
                IF for all index_k FROM zero TO number_of_rows MINUS one the element at position index_j of the element at position index_k of strs IS LESS THAN OR EQUAL TO the element at position index_i of the element at position index_k of strs  
                    SET dp at position index_i TO the GREATER OF dp at position index_i AND dp at position index_j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence TO the MAXIMUM VALUE in dp  
        RETURN number_of_columns MINUS longest_non_decreasing_subsequence  
    END FUNCTION  
END CLASS
