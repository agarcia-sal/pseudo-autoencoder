CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO a new Counter object initialized with s  
        SET sorted_characters TO the elements that are keys of frequency sorted so that for any two elements the one with frequency greater THAN the other comes first  
        SET result TO the empty string  
        FOR each character IN sorted_characters  
            APPEND character REPEATED frequency element at position character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
