CLASS Solution  
    FUNCTION maxProfit(prices, fee)  
        SET cash TO zero  
        SET hold TO negative infinity  
        FOR each price IN prices  
            IF cash GREATER THAN hold PLUS price MINUS fee  
                SET cash TO cash  
            ELSE  
                SET cash TO hold PLUS price MINUS fee  
            END IF  
            IF hold GREATER THAN cash MINUS price  
                SET hold TO hold  
            ELSE  
                SET hold TO cash MINUS price  
            END IF  
        END FOR  
        RETURN cash  
    END FUNCTION  
END CLASS
