CLASS Solution  
    FUNCTION numWays(n, k)  
        IF n EQUALS zero THEN  
            RETURN zero  
        END IF  
        IF n EQUALS one THEN  
            RETURN k  
        END IF  
        IF n EQUALS two THEN  
            RETURN k MULTIPLIED BY k  
        END IF  
        
        SET same TO k  
        SET diff TO k MULTIPLIED BY the difference of k MINUS one  
        SET total TO the sum of same PLUS diff  
        
        FOR i FROM three TO the sum of n DO  
            SET new_same TO diff  
            SET new_diff TO the product of total MULTIPLIED BY the difference of k MINUS one  
            
            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO the sum of same PLUS diff  
        END FOR  
        
        RETURN total  
    END FUNCTION  
END CLASS
