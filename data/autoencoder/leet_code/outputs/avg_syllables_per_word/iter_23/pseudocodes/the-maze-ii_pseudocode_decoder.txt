CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET number_of_rows TO the LENGTH OF maze  
        SET number_of_columns TO the LENGTH OF element at position zero of maze  
        SET directions TO a collection containing the pair of one and zero, the pair of negative one and zero, the pair of zero and one, and the pair of zero and negative one  
        SET start TO the tuple formed by the elements of start  
        SET destination TO the tuple formed by the elements of destination  
        SET visited TO an empty set  
        SET priority_queue TO a list containing the pair of zero and start  

        WHILE priority_queue is not empty  
            REMOVE the element with the smallest first component from priority_queue and ASSIGN its first component to dist and its second component to a pair containing x and y  

            IF the pair containing x and y EQUALS destination  
                RETURN dist  
            END IF  

            IF the pair containing x and y IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  

            ADD the pair containing x and y TO visited  

            FOR each pair containing dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN number_of_rows AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN number_of_columns AND the element at position new_x PLUS dx of maze and then at position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                END WHILE  

                IF the pair containing new_x and new_y IS NOT IN visited  
                    ADD the pair containing dist PLUS count and the pair containing new_x and new_y TO priority_queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
