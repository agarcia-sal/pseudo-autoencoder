CLASS Solution  
    FUNCTION countSubstrings WITH PARAMETER s  
        FUNCTION expand_around_center WITH PARAMETERS left AND right  
            SET count TO zero  
            WHILE left GREATER THAN OR EQUAL TO zero AND right LESS THAN the LENGTH OF s AND the element at position left of s EQUALS the element at position right of s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        
        SET total_palindromes TO zero  
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            SET total_palindromes TO total_palindromes PLUS the RESULT OF expand_around_center WITH PARAMETERS i AND i  
            SET total_palindromes TO total_palindromes PLUS the RESULT OF expand_around_center WITH PARAMETERS i AND i PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
