CLASS Solution  
    FUNCTION minMutation(start string  end string  bank list of strings) RETURNS integer  
        SET s TO the set of elements in bank  
        SET q TO a double-ended queue with a tuple consisting of start paired with zero  
        SET mp TO a mapping of character A to string TCG AND character T to string ACG AND character C to string ATG AND character G to string ATC  
        WHILE the queue q is not empty  
            REMOVE the element from the left side of queue q AND ASSIGN the first item of the tuple to variable t AND ASSIGN the second item of the tuple to variable step  
            IF variable t EQUALS variable end  
                RETURN the value of step  
            END IF  
            FOR index i FROM zero TO the length of string t MINUS one  
                SET variable v TO the character at position i of string t  
                FOR each character j IN the string mapped by key v in mapping mp  
                    SET next TO the concatenation of substring from position zero TO position i MINUS one of string t PLUS character j PLUS substring from position i PLUS one TO the end of string t  
                    IF string next IS an element of set s  
                        APPEND a tuple consisting of next paired with step PLUS one TO the right side of queue q  
                        REMOVE element next FROM set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
