CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings from one TO n  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO string concatenation of an open parenthesis CONCATENATED WITH element at position i of teams CONCATENATED WITH a comma CONCATENATED WITH element at position n MINUS one MINUS i of teams CONCATENATED WITH a close parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
