CLASS Solution  
    FUNCTION fallingSquares(positions)  
        SET max_heights TO empty list  
        SET intervals TO empty list  
        SET current_max_height TO zero  
        FOR each pair of left AND side_length IN positions  
            SET right TO the value of left PLUS the value of side_length MINUS one  
            SET height TO the value of side_length  
            FOR each triple of interval_left interval_right AND interval_height IN intervals  
                IF the value of left LESS THAN OR EQUAL TO the value of interval_right AND the value of right GREATER THAN OR EQUAL TO the value of interval_left  
                    SET height TO the greater value BETWEEN height AND the sum of side_length PLUS interval_height  
                END IF  
            END FOR  
            APPEND the triple of left right AND height TO intervals  
            SET current_max_height TO the greater value BETWEEN current_max_height AND height  
            APPEND current_max_height TO max_heights  
        END FOR  
        RETURN max_heights  
    END FUNCTION  
END CLASS
