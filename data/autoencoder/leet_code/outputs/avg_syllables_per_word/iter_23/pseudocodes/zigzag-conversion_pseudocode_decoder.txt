CLASS Solution  
    FUNCTION convert(s parameter s TO STRING, numRows parameter numRows TO INTEGER) RETURNS STRING  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list containing numRows elements each initialized to an empty string  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            SET element at position current_row of rows TO element at position current_row of rows PLUS character  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO the negation of going_down  
            END IF  
            IF going_down EQUALS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        RETURN the concatenation of all elements in rows with no separator  
    END FUNCTION  
END CLASS
