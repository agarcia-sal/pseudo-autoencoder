CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position index of nums NOT EQUALS the element at position the element at position index of nums MINUS one of nums  
                SET temporary_value TO the element at position index of nums  
                SET element at position index of nums TO the element at position temporary_value MINUS one of nums  
                SET element at position temporary_value MINUS one of nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each position index AND value element IN nums WITH positions from zero onward  
            IF element NOT EQUALS index PLUS one  
                APPEND element TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
