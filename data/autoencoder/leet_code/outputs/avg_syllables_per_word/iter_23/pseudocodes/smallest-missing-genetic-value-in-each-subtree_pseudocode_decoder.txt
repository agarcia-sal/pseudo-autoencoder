CLASS Solution  
    FUNCTION smallestMissingValueSubtree(parents AS List OF Integer, nums AS List OF Integer) RETURNS List OF Integer  
        SET n TO the LENGTH OF parents  
        SET tree TO a new mapping from Integer to List OF Integer with default empty list  
        
        FOR i FROM one TO n MINUS one  
            APPEND the element at position i of list parents TO the list at key the element at position i of list parents in tree  
        END FOR  
        
        SET node_with_one TO negative one  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of list nums EQUALS one THEN  
                SET node_with_one TO i  
                BREAK  
            END IF  
        END FOR  
        
        IF node_with_one EQUALS negative one THEN  
            RETURN a list consisting of one repeated n times  
        END IF  
        
        SET result TO a list consisting of one repeated n times  
        
        FUNCTION dfs(node) RETURNS a set of Integer  
            ADD node TO the set visited  
            SET genetic_values TO a new set containing the element at position node of list nums  
            FOR each child IN the list at key node in tree  
                IF child is NOT IN the set visited THEN  
                    SET genetic_values TO the union of genetic_values AND the result of dfs with argument child  
                END IF  
            END FOR  
            SET smallest_missing TO one  
            WHILE smallest_missing IN genetic_values  
                INCREMENT smallest_missing BY one  
            END WHILE  
            SET the element at position node of list result TO smallest_missing  
            RETURN genetic_values  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET path TO an empty list  
        SET current TO node_with_one  
        WHILE current NOT EQUALS negative one  
            APPEND current TO path  
            SET current TO the element at position current of list parents  
        END WHILE  
        
        SET path_set TO the set containing all elements of path  
        FOR each node IN the reverse of path  
            IF node IN visited THEN  
                CONTINUE  
            END IF  
            CALL dfs with argument node  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
