CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character IN expression  
            IF character EQUALS the letter t OR character EQUALS the letter f OR character EQUALS the symbol exclamation mark OR character EQUALS the symbol ampersand OR character EQUALS the symbol vertical bar  
                APPEND character TO stk  
            ELSE IF character EQUALS the symbol closing parenthesis  
                SET t TO zero  
                SET f TO zero  
                WHILE the last element of stk EQUALS the letter t OR the last element of stk EQUALS the letter f  
                    INCREMENT t BY one IF the last element of stk EQUALS the letter t  
                    INCREMENT f BY one IF the last element of stk EQUALS the letter f  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET op TO the last element FROM stk  
                REMOVE the last element FROM stk  
                IF op EQUALS the symbol exclamation mark  
                    SET character TO the letter t IF f GREATER THAN zero ELSE the letter f  
                ELSE IF op EQUALS the symbol ampersand  
                    SET character TO the letter f IF f GREATER THAN zero ELSE the letter t  
                ELSE IF op EQUALS the symbol vertical bar  
                    SET character TO the letter t IF t GREATER THAN zero ELSE the letter f  
                END IF  
                APPEND character TO stk  
            END IF  
        END FOR  
        RETURN whether the first element of stk EQUALS the letter t  
    END FUNCTION  
END CLASS
