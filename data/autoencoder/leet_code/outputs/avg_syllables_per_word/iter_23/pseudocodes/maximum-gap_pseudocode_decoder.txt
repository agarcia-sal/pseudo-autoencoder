CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN zero  
        END IF  

        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros of LENGTH n  
            SET count TO a list of zeros of LENGTH ten  

            FOR index_iterator FROM zero TO n MINUS one  
                SET index TO the INTEGER DIVISION OF element at position index_iterator of arr DIVIDED BY exp  
                INCREMENT element at position the REMAINDER OF index DIVIDED BY ten of count BY one  
            END FOR  

            FOR index_iterator FROM one TO nine  
                SET element at position index_iterator of count TO element at position index_iterator of count PLUS element at position index_iterator MINUS one of count  
            END FOR  

            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET index TO the INTEGER DIVISION OF element at position i of arr DIVIDED BY exp  
                SET element at position element at position the REMAINDER OF index DIVIDED BY ten of count MINUS one of output TO element at position i of arr  
                DECREMENT element at position the REMAINDER OF index DIVIDED BY ten of count BY one  
                DECREMENT i BY one  
            END WHILE  

            FOR index_iterator FROM zero TO n MINUS one  
                SET element at position index_iterator of arr TO element at position index_iterator of output  
            END FOR  
        END FUNCTION  

        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element of arr  
            SET exp TO one  
            WHILE the INTEGER DIVISION OF max1 BY exp GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  

        CALL radixSort WITH nums  

        SET max_diff TO zero  
        FOR index_iterator FROM one TO the LENGTH OF nums MINUS one  
            SET current_diff TO element at position index_iterator of nums MINUS element at position index_iterator MINUS one of nums  
            IF current_diff GREATER THAN max_diff THEN  
                SET max_diff TO current_diff  
            END IF  
        END FOR  

        RETURN max_diff  
    END FUNCTION  
END CLASS
