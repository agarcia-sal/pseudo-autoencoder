CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the substring of frag FROM position one TO position d  
                SET right TO the substring of frag FROM position d PLUS one TO the end of frag  
                IF the first element of left NOT EQUALS the character zero OR left EQUALS the character zero AND the last element of right NOT EQUALS the character zero  
                    YIELD the concatenation of left PLUS if d NOT EQUALS n THEN the single character decimal point ELSE the empty string PLUS right  
                END IF  
            END FOR  
        END FUNCTION  
        SET s TO the substring of s FROM position two TO the penultimate position of s  
        RETURN the list constructed by FOR i FROM one TO the LENGTH OF s MINUS one  
            FOR each a and b IN the cartesian product OF make OF the substring of s FROM position one TO position i AND make OF the substring of s FROM position i PLUS one TO the end of s  
                FORM the string OF an opening parenthesis PLUS a PLUS a comma AND a space PLUS b PLUS a closing parenthesis  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
