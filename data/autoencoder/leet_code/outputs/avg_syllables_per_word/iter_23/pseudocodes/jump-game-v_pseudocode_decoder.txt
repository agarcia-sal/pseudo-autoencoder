CLASS Solution  
    FUNCTION maxJumps(arr parameter representing list of integers d parameter representing integer)  
        FUNCTION dp(i parameter representing integer)  
            SET max_jumps TO one  
            FOR x parameter representing integer FROM one TO d amount inclusive  
                IF i PLUS x LESS THAN the LENGTH OF arr THEN  
                    IF element at position i PLUS x of arr GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK out of loop  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS result of calling dp with parameter i PLUS x  
                END IF  
            END FOR  
            FOR x parameter representing integer FROM one TO d amount inclusive  
                IF i MINUS x GREATER THAN OR EQUAL TO zero THEN  
                    IF element at position i MINUS x of arr GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK out of loop  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS result of calling dp with parameter i MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the GREATER OF result of calling dp with each i parameter representing integer FROM zero TO the LENGTH OF arr MINUS one inclusive  
    END FUNCTION  
END CLASS
