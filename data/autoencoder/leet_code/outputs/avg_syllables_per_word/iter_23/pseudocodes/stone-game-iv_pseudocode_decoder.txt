CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of length n PLUS one filled with False  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF dp at position i MINUS j MULTIPLIED BY j EQUALS False  
                    SET dp at position i TO True  
                    BREAK out of the loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN dp at position n  
    END FUNCTION  
END CLASS
