CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the values collection EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode object INITIALIZED WITH the element at position zero of values  
    SET pointer p TO head  
    FOR each element val IN the elements of values from position one TO the end  
        SET node TO a new ListNode object INITIALIZED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the condition that the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH PARAMETERS self AND head  
        SET dummy TO a new ListNode object INITIALIZED WITH value zero  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        WHILE the next attribute of current IS NOT none AND the next attribute of the next attribute of current IS NOT none  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
