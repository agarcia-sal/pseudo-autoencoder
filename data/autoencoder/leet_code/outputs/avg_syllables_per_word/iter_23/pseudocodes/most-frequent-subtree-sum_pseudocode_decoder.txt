CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values AS val and no left and no right  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode with the element at position i of values AS val and no left and no right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode with the element at position i of values AS val and no left and no right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS no value  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS the result of subtree_sum with argument node.left PLUS the result of subtree_sum with argument node.right  
            INCREMENT the value associated with current_sum in count BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO a new default dictionary with zero as default integer value  
        CALL subtree_sum with argument root  
        
        SET max_freq TO the maximum value among all values in count with zero as default if count is empty  
        
        SET result TO an empty list  
        FOR each key and freq IN each key value pair in count  
            IF freq EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
