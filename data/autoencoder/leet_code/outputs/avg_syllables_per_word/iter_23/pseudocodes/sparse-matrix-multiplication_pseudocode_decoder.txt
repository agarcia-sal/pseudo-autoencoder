CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position one of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position one of mat2  
        INITIALIZE result AS a LIST of m ELEMENTS each being a LIST of n ELEMENTS each SET TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                FOR p FROM one TO k  
                    INCREMENT the element at position j of the element at position i of result BY the element at position p of the element at position i of mat1 MULTIPLIED BY the element at position j of the element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
