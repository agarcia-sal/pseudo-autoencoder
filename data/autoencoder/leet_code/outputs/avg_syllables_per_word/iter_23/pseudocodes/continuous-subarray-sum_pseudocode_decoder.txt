CLASS Solution  
    FUNCTION checkSubarraySum(nums k) RETURNS boolean  
        SET remainder_map TO dictionary WITH key zero MAPPING TO value minus one  
        SET current_sum TO zero  
        FOR each position i AND element num IN nums  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF remainder_map CONTAINS key current_sum  
                IF i MINUS remainder_map AT key current_sum GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                SET remainder_map AT key current_sum TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
