CLASS Solution  
    FUNCTION reverse WITH parameter x  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET reversed_x TO integer conversion OF the string conversion OF the absolute value OF x reversed in order of digits  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO THE POWER OF thirty one OR result GREATER THAN two RAISED TO THE POWER OF thirty one MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
