CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as the element at position one of values and left as None and right as None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode with value as the element at position i of values and left as None and right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode with value as the element at position i of values and left as None and right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left property of p, the left property of q) AND is_same_tree(the right property of p, the right property of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS None THEN  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO helper(the left property of node)  
            SET right_max, right_gain TO helper(the right property of node)  

            SET current_max TO the value of node PLUS the GREATER of left_gain and zero PLUS the GREATER of right_gain and zero  

            SET current_gain TO the value of node PLUS the GREATER among left_gain, right_gain, and zero  

            SET global_max TO the GREATER among left_max, right_max, and current_max  

            RETURN global_max, current_gain  
        END FUNCTION  

        RETURN the first element of the tuple returned by helper with argument root  
    END FUNCTION  
END CLASS
