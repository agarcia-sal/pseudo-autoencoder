CLASS Solution  
    FUNCTION myPow(x float n int) RETURNS float  
        FUNCTION qpow(a float n int) RETURNS float  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF n BITWISE AND one NOT EQUALS zero  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two DISCARDING REMAINDER  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH PARAMETERS x AND n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH PARAMETERS x AND n MINUS n MULTIPLIED BY two  
        END IF  
    END FUNCTION  
END CLASS
