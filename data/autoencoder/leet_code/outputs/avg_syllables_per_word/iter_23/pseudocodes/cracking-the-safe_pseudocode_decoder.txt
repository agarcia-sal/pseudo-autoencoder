CLASS Solution  
    FUNCTION crackSafe WITH PARAMETERS n k  
        SET visited TO an empty set  
        SET sequence TO an empty list  
        
        FUNCTION dfs WITH PARAMETER node  
            FOR each index i FROM zero TO k MINUS one  
                SET edge TO the concatenation of node AND the string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH the substring of edge FROM position one TO the end  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO a string consisting of the character zero repeated n MINUS one times  
        CALL dfs WITH start_node  
        SET result TO the concatenation of all elements in sequence in reverse order FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
