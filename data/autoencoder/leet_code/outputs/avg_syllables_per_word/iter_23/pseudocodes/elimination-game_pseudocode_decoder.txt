CLASS Solution  
    FUNCTION lastRemaining(n)  
        SET remaining TO n  
        SET step TO one  
        SET left_to_right TO true  
        SET head TO one  
        WHILE remaining GREATER THAN one  
            IF left_to_right EQUALS true OR remaining MODULO two EQUALS one  
                INCREMENT head BY step  
            END IF  
            SET step TO step MULTIPLIED BY two  
            SET remaining TO remaining DIVIDED BY two  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN head  
    END FUNCTION  
END CLASS
