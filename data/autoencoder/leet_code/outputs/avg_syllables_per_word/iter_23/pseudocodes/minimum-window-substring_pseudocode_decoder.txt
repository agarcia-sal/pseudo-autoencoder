CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s t  
        IMPORT collections Counter FROM collections  
        
        SET t_count TO Counter OF t  
        SET required TO the LENGTH OF t_count  
        
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO Counter OF no elements  
        
        SET min_length TO positive infinity  
        SET min_start TO zero  
        
        WHILE right IS LESS THAN the LENGTH OF s  
            SET current_character TO element at position right of s  
            INCREMENT the count of current_character IN window_counts BY one  
            
            IF current_character EXISTS IN t_count AND the count of current_character IN window_counts EQUALS the count of current_character IN t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET character_at_left TO element at position left of s  
                
                IF right MINUS left PLUS one IS LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count of character_at_left IN window_counts BY one  
                
                IF character_at_left EXISTS IN t_count AND the count of character_at_left IN window_counts IS LESS THAN the count of character_at_left IN t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length IS EQUAL TO positive infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s FROM position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
