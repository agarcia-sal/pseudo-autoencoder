CLASS Solution  
    FUNCTION multiply(num1 str num2 str) RETURNS str  
        IF num1 EQUALS zero OR num2 EQUALS zero  
            RETURN zero  
        END IF  
        
        SET result TO list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
        
        SET num1 TO the REVERSE OF num1  
        SET num2 TO the REVERSE OF num2  
        
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the NUMERIC VALUE OF the ELEMENT at POSITION i of num1 MINUS the NUMERIC VALUE OF zero MULTIPLIED BY the NUMERIC VALUE OF the ELEMENT at POSITION j of num2 MINUS the NUMERIC VALUE OF zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum TO mul PLUS the ELEMENT at POSITION p1 of result  
                
                SET the ELEMENT at POSITION p1 of result TO sum MODULO ten  
                INCREMENT the ELEMENT at POSITION p2 of result BY sum DIVIDED BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the CONCATENATION OF the CONVERSION TO STRING OF each ELEMENT OF the REVERSE OF result  
        RETURN the SUBSTRING of result_str AFTER REMOVING all LEADING zeros  
    END FUNCTION  
END CLASS
