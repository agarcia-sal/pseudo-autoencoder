CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a new dictionary that provides a new empty list for each missing key  
        FOR each string s IN strs  
            SET sorted_list_of_characters TO the characters of s arranged in ascending alphabetical order  
            SET sorted_str TO the concatenation of all characters in sorted_list_of_characters in sequence  
            APPEND the string s TO the list in anagrams corresponding to the key sorted_str  
        END FOR  
        SET result_list TO the list of all values contained in the dictionary anagrams  
        RETURN result_list  
    END FUNCTION  
END CLASS
