CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO an empty mapping from each integer from one TO n TO an empty list  
        FOR each element WITH elements named u, v, weight IN times  
            APPEND a pair consisting of neighbor v AND weight TO the list associated WITH key u IN graph  
        END FOR  
  
        SET dist TO a mapping from each integer from one TO n TO infinity  
        SET the element associated WITH key k IN dist TO zero  
  
        SET min_heap TO a list containing a pair of zero AND k  
  
        WHILE min_heap is not empty  
            SET current_dist AND node TO the result of removing the smallest item FROM min_heap  
  
            IF current_dist GREATER THAN the element associated WITH key node IN dist  
                CONTINUE to the next iteration  
            END IF  
  
            FOR each neighbor AND weight IN the list associated WITH key node IN graph  
                SET distance TO current_dist PLUS weight  
  
                IF distance LESS THAN the element associated WITH key neighbor IN dist  
                    SET the element associated WITH key neighbor IN dist TO distance  
                    ADD the pair of distance AND neighbor TO min_heap maintaining min_heap property  
                END IF  
            END FOR  
        END WHILE  
  
        SET max_time TO the maximum value FROM all values IN dist  
  
        IF max_time EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
