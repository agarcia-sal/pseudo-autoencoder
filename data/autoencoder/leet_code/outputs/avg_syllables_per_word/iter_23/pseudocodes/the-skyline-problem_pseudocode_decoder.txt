CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each building DESCRIBED BY left height AND right IN buildings  
            APPEND a tuple DESCRIBED BY left AND negative height AND right TO events  
            APPEND a tuple DESCRIBED BY right AND zero AND zero TO events  
        END FOR  
        SORT events BY the first element THEN BY the second element THEN BY the third element IN ascending order  
        SET result TO a list CONTAINING a single list OF zero AND zero  
        SET max_heap TO a list CONTAINING a single tuple OF zero AND positive infinity  
        FOR each event DESCRIBED BY x negH AND r IN events  
            WHILE the second element OF the first tuple IN max_heap IS LESS THAN OR EQUAL TO x  
                REMOVE the first tuple FROM max_heap  
            END WHILE  
            IF negH IS NOT EQUALS zero  
                ADD a tuple DESCRIBED BY negH AND r TO max_heap  
            END IF  
            IF the second element OF the last list IN result IS NOT EQUALS the negative of the first element OF the first tuple IN max_heap  
                APPEND a list DESCRIBED BY x AND the negative of the first element OF the first tuple IN max_heap TO result  
            END IF  
        END FOR  
        IF the first list IN result IS EQUALS a list CONTAINING zero AND zero  
            REMOVE the first list FROM result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
