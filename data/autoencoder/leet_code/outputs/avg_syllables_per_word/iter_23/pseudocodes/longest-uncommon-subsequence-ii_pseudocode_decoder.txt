CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO iterator of t  
            RETURN boolean that all characters in s appear in it in order  
        END FUNCTION  
      
        SORT strs by length from largest to smallest AND lexicographically ascending  
      
        FOR each index i AND corresponding word IN strs  
            SET unique TO true  
            FOR each index j AND corresponding other IN strs  
                IF i NOT EQUALS j AND is_subsequence of word IN other holds true  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique holds true  
                RETURN length of word  
            END IF  
        END FOR  
      
        RETURN negative one  
    END FUNCTION  
END CLASS
