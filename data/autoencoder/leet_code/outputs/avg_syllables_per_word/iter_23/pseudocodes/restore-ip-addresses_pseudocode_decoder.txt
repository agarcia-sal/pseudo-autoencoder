CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            IF the LENGTH OF segment EQUALS one OR (the element at position one of segment NOT EQUALS zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty-five)  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the JOINED elements of path separated by a dot TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING OF s FROM position start TO position start PLUS length MINUS one  
                    IF is_valid(segment) EQUALS true  
                        CALL backtrack WITH start PLUS length AND path PLUS a list containing segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH zero AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
