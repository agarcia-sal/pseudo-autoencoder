CLASS Solution  
    FUNCTION canTransform(start parameter named start WITH type string, end parameter named end WITH type string) RETURNS boolean  
        SET start_filtered TO empty list  
        FOR each index named i AND each character named c IN start  
            IF character c NOT EQUALS the letter X  
                APPEND the tuple containing character c AND index i TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO empty list  
        FOR each index named i AND each character named c IN end  
            IF character c NOT EQUALS the letter X  
                APPEND the tuple containing character c AND index i TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair of elements named tuple one AND tuple two FROM start_filtered AND end_filtered  
            SET c1 TO the first element of tuple one  
            SET i1 TO the second element of tuple one  
            SET c2 TO the first element of tuple two  
            SET i2 TO the second element of tuple two  
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            IF c1 EQUALS the letter L AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 EQUALS the letter R AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
