CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL sort ON nums  
        SET n TO the LENGTH OF nums  
        SET first TO element at position one of nums  
        FOR i FROM two TO n  
            SET k TO the INTEGER DIVISION of the DIFFERENCE between element at position i of nums AND first BY two  
            IF k LESS THAN OR EQUAL TO zero OR the SUM of first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE to the next iteration  
            END IF  
            SET count TO a NEW counter constructed from nums  
            SET arr TO an EMPTY list  
            FOR each x IN nums  
                IF the value of count for key x EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the value of count for key the SUM of x PLUS two MULTIPLIED BY k EQUALS zero  
                    BREAK out of the loop  
                END IF  
                APPEND the SUM of x PLUS k TO arr  
                DECREMENT the value of count for key x BY one  
                DECREMENT the value of count for key the SUM of x PLUS two MULTIPLIED BY k BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS the INTEGER DIVISION of n BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an EMPTY list  
    END FUNCTION  
END CLASS
