CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS addition symbol OR token EQUALS subtraction symbol OR token EQUALS multiplication symbol OR token EQUALS division symbol  
                SET b TO last element of stack removed from stack  
                SET a TO last element of stack removed from stack  
                IF token EQUALS addition symbol  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS subtraction symbol  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS multiplication symbol  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS division symbol  
                    APPEND integer of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND integer converted from token TO stack  
            END IF  
        END FOR  
        RETURN first element at position zero of stack  
    END FUNCTION  
END CLASS
