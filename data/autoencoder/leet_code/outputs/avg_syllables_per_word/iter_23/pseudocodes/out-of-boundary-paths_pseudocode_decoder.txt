CLASS Solution  
    FUNCTION findPaths(m n maxMove startRow startColumn)  
        SET MOD TO the number one billion PLUS seven  
        SET directions TO the list of pairs each representing zero and one zero and minus one one and zero minus one and zero  
        SET dp TO a three dimensional list of zero initialized for each number FROM zero TO m MINUS one for each number FROM zero TO n MINUS one for each number FROM zero TO maxMove  
        FOR k FROM one TO maxMove  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    FOR each pair di dj IN directions  
                        SET ni TO i PLUS di  
                        SET nj TO j PLUS dj  
                        IF ni LESS THAN zero OR ni GREATER THAN OR EQUAL TO m OR nj LESS THAN zero OR nj GREATER THAN OR EQUAL TO n  
                            SET dp element at position i then position j then position k TO dp element at position i then position j then position k PLUS one MODULO MOD  
                        ELSE  
                            SET dp element at position i then position j then position k TO dp element at position i then position j then position k PLUS dp element at position ni then position nj then position k MINUS one MODULO MOD  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position startRow then position startColumn then position maxMove  
    END FUNCTION  
END CLASS
