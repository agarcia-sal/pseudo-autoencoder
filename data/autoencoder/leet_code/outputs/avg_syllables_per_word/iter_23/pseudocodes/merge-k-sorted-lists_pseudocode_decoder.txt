CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode initialized with element at position one of values  
    SET p TO head  
    FOR each val IN the slice of values from position two TO the last position  
        SET node TO new ListNode initialized with val  
        SET next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the condition that element val of p1 EQUALS element val of p2 AND the result of is_same_list applied to element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self, lists)  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists using enumeration  
            IF lst NOT EQUALS None THEN  
                PUSH tuple of element val of lst AND i AND lst INTO min_heap maintaining heap order  
            END IF  
        END FOR  
        SET dummy TO new ListNode with default initialization  
        SET current TO dummy  
        WHILE min_heap NOT EQUALS empty list  
            POP the tuple with smallest first element FROM min_heap AND unpack into val, i, node  
            SET next attribute of current TO node  
            SET current TO element next of current  
            IF element next of node NOT EQUALS None THEN  
                PUSH tuple of element val of element next of node AND i AND element next of node INTO min_heap maintaining heap order  
            END IF  
        END WHILE  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
