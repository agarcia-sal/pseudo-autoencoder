CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeroes with length n  
        SET stack TO an empty list  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack is greater than zero AND the element at position of the last element in stack of heights is LESS THAN the element at position i of heights  
                INCREMENT the element at position of the last element popped from stack in answer BY one  
            END WHILE  
            IF the LENGTH OF stack is greater than zero  
                INCREMENT the element at position of the last element in stack in answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
