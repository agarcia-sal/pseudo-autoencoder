CLASS Solution  
    FUNCTION minmaxGasDist(stations parameter AS list of integers, k parameter AS integer) RETURNS floating point number  
        FUNCTION possible(max_dist parameter AS floating point number) RETURNS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the difference between the element at position index of stations MINUS the element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one hundred million  
        WHILE the difference between right MINUS left GREATER THAN one divided by ten TO THE POWER OF six  
            SET mid TO the sum of left PLUS right DIVIDED BY two  
            IF possible(mid) THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
