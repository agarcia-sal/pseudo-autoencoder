CLASS Solution  
    FUNCTION minSkips(dist AS List of integers speed AS integer hoursBefore AS integer) RETURNS integer  
        SET n TO the LENGTH OF dist  
        SET dp TO a two dimensional list of size n BY n PLUS one WHERE each element IS infinite value  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET the first element at position zero of dp at position zero TO the element at position zero of dist  
        FOR each integer j FROM one TO n MINUS one  
            SET the element at position zero of dp at position j TO the element at position zero of dist  
        END FOR  
        
        FOR each integer i FROM one TO n MINUS one  
            FOR each integer j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET the element at position i of dp at position j TO the CEILING OF the element at position i MINUS one of dp at position j DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET the element at position i of dp at position j TO the MINIMUM OF the current value of the element at position i of dp at position j AND the element at position i MINUS one of dp at position j MINUS one PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer j FROM zero TO n MINUS one  
            IF the element at position n MINUS one of dp at position j IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
