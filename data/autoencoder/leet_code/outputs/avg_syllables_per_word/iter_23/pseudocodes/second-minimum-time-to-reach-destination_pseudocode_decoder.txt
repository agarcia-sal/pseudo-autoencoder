CLASS Solution  
    FUNCTION secondMinimum(n, edges, time, change)  
        SET graph TO an empty list of empty lists with length equal to n PLUS one  
        FOR each pair u AND v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        ASSIGN reference TO a deque initialized with a tuple containing the number one the number zero and the number zero  
        SET minTimes TO a list of pairs each pair containing two infinite values with length equal to n PLUS one  
        SET element at position one zero of minTimes TO zero  
        
        WHILE queue is not empty  
            REMOVE the leftmost element FROM queue AND ASSIGN to current currentTime count  
            
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
            
            SET divisionResult TO currentTime DIVIDED BY change  
            SET remainderCheck TO divisionResult MODULO two  
            
            IF remainderCheck EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO divisionResult PLUS one MULTIPLIED BY change PLUS time  
            END IF  
            
            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor zero of minTimes GREATER THAN nextTime  
                    SET element at position neighbor one of minTimes TO element at position neighbor zero of minTimes  
                    SET element at position neighbor zero of minTimes TO nextTime  
                    APPEND a tuple containing neighbor nextTime and zero TO queue  
                ELSE IF element at position neighbor zero of minTimes LESS THAN nextTime AND nextTime LESS THAN element at position neighbor one of minTimes  
                    SET element at position neighbor one of minTimes TO nextTime  
                    APPEND a tuple containing neighbor nextTime and one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
