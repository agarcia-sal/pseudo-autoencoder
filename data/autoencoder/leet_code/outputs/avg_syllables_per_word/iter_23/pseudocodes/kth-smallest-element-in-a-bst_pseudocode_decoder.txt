CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from front of queue  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values is not None  
            SET node.left TO NEW TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values is not None  
            SET node.right TO NEW TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack is not empty OR current is not None  
            WHILE current is not None  
                APPEND current TO stack  
                SET current TO left child of current  
            END WHILE  
            SET current TO element removed from end of stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN value of current  
            END IF  
            SET current TO right child of current  
        END WHILE  
    END FUNCTION  
END CLASS
