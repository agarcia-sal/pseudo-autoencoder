CLASS Solution  
    FUNCTION numWays(steps arrLen)  
        SET MOD TO one billion PLUS seven  
        SET max_index TO the MINIMUM OF steps DIVIDED BY two AND arrLen MINUS one  
        SET dp TO a list of zeros with LENGTH max_index PLUS one  
        SET element at position zero of dp TO one  
        FOR each value FROM one TO steps  
            SET prev_dp TO a COPY of dp  
            FOR each index FROM zero TO max_index  
                SET element at position index of dp TO element at position index of prev_dp  
                IF index GREATER THAN zero THEN  
                    SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of prev_dp MODULO MOD  
                END IF  
                IF index LESS THAN max_index THEN  
                    SET element at position index of dp TO element at position index of dp PLUS element at position index PLUS one of prev_dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
