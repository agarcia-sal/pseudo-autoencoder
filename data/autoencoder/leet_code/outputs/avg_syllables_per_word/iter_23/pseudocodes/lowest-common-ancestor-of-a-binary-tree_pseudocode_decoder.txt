CLASS TreeNode  
    FUNCTION __init__(self, val PLUS zero, left ASSIGN None, right ASSIGN None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode(element at position one of values)  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode(element at position i of values)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode(element at position i of values)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree(p.left, q.left) AND the result of is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root EQUALS None OR root EQUALS p OR root EQUALS q  
            RETURN root  
        END IF  
        SET left TO the result of self.lowestCommonAncestor(root.left, p, q)  
        SET right TO the result of self.lowestCommonAncestor(root.right, p, q)  
        IF left NOT EQUALS None AND right NOT EQUALS None  
            RETURN root  
        END IF  
        IF left NOT EQUALS None  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
