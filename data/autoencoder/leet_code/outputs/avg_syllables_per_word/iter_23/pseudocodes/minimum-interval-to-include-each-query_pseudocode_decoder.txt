CLASS Solution  
    FUNCTION minInterval(intervals LIST OF LIST OF INTEGER queries LIST OF INTEGER) RETURNS LIST OF INTEGER  
        SORT intervals BY the first element of each interval ASCENDING  
        SET sorted_queries TO a list of pairs each consisting of element at position zero of tuple q and element at position one of tuple q FOR each tuple q and its index i IN enumeration of queries  
        SORT sorted_queries BY element at position zero OF each pair ASCENDING  
        SET min_heap TO an empty list  
        SET result TO a list WHERE each element is negative one WITH LENGTH equal to the LENGTH OF queries  
        SET interval_index TO zero  
        FOR each pair consisting of query and original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND element at position zero of element at position interval_index OF intervals LESS THAN OR EQUAL TO query  
                SET start TO element at position zero of element at position interval_index OF intervals  
                SET end TO element at position one of element at position interval_index OF intervals  
                SET interval_size TO end MINUS start PLUS one  
                ADD pair consisting of interval_size and end TO min_heap USING a method that keeps min_heap sorted by first element ASCENDING  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT empty AND element at position one of element at position zero OF min_heap LESS THAN query  
                REMOVE the element with smallest first value FROM min_heap  
            END WHILE  
            IF min_heap IS NOT empty  
                SET element at position original_index OF result TO element at position zero of element at position zero OF min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
