CLASS Solution  
    FUNCTION combinationSum4(nums target)  
        SET dp TO a list of zeros with the LENGTH OF target PLUS one  
        SET the element at position zero of dp TO one  
        FOR amount FROM one TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    SET the element at position amount of dp TO the element at position amount of dp PLUS the element at position amount MINUS num of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position target of dp  
    END FUNCTION  
END CLASS
