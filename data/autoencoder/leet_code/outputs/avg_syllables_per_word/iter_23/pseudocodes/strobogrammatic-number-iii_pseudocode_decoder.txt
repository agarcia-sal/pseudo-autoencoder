CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN a list containing the empty string  
            END IF  
            IF n EQUALS one  
                RETURN a list containing the strings zero, one, and eight  
            END IF  
            SET middles TO the result of calling generate_strobogrammatic WITH n MINUS two AND final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND the string formed by the character zero plus middle plus zero TO result  
                END IF  
                APPEND the string formed by the character one plus middle plus one TO result  
                APPEND the string formed by the character six plus middle plus nine TO result  
                APPEND the string formed by the character eight plus middle plus eight TO result  
                APPEND the string formed by the character nine plus middle plus six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM the length of low TO the length of high INCLUSIVE  
                FOR each num IN the result of calling generate_strobogrammatic WITH length AND length  
                    IF length EQUALS the length of low AND num LESS THAN low OR length EQUALS the length of high AND num GREATER THAN high  
                        CONTINUE to the next iteration of the loop  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN the result of calling count_strobogrammatic WITH low AND high  
    END FUNCTION  
END CLASS
