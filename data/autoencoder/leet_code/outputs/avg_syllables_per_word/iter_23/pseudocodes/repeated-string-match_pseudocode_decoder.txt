CLASS Solution  
    FUNCTION repeatedStringMatch(a string_parameter b string_parameter) RETURNS integer  
        SET minimum_repetitions TO the smallest integer greater than or equal to the length of b DIVIDED BY the length of a  
        
        IF the string b IS a substring of the string a repeated minimum_repetitions times  
            RETURN minimum_repetitions  
        END IF  
        
        IF the string b IS a substring of the string a repeated minimum_repetitions PLUS one times  
            RETURN minimum_repetitions PLUS one  
        END IF  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
