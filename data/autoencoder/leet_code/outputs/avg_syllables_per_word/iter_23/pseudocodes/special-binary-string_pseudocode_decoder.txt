CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each position i AND character char IN the enumeration of string s  
                IF character EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the character one CONCATENATED WITH the result of dfs applied to the substring of s from the position start PLUS one TO the position i MINUS one CONCATENATED WITH the character zero TO the list specials  
                    SET start TO position i PLUS one  
                END IF  
            END FOR  
            SORT the list specials in reverse lexicographical order  
            RETURN the concatenation of all strings in the list specials  
        END FUNCTION  
        RETURN the result of dfs applied to s  
    END FUNCTION  
END CLASS
