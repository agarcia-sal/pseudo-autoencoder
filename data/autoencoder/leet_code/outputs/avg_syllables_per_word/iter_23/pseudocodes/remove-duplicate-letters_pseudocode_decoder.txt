CLASS Solution  
    FUNCTION removeDuplicateLetters WITH PARAMETER s  
        SET last_occurrence TO empty dictionary  
        FOR each index AND character IN the enumeration of s  
            SET the value associated to character in last_occurrence TO index  
        END FOR  
        
        SET stack TO empty list  
        SET in_stack TO empty set  
        
        FOR each index AND character IN the enumeration of s  
            IF character IS IN in_stack THEN  
                CONTINUE to next iteration  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND character IS LESS THAN the element at the last position of stack AND index IS LESS THAN the value associated to the element at the last position of stack in last_occurrence  
                SET removed_char TO the element popped from the last position of stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result_string TO empty string  
        FOR each character IN stack  
            SET result_string TO result_string PLUS character  
        END FOR  
        
        RETURN result_string  
    END FUNCTION  
END CLASS
