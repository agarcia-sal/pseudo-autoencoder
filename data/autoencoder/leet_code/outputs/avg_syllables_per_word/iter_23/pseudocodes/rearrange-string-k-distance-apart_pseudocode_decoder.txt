CLASS Solution  
    FUNCTION rearrangeString(s PARAMETER k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of each character in s  
        
        SET max_heap TO list of pairs where each pair is negative frequency count and corresponding character from freq  
        TRANSFORM max_heap into a heap structure based on the first element of each pair  
        
        SET wait_queue TO empty list  
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            REMOVE and RETURN element with smallest first value from max_heap INTO count AND char  
            APPEND char TO result  
            
            APPEND pair of count PLUS one and char TO wait_queue  
            
            IF length of wait_queue EQUALS k  
                REMOVE first element from wait_queue INTO count AND char  
                IF count LESS THAN zero  
                    ADD pair of count and char TO max_heap maintaining heap property  
                END IF  
            END IF  
        END WHILE  
        
        IF length of result NOT EQUALS length of s  
            RETURN empty string  
        END IF  
        
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
