CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty OR values NOT truthy  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node IS None  
                RETURN minus one  
            END IF  
            SET left_height TO collectLeaves(the left child of node)  
            SET right_height TO collectLeaves(the right child of node)  
            SET current_height TO the maximum of left_height AND right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO the length of result  
                APPEND an empty list TO result  
            END IF  
            APPEND the value of node TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO an empty list  
        CALL collectLeaves with root  
        RETURN result  
    END FUNCTION  
END CLASS
