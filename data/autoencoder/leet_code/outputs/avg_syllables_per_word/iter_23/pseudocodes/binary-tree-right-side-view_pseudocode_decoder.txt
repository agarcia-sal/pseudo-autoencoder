CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with the element at position zero of values AS val  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node dot left TO new TreeNode with the element at position i of values AS val  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node dot right TO new TreeNode with the element at position i of values AS val  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to p dot left AND q dot left AND the result of is_same_tree applied to p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO new deque containing root  
        WHILE queue is not empty  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element removed from the left of queue  
                IF i EQUALS level_length MINUS one  
                    APPEND node dot val TO right_side_view  
                END IF  
                IF node dot left NOT EQUALS None  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right NOT EQUALS None  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
