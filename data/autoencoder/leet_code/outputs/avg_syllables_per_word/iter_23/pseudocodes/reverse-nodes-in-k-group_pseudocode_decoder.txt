CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last position of values  
        SET node TO a new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup with parameters head AND k  
        FUNCTION reverseLinkedList with parameters start AND end  
            SET prev TO None  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO the next property of curr  
                SET the next property of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the next property of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO a new ListNode with zero as value  
        SET the next property of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR each iteration FROM one TO k  
                SET kth_node TO the next property of kth_node  
                IF kth_node EQUALS None  
                    RETURN the next property of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the next property of kth_node  
            SET the next property of kth_node TO None  
            SET new_group_start AND new_group_end TO reverseLinkedList called with the next property of prev_group_end AND kth_node  
            SET the next property of prev_group_end TO new_group_start  
            SET the next property of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
