CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr parameter, k parameter)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty collection for minimum heap storage  
        
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH onto min_heap the triple consisting of the fraction element at position i of arr DIVIDED BY element at position n MINUS one of arr the integer i and the integer n MINUS one  
        END FOR  
        
        FOR loop_variable FROM zero TO k MINUS one MINUS one  
            POP from min_heap the triple to variables discard_variable i j  
            IF j MINUS one GREATER THAN i THEN  
                PUSH onto min_heap the triple consisting of the fraction element at position i of arr DIVIDED BY element at position j MINUS one of arr the integer i and the integer j MINUS one  
            END IF  
        END FOR  
        
        POP from min_heap the triple to variables discard_variable i j  
        RETURN the list containing element at position i of arr and element at position j of arr  
    END FUNCTION  
END CLASS
