CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH arguments t AND s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff GREATER THAN one  
            RETURN false  
        END IF  
        
        IF diff EQUALS zero  
            RETURN the RESULT OF the SUM OVER all pairs OF elements c1 AND c2 FROM s AND t RESPECTIVELY WHERE c1 NOT EQUALS c2 EQUALS one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    RETURN whether the substring of s FROM position index TO THE END EQUALS the substring of t FROM position index PLUS one TO THE END  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
