CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a deque containing a tuple consisting of zero zero zero one zero  
        SET visited TO a set containing a tuple consisting of zero zero zero one  
        
        WHILE the LENGTH OF queue is greater than zero  
            SET r1 comma c1 comma r2 comma c2 comma steps TO the REMOVAL OF the first element OF queue  
            
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND a tuple consisting of r1 comma c1 PLUS one comma r2 comma c2 PLUS one NOT IN visited  
                ADD a tuple consisting of r1 comma c1 PLUS one comma r2 comma c2 PLUS one TO visited  
                APPEND a tuple consisting of r1 comma c1 PLUS one comma r2 comma c2 PLUS one comma steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r2 PLUS one of grid at position c2 EQUALS zero AND a tuple consisting of r1 PLUS one comma c1 comma r2 PLUS one comma c2 NOT IN visited  
                ADD a tuple consisting of r1 PLUS one comma c1 comma r2 PLUS one comma c2 TO visited  
                APPEND a tuple consisting of r1 PLUS one comma c1 comma r2 PLUS one comma c2 comma steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r1 PLUS one of grid at position c2 EQUALS zero AND a tuple consisting of r1 comma c1 comma r1 PLUS one comma c1 NOT IN visited  
                ADD a tuple consisting of r1 comma c1 comma r1 PLUS one comma c1 TO visited  
                APPEND a tuple consisting of r1 comma c1 comma r1 PLUS one comma c1 comma steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND a tuple consisting of r1 comma c1 comma r1 comma c1 PLUS one NOT IN visited  
                ADD a tuple consisting of r1 comma c1 comma r1 comma c1 PLUS one TO visited  
                APPEND a tuple consisting of r1 comma c1 comma r1 comma c1 PLUS one comma steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
