CLASS Solution  
    FUNCTION findMaxForm(strs m n)  
        SET dp TO two dimensional list with the number of rows as m PLUS one and the number of columns as n PLUS one where each element is zero  
        FOR each string s IN strs  
            SET count_0 TO the number of elements equal to zero in string s  
            SET count_1 TO the number of elements equal to one in string s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    IF element at position i of dp AND element at position j OF that element LESS THAN element at position i MINUS count_0 of dp AND element at position j MINUS count_1 OF that element PLUS one  
                        SET element at position i of dp AND element at position j of that element TO element at position i MINUS count_0 of dp AND element at position j MINUS count_1 of that element PLUS one  
                    ELSE  
                        KEEP element at position i of dp AND element at position j of that element unchanged  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position m of dp AND element at position n of that element  
    END FUNCTION  
END CLASS
