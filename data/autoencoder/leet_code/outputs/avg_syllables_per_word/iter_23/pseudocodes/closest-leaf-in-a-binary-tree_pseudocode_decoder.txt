CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode CREATED WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the LEFT of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH parameters p.left AND q.left AND the result of is_same_tree WITH parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left EQUALS None AND root.right EQUALS None  
            RETURN root.val  
        END IF  
        SET graph TO an empty map FROM values TO lists  
        FUNCTION dfs(node, parent)  
            IF parent IS NOT None  
                APPEND parent.val TO the list corresponding to node.val in graph  
                APPEND node.val TO the list corresponding to parent.val in graph  
            END IF  
            IF node.left IS NOT None  
                CALL dfs WITH parameters node.left AND node  
            END IF  
            IF node.right IS NOT None  
                CALL dfs WITH parameters node.right AND node  
            END IF  
        END FUNCTION  
        CALL dfs WITH parameters root AND None  
        SET queue TO a new double ended queue INITIALIZED WITH element k  
        SET visited TO an empty set  
        WHILE queue IS NOT empty  
            SET current TO the element REMOVED FROM the LEFT of queue  
            ADD current TO visited  
            IF the LENGTH OF the list corresponding to current in graph EQUALS one AND current NOT EQUALS root.val  
                RETURN current  
            END IF  
            FOR each neighbor IN the list corresponding to current in graph  
                IF neighbor NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
