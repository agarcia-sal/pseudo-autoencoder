CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode of element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended collection  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NO VALUE  
            SET node.left TO a new TreeNode of element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NO VALUE  
            SET node.right TO a new TreeNode of element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS NO VALUE AND q IS NO VALUE  
        RETURN True  
    ELSE IF p IS NO VALUE OR q IS NO VALUE  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree on p.left AND q.left AND is_same_tree on p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF the LENGTH OF inorder EQUALS zero OR the LENGTH OF postorder EQUALS zero  
            RETURN no value  
        END IF  
        SET root_val TO the element removed from the end of postorder  
        SET root TO a new TreeNode of root_val  
        SET root_index TO the position of root_val in inorder  
        SET root.right TO the result of calling buildTree with the sublist of inorder from the position root_index PLUS one TO the end AND postorder  
        SET root.left TO the result of calling buildTree with the sublist of inorder from the start TO the position root_index MINUS one AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
