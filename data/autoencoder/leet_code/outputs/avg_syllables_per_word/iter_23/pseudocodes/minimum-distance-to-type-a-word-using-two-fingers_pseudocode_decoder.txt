CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1 c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the integer division of the numerical value of c1 MINUS the numerical value of the capital letter A DIVIDED BY six  
            SET y1 TO the remainder of the numerical value of c1 MINUS the numerical value of the capital letter A DIVIDED BY six  
            SET x2 TO the integer division of the numerical value of c2 MINUS the numerical value of the capital letter A DIVIDED BY six  
            SET y2 TO the remainder of the numerical value of c2 MINUS the numerical value of the capital letter A DIVIDED BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  

        FUNCTION dp(i f1 f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO distance of f1 AND c PLUS dp of i PLUS one AND c AND f2  
            SET d2 TO distance of f2 AND c PLUS dp of i PLUS one AND f1 AND c  
            RETURN the minimum of d1 AND d2  
        END FUNCTION  

        RETURN dp of zero AND None AND None  
    END FUNCTION  
END CLASS
