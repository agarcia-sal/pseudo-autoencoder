CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the first element of days  
        
        SET dp TO a list of size n FILLED WITH negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO an empty mapping from integer TO list  
        FOR each integer i FROM zero TO n MINUS one  
            SET the element at key i of graph TO an empty list  
        END FOR  
        
        FOR each integer i FROM zero TO n MINUS one  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of element at position i of flights EQUALS one THEN  
                    APPEND j TO the element at key i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer week FROM zero TO k MINUS one  
            SET new_dp TO a list of size n FILLED WITH negative one  
            FOR each integer i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one THEN  
                    CONTINUE to the next iteration  
                END IF  
                SET the element at position i of new_dp TO the MAXIMUM OF the element at position i of new_dp AND the element at position i of dp PLUS the element at position week of element at position i of days  
                FOR each city j IN the element at key i of graph  
                    SET the element at position j of new_dp TO the MAXIMUM OF the element at position j of new_dp AND the element at position i of dp PLUS the element at position week of element at position j of days  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MAXIMUM VALUE of dp  
    END FUNCTION  
END CLASS
