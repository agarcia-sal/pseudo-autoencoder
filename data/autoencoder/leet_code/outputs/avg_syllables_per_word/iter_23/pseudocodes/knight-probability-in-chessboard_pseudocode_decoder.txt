CLASS Solution  
    FUNCTION knightProbability(n k row column)  
        SET moves TO list of pairs representing two less and one less two more and one more one less and two less one less and two more one more and two less one more and two more two less and one less two less and one more  
        SET dp TO two dimensional list of zeros with dimensions n and n  
        SET element at position row and position column of dp TO one  
        FOR each step FROM one TO k  
            SET new_dp TO two dimensional list of zeros with dimensions n and n  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r and position c of dp EQUALS zero THEN  
                        CONTINUE to next iteration  
                    END IF  
                    FOR each pair dr dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n THEN  
                            SET element at position nr and position nc of new_dp TO element at position nr and position nc of new_dp PLUS element at position r and position c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each value IN row_list  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
