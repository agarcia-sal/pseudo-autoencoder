CLASS Solution  
    FUNCTION expressiveWords(s, words)  
        FUNCTION compress(word)  
            IF word EQUALS empty  
                RETURN empty list  
            END IF  
            SET count TO one  
            SET compressed TO empty list  
            FOR index FROM one TO the LENGTH OF word MINUS one  
                IF element at position index of word EQUALS element at position index MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND pair of element at position index MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND pair of element at the last position of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy(s_compressed, word_compressed)  
            IF the LENGTH OF s_compressed NOT EQUALS the LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair of s_character AND s_count IN s_compressed AND each pair of w_character AND w_count IN word_compressed CORRESPONDINGLY  
                IF s_character NOT EQUALS w_character  
                    RETURN false  
                END IF  
                IF s_count NOT EQUALS w_count AND (s_count LESS THAN three OR s_count LESS THAN w_count)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET s_compressed TO compress applied TO s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO compress applied TO word  
            IF is_stretchy applied TO s_compressed AND word_compressed  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
