CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the result of two MULTIPLIED BY n power MINUS one  
            SET good TO a list of n elements each set to false  
            SET bad TO a list of n elements each set to false  
            FOR j FROM zero TO n MINUS one  
                IF the result of i BITWISE AND the result of two MULTIPLIED BY j power NOT EQUALS zero  
                    SET element at position j of good TO true  
                ELSE  
                    SET element at position j of bad TO true  
                END IF  
            END FOR  
            SET valid TO true  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of good EQUALS true  
                    FOR k FROM zero TO n MINUS one  
                        IF element at position k of statements at position j EQUALS zero AND element at position k of good EQUALS true OR element at position k of statements at position j EQUALS one AND element at position k of bad EQUALS true  
                            SET valid TO false  
                            BREAK the inner loop  
                        END IF  
                    END FOR  
                    IF valid EQUALS false  
                        BREAK the outer loop  
                    END IF  
                END IF  
            END FOR  
            IF valid EQUALS true  
                SET max_good TO the GREATER THAN value BETWEEN max_good AND the SUM OF elements in good WHERE element EQUALS true  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
