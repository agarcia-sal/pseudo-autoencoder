CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of elements FROM zero TO size MINUS one  
        SET self.rank TO list of zeros OF LENGTH size  
        SET self.size TO list of ones OF LENGTH size  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO the result of calling find WITH argument element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find WITH argument u  
        SET rootV TO the result of calling find WITH argument v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
                INCREMENT element at position rootV of self.size BY element at position rootU of self.size  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one SHIFTED LEFT BY the numeric code of character MINUS the numeric code of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a default dictionary with list as default type  
        SET n TO the LENGTH OF words  
        SET uf TO a new instance of UnionFind WITH argument n  

        FOR each index i AND word IN enumeration of words  
            SET mask TO bitmask WITH argument word  
            APPEND i TO element at key mask of word_map  
        END FOR  

        FOR each index i AND word IN enumeration of words  
            SET mask TO bitmask WITH argument word  
            FOR each integer j FROM zero TO twenty five  
                SET new_mask TO mask XOR one SHIFTED LEFT BY j  
                IF new_mask IS A KEY IN word_map  
                    FOR each element k IN element at key new_mask of word_map  
                        CALL union ON uf WITH arguments i AND k  
                    END FOR  
                END IF  

                IF mask AND one SHIFTED LEFT BY j EQUALS zero  
                    SET new_mask TO mask OR one SHIFTED LEFT BY j  
                    IF new_mask IS A KEY IN word_map  
                        FOR each element k IN element at key new_mask of word_map  
                            CALL union ON uf WITH arguments i AND k  
                        END FOR  
                    END IF  
                END IF  

                IF mask AND one SHIFTED LEFT BY j NOT EQUALS zero  
                    FOR each integer k FROM zero TO twenty five  
                        IF mask AND one SHIFTED LEFT BY k EQUALS zero  
                            SET new_mask TO mask XOR one SHIFTED LEFT BY j OR one SHIFTED LEFT BY k  
                            IF new_mask IS A KEY IN word_map  
                                FOR each element l IN element at key new_mask of word_map  
                                    CALL union ON uf WITH arguments i AND l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a default dictionary with integer zero as default value  
        FOR each integer i FROM zero TO n MINUS one  
            SET root TO uf.find WITH argument i  
            INCREMENT element at key root of root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value among values of root_sizes  
        SET num_groups TO the number of keys in root_sizes  

        RETURN a list containing num_groups AND max_group_size  
    END FUNCTION  
END CLASS
