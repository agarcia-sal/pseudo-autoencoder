CLASS Solution  
    FUNCTION advantageCount(nums1 LIST OF INTEGER nums2 LIST OF INTEGER) RETURNS LIST OF INTEGER  
        SORT nums1 IN NON-DECREASING ORDER  
        SET nums2_with_indices TO A LIST OF PAIRS WHERE EACH PAIR CONSISTS OF element AT position value OF nums2 AND CORRESPONDING position index OF nums2 SORTED IN DECREASING ORDER BY THE FIRST ELEMENT OF EACH PAIR  
        SET result TO A LIST OF ZEROES WITH THE LENGTH EQUAL TO THE LENGTH OF nums1  
        SET left TO zero  
        SET right TO THE LENGTH OF nums1 MINUS one  
        FOR each pair OF value AND index IN nums2_with_indices  
            IF element AT position right OF nums1 GREATER THAN value  
                SET element AT position index OF result TO element AT position right OF nums1  
                DECREMENT right BY one  
            ELSE  
                SET element AT position index OF result TO element AT position left OF nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
