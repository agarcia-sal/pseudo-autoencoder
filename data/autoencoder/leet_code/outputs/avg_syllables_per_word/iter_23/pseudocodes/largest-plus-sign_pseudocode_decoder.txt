CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine_element IN mines  
            ADD the tuple containing elements of mine_element TO mines_set  
        END FOR  
        SET dp TO a two dimensional list with size n by n where each element is a list containing four zeros representing counts in four directions  
        SET directions TO a list containing tuples corresponding to right zero one, down one zero, left zero negative one, up negative one zero directions  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF tuple containing i and j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at position zero of list at position j of list at position i of dp TO element at position zero of list at position j MINUS one of list at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at position one of list at position j of list at position i of dp TO element at position one of list at position j of list at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF tuple containing i and j NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at position two of list at position j of list at position i of dp TO element at position two of list at position j PLUS one of list at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at position three of list at position j of list at position i of dp TO element at position three of list at position j of list at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the four elements in list at position j of list at position i of dp  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
