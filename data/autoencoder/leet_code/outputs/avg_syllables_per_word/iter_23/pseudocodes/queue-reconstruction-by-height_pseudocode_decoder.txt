CLASS Solution  
    FUNCTION reconstructQueue(people)  
        SORT people BY the element at position zero of person MULTIPLIED BY negative one IN DESCENDING ORDER AND BY the element at position one of person IN ASCENDING ORDER  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT position corresponding to the element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
