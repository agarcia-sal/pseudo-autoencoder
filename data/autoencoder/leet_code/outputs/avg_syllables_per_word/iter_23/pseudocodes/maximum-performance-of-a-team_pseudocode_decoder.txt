CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET list_of_engineers TO a new empty list  
        FOR index FROM zero TO n MINUS one  
            SET current_efficiency TO element at position index of efficiency  
            SET current_speed TO element at position index of speed  
            APPEND the pair of current_efficiency and current_speed TO list_of_engineers  
        END FOR  
        SORT list_of_engineers IN DESCENDING ORDER BY the first element of each pair  
        SET speed_min_heap TO a new empty list  
        SET sum_of_speeds TO zero  
        SET maximum_performance TO zero  
        FOR each pair of efficiency_value and speed_value IN list_of_engineers  
            ADD speed_value TO speed_min_heap maintaining heap structure  
            INCREMENT sum_of_speeds BY speed_value  
            IF the LENGTH OF speed_min_heap GREATER THAN k  
                REMOVE the smallest element from speed_min_heap  
                DECREMENT sum_of_speeds BY that smallest element  
            END IF  
            SET current_performance TO sum_of_speeds MULTIPLIED BY efficiency_value  
            IF current_performance GREATER THAN maximum_performance  
                SET maximum_performance TO current_performance  
            END IF  
        END FOR  
        RETURN the remainder of maximum_performance DIVIDED BY one billion PLUS seven  
    END FUNCTION  
END CLASS
