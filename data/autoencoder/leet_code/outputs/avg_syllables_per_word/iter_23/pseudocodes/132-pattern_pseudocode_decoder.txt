CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums in reverse order  
            IF num LESS THAN second  
                RETURN True  
            END IF  
            WHILE the LENGTH OF stack is greater than zero AND element at position LENGTH OF stack MINUS one of stack LESS THAN num  
                SET second TO the element removed from the end of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
