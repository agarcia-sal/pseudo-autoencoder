CLASS Solution  
    FUNCTION pacificAtlantic(heights)  
        FUNCTION bfs(queue_parameter, visited_set)  
            WHILE the LENGTH OF queue_parameter is GREATER THAN zero  
                FOR iteration FROM zero TO the LENGTH OF queue_parameter MINUS one  
                    SET i, j TO the FIRST TWO elements REMOVED FROM the FRONT OF queue_parameter  
                    FOR each pair a AND b IN the collection of pairs where a and b are ZERO AND MINUS ONE, ZERO AND ONE, ONE AND ZERO, MINUS ONE AND ZERO respectively  
                        SET x TO i PLUS a  
                        SET y TO j PLUS b  
                        IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN m AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN n AND the pair x AND y IS NOT IN visited_set AND the element at position x AND y OF heights IS GREATER THAN OR EQUAL TO the element at position i AND j OF heights  
                            ADD the pair x AND y TO visited_set  
                            ADD the pair x AND y TO the END OF queue_parameter  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
        END FUNCTION  

        SET m TO the LENGTH OF heights  
        SET n TO the LENGTH OF the element at position zero OF heights  
        SET vis1 TO an EMPTY SET  
        SET vis2 TO an EMPTY SET  
        SET q1 TO an EMPTY DOUBLE ENDED QUEUE  
        SET q2 TO an EMPTY DOUBLE ENDED QUEUE  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF i EQUALS zero OR j EQUALS zero  
                    ADD the pair i AND j TO vis1  
                    ADD the pair i AND j TO the END OF q1  
                END IF  
                IF i EQUALS m MINUS one OR j EQUALS n MINUS one  
                    ADD the pair i AND j TO vis2  
                    ADD the pair i AND j TO the END OF q2  
                END IF  
            END FOR  
        END FOR  
        CALL bfs WITH PARAMETERS q1 AND vis1  
        CALL bfs WITH PARAMETERS q2 AND vis2  
        SET result_list TO an EMPTY LIST  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair i AND j IS IN vis1 AND the pair i AND j IS IN vis2  
                    APPEND the pair i AND j TO result_list  
                END IF  
            END FOR  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
