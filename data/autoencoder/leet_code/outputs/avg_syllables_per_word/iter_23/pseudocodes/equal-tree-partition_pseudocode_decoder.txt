CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO TreeNode with value equal to element at position one of values and left equal to None and right equal to None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue HAS elements  
        SET node TO the element removed from the left of queue  
        IF i IS less than the length of values AND element at position i of values IS NOT None  
            SET node.left TO TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the length of values AND element at position i of values IS NOT None  
            SET node.right TO TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            RETURN value of node PLUS subtree_sum(left child of node) PLUS subtree_sum(right child of node)  
        END FUNCTION  
        
        SET total_sum TO subtree_sum(root)  
        
        IF total_sum MODULO two IS NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing one element False  
        
        FUNCTION can_split(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO value of node PLUS can_split(left child of node) PLUS can_split(right child of node)  
            IF current_sum EQUALS target AND node IS NOT the same as root  
                SET element at position one of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  
        
        CALL can_split(root)  
        RETURN element at position one of found  
    END FUNCTION  
END CLASS
