CLASS Solution  
    FUNCTION kEmptySlots bulbs k  
        SET bloom_days TO empty list  
        
        FOR each day position IN bulbs enumerated starting from one  
            SET i TO the position for inserting position into bloom_days to keep bloom_days sorted using left bisect  
            
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            
            INSERT position INTO bloom_days AT position i  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
