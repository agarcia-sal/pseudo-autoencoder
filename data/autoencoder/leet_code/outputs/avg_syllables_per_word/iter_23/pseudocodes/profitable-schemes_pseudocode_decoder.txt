CLASS Solution  
    FUNCTION profitableSchemes(n profit MINUS n members minProfit minimum profit group list of integers group profit list of integers profit) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        CREATE two dimensional list dp with dimensions of n PLUS one by minProfit PLUS one and FILL with zeros  
        FOR index i FROM zero TO n INCLUSIVE  
            SET element at position zero of dp at position i TO one  
        END FOR  
        FOR each element p AND element g FROM simultaneous traversal of profit and group  
            FOR index i FROM n DOWNTO g  
                FOR index j FROM minProfit DOWNTO zero  
                    SET temporary profit index TO j MINUS p  
                    IF temporary profit index LESS THAN zero  
                        SET temporary profit index TO zero  
                    END IF  
                    SET element at position j of dp at position i TO the sum of element at position j of dp at position i AND element at position temporary profit index of dp at position i MINUS g modulo MOD  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position minProfit of dp at position n  
    END FUNCTION  
END CLASS
