CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS empty OR element at position zero of board EQUALS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        SET directions TO a list containing the pairs consisting of negative one and negative one negative one and zero negative one and one zero and negative one zero and one one and negative one one and zero one and one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each dx and dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND (element at position ni of board AND element at position nj of that element EQUALS one OR element at position ni of board AND element at position nj of that element EQUALS three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board AND element at position j of that element EQUALS one  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three  
                        SET element at position i of board AND element at position j of that element TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three  
                        SET element at position i of board AND element at position j of that element TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board AND element at position j of that element EQUALS two  
                    SET element at position i of board AND element at position j of that element TO one  
                ELSE IF element at position i of board AND element at position j of that element EQUALS three  
                    SET element at position i of board AND element at position j of that element TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
