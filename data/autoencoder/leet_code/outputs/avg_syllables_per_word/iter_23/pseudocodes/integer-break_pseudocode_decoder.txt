CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO a list of zero repeated the sum of n plus one times  
        SET the element at position one of dp TO one  
        FOR i FROM two TO the value of n  
            SET max_product TO zero  
            FOR j FROM one TO the value of i MINUS one  
                SET candidate_one TO the product of j AND the element at position i MINUS j of dp  
                SET candidate_two TO the product of j AND the difference of i MINUS j  
                SET max_product TO the greatest value among max_product AND candidate_one AND candidate_two  
            END FOR  
            SET the element at position i of dp TO max_product  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
