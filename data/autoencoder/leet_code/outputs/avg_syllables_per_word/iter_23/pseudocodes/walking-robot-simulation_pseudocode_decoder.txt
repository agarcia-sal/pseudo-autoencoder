CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of pairs where first pair is zero PLUS one for north direction second pair is one PLUS zero for east direction third pair is zero MINUS one for south direction fourth pair is minus one PLUS zero for west direction  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO set obtained by converting each element of obstacles to a tuple  
        SET max_distance TO zero  
        FOR each command IN commands  
            IF command EQUALS minus two  
                SET dir_index TO dir_index MINUS one MODULO four  
            ELSE IF command EQUALS minus one  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO first element of the pair at position dir_index of directions  
                SET dy TO second element of the pair at position dir_index of directions  
                FOR each iteration FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF pair of next_x and next_y IS IN obstacles_set  
                        BREAK the loop  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance  
                    SET max_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
