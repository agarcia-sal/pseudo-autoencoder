CLASS Solution  
    FUNCTION maxSum(nums1 nums1 OF TYPE List OF integers nums2 nums2 OF TYPE List OF integers) RETURNS integer  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS one  
        
        WHILE i LESS THAN the LENGTH OF nums1 AND j LESS THAN the LENGTH OF nums2  
            IF element at position i of nums1 LESS THAN element at position j of nums2  
                SET sum1 TO sum1 PLUS element at position i of nums1  
                INCREMENT i BY one  
            ELSE IF element at position i of nums1 GREATER THAN element at position j of nums2  
                SET sum2 TO sum2 PLUS element at position j of nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO the MAXIMUM OF sum1 AND sum2 PLUS element at position i of nums1  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i LESS THAN the LENGTH OF nums1  
            SET sum1 TO sum1 PLUS element at position i of nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j LESS THAN the LENGTH OF nums2  
            SET sum2 TO sum2 PLUS element at position j of nums2  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN the REMAINDER OF the MAXIMUM OF sum1 AND sum2 DIVIDED BY MOD  
    END FUNCTION  
END CLASS
