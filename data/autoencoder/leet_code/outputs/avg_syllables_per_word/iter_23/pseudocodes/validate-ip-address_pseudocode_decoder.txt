CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONTAINS only digits  
                SET num TO the integer value of block  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND the string representation of num EQUALS block  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the length of block GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO an integer assuming base sixteen  
                    RETURN true  
                CATCH a value error occurs  
                    RETURN false  
                END TRY  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the result of splitting queryIP by the period character  
        SET parts_ipv6 TO the result of splitting queryIP by the colon character  

        IF the length of parts_ipv4 EQUALS four AND all parts in parts_ipv4 SATISFY is_valid_ipv4_block  
            RETURN the string indicating Internet Protocol version four  
        ELSE IF the length of parts_ipv6 EQUALS eight AND all parts in parts_ipv6 SATISFY is_valid_ipv6_block  
            RETURN the string indicating Internet Protocol version six  
        ELSE  
            RETURN the string indicating neither Internet Protocol version four nor six  
        END IF  
    END FUNCTION  
END CLASS
