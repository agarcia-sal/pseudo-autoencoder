CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        
        SET lis TO a list of LENGTH n with each element SET TO one  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums GREATER THAN the element at position j of nums  
                    SET lis at position i TO the GREATER OF the current value of lis at position i AND the value of lis at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET lds TO a list of LENGTH n with each element SET TO one  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                IF the element at position i of nums GREATER THAN the element at position j of nums  
                    SET lds at position i TO the GREATER OF the current value of lds at position i AND the value of lds at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET max_mountain_length TO zero  
        FOR i FROM one TO n MINUS two  
            IF the value of lis at position i GREATER THAN one AND the value of lds at position i GREATER THAN one  
                SET max_mountain_length TO the GREATER OF the current value of max_mountain_length AND the value of lis at position i PLUS the value of lds at position i MINUS one  
            END IF  
        END FOR  
        
        RETURN the value of n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
