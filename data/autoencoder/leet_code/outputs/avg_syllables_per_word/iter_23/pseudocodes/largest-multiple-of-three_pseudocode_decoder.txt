CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency counter of digits  
        SET total_sum TO the sum of digits  
        SET remainder TO the total_sum MODULO three  
        IF remainder EQUALS one  
            FOR each element d IN the list containing one four seven  
                IF the count at key d IS GREATER THAN zero  
                    DECREMENT the count at key d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred in previous loop  
                FOR each element d IN the list containing two five eight  
                    IF the count at key d IS GREATER THAN one  
                        DECREMENT the count at key d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        IF remainder EQUALS two  
            FOR each element d IN the list containing two five eight  
                IF the count at key d IS GREATER THAN zero  
                    DECREMENT the count at key d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred in previous loop  
                FOR each element d IN the list containing one four seven  
                    IF the count at key d IS GREATER THAN one  
                        DECREMENT the count at key d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        SET result TO an empty list  
        FOR d FROM nine TO zero DECREMENT BY one  
            APPEND the string representation of d REPEATED count at key d TIMES TO result  
        END FOR  
        SET final_number TO the concatenation of all elements in result  
        IF the final_number IS EMPTY OR the character at position one of final_number EQUALS zero  
            RETURN zero as a string  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
