CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of zeroes of length n  
        SET stack TO an empty list  
        FOR each log IN logs  
            SET function_id, action, timestamp TO the parts of log separated by the word colon  
            SET function_id TO the integer value of function_id  
            SET timestamp TO the integer value of timestamp  
            IF action EQUALS the word start  
                IF the LENGTH OF stack is greater than zero  
                    SET prev_function_id, prev_timestamp TO the last element of stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND the tuple of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id, prev_timestamp TO the result of removing and returning the last element from stack  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the LENGTH OF stack is greater than zero  
                    SET the last element of stack TO the tuple of its first element and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
