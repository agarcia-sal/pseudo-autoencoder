CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty collection  
            RETURN zero  
        END IF  
        SORT intervals BY the end element of each sub-collection  
        SET end TO the end element of the first sub-collection in intervals  
        SET count TO zero  
        FOR index FROM two TO the LENGTH OF intervals  
            IF the start element of the sub-collection at position index of intervals GREATER THAN OR EQUAL TO end  
                SET end TO the end element of the sub-collection at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
