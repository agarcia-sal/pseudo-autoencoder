CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO a new mapping from keys to empty lists  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position i of nums TO the list in groups assigned to key the remainder WHEN i IS DIVIDED BY k  
        END FOR  
        SET freq TO a list containing k new mappings from keys to zero  
        SET min_size TO a list containing k occurrences of positive infinity  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN the list assigned to key i in groups  
                INCREMENT the value in freq at index i for key num BY one  
            END FOR  
            SET element at position i of min_size TO the SIZE OF the list assigned to key i in groups  
        END FOR  
        SET dp TO a list containing positive infinity repeated two TO THE POWER OF ten TIMES  
        SET element at position zero of dp TO zero  
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO a list containing positive infinity repeated two TO THE POWER OF ten TIMES  
            SET min_dp TO the MINIMUM value contained in dp  
            FOR j FROM zero TO two TO THE POWER OF ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each key num AND value count IN the mapping freq at index i  
                    SET candidate TO element at position (the BITWISE EXCLUSIVE OR of j AND num) of dp PLUS element at position i of min_size MINUS count  
                    IF candidate IS LESS THAN element at position j of new_dp THEN  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
