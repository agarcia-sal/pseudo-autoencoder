CLASS Solution  
    FUNCTION isNStraightHand(hand WITH TYPE List of integers groupSize WITH TYPE integer) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        SET count TO a mapping of elements to their frequency in hand  
        SET unique_cards TO the list of unique keys in count arranged in ascending order  
        FOR each card IN unique_cards  
            IF the value mapped by card in count GREATER THAN zero THEN  
                SET needed TO the value mapped by card in count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the value mapped by the sum of card PLUS index in count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    DECREMENT the value mapped by the sum of card PLUS index in count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
