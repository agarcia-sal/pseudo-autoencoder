CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO the iterator over s  
            FOR each character IN word  
                IF character is NOT contained in the remainder of iterator  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT the dictionary so that the words with longer length come before words with shorter length AND for words of equal length, words appear in ascending alphabetical order  
        
        FOR each word IN dictionary  
            IF the call to is_subsequence with parameters word AND s RETURNS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN the empty string  
    END FUNCTION  
END CLASS
