CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO infinity value representing positive infinity  
        SET Y1 TO infinity value representing positive infinity  
        SET X2 TO infinity value representing negative infinity  
        SET Y2 TO infinity value representing negative infinity  
        SET points TO an empty set  
        SET area TO zero  
        FOR each rectangle described by values x1 y1 x2 y2 IN rectangles  
            IF X1 GREATER THAN x1 THEN  
                SET X1 TO x1  
            END IF  
            IF Y1 GREATER THAN y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF X2 LESS THAN x2 THEN  
                SET X2 TO x2  
            END IF  
            IF Y2 LESS THAN y2 THEN  
                SET Y2 TO y2  
            END IF  
            SET rectangle_area TO (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)  
            INCREMENT area BY rectangle_area  
            FOR each point p IN the list of points consisting of point formed by x1 and y1 point formed by x1 and y2 point formed by x2 and y1 point formed by x2 and y2  
                IF point p IS IN points THEN  
                    REMOVE point p FROM points  
                ELSE  
                    ADD point p TO points  
                END IF  
            END FOR  
        END FOR  
        SET large_rectangle_area TO (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1)  
        IF area NOT EQUALS large_rectangle_area THEN  
            RETURN False  
        END IF  
        SET expected_corners TO the set consisting of point formed by X1 and Y1 point formed by X1 and Y2 point formed by X2 and Y1 point formed by X2 and Y2  
        IF points NOT EQUALS expected_corners THEN  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
