CLASS Solution  
    FUNCTION compareVersion(version1 version1 AS string, version2 version2 AS string) RETURNS integer  
        SET v1_revisions TO the result of splitting version1 by the period character  
        SET v2_revisions TO the result of splitting version2 by the period character  
        SET max_length TO the maximum of the length of v1_revisions AND the length of v2_revisions  
        FOR index i FROM zero TO max_length MINUS one  
            IF i LESS THAN the length of v1_revisions  
                SET v1 TO the integer value of the element at position i of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF i LESS THAN the length of v2_revisions  
                SET v2 TO the integer value of the element at position i of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2  
                RETURN minus one  
            ELSE IF v1 GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
