CLASS Solution  
    FUNCTION findMaximizedCapital(k w profits capital)  
        SET projects TO list of pairs of elements at same positions from capital and profits  
        SORT projects IN order of the element at position zero of each pair from smallest to largest  
        SET max_heap TO empty list  
        SET i TO zero  
        SET n TO the number of elements in projects  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND element at position zero of element at position i of projects LESS THAN OR EQUAL TO w  
                ADD negative of element at position one of element at position i of projects TO max_heap USING heap push  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
            SET w TO w PLUS the negative of the element popped from max_heap USING heap pop  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
