CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO an empty list  
        SET score TO zero  
        FOR each char IN s  
            IF char EQUALS the opening parenthesis symbol  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO the element removed from stack PLUS if score EQUALS zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
