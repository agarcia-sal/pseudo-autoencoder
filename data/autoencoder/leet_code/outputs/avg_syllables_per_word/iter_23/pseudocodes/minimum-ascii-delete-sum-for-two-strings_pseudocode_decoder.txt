CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        CREATE dp AS a two dimensional collection of size m PLUS one by n PLUS one filled with zeroes  
        
        FOR j FROM one TO n  
            SET element at position zero at position j of dp TO element at position zero at position j MINUS one of dp PLUS the ASCII value of the character at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position i at position zero of dp TO element at position i MINUS one at position zero of dp PLUS the ASCII value of the character at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the character at position i MINUS one of s1 EQUALS the character at position j MINUS one of s2  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j MINUS one of dp  
                ELSE  
                    SET element at position i at position j of dp TO the minimum value BETWEEN element at position i MINUS one at position j of dp PLUS the ASCII value of the character at position i MINUS one of s1 AND element at position i at position j MINUS one of dp PLUS the ASCII value of the character at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m at position n of dp  
    END FUNCTION  
END CLASS
