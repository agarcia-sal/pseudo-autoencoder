CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER n  
        SET self.parent TO list of integers from zero TO n inclusive  
        SET self.rank TO list of zeros with length n PLUS one  
        SET self.count TO n  
    END FUNCTION  

    FUNCTION find WITH PARAMETER u  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO CALL self.find WITH argument element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union WITH PARAMETERS u v  
        SET rootU TO CALL self.find WITH argument u  
        SET rootV TO CALL self.find WITH argument v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove WITH PARAMETERS n edges  
        SET uf_common TO new instance of UnionFind WITH argument n  
        SET uf_alice TO new instance of UnionFind WITH argument n  
        SET uf_bob TO new instance of UnionFind WITH argument n  
        SET used_edges TO zero  

        FOR each element consisting of t u v IN edges  
            IF t EQUALS three  
                IF CALL uf_common.union WITH arguments u v RETURNS True  
                    CALL uf_alice.union WITH arguments u v  
                    CALL uf_bob.union WITH arguments u v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  

        FOR each element consisting of t u v IN edges  
            IF t EQUALS one AND CALL uf_alice.union WITH arguments u v RETURNS True  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND CALL uf_bob.union WITH arguments u v RETURNS True  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  

        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
