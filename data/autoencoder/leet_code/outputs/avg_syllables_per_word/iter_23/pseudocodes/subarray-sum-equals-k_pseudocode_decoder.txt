CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary with key zero mapped TO one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            IF cumulative_sum_freq contains key current_sum MINUS k  
                INCREMENT count BY the value at key current_sum MINUS k of cumulative_sum_freq  
            END IF  
            
            IF cumulative_sum_freq contains key current_sum  
                INCREMENT the value at key current_sum of cumulative_sum_freq BY one  
            ELSE  
                SET the value at key current_sum of cumulative_sum_freq TO one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
