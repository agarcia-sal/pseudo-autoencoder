CLASS Solution  
    FUNCTION maxDotProduct(nums1 PARAMETER, nums2 PARAMETER)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  

        INITIALIZE dp AS a TWO DIMENSIONAL LIST  
        FOR each index i FROM zero TO m INCLUSIVE  
            INITIALIZE row AS an EMPTY LIST  
            FOR each index j FROM zero TO n INCLUSIVE  
                APPEND negative infinity TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  

        FOR each index i FROM one TO m INCLUSIVE  
            FOR each index j FROM one TO n INCLUSIVE  
                SET current_product TO element at position i MINUS one OF nums1 MULTIPLIED BY element at position j MINUS one OF nums2  

                SET dp at position i at position j TO the MAXIMUM OF  
                    current_product  
                    dp at position i MINUS one at position j  
                    dp at position i at position j MINUS one  
                    dp at position i MINUS one at position j MINUS one PLUS current_product  
            END FOR  
        END FOR  

        RETURN dp at position m at position n  
    END FUNCTION  
END CLASS
