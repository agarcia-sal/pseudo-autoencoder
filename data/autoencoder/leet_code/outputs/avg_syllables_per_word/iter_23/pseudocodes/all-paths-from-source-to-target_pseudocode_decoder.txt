CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        ASSIGN reference q TO a new empty double ended queue containing a list with element zero  
        ASSIGN reference ans TO an empty list  
        WHILE q is not empty  
            SET path TO the first element removed from the left side of q  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND a new list formed by appending v to path TO the right side of q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
