CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO an empty list  
        FOR each index FROM zero TO n MINUS one  
            APPEND a list of zero repeated n times TO matrix  
        END FOR  
        
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        
        WHILE the value of top IS LESS THAN OR EQUAL TO the value of bottom AND the value of left IS LESS THAN OR EQUAL TO the value of right  
            FOR each i FROM the value of left TO the value of right  
                SET the element at position top of matrix AT position i TO the value of num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            
            FOR each i FROM the value of top TO the value of bottom  
                SET the element at position i of matrix AT position right TO the value of num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            
            IF the value of top IS LESS THAN OR EQUAL TO the value of bottom  
                FOR each i FROM the value of right DOWN TO the value of left  
                    SET the element at position bottom of matrix AT position i TO the value of num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            
            IF the value of left IS LESS THAN OR EQUAL TO the value of right  
                FOR each i FROM the value of bottom DOWN TO the value of top  
                    SET the element at position i of matrix AT position left TO the value of num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
