CLASS Solution  
    FUNCTION minimumTime WITH PARAMETERS n relations time  
        ASSIGN reference TO an empty default dictionary that returns an empty list WHEN a key is accessed, NAMED graph  
        FOR each pair prevCourse AND nextCourse IN relations  
            APPEND prevCourse TO the list referenced by the element at position nextCourse of graph  
        END FOR  
        SET memo TO a list of size n PLUS the number one FILLED with the value minus one  
        FUNCTION dfs WITH PARAMETER course  
            IF the element at position course of memo NOT EQUALS minus one THEN  
                RETURN the element at position course of memo  
            END IF  
            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN the list referenced by the element at position course of graph  
                SET candidate_time TO the result of calling dfs WITH prevCourse  
                IF candidate_time GREATER THAN max_prerequisite_time THEN  
                    SET max_prerequisite_time TO candidate_time  
                END IF  
            END FOR  
            SET the element at position course of memo TO max_prerequisite_time PLUS the element at position course MINUS one of time  
            RETURN the element at position course of memo  
        END FUNCTION  
        SET total_minimum_time TO zero  
        FOR course FROM one TO n  
            SET candidate_time TO the result of calling dfs WITH course  
            IF candidate_time GREATER THAN total_minimum_time THEN  
                SET total_minimum_time TO candidate_time  
            END IF  
        END FOR  
        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
