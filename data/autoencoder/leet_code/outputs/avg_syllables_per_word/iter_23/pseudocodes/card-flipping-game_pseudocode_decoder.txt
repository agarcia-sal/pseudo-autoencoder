CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of elements x and y FROM fronts and backs IN PARALLEL  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO empty set  
        FOR each element IN fronts  
            ADD element TO all_numbers  
        END FOR  
        FOR each element IN backs  
            ADD element TO all_numbers  
        END FOR  
        SET good_numbers TO empty set  
        FOR each element IN all_numbers  
            IF element NOT IN same  
                ADD element TO good_numbers  
            END IF  
        END FOR  
        IF the LENGTH OF good_numbers GREATER THAN zero  
            SET minimum_good_number TO the smallest element IN good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
