CLASS Solution  
    FUNCTION kthSmallest(matrix param_kthSmallest, k param_kthSmallest)  
        FUNCTION countLessEqual(mid param_countLessEqual)  
            SET count TO zero  
            SET row TO the LENGTH OF matrix param_kthSmallest MINUS one  
            SET col TO zero  
            WHILE row GREATER THAN OR EQUAL TO zero AND col LESS THAN the LENGTH OF matrix param_kthSmallest  
                IF element at position row of matrix param_kthSmallest at position col LESS THAN OR EQUAL TO mid param_countLessEqual  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET n TO the LENGTH OF matrix param_kthSmallest  
        SET low TO element at position zero of matrix param_kthSmallest at position zero  
        SET high TO element at position n MINUS one of matrix param_kthSmallest at position n MINUS one  
        WHILE low LESS THAN high  
            SET mid TO the integer division of low PLUS high by two  
            IF the result of countLessEqual with argument mid LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
