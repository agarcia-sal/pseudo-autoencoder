CLASS Solution  
    FUNCTION shiftingLetters(s string shifts List OF integer) RETURNS string  
        SET total_shifts TO zero  
        FOR i FROM the LENGTH OF shifts MINUS one TO zero DECREMENT BY one  
            INCREMENT total_shifts BY element at position i of shifts  
            SET element at position i of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            SET shift_value TO element at position i of shifts  
            SET original_char_position TO the position of element at position i of s IN the alphabet starting from the letter a  
            SET new_char_position TO original_char_position PLUS shift_value MODULO the number of letters in the alphabet  
            SET new_character TO the letter at position new_char_position in the alphabet starting from the letter a  
            APPEND new_character TO result  
        END FOR  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
