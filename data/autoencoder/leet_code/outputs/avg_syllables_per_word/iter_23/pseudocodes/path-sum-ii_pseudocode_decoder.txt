CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node EQUALS None  
                RETURN  
            END IF  
            APPEND the value of node TO path  
            SET current_sum TO current_sum PLUS the value of node  
            IF left child of node EQUALS None AND right child of node EQUALS None AND current_sum EQUALS targetSum  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs(left child of node, current_sum, path)  
            CALL dfs(right child of node, current_sum, path)  
            REMOVE the last element FROM path  
        END FUNCTION  
        SET result TO an empty list  
        CALL dfs(root, zero, an empty list)  
        RETURN result  
    END FUNCTION  
END CLASS
