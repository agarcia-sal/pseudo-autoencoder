CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs formed by element at position zero of interval and corresponding index for each index and interval in enumeration of intervals  
        SET result TO empty list  
        FOR each unused_variable and end IN intervals  
            SET idx TO position found by binary search for tuple with element end in start_times  
            IF idx LESS THAN the LENGTH OF start_times  
                APPEND element at position one of element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
