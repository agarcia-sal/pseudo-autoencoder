CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze start destination  
        SET directions TO list containing the pairs of zero and one one and zero zero and negative one negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs WITH PARAMETERS x y  
            IF the tuple consisting of x and y IS IN visited THEN  
                RETURN False  
            END IF  
            IF the list consisting of x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD the tuple consisting of x and y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE the sum of nx and dx IS GREATER THAN OR EQUAL TO zero AND the sum of nx and dx IS LESS THAN m AND the sum of ny and dy IS GREATER THAN OR EQUAL TO zero AND the sum of ny and dy IS LESS THAN n AND element at position the sum of nx and dx of maze AND at position the sum of ny and dy EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs WITH nx ny RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs WITH element at position zero of start element at position one of start  
    END FUNCTION  
END CLASS
