CLASS Solution  
    FUNCTION findAllPeople(n  
                           meetings  
                           firstPerson)  
        SET know_secret TO the set containing zero AND firstPerson  
        SORT meetings BY the element at position three of each meeting  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position three of the meeting at position i of meetings  
            CREATE an empty mapping graph from a default list  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position three of the meeting at position i of meetings EQUALS current_time  
                SET x TO the element at position one of the meeting at position i of meetings  
                SET y TO the element at position two of the meeting at position i of meetings  
                APPEND y TO the list at key x of graph  
                APPEND x TO the list at key y of graph  
                INCREMENT i BY one  
            END WHILE  
            CREATE a queue containing each person from know_secret who is a key in graph  
            CREATE an empty set visited  
            WHILE the queue IS NOT empty  
                REMOVE the first person FROM queue AND SET person TO it  
                IF person IS IN visited  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person of graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list converted from know_secret  
    END FUNCTION  
END CLASS
