CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO stations  
        
        FOR each position fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is LESS THAN position  
                SET current_fuel TO current_fuel MINUS the largest fuel amount removed from max_heap  
                INCREMENT stops BY one  
            END WHILE  
            
            IF current_fuel is LESS THAN position  
                RETURN negative one  
            END IF  
            
            ADD the negative of fuel TO max_heap  
        END FOR  
        
        RETURN stops  
    END FUNCTION  
END CLASS
