CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one THEN  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF used at position i EQUALS False AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero) THEN  
                    SET used at position i TO True  
                    INCREMENT count BY backtrack with parameters used AND index PLUS one  
                    SET used at position i TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO list of False values with length n PLUS one  
        RETURN backtrack with parameters used AND one  
    END FUNCTION  
END CLASS
