CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO A new instance of ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO A new instance of ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list(the next attribute of p1 the next attribute of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self head x)  
        SET less_head TO A new instance of ListNode with zero  
        SET greater_head TO A new instance of ListNode with zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current is not None  
            IF the value attribute of current LESS THAN x  
                SET the next attribute of less TO current  
                SET less TO the next attribute of less  
            ELSE  
                SET the next attribute of greater TO current  
                SET greater TO the next attribute of greater  
            END IF  
            SET current TO the next attribute of current  
        END WHILE  
        SET the next attribute of less TO the next attribute of greater_head  
        SET the next attribute of greater TO None  
        RETURN the next attribute of less_head  
    END FUNCTION  
END CLASS
