CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position zero of each element  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF length of min_heap GREATER THAN zero AND the element at position zero of meeting GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap as the smallest element  
        END FOR  
        RETURN length of min_heap  
    END FUNCTION  
END CLASS
