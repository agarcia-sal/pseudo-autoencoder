CLASS Solution  
    FUNCTION maxProfit(prices)  
        IF the LENGTH OF prices EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET min_price1 TO infinite  
        SET min_price2 TO infinite  
        SET max_profit1 TO zero  
        SET max_profit2 TO zero  
        FOR each price IN prices  
            IF min_price1 GREATER THAN price THEN  
                SET min_price1 TO price  
            END IF  
            IF max_profit1 LESS THAN price MINUS min_price1 THEN  
                SET max_profit1 TO price MINUS min_price1  
            END IF  
            IF min_price2 GREATER THAN price MINUS max_profit1 THEN  
                SET min_price2 TO price MINUS max_profit1  
            END IF  
            IF max_profit2 LESS THAN price MINUS min_price2 THEN  
                SET max_profit2 TO price MINUS min_price2  
            END IF  
        END FOR  
        RETURN max_profit2  
    END FUNCTION  
END CLASS
