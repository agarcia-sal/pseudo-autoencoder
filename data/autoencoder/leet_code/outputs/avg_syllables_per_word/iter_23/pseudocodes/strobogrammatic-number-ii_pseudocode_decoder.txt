CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN a list containing an empty string  
            END IF  
            IF u EQUALS one  
                RETURN a list containing the string zero one and eight  
            END IF  
            SET ans TO an empty list  
            FOR each v IN the result of dfs with parameter u MINUS two  
                FOR each pair l r IN the list of string pairs eleven eighty-eight sixty-nine ninety-six  
                    APPEND the concatenation of l and v and r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND the concatenation of zero and v and zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs with parameter n  
    END FUNCTION  
END CLASS
