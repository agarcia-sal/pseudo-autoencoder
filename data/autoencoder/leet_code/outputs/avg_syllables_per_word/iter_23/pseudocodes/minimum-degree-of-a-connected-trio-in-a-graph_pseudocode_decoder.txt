CLASS Solution  
    FUNCTION minTrioDegree WITH PARAMETERS n edges  
        SET graph TO an empty mapping from integers to sets  
        FOR each integer i FROM one TO n  
            SET element at position i of graph TO an empty set  
        END FOR  
        FOR each pair u v IN edges  
            ADD v TO element at position u of graph  
            ADD u TO element at position v of graph  
        END FOR  
        SET min_degree TO infinity  
        SET trio_found TO false  
        FOR each integer i FROM one TO n  
            FOR each integer j FROM i PLUS one TO n  
                FOR each integer k FROM j PLUS one TO n  
                    IF element j IS IN element at position i of graph AND element k IS IN element at position i of graph AND element k IS IN element at position j of graph THEN  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF element at position i of graph PLUS the LENGTH OF element at position j of graph PLUS the LENGTH OF element at position k of graph MINUS six  
                        IF trio_degree LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found EQUALS true THEN  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
