CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new mapping from value to an EMPTY list  
        FOR each position index FROM zero TO n MINUS one  
            SET value TO element at position index of arr  
            APPEND position index TO the list associated WITH value IN value_indices  
        END FOR  
        
        SET queue TO a new double ended queue containing a pair of zero and zero representing current position and steps taken  
        SET visited TO a new set containing zero  
        
        WHILE queue IS NOT empty  
            REMOVE the first element FROM queue AND UNPACK INTO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list containing current_index PLUS one AND current_index MINUS one  
            APPEND all elements from the list associated WITH element at position current_index of arr IN value_indices TO next_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list associated WITH element at position current_index of arr IN value_indices TO an EMPTY list  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
