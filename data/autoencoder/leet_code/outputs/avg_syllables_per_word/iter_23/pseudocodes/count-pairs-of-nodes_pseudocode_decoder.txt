CLASS Solution  
    FUNCTION countPairs(n, edges, queries)  
        SET degree TO a list of zeros with length n PLUS one  
        FOR each pair u AND v IN edges  
            INCREMENT the element at position u of degree BY one  
            INCREMENT the element at position v of degree BY one  
        END FOR  
        
        SET shared TO an empty mapping with default integer zero  
        FOR each pair u AND v IN edges  
            IF u GREATER THAN v THEN  
                SET temporary variable temp_u TO v  
                SET v TO u  
                SET u TO temp_u  
            END IF  
            INCREMENT the count associated with the pair u AND v IN shared BY one  
        END FOR  
        
        SET sorted_degree TO the list degree sorted in non-decreasing order  
        
        SET answers TO an empty list  
        FOR each query q IN queries  
            SET count TO zero  
            SET left TO one  
            SET right TO n  
            WHILE left LESS THAN right  
                IF the element at position left of sorted_degree PLUS the element at position right of sorted_degree GREATER THAN q THEN  
                    INCREMENT count BY right MINUS left  
                    DECREMENT right BY one  
                ELSE  
                    INCREMENT left BY one  
                END IF  
            END WHILE  
            
            FOR each pair (u, v) AND associated count_uv IN shared  
                SET degree_uv TO the element at position u of degree PLUS the element at position v of degree  
                IF degree_uv GREATER THAN q AND degree_uv MINUS count_uv LESS THAN OR EQUAL TO q THEN  
                    DECREMENT count BY one  
                END IF  
            END FOR  
            
            APPEND count TO answers  
        END FOR  
        
        RETURN answers  
    END FUNCTION  
END CLASS
