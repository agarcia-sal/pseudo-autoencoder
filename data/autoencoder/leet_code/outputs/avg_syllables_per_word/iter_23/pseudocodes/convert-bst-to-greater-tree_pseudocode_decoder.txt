CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with the first element of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot left TO new TreeNode with element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot right TO new TreeNode with element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with p dot left and q dot left AND the result of is_same_tree with p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION convertBST WITH PARAMETERS self root  
        SET self dot sum TO zero  
        
        FUNCTION traverse WITH PARAMETER node  
            IF node EQUALS none  
                RETURN  
            END IF  
            CALL traverse with node dot right  
            INCREMENT self dot sum BY node dot val  
            SET node dot val TO self dot sum  
            CALL traverse with node dot left  
        END FUNCTION  
        
        CALL traverse with root  
        RETURN root  
    END FUNCTION  
END CLASS
