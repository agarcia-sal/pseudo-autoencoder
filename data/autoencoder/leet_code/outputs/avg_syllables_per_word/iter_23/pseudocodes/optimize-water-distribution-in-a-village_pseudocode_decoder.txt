CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND a list containing zero and the sum of index PLUS one and the element at position index of wells TO pipes  
        END FOR  
        
        SORT pipes BY the element at position two of each element in ascending order  
        
        SET parent TO a list containing the range from zero TO n PLUS one MINUS one  
        
        FUNCTION find(x)  
            IF the element at position x of parent NOT EQUALS x  
                SET the element at position x of parent TO the result of find with the element at position x of parent  
            END IF  
            RETURN the element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO the result of find with x  
            SET rootY TO the result of find with y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET min_cost TO zero  
        FOR each element containing house1, house2, cost IN pipes  
            IF union called with house1 and house2 RETURNS True  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
