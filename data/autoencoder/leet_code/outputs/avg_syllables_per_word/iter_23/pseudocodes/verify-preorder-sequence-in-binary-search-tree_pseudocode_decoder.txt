CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value LESS THAN lower_bound THEN  
                RETURN False  
            END IF  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND value GREATER THAN element at the last position of stack  
                SET lower_bound TO element removed from the last position of stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
