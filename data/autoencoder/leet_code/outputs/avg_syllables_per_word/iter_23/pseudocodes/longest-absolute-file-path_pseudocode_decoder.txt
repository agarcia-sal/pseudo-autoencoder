CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the list of substrings obtained by splitting input at each newline character  
        SET stack TO an empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters contained in line  
            SET name TO the substring of line starting from position depth plus one to the end of line  
            
            WHILE the length of stack is GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF stack is not empty  
                SET current_length TO the last element of stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF the character representing a dot IS CONTAINED IN name  
                SET max_length TO the greater value BETWEEN max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
