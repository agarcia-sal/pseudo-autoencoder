CLASS Solution  
    FUNCTION getHint(secret parameter, guess parameter)  
        SET bulls TO zero  
        SET cows TO zero  
        ASSIGN secret_count TO a new mapping with default zero for missing keys  
        ASSIGN guess_count TO a new mapping with default zero for missing keys  
        FOR each pair OF characters s AND g IN parallel iteration over secret parameter AND guess parameter  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value AT key s IN secret_count BY one  
                INCREMENT the value AT key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys OF secret_count  
            INCREMENT cows BY the smaller value BETWEEN the value at key digit IN secret_count AND the value at key digit IN guess_count  
        END FOR  
        RETURN the concatenation of the text form of bulls THEN the character A THEN the text form of cows THEN the character B  
    END FUNCTION  
END CLASS
