CLASS Solution  
    FUNCTION countSmaller(nums)  
        SET sorted_nums TO the sorted list of unique elements from nums  
        SET rank TO an empty dictionary  
        SET index TO zero  
        FOR each value IN sorted_nums  
            INCREMENT index BY one  
            SET the element at key value in rank TO index  
        END FOR  
        
        SET bit TO a list of zeros with length the length of sorted_nums PLUS one  
        
        FUNCTION get_sum(idx)  
            SET res TO zero  
            WHILE idx GREATER THAN zero  
                INCREMENT res BY the element at position idx of bit  
                SET idx TO idx MINUS the bitwise AND of idx and the negative of idx interpreted as an integer  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        FUNCTION update(idx delta)  
            WHILE idx LESS THAN the length of bit  
                INCREMENT the element at position idx of bit BY delta  
                SET idx TO idx PLUS the bitwise AND of idx and the negative of idx interpreted as an integer  
            END WHILE  
        END FUNCTION  
        
        SET counts TO an empty list  
        FOR each num IN nums iterated from the last element to the first element  
            SET r TO the element at key num in rank  
            APPEND the result of get_sum with argument r MINUS one TO counts  
            CALL update with arguments r and one  
        END FOR  
        
        RETURN the reversed list counts  
    END FUNCTION  
END CLASS
