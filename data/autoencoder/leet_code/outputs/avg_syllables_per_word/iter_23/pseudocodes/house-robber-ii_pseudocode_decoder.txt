CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN the element at position zero of houses  
            END IF  
            SET dp TO a list of zeros WITH the LENGTH OF houses  
            SET the element at position zero of dp TO the element at position zero of houses  
            IF the element at position zero of houses GREATER THAN OR EQUAL TO the element at position one of houses  
                SET the element at position one of dp TO the element at position zero of houses  
            ELSE  
                SET the element at position one of dp TO the element at position one of houses  
            END IF  
            FOR i FROM two TO the LENGTH OF houses MINUS one  
                IF the element at position i MINUS one of dp GREATER THAN OR EQUAL TO the element at position i MINUS two of dp PLUS the element at position i of houses  
                    SET the element at position i of dp TO the element at position i MINUS one of dp  
                ELSE  
                    SET the element at position i of dp TO the element at position i MINUS two of dp PLUS the element at position i of houses  
                END IF  
            END FOR  
            RETURN the element at the last position of dp  
        END FUNCTION  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        SET result_one TO rob_linear(all elements of nums EXCEPT the last element)  
        SET result_two TO rob_linear(all elements of nums EXCEPT the first element)  
        IF result_one GREATER THAN OR EQUAL TO result_two  
            RETURN result_one  
        ELSE  
            RETURN result_two  
        END IF  
    END FUNCTION  
END CLASS
