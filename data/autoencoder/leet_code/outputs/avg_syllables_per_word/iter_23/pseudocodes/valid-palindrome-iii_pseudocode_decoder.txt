CLASS Solution  
    FUNCTION isValidPalindrome(s parameter, k parameter)  
        FUNCTION longest_palindromic_subseq(s parameter)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional list where each element is zero with the number of rows equal to n and the number of columns equal to n  
            
            FOR i FROM zero TO n MINUS one  
                SET element at position i of element at position i of dp TO one  
            END FOR  
            
            FOR length FROM two TO n PLUS one  
                FOR i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF element at position i of s EQUALS element at position j of s  
                        SET element at position i of element at position j of dp TO element at position i PLUS one of element at position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i of element at position j of dp TO the maximum value BETWEEN element at position i PLUS one of element at position j of dp AND element at position i of element at position j MINUS one of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero of element at position n MINUS one of dp  
        END FUNCTION  
        
        SET longest_palindrome_length TO longest_palindromic_subseq CALLED WITH s  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
