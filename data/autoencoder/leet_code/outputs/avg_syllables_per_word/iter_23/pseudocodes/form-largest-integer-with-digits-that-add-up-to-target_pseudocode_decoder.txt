CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS cost AND target  
        SET cost_to_digit TO an empty mapping from cost values TO corresponding digit strings from one TO the length of the cost list  
        SET dp TO a list where the first element is the empty string and the following elements up to position target are the string representing minus one  
        FOR t FROM one TO target  
            FOR each pair of cost_value AND digit_string IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO cost_value AND the element at position t MINUS cost_value of dp NOT EQUALS the string representing minus one  
                    SET candidate TO the concatenation of digit_string AND the element at position t MINUS cost_value of dp  
                    IF the element at position t of dp EQUALS the string representing minus one OR the LENGTH OF candidate GREATER THAN the LENGTH OF the element at position t of dp OR the LENGTH OF candidate EQUALS the LENGTH OF the element at position t of dp AND candidate GREATER THAN the element at position t of dp  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF the element at position target of dp NOT EQUALS the string representing minus one  
            RETURN the element at position target of dp  
        ELSE  
            RETURN the string representing zero  
        END IF  
    END FUNCTION  
END CLASS
