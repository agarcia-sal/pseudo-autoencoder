CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF the element at position one of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the FIRST element of nums AS STRING FOLLOWED BY the WORD divided by FOLLOWED BY the SECOND element of nums AS STRING  
        END IF  
        RETURN the FIRST element of nums AS STRING FOLLOWED BY the WORD divided by FOLLOWED BY an OPEN PARENTHESIS FOLLOWED BY the STRING REPRESENTATION OF the element at position two of nums FOLLOWED BY the WORD divided by FOLLOWED BY the STRING REPRESENTATION OF the element at position three of nums AND SO ON UNTIL the LAST element of nums FOLLOWED BY a CLOSE PARENTHESIS  
    END FUNCTION  
END CLASS
