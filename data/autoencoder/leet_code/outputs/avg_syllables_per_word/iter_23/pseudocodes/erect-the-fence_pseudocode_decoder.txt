CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION cross(i j k)  
            ASSIGN a TO element at position i of trees  
            ASSIGN b TO element at position j of trees  
            ASSIGN c TO element at position k of trees  
            RETURN the quantity of the difference of the first element of b MINUS the first element of a MULTIPLIED BY the difference of the second element of c MINUS the second element of b MINUS the difference of the second element of b MINUS the second element of a MULTIPLIED BY the difference of the first element of c MINUS the first element of b  
        END FUNCTION  

        ASSIGN n TO the LENGTH OF trees  
        IF n LESS THAN four  
            RETURN trees  
        END IF  

        PERFORM an in place sort of trees in ascending order  
        ASSIGN vis TO a list of BOOLEAN false repeated n times  
        ASSIGN stk TO a list containing zero  
        
        FOR i FROM one TO n MINUS one  
            WHILE the LENGTH OF stk GREATER THAN one AND cross of the element at position LENGTH OF stk MINUS two of stk OFF THE TOP OF stk AND i LESS THAN zero holds  
                ASSIGN to the element at position popped value from stk in vis BOOLEAN false  
            END WHILE  
            ASSIGN to the element at position i in vis BOOLEAN true  
            APPEND i TO stk  
        END FOR  

        ASSIGN m TO the LENGTH OF stk  
        FOR i FROM n MINUS two DOWN TO zero  
            IF the element at position i in vis EQUALS true  
                CONTINUE TO the next iteration of the FOR loop  
            END IF  
            WHILE the LENGTH OF stk GREATER THAN m AND cross of element at position LENGTH OF stk MINUS two of stk OFF THE TOP OF stk AND i LESS THAN zero holds  
                REMOVE the element at the top of stk  
            END WHILE  
            APPEND i TO stk  
        END FOR  

        REMOVE the top element from stk  
        RETURN a list of elements at positions described by elements in stk from trees  
    END FUNCTION  
END CLASS
