CLASS Solution  
    FUNCTION addBoldTag WITH PARAMETERS s words  
        IF the LENGTH OF words EQUALS zero THEN  
            RETURN s  
        END IF  
  
        SET n TO the LENGTH OF s  
        SET mask TO a list of BOOLEAN FALSE values repeated n times  
  
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word DO  
                SET start TO the position of the first occurrence of word in s starting from start  
                IF start EQUALS negative one THEN  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one DO  
                    SET element at position i of mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN n DO  
            IF element at position i of mask IS TRUE THEN  
                APPEND the string bold opening tag TO result  
                WHILE i LESS THAN n AND element at position i of mask IS TRUE DO  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string bold closing tag TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
