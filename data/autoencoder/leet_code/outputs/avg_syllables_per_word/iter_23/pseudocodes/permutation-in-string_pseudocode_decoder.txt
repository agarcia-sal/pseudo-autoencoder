CLASS Solution  
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean  
        SET length_of_s1 TO the LENGTH OF s1  
        SET length_of_s2 TO the LENGTH OF s2  
        IF length_of_s1 GREATER THAN length_of_s2  
            RETURN false  
        END IF  
        SET count_of_s1 TO the frequency count of each character in s1  
        SET count_of_s2 TO the frequency count of each character in the substring consisting of the first length_of_s1 characters of s2  
        IF count_of_s1 EQUALS count_of_s2  
            RETURN true  
        END IF  
        FOR i FROM length_of_s1 TO length_of_s2 MINUS one  
            INCREMENT the count of the character at position i of s2 in count_of_s2 BY one  
            DECREMENT the count of the character at position i MINUS length_of_s1 of s2 in count_of_s2 BY one  
            IF the count of the character at position i MINUS length_of_s1 of s2 in count_of_s2 EQUALS zero  
                REMOVE the character at position i MINUS length_of_s1 of s2 FROM count_of_s2  
            END IF  
            IF count_of_s1 EQUALS count_of_s2  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
