CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO list containing single element negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND element h IN enumeration of heights  
            WHILE element at position last of stack NOT EQUALS negative one AND h LESS THAN element at position element at position last of stack of heights  
                SET height TO element at position result of removing last element from stack of heights  
                SET width TO i MINUS element at position last of stack MINUS one  
                SET max_area TO GREATER OF max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
