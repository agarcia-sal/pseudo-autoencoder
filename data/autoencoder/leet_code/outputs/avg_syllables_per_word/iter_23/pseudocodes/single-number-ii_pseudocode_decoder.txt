CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO the result of bitwise XOR between ones AND num AND the result of bitwise NOT of twos  
            SET twos TO the result of bitwise XOR between twos AND num AND the result of bitwise NOT of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
