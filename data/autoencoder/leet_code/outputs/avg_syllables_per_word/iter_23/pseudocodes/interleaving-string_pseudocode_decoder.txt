CLASS Solution  
    FUNCTION isInterleave(s1 s1 STRING s2 s2 STRING s3 s3 STRING) RETURNS BOOLEAN  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN false  
        END IF  
        SET dp TO a list of false BOOLEAN values with the LENGTH OF s2 PLUS one elements  
        SET element at position zero of dp TO true  
        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET element at position j of dp TO the logical AND of element at position j of dp AND the EQUALITY of element at position i MINUS one of s1 AND element at position i PLUS j MINUS one of s3  
                END IF  
                IF j GREATER THAN zero  
                    SET element at position j of dp TO the logical OR of element at position j of dp AND the logical AND of element at position j MINUS one of dp AND the EQUALITY of element at position j MINUS one of s2 AND element at position i PLUS j MINUS one of s3  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
