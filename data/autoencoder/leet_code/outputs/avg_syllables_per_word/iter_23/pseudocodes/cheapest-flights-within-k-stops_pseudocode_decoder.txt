CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO empty mapping from integer to empty list  
        FOR each integer i FROM zero TO n MINUS one  
            SET element at position i of graph TO empty list  
        END FOR  
        FOR each triple of values u v w IN flights  
            APPEND pair of values v and w TO element at position u of graph  
        END FOR  
        
        SET min_heap TO list containing a triple of zero cost negative one stops and src city  
        SET min_cost TO mapping with src key assigned negative one value  
        
        WHILE the LENGTH OF min_heap is greater than zero  
            REMOVE and RETURN smallest element FROM min_heap AND ASSIGN to cost stops city  
            
            IF city EQUALS dst THEN  
                RETURN cost  
            END IF  
            
            IF stops GREATER THAN OR EQUAL TO k THEN  
                CONTINUE to next iteration of WHILE  
            END IF  
            
            FOR each pair of neighbor price IN element at position city of graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor NOT IN min_cost OR next_stops LESS THAN element at key neighbor of min_cost OR next_cost LESS THAN cost element at position zero of smallest element in min_heap THEN  
                    SET element at key neighbor of min_cost TO next_stops  
                    INSERT triple of next_cost next_stops neighbor INTO min_heap maintaining ORDER by first element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
