CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the left property of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the right property of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val property of p NOT EQUALS the val property of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS None THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue initialized with root  
        WHILE the queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each number in the range from one TO level_size  
                SET node TO the element removed from the front of queue  
                APPEND the val property of node TO current_level  
                IF the left property of node IS NOT None THEN  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node IS NOT None THEN  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
