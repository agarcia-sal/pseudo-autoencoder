CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY one hundred million PLUS seven  
        SET dp TO LIST OF zero WITH LENGTH the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character representing an asterisk  
                RETURN nine  
            ELSE IF char EQUALS the character representing zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character representing an asterisk AND char2 EQUALS the character representing an asterisk  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character representing an asterisk  
                IF char2 LESS THAN OR EQUAL TO the character representing six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character representing an asterisk  
                IF char1 EQUALS the character representing one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character representing two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer formed by concatenating char1 THEN char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION
        
        FOR i FROM one TO the LENGTH OF s  
            SET dp at position i TO the element at position i of dp PLUS the element at position i minus one of dp MULTIPLIED BY ways_to_decode_single(element at position i minus one of s) MODULO MOD  
            
            IF i GREATER THAN one  
                SET dp at position i TO the element at position i of dp PLUS the element at position i minus two of dp MULTIPLIED BY ways_to_decode_pair(element at position i minus two of s, element at position i minus one of s) MODULO MOD  
            END IF  
        END FOR  
        
        RETURN the element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
