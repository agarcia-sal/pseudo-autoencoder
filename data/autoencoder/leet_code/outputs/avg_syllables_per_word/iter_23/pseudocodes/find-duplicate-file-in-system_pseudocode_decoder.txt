CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new mapping from content TO list of file paths  
        FOR each path IN paths  
            SET parts TO list of substrings obtained by splitting path by whitespace  
            SET directory TO the element at position one of parts  
            FOR each file_info IN parts from position two to the last element  
                SET file_name TO substring of file_info from start TO the position before the first opening parenthesis  
                SET content TO substring of file_info from the position after the first opening parenthesis TO the position before the last character  
                APPEND the concatenation of directory PLUS a forward slash PLUS file_name TO the list at content_to_paths for key content  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each paths_group IN the list of all values from content_to_paths  
            IF the length of paths_group GREATER THAN one  
                APPEND paths_group TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
