CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new double ended queue initialized with a single element containing zero for position one for speed zero for steps  
        ASSIGN reference TO a new set initialized with a single element containing zero for position and one for speed representing visited states  
        WHILE the queue is not empty  
            REMOVE the first element from the queue and destructure into position speed steps  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed is not contained in visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND a tuple of next_position next_speed steps PLUS one TO the queue  
            END IF  
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed is not contained in visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND a tuple of position next_speed steps PLUS one TO the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
