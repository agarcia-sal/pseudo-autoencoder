CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO the list of words representing numbers from empty string up to Nineteen as specified  
        SET tens TO the list of words representing multiples of Ten from empty strings up to Ninety as specified  
        SET thousands TO the list of words representing powers of Thousand including empty string, Thousand, Million, Billion  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN element at position n of below_twenty concatenated with a space  
            ELSE IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten of tens concatenated with a space concatenated with helper called with the remainder of n DIVIDED BY ten  
            ELSE  
                RETURN element at position n DIVIDED BY one hundred of below_twenty concatenated with the word Hundred concatenated with a space concatenated with helper called with the remainder of n DIVIDED BY one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF remainder of num DIVIDED BY one thousand NOT EQUALS zero  
                SET result TO helper called with remainder of num DIVIDED BY one thousand concatenated with element at position i of thousands concatenated with a space concatenated with result  
            END IF  
            SET num TO num DIVIDED BY one thousand as integer division  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
