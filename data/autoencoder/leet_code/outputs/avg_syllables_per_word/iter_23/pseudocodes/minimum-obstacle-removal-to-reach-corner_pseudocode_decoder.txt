CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the COLLECTION of pairs representing zero and one zero and zero and one zero and minus one and zero and minus one and zero  
        SET min_heap TO a COLLECTION containing a triple of zero zero zero representing cost and x and y  
        SET visited TO a SET containing the pair of zero and zero  
        
        WHILE min_heap is not empty  
            REMOVE and RETURN the triple with the smallest cost from min_heap and ASSIGN cost and x and y respectively  
            
            IF pair of x and y EQUALS pair of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND pair of nx and ny NOT IN visited  
                    ADD pair of nx and ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of element at position ny of grid  
                    ADD triple of new_cost nx ny TO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
