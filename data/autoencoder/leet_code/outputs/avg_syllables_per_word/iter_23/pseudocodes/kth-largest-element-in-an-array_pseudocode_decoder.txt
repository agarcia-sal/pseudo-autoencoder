CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO empty list  
        FOR each num IN nums  
            ADD num TO min_heap USING heap push operation  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE smallest element FROM min_heap USING heap pop operation  
            END IF  
        END FOR  
        RETURN element at position one of min_heap  
    END FUNCTION  
END CLASS
