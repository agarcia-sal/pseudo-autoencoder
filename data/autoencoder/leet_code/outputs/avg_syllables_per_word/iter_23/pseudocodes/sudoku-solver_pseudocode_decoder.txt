CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO nine MINUS one  
                IF element at position row of board element at position i EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO nine MINUS one  
                IF element at position i of board element at position col EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY row DIVIDED BY three  
            SET start_col TO three MULTIPLIED BY col DIVIDED BY three  
            FOR i FROM zero TO three MINUS one  
                FOR j FROM zero TO three MINUS one  
                    IF element at position start_row PLUS i of board element at position start_col PLUS j EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        FUNCTION solve(board)  
            FOR row FROM zero TO nine MINUS one  
                FOR col FROM zero TO nine MINUS one  
                    IF element at position row of board element at position col EQUALS the character DOT  
                        FOR num IN the string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position row of board element at position col TO num  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position row of board element at position col TO the character DOT  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        CALL solve(board)  
    END FUNCTION  
END CLASS
