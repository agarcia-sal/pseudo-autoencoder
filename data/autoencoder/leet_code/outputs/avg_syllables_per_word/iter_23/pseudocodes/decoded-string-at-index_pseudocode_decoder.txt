CLASS Solution  
    FUNCTION decodeAtIndex(s string, k integer) RETURNS string  
        SET length TO zero  
        FOR each char IN s  
            IF char IS A DIGIT  
                SET length TO length MULTIPLIED BY the numeric value of char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN s TRAVERSED IN REVERSE ORDER  
            SET k TO the remainder of k DIVIDED BY length  
            IF k EQUALS zero AND char IS A LETTER  
                RETURN char  
            END IF  
            IF char IS A DIGIT  
                SET length TO length DIVIDED BY the numeric value of char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
