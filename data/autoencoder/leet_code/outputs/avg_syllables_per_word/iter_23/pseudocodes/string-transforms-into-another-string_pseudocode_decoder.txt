CLASS Solution  
    FUNCTION canConvert(str1 str2) RETURNS boolean  
        IF str1 EQUALS str2  
            RETURN true  
        END IF  
        
        SET conversion_map TO empty mapping  
        
        FOR each pair of characters char1 and char2 FROM the simultaneous iteration of str1 and str2  
            IF char1 EXISTS AS a key IN conversion_map  
                IF the value mapped from char1 in conversion_map NOT EQUALS char2  
                    RETURN false  
                END IF  
            ELSE  
                SET the value mapped from char1 in conversion_map TO char2  
            END IF  
        END FOR  
        
        IF the number of unique characters in str2 EQUALS the total number of characters in the alphabet twenty six  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
