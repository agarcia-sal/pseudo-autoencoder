CLASS Solution  
    FUNCTION findTargetSumWays(nums target)  
        FUNCTION dfs(index current_sum memo)  
            IF index EQUALS the LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the PAIR of index AND current_sum IS IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
            END IF  
            SET positive TO CALL dfs WITH index PLUS one current_sum PLUS element at position index of nums memo  
            SET negative TO CALL dfs WITH index PLUS one current_sum MINUS element at position index of nums memo  
            SET the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
        END FUNCTION  
        RETURN CALL dfs WITH zero zero an EMPTY collection  
    END FUNCTION  
END CLASS
