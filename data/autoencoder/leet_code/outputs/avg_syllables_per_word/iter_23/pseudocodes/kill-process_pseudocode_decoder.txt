CLASS Solution  
    FUNCTION killProcess(pid AS List of integers ppid AS List of integers kill AS integer) RETURNS List of integers  
        SET tree TO a dictionary with default empty list  
        FOR each pair of child and parent IN pairs of elements from pid AND ppid  
            APPEND child TO the list stored at key parent in tree  
        END FOR  
        SET queue TO a list containing the element kill  
        SET killed_processes TO an empty list  
        WHILE the list queue is not empty  
            REMOVE the first element from queue AND ASSIGN it TO current  
            APPEND current TO killed_processes  
            FOR each child IN the list stored at key current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
