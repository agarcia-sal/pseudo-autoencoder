CLASS Solution  
    FUNCTION movesToChessboard WITH PARAMETER board  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF element at position i of board MINUS ONE AND element at position j of board MINUS ONE XOR element at position i of board MINUS ONE AND element at position zero of board MINUS ONE XOR element at position zero of board AND element at position j of board MINUS ONE XOR element at position zero of board AND element at position zero of board NOT EQUALS zero  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY element at position i of board MINUS ONE AND element at position zero of board MINUS ONE  
            INCREMENT colSum BY element at position zero of board AND element at position i of board MINUS ONE  
            INCREMENT rowSwap BY the expression that element at position i of board MINUS ONE AND element at position zero of board MINUS ONE EQUALS the remainder of dividing i BY two  
            INCREMENT colSwap BY the expression that element at position zero of board AND element at position i of board MINUS ONE EQUALS the remainder of dividing i BY two  
        END FOR  
        IF the integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of dividing N BY two NOT EQUALS zero  
            IF the remainder of dividing rowSwap BY two NOT EQUALS zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of dividing colSwap BY two NOT EQUALS zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum of colSwap AND N MINUS colSwap  
        END IF  
        RETURN the integer division of the sum of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
