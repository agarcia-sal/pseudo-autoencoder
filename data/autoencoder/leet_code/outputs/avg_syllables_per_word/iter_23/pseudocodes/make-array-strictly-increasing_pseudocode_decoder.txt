CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2 IN ASCENDING ORDER  
        SET dp TO a dictionary with key negative one AND value zero  
        
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each pair of last_num AND ops IN dp items  
                IF number GREATER THAN last_num  
                    SET current_min TO the value in new_dp at position number OR positive infinity  
                    SET new_dp at position number TO the smaller value BETWEEN current_min AND ops  
                END IF  
                
                SET idx TO the position obtained by finding the smallest index in arr2 which has an element GREATER THAN last_num  
                IF idx LESS THAN the LENGTH OF arr2  
                    SET replace_num TO element at position idx in arr2  
                    SET current_min_replace TO the value in new_dp at position replace_num OR positive infinity  
                    SET new_dp at position replace_num TO the smaller value BETWEEN current_min_replace AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp IS empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the smallest value among all values in dp  
    END FUNCTION  
END CLASS
