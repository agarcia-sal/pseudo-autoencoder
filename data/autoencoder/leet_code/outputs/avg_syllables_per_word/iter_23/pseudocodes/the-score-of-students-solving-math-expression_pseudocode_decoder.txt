CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character represents a digit  
                    SET num TO num MULTIPLIED BY ten PLUS numeric value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the multiplication symbol  
                    SET element at position i minus one of tokens TO element at position i minus one of tokens MULTIPLIED BY element at position i plus one of tokens  
                    REMOVE two elements from tokens starting at position i  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the addition symbol  
                    SET result TO result PLUS element at position i plus one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF expression consists only of digits  
                RETURN set containing numeric value of expression  
            END IF  
            
            SET results TO empty set  
            FOR index i FROM one TO the LENGTH OF expression MINUS two STEP two  
                SET left_results TO get_possible_answers(substring of expression from position zero TO position i minus one)  
                SET right_results TO get_possible_answers(substring of expression from position i plus one TO the last position of expression)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF character at position i of expression EQUALS the addition symbol  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
