CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with value as element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO the end  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the condition that element val of p1 EQUALS element val of p2 AND the result of is_same_list with p1.next and p2.next  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value as element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the length of queue is greater than zero  
        SET node TO the first element removed from queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF element val of p NOT EQUALS element val of q THEN  
        RETURN False  
    ELSE  
        RETURN the condition that is_same_tree with p.left and q.left AND is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the length of s THEN  
                RETURN None and index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the length of s AND (element at position index of s IS a digit OR element at position index of s EQUALS the character minus)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO new TreeNode with value converted to integer from substring of s from position start TO position index minus one  
            IF index LESS THAN the length of s AND element at position index of s EQUALS the character opening parenthesis THEN  
                SET node.left AND index TO the results of parse_tree with argument index PLUS one  
            END IF  
            IF index LESS THAN the length of s AND element at position index of s EQUALS the character opening parenthesis THEN  
                SET node.right AND index TO the results of parse_tree with argument index PLUS one  
            END IF  
            IF index LESS THAN the length of s AND element at position index of s EQUALS the character closing parenthesis THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node and index  
        END FUNCTION  
        SET root AND _ TO the results of parse_tree with argument zero  
        RETURN root  
    END FUNCTION  
END CLASS
