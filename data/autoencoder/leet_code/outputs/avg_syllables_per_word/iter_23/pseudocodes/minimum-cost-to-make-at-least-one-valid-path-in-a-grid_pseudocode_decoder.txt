CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the list consisting of element zero and one element zero and negative one element one and zero element negative one and zero  
        SET costs TO a list of length m where each element is a list of length n with all elements set to positive infinity  
        SET element at position zero and zero of costs TO zero  
        
        SET priority queue pq TO the list containing a tuple of zero zero and zero  
        SET visited TO an empty set  
        
        WHILE the LENGTH OF pq is greater than zero  
            EXTRACT the smallest element from pq and ASSIGN its values to cost i and j  
            
            IF the pair of i and j is contained in visited  
                CONTINUE to the next iteration  
            END IF  
            ADD the pair of i and j to visited  
            
            IF i EQUALS m MINUS one AND j EQUALS n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each index k and pair di and dj IN the enumeration of directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                    SET new_cost TO cost PLUS one IF k PLUS one NOT EQUALS element at position i and j of grid OTHERWISE zero  
                    IF new_cost IS LESS THAN element at position ni and nj of costs  
                        SET element at position ni and nj of costs TO new_cost  
                        INSERT into pq the tuple of new_cost ni and nj maintaining the priority queue order  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
