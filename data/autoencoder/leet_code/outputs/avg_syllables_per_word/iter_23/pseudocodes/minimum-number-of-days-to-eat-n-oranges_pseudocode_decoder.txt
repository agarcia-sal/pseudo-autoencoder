CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS the MINIMUM OF the sum of the remainder of remaining DIVIDED BY two AND the result of dp WITH the quotient of remaining DIVIDED BY two AND the sum of the remainder of remaining DIVIDED BY three AND the result of dp WITH the quotient of remaining DIVIDED BY three  
        END FUNCTION  
        RETURN dp WITH PARAMETER n  
END CLASS
