CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        
        FOR index FROM zero TO the LENGTH OF wordsDict MINUS one  
            SET word TO element at position index of wordsDict  
            
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO index  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO index  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET difference TO prev_word1 MINUS prev_word2  
                IF difference LESS THAN zero  
                    SET difference TO negative one MULTIPLIED BY difference  
                END IF  
                IF difference LESS THAN min_distance  
                    SET min_distance TO difference  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
