CLASS Solution  
    FUNCTION areConnected(n threshold queries)  
        IF threshold EQUALS zero THEN  
            RETURN a list of True with the LENGTH OF queries  
        END IF  
        
        SET parent TO a list of integers FROM zero TO n PLUS one  
        
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x THEN  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY THEN  
                SET element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS one TO n PLUS one  
            FOR j FROM two TIMES i TO n PLUS one STEP i  
                CALL union(i j)  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        FOR each pair a b IN queries  
            APPEND the condition find(a) EQUALS find(b) TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
