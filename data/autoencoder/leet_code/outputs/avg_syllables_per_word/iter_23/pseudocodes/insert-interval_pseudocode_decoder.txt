CLASS Solution  
    FUNCTION insert(intervals PARAMETER, newInterval PARAMETER)  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN a list consisting of newInterval  
        END IF  
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        WHILE i LESS THAN n AND the second element at position i of intervals LESS THAN the first element of newInterval  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        WHILE i LESS THAN n AND the first element at position i of intervals LESS THAN OR EQUAL TO the second element of newInterval  
            SET the first element of newInterval TO the minimum of the first element of newInterval AND the first element at position i of intervals  
            SET the second element of newInterval TO the maximum of the second element of newInterval AND the second element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        APPEND newInterval TO merged  
        WHILE i LESS THAN n  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        RETURN merged  
    END FUNCTION  
END CLASS
