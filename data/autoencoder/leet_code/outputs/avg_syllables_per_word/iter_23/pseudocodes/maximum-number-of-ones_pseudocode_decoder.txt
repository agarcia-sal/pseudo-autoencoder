CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO a list of zero repeated sideLength MULTIPLIED BY sideLength times  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET row_position TO i MODULO sideLength  
                SET column_position TO j MODULO sideLength  
                SET pos TO row_position MULTIPLIED BY sideLength PLUS column_position  
                INCREMENT element at position pos of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN order from highest to lowest  
        SET result TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT result BY element at position index of freq  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
