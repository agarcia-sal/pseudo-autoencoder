CLASS Solution  
    FUNCTION minBuildTime(blocks WITH type list OF integers split WITH type integer) RETURNS integer  
        SET heap_structure TO the list blocks arranged AS a MINIMUM HEAP  
        WHILE the length OF blocks IS GREATER THAN one  
            SET first_element TO the SMALLEST element REMOVED FROM the heap_structure  
            SET second_element TO the SMALLEST element REMOVED FROM the heap_structure  
            SET combined_time TO the value split PLUS the value second_element  
            INSERT the value combined_time INTO the heap_structure MAINTAINING the MINIMUM HEAP PROPERTY  
        END WHILE  
        RETURN the element AT position one OF the heap_structure  
    END FUNCTION  
END CLASS
