CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum BETWEEN two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY x MINUS target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET result TO dp OF target MINUS x RAISED TO THE power PLUS power  
            IF x RAISED TO THE power PLUS one MINUS target LESS THAN target  
                SET result TO the minimum BETWEEN result AND dp OF x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN dp OF target  
    END FUNCTION  
END CLASS
