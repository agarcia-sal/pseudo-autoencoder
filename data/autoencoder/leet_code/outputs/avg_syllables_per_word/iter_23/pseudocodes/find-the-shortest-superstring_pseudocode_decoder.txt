CLASS Solution  
    FUNCTION shortestSuperstring WITH PARAMETER words  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap WITH PARAMETERS i AND j  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one BY minus one  
                IF the substring from position the LENGTH OF element at position i of words MINUS k PLUS one TO the LENGTH OF element at position i of words of element at position i of words EQUALS the substring from position one TO k of element at position j of words  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp WITH PARAMETERS mask AND i  
            IF mask EQUALS two RAISED TO THE POWER OF n MINUS one  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND of mask AND two RAISED TO THE POWER OF j EQUALS zero  
                    SET length AND path TO the RESULT OF dp WITH ARGUMENTS bitwise OR of mask AND two RAISED TO THE POWER OF j AND j  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS the RESULT OF overlap WITH ARGUMENTS i AND j  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the substring from position overlap WITH ARGUMENTS i AND j PLUS one TO the LENGTH OF element at position j of words of element at position j of words FOLLOWED BY path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length AND path TO the RESULT OF dp WITH ARGUMENTS two RAISED TO THE POWER OF i AND i  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i of words FOLLOWED BY path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
