CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the position to insert num into dp to keep dp sorted in ascending order using binary search  
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
