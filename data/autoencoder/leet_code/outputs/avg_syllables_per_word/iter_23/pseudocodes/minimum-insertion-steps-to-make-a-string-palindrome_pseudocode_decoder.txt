CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            CREATE dp AS a two dimensional array of size n by n initialised TO zero  
            FOR i FROM zero TO n MINUS one  
                SET element at position i and position i OF dp TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two  
                        SET element at position i and position j OF dp TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s  
                        SET element at position i and position j OF dp TO element at position i PLUS one and position j MINUS one OF dp PLUS two  
                    ELSE  
                        SET element at position i and position j OF dp TO the maximum BETWEEN element at position i and position j MINUS one OF dp AND element at position i PLUS one and position j OF dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero and position n MINUS one OF dp  
        END FUNCTION  
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
