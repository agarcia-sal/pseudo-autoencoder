CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS list of strings  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO the substring of num from position index TO position i MINUS one  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS the character zero  
                    CONTINUE TO next iteration  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, the integer value of operand, the integer value of operand  
                ELSE  
                    CALL dfs WITH arguments i, the concatenation of path AND the character plus AND operand, value PLUS the integer value of operand, the integer value of operand  
                    CALL dfs WITH arguments i, the concatenation of path AND the character minus AND operand, value MINUS the integer value of operand, the negative of the integer value of operand  
                    CALL dfs WITH arguments i, the concatenation of path AND the character multiplied by AND operand, value MINUS prev PLUS prev MULTIPLIED BY the integer value of operand, prev MULTIPLIED BY the integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH arguments zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
