CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adjacency list TO an empty dictionary  
        FOR each course FROM zero TO numCourses MINUS one  
            SET element at position course of adjacency list TO an empty list  
        END FOR  
        
        SET in degrees TO an empty list  
        FOR each index FROM zero TO numCourses MINUS one  
            APPEND zero TO in degrees  
        END FOR  
        
        FOR each pair of destination AND source IN prerequisites  
            APPEND destination TO element at position source of adjacency list  
            INCREMENT element at position destination of in degrees BY one  
        END FOR  
        
        SET queue TO an empty deque  
        FOR each course FROM zero TO numCourses MINUS one  
            IF element at position course of in degrees EQUALS zero  
                APPEND course TO queue  
            END IF  
        END FOR  
        
        SET topological order TO an empty list  
        
        WHILE queue IS NOT empty  
            REMOVE and RETURN the element at the front of queue AND ASSIGN TO current  
            APPEND current TO topological order  
            
            FOR each neighbor IN element at position current of adjacency list  
                DECREMENT element at position neighbor of in degrees BY one  
                IF element at position neighbor of in degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological order EQUALS numCourses  
            RETURN topological order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
