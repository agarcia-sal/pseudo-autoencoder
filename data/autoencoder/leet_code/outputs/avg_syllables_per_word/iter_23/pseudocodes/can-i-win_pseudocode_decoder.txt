CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  

        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  

        SET memo TO an empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers EXISTS IN memo  
                RETURN the value from memo for the pair of current_total AND used_numbers  
            END IF  

            FOR number FROM one TO maxChoosableInteger  
                IF used_numbers AND the bitwise representation with number NOT EQUALS true  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, used_numbers WITH bitwise addition of number)  
                        SET the value for the pair of current_total AND used_numbers IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  

            SET the value for the pair of current_total AND used_numbers IN memo TO false  
            RETURN false  
        END FUNCTION  

        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
