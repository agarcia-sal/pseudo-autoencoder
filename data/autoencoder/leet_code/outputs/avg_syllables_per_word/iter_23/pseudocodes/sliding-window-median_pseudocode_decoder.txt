CLASS Solution  
    FUNCTION medianSlidingWindow(nums k)  
        SET lo TO empty list  
        SET hi TO empty list  

        FUNCTION balance_heaps()  
            IF the LENGTH OF lo GREATER THAN the LENGTH OF hi PLUS one  
                REMOVE the smallest element FROM lo AND NEGATE IT  
                ADD the removed element TO hi  
            ELSE IF the LENGTH OF hi GREATER THAN the LENGTH OF lo  
                REMOVE the smallest element FROM hi  
                NEGATE the removed element  
                ADD the negated element TO lo  
            END IF  
        END FUNCTION  

        FUNCTION add_num(num)  
            IF lo is empty OR num LESS THAN OR EQUAL TO the NEGATION OF element at position zero of lo  
                NEGATE num  
                ADD the negated num TO lo  
            ELSE  
                ADD num TO hi  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION remove_num(num)  
            IF num LESS THAN OR EQUAL TO the NEGATION OF element at position zero of lo  
                REMOVE the NEGATION of num FROM lo  
                IF lo is not empty  
                    REORGANIZE lo TO SATISFY heap properties  
                END IF  
            ELSE  
                REMOVE num FROM hi  
                IF hi is not empty  
                    REORGANIZE hi TO SATISFY heap properties  
                END IF  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION get_median()  
            IF the LENGTH OF lo GREATER THAN the LENGTH OF hi  
                RETURN the NEGATION OF element at position zero of lo  
            ELSE  
                ADD the NEGATION OF element at position zero of lo TO element at position zero of hi  
                DIVIDE the sum BY two  
                RETURN the result  
            END IF  
        END FUNCTION  

        FOR i FROM zero TO k MINUS one  
            CALL add_num with element at position i of nums  
        END FOR  

        SET medians TO list containing the result of get_median()  

        FOR i FROM k TO the LENGTH OF nums MINUS one  
            CALL add_num with element at position i of nums  
            CALL remove_num with element at position i MINUS k of nums  
            APPEND the result of get_median() TO medians  
        END FOR  

        RETURN medians  
    END FUNCTION  
END CLASS
