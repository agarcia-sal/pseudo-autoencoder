CLASS Solution  
    FUNCTION stoneGameIII(stoneValue)  
        SET n TO the LENGTH OF stoneValue  
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        FOR i FROM n MINUS one TO zero STEP minus one  
            SET dp at position i TO the element at position i of stoneValue MINUS the element at position i PLUS one of dp  
            IF i PLUS two LESS THAN OR EQUAL TO n THEN  
                SET dp at position i TO the GREATER OF the element at position i of dp AND the SUM of the elements at positions i AND i PLUS one of stoneValue MINUS the element at position i PLUS two of dp  
            END IF  
            IF i PLUS three LESS THAN OR EQUAL TO n THEN  
                SET dp at position i TO the GREATER OF the element at position i of dp AND the SUM of the elements at positions i AND i PLUS one AND i PLUS two of stoneValue MINUS the element at position i PLUS three of dp  
            END IF  
        END FOR  
        IF the element at position zero of dp GREATER THAN zero THEN  
            RETURN Alice  
        ELSE IF the element at position zero of dp LESS THAN zero THEN  
            RETURN Bob  
        ELSE  
            RETURN Tie  
        END IF  
    END FUNCTION  
END CLASS
