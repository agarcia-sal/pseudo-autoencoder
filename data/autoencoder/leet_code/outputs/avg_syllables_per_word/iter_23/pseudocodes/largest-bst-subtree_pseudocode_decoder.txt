CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with the left attribute of p and the left attribute of q AND the result of is_same_tree called with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION largestBSTSubtree(self root)  
        FUNCTION dfs(node)  
            IF node EQUALS none  
                RETURN positive infinity AND negative infinity AND zero representing minimum maximum AND size respectively  
            END IF  
            SET left_min left_max left_size TO the result of dfs called with the left attribute of node  
            SET right_min right_max right_size TO the result of dfs called with the right attribute of node  
            IF left_max LESS THAN the val attribute of node AND the val attribute of node LESS THAN right_min  
                RETURN the minimum between the val attribute of node AND left_min the maximum between the val attribute of node AND right_max AND the sum of left_size right_size AND one representing minimum maximum AND size respectively  
            END IF  
            RETURN negative infinity AND positive infinity AND the maximum between left_size AND right_size representing minimum maximum AND size respectively  
        END FUNCTION  
        RETURN the third element of the tuple returned by dfs called with root  
    END FUNCTION  
END CLASS
