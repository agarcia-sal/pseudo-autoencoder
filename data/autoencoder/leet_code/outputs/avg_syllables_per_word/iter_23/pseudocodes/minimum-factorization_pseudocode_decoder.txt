CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO empty list  
        FOR digit FROM nine TO two DECREMENT BY one  
            WHILE num MODULO digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        SET result_string TO concatenation of string representations of elements in factors  
        SET result TO integer value of result_string  
        
        IF result GREATER THAN two POWER thirty-one MINUS one  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
