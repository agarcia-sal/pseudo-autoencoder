CLASS Solution  
    FUNCTION findNumOfValidWords(words AS list of strings, puzzles AS list of strings) RETURNS list of integers  
        FUNCTION to_mask(s AS string) RETURNS integer  
            SET mask TO zero  
            FOR each character IN s  
                SET mask TO bitwise OR of mask AND one LEFT SHIFTED BY the difference BETWEEN the ordinal value of character AND the ordinal value of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  
        
        SET word_count TO a mapping from integer to integer with default value of zero  
        
        FOR each word IN words  
            SET mask TO to_mask(word)  
            IF the NUMBER OF BITS SET IN mask IS LESS THAN OR EQUAL TO seven  
                INCREMENT the value associated with mask in word_count BY one  
            END IF  
        END FOR  
        
        SET result TO an empty list  
        
        FOR each puzzle IN puzzles  
            SET first_char_mask TO one LEFT SHIFTED BY the difference BETWEEN the ordinal value of the element at position one of puzzle AND the ordinal value of letter a  
            SET puzzle_mask TO to_mask(the substring of puzzle from position two TO the end)  
            SET count TO the value associated with first_char_mask in word_count  
            
            SET subset TO puzzle_mask  
            WHILE subset IS NOT EQUAL TO zero  
                SET count TO count PLUS the value associated with the bitwise OR of subset AND first_char_mask in word_count  
                SET subset TO the bitwise AND of the difference BETWEEN subset AND one AND puzzle_mask  
            END WHILE  
            
            APPEND count TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
