CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals.sort BY natural order of element at position two of interval AND in descending natural order of element at position one of interval  
        SET first TO element at position two of element at position one of intervals MINUS one  
        SET second TO element at position two of element at position one of intervals  
        SET count TO two  
        FOR each element with names start and end IN elements of intervals FROM position two TO end  
            IF start GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
