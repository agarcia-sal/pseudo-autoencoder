CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        CALL courses BEING SORTED ACCORDING TO the element at position two of each course  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each duration AND last_day IN courses  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                CALL PUSH the negative value of duration ONTO max_heap  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND the negative value of the element at position one of max_heap GREATER THAN duration  
                INCREMENT current_end_day BY duration PLUS the element REMOVED FROM max_heap THAT HAS the smallest value  
                CALL PUSH the negative value of duration ONTO max_heap  
            END IF  
        END FOR  
        RETURN the LENGTH OF max_heap  
    END FUNCTION  
END CLASS
