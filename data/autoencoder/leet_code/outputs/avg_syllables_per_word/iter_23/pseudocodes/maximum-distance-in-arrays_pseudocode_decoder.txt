CLASS Solution  
    FUNCTION maxDistance WITH parameter arrays  
        SET min_val TO element at position zero of element at position zero of arrays  
        SET max_val TO element at position zero of element at position equal to the length of element at position zero of arrays MINUS one of arrays  
        SET max_distance TO zero  
        FOR index i FROM one TO the length of arrays MINUS one  
            SET current_array TO element at position i of arrays  
            SET current_min TO element at position zero of current_array  
            SET current_max TO element at position zero of current_array PLUS the length of current_array MINUS one of current_array  
            SET max_distance TO the maximum value AMONG max_distance AND the absolute value of current_max MINUS min_val AND the absolute value of max_val MINUS current_min  
            SET min_val TO the minimum value AMONG min_val AND current_min  
            SET max_val TO the maximum value AMONG max_val AND current_max  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
