CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d and value y IN dp  
                SET key_for_taller_support TO d PLUS rod  
                IF key_for_taller_support IN new_dp  
                    SET new_dp_value TO new_dp at key_for_taller_support  
                ELSE  
                    SET new_dp_value TO zero  
                END IF  
                IF y GREATER THAN new_dp_value  
                    SET new_dp at key_for_taller_support TO y  
                END IF  
                SET absolute_difference TO d MINUS rod  
                IF absolute_difference LESS THAN zero  
                    SET absolute_difference TO rod MINUS d  
                END IF  
                SET value_for_shorter_support TO y PLUS the minimum of d and rod  
                IF absolute_difference IN new_dp  
                    SET new_dp_value2 TO new_dp at absolute_difference  
                ELSE  
                    SET new_dp_value2 TO zero  
                END IF  
                IF value_for_shorter_support GREATER THAN new_dp_value2  
                    SET new_dp at absolute_difference TO value_for_shorter_support  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value at key zero in dp  
    END FUNCTION  
END CLASS
