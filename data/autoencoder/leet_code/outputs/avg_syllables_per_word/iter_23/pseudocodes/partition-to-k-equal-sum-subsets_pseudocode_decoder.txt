CLASS Solution  
    FUNCTION canPartitionKSubsets(nums k)  
        SET total_sum TO the sum of all elements in nums  
        IF total_sum MODULO k NOT EQUALS zero  
            RETURN false  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums in descending order  
        
        FUNCTION can_partition(index k current_sum used)  
            IF k EQUALS zero  
                RETURN true  
            END IF  
            IF current_sum EQUALS target  
                RETURN the result of calling can_partition with zero as index k MINUS one as k zero as current_sum and used unchanged  
            END IF  
            
            FOR i FROM index TO the length of nums MINUS one  
                IF the element at position i of used EQUALS false AND current_sum PLUS the element at position i of nums LESS THAN OR EQUAL TO target  
                    SET the element at position i of used TO true  
                    IF the result of calling can_partition with i PLUS one as index k unchanged current_sum PLUS the element at position i of nums and used is true  
                        RETURN true  
                    END IF  
                    SET the element at position i of used TO false  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        SET used TO a list of length equal to the length of nums with all elements false  
        RETURN the result of calling can_partition with zero as index k unchanged zero as current_sum and used  
    END FUNCTION  
END CLASS
