CLASS UnionFind
    FUNCTION __init__(self, size)
        SET self.parent TO list of elements from zero TO size MINUS one
        SET self.rank TO list containing size elements all set to one
    END FUNCTION

    FUNCTION find(self, p)
        IF self.parent at index p NOT EQUALS p THEN
            SET self.parent at index p TO self.find of self.parent at index p
        END IF
        RETURN self.parent at index p
    END FUNCTION

    FUNCTION union(self, p, q)
        SET rootP TO self.find of p
        SET rootQ TO self.find of q
        IF rootP EQUALS rootQ THEN
            RETURN False
        END IF
        IF self.rank at index rootP GREATER THAN self.rank at index rootQ THEN
            SET self.parent at index rootQ TO rootP
        ELSE IF self.rank at index rootP LESS THAN self.rank at index rootQ THEN
            SET self.parent at index rootP TO rootQ
        ELSE
            SET self.parent at index rootQ TO rootP
            INCREMENT self.rank at index rootP BY one
        END IF
        RETURN True
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION findRedundantConnection(self, edges)
        SET uf TO new UnionFind initialized with length of edges PLUS one
        FOR each pair u, v IN edges
            IF uf.union of u and v EQUALS False THEN
                RETURN list containing u and v
            END IF
        END FOR
    END FUNCTION
END CLASS
