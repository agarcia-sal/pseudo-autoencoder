CLASS Solution  
    FUNCTION largestDivisibleSubset(self, nums)  
        CALL sort ON nums  
        SET n TO length of nums  
        SET f TO a list of length n with all elements SET TO 1  
        SET k TO 0  
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF nums at i MODULO nums at j EQUALS 0 THEN  
                    SET f at i TO the greater value BETWEEN f at i AND f at j PLUS 1  
                END IF  
            END FOR  
            IF f at k LESS THAN f at i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f at k  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m GREATER THAN 0  
            IF nums at k MODULO nums at i EQUALS 0 AND f at i EQUALS m THEN  
                APPEND nums at i TO ans  
                SET k TO i  
                SET m TO m MINUS 1  
            END IF  
            DECREMENT i BY 1  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
