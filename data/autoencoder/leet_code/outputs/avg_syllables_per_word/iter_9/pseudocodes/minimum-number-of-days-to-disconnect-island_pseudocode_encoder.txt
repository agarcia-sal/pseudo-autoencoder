CLASS Solution  
    FUNCTION minDays(self, grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN 0 OR x GREATER THAN OR EQUAL TO length of grid OR y LESS THAN 0 OR y GREATER THAN OR EQUAL TO length of grid at index zero OR pair of x and y IN visited OR grid at x and y EQUALS 0 THEN  
                RETURN  
            END IF  
            ADD pair of x and y TO visited  
            FOR each pair dx and dy IN list of pairs: negative one and zero, one and zero, zero and negative one, zero and one  
                CALL dfs with x plus dx, y plus dy, and visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO empty set  
            SET land_cells TO list of pairs i and j FOR each i FROM zero TO length of grid MINUS one FOR each j FROM zero TO length of grid at index zero MINUS one WHERE grid at i and j EQUALS one  
            IF land_cells IS empty THEN  
                RETURN True  
            END IF  
            CALL dfs with first element of land_cells at index zero, first element of land_cells at index one, and visited  
            RETURN length of visited EQUALS length of land_cells  
        END FUNCTION  

        IF is_connected() NOT True THEN  
            RETURN zero  
        END IF  

        FOR i FROM zero TO length of grid MINUS one  
            FOR j FROM zero TO length of grid at index zero MINUS one  
                IF grid at i and j EQUALS one THEN  
                    SET grid at i and j TO zero  
                    IF is_connected() NOT True THEN  
                        RETURN one  
                    END IF  
                    SET grid at i and j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
