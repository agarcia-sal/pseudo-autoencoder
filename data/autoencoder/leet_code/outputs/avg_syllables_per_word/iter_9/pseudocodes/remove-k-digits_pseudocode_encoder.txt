CLASS Solution  
    FUNCTION removeKdigits(self, num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND last element OF stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero THEN  
            SET final_stack TO elements of stack excluding the last k elements  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET result TO concatenation of elements in final_stack  
        REMOVE all leading zeros FROM result  
        
        IF result IS empty THEN  
            RETURN string "0"  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
