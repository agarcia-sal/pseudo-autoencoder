CLASS Solution  
    FUNCTION leastBricks(self, wall)  
        SET gap_counts TO a default dictionary with default value zero  
        
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row except the last element  
                SET total TO total PLUS brick  
                INCREMENT gap_counts at key total BY one  
            END FOR  
        END FOR  
        
        SET max_gaps TO the maximum value in gap_counts values, or zero if gap_counts is empty  
        
        RETURN the number of elements in wall MINUS max_gaps  
    END FUNCTION  
END CLASS
