CLASS Solution  
    FUNCTION productExceptSelf(self, nums)  
        SET n TO length of nums  
        SET answer TO list of size n filled with 1  
        
        SET left_product TO 1  
        FOR i FROM 0 TO n MINUS 1  
            SET answer at index i TO left_product  
            SET left_product TO left_product MULTIPLIED BY nums at index i  
        END FOR  
        
        SET right_product TO 1  
        FOR i FROM n MINUS 1 DOWN TO 0  
            SET answer at index i TO answer at index i MULTIPLIED BY right_product  
            SET right_product TO right_product MULTIPLIED BY nums at index i  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
