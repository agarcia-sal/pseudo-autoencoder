CLASS Solution  
    FUNCTION escapeGhosts(self, ghosts, target)  
        SET player_distance TO absolute value of target at index zero PLUS absolute value of target at index one  
        
        FOR each ghost IN ghosts  
            SET ghost_distance TO absolute value of ghost at index zero MINUS target at index zero PLUS absolute value of ghost at index one MINUS target at index one  
            IF ghost_distance LESS THAN OR EQUAL TO player_distance THEN  
                RETURN False  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
