CLASS Solution  
    FUNCTION isValidSerialization(self, preorder)  
        SET nodes TO the list obtained by splitting preorder by comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero THEN  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the symbol representing null THEN  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN whether slots EQUALS zero  
    END FUNCTION  
END CLASS
