CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as first element of values  
    SET i TO 1  
    INITIALIZE queue as empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val as values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val as values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  

        INITIALIZE node_dict as empty mapping from keys to list  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node IS None THEN  
                RETURN  
            END IF  
            APPEND tuple of row and node.val TO node_dict at key col  
            SET min_col TO minimum of min_col and col  
            SET max_col TO maximum of max_col and col  
            CALL dfs with node.left, row incremented by one, col decremented by one  
            CALL dfs with node.right, row incremented by one, col incremented by one  
        END FUNCTION  

        CALL dfs with root, zero, zero  

        INITIALIZE result as empty list  
        FOR col FROM min_col TO max_col DO  
            APPEND to result a list constructed from values val extracted from node_dict at col sorted by row then val  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
