CLASS Solution  
    FUNCTION findShortestWay(self, maze, ball, hole)  
        SET directions TO list containing (zero, one, 'r'), (one, zero, 'd'), (zero, minus one, 'l'), (minus one, zero, 'u')  
        SET m TO length of maze  
        SET n TO length of first element in maze  
        SET start TO tuple formed from ball  
        SET hole TO tuple formed from hole  
        SET priority queue pq TO list containing tuple (zero, empty string, start)  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            SET dist, path, position (x, y) TO remove smallest element from pq  
            
            IF position (x, y) EQUALS hole THEN  
                RETURN path  
            END IF  
            
            IF position (x, y) IS IN visited THEN  
                CONTINUE to next iteration  
            END IF  
            
            ADD position (x, y) TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx GREATER THAN OR EQUAL TO zero AND new_x plus dx LESS THAN m AND new_y plus dy GREATER THAN OR EQUAL TO zero AND new_y plus dy LESS THAN n AND maze element at position (new_x plus dx, new_y plus dy) EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    
                    IF position (new_x, new_y) EQUALS hole THEN  
                        BREAK out of loop  
                    END IF  
                END WHILE  
                
                IF position (new_x, new_y) IS NOT IN visited THEN  
                    ADD tuple (dist plus count, concatenation of path and direction, position (new_x, new_y)) TO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN string "impossible"  
    END FUNCTION  
END CLASS
