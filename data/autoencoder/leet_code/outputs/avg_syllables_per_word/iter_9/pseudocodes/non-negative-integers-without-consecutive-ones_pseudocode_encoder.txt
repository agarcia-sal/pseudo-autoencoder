CLASS Solution  
    FUNCTION findIntegers(self, n)  
        SET bin_rep TO binary representation of n starting from second character  
        SET length TO length of bin_rep  
        
        INITIALIZE dp as a list of zeros with size length plus one  
        SET dp at index zero TO one  
        SET dp at index one TO two  
        
        FOR index FROM two TO length  
            SET dp at index TO dp at index minus one PLUS dp at index minus two  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR index FROM zero TO length minus one  
            IF character at index of bin_rep EQUALS '1' THEN  
                INCREMENT result BY dp at position length MINUS index MINUS one  
                IF prev_bit EQUALS one THEN  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            
            SET prev_bit TO one IF character at index of bin_rep EQUALS '1' ELSE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
