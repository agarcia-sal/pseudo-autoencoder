CLASS Solution  
    FUNCTION networkDelayTime(self, times, n, k)  
        SET graph TO a new mapping from each integer from 1 TO n mapped to an empty list  
        FOR each element u, v, w IN times  
            APPEND the pair (v, w) TO graph at key u  
        END FOR  

        SET dist TO a new mapping from each integer from 1 TO n mapped to positive infinity  
        SET dist at key k TO zero  

        SET min_heap TO a new list containing a single pair of zero and k  

        WHILE min_heap is not empty  
            REMOVE from min_heap the pair with the smallest first element and assign to current_dist, node  

            IF current_dist GREATER THAN dist at key node THEN  
                CONTINUE to next iteration of the WHILE loop  
            END IF  

            FOR each pair neighbor, weight IN graph at key node  
                SET distance TO current_dist PLUS weight  

                IF distance LESS THAN dist at key neighbor THEN  
                    SET dist at key neighbor TO distance  
                    ADD the pair of distance and neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  

        SET max_time TO the maximum value among all values in dist  

        IF max_time EQUALS positive infinity THEN  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
