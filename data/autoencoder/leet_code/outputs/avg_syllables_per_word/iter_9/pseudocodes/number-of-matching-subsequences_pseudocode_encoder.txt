CLASS Solution  
    FUNCTION numMatchingSubseq(self, s, words)  
        SET waiting TO a dictionary mapping each unique character in s TO an empty double-ended queue  
        
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element from it  
            IF first_letter EXISTS in waiting THEN  
                APPEND it TO the queue at waiting for first_letter  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN s  
            IF c EXISTS in waiting THEN  
                SET current_iterators TO the queue at waiting for c  
                SET the queue at waiting for c TO an empty double-ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next element from it OR none if exhausted  
                    IF next_char IS none THEN  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char EXISTS in waiting THEN  
                            APPEND it TO the queue at waiting for next_char  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
