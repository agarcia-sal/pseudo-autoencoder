CLASS Solution  
    FUNCTION convert(self, s, numRows)  
        IF numRows EQUALS 1 THEN  
            RETURN s  
        END IF  
        
        SET rows TO a list of empty strings with length numRows  
        SET current_row TO 0  
        SET going_down TO false  
        
        FOR each char IN s  
            APPEND char TO the element in rows at position current_row  
            IF current_row EQUALS 0 OR current_row EQUALS numRows MINUS 1 THEN  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down IS true THEN  
                INCREMENT current_row BY 1  
            ELSE  
                DECREMENT current_row BY 1  
            END IF  
        END FOR  
        
        RETURN the concatenation of all elements in rows  
    END FUNCTION  
END CLASS
