CLASS Solution  
    FUNCTION canJump(self, nums)  
        SET farthest TO zero  
        FOR i FROM zero TO length of nums MINUS one  
            IF i GREATER THAN farthest THEN  
                RETURN False  
            END IF  
            SET farthest TO the greater value BETWEEN farthest AND the sum of i AND the element at index i in nums  
        END FOR  
        RETURN farthest GREATER THAN OR EQUAL TO length of nums MINUS one  
    END FUNCTION  
END CLASS
