CLASS Solution  
    FUNCTION removeComments(self, source)  
        SET in_block_comment TO false  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN length of line  
                IF in_block_comment THEN  
                    IF substring of line from i to i PLUS two EQUALS "*/" THEN  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF substring of line from i to i PLUS two EQUALS "//" THEN  
                        BREAK from while loop  
                    ELSE IF substring of line from i to i PLUS two EQUALS "/*" THEN  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND character at line index i TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line NOT empty AND in_block_comment IS false THEN  
                APPEND concatenation of characters in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
