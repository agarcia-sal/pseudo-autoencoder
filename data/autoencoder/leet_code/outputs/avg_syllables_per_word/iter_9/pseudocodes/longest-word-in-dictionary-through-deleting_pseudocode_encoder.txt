CLASS Solution  
    FUNCTION findLongestWord(self, s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO iterator over s  
            RETURN whether all characters in word appear in order within iterator  
        END FUNCTION  
        
        SORT dictionary by descending length of each element AND ascending lexicographical order of each element  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) IS true THEN  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
