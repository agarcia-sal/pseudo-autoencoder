CLASS Solution
    FUNCTION judgeSquareSum(self, c)
        SET left TO 0
        SET right TO the integer part of the square root of c
        WHILE left less than or equal to right
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right
            IF current_sum EQUALS c THEN
                RETURN True
            ELSE IF current_sum LESS THAN c THEN
                INCREMENT left BY 1
            ELSE
                DECREMENT right BY 1
            END IF
        END WHILE
        RETURN False
    END FUNCTION
END CLASS
