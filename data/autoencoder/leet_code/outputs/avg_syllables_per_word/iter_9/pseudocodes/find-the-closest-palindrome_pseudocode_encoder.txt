CLASS Solution  
    FUNCTION nearestPalindromic(self, n)  
        SET length TO the number of characters in n  
        IF length EQUALS 1 THEN  
            RETURN the string form of the integer form of n MINUS one  
        END IF  
        
        SET candidates TO an empty set  
        
        APPEND the number formed by TEN RAISED TO THE POWER length PLUS one TO candidates  
        APPEND the number formed by TEN RAISED TO THE POWER length MINUS one MINUS one TO candidates  
        
        SET prefix TO the integer value of the substring of n from start to the integer division of (length PLUS one) BY two  
        
        FOR i FROM negative one TO one  
            SET new_prefix TO the string form of (prefix PLUS i)  
            IF length MODULO two EQUALS zero THEN  
                SET candidate TO new_prefix CONCATENATED WITH the reverse of new_prefix  
            ELSE  
                SET candidate TO new_prefix CONCATENATED WITH the reverse of new_prefix excluding its last character  
            END IF  
            APPEND the integer form of candidate TO candidates  
        END FOR  
        
        SET original_number TO the integer form of n  
        REMOVE original_number FROM candidates IF PRESENT  
        
        SET closest TO the element in candidates with the minimum value according to the tuple of (the absolute value of element MINUS original_number, element)  
        
        RETURN the string form of closest  
    END FUNCTION  
END CLASS
