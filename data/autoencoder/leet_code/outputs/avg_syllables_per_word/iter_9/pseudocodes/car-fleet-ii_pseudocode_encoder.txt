CLASS Solution  
    FUNCTION getCollisionTimes(self, cars)  
        SET n TO length of cars  
        SET result TO list of length n with all elements set to negative one point zero  
        SET stack TO empty list  

        FOR i FROM n MINUS one DOWNTO zero  
            SET pos TO first element of cars at index i  
            SET speed TO second element of cars at index i  
            
            WHILE stack is not empty  
                SET next_index TO last element of stack  
                SET next_pos TO first element of cars at index next_index  
                SET next_speed TO second element of cars at index next_index  
                
                IF speed less than or equals next_speed THEN  
                    REMOVE last element FROM stack  
                    CONTINUE  
                END IF  
                
                SET collision_time TO difference of next_pos and pos DIVIDED BY difference of speed and next_speed  
                
                IF element at index next_index in result greater than zero AND collision_time greater than or equals element at index next_index in result THEN  
                    REMOVE last element FROM stack  
                    CONTINUE  
                END IF  
                
                SET element at index i in result TO collision_time  
                BREAK  
            END WHILE  
            
            APPEND i TO stack  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
