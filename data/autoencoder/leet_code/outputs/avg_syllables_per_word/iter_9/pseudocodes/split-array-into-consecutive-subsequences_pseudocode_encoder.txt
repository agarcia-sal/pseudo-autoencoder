CLASS Solution  
    FUNCTION isPossible(self, nums)  
        SET num_count TO frequency count of elements in nums  
        SET end_with TO frequency count of subsequences ending with a number  

        FOR each num IN nums  
            IF num_count at num EQUALS zero THEN  
                CONTINUE to next iteration  
            END IF  

            IF end_with at (num MINUS one) GREATER THAN zero THEN  
                DECREMENT end_with at (num MINUS one) BY one  
                INCREMENT end_with at num BY one  
            ELSE IF num_count at (num PLUS one) GREATER THAN zero AND num_count at (num PLUS two) GREATER THAN zero THEN  
                DECREMENT num_count at (num PLUS one) BY one  
                DECREMENT num_count at (num PLUS two) BY one  
                INCREMENT end_with at (num PLUS two) BY one  
            ELSE  
                RETURN False  
            END IF  

            DECREMENT num_count at num BY one  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
