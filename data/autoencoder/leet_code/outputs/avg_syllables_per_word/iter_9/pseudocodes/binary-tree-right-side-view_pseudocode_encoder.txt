CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the first element of values  
    SET i TO 1  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO the first element removed from queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO a new TreeNode initialized with values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO a new TreeNode initialized with values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS None THEN  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a deque initialized with root  
        WHILE queue IS NOT empty DO  
            SET level_length TO length of queue  
            FOR i FROM 0 TO level_length MINUS 1 DO  
                SET node TO the first element removed from queue  
                IF i EQUALS level_length MINUS 1 THEN  
                    APPEND node.val TO right_side_view  
                END IF  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
