CLASS Solution  
    FUNCTION numTilings(self, n)  
        SET MOD TO one billion plus seven  
        
        INITIALIZE dp AS list of zeros with length n plus one  
        INITIALIZE dp1 AS list of zeros with length n plus one  
        
        SET dp at position zero TO one  
        SET dp at position one TO one  
        
        IF n GREATER THAN OR EQUALS two THEN  
            SET dp at position two TO two  
            SET dp1 at position two TO one  
        END IF  
        
        FOR i FROM three TO n DO  
            SET dp at position i TO remainder of (dp at position i minus one PLUS dp at position i minus two PLUS two MULTIPLIED BY dp1 at position i minus one) DIVIDED BY MOD  
            SET dp1 at position i TO remainder of (dp at position i minus two PLUS dp1 at position i minus one) DIVIDED BY MOD  
        END FOR  
        
        RETURN dp at position n  
    END FUNCTION  
END CLASS
