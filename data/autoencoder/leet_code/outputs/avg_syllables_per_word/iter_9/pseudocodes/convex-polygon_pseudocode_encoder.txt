CLASS Solution  
    FUNCTION isConvex(self, points)  
        FUNCTION cross_product(o, a, b)  
            RETURN (value at index zero of a MINUS value at index zero of o) MULTIPLIED BY (value at index one of b MINUS value at index one of o) MINUS (value at index one of a MINUS value at index one of o) MULTIPLIED BY (value at index zero of b MINUS value at index zero of o)  
        END FUNCTION  
      
        SET n TO length of points  
        IF n LESS THAN OR EQUAL TO three THEN  
            RETURN True  
        END IF  
      
        SET orientation TO zero  
      
        FOR i FROM zero TO n MINUS one  
            SET o TO element at index i of points  
            SET a TO element at index remainder of i plus one divided by n of points  
            SET b TO element at index remainder of i plus two divided by n of points  
            SET cross TO cross_product(o, a, b)  
          
            IF cross NOT EQUALS zero THEN  
                IF orientation EQUALS zero THEN  
                    IF cross GREATER THAN zero THEN  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF (orientation EQUALS one AND cross LESS THAN zero) OR (orientation EQUALS negative one AND cross GREATER THAN zero) THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
      
        RETURN True  
    END FUNCTION  
END CLASS
