CLASS Solution  
    FUNCTION findMaximizedCapital(self, k, w, profits, capital)  
        SET projects TO list of pairs from combining capital and profits  
        SORT projects by the capital value in each pair  
        
        SET max_heap TO empty list  
        
        SET i TO zero  
        SET n TO length of projects  
        
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND capital value of projects at index i LESS THAN OR EQUAL TO w  
                ADD negative of profit value of projects at index i TO max_heap  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS empty THEN  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS negative of removing the top element from max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
