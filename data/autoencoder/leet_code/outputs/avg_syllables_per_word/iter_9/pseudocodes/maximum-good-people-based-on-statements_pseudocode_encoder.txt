CLASS Solution  
    FUNCTION maximumGood(self, statements)  
        SET n TO length of statements  
        SET max_good TO zero  
          
        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO list of n elements all set to False  
            SET bad TO list of n elements all set to False  
              
            FOR j FROM zero TO n MINUS one  
                IF i BITWISE AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero THEN  
                    SET good at index j TO True  
                ELSE  
                    SET bad at index j TO True  
                END IF  
            END FOR  
              
            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF good at index j IS True THEN  
                    FOR k FROM zero TO n MINUS one  
                        IF (statements at index j and k EQUALS zero AND good at index k IS True) OR (statements at index j and k EQUALS one AND bad at index k IS True) THEN  
                            SET valid TO False  
                            BREAK inner FOR  
                        END IF  
                    END FOR  
                    IF valid IS False THEN  
                        BREAK outer FOR  
                    END IF  
                END IF  
            END FOR  
              
            IF valid IS True THEN  
                SET max_good TO maximum OF max_good AND sum of good list where True counts as one  
            END IF  
        END FOR  
          
        RETURN max_good  
    END FUNCTION  
END CLASS
