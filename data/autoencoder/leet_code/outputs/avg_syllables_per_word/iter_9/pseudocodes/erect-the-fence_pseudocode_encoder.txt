CLASS Solution  
    FUNCTION outerTrees(self, trees)  
        FUNCTION cross(i, j, k)  
            SET a TO trees at position i  
            SET b TO trees at position j  
            SET c TO trees at position k  
            RETURN (b at first coordinate MINUS a at first coordinate) MULTIPLIED BY (c at second coordinate MINUS b at second coordinate) MINUS (b at second coordinate MINUS a at second coordinate) MULTIPLIED BY (c at first coordinate MINUS b at first coordinate)  
        END FUNCTION  

        SET n TO length of trees  
        IF n LESS THAN 4 THEN  
            RETURN trees  
        END IF  
        SORT trees in ascending order  
        SET vis TO list of False repeated n times  
        SET stk TO list containing zero  
        FOR i FROM one TO n MINUS one  
            WHILE length of stk GREATER THAN one AND cross(element at second last index in stk, element at last index in stk, i) LESS THAN zero  
                SET vis at position popped last element FROM stk TO False  
            END WHILE  
            SET vis at position i TO True  
            APPEND i TO stk  
        END FOR  
        SET m TO length of stk  
        FOR i FROM n MINUS two DOWN TO zero  
            IF vis at position i IS True THEN  
                CONTINUE to next iteration  
            END IF  
            WHILE length of stk GREATER THAN m AND cross(element at second last index in stk, element at last index in stk, i) LESS THAN zero  
                POP last element FROM stk  
            END WHILE  
            APPEND i TO stk  
        END FOR  
        POP last element FROM stk  
        RETURN list of trees elements at indices in stk  
    END FUNCTION  
END CLASS
