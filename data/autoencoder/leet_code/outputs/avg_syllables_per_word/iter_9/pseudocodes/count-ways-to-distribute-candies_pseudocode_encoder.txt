CLASS Solution  
    FUNCTION waysToDistribute(self, n, k)  
        SET MOD TO one billion PLUS seven  
        
        INITIALIZE dp AS two dimensional list with (n PLUS one) rows AND (k PLUS one) columns FILLED WITH zero  
        
        SET dp at position zero zero TO one  
        
        FOR i FROM one TO n DO  
            FOR j FROM one TO k DO  
                SET dp at position i j TO j MULTIPLIED BY dp at position i MINUS one j PLUS dp at position i MINUS one j MINUS one  
                SET dp at position i j TO dp at position i j MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN dp at position n k  
    END FUNCTION  
END CLASS
