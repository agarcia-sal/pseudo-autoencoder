CLASS Solution  
    FUNCTION findMinDifference(self, timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO first part of time split by colon converted to integer  
            SET minutes TO second part of time split by colon converted to integer  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  

        SET minutes_list TO result of applying to_minutes to each element in timePoints  
        SORT minutes_list in ascending order  

        SET min_diff TO infinity  
        FOR index FROM one TO length of minutes_list MINUS one  
            SET diff TO value at minutes_list at index MINUS value at minutes_list at (index MINUS one)  
            IF diff LESS THAN min_diff THEN  
                SET min_diff TO diff  
            END IF  
        END FOR  

        SET circular_diff TO result of subtracting value at minutes_list at last index FROM one thousand four hundred forty PLUS value at minutes_list at first index  
        IF circular_diff LESS THAN min_diff THEN  
            SET min_diff TO circular_diff  
        END IF  

        RETURN min_diff  
    END FUNCTION  
END CLASS
