CLASS Solution  
    FUNCTION uniquePaths(self, m, n)  
        CREATE two_dimensional_array dp WITH dimensions m BY n FILLED WITH value one  
        FOR index i FROM one TO m MINUS one  
            FOR index j FROM one TO n MINUS one  
                SET dp at position i, j TO dp at position i MINUS one, j PLUS dp at position i, j MINUS one  
            END FOR  
        END FOR  
        RETURN dp at position m MINUS one, n MINUS one  
    END FUNCTION  
END CLASS
