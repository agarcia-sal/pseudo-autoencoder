CLASS Solution  
    FUNCTION isMatch(self, s, p)  
        SET dp TO a two-dimensional list of False values with dimensions (length of s PLUS one) by (length of p PLUS one)  
        
        SET dp at position zero zero TO True  
        
        FOR each index j FROM one TO length of p  
            IF character at position j MINUS one in p EQUALS the character representing star THEN  
                SET dp at position zero j TO dp at position zero j MINUS one  
            END IF  
        END FOR  
        
        FOR each index i FROM one TO length of s  
            FOR each index j FROM one TO length of p  
                IF character at position j MINUS one in p EQUALS star THEN  
                    SET dp at position i j TO dp at position i MINUS one j OR dp at position i j MINUS one  
                ELSE IF character at position j MINUS one in p EQUALS question mark OR character at position i MINUS one in s EQUALS character at position j MINUS one in p THEN  
                    SET dp at position i j TO dp at position i MINUS one j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position length of s length of p  
    END FUNCTION  
END CLASS
