CLASS Solution  
    FUNCTION minimumTotal(self, triangle)  
        FOR row FROM length of triangle MINUS two DOWN TO zero  
            FOR col FROM zero TO length of triangle at row MINUS one  
                SET triangle at row and col TO triangle at row and col PLUS minimum of triangle at row plus one and col, triangle at row plus one and col plus one  
            END FOR  
        END FOR  
        RETURN triangle at zero and zero  
    END FUNCTION  
END CLASS
