CLASS Solution  
    FUNCTION makeLargestSpecial(self, s)  
        FUNCTION dfs(s)  
            SET count TO 0  
            SET start TO 0  
            SET specials TO empty list  
            
            FOR each index, character IN s  
                IF character EQUALS '1' THEN  
                    INCREMENT count BY 1  
                ELSE  
                    DECREMENT count BY 1  
                END IF  
                
                IF count EQUALS 0 THEN  
                    APPEND concatenation of '1', dfs(substring of s from start plus one to index exclusive), and '0' TO specials  
                    SET start TO index plus one  
                END IF  
            END FOR  
            
            SORT specials in reverse lexicographical order  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        
        RETURN dfs(s)  
    END FUNCTION  
END CLASS
