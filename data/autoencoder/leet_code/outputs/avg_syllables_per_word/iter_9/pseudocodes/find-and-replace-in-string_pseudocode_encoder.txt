CLASS Solution  
    FUNCTION findReplaceString(self, s, indices, sources, targets)  
        SET replacements TO a list of tuples formed by combining indices, sources, and targets  
        SORT replacements by the first element of each tuple (index) in ascending order  
        
        SET result TO an empty list  
        SET prev_end TO zero  
        
        FOR each tuple index, source, target IN replacements  
            APPEND the substring of s from prev_end to index TO result  
            
            IF the substring of s from index to index PLUS the length of source EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from index to index PLUS the length of source TO result  
            END IF  
            
            SET prev_end TO index PLUS the length of source  
        END FOR  
        
        APPEND the substring of s from prev_end to the end TO result  
        
        RETURN the concatenation of all strings in result  
    END FUNCTION  
END CLASS
