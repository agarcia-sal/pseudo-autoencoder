CLASS Solution  
    FUNCTION checkRecord(self, n)  
        SET MOD TO one billion PLUS seven  
        
        DEFINE FUNCTION initialize_dp(size)  
            SET dp TO empty list  
            FOR index FROM zero TO size DO  
                SET level_one TO empty list  
                FOR count_A FROM zero TO one DO  
                    SET level_two TO empty list  
                    FOR count_L FROM zero TO two DO  
                        APPEND zero TO level_two  
                    END FOR  
                    APPEND level_two TO level_one  
                END FOR  
                APPEND level_one TO dp  
            END FOR  
            RETURN dp  
        END FUNCTION  
        
        SET dp TO initialize_dp(n PLUS one)  
        
        SET dp zero at zero at zero TO one  
        
        FOR i FROM one TO n DO  
            FOR j FROM zero TO one DO  
                FOR k FROM zero TO two DO  
                    SET dp i at j at zero TO dp i at j at zero PLUS dp (i MINUS one) at j at k  
                    IF dp i at j at zero GREATER THAN OR EQUAL TO MOD THEN  
                        SET dp i at j at zero TO dp i at j at zero MINUS MOD  
                    END IF  
                    
                    IF j EQUALS zero THEN  
                        SET dp i at one at zero TO dp i at one at zero PLUS dp (i MINUS one) at zero at k  
                        IF dp i at one at zero GREATER THAN OR EQUAL TO MOD THEN  
                            SET dp i at one at zero TO dp i at one at zero MINUS MOD  
                        END IF  
                    END IF  
                    
                    IF k LESS THAN two THEN  
                        SET dp i at j at k PLUS one TO dp i at j at k PLUS one PLUS dp (i MINUS one) at j at k  
                        IF dp i at j at k PLUS one GREATER THAN OR EQUAL TO MOD THEN  
                            SET dp i at j at k PLUS one TO dp i at j at k PLUS one MINUS MOD  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM zero TO one DO  
            FOR k FROM zero TO two DO  
                SET result TO result PLUS dp n at j at k  
                IF result GREATER THAN OR EQUAL TO MOD THEN  
                    SET result TO result MINUS MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
