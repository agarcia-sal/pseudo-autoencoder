CLASS Solution  
    FUNCTION isNStraightHand(self, hand, groupSize)  
        IF the length of hand modulo groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        SET count TO the frequency count of elements in hand  
        SET unique_cards TO the sorted list of keys in count  
        
        FOR each card IN unique_cards  
            IF count at card GREATER THAN zero THEN  
                SET needed TO count at card  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF count at card PLUS index LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT count at card PLUS index BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
