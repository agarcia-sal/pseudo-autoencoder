CLASS Solution  
    FUNCTION triangleNumber(self, nums)  
        CALL method TO SORT nums IN ASCENDING ORDER  
        SET count TO zero  
        SET n TO LENGTH OF nums  
        
        FOR k FROM n MINUS one DOWN TO two STEP minus one  
            SET i TO zero  
            SET j TO k MINUS one  
            WHILE i LESS THAN j  
                IF VALUE OF nums AT i PLUS VALUE OF nums AT j GREATER THAN VALUE OF nums AT k THEN  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY one  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
