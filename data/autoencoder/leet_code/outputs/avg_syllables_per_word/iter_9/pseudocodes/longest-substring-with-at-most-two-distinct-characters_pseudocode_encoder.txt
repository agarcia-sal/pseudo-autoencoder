CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(self, s)  
        DEFINE FUNCTIONALITY TO CREATE A DICTIONARY THAT RETURNS ZERO FOR MISSING KEYS AND ASSIGN TO char_count  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO LENGTH OF s MINUS one  
            INCREMENT char_count VALUE FOR CHARACTER AT POSITION right IN s BY one  
            
            WHILE COUNT OF DISTINCT KEYS IN char_count GREATER THAN two  
                DECREMENT char_count VALUE FOR CHARACTER AT POSITION left IN s BY one  
                IF char_count VALUE FOR CHARACTER AT POSITION left IN s EQUALS zero THEN  
                    REMOVE ENTRY FOR CHARACTER AT POSITION left IN s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO GREATER OF max_length AND (right MINUS left PLUS one)  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
