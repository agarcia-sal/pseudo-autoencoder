CLASS Solution  
    FUNCTION racecar(self, target)  
        INITIALIZE queue WITH a double-ended queue containing a tuple of zero position, speed one, and zero steps  
        INITIALIZE visited WITH a set containing a tuple of zero position and speed one  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET position, speed, steps FROM that element  
            
            IF position EQUALS target THEN  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF tuple of next_position and next_speed NOT IN visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target THEN  
                ADD tuple of next_position and next_speed TO visited  
                APPEND tuple of next_position, next_speed, and steps PLUS one TO queue  
            END IF  
            
            IF speed GREATER THAN zero THEN  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            
            IF tuple of position and next_speed NOT IN visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target THEN  
                ADD tuple of position and next_speed TO visited  
                APPEND tuple of position, next_speed, and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
