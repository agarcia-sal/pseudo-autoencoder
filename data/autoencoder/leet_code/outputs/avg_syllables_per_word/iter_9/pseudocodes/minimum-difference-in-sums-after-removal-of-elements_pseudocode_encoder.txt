CLASS Solution  
    FUNCTION minimumDifference(self, nums)  
        SET n TO the length of nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with length equal to two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            ADD the negation of nums at index i TO min_heap using heap addition  
            IF the length of min_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM min_heap using heap removal  
            END IF  
            IF the length of min_heap EQUALS n THEN  
                SET min_sums at index i TO the negation of the sum of elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with length equal to two MULTIPLIED BY n PLUS one  
        
        FOR i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            ADD nums at index i TO max_heap using heap addition  
            IF the length of max_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM max_heap using heap removal  
            END IF  
            IF the length of max_heap EQUALS n THEN  
                SET max_sums at index i TO the sum of elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET min_diff TO the smaller value BETWEEN min_diff AND min_sums at index i MINUS max_sums at index i PLUS one  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
