CLASS Solution  
    FUNCTION diffWaysToCompute(self, expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS plus sign THEN  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus sign THEN  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS multiplication sign THEN  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
          
        FUNCTION helper(sub_expr)  
            IF sub_expr IS composed only of digits THEN  
                RETURN list containing the integer value of sub_expr  
            END IF  
              
            SET results TO empty list  
            FOR index FROM zero TO length of sub_expr MINUS one  
                IF character at sub_expr at index IS one of plus sign, minus sign, multiplication sign THEN  
                    SET left_results TO helper(substring of sub_expr from start TO index MINUS one)  
                    SET right_results TO helper(substring of sub_expr from index PLUS one TO end)  
                    EXTEND results BY compute(left_results, right_results, character at sub_expr at index)  
                END IF  
            END FOR  
              
            RETURN results  
        END FUNCTION  
          
        RETURN helper(expression)  
    END FUNCTION  
END CLASS
