CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val SET TO first element of values  
    SET i TO one  
    CREATE empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND element at i of values IS NOT None THEN  
            SET node.left TO new TreeNode with val SET TO element at i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at i of values IS NOT None THEN  
            SET node.right TO new TreeNode with val SET TO element at i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left and q.left AND is_same_tree of p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        CREATE empty list result  
        CREATE queue with root as initial element  
        SET left_to_right TO True  
        WHILE queue IS NOT empty DO  
            SET level_size TO length of queue  
            CREATE empty deque current_level  
            FOR each index FROM zero TO level_size MINUS one DO  
                SET node TO first element removed FROM queue  
                IF left_to_right THEN  
                    APPEND node.val TO current_level  
                ELSE  
                    APPEND node.val TO the front of current_level  
                END IF  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND list converted from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
