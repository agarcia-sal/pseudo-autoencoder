CLASS Solution  
    FUNCTION minimumTime(self, s)  
        SET n TO length of s  
        SET left_costs TO list of size n with all elements zero  
        SET right_costs TO list of size n with all elements zero  

        IF character at position zero in s EQUALS '1' THEN  
            SET left_costs at position zero TO one  
        ELSE  
            SET left_costs at position zero TO zero  
        END IF  

        FOR i FROM one TO n MINUS one  
            IF character at position i in s EQUALS '0' THEN  
                SET left_costs at position i TO left_costs at position i MINUS one  
            ELSE  
                SET left_costs at position i TO minimum of (i PLUS one) AND (left_costs at position i MINUS one PLUS two)  
            END IF  
        END FOR  

        IF character at last position in s EQUALS '1' THEN  
            SET right_costs at last position TO one  
        ELSE  
            SET right_costs at last position TO zero  
        END IF  

        FOR i FROM n MINUS two DOWN TO zero  
            IF character at position i in s EQUALS '0' THEN  
                SET right_costs at position i TO right_costs at position i PLUS one  
            ELSE  
                SET right_costs at position i TO minimum of (n MINUS i) AND (right_costs at position i PLUS one PLUS two)  
            END IF  
        END FOR  

        SET min_time TO minimum of left_costs at last position AND right_costs at position zero  

        FOR i FROM zero TO n MINUS two  
            SET min_time TO minimum of min_time AND (left_costs at position i PLUS right_costs at position i PLUS one)  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
