CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(self, s, k)  
        IF k EQUALS 0 THEN  
            RETURN 0  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO 0  
        SET max_length TO 0  
        
        FOR right FROM 0 TO length of s MINUS 1  
            SET character AT s at position right IN char_map TO right  
            
            IF size of char_map GREATER THAN k THEN  
                SET leftmost_char TO key in char_map WITH minimum value  
                SET left TO value associated with leftmost_char IN char_map PLUS 1  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO maximum of max_length AND right MINUS left PLUS 1  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
