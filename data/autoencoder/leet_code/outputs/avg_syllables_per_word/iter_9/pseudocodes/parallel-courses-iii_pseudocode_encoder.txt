CLASS Solution  
    FUNCTION minimumTime(self, n, relations, time)  
        SET graph TO an empty map with default value as empty list  
        FOR each pair prevCourse, nextCourse IN relations  
            APPEND prevCourse TO graph at key nextCourse  
        END FOR  

        SET memo TO a list of length n plus one filled with negative one  

        FUNCTION dfs(course)  
            IF memo at index course NOT EQUALS negative one THEN  
                RETURN memo at index course  
            END IF  

            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN graph at key course  
                SET candidate_time TO dfs(prevCourse)  
                IF candidate_time GREATER THAN max_prerequisite_time THEN  
                    SET max_prerequisite_time TO candidate_time  
                END IF  
            END FOR  

            SET memo at index course TO max_prerequisite_time PLUS time at index course MINUS one  
            RETURN memo at index course  
        END FUNCTION  

        SET total_minimum_time TO zero  
        FOR course FROM one TO n  
            SET course_time TO dfs(course)  
            IF course_time GREATER THAN total_minimum_time THEN  
                SET total_minimum_time TO course_time  
            END IF  
        END FOR  

        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
