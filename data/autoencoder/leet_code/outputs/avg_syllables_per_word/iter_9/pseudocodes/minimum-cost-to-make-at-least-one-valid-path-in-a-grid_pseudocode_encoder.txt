CLASS Solution  
    FUNCTION minCost(self, grid)  
        SET m TO length of grid  
        SET n TO length of first element in grid  
        SET directions TO list of (0, 1), (0, -1), (1, 0), (-1, 0)  
        SET costs TO a two-dimensional list of size m by n with each element set to infinity  
        SET costs at position (0, 0) TO 0  
        
        SET pq TO a list containing tuple of 0, 0, 0  
        SET visited TO an empty set  
        
        WHILE pq is not empty DO  
            REMOVE smallest element from pq and unpack as cost, i, j  
            
            IF tuple (i, j) is in visited THEN  
                CONTINUE to next iteration  
            END IF  
            ADD tuple (i, j) to visited  
            
            IF i EQUALS m MINUS 1 AND j EQUALS n MINUS 1 THEN  
                RETURN cost  
            END IF  
            
            FOR each index k and tuple (di, dj) IN directions DO  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO 0 AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO 0 AND nj LESS THAN n THEN  
                    SET new_cost TO cost PLUS (one IF k PLUS one NOT EQUALS grid at position (i, j) ELSE zero)  
                    IF new_cost LESS THAN costs at position (ni, nj) THEN  
                        SET costs at position (ni, nj) TO new_cost  
                        ADD tuple (new_cost, ni, nj) to pq  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
