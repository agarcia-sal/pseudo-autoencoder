CLASS Solution  
    FUNCTION maximumScore(self, nums, multipliers)  
        SET m TO length of multipliers  
        SET n TO length of nums  
        
        INITIALIZE dp AS list of zeroes with length m PLUS one  
        
        FOR i FROM m MINUS one DOWN TO zero  
            INITIALIZE new_dp AS list of zeroes with length m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS (i MINUS left)  
                SET mul TO element at index i in multipliers  
                SET new_dp at index left TO maximum of (mul MULTIPLIED BY element at index left in nums PLUS dp at index left PLUS one) AND (mul MULTIPLIED BY element at index right in nums PLUS dp at index left)  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at index zero in dp  
    END FUNCTION  
END CLASS
