CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode initialized with first element of values  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO new ListNode initialized with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node IS NOT None DO  
                APPEND node.val TO stack  
                SET node TO node.next  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO pushIntoStack(l1)  
        SET stack2 TO pushIntoStack(l2)  

        SET carry TO 0  
        SET result TO None  

        WHILE stack1 IS NOT empty OR stack2 IS NOT empty OR carry GREATER THAN 0 DO  
            IF stack1 IS NOT empty THEN  
                SET val1 TO last element removed FROM stack1  
            ELSE  
                SET val1 TO 0  
            END IF  

            IF stack2 IS NOT empty THEN  
                SET val2 TO last element removed FROM stack2  
            ELSE  
                SET val2 TO 0  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY 10  
            SET digit TO total MODULO 10  

            SET new_node TO new ListNode initialized with digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
