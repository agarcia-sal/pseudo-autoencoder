CLASS Solution  
    FUNCTION reorderedPowerOf2(self, n)  
        INITIALIZE power_of_2_sorted_digits AS empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO ten raised to the power of nine DO  
            ADD tuple of digits of power_of_2 sorted in ascending order TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  

        SET sorted_n_digits TO tuple of digits of n sorted in ascending order  

        RETURN whether sorted_n_digits IS IN power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
