CLASS Solution  
    FUNCTION canWin(self, currentState)  
        SET memo TO empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN memo[state]  
            END IF  
            
            FOR index FROM zero TO length of state MINUS one MINUS one DO  
                IF substring of state FROM index TO index PLUS two EQUALS string "PLUS PLUS" THEN  
                    SET new_state TO concatenation of substring of state FROM start TO index, string "MINUS MINUS", and substring of state FROM index PLUS two TO end  
                    IF NOT can_win_helper(new_state) THEN  
                        SET memo[state] TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo[state] TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
