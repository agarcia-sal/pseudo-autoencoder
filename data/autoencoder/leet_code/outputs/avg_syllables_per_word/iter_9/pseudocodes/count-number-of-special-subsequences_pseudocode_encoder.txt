CLASS Solution  
    FUNCTION countSpecialSubsequences(self, nums)  
        SET MODULO TO ten raised to the power of nine PLUS seven  
        SET dp_at_zero TO zero  
        SET dp_at_one TO zero  
        SET dp_at_two TO zero  
  
        FOR each num IN nums  
            IF num EQUALS zero THEN  
                SET dp_at_zero TO (two MULTIPLIED BY dp_at_zero) PLUS one  
                SET dp_at_zero TO dp_at_zero MODULO MODULO  
            ELSE IF num EQUALS one THEN  
                SET dp_at_one TO (two MULTIPLIED BY dp_at_one) PLUS dp_at_zero  
                SET dp_at_one TO dp_at_one MODULO MODULO  
            ELSE IF num EQUALS two THEN  
                SET dp_at_two TO (two MULTIPLIED BY dp_at_two) PLUS dp_at_one  
                SET dp_at_two TO dp_at_two MODULO MODULO  
            END IF  
        END FOR  
  
        RETURN dp_at_two  
    END FUNCTION  
END CLASS
