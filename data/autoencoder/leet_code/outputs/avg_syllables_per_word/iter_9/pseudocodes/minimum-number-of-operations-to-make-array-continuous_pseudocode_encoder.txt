CLASS Solution  
    FUNCTION minOperations(self, nums)  
        SET n TO length of nums  
        SET unique_nums TO sorted list of unique elements from nums  
        
        SET min_operations TO n  
        SET j TO 0  
        FOR i FROM 0 TO length of unique_nums MINUS 1  
            WHILE j LESS THAN length of unique_nums AND unique_nums at position j LESS THAN unique_nums at position i PLUS n  
                INCREMENT j BY 1  
            END WHILE  
            SET min_operations TO minimum of min_operations AND n MINUS (j MINUS i)  
        END FOR  
        
        RETURN min_operations  
    END FUNCTION  
END CLASS
