CLASS Solution  
    FUNCTION findLonelyPixel(self, picture)  
        IF picture IS empty OR first element of picture IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of picture  
        SET n TO length of first element of picture  
        SET row_count TO list of zeros of length m  
        SET col_count TO list of zeros of length n  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at picture position i j EQUALS character B THEN  
                    INCREMENT element at row_count position i BY one  
                    INCREMENT element at col_count position j BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixel_count TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at picture position i j EQUALS character B AND element at row_count position i EQUALS one AND element at col_count position j EQUALS one THEN  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
