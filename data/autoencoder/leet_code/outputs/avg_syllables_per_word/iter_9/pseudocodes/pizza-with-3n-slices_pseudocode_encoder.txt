CLASS Solution  
    FUNCTION maxSizeSlices(self, slices)  
        SET n TO length of slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            
            SET pick TO value at slices position start PLUS dp(start PLUS two, end, k MINUS one)  
            SET skip TO dp(start PLUS one, end, k)  
            RETURN the maximum of pick AND skip  
        END FUNCTION  
        
        SET case1 TO dp(one, length of slices MINUS one, n)  
        SET case2 TO dp(zero, length of slices MINUS two, n)  
        
        RETURN the maximum of case1 AND case2  
    END FUNCTION  
END CLASS
