CLASS Solution  
    FUNCTION distinctEchoSubstrings(self, text)  
        SET n TO length of text  
        SET substrings TO empty set  

        FOR length FROM one TO integer division of n BY two PLUS one  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one  
                SET a TO substring of text from start TO start PLUS length  
                SET b TO substring of text from start PLUS length TO start PLUS two MULTIPLIED BY length  
                IF a EQUALS b THEN  
                    ADD concatenation of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  

        RETURN size of substrings  
    END FUNCTION  
END CLASS
