CLASS Solution  
    FUNCTION minimumDeleteSum(self, s1, s2)  
        SET m TO length of s1  
        SET n TO length of s2  
        
        SET dp TO two dimensional array of size (m plus one) by (n plus one) initialized with zeros  
        
        FOR j FROM one TO n DO  
            SET dp at row zero column j TO dp at row zero column (j minus one) PLUS ASCII value of character at position (j minus one) in s2  
        END FOR  
        
        FOR i FROM one TO m DO  
            SET dp at row i column zero TO dp at row (i minus one) column zero PLUS ASCII value of character at position (i minus one) in s1  
        END FOR  
        
        FOR i FROM one TO m DO  
            FOR j FROM one TO n DO  
                IF character at position (i minus one) in s1 EQUALS character at position (j minus one) in s2 THEN  
                    SET dp at row i column j TO dp at row (i minus one) column (j minus one)  
                ELSE  
                    SET dp at row i column j TO minimum of (dp at row (i minus one) column j PLUS ASCII value of character at position (i minus one) in s1) AND (dp at row i column (j minus one) PLUS ASCII value of character at position (j minus one) in s2)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at row m column n  
    END FUNCTION  
END CLASS
