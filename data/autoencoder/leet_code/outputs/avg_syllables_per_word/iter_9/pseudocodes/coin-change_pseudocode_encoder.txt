CLASS Solution  
    FUNCTION coinChange(self, coins, amount)  
        SET dp TO a list of length amount PLUS one, each element initialized to amount PLUS one  
        SET dp at position zero TO zero  

        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin GREATER THAN OR EQUAL TO zero THEN  
                    SET dp at position a TO the smaller value BETWEEN dp at position a AND dp at position a MINUS coin PLUS one  
                END IF  
            END FOR  
        END FOR  

        IF dp at position amount EQUALS amount PLUS one THEN  
            RETURN negative one  
        ELSE  
            RETURN dp at position amount  
        END IF  
    END FUNCTION  
END CLASS
