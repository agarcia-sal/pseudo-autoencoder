CLASS Solution  
    FUNCTION findUnsortedSubarray(self, nums)  
        SET sorted_nums TO sorted version of nums  
        SET start TO negative one  
        SET end TO negative two  
        FOR i FROM zero TO length of nums MINUS one  
            IF element at index i in nums NOT EQUALS element at index i in sorted_nums THEN  
                IF start EQUALS negative one THEN  
                    SET start TO i  
                END IF  
                SET end TO i  
            END IF  
        END FOR  
        RETURN end MINUS start PLUS one  
    END FUNCTION  
END CLASS
