CLASS Solution  
    FUNCTION maxChunksToSorted(self, arr)  
        SET max_so_far TO empty list  
        SET current_max TO first element IN arr  
        FOR each num IN arr  
            SET current_max TO greater value BETWEEN current_max AND num  
            APPEND current_max TO max_so_far  
        END FOR  
        
        SET sorted_arr TO arr sorted in ascending order  
        SET chunks TO zero  
        SET max_chunk TO zero  
        
        FOR index FROM zero TO length of arr MINUS one  
            SET max_chunk TO greater value BETWEEN max_chunk AND element at index IN arr  
            IF max_chunk EQUALS element at index IN sorted_arr THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
