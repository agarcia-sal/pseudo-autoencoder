CLASS Solution  
    FUNCTION threeEqualParts(self, arr)  
        SET total_ones TO the sum of all elements in arr  
        
        IF total_ones EQUALS zero THEN  
            RETURN a list containing zero and two  
        END IF  
        
        IF the remainder of total_ones divided by three NOT EQUALS zero THEN  
            RETURN a list containing negative one and negative one  
        END IF  
        
        SET part_ones TO total_ones divided by three  
        
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the length of arr MINUS one  
            IF the element at arr at position index EQUALS one THEN  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS one THEN  
                    SET first_part_index TO index  
                END IF  
                IF count_ones EQUALS part_ones THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the length of arr MINUS one  
            IF the element at arr at position index EQUALS one THEN  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS part_ones PLUS one THEN  
                    SET second_part_index TO index  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the length of arr MINUS one  
            IF the element at arr at position index EQUALS one THEN  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS two MULTIPLIED BY part_ones PLUS one THEN  
                    SET third_part_index TO index  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET n TO the length of arr MINUS third_part_index  
        IF first_part_index PLUS n LESS THAN OR EQUAL TO the length of arr AND  
           second_part_index PLUS n LESS THAN OR EQUAL TO the length of arr AND  
           the sublist of arr from first_part_index to first_part_index PLUS n MINUS one IS EQUAL TO  
           the sublist of arr from second_part_index to second_part_index PLUS n MINUS one AND  
           the sublist of arr from second_part_index to second_part_index PLUS n MINUS one IS EQUAL TO  
           the sublist of arr from third_part_index to the end THEN  
            RETURN a list containing first_part_index PLUS n MINUS one AND second_part_index PLUS n  
        END IF  
        
        RETURN a list containing negative one AND negative one  
    END FUNCTION  
END CLASS
