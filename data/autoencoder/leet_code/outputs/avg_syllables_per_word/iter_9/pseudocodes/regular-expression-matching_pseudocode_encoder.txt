CLASS Solution  
    FUNCTION isMatch(self, s, p)  
        INITIALIZE dp AS two-dimensional list of False with dimensions (length of s PLUS one) BY (length of p PLUS one)  
        
        SET dp at zero zero TO True  
        
        FOR j FROM 1 TO length of p  
            IF character at position j MINUS one in p EQUALS asterisk THEN  
                SET dp at zero j TO dp at zero (j MINUS two)  
            END IF  
        END FOR  
        
        FOR i FROM 1 TO length of s  
            FOR j FROM 1 TO length of p  
                IF character at position j MINUS one in p EQUALS character at position i MINUS one in s OR character at position j MINUS one in p EQUALS dot THEN  
                    SET dp at i j TO dp at (i MINUS one) (j MINUS one)  
                ELSE IF character at position j MINUS one in p EQUALS asterisk THEN  
                    SET dp at i j TO dp at i (j MINUS two)  
                    IF character at position j MINUS two in p EQUALS character at position i MINUS one in s OR character at position j MINUS two in p EQUALS dot THEN  
                        SET dp at i j TO dp at i j OR dp at (i MINUS one) j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at (length of s) (length of p)  
    END FUNCTION  
END CLASS
