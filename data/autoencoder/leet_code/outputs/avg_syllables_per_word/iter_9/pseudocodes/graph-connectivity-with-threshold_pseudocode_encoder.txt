CLASS Solution  
    FUNCTION areConnected(self, n, threshold, queries)  
        IF threshold EQUALS zero THEN  
            RETURN list of True repeated length of queries times  
        END IF  
        
        SET parent TO list of integers from zero TO n inclusive  
        
        FUNCTION find(x)  
            IF value at parent position x NOT EQUALS x THEN  
                SET value at parent position x TO result of find(value at parent position x)  
            END IF  
            RETURN value at parent position x  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO result of find(x)  
            SET rootY TO result of find(y)  
            IF rootX NOT EQUALS rootY THEN  
                SET value at parent position rootY TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS one TO n inclusive  
            FOR j FROM i MULTIPLIED BY two TO n inclusive STEP i  
                CALL union(i, j)  
            END FOR  
        END FOR  
        
        SET result TO empty list  
        FOR each pair a, b IN queries  
            APPEND EQUALITY of find(a) AND find(b) TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
