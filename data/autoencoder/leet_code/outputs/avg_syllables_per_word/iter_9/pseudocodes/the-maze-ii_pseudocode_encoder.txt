CLASS Solution  
    FUNCTION shortestDistance(self, maze, start, destination)  
        SET m TO length of maze  
        SET n TO length of first element in maze  
        SET directions TO list of pairs: (one, zero), (negative one, zero), (zero, one), (zero, negative one)  
        SET start TO tuple of start  
        SET destination TO tuple of destination  
        SET visited TO empty set  
        SET priority queue pq TO list containing pair of zero and start  
        
        WHILE pq is not empty  
            REMOVE element with smallest distance FROM pq and unpack as dist and position (x, y)  
            
            IF position (x, y) EQUALS destination THEN  
                RETURN dist  
            END IF  
            
            IF position (x, y) IS IN visited THEN  
                CONTINUE to next iteration of loop  
            END IF  
            
            ADD position (x, y) TO visited  
            
            FOR each pair (dx, dy) IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND  
                      new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND  
                      maze at position (new_x plus dx, new_y plus dy) EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF position (new_x, new_y) IS NOT IN visited THEN  
                    ADD pair of dist plus count and position (new_x, new_y) TO pq using heap insertion  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
