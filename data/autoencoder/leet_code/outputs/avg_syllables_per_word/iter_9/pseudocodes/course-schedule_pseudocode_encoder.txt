CLASS Solution  
    FUNCTION canFinish(self, numCourses, prerequisites)  
        IMPORT defaultdict, deque FROM collections  
        
        SET graph TO a default dictionary with empty list as default value  
        SET indegree TO a list of zeros with length numCourses  
        
        FOR each pair course, prereq IN prerequisites  
            APPEND course TO graph value at key prereq  
            INCREMENT indegree at index course BY one  
        END FOR  
        
        SET queue TO a deque containing all courses from zero to numCourses minus one where indegree equals zero  
        
        WHILE queue is not empty  
            SET current_course TO remove element from front of queue  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN graph value at key current_course  
                DECREMENT indegree at index next_course BY one  
                IF indegree at index next_course EQUALS zero THEN  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
