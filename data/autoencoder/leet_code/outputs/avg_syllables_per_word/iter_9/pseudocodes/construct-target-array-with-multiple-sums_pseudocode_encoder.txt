CLASS Solution  
    FUNCTION isPossible(self, target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative num TO max_heap  
        END FOR  
        CALL heapify ON max_heap  

        SET total_sum TO sum of all elements in target  

        WHILE first element of max_heap LESS THAN negative one  
            SET largest TO negative of pop smallest element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest LESS THAN OR EQUALS rest_sum OR rest_sum EQUALS zero THEN  
                RETURN False  
            END IF  
            
            SET new_value TO largest MODULO rest_sum  
            
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one THEN  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            PUSH negative new_value INTO max_heap  
        END WHILE  

        RETURN True  
    END FUNCTION  
END CLASS
