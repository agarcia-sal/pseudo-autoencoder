CLASS Solution  
    FUNCTION distanceToCycle(self, n, edges)  
        SET graph TO a default dictionary mapping to empty list  
        FOR each pair u, v IN edges  
            APPEND v TO graph at u  
            APPEND u TO graph at v  
        END FOR  

        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO position of node IN path  
                RETURN sublist of path starting at cycle_start  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN graph at node  
                IF neighbor NOT EQUALS parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT None THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE last element FROM path  
            RETURN None  
        END FUNCTION  

        SET visited TO empty set  
        SET cycle TO None  
        FOR node FROM zero TO n MINUS one  
            IF node NOT IN visited THEN  
                SET cycle TO dfs(node, MINUS one, empty list, visited)  
                IF cycle IS NOT None THEN  
                    BREAK  
                END IF  
            END IF  
        END FOR  

        SET distance TO list of zeros with length n  
        SET queue TO a deque initialized with elements of cycle  
        SET visited TO a set initialized with elements of cycle  

        SET level TO zero  
        WHILE queue IS NOT empty  
            FOR count FROM zero TO length of queue MINUS one  
                SET node TO element removed from the left of queue  
                SET distance at node TO level  
                FOR each neighbor IN graph at node  
                    IF neighbor NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  

        RETURN distance  
    END FUNCTION  
END CLASS
