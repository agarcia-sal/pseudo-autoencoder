CLASS Solution  
    FUNCTION findPeakElement(self, nums)  
        SET left TO 0  
        SET right TO length of nums MINUS 1  
        
        WHILE left LESS THAN right  
            SET mid TO the integer division of left PLUS right BY 2  
            IF element at index mid in nums GREATER THAN element at index mid PLUS 1 in nums THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS 1  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
