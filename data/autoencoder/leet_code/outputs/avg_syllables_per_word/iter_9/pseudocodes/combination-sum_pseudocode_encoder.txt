CLASS Solution  
    FUNCTION combinationSum(self, candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero THEN  
                RETURN  
            END IF  
            FOR i FROM start TO length of candidates MINUS one  
                CALL backtrack WITH i, target MINUS candidates AT i, path APPENDED WITH candidates AT i  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH zero, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
