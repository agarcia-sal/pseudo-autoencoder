CLASS Solution  
    FUNCTION checkSubarraySum(self, nums, k)  
        SET remainder_map TO a dictionary with key zero mapped to negative one  
        SET current_sum TO zero  
        
        FOR each index i AND element num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            IF k NOT EQUALS zero THEN  
                SET current_sum TO current_sum MODULO k  
            END IF  
            
            IF current_sum EXISTS AS key IN remainder_map THEN  
                IF i MINUS value mapped by current_sum IN remainder_map GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                SET remainder_map AT current_sum TO i  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
