CLASS Solution  
    FUNCTION topKFrequent(self, words, k)  
        SET count TO FUNCTION Counter WITH words  
        SET heap TO EMPTY LIST  
        FOR each word AND freq IN items OF count  
            APPEND tuple OF negative freq AND word TO heap  
        END FOR  
        CALL FUNCTION heapify WITH heap  
        SET result TO EMPTY LIST  
        FOR each number FROM one TO k  
            SET element TO CALL FUNCTION heappop WITH heap  
            APPEND second element OF element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
