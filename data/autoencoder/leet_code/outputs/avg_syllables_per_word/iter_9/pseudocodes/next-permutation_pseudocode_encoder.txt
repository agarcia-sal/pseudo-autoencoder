CLASS Solution  
    FUNCTION nextPermutation(self, nums)  
        SET i TO length of nums MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND element at position i IN nums GREATER THAN OR EQUALS element at position i PLUS 1 IN nums  
            DECREMENT i BY 1  
        END WHILE  

        IF i GREATER THAN OR EQUALS 0 THEN  
            SET j TO length of nums MINUS 1  
            WHILE element at position j IN nums LESS THAN OR EQUALS element at position i IN nums  
                DECREMENT j BY 1  
            END WHILE  
            SET temporary variable TO element at position i IN nums  
            SET element at position i IN nums TO element at position j IN nums  
            SET element at position j IN nums TO temporary variable  
        END IF  

        SET sublist TO elements of nums starting from position i PLUS 1 TO end  
        REVERSE sublist  
        REPLACE elements of nums starting from position i PLUS 1 TO end WITH sublist  
    END FUNCTION  
END CLASS
