CLASS Solution  
    FUNCTION pourWater(self, heights, volume, k)  
        SET n TO length of heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN zero AND element at position i MINUS one IN heights LESS THAN OR EQUAL TO element at position i IN heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i LESS THAN k AND element at position i IN heights EQUALS element at position i PLUS one IN heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i LESS THAN k THEN  
                INCREMENT element at position i IN heights BY one  
                CONTINUE to next iteration of FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND element at position i PLUS one IN heights LESS THAN OR EQUAL TO element at position i IN heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i GREATER THAN k AND element at position i IN heights EQUALS element at position i MINUS one IN heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i GREATER THAN k THEN  
                INCREMENT element at position i IN heights BY one  
                CONTINUE to next iteration of FOR loop  
            END IF  
            
            INCREMENT element at position k IN heights BY one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
