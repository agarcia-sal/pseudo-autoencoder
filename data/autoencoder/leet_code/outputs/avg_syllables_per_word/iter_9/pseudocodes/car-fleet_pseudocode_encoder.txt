CLASS Solution  
    FUNCTION carFleet(self, target, position, speed)  
        SET paired_list TO list of pairs of elements from position and speed sorted in descending order by position  
        SET time_to_target TO empty list  
        FOR each pair p, s IN paired_list  
            SET time_to_reach TO target MINUS p DIVIDED BY s  
            APPEND time_to_reach TO time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
