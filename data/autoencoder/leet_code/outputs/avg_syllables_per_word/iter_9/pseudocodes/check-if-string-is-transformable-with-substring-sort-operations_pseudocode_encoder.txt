CLASS Solution  
    FUNCTION isTransformable(self, s, t)  
        DEFINE pos AS mapping from digit TO list of positions  
        FOR each index i AND character c IN s  
            SET digit TO integer value of c  
            ADD i TO heap list corresponding to digit in pos  
        END FOR  
        
        FOR each index i AND character c IN t  
            SET digit TO integer value of c  
            IF heap list corresponding to digit in pos IS empty THEN  
                RETURN False  
            END IF  
            
            FOR each smaller FROM zero TO digit MINUS one  
                IF heap list corresponding to smaller in pos IS NOT empty AND first element IN heap list corresponding to smaller IS LESS THAN first element IN heap list corresponding to digit THEN  
                    RETURN False  
                END IF  
            END FOR  
            
            REMOVE first element FROM heap list corresponding to digit IN pos  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
