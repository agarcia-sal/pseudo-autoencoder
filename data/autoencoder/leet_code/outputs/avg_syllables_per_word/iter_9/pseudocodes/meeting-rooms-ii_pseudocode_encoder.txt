CLASS Solution  
    FUNCTION minMeetingRooms(self, intervals)  
        IF intervals EQUALS empty THEN  
            RETURN zero  
        END IF  

        SORT intervals BY the first element of each sublist  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF min_heap is not empty AND the first element of meeting GREATER THAN OR EQUAL TO the first element of min_heap THEN  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the second element of meeting TO min_heap, maintaining min_heap as a structure ordered by smallest element  
        END FOR  

        RETURN the number of elements IN min_heap  
    END FUNCTION  
END CLASS
