CLASS Solution  
    FUNCTION threeSumClosest(self, nums, target)  
        CALL nums SORT  
        SET closest_sum TO infinity  
        
        FOR i FROM zero TO length of nums MINUS two  
            SET left TO i PLUS one  
            SET right TO length of nums MINUS one  
            
            WHILE left LESS THAN right  
                SET current_sum TO value at nums index i PLUS value at nums index left PLUS value at nums index right  
                
                IF absolute value of current_sum MINUS target LESS THAN absolute value of closest_sum MINUS target THEN  
                    SET closest_sum TO current_sum  
                END IF  
                
                IF current_sum LESS THAN target THEN  
                    INCREMENT left BY one  
                ELSE IF current_sum GREATER THAN target THEN  
                    DECREMENT right BY one  
                ELSE  
                    RETURN current_sum  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN closest_sum  
    END FUNCTION  
END CLASS
