CLASS Solution  
    FUNCTION xorGame(self, nums)  
        SET total_xor TO zero  
        FOR each num IN nums  
            SET total_xor TO total_xor EXCLUSIVE OR num  
        END FOR  
        IF total_xor EQUALS zero THEN  
            RETURN true  
        END IF  
        IF length of nums MODULO two EQUALS zero THEN  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS
