CLASS Solution  
    FUNCTION getHint(self, secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        CALL auxiliary FUNCTION defaultdict WITH parameter integer TO CREATE secret_count  
        CALL auxiliary FUNCTION defaultdict WITH parameter integer TO CREATE guess_count  
        
        FOR each pair s and g IN corresponding elements of secret and guess  
            IF s EQUALS g THEN  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT value of secret_count at key s BY one  
                INCREMENT value of guess_count at key g BY one  
            END IF  
        END FOR  
        
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY minimum of secret_count at digit AND guess_count at digit  
        END FOR  
        
        RETURN concatenation of bulls, the letter A, cows, and the letter B in the format "xAyB"  
    END FUNCTION  
END CLASS
