CLASS Solution  
    FUNCTION appealSum(self, s)  
        SET last TO list of twenty-six elements each initialized to minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each index i and character char IN enumeration of s  
            SET index TO numerical position of char MINUS numerical position of character 'a'  
            INCREMENT current_appeal BY difference of i and element at position index in last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index in last TO i  
        END FOR  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
