CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(self, binary)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        
        SET has_zero TO whether the binary string CONTAINS the character zero  
        
        FOR each bit IN binary  
            IF bit EQUALS one THEN  
                SET end_with_1 TO remainder of the sum of end_with_0 PLUS end_with_1 PLUS one DIVIDED BY MOD  
            ELSE  
                SET end_with_0 TO remainder of the sum of end_with_0 PLUS end_with_1 DIVIDED BY MOD  
            END IF  
        END FOR  
        
        SET result TO remainder of the sum of end_with_0 PLUS end_with_1 PLUS has_zero DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
