CLASS Solution  
    FUNCTION countArrangement(self, n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one THEN  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF used at i EQUALS false AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero) THEN  
                    SET used at i TO true  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET used at i TO false  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
      
        SET used TO list of false repeated n PLUS one times  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
