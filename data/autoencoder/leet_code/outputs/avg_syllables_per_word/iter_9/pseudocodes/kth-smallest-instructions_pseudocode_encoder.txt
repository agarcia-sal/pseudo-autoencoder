CLASS Solution  
    FUNCTION kthSmallestPath(self, destination, k)  
        SET row TO first element of destination  
        SET col TO second element of destination  
        SET path TO empty list  
        SET total_moves TO col PLUS row  

        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero THEN  
                SET paths_with_H TO number of combinations of row PLUS col MINUS one choose col MINUS one  
                IF k LESS THAN OR EQUALS paths_with_H THEN  
                    APPEND character 'H' TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND character 'V' TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND character 'V' TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  

        RETURN concatenation of elements in path as a string  
    END FUNCTION  
END CLASS
