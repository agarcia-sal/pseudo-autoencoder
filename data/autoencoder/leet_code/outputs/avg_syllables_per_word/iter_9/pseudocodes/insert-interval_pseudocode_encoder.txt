CLASS Solution  
    FUNCTION insert(self, intervals, newInterval)  
        IF intervals EQUALS empty THEN  
            RETURN a list containing newInterval  
        END IF  
        
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the length of intervals  
        
        WHILE i LESS THAN n AND the second element of intervals at i LESS THAN the first element of newInterval DO  
            APPEND intervals at i TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i LESS THAN n AND the first element of intervals at i LESS THAN OR EQUAL TO the second element of newInterval DO  
            SET the first element of newInterval TO the minimum of the first element of newInterval AND the first element of intervals at i  
            SET the second element of newInterval TO the maximum of the second element of newInterval AND the second element of intervals at i  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n DO  
            APPEND intervals at i TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
