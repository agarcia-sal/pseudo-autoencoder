CLASS Solution  
    FUNCTION minCost(self, costs)  
        IF costs EQUALS empty THEN  
            RETURN 0  
        END IF  
        
        SET prev_red TO first element's first cost in costs  
        SET prev_blue TO first element's second cost in costs  
        SET prev_green TO first element's third cost in costs  
        
        FOR i FROM 1 TO length of costs MINUS 1  
            SET current_red TO costs at index i's first cost PLUS minimum of prev_blue AND prev_green  
            SET current_blue TO costs at index i's second cost PLUS minimum of prev_red AND prev_green  
            SET current_green TO costs at index i's third cost PLUS minimum of prev_red AND prev_blue  
            
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        
        RETURN minimum of prev_red, prev_blue, AND prev_green  
    END FUNCTION  
END CLASS
