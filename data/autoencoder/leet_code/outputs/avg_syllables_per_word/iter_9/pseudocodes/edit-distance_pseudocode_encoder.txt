CLASS Solution  
    FUNCTION minDistance(self, word1, word2)  
        SET m TO length of word1  
        SET n TO length of word2  
        
        CREATE dp AS two dimensional table with (m PLUS one) rows and (n PLUS one) columns filled with zero  
        
        FOR i FROM zero TO m  
            SET dp at position i and zero TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET dp at position zero and j TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position (i MINUS one) in word1 EQUALS character at position (j MINUS one) in word2 THEN  
                    SET dp at position i and j TO dp at position (i MINUS one) and (j MINUS one)  
                ELSE  
                    SET dp at position i and j TO one PLUS minimum of dp at position (i MINUS one) and j, dp at position i and (j MINUS one), dp at position (i MINUS one) and (j MINUS one)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
