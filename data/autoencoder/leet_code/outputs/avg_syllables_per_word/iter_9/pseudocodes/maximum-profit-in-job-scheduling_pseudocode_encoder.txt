CLASS Solution  
    FUNCTION jobScheduling(self, startTime, endTime, profit)  
        SET jobs TO list of tuples combining startTime, endTime, and profit, sorted by endTime of each tuple in ascending order  
        SET end_times TO list of endTime elements extracted from each job in jobs  
        SET dp TO list of zeros with length equal to length of jobs plus one  

        FOR i FROM 1 TO length of jobs  
            SET j TO position found by binary searching for right insertion point of startTime of job at index i minus one in end_times  
            SET dp at i TO maximum of dp at i minus one and sum of dp at j and profit of job at index i minus one  
        END FOR  

        RETURN last element of dp  
    END FUNCTION  
END CLASS
