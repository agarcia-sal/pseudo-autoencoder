CLASS Solution  
    FUNCTION numOfWays(self, n)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET a TO six  
        SET b TO six  
        FOR index FROM two TO n  
            SET new_a TO remainder of three MULTIPLIED BY a PLUS two MULTIPLIED BY b DIVIDED BY MODULO MOD  
            SET new_b TO remainder of two MULTIPLIED BY a PLUS two MULTIPLIED BY b DIVIDED BY MODULO MOD  
            SET a TO new_a  
            SET b TO new_b  
        END FOR  
        RETURN remainder of a PLUS b DIVIDED BY MODULO MOD  
    END FUNCTION  
END CLASS
