CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH value as first element of values  
    SET i TO one  
    SET queue TO new double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode WITH value as values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode WITH value as values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        SET column_table TO mapping from keys to empty lists  
        SET queue TO new double-ended queue with tuple containing root and zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue IS NOT empty DO  
            SET node, column_index TO first element removed FROM queue  
            IF node IS NOT None THEN  
                APPEND node.val TO list at column_index in column_table  
                IF column_index LESS THAN min_column THEN  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column THEN  
                    SET max_column TO column_index  
                END IF  
                APPEND tuple containing node.left and column_index minus one TO queue  
                APPEND tuple containing node.right and column_index plus one TO queue  
            END IF  
        END WHILE  
        SET result TO empty list  
        FOR x FROM min_column TO max_column DO  
            APPEND list at x in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
