CLASS Solution  
    FUNCTION minStickers(self, stickers, target)  
        SET stickers TO list of Counters each created from each element in stickers  
        SORT stickers by the negative of the maximum value present in each Counter  
        
        SET queue TO a new deque containing a tuple of target and zero  
        SET memo TO a dictionary with target mapped to zero  
        
        WHILE queue is not empty  
            REMOVE the leftmost element from queue INTO current_target and used_stickers  
            
            FOR each sticker IN stickers  
                IF the first character in current_target is NOT IN sticker THEN  
                    CONTINUE to next sticker  
                END IF  
                
                SET remaining TO a Counter created from current_target  
                DECREASE remaining by the counts in sticker  
                
                SET new_target TO the string created by joining the sorted list of elements from remaining  
                
                IF new_target is an empty string THEN  
                    RETURN used_stickers PLUS one  
                END IF  
                
                IF new_target is NOT IN memo THEN  
                    SET memo[new_target] TO used_stickers PLUS one  
                    APPEND a tuple of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
