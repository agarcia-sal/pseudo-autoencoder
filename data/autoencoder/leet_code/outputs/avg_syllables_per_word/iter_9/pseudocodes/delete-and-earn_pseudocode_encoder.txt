CLASS Solution  
    FUNCTION deleteAndEarn(self, nums)  
        SET count TO a mapping of each number in nums to its number of occurrences  
        SET unique_nums TO the sorted list of keys from count  

        SET dp TO a list of zeros with length equal to the length of unique_nums PLUS one  

        SET the element at index one in dp TO the first element in unique_nums MULTIPLIED BY the count of the first element in unique_nums  

        FOR index FROM two TO the length of unique_nums PLUS one  
            SET num TO the element at position index MINUS one in unique_nums  
            IF num EQUALS the element at position index MINUS two in unique_nums PLUS one THEN  
                SET the element at position index in dp TO the greater of the element at position index MINUS one in dp AND the element at position index MINUS two in dp PLUS num MULTIPLIED BY the count of num  
            ELSE  
                SET the element at position index in dp TO the element at position index MINUS one in dp PLUS num MULTIPLIED BY the count of num  
            END IF  
        END FOR  

        RETURN the last element in dp  
    END FUNCTION  
END CLASS
