CLASS Solution  
    FUNCTION search(self, nums, target)  
        SET left TO zero  
        SET right TO length of nums MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS (right MINUS left) DIVIDED BY two  
            
            IF element at index mid in nums EQUALS target THEN  
                RETURN mid  
            END IF  
            
            IF element at index left in nums LESS THAN OR EQUALS element at index mid in nums THEN  
                IF element at index left in nums LESS THAN OR EQUALS target AND target LESS THAN element at index mid in nums THEN  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF element at index mid in nums LESS THAN target AND target LESS THAN OR EQUALS element at index right in nums THEN  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
