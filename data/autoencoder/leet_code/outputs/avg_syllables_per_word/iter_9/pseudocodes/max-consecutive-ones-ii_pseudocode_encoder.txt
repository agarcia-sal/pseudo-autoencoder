CLASS Solution  
    FUNCTION findMaxConsecutiveOnes(self, nums)  
        SET max_count TO zero  
        SET current_count TO zero  
        SET zero_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO length of nums MINUS one  
            IF element at nums indexed by right EQUALS one THEN  
                INCREMENT current_count BY one  
            ELSE IF zero_count LESS THAN one THEN  
                INCREMENT zero_count BY one  
                INCREMENT current_count BY one  
            ELSE  
                SET max_count TO maximum of max_count AND current_count  
                WHILE zero_count EQUALS one  
                    IF element at nums indexed by left EQUALS zero THEN  
                        DECREMENT zero_count BY one  
                    END IF  
                    DECREMENT current_count BY one  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT current_count BY one  
            END IF  
        END FOR  
        
        SET max_count TO maximum of max_count AND current_count  
        RETURN max_count  
    END FUNCTION  
END CLASS
