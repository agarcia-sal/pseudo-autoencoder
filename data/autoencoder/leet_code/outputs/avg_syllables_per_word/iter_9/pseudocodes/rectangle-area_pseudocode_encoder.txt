CLASS Solution  
    FUNCTION computeArea(self, ax1, ay1, ax2, ay2, bx1, by1, bx2, by2)  
        SET area1 TO (ax2 MINUS ax1) MULTIPLIED BY (ay2 MINUS ay1)  
        SET area2 TO (bx2 MINUS bx1) MULTIPLIED BY (by2 MINUS by1)  
        SET overlap_width TO maximum BETWEEN zero AND (minimum BETWEEN ax2 AND bx2) MINUS (maximum BETWEEN ax1 AND bx1)  
        SET overlap_height TO maximum BETWEEN zero AND (minimum BETWEEN ay2 AND by2) MINUS (maximum BETWEEN ay1 AND by1)  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
