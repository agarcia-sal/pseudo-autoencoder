CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at first element of values  
    SET i TO 1  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed from queue  
        IF i LESS THAN length of values AND value at i in values IS NOT None THEN  
            SET node.left TO new TreeNode with value at i in values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND value at i in values IS NOT None THEN  
            SET node.right TO new TreeNode with value at i in values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS None THEN  
            RETURN None  
        END IF  

        IF root.val LESS THAN low THEN  
            RETURN self.trimBST(root.right, low, high)  
        END IF  

        IF root.val GREATER THAN high THEN  
            RETURN self.trimBST(root.left, low, high)  
        END IF  

        SET root.left TO self.trimBST(root.left, low, high)  
        SET root.right TO self.trimBST(root.right, low, high)  

        RETURN root  
    END FUNCTION  
END CLASS
