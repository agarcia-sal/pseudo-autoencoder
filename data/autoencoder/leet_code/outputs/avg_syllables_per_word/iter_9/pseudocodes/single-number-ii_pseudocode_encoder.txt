CLASS Solution  
    FUNCTION singleNumber(self, nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO (bits in ones XOR bits in num) AND bits NOT IN twos  
            SET twos TO (bits in twos XOR bits in num) AND bits NOT IN ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
