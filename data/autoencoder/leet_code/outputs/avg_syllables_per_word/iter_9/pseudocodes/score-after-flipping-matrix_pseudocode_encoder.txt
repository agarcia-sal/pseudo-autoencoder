CLASS Solution  
    FUNCTION matrixScore(self, grid)  
        FOR each row IN grid  
            IF first element of row EQUALS zero THEN  
                FOR index FROM zero TO length of row MINUS one  
                    SET element at index in row TO one MINUS element at index in row  
                END FOR  
            END IF  
        END FOR  

        FOR col FROM one TO length of first row in grid MINUS one  
            SET count_ones TO zero  
            FOR row_index FROM zero TO length of grid MINUS one  
                INCREMENT count_ones BY element at position col in grid at row_index  
            END FOR  
            IF count_ones LESS THAN length of grid DIVIDED BY two THEN  
                FOR row_index FROM zero TO length of grid MINUS one  
                    SET element at position col in grid at row_index TO one MINUS element at position col in grid at row_index  
                END FOR  
            END IF  
        END FOR  

        SET score TO zero  
        FOR each row IN grid  
            SET binary_number TO concatenation of string representations of elements in row  
            INCREMENT score BY integer value of binary_number interpreted as base two  
        END FOR  

        RETURN score  
    END FUNCTION  
END CLASS
