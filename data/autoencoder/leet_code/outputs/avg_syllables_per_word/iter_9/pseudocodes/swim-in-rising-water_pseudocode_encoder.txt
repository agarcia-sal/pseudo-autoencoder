CLASS Solution  
    FUNCTION swimInWater(self, grid)  
        SET n TO length of grid  
        SET visited TO empty set  
        SET min_heap TO list containing tuple of grid at position zero zero, zero, zero  
        ADD tuple zero zero TO visited  
          
        SET directions TO list containing tuples zero one, one zero, zero negative one, negative one zero  
          
        WHILE min_heap is not empty  
            REMOVE smallest element FROM min_heap AND SET to t, i, j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one THEN  
                RETURN t  
            END IF  
              
            FOR each di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND tuple ni nj NOT IN visited THEN  
                    ADD tuple ni nj TO visited  
                    ADD tuple where first element is maximum of t AND grid at ni nj, second element ni, third element nj TO min_heap  
                END IF  
            END FOR  
        END WHILE  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
