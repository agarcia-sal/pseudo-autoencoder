CLASS Solution  
    FUNCTION searchMatrix(self, matrix, target)  
        IF matrix IS EMPTY OR first element in matrix IS EMPTY THEN  
            RETURN False  
        END IF  

        SET row TO zero  
        SET col TO length of first element in matrix MINUS one  

        WHILE row LESS THAN length of matrix AND col GREATER THAN OR EQUAL TO zero DO  
            IF element at matrix[row][col] EQUALS target THEN  
                RETURN True  
            ELSE IF element at matrix[row][col] GREATER THAN target THEN  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  

        RETURN False  
    END FUNCTION  
END CLASS
