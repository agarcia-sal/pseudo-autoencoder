CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as first element of values  
    SET i TO 1  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed from queue  
        IF i LESS THAN length of values AND element at index i of values IS NOT None THEN  
            SET node.left TO new TreeNode with val as element at index i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND element at index i of values IS NOT None THEN  
            SET node.right TO new TreeNode with val as element at index i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self, root)  
        IF root IS None THEN  
            RETURN None  
        END IF  
        SET root.left TO self.pruneTree(root.left)  
        SET root.right TO self.pruneTree(root.right)  
        IF root.val EQUALS 0 AND root.left IS None AND root.right IS None THEN  
            RETURN None  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
