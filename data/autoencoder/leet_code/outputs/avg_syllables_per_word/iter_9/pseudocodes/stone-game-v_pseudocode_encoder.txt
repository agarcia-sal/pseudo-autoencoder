CLASS Solution  
    FUNCTION stoneGameV(self, stoneValue)  
        FUNCTION get_range_sum(left, right)  
            RETURN prefix_sum at position right plus one MINUS prefix_sum at position left  
        END FUNCTION  
        
        FUNCTION dp(left, right)  
            IF left EQUALS right THEN  
                RETURN zero  
            END IF  
            SET max_score TO zero  
            FOR each index i FROM left TO right MINUS one  
                SET left_sum TO result of get_range_sum with left and i  
                SET right_sum TO result of get_range_sum with i plus one and right  
                IF left_sum LESS THAN right_sum THEN  
                    SET max_score TO maximum of max_score AND left_sum PLUS result of dp with left and i  
                ELSE IF left_sum GREATER THAN right_sum THEN  
                    SET max_score TO maximum of max_score AND right_sum PLUS result of dp with i plus one and right  
                ELSE  
                    SET max_score TO maximum of max_score, left_sum PLUS result of dp with left and i, AND right_sum PLUS result of dp with i plus one and right  
                END IF  
            END FOR  
            RETURN max_score  
        END FUNCTION  
        
        SET prefix_sum TO list starting with zero and extended by sum accumulation over stoneValue  
        RETURN result of dp with zero and length of stoneValue MINUS one  
    END FUNCTION  
END CLASS
