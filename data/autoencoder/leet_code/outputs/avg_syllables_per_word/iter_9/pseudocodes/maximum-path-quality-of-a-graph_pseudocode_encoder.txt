CLASS Solution  
    FUNCTION maximalPathQuality(self, values, edges, maxTime)  
        CREATE graph AS a mapping from node to list of neighbor and travel time pairs  
        FOR each u, v, time IN edges  
            APPEND pair of v and time TO graph at key u  
            APPEND pair of u and time TO graph at key v  
        END FOR  
        
        CREATE visited AS an empty set  
        SET max_quality TO zero  
        
        FUNCTION dfs(node, current_time, current_quality)  
            DECLARE max_quality AS nonlocal  
            
            IF node EQUALS zero THEN  
                SET max_quality TO the greater of max_quality AND current_quality  
            END IF  
            
            FOR each neighbor, time IN graph at key node  
                IF time LESS THAN OR EQUAL TO current_time THEN  
                    IF neighbor NOT IN visited THEN  
                        ADD neighbor TO visited  
                        CALL dfs WITH neighbor, current_time MINUS time, current_quality PLUS values at neighbor  
                        REMOVE neighbor FROM visited  
                    ELSE  
                        CALL dfs WITH neighbor, current_time MINUS time, current_quality  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        ADD zero TO visited  
        CALL dfs WITH zero, maxTime, values at zero  
        
        RETURN max_quality  
    END FUNCTION  
END CLASS
