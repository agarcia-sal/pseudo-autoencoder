CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        CREATE mapping d FROM each skill s WITH its index i IN enumeration of req_skills  
        SET m TO length of req_skills  
        SET n TO length of people  
        INITIALIZE list p WITH size n FILLED WITH zero  
        FOR each index i AND skill set ss IN enumeration of people  
            FOR each skill s IN ss  
                SET p at index i TO bitwise OR of p at index i AND bit shifted left by d at s  
            END FOR  
        END FOR  
        INITIALIZE list f WITH size two raised to m FILLED WITH infinity  
        INITIALIZE list g WITH size two raised to m FILLED WITH zero  
        INITIALIZE list h WITH size two raised to m FILLED WITH zero  
        SET f at index zero TO zero  
        FOR each integer i FROM zero TO two raised to m MINUS one  
            IF f at index i EQUALS infinity THEN  
                CONTINUE to next iteration  
            END IF  
            FOR each integer j FROM zero TO n MINUS one  
                IF f at index i PLUS one LESS THAN f at index bitwise OR of i and p at index j THEN  
                    SET f at index bitwise OR of i and p at index j TO f at index i PLUS one  
                    SET g at index bitwise OR of i and p at index j TO j  
                    SET h at index bitwise OR of i and p at index j TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two raised to m MINUS one  
        INITIALIZE list ans AS empty  
        WHILE i NOT EQUAL zero  
            APPEND g at index i TO ans  
            SET i TO h at index i  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
