CLASS Solution  
    FUNCTION smallestFactorization(self, num)  
        IF num EQUALS 1 THEN  
            RETURN 1  
        END IF  
        
        SET factors TO empty list  
        FOR digit FROM 9 DOWN TO 2 STEP negative one  
            WHILE num MODULO digit EQUALS 0 DO  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS 1 THEN  
            RETURN 0  
        END IF  
        
        SORT factors IN ascending order  
        SET result TO integer formed by concatenating all elements of factors as strings in order  
        
        IF result GREATER THAN two raised to the power of thirty one MINUS one THEN  
            RETURN 0  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
