CLASS Solution  
    FUNCTION findMin(self, nums)  
        SET left TO 0  
        SET right TO length of nums MINUS 1  
        
        IF element at index left in nums LESS THAN element at index right in nums THEN  
            RETURN element at index left in nums  
        END IF  
        
        WHILE left LESS THAN right  
            SET mid TO the integer division result of left PLUS right by 2  
            
            IF element at index mid in nums GREATER THAN element at index right in nums THEN  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN element at index left in nums  
    END FUNCTION  
END CLASS
