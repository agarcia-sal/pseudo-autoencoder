CLASS Solution  
    FUNCTION minimumXORSum(self, nums1, nums2)  
        SET n TO length of nums1  
        SET dp TO list of size two raised to power n filled with positive infinity  
        SET first element of dp TO zero  
        
        FOR each mask FROM zero TO two raised to power n MINUS one  
            SET bit_count TO count of ones in binary representation of mask  
            FOR each j FROM zero TO n MINUS one  
                IF mask BITWISE AND two raised to power j IS NOT ZERO THEN  
                    SET dp at index mask TO minimum of dp at index mask AND dp at index mask BITWISE XOR two raised to power j PLUS bitwise XOR of element at index bit_count MINUS one in nums1 AND element at index j in nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at index two raised to power n MINUS one  
    END FUNCTION  
END CLASS
