CLASS Solution  
    FUNCTION fourSumCount(self, nums1, nums2, nums3, nums4)  
        SET sum_ab TO an empty dictionary  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EXISTS AS key IN sum_ab THEN  
                    INCREMENT sum_ab[current_sum] BY one  
                ELSE  
                    SET sum_ab[current_sum] TO one  
                END IF  
            END FOR  
        END FOR  
        
        SET count TO zero  
        
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO NEGATIVE of (c PLUS d)  
                IF target EXISTS AS key IN sum_ab THEN  
                    INCREMENT count BY sum_ab[target]  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
