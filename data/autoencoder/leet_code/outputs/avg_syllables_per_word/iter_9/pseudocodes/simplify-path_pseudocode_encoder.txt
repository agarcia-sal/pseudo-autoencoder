CLASS Solution  
    FUNCTION simplifyPath(self, path)  
        SET components TO the list obtained by splitting path by the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS dot THEN  
                CONTINUE to next iteration  
            ELSE IF component EQUALS two dots THEN  
                IF stack is not empty THEN  
                    REMOVE last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character slash concatenated WITH all elements in stack joined by the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
