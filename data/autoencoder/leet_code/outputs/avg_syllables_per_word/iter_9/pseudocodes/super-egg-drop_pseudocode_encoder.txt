CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        SET dp TO a list of zeros with length k PLUS one  
        SET moves TO zero  
        
        WHILE element at index k in dp LESS THAN n  
            INCREMENT moves BY one  
            FOR eggs FROM k DOWN TO one  
                SET element at index eggs in dp TO one PLUS element at index eggs MINUS one in dp PLUS element at index eggs in dp  
            END FOR  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
