CLASS Solution  
    FUNCTION findItinerary(self, tickets)  
        SET graph TO a new mapping from keys to lists with default empty list  
        FOR each pair FROM to IN the list of tickets sorted in ascending order and then reversed  
            APPEND to TO the list at graph indexed by FROM  
        END FOR  

        SET itinerary TO an empty list  

        FUNCTION visit(airport)  
            WHILE the list at graph indexed by airport is not empty  
                CALL visit WITH the last element popped from the list at graph indexed by airport  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  

        CALL visit WITH "JFK"  
        RETURN the reversed itinerary  
    END FUNCTION  
END CLASS
