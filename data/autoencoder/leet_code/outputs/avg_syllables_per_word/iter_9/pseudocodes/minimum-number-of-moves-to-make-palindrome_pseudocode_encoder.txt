CLASS Solution  
    FUNCTION minMovesToMakePalindrome(self, s)  
        SET s TO list of characters in s  
        SET n TO length of s  
        SET moves TO zero  
        
        WHILE length of s GREATER THAN one DO  
            FOR i FROM length of s MINUS one DOWN TO one STEP negative one DO  
                IF element at index i of s EQUALS element at index zero of s THEN  
                    FOR j FROM i TO length of s MINUS two DO  
                        SWAP element at index j of s WITH element at index j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE first element FROM s  
                    REMOVE last element FROM s  
                    BREAK out of FOR loop  
                END IF  
            END FOR  
            IF loop completed without BREAK THEN  
                REMOVE first element FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
