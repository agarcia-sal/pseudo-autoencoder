CLASS Solution  
    FUNCTION findRightInterval(self, intervals)  
        SET start_times TO sorted list of pairs where each pair is the first element of interval and its index in intervals  
        SET result TO empty list  
        FOR each interval IN intervals  
            SET end TO the second element of interval  
            SET idx TO position found by binary search for the smallest start time greater than or equal to end in start_times  
            IF idx LESS THAN length of start_times THEN  
                APPEND the index part of start_times at position idx TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
