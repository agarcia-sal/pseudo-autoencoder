CLASS Solution  
    FUNCTION makeArrayIncreasing(self, arr1, arr2)  
        CALL sort on arr2 to arrange elements in ascending order  
        
        SET dp TO a dictionary mapping -1 TO 0  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each last_num AND ops IN dp  
                IF num GREATER THAN last_num THEN  
                    SET new_dp at key num TO the minimum of new_dp at key num IF EXISTS OTHERWISE infinity, AND ops  
                END IF  
                
                SET idx TO the position to insert last_num on the right side in arr2 to maintain order  
                IF idx LESS THAN length of arr2 THEN  
                    SET new_dp at key arr2 at idx TO the minimum of new_dp at key arr2 at idx IF EXISTS OTHERWISE infinity, AND ops PLUS 1  
                END IF  
            END FOR  
            
            IF new_dp is empty THEN  
                RETURN negative 1  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
