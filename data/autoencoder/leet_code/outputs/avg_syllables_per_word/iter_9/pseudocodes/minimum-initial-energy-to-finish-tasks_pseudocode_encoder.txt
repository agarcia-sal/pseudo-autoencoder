CLASS Solution  
    FUNCTION minimumEffort(self, tasks)  
        SORT tasks BY difference between second element MINUS first element IN descending order  
        
        SET current_energy TO zero  
        SET required_initial_energy TO zero  
        
        FOR each pair actual AND minimum IN tasks  
            IF current_energy LESS THAN minimum THEN  
                INCREMENT required_initial_energy BY difference between minimum MINUS current_energy  
                SET current_energy TO minimum  
            END IF  
            
            DECREMENT current_energy BY actual  
        END FOR  
        
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
