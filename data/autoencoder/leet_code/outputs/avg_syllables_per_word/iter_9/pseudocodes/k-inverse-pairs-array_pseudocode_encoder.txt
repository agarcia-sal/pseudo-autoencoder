CLASS Solution  
    FUNCTION kInversePairs(self, n, k)  
        SET MOD TO ten to the power of nine PLUS seven  
        INITIALIZE dp AS a two dimensional list of size (n PLUS one) BY (k PLUS one) filled with zeroes  
        SET dp at position zero zero TO one  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS dp at position (i MINUS one) j, MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS dp at position (i MINUS one) (j MINUS i) PLUS MOD, MODULO MOD  
                END IF  
                SET dp at position i j TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN dp at position n k  
    END FUNCTION  
END CLASS
