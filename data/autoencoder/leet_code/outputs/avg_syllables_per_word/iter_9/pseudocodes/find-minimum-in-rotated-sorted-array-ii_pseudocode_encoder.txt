CLASS Solution  
    FUNCTION findMin(self, nums)  
        SET left TO 0  
        SET right TO length of nums MINUS 1  
        
        WHILE left LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY 2 (using integer division)  
            
            IF element at index mid of nums GREATER THAN element at index right of nums THEN  
                SET left TO mid PLUS 1  
            ELSE IF element at index mid of nums LESS THAN element at index right of nums THEN  
                SET right TO mid  
            ELSE  
                DECREMENT right BY 1  
            END IF  
        END WHILE  
        
        RETURN element at index left of nums  
    END FUNCTION  
END CLASS
