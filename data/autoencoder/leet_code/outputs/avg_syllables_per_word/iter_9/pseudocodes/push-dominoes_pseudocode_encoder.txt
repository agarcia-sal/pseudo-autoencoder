CLASS Solution  
    FUNCTION pushDominoes(self, dominoes)  
        SET n TO length of dominoes  
        SET forces TO list of zeros with length n  

        SET force TO 0  
        FOR i FROM 0 TO n MINUS 1  
            IF dominoes at position i EQUALS 'R' THEN  
                SET force TO n  
            ELSE IF dominoes at position i EQUALS 'L' THEN  
                SET force TO 0  
            ELSE  
                SET force TO the greater value between force MINUS 1 and 0  
            END IF  
            INCREMENT forces at position i BY force  
        END FOR  

        SET force TO 0  
        FOR i FROM n MINUS 1 DOWN TO 0  
            IF dominoes at position i EQUALS 'L' THEN  
                SET force TO n  
            ELSE IF dominoes at position i EQUALS 'R' THEN  
                SET force TO 0  
            ELSE  
                SET force TO the greater value between force MINUS 1 and 0  
            END IF  
            DECREMENT forces at position i BY force  
        END FOR  

        SET result TO empty list  
        FOR each f IN forces  
            IF f GREATER THAN 0 THEN  
                APPEND 'R' TO result  
            ELSE IF f LESS THAN 0 THEN  
                APPEND 'L' TO result  
            ELSE  
                APPEND '.' TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
