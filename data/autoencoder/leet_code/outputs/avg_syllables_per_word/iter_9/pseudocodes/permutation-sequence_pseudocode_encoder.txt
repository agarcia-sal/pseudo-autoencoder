CLASS Solution  
    FUNCTION getPermutation(self, n, k)  
        IMPORT math  
        SET numbers TO a list of numbers from one TO n  
        SET k TO k MINUS one  
        SET result TO an empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO factorial of (i MINUS one)  
            SET index TO integer division of k BY fact  
            APPEND string representation of numbers at position index TO result  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        RETURN concatenation of elements in result into a string  
    END FUNCTION  
END CLASS
