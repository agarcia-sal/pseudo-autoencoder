CLASS Solution  
    FUNCTION maxProduct(self, words)  
        SET word_sets TO empty list  
        FOR each word IN words  
            APPEND set of characters in word TO word_sets  
        END FOR  

        SET max_product TO zero  

        FOR i FROM zero TO length of words MINUS one  
            FOR j FROM i PLUS one TO length of words MINUS one  
                IF intersection of word_sets at i AND word_sets at j IS empty THEN  
                    SET product TO length of word at i MULTIPLIED BY length of word at j  
                    IF product GREATER THAN max_product THEN  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN max_product  
    END FUNCTION  
END CLASS
