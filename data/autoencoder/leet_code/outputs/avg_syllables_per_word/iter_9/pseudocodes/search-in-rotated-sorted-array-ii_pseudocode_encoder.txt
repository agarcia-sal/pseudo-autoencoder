CLASS Solution  
    FUNCTION search(self, nums, target)  
        IF nums EQUALS empty THEN  
            RETURN False  
        END IF  
        
        SET left TO zero  
        SET right TO length of nums MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the integer division of (right MINUS left) BY two  
            
            IF the element of nums at index mid EQUALS target THEN  
                RETURN True  
            END IF  
            
            IF the element of nums at index left EQUALS the element of nums at index mid AND the element of nums at index mid EQUALS the element of nums at index right THEN  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF the element of nums at index left LESS THAN OR EQUAL TO the element of nums at index mid THEN  
                IF the element of nums at index left LESS THAN OR EQUAL TO target AND target LESS THAN the element of nums at index mid THEN  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF the element of nums at index mid LESS THAN target AND target LESS THAN OR EQUAL TO the element of nums at index right THEN  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
