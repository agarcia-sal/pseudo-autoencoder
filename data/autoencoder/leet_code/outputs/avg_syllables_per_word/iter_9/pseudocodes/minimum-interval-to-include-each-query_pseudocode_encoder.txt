CLASS Solution  
    FUNCTION minInterval(self, intervals, queries)  
        SORT intervals BY their first element  
        SET sorted_queries TO list of pairs of each element in queries with its index, sorted BY the query value  
  
        SET min_heap TO empty list  
        SET result TO list of length equal to queries filled with minus one  
  
        SET interval_index TO zero  
        FOR each query_and_index IN sorted_queries  
            SET query TO first element of query_and_index  
            SET original_index TO second element of query_and_index  
            
            WHILE interval_index LESS THAN length of intervals AND first element of intervals at interval_index LESS THAN OR EQUAL TO query  
                SET start TO first element of intervals at interval_index  
                SET end TO second element of intervals at interval_index  
                SET interval_size TO end MINUS start PLUS one  
                INSERT tuple of interval_size and end INTO min_heap maintaining heap property  
                INCREMENT interval_index BY one  
            END WHILE  
            
            WHILE min_heap NOT EMPTY AND second element of the smallest element in min_heap LESS THAN query  
                REMOVE the smallest element FROM min_heap maintaining heap property  
            END WHILE  
            
            IF min_heap NOT EMPTY THEN  
                SET element at original_index in result TO first element of the smallest element in min_heap  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
