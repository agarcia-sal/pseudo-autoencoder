CLASS Solution  
    FUNCTION minAbsDifference(self, nums, goal)  
        SET n TO length of nums  
        SET mid TO n DIVIDED BY 2  
        
        SET left_sums TO list containing zero  
        FOR each num IN first mid elements of nums  
            SET new_sums TO empty list  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND left_sums BY new_sums  
        END FOR  
        
        SET right_sums TO list containing zero  
        FOR each num IN elements of nums from mid to end  
            SET new_sums TO empty list  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND right_sums BY new_sums  
        END FOR  
        
        SORT right_sums  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO result of binary searching for target in right_sums to find leftmost insertion point  
            
            IF pos LESS THAN length of right_sums THEN  
                SET candidate_diff TO absolute value of left_sum PLUS right_sums at position pos MINUS goal  
                IF candidate_diff LESS THAN min_diff THEN  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            
            IF pos GREATER THAN zero THEN  
                SET candidate_diff TO absolute value of left_sum PLUS right_sums at position pos MINUS 1 MINUS goal  
                IF candidate_diff LESS THAN min_diff THEN  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
