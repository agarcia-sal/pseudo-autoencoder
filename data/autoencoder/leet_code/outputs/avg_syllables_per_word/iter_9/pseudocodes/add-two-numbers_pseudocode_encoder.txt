CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH val AS first element IN values AND next AS None  
    SET p TO head  
    FOR each val IN values FROM second element TO last element  
        SET node TO new ListNode WITH val AS val AND next AS None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        SET dummy_head TO new ListNode WITH val AS zero AND next AS None  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry GREATER THAN zero  
            IF l1 IS NOT None THEN  
                SET val1 TO l1.val  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None THEN  
                SET val2 TO l2.val  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode WITH val AS remainder of total DIVIDED BY ten AND next AS None  
            SET current TO current.next  
            
            IF l1 IS NOT None THEN  
                SET l1 TO l1.next  
            END IF  
            IF l2 IS NOT None THEN  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
