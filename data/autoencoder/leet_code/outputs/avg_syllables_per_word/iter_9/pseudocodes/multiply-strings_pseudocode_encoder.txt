CLASS Solution  
    FUNCTION multiply(self, num1, num2)  
        IF num1 EQUALS "0" OR num2 EQUALS "0" THEN  
            RETURN "0"  
        END IF  
        
        SET result TO list of zeros WITH length EQUAL TO sum of length of num1 AND length of num2  
        
        REVERSE num1  
        REVERSE num2  
        
        FOR i FROM zero TO length of num1 MINUS one  
            FOR j FROM zero TO length of num2 MINUS one  
                SET mul TO digit value of character at position i in num1 MULTIPLIED BY digit value of character at position j in num2  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS result at position p1  
                
                SET result at position p1 TO sum_ MODULO ten  
                INCREMENT result at position p2 BY integer division of sum_ BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO concatenation of all elements in result reversed, each converted to string  
        RETURN result_str with all leading zeros removed  
    END FUNCTION  
END CLASS
