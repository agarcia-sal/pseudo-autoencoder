CLASS Solution  
    FUNCTION replaceNonCoprimes(self, nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE stack IS NOT empty  
                SET g TO greatest common divisor OF last element IN stack AND num  
                IF g EQUALS one THEN  
                    BREAK  
                END IF  
                SET num TO multiplication OF last element IN stack AND num DIVIDED BY g  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
