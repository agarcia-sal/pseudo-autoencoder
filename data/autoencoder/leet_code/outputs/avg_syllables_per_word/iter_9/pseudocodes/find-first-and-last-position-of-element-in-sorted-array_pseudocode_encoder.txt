CLASS Solution  
    FUNCTION searchRange(self, nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO 0  
            SET right TO length of nums MINUS 1  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY 2, INTEGER DIVISION  
                IF nums at mid LESS THAN target THEN  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
                END IF  
            END WHILE  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO 0  
            SET right TO length of nums MINUS 1  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY 2, INTEGER DIVISION  
                IF nums at mid LESS THAN OR EQUAL TO target THEN  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
                END IF  
            END WHILE  
            RETURN right  
        END FUNCTION  

        SET left_index TO find_left(nums, target)  
        SET right_index TO find_right(nums, target)  

        IF left_index LESS THAN OR EQUAL TO right_index AND right_index GREATER THAN OR EQUAL TO 0 AND right_index LESS THAN length of nums AND nums at left_index EQUALS target THEN  
            RETURN list containing left_index and right_index  
        ELSE  
            RETURN list containing negative one and negative one  
        END IF  
    END FUNCTION  
END CLASS
