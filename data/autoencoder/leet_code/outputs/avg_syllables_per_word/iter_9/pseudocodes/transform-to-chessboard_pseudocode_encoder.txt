CLASS Solution  
    FUNCTION movesToChessboard(self, board)  
        SET N TO length of board  
        
        SET rowSum TO 0  
        SET colSum TO 0  
        SET rowSwap TO 0  
        SET colSwap TO 0  
        
        FOR i FROM 0 TO N MINUS 1  
            FOR j FROM 0 TO N MINUS 1  
                IF board at i and j XOR board at i and zero XOR board at zero and j XOR board at zero and zero IS NOT EQUAL TO zero THEN  
                    RETURN negative one  
                END IF  
            END FOR  
            
            SET rowSum TO rowSum PLUS board at i and zero  
            SET colSum TO colSum PLUS board at zero and i  
            SET rowSwap TO rowSwap PLUS comparison of board at i and zero EQUALS i MODULO two  
            SET colSwap TO colSwap PLUS comparison of board at zero and i EQUALS i MODULO two  
        END FOR  
        
        IF integer division of N by two GREATER THAN rowSum OR rowSum GREATER THAN integer division of N plus one by two THEN  
            RETURN negative one  
        END IF  
        
        IF integer division of N by two GREATER THAN colSum OR colSum GREATER THAN integer division of N plus one by two THEN  
            RETURN negative one  
        END IF  
        
        IF N MODULO two IS EQUAL TO one THEN  
            IF rowSwap MODULO two IS EQUAL TO one THEN  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF colSwap MODULO two IS EQUAL TO one THEN  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO minimum of rowSwap and N MINUS rowSwap  
            SET colSwap TO minimum of colSwap and N MINUS colSwap  
        END IF  
        
        RETURN integer division of rowSwap PLUS colSwap by two  
    END FUNCTION  
END CLASS
