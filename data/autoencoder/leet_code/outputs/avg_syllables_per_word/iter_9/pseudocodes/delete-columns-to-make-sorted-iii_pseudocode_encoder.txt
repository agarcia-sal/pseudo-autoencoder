CLASS Solution  
    FUNCTION minDeletionSize(self, strs)  
        SET m TO length of strs  
        SET n TO length of first element in strs  
        SET dp TO list of n elements each set to one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for all k FROM zero TO m MINUS one the element at position j in strs at index k LESS THAN OR EQUAL TO element at position i in strs at index k THEN  
                    SET dp at position i TO the greater value between dp at position i and dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the greatest value in dp  
    END FUNCTION  
END CLASS
