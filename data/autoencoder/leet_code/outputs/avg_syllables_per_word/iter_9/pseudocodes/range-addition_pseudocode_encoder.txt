CLASS Solution  
    FUNCTION getModifiedArray(self, length, updates)  
        SET diff TO a list of zeros with size length PLUS one  
        FOR each startIdx, endIdx, inc IN updates  
            INCREMENT element at position startIdx in diff BY inc  
            IF endIdx PLUS one LESS THAN length THEN  
                DECREMENT element at position endIdx PLUS one in diff BY inc  
            END IF  
        END FOR  
        SET arr TO a list of zeros with size length  
        SET element at position zero in arr TO element at position zero in diff  
        FOR i FROM one TO length MINUS one  
            SET element at position i in arr TO element at position i MINUS one in arr PLUS element at position i in diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
