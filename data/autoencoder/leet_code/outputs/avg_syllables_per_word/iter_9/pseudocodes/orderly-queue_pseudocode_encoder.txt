CLASS Solution  
    FUNCTION orderlyQueue(self, s, k)  
        IF k GREATER THAN 1 THEN  
            RETURN concatenation of characters in s sorted in lexicographical order  
        ELSE  
            SET n TO length of s  
            SET smallest TO s  
            FOR i FROM 1 TO n MINUS 1  
                SET rotated TO substring of s from index i to end CONCATENATED WITH substring of s from start to index i MINUS 1  
                IF rotated LESS THAN smallest THEN  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
