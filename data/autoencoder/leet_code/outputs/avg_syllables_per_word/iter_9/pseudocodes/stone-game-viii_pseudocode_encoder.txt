CLASS Solution  
    FUNCTION stoneGameVIII(self, stones)  
        SET prefix_sum TO a list of zeroes with length equal to length of stones  
        SET prefix_sum at index zero TO stones at index zero  
        FOR index FROM one TO length of stones MINUS one  
            SET prefix_sum at index TO prefix_sum at index MINUS one PLUS stones at index  
        END FOR  
        
        SET dp TO last element in prefix_sum  
        
        FOR index FROM length of stones MINUS two DOWN TO one  
            SET dp TO the greater value BETWEEN dp AND prefix_sum at index MINUS dp  
        END FOR  
        
        RETURN dp  
    END FUNCTION  
END CLASS
