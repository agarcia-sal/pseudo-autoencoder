CLASS Solution  
    FUNCTION findStrobogrammatic(self, n)  
        FUNCTION dfs(u)  
            IF u EQUALS 0 THEN  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS 1 THEN  
                RETURN list containing string "0" , string "1" , string "8"  
            END IF  
            SET ans TO empty list  
            FOR each v IN dfs(u MINUS 2)  
                FOR each pair (l, r) IN list of pairs ("1", "1"), ("8", "8"), ("6", "9"), ("9", "6")  
                    APPEND concatenation of l, v, and r TO ans  
                END FOR  
                IF u NOT EQUALS n THEN  
                    APPEND concatenation of "0", v, and "0" TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs(n)  
    END FUNCTION  
END CLASS
