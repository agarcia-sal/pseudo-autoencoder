CLASS Solution  
    FUNCTION reachingPoints(self, sx, sy, tx, ty)  
        WHILE tx GREATER THAN OR EQUAL TO sx AND ty GREATER THAN OR EQUAL TO sy  
            IF tx EQUALS sx AND ty EQUALS sy THEN  
                RETURN True  
            END IF  
            IF tx GREATER THAN ty THEN  
                IF ty GREATER THAN sy THEN  
                    SET tx TO tx MODULO ty  
                ELSE  
                    RETURN the result of (tx MINUS sx) MODULO ty EQUALS zero  
                END IF  
            ELSE  
                IF tx GREATER THAN sx THEN  
                    SET ty TO ty MODULO tx  
                ELSE  
                    RETURN the result of (ty MINUS sy) MODULO tx EQUALS zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
