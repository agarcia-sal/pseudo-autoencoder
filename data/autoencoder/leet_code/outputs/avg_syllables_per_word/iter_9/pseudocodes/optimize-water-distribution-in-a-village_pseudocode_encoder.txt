CLASS Solution
    FUNCTION minCostToSupplyWater(self, n, wells, pipes)
        FOR i FROM zero TO n MINUS one
            APPEND list containing zero, i PLUS one, wells AT i TO pipes
        END FOR

        SORT pipes BY third element of each item

        SET parent TO list of integers FROM zero TO n

        FUNCTION find(x)
            IF parent AT x NOT EQUALS x THEN
                SET parent AT x TO find(parent AT x)
            END IF
            RETURN parent AT x
        END FUNCTION

        FUNCTION union(x, y)
            SET rootX TO find(x)
            SET rootY TO find(y)
            IF rootX NOT EQUALS rootY THEN
                SET parent AT rootX TO rootY
                RETURN True
            END IF
            RETURN False
        END FUNCTION

        SET min_cost TO zero
        FOR each element IN pipes
            DESTRUCTURE element INTO house1, house2, cost
            IF union(house1, house2) THEN
                SET min_cost TO min_cost PLUS cost
            END IF
        END FOR

        RETURN min_cost
    END FUNCTION
END CLASS
