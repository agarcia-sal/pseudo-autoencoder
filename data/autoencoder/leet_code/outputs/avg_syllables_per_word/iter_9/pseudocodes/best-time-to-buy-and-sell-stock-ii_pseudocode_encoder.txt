CLASS Solution  
    FUNCTION maxProfit(self, prices)  
        SET max_profit TO zero  
        FOR index FROM one TO length of prices MINUS one  
            IF element at prices at index GREATER THAN element at prices at index MINUS one THEN  
                INCREMENT max_profit BY element at prices at index MINUS element at prices at index MINUS one  
            END IF  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
