CLASS Solution  
    FUNCTION searchMatrix(self, matrix, target)  
        IF matrix IS empty OR first element of matrix IS empty THEN  
            RETURN False  
        END IF  

        SET m TO length of matrix  
        SET n TO length of first element of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  

        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO integer division of (left PLUS right) BY two  
            SET mid_value TO element in matrix at row integer division of mid BY n and column mid MODULO n  

            IF mid_value EQUALS target THEN  
                RETURN True  
            ELSE IF mid_value LESS THAN target THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  

        RETURN False  
    END FUNCTION  
END CLASS
