CLASS Solution  
    FUNCTION kIncreasing(self, arr, k)  
        FUNCTION longest_increasing_subsequence(subarr)  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis is empty OR num GREATER THAN OR EQUALS last element in lis THEN  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO position to insert num to the right of equals in lis  
                    SET element at position pos in lis TO num  
                END IF  
            END FOR  
            RETURN length of lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO length of arr  
        
        FOR start FROM zero TO one less than k  
            SET subarr TO empty list  
            FOR i FROM start TO less than n STEP BY k  
                APPEND arr at index i TO subarr  
            END FOR  
            
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY length of subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
