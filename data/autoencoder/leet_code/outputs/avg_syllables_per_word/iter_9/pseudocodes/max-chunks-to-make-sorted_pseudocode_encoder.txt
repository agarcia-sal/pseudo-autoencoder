CLASS Solution  
    FUNCTION maxChunksToSorted(self, arr)  
        SET max_so_far TO zero  
        SET chunks TO zero  
        
        FOR i FROM zero TO length of arr MINUS one  
            SET max_so_far TO greater value BETWEEN max_so_far AND element at index i IN arr  
            IF max_so_far EQUALS i THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
