CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val SET TO first element of values  
    SET i TO one  
    INITIALIZE queue AS an empty dequeue structure  
    APPEND root TO queue  
    WHILE queue IS not empty DO  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val SET TO values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val SET TO values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        INITIALIZE result AS empty list  
        INITIALIZE queue AS dequeue containing root  
        WHILE queue IS not empty DO  
            SET level_size TO length of queue  
            INITIALIZE current_level AS empty list  
            FOR counter FROM zero TO one LESS THAN level_size DO  
                SET node TO element removed from front of queue  
                APPEND node.val TO current_level  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result reversed in order  
    END FUNCTION  
END CLASS
