CLASS Solution  
    FUNCTION nextClosestTime(self, time)  
        SET digits TO the set of characters from time after removing the colon  
        SET hour AND minute TO the integer values obtained by splitting time by colon  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET hour TO the integer value of concatenating h1 and h2  
                            SET minute TO the integer value of concatenating m1 and m2  
                            IF hour LESS THAN 24 AND minute LESS THAN 60 THEN  
                                APPEND the tuple of (hour, minute) TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the sorted list times  
        END FUNCTION  

        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the index of the tuple (hour, minute) in all_times  

        IF current_time_index PLUS 1 LESS THAN the length of all_times THEN  
            SET next_hour AND next_minute TO the element in all_times at position current_time_index PLUS 1  
        ELSE  
            SET next_hour AND next_minute TO the first element in all_times  
        END IF  

        RETURN the string of next_hour formatted as two digits, concatenated with a colon, concatenated with next_minute formatted as two digits  
    END FUNCTION  
END CLASS
