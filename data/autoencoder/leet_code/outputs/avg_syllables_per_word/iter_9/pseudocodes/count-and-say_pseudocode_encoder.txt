CLASS Solution  
    FUNCTION countAndSay(self, n)  
        IF n EQUALS 1 THEN  
            RETURN the string "1"  
        END IF  
        
        SET current_sequence TO the string "1"  
        
        FOR each number FROM 1 TO n MINUS 1 DO  
            SET next_sequence TO the empty string  
            SET count TO 1  
            SET previous_char TO the first character of current_sequence  
            
            FOR index FROM 1 TO the length of current_sequence MINUS 1 DO  
                IF the character at current_sequence at position index EQUALS previous_char THEN  
                    INCREMENT count BY 1  
                ELSE  
                    APPEND the string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO the character at current_sequence at position index  
                    SET count TO 1  
                END IF  
            END FOR  
            
            APPEND the string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            
            SET current_sequence TO next_sequence  
        END FOR  
        
        RETURN current_sequence  
    END FUNCTION  
END CLASS
