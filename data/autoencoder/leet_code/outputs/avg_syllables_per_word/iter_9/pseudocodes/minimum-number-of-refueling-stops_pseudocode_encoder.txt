CLASS Solution  
    FUNCTION minRefuelStops(self, target, startFuel, stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a pair of target and zero TO stations  

        FOR each position, fuel IN stations  
            WHILE max_heap is not empty AND current_fuel LESS THAN position  
                REMOVE the largest available fuel amount FROM max_heap  
                INCREMENT current_fuel BY removed largest available fuel amount  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position THEN  
                RETURN negative one  
            END IF  
            ADD negative of fuel TO max_heap to simulate max-heap behavior  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
