CLASS Solution  
    FUNCTION generatePalindromes(self, s)  
        FUNCTION dfs(t)  
            IF length of t EQUALS length of s THEN  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND value v IN cnt  
                IF v GREATER THAN 1 THEN  
                    DECREMENT cnt[c] BY 2  
                    CALL dfs WITH concatenation of c, t, and c  
                    INCREMENT cnt[c] BY 2  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO a count mapping of characters in s  
        SET mid TO empty string  
        FOR each character c AND value v IN cnt  
            IF v modulo 2 NOT EQUALS 0 THEN  
                IF mid IS NOT empty THEN  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT cnt[c] BY 1  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
