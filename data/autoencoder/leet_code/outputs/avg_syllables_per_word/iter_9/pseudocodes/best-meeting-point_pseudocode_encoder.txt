CLASS Solution  
    FUNCTION minTotalDistance(self, grid)  
        SET rows TO empty list  
        SET cols TO empty list  
        
        FOR i FROM zero TO length of grid MINUS one  
            FOR j FROM zero TO length of first element of grid MINUS one  
                IF element at grid position i and j EQUALS one THEN  
                    APPEND i TO rows  
                    APPEND j TO cols  
                END IF  
            END FOR  
        END FOR  
        
        SORT rows  
        SORT cols  
        
        SET median_row TO element of rows at position length of rows DIVIDED BY two  
        SET median_col TO element of cols at position length of cols DIVIDED BY two  
        
        SET min_distance TO zero  
        
        FOR each row IN rows  
            SET min_distance TO min_distance PLUS absolute difference between row and median_row  
        END FOR  
        
        FOR each col IN cols  
            SET min_distance TO min_distance PLUS absolute difference between col and median_col  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
