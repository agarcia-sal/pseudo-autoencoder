CLASS Solution  
    FUNCTION twoSum(self, numbers, target)  
        SET left TO zero  
        SET right TO length of numbers MINUS one  
  
        WHILE left LESS THAN right  
            SET current_sum TO element at numbers indexed by left PLUS element at numbers indexed by right  
            IF current_sum EQUALS target THEN  
                RETURN list containing left PLUS one and right PLUS one  
            ELSE IF current_sum LESS THAN target THEN  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
