CLASS Solution  
    FUNCTION maxPerformance(self, n, speed, efficiency, k)  
        SET engineers TO list of pairs combining efficiency and speed elements element-wise  
        SORT engineers in order of efficiency from highest to lowest  
        
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        
        FOR each pair of eff and spd IN engineers  
            ADD spd TO speed_heap in a way that maintains minimum at top  
            INCREMENT speed_sum BY spd  
            
            IF size of speed_heap GREATER THAN k THEN  
                REMOVE the smallest speed from speed_heap and STORE it as removed_speed  
                DECREMENT speed_sum BY removed_speed  
            END IF  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the greater value between max_performance and current_performance  
        END FOR  
        
        RETURN max_performance MODULO (10 raised to the power of nine PLUS seven)  
    END FUNCTION  
END CLASS
