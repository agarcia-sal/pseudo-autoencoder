CLASS Solution  
    FUNCTION minimumOneBitOperations(self, n)  
        IF n EQUALS 0 THEN  
            RETURN 0  
        END IF  

        SET k TO 0  
        WHILE one shifted left by k PLUS 1 LESS THAN OR EQUAL TO n DO  
            INCREMENT k BY 1  
        END WHILE  

        SET most_significant_bit TO one shifted left by k  

        RETURN one shifted left by k PLUS 1 MINUS 1 MINUS self.minimumOneBitOperations(n EXCLUSIVE OR most_significant_bit)  
    END FUNCTION  
END CLASS
