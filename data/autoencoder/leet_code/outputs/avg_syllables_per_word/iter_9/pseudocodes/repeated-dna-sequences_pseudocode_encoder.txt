CLASS Solution  
    FUNCTION findRepeatedDnaSequences(self, s)  
        SET sequence_count TO an empty dictionary with default integer value  
        
        FOR index FROM zero TO length of s MINUS nine DO  
            SET sequence TO substring of s from index to index PLUS ten  
            INCREMENT the value of sequence_count at sequence BY one  
        END FOR  
        
        SET repeated_sequences TO an empty list  
        FOR each pair of sequence and count IN sequence_count items DO  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
