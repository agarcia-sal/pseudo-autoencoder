CLASS Solution  
    FUNCTION shortestSubarray(self, nums, k)  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            SET last_element TO the last element in prefix_sum  
            SET new_sum TO last_element PLUS num  
            APPEND new_sum TO prefix_sum  
        END FOR  
        
        INITIALIZE dq AS an empty double ended queue  
        SET min_length TO infinity  
        
        FOR each index i AND element current_sum IN prefix_sum WITH indices  
            
            WHILE dq is not empty AND current_sum MINUS prefix_sum at the first element of dq IS GREATER THAN OR EQUAL TO k  
                SET first_index TO the first element removed from dq  
                SET candidate_length TO i MINUS first_index  
                IF candidate_length LESS THAN min_length THEN  
                    SET min_length TO candidate_length  
                END IF  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO prefix_sum at the last element of dq  
                REMOVE the last element FROM dq  
            END WHILE  
            
            ADD i TO the end of dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO infinity THEN  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
