CLASS Solution  
    FUNCTION findSubstringInWraproundString(self, s)  
        SET max_length TO dictionary mapping each character from 'a' to 'z' TO zero  
        SET current_length TO zero  
        
        FOR index FROM zero TO length of s MINUS one  
            IF index GREATER THAN zero AND (difference between ordinal value of character at s[index] and character at s[index minus one] EQUALS one OR character at s[index] EQUALS 'a' AND character at s[index minus one] EQUALS 'z') THEN  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            
            SET max_length at key character s[index] TO the greater value between current max_length at character s[index] AND current_length  
        END FOR  
        
        SET total TO zero  
        FOR each value IN max_length values  
            INCREMENT total BY value  
        END FOR  
        
        RETURN total  
    END FUNCTION  
END CLASS
