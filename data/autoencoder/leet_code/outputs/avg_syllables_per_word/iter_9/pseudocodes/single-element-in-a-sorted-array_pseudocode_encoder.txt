CLASS Solution  
    FUNCTION singleNonDuplicate(self, nums)  
        SET left TO zero  
        SET right TO length of nums MINUS one  
        
        WHILE left LESS THAN right  
            SET mid TO left PLUS the integer division of (right MINUS left) BY two  
            
            IF mid MODULO two EQUALS one THEN  
                SET mid TO mid MINUS one  
            END IF  
            
            IF element at index mid in nums EQUALS element at index mid PLUS one in nums THEN  
                SET left TO mid PLUS two  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN element at index left in nums  
    END FUNCTION  
END CLASS
