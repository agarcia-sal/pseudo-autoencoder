CLASS Solution  
    FUNCTION solveEquation(self, equation)  
        FUNCTION parse_side(side)  
            SET tokens TO side with all plus signs replaced by delimiter plus sign and minus signs replaced by delimiter minus sign, then split by delimiter  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS empty THEN  
                    CONTINUE to next iteration  
                END IF  
                IF token CONTAINS character x THEN  
                    IF token EQUALS character x OR token EQUALS plus sign FOLLOWED BY character x THEN  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS minus sign FOLLOWED BY character x THEN  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET x_count TO x_count PLUS integer value of token WITH character x REMOVED  
                    END IF  
                ELSE  
                    SET num_sum TO num_sum PLUS integer value of token  
                END IF  
            END FOR  
            RETURN x_count, num_sum  
        END FUNCTION  

        SPLIT equation INTO left_side AND right_side BY character equals sign  
        SET left_x, left_num TO parse_side(left_side)  
        SET right_x, right_num TO parse_side(right_side)  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero THEN  
            IF total_num EQUALS zero THEN  
                RETURN string "Infinite solutions"  
            ELSE  
                RETURN string "No solution"  
            END IF  
        ELSE  
            RETURN string "x=" CONCATENATED WITH string value of total_num DIVIDED BY total_x  
        END IF  
    END FUNCTION  
END CLASS
