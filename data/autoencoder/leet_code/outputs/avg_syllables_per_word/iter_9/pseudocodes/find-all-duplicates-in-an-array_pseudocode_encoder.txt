CLASS Solution  
    FUNCTION findDuplicates(self, nums)  
        FOR i FROM zero TO length of nums MINUS one  
            WHILE value at index i of nums NOT EQUALS value at index (value at index i of nums MINUS one) of nums  
                SET temporary value TO value at index i of nums  
                SET value at index i of nums TO value at index (temporary value MINUS one) of nums  
                SET value at index (temporary value MINUS one) of nums TO temporary value  
            END WHILE  
        END FOR  
        SET result TO empty list  
        FOR each index i and value v IN nums by enumeration  
            IF value v NOT EQUALS index i PLUS one THEN  
                APPEND value v TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
