CLASS Solution  
    FUNCTION largestNumber(self, cost, target)  
        CREATE an empty mapping named cost_to_digit  
        FOR each index i and element c IN cost  
            SET cost_to_digit AT key c TO string representation of i PLUS one  
        END FOR  
        
        CREATE a list named dp with first element as empty string  
        FOR index from one TO target  
            APPEND string representation of minus one TO dp  
        END FOR  
        
        FOR each integer t FROM one TO target  
            FOR each key c and value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND dp element at position (t MINUS c) NOT EQUALS to string minus one THEN  
                    SET candidate TO concatenation of d FOLLOWED BY dp element at position (t MINUS c)  
                    IF dp element at position t EQUALS to string minus one OR length of candidate GREATER THAN length of dp element at position t OR (length of candidate EQUALS length of dp element at position t AND candidate GREATER THAN dp element at position t) THEN  
                        SET dp element at position t TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF dp element at position target NOT EQUALS to string minus one THEN  
            RETURN dp element at position target  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
