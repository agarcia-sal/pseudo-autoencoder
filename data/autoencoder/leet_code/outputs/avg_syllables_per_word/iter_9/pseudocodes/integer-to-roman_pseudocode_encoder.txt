CLASS Solution  
    FUNCTION intToRoman(self, num)  
        SET val TO list of values: one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, one  
        SET syms TO list of symbols: "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"  
        
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            SET count TO integer division of num BY val at position i  
            FOR each _ IN range from zero TO count MINUS one  
                APPEND syms at position i TO roman_numeral  
                SET num TO num MINUS val at position i  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
