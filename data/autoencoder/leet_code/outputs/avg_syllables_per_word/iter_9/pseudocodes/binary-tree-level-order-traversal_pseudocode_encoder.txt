CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO TreeNode with val AS first element in values  
    SET i TO one  
    INITIALIZE queue AS empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN size of values AND value at position i IS NOT none THEN  
            SET node.left TO TreeNode with val AS value at position i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN size of values AND value at position i IS NOT none THEN  
            SET node.right TO TreeNode with val AS value at position i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS none THEN  
            RETURN empty list  
        END IF  
        INITIALIZE result AS empty list  
        INITIALIZE queue AS double-ended queue containing root  
        WHILE queue IS NOT empty DO  
            SET level_size TO length of queue  
            INITIALIZE current_level AS empty list  
            FOR counter FROM one TO level_size DO  
                SET node TO first element removed FROM queue  
                APPEND value of node TO current_level  
                IF left child of node EXISTS THEN  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node EXISTS THEN  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
