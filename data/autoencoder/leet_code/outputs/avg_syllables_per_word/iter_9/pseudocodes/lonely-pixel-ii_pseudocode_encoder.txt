CLASS Solution  
    FUNCTION findBlackPixel(self, picture, target)  
        IF picture IS empty OR first element of picture IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of picture  
        SET n TO length of first element of picture  
        SET row_count TO list of m zeros  
        SET col_count TO list of n zeros  
        SET row_patterns TO a mapping with default integer zero  
        
        FOR each r FROM zero TO m MINUS one  
            SET pattern TO concatenation of all elements in picture at index r  
            INCREMENT value of row_patterns at key pattern BY one  
            FOR each c FROM zero TO n MINUS one  
                IF element in picture at row r and column c EQUALS character B THEN  
                    INCREMENT element in row_count at index r BY one  
                    INCREMENT element in col_count at index c BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR each r FROM zero TO m MINUS one  
            FOR each c FROM zero TO n MINUS one  
                IF element in picture at row r and column c EQUALS character B THEN  
                    SET pattern TO concatenation of all elements in picture at index r  
                    IF element in row_count at index r EQUALS target AND element in col_count at index c EQUALS target AND value of row_patterns at key pattern EQUALS target THEN  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
