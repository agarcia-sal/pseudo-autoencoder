CLASS Solution  
    FUNCTION minWindow(self, s, t)  
        CALL function Counter WITH t RETURNING t_count  
        SET required TO length OF t_count  
        SET left TO 0  
        SET right TO 0  
        SET formed TO 0  
        CALL function Counter WITH no argument RETURNING window_counts  
        SET min_length TO positive infinity  
        SET min_start TO 0  
        
        WHILE right LESS THAN length OF s DO  
            INCREMENT count OF s AT position right IN window_counts BY 1  
            
            IF s AT position right IS IN t_count AND count OF s AT position right IN window_counts EQUALS count OF s AT position right IN t_count THEN  
                INCREMENT formed BY 1  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required DO  
                SET char TO s AT position left  
                
                IF right MINUS left PLUS 1 LESS THAN min_length THEN  
                    SET min_length TO right MINUS left PLUS 1  
                    SET min_start TO left  
                END IF  
                
                DECREMENT count OF char IN window_counts BY 1  
                IF char IS IN t_count AND count OF char IN window_counts LESS THAN count OF char IN t_count THEN  
                    DECREMENT formed BY 1  
                END IF  
                
                INCREMENT left BY 1  
            END WHILE  
            
            INCREMENT right BY 1  
        END WHILE  
        
        IF min_length EQUALS positive infinity THEN  
            RETURN empty string  
        ELSE  
            RETURN substring OF s FROM min_start TO min_start PLUS min_length  
        END IF  
    END FUNCTION  
END CLASS
