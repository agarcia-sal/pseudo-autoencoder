CLASS Solution  
    FUNCTION largestMultipleOfThree(self, digits)  
        SET count TO frequency count of digits  
        SET total_sum TO summation of digits  
        SET remainder TO total_sum modulo 3  
        
        IF remainder EQUALS 1 THEN  
            FOR each digit d IN list of 1, 4, 7  
                IF count of d GREATER THAN 0 THEN  
                    DECREMENT count of d BY 1  
                    BREAK loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN list of 2, 5, 8  
                    IF count of d GREATER THAN 1 THEN  
                        DECREMENT count of d BY 2  
                        BREAK loop  
                    END IF  
                END FOR  
        END IF  
        
        IF remainder EQUALS 2 THEN  
            FOR each digit d IN list of 2, 5, 8  
                IF count of d GREATER THAN 0 THEN  
                    DECREMENT count of d BY 1  
                    BREAK loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN list of 1, 4, 7  
                    IF count of d GREATER THAN 1 THEN  
                        DECREMENT count of d BY 2  
                        BREAK loop  
                    END IF  
                END FOR  
        END IF  
        
        SET result TO empty list  
        FOR digit d FROM 9 DOWN TO 0  
            APPEND string representation of d repeated count of d times TO result  
        END FOR  
        
        SET final_number TO concatenation of all elements in result  
        
        IF final_number IS NOT empty AND first character of final_number EQUALS character zero THEN  
            RETURN character zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
