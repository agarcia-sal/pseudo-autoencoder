CLASS Solution  
    FUNCTION minTaps(self, n, ranges)  
        SET max_range TO a list of zeros with length n PLUS one  
        
        FOR index FROM zero TO n  
            IF value at ranges at index GREATER THAN zero THEN  
                SET left TO the greater value between zero AND index MINUS value at ranges at index  
                SET right TO the lesser value between n AND index PLUS value at ranges at index  
                SET value at max_range at left TO the greater value between value at max_range at left AND right  
            END IF  
        END FOR  
        
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR index FROM zero TO n  
            IF index GREATER THAN farthest THEN  
                RETURN negative one  
            END IF  
            IF index GREATER THAN current_end THEN  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the greater value between farthest AND value at max_range at index  
        END FOR  
        
        RETURN taps  
    END FUNCTION  
END CLASS
