CLASS Solution  
    FUNCTION findKthNumber(self, m, n, k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m inclusive  
                SET count TO count PLUS the minimum between the integer division of x by i AND n  
            END FOR  
            IF count GREATER THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET low TO one  
        SET high TO the product of m AND n  
        WHILE low LESS THAN high  
            SET mid TO the integer division of the sum of low AND high by two  
            IF enough(mid) IS true THEN  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
