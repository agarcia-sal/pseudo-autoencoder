CLASS Solution  
    FUNCTION validArrangement(self, pairs)  
        SET graph TO a mapping with default empty list for missing keys  
        SET out_degree TO a counting mapping with default zero for missing keys  
        SET in_degree TO a counting mapping with default zero for missing keys  
        
        FOR each pair consisting of u and v IN pairs  
            APPEND v TO the list at key u in graph  
            INCREMENT out_degree at key u BY one  
            INCREMENT in_degree at key v BY one  
        END FOR  
        
        SET start_node TO none  
        FOR each node IN graph  
            IF out_degree at node EQUALS in_degree at node PLUS one THEN  
                SET start_node TO node  
                BREAK out of loop  
            END IF  
        END FOR  
        IF start_node EQUALS none THEN  
            SET start_node TO the first node in graph  
        END IF  
        
        SET path TO empty list  
        SET stack TO a list containing start_node  
        
        WHILE stack is not empty  
            SET u TO the last element in stack  
            IF the list at key u in graph is not empty THEN  
                REMOVE the last element from the list at key u in graph and assign it to v  
                APPEND v TO stack  
            ELSE  
                REMOVE the last element from stack and APPEND it TO path  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO empty list  
        FOR index FROM zero TO length of path MINUS one EXCLUSIVE  
            APPEND the pair consisting of element at index in path and element at index PLUS one in path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
