CLASS Solution  
    FUNCTION primePalindrome(self, n)  
        FUNCTION is_prime(num)  
            IF num LESS THAN 2 THEN  
                RETURN False  
            END IF  
            IF num EQUALS 2 OR num EQUALS 3 THEN  
                RETURN True  
            END IF  
            IF num MODULO 2 EQUALS 0 OR num MODULO 3 EQUALS 0 THEN  
                RETURN False  
            END IF  
            SET i TO 5  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num DO  
                IF num MODULO i EQUALS 0 OR num MODULO (i PLUS 2) EQUALS 0 THEN  
                    RETURN False  
                END IF  
                INCREMENT i BY 6  
            END WHILE  
            RETURN True  
        END FUNCTION  

        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reverse of string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO 2 THEN  
            RETURN 2  
        END IF  

        IF n GREATER THAN OR EQUAL TO 10 AND n LESS THAN OR EQUAL TO 11 THEN  
            RETURN 11  
        END IF  

        WHILE True DO  
            IF n MODULO 2 EQUALS 0 THEN  
                INCREMENT n BY 1  
                CONTINUE TO NEXT ITERATION  
            END IF  
            IF is_palindrome(n) AND is_prime(n) THEN  
                RETURN n  
            END IF  
            IF length of string representation of n MODULO 2 EQUALS 0 THEN  
                SET n TO 10 RAISED TO power of length of string representation of n PLUS 1  
            ELSE  
                INCREMENT n BY 2  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
