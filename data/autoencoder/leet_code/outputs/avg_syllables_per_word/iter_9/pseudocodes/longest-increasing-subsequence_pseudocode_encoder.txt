CLASS Solution  
    FUNCTION lengthOfLIS(self, nums)  
        IF nums IS empty THEN  
            RETURN zero  
        END IF  

        SET dp TO empty list  

        FOR each num IN nums  
            SET idx TO position found by binary searching the leftmost insertion point of num in dp  

            IF idx EQUALS length of dp THEN  
                APPEND num TO dp  
            ELSE  
                SET element at index idx in dp TO num  
            END IF  
        END FOR  

        RETURN length of dp  
    END FUNCTION  
END CLASS
