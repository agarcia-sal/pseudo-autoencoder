CLASS Solution  
    FUNCTION characterReplacement(self, s, k)  
        IMPORT a dictionary type that returns zero for missing keys AS count  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a new dictionary that returns zero for missing keys  
        
        FOR right FROM zero TO length of s MINUS one  
            INCREMENT count of character at position right in s BY one  
            SET max_count TO the greater value BETWEEN max_count AND count of character at position right in s  
            
            IF the window size MINUS max_count GREATER THAN k THEN  
                DECREMENT count of character at position left in s BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the greater value BETWEEN max_length AND window size  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
