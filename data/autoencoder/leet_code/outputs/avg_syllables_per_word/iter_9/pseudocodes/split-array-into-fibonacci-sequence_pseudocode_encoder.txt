CLASS Solution  
    FUNCTION splitIntoFibonacci(self, num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS length of num AND length of path GREATER THAN OR EQUAL TO three THEN  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO length of num PLUS one  
                IF character at num index start EQUALS zero AND end GREATER THAN start PLUS one THEN  
                    CONTINUE to next iteration  
                END IF  
                SET next_num TO integer conversion of substring of num from start TO end  
                IF next_num GREATER THAN OR EQUAL TO two raised to the power of thirty one THEN  
                    BREAK loop  
                END IF  
                IF length of path LESS THAN two OR next_num EQUALS sum of last element of path AND second last element of path THEN  
                    SET result TO backtrack(end, concatenation of path AND list containing next_num)  
                    IF result IS NOT empty THEN  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN empty list  
        END FUNCTION  

        RETURN backtrack(zero, empty list)  
    END FUNCTION  
END CLASS
