CLASS Solution  
    FUNCTION findDuplicate(self, paths)  
        SET content_to_paths TO empty dictionary mapping to empty list  
        
        FOR each path IN paths  
            SET parts TO list of elements from splitting path by space  
            SET directory TO first element in parts  
            
            FOR each file_info IN elements of parts starting from second element  
                SET file_name TO substring before the first opening parenthesis in file_info  
                SET content TO substring inside the parentheses in file_info  
                
                APPEND concatenation of directory, forward slash, and file_name TO list at content_to_paths with key content  
            END FOR  
        END FOR  
        
        SET result TO empty list  
        FOR each paths_list IN values of content_to_paths  
            IF length of paths_list GREATER THAN one THEN  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
