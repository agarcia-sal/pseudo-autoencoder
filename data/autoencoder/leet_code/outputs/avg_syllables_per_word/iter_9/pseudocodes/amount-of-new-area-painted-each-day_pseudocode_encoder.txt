CLASS Solution  
    FUNCTION amountPainted(self, paint)  
        SET max_end TO the maximum end value among all intervals in paint  
        SET painted TO a list of size max_end PLUS one filled with value negative one  
        SET worklog TO an empty list  

        FOR each start, end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  

            WHILE current_start LESS THAN end  
                IF painted at position current_start EQUALS negative one THEN  
                    SET painted at position current_start TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO painted at position current_start  
                END IF  
            END WHILE  

            APPEND new_paint TO worklog  
        END FOR  

        RETURN worklog  
    END FUNCTION  
END CLASS
