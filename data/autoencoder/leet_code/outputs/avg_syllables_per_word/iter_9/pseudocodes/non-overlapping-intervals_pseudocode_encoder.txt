CLASS Solution  
    FUNCTION eraseOverlapIntervals(self, intervals)  
        IF intervals EQUALS empty THEN  
            RETURN 0  
        END IF  
        
        SORT intervals BY the second element in each interval  
        
        SET end TO the second element of the first interval in intervals  
        SET count TO 0  
        
        FOR index FROM 1 TO the length of intervals MINUS 1  
            IF the first element of intervals at index GREATER THAN OR EQUAL TO end THEN  
                SET end TO the second element of intervals at index  
            ELSE  
                INCREMENT count BY 1  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
