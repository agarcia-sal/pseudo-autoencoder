CLASS Solution  
    FUNCTION minJumps(self, arr)  
        SET n TO length of arr  
        IF n EQUALS 1 THEN  
            RETURN 0  
        END IF  
        
        SET value_indices TO a new mapping from value to list of indices  
        FOR each index i and corresponding value in arr  
            APPEND i TO value_indices at key value  
        END FOR  
        
        SET queue TO a new double-ended queue containing a tuple of zero and zero representing current index and steps  
        SET visited TO a new set containing zero  
        
        WHILE queue is not empty DO  
            REMOVE the front element from queue and UNPACK into current_index and steps  
            
            IF current_index EQUALS n MINUS 1 THEN  
                RETURN steps  
            END IF  
            
            SET next_indices TO list containing current_index PLUS one, current_index MINUS one concatenated with the list at value_indices for value arr at current_index  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited THEN  
                    ADD next_index TO visited  
                    APPEND tuple of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list at value_indices for value arr at current_index TO empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
