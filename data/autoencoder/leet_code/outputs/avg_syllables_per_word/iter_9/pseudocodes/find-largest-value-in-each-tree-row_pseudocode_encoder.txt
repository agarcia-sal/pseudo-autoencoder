CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH val AS first element of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode WITH val AS values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode WITH val AS values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root IS none THEN  
            RETURN empty list  
        END IF  
        SET queue TO empty double ended queue containing root  
        SET largest_values TO empty list  
        WHILE queue IS NOT empty  
            SET level_size TO length of queue  
            SET max_value TO negative infinity  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO first element removed FROM queue  
                IF node.val GREATER THAN max_value THEN  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
