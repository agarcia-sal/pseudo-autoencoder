CLASS Solution  
    FUNCTION removeDuplicateLetters(self, s)  
        SET last_occurrence TO an empty mapping  
        FOR each index and character in s  
            SET last_occurrence at character TO index  
        END FOR  
        
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index and character in s  
            IF character IS IN in_stack THEN  
                CONTINUE to next iteration  
            END IF  
            
            WHILE stack IS NOT empty AND character LESS THAN last element in stack AND index LESS THAN last_occurrence at last element in stack DO  
                SET removed_char TO last element popped from stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        RETURN concatenation of elements in stack  
    END FUNCTION  
END CLASS
