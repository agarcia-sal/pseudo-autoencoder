CLASS Solution  
    FUNCTION longestPalindrome(self, s)  
        SET n TO length of s  
        SET f TO a two dimensional table of size n by n with all elements set to True  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET f at position i and j TO False  
                IF character at position i in s EQUALS character at position j in s THEN  
                    SET f at position i and j TO f at position i plus one and j minus one  
                    IF f at position i and j IS True AND mx LESS THAN j MINUS i PLUS one THEN  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k to position k PLUS mx MINUS one  
    END FUNCTION  
END CLASS
