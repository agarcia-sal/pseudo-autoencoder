CLASS Solution
    FUNCTION myAtoi(self, s)
        SET INT_MAX TO two raised to the power thirty ONE MINUS one
        SET INT_MIN TO negative two raised to the power thirty ONE

        SET result TO zero
        SET index TO zero
        SET n TO length of s
        SET sign TO one

        WHILE index LESS THAN n AND character at s at position index EQUALS space
            INCREMENT index BY one
        END WHILE

        IF index LESS THAN n AND (character at s at position index EQUALS plus OR character at s at position index EQUALS minus) THEN
            IF character at s at position index EQUALS minus THEN
                SET sign TO negative one
            ELSE
                SET sign TO one
            END IF
            INCREMENT index BY one
        END IF

        WHILE index LESS THAN n AND character at s at position index IS a digit
            SET digit TO integer value of character at s at position index
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten THEN
                IF sign EQUALS one THEN
                    RETURN INT_MAX
                ELSE
                    RETURN INT_MIN
                END IF
            END IF
            SET result TO result MULTIPLIED BY ten PLUS digit
            INCREMENT index BY one
        END WHILE

        RETURN sign MULTIPLIED BY result
    END FUNCTION
END CLASS
