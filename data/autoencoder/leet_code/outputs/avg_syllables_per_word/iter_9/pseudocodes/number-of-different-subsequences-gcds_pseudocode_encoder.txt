CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(self, nums)  
        SET max_num TO the maximum element in nums  
        SET num_set TO the set of elements in nums  
        SET count TO zero  
        
        FOR g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            
            FOR m FROM g TO max_num IN STEPS OF g  
                IF m IS IN num_set THEN  
                    SET gcd_val TO the greatest common divisor of gcd_val AND m  
                    
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY one  
                        EXIT the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
