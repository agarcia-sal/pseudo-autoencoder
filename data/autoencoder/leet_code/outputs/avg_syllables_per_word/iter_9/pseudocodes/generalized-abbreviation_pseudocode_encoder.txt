CLASS Solution  
    FUNCTION generateAbbreviations(self, word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS length of word THEN  
                IF count GREATER THAN zero THEN  
                    APPEND string representation of count TO path  
                END IF  
                APPEND concatenation of path elements TO result  
            ELSE  
                IF count GREATER THAN zero THEN  
                    CALL backtrack with index INCREMENTED BY one, path EXTENDED BY string representation of count AND character at word position index, count SET TO zero  
                ELSE  
                    CALL backtrack with index INCREMENTED BY one, path EXTENDED BY character at word position index, count SET TO zero  
                END IF  
                CALL backtrack with index INCREMENTED BY one, path UNCHANGED, count INCREMENTED BY one  
            END IF  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack with zero, empty list, zero  
        RETURN result  
    END FUNCTION  
END CLASS
