CLASS Solution  
    FUNCTION permute(self, nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining IS empty THEN  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO length of remaining MINUS one  
                SET new_path TO path PLUS element at index i of remaining  
                SET new_remaining TO elements of remaining BEFORE index i CONCATENATED WITH elements of remaining AFTER index i  
                CALL backtrack(new_path, new_remaining)  
            END FOR  
        END FUNCTION  

        SET permutations TO empty list  
        CALL backtrack(empty list, nums)  
        RETURN permutations  
    END FUNCTION  
END CLASS
