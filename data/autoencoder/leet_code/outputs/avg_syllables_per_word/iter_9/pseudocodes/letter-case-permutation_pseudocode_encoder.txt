CLASS Solution  
    FUNCTION letterCasePermutation(self, s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUALS TO LENGTH OF t THEN  
                APPEND concatenation of all elements in t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i INCREMENTED BY one  
            IF element at index i in t IS alphabetic THEN  
                SET element at index i in t TO character obtained by toggling case of current character at t[i]  
                CALL dfs WITH i INCREMENTED BY one  
            END IF  
        END FUNCTION  

        SET t TO list representation of s  
        SET ans TO empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
