CLASS Solution  
    FUNCTION champagneTower(self, poured, query_row, query_glass)  
        SET current_row TO a list containing poured  
        
        FOR each row FROM one TO query_row INCLUSIVE  
            SET next_row TO a list of zeros with length equal to row PLUS one  
            
            FOR each index i FROM zero TO the length of current_row MINUS one  
                IF the element at current_row at index i GREATER THAN one THEN  
                    SET excess TO the element at current_row at index i MINUS one DIVIDED BY two  
                    INCREMENT the element at next_row at index i BY excess  
                    INCREMENT the element at next_row at index i PLUS one BY excess  
                END IF  
            END FOR  
            
            SET current_row TO next_row  
        END FOR  
        
        IF the element at current_row at index query_glass GREATER THAN one THEN  
            RETURN one  
        ELSE  
            RETURN the element at current_row at index query_glass  
        END IF  
    END FUNCTION  
END CLASS
