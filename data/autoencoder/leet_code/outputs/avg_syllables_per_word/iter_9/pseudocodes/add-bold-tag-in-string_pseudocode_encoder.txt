CLASS Solution  
    FUNCTION addBoldTag(self, s, words)  
        IF words IS empty THEN  
            RETURN s  
        END IF  
        
        SET n TO length of s  
        SET mask TO a list of n times False  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS length of word  
                SET start TO position of word found in s starting from start  
                IF start EQUALS negative one THEN  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS length of word MINUS one  
                    SET mask at position i TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF mask at position i IS True THEN  
                APPEND the string "<b>" TO result  
                WHILE i LESS THAN n AND mask at position i IS True  
                    APPEND the character s at position i TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string "</b>" TO result  
            ELSE  
                APPEND the character s at position i TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
