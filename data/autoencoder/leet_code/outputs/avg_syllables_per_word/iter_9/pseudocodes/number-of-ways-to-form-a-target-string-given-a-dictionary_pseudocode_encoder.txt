CLASS Solution  
    FUNCTION numWays(self, words, target)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SET n TO length of the first element in words  
        SET m TO length of target  
        
        SET freq TO a list containing n elements where each element is a mapping from characters to integers  
        FOR each word IN words  
            FOR each index j and character char in word  
                INCREMENT freq at position j and character char BY one  
            END FOR  
        END FOR  
        
        SET dp TO a two-dimensional list with m PLUS one rows and n PLUS one columns, all initialized to zero  
        
        FOR each index j FROM zero TO n inclusive  
            SET dp at row zero and column j TO one  
        END FOR  
        
        FOR each index i FROM one TO m inclusive  
            FOR each index j FROM one TO n inclusive  
                SET dp at row i and column j TO dp at row i and column j MINUS one  
                IF freq at position j MINUS one contains the character target at position i MINUS one AND its value GREATER THAN zero THEN  
                    SET dp at row i and column j TO dp at row i and column j PLUS dp at row i MINUS one and column j MINUS one MULTIPLIED BY freq at position j MINUS one and character target at position i MINUS one  
                    SET dp at row i and column j TO dp at row i and column j MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at row m and column n  
    END FUNCTION  
END CLASS
