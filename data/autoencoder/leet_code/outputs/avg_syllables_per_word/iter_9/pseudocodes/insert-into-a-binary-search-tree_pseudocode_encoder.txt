CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val AS first element of values  
    SET i TO one  
    INITIALIZE queue AS deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val AS values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val AS values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root IS None THEN  
                RETURN new TreeNode with val AS val  
            END IF  
            IF root.val LESS THAN val THEN  
                SET root.right TO dfs(root.right)  
            ELSE  
                SET root.left TO dfs(root.left)  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs(root)  
    END FUNCTION  
END CLASS
