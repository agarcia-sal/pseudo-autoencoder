CLASS Solution  
    FUNCTION monotoneIncreasingDigits(self, n)  
        SET digits TO list of characters in string representation of n  
        SET length TO number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one DOWN TO one  
            IF character at digits index i LESS THAN character at digits index i MINUS one THEN  
                SET character at digits index i MINUS one TO string representation of integer value of character at digits index i MINUS one MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET character at digits index i TO character nine  
        END FOR  
        RETURN integer value of concatenation of all characters in digits  
    END FUNCTION  
END CLASS
