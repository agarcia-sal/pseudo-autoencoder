CLASS Solution  
    FUNCTION rotatedDigits(self, n)  
        SET change_digits TO collection of characters "2", "5", "6", "9"  
        SET same_digits TO collection of characters "0", "1", "8"  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF all characters in num_str are elements of the union of change_digits and same_digits THEN  
                IF any character in num_str is an element of change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each i FROM one TO n DO  
            IF is_good_number(i) THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
