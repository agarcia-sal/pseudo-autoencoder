CLASS Solution  
    FUNCTION hasPath(self, maze, start, destination)  
        SET directions TO list of pairs: zero and one, one and zero, zero and minus one, minus one and zero  
        SET m TO length of maze  
        SET n TO length of first element in maze  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF pair of x and y IS IN visited THEN  
                RETURN False  
            END IF  
            IF list consisting of x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD pair of x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx plus dx GREATER THAN OR EQUAL TO zero AND nx plus dx LESS THAN m AND ny plus dy GREATER THAN OR EQUAL TO zero AND ny plus dy LESS THAN n AND maze element at nx plus dx and ny plus dy EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs(nx, ny) THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs element zero of start and dfs element one of start  
    END FUNCTION  
END CLASS
