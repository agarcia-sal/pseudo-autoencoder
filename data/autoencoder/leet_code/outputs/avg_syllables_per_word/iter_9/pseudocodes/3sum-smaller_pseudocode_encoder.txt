CLASS Solution
    FUNCTION threeSumSmaller(self, nums, target)
        SORT nums
        SET count TO zero
        SET n TO LENGTH of nums

        FOR i FROM zero TO n MINUS two
            SET left TO i PLUS one
            SET right TO n MINUS one

            WHILE left LESS THAN right
                SET current_sum TO nums AT i PLUS nums AT left PLUS nums AT right

                IF current_sum LESS THAN target THEN
                    INCREMENT count BY right MINUS left
                    INCREMENT left BY one
                ELSE
                    DECREMENT right BY one
                END IF
            END WHILE
        END FOR

        RETURN count
    END FUNCTION
END CLASS
