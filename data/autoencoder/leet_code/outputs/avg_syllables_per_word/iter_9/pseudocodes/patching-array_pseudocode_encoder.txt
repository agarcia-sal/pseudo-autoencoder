CLASS Solution  
    FUNCTION minPatches(nums, n)  
        SET missing TO 1  
        SET patches TO 0  
        SET i TO 0  
        
        WHILE missing LESS THAN OR EQUAL TO n  
            IF i LESS THAN length of nums AND nums at index i LESS THAN OR EQUAL TO missing THEN  
                SET missing TO missing PLUS nums at index i  
                INCREMENT i BY 1  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY 1  
            END IF  
        END WHILE  
        
        RETURN patches  
    END FUNCTION  
END CLASS
