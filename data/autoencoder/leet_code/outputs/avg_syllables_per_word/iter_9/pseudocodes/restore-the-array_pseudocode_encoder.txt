CLASS Solution  
    FUNCTION numberOfArrays(self, s, k)  
        SET MOD TO one billion plus seven  
        SET n TO length of s  
        
        SET dp TO list of size n plus one filled with zeros  
        SET dp at position zero TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO i minus one  
                IF character at position j in s NOT EQUALS zero AND integer value of substring of s from j to i GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k THEN  
                    SET dp at position i TO dp at position i PLUS dp at position j  
                    SET dp at position i TO dp at position i MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position n  
    END FUNCTION  
END CLASS
