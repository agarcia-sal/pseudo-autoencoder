CLASS Solution  
    FUNCTION solveSudoku(self, board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF board at row and i EQUALS num THEN  
                    RETURN False  
                END IF  
            END FOR  

            FOR i FROM zero TO eight  
                IF board at i and col EQUALS num THEN  
                    RETURN False  
                END IF  
            END FOR  

            SET start_row TO three MULTIPLIED BY the integer division of row BY three  
            SET start_col TO three MULTIPLIED BY the integer division of col BY three  

            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF board at (start_row PLUS i) and (start_col PLUS j) EQUALS num THEN  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  

            RETURN True  
        END FUNCTION  

        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF board at row and col EQUALS the character period THEN  
                        FOR num IN the string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num) THEN  
                                SET board at row and col TO num  
                                IF solve(board) THEN  
                                    RETURN True  
                                END IF  
                                SET board at row and col TO the character period  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  

        CALL solve(board)  
    END FUNCTION  
END CLASS
