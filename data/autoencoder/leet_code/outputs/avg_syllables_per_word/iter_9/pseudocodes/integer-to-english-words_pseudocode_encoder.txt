CLASS Solution  
    FUNCTION numberToWords(self, num)  
        IF num EQUALS 0 THEN  
            RETURN "Zero"  
        END IF  
        
        SET below_twenty TO list of strings representing numbers from empty string to "Nineteen"  
        SET tens TO list of strings representing tens multiples from two to nine  
        SET thousands TO list of strings representing thousand units from empty string to "Billion"  
        
        FUNCTION helper(n)  
            IF n EQUALS 0 THEN  
                RETURN empty string  
            ELSE IF n LESS THAN 20 THEN  
                RETURN element at index n in below_twenty PLUS space  
            ELSE IF n LESS THAN 100 THEN  
                RETURN element at index (n divided by ten) in tens PLUS space PLUS helper of (n modulo ten)  
            ELSE  
                RETURN element at index (n divided by one hundred) in below_twenty PLUS space PLUS "Hundred" PLUS space PLUS helper of (n modulo one hundred)  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO 0  
        
        WHILE num GREATER THAN 0 DO  
            IF (num modulo one thousand) NOT EQUALS 0 THEN  
                SET result TO helper of (num modulo one thousand) PLUS element at index i in thousands PLUS space PLUS result  
            END IF  
            SET num TO num divided by one thousand (integer division)  
            INCREMENT i BY 1  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
