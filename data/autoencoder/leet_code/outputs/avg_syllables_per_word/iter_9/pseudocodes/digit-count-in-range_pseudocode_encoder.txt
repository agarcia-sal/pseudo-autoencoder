CLASS Solution  
    FUNCTION digitsCount(self, d, low, high)  
        FUNCTION count_digit(n, digit)  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero  
                SET last TO current modulo ten  
                SET current TO current divided by ten, discarding remainder  
                SET full_cycles TO current  
                IF digit EQUALS zero THEN  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit THEN  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUALS digit THEN  
                    INCREMENT count BY remainder of n divided by power_of_10 PLUS one  
                END IF  
                MULTIPLY power_of_10 BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN result of count_digit with parameters high and d MINUS result of count_digit with parameters low MINUS one and d  
    END FUNCTION  
END CLASS
