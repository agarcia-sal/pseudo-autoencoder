CLASS Solution  
    FUNCTION maxNumber(self, nums1, nums2, k)  
        
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO length of nums MINUS k  
            SET stack TO empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND stack NOT empty AND last element in stack LESS THAN num  
                    REMOVE last element FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN first k elements FROM stack  
        END FUNCTION  
        
        FUNCTION merge(arr1, arr2)  
            SET result TO empty list  
            FOR each index FROM zero TO sum of lengths of arr1 and arr2 MINUS one  
                IF arr1 GREATER THAN arr2 THEN  
                    APPEND first element FROM arr1 TO result  
                    REMOVE first element FROM arr1  
                ELSE  
                    APPEND first element FROM arr2 TO result  
                    REMOVE first element FROM arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO length of nums1 AND k MINUS i LESS THAN OR EQUAL TO length of nums2 THEN  
                SET candidate TO merge(maxSingleArray(nums1, i), maxSingleArray(nums2, k MINUS i))  
                IF max_number LESS THAN candidate THEN  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
