CLASS Solution  
    FUNCTION asteroidCollision(self, asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE stack is not empty AND asteroid LESS THAN zero AND first element from end of stack GREATER THAN zero  
                IF first element from end of stack LESS THAN negative asteroid THEN  
                    REMOVE last element FROM stack  
                    CONTINUE to next iteration of WHILE  
                ELSE IF first element from end of stack EQUALS negative asteroid THEN  
                    REMOVE last element FROM stack  
                END IF  
                BREAK from WHILE  
            END WHILE  
            ELSE  
                APPEND asteroid TO stack  
            END ELSE  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
