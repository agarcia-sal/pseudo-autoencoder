CLASS Solution  
    FUNCTION reorganizeString(self, s)  
        SET count TO a mapping from characters TO their frequency in s  
        SET max_heap TO an empty list  
        FOR each character and frequency IN count  
            APPEND pair of negative frequency and character TO max_heap  
        END FOR  
        CALL heapify on max_heap  
        
        SET prev_char TO None  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero THEN  
                RETURN empty string  
            END IF  
            
            REMOVE top element FROM max_heap AND SET freq and char TO its values  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero THEN  
                ADD pair of prev_freq and prev_char TO max_heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation of elements in result  
    END FUNCTION  
END CLASS
