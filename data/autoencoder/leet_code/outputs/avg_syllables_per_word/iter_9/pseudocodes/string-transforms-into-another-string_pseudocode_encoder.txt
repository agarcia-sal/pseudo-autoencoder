CLASS Solution  
    FUNCTION canConvert(self, str1, str2)  
        IF str1 EQUALS str2 THEN  
            RETURN True  
        END IF  
        
        SET conversion_map TO empty mapping  
        
        FOR each pair char1, char2 FROM corresponding elements of str1 and str2  
            IF char1 EXISTS IN conversion_map THEN  
                IF conversion_map at char1 NOT EQUALS char2 THEN  
                    RETURN False  
                END IF  
            ELSE  
                SET conversion_map at char1 TO char2  
            END IF  
        END FOR  
        
        IF size of unique characters in str2 EQUALS 26 THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
