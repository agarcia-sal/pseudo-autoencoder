CLASS Solution  
    FUNCTION maxVacationDays(self, flights, days)  
        SET n TO length of flights  
        SET k TO length of days at first index  
        
        SET dp TO list of size n filled with negative one  
        SET dp at index zero TO zero  
        
        SET graph TO a mapping from each index in range zero to n minus one to an empty list  
        FOR each i IN range zero to n minus one  
            FOR each j IN range zero to n minus one  
                IF value of flights at i and j EQUALS one THEN  
                    APPEND j TO graph at key i  
                END IF  
            END FOR  
        END FOR  
        
        FOR each week IN range zero to k minus one  
            SET new_dp TO list of size n filled with negative one  
            FOR each i IN range zero to n minus one  
                IF value of dp at i EQUALS negative one THEN  
                    CONTINUE TO next iteration  
                END IF  
                SET new_dp at i TO the greater value between new_dp at i and dp at i PLUS days at i and week  
                FOR each j IN graph at key i  
                    SET new_dp at j TO the greater value between new_dp at j and dp at i PLUS days at j and week  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum value in dp  
    END FUNCTION  
END CLASS
