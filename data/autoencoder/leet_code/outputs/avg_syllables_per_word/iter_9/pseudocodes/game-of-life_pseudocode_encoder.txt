CLASS Solution  
    FUNCTION gameOfLife(self, board)  
        IF board IS empty OR first element of board IS empty THEN  
            RETURN  
        END IF  

        SET m TO length of board  
        SET n TO length of first element of board  

        SET directions TO list of pairs consisting of all combinations of minus one, zero, and one excluding zero-zero pair for eight neighbor positions  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx, dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND (board at ni, nj EQUALS one OR board at ni, nj EQUALS three) THEN  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  

                IF board at i, j EQUALS one THEN  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three THEN  
                        SET board at i, j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three THEN  
                        SET board at i, j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF board at i, j EQUALS two THEN  
                    SET board at i, j TO one  
                ELSE IF board at i, j EQUALS three THEN  
                    SET board at i, j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
