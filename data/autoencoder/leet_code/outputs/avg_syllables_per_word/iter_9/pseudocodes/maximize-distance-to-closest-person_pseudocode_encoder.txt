CLASS Solution  
    FUNCTION maxDistToClosest(self, seats)  
        SET max_distance TO zero  
        SET start TO zero  
        SET end TO zero  
        SET n TO length of seats  
        
        WHILE start LESS THAN n AND element at index start in seats EQUALS zero  
            INCREMENT start BY one  
        END WHILE  
        
        SET end TO n MINUS one  
        WHILE end GREATER THAN OR EQUAL TO zero AND element at index end in seats EQUALS zero  
            DECREMENT end BY one  
        END WHILE  
        
        IF max_distance LESS THAN start THEN  
            SET max_distance TO start  
        END IF  
        
        IF max_distance LESS THAN n MINUS end MINUS one THEN  
            SET max_distance TO n MINUS end MINUS one  
        END IF  
        
        SET current_distance TO zero  
        FOR i FROM start TO end  
            IF element at index i in seats EQUALS zero THEN  
                INCREMENT current_distance BY one  
                SET candidate_distance TO integer division of current_distance PLUS one BY two  
                IF max_distance LESS THAN candidate_distance THEN  
                    SET max_distance TO candidate_distance  
                END IF  
            ELSE  
                SET current_distance TO zero  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
