CLASS Solution  
    FUNCTION rotate(self, nums, k)  
        SET n TO length of nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start, end)  
            WHILE start LESS THAN end  
                SET temp TO nums[start]  
                SET nums[start] TO nums[end]  
                SET nums[end] TO temp  
                INCREMENT start BY 1  
                DECREMENT end BY 1  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH 0 AND n MINUS 1  
        CALL reverse WITH 0 AND k MINUS 1  
        CALL reverse WITH k AND n MINUS 1  
    END FUNCTION  
END CLASS
