CLASS Solution  
    FUNCTION canIWin(self, maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS 1) DIVIDED BY 2 LESS THAN desiredTotal THEN  
            RETURN False  
        END IF  

        IF desiredTotal LESS THAN OR EQUAL TO 0 THEN  
            RETURN True  
        END IF  

        SET memo TO empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF (current_total, used_numbers) IS IN memo THEN  
                RETURN memo VALUE FOR (current_total, used_numbers)  
            END IF  

            FOR number FROM 1 TO maxChoosableInteger DO  
                IF used_numbers WITH BIT REPRESENTING number NOT SET THEN  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, used_numbers WITH BIT FOR number SET) THEN  
                        SET memo VALUE FOR (current_total, used_numbers) TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET memo VALUE FOR (current_total, used_numbers) TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_win(0, 0)  
    END FUNCTION  
END CLASS
