CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(self, grid)  
        SET n TO length of grid  
        SET row_maxes TO empty list  
        FOR each row IN grid  
            APPEND maximum value in row TO row_maxes  
        END FOR  
        SET col_maxes TO empty list  
        FOR each column obtained by grouping elements at each index from grid  
            APPEND maximum value in column TO col_maxes  
        END FOR  

        SET total_increase TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET max_height TO minimum value between row_maxes at i and col_maxes at j  
                SET increase TO maximum value between zero and max_height MINUS grid element at row i column j  
                INCREMENT total_increase BY increase  
            END FOR  
        END FOR  

        RETURN total_increase  
    END FUNCTION  
END CLASS
