CLASS Solution
    FUNCTION minEatingSpeed(self, piles, h)
        FUNCTION canFinish(k)
            SET hours TO 0
            FOR each pile IN piles
                SET hours TO hours PLUS ceiling of (pile DIVIDED BY k)
            END FOR
            RETURN hours LESS THAN OR EQUAL TO h
        END FUNCTION

        SET left TO 1
        SET right TO maximum value IN piles

        WHILE left LESS THAN OR EQUAL TO right
            SET mid TO integer division of (left PLUS right) BY 2
            IF canFinish(mid) THEN
                SET right TO mid MINUS 1
            ELSE
                SET left TO mid PLUS 1
            END IF
        END WHILE

        RETURN left
    END FUNCTION
END CLASS
