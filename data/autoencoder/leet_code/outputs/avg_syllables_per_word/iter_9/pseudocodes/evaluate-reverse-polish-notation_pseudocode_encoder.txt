CLASS Solution  
    FUNCTION evalRPN(self, tokens)  
        SET stack TO empty list  
        
        FOR each token IN tokens  
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied by OR token EQUALS divided by THEN  
                SET b TO last element removed FROM stack  
                SET a TO last element removed FROM stack  
                IF token EQUALS plus THEN  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS minus THEN  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS multiplied by THEN  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS divided by THEN  
                    APPEND integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND integer conversion of token TO stack  
            END IF  
        END FOR  
        
        RETURN first element FROM stack  
    END FUNCTION  
END CLASS
