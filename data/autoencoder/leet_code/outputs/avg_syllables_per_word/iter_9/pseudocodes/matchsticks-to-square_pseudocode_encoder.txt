CLASS Solution  
    FUNCTION makesquare(self, matchsticks)  
        SET total_length TO sum of all elements in matchsticks  
        IF total_length modulo 4 NOT EQUALS 0 THEN  
            RETURN False  
        END IF  

        SET side_length TO total_length DIVIDED BY 4  
        SORT matchsticks in descending order  

        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS length of matchsticks THEN  
                FOR each side IN sides  
                    IF side NOT EQUALS side_length THEN  
                        RETURN False  
                    END IF  
                END FOR  
                RETURN True  
            END IF  

            FOR position FROM zero TO three  
                IF sides[position] PLUS matchsticks[index] LESS THAN OR EQUAL TO side_length THEN  
                    INCREMENT sides[position] BY matchsticks[index]  
                    IF can_form_sides(sides, index PLUS one) THEN  
                        RETURN True  
                    END IF  
                    DECREMENT sides[position] BY matchsticks[index]  
                END IF  

                IF sides[position] EQUALS zero THEN  
                    BREAK  
                END IF  
            END FOR  

            RETURN False  
        END FUNCTION  

        RETURN can_form_sides(list of four zeros, zero)  
    END FUNCTION  
END CLASS
