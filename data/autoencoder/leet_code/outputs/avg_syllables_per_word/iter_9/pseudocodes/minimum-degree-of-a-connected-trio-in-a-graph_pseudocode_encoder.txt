CLASS Solution  
    FUNCTION minTrioDegree(self, n, edges)  
        CREATE graph AS mapping FROM integer TO set  
        FOR each integer i FROM 1 TO n  
            SET graph at i TO empty set  
        END FOR  
        
        FOR each pair u, v IN edges  
            ADD v TO graph at u  
            ADD u TO graph at v  
        END FOR  
        
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        
        FOR each integer i FROM 1 TO n  
            FOR each integer j FROM i plus 1 TO n  
                FOR each integer k FROM j plus 1 TO n  
                    IF j IS IN graph at i AND k IS IN graph at i AND k IS IN graph at j THEN  
                        SET trio_found TO true  
                        SET trio_degree TO (size of graph at i PLUS size of graph at j PLUS size of graph at k MINUS 6)  
                        IF trio_degree LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found IS true THEN  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
