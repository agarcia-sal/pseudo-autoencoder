CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as first element of values  
    SET i TO one  
    INITIALIZE queue as a new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND element at i in values IS NOT None THEN  
            SET node.left TO new TreeNode with val as element at i in values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at i in values IS NOT None THEN  
            SET node.right TO new TreeNode with val as element at i in values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root IS None THEN  
            RETURN zero  
        END IF  

        FUNCTION dfs(node, parent, current_length)  
            IF node IS None THEN  
                RETURN current_length  
            END IF  

            IF parent IS NOT None AND value of node EQUALS parent PLUS one THEN  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  

            SET left_length TO dfs(left child of node, value of node, current_length)  
            SET right_length TO dfs(right child of node, value of node, current_length)  

            RETURN maximum of current_length, left_length, right_length  
        END FUNCTION  

        RETURN dfs(root, None, zero)  
    END FUNCTION  
END CLASS
