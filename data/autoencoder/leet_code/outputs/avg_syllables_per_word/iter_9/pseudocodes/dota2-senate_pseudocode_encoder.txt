CLASS Solution  
    FUNCTION predictPartyVictory(self, senate)  
        IMPORT deque FROM collections  
        
        SET radiant TO a new empty deque  
        SET dire TO a new empty deque  
        
        FOR each index i AND character s IN senate  
            IF s EQUALS 'R' THEN  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
            
            IF r_index LESS THAN d_index THEN  
                APPEND r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant IS NOT empty THEN  
            RETURN "Radiant"  
        ELSE  
            RETURN "Dire"  
        END IF  
    END FUNCTION  
END CLASS
