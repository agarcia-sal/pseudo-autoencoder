CLASS Solution  
    FUNCTION new21Game(self, n, k, maxPts)  

        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k THEN  
                IF i LESS THAN OR EQUAL TO n THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  

            IF i EQUALS k MINUS one THEN  
                SET numerator TO minimum of (n MINUS k PLUS one) AND maxPts  
                RETURN numerator DIVIDED BY maxPts  
            END IF  

            SET first_term TO dfs(i PLUS one)  
            SET second_term TO dfs(i PLUS one) MINUS dfs(i PLUS maxPts PLUS one)  
            RETURN first_term PLUS second_term DIVIDED BY maxPts  
        END FUNCTION  

        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
