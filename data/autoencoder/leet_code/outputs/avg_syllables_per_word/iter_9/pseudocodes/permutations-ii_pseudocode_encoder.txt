CLASS Solution  
    FUNCTION permuteUnique(self, nums)  
        FUNCTION backtrack(path, used, res)  
            IF length of path EQUALS length of nums THEN  
                APPEND a copy of path TO res  
                RETURN  
            END IF  

            FOR index FROM zero TO length of nums MINUS one  
                IF used at index IS True OR (index GREATER THAN zero AND nums at index EQUALS nums at index MINUS one AND used at index MINUS one IS False) THEN  
                    CONTINUE  
                END IF  

                SET used at index TO True  
                APPEND nums at index TO path  
                CALL backtrack WITH path, used, res  
                REMOVE last element FROM path  
                SET used at index TO False  
            END FOR  
        END FUNCTION  

        SORT nums IN ascending order  
        SET res TO empty list  
        SET used TO list of False with length equal to length of nums  
        CALL backtrack WITH empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
