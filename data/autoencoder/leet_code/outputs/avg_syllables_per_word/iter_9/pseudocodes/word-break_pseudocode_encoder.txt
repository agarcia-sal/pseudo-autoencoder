CLASS Solution  
    FUNCTION wordBreak(self, s, wordDict)  
        SET wordSet TO set constructed from wordDict  
        SET dp TO list of False values with length equal to length of s PLUS one  
        SET first element of dp TO True  
        
        FOR i FROM one TO length of s DO  
            FOR j FROM zero TO i MINUS one DO  
                IF element at dp indexed by j IS True AND substring of s from j UP TO i IS IN wordSet THEN  
                    SET element at dp indexed by i TO True  
                    BREAK inner loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at dp indexed by length of s  
    END FUNCTION  
END CLASS
