CLASS Solution  
    FUNCTION wiggleMaxLength(self, nums)  
        IF length of nums LESS THAN 2 THEN  
            RETURN length of nums  
        END IF  

        SET up TO 1  
        SET down TO 1  

        FOR i FROM 1 TO length of nums MINUS 1  
            IF nums at position i GREATER THAN nums at position i MINUS 1 THEN  
                SET up TO down PLUS 1  
            ELSE IF nums at position i LESS THAN nums at position i MINUS 1 THEN  
                SET down TO up PLUS 1  
            END IF  
        END FOR  

        RETURN the greater value between up and down  
    END FUNCTION  
END CLASS
