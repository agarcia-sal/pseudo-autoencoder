CLASS Solution  
    FUNCTION checkInclusion(self, s1, s2)  
        DEFINE count_s1 AS frequency counter FOR characters IN s1  
        DEFINE count_s2 AS frequency counter FOR characters IN first substring OF s2 WITH length EQUAL TO length OF s1  
        SET len_s1 TO length OF s1  
        SET len_s2 TO length OF s2  
        
        IF len_s1 GREATER THAN len_s2 THEN  
            RETURN False  
        END IF  
        
        IF count_s1 EQUALS count_s2 THEN  
            RETURN True  
        END IF  
        
        FOR index FROM len_s1 TO len_s2 MINUS one DO  
            INCREMENT count_s2 AT character s2 AT position index BY one  
            DECREMENT count_s2 AT character s2 AT position index MINUS len_s1 BY one  
            
            IF count_s2 AT character s2 AT position index MINUS len_s1 EQUALS zero THEN  
                REMOVE character s2 AT position index MINUS len_s1 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2 THEN  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
