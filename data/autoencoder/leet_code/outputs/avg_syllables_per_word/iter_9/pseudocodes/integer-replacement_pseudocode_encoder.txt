CLASS Solution  
    FUNCTION integerReplacement(self, n)  
        SET count TO 0  
        WHILE n NOT EQUALS 1  
            IF n MODULO 2 EQUALS 0 THEN  
                SET n TO n DIVIDED BY 2  
            ELSE  
                IF n EQUALS 3 THEN  
                    SET n TO n MINUS 1  
                ELSE IF (n PLUS 1) MODULO 4 EQUALS 0 THEN  
                    SET n TO n PLUS 1  
                ELSE  
                    SET n TO n MINUS 1  
                END IF  
            END IF  
            INCREMENT count BY 1  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
