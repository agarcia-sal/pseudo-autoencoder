CLASS Solution  
    FUNCTION numPermsDISequence(self, s)  
        SET MOD TO one billion PLUS seven  
        SET n TO length of s  
        
        DEFINE dp AS two-dimensional list of size (n PLUS one) BY (n PLUS one) FILLED WITH zero  
        
        FOR each j FROM zero TO n DO  
            SET dp at position zero and j TO one  
        END FOR  
        
        FOR each i FROM one TO n DO  
            SET prefix_sum TO zero  
            
            IF character at position i MINUS one of s EQUALS 'I' THEN  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one DO  
                    SET prefix_sum TO prefix_sum PLUS dp at position i MINUS one and j  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET dp at position i and j TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero DO  
                    SET prefix_sum TO prefix_sum PLUS dp at position i MINUS one and j PLUS one  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET dp at position i and j TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO sum of elements in dp at position n  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
