CLASS Solution  
    FUNCTION minDifficulty(self, jobDifficulty, d)  
        SET n TO length of jobDifficulty  
        
        IF n LESS THAN d THEN  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one THEN  
                RETURN maximum value in jobDifficulty starting from index i to end  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d DO  
                SET current_max TO maximum of current_max AND jobDifficulty at index j  
                SET min_difficulty TO minimum of min_difficulty AND current_max PLUS dp(j PLUS one, d MINUS one)  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
END CLASS
