CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with val as first element of values and next as None  
    SET p TO head  
    FOR each val IN values from second element to last  
        SET node TO new ListNode with val as val and next as None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self, head)  
        SET dummy TO new ListNode with val as zero and next as head  
        SET prev TO dummy  
        WHILE head IS NOT None DO  
            IF head.next IS NOT None AND head.val EQUALS head.next.val THEN  
                WHILE head.next IS NOT None AND head.val EQUALS head.next.val DO  
                    SET head TO head.next  
                END WHILE  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            END IF  
            SET head TO head.next  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
