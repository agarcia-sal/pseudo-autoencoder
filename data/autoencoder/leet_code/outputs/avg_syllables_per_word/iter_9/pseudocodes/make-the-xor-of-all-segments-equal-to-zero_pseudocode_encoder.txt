CLASS Solution  
    FUNCTION minChanges(self, nums, k)  
        SET groups TO result of auxiliary FUNCTION defaultdict_of_lists  
        FOR index FROM zero TO length of nums MINUS one  
            APPEND element at index in nums TO groups at key index modulo k  
        END FOR  

        SET freq TO empty list  
        SET min_size TO empty list  
        FOR index FROM zero TO k MINUS one  
            APPEND result of auxiliary FUNCTION defaultdict_of_ints TO freq  
            APPEND infinity TO min_size  
        END FOR  

        FOR index FROM zero TO k MINUS one  
            FOR each num IN groups at index  
                INCREMENT freq at index at key num BY one  
            END FOR  
            SET min_size at index TO length of groups at index  
        END FOR  

        SET dp TO list of size two raised to the power of ten filled with infinity  
        SET dp at zero TO zero  

        FOR index FROM zero TO k MINUS one  
            SET new_dp TO list of size two raised to the power of ten filled with infinity  
            SET min_dp TO minimum value in dp  
            FOR j FROM zero TO two raised to the power of ten MINUS one  
                SET new_dp at j TO min_dp PLUS min_size at index  
                FOR each num and count IN items of freq at index  
                    SET candidate TO dp at position bitwise XOR of j and num PLUS min_size at index MINUS count  
                    IF candidate LESS THAN new_dp at j THEN  
                        SET new_dp at j TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN dp at zero  
    END FUNCTION  
END CLASS
