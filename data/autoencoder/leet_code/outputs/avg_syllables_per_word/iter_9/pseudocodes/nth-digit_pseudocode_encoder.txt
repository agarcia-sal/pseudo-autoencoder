CLASS Solution  
    FUNCTION findNthDigit(self, n)  
        SET length TO 1  
        SET count TO 9  
        SET start TO 1  
        
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY 1  
            SET count TO count MULTIPLIED BY 10  
            SET start TO start MULTIPLIED BY 10  
        END WHILE  
        
        SET start TO start PLUS (n MINUS 1) DIVIDED BY length  
        RETURN digit at position (n MINUS 1) MODULO length IN string representation of start converted to integer  
    END FUNCTION  
END CLASS
