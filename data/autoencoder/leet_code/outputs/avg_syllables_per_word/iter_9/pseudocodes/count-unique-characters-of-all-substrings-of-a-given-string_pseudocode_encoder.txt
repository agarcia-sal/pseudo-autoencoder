CLASS Solution  
    FUNCTION uniqueLetterString(self, s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO length of s  
        
        SET last TO a list of size twenty-six with all elements set to negative one  
        SET last2 TO a list of size twenty-six with all elements set to negative one  
        
        SET result TO zero  
        
        FOR each index i and character char in s with enumeration  
            SET idx TO the ordinal position of char MINUS the ordinal position of character A  
            
            SET result TO result PLUS the product of (i MINUS last at idx) AND (last at idx MINUS last2 at idx)  
            SET result TO result modulo MOD  
            
            SET last2 at idx TO last at idx  
            SET last at idx TO i  
        END FOR  
        
        FOR each index i FROM zero TO twenty-five  
            SET result TO result PLUS the product of (n MINUS last at i) AND (last at i MINUS last2 at i)  
            SET result TO result modulo MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
