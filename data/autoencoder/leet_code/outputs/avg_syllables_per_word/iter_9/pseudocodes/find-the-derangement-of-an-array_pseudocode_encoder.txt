CLASS Solution  
    FUNCTION findDerangement(self, n)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        
        IF n EQUALS one THEN  
            RETURN zero  
        END IF  
        
        IF n EQUALS two THEN  
            RETURN one  
        END IF  
        
        SET prev2 TO one  
        SET prev1 TO zero  
        
        FOR i FROM three TO n DO  
            SET current TO (i MINUS one) MULTIPLIED BY (prev1 PLUS prev2) MODULO MOD  
            SET prev2 TO prev1  
            SET prev1 TO current  
        END FOR  
        
        RETURN prev1  
    END FUNCTION  
END CLASS
