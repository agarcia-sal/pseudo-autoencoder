CLASS Solution  
    FUNCTION findPermutation(self, s)  
        SET result TO list of integers starting from one TO length of s PLUS one  
        SET start TO zero  
        FOR end FROM zero TO length of s  
            IF end EQUALS length of s OR character at position end in s EQUALS letter I THEN  
                REVERSE the sublist of result from position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
