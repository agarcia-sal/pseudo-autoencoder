CLASS Solution  
    FUNCTION complexNumberMultiply(self, num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO splitting num by the plus sign  
            SET real TO converting first element of parts to integer  
            SET imaginary TO converting second element of parts with last character removed to integer  
            RETURN real, imaginary  
        END FUNCTION  

        SET real1, imag1 TO calling parse_complex with num1  
        SET real2, imag2 TO calling parse_complex with num2  

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  

        RETURN concatenation of real_part, the plus sign, imaginary_part, and the character i  
    END FUNCTION  
END CLASS
