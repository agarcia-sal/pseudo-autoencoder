CLASS Solution  
    FUNCTION knightProbability(self, n, k, row, column)  
        SET moves TO list of all pairs of integers representing the eight knight moves as follows:  
            pair of negative two and negative one,  
            pair of negative two and one,  
            pair of negative one and negative two,  
            pair of negative one and two,  
            pair of one and negative two,  
            pair of one and two,  
            pair of two and negative one,  
            pair of two and one  
        
        SET dp TO two-dimensional list of size n by n where each element is zero  
        SET element at row and column indices in dp TO one  
        
        FOR each step FROM one TO k DO  
            SET new_dp TO two-dimensional list of size n by n where each element is zero  
            
            FOR each r FROM zero TO n MINUS one DO  
                FOR each c FROM zero TO n MINUS one DO  
                    IF element at r and c indices in dp EQUALS zero THEN  
                        CONTINUE to next iteration  
                    END IF  
                    FOR each dr, dc IN moves DO  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n THEN  
                            INCREMENT element at nr and nc indices in new_dp BY element at r and c indices in dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            
            SET dp TO new_dp  
        END FOR  
        
        SET total_probability TO zero  
        FOR each row IN dp DO  
            FOR each value IN row DO  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
        
        RETURN total_probability  
    END FUNCTION  
END CLASS
