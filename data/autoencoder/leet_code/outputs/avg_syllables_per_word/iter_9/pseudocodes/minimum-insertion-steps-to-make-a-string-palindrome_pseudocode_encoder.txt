CLASS Solution  
    FUNCTION minInsertions(self, s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO length of s  
            INITIALIZE dp AS two-dimensional array of size n by n with all elements set to zero  
            FOR i FROM zero TO n minus one  
                SET dp at position i, i TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n minus cl  
                    SET j TO i plus cl minus one  
                    IF character at s at position i EQUALS character at s at position j AND cl EQUALS two THEN  
                        SET dp at position i, j TO two  
                    ELSE IF character at s at position i EQUALS character at s at position j THEN  
                        SET dp at position i, j TO dp at position i plus one, j minus one PLUS two  
                    ELSE  
                        SET dp at position i, j TO maximum of dp at position i, j minus one AND dp at position i plus one, j  
                    END IF  
                END FOR  
            END FOR  
            RETURN dp at position zero, n minus one  
        END FUNCTION  

        RETURN length of s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
