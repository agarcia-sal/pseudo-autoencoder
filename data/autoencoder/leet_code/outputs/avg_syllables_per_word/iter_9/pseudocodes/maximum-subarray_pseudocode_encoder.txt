CLASS Solution  
    FUNCTION maxSubArray(self, nums)  
        SET max_sum TO first element of nums  
        SET current_sum TO first element of nums  
        FOR each num IN nums starting from second element  
            IF num GREATER THAN current_sum PLUS num THEN  
                SET current_sum TO num  
            ELSE  
                SET current_sum TO current_sum PLUS num  
            END IF  
            IF max_sum LESS THAN current_sum THEN  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
