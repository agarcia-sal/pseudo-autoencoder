CLASS Solution  
    FUNCTION grayCode(self, n)  
        IF n EQUALS 0 THEN  
            RETURN a list containing 0  
        END IF  
        
        SET prev_gray TO the result of calling grayCode on n MINUS 1  
        
        SET mask TO 1 MULTIPLIED BY 2 RAISED TO THE POWER OF (n MINUS 1)  
        
        SET current_gray TO concatenation of prev_gray and a list formed by  
            FOR each number IN prev_gray traversed in reverse order  
                APPLY logical OR between mask and number  
        
        RETURN current_gray  
    END FUNCTION  
END CLASS
