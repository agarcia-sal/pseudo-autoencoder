CLASS Solution  
    FUNCTION maximumScore(self, nums, k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at index k in nums  
        SET max_score TO current_min  
        
        WHILE left GREATER THAN zero OR right LESS THAN length of nums MINUS one  
            IF left EQUALS zero THEN  
                INCREMENT right BY one  
            ELSE IF right EQUALS length of nums MINUS one THEN  
                DECREMENT left BY one  
            ELSE IF element at index left MINUS one in nums GREATER THAN OR EQUALS element at index right PLUS one in nums THEN  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            
            SET current_min TO minimum of current_min, element at index left in nums, and element at index right in nums  
            SET score TO current_min MULTIPLIED BY (right MINUS left PLUS one)  
            SET max_score TO maximum of max_score and score  
        END WHILE  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
