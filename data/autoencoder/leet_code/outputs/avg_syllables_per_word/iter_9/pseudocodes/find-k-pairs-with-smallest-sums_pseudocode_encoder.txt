CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS empty OR nums2 IS empty OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        FOR index FROM zero TO the smaller value BETWEEN k MINUS one AND length of nums1 MINUS one  
            INSERT into min_heap the tuple containing the sum of nums1 AT index PLUS nums2 AT zero, index, and zero  
        END FOR  
        
        SET result TO empty list  
        WHILE min_heap IS not empty AND length of result LESS THAN k DO  
            REMOVE the smallest element FROM min_heap AND ASSIGN TO _, i, j  
            APPEND list containing nums1 AT i and nums2 AT j TO result  
            IF j PLUS one LESS THAN length of nums2 THEN  
                INSERT into min_heap the tuple containing the sum of nums1 AT i PLUS nums2 AT j PLUS one, i, and j PLUS one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
