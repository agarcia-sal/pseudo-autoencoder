CLASS Solution  
    FUNCTION arrayNesting(self, nums)  
        SET visited TO list of False values with length equal to length of nums  
        SET max_length TO zero  
        
        FOR index FROM zero TO length of nums MINUS one  
            IF visited at index EQUALS False THEN  
                SET current_length TO zero  
                SET k TO index  
                WHILE visited at k EQUALS False DO  
                    SET visited at k TO True  
                    SET k TO value at nums at k  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO greater of max_length AND current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
