CLASS Solution  
    FUNCTION findContestMatch(self, n)  
        SET teams TO list of strings representing numbers from one TO n inclusive  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n divided by two MINUS one  
                SET teams at position i TO string formed by opening parenthesis, teams at position i, comma, teams at position (n MINUS one MINUS i), and closing parenthesis  
            END FOR  
            SET n TO n divided by two  
        END WHILE  
        RETURN teams at position zero  
    END FUNCTION  
END CLASS
