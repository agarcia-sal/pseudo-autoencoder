CLASS Solution  
    FUNCTION longestSubstring(self, s, k)  
        IF length of s LESS THAN k THEN  
            RETURN 0  
        END IF  
  
        SET count TO collection mapping of each character's frequency in s  
  
        FOR each character IN keys of count  
            IF frequency of character IN count LESS THAN k THEN  
                RETURN maximum value of longestSubstring called on each substring obtained by splitting s by character WITH k  
            END IF  
        END FOR  
  
        RETURN length of s  
    END FUNCTION  
END CLASS
