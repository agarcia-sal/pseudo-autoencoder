CLASS Solution  
    FUNCTION kEmptySlots(self, bulbs, k)  
        SET bloom_days TO empty list  
        
        FOR each day and position IN bulbs with day starting from one  
            SET i TO position in bloom_days where position can be inserted to keep bloom_days sorted  
            
            IF i GREATER THAN zero AND position MINUS element before position in bloom_days MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            
            IF i LESS THAN length of bloom_days AND element at position i in bloom_days MINUS position MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            
            INSERT position INTO bloom_days AT index i  
        END FOR  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
