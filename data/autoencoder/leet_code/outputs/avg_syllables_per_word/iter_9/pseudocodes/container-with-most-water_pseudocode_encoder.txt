CLASS Solution  
    FUNCTION maxArea(self, height)  
        SET left TO zero  
        SET right TO length of height MINUS one  
        SET max_area TO zero  
        
        WHILE left LESS THAN right  
            SET current_area TO minimum of height at left AND height at right MULTIPLIED BY right MINUS left  
            SET max_area TO maximum of max_area AND current_area  
            
            IF height at left LESS THAN height at right THEN  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
