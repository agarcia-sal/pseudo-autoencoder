CLASS Solution  
    FUNCTION myPow(self, x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO 1  
            WHILE n NOT EQUALS 0  
                IF remainder of n divided by 2 EQUALS 1 THEN  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO integer division of n BY 2  
            END WHILE  
            RETURN ans  
        END FUNCTION  

        IF n GREATER THAN OR EQUALS 0 THEN  
            RETURN qpow(x, n)  
        ELSE  
            RETURN 1 DIVIDED BY qpow(x, NEGATION of n)  
        END IF  
    END FUNCTION  
END CLASS
