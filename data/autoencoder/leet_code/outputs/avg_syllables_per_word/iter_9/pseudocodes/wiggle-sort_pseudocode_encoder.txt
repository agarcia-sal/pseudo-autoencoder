CLASS Solution  
    FUNCTION wiggleSort(self, nums)  
        SET less TO true  
        
        FOR i FROM zero TO length of nums MINUS one  
            IF less THEN  
                IF element at index i in nums GREATER THAN element at index i plus one in nums THEN  
                    SWAP element at index i in nums WITH element at index i plus one in nums  
                END IF  
            ELSE  
                IF element at index i in nums LESS THAN element at index i plus one in nums THEN  
                    SWAP element at index i in nums WITH element at index i plus one in nums  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
