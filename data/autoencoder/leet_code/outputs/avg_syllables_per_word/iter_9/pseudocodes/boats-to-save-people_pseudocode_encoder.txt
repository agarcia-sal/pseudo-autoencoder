CLASS Solution  
    FUNCTION numRescueBoats(people, limit)  
        CALL sort ON people  
        SET left TO zero  
        SET right TO length of people MINUS one  
        SET boats TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            IF value at people at left PLUS value at people at right LESS THAN OR EQUAL TO limit THEN  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
