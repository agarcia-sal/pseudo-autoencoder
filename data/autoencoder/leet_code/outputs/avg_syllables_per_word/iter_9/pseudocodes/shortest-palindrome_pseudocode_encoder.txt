CLASS Solution  
    FUNCTION shortestPalindrome(self, s)  
        IF s EQUALS empty string THEN  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse of check_str  
        END FUNCTION  

        FOR i FROM length of s DOWN TO zero  
            IF is_palindrome(substring of s from start TO i) THEN  
                SET to_add TO reverse of substring of s from i TO end  
                RETURN concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
