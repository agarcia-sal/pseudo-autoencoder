CLASS Solution  
    FUNCTION maxCoins(self, nums)  
        SET nums TO list with value one PREPENDED and value one APPENDED to original nums  
        SET n TO length of nums  
        
        SET dp TO two dimensional list of zeros with dimension n by n  
        
        FOR length FROM two TO n MINUS one  
            FOR i FROM zero TO n MINUS length MINUS one  
                SET j TO i PLUS length  
                FOR k FROM i PLUS one TO j MINUS one  
                    SET coins TO nums at i MULTIPLIED BY nums at k MULTIPLIED BY nums at j PLUS dp at i and k PLUS dp at k and j  
                    SET dp at i and j TO maximum of dp at i and j AND coins  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at zero and n MINUS one  
    END FUNCTION  
END CLASS
