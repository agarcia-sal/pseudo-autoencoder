CLASS Solution  
    FUNCTION countVowelPermutation(self, n)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        
        SET a TO one  
        SET e TO one  
        SET i TO one  
        SET o TO one  
        SET u TO one  
        
        FOR each number from one TO n MINUS one  
            SET a_next TO e PLUS i PLUS u  
            SET e_next TO a PLUS i  
            SET i_next TO e PLUS o  
            SET o_next TO i  
            SET u_next TO i PLUS o  
            
            SET a TO a_next MODULO MOD  
            SET e TO e_next MODULO MOD  
            SET i TO i_next MODULO MOD  
            SET o TO o_next MODULO MOD  
            SET u TO u_next MODULO MOD  
        END FOR  
        
        RETURN the sum of a, e, i, o, u MODULO MOD  
    END FUNCTION  
END CLASS
