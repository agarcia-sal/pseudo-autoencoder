CLASS Solution  
    FUNCTION lengthOfLongestSubstring(self, s)  
        SET char_index_map TO empty map  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end, char IN enumeration of s  
            IF char EXISTS IN char_index_map AND char_index_map[char] GREATER THAN OR EQUALS start THEN  
                SET start TO char_index_map[char] PLUS one  
            END IF  
            SET char_index_map[char] TO end  
            SET max_length TO maximum of max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
