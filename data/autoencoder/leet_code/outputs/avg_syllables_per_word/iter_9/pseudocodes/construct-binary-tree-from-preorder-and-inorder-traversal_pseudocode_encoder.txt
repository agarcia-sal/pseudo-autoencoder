CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO TreeNode with val equal to first element of values, left as None, right as None  
    SET i TO 1  
    SET queue TO an empty deque structure  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO element removed from the left of queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO TreeNode with val equal to values at i, left as None, right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO TreeNode with val equal to values at i, left as None, right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty THEN  
            RETURN None  
        END IF  
        SET root_val TO first element of preorder  
        SET root TO TreeNode with val equal to root_val, left as None, right as None  
        SET root_index_inorder TO position of root_val in inorder  
        SET root.left TO self.buildTree(sublist of preorder from position one to root_index_inorder inclusive, sublist of inorder from start to position before root_index_inorder)  
        SET root.right TO self.buildTree(sublist of preorder from position after root_index_inorder to end, sublist of inorder from position after root_index_inorder to end)  
        RETURN root  
    END FUNCTION  
END CLASS
