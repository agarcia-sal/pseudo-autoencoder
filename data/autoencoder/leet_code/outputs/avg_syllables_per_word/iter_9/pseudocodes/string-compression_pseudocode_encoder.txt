CLASS Solution  
    FUNCTION compress(self, chars)  
        IF chars IS empty THEN  
            RETURN 0  
        END IF  
        
        SET write_index TO 0  
        SET read_index TO 0  
        
        WHILE read_index LESS THAN length of chars DO  
            SET char TO chars at position read_index  
            SET count TO 0  
            
            WHILE read_index LESS THAN length of chars AND chars at position read_index EQUALS char DO  
                INCREMENT read_index BY 1  
                INCREMENT count BY 1  
            END WHILE  
            
            SET chars at position write_index TO char  
            INCREMENT write_index BY 1  
            
            IF count GREATER THAN 1 THEN  
                FOR each digit IN string representation of count DO  
                    SET chars at position write_index TO digit  
                    INCREMENT write_index BY 1  
                END FOR  
            END IF  
        END WHILE  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
