CLASS Solution  
    FUNCTION find132pattern(self, nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        
        FOR each num IN nums traversed from last element to first element  
            IF num LESS THAN second THEN  
                RETURN true  
            END IF  
            
            WHILE stack is not empty AND last element of stack LESS THAN num  
                SET second TO last element removed FROM stack  
            END WHILE  
            
            APPEND num TO stack  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
