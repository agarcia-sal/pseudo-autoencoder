CLASS Solution  
    FUNCTION splitArraySameAverage(self, nums)  
        SET n TO length of nums  
        SET total_sum TO sum of nums  
        
        IF n EQUALS 1 THEN  
            RETURN False  
        END IF  
        
        SET dp TO list of sets with size n divided by 2 plus 1  
        SET dp at index zero TO set containing zero  
        
        FOR each num IN nums  
            FOR size FROM n divided by 2 DOWN TO one  
                FOR each s IN dp at index size minus one  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum THEN  
                        RETURN True  
                    END IF  
                    ADD new_sum TO dp at index size  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
