CLASS Solution  
    FUNCTION subarraySum(self, nums, k)  
        SET cumulative_sum_freq TO dictionary with key zero and value one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF cumulative_sum_freq contains key current_sum MINUS k THEN  
                INCREMENT count BY cumulative_sum_freq value at key current_sum MINUS k  
            END IF  
            IF cumulative_sum_freq contains key current_sum THEN  
                INCREMENT cumulative_sum_freq value at key current_sum BY one  
            ELSE  
                SET cumulative_sum_freq value at key current_sum TO one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
