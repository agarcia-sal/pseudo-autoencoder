CLASS Solution  
    FUNCTION smallestRange(self, nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        
        FOR i FROM zero TO length of nums MINUS one  
            PUSH tuple of (element at index zero in nums at i, i, zero) INTO min_heap  
            IF max_value LESS THAN element at index zero in nums at i THEN  
                SET max_value TO element at index zero in nums at i  
            END IF  
        END FOR  
        
        SET smallest_range TO positive infinity  
        SET result TO list containing negative one hundred thousand and positive one hundred thousand  
        
        WHILE min_heap IS NOT empty  
            REMOVE smallest tuple FROM min_heap AND UNPACK TO min_value, list_index, element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value and max_value  
            END IF  
            
            IF element_index PLUS one LESS THAN length of nums at list_index THEN  
                SET next_value TO element at position element_index PLUS one in nums at list_index  
                PUSH tuple of (next_value, list_index, element_index PLUS one) INTO min_heap  
                IF max_value LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
