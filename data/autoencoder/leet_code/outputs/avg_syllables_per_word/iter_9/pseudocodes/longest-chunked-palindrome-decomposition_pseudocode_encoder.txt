CLASS Solution  
    FUNCTION longestDecomposition(self, text)  
        SET left TO zero  
        SET right TO length of text MINUS one  
        SET left_part TO empty string  
        SET right_part TO empty string  
        SET k TO zero  
        
        WHILE left LESS THAN right  
            APPEND character at position left FROM text TO left_part  
            PREPEND character at position right FROM text TO right_part  
            
            IF left_part EQUALS right_part THEN  
                INCREMENT k BY two  
                SET left_part TO empty string  
                SET right_part TO empty string  
            END IF  
            
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        
        IF left_part IS NOT empty OR left EQUALS right THEN  
            INCREMENT k BY one  
        END IF  
        
        RETURN k  
    END FUNCTION  
END CLASS
