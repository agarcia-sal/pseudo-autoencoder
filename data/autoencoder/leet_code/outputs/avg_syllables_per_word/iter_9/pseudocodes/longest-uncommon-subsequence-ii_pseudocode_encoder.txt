CLASS Solution  
    FUNCTION findLUSlength(self, strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO iterator over t  
            RETURN result of verifying for every character c in s that c is contained in it  
        END FUNCTION  

        SORT strs by descending length and then lexicographically ascending  

        FOR each index i and element word IN strs  
            SET unique TO true  
            FOR each index j and element other IN strs  
                IF i NOT EQUALS j AND is_subsequence(word, other) THEN  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique THEN  
                RETURN length of word  
            END IF  
        END FOR  

        RETURN negative one  
    END FUNCTION  
END CLASS
