CLASS Solution  
    FUNCTION combinationSum3(self, k, n)  
        FUNCTION backtrack(start, path, target, k)  
            IF target EQUALS zero AND k EQUALS zero THEN  
                APPEND a copy of path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero OR k LESS THAN zero THEN  
                RETURN  
            END IF  
            FOR i FROM start TO nine  
                APPEND i TO path  
                CALL backtrack WITH parameters i PLUS one, path, target MINUS i, k MINUS one  
                REMOVE last element FROM path  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH parameters one, empty list, n, k  
        RETURN result  
    END FUNCTION  
END CLASS
