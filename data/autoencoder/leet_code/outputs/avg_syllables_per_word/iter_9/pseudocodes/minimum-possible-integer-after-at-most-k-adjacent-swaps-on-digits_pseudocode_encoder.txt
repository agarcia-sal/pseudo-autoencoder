CLASS Solution  
    FUNCTION minInteger(self, num, k)  
        SET num TO list of characters in num  
        SET n TO length of num  
        
        WHILE k GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS two  
                IF character at position i in num GREATER THAN character at position i PLUS one in num THEN  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO minimum of i PLUS k PLUS one AND n MINUS one  
                        IF character at position j in num LESS THAN min_digit THEN  
                            SET min_digit TO character at position j in num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    REARRANGE num FROM index i TO min_index TO place character at min_index at position i, followed by previous sublist from i TO min_index MINUS one  
                    DECREMENT k BY (min_index MINUS i)  
                    SET swapped TO true  
                    BREAK loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS false THEN  
                BREAK loop  
            END IF  
        END WHILE  
        
        RETURN concatenation of characters in num  
    END FUNCTION  
END CLASS
