CLASS Solution  
    FUNCTION maxDistance(self, arrays)  
        SET min_val TO first element of first array in arrays  
        SET max_val TO last element of first array in arrays  
        SET max_distance TO zero  
        
        FOR index FROM one TO length of arrays MINUS one  
            SET current_array TO element at index in arrays  
            SET current_min TO first element in current_array  
            SET current_max TO last element in current_array  
            
            SET max_distance TO largest among max_distance, absolute value of current_max MINUS min_val, absolute value of max_val MINUS current_min  
            
            SET min_val TO smaller between min_val and current_min  
            SET max_val TO larger between max_val and current_max  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
