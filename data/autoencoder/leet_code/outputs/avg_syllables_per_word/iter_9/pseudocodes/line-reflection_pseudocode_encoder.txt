CLASS Solution  
    FUNCTION isReflected(self, points)  
        SET unique_points TO the set of tuples created from points  
        SET min_x TO the minimum value of first element in each point in unique_points  
        SET max_x TO the maximum value of first element in each point in unique_points  
        SET line_of_reflection TO the sum of min_x and max_x DIVIDED BY two  
        
        FOR each x, y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple of reflected_x and y NOT IN unique_points THEN  
                RETURN False  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
