CLASS Solution  
    FUNCTION maximumSwap(self, num)  
        SET digits TO list of characters from string representation of num  
        SET last TO empty dictionary  
        FOR each index i AND element d IN enumeration of digits  
            SET last at integer value of d TO i  
        END FOR  
        
        FOR each index i AND element d IN enumeration of digits  
            FOR each integer d2 FROM nine DOWN TO integer value of d PLUS one STEP minus one  
                IF last at d2 EXISTS AND last at d2 GREATER THAN i THEN  
                    SET temporary variable TO digits at i  
                    SET digits at i TO digits at last at d2  
                    SET digits at last at d2 TO temporary variable  
                    RETURN integer value of concatenation of digits  
                END IF  
            END FOR  
        END FOR  
        
        RETURN num  
    END FUNCTION  
END CLASS
