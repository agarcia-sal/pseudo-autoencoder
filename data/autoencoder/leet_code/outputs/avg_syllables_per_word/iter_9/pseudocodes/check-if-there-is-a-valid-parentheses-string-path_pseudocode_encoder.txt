CLASS Solution  
    FUNCTION hasValidPath(self, grid)  
        SET m TO length of grid  
        SET n TO length of first element in grid  
        
        IF remainder of (m PLUS n MINUS 1) divided by 2 NOT EQUALS 0 THEN  
            RETURN False  
        END IF  
        
        INITIALIZE dp as three-dimensional structure of size (m PLUS 1) by (n PLUS 1) by (m PLUS n PLUS 1) filled with False  
        
        SET dp at position zero, one, zero TO True  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n DO  
                    IF dp at position i, j PLUS one, k OR dp at position i PLUS one, j, k IS True THEN  
                        IF element at grid position i, j EQUALS opening parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one THEN  
                            SET dp at position i PLUS one, j PLUS one, k PLUS one TO True  
                        ELSE IF element at grid position i, j EQUALS closing parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero THEN  
                            SET dp at position i PLUS one, j PLUS one, k MINUS one TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position m, n, zero  
    END FUNCTION  
END CLASS
