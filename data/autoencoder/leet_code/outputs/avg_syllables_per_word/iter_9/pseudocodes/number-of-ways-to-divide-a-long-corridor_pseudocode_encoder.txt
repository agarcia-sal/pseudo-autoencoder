CLASS Solution  
    FUNCTION numberOfWays(self, corridor)  
        SET MOD TO one billion plus seven  
        SET seats TO list of all indices i in corridor where character at i EQUALS uppercase S  
        IF length of seats modulo two NOT EQUALS zero OR length of seats LESS THAN two THEN  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO length of seats MINUS one, stepping by two  
            SET difference TO value at seats index MINUS value at seats index minus one  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways modulo MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
