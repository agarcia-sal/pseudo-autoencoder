CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value as first element of values, left as None, right as None  
    SET i TO one  
    CREATE an empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN length of values AND values at index i IS NOT None THEN  
            SET node.left TO new TreeNode with value as values at i, left as None, right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at index i IS NOT None THEN  
            SET node.right TO new TreeNode with value as values at i, left as None, right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters p.left and q.left AND is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root IS None THEN  
            RETURN  
        END IF  
        SET stack TO list containing root  
        WHILE stack IS NOT empty DO  
            REMOVE last element FROM stack AND SET TO node  
            IF node.right IS NOT None THEN  
                APPEND node.right TO stack  
            END IF  
            IF node.left IS NOT None THEN  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO None  
            IF stack IS NOT empty THEN  
                SET node.right TO last element in stack  
            ELSE  
                SET node.right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
