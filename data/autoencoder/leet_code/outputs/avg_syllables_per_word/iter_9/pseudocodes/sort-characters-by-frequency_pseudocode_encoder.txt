CLASS Solution  
    FUNCTION frequencySort(self, s)  
        CALL frequency TO THE RESULT OF Counter WITH s  
        CALL sorted_characters TO THE RESULT OF sorting the keys of frequency BY their frequency in descending order  
        CALL result TO empty string  
        FOR each character IN sorted_characters  
            APPEND the character REPEATED frequency of character TIMES TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
