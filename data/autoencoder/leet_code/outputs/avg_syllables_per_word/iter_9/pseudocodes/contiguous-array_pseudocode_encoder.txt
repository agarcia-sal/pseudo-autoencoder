CLASS Solution  
    FUNCTION findMaxLength(self, nums)  
        SET prefix_sum_index TO a dictionary with key zero mapped to minus one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i and element num IN enumeration of nums  
            IF num EQUALS one THEN  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS IN prefix_sum_index THEN  
                SET max_length TO the greater of max_length and (i MINUS prefix_sum_index at key count)  
            ELSE  
                SET prefix_sum_index at key count TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
