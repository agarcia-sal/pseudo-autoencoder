CLASS Solution  
    FUNCTION ladderLength(self, beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN 0  
        END IF  
        
        SET word_dict TO a default dictionary that maps each key TO an empty list  
        SET word_length TO the length of beginWord  
        
        FOR each word IN wordList  
            FOR each index i FROM 0 TO word_length MINUS 1  
                SET intermediate_key TO concatenation of substring of word from start TO i MINUS 1, followed by the character star, followed by substring of word from i PLUS 1 TO end  
                APPEND word TO the list at word_dict indexed by intermediate_key  
            END FOR  
        END FOR  
        
        SET queue TO a double ended queue containing a tuple of beginWord and the integer one  
        SET visited TO a set containing beginWord  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue and ASSIGN to current_word and current_length  
            
            FOR each index i FROM 0 TO word_length MINUS 1  
                SET intermediate_word TO concatenation of substring of current_word from start TO i MINUS 1, followed by the character star, followed by substring of current_word from i PLUS 1 TO end  
                FOR each word IN the list at word_dict indexed by intermediate_word  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND a tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN 0  
    END FUNCTION  
END CLASS
