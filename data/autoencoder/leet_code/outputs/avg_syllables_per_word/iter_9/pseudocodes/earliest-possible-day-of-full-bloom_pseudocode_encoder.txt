CLASS Solution  
    FUNCTION earliestFullBloom(self, plantTime, growTime)  
        SET tasks TO sorted list of pairs of corresponding elements from plantTime and growTime, sorted by second element in descending order  
        
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        
        FOR each pair (plant, grow) IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
