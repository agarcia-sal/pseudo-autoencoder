CLASS Solution  
    FUNCTION rob(self, nums)  
        IF nums IS empty THEN  
            RETURN zero  
        END IF  

        IF length of nums EQUALS one THEN  
            RETURN first element of nums  
        END IF  

        SET dp TO list of zeros with length equal to length of nums  
        SET first element of dp TO first element of nums  
        SET second element of dp TO greater value between first element of nums AND second element of nums  

        FOR index FROM two TO length of nums MINUS one  
            SET dp at index TO greater value between dp at previous index AND sum of dp at two indices before AND element at index of nums  
        END FOR  

        RETURN last element of dp  
    END FUNCTION  
END CLASS
