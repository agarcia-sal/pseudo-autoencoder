CLASS Solution  
    FUNCTION canCross(self, stones)  
        IF stones IS empty OR length of stones LESS THAN 2 THEN  
            RETURN False  
        END IF  
        
        SET stone_set TO a set containing all elements in stones  
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the last element in stones THEN  
                RETURN True  
            END IF  
            
            IF the pair of position and jump EXISTS in memo THEN  
                RETURN the value in memo for the pair of position and jump  
            END IF  
            
            FOR next_jump FROM jump MINUS 1 TO jump PLUS 1  
                IF next_jump GREATER THAN 0 AND position PLUS next_jump EXISTS in stone_set THEN  
                    IF can_jump_to(position PLUS next_jump, next_jump) THEN  
                        SET memo at pair of position and jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo at pair of position and jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the first element in stones, 0)  
    END FUNCTION  
END CLASS
