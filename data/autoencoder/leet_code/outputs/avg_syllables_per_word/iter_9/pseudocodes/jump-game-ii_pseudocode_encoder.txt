CLASS Solution  
    FUNCTION jump(self, nums)  
        IF length of nums LESS THAN OR EQUAL TO 1 THEN  
            RETURN 0  
        END IF  
        
        SET jumps TO 0  
        SET current_end TO 0  
        SET farthest TO 0  
        
        FOR i FROM 0 TO length of nums MINUS 2  
            SET farthest TO the greater of farthest AND i PLUS element at index i of nums  
            
            IF i EQUALS current_end THEN  
                INCREMENT jumps BY 1  
                SET current_end TO farthest  
                
                IF current_end GREATER THAN OR EQUAL TO length of nums MINUS 1 THEN  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        
        RETURN jumps  
    END FUNCTION  
END CLASS
