CLASS Solution  
    FUNCTION topKFrequent(self, nums, k)  
        SET frequency_map TO a mapping with default integer zero  
        FOR each num IN nums  
            INCREMENT frequency_map value corresponding to num BY one  
        END FOR  

        SET max_frequency TO the maximum value among frequency_map values  
        SET buckets TO a list of empty lists with length max_frequency PLUS one  

        FOR each pair of num and freq IN frequency_map items  
            APPEND num TO the list at position freq in buckets  
        END FOR  

        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one STEP minus one  
            IF the length of buckets at position freq GREATER THAN zero THEN  
                EXTEND result BY the elements in buckets at position freq  
                IF the length of result GREATER THAN OR EQUALS k THEN  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
