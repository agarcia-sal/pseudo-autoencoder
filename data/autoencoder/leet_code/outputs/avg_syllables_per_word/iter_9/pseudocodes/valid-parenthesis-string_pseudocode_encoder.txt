CLASS Solution  
    FUNCTION checkValidString(self, s)  
        SET low TO 0  
        SET high TO 0  

        FOR each char IN s  
            IF char EQUALS open parenthesis THEN  
                INCREMENT low BY 1  
                INCREMENT high BY 1  
            ELSE IF char EQUALS close parenthesis THEN  
                DECREMENT low BY 1  
                DECREMENT high BY 1  
            ELSE  
                DECREMENT low BY 1  
                INCREMENT high BY 1  
            END IF  

            IF high LESS THAN 0 THEN  
                RETURN false  
            END IF  

            IF low LESS THAN 0 THEN  
                SET low TO 0  
            END IF  
        END FOR  

        RETURN low EQUALS 0  
    END FUNCTION  
END CLASS
