CLASS Solution  
    FUNCTION generateMatrix(self, n)  
        SET matrix TO a two-dimensional list of size n by n with all elements set to zero  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right  
            FOR index FROM left TO right  
                SET element at matrix at row top and column index TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            
            FOR index FROM top TO bottom  
                SET element at matrix at row index and column right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            
            IF top LESS THAN OR EQUAL TO bottom THEN  
                FOR index FROM right DOWN TO left  
                    SET element at matrix at row bottom and column index TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            
            IF left LESS THAN OR EQUAL TO right THEN  
                FOR index FROM bottom DOWN TO top  
                    SET element at matrix at row index and column left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
