CLASS Solution  
    FUNCTION minSwapsCouples(self, row)  
        CREATE empty dictionary position  
        FOR each index i AND person IN row BY enumeration  
            SET position at key person TO i  
        END FOR  
        SET swaps TO zero  
        
        FOR index i FROM zero TO length of row MINUS one STEP two  
            SET correct_pair TO row element at i EXCLUSIVE OR with one  
            
            IF row element at i PLUS one NOT EQUALS correct_pair THEN  
                SET swap_idx TO position at key correct_pair  
                
                SWAP row element at i PLUS one WITH row element at swap_idx  
                
                SET position at key row element at swap_idx TO swap_idx  
                SET position at key row element at i PLUS one TO i PLUS one  
                
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
