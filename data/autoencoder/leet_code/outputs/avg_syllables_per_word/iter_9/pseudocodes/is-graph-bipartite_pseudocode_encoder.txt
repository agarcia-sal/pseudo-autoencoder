CLASS Solution  
    FUNCTION isBipartite(self, graph)  
        SET n TO length of graph  
        SET color TO list of size n with all elements set to zero  
        
        FUNCTION dfs(node, c)  
            SET color at position node TO c  
            FOR each neighbor IN graph at position node  
                IF color at position neighbor EQUALS c THEN  
                    RETURN False  
                END IF  
                IF color at position neighbor EQUALS zero THEN  
                    IF dfs(neighbor, negative c) EQUALS False THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF color at position i EQUALS zero THEN  
                IF dfs(i, one) EQUALS False THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
