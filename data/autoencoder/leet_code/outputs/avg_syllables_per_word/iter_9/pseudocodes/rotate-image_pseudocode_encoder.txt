CLASS Solution  
    FUNCTION rotate(self, matrix)  
        SET n TO length of matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO matrix at position i j  
                SET matrix at position i j TO matrix at position j i  
                SET matrix at position j i TO temporary_value  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            REVERSE elements in matrix at position i  
        END FOR  
    END FUNCTION  
END CLASS
