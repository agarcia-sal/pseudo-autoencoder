CLASS Solution  
    FUNCTION minSubArrayLen(self, target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinite  

        FOR right FROM zero TO length of nums MINUS one  
            SET current_sum TO current_sum PLUS element at index right in nums  

            WHILE current_sum GREATER THAN OR EQUALS target  
                SET min_length TO smaller value between min_length AND right MINUS left PLUS one  
                SET current_sum TO current_sum MINUS element at index left in nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  

        IF min_length NOT EQUALS infinite THEN  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
