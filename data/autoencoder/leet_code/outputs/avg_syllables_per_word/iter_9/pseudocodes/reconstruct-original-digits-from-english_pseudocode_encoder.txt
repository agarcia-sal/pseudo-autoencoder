CLASS Solution  
    FUNCTION originalDigits(self, s)  
        IMPORT Counter FROM collections  
        
        SET count TO Counter of characters in s  
        
        INITIALIZE digits AS list of ten zeros  
        
        SET digits at position zero TO count at character 'z'  
        SET digits at position two TO count at character 'w'  
        SET digits at position four TO count at character 'u'  
        SET digits at position six TO count at character 'x'  
        SET digits at position eight TO count at character 'g'  
        
        SET digits at position three TO count at character 'h' MINUS digits at position eight  
        SET digits at position five TO count at character 'f' MINUS digits at position four  
        SET digits at position seven TO count at character 's' MINUS digits at position six  
        SET digits at position nine TO count at character 'i' MINUS digits at position five MINUS digits at position six MINUS digits at position eight  
        SET digits at position one TO count at character 'o' MINUS digits at position zero MINUS digits at position two MINUS digits at position four  
        
        INITIALIZE result AS empty list  
        FOR i FROM zero TO nine  
            APPEND string representation of i REPEATED digits at position i TIMES TO result  
        END FOR  
        
        RETURN concatenation of elements in result  
    END FUNCTION  
END CLASS
