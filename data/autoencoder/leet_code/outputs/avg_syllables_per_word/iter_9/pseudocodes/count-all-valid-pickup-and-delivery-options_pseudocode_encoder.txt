CLASS Solution  
    FUNCTION countOrders(self, n)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SET count TO one  
        FOR i FROM two TO n  
            SET count TO count MULTIPLIED BY i MULTIPLIED BY (two MULTIPLIED BY i MINUS one) MODULO MOD  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
