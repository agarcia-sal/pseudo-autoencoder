CLASS Solution  
    FUNCTION fullJustify(self, words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF length of line EQUALS one OR is_last_line THEN  
                RETURN concatenation of words in line separated by single space  
                    FOLLOWED BY spaces repeated times of (maxWidth MINUS num_of_letters MINUS length of line PLUS one)  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO length of line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                
                FOR i FROM zero TO num_of_gaps MINUS one  
                    IF i LESS THAN extra_spaces THEN  
                        APPEND spaces repeated times of (spaces_between_words PLUS one) TO line at position i  
                    ELSE  
                        APPEND spaces repeated times of spaces_between_words TO line at position i  
                    END IF  
                END FOR  
                
                RETURN concatenation of all elements in line without separator  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS length of word PLUS length of line GREATER THAN maxWidth THEN  
                APPEND result WITH justify_line(line, num_of_letters, False)  
                SET line TO empty list  
                SET num_of_letters TO zero  
            END IF  
            
            APPEND word TO line  
            INCREMENT num_of_letters BY length of word  
        END FOR  
        
        APPEND result WITH justify_line(line, num_of_letters, True)  
        
        RETURN result  
    END FUNCTION  
END CLASS
