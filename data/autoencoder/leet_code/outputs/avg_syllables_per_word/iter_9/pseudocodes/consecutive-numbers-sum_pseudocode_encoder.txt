CLASS Solution  
    FUNCTION consecutiveNumbersSum(self, n)  
        SET count TO zero  
        FOR k FROM one TO n INCLUSIVE  
            IF two MULTIPLIED BY n MODULO k EQUALS zero THEN  
                SET m TO ((two MULTIPLIED BY n DIVIDED BY k) MINUS k PLUS one) DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUALS the integer conversion of m THEN  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
