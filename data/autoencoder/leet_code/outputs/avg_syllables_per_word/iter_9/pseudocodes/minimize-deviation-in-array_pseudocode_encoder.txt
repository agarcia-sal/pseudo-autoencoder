CLASS Solution  
    FUNCTION minimumDeviation(self, nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF num modulo 2 EQUALS 1 THEN  
                SET num TO num multiplied BY 2  
            END IF  
            IF num LESS THAN min_value THEN  
                SET min_value TO num  
            END IF  
            ADD negative of num TO max_heap  
        END FOR  
        
        SET min_deviation TO infinity  
        
        WHILE negative of first element in max_heap modulo 2 EQUALS 0  
            SET max_value TO negative of REMOVE first element FROM max_heap  
            IF max_value MINUS min_value LESS THAN min_deviation THEN  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value divided BY 2  
            IF max_value LESS THAN min_value THEN  
                SET min_value TO max_value  
            END IF  
            ADD negative of max_value TO max_heap  
        END WHILE  
        
        IF negative of first element in max_heap MINUS min_value LESS THAN min_deviation THEN  
            SET min_deviation TO negative of first element in max_heap MINUS min_value  
        END IF  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
