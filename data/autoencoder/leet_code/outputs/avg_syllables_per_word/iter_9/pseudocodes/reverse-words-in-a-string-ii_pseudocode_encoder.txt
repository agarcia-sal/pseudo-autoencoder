CLASS Solution  
    FUNCTION reverseWords(self, s)  
        CALL reverse ON s  
        SET start TO zero  
        FOR end FROM zero TO length of s MINUS one  
            IF element at s with index end EQUALS a space character THEN  
                CALL reverse ON elements of s from start TO end MINUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        CALL reverse ON elements of s from start TO end of s  
    END FUNCTION  
END CLASS
