CLASS Solution  
    FUNCTION totalHammingDistance(self, nums)  
        SET total_distance TO zero  
        FOR each bit_position FROM zero TO thirty-one  
            SET count_ones TO zero  
            FOR each number IN nums  
                INCREMENT count_ones BY the one bit value of the number shifted right by bit_position  
            END FOR  
            SET count_zeros TO the length of nums MINUS count_ones  
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros  
        END FOR  
        RETURN total_distance  
    END FUNCTION  
END CLASS
