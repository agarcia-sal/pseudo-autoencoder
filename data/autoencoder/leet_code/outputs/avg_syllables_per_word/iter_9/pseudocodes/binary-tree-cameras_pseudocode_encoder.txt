CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val as first element of values  
    SET i TO one  
    CREATE queue as deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        REMOVE first element FROM queue AND SET as node  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode with val as values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode with val as values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node IS none THEN  
                RETURN tuple of zero, zero, positive infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO dfs(node.left)  
            SET right_no_camera, right_monitored, right_camera TO dfs(node.right)  
            
            SET no_camera TO left_monitored PLUS right_monitored  
            
            SET monitored TO minimum of (left_camera PLUS minimum of right_monitored AND right_camera),  
                                        (right_camera PLUS minimum of left_monitored AND left_camera),  
                                        left_camera PLUS right_camera  
            
            SET camera TO one PLUS minimum of left_no_camera, left_monitored, left_camera PLUS minimum of right_no_camera, right_monitored, right_camera  
            
            RETURN tuple of no_camera, monitored, camera  
        END FUNCTION  
        
        SET root_no_camera, root_monitored, root_camera TO dfs(root)  
        RETURN minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
