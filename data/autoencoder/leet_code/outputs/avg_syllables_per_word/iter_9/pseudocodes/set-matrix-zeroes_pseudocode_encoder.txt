CLASS Solution  
    FUNCTION setZeroes(self, matrix)  
        IF matrix IS empty OR first element of matrix IS empty THEN  
            RETURN  
        END IF  

        SET m TO length of matrix  
        SET n TO length of first element of matrix  

        SET first_row_has_zero TO False  
        FOR each index j FROM zero TO n minus one  
            IF element at matrix zero row and j column EQUALS zero THEN  
                SET first_row_has_zero TO True  
                BREAK  
            END IF  
        END FOR  

        SET first_col_has_zero TO False  
        FOR each index i FROM zero TO m minus one  
            IF element at matrix i row and zero column EQUALS zero THEN  
                SET first_col_has_zero TO True  
                BREAK  
            END IF  
        END FOR  

        FOR each index i FROM one TO m minus one  
            FOR each index j FROM one TO n minus one  
                IF element at matrix i row and j column EQUALS zero THEN  
                    SET element at matrix i row and zero column TO zero  
                    SET element at matrix zero row and j column TO zero  
                END IF  
            END FOR  
        END FOR  

        FOR each index i FROM one TO m minus one  
            FOR each index j FROM one TO n minus one  
                IF element at matrix i row and zero column EQUALS zero OR element at matrix zero row and j column EQUALS zero THEN  
                    SET element at matrix i row and j column TO zero  
                END IF  
            END FOR  
        END FOR  

        IF first_row_has_zero THEN  
            FOR each index j FROM zero TO n minus one  
                SET element at matrix zero row and j column TO zero  
            END FOR  
        END IF  

        IF first_col_has_zero THEN  
            FOR each index i FROM zero TO m minus one  
                SET element at matrix i row and zero column TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
