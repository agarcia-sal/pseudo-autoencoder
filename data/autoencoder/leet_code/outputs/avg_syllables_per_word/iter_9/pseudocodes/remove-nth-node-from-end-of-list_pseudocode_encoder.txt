CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to first element of values and next equal to None  
    SET p TO head  
    FOR each val IN the elements of values starting from the second element  
        SET node TO new ListNode with val equal to val and next equal to None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(self, head, n)  
        SET dummy TO new ListNode with val equal to zero and next equal to None  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n INCLUSIVE  
            SET first TO first.next  
        END FOR  
        WHILE first IS NOT None  
            SET first TO first.next  
            SET second TO second.next  
        END WHILE  
        SET second.next TO second.next.next  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
