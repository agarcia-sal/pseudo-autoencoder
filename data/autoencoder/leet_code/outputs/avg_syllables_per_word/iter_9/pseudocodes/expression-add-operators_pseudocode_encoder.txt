CLASS Solution  
    FUNCTION addOperators(self, num, target)  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS length of num THEN  
                IF value EQUALS target THEN  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO length of num PLUS one  
                SET operand TO substring of num from index TO i MINUS one  
                IF length of operand GREATER THAN one AND first character of operand EQUALS zero THEN  
                    CONTINUE to next iteration  
                END IF  
                
                IF index EQUALS zero THEN  
                    CALL dfs WITH parameters i, operand, integer value of operand, integer value of operand  
                ELSE  
                    CALL dfs WITH parameters i, path CONCATENATED WITH plus SIGN CONCATENATED WITH operand, value PLUS integer value of operand, integer value of operand  
                    CALL dfs WITH parameters i, path CONCATENATED WITH minus SIGN CONCATENATED WITH operand, value MINUS integer value of operand, NEGATIVE integer value of operand  
                    CALL dfs WITH parameters i, path CONCATENATED WITH multiplication SIGN CONCATENATED WITH operand, value MINUS prev PLUS prev MULTIPLIED BY integer value of operand, prev MULTIPLIED BY integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL dfs WITH parameters zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
