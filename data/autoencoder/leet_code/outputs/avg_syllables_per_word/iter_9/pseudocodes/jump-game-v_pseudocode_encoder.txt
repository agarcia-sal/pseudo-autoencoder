CLASS Solution  
    FUNCTION maxJumps(self, arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO 1  
            FOR x FROM 1 TO d  
                IF i PLUS x LESS THAN length of arr THEN  
                    IF element at arr at position i PLUS x GREATER THAN OR EQUALS element at arr at position i THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the greater value BETWEEN max_jumps AND one PLUS result of dp called with argument i PLUS x  
                END IF  
            END FOR  
            FOR x FROM 1 TO d  
                IF i MINUS x GREATER THAN OR EQUALS zero THEN  
                    IF element at arr at position i MINUS x GREATER THAN OR EQUALS element at arr at position i THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the greater value BETWEEN max_jumps AND one PLUS result of dp called with argument i MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET result TO the greatest value of dp called with each i from zero TO length of arr MINUS one  
        RETURN result  
    END FUNCTION  
END CLASS
