CLASS Solution  
    FUNCTION numDistinct(self, s, t)  
        SET m TO length of s  
        SET n TO length of t  
        
        FUNCTION initializeDP(rows, columns)  
            CREATE a two dimensional list dp with rows number of lists each containing columns number of zeros  
            RETURN dp  
        END FUNCTION  
        
        SET dp TO result of initializeDP m PLUS one, n PLUS one  
        
        FOR i FROM zero TO m  
            SET dp at i, zero TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one in s EQUALS character at position j MINUS one in t THEN  
                    SET dp at i, j TO dp at i MINUS one, j MINUS one PLUS dp at i MINUS one, j  
                ELSE  
                    SET dp at i, j TO dp at i MINUS one, j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at m, n  
    END FUNCTION  
END CLASS
