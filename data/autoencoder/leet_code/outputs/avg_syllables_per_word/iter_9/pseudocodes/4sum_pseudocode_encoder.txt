CLASS Solution  
    FUNCTION fourSum(self, nums, target)  
        CALL nums sort  
        SET n TO length of nums  
        SET quadruplets TO empty list  
        
        FOR i FROM zero TO n MINUS three MINUS one  
            IF i GREATER THAN zero AND nums at i EQUALS nums at i MINUS one THEN  
                CONTINUE  
            END IF  
            
            FOR j FROM i PLUS one TO n MINUS two MINUS one  
                IF j GREATER THAN i PLUS one AND nums at j EQUALS nums at j MINUS one THEN  
                    CONTINUE  
                END IF  
                
                SET left TO j PLUS one  
                SET right TO n MINUS one  
                
                WHILE left LESS THAN right  
                    SET total TO nums at i PLUS nums at j PLUS nums at left PLUS nums at right  
                    
                    IF total EQUALS target THEN  
                        APPEND list of nums at i, nums at j, nums at left, nums at right TO quadruplets  
                        
                        WHILE left LESS THAN right AND nums at left EQUALS nums at left PLUS one  
                            INCREMENT left BY one  
                        END WHILE  
                        
                        WHILE left LESS THAN right AND nums at right EQUALS nums at right MINUS one  
                            DECREMENT right BY one  
                        END WHILE  
                        
                        INCREMENT left BY one  
                        DECREMENT right BY one  
                    
                    ELSE IF total LESS THAN target THEN  
                        INCREMENT left BY one  
                    
                    ELSE  
                        DECREMENT right BY one  
                    
                    END IF  
                
                END WHILE  
            
            END FOR  
        
        END FOR  
        
        RETURN quadruplets  
    END FUNCTION  
END CLASS
