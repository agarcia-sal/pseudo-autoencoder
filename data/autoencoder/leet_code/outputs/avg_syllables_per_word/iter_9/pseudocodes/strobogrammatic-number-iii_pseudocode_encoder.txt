CLASS Solution  
    FUNCTION strobogrammaticInRange(self, low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS 0 THEN  
                RETURN list containing empty string  
            END IF  
            
            IF n EQUALS 1 THEN  
                RETURN list containing "0", "1", and "8"  
            END IF  
            
            SET middles TO result of generate_strobogrammatic with n MINUS 2 and final_length  
            SET result TO empty list  
            
            FOR each middle IN middles  
                IF n NOT EQUALS final_length THEN  
                    APPEND concatenation of "0", middle, and "0" TO result  
                END IF  
                
                APPEND concatenation of "1", middle, and "1" TO result  
                APPEND concatenation of "6", middle, and "9" TO result  
                APPEND concatenation of "8", middle, and "8" TO result  
                APPEND concatenation of "9", middle, and "6" TO result  
            END FOR  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO 0  
            FOR length FROM length of low TO length of high  
                SET numbers TO generate_strobogrammatic with length and length  
                FOR each num IN numbers  
                    IF (length EQUALS length of low AND num LESS THAN low) OR (length EQUALS length of high AND num GREATER THAN high) THEN  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY 1  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN result of count_strobogrammatic with low and high  
    END FUNCTION  
END CLASS
