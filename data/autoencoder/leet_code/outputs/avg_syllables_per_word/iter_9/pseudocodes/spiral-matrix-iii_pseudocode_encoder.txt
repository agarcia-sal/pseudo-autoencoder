CLASS Solution  
    FUNCTION spiralMatrixIII(self, rows, cols, rStart, cStart)  
        SET directions TO list of four pairs: first pair is zero and one, second pair is one and zero, third pair is zero and negative one, fourth pair is negative one and zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE length of visited LESS THAN rows MULTIPLIED BY cols DO  
            IF increment MODULO two EQUALS zero THEN  
                INCREMENT steps BY one  
            END IF  
            FOR each iteration FROM one TO steps DO  
                IF x GREATER THAN OR EQUAL TO zero AND x LESS THAN rows AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN cols THEN  
                    APPEND list of x and y TO visited  
                END IF  
                SET dx TO first element of directions at position direction_index  
                SET dy TO second element of directions at position direction_index  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO (direction_index PLUS one) MODULO four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
