CLASS Solution  
    FUNCTION combinationSum4(self, nums, target)  
        SET dp TO list of zeros with length target PLUS one  
        SET dp at index zero TO one  
        FOR amount FROM one TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    INCREMENT dp at index amount BY dp at index amount MINUS num  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at index target  
    END FUNCTION  
END CLASS
