CLASS Solution  
    FUNCTION maximalSquare(self, matrix)  
        IF matrix IS empty OR first element of matrix IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of matrix  
        SET n TO length of first element of matrix  
        INITIALIZE dp AS list of lists with (m PLUS one) rows and (n PLUS one) columns FILLED with zero  
        SET max_side TO zero  
        
        FOR i FROM one TO m DO  
            FOR j FROM one TO n DO  
                IF element at row (i MINUS one) and column (j MINUS one) in matrix EQUALS character '1' THEN  
                    SET dp element at row i and column j TO minimum of dp element at (i MINUS one, j), dp element at (i, j MINUS one), dp element at (i MINUS one, j MINUS one) PLUS one  
                    SET max_side TO maximum of max_side AND dp element at row i and column j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
