CLASS Solution  
    FUNCTION groupAnagrams(self, strs)  
        SET anagrams TO a new mapping from key to list of strings with default empty list  
        FOR each string s IN strs  
            SET sorted_str TO the result of sorting characters of s and then joining them into a string  
            APPEND s TO the list in anagrams identified by key sorted_str  
        END FOR  
        RETURN the list of all values in anagrams  
    END FUNCTION  
END CLASS
