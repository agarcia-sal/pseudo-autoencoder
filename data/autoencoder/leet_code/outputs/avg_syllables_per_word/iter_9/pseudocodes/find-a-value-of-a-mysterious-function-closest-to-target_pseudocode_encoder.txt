CLASS Solution  
    FUNCTION closestToTarget(self, arr, target)  
        SET seen TO empty set  
        SET min_diff TO infinity  
        
        FOR each num IN arr  
            SET seen TO the union of the set created by bitwise AND of num and each element in seen AND the set containing only num  
            
            FOR each val IN seen  
                SET min_diff TO the smaller value between min_diff and the absolute value of val MINUS target  
            END FOR  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
