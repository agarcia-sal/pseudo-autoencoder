CLASS Solution  
    FUNCTION largestOverlap(self, img1, img2)  
        SET n TO length of img1  
        SET points1 TO empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at img1 at position i, j EQUALS one THEN  
                    APPEND pair of i, j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at img2 at position i, j EQUALS one THEN  
                    APPEND pair of i, j TO points2  
                END IF  
            END FOR  
        END FOR  
        
        SET overlap_count TO default dictionary with default zero  
        
        FOR each pair x1, y1 IN points1  
            FOR each pair x2, y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT value at key pair dx, dy IN overlap_count BY one  
            END FOR  
        END FOR  
        
        SET maximum_overlap TO zero  
        IF overlap_count has any values THEN  
            SET maximum_overlap TO maximum value among overlap_count values  
        END IF  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
