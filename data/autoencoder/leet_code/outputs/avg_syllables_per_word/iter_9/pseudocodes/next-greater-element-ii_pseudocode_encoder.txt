CLASS Solution  
    FUNCTION nextGreaterElements(self, nums)  
        SET n TO length of nums  
        SET result TO list of n elements each set to negative one  
        SET stack TO empty list  

        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack is not empty AND nums at position last element of stack LESS THAN nums at position index  
                REMOVE last element FROM stack AND SET as poppedIndex  
                SET result at position poppedIndex TO nums at position index  
            END WHILE  
            IF i LESS THAN n THEN  
                APPEND index TO stack  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
