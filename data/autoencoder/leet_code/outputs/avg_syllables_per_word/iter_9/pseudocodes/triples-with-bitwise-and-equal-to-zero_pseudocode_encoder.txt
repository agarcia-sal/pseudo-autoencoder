CLASS Solution
    FUNCTION countTriplets(self, nums)
        SET pair_and_count TO a new empty mapping from integer to integer with default value zero

        FOR each element i IN nums
            FOR each element j IN nums
                SET bitwise_and_result TO the bitwise AND of i and j
                INCREMENT pair_and_count at key bitwise_and_result BY one
            END FOR
        END FOR

        SET triplet_count TO zero
        FOR each element k IN nums
            FOR each key pair_and and value count IN pair_and_count
                IF the bitwise AND of pair_and and k EQUALS zero THEN
                    INCREMENT triplet_count BY count
                END IF
            END FOR
        END FOR

        RETURN triplet_count
    END FUNCTION
END CLASS
