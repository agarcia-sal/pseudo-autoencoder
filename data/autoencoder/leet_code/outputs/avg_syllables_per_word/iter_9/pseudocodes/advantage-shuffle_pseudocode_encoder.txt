CLASS Solution  
    FUNCTION advantageCount(self, nums1, nums2)  
        SORT nums1 IN ascending order  
        CREATE nums2_with_indices AS list of tuples containing each value from nums2 and its original index  
        SORT nums2_with_indices IN descending order BY the value element  
        CREATE result AS list of zeroes with length equal to length of nums1  
        SET left TO zero  
        SET right TO length of nums1 MINUS one  
        FOR each tuple (value, index) IN nums2_with_indices  
            IF element at position right in nums1 GREATER THAN value THEN  
                SET element at position index in result TO element at position right in nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index in result TO element at position left in nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
