CLASS Solution  
    FUNCTION encode(self, s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN length of string representation of count PLUS length of substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO portion of s from index i to index j inclusive  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five THEN  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR k FROM one TO integer division of n BY two PLUS one  
                IF n MODULO k EQUALS zero THEN  
                    SET repeat_count TO integer division of n BY k  
                    IF substring EQUALS substring of first k characters MULTIPLIED BY repeat_count THEN  
                        SET encoded_candidate TO string representation of repeat_count CONCATENATED WITH open bracket CONCATENATED WITH dp(i, i PLUS k MINUS one) CONCATENATED WITH close bracket  
                        IF length of encoded_candidate LESS THAN length of shortest THEN  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR k FROM i TO j MINUS one  
                SET left_encoded TO dp(i, k)  
                SET right_encoded TO dp(k PLUS one, j)  
                SET combined TO left_encoded CONCATENATED WITH right_encoded  
                IF length of combined LESS THAN length of shortest THEN  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN dp(zero, length of s MINUS one)  
    END FUNCTION  
END CLASS
