CLASS Solution  
    FUNCTION findKthLargest(self, nums, k)  
        SET min_heap TO empty list  
        
        FOR each num IN nums  
            ADD num INTO min_heap USING heap push operation  
            IF size of min_heap GREATER THAN k THEN  
                REMOVE smallest element FROM min_heap USING heap pop operation  
            END IF  
        END FOR  
        
        RETURN first element OF min_heap  
    END FUNCTION  
END CLASS
