CLASS Solution  
    FUNCTION rearrangeString(self, s, k)  
        IF k LESS THAN OR EQUAL TO 1 THEN  
            RETURN s  
        END IF  
        
        SET freq TO FUNCTION Counter taking s  
        
        SET max_heap TO list of pairs where each pair consists of negative frequency and character taken from freq items  
        CALL heapify on max_heap  
        
        SET wait_queue TO empty list  
        
        SET result TO empty list  
        
        WHILE max_heap is not empty DO  
            SET count, char TO result of popping the smallest element from max_heap  
            APPEND char TO result  
            
            APPEND pair of (count incremented by one, char) TO wait_queue  
            
            IF size of wait_queue EQUALS k THEN  
                SET count, char TO removing first element from wait_queue  
                IF count LESS THAN 0 THEN  
                    PUSH pair of (count, char) INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF size of result NOT EQUALS size of s THEN  
            RETURN empty string  
        END IF  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
