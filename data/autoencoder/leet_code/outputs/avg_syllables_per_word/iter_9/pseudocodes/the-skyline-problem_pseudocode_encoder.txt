CLASS Solution  
    FUNCTION getSkyline(self, buildings)  
        SET events TO empty list  
        FOR each left, right, height IN buildings  
            APPEND tuple of left, negative height, right TO events  
            APPEND tuple of right, zero, zero TO events  
        END FOR  

        SORT events BY first element ascending, then by second element ascending  

        SET result TO list containing list of zero and zero  
        SET max_heap TO list containing tuple of zero and infinite value  

        FOR each x, negH, r IN events  
            WHILE first element of max_heap has second value less than or equal to x  
                REMOVE first element FROM max_heap USING heap pop operation  
            END WHILE  

            IF negH NOT EQUAL TO zero THEN  
                ADD tuple of negH and r TO max_heap USING heap push operation  
            END IF  

            IF second value of last element in result NOT EQUAL TO negative of first value of first element in max_heap THEN  
                APPEND list of x and negative of first value in first element of max_heap TO result  
            END IF  
        END FOR  

        IF first element in result EQUALS list of zero and zero THEN  
            REMOVE first element FROM result  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
