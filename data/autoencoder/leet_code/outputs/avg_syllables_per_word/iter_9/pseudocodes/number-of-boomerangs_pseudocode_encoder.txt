CLASS Solution  
    FUNCTION numberOfBoomerangs(self, points)  
        SET ans TO zero  
        FOR each p1 IN points  
            SET cnt TO new Counter  
            FOR each p2 IN points  
                SET dx TO the first element of p1 MINUS the first element of p2  
                SET dy TO the second element of p1 MINUS the second element of p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the value in cnt at key d  
                INCREMENT the value in cnt at key d BY one  
            END FOR  
        END FOR  
        RETURN ans MULTIPLIED BY two  
    END FUNCTION  
END CLASS
