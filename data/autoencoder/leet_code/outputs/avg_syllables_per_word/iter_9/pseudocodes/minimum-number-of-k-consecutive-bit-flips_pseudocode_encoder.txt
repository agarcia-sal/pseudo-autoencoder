CLASS Solution  
    FUNCTION minKBitFlips(self, nums, k)  
        SET n TO length of nums  
        SET flip_count TO zero  
        SET flipped TO a list of zeros with length n plus one  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n minus one  
            SET flip_state TO flip_state exclusive or flipped at position i  
            
            IF value at nums position i EQUALS flip_state THEN  
                IF i plus k GREATER THAN n THEN  
                    RETURN negative one  
                END IF  
                SET flip_state TO flip_state exclusive or one  
                SET flipped at position i plus k TO flipped at position i plus k exclusive or one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
