CLASS Solution  
    FUNCTION isInterleave(self, s1, s2, s3)  
        IF length of s1 PLUS length of s2 NOT EQUALS length of s3 THEN  
            RETURN False  
        END IF  

        SET dp TO list of False repeated length of s2 PLUS one times  
        SET first element of dp TO True  

        FOR i FROM zero TO length of s1  
            FOR j FROM zero TO length of s2  
                IF i GREATER THAN zero THEN  
                    SET dp at position j TO dp at position j AND character at position i MINUS one in s1 EQUALS character at position i PLUS j MINUS one in s3  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET dp at position j TO dp at position j OR (dp at position j MINUS one AND character at position j MINUS one in s2 EQUALS character at position i PLUS j MINUS one in s3)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at position length of s2  
    END FUNCTION  
END CLASS
