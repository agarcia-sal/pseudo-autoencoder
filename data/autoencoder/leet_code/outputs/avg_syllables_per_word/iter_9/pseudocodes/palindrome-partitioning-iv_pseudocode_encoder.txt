CLASS Solution  
    FUNCTION checkPartitioning(self, s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS reverse of sub  
        END FUNCTION  
        
        SET n TO length of s  
        
        SET is_pal TO two-dimensional list of False values with dimensions n by n  
        
        FOR i FROM n MINUS 1 DOWN TO 0  
            FOR j FROM i TO n MINUS 1  
                IF s at position i EQUALS s at position j AND (j MINUS i LESS THAN OR EQUAL TO 1 OR is_pal at position i PLUS 1 and j MINUS 1) THEN  
                    SET is_pal at position i and j TO True  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM 1 TO n MINUS 2  
            IF is_pal at position 0 and i MINUS 1 THEN  
                FOR j FROM i TO n MINUS 2  
                    IF is_pal at position i and j AND is_pal at position j PLUS 1 and n MINUS 1 THEN  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
