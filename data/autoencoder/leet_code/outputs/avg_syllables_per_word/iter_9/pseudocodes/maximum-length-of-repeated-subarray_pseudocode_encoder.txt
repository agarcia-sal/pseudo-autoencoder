CLASS Solution  
    FUNCTION findLength(self, nums1, nums2)  
        SET dp TO a two dimensional list with dimensions length of nums1 plus one by length of nums2 plus one, initialized with zeros  
        SET max_length TO zero  
        FOR i FROM one TO length of nums1  
            FOR j FROM one TO length of nums2  
                IF element at index i minus one in nums1 EQUALS element at index j minus one in nums2 THEN  
                    SET dp at position i and j TO dp at position i minus one and j minus one PLUS one  
                    IF dp at position i and j GREATER THAN max_length THEN  
                        SET max_length TO dp at position i and j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
