CLASS Solution  
    FUNCTION findMaxForm(self, strs, m, n)  
        INITIALIZE dp AS two dimensional array with (m PLUS 1) rows and (n PLUS 1) columns FILLED WITH zero  
        
        FOR each s IN strs  
            SET count_0 TO number of times character zero OCCURS in s  
            SET count_1 TO number of times character one OCCURS in s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp at position i and j TO the greater value BETWEEN dp at position i and j AND dp at position i minus count_0 and j minus count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
