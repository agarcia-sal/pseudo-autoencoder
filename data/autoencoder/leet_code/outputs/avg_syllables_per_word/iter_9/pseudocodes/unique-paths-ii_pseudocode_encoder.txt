CLASS Solution  
    FUNCTION uniquePathsWithObstacles(self, obstacleGrid)  
        IF obstacleGrid EQUALS empty OR obstacleGrid at position zero zero EQUALS one THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of obstacleGrid  
        SET n TO length of first element in obstacleGrid  
        SET dp TO two dimensional collection of zeros with dimensions m by n  
        
        SET dp at position zero zero TO one  
        
        FOR each index j FROM one TO n MINUS one  
            IF obstacleGrid at position zero j EQUALS zero THEN  
                SET dp at position zero j TO dp at position zero j MINUS one  
            END IF  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            IF obstacleGrid at position i zero EQUALS zero THEN  
                SET dp at position i zero TO dp at position i MINUS one zero  
            END IF  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            FOR each index j FROM one TO n MINUS one  
                IF obstacleGrid at position i j EQUALS zero THEN  
                    SET dp at position i j TO dp at position i MINUS one j PLUS dp at position i j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position m MINUS one n MINUS one  
    END FUNCTION  
END CLASS
