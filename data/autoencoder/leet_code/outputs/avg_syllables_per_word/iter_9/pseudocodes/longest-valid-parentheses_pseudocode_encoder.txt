CLASS Solution  
    FUNCTION longestValidParentheses(self, s)  
        INITIALIZE stack WITH a list containing negative one  
        SET max_length TO zero  

        FOR each index i AND character char IN s  
            IF char EQUALS opening parenthesis THEN  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty THEN  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO greater value BETWEEN max_length AND difference BETWEEN i AND last element in stack  
                END IF  
            END IF  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
