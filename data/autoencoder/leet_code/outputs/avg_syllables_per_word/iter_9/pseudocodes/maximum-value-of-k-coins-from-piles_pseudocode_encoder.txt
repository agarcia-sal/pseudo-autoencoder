CLASS Solution  
    FUNCTION maxValueOfCoins(self, piles, k)  
        SET n TO length of piles  
        SET dp TO two dimensional list of zeros with dimensions n plus one by k plus one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO the element at index i minus one in piles  
                SET current_value TO zero  
                FOR l FROM zero TO minimum of j and length of current_pile  
                    IF l GREATER THAN zero THEN  
                        INCREMENT current_value BY the element at index l minus one in current_pile  
                    END IF  
                    SET dp at i and j TO the maximum of dp at i and j AND dp at i minus one and j minus l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at n and k  
    END FUNCTION  
END CLASS
