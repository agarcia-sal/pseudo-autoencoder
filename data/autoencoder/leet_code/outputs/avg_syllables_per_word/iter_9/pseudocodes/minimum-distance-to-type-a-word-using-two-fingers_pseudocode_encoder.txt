CLASS Solution  
    FUNCTION minimumDistance(self, word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None THEN  
                RETURN 0  
            END IF  
            SET x1 AND y1 TO the quotient AND remainder of the division of (the numerical code of c1 MINUS the numerical code of 'A') BY 6  
            SET x2 AND y2 TO the quotient AND remainder of the division of (the numerical code of c2 MINUS the numerical code of 'A') BY 6  
            RETURN the absolute value of (x1 MINUS x2) PLUS the absolute value of (y1 MINUS y2)  
        END FUNCTION  

        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word THEN  
                RETURN 0  
            END IF  
            SET c TO the character in word at position i  
            SET d1 TO the result of distance of f1 AND c PLUS dp of (i PLUS 1, c, f2)  
            SET d2 TO the result of distance of f2 AND c PLUS dp of (i PLUS 1, f1, c)  
            RETURN the minimum of d1 AND d2  
        END FUNCTION  

        RETURN dp(0, None, None)  
    END FUNCTION  
END CLASS
