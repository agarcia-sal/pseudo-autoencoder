CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO maximum 32 bits integer value  
        SET mask TO value that preserves 32 bits  

        WHILE b NOT EQUALS zero  
            SET carry TO bitwise AND of a and b shifted left by one position  
            SET a TO bitwise XOR of a and b then bitwise AND with mask  
            SET b TO bitwise AND of carry and mask  
        END WHILE  

        IF a LESS THAN OR EQUALS MAX THEN  
            RETURN a  
        ELSE  
            RETURN bitwise NOT of bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
