CLASS Solution  
    FUNCTION numTrees(self, n)  
        SET dp TO list of zeros with length n PLUS one  
        SET dp at position zero TO one  
        SET dp at position one TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i  
                SET dp at position i TO dp at position i PLUS dp at position j MINUS one MULTIPLIED BY dp at position i MINUS j  
            END FOR  
        END FOR  
        RETURN dp at position n  
    END FUNCTION  
END CLASS
