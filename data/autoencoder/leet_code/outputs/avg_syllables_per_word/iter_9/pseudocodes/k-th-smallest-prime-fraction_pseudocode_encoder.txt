CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO length of arr  
        SET min_heap TO empty list  
        
        FOR i FROM zero TO n MINUS one MINUS one  
            SET fraction TO arr at i DIVIDED BY arr at n MINUS one  
            PUSH tuple of fraction, i, n MINUS one TO min_heap  
        END FOR  
        
        FOR count FROM zero TO k MINUS one MINUS one  
            POP smallest_tuple FROM min_heap  
            SET _, i, j TO smallest_tuple  
            IF j MINUS one GREATER THAN i THEN  
                SET new_fraction TO arr at i DIVIDED BY arr at j MINUS one  
                PUSH tuple of new_fraction, i, j MINUS one TO min_heap  
            END IF  
        END FOR  
        
        POP kth_tuple FROM min_heap  
        SET _, i, j TO kth_tuple  
        RETURN list containing arr at i AND arr at j  
    END FUNCTION  
END CLASS
