CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
        SET self.rank TO list of zeros with length size  
        SET self.size TO list of ones with length size  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF self.parent at index u NOT EQUALS u THEN  
            SET self.parent at index u TO result of self.find called with self.parent at index u  
        END IF  
        RETURN self.parent at index u  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO result of self.find called with u  
        SET rootV TO result of self.find called with v  
        IF rootU NOT EQUALS rootV THEN  
            IF self.rank at index rootU GREATER THAN self.rank at index rootV THEN  
                SET self.parent at index rootV TO rootU  
                INCREMENT self.size at index rootU BY self.size at index rootV  
            ELSE IF self.rank at index rootU LESS THAN self.rank at index rootV THEN  
                SET self.parent at index rootU TO rootV  
                INCREMENT self.size at index rootV BY self.size at index rootU  
            ELSE  
                SET self.parent at index rootV TO rootU  
                INCREMENT self.size at index rootU BY self.size at index rootV  
                INCREMENT self.rank at index rootU BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each char IN word  
                SET mask TO mask BITWISE OR with one left-shifted by (numeric value of char MINUS numeric value of 'a')  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO default mapping of lists  
        SET n TO length of words  
        SET uf TO a new UnionFind instance initialized with n  

        FOR each index i and word IN words  
            SET mask TO result of bitmask called with word  
            APPEND i TO word_map at key mask  
        END FOR  

        FOR each index i and word IN words  
            SET mask TO result of bitmask called with word  
            FOR j FROM zero TO twenty-five  
                SET new_mask TO mask XOR with one left-shifted by j  
                IF new_mask EXISTS in word_map THEN  
                    FOR each k IN word_map at key new_mask  
                        CALL uf.union with i and k  
                    END FOR  
                END IF  

                IF mask BITWISE AND with one left-shifted by j EQUALS zero THEN  
                    SET new_mask TO mask BITWISE OR with one left-shifted by j  
                    IF new_mask EXISTS in word_map THEN  
                        FOR each k IN word_map at key new_mask  
                            CALL uf.union with i and k  
                        END FOR  
                    END IF  
                END IF  

                IF mask BITWISE AND with one left-shifted by j NOT EQUALS zero THEN  
                    FOR k FROM zero TO twenty-five  
                        IF mask BITWISE AND with one left-shifted by k EQUALS zero THEN  
                            SET new_mask TO mask XOR with one left-shifted by j BITWISE OR with one left-shifted by k  
                            IF new_mask EXISTS in word_map THEN  
                                FOR each l IN word_map at key new_mask  
                                    CALL uf.union with i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO default mapping of integers  
        FOR i FROM zero TO n MINUS one  
            SET root TO result of uf.find called with i  
            INCREMENT root_sizes at key root BY one  
        END FOR  

        SET max_group_size TO maximum value in root_sizes  
        SET num_groups TO number of keys in root_sizes  

        RETURN list containing num_groups and max_group_size  
    END FUNCTION  
END CLASS
