CLASS Solution  
    FUNCTION minimumMoves(self, arr)  
        SET n TO length of arr  
        SET dp TO two-dimensional list of size n by n with all elements set to zero  
        
        FOR i FROM zero TO n minus one  
            SET dp at position i, i TO one  
        END FOR  
        
        FOR i FROM zero TO n minus two  
            IF element at index i of arr EQUALS element at index i plus one of arr THEN  
                SET dp at position i, i plus one TO one  
            ELSE  
                SET dp at position i, i plus one TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n minus length  
                SET j TO i plus length minus one  
                IF element at index i of arr EQUALS element at index j of arr THEN  
                    SET dp at position i, j TO dp at position i plus one, j minus one  
                ELSE  
                    SET dp at position i, j TO infinity  
                    FOR k FROM i TO j minus one  
                        SET dp at position i, j TO minimum of dp at position i, j AND sum of dp at position i, k AND dp at position k plus one, j  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position zero, n minus one  
    END FUNCTION  
END CLASS
