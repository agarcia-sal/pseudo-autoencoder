CLASS Solution  
    FUNCTION minCostII(self, costs)  
        IF costs IS empty THEN  
            RETURN zero  
        END IF  
        
        SET n TO length of costs  
        SET k TO length of first element in costs  
        
        IF k EQUALS one THEN  
            IF n EQUALS one THEN  
                RETURN first element of first element in costs  
            ELSE  
                RETURN infinity  
            END IF  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            SET min_cost TO infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO infinity  
            
            FOR j FROM zero TO k MINUS one  
                IF element at [i MINUS one][j] in costs LESS THAN min_cost THEN  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO element at [i MINUS one][j] in costs  
                    SET min_cost_index TO j  
                ELSE IF element at [i MINUS one][j] in costs LESS THAN second_min_cost THEN  
                    SET second_min_cost TO element at [i MINUS one][j] in costs  
                END IF  
            END FOR  
            
            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index THEN  
                    INCREMENT element at [i][j] in costs BY second_min_cost  
                ELSE  
                    INCREMENT element at [i][j] in costs BY min_cost  
                END IF  
            END FOR  
        END FOR  
        
        RETURN minimum element in last row of costs  
    END FUNCTION  
END CLASS
