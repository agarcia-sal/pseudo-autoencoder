CLASS Solution  
    FUNCTION compareVersion(self, version1, version2)  
        SET v1_revisions TO result of splitting version1 by period  
        SET v2_revisions TO result of splitting version2 by period  
        SET max_length TO the greater of the length of v1_revisions and the length of v2_revisions  
        FOR i FROM zero TO max_length MINUS one  
            IF i LESS THAN length of v1_revisions THEN  
                SET v1 TO integer conversion of v1_revisions at position i  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF i LESS THAN length of v2_revisions THEN  
                SET v2 TO integer conversion of v2_revisions at position i  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2 THEN  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2 THEN  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
