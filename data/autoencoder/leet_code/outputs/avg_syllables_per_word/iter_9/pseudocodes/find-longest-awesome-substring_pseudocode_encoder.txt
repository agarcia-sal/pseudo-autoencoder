CLASS Solution  
    FUNCTION longestAwesome(self, s)  
        SET prefix_map TO a map containing key zero mapped to minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN s  
            SET digit TO integer value of char  
            TOGGLE the bit corresponding to digit in prefix_xor  
            
            IF prefix_xor EXISTS as a key IN prefix_map THEN  
                SET max_length TO the greater of max_length AND the difference of i MINUS prefix_map at prefix_xor  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor toggled at bit j  
                IF flipped EXISTS as a key IN prefix_map THEN  
                    SET max_length TO the greater of max_length AND the difference of i MINUS prefix_map at flipped  
                END IF  
            END FOR  
            
            IF prefix_xor NOT EXISTS as a key IN prefix_map THEN  
                MAP prefix_xor TO i in prefix_map  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
