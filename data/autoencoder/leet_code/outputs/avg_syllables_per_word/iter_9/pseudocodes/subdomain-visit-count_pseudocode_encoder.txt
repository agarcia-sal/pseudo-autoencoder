CLASS Solution  
    FUNCTION subdomainVisits(self, cpdomains)  
        SET domain_count TO empty mapping with default integer zero  
        
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count AND domain BY space  
            SET count TO count converted to integer  
            SPLIT domain INTO subdomains BY dot  
            
            FOR index FROM zero TO length of subdomains MINUS one  
                JOIN subdomains from index to end WITH dot INTO subdomain  
                INCREMENT domain_count at subdomain BY count  
            END FOR  
        END FOR  
        
        RETURN list constructed by FOR each domain, count IN domain_count items returning string formed by count FOLLOWED BY space FOLLOWED BY domain  
    END FUNCTION  
END CLASS
