CLASS Solution
    FUNCTION parseBoolExpr(self, expression)
        SET stk TO empty list
        FOR each character c IN expression
            IF c EQUALS any of 't', 'f', '!', '&', '|' THEN
                APPEND c TO stk
            ELSE IF c EQUALS closing parenthesis THEN
                SET t TO zero
                SET f TO zero
                WHILE last element of stk EQUALS any of 't' or 'f' DO
                    INCREMENT t BY (last element of stk EQUALS 't')
                    INCREMENT f BY (last element of stk EQUALS 'f')
                    REMOVE last element FROM stk
                END WHILE
                SET op TO last element removed FROM stk
                IF op EQUALS '!' THEN
                    SET c TO 't' IF f GREATER THAN zero ELSE 'f'
                ELSE IF op EQUALS '&' THEN
                    SET c TO 'f' IF f GREATER THAN zero ELSE 't'
                ELSE IF op EQUALS '|' THEN
                    SET c TO 't' IF t GREATER THAN zero ELSE 'f'
                END IF
                APPEND c TO stk
            END IF
        END FOR
        RETURN whether first element of stk EQUALS 't'
    END FUNCTION
END CLASS
