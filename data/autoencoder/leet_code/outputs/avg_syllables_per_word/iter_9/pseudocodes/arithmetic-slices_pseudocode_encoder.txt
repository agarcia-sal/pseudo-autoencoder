CLASS Solution
    FUNCTION numberOfArithmeticSlices(self, nums)
        IF length of nums LESS THAN 3 THEN
            RETURN 0
        END IF

        SET total_slices TO 0
        SET current_length TO 2

        FOR index FROM 2 TO length of nums MINUS 1
            IF difference between element at index in nums AND element at index MINUS 1 in nums EQUALS difference between element at index MINUS 1 in nums AND element at index MINUS 2 in nums THEN
                INCREMENT current_length BY 1
                INCREMENT total_slices BY current_length MINUS 2
            ELSE
                SET current_length TO 2
            END IF
        END FOR

        RETURN total_slices
    END FUNCTION
END CLASS
