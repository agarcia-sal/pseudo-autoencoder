CLASS Solution  
    FUNCTION decodeString(self, s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char is a digit THEN  
                APPEND char TO d  
            ELSE IF char is a letter THEN  
                APPEND char TO a  
            ELSE IF char EQUALS opening bracket THEN  
                APPEND tuple of (a, d converted to integer) TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS closing bracket THEN  
                REMOVE last element FROM stack INTO p and n  
                SET a TO p concatenated WITH a repeated n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
