CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list containing integers from zero TO n inclusive  
        SET self.rank TO a list containing zeros repeated n plus one times  
        SET self.count TO n  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF self.parent at index u NOT EQUALS u THEN  
            SET self.parent at index u TO the result of self.find called with self.parent at index u  
        END IF  
        RETURN self.parent at index u  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO the result of self.find called with u  
        SET rootV TO the result of self.find called with v  
        IF rootU NOT EQUALS rootV THEN  
            IF self.rank at index rootU GREATER THAN self.rank at index rootV THEN  
                SET self.parent at index rootV TO rootU  
            ELSE IF self.rank at index rootU LESS THAN self.rank at index rootV THEN  
                SET self.parent at index rootU TO rootV  
            ELSE  
                SET self.parent at index rootV TO rootU  
                INCREMENT self.rank at index rootU BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        SET uf_common TO a new instance of UnionFind with argument n  
        SET uf_alice TO a new instance of UnionFind with argument n  
        SET uf_bob TO a new instance of UnionFind with argument n  
        
        SET used_edges TO zero  
        
        FOR each tuple of values t, u, v IN edges  
            IF t EQUALS three THEN  
                IF uf_common.union called with u and v RETURNS True THEN  
                    CALL uf_alice.union with u and v  
                    CALL uf_bob.union with u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
        
        FOR each tuple of values t, u, v IN edges  
            IF t EQUALS one AND uf_alice.union called with u and v RETURNS True THEN  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND uf_bob.union called with u and v RETURNS True THEN  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
        
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one THEN  
            RETURN the length of edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
