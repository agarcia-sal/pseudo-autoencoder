CLASS Solution  
    FUNCTION spiralOrder(self, matrix)  
        IF matrix IS empty OR first element of matrix IS empty THEN  
            RETURN empty list  
        END IF  
        
        SET result TO empty list  
        SET top TO zero  
        SET bottom TO length of matrix MINUS one  
        SET left TO zero  
        SET right TO length of first element of matrix MINUS one  
        
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right DO  
            FOR index FROM left TO right DO  
                APPEND element at position top and index in matrix TO result  
            END FOR  
            INCREMENT top BY one  
            
            FOR index FROM top TO bottom DO  
                APPEND element at position index and right in matrix TO result  
            END FOR  
            DECREMENT right BY one  
            
            IF top LESS THAN OR EQUAL TO bottom THEN  
                FOR index FROM right DOWN TO left DO  
                    APPEND element at position bottom and index in matrix TO result  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            
            IF left LESS THAN OR EQUAL TO right THEN  
                FOR index FROM bottom DOWN TO top DO  
                    APPEND element at position index and left in matrix TO result  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
