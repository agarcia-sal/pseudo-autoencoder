CLASS Solution  
    FUNCTION findOrder(self, numCourses, prerequisites)  
        SET adj_list TO an empty mapping from each course in range from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros with length numCourses  
        
        FOR each pair dest, src IN prerequisites  
            APPEND dest TO the list at adj_list[src]  
            INCREMENT in_degrees[dest] BY one  
        END FOR  
        
        SET queue TO a new deque containing each course in range from zero TO numCourses MINUS one WHERE in_degrees[course] EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the leftmost element REMOVED from queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN adj_list[current]  
                DECREMENT in_degrees[neighbor] BY one  
                IF in_degrees[neighbor] EQUALS zero THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF length of topological_order EQUALS numCourses THEN  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
