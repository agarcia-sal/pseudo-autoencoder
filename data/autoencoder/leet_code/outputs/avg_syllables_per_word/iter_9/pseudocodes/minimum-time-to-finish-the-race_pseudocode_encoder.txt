CLASS Solution  
    FUNCTION minimumFinishTime(self, tires, changeTime, numLaps)  
        SET min_times TO list of length fifteen FILLED WITH infinity  
        FOR each pair f, r IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF index  
                IF lap_time GREATER THAN changeTime PLUS f THEN  
                    BREAK the inner loop  
                END IF  
                SET total_time TO total_time PLUS lap_time  
                SET min_times at position index TO the smaller value BETWEEN min_times at position index AND total_time  
            END FOR  
        END FOR  

        SET dp TO list of length numLaps PLUS one FILLED WITH infinity  
        SET dp at position zero TO zero  

        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET dp at position i TO the smaller value BETWEEN dp at position i AND dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  

        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
