CLASS ListNode  
    FUNCTION __init__(self, val DEFAULTED TO zero, next DEFAULTED TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET head TO new ListNode WITH val AS first element OF values  
    SET p TO head  
    FOR each val IN elements of values starting from second element  
        SET node TO new ListNode WITH val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none THEN  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none THEN  
        RETURN false  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head IS none OR head.next IS none THEN  
            RETURN head  
        END IF  
        SET dummy TO new ListNode WITH val AS negative five thousand and one  
        SET dummy.next TO head  
        SET current TO head.next  
        SET head.next TO none  
        WHILE current IS NOT none DO  
            SET next_node TO current.next  
            SET prev TO dummy  
            WHILE prev.next IS NOT none AND prev.next.val LESS THAN current.val DO  
                SET prev TO prev.next  
            END WHILE  
            SET current.next TO prev.next  
            SET prev.next TO current  
            SET current TO next_node  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
