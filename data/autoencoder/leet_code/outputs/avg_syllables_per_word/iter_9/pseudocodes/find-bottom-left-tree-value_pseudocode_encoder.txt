CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val SET TO first element OF values  
    SET i TO one  
    INITIALIZE queue AS empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO element REMOVED FROM left OF queue  
        IF i LESS THAN length OF values AND element AT index i OF values IS NOT None THEN  
            SET node.left TO new TreeNode WITH val SET TO element AT index i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length OF values AND element AT index i OF values IS NOT None THEN  
            SET node.right TO new TreeNode WITH val SET TO element AT index i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        INITIALIZE queue AS double-ended queue CONTAINING root  
        SET leftmost_value TO None  
        WHILE queue IS NOT empty DO  
            SET level_size TO length OF queue  
            FOR i FROM zero TO level_size MINUS one DO  
                SET node TO element REMOVED FROM left OF queue  
                IF i EQUALS zero THEN  
                    SET leftmost_value TO node.val  
                END IF  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
