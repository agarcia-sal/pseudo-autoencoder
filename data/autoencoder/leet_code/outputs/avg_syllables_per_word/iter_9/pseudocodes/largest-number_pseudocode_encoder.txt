CLASS Solution  
    FUNCTION largestNumber(self, nums)  
        SET nums_str TO list of string representations of each element in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x followed by y GREATER THAN concatenation of y followeded by x THEN  
                RETURN negative one  
            ELSE IF concatenation of x followed by y LESS THAN concatenation of y followed by x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str according to comparison using compare as comparator  
        
        IF first element of nums_str EQUALS string zero THEN  
            RETURN string zero  
        END IF  
        
        RETURN concatenation of all elements in nums_str  
    END FUNCTION  
END CLASS
