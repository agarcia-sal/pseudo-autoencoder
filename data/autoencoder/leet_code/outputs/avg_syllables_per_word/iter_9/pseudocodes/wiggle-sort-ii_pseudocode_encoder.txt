CLASS Solution  
    FUNCTION wiggleSort(self, nums)  
        CALL nums.sort  
        SET sorted_nums TO a copy of nums  
        SET n TO length of nums  
        SET left TO integer division of (n MINUS 1) BY 2  
        SET right TO n MINUS 1  
        FOR i FROM 0 TO n MINUS 1  
            IF i MODULO 2 EQUALS 0 THEN  
                SET nums at position i TO sorted_nums at position left  
                DECREMENT left BY 1  
            ELSE  
                SET nums at position i TO sorted_nums at position right  
                DECREMENT right BY 1  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
