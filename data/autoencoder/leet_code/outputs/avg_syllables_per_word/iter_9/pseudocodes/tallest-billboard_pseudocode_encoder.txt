CLASS Solution  
    FUNCTION tallestBillboard(self, rods)  
        SET dp TO a dictionary mapping zero TO zero  
        
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each pair of difference d AND value y IN dp  
                SET key1 TO d PLUS rod  
                SET existing_value1 TO value in new_dp at key1 OR zero if absent  
                SET new_dp at key1 TO the maximum of existing_value1 AND y  
                
                SET key2 TO the absolute value of d MINUS rod  
                SET existing_value2 TO value in new_dp at key2 OR zero if absent  
                SET new_value2 TO y PLUS the minimum of d AND rod  
                SET new_dp at key2 TO the maximum of existing_value2 AND new_value2  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN value in dp at zero  
    END FUNCTION  
END CLASS
