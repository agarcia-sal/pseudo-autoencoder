CLASS Solution  
    FUNCTION findSubsequences(self, nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS length of nums THEN  
                IF length of t GREATER THAN 1 THEN  
                    APPEND a copy of t TO ans  
                END IF  
                RETURN  
            END IF  
            IF element at index u in nums GREATER THAN OR EQUALS last THEN  
                APPEND element at index u in nums TO t  
                CALL dfs with parameters u PLUS 1, element at index u in nums, and t  
                REMOVE last element FROM t  
            END IF  
            IF element at index u in nums NOT EQUALS last THEN  
                CALL dfs with parameters u PLUS 1, last, and t  
            END IF  
        END FUNCTION  

        SET ans TO empty list  
        CALL dfs with parameters zero, negative one thousand, and empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
