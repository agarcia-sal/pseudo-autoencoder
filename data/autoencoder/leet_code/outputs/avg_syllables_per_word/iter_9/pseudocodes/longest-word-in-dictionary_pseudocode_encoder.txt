CLASS Solution  
    FUNCTION longestWord(self, words)  
        CALL words.sort BY length AND lexicographical order  
        SET valid_words TO empty set  
        SET longest_word TO empty string  
        FOR each word IN words  
            IF length of word EQUALS one OR word without last character IN valid_words THEN  
                ADD word TO valid_words  
                IF length of word GREATER THAN length of longest_word THEN  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
