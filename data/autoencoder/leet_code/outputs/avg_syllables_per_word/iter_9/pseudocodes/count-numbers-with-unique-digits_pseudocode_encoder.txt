CLASS Solution  
    FUNCTION countNumbersWithUniqueDigits(self, n)  
        IF n EQUALS 0 THEN  
            RETURN 1  
        END IF  
        IF n EQUALS 1 THEN  
            RETURN 10  
        END IF  
  
        SET count TO 10  
        SET available_digits TO 9  
        SET choices TO 9  
  
        FOR i FROM 2 TO n  
            SET choices TO choices MULTIPLIED BY available_digits  
            SET count TO count PLUS choices  
            SET available_digits TO available_digits MINUS 1  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
