CLASS Solution  
    FUNCTION isValidPalindrome(self, s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO length of s  
            SET dp TO a two-dimensional list of size n by n initialized to zero  
            
            FOR i FROM zero TO n MINUS one  
                SET dp at position i by i TO one  
            END FOR  
            
            FOR length FROM two TO n  
                FOR i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF character at s position i EQUALS character at s position j THEN  
                        SET dp at position i by j TO dp at position i PLUS one by j MINUS one PLUS two  
                    ELSE  
                        SET dp at position i by j TO the greater of dp at position i PLUS one by j and dp at position i by j MINUS one  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN dp at position zero by n MINUS one  
        END FUNCTION  
        
        SET longest_palindrome_length TO longest_palindromic_subseq(s)  
        RETURN length of s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
