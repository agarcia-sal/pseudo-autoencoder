CLASS Solution  
    FUNCTION circularArrayLoop(self, nums)  
        FUNCTION sameDirection(val1, val2)  
            IF val1 GREATER THAN 0 AND val2 GREATER THAN 0 THEN  
                RETURN True  
            ELSE IF val1 LESS THAN 0 AND val2 LESS THAN 0 THEN  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET n TO length of nums  

        FOR i FROM 0 TO n MINUS 1  
            SET slow TO i  
            SET fast TO i  

            IF nums AT i EQUALS 0 THEN  
                CONTINUE TO next iteration  
            END IF  

            WHILE True  
                SET slow TO remainder of (slow PLUS nums AT slow) DIVIDED BY n  
                IF slow LESS THAN 0 THEN  
                    INCREMENT slow BY n  
                END IF  

                SET fast TO remainder of (fast PLUS nums AT fast) DIVIDED BY n  
                IF fast LESS THAN 0 THEN  
                    INCREMENT fast BY n  
                END IF  

                SET fast TO remainder of (fast PLUS nums AT fast) DIVIDED BY n  
                IF fast LESS THAN 0 THEN  
                    INCREMENT fast BY n  
                END IF  

                IF NOT sameDirection(nums AT slow, nums AT i) OR NOT sameDirection(nums AT fast, nums AT i) THEN  
                    BREAK the WHILE loop  
                END IF  

                IF slow EQUALS fast THEN  
                    IF slow EQUALS remainder of (slow PLUS nums AT slow) DIVIDED BY n THEN  
                        BREAK the WHILE loop  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
