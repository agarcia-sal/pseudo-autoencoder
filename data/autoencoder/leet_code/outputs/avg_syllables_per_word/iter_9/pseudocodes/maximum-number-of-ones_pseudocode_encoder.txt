CLASS Solution  
    FUNCTION maximumNumberOfOnes(self, width, height, sideLength, maxOnes)  
        SET freq TO list of zeros with length sideLength MULTIPLIED BY sideLength  
        
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET pos TO remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT freq at position pos BY one  
            END FOR  
        END FOR  
        
        SORT freq in descending order  
        
        SET result TO sum of the first maxOnes elements of freq  
        RETURN result  
    END FUNCTION  
END CLASS
