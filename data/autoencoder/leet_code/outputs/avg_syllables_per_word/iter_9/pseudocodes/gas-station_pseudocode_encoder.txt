CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF sum of gas LESS THAN sum of cost THEN  
            RETURN negative one  
        END IF  
        
        SET start TO zero  
        SET total tank TO zero  
        
        FOR index FROM zero TO length of gas MINUS one  
            SET total tank TO total tank PLUS gas at index MINUS cost at index  
            
            IF total tank LESS THAN zero THEN  
                SET start TO index PLUS one  
                SET total tank TO zero  
            END IF  
        END FOR  
        
        RETURN start  
    END FUNCTION  
END CLASS
