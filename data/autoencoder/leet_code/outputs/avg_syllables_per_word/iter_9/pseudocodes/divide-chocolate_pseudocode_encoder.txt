CLASS Solution  
    FUNCTION maximizeSweetness(self, sweetness, k)  
        FUNCTION canDivide(min_sweet)  
            SET pieces TO 0  
            SET current_sweet TO 0  
            FOR each sweet IN sweetness  
                SET current_sweet TO current_sweet PLUS sweet  
                IF current_sweet GREATER THAN OR EQUAL TO min_sweet THEN  
                    INCREMENT pieces BY 1  
                    SET current_sweet TO 0  
                    IF pieces GREATER THAN k THEN  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            RETURN pieces GREATER THAN k  
        END FUNCTION  

        SET left TO minimum element IN sweetness  
        SET right TO sum of elements IN sweetness DIVIDED BY (k PLUS 1)  
        SET result TO 0  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS right DIVIDED BY 2  
            IF canDivide(mid) THEN  
                SET result TO mid  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid MINUS 1  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
