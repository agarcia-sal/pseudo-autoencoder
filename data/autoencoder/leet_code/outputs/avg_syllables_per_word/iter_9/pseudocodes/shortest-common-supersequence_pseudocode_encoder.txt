CLASS Solution  
    FUNCTION shortestCommonSupersequence(self, str1, str2)  
        SET m TO length of str1  
        SET n TO length of str2  
        
        CREATE dp AS two-dimensional list of size (m plus one) by (n plus one) filled with zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position (i minus one) in str1 EQUALS character at position (j minus one) in str2 THEN  
                    SET dp at position [i][j] TO dp at position [i minus one][j minus one] PLUS one  
                ELSE  
                    SET dp at position [i][j] TO maximum of dp at position [i minus one][j] AND dp at position [i][j minus one]  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        CREATE empty list named scs  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero DO  
            IF character at position (i minus one) in str1 EQUALS character at position (j minus one) in str2 THEN  
                APPEND character at position (i minus one) in str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF dp at position [i minus one][j] GREATER THAN dp at position [i][j minus one] THEN  
                APPEND character at position (i minus one) in str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND character at position (j minus one) in str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero DO  
            APPEND character at position (i minus one) in str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero DO  
            APPEND character at position (j minus one) in str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the list scs  
        
        RETURN concatenation of elements in list scs  
    END FUNCTION  
END CLASS
