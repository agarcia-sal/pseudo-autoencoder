CLASS Solution  
    FUNCTION canSeePersonsCount(self, heights)  
        SET n TO length of heights  
        SET answer TO list of n zeros  
        SET stack TO empty list  
          
        FOR i FROM zero TO n MINUS one  
            WHILE stack is not empty AND heights at last element in stack LESS THAN heights at i DO  
                INCREMENT answer at index popped from stack BY one  
            END WHILE  
            IF stack is not empty THEN  
                INCREMENT answer at last element in stack BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
          
        RETURN answer  
    END FUNCTION  
END CLASS
