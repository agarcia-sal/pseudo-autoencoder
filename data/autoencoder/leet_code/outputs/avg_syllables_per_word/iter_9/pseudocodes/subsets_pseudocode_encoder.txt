CLASS Solution  
    FUNCTION subsets(self, nums)  
        IF nums EQUALS empty list THEN  
            RETURN list containing empty list  
        END IF  
        
        SET subsets_without_first TO self.subsets(nums from second element TO end)  
        
        SET subsets_with_first TO empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO list containing first element of nums followed by subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        
        RETURN concatenation of subsets_without_first and subsets_with_first  
    END FUNCTION  
END CLASS
