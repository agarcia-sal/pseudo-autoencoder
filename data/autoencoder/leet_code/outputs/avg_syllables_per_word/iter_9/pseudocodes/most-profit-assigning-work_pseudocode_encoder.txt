CLASS Solution  
    FUNCTION maxProfitAssignment(self, difficulty, profit, worker)  
        SET jobs TO list of pairs combining difficulty and profit elements correspondingly  
        SORT jobs in ascending order by difficulty  
        SORT worker in ascending order by ability  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO length of jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND difficulty of jobs at job_index LESS THAN OR EQUAL TO ability  
                SET best_profit TO greater value BETWEEN best_profit AND profit of jobs at job_index  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
