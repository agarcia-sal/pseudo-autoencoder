CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(self, nums, k)  
        SET n TO length of nums  
        INITIALIZE sums AS a list of zeros with length equal to n MINUS k PLUS one  
        SET current_sum TO the sum of the first k elements in nums  
        SET sums at index zero TO current_sum  
        
        FOR i FROM one TO n MINUS k DO  
            INCREMENT current_sum BY nums at index i PLUS k MINUS one MINUS nums at index i MINUS one  
            SET sums at index i TO current_sum  
        END FOR  
        
        INITIALIZE left AS a list of zeros with length equal to n MINUS k PLUS one  
        INITIALIZE right AS a list with each element equal to n MINUS k and length equal to n MINUS k PLUS one  
        
        SET max_sum TO zero  
        FOR i FROM zero TO n MINUS k DO  
            IF sums at index i GREATER THAN sums at index max_sum THEN  
                SET max_sum TO i  
            END IF  
            SET left at index i TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWN TO zero DO  
            IF sums at index i GREATER THAN OR EQUAL TO sums at index max_sum THEN  
                SET max_sum TO i  
            END IF  
            SET right at index i TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a list containing zero, zero, zero  
        FOR j FROM k TO n MINUS two MULTIPLIED BY k DO  
            SET i TO left at index j MINUS k  
            SET l TO right at index j PLUS k  
            SET total TO sums at index i PLUS sums at index j PLUS sums at index l  
            IF total GREATER THAN max_total THEN  
                SET max_total TO total  
                SET result TO a list containing i, j, l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
