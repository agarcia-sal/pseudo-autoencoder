CLASS Solution  
    FUNCTION findMinHeightTrees(self, n, edges)  
        IF n EQUALS 1 THEN  
            RETURN list containing 0  
        END IF  

        SET graph TO a mapping from each node to a list of neighboring nodes  
        SET degree TO a list of n zeros  

        FOR each pair u, v IN edges  
            APPEND v TO graph at u  
            APPEND u TO graph at v  
            INCREMENT degree at u BY 1  
            INCREMENT degree at v BY 1  
        END FOR  

        SET leaves TO a double-ended queue containing all indices i from zero TO n minus one WHERE degree at i EQUALS 1  

        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN 2  
            SET leaves_count TO number of elements in leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each element FROM one TO leaves_count  
                REMOVE first element FROM leaves and CALL it leaf  
                FOR each neighbor IN graph at leaf  
                    DECREMENT degree at neighbor BY 1  
                    IF degree at neighbor EQUALS 1 THEN  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN list conversion of leaves  
    END FUNCTION  
END CLASS
