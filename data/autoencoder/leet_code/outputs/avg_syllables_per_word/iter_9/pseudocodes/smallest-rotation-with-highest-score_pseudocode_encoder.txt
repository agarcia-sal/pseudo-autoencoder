CLASS Solution  
    FUNCTION bestRotation(self, nums)  
        SET n TO length of nums  
        SET delta TO list of n zeros  
        
        FOR each index i and element num IN nums  
            IF num LESS THAN OR EQUAL TO i THEN  
                INCREMENT delta at position zero BY one  
                IF i MINUS num PLUS one LESS THAN n THEN  
                    DECREMENT delta at position (i MINUS num PLUS one) BY one  
                END IF  
            END IF  
            
            IF i PLUS one LESS THAN n THEN  
                INCREMENT delta at position (i PLUS one) BY one  
                IF i PLUS n MINUS num PLUS one LESS THAN n THEN  
                    DECREMENT delta at position (i PLUS n MINUS num PLUS one) BY one  
                END IF  
            END IF  
        END FOR  
        
        SET score TO zero  
        SET max_score TO zero  
        SET best_k TO zero  
        
        FOR k FROM zero TO n MINUS one  
            INCREMENT score BY delta at position k  
            IF score GREATER THAN max_score THEN  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
