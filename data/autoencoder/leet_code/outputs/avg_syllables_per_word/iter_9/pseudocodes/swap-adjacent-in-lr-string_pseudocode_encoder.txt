CLASS Solution  
    FUNCTION canTransform(self, start, end)  
        SET start_filtered TO empty list  
        FOR index FROM zero TO length of start MINUS one  
            SET character TO character at position index in start  
            IF character NOT EQUALS 'X' THEN  
                APPEND pair of character AND index TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO empty list  
        FOR index FROM zero TO length of end MINUS one  
            SET character TO character at position index in end  
            IF character NOT EQUALS 'X' THEN  
                APPEND pair of character AND index TO end_filtered  
            END IF  
        END FOR  
        
        IF length of start_filtered NOT EQUALS length of end_filtered THEN  
            RETURN False  
        END IF  
        
        FOR each pair of elements from start_filtered AND end_filtered as (c1, i1) AND (c2, i2)  
            IF c1 NOT EQUALS c2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUALS 'L' AND i1 LESS THAN i2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUALS 'R' AND i1 GREATER THAN i2 THEN  
                RETURN False  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
