CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val set to first element of values  
    SET i TO one  
    INITIALIZE queue as empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO element removed from the front of queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val set to values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val set to values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node IS None THEN  
                RETURN negative one  
            END IF  
            SET left_height TO collectLeaves(node.left)  
            SET right_height TO collectLeaves(node.right)  
            SET current_height TO maximum of left_height and right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO length of result THEN  
                APPEND empty list TO result  
            END IF  
            APPEND node.val TO result at current_height  
            RETURN current_height  
        END FUNCTION  
        SET result TO empty list  
        CALL collectLeaves WITH root  
        RETURN result  
    END FUNCTION  
END CLASS
