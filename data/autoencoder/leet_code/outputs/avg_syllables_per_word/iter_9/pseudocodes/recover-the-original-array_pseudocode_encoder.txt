CLASS Solution  
    FUNCTION recoverArray(self, nums)  
        CALL sort ON nums  
        SET n TO length of nums  
          
        SET first TO element at index zero of nums  
          
        FOR i FROM one TO n MINUS one  
            SET k TO (element at index i of nums MINUS first) DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at index i of nums THEN  
                CONTINUE to next iteration  
            END IF  
              
            SET count TO frequency counter constructed from nums  
            SET arr TO empty list  
              
            FOR each x IN nums  
                IF count value at x EQUALS zero THEN  
                    CONTINUE to next iteration  
                END IF  
                IF count value at x PLUS two MULTIPLIED BY k EQUALS zero THEN  
                    BREAK loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT count value at x BY one  
                DECREMENT count value at x PLUS two MULTIPLIED BY k BY one  
            END FOR  
              
            IF length of arr EQUALS n DIVIDED BY two THEN  
                RETURN arr  
            END IF  
        END FOR  
          
        RETURN empty list  
    END FUNCTION  
END CLASS
