CLASS Solution  
    FUNCTION reverse(self, x)  
        IF x LESS THAN 0 THEN  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  

        SET absolute_x TO absolute value of x  
        SET reversed_string TO the digits of absolute_x in reverse order  
        SET reversed_x TO the integer value represented by reversed_string  

        SET result TO sign MULTIPLIED BY reversed_x  

        IF result LESS THAN negative two RAISED TO THE power thirty one OR result GREATER THAN two RAISED TO THE power thirty one MINUS one THEN  
            RETURN zero  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
