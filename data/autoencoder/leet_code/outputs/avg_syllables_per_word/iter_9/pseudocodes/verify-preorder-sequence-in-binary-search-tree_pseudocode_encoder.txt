CLASS Solution  
    FUNCTION verifyPreorder(self, preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound THEN  
                RETURN False  
            END IF  
            
            WHILE stack IS NOT empty AND value GREATER THAN last element IN stack DO  
                SET lower_bound TO remove last element FROM stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
