CLASS Solution
    FUNCTION wallsAndGates(self, rooms)
        IF rooms EQUALS empty OR first element of rooms EQUALS empty THEN
            RETURN
        END IF

        SET rows TO length of rooms
        SET cols TO length of first element of rooms
        INITIALIZE queue AS empty double-ended queue

        FOR r FROM 0 TO rows MINUS 1
            FOR c FROM 0 TO cols MINUS 1
                IF element at rooms position r and c EQUALS 0 THEN
                    APPEND pair of r and c TO queue
                END IF
            END FOR
        END FOR

        SET directions TO list of pairs: (0, 1), (1, 0), (0, -1), (-1, 0)

        WHILE queue IS NOT empty
            REMOVE first element FROM queue AND SET TO pair r and c
            FOR each pair dr and dc IN directions
                SET nr TO r PLUS dr
                SET nc TO c PLUS dc
                IF nr GREATER THAN OR EQUAL TO 0 AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO 0 AND nc LESS THAN cols AND element at rooms position nr and nc EQUALS 2147483647 THEN
                    SET element at rooms position nr and nc TO element at rooms position r and c PLUS 1
                    APPEND pair of nr and nc TO queue
                END IF
            END FOR
        END WHILE
    END FUNCTION
END CLASS
