CLASS Solution  
    FUNCTION letterCombinations(self, digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        
        SET phone_map TO a mapping from digit characters to corresponding letter strings as follows:  
            digit character '2' MAPPED TO string 'abc'  
            digit character '3' MAPPED TO string 'def'  
            digit character '4' MAPPED TO string 'ghi'  
            digit character '5' MAPPED TO string 'jkl'  
            digit character '6' MAPPED TO string 'mno'  
            digit character '7' MAPPED TO string 'pqrs'  
            digit character '8' MAPPED TO string 'tuv'  
            digit character '9' MAPPED TO string 'wxyz'  
        
        FUNCTION backtrack(combination, next_digits)  
            IF length of next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map MAPPED BY first character of next_digits DO  
                    CALL backtrack WITH combination concatenated with letter, and substring of next_digits from second character onward  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
