CLASS Solution  
    FUNCTION countPrimes(self, n)  
        IF n LESS THAN OR EQUAL TO 2 THEN  
            RETURN 0  
        END IF  
        
        INITIALIZE is_prime TO list of size n WITH all elements SET TO true  
        SET is_prime at index 0 TO false  
        SET is_prime at index 1 TO false  
        
        FOR start FROM 2 TO integer part of square root of n PLUS 1 DO  
            IF is_prime at index start IS true THEN  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS 1 IN STEPS OF start DO  
                    SET is_prime at index multiple TO false  
                END FOR  
            END IF  
        END FOR  
        
        SET prime_count TO 0  
        FOR each element IN is_prime DO  
            IF element IS true THEN  
                INCREMENT prime_count BY 1  
            END IF  
        END FOR  
        
        RETURN prime_count  
    END FUNCTION  
END CLASS
