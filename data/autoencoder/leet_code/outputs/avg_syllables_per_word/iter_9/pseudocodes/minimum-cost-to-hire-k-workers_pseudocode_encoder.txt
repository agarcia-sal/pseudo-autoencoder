CLASS Solution  
    FUNCTION mincostToHireWorkers(self, quality, wage, k)  
        SET workers TO sorted list of pairs where each pair consists of wage divided by quality and quality for each pair of wage and quality taken together  
        
        SET max_heap TO an empty list  
        SET total_quality TO zero  
        SET min_cost TO infinity  
        
        FOR each ratio and quality_value IN workers  
            PUSH negative quantity of quality_value INTO max_heap  
            INCREMENT total_quality BY quality_value  
            
            IF size of max_heap GREATER THAN k THEN  
                INCREMENT total_quality BY the result of removing the largest element from max_heap  
            END IF  
            
            IF size of max_heap EQUALS k THEN  
                SET min_cost TO the lesser of min_cost and total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
