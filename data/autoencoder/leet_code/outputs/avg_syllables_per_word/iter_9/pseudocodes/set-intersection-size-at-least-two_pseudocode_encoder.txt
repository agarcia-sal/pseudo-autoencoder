CLASS Solution  
    FUNCTION intersectionSizeTwo(self, intervals)  
        SORT intervals BY interval end value ASCENDING AND interval start value DESCENDING  
        SET first TO value of end of first interval MINUS one  
        SET second TO value of end of first interval  
        SET count TO two  
        
        FOR each interval starting from second IN intervals  
            SET start TO interval start value  
            SET end TO interval end value  
            
            IF start GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
