CLASS Solution  
    FUNCTION boldWords(self, words, s)  
        SET bold TO list of boolean false values with length equal to length of s  
        
        FOR each word IN words  
            SET start TO position of word in s  
            WHILE start NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS length of word MINUS one  
                    SET bold at position i TO true  
                END FOR  
                SET start TO position of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN length of s  
            IF bold at position i IS true THEN  
                APPEND "<b>" TO result  
                WHILE i LESS THAN length of s AND bold at position i IS true  
                    APPEND character at position i in s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND "</b>" TO result  
            ELSE  
                APPEND character at position i in s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
