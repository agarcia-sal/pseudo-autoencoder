CLASS Solution  
    FUNCTION minimumMoves(self, grid)  
        SET n TO length of grid  
        SET queue TO new double-ended queue containing tuple (0, 0, 0, 1, 0)  
        SET visited TO new set containing tuple (0, 0, 0, 1)  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND UNPACK TO r1, c1, r2, c2, steps  
            
            IF r1 EQUALS n MINUS 1 AND c1 EQUALS n MINUS 2 AND r2 EQUALS n MINUS 1 AND c2 EQUALS n MINUS 1 THEN  
                RETURN steps  
            END IF  
            
            IF c2 PLUS 1 LESS THAN n AND grid at position r1 and c1 PLUS 1 EQUALS zero AND grid at position r2 and c2 PLUS 1 EQUALS zero AND tuple (r1, c1 PLUS 1, r2, c2 PLUS 1) NOT IN visited THEN  
                ADD tuple (r1, c1 PLUS 1, r2, c2 PLUS 1) TO visited  
                APPEND tuple (r1, c1 PLUS 1, r2, c2 PLUS 1, steps PLUS 1) TO queue  
            END IF  
            
            IF r2 PLUS 1 LESS THAN n AND grid at position r1 PLUS 1 and c1 EQUALS zero AND grid at position r2 PLUS 1 and c2 EQUALS zero AND tuple (r1 PLUS 1, c1, r2 PLUS 1, c2) NOT IN visited THEN  
                ADD tuple (r1 PLUS 1, c1, r2 PLUS 1, c2) TO visited  
                APPEND tuple (r1 PLUS 1, c1, r2 PLUS 1, c2, steps PLUS 1) TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS 1 LESS THAN n AND grid at position r1 PLUS 1 and c1 EQUALS zero AND grid at position r1 PLUS 1 and c2 EQUALS zero AND tuple (r1, c1, r1 PLUS 1, c1) NOT IN visited THEN  
                ADD tuple (r1, c1, r1 PLUS 1, c1) TO visited  
                APPEND tuple (r1, c1, r1 PLUS 1, c1, steps PLUS 1) TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS 1 LESS THAN n AND grid at position r1 and c1 PLUS 1 EQUALS zero AND grid at position r2 and c2 PLUS 1 EQUALS zero AND tuple (r1, c1, r1, c1 PLUS 1) NOT IN visited THEN  
                ADD tuple (r1, c1, r1, c1 PLUS 1) TO visited  
                APPEND tuple (r1, c1, r1, c1 PLUS 1, steps PLUS 1) TO queue  
            END IF  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
