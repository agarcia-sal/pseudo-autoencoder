CLASS Solution  
    FUNCTION minDays(self, n)  
        DEFINE FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one THEN  
                RETURN remaining  
            END IF  
            SET result TO one PLUS minimum OF (remaining MODULO two PLUS dp(remaining DIVIDED BY two)) AND (remaining MODULO three PLUS dp(remaining DIVIDED BY three))  
            RETURN result  
        END FUNCTION  
        RETURN dp(n)  
    END FUNCTION  
END CLASS
