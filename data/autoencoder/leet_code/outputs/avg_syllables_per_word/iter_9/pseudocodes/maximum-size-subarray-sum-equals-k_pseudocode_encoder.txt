CLASS Solution  
    FUNCTION maxSubArrayLen(self, nums, k)  
        SET cumulative_sum_index TO a dictionary with key zero mapped to value minus one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each index i AND each element num IN nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF cumulative_sum MINUS k EXISTS IN cumulative_sum_index THEN  
                SET max_length TO the greater value BETWEEN max_length AND i MINUS cumulative_sum_index VALUE AT cumulative_sum MINUS k  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST IN cumulative_sum_index THEN  
                SET cumulative_sum_index VALUE AT cumulative_sum TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
