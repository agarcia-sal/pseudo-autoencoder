CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val equal to first element of values, left none, right none  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND element at index i in values IS NOT None THEN  
            SET node.left TO new TreeNode with val equal to element at index i in values, left none, right none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND element at index i in values IS NOT None THEN  
            SET node.right TO new TreeNode with val equal to element at index i in values, left none, right none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None THEN  
                RETURN 0  
            END IF  
            RETURN node.val PLUS subtree_sum(node.left) PLUS subtree_sum(node.right)  
        END FUNCTION  

        SET total_sum TO subtree_sum(root)  

        IF total_sum MODULO 2 NOT EQUALS 0 THEN  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY 2  
        SET found TO list containing single False  

        FUNCTION can_split(node)  
            IF node IS None THEN  
                RETURN 0  
            END IF  
            SET current_sum TO node.val PLUS can_split(node.left) PLUS can_split(node.right)  
            IF current_sum EQUALS target AND node NOT EQUALS root THEN  
                SET found AT index 0 TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split(root)  
        RETURN found AT index 0  
    END FUNCTION  
END CLASS
