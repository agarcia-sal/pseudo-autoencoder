CLASS Solution  
    FUNCTION peakIndexInMountainArray(self, arr)  
        SET left TO 0  
        SET right TO length of arr MINUS 1  
        WHILE left LESS THAN right  
            SET mid TO integer division of (left PLUS right) BY 2  
            IF element at position mid in arr LESS THAN element at position mid PLUS 1 in arr THEN  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
