CLASS Solution  
    FUNCTION minimumWhiteTiles(self, floor, numCarpets, carpetLen)  
        SET n TO length of floor  
        
        INITIALIZE dp AS two dimensional list with dimensions n plus one BY numCarpets plus one FILLED WITH zeroes  
        
        FOR i FROM one TO n  
            SET dp at position i and zero TO dp at position i minus one and zero PLUS one IF character at position i minus one in floor EQUALS '1' OTHERWISE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i and j TO dp at position i minus one and j PLUS one IF character at position i minus one in floor EQUALS '1' OTHERWISE zero  
                SET dp at position i and j TO minimum of dp at position i and j AND dp at position maximum of zero AND i minus carpetLen and j minus one  
            END FOR  
        END FOR  
        
        RETURN dp at position n and numCarpets  
    END FUNCTION  
END CLASS
