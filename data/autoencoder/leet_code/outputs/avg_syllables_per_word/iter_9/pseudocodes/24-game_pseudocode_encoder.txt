CLASS Solution  
    FUNCTION judgePoint24(self, cards)  
        FUNCTION solve(nums)  
            IF length of nums EQUALS 1 THEN  
                RETURN absolute value of (first element of nums MINUS 24) LESS THAN point zero zero zero zero zero one  
            END IF  
            
            FOR each index i FROM zero TO length of nums MINUS one  
                FOR each index j FROM zero TO length of nums MINUS one  
                    IF i NOT EQUALS j THEN  
                        SET new_nums TO list of elements from nums at indices k where k NOT EQUALS i AND k NOT EQUALS j  
                        
                        FOR each operation op IN list of operations truediv, mul, add, sub  
                            IF (op EQUALS add OR op EQUALS mul) AND j GREATER THAN i THEN  
                                CONTINUE to next iteration  
                            END IF  
                            
                            IF op NOT EQUALS truediv OR element at index j of nums NOT EQUALS zero THEN  
                                APPEND result of op applied to element at index i of nums AND element at index j of nums TO new_nums  
                                IF solve(new_nums) IS True THEN  
                                    RETURN True  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        IMPORT operations truediv, mul, add, sub  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
