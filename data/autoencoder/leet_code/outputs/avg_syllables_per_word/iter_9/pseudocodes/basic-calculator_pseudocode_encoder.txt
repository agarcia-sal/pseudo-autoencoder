CLASS Solution  
    FUNCTION calculate(self, s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus symbol  
            SET num TO zero  
            
            WHILE length of s GREATER THAN zero  
                SET char TO first element removed from s  
                
                IF char is a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS numeric value of char  
                END IF  
                
                IF char EQUALS opening parenthesis THEN  
                    SET num TO result of calling helper with s  
                END IF  
                
                IF char is NOT a digit AND char is NOT a whitespace OR length of s EQUALS zero THEN  
                    IF sign EQUALS plus symbol THEN  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol THEN  
                        APPEND negative num TO stack  
                    ELSE IF sign EQUALS multiplication symbol THEN  
                        SET last element of stack TO last element MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol THEN  
                        SET last element of stack TO integer division of last element BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS closing parenthesis THEN  
                    BREAK the loop  
                END IF  
            END WHILE  
            
            RETURN sum of elements in stack  
        END FUNCTION  
        
        RETURN result of calling helper with list formed from s  
    END FUNCTION  
END CLASS
