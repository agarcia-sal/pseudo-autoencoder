CLASS Solution  
    FUNCTION canDistribute(self, nums, quantity)  
        SET count TO list of frequencies of each unique element in nums  
        SORT quantity in descending order  
        
        FUNCTION can_satisfy(index)  
            IF index EQUALS length of quantity THEN  
                RETURN True  
            END IF  
            FOR i FROM zero TO length of count MINUS one  
                IF count at position i GREATER THAN OR EQUALS quantity at position index THEN  
                    DECREMENT count at position i BY quantity at position index  
                    IF can_satisfy(index PLUS one) THEN  
                        RETURN True  
                    END IF  
                    INCREMENT count at position i BY quantity at position index  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN can_satisfy(zero)  
    END FUNCTION  
END CLASS
