CLASS Solution  
    FUNCTION splitArray(self, nums)  
        SET n TO length of nums  
        IF n LESS THAN 7 THEN  
            RETURN False  
        END IF  

        SET prefix_sum TO list of size n filled with zero  
        SET prefix_sum at index zero TO nums at index zero  
        FOR i FROM 1 TO n MINUS 1  
            SET prefix_sum at index i TO prefix_sum at index i MINUS 1 PLUS nums at index i  
        END FOR  

        FOR j FROM 3 TO n MINUS 4  
            SET seen TO empty set  
            FOR i FROM 1 TO j MINUS 2  
                IF prefix_sum at index i MINUS 1 EQUALS prefix_sum at index j MINUS 1 MINUS prefix_sum at index i THEN  
                    ADD prefix_sum at index i MINUS 1 TO seen  
                END IF  
            END FOR  

            FOR k FROM j PLUS 2 TO n MINUS 2  
                IF prefix_sum at index n MINUS 1 MINUS prefix_sum at index k EQUALS prefix_sum at index k MINUS 1 MINUS prefix_sum at index j AND prefix_sum at index k MINUS 1 MINUS prefix_sum at index j IS IN seen THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
