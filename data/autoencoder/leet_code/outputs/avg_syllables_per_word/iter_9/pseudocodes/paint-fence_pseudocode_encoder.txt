CLASS Solution  
    FUNCTION numWays(self, n, k)  
        IF n EQUALS 0 THEN  
            RETURN 0  
        END IF  

        IF n EQUALS 1 THEN  
            RETURN k  
        END IF  

        IF n EQUALS 2 THEN  
            RETURN k MULTIPLIED BY k  
        END IF  

        SET same TO k  
        SET diff TO k MULTIPLIED BY (k MINUS 1)  
        SET total TO same PLUS diff  

        FOR i FROM 3 TO n  
            SET new_same TO diff  
            SET new_diff TO total MULTIPLIED BY (k MINUS 1)  

            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO same PLUS diff  
        END FOR  

        RETURN total  
    END FUNCTION  
END CLASS
