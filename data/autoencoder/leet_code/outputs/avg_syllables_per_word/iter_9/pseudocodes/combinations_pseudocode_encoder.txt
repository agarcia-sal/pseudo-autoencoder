CLASS Solution  
    FUNCTION combine(self, n, k)  
        FUNCTION backtrack(start, path)  
            IF length of path EQUALS k THEN  
                APPEND copy of path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n  
                APPEND i TO path  
                CALL backtrack WITH i PLUS one AND path  
                REMOVE last element FROM path  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH one AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
