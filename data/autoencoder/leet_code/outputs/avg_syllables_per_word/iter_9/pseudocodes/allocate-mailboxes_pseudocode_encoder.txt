CLASS Solution  
    FUNCTION minDistance(self, houses, k)  
        CALL houses TO SORT IN ASCENDING ORDER  
        SET n TO LENGTH OF houses  
        
        SET cost TO TWO-DIMENSIONAL LIST OF SIZE n BY n WITH ALL ELEMENTS SET TO ZERO  
        
        FOR i FROM n MINUS ONE DOWN TO ZERO  
            FOR j FROM i PLUS ONE TO n MINUS ONE  
                SET cost AT i AND j TO cost AT i PLUS ONE AND j MINUS ONE PLUS houses AT j MINUS houses AT i  
            END FOR  
        END FOR  
        
        SET dp TO TWO-DIMENSIONAL LIST OF SIZE n PLUS ONE BY k PLUS ONE WITH ALL ELEMENTS SET TO POSITIVE INFINITY  
        SET dp AT ZERO AND ZERO TO ZERO  
        
        FOR i FROM ONE TO n  
            FOR j FROM ONE TO k  
                FOR p FROM ZERO TO i MINUS ONE  
                    SET dp AT i AND j TO MINIMUM OF dp AT i AND j AND dp AT p AND j MINUS ONE PLUS cost AT p AND i MINUS ONE  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp AT n AND k  
    END FUNCTION  
END CLASS
