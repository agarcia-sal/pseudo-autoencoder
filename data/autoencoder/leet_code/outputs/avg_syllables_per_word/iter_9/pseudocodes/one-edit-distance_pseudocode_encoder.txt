CLASS Solution  
    FUNCTION isOneEditDistance(self, s, t)  
        IF length of s GREATER THAN length of t THEN  
            RETURN result of self calling isOneEditDistance with parameters t and s  
        END IF  
        
        SET diff TO length of t MINUS length of s  
        
        IF diff GREATER THAN 1 THEN  
            RETURN False  
        END IF  
        
        IF diff EQUALS 0 THEN  
            RETURN whether the count of positions where characters from s and t differ EQUALS 1  
        ELSE  
            FOR index FROM zero TO length of s MINUS one  
                IF character at index in s NOT EQUALS character at index in t THEN  
                    RETURN whether the substring of s from index onward EQUALS the substring of t from index plus one onward  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
