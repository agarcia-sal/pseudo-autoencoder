CLASS Solution  
    FUNCTION rangeBitwiseAnd(self, left, right)  
        SET shift TO zero  
        WHILE left LESS THAN right DO  
            SET left TO left DIVIDED BY two (integer division)  
            SET right TO right DIVIDED BY two (integer division)  
            INCREMENT shift BY one  
        END WHILE  
        RETURN left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
