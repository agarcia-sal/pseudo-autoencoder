CLASS Solution  
    FUNCTION findSubstring(self, s, words)  
        IF s IS empty OR words IS empty OR first element of words IS empty THEN  
            RETURN empty list  
        END IF  
        
        SET word_length TO length of first element in words  
        SET num_words TO length of words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO count of occurrences of each word in words  
        SET result TO empty list  
        
        FOR i FROM zero TO length of s MINUS total_length PLUS one  
            SET seen TO empty count of words  
            FOR j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO substring of s from index j TO j PLUS word_length  
                INCREMENT count of word IN seen BY one  
                IF count of word IN seen GREATER THAN count of word IN word_count THEN  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF inner loop completed without break THEN  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
