CLASS Solution  
    FUNCTION findTargetSumWays(self, nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS length of nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN 1  
                ELSE  
                    RETURN 0  
                END IF  
            END IF  
            
            IF memo CONTAINS key composed of index AND current_sum THEN  
                RETURN value from memo at key composed of index AND current_sum  
            END IF  
            
            SET positive TO dfs called with index incremented by one, current_sum PLUS value at nums at index, and memo  
            SET negative TO dfs called with index incremented by one, current_sum MINUS value at nums at index, and memo  
            
            SET memo at key composed of index AND current_sum TO positive PLUS negative  
            RETURN memo at key composed of index AND current_sum  
        END FUNCTION  
        
        RETURN dfs called with zero, zero, and empty collection  
    END FUNCTION  
END CLASS
