CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO length of dist  
        SET dp TO a two dimensional list of size n by (n plus one) with all elements set to infinity  
        SET total_dist TO sum of elements in dist  
        
        IF total_dist divided by speed GREATER THAN hoursBefore THEN  
            RETURN minus one  
        END IF  
        
        SET dp at position zero zero TO dist at position zero  
        FOR j FROM one TO n MINUS one DO  
            SET dp at position zero j TO dist at position zero  
        END FOR  
        
        FOR i FROM one TO n MINUS one DO  
            FOR j FROM zero TO i DO  
                IF j LESS THAN i THEN  
                    SET dp at position i j TO ceiling of (dp at position i minus one j divided by speed) MULTIPLIED BY speed PLUS dist at position i  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET dp at position i j TO minimum of dp at position i j AND dp at position i minus one j minus one PLUS dist at position i  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one DO  
            IF dp at position n minus one j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
