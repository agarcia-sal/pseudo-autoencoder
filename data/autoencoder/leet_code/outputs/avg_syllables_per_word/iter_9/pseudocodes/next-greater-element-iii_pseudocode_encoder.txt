CLASS Solution  
    FUNCTION nextGreaterElement(self, n)  
        SET digits TO list of characters representing n  
        SET length TO number of elements in digits  
        
        SET i TO length MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND digit at position i GREATER THAN OR EQUAL TO digit at position i PLUS 1  
            DECREMENT i BY 1  
        END WHILE  
        
        IF i EQUALS -1 THEN  
            RETURN -1  
        END IF  
        
        SET j TO length MINUS 1  
        WHILE digit at position j LESS THAN OR EQUAL TO digit at position i  
            DECREMENT j BY 1  
        END WHILE  
        
        SWAP digit at position i WITH digit at position j  
        
        REVERSE the sublist of digits from position i PLUS 1 to the end  
        
        SET result TO integer value of concatenation of digits  
        
        IF result GREATER THAN (two raised to the power of thirty-one MINUS 1) THEN  
            RETURN -1  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
