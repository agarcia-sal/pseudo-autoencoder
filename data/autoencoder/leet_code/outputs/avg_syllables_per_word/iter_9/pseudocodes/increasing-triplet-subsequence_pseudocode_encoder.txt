CLASS Solution  
    FUNCTION increasingTriplet(self, nums)  
        SET first TO positive infinity  
        SET second TO positive infinity  

        FOR each num IN nums  
            IF num LESS THAN OR EQUALS first THEN  
                SET first TO num  
            ELSE IF num LESS THAN OR EQUALS second THEN  
                SET second TO num  
            ELSE  
                RETURN true  
            END IF  
        END FOR  

        RETURN false  
    END FUNCTION  
END CLASS
