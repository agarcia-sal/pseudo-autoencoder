CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO TreeNode(created with values AT first index)  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN length of values AND values AT i IS NOT none THEN  
            SET node.left TO TreeNode(created with values AT i)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values AT i IS NOT none THEN  
            SET node.right TO TreeNode(created with values AT i)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node IS none THEN  
                RETURN  
            END IF  
            APPEND node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left IS none AND node.right IS none AND current_sum EQUALS targetSum THEN  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs WITH node.left, current_sum, path  
            CALL dfs WITH node.right, current_sum, path  
            REMOVE last element FROM path  
        END FUNCTION  
        SET result TO an empty list  
        CALL dfs WITH root, zero, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
