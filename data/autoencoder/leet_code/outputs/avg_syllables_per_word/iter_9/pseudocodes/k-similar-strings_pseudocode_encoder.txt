CLASS Solution  
    FUNCTION kSimilarity(self, s1, s2)  
        DEFINE FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN length of s AND character at position i in s EQUALS character at position i in s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO length of s MINUS one  
                IF character at position j in s EQUALS character at position i in s2 AND character at position j in s NOT EQUALS character at position j in s2 THEN  
                    YIELD concatenation of substring of s from start to i, character at position j in s, substring of s from i plus one to j minus one, character at position i in s, and substring of s from j plus one to end  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET queue TO deque containing tuple of s1 and zero  
        SET visited TO set containing s1  
      
        WHILE queue is not empty  
            REMOVE leftmost element from queue and ASSIGN to s and swaps  
            IF s EQUALS s2 THEN  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited THEN  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor and swaps incremented by one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
