CLASS Solution  
    FUNCTION killProcess(self, pid, ppid, kill)  
        SET tree TO an empty mapping from keys to lists  
        FOR each child PROCESS paired with parent PROCESS IN the pairing of pid and ppid  
            APPEND child PROCESS TO the list associated with parent PROCESS in tree  
        END FOR  
        
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the first element removed FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY the list of children of current FROM tree  
        END WHILE  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
