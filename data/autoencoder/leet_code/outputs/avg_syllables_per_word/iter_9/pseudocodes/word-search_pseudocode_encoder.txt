CLASS Solution  
    FUNCTION exist(self, board, word)  
        IF board IS empty THEN  
            RETURN False  
        END IF  

        SET m TO length of board  
        SET n TO length of first element in board  
        SET visited TO a two-dimensional list of size m by n with all values set to False  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS length of word THEN  
                RETURN True  
            END IF  

            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR visited at position x y IS True OR board at position x y NOT EQUALS word at index THEN  
                RETURN False  
            END IF  

            SET visited at position x y TO True  
            FOR each pair dx dy IN the list of pairs consisting of negative one zero, one zero, zero negative one, zero one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one) RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            SET visited at position x y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero) RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
