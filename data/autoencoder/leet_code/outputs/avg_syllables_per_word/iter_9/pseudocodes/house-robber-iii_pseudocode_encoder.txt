CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as first element of values, left as None, right as None  
    SET i TO 1  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val as values at i, left as None, right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val as values at i, left as None, right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(self, root)  
        FUNCTION rob_from_node(node)  
            IF node IS None THEN  
                RETURN 0, 0  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node(node.left)  
            SET right_robbed, right_not_robbed TO rob_from_node(node.right)  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO maximum of left_robbed and left_not_robbed PLUS maximum of right_robbed and right_not_robbed  
            RETURN rob_this_node, not_rob_this_node  
        END FUNCTION  
        RETURN maximum value FROM rob_from_node(root)  
    END FUNCTION  
END CLASS
