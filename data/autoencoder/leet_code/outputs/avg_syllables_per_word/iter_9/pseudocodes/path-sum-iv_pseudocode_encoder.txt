CLASS Solution  
    FUNCTION pathSum(self, nums)  
        SET tree TO empty dictionary  
        
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO (num DIVIDED BY ten) MODULO ten  
            SET value TO num MODULO ten  
            SET tree at key (depth, pos) TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO value in tree at key (depth, pos)  
            IF current_value IS None THEN  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            
            SET left_child TO pair of (depth PLUS one, pos MULTIPLIED BY two MINUS one)  
            SET right_child TO pair of (depth PLUS one, pos MULTIPLIED BY two)  
            
            IF left_child NOT IN tree AND right_child NOT IN tree THEN  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO result of dfs called with left_child first element, left_child second element, new_path_sum  
            SET right_sum TO result of dfs called with right_child first element, right_child second element, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN result of dfs called with one, one, zero  
    END FUNCTION  
END CLASS
