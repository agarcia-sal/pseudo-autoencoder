CLASS Solution  
    FUNCTION maxSum(self, nums1, nums2)  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO one billion PLUS seven  
        
        WHILE i LESS THAN length of nums1 AND j LESS THAN length of nums2  
            IF element at index i in nums1 LESS THAN element at index j in nums2 THEN  
                SET sum1 TO sum1 PLUS element at index i in nums1  
                INCREMENT i BY one  
            ELSE IF element at index i in nums1 GREATER THAN element at index j in nums2 THEN  
                SET sum2 TO sum2 PLUS element at index j in nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO maximum of sum1 AND sum2 PLUS element at index i in nums1  
                SET sum2 TO maximum of sum1 AND sum2 PLUS element at index i in nums1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i LESS THAN length of nums1  
            SET sum1 TO sum1 PLUS element at index i in nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j LESS THAN length of nums2  
            SET sum2 TO sum2 PLUS element at index j in nums2  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN maximum of sum1 AND sum2 modulo MOD  
    END FUNCTION  
END CLASS
