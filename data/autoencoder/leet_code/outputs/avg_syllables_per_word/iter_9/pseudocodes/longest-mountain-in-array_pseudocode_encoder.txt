CLASS Solution  
    FUNCTION longestMountain(self, arr)  
        SET n TO length of arr  
        SET longest TO zero  
        
        SET i TO one  
        WHILE i LESS THAN n MINUS one  
            IF arr AT i MINUS one LESS THAN arr AT i AND arr AT i GREATER THAN arr AT i PLUS one THEN  
                SET left TO i MINUS one  
                SET right TO i PLUS one  
                
                WHILE left GREATER THAN zero AND arr AT left MINUS one LESS THAN arr AT left  
                    DECREMENT left BY one  
                END WHILE  
                
                WHILE right LESS THAN n MINUS one AND arr AT right GREATER THAN arr AT right PLUS one  
                    INCREMENT right BY one  
                END WHILE  
                
                IF longest LESS THAN right MINUS left PLUS one THEN  
                    SET longest TO right MINUS left PLUS one  
                END IF  
                
                SET i TO right  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN longest  
    END FUNCTION  
END CLASS
