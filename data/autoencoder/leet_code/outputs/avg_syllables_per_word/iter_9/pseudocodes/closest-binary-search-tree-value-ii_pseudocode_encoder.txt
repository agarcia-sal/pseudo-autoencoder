CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode USING first element of values  
    SET i TO one  
    INITIALIZE queue as an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode USING values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode USING values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF value of p IS NOT equal to value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS none THEN  
                RETURN  
            END IF  
            CALL inorder_traversal WITH left child of node  
            IF length of result IS LESS THAN k THEN  
                APPEND value of node TO result  
            ELSE  
                IF absolute value of (value of node MINUS target) IS LESS THAN absolute value of (first element of result MINUS target) THEN  
                    REMOVE first element FROM result  
                    APPEND value of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH right child of node  
        END FUNCTION  
        
        INITIALIZE result as an empty double-ended queue  
        CALL inorder_traversal WITH root  
        RETURN result converted to a list  
    END FUNCTION  
END CLASS
