CLASS Solution  
    FUNCTION reachNumber(self, target)  
        SET target TO absolute value of target  
        SET k TO zero  
        SET sum_steps TO zero  
        WHILE sum_steps LESS THAN target  
            INCREMENT k BY one  
            INCREMENT sum_steps BY k  
        END WHILE  
        IF (sum_steps MINUS target) modulo two EQUALS zero THEN  
            RETURN k  
        ELSE  
            IF k modulo two EQUALS zero THEN  
                RETURN k PLUS one  
            ELSE  
                RETURN k PLUS two  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
