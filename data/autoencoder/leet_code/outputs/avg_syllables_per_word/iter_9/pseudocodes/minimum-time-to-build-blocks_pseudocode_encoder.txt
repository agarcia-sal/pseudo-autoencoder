CLASS Solution  
    FUNCTION minBuildTime(self, blocks, split)  
        TRANSFORM blocks INTO a data structure that always provides the smallest element first  
        
        WHILE the number of elements in blocks IS GREATER THAN one  
            REMOVE the smallest element FROM blocks AND STORE IN first  
            REMOVE the smallest element FROM blocks AND STORE IN second  
            
            SET combined_time TO split PLUS second  
            
            INSERT combined_time INTO blocks maintaining the property of smallest element first  
        END WHILE  
        
        RETURN the only remaining element IN blocks  
    END FUNCTION  
END CLASS
