CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH val AS first element of values AND next AS None  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO new ListNode WITH val AS val AND next AS None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self, head)  
        SET dummy TO new ListNode WITH val AS zero AND next AS None  
        SET dummy.next TO head  
        SET current TO dummy  
        
        WHILE current.next IS NOT None AND current.next.next IS NOT None  
            SET first TO current.next  
            SET second TO current.next.next  
            
            SET current.next TO second  
            SET first.next TO second.next  
            SET second.next TO first  
            
            SET current TO current.next.next  
        END WHILE  
        
        RETURN dummy.next  
    END FUNCTION  
END CLASS
