CLASS Solution  
    FUNCTION strangePrinter(self, s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            
            IF memo CONTAINS key tuple of i and j THEN  
                RETURN memo value at key tuple of i and j  
            END IF  
            
            SET result TO call dp with parameters i, j MINUS one, and memo PLUS one  
            
            FOR k FROM i TO j MINUS one  
                IF character at position k IN s EQUALS character at position j IN s THEN  
                    SET temp_result TO call dp with parameters i, k MINUS one, and memo PLUS call dp with parameters k, j MINUS one, and memo  
                    IF temp_result LESS THAN result THEN  
                        SET result TO temp_result  
                    END IF  
                END IF  
            END FOR  
            
            SET memo at key tuple of i and j TO result  
            RETURN result  
        END FUNCTION  
        
        RETURN call dp with parameters zero, length of s MINUS one, and empty dictionary  
    END FUNCTION  
END CLASS
