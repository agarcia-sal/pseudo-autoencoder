CLASS Solution  
    FUNCTION canPartition(self, nums)  
        SET total_sum TO sum of elements in nums  
        
        IF total_sum modulo 2 NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        SET target TO total_sum divided by two  
        SET n TO length of nums  
        
        INITIALIZE dp AS list of False values with size target PLUS one  
        SET dp at position zero TO True  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO dp at position i OR dp at position i MINUS num  
            END FOR  
        END FOR  
        
        RETURN dp at position target  
    END FUNCTION  
END CLASS
