CLASS Solution  
    FUNCTION minMutation(self, start, end, bank)  
        SET s TO a set created from bank  
        SET q TO a deque initialized with a pair of start and zero  
        SET mp TO a mapping where key 'A' maps to the string 'TCG', key 'T' maps to 'ACG', key 'C' maps to 'ATG', and key 'G' maps to 'ATC'  
        WHILE q is not empty  
            REMOVE and SET t and step from the leftmost element of q  
            IF t EQUALS end THEN  
                RETURN step  
            END IF  
            FOR each index i and character v IN t  
                FOR each character j IN mp indexed by v  
                    SET next TO a new string formed by concatenating the substring of t from start to i, character j, and the substring of t from i plus one to end  
                    IF next IS IN s THEN  
                        APPEND the pair of next and step incremented by one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
