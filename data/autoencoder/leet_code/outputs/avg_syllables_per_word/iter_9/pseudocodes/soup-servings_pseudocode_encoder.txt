CLASS Solution  
    FUNCTION soupServings(self, n)  
        IF n GREATER THAN 4800 THEN  
            RETURN 1.0  
        END IF  
  
        SET n TO (n PLUS 24) DIVIDED BY 25  
  
        SET memo TO an empty collection  
  
        FUNCTION dp(a, b)  
            IF pair (a, b) EXISTS IN memo THEN  
                RETURN memo AT (a, b)  
            END IF  
  
            IF a LESS THAN OR EQUAL TO 0 AND b LESS THAN OR EQUAL TO 0 THEN  
                RETURN 0.5  
            END IF  
  
            IF a LESS THAN OR EQUAL TO 0 THEN  
                RETURN 1.0  
            END IF  
  
            IF b LESS THAN OR EQUAL TO 0 THEN  
                RETURN 0.0  
            END IF  
  
            SET prob TO one quarter MULTIPLIED BY (dp at (a MINUS 4, b) PLUS dp at (a MINUS 3, b MINUS 1) PLUS dp at (a MINUS 2, b MINUS 2) PLUS dp at (a MINUS 1, b MINUS 3))  
  
            SET memo AT (a, b) TO prob  
            RETURN prob  
        END FUNCTION  
  
        RETURN dp at (n, n)  
    END FUNCTION  
END CLASS
