CLASS Solution  
    FUNCTION maxDotProduct(self, nums1, nums2)  
        SET m TO length of nums1  
        SET n TO length of nums2  
        
        FUNCTION create_dp_table(rows, columns)  
            CREATE table with dimensions rows PLUS one by columns PLUS one  
            FOR each row FROM zero TO rows  
                FOR each column FROM zero TO columns  
                    SET table element at row and column TO negative infinity  
                END FOR  
            END FOR  
            RETURN table  
        END FUNCTION  
        
        SET dp TO create_dp_table(m, n)  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at index i MINUS one of nums1 MULTIPLIED BY element at index j MINUS one of nums2  
                
                SET dp element at i and j TO maximum of  
                    current_product,  
                    dp element at i MINUS one and j,  
                    dp element at i and j MINUS one,  
                    dp element at i MINUS one and j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        
        RETURN dp element at m and n  
    END FUNCTION  
END CLASS
