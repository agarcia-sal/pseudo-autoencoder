CLASS Solution  
    FUNCTION minWindow(self, s1, s2)  
        SET min_len TO infinity  
        SET min_start TO minus one  
        SET len1 TO length of s1  
        SET len2 TO length of s2  
          
        FOR i FROM zero TO len1 MINUS one  
            SET j TO zero  
            FOR k FROM i TO len1 MINUS one  
                IF character at index k of s1 EQUALS character at index j of s2 THEN  
                    INCREMENT j BY one  
                    IF j EQUALS len2 THEN  
                        IF k MINUS i PLUS one LESS THAN min_len THEN  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        IF min_start EQUALS minus one THEN  
            RETURN empty string  
        ELSE  
            RETURN substring of s1 from min_start TO min_start PLUS min_len MINUS one  
        END IF  
    END FUNCTION  
END CLASS
