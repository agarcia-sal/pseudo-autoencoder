CLASS Solution  
    FUNCTION threeSum(self, nums)  
        CALL sort ON nums  
        SET result TO empty list  
        SET n TO length of nums  
        
        FOR i FROM zero TO n MINUS two  
            IF i GREATER THAN zero AND element at index i IN nums EQUALS element at index i MINUS one IN nums THEN  
                CONTINUE to next iteration  
            END IF  
            
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            
            WHILE left LESS THAN right  
                SET total TO element at index i IN nums PLUS element at index left IN nums PLUS element at index right IN nums  
                
                IF total EQUALS zero THEN  
                    APPEND list containing element at index i IN nums, element at index left IN nums, element at index right IN nums TO result  
                    
                    WHILE left LESS THAN right AND element at index left IN nums EQUALS element at index left PLUS one IN nums  
                        INCREMENT left BY one  
                    END WHILE  
                    
                    WHILE left LESS THAN right AND element at index right IN nums EQUALS element at index right MINUS one IN nums  
                        DECREMENT right BY one  
                    END WHILE  
                    
                    INCREMENT left BY one  
                    DECREMENT right BY one  
                ELSE IF total LESS THAN zero THEN  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
