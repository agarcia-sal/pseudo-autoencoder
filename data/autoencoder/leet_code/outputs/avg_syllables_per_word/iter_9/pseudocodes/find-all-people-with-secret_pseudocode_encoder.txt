CLASS Solution  
    FUNCTION findAllPeople(self, n, meetings, firstPerson)  
        SET know_secret TO set containing zero and firstPerson  
        
        SORT meetings BY the third element of each meeting in ascending order  
        
        SET i TO zero  
        WHILE i LESS THAN length of meetings DO  
            SET current_time TO the third element of meetings at index i  
            
            SET graph TO a mapping with default empty list  
            WHILE i LESS THAN length of meetings AND third element of meetings at index i EQUALS current_time DO  
                SET x TO first element of meetings at index i  
                SET y TO second element of meetings at index i  
                APPEND y TO graph at key x  
                APPEND x TO graph at key y  
                INCREMENT i BY one  
            END WHILE  
            
            SET queue TO a double ended queue containing persons in know_secret who are keys in graph  
            SET visited TO empty set  
            WHILE queue is not empty DO  
                REMOVE first element FROM queue AND SET TO person  
                IF person IS IN visited THEN  
                    CONTINUE to next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN graph at key person DO  
                    IF neighbor NOT IN visited THEN  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        
        RETURN list constructed from know_secret  
    END FUNCTION  
END CLASS
