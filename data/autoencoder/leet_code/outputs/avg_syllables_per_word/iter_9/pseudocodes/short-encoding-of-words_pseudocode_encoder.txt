CLASS Solution  
    FUNCTION minimumLengthEncoding(self, words)  
        SET reversed_words TO list of each word in words reversed and sorted in ascending order  
        SET total_length TO zero  
        FOR index FROM zero TO length of reversed_words MINUS one  
            IF index PLUS one EQUALS length of reversed_words OR reversed_words at position index PLUS one DOES NOT start with reversed_words at position index THEN  
                INCREMENT total_length BY length of reversed_words at position index PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
