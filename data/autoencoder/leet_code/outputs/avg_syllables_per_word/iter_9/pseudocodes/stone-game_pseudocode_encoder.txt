CLASS Solution  
    FUNCTION stoneGame(self, piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN 0  
            END IF  
            RETURN the greater of (value at piles at i MINUS dfs(i PLUS 1, j)) AND (value at piles at j MINUS dfs(i, j MINUS 1))  
        END FUNCTION  

        RETURN dfs(0, length of piles MINUS 1) GREATER THAN 0  
    END FUNCTION  
END CLASS
