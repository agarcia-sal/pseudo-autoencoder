CLASS Solution  
    FUNCTION isNumber(self, s)  
        IMPORT regular_expression_module  
        SET number_regular_expression TO pattern representing optional plus or minus sign FOLLOWED BY either digits optionally followed by a decimal point and digits OR decimal point followed by digits FOLLOWED BY optional exponent part including 'e' with optional plus or minus sign and digits  
        SET match_result TO full match of number_regular_expression against s  
        IF match_result IS NOT NONE THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
