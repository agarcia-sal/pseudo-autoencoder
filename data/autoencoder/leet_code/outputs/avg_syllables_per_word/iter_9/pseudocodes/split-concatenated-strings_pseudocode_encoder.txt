CLASS Solution  
    FUNCTION splitLoopedString(self, strs)  
        FOR index FROM zero TO length of strs MINUS one  
            SET reversed_str TO strs at index reversed  
            IF reversed_str GREATER THAN strs at index THEN  
                SET strs at index TO reversed_str  
            END IF  
        END FOR  

        SET doubled_strs TO concatenation of strs with strs  

        SET max_string TO empty string  

        FOR index FROM zero TO length of strs MINUS one  
            FOR is_reversed IN list of False and True  
                IF is_reversed EQUALS True THEN  
                    SET current TO strs at index reversed  
                ELSE  
                    SET current TO strs at index  
                END IF  

                SET remaining TO concatenation of elements in doubled_strs from index plus one TO index plus length of strs MINUS one  

                FOR j FROM zero TO length of current  
                    SET candidate TO concatenation of substring of current from j TO end, remaining, and substring of current from start TO j  
                    IF candidate GREATER THAN max_string THEN  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN max_string  
    END FUNCTION  
END CLASS
