CLASS Solution  
    FUNCTION minMoves(self, nums, k)  
        SET positions TO empty list  
        FOR i FROM zero TO length of nums MINUS one  
            IF element at index i in nums EQUALS one THEN  
                APPEND i TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO integer division of sum of start AND end BY two  
            SET median TO element at index mid IN positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                INCREMENT cost BY absolute value of element at index i IN positions MINUS (median MINUS (mid MINUS i))  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO infinity  
        FOR i FROM zero TO length of positions MINUS k  
            SET min_cost TO minimum of min_cost AND calculate_cost(i, i PLUS k)  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
