CLASS Solution  
    FUNCTION minmaxGasDist(self, stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO length of stations MINUS one  
                SET distance TO stations at index MINUS stations at index MINUS one  
                INCREMENT additional_stations BY integer division of distance BY max_dist  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO ten raised to the power of eight  
        WHILE right MINUS left GREATER THAN zero point zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
