CLASS Solution  
    FUNCTION repeatedStringMatch(self, a, b)  
        SET min_repeats TO negative of negative length of b divided by length of a rounded down to nearest integer greater than or equal  
        IF b is substring of a repeated min_repeats times THEN  
            RETURN min_repeats  
        END IF  
        IF b is substring of a repeated min_repeats plus one times THEN  
            RETURN min_repeats plus one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
