CLASS Solution  
    FUNCTION maxRunTime(self, n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the smaller value BETWEEN battery AND time  
            END FOR  
            RETURN whether total IS GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  

        SET left TO zero  
        SET right TO sum of batteries DIVIDED BY n  
        WHILE left IS LESS THAN right  
            SET mid TO the integer division of the sum of left, right, and one BY two  
            IF canRunFor(mid) IS true THEN  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
