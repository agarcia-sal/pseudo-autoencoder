CLASS Solution  
    FUNCTION hIndex(self, citations)  
        SET n TO length of citations  
        SET left TO zero  
        SET right TO n MINUS one  
        SET h_index TO zero  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS (right MINUS left) DIVIDED BY two  
            IF citations AT mid GREATER THAN OR EQUAL TO n MINUS mid THEN  
                SET h_index TO n MINUS mid  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN h_index  
    END FUNCTION  
END CLASS
