CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image IS empty OR first element of image IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of image  
        SET n TO length of first element of image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN image at position row  
                IF pixel EQUALS character one THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR index FROM zero TO m MINUS one  
                IF image at position index and col EQUALS character one THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high DO  
                SET mid TO integer division of low PLUS high BY two  
                IF hasBlackPixelInRow(mid) THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high DO  
                SET mid TO integer division of low PLUS high PLUS one BY two  
                IF hasBlackPixelInRow(mid) THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high DO  
                SET mid TO integer division of low PLUS high BY two  
                IF hasBlackPixelInColumn(mid) THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high DO  
                SET mid TO integer division of low PLUS high PLUS one BY two  
                IF hasBlackPixelInColumn(mid) THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO result of findTop()  
        SET bottom TO result of findBottom()  
        SET left TO result of findLeft()  
        SET right TO result of findRight()  
        
        RETURN product of (bottom MINUS top PLUS one) AND (right MINUS left PLUS one)  
    END FUNCTION  
END CLASS
