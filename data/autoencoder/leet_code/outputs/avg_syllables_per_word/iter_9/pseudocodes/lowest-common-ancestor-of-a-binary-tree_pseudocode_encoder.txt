CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val as first element of values, left as none, right as none  
    SET i TO 1  
    INITIALIZE queue as empty collection supporting removal of first element  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        REMOVE first element FROM queue and SET TO node  
        IF i LESS THAN length of values AND values at index i IS NOT none THEN  
            SET node.left TO new TreeNode with val as values at index i, left as none, right as none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at index i IS NOT none THEN  
            SET node.right TO new TreeNode with val as values at index i, left as none, right as none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root IS none OR root EQUALS p OR root EQUALS q THEN  
            RETURN root  
        END IF  
        SET left TO self.lowestCommonAncestor(root.left, p, q)  
        SET right TO self.lowestCommonAncestor(root.right, p, q)  
        IF left IS NOT none AND right IS NOT none THEN  
            RETURN root  
        END IF  
        IF left IS NOT none THEN  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
