CLASS Solution  
    FUNCTION minPathSum(self, grid)  
        IF grid IS empty OR first element of grid IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of grid  
        SET n TO length of first element of grid  
        
        CALL a FUNCTION to create a 2D list dp with m rows and n columns initialized to zero  
        
        SET element at first row and first column of dp TO element at first row and first column of grid  
        
        FOR each index j FROM one TO n MINUS one  
            SET element at first row and j of dp TO element at first row and j MINUS one of dp PLUS element at first row and j of grid  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            SET element at i and first column of dp TO element at i MINUS one and first column of dp PLUS element at i and first column of grid  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            FOR each index j FROM one TO n MINUS one  
                SET element at i and j of dp TO the minimum of element at i MINUS one and j of dp AND element at i and j MINUS one of dp PLUS element at i and j of grid  
            END FOR  
        END FOR  
        
        RETURN element at m MINUS one and n MINUS one of dp  
    END FUNCTION  
END CLASS
