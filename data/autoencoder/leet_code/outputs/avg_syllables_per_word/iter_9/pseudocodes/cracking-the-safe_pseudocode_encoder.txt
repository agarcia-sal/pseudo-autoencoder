CLASS Solution  
    FUNCTION crackSafe(self, n, k)  
        SET visited TO empty set  
        SET sequence TO empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO node CONCATENATED WITH string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH argument substring of edge FROM second character TO end  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string of zero REPEATED n MINUS one times  
        CALL dfs WITH argument start_node  
        RETURN concatenation of reversed sequence elements FOLLOWED BY start_node  
    END FUNCTION  
END CLASS
