CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH val SET TO first element IN values AND next SET TO None  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO new ListNode WITH val SET TO val AND next SET TO None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val SET TO first element IN values AND left and right SET TO None  
    SET i TO 1  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY DO  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN length of values AND values[i] IS NOT None THEN  
            SET node.left TO new TreeNode WITH val SET TO values[i] AND left and right SET TO None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values[i] IS NOT None THEN  
            SET node.right TO new TreeNode WITH val SET TO values[i] AND left and right SET TO None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO length of s THEN  
                RETURN None, index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN length of s AND (character at s[index] IS digit OR character at s[index] EQUALS minus sign) DO  
                INCREMENT index BY 1  
            END WHILE  
            SET node TO new TreeNode WITH val SET TO integer parsed from s BETWEEN start AND index AND left and right SET TO None  
            IF index LESS THAN length of s AND character at s[index] EQUALS open parenthesis THEN  
                SET node.left, index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN length of s AND character at s[index] EQUALS open parenthesis THEN  
                SET node.right, index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN length of s AND character at s[index] EQUALS close parenthesis THEN  
                INCREMENT index BY 1  
            END IF  
            RETURN node, index  
        END FUNCTION  
        SET root, _ TO parse_tree(0)  
        RETURN root  
    END FUNCTION  
END CLASS
