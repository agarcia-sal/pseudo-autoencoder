CLASS Solution  
    FUNCTION orderOfLargestPlusSign(self, n, mines)  
        SET mines_set TO a set containing each element of mines converted to a tuple  
        
        INITIALIZE dp as a two-dimensional grid of size n by n  
        WHERE each cell contains a list of four zeroes representing counts for four directions  
        
        SET directions TO list containing (zero, one), (one, zero), (zero, negative one), and (negative one, zero)  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF pair (i, j) NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET dp at position i, j, first direction TO dp at position i, j minus one, first direction PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET dp at position i, j, second direction TO dp at position i minus one, j, second direction PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF pair (i, j) NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET dp at position i, j, third direction TO dp at position i, j plus one, third direction PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET dp at position i, j, fourth direction TO dp at position i plus one, j, fourth direction PLUS one  
                    END IF  
                    SET plus_sign_order TO minimum value of all four directions in dp at position i, j  
                    SET result TO maximum of result and plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
