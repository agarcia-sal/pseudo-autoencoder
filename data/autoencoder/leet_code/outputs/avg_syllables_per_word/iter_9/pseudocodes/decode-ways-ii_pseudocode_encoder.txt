CLASS Solution  
    FUNCTION numDecodings(self, s)  
        SET MOD TO ten raised to the power of nine PLUS seven  

        SET dp TO a list of zeros with length EQUALS length of s PLUS one  
        SET first element of dp TO one  

        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS star THEN  
                RETURN nine  
            ELSE IF char EQUALS zero THEN  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  

        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS star AND char2 EQUALS star THEN  
                RETURN fifteen  
            ELSE IF char1 EQUALS star THEN  
                IF char2 LESS THAN OR EQUAL TO six THEN  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS star THEN  
                IF char1 EQUALS one THEN  
                    RETURN nine  
                ELSE IF char1 EQUALS two THEN  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value of concatenating char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  

        FOR i FROM one TO length of s DO  
            SET dp at index i TO dp at index i PLUS dp at index i MINUS one MULTIPLIED BY ways_to_decode_single(s at index i MINUS one)  
            SET dp at index i TO dp at index i MODULO MOD  

            IF i GREATER THAN one THEN  
                SET dp at index i TO dp at index i PLUS dp at index i MINUS two MULTIPLIED BY ways_to_decode_pair(s at index i MINUS two, s at index i MINUS one)  
                SET dp at index i TO dp at index i MODULO MOD  
            END IF  
        END FOR  

        RETURN last element of dp  
    END FUNCTION  
END CLASS
