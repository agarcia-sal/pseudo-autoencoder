CLASS Solution  
    FUNCTION fullBloomFlowers(self, flowers, people)  
        SET start_times TO sorted list of start times extracted from flowers  
        SET end_times TO sorted list of end times extracted from flowers  
        
        SET result TO empty list  
        
        FOR each person IN people  
            SET started TO number of elements in start_times that are less than or equal to person  
            SET ended TO number of elements in end_times that are less than person  
            APPEND started MINUS ended TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
