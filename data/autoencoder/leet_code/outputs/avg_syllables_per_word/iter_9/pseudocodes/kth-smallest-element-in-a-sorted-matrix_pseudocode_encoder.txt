CLASS Solution  
    FUNCTION kthSmallest(self, matrix, k)  
        FUNCTION countLessEqual(mid)  
            SET count TO zero  
            SET row TO n MINUS one  
            SET col TO zero  
            WHILE row GREATER THAN OR EQUAL TO zero AND col LESS THAN n  
                IF element at matrix position row and col LESS THAN OR EQUAL TO mid THEN  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  

        SET n TO length of matrix  
        SET low TO element at matrix position zero and zero  
        SET high TO element at matrix position n MINUS one and n MINUS one  

        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two (integer division)  
            IF countLessEqual(mid) LESS THAN k THEN  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  

        RETURN low  
    END FUNCTION  
END CLASS
