CLASS Solution  
    FUNCTION maxRotateFunction(self, nums)  
        SET n TO length of nums  
        IF n EQUALS 1 THEN  
            RETURN 0  
        END IF  
        
        SET total_sum TO sum of elements in nums  
        SET current_sum TO sum of index MULTIPLIED BY element OVER each index and element in nums  
        
        SET max_value TO current_sum  
        
        FOR k FROM 1 TO n MINUS 1  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY element at position (n MINUS k) in nums  
            IF max_value LESS THAN current_sum THEN  
                SET max_value TO current_sum  
            END IF  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
