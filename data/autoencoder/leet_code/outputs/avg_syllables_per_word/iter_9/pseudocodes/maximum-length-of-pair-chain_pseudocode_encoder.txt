CLASS Solution  
    FUNCTION findLongestChain(self, pairs)  
        EXECUTE sorting pairs BY second element of each pair  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair WITH elements start and end IN pairs  
            IF start GREATER THAN current_end THEN  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
