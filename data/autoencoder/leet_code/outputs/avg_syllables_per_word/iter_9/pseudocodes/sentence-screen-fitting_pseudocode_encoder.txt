CLASS Solution  
    FUNCTION wordsTyping(self, sentence, rows, cols)  
        SET s TO join all elements in sentence separated by a single space, THEN append a single space  
        SET total_length TO the length of s  
        SET position TO zero  
        
        FOR each integer from zero TO rows MINUS one  
            INCREMENT position BY cols  
            
            IF character at index position MODULO total_length in s EQUALS single space THEN  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND character at index (position MINUS one) MODULO total_length in s NOT EQUALS single space DO  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN integer division of position BY total_length  
    END FUNCTION  
END CLASS
