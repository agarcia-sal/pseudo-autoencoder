CLASS Solution  
    FUNCTION kthGrammar(self, n, k)  
        IF n EQUALS 1 THEN  
            RETURN 0  
        END IF  

        SET length_of_previous_row TO two RAISED TO the power of (n MINUS two)  

        IF k LESS THAN OR EQUAL TO length_of_previous_row THEN  
            RETURN self.kthGrammar(n MINUS one, k)  
        ELSE  
            RETURN one MINUS self.kthGrammar(n MINUS one, k MINUS length_of_previous_row)  
        END IF  
    END FUNCTION  
END CLASS
