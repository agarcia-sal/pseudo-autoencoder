CLASS Solution
    FUNCTION findMaximumXOR(self, nums)
        SET max_xor TO zero
        
        FOR i FROM thirty-one DOWN TO zero
            SET max_xor TO max_xor MULTIPLIED BY two
            SET current_candidate TO max_xor PLUS one
            
            INITIALIZE an empty set called prefixes
            FOR each num IN nums
                SET prefix TO num DIVIDED BY two TO THE POWER OF i
                ADD prefix TO prefixes
            END FOR
            
            FOR each prefix IN prefixes
                IF the result of current_candidate XOR prefix IS IN prefixes THEN
                    SET max_xor TO current_candidate
                    EXIT the loop over prefixes
                END IF
            END FOR
        END FOR
        
        RETURN max_xor
    END FUNCTION
END CLASS
