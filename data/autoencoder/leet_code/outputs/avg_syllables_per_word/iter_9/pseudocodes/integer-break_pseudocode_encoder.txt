CLASS Solution  
    FUNCTION integerBreak(self, n)  
        SET dp TO list of zeros with length n PLUS one  
        SET dp at position one TO one  
        
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET max_product TO maximum of max_product, j MULTIPLIED BY dp at position i MINUS j, and j MULTIPLIED BY i MINUS j  
            END FOR  
            SET dp at position i TO max_product  
        END FOR  
        
        RETURN dp at position n  
    END FUNCTION  
END CLASS
