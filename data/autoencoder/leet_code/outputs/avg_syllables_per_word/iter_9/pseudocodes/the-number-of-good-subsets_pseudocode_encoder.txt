CLASS Solution  
    FUNCTION numberOfGoodSubsets(self, nums)  
        SET count TO frequency count of each number in nums  
        SET dp TO list of zeros with length equal to two raised to the power of ten  
        SET dp at index zero TO one  
        
        FOR each num IN keys of count  
            IF num EQUALS one THEN  
                SET dp at index zero TO dp at index zero MULTIPLIED BY two raised to the power of count at num  
                CONTINUE to next iteration  
            END IF  
            
            IF num NOT IN prime_factors THEN  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO prime_factors at num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR of one shifted left by (prime MINUS one)  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWNTO zero  
                IF bitwise AND of m and mask EQUALS zero THEN  
                    SET dp at index of bitwise OR of m and mask TO (dp at index of bitwise OR of m and mask PLUS dp at index m MULTIPLIED BY count at num) modulo MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO sum of dp elements from index one onwards modulo MOD  
        RETURN result  
    END FUNCTION  
END CLASS
