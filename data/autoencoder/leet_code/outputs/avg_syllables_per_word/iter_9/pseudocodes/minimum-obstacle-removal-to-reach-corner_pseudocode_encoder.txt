CLASS Solution  
    FUNCTION minimumObstacles(self, grid)  
        SET m TO length of grid  
        SET n TO length of first element in grid  
        SET directions TO list of four pairs representing right, down, left, and up moves  
        SET min_heap TO list containing a tuple of zero cost and starting coordinates zero, zero  
        SET visited TO set containing starting coordinates zero, zero  
        
        WHILE min_heap is not empty  
            REMOVE element with minimum cost, x coordinate, and y coordinate FROM min_heap  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one THEN  
                RETURN cost  
            END IF  
            
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND (nx, ny) NOT IN visited THEN  
                    ADD (nx, ny) TO visited  
                    SET new_cost TO cost PLUS value at grid location nx, ny  
                    INSERT tuple of new_cost, nx, ny INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
