CLASS Solution  
    FUNCTION numWays(self, steps, arrLen)  
        SET MOD TO one billion plus seven  
        SET max_index TO the smaller value between steps divided by two and arrLen minus one  
        
        SET dp TO a list of zeros with length max_index plus one  
        SET dp at position zero TO one  
        
        FOR each iteration FROM zero TO steps minus one  
            SET prev_dp TO a copy of dp  
            FOR each index i FROM zero TO max_index  
                SET dp at position i TO value of prev_dp at position i  
                IF i GREATER THAN zero THEN  
                    SET dp at position i TO the sum of dp at position i and prev_dp at position i minus one, modulo MOD  
                END IF  
                IF i LESS THAN max_index THEN  
                    SET dp at position i TO the sum of dp at position i and prev_dp at position i plus one, modulo MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position zero  
    END FUNCTION  
END CLASS
