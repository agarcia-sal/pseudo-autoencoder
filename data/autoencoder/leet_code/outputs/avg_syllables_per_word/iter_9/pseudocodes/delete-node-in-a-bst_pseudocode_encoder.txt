CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as first element of values  
    SET i TO one  
    INITIALIZE empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode with value as values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode with value as values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root IS none THEN  
            RETURN none  
        END IF  
        IF value of root GREATER THAN key THEN  
            SET left child of root TO result of deleteNode called with left child of root and key  
            RETURN root  
        END IF  
        IF value of root LESS THAN key THEN  
            SET right child of root TO result of deleteNode called with right child of root and key  
            RETURN root  
        END IF  
        IF left child of root IS none THEN  
            RETURN right child of root  
        END IF  
        IF right child of root IS none THEN  
            RETURN left child of root  
        END IF  
        SET node TO right child of root  
        WHILE left child of node IS NOT none DO  
            SET node TO left child of node  
        END WHILE  
        SET left child of node TO left child of root  
        SET root TO right child of root  
        RETURN root  
    END FUNCTION  
END CLASS
