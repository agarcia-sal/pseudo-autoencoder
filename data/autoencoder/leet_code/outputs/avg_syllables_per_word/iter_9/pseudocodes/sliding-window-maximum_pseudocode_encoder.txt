CLASS Solution  
    FUNCTION maxSlidingWindow(self, nums, k)  
        IF nums IS empty OR k EQUALS zero THEN  
            RETURN empty list  
        END IF  
        
        SET deq TO empty double-ended queue  
        SET result TO empty list  
        
        FOR i FROM zero TO length of nums MINUS one  
            IF deq is not empty AND front element of deq LESS THAN i MINUS k PLUS one THEN  
                REMOVE first element FROM deq  
            END IF  
            
            WHILE deq is not empty AND value at nums at last element of deq LESS THAN value at nums at i DO  
                REMOVE last element FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND value at nums at front element of deq TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
