CLASS Solution
    FUNCTION minSwap(self, nums1, nums2)
        SET n TO length of nums1
        INITIALIZE dp AS list of length n where each element is a list of two zeros

        SET dp at index zero and position zero TO zero
        SET dp at index zero and position one TO one

        FOR i FROM one TO n MINUS one
            IF element at nums1 index i GREATER THAN element at nums1 index i MINUS one AND element at nums2 index i GREATER THAN element at nums2 index i MINUS one THEN
                IF element at nums1 index i GREATER THAN element at nums2 index i MINUS one AND element at nums2 index i GREATER THAN element at nums1 index i MINUS one THEN
                    SET dp at index i and position zero TO minimum of dp at index i MINUS one and position zero AND dp at index i MINUS one and position one
                    SET dp at index i and position one TO value of dp at index i and position zero PLUS one
                ELSE
                    SET dp at index i and position zero TO value of dp at index i MINUS one and position zero
                    SET dp at index i and position one TO value of dp at index i MINUS one and position one PLUS one
                END IF
            ELSE
                IF element at nums1 index i GREATER THAN element at nums2 index i MINUS one AND element at nums2 index i GREATER THAN element at nums1 index i MINUS one THEN
                    SET dp at index i and position zero TO value of dp at index i MINUS one and position one
                    SET dp at index i and position one TO value of dp at index i MINUS one and position zero PLUS one
                ELSE
                    SET dp at index i and position zero TO value of dp at index i MINUS one and position one
                    SET dp at index i and position one TO value of dp at index i MINUS one and position zero PLUS one
                END IF
            END IF
        END FOR

        RETURN minimum of dp at index n MINUS one and position zero AND dp at index n MINUS one and position one
    END FUNCTION
END CLASS
