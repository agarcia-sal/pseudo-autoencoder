CLASS Solution  
    FUNCTION minOperations(self, target, arr)  
        CREATE empty mapping named index_map  
        FOR each index FROM zero TO length of target MINUS one  
            SET value TO element at target at index  
            SET index_map at value TO index  
        END FOR  
  
        CREATE empty list named transformed_arr  
        FOR each value IN arr  
            IF value EXISTS in index_map THEN  
                APPEND index_map at value TO transformed_arr  
            END IF  
        END FOR  
  
        CREATE empty list named lis  
  
        FOR each num IN transformed_arr  
            SET pos TO the position found by binary searching to the left in lis for num  
            IF pos EQUALS length of lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at lis at pos TO num  
            END IF  
        END FOR  
  
        SET lcs_length TO length of lis  
  
        RETURN length of target MINUS lcs_length  
    END FUNCTION  
END CLASS
