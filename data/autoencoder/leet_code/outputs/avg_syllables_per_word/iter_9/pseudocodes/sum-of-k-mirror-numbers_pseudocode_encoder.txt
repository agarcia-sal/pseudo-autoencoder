CLASS Solution  
    FUNCTION kMirror(self, k, n)  

        FUNCTION is_palindrome(s)  
            RETURN s EQUALS reverse of s  
        END FUNCTION  

        FUNCTION to_base_k(num, k)  
            IF num EQUALS zero THEN  
                RETURN string representation of zero  
            END IF  
            SET digits TO empty list  
            WHILE num GREATER THAN zero DO  
                APPEND remainder of num DIVIDED BY k TO digits  
                SET num TO quotient of num DIVIDED BY k  
            END WHILE  
            RETURN concatenation of string representations of digits in reverse order  
        END FUNCTION  

        FUNCTION generate_palindromes(length)  
            IF length EQUALS one THEN  
                FOR i FROM one TO nine DO  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO power of integer division of (length MINUS one) BY two  
                SET end TO ten RAISED TO power of integer division of (length PLUS one) BY two  
                FOR half FROM start TO one LESS THAN end DO  
                    SET half_str TO string representation of half  
                    IF length MODULO two EQUALS zero THEN  
                        SET full TO concatenation of half_str AND reverse of half_str  
                    ELSE  
                        SET full TO concatenation of half_str AND reverse of half_str EXCLUDING last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  

        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE size of k_mirror_numbers LESS THAN n DO  
            FOR pal IN generate_palindromes(length) DO  
                SET num TO integer value of pal  
                IF is_palindrome(to_base_k(num, k)) THEN  
                    APPEND num TO k_mirror_numbers  
                    IF size of k_mirror_numbers EQUALS n THEN  
                        BREAK from inner loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  

        RETURN sum of k_mirror_numbers  
    END FUNCTION  
END CLASS
