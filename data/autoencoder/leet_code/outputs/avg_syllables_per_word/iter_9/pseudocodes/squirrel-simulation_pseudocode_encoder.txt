CLASS Solution  
    FUNCTION minDistance(self, height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            RETURN absolute value of p1 at position zero MINUS p2 at position zero PLUS absolute value of p1 at position one MINUS p2 at position one  
        END FUNCTION  

        SET total_distance TO zero  
        SET max_save TO negative infinity  

        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance(nut, tree)  
            SET first_nut_distance TO distance(squirrel, nut) PLUS distance(nut, tree)  
            SET later_nut_distance TO two MULTIPLIED BY distance(nut, tree)  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save THEN  
                SET max_save TO save  
            END IF  
        END FOR  

        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
