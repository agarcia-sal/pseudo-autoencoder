CLASS Solution  
    FUNCTION wordSquares(self, words)  
        SET prefix_to_words TO a new dictionary mapping prefixes TO lists of words  
        
        FOR each word IN words  
            FOR index FROM zero TO length of word MINUS one  
                SET prefix TO substring of word FROM start UP TO index  
                APPEND word TO list at prefix_to_words[prefix]  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF length of square EQUALS length of first word in words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO concatenation of character at position length of square IN each word OF square  
            FOR each candidate IN prefix_to_words[prefix]  
                CALL backtrack WITH square concatenated WITH list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack WITH list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
