CLASS Solution  
    FUNCTION findMedianSortedArrays(self, nums1, nums2)  
        IF length of nums1 GREATER THAN length of nums2 THEN  
            SET temporary nums1 TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary nums1  
        END IF  
        
        SET x TO length of nums1  
        SET y TO length of nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low LESS THAN OR EQUAL TO high DO  
            SET partitionX TO integer division of (low PLUS high) BY two  
            SET partitionY TO integer division of (x PLUS y PLUS one) BY two MINUS partitionX  
            
            IF partitionX EQUALS zero THEN  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO element at index (partitionX MINUS one) OF nums1  
            END IF  
            
            IF partitionX EQUALS x THEN  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO element at index partitionX OF nums1  
            END IF  
            
            IF partitionY EQUALS zero THEN  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO element at index (partitionY MINUS one) OF nums2  
            END IF  
            
            IF partitionY EQUALS y THEN  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO element at index partitionY OF nums2  
            END IF  
            
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX THEN  
                IF modulo of (x PLUS y) BY two EQUALS zero THEN  
                    RETURN average of maximum of maxLeftX AND maxLeftY PLUS minimum of minRightX AND minRightY  
                ELSE  
                    RETURN maximum of maxLeftX AND maxLeftY  
                END IF  
            ELSE IF maxLeftX GREATER THAN minRightY THEN  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
