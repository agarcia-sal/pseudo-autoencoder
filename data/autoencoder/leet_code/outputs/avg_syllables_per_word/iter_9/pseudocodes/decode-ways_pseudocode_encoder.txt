CLASS Solution  
    FUNCTION numDecodings(self, s)  
        IF s IS empty OR the first character of s EQUALS character zero THEN  
            RETURN zero  
        END IF  
        
        SET dp TO a list of zeros with length equal to the length of s PLUS one  
        SET the first element of dp TO one  
        SET the second element of dp TO one  
        
        FOR index FROM two TO the length of s PLUS one  
            IF the character at position index MINUS one in s NOT EQUALS character zero THEN  
                INCREMENT dp at position index BY dp at position index MINUS one  
            END IF  
            
            SET two_digit TO the integer value of the substring of s from position index MINUS two TO index  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six THEN  
                INCREMENT dp at position index BY dp at position index MINUS two  
            END IF  
        END FOR  
        
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
