CLASS Solution  
    FUNCTION maxKilledEnemies(self, grid)  
        IF grid IS empty OR first element of grid IS empty THEN  
            RETURN zero  
        END IF  
        
        SET m TO length of grid  
        SET n TO length of first element of grid  
        SET max_kills TO zero  
        
        SET row_kills TO zero  
        SET col_kills TO list of zero with length n  
        
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF j EQUALS zero OR element at position (i, j MINUS one) in grid EQUALS character representing wall THEN  
                    SET row_kills TO zero  
                    FOR each k FROM j TO n MINUS one  
                        IF element at position (i, k) in grid EQUALS character representing enemy THEN  
                            INCREMENT row_kills BY one  
                        ELSE IF element at position (i, k) in grid EQUALS character representing wall THEN  
                            BREAK inner loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF i EQUALS zero OR element at position (i MINUS one, j) in grid EQUALS character representing wall THEN  
                    SET col_kills at index j TO zero  
                    FOR each k FROM i TO m MINUS one  
                        IF element at position (k, j) in grid EQUALS character representing enemy THEN  
                            INCREMENT col_kills at index j BY one  
                        ELSE IF element at position (k, j) in grid EQUALS character representing wall THEN  
                            BREAK inner loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF element at position (i, j) in grid EQUALS character representing empty cell THEN  
                    SET max_kills TO the greater value between max_kills AND sum of row_kills PLUS col_kills at index j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
