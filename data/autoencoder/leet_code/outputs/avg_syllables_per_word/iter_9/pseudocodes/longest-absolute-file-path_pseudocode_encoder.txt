CLASS Solution  
    FUNCTION lengthLongestPath(self, input)  
        SET lines TO the input SPLIT BY newline  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters in line  
            SET name TO the substring of line starting after depth tab characters  
            
            WHILE the size of stack GREATER THAN depth  
                REMOVE last element FROM stack  
            END WHILE  
            
            IF stack is not empty THEN  
                SET current_length TO the last element in stack PLUS the length of name PLUS one for the separator  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF name CONTAINS a dot character THEN  
                SET max_length TO the greater of max_length and current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
