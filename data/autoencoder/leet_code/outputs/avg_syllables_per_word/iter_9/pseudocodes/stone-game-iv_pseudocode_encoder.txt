CLASS Solution  
    FUNCTION winnerSquareGame(self, n)  
        CREATE list dp WITH n PLUS one elements ALL SET TO False  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF NOT dp AT POSITION i MINUS j MULTIPLIED BY j THEN  
                    SET dp AT POSITION i TO True  
                    BREAK  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN dp AT POSITION n  
    END FUNCTION  
END CLASS
