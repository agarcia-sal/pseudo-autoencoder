CLASS Solution  
    FUNCTION removeDuplicates(self, nums)  
        IF nums IS empty THEN  
            RETURN zero  
        END IF  
        
        SET write_index TO zero  
        
        FOR i FROM zero TO minimum of two AND length of nums MINUS one  
            SET element at write_index in nums TO element at i in nums  
            INCREMENT write_index BY one  
        END FOR  
        
        FOR i FROM two TO length of nums MINUS one  
            IF element at i in nums NOT EQUALS element at write_index MINUS two in nums THEN  
                SET element at write_index in nums TO element at i in nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
