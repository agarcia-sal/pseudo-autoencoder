CLASS Solution  
    FUNCTION optimalDivision(self, nums)  
        IF length of nums EQUALS 1 THEN  
            RETURN string representation of first element in nums  
        END IF  

        IF length of nums EQUALS 2 THEN  
            RETURN string concatenation of first element in nums, followed by the phrase " divided by ", followed by second element in nums  
        END IF  

        RETURN string concatenation of first element in nums, followed by the phrase " divided by ", followed by an opening parenthesis, followed by the sequence of string representations of elements from second to last in nums joined by the phrase " divided by ", followed by a closing parenthesis  
    END FUNCTION  
END CLASS
