CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(self, nums, k)  
        IF k LESS THAN OR EQUAL TO 1 THEN  
            RETURN 0  
        END IF  
  
        SET product TO 1  
        SET left TO 0  
        SET count TO 0  
  
        FOR each right FROM 0 TO length of nums MINUS 1  
            SET product TO product MULTIPLIED BY element at index right in nums  
  
            WHILE product GREATER THAN OR EQUAL TO k AND left LESS THAN OR EQUAL TO right  
                SET product TO product DIVIDED BY element at index left in nums  
                INCREMENT left BY 1  
            END WHILE  
  
            INCREMENT count BY right MINUS left PLUS 1  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
