CLASS Solution  
    FUNCTION numSubarrayBoundedMax(self, nums, left, right)  
        FUNCTION count(max_value)  
            SET count TO zero  
            SET current_length TO zero  
            FOR each num IN nums  
                IF num LESS THAN OR EQUAL TO max_value THEN  
                    INCREMENT current_length BY one  
                    INCREMENT count BY current_length  
                ELSE  
                    SET current_length TO zero  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  

        RETURN count(right) MINUS count(left MINUS one)  
    END FUNCTION  
END CLASS
