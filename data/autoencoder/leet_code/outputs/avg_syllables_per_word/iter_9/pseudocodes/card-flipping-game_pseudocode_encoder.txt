CLASS Solution  
    FUNCTION flipgame(self, fronts, backs)  
        SET same TO empty set  
        FOR each pair of elements x, y FROM fronts and backs TOGETHER  
            IF x EQUALS y THEN  
                APPEND x TO same  
            END IF  
        END FOR  

        SET all_numbers TO set of all elements from fronts and backs together  

        SET good_numbers TO all_numbers MINUS same  

        IF good_numbers IS NOT empty THEN  
            RETURN minimum element FROM good_numbers  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
