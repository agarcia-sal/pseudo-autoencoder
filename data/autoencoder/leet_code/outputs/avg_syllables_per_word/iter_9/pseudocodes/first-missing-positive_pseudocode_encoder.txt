CLASS Solution  
    FUNCTION firstMissingPositive(self, nums)  
        SET n TO length of nums  
        
        FOR i FROM zero TO n MINUS one  
            WHILE nums at index i GREATER THAN OR EQUAL TO one AND nums at index i LESS THAN OR EQUAL TO n AND nums at index nums at index i MINUS one NOT EQUALS nums at index i DO  
                SET correct_index TO nums at index i MINUS one  
                SET temp TO nums at index i  
                SET nums at index i TO nums at index correct_index  
                SET nums at index correct_index TO temp  
            END WHILE  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            IF nums at index i NOT EQUALS i PLUS one THEN  
                RETURN i PLUS one  
            END IF  
        END FOR  
        
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
