CLASS Solution  
    FUNCTION ambiguousCoordinates(self, s)  
        FUNCTION make(frag)  
            SET n TO length of frag  
            FOR d FROM one TO n  
                SET left TO substring of frag from start to position d  
                SET right TO substring of frag from position d to end  
                IF (left does not start with '0' OR left equals '0') AND (right does not end with '0') THEN  
                    YIELD concatenation of left, IF d not equals n THEN '.' ELSE empty string, and right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO substring of s from second character to second last character  
        RETURN list of formatted strings with format "(", a, ", ", b, ")" FOR each i from one TO length of s minus one, FOR each a, b IN Cartesian product of make applied to substring of s from start to i, and make applied to substring of s from i to end  
    END FUNCTION  
END CLASS
