CLASS Solution  
    FUNCTION canPartitionKSubsets(self, nums, k)  
        SET total_sum TO sum of all elements in nums  
        IF total_sum modulo k NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET target TO total_sum divided by k  
        SORT nums in descending order  

        FUNCTION can_partition(index, k, current_sum, used)  
            IF k EQUALS zero THEN  
                RETURN True  
            END IF  
            IF current_sum EQUALS target THEN  
                RETURN result of can_partition called with zero, k MINUS one, zero, used  
            END IF  

            FOR each i FROM index TO length of nums MINUS one  
                IF used at i EQUALS False AND current_sum PLUS nums at i LESS THAN OR EQUAL TO target THEN  
                    SET used at i TO True  
                    IF result of can_partition called with i PLUS one, k, current_sum PLUS nums at i, used EQUALS True THEN  
                        RETURN True  
                    END IF  
                    SET used at i TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET used TO list of False values with length equal to length of nums  
        RETURN result of can_partition called with zero, k, zero, used  
    END FUNCTION  
END CLASS
