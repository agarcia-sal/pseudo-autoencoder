CLASS Solution  
    FUNCTION shiftingLetters(self, s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM length of shifts MINUS one DOWN TO zero  
            SET total_shifts TO total_shifts PLUS shifts at index  
            SET shifts at index TO total_shifts  
        END FOR  

        SET result TO empty list  
        FOR index FROM zero TO length of s MINUS one  
            SET shifted_value TO ordinal position of character at index in s MINUS ordinal position of character 'a' PLUS shifts at index  
            SET wrapped_value TO remainder of shifted_value DIVIDED BY twenty six  
            SET new_character TO character with ordinal position of wrapped_value PLUS ordinal position of character 'a'  
            APPEND new_character TO result  
        END FOR  

        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
