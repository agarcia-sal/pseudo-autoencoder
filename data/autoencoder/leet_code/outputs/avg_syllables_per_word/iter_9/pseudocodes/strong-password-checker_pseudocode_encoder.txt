CLASS Solution  
    FUNCTION strongPasswordChecker(self, password)  
        SET n TO length of password  
        
        SET has_lower TO false  
        FOR each character IN password  
            IF character is lowercase THEN  
                SET has_lower TO true  
                EXIT FOR  
            END IF  
        END FOR  
        
        SET has_upper TO false  
        FOR each character IN password  
            IF character is uppercase THEN  
                SET has_upper TO true  
                EXIT FOR  
            END IF  
        END FOR  
        
        SET has_digit TO false  
        FOR each character IN password  
            IF character is digit THEN  
                SET has_digit TO true  
                EXIT FOR  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS (convert has_lower TO one PLUS convert has_upper TO one PLUS convert has_digit TO one)  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        
        WHILE i LESS THAN n DO  
            IF password at i EQUALS password at i minus one AND password at i EQUALS password at i minus two THEN  
                SET length TO two  
                WHILE i LESS THAN n AND password at i EQUALS password at i minus one DO  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three EQUALS zero THEN  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one THEN  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n LESS THAN six THEN  
            RETURN maximum of missing_types AND six MINUS n  
        ELSE IF n LESS THAN OR EQUAL TO twenty THEN  
            RETURN maximum of missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            SET replace TO replace MINUS minimum of delete AND one  
            SET replace TO replace MINUS (minimum of maximum of delete MINUS one AND zero AND two MULTIPLIED BY two) DIVIDED BY two  
            SET replace TO replace MINUS maximum of delete MINUS one MINUS two MULTIPLIED BY two AND zero DIVIDED BY three  
            
            RETURN delete PLUS maximum of missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
