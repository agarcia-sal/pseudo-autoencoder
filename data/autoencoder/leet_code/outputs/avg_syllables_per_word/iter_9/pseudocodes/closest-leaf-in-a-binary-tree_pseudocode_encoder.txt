CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at first element of values  
    SET i TO one  
    INITIALIZE queue as a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode with value at values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode with value at values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS none AND root.right IS none THEN  
            RETURN root.val  
        END IF  
        
        INITIALIZE graph as a default dictionary with empty list as default value  
        
        FUNCTION dfs(node, parent WITH DEFAULT none)  
            IF parent IS NOT none THEN  
                APPEND parent.val TO graph at key node.val  
                APPEND node.val TO graph at key parent.val  
            END IF  
            IF node.left IS NOT none THEN  
                CALL dfs(node.left, node)  
            END IF  
            IF node.right IS NOT none THEN  
                CALL dfs(node.right, node)  
            END IF  
        END FUNCTION  
        
        CALL dfs(root)  
        
        INITIALIZE queue as a new deque containing k  
        INITIALIZE visited as an empty set  
        
        WHILE queue IS NOT empty DO  
            SET current TO first element REMOVED FROM queue  
            ADD current TO visited  
            
            IF length of graph at current EQUALS one AND current NOT EQUALS root.val THEN  
                RETURN current  
            END IF  
            
            FOR each neighbor IN graph at current DO  
                IF neighbor NOT IN visited THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
