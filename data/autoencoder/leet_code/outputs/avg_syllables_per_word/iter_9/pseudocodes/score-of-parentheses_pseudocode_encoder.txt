CLASS Solution  
    FUNCTION scoreOfParentheses(self, s)  
        SET stack TO empty list  
        SET score TO zero  
        
        FOR each character IN s  
            IF character EQUALS opening parenthesis THEN  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO value popped from stack PLUS (one IF score EQUALS zero ELSE score MULTIPLIED BY two)  
            END IF  
        END FOR  
        
        RETURN score  
    END FUNCTION  
END CLASS
