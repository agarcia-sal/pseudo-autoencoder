CLASS Solution  
    FUNCTION getMinDistSum(self, positions)  
        SET n TO length of positions  
        SET x TO 0  
        SET y TO 0  
        FOR each (x1, y1) IN positions  
            INCREMENT x BY x1  
            INCREMENT y BY y1  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO zero point nine nine nine  
        SET eps TO zero point zero zero zero zero zero one  
        SET alpha TO zero point five  
        WHILE true  
            SET grad_x TO 0  
            SET grad_y TO 0  
            SET dist TO 0  
            FOR each (x1, y1) IN positions  
                SET a TO x MINUS x1  
                SET b TO y MINUS y1  
                SET c TO square root of (a MULTIPLIED BY a PLUS b MULTIPLIED BY b)  
                INCREMENT grad_x BY a DIVIDED BY (c PLUS zero point zero zero zero zero zero zero one)  
                INCREMENT grad_y BY b DIVIDED BY (c PLUS zero point zero zero zero zero zero zero one)  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            SET x TO x MINUS dx  
            SET y TO y MINUS dy  
            SET alpha TO alpha MULTIPLIED BY decay  
            IF absolute value of dx LESS THAN OR EQUAL TO eps AND absolute value of dy LESS THAN OR EQUAL TO eps THEN  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
