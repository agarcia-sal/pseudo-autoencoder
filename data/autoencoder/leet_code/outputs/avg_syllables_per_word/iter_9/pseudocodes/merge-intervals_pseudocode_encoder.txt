CLASS Solution  
    FUNCTION merge(self, intervals)  
        PERFORM sorting of intervals BASED ON first element of each interval  
        
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR second element of last element IN merged LESS THAN first element of interval THEN  
                APPEND interval TO merged  
            ELSE  
                SET second element of last element IN merged TO maximum of second element of last element IN merged AND second element of interval  
            END IF  
        END FOR  
        
        RETURN merged  
    END FUNCTION  
END CLASS
