CLASS Solution  
    FUNCTION smallestDistancePair(self, nums, k)  
        CALL nums SORT  
        SET n TO LENGTH of nums  
        
        FUNCTION count(mid)  
            SET count TO 0  
            SET left TO 0  
            FOR right FROM 0 TO n MINUS 1  
                WHILE nums AT right MINUS nums AT left GREATER THAN mid  
                    INCREMENT left BY 1  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO 0  
        SET right TO nums AT (n MINUS 1) MINUS nums AT 0  
        WHILE left LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            IF count(mid) LESS THAN k THEN  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
