CLASS Solution  
    FUNCTION findCheapestPrice(self, n, flights, src, dst, k)  
        BUILD graph AS a mapping FROM each integer i in range from zero TO n MINUS one TO an empty list  
        FOR each flight consisting of u, v, w IN flights  
            APPEND a pair consisting of v AND w TO graph AT key u  
        END FOR  

        SET min_heap TO a list containing a triple consisting of zero, MINUS one, AND src  
        SET min_cost TO a mapping FROM src TO MINUS one  

        WHILE min_heap is not empty  
            REMOVE and RETURN the triple with the smallest cost FROM min_heap AND SET to cost, stops, city  

            IF city EQUALS dst THEN  
                RETURN cost  
            END IF  

            IF stops GREATER THAN OR EQUAL TO k THEN  
                CONTINUE to the next iteration  
            END IF  

            FOR each pair of neighbor AND price IN graph AT city  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  

                IF neighbor NOT IN min_cost OR next_stops LESS THAN min_cost AT neighbor OR next_cost LESS THAN the cost part of the smallest element IN min_heap THEN  
                    SET min_cost AT neighbor TO next_stops  
                    ADD the triple consisting of next_cost, next_stops, neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  

        RETURN MINUS one  
    END FUNCTION  
END CLASS
