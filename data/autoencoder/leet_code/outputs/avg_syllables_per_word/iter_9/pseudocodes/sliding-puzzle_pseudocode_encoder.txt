CLASS Solution
    FUNCTION slidingPuzzle(self, board)
        SET start TO tuple of tuples created by converting each row in board into a tuple
        SET target TO tuple containing tuples (1, 2, 3) and (4, 5, 0)
        
        SET neighbors TO mapping of zero tile positions to sets of possible swap positions:
            position 0 maps to set containing 1 and 3
            position 1 maps to set containing 0, 2, and 4
            position 2 maps to set containing 1 and 5
            position 3 maps to set containing 0 and 4
            position 4 maps to set containing 1, 3, and 5
            position 5 maps to set containing 2 and 4
        
        FOR i FROM zero TO one
            FOR j FROM zero TO two
                IF element at start indexed by i and j EQUALS zero THEN
                    SET zero_pos TO i multiplied by three PLUS j
                END IF
            END FOR
        END FOR
        
        SET queue TO a double ended queue initialized with a tuple containing start, zero_pos, and zero
        SET visited TO a set initialized with start
        
        WHILE queue is not empty
            REMOVE first element from queue and unpack into state, zero_pos, moves
            
            IF state EQUALS target THEN
                RETURN moves
            END IF
            
            FOR each neighbor in neighbors at key zero_pos
                SET new_state TO a list of lists created by converting each row in state into a list
                SET i TO integer division of zero_pos by three
                SET j TO zero_pos modulo three
                SET ni TO integer division of neighbor by three
                SET nj TO neighbor modulo three
                
                SWAP elements at new_state indexed by i and j with element at new_state indexed by ni and nj
                
                SET new_tuple_state TO tuple of tuples created by converting each row in new_state into a tuple
                
                IF new_tuple_state NOT IN visited THEN
                    ADD new_tuple_state to visited
                    APPEND tuple containing new_tuple_state, neighbor, and moves plus one to queue
                END IF
            END FOR
        END WHILE
        
        RETURN negative one
    END FUNCTION
END CLASS
