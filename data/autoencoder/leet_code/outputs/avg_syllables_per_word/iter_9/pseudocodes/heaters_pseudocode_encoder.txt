CLASS Solution  
    FUNCTION findRadius(self, houses, heaters)  
        SORT houses IN ASCENDING ORDER  
        SORT heaters IN ASCENDING ORDER  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN length of heaters MINUS one AND absolute value of heaters at j MINUS house GREATER THAN OR EQUAL TO absolute value of heaters at j plus one MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO the greater of min_radius AND absolute value of heaters at j MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
