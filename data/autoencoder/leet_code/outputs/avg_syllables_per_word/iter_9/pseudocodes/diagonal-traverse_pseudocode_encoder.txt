CLASS Solution  
    FUNCTION findDiagonalOrder(self, mat)  
        IF mat IS empty OR first element of mat IS empty THEN  
            RETURN empty list  
        END IF  
          
        SET m TO length of mat  
        SET n TO length of first element of mat  
        SET result TO empty list  
        SET r TO zero  
        SET c TO zero  
        SET direction TO one  
          
        FOR counter FROM zero TO m MULTIPLIED BY n MINUS one  
            APPEND element at position r and c in mat TO result  
            IF direction EQUALS one THEN  
                SET new_r TO r MINUS one  
                SET new_c TO c PLUS one  
            ELSE  
                SET new_r TO r PLUS one  
                SET new_c TO c MINUS one  
            END IF  
              
            IF new_r LESS THAN zero OR new_r EQUALS m OR new_c LESS THAN zero OR new_c EQUALS n THEN  
                IF direction EQUALS one THEN  
                    IF new_c EQUALS n THEN  
                        SET new_r TO r PLUS one  
                        SET new_c TO c  
                    ELSE  
                        SET new_r TO zero  
                        SET new_c TO c PLUS one  
                    END IF  
                ELSE  
                    IF new_r EQUALS m THEN  
                        SET new_r TO r  
                        SET new_c TO c PLUS one  
                    ELSE  
                        SET new_r TO r PLUS one  
                        SET new_c TO zero  
                    END IF  
                END IF  
                SET direction TO direction MULTIPLIED BY minus one  
            END IF  
              
            SET r TO new_r  
            SET c TO new_c  
        END FOR  
          
        RETURN result  
    END FUNCTION  
END CLASS
