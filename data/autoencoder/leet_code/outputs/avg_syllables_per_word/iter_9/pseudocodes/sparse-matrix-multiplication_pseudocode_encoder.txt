CLASS Solution  
    FUNCTION multiply(self, mat1, mat2)  
        SET m TO length of mat1  
        SET k TO length of first element of mat1  
        SET k TO length of mat2  
        SET n TO length of first element of mat2  
        
        INITIALIZE result AS a matrix with m rows and n columns filled with zeros  
        
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    INCREMENT element at result position i, j BY product of element at mat1 position i, p AND element at mat2 position p, j  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
