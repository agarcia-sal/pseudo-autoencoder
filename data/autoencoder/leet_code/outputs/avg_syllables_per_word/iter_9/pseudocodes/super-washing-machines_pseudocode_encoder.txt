CLASS Solution  
    FUNCTION findMinMoves(self, machines)  
        SET total_dresses TO sum of all elements in machines  
        SET n TO length of machines  
        
        IF total_dresses modulo n NOT EQUALS zero THEN  
            RETURN negative one  
        END IF  
        
        SET target TO total_dresses divided by n  
        SET left_sum TO zero  
        SET max_moves TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET left_excess TO left_sum MINUS target MULTIPLIED BY i  
            SET right_excess TO total_dresses MINUS left_sum MINUS machines at index i MINUS target MULTIPLIED BY (n MINUS i MINUS one)  
            
            SET current_moves TO maximum of absolute value of left_excess, absolute value of right_excess, and machines at index i MINUS target  
            SET max_moves TO maximum of max_moves and current_moves  
            
            INCREMENT left_sum BY machines at index i  
        END FOR  
        
        RETURN max_moves  
    END FUNCTION  
END CLASS
