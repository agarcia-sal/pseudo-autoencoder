CLASS Solution  
    FUNCTION findClosestElements(self, arr, k, x)  
        SET left TO bisect_left position of x in arr MINUS one  
        SET right TO left PLUS one  
        
        WHILE right MINUS left MINUS one LESS THAN k  
            IF left EQUALS minus one OR (right LESS THAN length of arr AND absolute value of arr at right MINUS x LESS THAN absolute value of arr at left MINUS x) THEN  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN sublist of arr from position left PLUS one TO position right (not inclusive)  
    END FUNCTION  
END CLASS
