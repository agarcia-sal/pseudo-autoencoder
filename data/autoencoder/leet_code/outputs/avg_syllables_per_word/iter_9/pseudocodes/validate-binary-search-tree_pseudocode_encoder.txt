CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val EQUALS first element of values  
    SET i TO one  
    CREATE queue as empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO remove first element FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.left TO new TreeNode with val EQUALS values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT none THEN  
            SET node.right TO new TreeNode with val EQUALS values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low EQUALS negative infinity, high EQUALS positive infinity)  
            IF node IS none THEN  
                RETURN true  
            END IF  
            IF node.val LESS THAN OR EQUAL TO low OR node.val GREATER THAN OR EQUAL TO high THEN  
                RETURN false  
            END IF  
            RETURN validate(node.left, low, node.val) AND validate(node.right, node.val, high)  
        END FUNCTION  
        
        RETURN validate(root)  
    END FUNCTION  
END CLASS
