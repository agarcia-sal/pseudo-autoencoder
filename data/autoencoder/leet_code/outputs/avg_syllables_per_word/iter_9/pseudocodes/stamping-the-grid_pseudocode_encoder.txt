CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO length of grid  
        SET n TO length of first element in grid  
        
        FUNCTION prefix_sum(matrix)  
            SET p TO two-dimensional list of zeros with dimensions m plus one by n plus one  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET p at position r and c TO value at matrix position r minus one and c minus one PLUS p at position r minus one and c PLUS p at position r and c minus one MINUS p at position r minus one and c minus one  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            RETURN p at position r2 plus one and c2 plus one MINUS p at position r2 plus one and c1 MINUS p at position r1 and c2 plus one PLUS p at position r1 and c1  
        END FUNCTION  
        
        SET p_grid TO result of calling prefix_sum with grid  
        SET p_stamps TO two-dimensional list of zeros with dimensions m plus one by n plus one  
        
        FOR r FROM zero TO m minus one  
            FOR c FROM zero TO n minus one  
                IF value at grid position r and c EQUALS zero THEN  
                    IF r PLUS stampHeight LESS THAN OR EQUAL TO m AND c PLUS stampWidth LESS THAN OR EQUAL TO n THEN  
                        IF result of submatrix_sum called with p_grid, r, c, r PLUS stampHeight MINUS one, c PLUS stampWidth MINUS one EQUALS zero THEN  
                            INCREMENT p_stamps at position r and c BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO result of calling prefix_sum with p_stamps  
        
        FOR r FROM zero TO m minus one  
            FOR c FROM zero TO n minus one  
                IF value at grid position r and c EQUALS zero THEN  
                    IF result of submatrix_sum called with p_stamps, maximum of zero and r MINUS stampHeight PLUS one, maximum of zero and c MINUS stampWidth PLUS one, r, c EQUALS zero THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
