CLASS Solution  
    FUNCTION smallestGoodBase(self, n)  
        SET n TO integer conversion of n  
        SET max_length TO integer part of logarithm base 2 of n INCREMENTED BY one  
        
        FOR m FROM max_length DOWN TO two  
            SET k TO integer part of n RAISED TO the power of one DIVIDED BY (m MINUS one)  
            
            IF k LESS THAN two THEN  
                CONTINUE to next iteration  
            END IF  
            
            SET num TO (k RAISED TO the power of m MINUS one) DIVIDED BY (k MINUS one)  
            
            IF num EQUALS n THEN  
                RETURN string conversion of k  
            END IF  
        END FOR  
        
        RETURN string conversion of n MINUS one  
    END FUNCTION  
END CLASS
