CLASS Solution  
    FUNCTION combinationSum2(self, candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero THEN  
                RETURN  
            END IF  
            FOR i FROM start TO length of candidates MINUS one  
                IF i GREATER THAN start AND candidates AT i EQUALS candidates AT i MINUS one THEN  
                    CONTINUE  
                END IF  
                CALL backtrack WITH arguments i PLUS one, target MINUS candidates AT i, path APPENDED WITH candidates AT i  
            END FOR  
        END FUNCTION  

        SORT candidates  
        SET result TO empty list  
        CALL backtrack WITH arguments zero, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
