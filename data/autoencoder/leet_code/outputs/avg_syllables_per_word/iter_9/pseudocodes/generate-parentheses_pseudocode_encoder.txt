CLASS Solution  
    FUNCTION generateParenthesis(self, n)  
        FUNCTION backtrack(s, left, right)  
            IF length of s EQUALS two MULTIPLIED BY n THEN  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n THEN  
                CALL backtrack WITH s CONCATENATED WITH left parenthesis, left INCREMENTED BY one, right  
            END IF  
            IF right LESS THAN left THEN  
                CALL backtrack WITH s CONCATENATED WITH right parenthesis, left, right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
