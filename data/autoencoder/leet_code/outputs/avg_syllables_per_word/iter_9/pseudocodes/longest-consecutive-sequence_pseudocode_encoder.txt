CLASS Solution  
    FUNCTION longestConsecutive(self, nums)  
        IF nums IS empty THEN  
            RETURN zero  
        END IF  
        
        SET num_set TO the collection of unique elements in nums  
        SET longest_streak TO zero  
        
        FOR each num IN num_set  
            IF num MINUS one NOT IN num_set THEN  
                SET current_num TO num  
                SET current_streak TO one  
                
                WHILE current_num PLUS one IN num_set DO  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                
                SET longest_streak TO the maximum of longest_streak AND current_streak  
            END IF  
        END FOR  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
