CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value equal to first element of values  
    SET i TO 1  
    CREATE empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        REMOVE first element FROM queue AND SET TO node  
        IF i less than length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with value values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i less than length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with value values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree called with p.left and q.left AND is_same_tree called with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest(self, root)  
        FUNCTION dfs(node)  
            IF node IS None THEN  
                RETURN node, zero  
            END IF  
            SET left_node, left_depth TO dfs called with node.left  
            SET right_node, right_depth TO dfs called with node.right  
            IF left_depth GREATER THAN right_depth THEN  
                RETURN left_node, left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth THEN  
                RETURN right_node, right_depth PLUS one  
            ELSE  
                RETURN node, left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN first element of dfs called with root  
    END FUNCTION  
END CLASS
