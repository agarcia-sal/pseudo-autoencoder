CLASS Solution  
    FUNCTION validTree(self, n, edges)  
        IF length of edges NOT EQUALS n MINUS 1 THEN  
            RETURN False  
        END IF  
          
        SET adj_list TO a new mapping from each integer from zero TO n MINUS 1 mapped TO an empty list  
        FOR each pair a, b IN edges  
            APPEND b TO adj_list at key a  
            APPEND a TO adj_list at key b  
        END FOR  
          
        SET visited TO an empty set  
          
        FUNCTION dfs(node)  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN adj_list at key node  
                CALL dfs(neighbor)  
            END FOR  
        END FUNCTION  
          
        CALL dfs with argument zero  
          
        RETURN length of visited EQUALS n  
    END FUNCTION  
END CLASS
