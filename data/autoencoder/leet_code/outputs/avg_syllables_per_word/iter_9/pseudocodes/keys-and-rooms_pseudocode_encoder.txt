CLASS Solution  
    FUNCTION canVisitAllRooms(self, rooms)  
        SET visited TO empty set  
        SET stack TO list containing zero  
        
        WHILE stack is not empty  
            SET current_room TO remove last element FROM stack  
            ADD current_room TO visited  
            FOR each key IN rooms at index current_room  
                IF key NOT IN visited THEN  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN size of visited EQUALS size of rooms  
    END FUNCTION  
END CLASS
