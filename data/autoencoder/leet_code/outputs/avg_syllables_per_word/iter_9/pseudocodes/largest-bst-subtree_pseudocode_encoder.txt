CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val EQUALS first element of values  
    SET i TO one  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with val EQUALS values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with val EQUALS values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node IS None THEN  
                RETURN floating point infinity, floating point negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO result of calling dfs with node.left  
            SET right_min, right_max, right_size TO result of calling dfs with node.right  
            IF left_max LESS THAN node.val AND node.val LESS THAN right_min THEN  
                RETURN minimum of node.val and left_min, maximum of node.val and right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN floating point negative infinity, floating point infinity, maximum of left_size and right_size  
        END FUNCTION  
        RETURN third element of result of calling dfs with root  
    END FUNCTION  
END CLASS
