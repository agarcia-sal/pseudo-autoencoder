CLASS Solution  
    FUNCTION minDistance(self, word1, word2)  
        SET m TO length of word1  
        SET n TO length of word2  
        
        FUNCTION create_2D_array(rows, columns)  
            CREATE array with rows number of elements  
            FOR each index FROM zero TO rows MINUS one  
                CREATE row list with columns number of zeros  
                SET element at index in array TO row list  
            END FOR  
            RETURN array  
        END FUNCTION  
        
        SET dp TO create_2D_array(m PLUS one, n PLUS one)  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one in word1 EQUALS character at position j MINUS one in word2 THEN  
                    SET value at dp position i and j TO value at dp position i MINUS one and j MINUS one PLUS one  
                ELSE  
                    IF value at dp position i MINUS one and j GREATER THAN value at dp position i and j MINUS one THEN  
                        SET value at dp position i and j TO value at dp position i MINUS one and j  
                    ELSE  
                        SET value at dp position i and j TO value at dp position i and j MINUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO value at dp position m and n  
        
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
