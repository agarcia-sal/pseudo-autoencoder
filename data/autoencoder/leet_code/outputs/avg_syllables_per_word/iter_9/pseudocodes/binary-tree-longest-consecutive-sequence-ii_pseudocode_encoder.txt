CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value equal to first element of values  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO remove first element from queue  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.left TO new TreeNode with value values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None THEN  
            SET node.right TO new TreeNode with value values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        FUNCTION dfs(node)  
            IF node IS None THEN  
                RETURN 0, 0  
            END IF  
            SET inc_len TO 1  
            SET dec_len TO 1  
            SET left_inc, left_dec TO dfs(node.left)  
            SET right_inc, right_dec TO dfs(node.right)  
            IF node.left IS NOT None THEN  
                IF node.val EQUALS node.left.val PLUS 1 THEN  
                    SET dec_len TO left_dec PLUS 1  
                ELSE IF node.val EQUALS node.left.val MINUS 1 THEN  
                    SET inc_len TO left_inc PLUS 1  
                END IF  
            END IF  
            IF node.right IS NOT None THEN  
                IF node.val EQUALS node.right.val PLUS 1 THEN  
                    SET dec_len TO the maximum of dec_len AND right_dec PLUS 1  
                ELSE IF node.val EQUALS node.right.val MINUS 1 THEN  
                    SET inc_len TO the maximum of inc_len AND right_inc PLUS 1  
                END IF  
            END IF  
            UPDATE max_len TO the maximum of max_len AND inc_len PLUS dec_len MINUS 1  
            RETURN inc_len, dec_len  
        END FUNCTION  
        SET max_len TO 0  
        CALL dfs with root  
        RETURN max_len  
    END FUNCTION  
END CLASS
