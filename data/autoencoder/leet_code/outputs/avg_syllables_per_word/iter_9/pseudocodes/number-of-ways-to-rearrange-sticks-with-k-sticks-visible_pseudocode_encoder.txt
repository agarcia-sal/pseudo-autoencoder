CLASS Solution  
    FUNCTION rearrangeSticks(self, n, k)  
        SET MOD TO ten to the power of nine PLUS seven  
        
        INITIALIZE dp AS a two-dimensional list with (n PLUS one) rows and (k PLUS one) columns all set to zero  
        
        SET dp at zero zero TO one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET dp at i j TO (dp at (i MINUS one) (j MINUS one) PLUS (i MINUS one) MULTIPLIED BY dp at (i MINUS one) j) MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN dp at n k  
    END FUNCTION  
END CLASS
