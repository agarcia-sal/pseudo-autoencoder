CLASS Solution  
    FUNCTION partitionLabels(self, s)  
        SET last_occurrence TO an empty dictionary  
        FOR each index and character IN s with index  
            SET last_occurrence at character TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        
        FOR each index and character IN s with index  
            SET end TO the greater of end and last_occurrence at character  
            IF index EQUALS end THEN  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
