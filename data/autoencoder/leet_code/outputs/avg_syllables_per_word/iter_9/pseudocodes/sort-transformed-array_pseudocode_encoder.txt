CLASS Solution  
    FUNCTION sortTransformedArray(self, nums, a, b, c)  
        FUNCTION quadratic(x)  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO length of nums  
        SET result TO list of size n filled with zero  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a GREATER THAN OR EQUAL TO zero THEN  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right DO  
                IF quadratic of nums at left GREATER THAN quadratic of nums at right THEN  
                    SET element at result at index TO quadratic of nums at left  
                    INCREMENT left BY one  
                ELSE  
                    SET element at result at index TO quadratic of nums at right  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right DO  
                IF quadratic of nums at left LESS THAN quadratic of nums at right THEN  
                    SET element at result at index TO quadratic of nums at left  
                    INCREMENT left BY one  
                ELSE  
                    SET element at result at index TO quadratic of nums at right  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
