CLASS Solution  
    FUNCTION countBattleships(self, board)  
        SET count TO zero  
        FOR i FROM zero TO length of board MINUS one  
            FOR j FROM zero TO length of first element of board MINUS one  
                IF element at board at position i and j EQUALS character X THEN  
                    IF (i EQUALS zero OR element at board at position i MINUS one and j EQUALS character dot) AND (j EQUALS zero OR element at board at position i and j MINUS one EQUALS character dot) THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
