CLASS Solution  
    FUNCTION parseTernary(self, expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS length of expression MINUS one OR character at expression position index PLUS one EQUALS colon THEN  
                RETURN character at expression position index, index PLUS two  
            END IF  
          
            SET true_part, next_index TO result of calling evaluate with index PLUS two  
            SET false_part, final_index TO result of calling evaluate with next_index  
          
            IF character at expression position index EQUALS 'T' THEN  
                RETURN true_part, final_index  
            ELSE  
                RETURN false_part, final_index  
            END IF  
        END FUNCTION  
      
        SET result, unused_variable TO result of calling evaluate with zero  
        RETURN result  
    END FUNCTION  
END CLASS
