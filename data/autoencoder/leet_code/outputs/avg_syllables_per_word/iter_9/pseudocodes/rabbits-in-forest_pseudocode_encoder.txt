CLASS Solution  
    FUNCTION numRabbits(self, answers)  
        SET answer_counts TO result of counting occurrences of each element in answers  
        SET total_rabbits TO zero  
        FOR each pair of answer and count IN answer_counts  
            SET groups TO integer result of (count PLUS answer) DIVIDED BY (answer PLUS one)  
            INCREMENT total_rabbits BY groups MULTIPLIED BY (answer PLUS one)  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
