CLASS Solution  
    FUNCTION findCircleNum(self, isConnected)  
        FUNCTION dfs(city)  
            FOR neighbor FROM zero TO n MINUS one  
                IF isConnected AT city AND neighbor EQUALS one AND visited AT neighbor EQUALS false THEN  
                    SET visited AT neighbor TO true  
                    CALL dfs WITH neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO length of isConnected  
        SET visited TO list of false repeated n times  
        SET province_count TO zero  

        FOR city FROM zero TO n MINUS one  
            IF visited AT city EQUALS false THEN  
                CALL dfs WITH city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  

        RETURN province_count  
    END FUNCTION  
END CLASS
