CLASS Solution  
    FUNCTION findNumberOfLIS(self, nums)  
        IF nums IS empty THEN  
            RETURN zero  
        END IF  
        
        SET n TO the length of nums  
        SET lengths TO a list of n elements each initialized to one  
        SET counts TO a list of n elements each initialized to one  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at index i in nums GREATER THAN element at index j in nums THEN  
                    IF element at index j in lengths PLUS one GREATER THAN element at index i in lengths THEN  
                        SET element at index i in lengths TO element at index j in lengths PLUS one  
                        SET element at index i in counts TO element at index j in counts  
                    ELSE IF element at index j in lengths PLUS one EQUALS element at index i in lengths THEN  
                        INCREMENT element at index i in counts BY element at index j in counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET longest TO the maximum element in lengths  
        RETURN the sum of elements in counts at indices where the corresponding element in lengths EQUALS longest  
    END FUNCTION  
END CLASS
