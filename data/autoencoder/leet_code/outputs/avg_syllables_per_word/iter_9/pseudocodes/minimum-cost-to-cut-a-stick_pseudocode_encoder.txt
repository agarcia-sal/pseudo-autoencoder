CLASS Solution  
    FUNCTION minCost(self, n, cuts)  
        SET cuts TO sorted list formed by appending zero and n to cuts  
        
        DEFINE FUNCTION dp(left, right)  
            IF right MINUS left LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO cuts AT right MINUS cuts AT left PLUS dp(left, i) PLUS dp(i, right)  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(zero, length of cuts MINUS one)  
    END FUNCTION  
END CLASS
