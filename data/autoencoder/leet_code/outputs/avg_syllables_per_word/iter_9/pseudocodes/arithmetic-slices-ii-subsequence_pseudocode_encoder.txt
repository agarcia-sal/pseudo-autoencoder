CLASS Solution  
    FUNCTION numberOfArithmeticSlices(self, nums)  
        SET n TO length of nums  
        IF n LESS THAN 3 THEN  
            RETURN 0  
        END IF  
        
        SET dp TO list of empty mappings from difference to count, one for each index from zero TO n minus one  
        SET total_count TO 0  
        
        FOR i FROM 0 TO n minus one  
            FOR j FROM 0 TO i minus one  
                SET d TO nums at i MINUS nums at j  
                
                IF dp at j CONTAINS key d THEN  
                    INCREMENT dp at i at key d BY dp at j at key d PLUS 1  
                    INCREMENT total_count BY dp at j at key d  
                ELSE  
                    INCREMENT dp at i at key d BY 1  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
