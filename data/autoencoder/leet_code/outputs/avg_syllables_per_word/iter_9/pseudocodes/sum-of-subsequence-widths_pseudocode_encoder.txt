CLASS Solution  
    FUNCTION sumSubseqWidths(self, nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SORT nums IN ascending order  
        SET n TO length of nums  
        SET total_sum TO zero  
        SET power TO list of length n with all elements set to one  
        
        FOR i FROM one TO n MINUS one  
            SET power at index i TO power at index i MINUS one MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS nums at index i MULTIPLIED BY (power at index i MINUS power at index n MINUS i MINUS one) MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
