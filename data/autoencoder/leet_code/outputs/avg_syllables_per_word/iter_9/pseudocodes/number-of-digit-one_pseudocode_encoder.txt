CLASS Solution  
    FUNCTION countDigitOne(self, n)  
        IF n LESS THAN OR EQUAL TO 0 THEN  
            RETURN 0  
        END IF  
        
        SET count TO 0  
        SET factor TO 1  
        
        WHILE factor LESS THAN OR EQUAL TO n DO  
            SET lower_nums TO n MINUS (integer division of n BY factor MULTIPLIED BY factor)  
            SET cur_num TO (integer division of n BY factor) MODULO 10  
            SET higher_nums TO integer division of n BY (factor MULTIPLIED BY 10)  
            
            IF cur_num EQUALS 0 THEN  
                INCREMENT count BY higher_nums MULTIPLIED BY factor  
            ELSE IF cur_num EQUALS 1 THEN  
                INCREMENT count BY higher_nums MULTIPLIED BY factor PLUS lower_nums PLUS 1  
            ELSE  
                INCREMENT count BY (higher_nums PLUS 1) MULTIPLIED BY factor  
            END IF  
            
            SET factor TO factor MULTIPLIED BY 10  
        END WHILE  
        
        RETURN count  
    END FUNCTION  
END CLASS
