CLASS Solution  
    FUNCTION dailyTemperatures(self, temperatures)  
        SET n TO length of temperatures  
        SET answer TO list of n elements each set to zero  
        SET stack TO empty list  

        FOR i FROM zero TO n MINUS one  
            WHILE stack is not empty AND temperature at index i GREATER THAN temperature at index at the last element in stack  
                SET prev_index TO remove last element FROM stack  
                SET element at prev_index in answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  

        RETURN answer  
    END FUNCTION  
END CLASS
