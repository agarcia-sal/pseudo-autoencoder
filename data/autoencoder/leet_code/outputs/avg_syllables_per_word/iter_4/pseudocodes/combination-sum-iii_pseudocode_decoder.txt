CLASS Solution
    FUNCTION combinationSum3(k, n)
        FUNCTION backtrack(start, path, target, k)
            IF target EQUALS 0 AND k EQUALS 0 THEN
                APPEND COPY OF path TO result
                RETURN
            END IF
            IF target LESS THAN 0 OR k LESS THAN 0 THEN
                RETURN
            END IF
            FOR i FROM start TO 9
                APPEND i TO path
                CALL backtrack(i + 1, path, target MINUS i, k MINUS 1)
                REMOVE LAST ELEMENT FROM path
            END FOR
        END FUNCTION

        SET result TO EMPTY LIST
        CALL backtrack(1, EMPTY LIST, n, k)
        RETURN result
    END FUNCTION
END CLASS
