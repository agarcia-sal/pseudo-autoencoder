CLASS Solution
    FUNCTION findMin(nums)
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        WHILE left IS LESS THAN right
            SET mid TO (left PLUS right) DIVIDED BY 2
            IF nums[mid] IS GREATER THAN nums[right]
                SET left TO mid PLUS 1
            ELSE IF nums[mid] IS LESS THAN nums[right]
                SET right TO mid
            ELSE
                DECREMENT right BY 1
        RETURN nums[left]
    END FUNCTION
END CLASS
