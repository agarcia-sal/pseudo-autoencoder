CLASS Solution
    FUNCTION findSubstringInWraproundString(s)
        SET max_length TO a mapping from each character 'a' to 'z' with value 0
        SET current_length TO 0
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            IF i GREATER THAN 0 AND (ASCII VALUE OF s[i] MINUS ASCII VALUE OF s[i - 1] EQUALS 1 OR (s[i] EQUALS 'a' AND s[i - 1] EQUALS 'z'))
                INCREMENT current_length BY 1
            ELSE
                SET current_length TO 1
            END IF
            SET max_length[s[i]] TO THE GREATER OF max_length[s[i]] AND current_length
        END FOR
        RETURN THE SUM OF ALL VALUES IN max_length
    END FUNCTION
END CLASS
