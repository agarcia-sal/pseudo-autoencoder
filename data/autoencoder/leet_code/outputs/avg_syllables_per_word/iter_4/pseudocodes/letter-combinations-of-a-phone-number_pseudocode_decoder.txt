CLASS Solution
    FUNCTION letterCombinations(digits)
        IF digits IS EMPTY
            RETURN empty list
        END IF

        SET phone_map TO dictionary mapping:
            '2' TO "abc",
            '3' TO "def",
            '4' TO "ghi",
            '5' TO "jkl",
            '6' TO "mno",
            '7' TO "pqrs",
            '8' TO "tuv",
            '9' TO "wxyz"

        SET output TO empty list

        FUNCTION backtrack(combination, next_digits)
            IF LENGTH OF next_digits IS 0
                APPEND combination TO output
            ELSE
                FOR EACH letter IN phone_map[next_digits[0]]
                    CALL backtrack(combination CONCATENATED WITH letter, SUBSTRING OF next_digits FROM 1 ONWARD)
                END FOR
            END IF
        END FUNCTION

        CALL backtrack WITH empty string AND digits
        RETURN output
    END FUNCTION
END CLASS
