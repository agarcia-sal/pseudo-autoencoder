CLASS Solution
    FUNCTION optimalDivision(nums)
        IF LENGTH OF nums EQUALS 1
            RETURN STRING REPRESENTATION OF nums[0]
        END IF
        IF LENGTH OF nums EQUALS 2
            RETURN STRING REPRESENTATION OF nums[0] CONCATENATED WITH "/" CONCATENATED WITH STRING REPRESENTATION OF nums[1]
        END IF
        RETURN STRING REPRESENTATION OF nums[0] CONCATENATED WITH "/(" CONCATENATED WITH JOIN OF "/" APPLIED TO STRING REPRESENTATIONS OF nums FROM INDEX 1 TO END CONCATENATED WITH ")"
    END FUNCTION
END CLASS
