CLASS Solution
    FUNCTION minDistance(word1, word2)
        SET m TO LENGTH OF word1
        SET n TO LENGTH OF word2
        SET dp TO 2D ARRAY OF SIZE (m + 1) BY (n + 1) FILLED WITH 0

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                IF CHARACTER AT word1 AT (i - 1) EQUALS CHARACTER AT word2 AT (j - 1)
                    SET dp[i][j] TO dp[i - 1][j - 1] + 1
                ELSE
                    SET dp[i][j] TO MAXIMUM OF dp[i - 1][j] AND dp[i][j - 1]
        
        SET lcs_length TO dp[m][n]
        RETURN (m - lcs_length) + (n - lcs_length)
    END FUNCTION
END CLASS
