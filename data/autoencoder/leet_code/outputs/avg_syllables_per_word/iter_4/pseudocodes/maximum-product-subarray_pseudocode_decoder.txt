CLASS Solution
    FUNCTION maxProduct(nums)
        IF nums IS empty
            RETURN 0
        END IF

        SET max_product TO first element of nums
        SET min_product TO first element of nums
        SET result TO first element of nums

        FOR each num IN nums starting from second element
            IF num IS less than 0
                SWAP max_product AND min_product
            END IF

            SET max_product TO maximum of num AND max_product multiplied BY num
            SET min_product TO minimum of num AND min_product multiplied BY num

            SET result TO maximum of result AND max_product
        END FOR

        RETURN result
    END FUNCTION
END CLASS
