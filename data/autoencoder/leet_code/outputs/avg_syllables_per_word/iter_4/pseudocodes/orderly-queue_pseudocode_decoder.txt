CLASS Solution
    FUNCTION orderlyQueue(s, k)
        IF k > 1 THEN
            RETURN sorted characters of s joined as a string
        ELSE
            SET n TO length of s
            SET smallest TO s
            FOR i FROM 1 TO n MINUS 1 DO
                SET rotated TO substring of s from i to end concatenated with substring of s from start to i
                IF rotated is lexicographically less than smallest THEN
                    SET smallest TO rotated
            RETURN smallest
    END FUNCTION
END CLASS
