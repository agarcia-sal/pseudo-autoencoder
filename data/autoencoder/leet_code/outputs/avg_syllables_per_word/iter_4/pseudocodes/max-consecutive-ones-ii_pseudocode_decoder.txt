CLASS Solution
    FUNCTION findMaxConsecutiveOnes(nums)
        SET max_count TO 0
        SET current_count TO 0
        SET zero_count TO 0
        SET left TO 0
        
        FOR right FROM 0 TO LENGTH OF nums MINUS 1
            IF nums[right] EQUALS 1 THEN
                INCREMENT current_count BY 1
            ELSE IF zero_count LESS THAN 1 THEN
                INCREMENT zero_count BY 1
                INCREMENT current_count BY 1
            ELSE
                SET max_count TO MAXIMUM OF max_count AND current_count
                WHILE zero_count EQUALS 1
                    IF nums[left] EQUALS 0 THEN
                        DECREMENT zero_count BY 1
                    DECREMENT current_count BY 1
                    INCREMENT left BY 1
                END WHILE
                INCREMENT current_count BY 1
            END IF
        END FOR
        
        SET max_count TO MAXIMUM OF max_count AND current_count
        RETURN max_count
    END FUNCTION
END CLASS
