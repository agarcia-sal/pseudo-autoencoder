CLASS Solution
    FUNCTION minimumLengthEncoding(words)
        SET reversed_words TO sorted list of each word reversed in words
        SET total_length TO 0
        FOR i FROM 0 TO LENGTH OF reversed_words MINUS 1
            IF i + 1 EQUALS LENGTH OF reversed_words OR reversed_words[i + 1] DOES NOT START WITH reversed_words[i]
                INCREMENT total_length BY LENGTH OF reversed_words[i] PLUS 1
        RETURN total_length
    END FUNCTION
END CLASS
