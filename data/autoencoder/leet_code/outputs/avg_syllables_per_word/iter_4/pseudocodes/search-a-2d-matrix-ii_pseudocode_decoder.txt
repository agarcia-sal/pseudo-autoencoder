CLASS Solution
    FUNCTION searchMatrix(matrix, target)
        IF matrix IS EMPTY OR matrix[0] IS EMPTY THEN
            RETURN False
        SET row TO 0
        SET col TO LENGTH OF matrix[0] MINUS 1
        WHILE row LESS THAN LENGTH OF matrix AND col GREATER THAN OR EQUAL TO 0
            IF matrix[row][col] EQUALS target THEN
                RETURN True
            ELSE IF matrix[row][col] GREATER THAN target THEN
                DECREMENT col BY 1
            ELSE
                INCREMENT row BY 1
        RETURN False
    END FUNCTION
END CLASS
