CLASS Solution
    FUNCTION parseBoolExpr(expression)
        SET stk TO empty list
        FOR each character c IN expression
            IF c IS IN 'tf!&|'
                APPEND c TO stk
            ELSE IF c IS ')'
                SET t TO 0
                SET f TO 0
                WHILE top element of stk IS IN 'tf'
                    IF top element of stk IS 't'
                        INCREMENT t BY 1
                    ELSE IF top element of stk IS 'f'
                        INCREMENT f BY 1
                    REMOVE top element FROM stk
                SET op TO REMOVE top element FROM stk
                IF op IS '!'
                    IF f IS NOT zero
                        SET c TO 't'
                    ELSE
                        SET c TO 'f'
                ELSE IF op IS '&'
                    IF f IS NOT zero
                        SET c TO 'f'
                    ELSE
                        SET c TO 't'
                ELSE IF op IS '|'
                    IF t IS NOT zero
                        SET c TO 't'
                    ELSE
                        SET c TO 'f'
                APPEND c TO stk
        RETURN top element of stk EQUALS 't'
    END FUNCTION
END CLASS
