CLASS Solution
    FUNCTION numberOfWays(corridor)
        SET MOD TO 1000000007
        SET seats TO list of indices WHERE character IN corridor IS 'S'
        IF LENGTH OF seats MODULO 2 IS NOT 0 OR LENGTH OF seats LESS THAN 2
            RETURN 0
        SET ways TO 1
        FOR i FROM 2 TO LENGTH OF seats STEP 2
            SET ways TO (ways TIMES (seats[i] MINUS seats[i - 1])) MODULO MOD
        RETURN ways
    END FUNCTION
END CLASS
