CLASS Solution
    FUNCTION minimumObstacles(grid)
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]
        SET directions TO [(0, 1), (1, 0), (0, -1), (-1, 0)]
        SET min_heap TO [(0, 0, 0)]
        SET visited TO SET CONTAINING (0, 0)
        
        WHILE min_heap IS NOT EMPTY
            EXTRACT (cost, x, y) FROM min_heap
            
            IF (x, y) EQUALS (m - 1, n - 1)
                RETURN cost
            
            FOR EACH (dx, dy) IN directions
                SET nx TO x + dx
                SET ny TO y + dy
                IF nx IS AT LEAST 0 AND LESS THAN m AND ny IS AT LEAST 0 AND LESS THAN n AND (nx, ny) NOT IN visited
                    ADD (nx, ny) TO visited
                    SET new_cost TO cost + grid[nx][ny]
                    INSERT (new_cost, nx, ny) INTO min_heap
        END WHILE
        
        RETURN -1
    END FUNCTION
END CLASS
