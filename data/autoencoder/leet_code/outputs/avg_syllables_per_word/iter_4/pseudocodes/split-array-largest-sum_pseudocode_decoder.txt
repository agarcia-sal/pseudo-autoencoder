CLASS Solution
    FUNCTION splitArray(nums, k)
        FUNCTION canSplit(max_sum)
            SET current_sum TO 0
            SET subarrays TO 1
            FOR each num IN nums
                IF current_sum + num > max_sum THEN
                    INCREMENT subarrays BY 1
                    SET current_sum TO num
                ELSE
                    INCREMENT current_sum BY num
            RETURN subarrays <= k
        END FUNCTION

        SET left TO maximum value in nums
        SET right TO sum of nums
        WHILE left < right DO
            SET mid TO (left + right) DIVIDED BY 2
            IF canSplit(mid) THEN
                SET right TO mid
            ELSE
                SET left TO mid + 1
        RETURN left
    END FUNCTION
END CLASS
