CLASS Solution
    FUNCTION findBlackPixel(picture: List of List of String, target: Integer) RETURNS Integer
        IF picture is empty OR first row of picture is empty THEN
            RETURN 0
        END IF

        SET m TO LENGTH OF picture
        SET n TO LENGTH OF picture[0]
        SET row_count TO list of zeros with length m
        SET col_count TO list of zeros with length n
        SET row_patterns TO empty map with default Integer value 0

        FOR r FROM 0 TO m - 1
            SET pattern TO concatenation of elements in picture[r]
            INCREMENT row_patterns[pattern] BY 1
            FOR c FROM 0 TO n - 1
                IF picture[r][c] EQUALS 'B' THEN
                    INCREMENT row_count[r] BY 1
                    INCREMENT col_count[c] BY 1
                END IF
            END FOR
        END FOR

        SET lonely_pixels TO 0

        FOR r FROM 0 TO m - 1
            FOR c FROM 0 TO n - 1
                IF picture[r][c] EQUALS 'B' THEN
                    SET pattern TO concatenation of elements in picture[r]
                    IF row_count[r] EQUALS target AND col_count[c] EQUALS target AND row_patterns[pattern] EQUALS target THEN
                        INCREMENT lonely_pixels BY 1
                    END IF
                END IF
            END FOR
        END FOR

        RETURN lonely_pixels
    END FUNCTION
END CLASS
