CLASS Solution
    FUNCTION slidingPuzzle(board: List of List of int) RETURNS int
        SET start TO tuple of tuples from board rows
        SET target TO ((1, 2, 3), (4, 5, 0))
        
        SET neighbors TO dictionary:
            0 maps to set {1, 3}
            1 maps to set {0, 2, 4}
            2 maps to set {1, 5}
            3 maps to set {0, 4}
            4 maps to set {1, 3, 5}
            5 maps to set {2, 4}
        
        FOR i FROM 0 TO 1
            FOR j FROM 0 TO 2
                IF start at position (i, j) EQUALS 0
                    SET zero_pos TO i TIMES 3 PLUS j
        
        INITIALIZE queue AS deque containing tuple (start, zero_pos, 0)
        INITIALIZE visited AS set containing start
        
        WHILE queue is not empty
            SET state, zero_pos, moves TO dequeue from queue
            
            IF state EQUALS target
                RETURN moves
            
            FOR each neighbor IN neighbors at zero_pos
                SET new_state TO list of lists converted from state
                SET i, j TO zero_pos DIVIDED BY 3, zero_pos MODULO 3
                SET ni, nj TO neighbor DIVIDED BY 3, neighbor MODULO 3
                
                SWAP new_state[i][j] WITH new_state[ni][nj]
                SET new_tuple_state TO tuple of tuples created from new_state
                
                IF new_tuple_state NOT IN visited
                    ADD new_tuple_state TO visited
                    ENQUEUE (new_tuple_state, neighbor, moves PLUS 1) INTO queue
        
        RETURN -1
    END FUNCTION
END CLASS
