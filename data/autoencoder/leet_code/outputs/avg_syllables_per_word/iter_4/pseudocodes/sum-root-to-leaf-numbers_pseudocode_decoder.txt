CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION sumNumbers(root: Optional[TreeNode]) RETURNS int
        FUNCTION dfs(node, current_number)
            IF node IS None
                RETURN 0
            SET current_number TO current_number MULTIPLIED BY 10 PLUS node.val
            IF node.left IS None AND node.right IS None
                RETURN current_number
            SET left_sum TO dfs(node.left, current_number)
            SET right_sum TO dfs(node.right, current_number)
            RETURN left_sum PLUS right_sum
        END FUNCTION
        RETURN dfs(root, 0)
    END FUNCTION
END CLASS
