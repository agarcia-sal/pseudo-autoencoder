CLASS Solution
    FUNCTION killProcess(pid: List of integer, ppid: List of integer, kill: integer) RETURNS List of integer
        SET tree TO a dictionary with default value as empty list
        FOR each pair of child and parent FROM pid and ppid
            APPEND child TO tree[parent]
        END FOR

        SET queue TO list containing kill
        SET killed_processes TO empty list

        WHILE queue is not empty
            SET current TO first element removed from queue
            APPEND current TO killed_processes
            EXTEND queue BY tree[current]
        END WHILE

        RETURN killed_processes
    END FUNCTION
END CLASS
