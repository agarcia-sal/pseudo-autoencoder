CLASS Solution
    FUNCTION findShortestWay(maze, ball, hole)
        SET directions TO [(0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')]
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze[0]
        SET start TO T-UPLE OF ball
        SET hole TO T-UPLE OF hole
        SET pq TO LIST CONTAINING (0, "", start)
        SET visited TO EMPTY SET
        
        WHILE pq IS NOT EMPTY
            SET dist, path, (x, y) TO HEAP-POP FROM pq
            
            IF (x, y) EQUALS hole
                RETURN path
            
            IF (x, y) IN visited
                CONTINUE TO NEXT ITERATION
            
            ADD (x, y) TO visited
            
            FOR dx, dy, direction IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO 0
                
                WHILE new_x + dx IS WITHIN 0 AND m-1 AND new_y + dy IS WITHIN 0 AND n-1 AND maze[new_x + dx][new_y + dy] EQUALS 0
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY 1
                    IF (new_x, new_y) EQUALS hole
                        BREAK
                
                IF (new_x, new_y) NOT IN visited
                    HEAP-PUSH (dist + count, path + direction, (new_x, new_y)) TO pq
        
        RETURN "impossible"
    END FUNCTION
END CLASS
