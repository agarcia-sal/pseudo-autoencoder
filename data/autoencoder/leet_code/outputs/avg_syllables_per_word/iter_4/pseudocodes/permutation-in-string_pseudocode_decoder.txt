CLASS Solution
    FUNCTION checkInclusion(s1, s2)
        SET len_s1 TO LENGTH OF s1
        SET len_s2 TO LENGTH OF s2

        IF len_s1 > len_s2 THEN
            RETURN False
        END IF

        SET count_s1 TO FREQUENCY COUNTER OF CHARACTERS IN s1
        SET count_s2 TO FREQUENCY COUNTER OF CHARACTERS IN FIRST len_s1 CHARACTERS OF s2

        IF count_s1 EQUALS count_s2 THEN
            RETURN True
        END IF

        FOR i FROM len_s1 TO len_s2 - 1
            INCREMENT count_s2[s2[i]] BY 1
            DECREMENT count_s2[s2[i - len_s1]] BY 1

            IF count_s2[s2[i - len_s1]] IS 0 THEN
                REMOVE s2[i - len_s1] FROM count_s2
            END IF

            IF count_s1 EQUALS count_s2 THEN
                RETURN True
            END IF
        END FOR

        RETURN False
    END FUNCTION
END CLASS
