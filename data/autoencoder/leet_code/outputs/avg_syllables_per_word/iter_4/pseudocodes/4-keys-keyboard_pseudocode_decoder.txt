CLASS Solution
    FUNCTION maxA(n)
        IF n <= 3 THEN
            RETURN n
        END IF

        SET dp TO list of zeros with length n + 1

        FOR i FROM 1 TO 3
            SET dp[i] TO i
        END FOR

        FOR i FROM 4 TO n
            FOR j FROM i - 3 DOWNTO 1
                SET dp[i] TO maximum of dp[i] AND dp[j] MULTIPLIED BY (i - j - 1)
            END FOR
        END FOR

        RETURN dp[n]
    END FUNCTION
END CLASS
