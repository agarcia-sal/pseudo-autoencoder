CLASS Solution
    FUNCTION replaceNonCoprimes(nums)
        SET stack TO empty list
        FOR each num IN nums
            WHILE stack is not empty
                SET g TO gcd of last element in stack and num
                IF g EQUALS 1
                    BREAK loop
                SET num TO (last element in stack MULTIPLIED BY num) DIVIDED BY g
                REMOVE last element from stack
            APPEND num TO stack
        RETURN stack
    END FUNCTION
END CLASS
