CLASS Solution
    FUNCTION minimumOneBitOperations(n)
        IF n EQUALS 0 THEN
            RETURN 0
        END IF

        SET k TO 0
        WHILE (1 LEFT-SHIFTED BY (k + 1)) LESS THAN OR EQUAL TO n DO
            INCREMENT k BY 1
        END WHILE

        SET most_significant_bit TO 1 LEFT-SHIFTED BY k

        RETURN (1 LEFT-SHIFTED BY (k + 1)) MINUS 1 MINUS minimumOneBitOperations(n XOR most_significant_bit)
    END FUNCTION
END CLASS
