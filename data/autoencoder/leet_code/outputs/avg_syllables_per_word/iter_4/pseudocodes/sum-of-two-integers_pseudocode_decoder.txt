CLASS Solution
    FUNCTION getSum(a, b)
        SET MAX TO hexadecimal 7FFFFFFF
        SET mask TO hexadecimal FFFFFFFF
        
        WHILE b NOT EQUAL TO 0
            SET carry TO (a AND b) SHIFTED LEFT BY 1
            SET a TO (a XOR b) AND mask
            SET b TO carry AND mask
        
        IF a LESS THAN OR EQUAL TO MAX
            RETURN a
        ELSE
            RETURN BITWISE NOT OF (a XOR mask)
    END FUNCTION
END CLASS
