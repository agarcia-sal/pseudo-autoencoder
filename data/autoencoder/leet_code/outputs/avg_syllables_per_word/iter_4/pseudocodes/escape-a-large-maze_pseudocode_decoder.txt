CLASS Solution
    FUNCTION isEscapePossible(blocked, source, target)
        IF blocked IS EMPTY
            RETURN True
        
        SET blocked_set TO SET OF TUPLES FROM blocked
        SET n TO LENGTH OF blocked
        SET max_steps TO n MULTIPLIED BY (n MINUS 1) DIVIDED BY 2
        
        FUNCTION bfs(start, end)
            SET queue TO DEQUE CONTAINING TUPLE OF start
            SET visited TO SET CONTAINING ELEMENTS OF queue
            
            WHILE queue IS NOT EMPTY
                SET x, y TO LEFTMOST ELEMENT OF queue AND REMOVE IT
                
                IF LENGTH OF visited IS GREATER THAN max_steps OR (x EQUALS end[0] AND y EQUALS end[1])
                    RETURN True
                
                FOR dx, dy IN [(0, 1), (0, -1), (1, 0), (-1, 0)]
                    SET nx TO x PLUS dx
                    SET ny TO y PLUS dy
                    IF nx IS AT LEAST 0 AND LESS THAN 10^6 AND ny IS AT LEAST 0 AND LESS THAN 10^6 AND (nx, ny) NOT IN blocked_set AND (nx, ny) NOT IN visited
                        ADD (nx, ny) TO visited
                        APPEND (nx, ny) TO queue
            
            RETURN False
        
        RETURN bfs(source, target) AND bfs(target, source)
    END FUNCTION
END CLASS
