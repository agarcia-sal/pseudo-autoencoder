CLASS Solution
    FUNCTION nextClosestTime(time)
        SET digits TO set of characters in time with ":" removed
        SPLIT time by ":" INTO hour_string, minute_string
        SET hour TO integer value of hour_string
        SET minute TO integer value of minute_string

        FUNCTION generate_times(digits)
            SET times TO empty list
            FOR each h1 IN digits
                FOR each h2 IN digits
                    FOR each m1 IN digits
                        FOR each m2 IN digits
                            SET hour_val TO integer formed by concatenating h1 and h2
                            SET minute_val TO integer formed by concatenating m1 and m2
                            IF hour_val less than 24 AND minute_val less than 60
                                APPEND tuple (hour_val, minute_val) TO times
            RETURN times sorted in ascending order
        END FUNCTION

        SET all_times TO generate_times(digits)
        SET current_time_index TO index of tuple (hour, minute) IN all_times

        IF current_time_index plus 1 less than length of all_times
            SET next_hour, next_minute TO element at current_time_index plus 1 in all_times
        ELSE
            SET next_hour, next_minute TO first element in all_times

        RETURN string formatted as two-digit next_hour, colon, two-digit next_minute
    END FUNCTION
END CLASS
