CLASS Solution
    FUNCTION minTotalDistance(grid)
        SET rows TO empty list
        SET cols TO empty list
        
        FOR i FROM 0 TO LENGTH OF grid MINUS 1
            FOR j FROM 0 TO LENGTH OF grid AT 0 MINUS 1
                IF grid AT i AT j EQUALS 1
                    APPEND i TO rows
                    APPEND j TO cols
        
        SORT rows
        SORT cols
        
        SET median_row TO rows AT (LENGTH OF rows DIVIDED BY 2)
        SET median_col TO cols AT (LENGTH OF cols DIVIDED BY 2)
        
        SET min_distance TO 0
        FOR each row IN rows
            INCREMENT min_distance BY ABSOLUTE VALUE OF (row MINUS median_row)
        FOR each col IN cols
            INCREMENT min_distance BY ABSOLUTE VALUE OF (col MINUS median_col)
        
        RETURN min_distance
    END FUNCTION
END CLASS
