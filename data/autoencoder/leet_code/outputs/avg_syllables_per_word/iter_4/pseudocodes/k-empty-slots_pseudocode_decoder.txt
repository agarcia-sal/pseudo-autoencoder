CLASS Solution
    FUNCTION kEmptySlots(bulbs, k)
        SET bloom_days TO empty list
        FOR day FROM 1 TO LENGTH OF bulbs
            SET position TO bulbs AT INDEX day MINUS 1 PLUS 1
            SET i TO POSITION TO INSERT position IN bloom_days TO MAINTAIN SORT ORDER
            IF i GREATER THAN 0 AND position MINUS bloom_days AT i MINUS 1 MINUS 1 EQUALS k
                RETURN day
            IF i LESS THAN LENGTH OF bloom_days AND bloom_days AT i MINUS position MINUS 1 EQUALS k
                RETURN day
            INSERT position INTO bloom_days AT INDEX i
        RETURN -1
    END FUNCTION
END CLASS
