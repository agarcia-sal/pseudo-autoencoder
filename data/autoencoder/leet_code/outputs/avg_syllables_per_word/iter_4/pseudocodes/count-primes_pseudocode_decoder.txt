CLASS Solution
    FUNCTION countPrimes(n)
        IF n LESS THAN OR EQUAL TO 2
            RETURN 0
        END IF

        SET is_prime TO list of True repeated n times
        SET is_prime[0] TO False
        SET is_prime[1] TO False

        FOR start FROM 2 TO INTEGER PART OF square root of n INCLUSIVE
            IF is_prime[start] IS True
                FOR multiple FROM start TIMES start TO n EXCLUSIVE STEP start
                    SET is_prime[multiple] TO False
                END FOR
            END IF
        END FOR

        SET prime_count TO number of True elements in is_prime
        RETURN prime_count
    END FUNCTION
END CLASS
