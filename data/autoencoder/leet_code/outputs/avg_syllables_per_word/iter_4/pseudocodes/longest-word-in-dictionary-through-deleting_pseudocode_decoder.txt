CLASS Solution
    FUNCTION findLongestWord(s, dictionary)
        FUNCTION is_subsequence(word, s)
            SET iterator TO ITERATOR OF s
            RETURN ALL characters IN word ARE IN iterator IN ORDER
        END FUNCTION

        SORT dictionary BY descending LENGTH AND ascending LEXICOGRAPHICAL ORDER

        FOR each word IN dictionary
            IF is_subsequence(word, s) IS TRUE
                RETURN word
        RETURN empty string
    END FUNCTION
END CLASS
