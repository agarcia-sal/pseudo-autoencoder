CLASS Solution
    FUNCTION numberOfArrays(s, k)
        SET MOD TO 10 POWER 9 PLUS 7
        SET n TO LENGTH OF s
        
        INITIALIZE dp AS LIST OF SIZE n PLUS 1 FILLED WITH 0
        SET dp[0] TO 1
        
        FOR i FROM 1 TO n
            FOR j FROM 0 TO i MINUS 1
                IF s[j] IS NOT '0' AND INTEGER VALUE OF s FROM j TO i MINUS 1 IS AT LEAST 1 AND AT MOST k
                    SET dp[i] TO (dp[i] PLUS dp[j]) MODULO MOD
        
        RETURN dp[n]
    END FUNCTION
END CLASS
