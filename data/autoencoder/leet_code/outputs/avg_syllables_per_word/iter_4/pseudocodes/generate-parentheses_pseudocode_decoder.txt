CLASS Solution
    FUNCTION generateParenthesis(n)
        FUNCTION backtrack(s = empty string, left = 0, right = 0)
            IF LENGTH OF s EQUALS 2 TIMES n THEN
                APPEND s TO result
                RETURN
            END IF
            IF left LESS THAN n THEN
                CALL backtrack WITH (s CONCATENATED WITH '(', left INCREMENTED BY 1, right)
            END IF
            IF right LESS THAN left THEN
                CALL backtrack WITH (s CONCATENATED WITH ')', left, right INCREMENTED BY 1)
            END IF
        END FUNCTION

        SET result TO empty list
        CALL backtrack()
        RETURN result
    END FUNCTION
END CLASS
