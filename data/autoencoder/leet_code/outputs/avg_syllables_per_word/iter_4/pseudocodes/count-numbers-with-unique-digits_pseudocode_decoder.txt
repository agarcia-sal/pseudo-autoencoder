CLASS Solution
    FUNCTION countNumbersWithUniqueDigits(n)
        IF n EQUALS 0 THEN
            RETURN 1
        IF n EQUALS 1 THEN
            RETURN 10
        SET count TO 10
        SET available_digits TO 9
        SET choices TO 9
        FOR i FROM 2 TO n INCLUSIVE DO
            SET choices TO choices MULTIPLIED BY available_digits
            SET count TO count PLUS choices
            DECREMENT available_digits BY 1
        RETURN count
    END FUNCTION
END CLASS
