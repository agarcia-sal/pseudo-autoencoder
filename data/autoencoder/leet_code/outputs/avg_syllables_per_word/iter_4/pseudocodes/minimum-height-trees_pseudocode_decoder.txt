CLASS Solution
    FUNCTION findMinHeightTrees(n, edges)
        IF n EQUALS 1 THEN
            RETURN [0]
        END IF

        SET graph TO a map with default empty list
        SET degree TO a list of zeros with length n

        FOR each pair (u, v) IN edges
            APPEND v TO graph[u]
            APPEND u TO graph[v]
            INCREMENT degree[u] BY 1
            INCREMENT degree[v] BY 1
        END FOR

        SET leaves TO a queue containing all indices i FROM 0 TO n-1 WHERE degree[i] EQUALS 1

        SET remaining_nodes TO n
        WHILE remaining_nodes GREATER THAN 2
            SET leaves_count TO length of leaves
            DECREMENT remaining_nodes BY leaves_count

            FOR each _ FROM 1 TO leaves_count
                SET leaf TO REMOVE first element from leaves
                FOR each neighbor IN graph[leaf]
                    DECREMENT degree[neighbor] BY 1
                    IF degree[neighbor] EQUALS 1 THEN
                        ADD neighbor TO leaves
                    END IF
                END FOR
            END FOR
        END WHILE

        RETURN list converted from leaves
    END FUNCTION
END CLASS
