CLASS Solution
    FUNCTION minDays(n)
        DEFINE FUNCTION dp(remaining) WITH CACHE
            IF remaining <= 1
                RETURN remaining
            END IF
            SET option1 TO remaining MODULO 2 PLUS dp(remaining DIVIDED BY 2)
            SET option2 TO remaining MODULO 3 PLUS dp(remaining DIVIDED BY 3)
            RETURN 1 PLUS MINIMUM OF option1 AND option2
        END FUNCTION
        RETURN dp(n)
    END FUNCTION
END CLASS
