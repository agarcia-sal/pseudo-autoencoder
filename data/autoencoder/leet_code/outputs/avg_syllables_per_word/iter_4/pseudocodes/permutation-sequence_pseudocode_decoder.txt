CLASS Solution
    FUNCTION getPermutation(n, k)
        SET numbers TO list of integers from 1 to n
        DECREMENT k BY 1
        SET result TO empty list
        FOR i FROM n DOWNTO 1
            SET fact TO factorial of (i - 1)
            SET index TO k DIVIDED BY fact
            APPEND string representation of numbers at index TO result
            REMOVE element at index FROM numbers
            SET k TO k MODULO fact
        RETURN concatenation of elements in result
    END FUNCTION
END CLASS
