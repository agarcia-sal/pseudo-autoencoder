CLASS Solution
    FUNCTION evalRPN(tokens)
        SET stack TO empty list
        FOR each token IN tokens
            IF token IS one of "+", "-", "*", "/"
                SET b TO POP from stack
                SET a TO POP from stack
                IF token IS "+"
                    APPEND (a plus b) TO stack
                ELSE IF token IS "-"
                    APPEND (a minus b) TO stack
                ELSE IF token IS "*"
                    APPEND (a times b) TO stack
                ELSE IF token IS "/"
                    APPEND (integer division of a by b, truncating towards zero) TO stack
            ELSE
                APPEND (integer value of token) TO stack
        RETURN first element of stack
    END FUNCTION
END CLASS
