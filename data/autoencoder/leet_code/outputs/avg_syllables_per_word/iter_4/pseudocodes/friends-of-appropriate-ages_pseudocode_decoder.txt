CLASS Solution
    FUNCTION numFriendRequests(ages)
        FUNCTION request_will_be_sent(age_x, age_y)
            RETURN NOT (age_y <= 0.5 * age_x + 7 OR age_y > age_x OR (age_y > 100 AND age_x < 100))
        END FUNCTION

        SET count TO list of 121 zeros
        FOR each age IN ages
            INCREMENT count[age] BY 1

        SET total_requests TO 0
        FOR age_x FROM 1 TO 120
            FOR age_y FROM 1 TO 120
                IF request_will_be_sent(age_x, age_y)
                    INCREMENT total_requests BY count[age_x] * (count[age_y] - IF age_x EQUALS age_y THEN 1 ELSE 0)
        RETURN total_requests
    END FUNCTION
END CLASS
