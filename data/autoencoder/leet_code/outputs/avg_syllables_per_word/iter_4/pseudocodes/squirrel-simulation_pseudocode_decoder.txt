CLASS Solution
    FUNCTION minDistance(height, width, tree, squirrel, nuts)
        FUNCTION distance(p1, p2)
            RETURN absolute value of (p1[0] MINUS p2[0]) PLUS absolute value of (p1[1] MINUS p2[1])
        END FUNCTION

        SET total_distance TO 0
        SET max_save TO negative infinity

        FOR each nut IN nuts
            INCREMENT total_distance BY 2 TIMES distance between nut and tree
            SET first_nut_distance TO distance between squirrel and nut PLUS distance between nut and tree
            SET later_nut_distance TO 2 TIMES distance between nut and tree
            SET save TO later_nut_distance MINUS first_nut_distance
            IF save GREATER THAN max_save
                SET max_save TO save
            END IF
        END FOR

        RETURN total_distance MINUS max_save
    END FUNCTION
END CLASS
