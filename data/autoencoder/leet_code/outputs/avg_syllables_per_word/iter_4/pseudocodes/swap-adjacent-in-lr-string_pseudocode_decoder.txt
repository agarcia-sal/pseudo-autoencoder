CLASS Solution
    FUNCTION canTransform(start, end)
        SET start_filtered TO EMPTY LIST
        FOR EACH index, character IN start WITH INDICES
            IF character IS NOT 'X'
                APPEND (character, index) TO start_filtered
        SET end_filtered TO EMPTY LIST
        FOR EACH index, character IN end WITH INDICES
            IF character IS NOT 'X'
                APPEND (character, index) TO end_filtered
        IF LENGTH OF start_filtered IS NOT EQUAL TO LENGTH OF end_filtered
            RETURN False
        FOR EACH (char1, index1), (char2, index2) IN PAIRS FROM start_filtered AND end_filtered
            IF char1 IS NOT EQUAL TO char2
                RETURN False
            IF char1 IS 'L' AND index1 IS LESS THAN index2
                RETURN False
            IF char1 IS 'R' AND index1 IS GREATER THAN index2
                RETURN False
        RETURN True
    END FUNCTION
END CLASS
