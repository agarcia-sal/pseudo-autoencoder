CLASS Solution
    FUNCTION splitIntoFibonacci(num)
        FUNCTION backtrack(start, path)
            IF start EQUALS LENGTH OF num AND LENGTH OF path >= 3 THEN
                RETURN path
            END IF
            FOR end FROM start + 1 TO LENGTH OF num + 1 DO
                IF num[start] EQUALS '0' AND end > start + 1 THEN
                    CONTINUE TO NEXT ITERATION
                END IF
                SET next_num TO CONVERT substring of num FROM start TO end TO INTEGER
                IF next_num >= 2 TO THE POWER OF 31 THEN
                    BREAK LOOP
                END IF
                IF LENGTH OF path < 2 OR next_num EQUALS path[-1] + path[-2] THEN
                    SET result TO backtrack(end, path APPENDED WITH next_num)
                    IF result IS NOT EMPTY THEN
                        RETURN result
                    END IF
                END IF
            END FOR
            RETURN empty list
        END FUNCTION
        RETURN backtrack(0, empty list)
    END FUNCTION
END CLASS
