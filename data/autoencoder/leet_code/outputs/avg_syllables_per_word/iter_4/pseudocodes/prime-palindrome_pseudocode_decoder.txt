CLASS Solution
    FUNCTION primePalindrome(n)
        FUNCTION is_prime(num)
            IF num < 2 THEN
                RETURN False
            IF num EQUALS 2 OR num EQUALS 3 THEN
                RETURN True
            IF num MODULO 2 EQUALS 0 OR num MODULO 3 EQUALS 0 THEN
                RETURN False
            SET i TO 5
            WHILE i * i LESS THAN OR EQUAL TO num DO
                IF num MODULO i EQUALS 0 OR num MODULO (i + 2) EQUALS 0 THEN
                    RETURN False
                INCREMENT i BY 6
            RETURN True
        END FUNCTION

        FUNCTION is_palindrome(num)
            RETURN string representation of num EQUALS reverse of string representation of num
        END FUNCTION

        IF n LESS THAN OR EQUAL TO 2 THEN
            RETURN 2

        IF n GREATER THAN OR EQUAL TO 10 AND n LESS THAN OR EQUAL TO 11 THEN
            RETURN 11

        WHILE True DO
            IF n MODULO 2 EQUALS 0 THEN
                INCREMENT n BY 1
                CONTINUE
            IF is_palindrome(n) AND is_prime(n) THEN
                RETURN n
            IF LENGTH OF string representation of n MODULO 2 EQUALS 0 THEN
                SET n TO 10 RAISED TO LENGTH OF string representation of n PLUS 1
            ELSE
                INCREMENT n BY 2
        END WHILE
    END FUNCTION
END CLASS
