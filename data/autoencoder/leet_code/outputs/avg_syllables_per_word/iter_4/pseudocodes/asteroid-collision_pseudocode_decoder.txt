CLASS Solution
    FUNCTION asteroidCollision(asteroids)
        SET stack TO empty list
        FOR each asteroid IN asteroids
            WHILE stack IS NOT empty AND asteroid IS less than 0 AND last element of stack IS greater than 0
                IF last element of stack IS less than NEGATIVE of asteroid
                    REMOVE last element from stack
                    CONTINUE
                ELSE IF last element of stack IS equal to NEGATIVE of asteroid
                    REMOVE last element from stack
                END IF
                BREAK
            END WHILE
            IF loop did not BREAK
                APPEND asteroid TO stack
        RETURN stack
    END FUNCTION
END CLASS
