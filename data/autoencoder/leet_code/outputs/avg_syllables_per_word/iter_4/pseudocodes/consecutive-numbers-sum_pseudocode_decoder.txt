CLASS Solution
    FUNCTION consecutiveNumbersSum(n)
        SET count TO 0
        FOR k FROM 1 TO n
            IF (2 * n) MODULO k EQUALS 0
                SET m TO ((2 * n DIVIDED BY k) - k + 1) DIVIDED BY 2
                IF m GREATER THAN 0 AND m IS INTEGER
                    INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
