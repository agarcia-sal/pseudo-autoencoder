CLASS Solution
    FUNCTION decodeString(s)
        SET stack TO empty list
        SET a TO empty string
        SET d TO empty string

        FOR each char IN s
            IF char IS digit THEN
                APPEND char TO d
            ELSE IF char IS alphabetic THEN
                APPEND char TO a
            ELSE IF char IS '[' THEN
                APPEND tuple (a, CONVERT d TO integer) TO stack
                SET a TO empty string
                SET d TO empty string
            ELSE IF char IS ']' THEN
                SET p, n TO POP last element FROM stack
                SET a TO p CONCATENATED WITH a REPEATED n times
        RETURN a
    END FUNCTION
END CLASS
