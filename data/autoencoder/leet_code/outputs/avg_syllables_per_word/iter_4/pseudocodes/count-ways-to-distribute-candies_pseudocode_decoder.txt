CLASS Solution
    FUNCTION waysToDistribute(n, k)
        SET MOD TO 1000000007
        SET dp TO 2D list with dimensions (n + 1) by (k + 1) filled with 0
        SET dp[0][0] TO 1
        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                SET dp[i][j] TO (j MULTIPLIED BY dp[i - 1][j] PLUS dp[i - 1][j - 1]) MODULO MOD
        RETURN dp[n][k]
    END FUNCTION
END CLASS
