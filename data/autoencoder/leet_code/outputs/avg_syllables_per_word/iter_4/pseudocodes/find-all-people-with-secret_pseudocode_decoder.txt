CLASS Solution
    FUNCTION findAllPeople(n, meetings, firstPerson)
        SET know_secret TO SET containing 0 and firstPerson
        SORT meetings BY the third element (time) in ascending order
        SET i TO 0
        WHILE i LESS THAN LENGTH OF meetings
            SET current_time TO the third element of meetings[i]
            INITIALIZE graph AS empty mapping from person TO list of connected persons
            WHILE i LESS THAN LENGTH OF meetings AND third element of meetings[i] EQUALS current_time
                SET x TO first element of meetings[i]
                SET y TO second element of meetings[i]
                APPEND y TO graph[x]
                APPEND x TO graph[y]
                INCREMENT i BY 1
            INITIALIZE queue WITH persons IN know_secret THAT ARE KEYS IN graph
            SET visited TO empty set
            WHILE queue IS NOT empty
                REMOVE first person FROM queue AS person
                IF person IS IN visited
                    CONTINUE TO next iteration
                ADD person TO visited
                ADD person TO know_secret
                FOR each neighbor IN graph[person]
                    IF neighbor IS NOT IN visited
                        APPEND neighbor TO queue
        RETURN LIST of know_secret
    END FUNCTION
END CLASS
