CLASS Solution
    FUNCTION numRabbits(answers)
        SET answer_counts TO count occurrences of each element in answers
        SET total_rabbits TO 0
        FOR each answer, count IN answer_counts
            SET groups TO (count + answer) DIVIDED BY (answer + 1)
            INCREMENT total_rabbits BY groups TIMES (answer + 1)
        RETURN total_rabbits
    END FUNCTION
END CLASS
