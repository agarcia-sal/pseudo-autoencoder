CLASS Solution
    FUNCTION isRectangleCover(rectangles)
        SET X1 TO positive infinity
        SET Y1 TO positive infinity
        SET X2 TO negative infinity
        SET Y2 TO negative infinity
        
        SET points TO empty set
        SET area TO 0
        
        FOR each rectangle IN rectangles
            SET x1, y1, x2, y2 TO values from rectangle
            
            SET X1 TO minimum of X1 and x1
            SET Y1 TO minimum of Y1 and y1
            SET X2 TO maximum of X2 and x2
            SET Y2 TO maximum of Y2 and y2
            
            INCREMENT area BY (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)
            
            FOR each point p IN list of (x1,y1), (x1,y2), (x2,y1), (x2,y2)
                IF p IS IN points
                    REMOVE p FROM points
                ELSE
                    ADD p TO points
        
        IF area IS NOT EQUAL TO (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1)
            RETURN False
        
        IF points IS NOT EQUAL TO set of (X1,Y1), (X1,Y2), (X2,Y1), (X2,Y2)
            RETURN False
        
        RETURN True
    END FUNCTION
END CLASS
