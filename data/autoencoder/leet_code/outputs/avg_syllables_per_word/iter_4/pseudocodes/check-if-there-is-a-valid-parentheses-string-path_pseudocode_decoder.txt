CLASS Solution
    FUNCTION hasValidPath(grid)
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]
        
        IF (m + n - 1) MODULO 2 IS NOT 0
            RETURN False
        
        SET dp TO 3D LIST OF False WITH DIMENSIONS (m + 1) x (n + 1) x (m + n + 1)
        
        SET dp[0][1][0] TO True
        
        FOR i FROM 0 TO m - 1
            FOR j FROM 0 TO n - 1
                FOR k FROM 0 TO m + n
                    IF dp[i][j + 1][k] IS True OR dp[i + 1][j][k] IS True
                        IF grid[i][j] IS '(' AND k + 1 < m + n + 1
                            SET dp[i + 1][j + 1][k + 1] TO True
                        ELSE IF grid[i][j] IS ')' AND k - 1 >= 0
                            SET dp[i + 1][j + 1][k - 1] TO True
        
        RETURN dp[m][n][0]
    END FUNCTION
END CLASS
