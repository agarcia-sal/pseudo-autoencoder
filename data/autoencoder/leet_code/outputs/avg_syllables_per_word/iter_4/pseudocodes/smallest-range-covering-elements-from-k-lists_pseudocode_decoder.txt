CLASS Solution
    FUNCTION smallestRange(nums: List of List of Integer) RETURNS List of Integer
        SET min_heap TO empty list
        SET max_value TO negative infinity
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            PUSH (nums[i][0], i, 0) INTO min_heap
            IF nums[i][0] > max_value THEN
                SET max_value TO nums[i][0]
        
        SET smallest_range TO infinity
        SET result TO list containing [-100000, 100000]
        
        WHILE min_heap IS NOT empty
            REMOVE smallest element FROM min_heap INTO (min_value, list_index, element_index)
            
            IF (max_value - min_value) < smallest_range THEN
                SET smallest_range TO (max_value - min_value)
                SET result TO list containing [min_value, max_value]
            
            IF (element_index + 1) < LENGTH OF nums[list_index] THEN
                SET next_value TO nums[list_index][element_index + 1]
                PUSH (next_value, list_index, element_index + 1) INTO min_heap
                IF next_value > max_value THEN
                    SET max_value TO next_value
            ELSE
                BREAK
        
        RETURN result
    END FUNCTION
END CLASS
