CLASS Solution
    FUNCTION findDuplicate(paths: List of String) RETURNS List of List of String
        SET content_to_paths TO empty dictionary with default list values
        FOR each path IN paths
            SET parts TO SPLIT path BY spaces
            SET directory TO first element of parts
            FOR each file_info IN parts from second element onward
                SET file_name, content TO SPLIT file_info BY '('
                REMOVE last character from content
                APPEND CONCATENATE directory, '/', file_name TO content_to_paths at key content
        SET result TO empty list
        FOR each paths_list IN values of content_to_paths
            IF LENGTH of paths_list GREATER THAN 1
                APPEND paths_list TO result
        RETURN result
    END FUNCTION
END CLASS
