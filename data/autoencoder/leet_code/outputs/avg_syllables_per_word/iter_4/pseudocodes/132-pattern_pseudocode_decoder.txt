CLASS Solution
    FUNCTION find132pattern(nums)
        SET second TO negative infinity
        SET stack TO empty list

        FOR each num IN nums traversed from right to left
            IF num < second THEN
                RETURN true
            END IF

            WHILE stack is not empty AND top element of stack < num
                SET second TO pop from stack
            END WHILE

            APPEND num TO stack
        END FOR

        RETURN false
    END FUNCTION
END CLASS
