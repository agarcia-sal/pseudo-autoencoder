CLASS Solution
    FUNCTION minimumTotal(triangle)
        FOR row FROM LENGTH OF triangle MINUS 2 DOWNTO 0
            FOR col FROM 0 TO LENGTH OF triangle[row] MINUS 1
                SET triangle[row][col] TO triangle[row][col] PLUS MINIMUM OF triangle[row + 1][col] AND triangle[row + 1][col + 1]
        RETURN triangle[0][0]
    END FUNCTION
END CLASS
