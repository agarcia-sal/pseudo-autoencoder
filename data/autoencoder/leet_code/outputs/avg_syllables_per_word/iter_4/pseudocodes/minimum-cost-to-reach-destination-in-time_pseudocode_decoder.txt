CLASS Solution
    FUNCTION minCost(maxTime, edges, passingFees)
        SET n TO LENGTH OF passingFees
        SET graph TO LIST OF n EMPTY LISTS
        FOR each (x, y, time) IN edges
            APPEND (y, time) TO graph[x]
            APPEND (x, time) TO graph[y]
        
        SET pq TO LIST WITH (passingFees[0], 0, 0)
        SET visited_time TO LIST OF n VALUES ALL SET TO infinity
        SET visited_time[0] TO 0
        
        WHILE pq IS NOT EMPTY
            REMOVE smallest element FROM pq AND ASSIGN TO (current_cost, current_city, current_time)
            
            IF current_city EQUALS n MINUS 1 THEN
                RETURN current_cost
            
            FOR each (neighbor, travel_time) IN graph[current_city]
                SET new_time TO current_time PLUS travel_time
                SET new_cost TO current_cost PLUS passingFees[neighbor]
                
                IF new_time IS LESS THAN OR EQUAL TO maxTime AND (new_time IS LESS THAN visited_time[neighbor] OR new_cost IS LESS THAN current_cost) THEN
                    SET visited_time[neighbor] TO new_time
                    ADD (new_cost, neighbor, new_time) TO pq
        
        RETURN -1
    END FUNCTION
END CLASS
