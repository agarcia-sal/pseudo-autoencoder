CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION maxSumBST(root)
        SET self.max_sum TO 0

        FUNCTION helper(node)
            IF node IS None
                RETURN True, 0, positive infinity, negative infinity
            END IF

            SET left_is_bst, left_sum, left_min, left_max TO helper(node.left)
            SET right_is_bst, right_sum, right_min, right_max TO helper(node.right)

            IF left_is_bst AND right_is_bst AND left_max < node.val AND node.val < right_min
                SET current_sum TO left_sum + right_sum + node.val
                SET current_min TO the smaller of node.val and left_min
                SET current_max TO the larger of node.val and right_max
                SET self.max_sum TO the larger of self.max_sum and current_sum
                RETURN True, current_sum, current_min, current_max
            ELSE
                RETURN False, 0, positive infinity, negative infinity
            END IF
        END FUNCTION

        CALL helper(root)
        RETURN self.max_sum
    END FUNCTION
END CLASS
