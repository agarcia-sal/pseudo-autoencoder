CLASS Solution
    FUNCTION findMin(nums)
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        
        IF nums[left] < nums[right]
            RETURN nums[left]
        
        WHILE left < right
            SET mid TO (left PLUS right) DIVIDED BY 2
            
            IF nums[mid] > nums[right]
                SET left TO mid PLUS 1
            ELSE
                SET right TO mid
        
        RETURN nums[left]
    END FUNCTION
END CLASS
