CLASS Solution
    FUNCTION judgePoint24(cards)
        FUNCTION solve(nums)
            IF LENGTH OF nums IS 1 THEN
                RETURN absolute value of (nums[0] MINUS 24) LESS THAN 0.000001
            END IF

            FOR i FROM 0 TO LENGTH OF nums MINUS 1
                FOR j FROM 0 TO LENGTH OF nums MINUS 1
                    IF i IS NOT EQUAL TO j THEN
                        SET new_nums TO empty list
                        FOR k FROM 0 TO LENGTH OF nums MINUS 1
                            IF k IS NOT i AND k IS NOT j THEN
                                APPEND nums[k] TO new_nums
                            END IF
                        END FOR

                        FOR EACH op IN {division, multiplication, addition, subtraction}
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i THEN
                                CONTINUE TO NEXT op
                            END IF

                            IF op IS NOT division OR nums[j] IS NOT 0 THEN
                                APPEND result of op(nums[i], nums[j]) TO new_nums
                                IF solve(new_nums) IS TRUE THEN
                                    RETURN TRUE
                                END IF
                                REMOVE last element FROM new_nums
                            END IF
                        END FOR
                    END IF
                END FOR
            END FOR

            RETURN FALSE
        END FUNCTION

        RETURN solve(cards)
    END FUNCTION
END CLASS
