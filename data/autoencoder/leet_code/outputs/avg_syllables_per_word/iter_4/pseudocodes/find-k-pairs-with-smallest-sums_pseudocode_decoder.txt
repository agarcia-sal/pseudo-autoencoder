CLASS Solution
    FUNCTION kSmallestPairs(nums1, nums2, k)
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k <= 0
            RETURN empty list
        
        SET min_heap TO empty heap
        FOR i FROM 0 TO MINIMUM OF k AND LENGTH OF nums1 MINUS 1
            PUSH (nums1[i] + nums2[0], i, 0) INTO min_heap
        
        SET result TO empty list
        WHILE min_heap IS NOT EMPTY AND LENGTH OF result IS LESS THAN k
            POP smallest sum, i, j FROM min_heap
            APPEND [nums1[i], nums2[j]] TO result
            IF j + 1 IS LESS THAN LENGTH OF nums2
                PUSH (nums1[i] + nums2[j + 1], i, j + 1) INTO min_heap
        
        RETURN result
    END FUNCTION
END CLASS
