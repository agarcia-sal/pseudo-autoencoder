CLASS Solution
    FUNCTION shortestWordDistance(wordsDict, word1, word2)
        SET min_distance TO infinity
        SET prev_word1 TO -1
        SET prev_word2 TO -1
        SET same_word TO (word1 EQUALS word2)
        
        FOR EACH index, word IN wordsDict WITH INDICES
            IF word EQUALS word1 THEN
                IF same_word IS TRUE THEN
                    SET prev_word2 TO prev_word1
                END IF
                SET prev_word1 TO index
            END IF
            
            IF word EQUALS word2 THEN
                IF same_word IS FALSE THEN
                    SET prev_word2 TO index
                END IF
            END IF
            
            IF prev_word1 IS NOT -1 AND prev_word2 IS NOT -1 THEN
                SET min_distance TO MINIMUM OF min_distance AND ABSOLUTE VALUE OF (prev_word1 MINUS prev_word2)
            END IF
        END FOR
        
        RETURN min_distance
    END FUNCTION
END CLASS
