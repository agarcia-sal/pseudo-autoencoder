CLASS Solution
    FUNCTION makeLargestSpecial(s)
        FUNCTION dfs(substring)
            SET count TO 0
            SET start TO 0
            SET specials TO empty list
            
            FOR each index i AND character char IN substring
                IF char EQUALS '1'
                    INCREMENT count BY 1
                ELSE
                    DECREMENT count BY 1
                
                IF count EQUALS 0
                    APPEND '1' CONCATENATED WITH dfs(substring FROM start + 1 TO i) CONCATENATED WITH '0' TO specials
                    SET start TO i + 1
            
            SORT specials IN reverse lexicographical order
            RETURN concatenation OF all strings IN specials
        END FUNCTION
        
        RETURN dfs(s)
    END FUNCTION
END CLASS
