CLASS Solution
    FUNCTION removeKdigits(num, k)
        SET stack TO empty list
        FOR each digit IN num
            WHILE k GREATER THAN 0 AND stack IS NOT empty AND last element in stack GREATER THAN digit
                REMOVE last element FROM stack
                DECREMENT k BY 1
            END WHILE
            APPEND digit TO stack
        END FOR
        IF k GREATER THAN 0
            SET final_stack TO elements in stack EXCLUDING last k elements
        ELSE
            SET final_stack TO stack
        END IF
        SET result TO concatenation of elements in final_stack AFTER REMOVING leading zeros
        IF result IS empty
            RETURN "0"
        ELSE
            RETURN result
        END IF
    END FUNCTION
END CLASS
