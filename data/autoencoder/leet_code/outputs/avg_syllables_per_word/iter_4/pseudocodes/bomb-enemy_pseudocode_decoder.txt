CLASS Solution
    FUNCTION maxKilledEnemies(grid)
        IF grid IS EMPTY OR grid[0] IS EMPTY
            RETURN 0
        END IF

        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]
        SET max_kills TO 0

        SET row_kills TO 0
        SET col_kills TO LIST OF SIZE n WITH ALL ZEROS

        FOR i FROM 0 TO m - 1
            FOR j FROM 0 TO n - 1
                IF j EQUALS 0 OR grid[i][j - 1] EQUALS 'W'
                    SET row_kills TO 0
                    FOR k FROM j TO n - 1
                        IF grid[i][k] EQUALS 'E'
                            INCREMENT row_kills BY 1
                        ELSE IF grid[i][k] EQUALS 'W'
                            BREAK
                        END IF
                    END FOR
                END IF

                IF i EQUALS 0 OR grid[i - 1][j] EQUALS 'W'
                    SET col_kills[j] TO 0
                    FOR k FROM i TO m - 1
                        IF grid[k][j] EQUALS 'E'
                            INCREMENT col_kills[j] BY 1
                        ELSE IF grid[k][j] EQUALS 'W'
                            BREAK
                        END IF
                    END FOR
                END IF

                IF grid[i][j] EQUALS '0'
                    SET max_kills TO MAXIMUM OF max_kills AND (row_kills + col_kills[j])
                END IF
            END FOR
        END FOR

        RETURN max_kills
    END FUNCTION
END CLASS
