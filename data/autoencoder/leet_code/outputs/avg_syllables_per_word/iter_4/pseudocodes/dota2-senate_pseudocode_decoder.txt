CLASS Solution
    FUNCTION predictPartyVictory(senate)
        SET radiant TO empty queue
        SET dire TO empty queue
        FOR each index i AND character s IN senate
            IF s EQUALS 'R'
                APPEND i TO radiant
            ELSE
                APPEND i TO dire
        WHILE radiant IS NOT empty AND dire IS NOT empty
            SET r_index TO REMOVE first element FROM radiant
            SET d_index TO REMOVE first element FROM dire
            IF r_index < d_index
                APPEND (r_index + LENGTH OF senate) TO radiant
            ELSE
                APPEND (d_index + LENGTH OF senate) TO dire
        IF radiant IS NOT empty
            RETURN "Radiant"
        ELSE
            RETURN "Dire"
    END FUNCTION
END CLASS
