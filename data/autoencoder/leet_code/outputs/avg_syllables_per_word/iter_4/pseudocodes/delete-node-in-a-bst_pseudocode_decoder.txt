CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO new empty queue
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue REMOVE FRONT
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]
        IF root IS None
            RETURN None
        IF root.val > key
            SET root.left TO self.deleteNode(root.left, key)
            RETURN root
        IF root.val < key
            SET root.right TO self.deleteNode(root.right, key)
            RETURN root
        IF root.left IS None
            RETURN root.right
        IF root.right IS None
            RETURN root.left
        SET node TO root.right
        WHILE node.left IS NOT None
            SET node TO node.left
        SET node.left TO root.left
        SET root TO root.right
        RETURN root
    END FUNCTION
END CLASS
