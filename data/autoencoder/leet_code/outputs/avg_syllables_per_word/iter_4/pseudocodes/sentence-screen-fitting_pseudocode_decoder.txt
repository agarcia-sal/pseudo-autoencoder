CLASS Solution
    FUNCTION wordsTyping(sentence: List of String, rows: Integer, cols: Integer) RETURNS Integer
        SET s TO JOIN sentence WITH ' ' PLUS ' '
        SET total_length TO LENGTH OF s
        SET position TO 0
        
        FOR each row FROM 1 TO rows
            INCREMENT position BY cols
            
            IF CHARACTER AT (position MODULO total_length) IN s IS ' '
                INCREMENT position BY 1
            ELSE
                WHILE position > 0 AND CHARACTER AT ((position - 1) MODULO total_length) IN s IS NOT ' '
                    DECREMENT position BY 1
        
        RETURN position DIVIDED BY total_length
    END FUNCTION
END CLASS
