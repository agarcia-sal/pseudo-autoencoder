CLASS Solution
    FUNCTION spiralOrder(matrix)
        IF matrix IS EMPTY OR matrix[0] IS EMPTY
            RETURN empty list
        END IF

        SET result TO empty list
        SET top TO 0
        SET bottom TO LENGTH OF matrix MINUS 1
        SET left TO 0
        SET right TO LENGTH OF matrix[0] MINUS 1

        WHILE top <= bottom AND left <= right
            FOR i FROM left TO right
                APPEND matrix[top][i] TO result
            END FOR
            INCREMENT top BY 1

            FOR i FROM top TO bottom
                APPEND matrix[i][right] TO result
            END FOR
            DECREMENT right BY 1

            IF top <= bottom
                FOR i FROM right DOWN TO left
                    APPEND matrix[bottom][i] TO result
                END FOR
                DECREMENT bottom BY 1
            END IF

            IF left <= right
                FOR i FROM bottom DOWN TO top
                    APPEND matrix[i][left] TO result
                END FOR
                INCREMENT left BY 1
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
