CLASS ListNode
    FUNCTION __init__(val = 0, next = None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY
        RETURN None
    SET head TO new ListNode(values[0])
    SET p TO head
    FOR each val IN values FROM SECOND ELEMENT TO END
        SET node TO new ListNode(val)
        SET p.next TO node
        SET p TO node
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    IF p1 IS None OR p2 IS None
        RETURN False
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION sortedListToBST(head)
        IF head IS None
            RETURN None

        FUNCTION findMiddle(left, right)
            SET slow TO left
            SET fast TO left
            WHILE fast NOT EQUAL TO right AND fast.next NOT EQUAL TO right
                SET slow TO slow.next
                SET fast TO fast.next.next
            RETURN slow
        END FUNCTION

        FUNCTION convertListToBST(left, right)
            IF left EQUALS right
                RETURN None

            SET mid TO findMiddle(left, right)

            SET node TO new TreeNode(mid.val)
            SET node.left TO convertListToBST(left, mid)
            SET node.right TO convertListToBST(mid.next, right)
            RETURN node
        END FUNCTION

        RETURN convertListToBST(head, None)
    END FUNCTION
END CLASS
