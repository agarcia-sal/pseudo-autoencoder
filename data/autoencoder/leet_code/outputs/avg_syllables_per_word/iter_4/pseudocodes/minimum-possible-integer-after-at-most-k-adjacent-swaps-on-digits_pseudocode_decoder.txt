CLASS Solution
    FUNCTION minInteger(num AS STRING, k AS INTEGER) AS STRING
        SET num TO LIST OF CHARACTERS IN num
        SET n TO LENGTH OF num

        WHILE k GREATER THAN 0
            SET swapped TO FALSE
            FOR i FROM 0 TO n MINUS 2
                IF num AT i GREATER THAN num AT i PLUS 1
                    SET min_digit TO '9'
                    SET min_index TO -1
                    FOR j FROM i TO MINIMUM OF (i PLUS k PLUS 1) AND n MINUS 1
                        IF num AT j LESS THAN min_digit
                            SET min_digit TO num AT j
                            SET min_index TO j
                    SET num FROM i TO min_index INCLUSIVE TO LIST STARTING WITH num AT min_index FOLLOWED BY num FROM i TO min_index MINUS 1
                    DECREMENT k BY (min_index MINUS i)
                    SET swapped TO TRUE
                    BREAK FOR LOOP
            IF swapped IS FALSE
                BREAK WHILE LOOP

        RETURN CONCATENATION OF CHARACTERS IN num
    END FUNCTION
END CLASS
