CLASS Solution
    FUNCTION threeSumClosest(nums, target)
        CALL nums.sort()
        SET closest_sum TO infinity

        FOR i FROM 0 TO LENGTH OF nums MINUS 3
            SET left TO i PLUS 1
            SET right TO LENGTH OF nums MINUS 1

            WHILE left < right
                SET current_sum TO nums[i] PLUS nums[left] PLUS nums[right]

                IF ABSOLUTE VALUE OF (current_sum MINUS target) < ABSOLUTE VALUE OF (closest_sum MINUS target)
                    SET closest_sum TO current_sum
                END IF

                IF current_sum < target
                    INCREMENT left BY 1
                ELSE IF current_sum > target
                    DECREMENT right BY 1
                ELSE
                    RETURN current_sum
                END IF
            END WHILE
        END FOR

        RETURN closest_sum
    END FUNCTION
END CLASS
