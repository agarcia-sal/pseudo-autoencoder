CLASS ListNode
    FUNCTION __init__(val = 0, next = None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY THEN
        RETURN None
    SET head TO NEW ListNode(values[0])
    SET p TO head
    FOR val IN values FROM INDEX 1 TO END
        SET node TO NEW ListNode(val)
        SET p.next TO node
        SET p TO node
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None THEN
        RETURN True
    IF p1 IS None OR p2 IS None THEN
        RETURN False
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY THEN
        RETURN None
    SET root TO NEW TreeNode(values[0])
    SET i TO 1
    SET queue TO NEW deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO POP LEFT FROM queue
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None THEN
            SET node.left TO NEW TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None THEN
            SET node.right TO NEW TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None THEN
        RETURN True
    ELSE IF p IS None OR q IS None THEN
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val THEN
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION str2tree(s)
        FUNCTION parse_tree(index)
            IF index GREATER THAN OR EQUAL TO LENGTH OF s THEN
                RETURN None, index
            SET start TO index
            WHILE index LESS THAN LENGTH OF s AND (s[index] IS DIGIT OR s[index] IS '-')
                INCREMENT index BY 1
            SET node TO NEW TreeNode(CONVERT s[start TO index] TO INTEGER)
            IF index LESS THAN LENGTH OF s AND s[index] EQUALS '(' THEN
                SET node.left, index TO parse_tree(index + 1)
            IF index LESS THAN LENGTH OF s AND s[index] EQUALS '(' THEN
                SET node.right, index TO parse_tree(index + 1)
            IF index LESS THAN LENGTH OF s AND s[index] EQUALS ')' THEN
                INCREMENT index BY 1
            RETURN node, index
        END FUNCTION
        SET root, _ TO parse_tree(0)
        RETURN root
    END FUNCTION
END CLASS
