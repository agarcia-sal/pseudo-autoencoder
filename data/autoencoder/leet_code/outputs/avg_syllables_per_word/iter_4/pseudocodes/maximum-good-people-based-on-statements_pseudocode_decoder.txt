CLASS Solution
    FUNCTION maximumGood(statements: List OF List OF INT)
        SET n TO LENGTH OF statements
        SET max_good TO 0

        FOR i FROM 0 TO (2 RAISED TO n) MINUS 1
            SET good TO LIST OF False OF SIZE n
            SET bad TO LIST OF False OF SIZE n

            FOR j FROM 0 TO n MINUS 1
                IF BIT j OF i IS SET
                    SET good[j] TO True
                ELSE
                    SET bad[j] TO True

            SET valid TO True
            FOR j FROM 0 TO n MINUS 1
                IF good[j] IS True
                    FOR k FROM 0 TO n MINUS 1
                        IF (statements[j][k] EQUALS 0 AND good[k] IS True) OR (statements[j][k] EQUALS 1 AND bad[k] IS True)
                            SET valid TO False
                            BREAK FROM INNER LOOP
                    IF valid IS False
                        BREAK FROM OUTER LOOP

            IF valid IS True
                SET max_good TO MAXIMUM OF max_good AND SUM OF good

        RETURN max_good
    END FUNCTION
END CLASS
