CLASS Solution
    FUNCTION isNStraightHand(hand, groupSize)
        IF LENGTH OF hand MODULO groupSize IS NOT 0 THEN
            RETURN False
        SET count TO frequency count of elements in hand
        SET unique_cards TO sorted list of keys in count
        FOR each card IN unique_cards
            IF count[card] IS GREATER THAN 0 THEN
                SET needed TO count[card]
                FOR i FROM 0 TO groupSize MINUS 1
                    IF count[card + i] IS LESS THAN needed THEN
                        RETURN False
                    DECREMENT count[card + i] BY needed
        RETURN True
    END FUNCTION
END CLASS
