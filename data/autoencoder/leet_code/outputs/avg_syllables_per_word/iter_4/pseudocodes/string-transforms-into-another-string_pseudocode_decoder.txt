CLASS Solution
    FUNCTION canConvert(str1, str2)
        IF str1 IS EQUAL TO str2
            RETURN True
        END IF

        SET conversion_map TO empty map

        FOR EACH char1, char2 IN corresponding positions of str1 and str2
            IF char1 IS IN conversion_map
                IF conversion_map[char1] IS NOT EQUAL TO char2
                    RETURN False
                END IF
            ELSE
                SET conversion_map[char1] TO char2
            END IF
        END FOR

        IF SIZE OF unique characters IN str2 IS EQUAL TO 26
            RETURN False
        END IF

        RETURN True
    END FUNCTION
END CLASS
