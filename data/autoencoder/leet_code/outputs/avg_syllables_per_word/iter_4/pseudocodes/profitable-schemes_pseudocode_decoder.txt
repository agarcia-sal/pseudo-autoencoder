CLASS Solution
    FUNCTION profitableSchemes(n, minProfit, group, profit)
        SET MOD TO 10 POWER 9 PLUS 7
        SET dp TO a 2D list of size (n + 1) by (minProfit + 1) filled with 0
        FOR i FROM 0 TO n
            SET dp[i][0] TO 1
        FOR each pair (p, g) IN profit AND group
            FOR i FROM n DOWN TO g
                FOR j FROM minProfit DOWN TO 0
                    SET dp[i][j] TO (dp[i][j] PLUS dp[i - g][MAXIMUM OF 0 AND j MINUS p]) MODULO MOD
        RETURN dp[n][minProfit]
    END FUNCTION
END CLASS
