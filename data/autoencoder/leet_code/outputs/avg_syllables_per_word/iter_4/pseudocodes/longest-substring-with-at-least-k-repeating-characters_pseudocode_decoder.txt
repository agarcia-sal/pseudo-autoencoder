CLASS Solution
    FUNCTION longestSubstring(s, k)
        IF LENGTH OF s < k THEN
            RETURN 0
        END IF

        SET count TO FREQUENCY COUNT OF CHARACTERS IN s

        FOR EACH char IN count
            IF count[char] < k THEN
                RETURN MAXIMUM OF longestSubstring(substring, k) FOR EACH substring IN s SPLIT BY char
            END IF
        END FOR

        RETURN LENGTH OF s
    END FUNCTION
END CLASS
