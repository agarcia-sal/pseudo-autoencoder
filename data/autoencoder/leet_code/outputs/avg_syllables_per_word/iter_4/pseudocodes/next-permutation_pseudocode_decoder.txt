CLASS Solution
    FUNCTION nextPermutation(nums: List of Integer)
        SET i TO LENGTH OF nums MINUS 2
        WHILE i >= 0 AND nums[i] >= nums[i + 1]
            DECREMENT i BY 1
        IF i >= 0
            SET j TO LENGTH OF nums MINUS 1
            WHILE nums[j] <= nums[i]
                DECREMENT j BY 1
            SWAP nums[i] WITH nums[j]
        REVERSE nums FROM POSITION i + 1 TO END
    END FUNCTION
END CLASS
