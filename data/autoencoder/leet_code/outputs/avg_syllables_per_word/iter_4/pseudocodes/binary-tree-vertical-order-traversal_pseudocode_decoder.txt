CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO new TreeNode with values[0]
    SET i TO 1
    CREATE queue as empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO new TreeNode with values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO new TreeNode with values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION verticalOrder(self, root)
        IF root IS None
            RETURN empty list
        END IF
        CREATE column_table as a dictionary with default empty list
        CREATE queue as new deque containing (root, 0)
        SET min_column TO 0
        SET max_column TO 0
        WHILE queue IS NOT EMPTY
            SET node, column_index TO queue.popleft()
            IF node IS NOT None
                APPEND node.val TO column_table[column_index]
                SET min_column TO minimum of min_column and column_index
                SET max_column TO maximum of max_column and column_index
                APPEND (node.left, column_index - 1) TO queue
                APPEND (node.right, column_index + 1) TO queue
            END IF
        END WHILE
        RETURN list of column_table[x] FOR x IN RANGE from min_column TO max_column INCLUSIVE
    END FUNCTION
END CLASS
