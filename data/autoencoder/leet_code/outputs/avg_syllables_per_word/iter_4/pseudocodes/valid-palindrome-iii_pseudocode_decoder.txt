CLASS Solution
    FUNCTION isValidPalindrome(s, k)
        FUNCTION longest_palindromic_subseq(s)
            SET n TO LENGTH OF s
            SET dp TO TWO-DIMENSIONAL LIST OF SIZE n BY n FILLED WITH 0
            
            FOR i FROM 0 TO n - 1
                SET dp[i][i] TO 1
            
            FOR length FROM 2 TO n
                FOR i FROM 0 TO n - length
                    SET j TO i + length - 1
                    IF s[i] EQUALS s[j]
                        SET dp[i][j] TO dp[i + 1][j - 1] + 2
                    ELSE
                        SET dp[i][j] TO MAXIMUM OF dp[i + 1][j] AND dp[i][j - 1]
            
            RETURN dp[0][n - 1]
        
        SET longest_palindrome_length TO longest_palindromic_subseq(s)
        RETURN (LENGTH OF s MINUS longest_palindrome_length) LESS THAN OR EQUAL TO k
    END FUNCTION
END CLASS
