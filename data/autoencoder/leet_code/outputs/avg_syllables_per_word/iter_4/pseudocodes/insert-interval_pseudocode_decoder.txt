CLASS Solution
    FUNCTION insert(intervals, newInterval)
        IF intervals IS EMPTY
            RETURN LIST CONTAINING newInterval
        SET merged TO EMPTY LIST
        SET i TO 0
        SET n TO LENGTH OF intervals
        WHILE i < n AND intervals[i][1] < newInterval[0]
            APPEND intervals[i] TO merged
            INCREMENT i BY 1
        WHILE i < n AND intervals[i][0] <= newInterval[1]
            SET newInterval[0] TO MINIMUM OF newInterval[0] AND intervals[i][0]
            SET newInterval[1] TO MAXIMUM OF newInterval[1] AND intervals[i][1]
            INCREMENT i BY 1
        APPEND newInterval TO merged
        WHILE i < n
            APPEND intervals[i] TO merged
            INCREMENT i BY 1
        RETURN merged
    END FUNCTION
END CLASS
