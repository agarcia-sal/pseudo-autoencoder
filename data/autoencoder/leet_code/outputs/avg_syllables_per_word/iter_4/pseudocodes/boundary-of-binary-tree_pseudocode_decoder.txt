CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO NEW TreeNode WITH values[0]
    SET i TO 1
    INITIALIZE queue AS an empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.pop leftmost element
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION boundaryOfBinaryTree(root: Optional[TreeNode]) RETURNS List[int]
        IF root IS None
            RETURN empty list
        END IF

        FUNCTION is_leaf(node)
            RETURN node.left IS None AND node.right IS None
        END FUNCTION

        SET left_boundary TO empty list
        SET current TO root.left
        WHILE current IS NOT None
            IF NOT is_leaf(current)
                APPEND current.val TO left_boundary
            END IF
            IF current.left IS NOT None
                SET current TO current.left
            ELSE
                SET current TO current.right
            END IF
        END WHILE

        SET right_boundary TO empty list
        SET current TO root.right
        WHILE current IS NOT None
            IF NOT is_leaf(current)
                APPEND current.val TO right_boundary
            END IF
            IF current.right IS NOT None
                SET current TO current.right
            ELSE
                SET current TO current.left
            END IF
        END WHILE

        SET leaves TO empty list
        FUNCTION collect_leaves(node)
            IF node IS None
                RETURN
            END IF
            IF is_leaf(node)
                APPEND node.val TO leaves
            END IF
            CALL collect_leaves(node.left)
            CALL collect_leaves(node.right)
        END FUNCTION

        CALL collect_leaves(root)

        IF is_leaf(root)
            SET leaves TO empty list
        END IF

        RETURN list consisting of root.val concatenated with left_boundary concatenated with leaves concatenated with right_boundary reversed
    END FUNCTION
END CLASS
