CLASS Solution
    FUNCTION kthSmallest(matrix: List of List of Integer, k: Integer) RETURNS Integer
        FUNCTION countLessEqual(mid: Integer) RETURNS Integer
            SET count TO 0
            SET row TO n MINUS 1
            SET col TO 0
            WHILE row GREATER OR EQUAL TO 0 AND col LESS THAN n
                IF matrix[row][col] LESS OR EQUAL TO mid THEN
                    INCREMENT count BY row PLUS 1
                    INCREMENT col BY 1
                ELSE
                    DECREMENT row BY 1
            RETURN count
        END FUNCTION

        SET n TO LENGTH OF matrix
        SET low TO matrix[0][0]
        SET high TO matrix[n MINUS 1][n MINUS 1]

        WHILE low LESS THAN high
            SET mid TO (low PLUS high) DIVIDED BY 2
            IF countLessEqual(mid) LESS THAN k THEN
                SET low TO mid PLUS 1
            ELSE
                SET high TO mid

        RETURN low
    END FUNCTION
END CLASS
