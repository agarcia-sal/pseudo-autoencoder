CLASS Solution
    FUNCTION numDistinct(s, t)
        SET m TO LENGTH OF s
        SET n TO LENGTH OF t
        SET dp TO 2D array of size (m + 1) by (n + 1) filled with 0
        
        FOR i FROM 0 TO m
            SET dp[i][0] TO 1
        
        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                IF s[i - 1] EQUALS t[j - 1] THEN
                    SET dp[i][j] TO dp[i - 1][j - 1] PLUS dp[i - 1][j]
                ELSE
                    SET dp[i][j] TO dp[i - 1][j]
        
        RETURN dp[m][n]
    END FUNCTION
END CLASS
