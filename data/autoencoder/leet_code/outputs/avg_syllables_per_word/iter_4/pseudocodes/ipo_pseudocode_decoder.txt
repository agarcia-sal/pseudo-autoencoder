CLASS Solution
    FUNCTION findMaximizedCapital(k, w, profits, capital)
        SET projects TO list of pairs (capital, profits)
        SORT projects by capital ascending
        
        SET max_heap TO empty list
        SET i TO 0
        SET n TO length of projects
        
        FOR _ FROM 1 TO k
            WHILE i < n AND projects[i][0] <= w
                PUSH negative of projects[i][1] INTO max_heap
                INCREMENT i BY 1
            
            IF max_heap is empty
                BREAK loop
            
            SET w TO w PLUS negative of POP from max_heap
        
        RETURN w
    END FUNCTION
END CLASS
