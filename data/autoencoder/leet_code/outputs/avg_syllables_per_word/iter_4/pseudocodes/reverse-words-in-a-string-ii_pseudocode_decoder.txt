CLASS Solution
    FUNCTION reverseWords(s: List of characters)
        CALL s.reverse()
        SET start TO 0
        FOR end FROM 0 TO LENGTH OF s MINUS 1
            IF s[end] EQUALS ' '
                SET s FROM start TO end MINUS 1 TO THE REVERSED SUBLIST OF s FROM start TO end MINUS 1
                SET start TO end PLUS 1
        SET s FROM start TO END TO THE REVERSED SUBLIST OF s FROM start TO END
    END FUNCTION
END CLASS
