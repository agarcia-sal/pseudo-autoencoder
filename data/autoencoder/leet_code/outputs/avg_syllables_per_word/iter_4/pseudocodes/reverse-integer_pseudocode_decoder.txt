CLASS Solution
    FUNCTION reverse(x: integer) RETURNS integer
        IF x < 0 THEN
            SET sign TO -1
        ELSE
            SET sign TO 1
        SET reversed_x TO integer formed by reversing the digits of the absolute value of x
        SET result TO sign MULTIPLIED BY reversed_x
        IF result LESS THAN negative 2 TO THE POWER OF 31 OR result GREATER THAN 2 TO THE POWER OF 31 MINUS 1 THEN
            RETURN 0
        RETURN result
    END FUNCTION
END CLASS
