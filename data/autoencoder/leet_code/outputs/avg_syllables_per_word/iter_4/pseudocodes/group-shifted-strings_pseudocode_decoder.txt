CLASS Solution
    FUNCTION groupStrings(strings)
        FUNCTION get_canonical_form(s)
            IF s IS EMPTY
                RETURN empty string
            SET canonical_form TO empty list
            FOR each character c IN s
                SET difference TO (ASCII value of c MINUS ASCII value of first character in s) MODULO 26
                APPEND string representation of difference TO canonical_form
            RETURN JOIN elements of canonical_form WITH ","
        END FUNCTION

        SET groups TO empty dictionary with default list value
        FOR each string s IN strings
            SET canonical_form TO get_canonical_form(s)
            APPEND s TO groups[canonical_form]
        RETURN list of values in groups
    END FUNCTION
END CLASS
