CLASS Solution
    FUNCTION uniquePaths(m, n)
        SET dp TO 2D array with m rows and n columns, each element initialized to 1
        FOR i FROM 1 TO m - 1
            FOR j FROM 1 TO n - 1
                SET dp[i][j] TO dp[i - 1][j] PLUS dp[i][j - 1]
        RETURN dp[m - 1][n - 1]
    END FUNCTION
END CLASS
