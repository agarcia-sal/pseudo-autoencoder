CLASS Solution
    FUNCTION combinationSum4(nums, target)
        SET dp TO list of zeros with length target PLUS 1
        SET dp[0] TO 1
        FOR amount FROM 1 TO target INCLUSIVE
            FOR each num IN nums
                IF amount IS GREATER THAN OR EQUAL TO num THEN
                    INCREMENT dp[amount] BY dp[amount MINUS num]
        RETURN dp[target]
    END FUNCTION
END CLASS
