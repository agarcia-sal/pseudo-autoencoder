CLASS Solution
    FUNCTION isTransformable(s, t)
        IMPORT defaultdict FROM collections
        IMPORT heappop, heappush FROM heapq
        
        SET pos TO defaultdict OF list
        
        FOR i, c IN ENUMERATE s
            CALL heappush ON pos[INTEGER VALUE OF c], i
        
        FOR i, c IN ENUMERATE t
            SET digit TO INTEGER VALUE OF c
            
            IF pos[digit] IS EMPTY
                RETURN False
            
            FOR smaller FROM 0 TO digit MINUS 1
                IF pos[smaller] IS NOT EMPTY AND pos[smaller][0] LESS THAN pos[digit][0]
                    RETURN False
            
            CALL heappop ON pos[digit]
        
        RETURN True
    END FUNCTION
END CLASS
