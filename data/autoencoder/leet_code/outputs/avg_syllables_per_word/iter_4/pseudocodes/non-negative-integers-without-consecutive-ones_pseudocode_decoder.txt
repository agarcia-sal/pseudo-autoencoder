CLASS Solution
    FUNCTION findIntegers(n)
        SET bin_rep TO binary representation of n as string without '0b'
        SET length TO LENGTH OF bin_rep

        INITIALIZE dp AS list of zeros with size length + 1
        SET dp[0] TO 1
        SET dp[1] TO 2

        FOR i FROM 2 TO length INCLUSIVE
            SET dp[i] TO dp[i - 1] + dp[i - 2]

        SET result TO 0
        SET prev_bit TO 0

        FOR i FROM 0 TO length - 1
            IF bin_rep[i] EQUALS '1' THEN
                INCREMENT result BY dp[length - i - 1]
                IF prev_bit EQUALS 1 THEN
                    DECREMENT result BY 1
                    BREAK LOOP
            END IF
            SET prev_bit TO 1 IF bin_rep[i] EQUALS '1' ELSE 0
        END FOR

        RETURN result + 1
    END FUNCTION
END CLASS
