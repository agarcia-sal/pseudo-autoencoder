CLASS Solution
    FUNCTION minSkips(dist, speed, hoursBefore)
        SET n TO LENGTH OF dist
        SET dp TO 2D LIST OF SIZE n BY (n + 1) FILLED WITH infinity
        SET total_dist TO SUM OF dist

        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore
            RETURN -1

        SET dp[0][0] TO dist[0]
        FOR j FROM 1 TO n - 1
            SET dp[0][j] TO dist[0]

        FOR i FROM 1 TO n - 1
            FOR j FROM 0 TO i
                IF j LESS THAN i
                    SET dp[i][j] TO CEILING OF (dp[i-1][j] DIVIDED BY speed) MULTIPLIED BY speed PLUS dist[i]
                IF j GREATER THAN 0
                    SET dp[i][j] TO MINIMUM OF dp[i][j] AND dp[i-1][j-1] PLUS dist[i]

        FOR j FROM 0 TO n - 1
            IF dp[n-1][j] IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed
                RETURN j

        RETURN -1
END CLASS
