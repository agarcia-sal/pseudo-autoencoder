CLASS Solution
    FUNCTION numberOfBoomerangs(points)
        SET ans TO 0
        FOR each p1 IN points
            SET cnt TO empty counter
            FOR each p2 IN points
                SET dx TO p1[0] MINUS p2[0]
                SET dy TO p1[1] MINUS p2[1]
                SET d TO dx TIMES dx PLUS dy TIMES dy
                INCREMENT ans BY cnt VALUE AT d
                INCREMENT cnt VALUE AT d BY 1
        RETURN ans TIMES 2
    END FUNCTION
END CLASS
