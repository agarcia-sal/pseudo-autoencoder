CLASS Solution
    FUNCTION topKFrequent(words: List of String, k: Integer) RETURNS List of String
        SET count TO frequency count of each word in words
        SET heap TO empty list
        FOR each word, freq IN count
            APPEND tuple of (negative freq, word) TO heap
        CALL heapify on heap
        SET result TO empty list
        FOR i FROM 1 TO k
            SET top_element TO pop the smallest element from heap
            APPEND second item of top_element TO result
        RETURN result
    END FUNCTION
END CLASS
