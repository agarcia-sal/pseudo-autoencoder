CLASS Solution
    FUNCTION minWindow(s, t)
        SET t_count TO count frequencies of characters in t
        SET required TO number of unique characters in t_count

        SET left TO 0
        SET right TO 0
        SET formed TO 0
        SET window_counts TO count frequencies in empty window

        SET min_length TO infinity
        SET min_start TO 0

        WHILE right < length of s
            INCREMENT count of s[right] in window_counts

            IF s[right] is in t_count AND window_counts[s[right]] EQUALS t_count[s[right]]
                INCREMENT formed BY 1
            END IF

            WHILE left <= right AND formed EQUALS required
                SET char TO s[left]

                IF (right - left + 1) LESS THAN min_length
                    SET min_length TO (right - left + 1)
                    SET min_start TO left
                END IF

                DECREMENT count of char in window_counts

                IF char in t_count AND window_counts[char] LESS THAN t_count[char]
                    DECREMENT formed BY 1
                END IF

                INCREMENT left BY 1
            END WHILE

            INCREMENT right BY 1
        END WHILE

        IF min_length EQUALS infinity
            RETURN empty string
        ELSE
            RETURN substring of s from min_start TO (min_start + min_length)
        END IF
    END FUNCTION
END CLASS
