CLASS Solution
    FUNCTION findOrder(numCourses, prerequisites)
        SET adj_list TO DICTIONARY with keys from 0 to numCourses - 1 each mapped to empty LIST
        SET in_degrees TO LIST of zeroes with length numCourses
        
        FOR each pair (dest, src) IN prerequisites
            APPEND dest TO adj_list[src]
            INCREMENT in_degrees[dest] BY 1
        
        SET queue TO deque containing courses with in_degrees equal to 0
        SET topological_order TO empty LIST
        
        WHILE queue is not empty
            SET current TO queue.popleft()
            APPEND current TO topological_order
            
            FOR each neighbor IN adj_list[current]
                DECREMENT in_degrees[neighbor] BY 1
                IF in_degrees[neighbor] IS 0
                    APPEND neighbor TO queue
        
        IF LENGTH of topological_order IS numCourses
            RETURN topological_order
        ELSE
            RETURN empty LIST
    END FUNCTION
END CLASS
