CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    SET root TO NEW TreeNode WITH values[0]
    SET i TO 1
    INITIALIZE queue AS empty deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO REMOVE leftmost element FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH values[i]
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH values[i]
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION subtreeWithAllDeepest(root: TreeNode) -> TreeNode
        FUNCTION dfs(node)
            IF node IS None
                RETURN node, 0
            SET left_node, left_depth TO dfs(node.left)
            SET right_node, right_depth TO dfs(node.right)
            IF left_depth > right_depth
                RETURN left_node, left_depth + 1
            ELSE IF right_depth > left_depth
                RETURN right_node, right_depth + 1
            ELSE
                RETURN node, left_depth + 1
        END FUNCTION
        RETURN dfs(root)[0]
    END FUNCTION
END CLASS
