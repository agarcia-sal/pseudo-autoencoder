CLASS Solution
    FUNCTION findLength(nums1, nums2)
        SET dp TO a 2D array of zeros with dimensions (LENGTH OF nums1 + 1) by (LENGTH OF nums2 + 1)
        SET max_length TO 0
        FOR i FROM 1 TO LENGTH OF nums1
            FOR j FROM 1 TO LENGTH OF nums2
                IF nums1 at position (i - 1) EQUALS nums2 at position (j - 1)
                    SET dp at position [i][j] TO dp at position [i - 1][j - 1] PLUS 1
                    IF dp at position [i][j] GREATER THAN max_length
                        SET max_length TO dp at position [i][j]
        RETURN max_length
    END FUNCTION
END CLASS
