CLASS Solution
    FUNCTION findDiagonalOrder(mat)
        IF mat IS EMPTY OR mat[0] IS EMPTY
            RETURN EMPTY LIST
        
        SET m TO LENGTH OF mat
        SET n TO LENGTH OF mat[0]
        SET result TO EMPTY LIST
        SET r TO 0
        SET c TO 0
        SET direction TO 1
        
        FOR _ FROM 0 TO m TIMES n MINUS 1
            APPEND mat[r][c] TO result
            IF direction EQUALS 1
                SET new_r TO r MINUS 1
                SET new_c TO c PLUS 1
            ELSE
                SET new_r TO r PLUS 1
                SET new_c TO c MINUS 1
            
            IF new_r LESS THAN 0 OR new_r EQUALS m OR new_c LESS THAN 0 OR new_c EQUALS n
                IF direction EQUALS 1
                    IF new_c EQUALS n
                        SET new_r TO r PLUS 1
                        SET new_c TO c
                    ELSE
                        SET new_r TO 0
                        SET new_c TO c PLUS 1
                ELSE
                    IF new_r EQUALS m
                        SET new_r TO r
                        SET new_c TO c PLUS 1
                    ELSE
                        SET new_r TO r PLUS 1
                        SET new_c TO 0
                SET direction TO NEGATIVE direction
            
            SET r TO new_r
            SET c TO new_c
        
        RETURN result
    END FUNCTION
END CLASS
