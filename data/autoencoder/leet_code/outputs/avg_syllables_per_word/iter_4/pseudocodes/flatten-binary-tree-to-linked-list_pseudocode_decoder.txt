CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values is empty THEN
        RETURN None
    END IF
    SET root TO TreeNode with val = values[0]
    SET i TO 1
    INITIALIZE queue as an empty deque
    APPEND root TO queue
    WHILE queue is not empty DO
        SET node TO queue.popleft()
        IF i < LENGTH of values AND values[i] is not None THEN
            SET node.left TO TreeNode with val = values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH of values AND values[i] is not None THEN
            SET node.right TO TreeNode with val = values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None THEN
        RETURN True
    ELSE IF p IS None OR q IS None THEN
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val THEN
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION flatten(root: Optional[TreeNode])
        IF root IS None THEN
            RETURN
        END IF
        SET stack TO list containing root
        WHILE stack is not empty DO
            SET node TO stack.pop()
            IF node.right IS NOT None THEN
                APPEND node.right TO stack
            END IF
            IF node.left IS NOT None THEN
                APPEND node.left TO stack
            END IF
            SET node.left TO None
            IF stack is not empty THEN
                SET node.right TO stack LAST ELEMENT
            ELSE
                SET node.right TO None
            END IF
        END WHILE
    END FUNCTION
END CLASS
