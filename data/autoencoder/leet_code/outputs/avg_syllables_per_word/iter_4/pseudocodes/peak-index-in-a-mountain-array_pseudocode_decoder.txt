CLASS Solution
    FUNCTION peakIndexInMountainArray(arr)
        SET left TO 0
        SET right TO LENGTH OF arr MINUS 1
        WHILE left < right
            SET mid TO (left + right) DIVIDED BY 2
            IF arr[mid] < arr[mid + 1]
                SET left TO mid + 1
            ELSE
                SET right TO mid
        RETURN left
    END FUNCTION
END CLASS
