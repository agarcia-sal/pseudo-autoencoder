CLASS Solution
    FUNCTION longestMountain(arr)
        SET n TO LENGTH OF arr
        SET longest TO 0

        SET i TO 1
        WHILE i < n - 1
            IF arr[i - 1] < arr[i] AND arr[i] > arr[i + 1]
                SET left TO i - 1
                SET right TO i + 1

                WHILE left > 0 AND arr[left - 1] < arr[left]
                    DECREMENT left BY 1

                WHILE right < n - 1 AND arr[right] > arr[right + 1]
                    INCREMENT right BY 1

                SET current_length TO right - left + 1
                SET longest TO MAXIMUM OF longest AND current_length

                SET i TO right
            ELSE
                INCREMENT i BY 1

        RETURN longest
    END FUNCTION
END CLASS
