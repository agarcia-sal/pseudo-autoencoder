CLASS Solution
    FUNCTION getMoneyAmount(n)
        DEFINE FUNCTION dp(left, right) WITH CACHE
            IF left IS GREATER THAN OR EQUAL TO right
                RETURN 0
            SET min_cost TO INFINITY
            FOR pivot FROM left TO right
                SET cost TO pivot PLUS MAXIMUM OF dp(left, pivot MINUS 1) AND dp(pivot PLUS 1, right)
                SET min_cost TO MINIMUM OF min_cost AND cost
            RETURN min_cost
        END FUNCTION
        RETURN dp(1, n)
    END FUNCTION
END CLASS
