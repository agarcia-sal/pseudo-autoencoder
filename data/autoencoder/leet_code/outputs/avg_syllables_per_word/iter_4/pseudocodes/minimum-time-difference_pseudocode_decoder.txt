CLASS Solution
    FUNCTION findMinDifference(timePoints)
        FUNCTION to_minutes(time)
            SPLIT time BY ':' INTO hours AND minutes
            SET hours TO CONVERTED INTEGER OF hours
            SET minutes TO CONVERTED INTEGER OF minutes
            RETURN hours MULTIPLIED BY 60 PLUS minutes
        END FUNCTION

        SET minutes_list TO EMPTY LIST
        FOR each time IN timePoints
            APPEND to_minutes(time) TO minutes_list

        SORT minutes_list

        SET min_diff TO infinity
        FOR i FROM 1 TO LENGTH OF minutes_list MINUS 1
            SET current_diff TO minutes_list[i] MINUS minutes_list[i - 1]
            SET min_diff TO MINIMUM OF min_diff AND current_diff

        SET circular_diff TO (1440 MINUS minutes_list[-1]) PLUS minutes_list[0]
        SET min_diff TO MINIMUM OF min_diff AND circular_diff

        RETURN min_diff
    END FUNCTION
END CLASS
