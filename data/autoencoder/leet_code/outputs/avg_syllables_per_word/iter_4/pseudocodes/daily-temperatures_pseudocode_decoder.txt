CLASS Solution
    FUNCTION dailyTemperatures(temperatures)
        SET n TO LENGTH OF temperatures
        SET answer TO LIST OF 0s WITH LENGTH n
        SET stack TO EMPTY LIST

        FOR i FROM 0 TO n MINUS 1
            WHILE stack IS NOT EMPTY AND temperatures[i] GREATER THAN temperatures[LAST ELEMENT OF stack]
                SET prev_index TO REMOVE LAST ELEMENT FROM stack
                SET answer[prev_index] TO i MINUS prev_index
            END WHILE
            APPEND i TO stack
        END FOR

        RETURN answer
    END FUNCTION
END CLASS
