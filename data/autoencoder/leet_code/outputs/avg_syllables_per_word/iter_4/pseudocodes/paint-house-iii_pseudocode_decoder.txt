CLASS Solution
    FUNCTION minCost(houses, cost, m, n, target)
        DEFINE FUNCTION dp(i, prev_color, groups)
            IF i == m THEN
                IF groups == target THEN
                    RETURN 0
                ELSE
                    RETURN infinity
            END IF

            IF houses[i] != 0 THEN
                SET new_groups TO groups
                IF houses[i] != prev_color THEN
                    INCREMENT new_groups BY 1
                END IF
                RETURN dp(i + 1, houses[i], new_groups)
            END IF

            SET min_cost TO infinity
            FOR color FROM 1 TO n DO
                SET new_groups TO groups
                IF color != prev_color THEN
                    INCREMENT new_groups BY 1
                END IF
                SET current_cost TO cost[i][color - 1] + dp(i + 1, color, new_groups)
                IF current_cost < min_cost THEN
                    SET min_cost TO current_cost
                END IF
            END FOR

            RETURN min_cost
        END FUNCTION

        SET result TO dp(0, 0, 0)
        IF result == infinity THEN
            RETURN -1
        ELSE
            RETURN result
        END IF
    END FUNCTION
END CLASS
