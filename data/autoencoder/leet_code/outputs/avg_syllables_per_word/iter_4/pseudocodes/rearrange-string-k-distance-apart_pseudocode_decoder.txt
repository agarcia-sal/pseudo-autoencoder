CLASS Solution
    FUNCTION rearrangeString(s, k)
        IF k <= 1 THEN
            RETURN s
        END IF

        SET freq TO frequency count of characters in s

        SET max_heap TO list of pairs (negative count, character) for each character and count in freq
        CONVERT max_heap INTO a heap structure

        SET wait_queue TO empty list
        SET result TO empty list

        WHILE max_heap is not empty DO
            EXTRACT the pair (count, char) with the highest priority from max_heap
            APPEND char TO result

            APPEND (count + 1, char) TO wait_queue

            IF size of wait_queue EQUALS k THEN
                REMOVE the first element (count, char) FROM wait_queue
                IF count < 0 THEN
                    INSERT (count, char) INTO max_heap
                END IF
            END IF
        END WHILE

        IF length of result NOT EQUAL TO length of s THEN
            RETURN empty string
        END IF

        RETURN concatenation of characters in result
    END FUNCTION
END CLASS
