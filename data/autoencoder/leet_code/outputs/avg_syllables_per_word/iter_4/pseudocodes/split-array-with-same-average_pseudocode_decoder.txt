CLASS Solution
    FUNCTION splitArraySameAverage(nums)
        SET n TO LENGTH OF nums
        SET total_sum TO SUM OF nums

        IF n EQUALS 1 THEN
            RETURN False
        END IF

        SET dp TO LIST OF SETS WITH (n DIVIDED BY 2 + 1) ELEMENTS
        ADD 0 TO dp[0]

        FOR each num IN nums DO
            FOR size FROM (n DIVIDED BY 2) DOWN TO 1 DO
                FOR each s IN dp[size - 1] DO
                    SET new_sum TO s PLUS num
                    IF new_sum TIMES n EQUALS size TIMES total_sum THEN
                        RETURN True
                    END IF
                    ADD new_sum TO dp[size]
                END FOR
            END FOR
        END FOR

        RETURN False
    END FUNCTION
END CLASS
