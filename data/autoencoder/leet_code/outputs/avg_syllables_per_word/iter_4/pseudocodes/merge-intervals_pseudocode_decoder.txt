CLASS Solution
    FUNCTION merge(intervals)
        SORT intervals BY the first element of each interval
        SET merged TO empty list
        FOR each interval IN intervals
            IF merged is empty OR the end of the last interval in merged is less than the start of interval
                APPEND interval TO merged
            ELSE
                SET the end of the last interval in merged TO the maximum of the current end and interval's end
        RETURN merged
    END FUNCTION
END CLASS
