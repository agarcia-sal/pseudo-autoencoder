CLASS Solution
    FUNCTION minPatches(nums, n)
        SET missing TO 1
        SET patches TO 0
        SET i TO 0
        WHILE missing <= n
            IF i < LENGTH OF nums AND nums[i] <= missing
                SET missing TO missing + nums[i]
                INCREMENT i BY 1
            ELSE
                SET missing TO missing + missing
                INCREMENT patches BY 1
        RETURN patches
    END FUNCTION
END CLASS
