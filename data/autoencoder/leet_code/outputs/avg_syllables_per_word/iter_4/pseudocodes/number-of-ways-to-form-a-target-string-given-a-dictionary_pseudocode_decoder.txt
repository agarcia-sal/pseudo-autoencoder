CLASS Solution
    FUNCTION numWays(words: List of String, target: String) RETURNS Integer
        SET MOD TO 10^9 + 7
        SET n TO LENGTH OF words[0]
        SET m TO LENGTH OF target

        SET freq TO list of n dictionaries mapping character to integer, each initialized empty
        FOR each word IN words
            FOR each index j and character char IN word
                INCREMENT freq[j][char] BY 1

        SET dp TO a 2D list of size (m + 1) by (n + 1) filled with 0

        FOR j FROM 0 TO n
            SET dp[0][j] TO 1

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                SET dp[i][j] TO dp[i][j - 1]
                IF freq[j - 1][target[i - 1]] > 0 THEN
                    SET dp[i][j] TO (dp[i][j] + dp[i - 1][j - 1] * freq[j - 1][target[i - 1]]) MODULO MOD

        RETURN dp[m][n]
    END FUNCTION
END CLASS
