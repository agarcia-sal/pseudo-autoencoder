CLASS Solution
    FUNCTION minStickers(stickers: List of String, target: String) RETURNS Integer
        SET stickers TO list of frequency counters from each sticker
        SORT stickers by descending maximum frequency character count
        
        SET queue TO deque containing (target, 0)
        SET memo TO dictionary with key target and value 0
        
        WHILE queue is not empty
            SET current_target, used_stickers TO queue remove front element
            
            FOR each sticker IN stickers
                IF first character of current_target NOT IN sticker
                    CONTINUE to next sticker
                
                SET remaining TO frequency counter of current_target
                SUBTRACT frequency counter of sticker from remaining
                
                SET new_target TO concatenation of sorted characters from remaining elements
                
                IF new_target is empty
                    RETURN used_stickers + 1
                
                IF new_target NOT IN memo
                    SET memo[new_target] TO used_stickers + 1
                    APPEND (new_target, used_stickers + 1) TO queue
        
        RETURN -1
    END FUNCTION
END CLASS
