CLASS Solution
    FUNCTION jobScheduling(startTime: List of Integer, endTime: List of Integer, profit: List of Integer) RETURNS Integer
        SET jobs TO list of tuples combining startTime, endTime, and profit sorted by end time
        SET end_times TO list of end times extracted from jobs
        INITIALIZE dp AS list of zeros with length one greater than number of jobs
        FOR i FROM 1 TO LENGTH OF jobs
            SET j TO position found by binary searching right in end_times for jobs[i-1]'s start time
            SET dp[i] TO maximum of dp[i-1] and dp[j] plus jobs[i-1]'s profit
        RETURN last element in dp
    END FUNCTION
END CLASS
