CLASS Solution
    FUNCTION getFactors(n)
        FUNCTION backtrack(start, target)
            FOR i FROM start TO INTEGER SQUARE ROOT OF target INCLUSIVE
                IF target MODULO i EQUALS 0 THEN
                    APPEND path CONCATENATED WITH [i, target DIVIDED BY i] TO result
                    APPEND i TO path
                    CALL backtrack(i, target DIVIDED BY i)
                    REMOVE LAST ELEMENT FROM path
                END IF
            END FOR
        END FUNCTION

        SET result TO EMPTY LIST
        SET path TO EMPTY LIST
        CALL backtrack(2, n)
        RETURN result
    END FUNCTION
END CLASS
