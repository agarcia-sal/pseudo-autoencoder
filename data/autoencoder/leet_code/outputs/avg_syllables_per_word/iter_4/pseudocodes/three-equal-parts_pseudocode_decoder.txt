CLASS Solution
    FUNCTION threeEqualParts(arr)
        SET total_ones TO sum of all elements in arr
        
        IF total_ones EQUALS 0 THEN
            RETURN [0, 2]
        END IF
        
        IF total_ones MODULO 3 NOT EQUALS 0 THEN
            RETURN [-1, -1]
        END IF
        
        SET part_ones TO total_ones DIVIDED BY 3
        
        SET first_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] EQUALS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones EQUALS 1 THEN
                    SET first_part_index TO i
                END IF
                IF count_ones EQUALS part_ones THEN
                    BREAK LOOP
                END IF
            END IF
        END FOR
        
        SET second_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] EQUALS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones EQUALS part_ones PLUS 1 THEN
                    SET second_part_index TO i
                    BREAK LOOP
                END IF
            END IF
        END FOR
        
        SET third_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] EQUALS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones EQUALS 2 TIMES part_ones PLUS 1 THEN
                    SET third_part_index TO i
                    BREAK LOOP
                END IF
            END IF
        END FOR
        
        SET n TO LENGTH OF arr MINUS third_part_index
        
        IF first_part_index PLUS n LESS THAN OR EQUAL TO LENGTH OF arr AND
           second_part_index PLUS n LESS THAN OR EQUAL TO LENGTH OF arr AND
           SUBLIST OF arr FROM first_part_index TO first_part_index PLUS n MINUS 1 EQUALS
           SUBLIST OF arr FROM second_part_index TO second_part_index PLUS n MINUS 1 AND
           SUBLIST OF arr FROM second_part_index TO second_part_index PLUS n MINUS 1 EQUALS
           SUBLIST OF arr FROM third_part_index TO END THEN
            RETURN [first_part_index PLUS n MINUS 1, second_part_index PLUS n]
        END IF
        
        RETURN [-1, -1]
    END FUNCTION
END CLASS
