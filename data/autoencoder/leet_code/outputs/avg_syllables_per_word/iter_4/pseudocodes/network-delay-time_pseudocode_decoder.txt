CLASS Solution
    FUNCTION networkDelayTime(times, n, k)
        SET graph TO empty dictionary mapping nodes 1 to n to empty lists
        FOR each (u, v, w) IN times
            APPEND (v, w) TO graph[u]
        END FOR

        SET dist TO dictionary mapping nodes 1 to n to infinity
        SET dist[k] TO 0

        SET min_heap TO list containing tuple (0, k)

        WHILE min_heap is not empty
            REMOVE smallest element FROM min_heap AS (current_dist, node)

            IF current_dist > dist[node]
                CONTINUE

            FOR each (neighbor, weight) IN graph[node]
                SET distance TO current_dist + weight

                IF distance < dist[neighbor]
                    SET dist[neighbor] TO distance
                    ADD (distance, neighbor) TO min_heap
                END IF
            END FOR
        END WHILE

        SET max_time TO maximum value in dist values

        IF max_time IS infinity
            RETURN -1
        ELSE
            RETURN max_time
        END IF
    END FUNCTION
END CLASS
