CLASS Solution
    FUNCTION minMutation(start, end, bank)
        SET s TO SET OF bank
        SET q TO QUEUE WITH ITEM (start, 0)
        SET mp TO DICTIONARY MAPPING 
            'A' TO "TCG",
            'T' TO "ACG",
            'C' TO "ATG",
            'G' TO "ATC"
        WHILE q IS NOT EMPTY
            SET t, step TO REMOVE FIRST ITEM FROM q
            IF t IS EQUAL TO end
                RETURN step
            FOR EACH i, v IN ENUMERATE t
                FOR EACH j IN mp[v]
                    SET next TO CONCATENATION OF SUBSTRING t FROM START TO i, j, AND SUBSTRING t FROM i+1 TO END
                    IF next IS IN s
                        ADD (next, step + 1) TO q
                        REMOVE next FROM s
        RETURN -1
    END FUNCTION
END CLASS
