CLASS Solution
    FUNCTION tallestBillboard(rods)
        SET dp TO dictionary with key 0 and value 0
        
        FOR each rod IN rods
            SET new_dp TO copy of dp
            FOR each key d AND value y IN dp
                SET new_dp[d + rod] TO maximum of new_dp[d + rod] if exists otherwise 0, AND y
                SET new_dp[absolute value of (d - rod)] TO maximum of new_dp[absolute value of (d - rod)] if exists otherwise 0, AND y plus minimum of d AND rod
            END FOR
            SET dp TO new_dp
        END FOR
        
        RETURN dp value at key 0
    END FUNCTION
END CLASS
