CLASS Solution
    FUNCTION canFinish(numCourses, prerequisites)
        SET graph TO a mapping with default empty list
        SET indegree TO a list of zeros with length numCourses
        
        FOR each pair (course, prereq) IN prerequisites
            APPEND course TO graph at key prereq
            INCREMENT indegree at index course BY 1
        
        SET queue TO a list of courses WHERE indegree is 0
        
        WHILE queue is not empty
            REMOVE the first element FROM queue AND SET TO current_course
            DECREMENT numCourses BY 1
            
            FOR each next_course IN graph at key current_course
                DECREMENT indegree at index next_course BY 1
                IF indegree at index next_course IS 0
                    APPEND next_course TO queue
        
        RETURN True IF numCourses EQUALS 0 ELSE False
    END FUNCTION
END CLASS
