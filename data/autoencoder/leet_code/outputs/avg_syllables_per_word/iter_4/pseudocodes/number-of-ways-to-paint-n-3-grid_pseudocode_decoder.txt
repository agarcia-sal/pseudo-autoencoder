CLASS Solution
    FUNCTION numOfWays(n)
        SET MOD TO 10^9 + 7
        SET a TO 6
        SET b TO 6
        FOR i FROM 2 TO n
            SET new_a TO (3 * a + 2 * b) MODULO MOD
            SET new_b TO (2 * a + 2 * b) MODULO MOD
            SET a TO new_a
            SET b TO new_b
        RETURN (a + b) MODULO MOD
    END FUNCTION
END CLASS
