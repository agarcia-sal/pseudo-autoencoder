CLASS Solution
    FUNCTION rotatedDigits(n)
        SET change_digits TO SET OF '2', '5', '6', '9'
        SET same_digits TO SET OF '0', '1', '8'

        FUNCTION is_good_number(num)
            SET num_str TO STRING REPRESENTATION OF num
            IF ALL digits IN num_str ARE IN change_digits UNION same_digits THEN
                IF ANY digit IN num_str IS IN change_digits THEN
                    RETURN TRUE
            RETURN FALSE
        END FUNCTION

        SET good_count TO 0
        FOR i FROM 1 TO n DO
            IF is_good_number(i) THEN
                INCREMENT good_count BY 1
        RETURN good_count
    END FUNCTION
END CLASS
