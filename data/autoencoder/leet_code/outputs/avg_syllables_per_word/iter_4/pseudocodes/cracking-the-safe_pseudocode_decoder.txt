CLASS Solution
    FUNCTION crackSafe(n, k)
        SET visited TO empty set
        SET sequence TO empty list

        FUNCTION dfs(node)
            FOR i FROM 0 TO k - 1
                SET edge TO CONCATENATE node WITH STRING OF i
                IF edge NOT IN visited
                    ADD edge TO visited
                    CALL dfs(SUBSTRING OF edge FROM INDEX 1)
                    APPEND STRING OF i TO sequence
                END IF
            END FOR
        END FUNCTION

        SET start_node TO STRING OF '0' REPEATED (n - 1) TIMES
        CALL dfs(start_node)
        RETURN CONCATENATE ALL ELEMENTS IN REVERSED sequence FOLLOWED BY start_node
    END FUNCTION
END CLASS
