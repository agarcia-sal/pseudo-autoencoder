CLASS Solution
    FUNCTION pourWater(heights: List[int], volume: int, k: int) RETURNS List[int]
        SET n TO LENGTH OF heights

        FOR each droplet FROM 1 TO volume
            SET i TO k
            WHILE i > 0 AND heights[i - 1] <= heights[i]
                DECREMENT i BY 1
            END WHILE

            WHILE i < k AND heights[i] == heights[i + 1]
                INCREMENT i BY 1
            END WHILE

            IF i < k THEN
                INCREMENT heights[i] BY 1
                CONTINUE TO NEXT droplet
            END IF

            SET i TO k
            WHILE i < n - 1 AND heights[i + 1] <= heights[i]
                INCREMENT i BY 1
            END WHILE

            WHILE i > k AND heights[i] == heights[i - 1]
                DECREMENT i BY 1
            END WHILE

            IF i > k THEN
                INCREMENT heights[i] BY 1
                CONTINUE TO NEXT droplet
            END IF

            INCREMENT heights[k] BY 1
        END FOR

        RETURN heights
    END FUNCTION
END CLASS
