CLASS Solution
    FUNCTION maximumSwap(num)
        SET digits TO list of characters in string form of num
        SET last TO empty dictionary
        FOR each index i and digit d IN digits
            SET last[integer value of d] TO i
        FOR each index i and digit d IN digits
            FOR d2 FROM 9 DOWNTO integer value of d + 1
                IF last value for d2 exists AND last[d2] > i THEN
                    SWAP digits[i] WITH digits[last[d2]]
                    RETURN integer formed by concatenating digits
        RETURN num
    END FUNCTION
END CLASS
