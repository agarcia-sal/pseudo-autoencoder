CLASS Solution
    FUNCTION minArea(image, x, y)
        IF image IS EMPTY OR image[0] IS EMPTY
            RETURN 0
        
        SET m TO LENGTH OF image
        SET n TO LENGTH OF image[0]
        
        FUNCTION hasBlackPixelInRow(row)
            FOR EACH pixel IN image[row]
                IF pixel EQUALS '1'
                    RETURN True
            RETURN False
        END FUNCTION
        
        FUNCTION hasBlackPixelInColumn(col)
            FOR row FROM 0 TO m - 1
                IF image[row][col] EQUALS '1'
                    RETURN True
            RETURN False
        END FUNCTION
        
        FUNCTION findTop()
            SET low TO 0
            SET high TO x
            WHILE low LESS THAN high
                SET mid TO (low + high) DIVIDED BY 2
                IF hasBlackPixelInRow(mid)
                    SET high TO mid
                ELSE
                    SET low TO mid + 1
            RETURN low
        END FUNCTION
        
        FUNCTION findBottom()
            SET low TO x
            SET high TO m - 1
            WHILE low LESS THAN high
                SET mid TO (low + high + 1) DIVIDED BY 2
                IF hasBlackPixelInRow(mid)
                    SET low TO mid
                ELSE
                    SET high TO mid - 1
            RETURN low
        END FUNCTION
        
        FUNCTION findLeft()
            SET low TO 0
            SET high TO y
            WHILE low LESS THAN high
                SET mid TO (low + high) DIVIDED BY 2
                IF hasBlackPixelInColumn(mid)
                    SET high TO mid
                ELSE
                    SET low TO mid + 1
            RETURN low
        END FUNCTION
        
        FUNCTION findRight()
            SET low TO y
            SET high TO n - 1
            WHILE low LESS THAN high
                SET mid TO (low + high + 1) DIVIDED BY 2
                IF hasBlackPixelInColumn(mid)
                    SET low TO mid
                ELSE
                    SET high TO mid - 1
            RETURN low
        END FUNCTION
        
        SET top TO findTop()
        SET bottom TO findBottom()
        SET left TO findLeft()
        SET right TO findRight()
        
        RETURN (bottom - top + 1) TIMES (right - left + 1)
    END FUNCTION
END CLASS
