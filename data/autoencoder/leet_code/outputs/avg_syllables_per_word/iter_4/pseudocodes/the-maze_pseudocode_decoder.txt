CLASS Solution
    FUNCTION hasPath(maze, start, destination)
        SET directions TO [(0, 1), (1, 0), (0, -1), (-1, 0)]
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze[0]
        SET visited TO EMPTY SET
        
        FUNCTION dfs(x, y)
            IF (x, y) IN visited THEN
                RETURN False
            IF [x, y] EQUALS destination THEN
                RETURN True
            ADD (x, y) TO visited
            
            FOR EACH (dx, dy) IN directions DO
                SET nx, ny TO x, y
                WHILE nx + dx IS BETWEEN 0 AND m - 1 AND ny + dy IS BETWEEN 0 AND n - 1 AND maze[nx + dx][ny + dy] EQUALS 0 DO
                    SET nx TO nx + dx
                    SET ny TO ny + dy
                IF dfs(nx, ny) THEN
                    RETURN True
            RETURN False
        END FUNCTION
        
        RETURN dfs(start[0], start[1])
    END FUNCTION
END CLASS
