CLASS Solution
    FUNCTION fourSum(nums: List[int], target: int) RETURNS List[List[int]]
        CALL nums.sort()
        SET n TO LENGTH OF nums
        SET quadruplets TO empty list

        FOR i FROM 0 TO n - 3
            IF i > 0 AND nums[i] EQUALS nums[i - 1]
                CONTINUE
            FOR j FROM i + 1 TO n - 2
                IF j > i + 1 AND nums[j] EQUALS nums[j - 1]
                    CONTINUE
                SET left TO j + 1
                SET right TO n - 1
                WHILE left LESS THAN right
                    SET total TO nums[i] PLUS nums[j] PLUS nums[left] PLUS nums[right]
                    IF total EQUALS target
                        APPEND [nums[i], nums[j], nums[left], nums[right]] TO quadruplets
                        WHILE left LESS THAN right AND nums[left] EQUALS nums[left + 1]
                            INCREMENT left BY 1
                        WHILE left LESS THAN right AND nums[right] EQUALS nums[right - 1]
                            DECREMENT right BY 1
                        INCREMENT left BY 1
                        DECREMENT right BY 1
                    ELSE IF total LESS THAN target
                        INCREMENT left BY 1
                    ELSE
                        DECREMENT right BY 1

        RETURN quadruplets
    END FUNCTION
END CLASS
