CLASS Solution
    FUNCTION coinChange(coins, amount)
        SET dp TO list of size (amount + 1) filled with (amount + 1)
        SET dp[0] TO 0
        FOR a FROM 1 TO amount
            FOR each coin IN coins
                IF a - coin >= 0 THEN
                    SET dp[a] TO minimum of dp[a] AND (dp[a - coin] + 1)
        IF dp[amount] IS NOT EQUAL TO amount + 1 THEN
            RETURN dp[amount]
        ELSE
            RETURN -1
    END FUNCTION
END CLASS
