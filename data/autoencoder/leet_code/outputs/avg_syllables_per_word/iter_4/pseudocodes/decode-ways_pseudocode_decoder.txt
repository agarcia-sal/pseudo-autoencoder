CLASS Solution
    FUNCTION numDecodings(s)
        IF s IS empty OR s[0] EQUALS '0'
            RETURN 0
        END IF

        SET dp TO list of zeros with length LENGTH OF s plus 1
        SET dp[0] TO 1
        SET dp[1] TO 1

        FOR i FROM 2 TO LENGTH OF s
            IF s at position i minus 1 IS NOT '0'
                INCREMENT dp[i] BY dp[i minus 1]
            END IF

            SET two_digit TO integer value of substring s from i minus 2 to i
            IF two_digit IS GREATER THAN OR EQUAL TO 10 AND LESS THAN OR EQUAL TO 26
                INCREMENT dp[i] BY dp[i minus 2]
            END IF
        END FOR

        RETURN dp at last position
    END FUNCTION
END CLASS
