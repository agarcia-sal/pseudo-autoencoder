CLASS Solution
    FUNCTION minDeletionSize(strs)
        SET m TO LENGTH OF strs
        SET n TO LENGTH OF strs[0]
        SET dp TO LIST OF LENGTH n FILLED WITH 1

        FOR i FROM 1 TO n - 1
            FOR j FROM 0 TO i - 1
                IF FOR ALL k FROM 0 TO m - 1, strs[k][j] IS LESS THAN OR EQUAL TO strs[k][i]
                    SET dp[i] TO MAXIMUM OF dp[i] AND (dp[j] + 1)
        END FOR
        END FOR

        SET result TO n MINUS MAXIMUM VALUE IN dp
        RETURN result
    END FUNCTION
END CLASS
