CLASS Solution
    FUNCTION monotoneIncreasingDigits(n)
        SET digits TO list of characters in string representation of n
        SET length TO length of digits
        SET marker TO length
        FOR i FROM length - 1 DOWNTO 1
            IF digits[i] < digits[i - 1] THEN
                SET digits[i - 1] TO string of (integer value of digits[i - 1] MINUS 1)
                SET marker TO i
        FOR i FROM marker TO length - 1
            SET digits[i] TO '9'
        RETURN integer value of concatenation of digits
    END FUNCTION
END CLASS
