CLASS Solution
    FUNCTION minCost(costs)
        IF costs IS EMPTY THEN
            RETURN 0
        END IF

        SET prev_red, prev_blue, prev_green TO costs[0][0], costs[0][1], costs[0][2]

        FOR i FROM 1 TO LENGTH OF costs MINUS 1
            SET current_red TO costs[i][0] PLUS MINIMUM OF prev_blue AND prev_green
            SET current_blue TO costs[i][1] PLUS MINIMUM OF prev_red AND prev_green
            SET current_green TO costs[i][2] PLUS MINIMUM OF prev_red AND prev_blue

            SET prev_red, prev_blue, prev_green TO current_red, current_blue, current_green
        END FOR

        RETURN MINIMUM OF prev_red, prev_blue, prev_green
    END FUNCTION
END CLASS
