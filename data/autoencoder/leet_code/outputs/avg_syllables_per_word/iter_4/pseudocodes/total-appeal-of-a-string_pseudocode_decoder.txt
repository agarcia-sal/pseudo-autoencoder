CLASS Solution
    FUNCTION appealSum(s)
        SET last TO list of -1 repeated 26 times
        SET total_appeal TO 0
        SET current_appeal TO 0
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            SET char TO s AT POSITION i
            SET index TO numerical position of char in alphabet starting from 0
            INCREMENT current_appeal BY (i MINUS last AT POSITION index)
            INCREMENT total_appeal BY current_appeal
            SET last AT POSITION index TO i
        RETURN total_appeal
    END FUNCTION
END CLASS
