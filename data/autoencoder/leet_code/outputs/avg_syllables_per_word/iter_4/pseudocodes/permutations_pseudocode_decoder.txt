CLASS Solution
    FUNCTION permute(nums)
        FUNCTION backtrack(path, remaining)
            IF remaining IS EMPTY
                APPEND path TO permutations
                RETURN
            END IF
            FOR i FROM 0 TO LENGTH OF remaining MINUS 1
                SET new_path TO path PLUS remaining[i]
                SET new_remaining TO SLICE remaining FROM START TO i CONCATENATED WITH SLICE remaining FROM i+1 TO END
                CALL backtrack(new_path, new_remaining)
            END FOR
        END FUNCTION

        SET permutations TO EMPTY LIST
        CALL backtrack(EMPTY LIST, nums)
        RETURN permutations
    END FUNCTION
END CLASS
