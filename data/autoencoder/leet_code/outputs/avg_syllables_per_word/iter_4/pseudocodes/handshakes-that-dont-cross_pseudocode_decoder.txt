CLASS Solution
    FUNCTION numberOfWays(numPeople)
        SET MOD TO 10 POWER 9 PLUS 7
        SET dp TO LIST OF ZEROS OF SIZE numPeople PLUS 1
        SET dp[0] TO 1
        SET dp[2] TO 1

        FOR i FROM 4 TO numPeople STEP 2
            FOR j FROM 0 TO i MINUS 1 STEP 2
                SET dp[i] TO (dp[i] PLUS dp[j] TIMES dp[i MINUS 2 MINUS j]) MODULO MOD
        RETURN dp[numPeople]
    END FUNCTION
END CLASS
