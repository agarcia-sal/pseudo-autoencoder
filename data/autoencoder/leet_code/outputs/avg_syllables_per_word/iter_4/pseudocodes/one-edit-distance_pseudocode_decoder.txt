CLASS Solution
    FUNCTION isOneEditDistance(s, t)
        IF LENGTH OF s > LENGTH OF t THEN
            RETURN CALL isOneEditDistance WITH (t, s)
        SET diff TO LENGTH OF t MINUS LENGTH OF s
        IF diff > 1 THEN
            RETURN False
        IF diff == 0 THEN
            SET mismatch_count TO 0
            FOR EACH pair (c1, c2) IN ZIP OF s AND t
                IF c1 IS NOT EQUAL TO c2 THEN
                    INCREMENT mismatch_count BY 1
            RETURN mismatch_count EQUALS 1
        ELSE
            FOR i FROM 0 TO LENGTH OF s MINUS 1
                IF CHARACTER AT s[i] IS NOT EQUAL TO CHARACTER AT t[i] THEN
                    RETURN SUBSTRING OF s FROM i TO END IS EQUAL TO SUBSTRING OF t FROM i+1 TO END
            RETURN True
    END FUNCTION
END CLASS
