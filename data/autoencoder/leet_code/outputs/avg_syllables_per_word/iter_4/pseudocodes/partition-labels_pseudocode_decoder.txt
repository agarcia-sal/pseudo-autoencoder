CLASS Solution
    FUNCTION partitionLabels(s)
        SET last_occurrence TO empty dictionary
        FOR each index, character IN s
            SET last_occurrence[character] TO index

        SET start TO 0
        SET end TO 0
        SET partition_sizes TO empty list

        FOR each index, character IN s
            SET end TO maximum of end and last_occurrence[character]

            IF index EQUALS end THEN
                APPEND (end MINUS start PLUS 1) TO partition_sizes
                SET start TO index PLUS 1

        RETURN partition_sizes
    END FUNCTION
END CLASS
