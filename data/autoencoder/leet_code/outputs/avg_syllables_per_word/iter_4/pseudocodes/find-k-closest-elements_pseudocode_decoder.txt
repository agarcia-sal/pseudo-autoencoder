CLASS Solution
    FUNCTION findClosestElements(arr, k, x)
        SET left TO position where x would be inserted in arr MINUS 1
        SET right TO left PLUS 1
        WHILE right MINUS left MINUS 1 LESS THAN k
            IF left EQUALS -1 OR (right LESS THAN LENGTH OF arr AND absolute value of arr[right] MINUS x LESS THAN absolute value of arr[left] MINUS x)
                INCREMENT right BY 1
            ELSE
                DECREMENT left BY 1
        RETURN subarray of arr FROM left PLUS 1 TO right
    END FUNCTION
END CLASS
