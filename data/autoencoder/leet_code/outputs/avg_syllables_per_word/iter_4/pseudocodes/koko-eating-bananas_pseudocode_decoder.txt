CLASS Solution
    FUNCTION minEatingSpeed(piles, h)
        FUNCTION canFinish(k)
            SET hours TO 0
            FOR each pile IN piles
                SET hours TO hours PLUS CEILING OF (pile DIVIDED BY k)
            RETURN hours IS LESS THAN OR EQUAL TO h
        END FUNCTION

        SET left TO 1
        SET right TO MAXIMUM ELEMENT IN piles

        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO (left PLUS right) DIVIDED BY 2
            IF canFinish(mid) IS TRUE
                SET right TO mid MINUS 1
            ELSE
                SET left TO mid PLUS 1
        END WHILE

        RETURN left
    END FUNCTION
END CLASS
