CLASS Solution
    FUNCTION originalDigits(s: string) RETURNS string
        IMPORT Counter FROM collections
        
        SET count TO Counter OF characters IN s
        
        SET digits TO list OF size 10 FILLED WITH 0
        
        SET digits[0] TO count OF 'z'
        SET digits[2] TO count OF 'w'
        SET digits[4] TO count OF 'u'
        SET digits[6] TO count OF 'x'
        SET digits[8] TO count OF 'g'
        
        SET digits[3] TO count OF 'h' MINUS digits[8]
        SET digits[5] TO count OF 'f' MINUS digits[4]
        SET digits[7] TO count OF 's' MINUS digits[6]
        SET digits[9] TO count OF 'i' MINUS digits[5] MINUS digits[6] MINUS digits[8]
        SET digits[1] TO count OF 'o' MINUS digits[0] MINUS digits[2] MINUS digits[4]
        
        SET result TO empty list
        FOR i FROM 0 TO 9
            APPEND string REPEATED digits[i] TIMES OF i TO result
        
        RETURN concatenation OF elements IN result
    END FUNCTION
END CLASS
