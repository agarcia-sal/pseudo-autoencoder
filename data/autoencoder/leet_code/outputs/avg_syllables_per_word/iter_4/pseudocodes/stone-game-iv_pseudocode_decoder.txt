CLASS Solution
    FUNCTION winnerSquareGame(n)
        SET dp TO list of False with length n + 1
        FOR i FROM 1 TO n
            SET j TO 1
            WHILE j * j LESS THAN OR EQUAL TO i
                IF dp at position (i MINUS j * j) IS False
                    SET dp at position i TO True
                    BREAK out of WHILE loop
                INCREMENT j BY 1
        RETURN dp at position n
    END FUNCTION
END CLASS
