CLASS Solution
    FUNCTION recoverArray(nums)
        SORT nums
        SET n TO LENGTH OF nums
        
        SET first TO nums[0]
        
        FOR i FROM 1 TO n - 1
            SET k TO (nums[i] MINUS first) DIVIDED BY 2
            IF k <= 0 OR (first PLUS 2 TIMES k) NOT EQUAL TO nums[i]
                CONTINUE TO NEXT i
            
            SET count TO COUNTER OF nums
            SET arr TO EMPTY LIST
            
            FOR each x IN nums
                IF count[x] EQUALS 0
                    CONTINUE TO NEXT x
                IF count[x PLUS 2 TIMES k] EQUALS 0
                    BREAK FROM LOOP
                
                APPEND (x PLUS k) TO arr
                DECREMENT count[x] BY 1
                DECREMENT count[x PLUS 2 TIMES k] BY 1
            
            IF LENGTH OF arr EQUALS n DIVIDED BY 2
                RETURN arr
        
        RETURN EMPTY LIST
END CLASS
