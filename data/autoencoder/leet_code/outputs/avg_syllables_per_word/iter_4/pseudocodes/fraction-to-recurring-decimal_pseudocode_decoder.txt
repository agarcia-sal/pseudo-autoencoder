CLASS Solution
    FUNCTION fractionToDecimal(numerator, denominator)
        IF (numerator < 0) XOR (denominator < 0) THEN
            SET sign TO "-"
        ELSE
            SET sign TO ""
        END IF

        SET numerator TO absolute value of numerator
        SET denominator TO absolute value of denominator

        SET quotient, remainder TO INTEGER DIVISION and MODULO of numerator BY denominator
        SET integer_part TO STRING representation of quotient

        IF remainder EQUALS 0 THEN
            RETURN concatenation of sign and integer_part
        END IF

        SET seen_remainders TO empty dictionary
        SET fractional_part TO empty list

        WHILE remainder IS NOT 0 DO
            IF remainder EXISTS IN seen_remainders THEN
                SET start TO seen_remainders[remainder]
                SET non_repeating TO concatenation of fractional_part elements FROM start 0 UP TO start
                SET repeating TO concatenation of fractional_part elements FROM start TO END
                RETURN concatenation of sign, integer_part, ".", non_repeating, "(", repeating, ")"
            END IF

            SET seen_remainders[remainder] TO LENGTH of fractional_part

            MULTIPLY remainder BY 10
            SET quotient, remainder TO INTEGER DIVISION and MODULO of remainder BY denominator
            APPEND STRING representation of quotient TO fractional_part
        END WHILE

        RETURN concatenation of sign, integer_part, ".", concatenation of all elements in fractional_part
    END FUNCTION
END CLASS
