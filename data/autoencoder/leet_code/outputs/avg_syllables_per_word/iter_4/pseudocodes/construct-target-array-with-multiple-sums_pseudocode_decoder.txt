CLASS Solution
    FUNCTION isPossible(target: List of Integer) RETURNS Boolean
        SET max_heap TO empty list
        FOR each num IN target
            APPEND negative of num TO max_heap
        CALL heapify on max_heap
        
        SET total_sum TO sum of elements in target
        
        WHILE max_heap[0] < negative 1
            SET largest TO negative of heappop from max_heap
            SET rest_sum TO total_sum MINUS largest
            
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS 0
                RETURN False
            
            SET new_value TO largest MODULO rest_sum
            IF new_value EQUALS 0 AND rest_sum NOT EQUAL TO 1
                RETURN False
            
            SET total_sum TO rest_sum PLUS new_value
            CALL heappush max_heap WITH negative of new_value
        
        RETURN True
    END FUNCTION
END CLASS
