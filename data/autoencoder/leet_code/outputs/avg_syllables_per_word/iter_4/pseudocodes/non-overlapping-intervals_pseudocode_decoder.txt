CLASS Solution
    FUNCTION eraseOverlapIntervals(intervals)
        IF intervals IS empty
            RETURN 0
        END IF

        SORT intervals BY the end time of each interval

        SET end TO the end time of the first interval
        SET count TO 0

        FOR i FROM 1 TO length of intervals MINUS 1
            IF start time of intervals[i] IS GREATER THAN OR EQUAL TO end
                SET end TO the end time of intervals[i]
            ELSE
                INCREMENT count BY 1
            END IF
        END FOR

        RETURN count
    END FUNCTION
END CLASS
