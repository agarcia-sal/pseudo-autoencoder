CLASS Solution
    FUNCTION nextGreaterElement(n)
        SET digits TO list of characters from string representation of n
        SET length TO size of digits

        SET i TO length MINUS 2
        WHILE i IS GREATER THAN OR EQUAL TO 0 AND digits[i] IS GREATER THAN OR EQUAL TO digits[i PLUS 1]
            DECREMENT i BY 1

        IF i IS EQUAL TO -1
            RETURN -1

        SET j TO length MINUS 1
        WHILE digits[j] IS LESS THAN OR EQUAL TO digits[i]
            DECREMENT j BY 1

        SWAP digits[i] WITH digits[j]

        SET digits TO CONCATENATION OF digits UP TO AND INCLUDING i AND REVERSE OF digits FROM i PLUS 1 TO END

        SET result TO integer formed by concatenating characters in digits

        IF result IS GREATER THAN 2 TO THE POWER OF 31 MINUS 1
            RETURN -1

        RETURN result
    END FUNCTION
END CLASS
