CLASS Solution
    FUNCTION minCostII(costs)
        IF costs IS EMPTY
            RETURN 0
        SET n TO LENGTH OF costs
        SET k TO LENGTH OF costs[0]
        IF k EQUALS 1
            IF n EQUALS 1
                RETURN costs[0][0]
            ELSE
                RETURN infinity
        FOR i FROM 1 TO n MINUS 1
            SET min_cost TO infinity
            SET min_cost_index TO -1
            SET second_min_cost TO infinity
            FOR j FROM 0 TO k MINUS 1
                IF costs[i MINUS 1][j] LESS THAN min_cost
                    SET second_min_cost TO min_cost
                    SET min_cost TO costs[i MINUS 1][j]
                    SET min_cost_index TO j
                ELSE IF costs[i MINUS 1][j] LESS THAN second_min_cost
                    SET second_min_cost TO costs[i MINUS 1][j]
            FOR j FROM 0 TO k MINUS 1
                IF j EQUALS min_cost_index
                    SET costs[i][j] TO costs[i][j] PLUS second_min_cost
                ELSE
                    SET costs[i][j] TO costs[i][j] PLUS min_cost
        RETURN MINIMUM VALUE IN costs[n MINUS 1]
END CLASS
