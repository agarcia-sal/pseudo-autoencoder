CLASS Solution
    FUNCTION kthSmallestPrimeFraction(arr, k)
        SET n TO LENGTH OF arr
        SET min_heap TO empty list

        FOR i FROM 0 TO n - 2
            PUSH (arr[i] DIVIDED BY arr[n - 1], i, n - 1) INTO min_heap
        END FOR

        FOR count FROM 1 TO k - 1
            POP smallest_fraction, i, j FROM min_heap
            IF j - 1 > i THEN
                PUSH (arr[i] DIVIDED BY arr[j - 1], i, j - 1) INTO min_heap
            END IF
        END FOR

        POP value, i, j FROM min_heap
        RETURN [arr[i], arr[j]]
    END FUNCTION
END CLASS
