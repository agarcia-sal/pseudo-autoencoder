CLASS Solution
    FUNCTION findRightInterval(intervals)
        SET start_times TO SORTED list of (interval start, index) FOR each interval WITH index IN intervals
        SET result TO empty list
        FOR each (_, end) IN intervals
            SET idx TO POSITION in start_times WHERE (end,) CAN BE INSERTED TO MAINTAIN ORDER
            IF idx IS LESS THAN LENGTH OF start_times THEN
                APPEND start_times[idx][1] TO result
            ELSE
                APPEND -1 TO result
        RETURN result
    END FUNCTION
END CLASS
