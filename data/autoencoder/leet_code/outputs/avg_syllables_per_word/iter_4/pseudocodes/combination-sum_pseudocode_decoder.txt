CLASS Solution
    FUNCTION combinationSum(candidates, target)
        FUNCTION backtrack(start, target, path)
            IF target EQUALS 0 THEN
                APPEND path TO result
                RETURN
            END IF
            IF target LESS THAN 0 THEN
                RETURN
            END IF
            FOR i FROM start TO LENGTH OF candidates MINUS 1 DO
                CALL backtrack(i, target MINUS candidates[i], path PLUS [candidates[i]])
            END FOR
        END FUNCTION

        SET result TO empty list
        CALL backtrack(0, target, empty list)
        RETURN result
    END FUNCTION
END CLASS
