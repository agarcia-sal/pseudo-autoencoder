CLASS Solution
    FUNCTION shortestDistance(maze, start, destination)
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze[0]
        SET directions TO LIST OF (1, 0), (-1, 0), (0, 1), (0, -1)
        CONVERT start TO tuple
        CONVERT destination TO tuple
        SET visited TO empty set
        SET pq TO list containing (0, start)

        WHILE pq IS NOT empty
            REMOVE smallest element FROM pq AND DESTRUCTURE TO dist, (x, y)

            IF (x, y) EQUALS destination
                RETURN dist

            IF (x, y) IS IN visited
                CONTINUE

            ADD (x, y) TO visited

            FOR each (dx, dy) IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO 0

                WHILE new_x + dx IS AT LEAST 0 AND LESS THAN m
                      AND new_y + dy IS AT LEAST 0 AND LESS THAN n
                      AND maze[new_x + dx][new_y + dy] EQUALS 0
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY 1

                IF (new_x, new_y) IS NOT IN visited
                    ADD (dist + count, (new_x, new_y)) TO pq

        RETURN -1
    END FUNCTION
END CLASS
