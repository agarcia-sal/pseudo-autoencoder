CLASS Solution
    FUNCTION criticalConnections(n, connections)
        FUNCTION dfs(node, parent, disc, low, time, graph, critical)
            SET disc[node] TO time[0]
            SET low[node] TO time[0]
            INCREMENT time[0] BY 1

            FOR each neighbor IN graph[node]
                IF disc[neighbor] IS -1 THEN
                    CALL dfs(neighbor, node, disc, low, time, graph, critical)
                    SET low[node] TO MINIMUM OF low[node] AND low[neighbor]

                    IF low[neighbor] GREATER THAN disc[node] THEN
                        APPEND [node, neighbor] TO critical
                ELSE IF neighbor IS NOT parent THEN
                    SET low[node] TO MINIMUM OF low[node] AND disc[neighbor]
        END FUNCTION

        SET graph TO a list of empty lists with length n

        FOR each pair u, v IN connections
            APPEND v TO graph[u]
            APPEND u TO graph[v]

        SET disc TO a list of -1 values with length n
        SET low TO a list of -1 values with length n
        SET time TO a list containing 0

        SET critical TO an empty list

        FOR node FROM 0 TO n MINUS 1
            IF disc[node] IS -1 THEN
                CALL dfs(node, -1, disc, low, time, graph, critical)

        RETURN critical
    END FUNCTION
END CLASS
