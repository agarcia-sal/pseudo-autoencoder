CLASS Solution
    FUNCTION isSelfCrossing(distance)
        SET n TO LENGTH OF distance
        IF n LESS THAN 4 THEN
            RETURN False
        FOR i FROM 3 TO n MINUS 1
            IF distance[i] GREATER THAN OR EQUAL TO distance[i MINUS 2] AND distance[i MINUS 1] LESS THAN OR EQUAL TO distance[i MINUS 3] THEN
                RETURN True
            IF i GREATER THAN OR EQUAL TO 4 AND distance[i MINUS 1] EQUALS distance[i MINUS 3] AND distance[i] PLUS distance[i MINUS 4] GREATER THAN OR EQUAL TO distance[i MINUS 2] THEN
                RETURN True
            IF i GREATER THAN OR EQUAL TO 5 AND distance[i MINUS 2] GREATER THAN OR EQUAL TO distance[i MINUS 4] AND distance[i MINUS 3] GREATER THAN OR EQUAL TO distance[i MINUS 1] AND distance[i MINUS 1] PLUS distance[i MINUS 5] GREATER THAN OR EQUAL TO distance[i MINUS 3] AND distance[i] PLUS distance[i MINUS 4] GREATER THAN OR EQUAL TO distance[i MINUS 2] THEN
                RETURN True
        RETURN False
    END FUNCTION
END CLASS
