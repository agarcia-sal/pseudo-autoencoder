CLASS Solution
    FUNCTION splitArray(nums)
        SET n TO LENGTH OF nums
        IF n < 7 THEN
            RETURN False
        END IF

        SET prefix_sum TO LIST OF SIZE n WITH ALL ELEMENTS 0
        SET prefix_sum[0] TO nums[0]
        FOR i FROM 1 TO n - 1
            SET prefix_sum[i] TO prefix_sum[i - 1] + nums[i]
        END FOR

        FOR j FROM 3 TO n - 4
            SET seen TO EMPTY SET
            FOR i FROM 1 TO j - 2
                IF prefix_sum[i - 1] == prefix_sum[j - 1] - prefix_sum[i] THEN
                    ADD prefix_sum[i - 1] TO seen
                END IF
            END FOR
            FOR k FROM j + 2 TO n - 2
                IF prefix_sum[n - 1] - prefix_sum[k] == prefix_sum[k - 1] - prefix_sum[j] AND
                   prefix_sum[k - 1] - prefix_sum[j] IN seen THEN
                    RETURN True
                END IF
            END FOR
        END FOR

        RETURN False
    END FUNCTION
END CLASS
