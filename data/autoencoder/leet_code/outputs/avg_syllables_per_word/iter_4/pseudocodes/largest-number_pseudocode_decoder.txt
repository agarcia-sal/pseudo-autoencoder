CLASS Solution
    FUNCTION largestNumber(nums: List of Integer) RETURNS String
        SET nums_str TO list of strings converted from nums

        FUNCTION compare(x: String, y: String) RETURNS Integer
            IF x concatenated with y is greater than y concatenated with x THEN
                RETURN -1
            ELSE IF x concatenated with y is less than y concatenated with x THEN
                RETURN 1
            ELSE
                RETURN 0
        END FUNCTION

        SORT nums_str USING compare as the custom comparator

        IF first element of nums_str is "0" THEN
            RETURN "0"

        RETURN concatenation of all elements in nums_str
    END FUNCTION
END CLASS
