CLASS Solution
    FUNCTION isBipartite(graph)
        SET n TO LENGTH OF graph
        SET color TO LIST OF SIZE n FILLED WITH 0

        FUNCTION dfs(node, c)
            SET color[node] TO c
            FOR each neighbor IN graph[node]
                IF color[neighbor] IS c
                    RETURN False
                IF color[neighbor] IS 0
                    IF dfs(neighbor, NEGATE c) IS False
                        RETURN False
            RETURN True
        END FUNCTION

        FOR i FROM 0 TO n MINUS 1
            IF color[i] IS 0
                IF dfs(i, 1) IS False
                    RETURN False
        RETURN True
    END FUNCTION
END CLASS
