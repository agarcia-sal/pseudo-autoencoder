CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values is empty THEN
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS empty deque
    APPEND root TO queue
    WHILE queue is not empty DO
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] is not None THEN
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] is not None THEN
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p is None AND q is None THEN
        RETURN True
    ELSE IF p is None OR q is None THEN
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val THEN
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION lowestCommonAncestor(self, root: TreeNode, p: TreeNode, q: TreeNode) RETURNS TreeNode
        IF root is None OR root EQUALS p OR root EQUALS q THEN
            RETURN root
        END IF
        SET left TO self.lowestCommonAncestor(root.left, p, q)
        SET right TO self.lowestCommonAncestor(root.right, p, q)
        IF left is not None AND right is not None THEN
            RETURN root
        END IF
        IF left is not None THEN
            RETURN left
        ELSE
            RETURN right
        END IF
    END FUNCTION
END CLASS
