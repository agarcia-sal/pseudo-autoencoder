CLASS Solution
    FUNCTION findTargetSumWays(nums, target)
        FUNCTION dfs(index, current_sum, memo)
            IF index EQUALS LENGTH OF nums THEN
                IF current_sum EQUALS target THEN
                    RETURN 1
                ELSE
                    RETURN 0
            END IF
            
            IF (index, current_sum) EXISTS IN memo THEN
                RETURN memo[(index, current_sum)]
            END IF
            
            SET positive TO CALL dfs(index + 1, current_sum + nums[index], memo)
            SET negative TO CALL dfs(index + 1, current_sum - nums[index], memo)
            
            SET memo[(index, current_sum)] TO positive + negative
            RETURN memo[(index, current_sum)]
        END FUNCTION
        
        RETURN CALL dfs(0, 0, EMPTY DICTIONARY)
    END FUNCTION
END CLASS
