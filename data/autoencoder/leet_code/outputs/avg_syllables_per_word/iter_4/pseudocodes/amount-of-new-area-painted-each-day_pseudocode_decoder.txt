CLASS Solution
    FUNCTION amountPainted(paint: List of List of Integer) RETURNS List of Integer
        SET max_end TO maximum end value from each pair in paint
        SET painted TO list of length max_end + 1 filled with -1
        SET worklog TO empty list

        FOR each pair start, end IN paint
            SET current_start TO start
            SET new_paint TO 0

            WHILE current_start < end
                IF painted at current_start IS -1
                    SET painted at current_start TO end
                    INCREMENT new_paint BY 1
                    INCREMENT current_start BY 1
                ELSE
                    SET current_start TO painted at current_start

            APPEND new_paint TO worklog

        RETURN worklog
    END FUNCTION
END CLASS
