CLASS Solution
    FUNCTION integerBreak(n)
        SET dp TO list of zeros with length n plus 1
        SET dp[1] TO 1
        FOR i FROM 2 TO n
            SET max_product TO 0
            FOR j FROM 1 TO i MINUS 1
                SET max_product TO maximum of max_product, j multiplied by dp[i minus j], and j multiplied by (i minus j)
            SET dp[i] TO max_product
        RETURN dp[n]
    END FUNCTION
END CLASS
