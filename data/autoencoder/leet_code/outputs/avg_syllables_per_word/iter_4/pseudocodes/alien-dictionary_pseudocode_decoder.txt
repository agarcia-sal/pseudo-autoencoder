CLASS Solution
    FUNCTION alienOrder(words: List of String) RETURNS String
        SET graph TO a mapping from character to set of characters
        SET indegree TO a mapping from character to integer
        
        SET all_chars TO the set of all unique characters in words
        
        FOR i FROM 0 TO LENGTH OF words MINUS 2
            SET word1 TO words[i]
            SET word2 TO words[i + 1]
            SET min_length TO MINIMUM OF LENGTH OF word1 AND LENGTH OF word2
            SET found TO FALSE
            
            FOR j FROM 0 TO min_length MINUS 1
                IF character at position j in word1 IS NOT EQUAL TO character at j in word2
                    IF character at j in word2 IS NOT IN graph[word1[j]]
                        ADD character at j in word2 TO graph[word1[j]]
                        INCREMENT indegree[character at j in word2] BY 1
                    SET found TO TRUE
                    BREAK out of inner loop
            IF found IS FALSE AND LENGTH OF word1 IS GREATER THAN LENGTH OF word2
                RETURN empty string
        
        SET queue TO a deque containing all characters in all_chars WHERE indegree of character IS 0
        SET result TO empty list
        
        WHILE queue IS NOT empty
            SET char TO pop left element from queue
            APPEND char TO result
            FOR EACH neighbor IN graph[char]
                DECREMENT indegree[neighbor] BY 1
                IF indegree[neighbor] IS 0
                    APPEND neighbor TO queue
        
        IF LENGTH OF result IS NOT EQUAL TO LENGTH OF all_chars
            RETURN empty string
        
        RETURN concatenation of characters in result
    END FUNCTION
END CLASS
