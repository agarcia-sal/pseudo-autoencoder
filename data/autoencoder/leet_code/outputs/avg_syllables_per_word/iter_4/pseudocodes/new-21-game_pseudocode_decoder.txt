CLASS Solution
    FUNCTION new21Game(n, k, maxPts)
        FUNCTION dfs(i)
            IF i >= k THEN
                RETURN 1 IF i <= n ELSE 0
            END IF
            IF i == k - 1 THEN
                SET numerator TO MINIMUM OF (n - k + 1) AND maxPts
                RETURN numerator DIVIDED BY maxPts
            END IF
            SET result TO dfs(i + 1) PLUS (dfs(i + 1) MINUS dfs(i + maxPts + 1)) DIVIDED BY maxPts
            RETURN result
        END FUNCTION
        RETURN dfs(0)
END CLASS
