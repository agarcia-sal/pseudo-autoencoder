CLASS Solution
    FUNCTION checkPossibility(nums)
        SET modified TO False
        FOR i FROM 1 TO LENGTH OF nums MINUS 1
            IF nums[i] < nums[i - 1] THEN
                IF modified IS True THEN
                    RETURN False
                SET modified TO True
                IF i LESS THAN 2 OR nums[i] GREATER THAN OR EQUAL TO nums[i - 2] THEN
                    SET nums[i - 1] TO nums[i]
                ELSE
                    SET nums[i] TO nums[i - 1]
        RETURN True
