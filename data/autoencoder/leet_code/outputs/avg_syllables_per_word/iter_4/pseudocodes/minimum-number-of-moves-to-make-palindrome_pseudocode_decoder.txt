CLASS Solution
    FUNCTION minMovesToMakePalindrome(s)
        SET s TO list of characters in s
        SET n TO length of s
        SET moves TO 0
        
        WHILE length of s > 1
            FOR i FROM length of s MINUS 1 DOWN TO 1
                IF s[i] EQUALS s[0]
                    FOR j FROM i TO length of s MINUS 2
                        SWAP s[j] WITH s[j + 1]
                        INCREMENT moves BY 1
                    END FOR
                    REMOVE first element from s
                    REMOVE last element from s
                    BREAK out of FOR loop
                END IF
            END FOR
            IF no break occurred in FOR loop
                REMOVE first element from s
            END IF
        END WHILE
        
        RETURN moves
    END FUNCTION
END CLASS
