CLASS Solution
    FUNCTION maxSlidingWindow(nums, k)
        IF nums IS EMPTY OR k EQUALS 0 THEN
            RETURN empty list
        END IF

        SET deq TO empty double-ended queue
        SET result TO empty list

        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            IF deq IS NOT EMPTY AND front element of deq IS LESS THAN i MINUS k PLUS 1 THEN
                REMOVE front element from deq
            END IF

            WHILE deq IS NOT EMPTY AND nums at back element of deq IS LESS THAN nums at i DO
                REMOVE back element from deq
            END WHILE

            ADD i TO back of deq

            IF i IS GREATER THAN OR EQUAL TO k MINUS 1 THEN
                APPEND nums at front element of deq TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
