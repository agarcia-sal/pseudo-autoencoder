CLASS Solution
    FUNCTION findItinerary(tickets)
        SET graph TO empty mapping with default empty list
        FOR each pair (frm, to) IN tickets sorted in reverse order
            APPEND to TO graph[frm]
        SET itinerary TO empty list

        FUNCTION visit(airport)
            WHILE graph[airport] is not empty
                CALL visit with graph[airport].pop()
            APPEND airport TO itinerary
        END FUNCTION

        CALL visit WITH "JFK"
        RETURN itinerary reversed
    END FUNCTION
END CLASS
