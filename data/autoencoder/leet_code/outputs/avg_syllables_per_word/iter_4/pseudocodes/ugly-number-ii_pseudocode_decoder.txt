CLASS Solution
    FUNCTION nthUglyNumber(n)
        SET ugly_numbers TO [1]
        SET i2 TO 0
        SET i3 TO 0
        SET i5 TO 0
        WHILE LENGTH OF ugly_numbers LESS THAN n
            SET next_multiple_of_2 TO ugly_numbers[i2] MULTIPLIED BY 2
            SET next_multiple_of_3 TO ugly_numbers[i3] MULTIPLIED BY 3
            SET next_multiple_of_5 TO ugly_numbers[i5] MULTIPLIED BY 5
            SET next_ugly_number TO MINIMUM OF next_multiple_of_2, next_multiple_of_3, next_multiple_of_5
            APPEND next_ugly_number TO ugly_numbers
            IF next_ugly_number EQUALS next_multiple_of_2
                INCREMENT i2 BY 1
            IF next_ugly_number EQUALS next_multiple_of_3
                INCREMENT i3 BY 1
            IF next_ugly_number EQUALS next_multiple_of_5
                INCREMENT i5 BY 1
        RETURN ugly_numbers AT INDEX n MINUS 1
    END FUNCTION
END CLASS
