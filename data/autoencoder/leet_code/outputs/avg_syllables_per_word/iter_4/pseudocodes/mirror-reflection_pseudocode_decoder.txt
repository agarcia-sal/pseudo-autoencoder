CLASS Solution
    FUNCTION mirrorReflection(p, q)
        SET lcm TO (p MULTIPLIED BY q) DIVIDED BY gcd(p, q)
        SET vertical_reflections TO lcm DIVIDED BY p
        SET horizontal_reflections TO lcm DIVIDED BY q
        IF vertical_reflections MODULO 2 EQUALS 0 THEN
            RETURN 0
        ELSE IF horizontal_reflections MODULO 2 EQUALS 1 THEN
            RETURN 1
        ELSE
            RETURN 2
    END FUNCTION
END CLASS
