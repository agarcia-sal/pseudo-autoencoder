CLASS Solution
    FUNCTION flipgame(fronts, backs)
        SET same TO empty set
        FOR each pair (x, y) IN zip of fronts and backs
            IF x EQUALS y
                ADD x TO same
        SET all_numbers TO set of all elements in fronts and backs combined
        SET good_numbers TO all_numbers MINUS same
        IF good_numbers is not empty
            RETURN minimum value in good_numbers
        ELSE
            RETURN 0
    END FUNCTION
END CLASS
