CLASS Solution
    FUNCTION numTilings(n)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7
        SET dp TO LIST OF SIZE n + 1 WITH ALL VALUES 0
        SET dp1 TO LIST OF SIZE n + 1 WITH ALL VALUES 0

        SET dp[0] TO 1
        SET dp[1] TO 1
        
        IF n IS GREATER THAN OR EQUAL TO 2 THEN
            SET dp[2] TO 2
            SET dp1[2] TO 1
        END IF

        FOR i FROM 3 TO n INCLUSIVE
            SET dp[i] TO (dp[i - 1] PLUS dp[i - 2] PLUS 2 TIMES dp1[i - 1]) MODULO MOD
            SET dp1[i] TO (dp[i - 2] PLUS dp1[i - 1]) MODULO MOD
        END FOR

        RETURN dp[n]
    END FUNCTION
END CLASS
