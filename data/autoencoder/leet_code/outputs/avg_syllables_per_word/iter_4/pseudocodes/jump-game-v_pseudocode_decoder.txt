CLASS Solution
    FUNCTION maxJumps(arr, d)
        DEFINE dp(i) WITH CACHE
            SET max_jumps TO 1
            FOR x FROM 1 TO d
                IF i + x IS LESS THAN LENGTH OF arr THEN
                    IF arr[i + x] IS GREATER THAN OR EQUAL TO arr[i] THEN
                        BREAK LOOP
                    END IF
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 + dp(i + x))
                END IF
            END FOR
            FOR x FROM 1 TO d
                IF i - x IS GREATER THAN OR EQUAL TO 0 THEN
                    IF arr[i - x] IS GREATER THAN OR EQUAL TO arr[i] THEN
                        BREAK LOOP
                    END IF
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 + dp(i - x))
                END IF
            END FOR
            RETURN max_jumps
        END FUNCTION
        RETURN MAXIMUM OF dp(i) FOR i IN RANGE OF LENGTH OF arr
    END FUNCTION
END CLASS
