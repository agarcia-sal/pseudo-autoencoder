CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values : list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    CREATE empty queue
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION buildTree(inorder : List[int], postorder : List[int]) RETURNS TreeNode
        IF inorder IS EMPTY OR postorder IS EMPTY
            RETURN None
        SET root_val TO REMOVE LAST ELEMENT FROM postorder
        SET root TO TreeNode(root_val)
        SET root_index TO POSITION OF root_val IN inorder
        SET root.right TO SELF.buildTree(SUBLIST OF inorder FROM root_index + 1 TO END, postorder)
        SET root.left TO SELF.buildTree(SUBLIST OF inorder FROM START TO root_index, postorder)
        RETURN root
    END FUNCTION
END CLASS
