CLASS Solution
    FUNCTION generateAbbreviations(word)
        FUNCTION backtrack(index, path, count)
            IF index EQUALS LENGTH OF word THEN
                IF count GREATER THAN 0 THEN
                    APPEND STRING REPRESENTATION OF count TO path
                END IF
                APPEND CONCATENATION OF path TO result
            ELSE
                IF count GREATER THAN 0 THEN
                    CALL backtrack WITH (index + 1, path PLUS [STRING REPRESENTATION OF count, CHARACTER AT word[index]], 0)
                ELSE
                    CALL backtrack WITH (index + 1, path PLUS [CHARACTER AT word[index]], 0)
                END IF
                CALL backtrack WITH (index + 1, path, count + 1)
            END IF
        END FUNCTION

        SET result TO EMPTY LIST
        CALL backtrack WITH (0, EMPTY LIST, 0)
        RETURN result
    END FUNCTION
END CLASS
