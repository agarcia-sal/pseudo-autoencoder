CLASS Solution
    FUNCTION lengthOfLIS(nums: List of Integer) RETURNS Integer
        IF nums IS EMPTY
            RETURN 0
        SET dp TO empty list
        FOR each num IN nums
            SET idx TO position to insert num in dp preserving sorted order (using binary search)
            IF idx IS EQUAL TO LENGTH OF dp
                APPEND num TO dp
            ELSE
                SET dp[idx] TO num
        RETURN LENGTH OF dp
    END FUNCTION
END CLASS
