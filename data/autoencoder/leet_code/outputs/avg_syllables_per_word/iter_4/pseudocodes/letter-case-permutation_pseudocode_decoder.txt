CLASS Solution
    FUNCTION letterCasePermutation(s)
        FUNCTION dfs(i)
            IF i GREATER THAN OR EQUAL TO LENGTH OF t THEN
                APPEND JOINED t TO ans
                RETURN
            END IF
            CALL dfs(i + 1)
            IF t[i] IS ALPHABETIC THEN
                SET t[i] TO CHARACTER WITH CASE TOGGLED
                CALL dfs(i + 1)
            END IF
        END FUNCTION

        SET t TO LIST OF CHARACTERS IN s
        SET ans TO EMPTY LIST
        CALL dfs(0)
        RETURN ans
    END FUNCTION
END CLASS
