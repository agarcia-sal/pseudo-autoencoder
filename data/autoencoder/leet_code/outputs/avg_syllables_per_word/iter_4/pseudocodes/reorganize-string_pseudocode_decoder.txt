CLASS Solution
    FUNCTION reorganizeString(s)
        SET count TO frequency count of characters in s
        SET max_heap TO list of pairs (negative frequency, character) from count
        CALL heapify on max_heap

        SET prev_char TO None
        SET prev_freq TO 0
        SET result TO empty list

        WHILE max_heap is not empty OR prev_freq < 0
            IF max_heap is empty AND prev_freq < 0
                RETURN empty string

            SET freq, char TO element popped from max_heap heap
            APPEND char TO result

            INCREMENT freq BY 1

            IF prev_freq < 0
                PUSH (prev_freq, prev_char) INTO max_heap heap

            SET prev_char TO char
            SET prev_freq TO freq

        RETURN concatenation of characters in result
    END FUNCTION
END CLASS
