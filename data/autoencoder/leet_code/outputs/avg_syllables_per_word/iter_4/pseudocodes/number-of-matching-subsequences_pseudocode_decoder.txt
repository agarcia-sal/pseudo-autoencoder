CLASS Solution
    FUNCTION numMatchingSubseq(s, words)
        SET waiting TO dictionary mapping each unique character in s TO empty deque
        FOR each word IN words
            SET it TO iterator of word
            SET first_letter TO next element from it
            IF first_letter is in waiting
                APPEND it TO waiting[first_letter]
        SET match_count TO 0
        FOR each character c IN s
            IF c in waiting
                SET current_iterators TO waiting[c]
                SET waiting[c] TO empty deque
                FOR each it IN current_iterators
                    SET next_char TO next element from it or None if exhausted
                    IF next_char is None
                        INCREMENT match_count BY 1
                    ELSE IF next_char in waiting
                        APPEND it TO waiting[next_char]
        RETURN match_count
    END FUNCTION
END CLASS
