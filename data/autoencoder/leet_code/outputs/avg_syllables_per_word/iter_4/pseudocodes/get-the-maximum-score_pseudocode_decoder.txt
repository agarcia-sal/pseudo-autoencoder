CLASS Solution
    FUNCTION maxSum(nums1, nums2)
        SET i TO 0
        SET j TO 0
        SET sum1 TO 0
        SET sum2 TO 0
        SET MOD TO 1000000007

        WHILE i LESS THAN LENGTH OF nums1 AND j LESS THAN LENGTH OF nums2
            IF nums1[i] LESS THAN nums2[j]
                INCREMENT sum1 BY nums1[i]
                INCREMENT i BY 1
            ELSE IF nums1[i] GREATER THAN nums2[j]
                INCREMENT sum2 BY nums2[j]
                INCREMENT j BY 1
            ELSE
                SET sum1 TO MAXIMUM OF sum1 AND sum2 PLUS nums1[i]
                SET sum2 TO sum1
                INCREMENT i BY 1
                INCREMENT j BY 1

        WHILE i LESS THAN LENGTH OF nums1
            INCREMENT sum1 BY nums1[i]
            INCREMENT i BY 1

        WHILE j LESS THAN LENGTH OF nums2
            INCREMENT sum2 BY nums2[j]
            INCREMENT j BY 1

        RETURN MAXIMUM OF sum1 AND sum2 MODULO MOD
    END FUNCTION
END CLASS
