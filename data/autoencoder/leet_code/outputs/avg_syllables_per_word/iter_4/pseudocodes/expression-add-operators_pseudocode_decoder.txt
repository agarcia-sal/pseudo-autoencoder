CLASS Solution
    FUNCTION addOperators(num, target)
        FUNCTION dfs(index, path, value, prev)
            IF index == LENGTH OF num THEN
                IF value == target THEN
                    APPEND path TO result
                END IF
                RETURN
            END IF

            FOR i FROM index + 1 TO LENGTH OF num + 1
                SET operand TO SUBSTRING OF num FROM index TO i
                IF LENGTH OF operand > 1 AND FIRST CHARACTER OF operand IS '0' THEN
                    CONTINUE TO NEXT iteration
                END IF

                IF index == 0 THEN
                    CALL dfs(i, operand, CONVERT operand TO INTEGER, CONVERT operand TO INTEGER)
                ELSE
                    CALL dfs(i, CONCATENATE path, '+', operand, value + CONVERT operand TO INTEGER, CONVERT operand TO INTEGER)
                    CALL dfs(i, CONCATENATE path, '-', operand, value - CONVERT operand TO INTEGER, NEGATIVE OF CONVERT operand TO INTEGER)
                    CALL dfs(i, CONCATENATE path, '*', operand, value - prev + prev * CONVERT operand TO INTEGER, prev * CONVERT operand TO INTEGER)
                END IF
            END FOR
        END FUNCTION

        SET result TO EMPTY LIST
        CALL dfs(0, EMPTY STRING, 0, 0)
        RETURN result
    END FUNCTION
END CLASS
