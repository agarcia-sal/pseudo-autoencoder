CLASS Solution
    FUNCTION smallestGoodBase(n: string)
        SET n TO integer value of n
        SET max_length TO integer value of logarithm base 2 of n PLUS 1
        FOR m FROM max_length DOWNTO 2
            SET k TO integer value of the (m-1)th root of n
            IF k LESS THAN 2
                CONTINUE to next iteration
            SET num TO (k raised to power m MINUS 1) DIVIDED BY (k MINUS 1)
            IF num EQUALS n
                RETURN string value of k
        RETURN string value of (n MINUS 1)
    END FUNCTION
END CLASS
