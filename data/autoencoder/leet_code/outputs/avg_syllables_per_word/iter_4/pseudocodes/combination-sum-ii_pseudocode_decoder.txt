CLASS Solution
    FUNCTION combinationSum2(candidates, target)
        FUNCTION backtrack(start, target, path)
            IF target EQUALS 0
                APPEND path TO result
                RETURN
            IF target LESS THAN 0
                RETURN
            FOR i FROM start TO LENGTH OF candidates MINUS 1
                IF i GREATER THAN start AND candidates[i] EQUALS candidates[i - 1]
                    CONTINUE
                CALL backtrack(i + 1, target MINUS candidates[i], path WITH candidates[i] APPENDED)
        END FUNCTION
        SORT candidates IN ASCENDING ORDER
        SET result TO EMPTY LIST
        CALL backtrack(0, target, EMPTY LIST)
        RETURN result
    END FUNCTION
END CLASS
