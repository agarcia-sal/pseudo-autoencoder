CLASS TreeNode
    FUNCTION __init__(self, val=0, left=None, right=None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty queue
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION zigzagLevelOrder(self, root)
        IF root IS None
            RETURN empty list
        SET result TO empty list
        SET queue TO new queue WITH root
        SET left_to_right TO True
        WHILE queue IS NOT EMPTY
            SET level_size TO LENGTH OF queue
            SET current_level TO empty deque
            FOR count FROM 1 TO level_size
                SET node TO REMOVE FIRST ELEMENT FROM queue
                IF left_to_right IS True
                    APPEND node.val TO current_level
                ELSE
                    INSERT node.val AT BEGINNING OF current_level
                IF node.left IS NOT None
                    APPEND node.left TO queue
                IF node.right IS NOT None
                    APPEND node.right TO queue
            APPEND LIST OF current_level TO result
            SET left_to_right TO NOT left_to_right
        RETURN result
    END FUNCTION
END CLASS
