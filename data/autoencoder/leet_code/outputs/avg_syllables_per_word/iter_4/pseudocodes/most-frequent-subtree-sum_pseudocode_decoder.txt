CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION findFrequentTreeSum(self, root: Optional[TreeNode]) RETURNS List[int]
        FUNCTION subtree_sum(node)
            IF node IS None
                RETURN 0
            SET current_sum TO node.val PLUS subtree_sum(node.left) PLUS subtree_sum(node.right)
            INCREMENT count[current_sum] BY 1
            RETURN current_sum
        END FUNCTION

        SET count TO empty dictionary with default integer 0
        CALL subtree_sum(root)

        SET max_freq TO maximum value in count values, default 0 if empty

        SET result TO empty list
        FOR each key, freq IN count items
            IF freq EQUALS max_freq
                APPEND key TO result
        RETURN result
    END FUNCTION
END CLASS
