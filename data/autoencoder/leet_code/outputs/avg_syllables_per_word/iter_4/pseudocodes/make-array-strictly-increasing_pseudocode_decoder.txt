CLASS Solution
    FUNCTION makeArrayIncreasing(arr1, arr2)
        SORT arr2
        SET dp TO dictionary with key -1 and value 0
        
        FOR each num IN arr1
            SET new_dp TO empty dictionary
            FOR each last_num, ops IN dp
                IF num > last_num THEN
                    SET current_val TO new_dp value at key num OR infinity
                    SET new_dp[num] TO minimum of current_val and ops
                SET idx TO position after last_num in arr2 using binary search
                IF idx LESS THAN length of arr2 THEN
                    SET replace_val TO arr2 at idx
                    SET current_val TO new_dp value at key replace_val OR infinity
                    SET new_dp[replace_val] TO minimum of current_val and ops plus 1
            IF new_dp is empty THEN
                RETURN -1
            SET dp TO new_dp
        
        RETURN minimum value in dp
    END FUNCTION
END CLASS
