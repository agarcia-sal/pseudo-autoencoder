CLASS Solution
    FUNCTION findCheapestPrice(n: integer, flights: list of list of integer, src: integer, dst: integer, k: integer) RETURNS integer
        SET graph TO dictionary mapping each integer from 0 to n-1 TO empty list
        FOR each u, v, w IN flights
            APPEND (v, w) TO graph[u]
        
        SET min_heap TO list containing tuple (0, -1, src)
        SET min_cost TO dictionary with key src and value -1
        
        WHILE min_heap is not empty
            REMOVE and SET cost, stops, city TO the smallest element from min_heap
            
            IF city EQUALS dst THEN
                RETURN cost
            
            IF stops IS GREATER THAN OR EQUAL TO k THEN
                CONTINUE to next iteration
            
            FOR each neighbor, price IN graph[city]
                SET next_cost TO cost PLUS price
                SET next_stops TO stops PLUS 1
                
                IF neighbor NOT IN min_cost OR next_stops IS LESS THAN min_cost[neighbor] OR next_cost IS LESS THAN the cost part of the smallest element in min_heap THEN
                    SET min_cost[neighbor] TO next_stops
                    ADD (next_cost, next_stops, neighbor) TO min_heap
        
        RETURN -1
    END FUNCTION
END CLASS
