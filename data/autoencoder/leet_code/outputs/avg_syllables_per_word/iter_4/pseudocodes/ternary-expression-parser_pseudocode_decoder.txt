CLASS Solution
    FUNCTION parseTernary(expression)
        FUNCTION evaluate(index)
            IF index EQUALS LENGTH OF expression MINUS 1 OR character AT index PLUS 1 EQUALS ':'
                RETURN character AT index, index PLUS 2
            END IF
            
            SET true_part, next_index TO evaluate(index PLUS 2)
            SET false_part, final_index TO evaluate(next_index)
            
            IF character AT index EQUALS 'T'
                RETURN true_part, final_index
            ELSE
                RETURN false_part, final_index
            END IF
        END FUNCTION
        
        SET result, _ TO evaluate(0)
        RETURN result
    END FUNCTION
END CLASS
