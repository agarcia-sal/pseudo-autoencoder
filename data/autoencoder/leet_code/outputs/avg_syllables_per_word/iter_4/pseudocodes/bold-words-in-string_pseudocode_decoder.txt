CLASS Solution
    FUNCTION boldWords(words, s)
        SET bold TO list of False repeated LENGTH OF s times
        FOR each word IN words
            SET start TO POSITION OF word IN s
            WHILE start IS NOT -1
                FOR i FROM start TO start PLUS LENGTH OF word MINUS 1
                    SET bold[i] TO True
                SET start TO POSITION OF word IN s STARTING FROM start PLUS 1
        SET result TO empty list
        SET i TO 0
        WHILE i LESS THAN LENGTH OF s
            IF bold[i] IS True
                APPEND "<b>" TO result
                WHILE i LESS THAN LENGTH OF s AND bold[i] IS True
                    APPEND s[i] TO result
                    INCREMENT i BY 1
                APPEND "</b>" TO result
            ELSE
                APPEND s[i] TO result
                INCREMENT i BY 1
        RETURN CONCATENATION OF elements IN result
    END FUNCTION
END CLASS
