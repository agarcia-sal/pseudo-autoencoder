CLASS Solution
    FUNCTION findCircleNum(isConnected)
        FUNCTION dfs(city)
            FOR neighbor FROM 0 TO n MINUS 1
                IF isConnected[city][neighbor] EQUALS 1 AND visited[neighbor] IS FALSE
                    SET visited[neighbor] TO TRUE
                    CALL dfs(neighbor)
                END IF
            END FOR
        END FUNCTION

        SET n TO LENGTH OF isConnected
        SET visited TO LIST OF FALSE OF SIZE n
        SET province_count TO 0

        FOR city FROM 0 TO n MINUS 1
            IF visited[city] IS FALSE
                CALL dfs(city)
                INCREMENT province_count BY 1
            END IF
        END FOR

        RETURN province_count
    END FUNCTION
END CLASS
