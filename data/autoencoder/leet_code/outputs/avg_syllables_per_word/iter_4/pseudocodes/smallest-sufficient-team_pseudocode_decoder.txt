CLASS Solution
    FUNCTION smallestSufficientTeam(req_skills, people)
        SET d TO dictionary mapping each skill in req_skills to its index
        SET m TO length of req_skills
        SET n TO length of people
        SET p TO list of zeros with length n
        FOR i FROM 0 TO n - 1
            FOR each skill s IN people[i]
                SET p[i] TO p[i] OR (1 LEFT SHIFTED BY d[s])
        SET f TO list of infinity with length 2 RAISED TO m
        SET g TO list of zeros with length 2 RAISED TO m
        SET h TO list of zeros with length 2 RAISED TO m
        SET f[0] TO 0
        FOR i FROM 0 TO (2 RAISED TO m) - 1
            IF f[i] IS infinity
                CONTINUE to next iteration
            FOR j FROM 0 TO n - 1
                SET new_state TO i OR p[j]
                IF f[i] + 1 IS LESS THAN f[new_state]
                    SET f[new_state] TO f[i] + 1
                    SET g[new_state] TO j
                    SET h[new_state] TO i
        SET i TO (2 RAISED TO m) - 1
        SET ans TO empty list
        WHILE i IS NOT zero
            APPEND g[i] TO ans
            SET i TO h[i]
        RETURN ans
    END FUNCTION
END CLASS
