CLASS Solution
    FUNCTION findRadius(houses: List of integers, heaters: List of integers) RETURNS integer
        SORT houses
        SORT heaters
        
        SET min_radius TO 0
        SET j TO 0
        
        FOR each house IN houses
            WHILE j LESS THAN LENGTH OF heaters MINUS 1 AND 
                  ABSOLUTE VALUE OF (heaters[j] MINUS house) GREATER THAN OR EQUAL TO ABSOLUTE VALUE OF (heaters[j + 1] MINUS house)
                INCREMENT j BY 1
            
            SET min_radius TO MAXIMUM OF min_radius AND ABSOLUTE VALUE OF (heaters[j] MINUS house)
        
        RETURN min_radius
    END FUNCTION
END CLASS
