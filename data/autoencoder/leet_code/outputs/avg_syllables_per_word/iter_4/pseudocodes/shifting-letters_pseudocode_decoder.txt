CLASS Solution
    FUNCTION shiftingLetters(s, shifts)
        SET total_shifts TO 0
        FOR i FROM LENGTH OF shifts MINUS 1 DOWNTO 0
            INCREMENT total_shifts BY shifts[i]
            SET shifts[i] TO total_shifts
        SET result TO empty list
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            SET new_char_code TO (ASCII OF s[i] MINUS ASCII OF 'a' PLUS shifts[i]) MODULO 26 PLUS ASCII OF 'a'
            APPEND CHARACTER OF new_char_code TO result
        RETURN CONCATENATION OF result
    END FUNCTION
END CLASS
