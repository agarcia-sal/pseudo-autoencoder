CLASS Solution
    FUNCTION findMaximumXOR(nums: List of Integer) RETURNS Integer
        SET max_xor TO 0
        FOR i FROM 31 DOWNTO 0
            SET max_xor TO max_xor SHIFTED LEFT BY 1
            SET current_candidate TO max_xor BITWISE OR 1
            SET prefixes TO empty set
            FOR each num IN nums
                SET prefix TO num SHIFTED RIGHT BY i
                ADD prefix TO prefixes
            FOR each prefix IN prefixes
                IF (current_candidate BITWISE XOR prefix) IS IN prefixes THEN
                    SET max_xor TO current_candidate
                    BREAK
        RETURN max_xor
    END FUNCTION
END CLASS
