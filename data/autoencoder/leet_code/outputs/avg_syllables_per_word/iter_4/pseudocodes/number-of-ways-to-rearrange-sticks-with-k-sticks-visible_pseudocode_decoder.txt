CLASS Solution
    FUNCTION rearrangeSticks(n, k)
        SET MOD TO 10 POWER 9 PLUS 7
        CREATE dp AS 2D LIST OF SIZE (n + 1) BY (k + 1) FILLED WITH 0
        SET dp[0][0] TO 1
        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                SET dp[i][j] TO (dp[i - 1][j - 1] PLUS (i - 1) TIMES dp[i - 1][j]) MODULO MOD
        RETURN dp[n][k]
    END FUNCTION
END CLASS
