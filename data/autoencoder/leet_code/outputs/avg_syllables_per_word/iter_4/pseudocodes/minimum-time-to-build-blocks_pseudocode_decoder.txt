CLASS Solution
    FUNCTION minBuildTime(blocks, split)
        CONVERT blocks INTO a min-heap
        WHILE LENGTH OF blocks IS GREATER THAN 1
            SET first TO REMOVE smallest element FROM blocks
            SET second TO REMOVE next smallest element FROM blocks
            SET combined_time TO split PLUS second
            ADD combined_time INTO blocks heap
        RETURN the only element IN blocks
    END FUNCTION
END CLASS
