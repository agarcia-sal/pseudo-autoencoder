CLASS Solution
    FUNCTION isNumber(s)
        IMPORT regular_expression_module AS re
        SET number_pattern TO start_of_string FOLLOWED BY optional_plus_or_minus FOLLOWED BY (digits FOLLOWED BY optional_decimal_point_and_digits OR decimal_point FOLLOWED BY digits) FOLLOWED BY optional_exponent_pattern FOLLOWED BY end_of_string
        RETURN re.fullmatch(number_pattern, s) IS NOT NULL
    END FUNCTION
END CLASS
