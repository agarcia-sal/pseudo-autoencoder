CLASS UnionFind
    FUNCTION __init__(size)
        SET parent TO list of integers from 0 to size minus 1
    END FUNCTION

    FUNCTION find(u)
        IF parent[u] is not equal to u THEN
            SET parent[u] TO find(parent[u])
        END IF
        RETURN parent[u]
    END FUNCTION

    FUNCTION union(u, v)
        SET root_u TO find(u)
        SET root_v TO find(v)
        IF root_u is not equal to root_v THEN
            SET parent[root_u] TO root_v
        END IF
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION matrixRankTransform(matrix)
        SET m TO number of rows in matrix
        SET n TO number of columns in matrix
        SET value_to_indices TO an empty map from value to list

        FOR r FROM 0 TO m minus 1
            FOR c FROM 0 TO n minus 1
                APPEND (r, c) TO value_to_indices[matrix[r][c]]
            END FOR
        END FOR

        SET row_rank TO list of zeros with length m
        SET col_rank TO list of zeros with length n
        SET result TO m by n matrix filled with zeros

        FOR each value IN sorted keys of value_to_indices
            CREATE UnionFind instance uf with size m plus n
            SET rank TO an empty map from root to integer

            FOR each (r, c) IN value_to_indices[value]
                CALL uf.union(r, c plus m)
            END FOR

            FOR each (r, c) IN value_to_indices[value]
                SET root TO uf.find(r)
                SET rank[root] TO maximum of rank[root] and (maximum of row_rank[r] and col_rank[c]) plus 1
            END FOR

            FOR each (r, c) IN value_to_indices[value]
                SET root TO uf.find(r)
                SET result[r][c] TO rank[root]
                SET row_rank[r] TO rank[root]
                SET col_rank[c] TO rank[root]
            END FOR
        END FOR

        RETURN result
    END FUNCTION
END CLASS
