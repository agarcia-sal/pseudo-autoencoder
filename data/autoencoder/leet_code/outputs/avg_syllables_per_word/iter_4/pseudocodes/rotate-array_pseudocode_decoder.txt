CLASS Solution
    FUNCTION rotate(nums, k)
        SET n TO LENGTH OF nums
        SET k TO k MODULO n

        FUNCTION reverse(start, end)
            WHILE start LESS THAN end
                SWAP nums[start] WITH nums[end]
                INCREMENT start BY 1
                DECREMENT end BY 1
            END WHILE
        END FUNCTION

        CALL reverse WITH PARAMETERS (0, n MINUS 1)
        CALL reverse WITH PARAMETERS (0, k MINUS 1)
        CALL reverse WITH PARAMETERS (k, n MINUS 1)
    END FUNCTION
END CLASS
