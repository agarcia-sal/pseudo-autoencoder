CLASS Solution
    FUNCTION minOperations(target: List of Integer, arr: List of Integer) RETURNS Integer
        SET index_map TO empty map
        FOR each index, value IN enumeration of target
            SET index_map[value] TO index
        
        SET transformed_arr TO empty list
        FOR each value IN arr
            IF value EXISTS IN index_map
                APPEND index_map[value] TO transformed_arr
        
        SET lis TO empty list
        FOR each num IN transformed_arr
            SET pos TO POSITION TO INSERT num IN lis USING binary search for left insertion
            IF pos EQUALS LENGTH OF lis
                APPEND num TO lis
            ELSE
                SET lis[pos] TO num
        
        SET lcs_length TO LENGTH OF lis
        RETURN LENGTH OF target MINUS lcs_length
    END FUNCTION
END CLASS
