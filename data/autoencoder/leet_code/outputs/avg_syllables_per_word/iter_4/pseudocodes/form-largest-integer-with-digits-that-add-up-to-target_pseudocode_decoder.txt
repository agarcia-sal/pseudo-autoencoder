CLASS Solution
    FUNCTION largestNumber(cost: List[int], target: int) RETURNS str
        SET cost_to_digit TO DICTIONARY mapping each cost value TO its corresponding digit string starting from "1"
        SET dp TO LIST with EMPTY STRING at index 0 and "-1" repeated target times
        
        FOR t FROM 1 TO target
            FOR each cost c AND digit d IN cost_to_digit
                IF t >= c AND dp[t - c] IS NOT "-1"
                    SET candidate TO CONCATENATION of d AND dp[t - c]
                    IF dp[t] IS "-1" OR LENGTH OF candidate GREATER THAN LENGTH OF dp[t] OR (LENGTHS EQUAL AND candidate GREATER THAN dp[t])
                        SET dp[t] TO candidate
        END FOR
        END FOR

        IF dp[target] IS NOT "-1"
            RETURN dp[target]
        ELSE
            RETURN "0"
    END FUNCTION
END CLASS
