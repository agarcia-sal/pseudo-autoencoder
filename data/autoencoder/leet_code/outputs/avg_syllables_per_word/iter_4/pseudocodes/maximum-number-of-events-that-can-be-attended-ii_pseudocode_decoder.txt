CLASS Solution
    FUNCTION maxValue(events, k)
        SORT events BY their end day
        SET n TO LENGTH OF events
        INITIALIZE dp AS 2D list WITH (n + 1) rows AND (k + 1) columns FILLED WITH 0

        FOR i FROM 1 TO n
            SET start, end, value TO events[i - 1]

            FIND index j WHERE the event at j ENDS BEFORE start USING binary search ON events' end days

            FOR l FROM 1 TO k
                SET dp[i][l] TO MAXIMUM OF dp[i - 1][l] AND dp[j][l - 1] PLUS value
        RETURN dp[n][k]
    END FUNCTION
END CLASS
