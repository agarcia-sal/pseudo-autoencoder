CLASS Solution
    FUNCTION canSeePersonsCount(heights)
        SET n TO LENGTH OF heights
        SET answer TO LIST OF ZEROES WITH SIZE n
        SET stack TO EMPTY LIST
        
        FOR i FROM 0 TO n MINUS 1
            WHILE stack IS NOT EMPTY AND heights[stack LAST ELEMENT] < heights[i]
                INCREMENT answer VALUE AT index stack POPPED BY 1
            IF stack IS NOT EMPTY
                INCREMENT answer VALUE AT index stack LAST ELEMENT BY 1
            APPEND i TO stack
        
        RETURN answer
    END FUNCTION
END CLASS
