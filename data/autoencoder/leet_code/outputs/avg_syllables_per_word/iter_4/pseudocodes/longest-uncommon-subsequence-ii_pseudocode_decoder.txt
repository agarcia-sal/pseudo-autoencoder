CLASS Solution
    FUNCTION findLUSlength(strs)
        FUNCTION is_subsequence(s, t)
            SET iterator_it TO iterator over t
            FOR each character c IN s
                IF c NOT IN iterator_it THEN
                    RETURN False
            RETURN True
        END FUNCTION

        SORT strs BY descending length AND lexicographically ascending

        FOR each index i AND word IN strs
            SET unique TO True
            FOR each index j AND other IN strs
                IF i NOT EQUAL TO j AND is_subsequence(word, other) THEN
                    SET unique TO False
                    BREAK inner loop
            IF unique THEN
                RETURN length of word
        RETURN -1
END CLASS
