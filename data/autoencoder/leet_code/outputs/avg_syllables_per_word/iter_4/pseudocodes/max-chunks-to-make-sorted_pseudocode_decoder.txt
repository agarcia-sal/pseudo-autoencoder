CLASS Solution
    FUNCTION maxChunksToSorted(arr)
        SET max_so_far TO 0
        SET chunks TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            SET max_so_far TO MAXIMUM OF max_so_far AND arr[i]
            IF max_so_far EQUALS i
                INCREMENT chunks BY 1
        RETURN chunks
    END FUNCTION
END CLASS
