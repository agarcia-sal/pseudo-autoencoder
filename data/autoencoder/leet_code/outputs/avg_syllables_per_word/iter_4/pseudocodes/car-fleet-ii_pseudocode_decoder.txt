CLASS Solution
    FUNCTION getCollisionTimes(cars: List of List of Integer) RETURNS List of Float
        SET n TO LENGTH OF cars
        SET result TO LIST OF SIZE n FILLED WITH -1.0
        SET stack TO EMPTY LIST

        FOR i FROM n MINUS 1 DOWNTO 0
            SET pos, speed TO cars[i]

            WHILE stack IS NOT EMPTY
                SET next_index TO LAST ELEMENT IN stack
                SET next_pos, next_speed TO cars[next_index]

                IF speed IS LESS THAN OR EQUAL TO next_speed
                    REMOVE LAST ELEMENT FROM stack
                    CONTINUE WHILE LOOP
                END IF

                SET collision_time TO (next_pos MINUS pos) DIVIDED BY (speed MINUS next_speed)

                IF result[next_index] IS GREATER THAN 0 AND collision_time IS GREATER THAN OR EQUAL TO result[next_index]
                    REMOVE LAST ELEMENT FROM stack
                    CONTINUE WHILE LOOP
                END IF

                SET result[i] TO collision_time
                BREAK WHILE LOOP
            END WHILE

            APPEND i TO stack
        END FOR

        RETURN result
    END FUNCTION
END CLASS
