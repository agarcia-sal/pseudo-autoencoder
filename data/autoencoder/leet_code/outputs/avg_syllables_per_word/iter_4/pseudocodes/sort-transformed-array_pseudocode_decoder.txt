CLASS Solution
    FUNCTION sortTransformedArray(nums, a, b, c)
        FUNCTION quadratic(x)
            RETURN a TIMES x TIMES x PLUS b TIMES x PLUS c
        END FUNCTION

        SET n TO LENGTH OF nums
        SET result TO LIST OF SIZE n FILLED WITH 0
        SET left TO 0
        SET right TO n MINUS 1

        IF a GREATER THAN OR EQUAL TO 0 THEN
            SET index TO n MINUS 1
            WHILE left LESS THAN OR EQUAL TO right
                IF quadratic(nums[left]) GREATER THAN quadratic(nums[right]) THEN
                    SET result[index] TO quadratic(nums[left])
                    INCREMENT left BY 1
                ELSE
                    SET result[index] TO quadratic(nums[right])
                    DECREMENT right BY 1
                END IF
                DECREMENT index BY 1
            END WHILE
        ELSE
            SET index TO 0
            WHILE left LESS THAN OR EQUAL TO right
                IF quadratic(nums[left]) LESS THAN quadratic(nums[right]) THEN
                    SET result[index] TO quadratic(nums[left])
                    INCREMENT left BY 1
                ELSE
                    SET result[index] TO quadratic(nums[right])
                    DECREMENT right BY 1
                END IF
                INCREMENT index BY 1
            END WHILE
        END IF

        RETURN result
    END FUNCTION
END CLASS
