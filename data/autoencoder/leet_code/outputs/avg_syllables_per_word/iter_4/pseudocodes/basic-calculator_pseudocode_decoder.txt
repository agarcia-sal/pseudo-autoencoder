CLASS Solution
    FUNCTION calculate(s)
        FUNCTION helper(s_list)
            SET stack TO empty list
            SET sign TO '+'
            SET num TO 0
            
            WHILE LENGTH OF s_list > 0
                SET char TO REMOVE FIRST ELEMENT FROM s_list
                
                IF char IS A DIGIT
                    SET num TO num MULTIPLIED BY 10 PLUS integer value of char
                
                IF char IS '('
                    SET num TO CALL helper WITH s_list
                
                IF char IS NOT A DIGIT AND char IS NOT WHITESPACE OR LENGTH OF s_list IS 0
                    IF sign IS '+'
                        APPEND num TO stack
                    ELSE IF sign IS '-'
                        APPEND negative num TO stack
                    ELSE IF sign IS '*'
                        SET last element of stack TO last element MULTIPLIED BY num
                    ELSE IF sign IS '/'
                        SET last element of stack TO integer division of last element BY num
                    
                    SET sign TO char
                    SET num TO 0
                
                IF char IS ')'
                    BREAK
            
            RETURN SUM OF stack
        
        RETURN helper(CONVERT s TO LIST)
