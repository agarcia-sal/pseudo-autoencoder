CLASS Solution
    FUNCTION fullBloomFlowers(flowers WITH TYPE List of List of int, people WITH TYPE List of int) RETURNS List of int
        SET start_times TO sorted list of the first element of each flower interval in flowers
        SET end_times TO sorted list of the second element of each flower interval in flowers
        SET result TO empty list
        FOR each person IN people
            SET started TO number of elements in start_times less than or equal to person
            SET ended TO number of elements in end_times strictly less than person
            APPEND started MINUS ended TO result
        RETURN result
    END FUNCTION
END CLASS
