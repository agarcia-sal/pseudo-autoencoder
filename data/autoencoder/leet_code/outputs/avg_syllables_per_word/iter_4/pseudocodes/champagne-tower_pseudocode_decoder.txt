CLASS Solution
    FUNCTION champagneTower(poured, query_row, query_glass)
        SET current_row TO [poured]
        FOR row FROM 1 TO query_row
            SET next_row TO list of zeros with length (row + 1)
            FOR i FROM 0 TO LENGTH OF current_row MINUS 1
                IF current_row[i] GREATER THAN 1
                    SET excess TO (current_row[i] MINUS 1) DIVIDED BY 2
                    INCREMENT next_row[i] BY excess
                    INCREMENT next_row[i + 1] BY excess
            SET current_row TO next_row
        RETURN MINIMUM OF current_row[query_glass] AND 1
    END FUNCTION
END CLASS
