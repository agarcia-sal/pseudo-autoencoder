CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND the element at position i of nums IS GREATER THAN OR EQUAL TO the element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        
        IF i IS GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE the element at position j of nums IS LESS THAN OR EQUAL TO the element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary TO the element at position i of nums  
            SET the element at position i of nums TO the element at position j of nums  
            SET the element at position j of nums TO temporary  
        END IF  
        
        SET start_position TO i PLUS one  
        SET end_position TO the LENGTH OF nums MINUS one  
        WHILE start_position IS LESS THAN end_position  
            SET temporary TO the element at position start_position of nums  
            SET the element at position start_position of nums TO the element at position end_position of nums  
            SET the element at position end_position of nums TO temporary  
            INCREMENT start_position BY one  
            DECREMENT end_position BY one  
        END WHILE  
    END FUNCTION  
END CLASS
