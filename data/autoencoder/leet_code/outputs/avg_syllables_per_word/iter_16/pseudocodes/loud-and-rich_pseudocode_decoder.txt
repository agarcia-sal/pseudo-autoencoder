CLASS Solution  
    FUNCTION loudAndRich WITH PARAMETERS richer OF LIST OF LIST OF INTEGER, quiet OF LIST OF INTEGER RETURNS LIST OF INTEGER  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair of integers a AND b IN richer  
            APPEND integer a TO the list element at key b OF graph  
        END FOR  
        
        SET n TO the LENGTH OF quiet  
        SET answer TO a list of size n WITH all elements set TO negative one  
        
        FUNCTION dfs WITH PARAMETER person OF INTEGER RETURNS INTEGER  
            IF element at position person OF answer EQUALS negative one  
                SET element at position person OF answer TO person  
                FOR each neighbor IN element at key person OF graph  
                    SET candidate TO dfs WITH neighbor AS PARAMETER  
                    IF element at position candidate OF quiet LESS THAN element at position element at position person OF answer OF quiet  
                        SET element at position person OF answer TO candidate  
                    END IF  
                END FOR  
            END IF  
            RETURN element at position person OF answer  
        END FUNCTION  
        
        FOR each integer i FROM zero TO n MINUS one  
            CALL dfs WITH parameter i  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
