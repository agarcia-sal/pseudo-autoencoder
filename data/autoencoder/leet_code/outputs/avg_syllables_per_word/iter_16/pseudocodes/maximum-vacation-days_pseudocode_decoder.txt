CLASS Solution  
    FUNCTION maxVacationDays(flights AS List OF List OF integers, days AS List OF List OF integers) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
        
        SET dp TO a list of length n with all elements set TO negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a mapping from each integer i from zero TO n MINUS one TO an empty list  
        FOR each integer i FROM zero TO n MINUS one  
            FOR each integer j FROM zero TO n MINUS one  
                IF element at position j OF element at position i OF flights EQUALS one THEN  
                    APPEND j TO element at position i OF graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n with all elements set TO negative one  
            FOR each integer i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one THEN  
                    CONTINUE TO next iteration  
                END IF  
                SET element at position i of new_dp TO the GREATER VALUE BETWEEN element at position i of new_dp AND element at position i of dp PLUS element at position week of element at position i of days  
                FOR each city j IN element at position i of graph  
                    SET element at position j of new_dp TO the GREATER VALUE BETWEEN element at position j of new_dp AND element at position i of dp PLUS element at position week of element at position j of days  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN the GREATER VALUE AMONG ALL elements OF dp  
    END FUNCTION  
END CLASS
