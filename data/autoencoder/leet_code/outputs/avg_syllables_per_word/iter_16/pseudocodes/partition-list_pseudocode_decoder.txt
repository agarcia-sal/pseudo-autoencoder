CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW INSTANCE OF ListNode WITH val EQUAL TO element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one to the end  
        SET node TO A NEW INSTANCE OF ListNode WITH val EQUAL TO val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list WITH PARAMETERS next of p1 AND next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition WITH PARAMETERS head AS Optional OF ListNode AND x AS integer RETURNING Optional OF ListNode  
        SET less_head TO A NEW INSTANCE OF ListNode WITH val EQUAL TO zero  
        SET greater_head TO A NEW INSTANCE OF ListNode WITH val EQUAL TO zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT none  
            IF value at current IS LESS THAN x  
                SET next of less TO current  
                SET less TO next of less  
            ELSE  
                SET next of greater TO current  
                SET greater TO next of greater  
            END IF  
            SET current TO next of current  
        END WHILE  
        SET next of less TO next of greater_head  
        SET next of greater TO none  
        RETURN next of less_head  
    END FUNCTION  
END CLASS
