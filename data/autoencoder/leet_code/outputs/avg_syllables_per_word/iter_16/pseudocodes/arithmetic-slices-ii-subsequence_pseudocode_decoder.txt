CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a LIST containing for each index FROM zero TO n MINUS one a new DEFAULT DICTIONARY mapping INTEGER keys TO INTEGER values  
        SET total_count TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET d TO element at position i of nums MINUS element at position j of nums  
                
                IF dp at position j HAS KEY d  
                    INCREMENT dp at position i for key d BY the VALUE at dp position j for key d PLUS one  
                    INCREMENT total_count BY the VALUE at dp position j for key d  
                ELSE  
                    INCREMENT dp at position i for key d BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
