CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF element at position zero of dungeon  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns each initialized TO infinity  
        SET element at row position m column position n MINUS one of dp TO one  
        SET element at row position m MINUS one column position n of dp TO one  
        FOR i FROM m MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                SET minimum_health TO the MINIMUM OF element at row position i PLUS one column position j of dp AND element at row position i column position j PLUS one of dp MINUS element at row position i column position j of dungeon  
                IF minimum_health IS LESS THAN one  
                    SET element at row position i column position j of dp TO one  
                ELSE  
                    SET element at row position i column position j of dp TO minimum_health  
                END IF  
            END FOR  
        END FOR  
        RETURN element at row position zero column position zero of dp  
    END FUNCTION  
END CLASS
