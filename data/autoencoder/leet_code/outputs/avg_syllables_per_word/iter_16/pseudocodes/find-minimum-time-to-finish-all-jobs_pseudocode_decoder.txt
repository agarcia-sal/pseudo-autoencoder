CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        
        SET subset_sums TO a list of zeros with LENGTH EQUAL TO two RAISED TO THE POWER OF n  
        
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET element at position the bitwise OR of mask AND two RAISED TO THE POWER OF i TO the sum of element at position mask of subset_sums PLUS element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN element at position mask of subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS GREATER THAN zero  
                SET temp_max TO the GREATER OF dp the result of k MINUS one AND the bitwise exclusive OR of mask AND submask AND element at position submask of subset_sums  
                IF temp_max IS LESS THAN res THEN  
                    SET res TO temp_max  
                END IF  
                SET submask TO the bitwise AND of the result of submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp the result of k AND two RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
