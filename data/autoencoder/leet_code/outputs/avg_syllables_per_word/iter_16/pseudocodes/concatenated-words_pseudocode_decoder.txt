CLASS Trie  
    FUNCTION __init__()  
        SET children TO a list of length twenty six filled with None elements  
        SET is_end TO False  
    END FUNCTION  

    FUNCTION insert(w)  
        SET node TO the current Trie instance  
        FOR each character c IN the string w  
            SET idx TO the numeric position of character c in the alphabet counted from zero for letter a  
            IF element at position idx of node children IS None  
                SET element at position idx of node children TO a new Trie instance  
            END IF  
            SET node TO element at position idx of node children  
        END FOR  
        SET node is_end TO True  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the LENGTH OF string w  
                RETURN True  
            END IF  
            SET node TO the trie instance accessible from outer scope  
            FOR each index i FROM start TO one LESS THAN the LENGTH OF string w  
                SET idx TO the numeric position of character at position i of string w in the alphabet counted from zero for letter a  
                IF element at position idx of node children IS None  
                    RETURN False  
                END IF  
                SET node TO element at position idx of node children  
                IF node is_end EQUALS True AND dfs(w, i PLUS one) EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET trie TO a new Trie instance  
        SET ans TO an empty list  
        SORT words IN PLACE BY the LENGTH OF each word in ascending order  
        FOR each string w IN words  
            IF w IS NOT an empty string  
                IF dfs(w, zero) EQUALS True  
                    APPEND w TO ans  
                ELSE  
                    CALL insert method of trie WITH argument w  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
