CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN list containing zero  
        END IF  
        
        SET graph TO a new default dictionary that maps each key to an empty list  
        SET degree TO a list of n elements each initialized TO zero  
        
        FOR each pair u and v IN edges  
            APPEND v TO the list associated with key u in graph  
            APPEND u TO the list associated with key v in graph  
            INCREMENT degree at position u BY one  
            INCREMENT degree at position v BY one  
        END FOR  
        
        SET leaves TO a new double ended queue initialized with all indices i from zero TO n MINUS one for which degree at position i EQUALS one  
        
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the LENGTH OF leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each element in the range from zero TO leaves_count MINUS one  
                SET leaf TO the element removed from the left end of leaves  
                FOR each neighbor IN the list associated with key leaf in graph  
                    DECREMENT degree at position neighbor BY one  
                    IF degree at position neighbor EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN the list constructed from leaves  
    END FUNCTION  
END CLASS
