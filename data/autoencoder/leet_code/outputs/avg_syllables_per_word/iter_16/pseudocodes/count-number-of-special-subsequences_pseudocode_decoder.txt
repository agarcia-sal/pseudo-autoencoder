CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of zero zero zero representing counts of special subsequences ending with zero one and two  

        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp element at position zero TO the REMAINDER AFTER DIVIDING the SUM of two MULTIPLIED BY the value at dp element at position zero PLUS one BY MOD  
            ELSE IF num EQUALS one  
                SET dp element at position one TO the REMAINDER AFTER DIVIDING the SUM of two MULTIPLIED BY the value at dp element at position one PLUS the value at dp element at position zero BY MOD  
            ELSE IF num EQUALS two  
                SET dp element at position two TO the REMAINDER AFTER DIVIDING the SUM of two MULTIPLIED BY the value at dp element at position two PLUS the value at dp element at position one BY MOD  
            END IF  
        END FOR  
        RETURN the value at dp element at position two  
    END FUNCTION  
END CLASS
