CLASS Solution  
    FUNCTION findDiagonalOrder(mat)  
        IF mat IS EMPTY OR element at position zero of mat IS EMPTY  
            RETURN empty list  
        END IF  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET result TO empty list  
        SET r TO zero  
        SET c TO zero  
        SET direction TO one  
        
        FOR each iteration FROM zero TO m MULTIPLIED BY n MINUS one  
            APPEND element at position r of mat at position c TO result  
            SET new_r TO r PLUS negative one IF direction EQUALS one OTHERWISE r PLUS one  
            SET new_c TO c PLUS one IF direction EQUALS one OTHERWISE c MINUS one  
            
            IF new_r LESS THAN zero OR new_r EQUALS m OR new_c LESS THAN zero OR new_c EQUALS n  
                IF direction EQUALS one  
                    IF new_c EQUALS n  
                        SET new_r TO r PLUS one  
                        SET new_c TO c  
                    ELSE  
                        SET new_r TO zero  
                        SET new_c TO c PLUS one  
                    END IF  
                ELSE  
                    IF new_r EQUALS m  
                        SET new_r TO r  
                        SET new_c TO c PLUS one  
                    ELSE  
                        SET new_r TO r PLUS one  
                        SET new_c TO zero  
                    END IF  
                END IF  
                
                SET direction TO direction MULTIPLIED BY negative one  
            END IF  
            
            SET r TO new_r  
            SET c TO new_c  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
