CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER VALUE OF n MINUS one  
        END IF  
        
        SET candidates TO an empty SET  
        
        ADD ten RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD ten RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER VALUE OF the SUBSTRING of n FROM POSITION one TO the INTEGER RESULT OF the SUM of length PLUS one DIVIDED BY two  
        FOR i FROM minus one TO two MINUS one  
            SET new_prefix TO the STRING REPRESENTATION OF the SUM of prefix AND i  
            IF the REMAINDER of length DIVIDED BY two EQUALS zero  
                SET candidate TO the CONCATENATION of new_prefix AND the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION of new_prefix AND the REVERSE OF new_prefix FROM THE SECOND LAST CHARACTER TO the FIRST CHARACTER  
            END IF  
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF IT IS PRESENT  
        
        SET closest TO the ELEMENT IN candidates WITH the MINIMUM VALUE ACCORDING TO the TUPLE COMPRISING the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN the ELEMENT AND original_number AND the ELEMENT ITSELF  
        
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
