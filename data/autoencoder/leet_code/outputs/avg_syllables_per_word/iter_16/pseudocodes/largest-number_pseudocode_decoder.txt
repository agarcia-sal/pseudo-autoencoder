CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a new list containing the string representation of each element in nums  
        
        FUNCTION compare(x, y)  
            IF the concatenation of x followed by y IS GREATER THAN the concatenation of y followed by x  
                RETURN negative one  
            ELSE IF the concatenation of x followed by y IS LESS THAN the concatenation of y followed by x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT the list nums_str USING compare as the comparison key  
        
        IF the element at position zero of nums_str IS EQUAL TO zero as a string  
            RETURN zero as a string  
        END IF  
        
        RETURN the concatenation of all elements in nums_str in order  
    END FUNCTION  
END CLASS
