CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EMPTY OR element at position zero of matrix IS EMPTY  
            RETURN False  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER PART OF left PLUS right DIVIDED BY two  
            SET mid_value TO element at position the INTEGER PART OF mid DIVIDED BY n of matrix AT position mid MINUS the INTEGER PART OF mid DIVIDED BY n MULTIPLIED BY n of element at position the INTEGER PART OF mid DIVIDED BY n of matrix  
            
            IF mid_value EQUALS target  
                RETURN True  
            ELSE IF mid_value IS LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
