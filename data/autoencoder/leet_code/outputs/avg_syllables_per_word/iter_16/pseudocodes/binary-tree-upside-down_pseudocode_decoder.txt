CLASS TreeNode  
    FUNCTION __init__ with parameters self val EQUALS zero left EQUALS None right EQUALS None  
        SET the val attribute of self TO val  
        SET the left attribute of self TO left  
        SET the right attribute of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values list  
    IF the values list IS empty  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the front of the queue  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET the left attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET the right attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUAL TO the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with the left attributes of p and q AND the result of is_same_tree called with the right attributes of p and q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree with parameters self root optional TreeNode  
        IF root IS None OR the left attribute of root IS None  
            RETURN root  
        END IF  
        SET new_root TO the result of calling upsideDownBinaryTree on the left attribute of root  
        SET the left attribute of the left attribute of root TO the right attribute of root  
        SET the right attribute of the left attribute of root TO root  
        SET the left attribute of root TO None  
        SET the right attribute of root TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
