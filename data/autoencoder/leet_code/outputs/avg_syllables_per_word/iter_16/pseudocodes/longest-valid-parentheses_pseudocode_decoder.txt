CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list initialized with negative one as a single element  
        SET max_length TO zero  
        FOR each index i AND character char IN the enumeration of the string s  
            IF character char EQUALS open parenthesis  
                APPEND the index i TO the stack  
            ELSE  
                REMOVE the last element FROM the stack  
                IF the stack is empty  
                    APPEND the index i TO the stack  
                ELSE  
                    SET difference TO the index i MINUS the last element at top of the stack  
                    SET max_length TO the greater of max_length AND difference  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
