CLASS Solution  
    FUNCTION isNStraightHand(hand AS List of Integer, groupSize AS Integer) RETURNS Boolean  
        IF the LENGTH OF hand MODULO groupSize IS NOT EQUAL TO zero  
            RETURN false  
        END IF  
        
        SET count TO a COLLECTION REPRESENTING FREQUENCY OF EACH ELEMENT IN hand  
        SET unique_cards TO A LIST OF UNIQUE ELEMENTS IN count ORDERED FROM LOWEST TO HIGHEST  
        
        FOR each card IN unique_cards  
            IF the FREQUENCY OF card IN count IS GREATER THAN zero  
                SET needed TO the FREQUENCY OF card IN count  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF the FREQUENCY OF the ELEMENT AT POSITION card PLUS index IN count IS LESS THAN needed  
                        RETURN false  
                    END IF  
                    DECREMENT the FREQUENCY OF the ELEMENT AT POSITION card PLUS index IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
