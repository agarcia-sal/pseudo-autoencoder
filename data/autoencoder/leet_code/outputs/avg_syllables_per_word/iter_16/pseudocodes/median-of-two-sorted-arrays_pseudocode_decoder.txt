CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1 LIST OF INTEGER nums2 LIST OF INTEGER) RETURNS FLOAT  
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2  
            SET temporary_variable TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary_variable  
        END IF  
          
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
          
        WHILE low IS LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION OF the SUM OF low AND high BY two  
            SET partitionY TO the INTEGER DIVISION OF the SUM OF x AND y AND one BY two MINUS partitionX  
              
            IF partitionX EQUALS zero  
                SET maxLeftX TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftX TO element at position partitionX MINUS one OF nums1  
            END IF  
              
            IF partitionX EQUALS x  
                SET minRightX TO POSITIVE_INFINITY  
            ELSE  
                SET minRightX TO element at position partitionX OF nums1  
            END IF  
              
            IF partitionY EQUALS zero  
                SET maxLeftY TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftY TO element at position partitionY MINUS one OF nums2  
            END IF  
              
            IF partitionY EQUALS y  
                SET minRightY TO POSITIVE_INFINITY  
            ELSE  
                SET minRightY TO element at position partitionY OF nums2  
            END IF  
              
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX  
                IF the MODULO OF the SUM OF x AND y BY two EQUALS zero  
                    RETURN the SUM OF the GREATER OF maxLeftX AND maxLeftY AND the LESSER OF minRightX AND minRightY DIVIDED BY two  
                ELSE  
                    RETURN the GREATER OF maxLeftX AND maxLeftY  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
