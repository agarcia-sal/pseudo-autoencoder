CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a list of size size1 plus one where each element is a list of size two RAISED TO THE POWER OF size2 filled with infinity values  
        SET element at position zero of element at position zero of dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET combined_mask TO mask OR THE VALUE OF one LEFT SHIFTED BY j  
                    SET current_value TO element at position i plus one of dp at position combined_mask  
                    SET candidate_value TO element at position i of dp at position mask PLUS element at position i of cost at position j  
                    IF candidate_value LESS THAN current_value  
                        SET element at position i plus one of dp at position combined_mask TO candidate_value  
                    END IF  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    SET bit_mask TO the VALUE OF one LEFT SHIFTED BY j  
                    IF bit_mask AND mask IS NOT EQUAL TO zero  
                        SET current_value TO element at position i plus one of dp at position mask  
                        SET alternative_value TO element at position i plus one of dp at position mask WITH bit at position j TO zero PLUS element at position i of cost at position j  
                        IF alternative_value LESS THAN current_value  
                            SET element at position i plus one of dp at position mask TO alternative_value  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET full_mask TO two RAISED TO THE POWER OF size2 MINUS one  
        RETURN element at position size1 of dp at position full_mask  
    END FUNCTION  
END CLASS
