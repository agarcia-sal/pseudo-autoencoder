CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO an empty mapping from character TO set of characters with default empty set  
        SET indegree TO an empty mapping from character TO integer with default zero  
        SET all_chars TO the set of all characters appearing in all elements of words  
        
        FOR index FROM zero TO the LENGTH OF words MINUS one  
            SET word1 TO element at position index of words  
            SET word2 TO element at position index PLUS one of words  
            SET min_length TO the minimum of the LENGTH OF word1 AND the LENGTH OF word2  
            SET found TO false  
            FOR position FROM zero TO min_length MINUS one  
                IF element at position position of word1 IS NOT EQUAL TO element at position position of word2  
                    IF element at position position of word2 IS NOT IN the set mapped by element at position position of word1 in graph  
                        ADD element at position position of word2 TO the set mapped by element at position position of word1 in graph  
                        INCREMENT the integer mapped by element at position position of word2 in indegree BY one  
                    SET found TO true  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF found IS false AND the LENGTH OF word1 IS GREATER THAN the LENGTH OF word2  
                RETURN an empty string  
            END IF  
        END FOR  
        
        SET queue TO a double ended queue containing all characters in all_chars whose mapped integer in indegree IS EQUAL TO zero  
        SET result TO an empty list  
        
        WHILE queue is not empty  
            REMOVE and RETURN the element at the front of queue INTO char  
            APPEND char TO result  
            FOR each neighbor IN the set mapped by char in graph  
                DECREMENT the integer mapped by neighbor in indegree BY one  
                IF the integer mapped by neighbor in indegree IS EQUAL TO zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF result IS NOT EQUAL TO the LENGTH OF all_chars  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters in the list result  
    END FUNCTION  
END CLASS
