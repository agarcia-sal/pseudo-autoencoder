CLASS Solution  
    FUNCTION shortestSubarray(nums AS List OF integers, k AS integer) RETURNS integer  
        SET prefix_sum TO a new list containing zero as its first element  
        FOR each num IN nums  
            APPEND the sum of the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        SET dq TO a new deque data structure  
        SET min_length TO positive infinity  
        
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND the difference between current_sum AND the element at position at the first element of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value BETWEEN min_length AND the difference BETWEEN i AND the element removed from the left end of dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at position at the last element of dq in prefix_sum  
                REMOVE the element from the right end of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
