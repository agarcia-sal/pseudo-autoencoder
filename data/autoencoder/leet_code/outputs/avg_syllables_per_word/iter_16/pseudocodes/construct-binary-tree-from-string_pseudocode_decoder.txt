CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO new ListNode constructed with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the last position  
        SET node TO new ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list called with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    INITIALIZE queue as an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the left attribute of node TO new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the right attribute of node TO new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree called with the left attribute of p AND the left attribute of q AND is_same_tree called with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (the element at position index of s IS A DIGIT OR the element at position index of s EQUALS the character MINUS)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO new TreeNode constructed with the integer conversion of the substring from position start TO one LESS THAN index of s  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character OPENING PARENTHESIS  
                SET the left attribute of node AND index TO the result of parse_tree called with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character OPENING PARENTHESIS  
                SET the right attribute of node AND index TO the result of parse_tree called with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character CLOSING PARENTHESIS  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND discard the second returned value TO the result of parse_tree called with zero  
        RETURN root  
    END FUNCTION  
END CLASS
