CLASS Solution  
    FUNCTION rotate(nums AS List OF integers, k AS integer) RETURNS None  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  

        FUNCTION reverse(start AS integer, end AS integer) RETURNS None  
            WHILE start IS LESS THAN end  
                SET temporary TO the element at position start of nums  
                SET the element at position start of nums TO the element at position end of nums  
                SET the element at position end of nums TO temporary  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  

        CALL reverse WITH start zero AND end n MINUS one  
        CALL reverse WITH start zero AND end k MINUS one  
        CALL reverse WITH start k AND end n MINUS one  
    END FUNCTION  
END CLASS
