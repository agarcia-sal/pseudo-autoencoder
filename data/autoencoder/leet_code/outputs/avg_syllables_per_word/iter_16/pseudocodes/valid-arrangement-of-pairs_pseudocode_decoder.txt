CLASS Solution  
    FUNCTION validArrangement(pairs)  
        SET graph TO a default dictionary with list as default value  
        SET out_degree TO a counter dictionary  
        SET in_degree TO a counter dictionary  
        
        FOR each element u AND element v IN pairs  
            APPEND element v TO the list at element u in graph  
            INCREMENT the count of element u in out_degree BY one  
            INCREMENT the count of element v in in_degree BY one  
        END FOR  
        
        SET start_node TO none  
        FOR each element node IN graph  
            IF the count of node in out_degree EQUALS the count of node in in_degree PLUS one  
                SET start_node TO node  
                BREAK the loop  
            END IF  
        END FOR  
        
        IF start_node IS none  
            SET start_node TO the first element of the sequence of keys in graph  
        END IF  
        
        SET path TO an empty list  
        SET stack TO a list containing start_node  
        
        WHILE the stack is not empty  
            SET u TO the last element of stack  
            IF the list at element u in graph is not empty  
                REMOVE the last element v FROM the list at element u in graph  
                APPEND element v TO stack  
            ELSE  
                REMOVE the last element from stack AND APPEND it TO path  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO an empty list  
        FOR index FROM zero TO the length of path MINUS one  
            APPEND a list containing element at index of path AND element at index PLUS one of path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
