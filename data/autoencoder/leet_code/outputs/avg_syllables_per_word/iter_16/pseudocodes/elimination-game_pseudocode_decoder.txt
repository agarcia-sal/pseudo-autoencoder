CLASS Solution  
    FUNCTION lastRemaining(n)  
        SET remaining TO n  
        SET step TO one  
        SET left_to_right TO true  
        SET head TO one  
        
        WHILE remaining GREATER THAN one holds  
            IF left_to_right EQUALS true OR remaining MODULO two EQUALS one  
                INCREMENT head BY step  
            END IF  
            MULTIPLY step BY two AND ASSIGN TO step  
            ASSIGN remaining DIVIDED BY two TO remaining AS INTEGER DIVISION  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        
        RETURN head  
    END FUNCTION  
END CLASS
