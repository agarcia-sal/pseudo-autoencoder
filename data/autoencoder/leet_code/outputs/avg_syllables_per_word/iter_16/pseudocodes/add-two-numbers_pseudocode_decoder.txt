CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val WITH DEFAULT ZERO AND next WITH DEFAULT NONE  
        SET the val ATTRIBUTE OF this OBJECT TO val  
        SET the next ATTRIBUTE OF this OBJECT TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN NONE  
    END IF  
    SET head TO A NEW ListNode WITH THE ELEMENT at POSITION zero OF values  
    SET p TO head  
    FOR each val IN VALUES STARTING FROM POSITION one TO THE END  
        SET node TO A NEW ListNode WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS NONE AND p2 IS NONE  
        RETURN TRUE  
    END IF  
    IF p1 IS NONE OR p2 IS NONE  
        RETURN FALSE  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND CALL is_same_list WITH ARGUMENTS the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
        SET dummy_head TO A NEW ListNode WITH val ZERO  
        SET current TO dummy_head  
        SET carry TO ZERO  
        
        WHILE l1 IS NOT NONE OR l2 IS NOT NONE OR carry IS NOT ZERO  
            IF l1 IS NOT NONE  
                SET val1 TO the val ATTRIBUTE OF l1  
            ELSE  
                SET val1 TO ZERO  
            END IF  
            IF l2 IS NOT NONE  
                SET val2 TO the val ATTRIBUTE OF l2  
            ELSE  
                SET val2 TO ZERO  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY TEN  
            SET current dot next TO A NEW ListNode WITH val EQUAL TO total MODULO TEN  
            SET current TO the next ATTRIBUTE OF current  
            
            IF l1 IS NOT NONE  
                SET l1 TO the next ATTRIBUTE OF l1  
            END IF  
            IF l2 IS NOT NONE  
                SET l2 TO the next ATTRIBUTE OF l2  
            END IF  
        END WHILE  
        
        RETURN the next ATTRIBUTE OF dummy_head  
    END FUNCTION  
END CLASS
