CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a collection that counts each occurrence of elements in nums  
        SET unique_nums TO the sorted list of the unique elements in count  
        SET dp TO a list of zeros with a length equal to the length of unique_nums plus one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the count of the element at position zero of unique_nums  
        FOR index FROM two TO the length of unique_nums plus one  
            SET num TO the element at the position index minus one of unique_nums  
            IF num EQUALS the element at the position index minus two of unique_nums PLUS one  
                SET the element at position index of dp TO the maximum of the element at position index minus one of dp AND the element at position index minus two of dp PLUS num MULTIPLIED BY the count of num  
            ELSE  
                SET the element at position index of dp TO the element at position index minus one of dp PLUS num MULTIPLIED BY the count of num  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
