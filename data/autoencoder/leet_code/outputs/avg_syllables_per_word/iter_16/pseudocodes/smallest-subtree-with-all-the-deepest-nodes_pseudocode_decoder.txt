CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW DEQUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE element REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right q.right USING LOGICAL AND  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS self root AS TreeNode  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO THE RESULT OF dfs WITH PARAMETER node.left  
            SET right_node AND right_depth TO THE RESULT OF dfs WITH PARAMETER node.right  
            IF left_depth IS GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth IS GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN THE FIRST ELEMENT OF THE RESULT OF dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
