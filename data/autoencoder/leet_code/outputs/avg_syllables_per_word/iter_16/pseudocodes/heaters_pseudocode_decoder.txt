CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        CALL sort ON houses  
        CALL sort ON heaters  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN LENGTH OF heaters MINUS one AND  
                  ABSOLUTE VALUE OF element at position j of heaters MINUS house GREATER THAN OR EQUAL TO  
                  ABSOLUTE VALUE OF element at position j plus one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO THE GREATER VALUE BETWEEN min_radius AND ABSOLUTE VALUE OF element at position j of heaters MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
