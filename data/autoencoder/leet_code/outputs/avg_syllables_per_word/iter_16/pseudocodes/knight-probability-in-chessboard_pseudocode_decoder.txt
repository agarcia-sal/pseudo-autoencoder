CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO a list of pairs consisting of negative two and negative one negative two and one negative one and negative two negative one and two one and negative two one and two two and negative one two and one  
        SET dp TO a two dimensional list of dimensions n by n with all elements set to zero  
        SET element at position row and column of dp TO one  
        FOR each move iteration FROM one TO k  
            SET new_dp TO a two dimensional list of dimensions n by n with all elements set to zero  
            FOR r FROM zero TO n MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF element at position r and c of dp EQUALS zero  
                        CONTINUE to next iteration of inner loop  
                    END IF  
                    FOR each pair dr and dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr and nc of new_dp BY element at position r and c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each probability_value IN row_list  
                INCREMENT total_probability BY probability_value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
