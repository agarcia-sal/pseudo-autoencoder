CLASS Solution  
    FUNCTION merge(intervals)  
        CALL sort intervals BY the element at position zero of each sublist  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the element at position one of the element at the last position of merged IS LESS THAN the element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one of the element at the last position of merged TO maximum of the element at position one of the element at the last position of merged AND the element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
