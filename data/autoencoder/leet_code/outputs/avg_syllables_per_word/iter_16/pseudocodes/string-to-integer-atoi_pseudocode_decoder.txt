CLASS Solution  
    FUNCTION myAtoi(s as string) RETURNS integer  
        SET INT_MAX TO two RAISED TO THE POWER OF thirty ONE MINUS one  
        SET INT_MIN TO negative two RAISED TO THE POWER OF thirty ONE  
        
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        
        WHILE index IS LESS THAN n AND the element at position index of s EQUALS a space character  
            INCREMENT index BY one  
        END WHILE  
        
        IF index IS LESS THAN n AND (the element at position index of s EQUALS the plus character OR the element at position index of s EQUALS the minus character)  
            SET sign TO negative one IF the element at position index of s EQUALS the minus character OTHERWISE one  
            INCREMENT index BY one  
        END IF  
        
        WHILE index IS LESS THAN n AND the element at position index of s IS A DIGIT  
            SET digit TO the integer value of the element at position index of s  
            IF result IS GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
