CLASS Solution  
    FUNCTION canDistribute(nums AS List OF Integer, quantity AS List OF Integer) RETURNS Boolean  
        SET count TO the list of frequencies of each distinct element in nums  
        SORT quantity in descending order to prioritize satisfying larger quantities first  
        FUNCTION can_satisfy(index) RETURNS Boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN True  
            END IF  
            FOR position FROM zero TO the LENGTH OF count MINUS one  
                IF the element at position in count IS GREATER THAN OR EQUAL TO the element at position index in quantity THEN  
                    DECREMENT the element at position in count BY the element at position index in quantity  
                    IF calling can_satisfy with argument index PLUS one RETURNS True THEN  
                        RETURN True  
                    END IF  
                    INCREMENT the element at position in count BY the element at position index in quantity  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN the result of calling can_satisfy with argument zero  
    END FUNCTION  
END CLASS
