CLASS Solution  
    FUNCTION kInversePairs WITH PARAMETERS n AND k  
        SET MOD TO one billion PLUS seven  
        SET dp TO two dimensional list OF zeros WITH n PLUS one rows AND k PLUS one columns  
        SET element at position zero of row zero of dp TO one  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position j of row i MINUS one of dp  
                COMPUTE prefix_sum MODULO MOD  
                IF j IS GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position j MINUS i of row i MINUS one of dp PLUS MOD  
                    COMPUTE prefix_sum MODULO MOD  
                END IF  
                SET element at position j of row i of dp TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN element at position k of row n of dp  
    END FUNCTION  
END CLASS
