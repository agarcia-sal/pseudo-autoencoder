CLASS Solution  
    FUNCTION longestPalindrome(word1, word2)  
        SET s TO concatenation of word1 and word2  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list OF zero WITH size n BY n  
        
        FOR index i FROM zero TO n MINUS one  
            SET element at position i AT position i OF dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR index i FROM zero TO n MINUS length  
                SET index j TO i PLUS length MINUS one  
                IF element at position i OF s EQUALS element at position j OF s  
                    SET element at position i AT position j OF dp TO element at position i PLUS one AT position j MINUS one OF dp PLUS two  
                    IF i LESS THAN the LENGTH OF word1 AND j GREATER THAN OR EQUAL TO the LENGTH OF word1  
                        SET max_len TO the GREATER VALUE OF max_len AND element at position i AT position j OF dp  
                    END IF  
                ELSE  
                    SET element at position i AT position j OF dp TO the GREATER VALUE OF element at position i PLUS one AT position j OF dp AND element at position i AT position j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
