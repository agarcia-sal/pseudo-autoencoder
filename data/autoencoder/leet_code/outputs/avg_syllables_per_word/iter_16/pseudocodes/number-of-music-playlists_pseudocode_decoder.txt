CLASS Solution  
    FUNCTION numMusicPlaylists with parameters n integer goal integer k integer RETURNS integer  
        SET MODULO TO ten POWERED BY nine PLUS seven  
        SET dp TO a two dimensional list with goal PLUS one rows and n PLUS one columns filled with zero  
        SET element at position zero zero of dp TO one  
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                INCREMENT element at position i j of dp BY element at position i MINUS one j MINUS one of dp MULTIPLIED BY n MINUS j MINUS one  
                SET element at position i j of dp TO element at position i j of dp MODULO MODULO  
                IF j GREATER THAN k THEN  
                    INCREMENT element at position i j of dp BY element at position i MINUS one j of dp MULTIPLIED BY j MINUS k  
                    SET element at position i j of dp TO element at position i j of dp MODULO MODULO  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position goal n of dp  
    END FUNCTION  
END CLASS
