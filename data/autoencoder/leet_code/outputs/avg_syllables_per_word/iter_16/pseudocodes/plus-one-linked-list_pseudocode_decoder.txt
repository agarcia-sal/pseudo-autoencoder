CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULTING TO zero AND next DEFAULTING TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO A NEW INSTANCE OF ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO A NEW INSTANCE OF ListNode INITIALIZED WITH val  
        SET the element referenced by p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the VALUE OF p1.val EQUALS the VALUE OF p2.val AND is_same_list WITH PARAMETERS the element referenced by p1.next AND the element referenced by p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETERS head OPTIONAL OF TYPE ListNode  
        FUNCTION reverseList WITH PARAMETER node OF TYPE ListNode  
            SET prev TO none  
            WHILE node HOLDS a non none value  
                SET next_node TO the element referenced by node.next  
                SET node.next TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList WITH PARAMETER head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current HOLDS a non none value AND carry EQUALS one  
            SET current.val TO current.val PLUS carry  
            SET carry TO current.val DIVIDED BY ten  
            SET current.val TO current.val MODULO ten  
            
            IF the element referenced by current.next EQUALS none AND carry EQUALS one  
                SET current.next TO A NEW INSTANCE OF ListNode INITIALIZED WITH carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the element referenced by current.next  
        END WHILE  
        
        RETURN reverseList WITH PARAMETER reversed_head  
    END FUNCTION  
END CLASS
