CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN WHETHER sub IS EQUAL TO THE REVERSED ORDER OF sub  
        END FUNCTION  
        
        SET n TO THE LENGTH OF s  
        
        SET is_pal TO A TWO DIMENSIONAL COLLECTION OF BOOLEAN VALUES INITIALIZED TO FALSE WITH DIMENSIONS n BY n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF THE ELEMENT AT POSITION i OF s IS EQUAL TO THE ELEMENT AT POSITION j OF s AND EITHER THE DIFFERENCE BETWEEN j AND i IS LESS THAN OR EQUAL TO one OR THE VALUE AT POSITION i PLUS one AND j MINUS one OF is_pal IS TRUE  
                    SET THE VALUE AT POSITION i AND j OF is_pal TO TRUE  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF THE VALUE AT POSITION zero AND i MINUS one OF is_pal IS TRUE  
                FOR j FROM i TO n MINUS two  
                    IF THE VALUE AT POSITION i AND j OF is_pal IS TRUE AND THE VALUE AT POSITION j PLUS one AND n MINUS one OF is_pal IS TRUE  
                        RETURN TRUE  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
