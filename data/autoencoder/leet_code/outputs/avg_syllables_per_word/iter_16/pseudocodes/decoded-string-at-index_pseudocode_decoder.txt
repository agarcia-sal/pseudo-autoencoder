CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each char IN s  
            IF the character is a digit  
                SET length TO length MULTIPLIED BY the numerical value of the character  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN the reverse order of s  
            SET k TO the remainder of k DIVIDED BY length  
            IF k EQUALS zero AND the character is an alphabetic letter  
                RETURN the character  
            END IF  
            IF the character is a digit  
                SET length TO length DIVIDED BY the numerical value of the character with integer division  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
