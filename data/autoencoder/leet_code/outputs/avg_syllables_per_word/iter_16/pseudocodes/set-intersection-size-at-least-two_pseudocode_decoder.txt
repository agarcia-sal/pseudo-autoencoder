CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals SORTED BY the element at position one of each interval AND IF those values ARE EQUAL THEN BY the NEGATION OF the element at position zero of each interval DESCENDING  
        SET first TO the element at position one of the interval at position zero MINUS one  
        SET second TO the element at position one of the interval at position zero  
        SET count TO two  
        FOR each start comma end IN the intervals from position one TO the end  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
