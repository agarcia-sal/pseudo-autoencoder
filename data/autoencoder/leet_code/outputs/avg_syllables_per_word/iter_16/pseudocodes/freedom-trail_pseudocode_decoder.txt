CLASS Solution  
    FUNCTION findRotateSteps(ring, key)  
        SET char_positions TO a new dictionary with default value as an empty list  
        FOR each index_from_zero AND character IN ring  
            APPEND index_from_zero TO the element at character of char_positions  
        END FOR  
        
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i, prev_pos)  
            IF i EQUALS the LENGTH OF key THEN  
                RETURN zero  
            END IF  
            
            SET min_steps TO positive infinity  
            FOR each pos IN the element at position i of key within char_positions  
                SET clockwise TO the remainder of pos MINUS prev_pos DIVIDED BY n  
                SET anticlockwise TO the remainder of prev_pos MINUS pos DIVIDED BY n  
                SET steps TO the smaller between clockwise AND anticlockwise PLUS one PLUS dp with parameters i INCREMENTED BY one AND pos  
                SET min_steps TO the smaller between min_steps AND steps  
            END FOR  
            
            RETURN min_steps  
        END FUNCTION  
        
        RETURN dp with parameters zero AND zero  
    END FUNCTION  
END CLASS
