CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
  
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with LENGTH equal to two TIMES n PLUS one  
  
        FOR each index i FROM zero TO two TIMES n MINUS one  
            PUSH the NEGATION OF element at position i of nums INTO min_heap USING heap operation  
            IF the LENGTH OF min_heap IS GREATER THAN n  
                REMOVE the smallest element FROM min_heap USING heap operation  
            END IF  
            IF the LENGTH OF min_heap IS EQUAL TO n  
                SET element at position i of min_sums TO the NEGATION OF the SUM OF all elements in min_heap  
            END IF  
        END FOR  
  
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with LENGTH equal to two TIMES n PLUS one  
  
        FOR each index i FROM three TIMES n MINUS one DOWN TO n  
            PUSH element at position i of nums INTO max_heap USING heap operation  
            IF the LENGTH OF max_heap IS GREATER THAN n  
                REMOVE the smallest element FROM max_heap USING heap operation  
            END IF  
            IF the LENGTH OF max_heap IS EQUAL TO n  
                SET element at position i of max_sums TO the SUM OF all elements in max_heap  
            END IF  
        END FOR  
  
        SET min_diff TO infinity  
  
        FOR each index i FROM n MINUS one TO two TIMES n MINUS one  
            SET min_diff TO the SMALLER VALUE BETWEEN min_diff AND element at position i of min_sums MINUS element at position i PLUS one of max_sums  
        END FOR  
  
        RETURN min_diff  
    END FUNCTION  
END CLASS
