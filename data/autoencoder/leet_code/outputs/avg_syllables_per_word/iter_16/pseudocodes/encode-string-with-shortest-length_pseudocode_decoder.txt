CLASS Solution  
    FUNCTION encode(s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN the LENGTH OF the STRING REPRESENTATION OF count PLUS the LENGTH OF substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO the substring from position i TO j of s  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five THEN  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR each k FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
                IF the REMAINDER OF n DIVIDED BY k EQUALS zero THEN  
                    SET repeat_count TO the INTEGER DIVISION OF n BY k  
                    IF substring EQUALS the substring from position zero TO k MINUS one of substring MULTIPLIED BY repeat_count THEN  
                        SET encoded_candidate TO the CONCATENATION OF the STRING REPRESENTATION OF repeat_count PLUS an opening bracket PLUS the RESULT OF dp WITH PARAMETERS i AND i PLUS k MINUS one PLUS a closing bracket  
                        IF the LENGTH OF encoded_candidate LESS THAN the LENGTH OF shortest THEN  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR each k FROM i TO j MINUS one  
                SET left_encoded TO the RESULT OF dp WITH PARAMETERS i AND k  
                SET right_encoded TO the RESULT OF dp WITH PARAMETERS k PLUS one AND j  
                SET combined TO the CONCATENATION OF left_encoded PLUS right_encoded  
                IF the LENGTH OF combined LESS THAN the LENGTH OF shortest THEN  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN the RESULT OF dp WITH PARAMETERS zero AND the LENGTH OF s MINUS one  
    END FUNCTION  
END CLASS
