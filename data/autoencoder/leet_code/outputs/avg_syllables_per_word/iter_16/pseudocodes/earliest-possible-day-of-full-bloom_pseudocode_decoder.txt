CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO sorted list of pairs combining element at position index of plantTime AND element at position index of growTime with index ranging over all valid indices sorted BY grow time in descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair consisting of plant AND grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            SET earliest_bloom TO the maximum of earliest_bloom AND bloom_day  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
