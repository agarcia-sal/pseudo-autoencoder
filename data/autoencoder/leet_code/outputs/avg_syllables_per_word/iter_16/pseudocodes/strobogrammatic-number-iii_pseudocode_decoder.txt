CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN a list containing an empty string  
            END IF  
            IF n EQUALS one  
                RETURN a list containing the strings zero one and eight  
            END IF  
            SET middles TO generate_strobogrammatic n MINUS two AND final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUAL TO final_length  
                    APPEND the string consisting of zero concatenated with middle concatenated with zero TO result  
                END IF  
                APPEND the string consisting of one concatenated with middle concatenated with one TO result  
                APPEND the string consisting of six concatenated with middle concatenated with nine TO result  
                APPEND the string consisting of eight concatenated with middle concatenated with eight TO result  
                APPEND the string consisting of nine concatenated with middle concatenated with six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM the LENGTH OF low TO the LENGTH OF high  
                FOR each num IN generate_strobogrammatic length AND length  
                    IF length EQUALS the LENGTH OF low AND num IS lexicographically less than low  
                        CONTINUE to next iteration  
                    END IF  
                    IF length EQUALS the LENGTH OF high AND num IS lexicographically greater than high  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic low AND high  
    END FUNCTION  
END CLASS
