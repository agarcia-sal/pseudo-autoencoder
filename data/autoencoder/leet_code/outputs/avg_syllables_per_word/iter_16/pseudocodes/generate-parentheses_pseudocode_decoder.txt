CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s INITIALIZED TO empty string left INITIALIZED TO zero right INITIALIZED TO zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis left INCREMENTED BY one right UNCHANGED  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis left UNCHANGED right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
