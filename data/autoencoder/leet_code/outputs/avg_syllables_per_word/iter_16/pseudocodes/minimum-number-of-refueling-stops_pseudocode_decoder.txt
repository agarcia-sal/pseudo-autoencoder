CLASS Solution  
    FUNCTION minRefuelStops(target INTEGER, startFuel INTEGER, stations LIST OF LIST OF INTEGER)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list containing target and zero TO stations  

        FOR each position AND fuel IN stations  
            WHILE max_heap is not empty AND current_fuel IS LESS THAN position  
                SET current_fuel TO current_fuel MINUS the element REMOVED from max_heap with the largest value  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel IS LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative fuel TO max_heap preserving max-heap behavior  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
