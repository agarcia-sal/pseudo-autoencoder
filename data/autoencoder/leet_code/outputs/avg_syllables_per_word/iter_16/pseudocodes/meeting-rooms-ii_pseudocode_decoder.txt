CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  

        SORT intervals BY element at position zero of interval  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF min_heap NOT EMPTY AND element at position zero of meeting GREATER THAN OR EQUAL TO element at position zero of min_heap  
                CALL remove the smallest element from min_heap  
            END IF  
            CALL add the element at position one of meeting TO min_heap  
        END FOR  

        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
