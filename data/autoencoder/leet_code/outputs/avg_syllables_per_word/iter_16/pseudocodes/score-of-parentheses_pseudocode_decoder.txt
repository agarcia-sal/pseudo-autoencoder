CLASS Solution  
    FUNCTION scoreOfParentheses WITH PARAMETER s  
        SET stack TO an empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS open parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET popped_value TO the element removed from the end of stack  
                IF score EQUALS zero  
                    SET score TO popped_value PLUS one  
                ELSE  
                    SET score TO popped_value PLUS two MULTIPLIED BY score  
                END IF  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
