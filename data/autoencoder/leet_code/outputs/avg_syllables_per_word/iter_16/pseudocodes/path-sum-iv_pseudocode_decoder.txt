CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the result of num DIVIDED BY ten THEN MODULO by ten  
            SET value TO num MODULO by ten  
            SET the element at position depth and pos of tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at position depth and pos of tree if it exists otherwise nothing  
            IF current_value is nothing  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair consisting of depth PLUS one and two MULTIPLIED BY pos MINUS one  
            SET right_child TO a pair consisting of depth PLUS one and two MULTIPLIED BY pos  
            
            IF left_child is not a key in tree AND right_child is not a key in tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO the result of dfs with depth as the first element of left_child, pos as the second element of left_child, and new_path_sum as path_sum  
            SET right_sum TO the result of dfs with depth as the first element of right_child, pos as the second element of right_child, and new_path_sum as path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN the result of dfs with one as depth, one as pos, and zero as path_sum  
    END FUNCTION  
END CLASS
