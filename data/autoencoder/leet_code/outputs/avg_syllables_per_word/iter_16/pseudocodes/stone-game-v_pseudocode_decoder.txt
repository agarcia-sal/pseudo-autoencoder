CLASS Solution  
    FUNCTION stoneGameV(stoneValue)  
        SET prefix_sum TO a list of zero elements with LENGTH EQUAL TO the LENGTH OF stoneValue PLUS one  
        FOR each index i FROM zero TO the LENGTH OF stoneValue MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of stoneValue  
        END FOR  
  
        FUNCTION get_range_sum(left, right)  
            RETURN element at position right PLUS one of prefix_sum MINUS element at position left of prefix_sum  
        END FUNCTION  
  
        FUNCTION dp(left, right)  
            IF left EQUALS right THEN  
                RETURN zero  
            END IF  
  
            SET max_score TO zero  
            FOR each index i FROM left TO right MINUS one  
                SET left_sum TO the result of get_range_sum with left AND i  
                SET right_sum TO the result of get_range_sum with i PLUS one AND right  
  
                IF left_sum LESS THAN right_sum THEN  
                    SET max_score TO the maximum value BETWEEN max_score AND left_sum PLUS the result of dp with left AND i  
                ELSE IF left_sum GREATER THAN right_sum THEN  
                    SET max_score TO the maximum value BETWEEN max_score AND right_sum PLUS the result of dp with i PLUS one AND right  
                ELSE  
                    SET max_score TO the maximum value among max_score AND left_sum PLUS the result of dp with left AND i AND right_sum PLUS the result of dp with i PLUS one AND right  
                END IF  
            END FOR  
  
            RETURN max_score  
        END FUNCTION  
  
        RETURN the result of dp with zero AND the LENGTH OF stoneValue MINUS one  
    END FUNCTION  
END CLASS
