CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO list of sets of characters for each word in words  
        SET max_product TO zero  
        FOR index i FROM zero TO the LENGTH OF words MINUS one  
            FOR index j FROM i PLUS one TO the LENGTH OF words MINUS one  
                IF the intersection of element at position i of word_sets AND element at position j of word_sets EQUALS the empty set  
                    SET product TO the LENGTH OF element at position i of words MULTIPLIED BY the LENGTH OF element at position j of words  
                    IF max_product LESS THAN product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  
END CLASS
