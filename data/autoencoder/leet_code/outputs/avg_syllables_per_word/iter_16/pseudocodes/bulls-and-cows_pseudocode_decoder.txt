CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO DICTIONARY WITH DEFAULT VALUE zero  
        SET guess_count TO DICTIONARY WITH DEFAULT VALUE zero  
        FOR each pair consisting of character s FROM secret AND character g FROM guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value FOR key s IN secret_count BY one  
                INCREMENT the value FOR key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN the keys OF secret_count  
            SET cows TO cows PLUS the MINIMUM OF the value FOR digit IN secret_count AND the value FOR digit IN guess_count  
        END FOR  
        RETURN the concatenation OF the string REPRESENTING bulls FOLLOWED BY letter A FOLLOWED BY the string REPRESENTING cows FOLLOWED BY letter B  
    END FUNCTION  
END CLASS
