CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze AS list of list of integer AND start AS list of integer AND destination AS list of integer  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO list containing tuple of one and zero AND tuple of MINUS one and zero AND tuple of zero and one AND tuple of zero and MINUS one  
        SET start TO tuple converted FROM start  
        SET destination TO tuple converted FROM destination  
        SET visited TO empty set  
        SET priority queue TO list containing tuple of zero AND start  

        WHILE priority queue is not empty  
            REMOVE and OBTAIN the element with the smallest first component FROM priority queue ASSIGN first component TO dist AND second component TO tuple ASSIGN first element of tuple TO x AND second element of tuple TO y  
            
            IF tuple of x and y EQUALS destination  
                RETURN dist  
            END IF  
            
            IF tuple of x and y IS IN visited  
                CONTINUE TO the next iteration of the loop  
            END IF  
            
            ADD tuple of x and y TO visited  
            
            FOR each tuple of dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze at position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF tuple of new_x and new_y IS NOT IN visited  
                    ADD tuple of dist PLUS count AND tuple of new_x and new_y TO priority queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
