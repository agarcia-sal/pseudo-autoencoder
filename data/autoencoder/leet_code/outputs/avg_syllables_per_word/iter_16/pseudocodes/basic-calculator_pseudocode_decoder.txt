CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO an empty list  
            SET sign TO the plus character  
            SET num TO zero  
            
            WHILE the LENGTH OF s is GREATER THAN zero holds  
                SET char TO the element at position zero of s and REMOVE element at position zero from s  
                
                IF the char IS a digit holds  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                
                IF char EQUALS the open parenthesis character holds  
                    SET num TO the result of calling helper with s  
                END IF  
                
                IF char IS NOT a digit AND char IS NOT a whitespace character OR the LENGTH OF s EQUALS zero holds  
                    IF sign EQUALS the plus character holds  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character holds  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the multiplication character holds  
                        SET the element at position the LENGTH OF stack MINUS one of stack TO the element at position the LENGTH OF stack MINUS one of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character holds  
                        SET the element at position the LENGTH OF stack MINUS one of stack TO the integer result of the element at position the LENGTH OF stack MINUS one of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS the close parenthesis character holds  
                    BREAK  
                END IF  
            END WHILE  
            
            RETURN the SUM OF elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper with the list formed from s  
    END FUNCTION  
END CLASS
