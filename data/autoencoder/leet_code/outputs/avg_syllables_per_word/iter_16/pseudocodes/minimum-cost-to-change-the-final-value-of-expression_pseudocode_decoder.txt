CLASS Solution  
    FUNCTION minOperationsToFlip(expression)  
        SET stack TO an empty list  
        
        FOR each element e IN expression  
            IF element e IS one of opening parenthesis or ampersand or vertical bar  
                APPEND a pair of element e and zero TO stack  
                CONTINUE to next iteration  
            ELSE IF element e IS a closing parenthesis  
                SET lastPair TO the result of removing the last element FROM stack  
                REMOVE the last element FROM stack  
            ELSE  
                SET lastPair TO a pair of element e and one  
            END IF  
            
            IF stack is not empty AND the first element of the last pair in stack IS one of ampersand or vertical bar  
                SET op TO the first element of the result of removing the last element FROM stack  
                SET a and costA TO the pair of the result of removing the last element FROM stack  
                SET b and costB TO lastPair  
                
                IF operator op IS ampersand  
                    IF a IS zero AND b IS zero  
                        SET lastPair TO a pair of zero and one PLUS the smaller of costA and costB  
                    ELSE IF a IS zero AND b IS one  
                        SET lastPair TO a pair of zero and one  
                    ELSE IF a IS one AND b IS zero  
                        SET lastPair TO a pair of zero and one  
                    ELSE  
                        SET lastPair TO a pair of one and the smaller of costA and costB  
                    END IF  
                ELSE  
                    IF a IS zero AND b IS zero  
                        SET lastPair TO a pair of zero and the smaller of costA and costB  
                    ELSE IF a IS zero AND b IS one  
                        SET lastPair TO a pair of one and one  
                    ELSE IF a IS one AND b IS zero  
                        SET lastPair TO a pair of one and one  
                    ELSE  
                        SET lastPair TO a pair of one and one PLUS the smaller of costA and costB  
                    END IF  
                END IF  
            END IF  
            
            APPEND lastPair TO stack  
        END FOR  
        
        RETURN the second element of the last pair in stack  
    END FUNCTION  
END CLASS
