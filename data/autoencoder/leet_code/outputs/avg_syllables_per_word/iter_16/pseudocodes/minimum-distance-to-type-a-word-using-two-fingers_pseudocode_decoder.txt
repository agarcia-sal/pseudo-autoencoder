CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS no value  
                RETURN zero  
            END IF  
            SET x1 TO the quotient of the Unicode code of character c1 minus the Unicode code of character A DIVIDED BY six  
            SET y1 TO the remainder of the Unicode code of character c1 minus the Unicode code of character A DIVIDED BY six  
            SET x2 TO the quotient of the Unicode code of character c2 minus the Unicode code of character A DIVIDED BY six  
            SET y2 TO the remainder of the Unicode code of character c2 minus the Unicode code of character A DIVIDED BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  
  
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO distance of f1 TO c PLUS dp of i INCREMENTED BY one COMMA c COMMA f2  
            SET d2 TO distance of f2 TO c PLUS dp of i INCREMENTED BY one COMMA f1 COMMA c  
            RETURN the lesser of d1 AND d2  
        END FUNCTION  
  
        RETURN dp of zero COMMA no value COMMA no value  
    END FUNCTION  
END CLASS
