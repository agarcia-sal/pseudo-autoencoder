CLASS Solution  
    FUNCTION isMatch with parameters s and p  
        SET dp TO a list of lists of False with dimensions the length of s plus one by the length of p plus one  
        SET element at position zero zero of dp TO True  
        FOR index j FROM one TO the length of p  
            IF element at position j minus one of p EQUALS the character star  
                SET element at position zero position j of dp TO element at position zero position j minus one of dp  
            END IF  
        END FOR  
        FOR index i FROM one TO the length of s  
            FOR index j FROM one TO the length of p  
                IF element at position j minus one of p EQUALS the character star  
                    SET element at position i position j of dp TO element at position i minus one position j of dp OR element at position i position j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS the character question mark OR element at position i minus one of s EQUALS element at position j minus one of p  
                    SET element at position i position j of dp TO element at position i minus one position j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s position the length of p of dp  
    END FUNCTION  
END CLASS
