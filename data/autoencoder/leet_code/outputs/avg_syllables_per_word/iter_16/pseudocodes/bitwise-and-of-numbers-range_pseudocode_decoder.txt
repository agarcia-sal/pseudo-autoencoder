CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER, right INTEGER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE left IS LESS THAN right  
            SET left TO left DIVIDED BY two USING INTEGER DIVISION  
            SET right TO right DIVIDED BY two USING INTEGER DIVISION  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY two RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
