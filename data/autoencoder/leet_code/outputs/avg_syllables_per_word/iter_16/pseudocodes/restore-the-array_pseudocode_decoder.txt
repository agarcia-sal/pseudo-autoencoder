CLASS Solution  
    FUNCTION numberOfArrays(s AS string, k AS integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zero repeated n PLUS one times  
        SET the element at position zero of dp TO one  
        FOR each integer i FROM one TO n  
            FOR each integer j FROM zero TO i MINUS one  
                IF the character at position j of s NOT EQUALS zero AND the integer value of the substring from position j PLUS one TO i of s IS GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET the element at position i of dp TO the sum of the element at position i of dp PLUS the element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
