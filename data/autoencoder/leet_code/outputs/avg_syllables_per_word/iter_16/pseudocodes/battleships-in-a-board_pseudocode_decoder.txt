CLASS Solution  
    FUNCTION countBattleships WITH PARAMETER board  
        SET count TO zero  
        FOR each index i FROM zero TO the LENGTH OF board MINUS one  
            FOR each index j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF the element at position i of board at position j EQUALS the letter X  
                    IF the index i EQUALS zero OR the element at position i MINUS one of board at position j EQUALS the period character AND the index j EQUALS zero OR the element at position i of board at position j MINUS one EQUALS the period character  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
