CLASS Solution  
    FUNCTION checkSubarraySum(nums AS list of integers, k AS integer) RETURNS boolean  
        CREATE a dictionary named remainder_map AND SET the element with key zero TO negative one  
        SET current_sum TO zero  
        FOR each index i AND corresponding element num IN nums  
            INCREMENT current_sum BY num  
            IF k DOES NOT EQUAL zero THEN  
                SET current_sum TO the remainder of current_sum DIVIDED BY k  
            END IF  
            IF remainder_map CONTAINS an element with key current_sum THEN  
                IF the difference between i AND element at key current_sum OF remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                SET the element with key current_sum OF remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
