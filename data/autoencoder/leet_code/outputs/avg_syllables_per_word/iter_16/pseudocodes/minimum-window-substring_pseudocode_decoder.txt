CLASS Solution  
    FUNCTION minWindow(s parameter OF TYPE string t parameter OF TYPE string) RETURNS string  
        SET t_count TO THE RESULT OF Counter APPLIED TO t  
        SET required TO THE LENGTH OF t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO AN EMPTY Counter  
        SET min_length TO positive INFINITY  
        SET min_start TO zero  
        
        WHILE right IS LESS THAN THE LENGTH OF s  
            INCREMENT BY one the FREQUENCY OF THE CHARACTER AT POSITION right OF s IN window_counts  
            
            IF THE CHARACTER AT POSITION right OF s IS IN t_count AND THE FREQUENCY OF THAT CHARACTER IN window_counts EQUALS THE FREQUENCY OF THAT CHARACTER IN t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO THE CHARACTER AT POSITION left OF s  
                
                IF THE NUMBER OF ELEMENTS FROM left TO right INCLUSIVE IS LESS THAN min_length  
                    SET min_length TO THE NUMBER OF ELEMENTS FROM left TO right INCLUSIVE  
                    SET min_start TO left  
                END IF  
                
                DECREMENT BY one THE FREQUENCY OF char IN window_counts  
                IF char IS IN t_count AND THE FREQUENCY OF char IN window_counts IS LESS THAN THE FREQUENCY OF char IN t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS positive INFINITY  
            RETURN AN EMPTY string  
        ELSE  
            RETURN THE SUBSTRING OF s FROM POSITION min_start TO POSITION min_start PLUS min_length MINUS ONE  
        END IF  
    END FUNCTION  
END CLASS
