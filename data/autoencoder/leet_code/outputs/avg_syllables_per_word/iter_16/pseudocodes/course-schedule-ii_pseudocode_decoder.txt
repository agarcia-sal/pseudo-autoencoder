CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adjacency list TO a map from each course in the range from zero to numCourses MINUS one TO an empty list  
        SET in degrees TO a list of zeros with the length equal to numCourses  
        
        FOR each pair of destination and source IN prerequisites  
            APPEND destination TO the list associated with source in adjacency list  
            INCREMENT the element at position destination of in degrees BY one  
        END FOR  
        
        SET queue TO a double ended queue initialized with all courses in the range from zero to numCourses MINUS one WHERE the element at position course of in degrees EQUALS zero  
        SET topological order TO an empty list  
        
        WHILE the queue is not empty  
            SET current TO the first element removed from the queue  
            APPEND current TO topological order  
            
            FOR each neighbor IN the list associated with current in adjacency list  
                DECREMENT the element at position neighbor of in degrees BY one  
                IF the element at position neighbor of in degrees EQUALS zero  
                    APPEND neighbor TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the length of topological order EQUALS numCourses  
            RETURN topological order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
