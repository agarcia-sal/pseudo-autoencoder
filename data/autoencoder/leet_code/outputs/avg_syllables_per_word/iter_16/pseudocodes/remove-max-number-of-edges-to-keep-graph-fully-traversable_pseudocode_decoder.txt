CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list of numbers from zero TO n inclusive  
        SET self.rank TO a list of zeros with length n PLUS one  
        SET self.count TO n  
    END FUNCTION

    FUNCTION find(self, u)  
        IF the element at position u of self.parent DOES NOT EQUAL u  
            SET the element at position u of self.parent TO the result of calling find on the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION

    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find on u  
        SET rootV TO the result of calling find on v  
        IF rootU DOES NOT EQUAL rootV  
            IF the element at position rootU of self.rank IS GREATER THAN the element at position rootV of self.rank  
                SET the element at position rootV of self.parent TO rootU  
            ELSE IF the element at position rootU of self.rank IS LESS THAN the element at position rootV of self.rank  
                SET the element at position rootU of self.parent TO rootV  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS

CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        SET uf_common TO a new instance of UnionFind initialized with n  
        SET uf_alice TO a new instance of UnionFind initialized with n  
        SET uf_bob TO a new instance of UnionFind initialized with n  

        SET used_edges TO zero  

        FOR each t u v IN edges  
            IF t EQUALS three  
                IF calling union on uf_common with parameters u and v RETURNS true  
                    CALL union on uf_alice with parameters u and v  
                    CALL union on uf_bob with parameters u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  

        FOR each t u v IN edges  
            IF t EQUALS one AND calling union on uf_alice with parameters u and v RETURNS true  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND calling union on uf_bob with parameters u and v RETURNS true  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  

        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
