CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeros with the LENGTH OF batchSize  
        FOR each group IN groups  
            INCREMENT the element at position group MODULO batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, SEQUENCE OF remainder_count)  
            CONVERT remainder_count TO a list  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF the element at position index of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT the element at position index of remainder_count BY one  
                SET new_remainder TO remainder MINUS index MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, SEQUENCE OF remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, SEQUENCE OF remainder_count)  
                END IF  
                
                SET max_happy TO the GREATER VALUE BETWEEN max_happy AND happy_groups  
                
                INCREMENT the element at position index of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, SEQUENCE OF remainder_count)  
END CLASS
