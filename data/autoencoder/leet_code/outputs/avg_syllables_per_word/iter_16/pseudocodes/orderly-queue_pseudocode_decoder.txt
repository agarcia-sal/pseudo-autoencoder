CLASS Solution  
    FUNCTION orderlyQueue WITH PARAMETERS s AND k  
        IF k GREATER THAN one  
            RETURN the JOINED STRING OF THE SORTED CHARACTERS OF s  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR each i FROM one TO n MINUS one  
                SET rotated TO the SUBSTRING FROM POSITION i TO THE END OF s CONCATENATED WITH the SUBSTRING FROM POSITION zero TO i OF s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
