CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a new mapping from default key to empty list  
        FOR each pair of elements u and v IN edges  
            APPEND v TO the list corresponding to key u in graph  
            APPEND u TO the list corresponding to key v in graph  
        END FOR  

        FUNCTION dfs(node, parent, path, visited)  
            IF node is AN ELEMENT IN visited  
                SET cycle_start TO the position of node in path  
                RETURN the subsequence of path from cycle_start to the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list corresponding to key node in graph  
                IF neighbor is NOT EQUAL TO parent  
                    SET cycle TO the result of calling dfs with neighbor, node, path, and visited  
                    IF cycle is truthy  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN None  
        END FUNCTION  

        SET visited TO an empty set  
        SET cycle TO None  
        FOR node FROM zero TO n MINUS one  
            IF node is NOT AN ELEMENT IN visited  
                SET cycle TO the result of calling dfs with node, negative one, an empty list, and visited  
                IF cycle is truthy  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  

        SET distance TO a list of zeros with length n  
        SET queue TO a new double ended queue initialized with elements of cycle  
        SET visited TO a new set initialized with elements of cycle  

        SET level TO zero  
        WHILE queue is NOT empty  
            SET current_length TO the number of elements in queue  
            FOR each index FROM zero TO current_length MINUS one  
                REMOVE and STORE the element from the front of queue IN node  
                SET the element at position node of distance TO level  
                FOR each neighbor IN the list corresponding to key node in graph  
                    IF neighbor is NOT AN ELEMENT IN visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  

        RETURN distance  
    END FUNCTION  
END CLASS
