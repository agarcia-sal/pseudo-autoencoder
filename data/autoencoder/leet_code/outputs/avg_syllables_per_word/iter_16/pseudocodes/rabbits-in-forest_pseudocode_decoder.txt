CLASS Solution  
    FUNCTION numRabbits WITH parameter answers AS list of integers  
        SET answer_counts TO the result of counting each element's occurrences in answers  
        SET total_rabbits TO zero  
        FOR each pair of answer and count IN answer_counts items  
            SET group_size TO answer PLUS one  
            SET groups TO the result of the integer division of the sum of count and answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
