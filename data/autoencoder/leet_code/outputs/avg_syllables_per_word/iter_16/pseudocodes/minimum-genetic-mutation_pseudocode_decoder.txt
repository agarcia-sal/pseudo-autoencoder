CLASS Solution  
    FUNCTION minMutation(start string, end string, bank list of strings) RETURNS integer  
        SET s TO the set created from the elements of bank  
        SET q TO a double ended queue initialized with the tuple consisting of start and zero  
        SET mp TO the mapping from the letter A to the string consisting of letters T C G the letter T to the string consisting of letters A C G the letter C to the string consisting of letters A T G the letter G to the string consisting of letters A T C  
        WHILE the queue q is not empty holds  
            REMOVE and ASSIGN the first element of q TO the tuple consisting of t and step  
            IF t EQUALS end THEN  
                RETURN step  
            END IF  
            FOR each index i and character v IN the elements of string t enumerated  
                FOR each character j IN the string mapped by key v in mp  
                    SET next TO the string formed by concatenating the substring from position zero TO position i minus one of t the character j the substring from position i plus one TO the end of t  
                    IF next IS an element of set s THEN  
                        APPEND the tuple consisting of next and step PLUS one TO the queue q  
                        REMOVE the element next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
