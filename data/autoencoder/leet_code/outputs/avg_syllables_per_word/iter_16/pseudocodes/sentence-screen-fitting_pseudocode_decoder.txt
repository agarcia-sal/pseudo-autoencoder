CLASS Solution  
    FUNCTION wordsTyping(sentence AS list of strings, rows AS integer, cols AS integer) RETURNS integer  
        SET s TO a single string formed by concatenating each word in sentence separated by a space FOLLOWED BY a space character  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the element at position MODULO total_length of s EQUALS a space character THEN  
                INCREMENT position BY one  
            ELSE  
                WHILE position IS GREATER THAN zero AND the element at position MINUS one MODULO total_length of s DOES NOT EQUAL a space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN the INTEGER DIVISION of position BY total_length  
    END FUNCTION  
END CLASS
