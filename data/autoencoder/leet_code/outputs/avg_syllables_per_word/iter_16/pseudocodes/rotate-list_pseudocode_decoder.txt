CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a NEW INSTANCE OF ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position one TO the end  
        SET node TO a NEW INSTANCE OF ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the RESULT OF is_same_list APPLIED TO the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight WITH PARAMETERS head AND k  
        IF head IS None OR the next ATTRIBUTE OF head IS None OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next ATTRIBUTE OF tail IS NOT None  
            SET tail TO the next ATTRIBUTE OF tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next ATTRIBUTE OF tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each iteration FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the next ATTRIBUTE OF new_tail  
        END FOR  
        SET new_head TO the next ATTRIBUTE OF new_tail  
        SET the next ATTRIBUTE OF new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
