CLASS Solution  
    FUNCTION findItinerary WITH parameter tickets AS list of list of strings  
        SET graph TO a new defaultdict mapping keys to empty lists  
        FOR each pair consisting of departure AND arrival IN the list tickets sorted in ascending lexical order then reversed  
            APPEND arrival TO the list at key departure IN graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit WITH parameter airport  
            WHILE the list at key airport IN graph is not empty  
                CALL visit WITH argument the last element removed from the list at key airport IN graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit WITH argument the string airport called J F K  
        RETURN the list itinerary reversed in order  
    END FUNCTION  
END CLASS
