CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT TO zero AND left DEFAULT TO none AND right DEFAULT TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero OF values  
    SET i TO one  
    SET queue TO A new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVAL OF element from the left end OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node dot left TO A NEW TreeNode INITIALIZED WITH element at position i OF values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node dot right TO A NEW TreeNode INITIALIZED WITH element at position i OF values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest WITH PARAMETERS root AND k  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        
        WHILE stack IS NOT empty OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO current dot left  
            END WHILE  
            SET current TO REMOVAL OF last element FROM stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current dot val  
            END IF  
            SET current TO current dot right  
        END WHILE  
    END FUNCTION  
END CLASS
