CLASS Solution  
    FUNCTION findLUSlength WITH parameter strs which is a list of strings  
        FUNCTION is_subsequence WITH parameters s and t  
            SET iterator TO an iterator over t  
            RETURN whether for every character of s the character appears in the remaining elements of iterator in order  
        END FUNCTION  
        
        SORT the list strs by descending order of the length of each element and then by lexicographical order of each element  
        
        FOR each index i and element word IN strs  
            SET unique TO true  
            FOR each index j and element other IN strs  
                IF i is not equal to j AND is_subsequence with s being word and t being other returns true  
                    SET unique TO false  
                    EXIT the inner loop  
                END IF  
            END FOR  
            
            IF unique is true  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
