CLASS Solution  
    FUNCTION findMaximizedCapital(k AS integer, w AS integer, profits AS list OF integers, capital AS list OF integers) RETURNS integer  
        SET projects TO A LIST OF TUPLES EACH CONTAINING element at the same position of capital AND element at the same position of profits  
        SORT projects BY the FIRST element OF each tuple IN ascending order  
        SET max_heap TO AN EMPTY LIST  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR each number FROM one TO k  
            WHILE i IS LESS THAN n AND the FIRST element OF element at position i OF projects IS LESS THAN OR EQUAL TO w  
                ADD THE NEGATION OF the SECOND element OF element at position i OF projects TO max_heap USING HEAP PUSH OPERATION  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS EMPTY  
                BREAK THE LOOP  
            END IF  
            SET w TO w PLUS THE NEGATION OF THE RESULT OF HEAP POP OPERATION ON max_heap  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
