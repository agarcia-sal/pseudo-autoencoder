CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS List of string  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO the SUBSTRING FROM POSITION index TO i MINUS one OF num  
                IF the LENGTH OF operand IS GREATER THAN one AND the ELEMENT AT POSITION zero OF operand EQUALS zero  
                    CONTINUE TO the NEXT ITERATION OF THE LOOP  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH PARAMETERS i, operand, CONVERT operand TO integer, CONVERT operand TO integer  
                ELSE  
                    CALL dfs WITH PARAMETERS i, path CONCATENATED WITH the CHARACTER PLUS AND CONCATENATED WITH operand, value PLUS CONVERT operand TO integer, CONVERT operand TO integer  
                    CALL dfs WITH PARAMETERS i, path CONCATENATED WITH the CHARACTER MINUS AND CONCATENATED WITH operand, value MINUS CONVERT operand TO integer, NEGATIVE OF CONVERT operand TO integer  
                    CALL dfs WITH PARAMETERS i, path CONCATENATED WITH the CHARACTER MULTIPLICATION AND CONCATENATED WITH operand, value MINUS prev PLUS prev MULTIPLIED BY CONVERT operand TO integer, prev MULTIPLIED BY CONVERT operand TO integer  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH PARAMETERS zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
