CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of length n where each element is one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR each k FROM zero TO m MINUS one element at position k at position j of strs IS LESS THAN OR EQUAL TO element at position k at position i of strs HOLDS  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the GREATER OF all elements IN dp  
    END FUNCTION  
END CLASS
