CLASS Solution  
    FUNCTION topKFrequent(words LIST OF STRING, k INTEGER) RETURNS LIST OF STRING  
        SET count TO RESULT OF counting frequencies of each word IN words  
        SET heap TO A NEW LIST formed by FOR each word and frequency IN count items CONSTRUCT A PAIR consisting of negative frequency AND word  
        TRANSFORM heap INTO A HEAP STRUCTURE preserving heap property  
        SET result TO A NEW LIST formed by FOR each number FROM one TO k EXTRACT the second element of the minimum element FROM heap  
        RETURN result  
    END FUNCTION  
END CLASS
