CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1, p2)  
            IF the element at position zero of p1 EQUALS the element at position zero of p2  
                RETURN positive infinity value  
            END IF  
            RETURN the element at position one of p1 MINUS the element at position one of p2 DIVIDED BY the element at position zero of p1 MINUS the element at position zero of p2  
        END FUNCTION  
      
        SET max_points TO zero  
      
        FOR index i FROM zero TO the LENGTH OF points MINUS one  
            SET slopes TO a new mapping with default integer zero  
            FOR index j FROM zero TO the LENGTH OF points MINUS one  
                IF index i NOT EQUALS index j  
                    SET s TO the result of calling slope with the element at position i of points AND the element at position j of points  
                    INCREMENT the value at key s in slopes BY one  
                    IF the value at key s in slopes GREATER THAN max_points  
                        SET max_points TO the value at key s in slopes  
                    END IF  
                END IF  
            END FOR  
        END FOR  
      
        IF the LENGTH OF points GREATER THAN zero  
            RETURN max_points PLUS one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
