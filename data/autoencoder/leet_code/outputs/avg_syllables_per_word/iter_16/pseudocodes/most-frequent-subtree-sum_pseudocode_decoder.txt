CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments node left of p AND node left of q AND the result of is_same_tree with arguments node right of p AND node right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO the value of node PLUS the result of subtree_sum with node left of node PLUS the result of subtree_sum with node right of node  
            INCREMENT the frequency count associated with current_sum in count by one  
            RETURN current_sum  
        END FUNCTION  
        SET count TO a new default dictionary that returns zero for missing keys  
        CALL subtree_sum with argument root  
        SET max_freq TO the maximum value among the frequency values in count or zero if count is empty  
        SET result TO a new list containing all keys from count for which their associated frequency EQUALS max_freq  
        RETURN result  
    END FUNCTION  
END CLASS
