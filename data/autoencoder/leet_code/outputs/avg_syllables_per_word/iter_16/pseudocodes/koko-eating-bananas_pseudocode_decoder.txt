CLASS Solution  
    FUNCTION minEatingSpeed WITH parameters piles AND h  
        FUNCTION canFinish WITH parameter k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the division of pile BY k  
            END FOR  
            RETURN hours LESS THAN OR EQUAL TO h  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the maximum element IN piles  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of the sum of left AND right BY two  
            IF canFinish WITH mid RETURNS true  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
