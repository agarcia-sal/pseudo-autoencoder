CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of length n filled with negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND the element at position last element of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_index TO the last element removed from stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
