CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero of mat2  
        SET result TO a new two dimensional collection of m rows each containing n elements all set TO zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    SET element at position i of result TO element at position i of result  
                    SET element at position j TO element at position j of result at position i PLUS element at position i of mat1 MULTIPLIED BY element at position p MULTIPLIED BY element at position j of mat2 at position p  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
