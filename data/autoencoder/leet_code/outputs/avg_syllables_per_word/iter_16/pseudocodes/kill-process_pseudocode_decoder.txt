CLASS Solution  
    FUNCTION killProcess(pid AS List OF integers, ppid AS List OF integers, kill AS integer) RETURNS List OF integers  
        SET tree TO call defaultdict function WITH argument list  
        FOR each pair consisting of child process pid_element and parent process ppid_element IN parallel from pid and ppid  
            APPEND child process pid_element TO element at position parent process ppid_element of tree  
        END FOR  
        SET queue TO list containing the single element kill  
        SET killed_processes TO empty list  
        WHILE queue is not empty  
            REMOVE element at position zero from queue AND ASSIGN TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY all elements of element at position current of tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
