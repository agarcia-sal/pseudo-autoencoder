CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        SET dp TO a two dimensional list OF size m PLUS one BY n PLUS one FILLED WITH zeroes  
        
        FOR j FROM one TO n  
            SET element at position j of the first row of dp TO element at position j MINUS one of the first row of dp PLUS ASCII VALUE OF character at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position i of the first column of dp TO element at position i MINUS one of the first column of dp PLUS ASCII VALUE OF character at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s1 EQUALS character at position j MINUS one of s2  
                    SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp  
                ELSE  
                    SET element at position i j of dp TO the MINIMUM OF element at position i MINUS one j of dp PLUS ASCII VALUE OF character at position i MINUS one of s1 AND element at position i j MINUS one of dp PLUS ASCII VALUE OF character at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m n of dp  
    END FUNCTION  
END CLASS
