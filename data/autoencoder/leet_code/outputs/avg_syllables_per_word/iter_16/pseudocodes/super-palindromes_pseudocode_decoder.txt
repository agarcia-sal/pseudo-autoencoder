CLASS Solution  
    FUNCTION superpalindromesInRange(left AS string, right AS string) RETURNS integer  
        FUNCTION is_palindrome(num) RETURNS boolean  
            RETURN the string representation of num EQUALS the string representation of num REVERSED  
        END FUNCTION  
  
        FUNCTION generate_palindromes(limit) RETURNS list of integers  
            SET palindromes TO an empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO the string representation of i  
                APPEND the integer conversion of s CONCATENATED WITH the substring of s from second last position TO first position reversed TO palindromes  
                APPEND the integer conversion of s CONCATENATED WITH the string s reversed TO palindromes  
            END FOR  
            RETURN filtered list of palindromes WHERE each element IS LESS THAN OR EQUAL TO limit  
        END FUNCTION  
  
        SET left TO the integer conversion of left  
        SET right TO the integer conversion of right  
        SET limit TO the integer conversion of the square root of right PLUS one  
        SET palindromes TO generate_palindromes(limit)  
  
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND is_palindrome(square)  
                INCREMENT count BY one  
            END IF  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
