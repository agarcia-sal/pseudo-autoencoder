CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary with keys each character from a to z and values zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index IS GREATER THAN zero AND the UNICODE code of the character at position index of s MINUS the UNICODE code of the character at position index MINUS one of s EQUALS one OR the character at position index of s EQUALS character a AND the character at position index MINUS one of s EQUALS character z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key character at position index of s TO the GREATER value BETWEEN max_length at key character at position index of s AND current_length  
        END FOR  
        SET total_unique_substrings TO the SUM of all values in max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
