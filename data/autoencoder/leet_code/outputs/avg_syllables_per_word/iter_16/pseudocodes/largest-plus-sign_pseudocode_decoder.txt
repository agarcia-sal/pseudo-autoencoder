CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        CREATE a set called mines_set containing each element of mines converted into a tuple to allow constant time lookup  
        
        INITIALIZE a three dimensional list called dp with dimensions n by n where each element is a list of four zeros representing counts for right down left and up directions  
        
        SET directions TO a list containing the tuples representing right zero one down one zero left zero negative one and up negative one zero  
        
        FOR each i FROM zero TO n minus one  
            FOR each j FROM zero TO n minus one  
                IF the tuple containing i and j IS NOT an element of mines_set  
                    IF j IS GREATER THAN zero  
                        SET the element at position zero of the list at position j of the list at position i of dp TO the element at position zero of the list at position j minus one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero  
                        SET the element at position one of the list at position j of the list at position i of dp TO the element at position one of the list at position j of the list at position i minus one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR each i FROM n minus one DOWN TO zero  
            FOR each j FROM n minus one DOWN TO zero  
                IF the tuple containing i and j IS NOT an element of mines_set  
                    IF j IS LESS THAN n MINUS one  
                        SET the element at position two of the list at position j of the list at position i of dp TO the element at position two of the list at position j plus one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one  
                        SET the element at position three of the list at position j of the list at position i of dp TO the element at position three of the list at position j of the list at position i plus one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value of the list at position j of the list at position i of dp  
                    SET result TO the maximum value BETWEEN result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
