CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_nums TO the RESULT OF sorting nums in non-decreasing order  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index of nums IS NOT EQUAL TO element at position index of sorted_nums  
                IF start EQUALS negative one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        SET result_length TO end MINUS start PLUS one  
        RETURN result_length  
    END FUNCTION  
END CLASS
