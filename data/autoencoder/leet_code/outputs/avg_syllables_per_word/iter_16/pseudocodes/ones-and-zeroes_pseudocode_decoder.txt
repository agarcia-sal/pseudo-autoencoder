CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        INITIALIZE dp TO a two dimensional list with m plus one rows and n plus one columns each filled with zero  
        
        FOR each string s IN strs  
            SET count_0 TO the number of zero characters in string s  
            SET count_1 TO the number of one characters in string s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i and j TO the maximum of dp element at position i and j AND dp element at position i minus count_0 and j minus count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position m and n of dp  
    END FUNCTION  
END CLASS
