CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO infinity  
        SET Y1 TO infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        INITIALIZE points AS empty set  
        SET area TO zero  
        FOR each rectangle IN rectangles  
            SET x1 TO element at position zero of rectangle  
            SET y1 TO element at position one of rectangle  
            SET x2 TO element at position two of rectangle  
            SET y2 TO element at position three of rectangle  
            IF X1 GREATER THAN x1 THEN  
                SET X1 TO x1  
            END IF  
            IF Y1 GREATER THAN y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF X2 LESS THAN x2 THEN  
                SET X2 TO x2  
            END IF  
            IF Y2 LESS THAN y2 THEN  
                SET Y2 TO y2  
            END IF  
            SET area TO area PLUS (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)  
            SET corner_points TO list containing tuples of (x1 y1) (x1 y2) (x2 y1) and (x2 y2)  
            FOR each point IN corner_points  
                IF point IS IN points THEN  
                    REMOVE point FROM points  
                ELSE  
                    ADD point TO points  
                END IF  
            END FOR  
        END FOR  
        IF area NOT EQUALS (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1) THEN  
            RETURN false  
        END IF  
        SET expected_corners TO set containing tuples of (X1 Y1) (X1 Y2) (X2 Y1) and (X2 Y2)  
        IF points NOT EQUALS expected_corners THEN  
            RETURN false  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS
