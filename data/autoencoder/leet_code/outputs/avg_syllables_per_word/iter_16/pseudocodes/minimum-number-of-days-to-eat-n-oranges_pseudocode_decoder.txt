CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining IS LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO the REMAINDER OF remaining DIVIDED BY two PLUS CALL dp WITH the INTEGER PART OF remaining DIVIDED BY two  
            SET option_two TO the REMAINDER OF remaining DIVIDED BY three PLUS CALL dp WITH the INTEGER PART OF remaining DIVIDED BY three  
            RETURN one PLUS the MINIMUM VALUE BETWEEN option_one AND option_two  
        END FUNCTION  
        RETURN CALL dp WITH n  
    END FUNCTION  
END CLASS
