CLASS Solution  
    FUNCTION subdomainVisits WITH PARAMETER cpdomains AS List of strings  
        SET domain_count TO a new default dictionary with integer default values  
        
        FOR each cpdomain IN cpdomains  
            SET count TO the first part of splitting cpdomain by whitespace converted to integer  
            SET domain TO the second part of splitting cpdomain by whitespace  
            SET subdomains TO the list of parts from splitting domain by the dot character  
            
            FOR each index FROM zero TO the length of subdomains MINUS one  
                SET subdomain TO the concatenation of elements from position index TO the last position of subdomains joined by the dot character  
                INCREMENT the element at key subdomain of domain_count BY count  
            END FOR  
        END FOR  
        
        RETURN a new list formed by concatenating in string format the count followed by a space followed by the domain FOR each domain and count pair in the items of domain_count  
    END FUNCTION  
END CLASS
