CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO the list of words for numbers from zero to nineteen in order  
        SET tens TO the list of words for the multiples of ten from zero to ninety in order  
        SET thousands TO the list of words for the powers of thousand including empty string Thousand Million and Billion  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN the empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n by ten of tens concatenated with a single space concatenated with the result of helper applied to the remainder of n divided by ten  
            ELSE  
                RETURN the element at position the integer division of n by one hundred of below_twenty concatenated with the word Hundred and a single space concatenated with the result of helper applied to the remainder of n divided by one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO the empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF the remainder of num divided by one thousand NOT EQUALS zero  
                SET result TO the result of helper applied to the remainder of num divided by one thousand concatenated with the element at position i of thousands concatenated with a single space concatenated with result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing whitespace removed  
    END FUNCTION  
END CLASS
