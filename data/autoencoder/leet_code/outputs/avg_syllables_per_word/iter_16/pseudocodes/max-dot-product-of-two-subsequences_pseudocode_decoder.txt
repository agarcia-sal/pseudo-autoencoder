CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        SET dp TO a LIST of LISTS where each inner LIST has n PLUS one elements all SET TO negative infinity and there are m PLUS one such inner LISTS initialized  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO the element at position i MINUS one of nums1 MULTIPLIED BY the element at position j MINUS one of nums2  
                
                SET dp element at position i element at position j TO the MAXIMUM OF  
                    current_product  
                    the element at position i MINUS one element at position j of dp  
                    the element at position i element at position j MINUS one of dp  
                    the element at position i MINUS one element at position j MINUS one of dp PLUS current_product  
            END FOR  
        END FOR  
        
        RETURN the element at position m element at position n of dp  
    END FUNCTION  
END CLASS
