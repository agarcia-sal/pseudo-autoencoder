CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO a dictionary mapping each distinct character in s TO an empty double-ended queue  
        FOR each word IN words  
            ASSIGN iterator TO an iterator over word  
            SET first_letter TO the next character from iterator  
            IF first_letter EXISTS IN waiting  
                APPEND iterator TO the double-ended queue corresponding to first_letter in waiting  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN s  
            IF c EXISTS IN waiting  
                ASSIGN current_iterators TO the double-ended queue corresponding to c in waiting  
                SET the double-ended queue for c in waiting TO a new empty double-ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next character from it OR none if exhausted  
                    IF next_char IS none  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char EXISTS IN waiting  
                            APPEND it TO the double-ended queue corresponding to next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
