CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO NEW ListNode WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN whether the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the result of calling is_same_list WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS none AND right EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left ATTRIBUTE OF node TO NEW TreeNode WITH the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right ATTRIBUTE OF node TO NEW TreeNode WITH the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUAL TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the result of calling is_same_tree WITH the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS self AND head AS an optional ListNode  
        IF head IS none  
            RETURN none  
        END IF  

        FUNCTION findMiddle WITH PARAMETERS left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast IS NOT EQUAL TO right AND the next ATTRIBUTE OF fast IS NOT EQUAL TO right  
                SET slow TO the next ATTRIBUTE OF slow  
                SET fast TO the next ATTRIBUTE OF the next ATTRIBUTE OF fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST WITH PARAMETERS left AND right  
            IF left IS EQUAL TO right  
                RETURN none  
            END IF  
            SET mid TO the result of calling findMiddle WITH left AND right  
            SET node TO NEW TreeNode WITH the val ATTRIBUTE OF mid  
            SET the left ATTRIBUTE OF node TO the result of calling convertListToBST WITH left AND mid  
            SET the right ATTRIBUTE OF node TO the result of calling convertListToBST WITH the next ATTRIBUTE OF mid AND right  
            RETURN node  
        END FUNCTION  

        RETURN the result of calling convertListToBST WITH head AND none  
    END FUNCTION  
END CLASS
