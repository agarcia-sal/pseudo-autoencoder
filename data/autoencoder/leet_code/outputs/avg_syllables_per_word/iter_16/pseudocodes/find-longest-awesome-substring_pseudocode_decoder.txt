CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO an empty mapping from zero TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN s  
            SET digit TO the integer value of char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR with one LEFT SHIFTED BY digit  
            
            IF prefix_xor IS A KEY IN prefix_map  
                SET max_length TO the greater OF max_length AND the result of i MINUS the element at key prefix_xor of prefix_map  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR with one LEFT SHIFTED BY j  
                IF flipped IS A KEY IN prefix_map  
                    SET max_length TO the greater OF max_length AND the result of i MINUS the element at key flipped of prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor IS NOT A KEY IN prefix_map  
                SET the element at key prefix_xor of prefix_map TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
