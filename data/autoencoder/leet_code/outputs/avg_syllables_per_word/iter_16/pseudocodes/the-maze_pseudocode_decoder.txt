CLASS Solution  
    FUNCTION hasPath maze start destination  
        SET directions TO a list of pairs representing zero AND one, zero AND one zero AND negative one, negative one AND zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs x y  
            IF the pair of x AND y IS IN visited  
                RETURN false  
            END IF  
            IF a list of x AND y EQUALS destination  
                RETURN true  
            END IF  
            ADD the pair of x AND y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE the sum of nx AND dx IS GREATER THAN OR EQUAL TO zero AND the sum of nx AND dx IS LESS THAN m AND the sum of ny AND dy IS GREATER THAN OR EQUAL TO zero AND the sum of ny AND dy IS LESS THAN n AND the element at position the sum of nx AND dx of maze AND at position the sum of ny AND dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs called with nx AND ny RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs called with element at position zero of start AND element at position one of start  
    END FUNCTION  
END CLASS
