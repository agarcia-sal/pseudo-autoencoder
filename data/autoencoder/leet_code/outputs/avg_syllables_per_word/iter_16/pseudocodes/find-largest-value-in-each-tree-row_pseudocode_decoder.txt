CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulted to zero left defaulted to None right defaulted to None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values list  
    IF the values list is empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values list  
    SET i TO one  
    INITIALIZE queue AS a deque data structure  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i is less than the LENGTH OF values list AND the element at position i of values is not None  
            SET node dot left TO a new TreeNode initialized with the element at position i of values list  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the LENGTH OF values list AND the element at position i of values is not None  
            SET node dot right TO a new TreeNode initialized with the element at position i of values list  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF p dot val is NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with parameters p dot left and q dot left AND the result of calling is_same_tree with parameters p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues with parameters self and root Optional TreeNode  
        IF root is None  
            RETURN an empty list  
        END IF  
        INITIALIZE queue AS a deque data structure containing root  
        INITIALIZE largest_values as an empty list  
        WHILE queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each count FROM one TO level_size  
                SET node TO the first element removed from queue  
                IF max_value is less than node dot val  
                    SET max_value TO node dot val  
                END IF  
                IF node dot left is not None  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right is not None  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
