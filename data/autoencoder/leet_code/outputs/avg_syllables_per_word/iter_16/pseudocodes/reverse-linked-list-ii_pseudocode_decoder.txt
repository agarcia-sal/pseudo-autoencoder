CLASS ListNode  
    FUNCTION __init__ with parameters val DEFAULTING TO zero AND next DEFAULTING TO none  
        SET the attribute val TO the parameter val  
        SET the attribute next TO the parameter next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one TO the end  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN false  
    END IF  
    RETURN the logical AND of the equality of the val attribute of p1 AND the val attribute of p2 AND the recursive call to is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween with parameters head AND left AND right  
        IF head EQUALS nothing OR left EQUALS right  
            RETURN head  
        END IF  
        
        SET dummy TO a new ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET prev TO dummy  
        
        FOR each iteration FROM zero TO left MINUS one MINUS one  
            SET prev TO the next attribute of prev  
        END FOR  
        
        SET current TO the next attribute of prev  
        FOR each iteration FROM zero TO right MINUS left MINUS one  
            SET next_node TO the next attribute of current  
            SET the next attribute of current TO the next attribute of next_node  
            SET the next attribute of next_node TO the next attribute of prev  
            SET the next attribute of prev TO next_node  
        END FOR  
        
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
