CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of integer minus one repeated twenty six times  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position and character IN s  
            SET index TO the character code of character MINUS the character code of letter a  
            INCREMENT current_appeal BY position MINUS element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of last TO position  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
