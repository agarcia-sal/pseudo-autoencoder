CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the STRING AT position index of strs REVERSED  
            IF reversed_str GREATER THAN the STRING AT position index of strs  
                SET the STRING AT position index of strs TO reversed_str  
            END IF  
        END FOR  
  
        SET doubled_strs TO strs CONCATENATED WITH strs  
  
        SET max_string TO EMPTY STRING  
  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN collection of Boolean FALSE and TRUE  
                IF is_reversed IS TRUE  
                    SET current TO the STRING AT position index of strs REVERSED  
                ELSE  
                    SET current TO the STRING AT position index of strs  
                END IF  
  
                SET remaining TO the CONCATENATION OF the STRINGS from position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one in doubled_strs  
  
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the SUBSTRING of current FROM cut_position TO the END PLUS remaining PLUS the SUBSTRING of current FROM the START TO cut_position MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
  
        RETURN max_string  
    END FUNCTION  
END CLASS
