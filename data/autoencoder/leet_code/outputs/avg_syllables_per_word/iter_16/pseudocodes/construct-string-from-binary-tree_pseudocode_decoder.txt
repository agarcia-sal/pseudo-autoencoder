CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET the property val OF self TO val  
        SET the property left OF self TO left  
        SET the property right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the collection values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode INITIALIZED WITH the element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double-ended queue data structure  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none  
            SET the property left OF node TO a new instance of TreeNode INITIALIZED WITH the element at position i OF values  
            APPEND the property left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none  
            SET the property right OF node TO a new instance of TreeNode INITIALIZED WITH the element at position i OF values  
            APPEND the property right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the property val OF p IS NOT EQUAL TO the property val OF q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the property left OF p AND the property left OF q AND the result of is_same_tree WITH PARAMETERS the property right OF p AND the property right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str WITH PARAMETERS self AND root  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN an empty string  
            END IF  
            SET result TO the string representation of the property val OF node  
            IF the property left OF node EXISTS OR the property left OF node DOES NOT EXIST AND the property right OF node EXISTS  
                SET result TO result CONCATENATED WITH an opening parenthesis CONCATENATED WITH the result of helper WITH PARAMETER the property left OF node CONCATENATED WITH a closing parenthesis  
            END IF  
            IF the property right OF node EXISTS  
                SET result TO result CONCATENATED WITH an opening parenthesis CONCATENATED WITH the result of helper WITH PARAMETER the property right OF node CONCATENATED WITH a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper WITH PARAMETER root  
    END FUNCTION  
END CLASS
