CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero IN DECREASING ORDER  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_character_position TO the Unicode code point of element at position index of s MINUS the Unicode code point of lowercase letter a  
            SET shifted_character_position TO the remainder of original_character_position PLUS element at position index of shifts DIVIDED BY twenty six  
            SET new_character_code TO shifted_character_position PLUS the Unicode code point of lowercase letter a  
            SET new_character TO the character represented by the Unicode value new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
