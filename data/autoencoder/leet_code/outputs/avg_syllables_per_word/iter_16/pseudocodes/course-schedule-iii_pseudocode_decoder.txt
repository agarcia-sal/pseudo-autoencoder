CLASS Solution  
    FUNCTION scheduleCourse(courses AS List OF List OF Integer) RETURNS Integer  
        CALL the METHOD sort ON courses USING a KEY FUNCTION THAT RETURNS the element at position one of each course  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each pair of duration and last_day IN courses  
            IF current_end_day PLUS duration IS LESS THAN OR EQUAL TO last_day THEN  
                CALL the METHOD to PUSH negative duration ONTO max_heap  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND the NEGATIVE of the element at position zero of max_heap IS GREATER THAN duration THEN  
                INCREMENT current_end_day BY duration PLUS the result of the METHOD to POP the largest element FROM max_heap  
                CALL the METHOD to PUSH negative duration ONTO max_heap  
            END IF  
        END FOR  
        RETURN the LENGTH OF max_heap  
    END FUNCTION  
END CLASS
