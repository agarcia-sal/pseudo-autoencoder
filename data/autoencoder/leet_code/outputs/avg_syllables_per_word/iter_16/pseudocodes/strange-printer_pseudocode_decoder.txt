CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i IS GREATER THAN j  
                RETURN zero  
            END IF  
            IF tuple composed of i and j IS IN memo  
                RETURN element at tuple composed of i and j IN memo  
            END IF  
            SET result TO CALL dp WITH i AND j MINUS one AND memo PLUS one  
            FOR each index k FROM i TO j MINUS one  
                IF character at position k OF s EQUALS character at position j OF s  
                    SET result TO the MINIMUM OF result AND THE SUM OF dp CALLED WITH i AND k MINUS one AND memo AND dp CALLED WITH k AND j MINUS one AND memo  
                END IF  
            END FOR  
            SET element at tuple composed of i and j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN CALL dp WITH zero AND THE LENGTH OF s MINUS one AND AN EMPTY COLLECTION  
    END FUNCTION  
END CLASS
