CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO an iterator over the characters of s  
            RETURN the result of the test that for every character in word the character appears in iterator in order  
        END FUNCTION  
        
        SORT dictionary BY the LENGTH OF each word in descending order AND BY the alphabetical order of words in ascending order  
        
        FOR each word IN dictionary  
            IF the result of is_subsequence with parameters word and s is true THEN  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN an empty string  
    END FUNCTION  
END CLASS
