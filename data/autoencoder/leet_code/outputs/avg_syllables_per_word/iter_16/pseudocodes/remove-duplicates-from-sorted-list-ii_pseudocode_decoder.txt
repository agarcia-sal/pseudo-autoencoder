CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF SELF TO val  
        SET the next ATTRIBUTE OF SELF TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the last position  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND CALL is_same_list WITH PARAMETERS the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS self AND head AS Optional OF ListNode  
        SET dummy TO A NEW ListNode INITIALIZED WITH zero AND head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF the next ATTRIBUTE OF head IS NOT none AND the val ATTRIBUTE OF head EQUALS the val ATTRIBUTE OF the next ATTRIBUTE OF head  
                WHILE the next ATTRIBUTE OF head IS NOT none AND the val ATTRIBUTE OF head EQUALS the val ATTRIBUTE OF the next ATTRIBUTE OF head  
                    SET head TO the next ATTRIBUTE OF head  
                END WHILE  
                SET the next ATTRIBUTE OF prev TO the next ATTRIBUTE OF head  
            ELSE  
                SET prev TO the next ATTRIBUTE OF prev  
            END IF  
            SET head TO the next ATTRIBUTE OF head  
        END WHILE  
        RETURN the next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
