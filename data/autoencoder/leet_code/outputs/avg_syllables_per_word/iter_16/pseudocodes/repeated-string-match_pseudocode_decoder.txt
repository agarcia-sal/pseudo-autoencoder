CLASS Solution  
    FUNCTION repeatedStringMatch(a STRING, b STRING) RETURNS INTEGER  
        SET min_repeats TO the smallest integer greater than or equal to the LENGTH OF string b DIVIDED BY the LENGTH OF string a  
        IF string b IS A SUBSTRING OF string a REPEATED min_repeats TIMES  
            RETURN min_repeats  
        END IF  
        IF string b IS A SUBSTRING OF string a REPEATED min_repeats PLUS one TIMES  
            RETURN min_repeats PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
