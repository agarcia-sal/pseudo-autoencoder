CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the NUMBER OF occurrences of letter X in row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the NUMBER OF occurrences of letter O in row  
        END FOR  

        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN false  
        END IF  

        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row EQUALS player CONCATENATED WITH player CONCATENATED WITH player  
                    RETURN true  
                END IF  
            END FOR  
            FOR each column_index FROM zero TO two  
                IF element at position column_index of element at position zero of board EQUALS player AND element at position column_index of element at position one of board EQUALS player AND element at position column_index of element at position two of board EQUALS player  
                    RETURN true  
                END IF  
            END FOR  
            IF element at position zero of element at position zero of board EQUALS player AND element at position one of element at position one of board EQUALS player AND element at position two of element at position two of board EQUALS player  
                RETURN true  
            END IF  
            IF element at position two of element at position zero of board EQUALS player AND element at position one of element at position one of board EQUALS player AND element at position zero of element at position two of board EQUALS player  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  

        SET winX TO result of check_win with player as letter X  
        SET winO TO result of check_win with player as letter O  

        IF winX AND winO  
            RETURN false  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN false  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN false  
        END IF  

        RETURN true  
    END FUNCTION  
END CLASS
