CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO a list combining elements from difficulty AND profit INTO pairs associating difficulty WITH profit  
        SORT jobs by the difficulty value in each pair IN ascending order  
        SORT worker by ability IN ascending order  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of jobs  
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND the difficulty value of the element at position job_index of jobs IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO the greater value BETWEEN best_profit AND the profit value of the element at position job_index of jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
