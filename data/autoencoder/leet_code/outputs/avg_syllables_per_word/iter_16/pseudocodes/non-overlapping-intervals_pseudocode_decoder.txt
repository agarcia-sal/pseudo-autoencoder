CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the end value of each interval  
        SET end TO the end value of element at position zero of intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the start value of element at position index of intervals GREATER THAN OR EQUAL TO end  
                SET end TO the end value of element at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
