CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO collection of characters two five six nine  
        SET same_digits TO collection of characters zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string converted from num  
            IF all digits IN num_str belong to the union of change_digits and same_digits  
                IF any digit IN num_str belongs to change_digits  
                    RETURN true  
                END IF  
                RETURN false  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO the sum of is_good_number(i) for each integer i FROM one TO n inclusive  
        RETURN good_count  
    END FUNCTION  
END CLASS
