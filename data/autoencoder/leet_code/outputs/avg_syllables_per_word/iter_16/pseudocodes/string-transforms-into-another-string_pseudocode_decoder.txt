CLASS Solution  
    FUNCTION canConvert str1 str2  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        
        SET conversion_map TO empty mapping  
        
        FOR each pair char1 char2 IN positions of str1 and str2 simultaneously  
            IF char1 EXISTS AS key IN conversion_map  
                IF value at key char1 in conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET value at key char1 in conversion_map TO char2  
            END IF  
        END FOR  
        
        IF the SIZE OF the SET OF characters in str2 EQUALS twenty six  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
