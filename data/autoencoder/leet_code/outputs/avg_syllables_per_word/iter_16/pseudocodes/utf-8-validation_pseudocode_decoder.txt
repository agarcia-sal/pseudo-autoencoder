CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the result of shifting byte to the right by seven positions EQUALS zero  
                RETURN one  
            ELSE IF the result of shifting byte to the right by five positions EQUALS the binary value one one zero  
                RETURN two  
            ELSE IF the result of shifting byte to the right by four positions EQUALS the binary value one one one zero  
                RETURN three  
            ELSE IF the result of shifting byte to the right by three positions EQUALS the binary value one one one one zero  
                RETURN four  
            ELSE  
                RETURN minus one  
            END IF  
        END FUNCTION  

        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH of data  
            SET byte_count TO the result of the FUNCTION get_byte_count applied to the element at position i of data  
            IF byte_count EQUALS minus one  
                RETURN False  
            END IF  
            IF the sum of i PLUS byte_count IS GREATER THAN the LENGTH of data  
                RETURN False  
            END IF  
            FOR j FROM i PLUS one TO i PLUS byte_count MINUS one  
                IF the result of shifting the element at position j of data to the right by six positions IS NOT EQUAL to the binary value one zero  
                    RETURN False  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
