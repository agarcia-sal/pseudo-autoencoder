CLASS Solution  
    FUNCTION minKBitFlips(nums, k)  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a list of zeros with LENGTH n PLUS one  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET flip_state TO the VALUE OF flip_state XORED WITH the element at position i of flipped  
            
            IF the element at position i of nums EQUALS flip_state THEN  
                IF i PLUS k GREATER THAN n THEN  
                    RETURN the INTEGER minus one  
                END IF  
                SET flip_state TO the VALUE OF flip_state XORED WITH one  
                SET the element at position i PLUS k of flipped TO the VALUE OF the element at position i PLUS k of flipped XORED WITH one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
