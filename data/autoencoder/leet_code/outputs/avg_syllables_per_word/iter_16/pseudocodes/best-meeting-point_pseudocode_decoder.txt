CLASS Solution  
    FUNCTION minTotalDistance(grid)  
        SET rows TO an empty list  
        SET cols TO an empty list  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF the element at position zero of grid MINUS one  
                IF the element at position i of grid at position j EQUALS one  
                    APPEND i TO rows  
                    APPEND j TO cols  
                END IF  
            END FOR  
        END FOR  
        SORT the list rows in ascending order  
        SORT the list cols in ascending order  
        SET median_row TO the element at position the LENGTH OF rows DIVIDED BY two of rows  
        SET median_col TO the element at position the LENGTH OF cols DIVIDED BY two of cols  
        SET min_distance TO zero  
        FOR each element row IN rows  
            INCREMENT min_distance BY the ABSOLUTE VALUE of row MINUS median_row  
        END FOR  
        FOR each element col IN cols  
            INCREMENT min_distance BY the ABSOLUTE VALUE of col MINUS median_col  
        END FOR  
        RETURN min_distance  
    END FUNCTION  
END CLASS
