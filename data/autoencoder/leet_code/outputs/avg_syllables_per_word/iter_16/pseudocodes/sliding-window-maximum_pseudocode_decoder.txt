CLASS Solution  
    FUNCTION maxSlidingWindow(nums AS List of integers, k AS integer) RETURNS List of integers  
        IF nums IS empty OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        
        SET deq TO an empty double ended queue  
        SET result TO an empty list  
        
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the element from the front of deq  
            END IF  
            
            WHILE deq IS NOT empty AND the element at position the last of deq in nums IS LESS THAN the element at position i in nums  
                REMOVE the element from the end of deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position the first of deq in nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
