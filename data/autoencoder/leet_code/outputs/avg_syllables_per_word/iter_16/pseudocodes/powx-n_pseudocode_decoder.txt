CLASS Solution  
    FUNCTION myPow(x FLOAT, n INTEGER) RETURNS FLOAT  
        FUNCTION qpow(a FLOAT, n INTEGER) RETURNS FLOAT  
            SET ans TO one  
            WHILE n IS NOT EQUAL TO zero  
                IF the result of bitwise AND operation between n and one EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the result of unsigned right shift of n by one  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN the result of qpow with parameters x and n  
        ELSE  
            RETURN one DIVIDED BY the result of qpow with parameters x and the negation of n  
        END IF  
    END FUNCTION  
END CLASS
