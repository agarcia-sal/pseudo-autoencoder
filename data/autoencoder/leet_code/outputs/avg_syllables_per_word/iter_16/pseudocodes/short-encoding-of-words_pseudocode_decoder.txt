CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO the RESULT OF sorting each word REVERSED in words  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF the index PLUS one EQUALS the LENGTH OF reversed_words OR the word at position index PLUS one in reversed_words DOES NOT START WITH the word at position index in reversed_words  
                INCREMENT total_length BY the LENGTH OF the word at position index in reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
