CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t THEN  
                APPEND the JOIN of all elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs with the argument i PLUS one  
            IF the element at position i of t IS an alphabetic character THEN  
                SET the element at position i of t TO the character REPRESENTED BY the ASCII code of the element at position i of t EXCLUSIVE ORED BY thirty two  
                CALL dfs with the argument i PLUS one  
            END IF  
        END FUNCTION  
        
        SET t TO the list CONSTRUCTED FROM s  
        SET ans TO an empty list  
        CALL dfs with the argument zero  
        RETURN ans  
    END FUNCTION  
END CLASS
