CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        CREATE a list named dp to contain boolean values initialized to FALSE with the size equal to the total count of n PLUS one  
        FOR index i FROM one TO n  
            SET index j TO one  
            WHILE index j MULTIPLIED BY index j IS LESS THAN OR EQUAL TO i  
                IF dp at position i MINUS index j MULTIPLIED BY index j IS EQUAL TO FALSE  
                    SET dp at position i TO TRUE  
                    BREAK out of the current WHILE loop  
                END IF  
                INCREMENT index j BY one  
            END WHILE  
        END FOR  
        RETURN dp at position n  
    END FUNCTION  
END CLASS
