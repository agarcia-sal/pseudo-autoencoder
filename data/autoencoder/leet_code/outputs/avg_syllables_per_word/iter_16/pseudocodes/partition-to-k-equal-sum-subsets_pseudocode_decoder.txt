CLASS Solution  
    FUNCTION canPartitionKSubsets(nums AS List OF integers, k AS integer) RETURNS boolean  
        SET total_sum TO the SUM OF all elements IN nums  
        IF the REMAINDER OF total_sum DIVIDED BY k IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO the QUOTIENT OF total_sum DIVIDED BY k  
        SORT nums IN DESCENDING ORDER  
        
        FUNCTION can_partition(index AS integer, k AS integer, current_sum AS integer, used AS list of boolean) RETURNS boolean  
            IF k IS EQUAL TO zero  
                RETURN True  
            END IF  
            IF current_sum IS EQUAL TO target  
                RETURN the RESULT OF can_partition WITH parameters zero, the DIFFERENCE OF k AND one, zero, used  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF the element at position i of used IS False AND current_sum PLUS the element at position i of nums IS LESS THAN OR EQUAL TO target  
                    SET the element at position i of used TO True  
                    IF the RESULT OF can_partition WITH parameters the SUM OF i AND one, k, current_sum PLUS the element at position i of nums, used IS True  
                        RETURN True  
                    END IF  
                    SET the element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list of boolean False values WITH length EQUAL TO the LENGTH OF nums  
        RETURN the RESULT OF can_partition WITH parameters zero, k, zero, used  
    END FUNCTION  
END CLASS
