CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUALS the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the LEFT SIDE OF queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO A NEW TreeNode WITH val EQUALS the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO A NEW TreeNode WITH val EQUALS the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val DOES NOT EQUAL q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root IS None THEN  
            RETURN None  
        END IF  
        IF the val property of root GREATER THAN key THEN  
            SET the left child of root TO the RESULT OF self.deleteNode(the left child of root, key)  
            RETURN root  
        END IF  
        IF the val property of root LESS THAN key THEN  
            SET the right child of root TO the RESULT OF self.deleteNode(the right child of root, key)  
            RETURN root  
        END IF  
        IF the left child of root IS None THEN  
            RETURN the right child of root  
        END IF  
        IF the right child of root IS None THEN  
            RETURN the left child of root  
        END IF  
        SET node TO the right child of root  
        WHILE the left child of node IS NOT None  
            SET node TO the left child of node  
        END WHILE  
        SET the left child of node TO the left child of root  
        SET root TO the right child of root  
        RETURN root  
    END FUNCTION  
END CLASS
