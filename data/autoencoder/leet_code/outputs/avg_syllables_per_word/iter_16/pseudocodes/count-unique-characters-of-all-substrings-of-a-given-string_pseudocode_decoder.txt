CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO a list of twenty six elements each EQUAL TO negative one  
        SET last2 TO a list of twenty six elements each EQUAL TO negative one  
        SET result TO zero  
        FOR each index i AND character char IN s WITH enumeration  
            SET idx TO the Unicode code point of char MINUS the Unicode code point of uppercase letter A  
            SET contribution TO the DIFFERENCE BETWEEN i AND element at position idx of last MULTIPLIED BY the DIFFERENCE BETWEEN element at position idx of last AND element at position idx of last2  
            SET result TO the sum of result AND contribution MODULO MOD  
            SET element at position idx of last2 TO element at position idx of last  
            SET element at position idx of last TO i  
        END FOR  
        FOR each index i FROM zero TO twenty five  
            SET contribution TO the DIFFERENCE BETWEEN n AND element at position i of last MULTIPLIED BY the DIFFERENCE BETWEEN element at position i of last AND element at position i of last2  
            SET result TO the sum of result AND contribution MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
