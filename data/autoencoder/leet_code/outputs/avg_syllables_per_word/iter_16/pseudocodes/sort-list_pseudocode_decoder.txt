CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF the collection values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH the element at position one of values  
    SET pointer p TO head  
    FOR each element val IN the elements of values FROM position two TO the END  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN truth value true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN truth value false  
    END IF  
    RETURN THE RESULT OF the LOGICAL AND of THE CONDITION that the val attribute of p1 EQUALS the val attribute of p2 AND THE RESULT OF the FUNCTION is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList WITH PARAMETERS self AND head AS optional ListNode  
        IF head EQUALS none OR the next attribute of head EQUALS none  
            RETURN head  
        END IF  

        FUNCTION split WITH PARAMETER head  
            SET slow TO head  
            SET fast TO the next attribute of head  
            WHILE fast DOES NOT EQUAL none AND the next attribute of fast DOES NOT EQUAL none  
                SET slow TO the next attribute of slow  
                SET fast TO the next attribute of the next attribute of fast  
            END WHILE  
            SET mid TO the next attribute of slow  
            SET the next attribute of slow TO none  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge WITH PARAMETERS l1 AND l2  
            SET dummy TO A NEW ListNode WITH default initialization  
            SET current TO dummy  
            WHILE l1 DOES NOT EQUAL none AND l2 DOES NOT EQUAL none  
                IF the val attribute of l1 IS LESS THAN the val attribute of l2  
                    SET the next attribute of current TO l1  
                    SET l1 TO the next attribute of l1  
                ELSE  
                    SET the next attribute of current TO l2  
                    SET l2 TO the next attribute of l2  
                END IF  
                SET current TO the next attribute of current  
            END WHILE  
            IF l1 DOES NOT EQUAL none  
                SET the next attribute of current TO l1  
            ELSE  
                SET the next attribute of current TO l2  
            END IF  
            RETURN the next attribute of dummy  
        END FUNCTION  

        SET left AND right TO THE RESULT OF THE FUNCTION split WITH PARAMETER head  
        SET left TO THE RESULT OF THE METHOD sortList CALLED ON self WITH PARAMETER left  
        SET right TO THE RESULT OF THE METHOD sortList CALLED ON self WITH PARAMETER right  

        RETURN THE RESULT OF THE FUNCTION merge WITH PARAMETERS left AND right  
    END FUNCTION  
END CLASS
