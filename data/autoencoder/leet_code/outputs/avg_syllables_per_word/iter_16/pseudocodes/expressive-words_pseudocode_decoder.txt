CLASS Solution  
    FUNCTION expressiveWords(s, words)  
        FUNCTION compress(word)  
            IF word EQUALS empty  
                RETURN empty list  
            END IF  
            SET count TO one  
            SET compressed TO empty list  
            FOR index FROM one TO LENGTH OF word MINUS one  
                IF element at position index of word EQUALS element at position index MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND pair of element at position index MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND pair of element at last position of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy(s_compressed, word_compressed)  
            IF LENGTH OF s_compressed NOT EQUALS LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair of (character_sc, count_sc) AND (character_wc, count_wc) FROM paired elements of s_compressed AND word_compressed  
                IF character_sc NOT EQUALS character_wc  
                    RETURN false  
                END IF  
                IF count_sc NOT EQUALS count_wc AND (count_sc LESS THAN three OR count_sc LESS THAN count_wc)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET s_compressed TO RESULT OF compress WITH s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO RESULT OF compress WITH word  
            IF RESULT OF is_stretchy WITH s_compressed AND word_compressed  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
