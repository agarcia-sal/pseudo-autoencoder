CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO a new list consisting of the negation of each number in target  
        CALL heapify FUNCTION ON max_heap  
        SET total_sum TO the summation of all numbers in target  
        WHILE the element at position zero of max_heap is less than negative one  
            SET largest TO the negation of the result of removing the smallest element from max_heap by heap pop operation  
            SET rest_sum TO total_sum MINUS largest  
            IF largest is less than or equal to rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
            SET new_value TO the remainder when largest is divided by rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUAL TO one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            ADD the negation of new_value TO max_heap using heap push operation  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
