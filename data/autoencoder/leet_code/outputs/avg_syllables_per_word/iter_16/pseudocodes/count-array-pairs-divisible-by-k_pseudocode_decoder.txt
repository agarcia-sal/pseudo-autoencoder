CLASS Solution  
    FUNCTION countPairs(nums AS List OF INTEGER, k AS INTEGER) RETURNS INTEGER  
        SET gcd_count TO AN EMPTY COUNTER MAPPING FROM INTEGER TO INTEGER  
        
        FOR each num IN nums  
            INCREMENT the VALUE ASSOCIATED WITH the greatest common divisor OF num AND k IN gcd_count BY one  
        END FOR  
        
        SET total_pairs TO zero  
        
        SET gcd_values TO A LIST CONTAINING ALL KEYS IN gcd_count  
        SET n TO THE LENGTH OF gcd_values  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET gcd1 TO the ELEMENT AT position i OF gcd_values  
                SET gcd2 TO the ELEMENT AT position j OF gcd_values  
                IF the PRODUCT OF gcd1 AND gcd2 MODULO k EQUALS zero THEN  
                    IF i EQUALS j THEN  
                        INCREMENT total_pairs BY THE VALUE ASSOCIATED WITH gcd1 IN gcd_count MULTIPLIED BY THE VALUE ASSOCIATED WITH gcd1 IN gcd_count MINUS one DIVIDED BY two  
                    ELSE  
                        INCREMENT total_pairs BY THE VALUE ASSOCIATED WITH gcd1 IN gcd_count MULTIPLIED BY THE VALUE ASSOCIATED WITH gcd2 IN gcd_count  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_pairs  
    END FUNCTION  
END CLASS
