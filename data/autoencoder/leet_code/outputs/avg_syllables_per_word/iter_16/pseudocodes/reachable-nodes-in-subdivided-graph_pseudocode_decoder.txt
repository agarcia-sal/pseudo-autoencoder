CLASS Solution  
    FUNCTION reachableNodes(edges AS List OF List OF integer, maxMoves AS integer, n AS integer) AS integer  
        SET graph TO an empty mapping from integer TO empty list  
        FOR index FROM zero TO n MINUS one  
            SET element at position index of graph TO an empty list  
        END FOR  
        FOR each u AND v AND cnt IN edges  
            APPEND the pair of v AND cnt TO the list element at position u of graph  
            APPEND the pair of u AND cnt TO the list element at position v of graph  
        END FOR  
        
        SET distances TO a list of length n where each element is infinity  
        SET element at position zero of distances TO zero  
        SET priority_queue TO a list containing the pair of zero AND zero  
        
        WHILE priority_queue is not empty  
            REMOVE and RETURN the smallest first element pair from priority_queue AS dist AND node  
            IF dist IS GREATER THAN element at position node of distances  
                CONTINUE to next iteration of WHILE  
            END IF  
            FOR each neighbor AND cnt IN element at position node of graph  
                SET new_dist TO dist PLUS cnt PLUS one  
                IF new_dist IS LESS THAN element at position neighbor of distances  
                    SET element at position neighbor of distances TO new_dist  
                    INSERT the pair of new_dist AND neighbor INTO priority_queue maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        SET reachable_nodes TO zero  
        
        FOR each d IN distances  
            IF d IS LESS THAN OR EQUAL TO maxMoves  
                INCREMENT reachable_nodes BY one  
            END IF  
        END FOR  
        
        SET used_edges TO an empty mapping  
        FOR each u AND v AND cnt IN edges  
            SET moves_u TO the maximum OF zero AND the difference of maxMoves MINUS the element at position u of distances  
            SET moves_v TO the maximum OF zero AND the difference of maxMoves MINUS the element at position v of distances  
            SET reachable_new_nodes TO the minimum OF cnt AND the sum of moves_u AND moves_v  
            INCREMENT reachable_nodes BY reachable_new_nodes  
            SET the element identified by the pair u AND v of used_edges TO reachable_new_nodes  
        END FOR  
        
        RETURN reachable_nodes  
    END FUNCTION  
END CLASS
