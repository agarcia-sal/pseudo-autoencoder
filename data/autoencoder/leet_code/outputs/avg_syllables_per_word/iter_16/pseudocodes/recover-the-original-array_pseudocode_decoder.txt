CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL the SORT method on nums to arrange elements in ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two with integer division  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums THEN  
                CONTINUE to next iteration of the loop  
            END IF  
            SET count TO a new Counter initialized with nums  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF the value at key x in count EQUALS zero THEN  
                    CONTINUE to next iteration of the loop  
                END IF  
                IF the value at key x PLUS two MULTIPLIED BY k in count EQUALS zero THEN  
                    BREAK from the loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the value at key x in count BY one  
                DECREMENT the value at key x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two THEN  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
