CLASS Solution  
    FUNCTION findAllPeople(n AS integer, meetings AS list of list of integer, firstPerson AS integer) RETURNS list of integer  
        SET know_secret TO a set containing zero and firstPerson  
        SORT meetings BY the element at position three of the meeting in ascending order  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position three of the meeting at position i of meetings  
            INITIALIZE graph AS a mapping with default empty list of neighbors  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the element at position three of the meeting at position i of meetings EQUALS current_time  
                SET x TO the element at position one of the meeting at position i of meetings  
                SET y TO the element at position two of the meeting at position i of meetings  
                APPEND y TO the list of neighbors mapped to x in graph  
                APPEND x TO the list of neighbors mapped to y in graph  
                INCREMENT i BY one  
            END WHILE  
            INITIALIZE queue AS a double-ended queue containing each person IN know_secret WHO is a key IN graph  
            INITIALIZE visited AS an empty set  
            WHILE queue IS NOT empty  
                REMOVE the element from the left side of queue AND SET TO person  
                IF person IS IN visited THEN  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list of neighbors mapped to person in graph  
                    IF neighbor IS NOT IN visited THEN  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list containing all elements IN know_secret  
    END FUNCTION  
END CLASS
