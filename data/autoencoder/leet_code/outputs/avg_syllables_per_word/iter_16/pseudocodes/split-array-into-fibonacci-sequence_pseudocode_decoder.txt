CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path IS GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS zero AND end IS GREATER THAN start PLUS one  
                    CONTINUE to the next iteration  
                END IF  
                SET next_num TO the integer value represented by the substring from position start TO end MINUS one of num  
                IF next_num IS GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path IS LESS THAN two OR next_num EQUALS the element at the last position of path PLUS the element at the second last position of path  
                    SET result TO the RETURN VALUE OF backtrack WITH PARAMETERS end AND the LIST path WITH next_num APPENDED  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an EMPTY list  
        END FUNCTION  
        RETURN the RETURN VALUE OF backtrack WITH PARAMETERS zero AND an EMPTY list  
    END FUNCTION  
END CLASS
