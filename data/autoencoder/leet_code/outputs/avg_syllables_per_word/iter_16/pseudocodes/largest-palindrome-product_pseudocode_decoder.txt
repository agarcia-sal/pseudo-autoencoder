CLASS Solution  
    FUNCTION largestPalindrome with parameter n  
        IF n EQUALS one  
            RETURN nine  
        END IF  
        SET upper_limit TO ten RAISED TO THE POWER OF n MINUS one  
        SET lower_limit TO ten RAISED TO THE POWER OF n MINUS one  
        SET max_palindrome TO zero  
        FOR i FROM upper_limit DOWN TO lower_limit  
            FOR j FROM i DOWN TO lower_limit  
                SET product TO i MULTIPLIED BY j  
                IF product LESS THAN OR EQUAL TO max_palindrome  
                    BREAK  
                END IF  
                SET str_product TO CONVERSION OF product TO STRING  
                IF str_product EQUALS THE REVERSE OF str_product  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  
        RETURN max_palindrome MODULO one thousand three hundred thirty seven  
    END FUNCTION  
END CLASS
