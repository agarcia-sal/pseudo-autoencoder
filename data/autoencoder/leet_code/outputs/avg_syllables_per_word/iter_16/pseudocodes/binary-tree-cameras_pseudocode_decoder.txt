CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET the val attribute OF self TO val  
        SET the left attribute OF self TO left  
        SET the right attribute OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO the element removed FROM the left OF queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the left attribute OF node TO a new TreeNode initialized WITH the element at position i of values  
            APPEND the left attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the right attribute OF node TO a new TreeNode initialized WITH the element at position i of values  
            APPEND the right attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val attribute OF p NOT EQUALS the val attribute OF q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree CALLED WITH the left attribute OF p AND the left attribute OF q AND the result of is_same_tree CALLED WITH the right attribute OF p AND the right attribute OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node IS None THEN  
                RETURN zero TO represent no camera monitored count, zero TO represent monitored count, and positive infinity TO represent camera count  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO the results of dfs CALLED WITH the left attribute OF node  
            SET right_no_camera, right_monitored, right_camera TO the results of dfs CALLED WITH the right attribute OF node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the smallest value AMONG left_camera PLUS the smallest value AMONG right_monitored AND right_camera, right_camera PLUS the smallest value AMONG left_monitored AND left_camera, AND left_camera PLUS right_camera  
            SET camera TO one PLUS the smallest value AMONG left_no_camera, left_monitored, AND left_camera PLUS the smallest value AMONG right_no_camera, right_monitored, AND right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO dfs CALLED WITH root  
        RETURN the smallest value AMONG root_monitored AND root_camera  
    END FUNCTION  
END CLASS
