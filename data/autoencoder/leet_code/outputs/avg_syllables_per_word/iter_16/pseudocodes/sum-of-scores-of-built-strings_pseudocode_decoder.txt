CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        SET lcp TO a list of zeros with the LENGTH OF s  
        SET l TO zero  
        SET r TO zero  
        FOR index i FROM one TO n MINUS one  
            IF i IS LESS THAN OR EQUAL TO r  
                SET lcp at position i TO the MINIMUM OF r MINUS i PLUS one AND lcp at position i MINUS l  
            END IF  
            WHILE i PLUS lcp at position i IS LESS THAN n AND the character at position lcp at position i of s EQUALS the character at position i PLUS lcp at position i of s  
                INCREMENT lcp at position i BY one  
            END WHILE  
            IF i PLUS lcp at position i MINUS one IS GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp at position i MINUS one  
            END IF  
        END FOR  
        SET total_sum TO the SUM OF all elements IN lcp PLUS n  
        RETURN total_sum  
    END FUNCTION  
END CLASS
