CLASS Solution  
    FUNCTION numOfArrays(n INTEGER, m INTEGER, k INTEGER) RETURNS INTEGER  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize three dimensional list dp with dimensions n PLUS one BY m PLUS one BY k PLUS one FILLED WITH zero  
        
        FOR j FROM one TO m  
            SET element at position one BY j BY one of dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position i BY j BY p of dp TO element at position i BY j BY p of dp PLUS element at position i MINUS one BY x BY p MINUS one of dp  
                        SET element at position i BY j BY p of dp TO element at position i BY j BY p of dp MODULO MOD  
                    END FOR  
                    SET element at position i BY j BY p of dp TO element at position i BY j BY p of dp PLUS element at position i MINUS one BY j BY p of dp MULTIPLIED BY j  
                    SET element at position i BY j BY p of dp TO element at position i BY j BY p of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position n BY j BY k of dp  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
