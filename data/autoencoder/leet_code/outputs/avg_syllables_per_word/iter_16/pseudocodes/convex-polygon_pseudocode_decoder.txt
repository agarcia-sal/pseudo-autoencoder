CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN the difference of the product of the difference between element at position zero of a and element at position zero of o and the difference between element at position one of b and element at position one of o AND the product of the difference between element at position one of a and element at position one of o and the difference between element at position zero of b and element at position zero of o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN true  
        END IF  
        
        SET orientation TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET o TO element at position index of points  
            SET a TO element at position of the remainder of the division of index PLUS one BY n of points  
            SET b TO element at position of the remainder of the division of index PLUS two BY n of points  
            SET cross TO cross_product WITH arguments o a b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    SET orientation TO one IF cross GREATER THAN zero ELSE MINUS one  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS MINUS one AND cross GREATER THAN zero  
                    RETURN false  
                END IF  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
