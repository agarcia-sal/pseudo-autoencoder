CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs  
        SET anagrams TO a new collection that creates an empty list for any new key  
        FOR each string s IN strs  
            SET sorted_str TO the result of joining all characters of string s arranged in ascending alphabetical order  
            APPEND string s TO the list corresponding to the key sorted_str in anagrams  
        END FOR  
        RETURN a list formed by all values contained in anagrams  
    END FUNCTION  
END CLASS
