CLASS Solution  
    FUNCTION frequencySort(s)  
        ASSIGN pointer TO Counter OBJECT initialized with s  
        ASSIGN frequency TO pointer  
        ASSIGN list_of_characters TO the collection of keys of frequency  
        ASSIGN sorted_characters TO list_of_characters SORTED BY the frequency of each character IN descending order  
        SET result TO the empty string  
        FOR each character IN sorted_characters  
            REPEAT character BY the frequency of character AND APPEND TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
