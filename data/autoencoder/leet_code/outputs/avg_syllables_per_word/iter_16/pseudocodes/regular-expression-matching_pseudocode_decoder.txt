CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp AS two dimensional list sized length of s plus one BY length of p plus one FILLED WITH False  
        SET element at position zero zero of dp TO True  
        FOR each variable j FROM one TO length of p inclusive  
            IF element at position j minus one of p EQUALS star  
                SET element at position zero j of dp TO element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR each variable i FROM one TO length of s inclusive  
            FOR each variable j FROM one TO length of p inclusive  
                IF element at position j minus one of p EQUALS element at position i minus one of s OR element at position j minus one of p EQUALS dot  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS star  
                    SET element at position i j of dp TO element at position i j minus two of dp  
                    IF element at position j minus two of p EQUALS element at position i minus one of s OR element at position j minus two of p EQUALS dot  
                        SET element at position i j of dp TO element at position i j of dp OR element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position length of s length of p of dp  
    END FUNCTION  
END CLASS
