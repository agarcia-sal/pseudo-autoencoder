CLASS Solution  
    FUNCTION findMaxValueOfEquation(points AS List OF List OF integer, k AS integer) RETURNS integer  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        
        FOR each element consisting of xj and yj IN points  
            WHILE deque_ is not empty AND the difference of xj MINUS the element at position zero of deque_ at position one is GREATER THAN k  
                REMOVE the element at position zero FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value between max_value AND the sum of yj PLUS xj PLUS the element at position zero of deque_ at position zero  
            END IF  
            
            WHILE deque_ is not empty AND the difference of yj MINUS xj IS GREATER THAN OR EQUAL TO the element at the last position of deque_ at position zero  
                REMOVE the element at the last position FROM deque_  
            END WHILE  
            
            APPEND the pair consisting of the difference of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
