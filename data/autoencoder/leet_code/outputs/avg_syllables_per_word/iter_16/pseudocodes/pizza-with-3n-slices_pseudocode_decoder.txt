CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end  
                RETURN zero  
            END IF  
            
            SET pick TO the element at position start of slices PLUS the RESULT of dp with parameters start PLUS two END end k MINUS one  
            SET skip TO the RESULT of dp with parameters start PLUS one END end k  
            IF pick GREATER THAN skip  
                RETURN pick  
            ELSE  
                RETURN skip  
            END IF  
        END FUNCTION  
        
        SET case1 TO the RESULT of dp with parameters one END the LENGTH OF slices MINUS one n  
        SET case2 TO the RESULT of dp with parameters zero END the LENGTH OF slices MINUS two n  
        IF case1 GREATER THAN case2  
            RETURN case1  
        ELSE  
            RETURN case2  
        END IF  
    END FUNCTION  
END CLASS
