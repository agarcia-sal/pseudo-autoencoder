CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO a frequency COUNTER of digits  
        SET total_sum TO the SUM of all elements in digits  
        SET remainder TO the value of total_sum MODULO three  
        IF remainder EQUALS one  
            FOR each digit IN the list of digits one FOUR seven  
                IF the frequency of digit in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            IF no break occurred in the previous loop  
                FOR each digit IN the list of digits two FIVE eight  
                    IF the frequency of digit in count IS GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        IF remainder EQUALS two  
            FOR each digit IN the list of digits two FIVE eight  
                IF the frequency of digit in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            IF no break occurred in the previous loop  
                FOR each digit IN the list of digits one FOUR seven  
                    IF the frequency of digit in count IS GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        SET result TO an empty list  
        FOR digit FROM nine DOWN TO zero  
            APPEND the string representation of digit repeated the frequency of digit in count times TO result  
        END FOR  
        SET final_number TO the concatenation of all elements in result  
        IF final_number IS NOT empty AND the element at position zero of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
