CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO a set containing all elements from deadends  
        
        IF the string zero zero zero zero is an element of deadends_set  
            RETURN negative one  
        END IF  
        
        IF target EQUALS the string zero zero zero zero  
            RETURN zero  
        END IF  
        
        INITIALIZE queue AS a deque containing a single element consisting of the string zero zero zero zero AND the integer zero  
        SET visited TO a set containing the string zero zero zero zero  
        
        FUNCTION neighbors(state)  
            INITIALIZE result AS an empty list  
            FOR index FROM zero TO three  
                SET digit TO the integer value of the character at position index of state  
                FOR move IN the list containing negative one AND positive one  
                    SET new_digit TO the sum of digit AND move MODULO ten  
                    SET new_state TO the concatenation of the substring from position zero TO one BEFORE index of state, THE string representation of new_digit, AND the substring from position one AFTER index TO the end of state  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE the queue is not empty  
            REMOVE AND ASSIGN current_state AND steps FROM the first element of queue  
            
            FOR each neighbor IN neighbors(current_state)  
                IF neighbor is not an element of visited AND neighbor is not an element of deadends_set  
                    IF neighbor EQUALS target  
                        RETURN the sum of steps AND one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND a tuple consisting of neighbor AND the sum of steps AND one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
