CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN pointer/reference TO a new double ended queue initialized with one element that is a tuple containing zero position one speed and zero steps  
        ASSIGN pointer/reference TO a new set initialized with one tuple containing zero position and one speed to keep track of visited states to avoid reprocessing  
        
        WHILE the queue contains any element  
            REMOVE the element from the front of the queue and ASSIGN its components TO position speed and steps respectively  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple containing next_position and next_speed is not in visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple containing next_position and next_speed TO visited  
                APPEND the tuple containing next_position next_speed and steps PLUS one TO the queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            
            IF the tuple containing position and next_speed is not in visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple containing position and next_speed TO visited  
                APPEND the tuple containing position next_speed and steps PLUS one TO the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
