CLASS Solution  
    FUNCTION smallestMissingValueSubtree(parents AS List of integer, nums AS List of integer) RETURNS List of integer  
        SET n TO the LENGTH OF parents  
        SET tree TO an empty mapping from integer TO list of integers with default empty list  
        
        FOR index FROM one TO n MINUS one  
            APPEND index TO the list at key equal to element at position index of parents in tree  
        END FOR  
        
        SET node_with_one TO minus one  
        FOR index FROM zero TO n MINUS one  
            IF element at position index of nums EQUALS one  
                SET node_with_one TO index  
                BREAK the loop  
            END IF  
        END FOR  
        
        IF node_with_one EQUALS minus one  
            SET result TO a list consisting of one repeated n times  
            RETURN result  
        END IF  
        
        SET result TO a list consisting of one repeated n times  
        
        FUNCTION dfs(node AS integer) RETURNS set of integers  
            ADD node TO the set named visited  
            SET genetic_values TO a set containing element at position node of nums  
            FOR each child IN the list at key equal to node in tree  
                IF child NOT IN visited  
                    SET child_values TO dfs(child)  
                    ADD all elements of child_values TO genetic_values  
                END IF  
            END FOR  
            
            SET smallest_missing TO one  
            WHILE smallest_missing IS IN genetic_values  
                INCREMENT smallest_missing BY one  
            END WHILE  
            SET element at position node of result TO smallest_missing  
            RETURN genetic_values  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET path TO an empty list  
        SET current TO node_with_one  
        WHILE current NOT EQUALS minus one  
            APPEND current TO path  
            SET current TO element at position current of parents  
        END WHILE  
        
        SET path_set TO a set created from path  
        FOR each node IN the reverse order of path  
            IF node IS IN visited  
                CONTINUE to the next iteration  
            END IF  
            CALL dfs WITH node  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
