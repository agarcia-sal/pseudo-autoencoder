CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters obtained by replacing the colon in time with an empty string  
        SET hour TO the integer value of the substring from position zero TO one of time  
        SET minute TO the integer value of the substring from position three TO four of time  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each first_hour_digit IN digits  
                FOR each second_hour_digit IN digits  
                    FOR each first_minute_digit IN digits  
                        FOR each second_minute_digit IN digits  
                            SET combined_hour_string TO the concatenation of first_hour_digit WITH second_hour_digit  
                            SET combined_minute_string TO the concatenation of first_minute_digit WITH second_minute_digit  
                            SET generated_hour TO the integer value of combined_hour_string  
                            SET generated_minute TO the integer value of combined_minute_string  
                            IF generated_hour IS LESS THAN twenty four AND generated_minute IS LESS THAN sixty  
                                APPEND the pair of generated_hour AND generated_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times WITH digits  
        
        SET current_time_index TO the position of the pair consisting of hour AND minute IN all_times  
        
        IF current_time_index PLUS one IS LESS THAN the LENGTH OF all_times  
            SET next_hour AND next_minute TO the element at position current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour AND next_minute TO the element at position zero OF all_times  
        END IF  
        
        RETURN the string formed by formatting next_hour AS a two digit number FOLLOWED BY a colon FOLLOWED BY next_minute AS a two digit number  
    END FUNCTION  
END CLASS
