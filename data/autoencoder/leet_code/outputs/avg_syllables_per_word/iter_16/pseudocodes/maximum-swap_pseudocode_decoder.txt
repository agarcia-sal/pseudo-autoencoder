CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of characters representing num converted to string  
        SET last TO a mapping from digit converted to integer TO its last position as index in digits  
        FOR each position i and digit character d IN digits, iterated by position and element  
            FOR each digit d2 FROM nine DOWN TO the integer value of d PLUS one, decreasing by one each step  
                IF the value from last for key d2 with default zero GREATER THAN i  
                    SET temporary variable to element at position i of digits  
                    SET element at position i of digits TO element at position from last for key d2 of digits  
                    SET element at position from last for key d2 of digits TO temporary variable  
                    SET swapped_number TO the integer converted from concatenation of all characters in digits in order  
                    RETURN swapped_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
