CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO the LIST OF TUPLES consisting of elements FROM indices, sources, AND targets SORTED BY the FIRST element OF each tuple  
        SET result TO an empty LIST  
        SET prev_end TO zero  
        FOR each tuple consisting of index, source, AND target IN replacements  
            APPEND the SUBSTRING of s FROM position prev_end TO position index TO result  
            IF the SUBSTRING of s FROM position index TO position index PLUS the LENGTH OF source EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the SUBSTRING of s FROM position index TO position index PLUS the LENGTH OF source TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the SUBSTRING of s FROM position prev_end TO the END OF s TO result  
        RETURN the STRING CREATED BY JOINING all ELEMENTS OF result WITH NO SEPARATOR  
    END FUNCTION  
END CLASS
