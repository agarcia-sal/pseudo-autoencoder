CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums CONVERTED TO string  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the element at position zero of nums CONCATENATED WITH the character FOR division CONCATENATED WITH the element at position one of nums CONVERTED TO string  
        END IF  
        RETURN the element at position zero of nums CONCATENATED WITH the character FOR division CONCATENATED WITH the character FOR left parenthesis CONCATENATED WITH all elements from position one TO the last position of nums EACH CONVERTED TO string AND JOINED WITH the character FOR division CONCATENATED WITH the character FOR right parenthesis  
    END FUNCTION  
END CLASS
