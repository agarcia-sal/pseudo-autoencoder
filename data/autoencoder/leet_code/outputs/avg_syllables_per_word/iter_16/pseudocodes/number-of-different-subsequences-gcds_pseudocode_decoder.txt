CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element IN nums  
        SET num_set TO a set containing all elements of nums  
        SET count TO zero  
        FOR each integer g FROM one TO max_num  
            SET gcd_val TO zero  
            FOR each integer m FROM g TO max_num IN STEPS OF g  
                IF element m IS IN num_set THEN  
                    SET gcd_val TO the greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY one  
                        EXIT THE INNER LOOP  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
