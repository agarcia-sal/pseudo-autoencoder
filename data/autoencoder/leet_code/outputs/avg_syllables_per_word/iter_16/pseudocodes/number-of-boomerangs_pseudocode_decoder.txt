CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO zero  
        FOR each point p1 IN points  
            SET cnt TO a new counting map  
            FOR each point p2 IN points  
                SET dx TO element at position zero of point p1 MINUS element at position zero of point p2  
                SET dy TO element at position one of point p1 MINUS element at position one of point p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the COUNT from cnt at key d  
                INCREMENT the COUNT in cnt at key d BY one  
            END FOR  
        END FOR  
        RETURN ans MULTIPLIED BY two  
    END FUNCTION  
END CLASS
