CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO one billion PLUS seven  
        
        IF primeFactors LESS THAN OR EQUAL TO three  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO integer division of primeFactors BY three  
        SET remainder TO primeFactors MINUS num_threes MULTIPLIED BY three  
        
        IF remainder EQUALS zero  
            RETURN the result of raising three to the power of num_threes then modulo MOD  
        ELSE IF remainder EQUALS one  
            RETURN the result of raising three to the power of num_threes MINUS one then modulo MOD MULTIPLIED BY four then modulo MOD  
        ELSE  
            RETURN the result of raising three to the power of num_threes then modulo MOD MULTIPLIED BY two then modulo MOD  
        END IF  
    END FUNCTION  
END CLASS
