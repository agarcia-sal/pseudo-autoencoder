CLASS Solution  
    FUNCTION getXORSum WITH PARAMETERS arr1 AND arr2  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO xor_sum_arr1 EXCLUSIVE OR num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO xor_sum_arr2 EXCLUSIVE OR num  
        END FOR  
        RETURN bitwise AND of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
