CLASS Solution  
    FUNCTION reachingPoints WITH PARAMETERS sx sy tx ty  
        WHILE the VALUE OF tx IS GREATER THAN OR EQUAL TO the VALUE OF sx AND the VALUE OF ty IS GREATER THAN OR EQUAL TO the VALUE OF sy HOLDS  
            IF the VALUE OF tx EQUALS the VALUE OF sx AND the VALUE OF ty EQUALS the VALUE OF sy  
                RETURN True  
            END IF  
            IF the VALUE OF tx IS GREATER THAN the VALUE OF ty  
                IF the VALUE OF ty IS GREATER THAN the VALUE OF sy  
                    SET tx TO the REMAINDER AFTER DIVIDING the VALUE OF tx BY the VALUE OF ty  
                ELSE  
                    RETURN whether the REMAINDER AFTER SUBTRACTING the VALUE OF sx FROM the VALUE OF tx IS EQUAL TO zero WHEN DIVIDED BY the VALUE OF ty  
                END IF  
            ELSE  
                IF the VALUE OF tx IS GREATER THAN the VALUE OF sx  
                    SET ty TO the REMAINDER AFTER DIVIDING the VALUE OF ty BY the VALUE OF tx  
                ELSE  
                    RETURN whether the REMAINDER AFTER SUBTRACTING the VALUE OF sy FROM the VALUE OF ty IS EQUAL TO zero WHEN DIVIDED BY the VALUE OF tx  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
