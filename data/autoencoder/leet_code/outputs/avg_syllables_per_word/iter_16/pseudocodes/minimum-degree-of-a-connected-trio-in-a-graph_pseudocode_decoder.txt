CLASS Solution  
    FUNCTION minTrioDegree WITH PARAMETERS n AND edges  
        SET graph TO a new mapping from integer to set of integers  
        FOR each integer i FROM one TO n  
            ASSIGN element at integer i of graph TO an empty set  
        END FOR  
        
        FOR each pair of integers u and v IN edges  
            ADD integer v TO the set at element integer u of graph  
            ADD integer u TO the set at element integer v of graph  
        END FOR  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR each integer i FROM one TO n  
            FOR each integer j FROM i PLUS one TO n  
                FOR each integer k FROM j PLUS one TO n  
                    IF the set at element integer i of graph CONTAINS integer j AND the set at element integer i of graph CONTAINS integer k AND the set at element integer j of graph CONTAINS integer k THEN  
                        SET trio_found TO true  
                        SET trio_degree TO the length of the set at element integer i of graph PLUS the length of the set at element integer j of graph PLUS the length of the set at element integer k of graph MINUS six  
                        IF trio_degree LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found THEN  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
