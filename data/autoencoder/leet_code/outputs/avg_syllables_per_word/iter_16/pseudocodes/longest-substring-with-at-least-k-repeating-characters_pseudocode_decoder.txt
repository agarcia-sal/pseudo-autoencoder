CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s AND k  
        IF the LENGTH OF s LESS THAN k THEN  
            RETURN zero  
        END IF  
        SET count TO CALL Counter FUNCTION WITH ARGUMENT s  
        FOR each character IN count  
            IF the FREQUENCY OF character IN count LESS THAN k THEN  
                RETURN the MAXIMUM OF CALL longestSubstring WITH PARAMETERS each substring OBTAINED BY SPLITTING s BY character AND k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
