CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH parameter s  
        SET sequence_count TO an empty dictionary with default integer zero values  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO substring from position index TO position index PLUS ten MINUS one of s  
            INCREMENT the element at key sequence of sequence_count BY one  
        END FOR  
        SET repeated_sequences TO a list of all sequences for each sequence and count in the elements of sequence_count WHERE count IS GREATER THAN one  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
