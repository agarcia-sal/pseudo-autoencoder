CLASS Solution  
    FUNCTION rearrangeSticks(n INTEGER, k INTEGER) RETURNS INTEGER  
        SET MOD TO ten POWER nine PLUS seven  
        
        INITIALIZE dp AS a two dimensional list of zero with dimensions n PLUS one BY k PLUS one  
        
        SET element at position zero of dp TO a list of zero with length k PLUS one  
        SET element at position zero comma zero of dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position i comma j of dp TO the REMAINDER WHEN the SUM OF element at position i MINUS one comma j MINUS one of dp PLUS the PRODUCT OF i MINUS one AND element at position i MINUS one comma j of dp IS DIVIDED BY MOD  
            END FOR  
        END FOR  
        
        RETURN element at position n comma k of dp  
    END FUNCTION  
END CLASS
