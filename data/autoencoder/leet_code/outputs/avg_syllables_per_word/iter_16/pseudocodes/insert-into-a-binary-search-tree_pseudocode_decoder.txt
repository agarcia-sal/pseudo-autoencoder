CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A new TreeNode initialized WITH element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the front OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO A new TreeNode initialized WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO A new TreeNode initialized WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q dot left AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST WITH PARAMETERS self root val  
        FUNCTION dfs WITH PARAMETER root  
            IF root IS none  
                RETURN A new TreeNode initialized WITH val  
            END IF  
            IF root dot val IS LESS THAN val  
                SET root dot right TO dfs WITH PARAMETER root dot right  
            ELSE  
                SET root dot left TO dfs WITH PARAMETER root dot left  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
