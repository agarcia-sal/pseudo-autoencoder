CLASS Solution  
    FUNCTION eventualSafeNodes(graph)  
        FUNCTION is_safe(node)  
            IF the element at position node of visited EQUALS one THEN  
                RETURN false  
            END IF  
            IF the element at position node of visited EQUALS two THEN  
                RETURN true  
            END IF  
            SET the element at position node of visited TO one  
            FOR each neighbor IN the element at position node of graph  
                IF is_safe(neighbor) EQUALS false THEN  
                    RETURN false  
                END IF  
            END FOR  
            SET the element at position node of visited TO two  
            RETURN true  
        END FUNCTION  
        
        SET n TO the LENGTH OF graph  
        SET visited TO a list of zeros of LENGTH n  
        SET safe_nodes TO an empty list  
        
        FOR node FROM zero TO n MINUS one  
            IF is_safe(node) EQUALS true THEN  
                APPEND node TO safe_nodes  
            END IF  
        END FOR  
        
        RETURN safe_nodes  
    END FUNCTION  
END CLASS
