CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        
        FOR each integer i FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET good TO a new LIST OF BOOLEAN FALSE VALUES with the LENGTH n  
            SET bad TO a new LIST OF BOOLEAN FALSE VALUES with the LENGTH n  
            
            FOR each integer j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND TWO RAISED TO THE POWER OF j IS NOT EQUALS zero  
                    SET the element at position j of good TO TRUE  
                ELSE  
                    SET the element at position j of bad TO TRUE  
                END IF  
            END FOR  
            
            SET valid TO TRUE  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of good IS TRUE  
                    FOR each integer k FROM zero TO n MINUS one  
                        IF the element at position j and position k of statements EQUALS zero AND the element at position k of good IS TRUE  
                            SET valid TO FALSE  
                            BREAK out of the inner loop  
                        ELSE IF the element at position j and position k of statements EQUALS one AND the element at position k of bad IS TRUE  
                            SET valid TO FALSE  
                            BREAK out of the inner loop  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        BREAK out of the outer loop  
                    END IF  
                END IF  
            END FOR  
            
            IF valid IS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF all TRUE elements in good  
            END IF  
        END FOR  
        
        RETURN max_good  
    END FUNCTION  
END CLASS
