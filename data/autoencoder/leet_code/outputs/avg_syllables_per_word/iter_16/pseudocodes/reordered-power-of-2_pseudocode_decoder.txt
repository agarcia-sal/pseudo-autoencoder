CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH parameter n OF TYPE integer  
        SET power_of_2_sorted_digits TO empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO ten RAISED TO THE NINTH POWER  
            ADD the tuple of the digits sorted in ascending order of the string representation of power_of_2 TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the digits sorted in ascending order of the string representation of n  
        IF sorted_n_digits IS IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
