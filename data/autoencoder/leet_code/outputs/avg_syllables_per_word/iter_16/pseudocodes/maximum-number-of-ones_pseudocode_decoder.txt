CLASS Solution  
    FUNCTION maximumNumberOfOnes(width INTEGER, height INTEGER, sideLength INTEGER, maxOnes INTEGER) RETURNS INTEGER  
        SET freq TO a list of zeros with the LENGTH EQUAL TO sideLength MULTIPLIED BY sideLength  
        FOR each integer i FROM zero TO width MINUS one  
            FOR each integer j FROM zero TO height MINUS one  
                SET position TO the remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of j DIVIDED BY sideLength  
                INCREMENT the element at position position of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET sum_of_top_frequencies TO the sum of the first maxOnes elements of freq  
        RETURN sum_of_top_frequencies  
    END FUNCTION  
END CLASS
