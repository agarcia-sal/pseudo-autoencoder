CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid IS empty OR the element at position zero of grid IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        SET max_kills TO zero  
        
        SET row_kills TO zero  
        SET col_kills TO a list of zeroes of LENGTH n  
        
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF j EQUALS zero OR the element at position j MINUS one at position i of grid EQUALS the character representing a wall  
                    SET row_kills TO zero  
                    FOR each k FROM j TO n MINUS one  
                        IF the element at position k at position i of grid EQUALS the character representing an enemy  
                            INCREMENT row_kills BY one  
                        ELSE IF the element at position k at position i of grid EQUALS the character representing a wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF i EQUALS zero OR the element at position j at position i MINUS one of grid EQUALS the character representing a wall  
                    SET the element at position j of col_kills TO zero  
                    FOR each k FROM i TO m MINUS one  
                        IF the element at position j at position k of grid EQUALS the character representing an enemy  
                            INCREMENT the element at position j of col_kills BY one  
                        ELSE IF the element at position j at position k of grid EQUALS the character representing a wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF the element at position j at position i of grid EQUALS the character representing zero  
                    SET max_kills TO the GREATER OF max_kills AND the SUM OF row_kills AND the element at position j of col_kills  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
