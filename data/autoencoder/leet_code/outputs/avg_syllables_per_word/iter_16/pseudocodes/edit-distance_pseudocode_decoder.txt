CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        ASSIGN pointer TO two-dimensional list dp WITH the DIMENSION from zero TO m INCLUSIVE AND zero TO n INCLUSIVE INITIATED WITH zero  
        FOR each index i FROM zero TO m INCLUSIVE  
            SET element at position zero of element at position i of dp TO i  
        END FOR  
        FOR each index j FROM zero TO n INCLUSIVE  
            SET element at position j of element at position zero of dp TO j  
        END FOR  
        FOR each index i FROM one TO m INCLUSIVE  
            FOR each index j FROM one TO n INCLUSIVE  
                IF character at position i MINUS one of word1 EQUALS character at position j MINUS one of word2  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET deletion_cost TO element at position j of element at position i MINUS one of dp  
                    SET insertion_cost TO element at position j MINUS one of element at position i of dp  
                    SET replacement_cost TO element at position j MINUS one of element at position i MINUS one of dp  
                    SET element at position j of element at position i of dp TO one PLUS the MINIMUM OF deletion_cost insertion_cost replacement_cost  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
