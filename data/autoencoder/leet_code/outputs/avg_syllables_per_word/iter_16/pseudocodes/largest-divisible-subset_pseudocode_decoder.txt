CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        PERFORM sorting on nums in ascending order  
        SET n TO the LENGTH OF nums  
        INITIALIZE f AS a list consisting of n elements each being one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums DIVIDED BY the element at position j of nums HAS zero remainder THEN  
                    SET f at position i TO the GREATER VALUE BETWEEN f at position i AND f at position j PLUS one  
                END IF  
            END FOR  
            IF f at position k IS LESS THAN f at position i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f at position k  
        SET i TO k  
        INITIALIZE ans AS an empty list  
        WHILE m IS GREATER THAN zero  
            IF the element at position k of nums DIVIDED BY the element at position i of nums HAS zero remainder AND f at position i EQUALS m THEN  
                APPEND the element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
