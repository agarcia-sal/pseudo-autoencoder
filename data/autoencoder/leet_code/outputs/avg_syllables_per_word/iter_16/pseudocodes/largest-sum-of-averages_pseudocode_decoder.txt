CLASS Solution  
    FUNCTION largestSumOfAverages(nums as List of integers, k as integer) RETURNS float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zero repeated n PLUS one times  
        FOR index FROM zero TO n MINUS one  
            SET element at position index PLUS one of prefix_sum TO element at position index of prefix_sum PLUS element at position index of nums  
        END FOR  
        
        FUNCTION average(i as integer, j as integer) RETURNS float  
            SET numerator TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET dp TO a two dimensional list of zero repeated n PLUS one times by k PLUS one times  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at partition one of dp TO CALL average WITH i AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET candidate_value TO element at position j at partition partitions MINUS one of dp PLUS CALL average WITH i AND j MINUS one  
                    SET element at position i at partition partitions of dp TO THE MAXIMUM OF element at position i at partition partitions of dp AND candidate_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero at partition k of dp  
    END FUNCTION  
END CLASS
