CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO side WHERE each plus SIGN IS REPLACED BY a separator FOLLOWED BY plus SIGN AND each minus SIGN IS REPLACED BY a separator FOLLOWED BY minus SIGN THEN SPLIT by the separator  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS empty  
                    CONTINUE to next token  
                END IF  
                IF the character x IS IN token  
                    IF token IS exactly x OR token IS exactly plus SIGN FOLLOWED BY x  
                        INCREMENT x_count BY one  
                    ELSE IF token IS exactly minus SIGN FOLLOWED BY x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET coefficient_string TO token WITH the character x REMOVED  
                        SET coefficient_number TO the integer value OF coefficient_string  
                        INCREMENT x_count BY coefficient_number  
                    END IF  
                ELSE  
                    SET number_value TO the integer value OF token  
                    INCREMENT num_sum BY number_value  
                END IF  
            END FOR  
            RETURN the pair x_count AND num_sum  
        END FUNCTION  

        SPLIT equation INTO two parts left_side AND right_side AROUND the equal SIGN  
        SET left_x AND left_num TO THE RESULT OF parse_side APPLIED TO left_side  
        SET right_x AND right_num TO THE RESULT OF parse_side APPLIED TO right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN Infinite solutions  
            ELSE  
                RETURN No solution  
            END IF  
        ELSE  
            SET quotient TO total_num DIVIDED BY total_x USING integer DIVISION  
            RETURN the STRING concatenation OF the character x EQUALS AND quotient  
        END IF  
    END FUNCTION  
END CLASS
