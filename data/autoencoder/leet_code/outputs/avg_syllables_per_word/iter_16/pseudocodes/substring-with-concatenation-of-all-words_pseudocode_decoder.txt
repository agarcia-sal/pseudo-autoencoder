CLASS Solution  
    FUNCTION findSubstring WITH PARAMETERS s AND words  
        IF s IS EMPTY OR words IS EMPTY OR element at position zero of words IS EMPTY  
            RETURN empty list  
        END IF  
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO a new Counter initialized WITH words  
        SET result TO empty list  
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO a new Counter  
            FOR index j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO the substring from position j TO j PLUS word_length of s  
                INCREMENT count of word in seen BY one  
                IF count of word in seen IS GREATER THAN count of word in word_count  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
            IF the inner loop did not break  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
