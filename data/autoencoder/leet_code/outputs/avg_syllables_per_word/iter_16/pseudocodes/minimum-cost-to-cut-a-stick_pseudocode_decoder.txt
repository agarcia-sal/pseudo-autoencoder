CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO cuts PLUS a list containing zero AND n  
        SET cuts TO cuts SORTED IN ASCENDING ORDER  
      
        FUNCTION dp(left, right)  
            IF right MINUS left IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
          
            SET min_cost TO infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right of cuts MINUS element at position left of cuts PLUS dp(left, i) PLUS dp(i, right)  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
          
            RETURN min_cost  
        END FUNCTION  
      
        RETURN dp(zero, the LENGTH OF cuts MINUS one)  
END CLASS
