CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO an empty list of length n where each element is an empty list  
        FOR each element consisting of x y and time IN edges  
            APPEND a tuple of element y and element time TO element at position x of graph  
            APPEND a tuple of element x and element time TO element at position y of graph  
        END FOR  

        SET pq TO a list containing a tuple of element at position zero of passingFees current city zero and current time zero  
        SET visited_time TO a list of length n where each element is infinity  
        SET element at position zero of visited_time TO zero  

        WHILE pq is not empty  
            REMOVE and RETURN the smallest element FROM pq ASSIGN TO current_cost current_city current_time  

            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  

            FOR each element consisting of neighbor and travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  

                IF new_time IS LESS THAN OR EQUAL TO maxTime AND  
                   new_time IS LESS THAN element at position neighbor of visited_time OR new_cost IS LESS THAN current_cost  
                    SET element at position neighbor of visited_time TO new_time  
                    INSERT the tuple of new_cost neighbor and new_time INTO pq maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
