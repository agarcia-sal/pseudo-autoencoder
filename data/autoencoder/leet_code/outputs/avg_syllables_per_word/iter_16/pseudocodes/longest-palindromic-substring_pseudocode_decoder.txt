CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional list OF boolean values where each element is TRUE for each pair of indices FROM zero TO n MINUS one  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and j of f TO FALSE  
                IF character at position i of s EQUALS character at position j of s  
                    SET element at position i and j of f TO element at position i PLUS one and j MINUS one of f  
                    IF element at position i and j of f IS TRUE AND mx IS LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN substring from position k TO k PLUS mx of s  
    END FUNCTION  
END CLASS
