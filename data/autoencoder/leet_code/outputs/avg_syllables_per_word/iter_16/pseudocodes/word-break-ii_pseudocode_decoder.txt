CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the CONVERSION OF wordDict INTO A SET TO ALLOW CONSTANT TIME MEMBERSHIP TESTS  
        SET memo TO an EMPTY DICTIONARY TO STORE SENTENCES FOR EACH STARTING INDEX  
        
        FUNCTION backtrack WITH PARAMETER start  
            IF start IS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH start IN memo  
            END IF  
            IF start IS EQUAL TO the LENGTH OF s  
                RETURN A LIST CONTAINING AN EMPTY STRING  
            END IF  
            
            SET res TO an EMPTY LIST  
            FOR end FROM start PLUS ONE TO the LENGTH OF s INCLUSIVE  
                SET word TO the SUBSTRING FROM POSITION start TO end MINUS ONE OF s  
                IF word IS AN ELEMENT OF wordSet  
                    FOR each sentence IN backtrack WITH ARGUMENT end  
                        IF sentence IS NOT EMPTY  
                            APPEND THE CONCATENATION OF word, A SINGLE SPACE, AND sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET memo AT KEY start TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack WITH ARGUMENT zero  
    END FUNCTION  
END CLASS
