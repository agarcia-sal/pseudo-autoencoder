CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS check_str reversed  
        END FUNCTION  

        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(substring from position zero TO i minus one of s)  
                SET to_add TO substring from position i TO the end of s reversed  
                RETURN concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
