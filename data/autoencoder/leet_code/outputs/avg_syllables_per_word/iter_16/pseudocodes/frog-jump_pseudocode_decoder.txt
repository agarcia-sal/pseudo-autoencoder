CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  

        SET stone_set TO a set containing all elements of stones  
        SET memo TO an empty mapping from pairs of position and jump TO boolean  

        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  

            IF the pair of position AND jump IS in memo  
                RETURN the value associated with the pair of position AND jump in memo  
            END IF  

            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND the element at position PLUS next_jump IS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump) RETURNS True  
                        SET the value associated with the pair of position AND jump in memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET the value associated with the pair of position AND jump in memo TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_jump_to(the element at position zero of stones, zero)  
    END FUNCTION  
END CLASS
