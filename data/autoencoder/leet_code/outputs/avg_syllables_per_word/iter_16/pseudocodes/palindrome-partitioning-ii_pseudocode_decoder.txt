CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET is_palindrome TO a two dimensional collection of size n BY n WHERE each element IS false  
        
        FOR each i FROM zero TO n MINUS one  
            SET element at position i AT position i of is_palindrome TO true  
        END FOR  
        
        FOR each length FROM two TO n  
            FOR each start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start AT position end of is_palindrome TO the condition that the element at position start of s EQUALS the element at position end of s  
                ELSE  
                    SET element at position start AT position end of is_palindrome TO the condition that the element at position start of s EQUALS the element at position end of s AND the value of element at position start PLUS one AT position end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        SET min_cuts TO a collection of size n WHERE each element IS zero  
        FOR each i FROM zero TO n MINUS one  
            IF the value of element at position zero AT position i of is_palindrome  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity  
                FOR each j FROM one TO i  
                    IF the value of element at position j AT position i of is_palindrome AND the value of element at position j MINUS one of min_cuts PLUS one LESS THAN the value of element at position i of min_cuts  
                        SET element at position i of min_cuts TO the value of element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN the value of element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
