CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN the element at position zero of nums AND twenty four IS LESS THAN zero POINT zero zero zero zero zero one  
            END IF  
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF index i NOT EQUALS index j  
                        SET new_nums TO an EMPTY LIST  
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF index k NOT EQUALS index i AND index k NOT EQUALS index j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        FOR each op IN the collection of operators true division multiplication addition subtraction  
                            IF the operator op IS addition OR the operator op IS multiplication AND index j GREATER THAN index i  
                                CONTINUE to the next iteration of the loop  
                            END IF  
                            IF the operator op IS NOT true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the RESULT OF applying operator op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF the RESULT OF solve CALLED WITH new_nums IS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN false  
        END FUNCTION  
        RETURN solve CALLED WITH cards  
    END FUNCTION  
END CLASS
