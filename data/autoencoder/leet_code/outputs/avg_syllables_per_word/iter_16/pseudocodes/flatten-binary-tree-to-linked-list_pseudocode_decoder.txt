CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the values LIST IS empty  
        RETURN None  
    END IF  
    SET root TO a NEW instance of TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element popped from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node dot left TO a NEW instance of TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node dot right TO a NEW instance of TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p dot left AND q dot left AND the result of is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten WITH PARAMETERS self root  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO a list THAT CONTAINS root AS only element  
        WHILE the stack IS NOT empty  
            SET node TO the element popped from the end of stack  
            IF node dot right IS NOT None  
                APPEND node dot right TO stack  
            END IF  
            IF node dot left IS NOT None  
                APPEND node dot left TO stack  
            END IF  
            SET node dot left TO None  
            IF the stack IS NOT empty  
                SET node dot right TO the element at the last position of stack  
            ELSE  
                SET node dot right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
