CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list of size n plus one BY numCarpets plus one with all elements initialized TO zero  
        
        FOR i FROM one TO n  
            SET dp element at position i zero TO dp element at position i minus one zero PLUS one IF the element at position i minus one of floor EQUALS one ELSE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at position i j TO dp element at position i minus one j PLUS one IF the element at position i minus one of floor EQUALS one ELSE zero  
                SET dp element at position i j TO the MINIMUM OF dp element at position i j AND dp element at position MAXIMUM OF zero AND i minus carpetLen j minus one  
            END FOR  
        END FOR  
        
        RETURN dp element at position n numCarpets  
    END FUNCTION  
END CLASS
