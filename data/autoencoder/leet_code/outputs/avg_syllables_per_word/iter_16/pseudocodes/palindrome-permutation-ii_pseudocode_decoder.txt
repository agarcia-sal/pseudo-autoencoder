CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND value v IN the ITEMS OF cnt  
                IF v GREATER THAN one  
                    DECREMENT the VALUE ASSOCIATED WITH c IN cnt BY two  
                    CALL dfs WITH ARGUMENT character c CONCATENATED WITH t CONCATENATED WITH character c  
                    INCREMENT the VALUE ASSOCIATED WITH c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  
        SET cnt TO A COUNT OF ALL CHARACTERS IN s  
        SET mid TO the EMPTY STRING  
        FOR each character c AND value v IN the ITEMS OF cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT the EMPTY STRING  
                    RETURN the EMPTY LIST  
                END IF  
                SET mid TO character c  
                DECREMENT the VALUE ASSOCIATED WITH c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO the EMPTY LIST  
        CALL dfs WITH ARGUMENT mid  
        RETURN ans  
    END FUNCTION  
END CLASS
