CLASS Solution  
    FUNCTION subarraySum(nums AS list of integers, k AS integer) RETURNS integer  
        SET cumulative_sum_freq TO dictionary with key zero assigned value one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF current_sum MINUS k EXISTS AS key IN cumulative_sum_freq  
                INCREMENT count BY value at key current_sum MINUS k OF cumulative_sum_freq  
            END IF  
            IF current_sum EXISTS AS key IN cumulative_sum_freq  
                INCREMENT value at key current_sum OF cumulative_sum_freq BY one  
            ELSE  
                SET value at key current_sum OF cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
