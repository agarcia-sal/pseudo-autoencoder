CLASS Solution  
    FUNCTION grayCode WITH parameter n  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode WITH n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one positions  
        SET current_gray TO the concatenation of prev_gray WITH a new list constructed by FOR each number IN the reverse of prev_gray CREATE the element as mask OR number  
        RETURN current_gray  
    END FUNCTION  
END CLASS
