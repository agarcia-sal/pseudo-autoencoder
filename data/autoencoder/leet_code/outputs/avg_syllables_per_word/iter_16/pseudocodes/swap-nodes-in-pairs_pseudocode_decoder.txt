CLASS ListNode  
    FUNCTION __init__ WITH PARAMETER val EQUALS zero AND PARAMETER next EQUALS none  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values OF TYPE list  
    IF the input values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO THE END   
        SET node TO A NEW ListNode CREATED WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the condition that the val property of p1 EQUALS the val property of p2 AND the result of calling is_same_list WITH PARAMETERS the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH PARAMETERS self AND head OF TYPE Optional ListNode  
        SET dummy TO A NEW ListNode CREATED WITH zero  
        SET the next property of dummy TO head  
        SET current TO dummy  
        WHILE the next property of current IS NOT none AND the next property of the next property of current IS NOT none  
            SET first TO the next property of current  
            SET second TO the next property of the next property of current  
            SET the next property of current TO second  
            SET the next property of first TO the next property of second  
            SET the next property of second TO first  
            SET current TO the next property of the next property of current  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
