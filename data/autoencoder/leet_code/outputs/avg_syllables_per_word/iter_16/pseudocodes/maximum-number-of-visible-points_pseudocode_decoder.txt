CLASS Solution  
    FUNCTION visiblePoints(points AS List OF List OF Integer comma angle AS Integer comma location AS List OF Integer) RETURNS Integer  
        SET angles TO an empty list  
        SET extra_points TO zero  
          
        FOR each pair x comma y IN points  
            IF x EQUALS element at position zero of location AND y EQUALS element at position one of location  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_rad TO the arctangent of the difference between y and element at position one of location DIVIDED BY the difference between x and element at position zero of location  
            SET angle_deg TO convert angle_rad FROM radians TO degrees  
            APPEND angle_deg TO angles  
        END FOR  
          
        SORT angles in ascending order  
        SET angles TO the concatenation of angles WITH each element IN angles INCREMENTED BY three hundred sixty  
          
        SET max_visible TO zero  
        SET left TO zero  
          
        FOR each right FROM zero TO the LENGTH OF angles MINUS one  
            WHILE the difference between element at position right of angles AND element at position left of angles IS GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET max_visible TO the maximum value BETWEEN max_visible AND right MINUS left PLUS one  
        END FOR  
          
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
