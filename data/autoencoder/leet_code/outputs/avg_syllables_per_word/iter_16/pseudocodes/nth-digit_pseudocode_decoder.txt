CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n IS GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        INCREMENT start BY the INTEGER DIVISION OF n MINUS one BY length  
        SET digit_string TO the STRING REPRESENTATION OF start  
        SET digit_index TO the REMAINDER OF n MINUS one DIVIDED BY length  
        SET result_digit TO the CHARACTER AT position digit_index OF digit_string CONVERTED TO INTEGER  
        RETURN result_digit  
    END FUNCTION  
END CLASS
