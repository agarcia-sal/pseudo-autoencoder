CLASS Solution  
    FUNCTION slidingPuzzle WITH board  
        SET start TO the tuple of tuples each constructed from each row of board  
        SET target TO the tuple consisting of the tuples one two three and four five zero  
        SET neighbors TO a mapping where zero maps to the set containing one and three  
            one maps to the set containing zero two and four  
            two maps to the set containing one and five  
            three maps to the set containing zero and four  
            four maps to the set containing one three and five  
            five maps to the set containing two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF the element at position j of the element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a double-ended queue initialized with a tuple of start zero_pos zero  
        SET visited TO a set containing start  
        WHILE queue is not empty  
            REMOVE the element from the left of queue and unpack into state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set at key zero_pos in neighbors  
                SET new_state TO a list of lists constructed by converting each tuple row of state into a list  
                SET i TO the integer division of zero_pos BY three  
                SET j TO the remainder of zero_pos DIVIDED BY three  
                SET ni TO the integer division of neighbor BY three  
                SET nj TO the remainder of neighbor DIVIDED BY three  
                SET temp TO the element at position j of the element at position i of new_state  
                SET the element at position j of the element at position i of new_state TO the element at position nj of the element at position ni of new_state  
                SET the element at position nj of the element at position ni of new_state TO temp  
                SET new_tuple_state TO the tuple of tuples each constructed from each row of new_state  
                IF new_tuple_state is not in visited  
                    ADD new_tuple_state to visited  
                    APPEND the tuple consisting of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
