CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        DEFINE dp AS FUNCTION OF target WITH LRU CACHE  
            IF target EQUALS zero  
                RETURN minus one  
            END IF  
            IF target LESS THAN x  
                RETURN the MINIMUM BETWEEN two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY x MINUS target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power LESS THAN OR EQUAL TO target HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp WITH parameter target MINUS x RAISED TO THE power PLUS power  
            IF x RAISED TO THE power PLUS one MINUS target LESS THAN target  
                SET res TO the MINIMUM BETWEEN res AND dp WITH parameter x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH parameter target  
    END FUNCTION  
END CLASS
