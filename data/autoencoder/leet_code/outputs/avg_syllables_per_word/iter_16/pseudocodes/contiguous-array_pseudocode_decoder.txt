CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with element zero key SET TO value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS AS a key IN prefix_sum_index  
                SET current_length TO index i MINUS the element at key count of prefix_sum_index  
                SET max_length TO the GREATER VALUE BETWEEN max_length AND current_length  
            ELSE  
                SET the element at key count of prefix_sum_index TO index i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
