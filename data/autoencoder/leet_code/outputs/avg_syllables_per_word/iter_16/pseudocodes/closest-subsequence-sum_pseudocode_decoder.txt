CLASS Solution  
    FUNCTION minAbsDifference(nums AS List OF integers, goal AS integer) RETURNS integer  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums FROM first position TO position mid minus one  
            SET new_sums TO an empty list  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND left_sums BY new_sums  
        END FOR  
        
        SET right_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums FROM position mid TO the last position  
            SET new_sums TO an empty list  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND right_sums BY new_sums  
        END FOR  
        
        SORT right_sums IN ASCENDING ORDER  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the POSITION FOUND BY binary search of target IN right_sums using left bound  
            
            IF pos IS LESS THAN the LENGTH OF right_sums  
                SET candidate_diff TO THE ABSOLUTE VALUE OF left_sum PLUS THE ELEMENT at position pos IN right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            
            IF pos IS GREATER THAN zero  
                SET candidate_diff TO THE ABSOLUTE VALUE OF left_sum PLUS THE ELEMENT at position pos MINUS one IN right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
