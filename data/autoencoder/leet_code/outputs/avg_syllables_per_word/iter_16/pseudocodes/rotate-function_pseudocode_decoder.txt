CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET total_sum TO the SUM OF all elements IN nums  
        SET current_sum TO the SUM OF each index MULTIPLIED BY element at position index OF nums FOR each index and element IN nums  
        SET max_value TO current_sum  
        FOR k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY element at position n MINUS k OF nums  
            SET max_value TO THE GREATER OF max_value AND current_sum  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
