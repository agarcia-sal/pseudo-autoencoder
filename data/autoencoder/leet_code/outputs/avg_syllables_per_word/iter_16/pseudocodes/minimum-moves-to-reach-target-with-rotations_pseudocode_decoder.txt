CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a new empty double ended queue initialized with a tuple containing zero zero zero one zero representing the starting tail row tail column head row head column and step count  
        SET visited TO a new empty set initialized with a tuple containing zero zero zero one representing the visited states  

        WHILE queue is not empty  
            REMOVE the element from the LEFT END of queue and ASSIGN the values to r1 c1 r2 c2 steps  
            
            IF the tuple r1 c1 r2 c2 EQUALS the tuple n MINUS one n MINUS two n MINUS one n MINUS one  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple r1 c1 PLUS one r2 c2 PLUS one is NOT IN visited  
                ADD the tuple r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r2 PLUS one and position c2 of grid EQUALS zero AND the tuple r1 PLUS one c1 r2 PLUS one c2 is NOT IN visited  
                ADD the tuple r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r1 PLUS one and position c2 of grid EQUALS zero AND the tuple r1 c1 r1 PLUS one c1 is NOT IN visited  
                ADD the tuple r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple r1 c1 r1 c1 PLUS one is NOT IN visited  
                ADD the tuple r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
