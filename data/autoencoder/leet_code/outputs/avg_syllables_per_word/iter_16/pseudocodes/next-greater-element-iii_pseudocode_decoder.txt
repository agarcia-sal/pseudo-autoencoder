CLASS Solution  
    FUNCTION nextGreaterElement WITH parameter n AS integer  
        SET digits TO the list of characters representing the decimal representation of n  
        SET length TO the number of elements in digits  
        
        SET i TO length MINUS two  
        WHILE i greater than or equal to zero AND the element at position i of digits GREATER THAN OR EQUAL TO the element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        
        SET j TO length MINUS one  
        WHILE the element at position j of digits LESS THAN OR EQUAL TO the element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        
        SET temporary TO the element at position i of digits  
        SET the element at position i of digits TO the element at position j of digits  
        SET the element at position j of digits TO temporary  
        
        SET the part of digits from the position zero TO i inclusive TO itself concatenated with the reverse of the part of digits from position i PLUS one TO the last position  
        
        SET result TO the integer value obtained by concatenating all elements of digits in order  
        
        IF result GREATER THAN the number two RAISED TO THE power of thirty ONE MINUS one  
            RETURN negative one  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
