CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one point zero  
        END IF  
        
        SET n TO the result of integer division of n PLUS twenty four BY twenty five  
        
        SET memo TO an empty mapping from pair of integers TO floating point number  
        
        FUNCTION dp(a, b)  
            IF the pair of a AND b EXISTS AS a key IN memo  
                RETURN the value corresponding TO the pair of a AND b IN memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            
            SET probability TO zero point two five MULTIPLIED BY the sum of dp called with a MINUS four AND b PLUS dp called with a MINUS three AND b MINUS one PLUS dp called with a MINUS two AND b MINUS two PLUS dp called with a MINUS one AND b MINUS three  
            
            ASSIGN the key of the pair of a AND b IN memo TO probability  
            
            RETURN probability  
        END FUNCTION  
        
        RETURN dp called with n AND n  
    END FUNCTION  
END CLASS
