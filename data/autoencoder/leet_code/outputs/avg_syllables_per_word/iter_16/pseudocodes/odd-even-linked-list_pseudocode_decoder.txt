CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO A new ListNode INITIALIZED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the SUBSEQUENCE of values FROM position two TO the end  
        SET node TO A new ListNode INITIALIZED WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN TRUE  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN FALSE  
    END IF  
    RETURN the VALUE of p1 EQUALS the VALUE of p2 AND the RESULT OF is_same_list WITH PARAMETERS the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self head OPTIONAL ListNode  
        IF head IS none OR the next element of head IS none OR the next element of the next element of head IS none  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next element of head  
        SET even_head TO even  
        WHILE even IS NOT none AND the next element of even IS NOT none  
            SET the next element of odd TO the next element of even  
            SET odd TO the next element of odd  
            SET the next element of even TO the next element of odd  
            SET even TO the next element of even  
        END WHILE  
        SET the next element of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
