CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty set  
        FOR each length FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR each start FROM zero TO n MINUS two TIMES length INCLUSIVE  
                SET a TO substring from position start TO start PLUS length MINUS one of text  
                SET b TO substring from position start PLUS length TO start PLUS two TIMES length MINUS one of text  
                IF a EQUALS b  
                    ADD the concatenation of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF substrings  
    END FUNCTION  
END CLASS
