CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO an empty set  
        SET stack TO a list containing zero  
        WHILE the stack is not empty  
            SET current_room TO the element removed from the end of the stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room of rooms  
                IF key is not an element of visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the number of elements in visited IS EQUAL TO the number of elements in rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
