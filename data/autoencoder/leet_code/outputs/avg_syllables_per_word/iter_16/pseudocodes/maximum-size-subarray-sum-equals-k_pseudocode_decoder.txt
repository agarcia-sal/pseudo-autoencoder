CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative sum index TO a dictionary mapping zero TO negative one  
        SET cumulative sum TO zero  
        SET max length TO zero  
        FOR each index and number IN the enumeration of nums  
            SET cumulative sum TO cumulative sum PLUS number  
            IF cumulative sum MINUS k EXISTS AS A KEY IN cumulative sum index  
                SET difference index TO the element AT cumulative sum MINUS k IN cumulative sum index  
                SET max length TO the GREATER VALUE BETWEEN max length AND index MINUS difference index  
            END IF  
            IF cumulative sum DOES NOT EXIST AS A KEY IN cumulative sum index  
                SET the element AT cumulative sum IN cumulative sum index TO index  
            END IF  
        END FOR  
        RETURN max length  
    END FUNCTION  
END CLASS
