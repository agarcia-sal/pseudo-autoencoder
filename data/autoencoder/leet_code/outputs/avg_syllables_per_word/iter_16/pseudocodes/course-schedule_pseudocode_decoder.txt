CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        ASSIGN pointer TO a new mapping from keys to empty lists called graph  
        SET indegree TO a list of zeros with the LENGTH EQUAL TO numCourses  
        
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO the list at key prereq of graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        
        ASSIGN pointer TO a new double-ended queue containing all courses from zero TO numCourses MINUS one where the element at position course of indegree EQUALS zero called queue  
        
        WHILE the LENGTH OF queue is GREATER THAN zero  
            REMOVE and RETURN the element from the left end of queue called current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course of graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
