CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board IS empty OR the element at position zero of board IS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero of board  
        
        SET directions TO the COLLECTION of pairs consisting of minus one AND minus one a pair consisting of minus one AND zero a pair consisting of minus one AND one a pair consisting of zero AND minus one a pair consisting of zero AND one a pair consisting of one AND minus one a pair consisting of one AND zero a pair consisting of one AND one  
        
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx AND dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND (the element at position ni of board at position nj EQUALS one OR the element at position ni of board at position nj EQUALS three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF the element at position i of board at position j EQUALS one  
                    IF live_neighbors IS LESS THAN two OR live_neighbors IS GREATER THAN three  
                        SET the element at position i of board at position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three  
                        SET the element at position i of board at position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF the element at position i of board at position j EQUALS two  
                    SET the element at position i of board at position j TO one  
                ELSE IF the element at position i of board at position j EQUALS three  
                    SET the element at position i of board at position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
