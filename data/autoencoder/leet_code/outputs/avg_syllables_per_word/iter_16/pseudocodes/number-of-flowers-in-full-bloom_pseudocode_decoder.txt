CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the SORTED list of the first element of each sublist in flowers  
        SET end_times TO the SORTED list of the second element of each sublist in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the position to insert person to the right in start_times  
            SET ended TO the position to insert person to the left in end_times  
            APPEND the value of started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
