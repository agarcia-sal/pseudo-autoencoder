CLASS Solution  
    FUNCTION fullJustify(words AS List of strings, maxWidth AS integer) RETURNS List of strings  
        FUNCTION justify_line(line AS List of strings, num_of_letters AS integer, is_last_line AS boolean) RETURNS string  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN the concatenation of all words in line separated by single space FOLLOWED BY SPACES repeated to fill maxWidth minus num_of_letters minus the LENGTH OF line plus one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO the REMAINDER of total_spaces DIVIDED BY num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND spaces_between_words PLUS one number of SPACES TO the element at position index of line  
                    ELSE  
                        APPEND spaces_between_words number of SPACES TO the element at position index of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements of line without separator  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of justify_line with line num_of_letters and false TO result  
                SET line TO empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result of justify_line with line num_of_letters and true TO result  
        RETURN result  
    END FUNCTION  
END CLASS
