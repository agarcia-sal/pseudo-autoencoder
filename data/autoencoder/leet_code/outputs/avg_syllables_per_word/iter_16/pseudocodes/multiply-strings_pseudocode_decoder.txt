CLASS Solution  
    FUNCTION multiply(num1 STRING, num2 STRING) RETURNS STRING  
        IF variable num1 EQUALS string zero OR variable num2 EQUALS string zero  
            RETURN string zero  
        END IF  
        
        SET variable result TO list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
        
        SET variable num1 TO the REVERSED sequence OF characters IN num1  
        SET variable num2 TO the REVERSED sequence OF characters IN num2  
        
        FOR each index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR each index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET variable mul TO the product OF the numeric value OF character at position i OF num1 MINUS the numeric value OF character zero AND the numeric value OF character at position j OF num2 MINUS the numeric value OF character zero  
                SET variable p1 TO i PLUS j  
                SET variable p2 TO i PLUS j PLUS one  
                SET variable sum_ TO the addition OF mul AND element at position p1 OF result  
                
                SET element at position p1 OF result TO the remainder WHEN sum_ IS DIVIDED BY ten  
                INCREMENT element at position p2 OF result BY the integer division OF sum_ BY ten  
            END FOR  
        END FOR  
        
        SET variable result_str TO the concatenation of the string representations OF elements in result IN REVERSED order  
        RETURN result_str AFTER REMOVING all leading string zero characters  
    END FUNCTION  
END CLASS
