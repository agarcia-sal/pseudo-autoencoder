CLASS Solution  
    FUNCTION minMoves(nums LIST OF INTEGER, k INTEGER) RETURNS INTEGER  
        SET positions TO an empty list  
        FOR each index AND element IN nums  
            IF element EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start INTEGER, end INTEGER) RETURNS INTEGER  
            SET mid TO the integer division of start PLUS end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR each index FROM start TO end MINUS one  
                SET distance TO the absolute value of element at position index of positions MINUS median PLUS (mid MINUS index)  
                INCREMENT cost BY distance  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR each index FROM zero TO the length of positions MINUS k  
            SET current_cost TO the result of calculate_cost with index AND index PLUS k  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
