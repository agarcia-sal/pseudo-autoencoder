CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF i IS GREATER THAN farthest  
                RETURN False  
            END IF  
            SET farthest TO the GREATER OF farthest AND the SUM OF i AND the element at position i of nums  
        END FOR  
        IF farthest IS GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
            RETURN True  
        ELSE  
            RETURN False  
        END IF  
    END FUNCTION  
END CLASS
