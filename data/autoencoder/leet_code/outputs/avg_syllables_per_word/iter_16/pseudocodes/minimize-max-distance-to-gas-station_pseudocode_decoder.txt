CLASS Solution  
    FUNCTION minmaxGasDist WITH PARAMETERS stations AND k  
        FUNCTION possible WITH PARAMETER max_dist  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO the element at position index of stations MINUS the element at position index MINUS one of stations  
                SET additional_needed TO the INTEGER PART OF distance_difference DIVIDED BY max_dist  
                INCREMENT additional_stations BY additional_needed  
            END FOR  
            IF additional_stations IS LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE the difference BETWEEN right AND left IS GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO the SUM OF left AND right DIVIDED BY two  
            IF possible WITH mid RETURNS true THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
