CLASS Solution  
    FUNCTION minArea(image AS List OF List OF string, x AS integer, y AS integer) RETURNS integer  
        IF the image IS EMPTY OR the element at position zero of image IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  
        
        FUNCTION hasBlackPixelInRow(row AS integer) RETURNS boolean  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS the character representing one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col AS integer) RETURNS boolean  
            FOR index_row FROM zero TO m MINUS one  
                IF the element at position col of element at position index_row of image EQUALS the character representing one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop() RETURNS integer  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the sum of low AND high BY two  
                IF hasBlackPixelInRow(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom() RETURNS integer  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the sum of low, high AND one BY two  
                IF hasBlackPixelInRow(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft() RETURNS integer  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the sum of low AND high BY two  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight() RETURNS integer  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the sum of low, high AND one BY two  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  
        
        RETURN the product of the difference of bottom MINUS top PLUS one AND the difference of right MINUS left PLUS one  
    END FUNCTION  
END CLASS
