CLASS Solution  
    FUNCTION minSteps(n)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET operations TO zero  
        SET factor TO two  
        WHILE n IS GREATER THAN one  
            WHILE the REMAINDER of n DIVIDED BY factor EQUALS zero  
                INCREMENT operations BY factor  
                SET n TO the INTEGER PART of n DIVIDED BY factor  
            END WHILE  
            INCREMENT factor BY one  
        END WHILE  
        RETURN operations  
    END FUNCTION  
END CLASS
