CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF all elements IN nums  
        IF n EQUALS one  
            RETURN False  
        END IF  
        SET dp TO a list of empty sets FOR each integer FROM zero TO n DIVIDED BY two INCLUSIVE  
        APPEND zero TO the set at position zero of dp  
        FOR each num IN nums  
            FOR size FROM n DIVIDED BY two DOWN TO one  
                FOR each s IN the set at position size MINUS one of dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    APPEND new_sum TO the set at position size of dp  
                END FOR  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
