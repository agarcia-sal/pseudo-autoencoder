CLASS Solution  
    FUNCTION numBusesToDestination(routes, source, target)  
        IF source EQUALS target  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a mapping from a stop TO list of buses that visit that stop  
        FOR each bus and route combination enumerated FROM zero IN routes  
            FOR each stop IN route  
                APPEND bus TO element at position stop of stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO a double ended queue for processing buses  
        SET visited_buses TO an empty set of buses already visited  
        SET visited_stops TO an empty set of stops already visited  

        FOR each bus IN element at position source of stop_to_buses  
            APPEND tuple of bus and one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE and ASSIGN first element of bus_queue TO current_bus and bus_count  

            FOR each stop IN element at position current_bus of routes  
                IF stop EQUALS target  
                    RETURN bus_count  
                END IF  

                IF stop NOT IN visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN element at position stop of stop_to_buses  
                        IF next_bus NOT IN visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND tuple of next_bus and bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
