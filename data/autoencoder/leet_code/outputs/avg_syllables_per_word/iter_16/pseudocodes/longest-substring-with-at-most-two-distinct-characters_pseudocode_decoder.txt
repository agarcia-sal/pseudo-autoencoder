CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        SET char_count TO a new dictionary that returns zero for missing keys  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of the element at position right of string s IN char_count BY one  
            WHILE the NUMBER OF KEYS IN char_count IS GREATER THAN two  
                DECREMENT the count of the element at position left of string s IN char_count BY one  
                IF the count of the element at position left of string s IN char_count EQUALS zero  
                    REMOVE the element at position left of string s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            SET max_length TO the GREATER VALUE BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
