CLASS Solution  
    FUNCTION wordSquares(words)  
        CREATE prefix_to_words AS A DEFAULT DICTIONARY THAT RETURNS AN EMPTY LIST FOR NEW KEYS  
        FOR each word IN words  
            FOR index FROM zero TO THE LENGTH OF word MINUS ONE  
                SET prefix TO SUBSTRING FROM POSITION zero TO index MINUS ONE OF word  
                APPEND word TO THE LIST ASSOCIATED WITH prefix IN prefix_to_words  
            END FOR  
        END FOR  
        
        SET results TO AN EMPTY LIST  
        
        FUNCTION backtrack(square)  
            IF THE LENGTH OF square EQUALS THE LENGTH OF THE ELEMENT AT POSITION zero OF words  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO THE CONCATENATION OF THE ELEMENT AT THE POSITION EQUAL TO THE LENGTH OF square OF EACH word IN square  
            FOR each candidate IN THE LIST ASSOCIATED WITH prefix IN prefix_to_words  
                CALL backtrack WITH THE ARGUMENT AS square WITH candidate APPENDED  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack WITH A LIST CONTAINING word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
