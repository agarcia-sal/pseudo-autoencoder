CLASS Solution  
    FUNCTION asteroidCollision WITH PARAMETERS asteroids  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND element at position LENGTH OF stack MINUS one of stack IS GREATER THAN zero  
                IF element at position LENGTH OF stack MINUS one of stack IS LESS THAN the NEGATION OF asteroid  
                    REMOVE element at position LENGTH OF stack MINUS one FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF element at position LENGTH OF stack MINUS one of stack EQUALS the NEGATION OF asteroid  
                    REMOVE element at position LENGTH OF stack MINUS one FROM stack  
                END IF  
                BREAK from the WHILE loop  
            END WHILE  
            ELSE  
                APPEND asteroid TO stack  
            END ELSE  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
