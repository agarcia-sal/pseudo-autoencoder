CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR each row FROM the LENGTH OF triangle MINUS two DOWN TO zero  
            FOR each col FROM zero TO the LENGTH OF element at position row of triangle MINUS one  
                SET element at position col of element at position row of triangle TO element at position col of element at position row of triangle PLUS the MINIMUM OF element at position col of element at position row PLUS one of triangle AND element at position col PLUS one of element at position row PLUS one of triangle  
            END FOR  
        END FOR  
        RETURN element at position zero of element at position zero of triangle  
    END FUNCTION  
END CLASS
