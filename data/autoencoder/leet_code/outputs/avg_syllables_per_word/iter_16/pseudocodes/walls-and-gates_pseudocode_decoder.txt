    FUNCTION wallsAndGates(rooms)  
        IF rooms IS EMPTY OR element at position zero of rooms IS EMPTY  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero of rooms  
        ASSIGN pointer TO a new double ended queue named queue  
        FOR each r FROM zero TO rows MINUS one  
            FOR each c FROM zero TO cols MINUS one  
                IF element at position r of rooms AND element at position c of element at position r of rooms EQUALS zero  
                    APPEND pair consisting of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO the list containing pair zero AND one, pair one AND zero, pair zero AND negative one, pair negative one AND zero  
        WHILE queue IS NOT EMPTY  
            REMOVE and RETURN the element FROM the front of queue AND ASSIGN IT TO r AND c  
            FOR each dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr of rooms AND element at position nc of element at position nr of rooms EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven  
                    SET element at position nr of rooms AND element at position nc of element at position nr of rooms TO element at position r of rooms AND element at position c of element at position r of rooms PLUS one  
                    APPEND pair consisting of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
