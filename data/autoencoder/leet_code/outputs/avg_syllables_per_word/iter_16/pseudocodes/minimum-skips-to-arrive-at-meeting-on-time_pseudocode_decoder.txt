CLASS Solution  
    FUNCTION minSkips WITH PARAMETERS dist AS list of integers speed AS integer hoursBefore AS integer  
        SET n TO the LENGTH OF dist  
        SET dp TO a list of lists each containing n PLUS one copies of infinity for each of n times  

        SET total_dist TO the SUM OF all elements IN dist  

        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  

        SET element at position zero of list at position zero of dp TO element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero of list at position j of dp TO element at position zero of dist  
        END FOR  

        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET dp element at position i list at position j TO the CEILING OF the element at position i MINUS one list at position j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET dp element at position i list at position j TO the MINIMUM OF its current value AND the element at position i MINUS one list at position j MINUS one of dp PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  

        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one list at position j of dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  

        RETURN negative one  
    END FUNCTION  
END CLASS
