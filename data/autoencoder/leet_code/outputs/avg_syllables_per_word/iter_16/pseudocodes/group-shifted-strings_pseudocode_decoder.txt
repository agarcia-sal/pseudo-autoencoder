CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS an empty string  
                RETURN an empty string  
            END IF  
            SET canonical_form TO an empty list  
            FOR each character c IN s  
                SET difference TO the numeric position of character c MINUS the numeric position of the character at position one of s  
                SET difference TO the remainder of difference DIVIDED BY twenty six  
                APPEND the string representation of difference TO canonical_form  
            END FOR  
            SET result TO the concatenation of elements in canonical_form SEPARATED BY a comma  
            RETURN result  
        END FUNCTION  
      
        SET groups TO a new mapping that returns an empty list for nonexistent keys  
      
        FOR each string s IN strings  
            SET canonical_form TO the result of calling get_canonical_form WITH s  
            APPEND s TO the list at key canonical_form in groups  
        END FOR  
      
        RETURN the list of all values contained in groups  
    END FUNCTION  
END CLASS
