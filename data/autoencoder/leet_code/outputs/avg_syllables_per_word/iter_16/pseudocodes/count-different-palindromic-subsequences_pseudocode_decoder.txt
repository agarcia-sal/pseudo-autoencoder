CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET next_occurrence TO a two dimensional list with four rows and n columns FILLED WITH negative one  
        SET prev_occurrence TO a two dimensional list with four rows and n columns FILLED WITH negative one  
        
        FOR each c FROM zero TO three  
            SET char TO the character at position ordinal value of lowercase a PLUS c  
            SET last TO negative one  
            FOR each i FROM zero TO n MINUS one  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position c of row i of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR each i FROM n MINUS one DOWN TO zero  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position c of row i of next_occurrence TO last  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list with n rows and n columns FILLED WITH zero  
        
        FOR each i FROM zero TO n MINUS one  
            SET element at position i of row i of dp TO one  
        END FOR  
        
        FOR each length FROM two TO n  
            FOR each i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                FOR each c FROM zero TO three  
                    SET char TO the character at position ordinal value of lowercase a PLUS c  
                    SET left TO element at position c of row i of next_occurrence  
                    SET right TO element at position c of row j of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i THEN  
                        CONTINUE TO next iteration  
                    ELSE IF left EQUALS right THEN  
                        SET element at position i of row j of dp TO element at position i of row j of dp PLUS one MODULO MOD  
                    ELSE  
                        SET element at position i of row j of dp TO element at position i of row j of dp PLUS element at position left PLUS one of row right MINUS one of dp PLUS two MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero of row n MINUS one of dp  
    END FUNCTION  
END CLASS
