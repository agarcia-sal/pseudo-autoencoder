CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of infinite values with length fifteen  
        FOR each pair consisting of first_factor AND rate IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO first_factor MULTIPLIED BY rate RAISED TO THE POWER OF index  
                IF lap_time IS GREATER THAN changeTime PLUS first_factor  
                    EXIT THE LOOP  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position index TO the smaller value of min_times at position index AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of infinite values with length numLaps PLUS one  
        SET dp at position zero TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value of i MINUS one AND fourteen  
                SET dp at position i TO the smaller value of dp at position i AND the sum of dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
