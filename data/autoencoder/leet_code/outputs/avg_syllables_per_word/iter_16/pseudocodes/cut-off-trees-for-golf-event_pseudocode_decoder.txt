CLASS Solution  
    FUNCTION cutOffTree(forest)  
        IF forest EQUALS empty OR element at position zero of forest EQUALS empty  
            RETURN negative one  
        END IF  

        SET trees TO a list containing tuples of the element at position i of forest at position j of forest, i, j FOR each i FROM zero TO the LENGTH OF forest MINUS one FOR each j FROM zero TO the LENGTH OF element at position zero of forest MINUS one WHERE the element at position i of forest at position j of forest IS GREATER THAN one  
        SORT trees IN ASCENDING ORDER  

        FUNCTION bfs(start, end)  
            IF start EQUALS end  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF forest  
            SET n TO the LENGTH OF element at position zero of forest  
            ASSIGN queue TO a deque initialized with start  
            ASSIGN visited TO a set initialized with start  
            SET steps TO zero  

            WHILE queue IS NOT EMPTY  
                FOR each iteration FROM zero TO the LENGTH OF queue MINUS one  
                    REMOVE element from the front of queue ASSIGN TO x, y  
                    FOR each dx, dy IN the list containing negative one and zero, one and zero, zero and negative one, zero and one  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND tuple of nx, ny IS NOT IN visited AND the element at position nx of forest at position ny of forest IS NOT EQUAL TO zero  
                            IF the tuple of nx, ny EQUALS end  
                                RETURN steps PLUS one  
                            END IF  
                            ADD the tuple of nx, ny TO visited  
                            APPEND the tuple of nx, ny TO queue  
                        END IF  
                    END FOR  
                END FOR  
                INCREMENT steps BY one  
            END WHILE  

            RETURN negative one  
        END FUNCTION  

        SET x TO zero  
        SET y TO zero  
        SET total_steps TO zero  

        FOR each height, tx, ty IN trees  
            SET steps TO the result of bfs called with the tuple of x, y AND the tuple of tx, ty  
            IF steps EQUALS negative one  
                RETURN negative one  
            END IF  
            INCREMENT total_steps BY steps  
            SET x TO tx  
            SET y TO ty  
        END FOR  

        RETURN total_steps  
    END FUNCTION  
END CLASS
