CLASS Solution  
    FUNCTION findCircleNum WITH PARAMETER isConnected  
        FUNCTION dfs WITH PARAMETER city  
            FOR each neighbor FROM zero TO n MINUS one  
                IF element at position city of isConnected at position neighbor EQUALS one AND element at position neighbor of visited EQUALS false  
                    SET element at position neighbor of visited TO true  
                    CALL dfs WITH neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO the LENGTH OF isConnected  
        SET visited TO a list of boolean false values with LENGTH n  
        SET province_count TO zero  

        FOR each city FROM zero TO n MINUS one  
            IF element at position city of visited EQUALS false  
                CALL dfs WITH city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  

        RETURN province_count  
    END FUNCTION  
END CLASS
