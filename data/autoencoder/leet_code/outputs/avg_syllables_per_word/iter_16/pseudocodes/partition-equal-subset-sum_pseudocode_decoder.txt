CLASS Solution  
    FUNCTION canPartition WITH parameter nums  
        SET total_sum TO the SUM OF all elements IN nums  
        IF the REMAINDER WHEN total_sum IS DIVIDED BY two IS NOT EQUALS TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        INITIALIZE dp AS a LIST OF False VALUES WITH LENGTH target PLUS one  
        SET element at position zero of dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET element at position i of dp TO the LOGICAL OR OF element at position i of dp AND element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
