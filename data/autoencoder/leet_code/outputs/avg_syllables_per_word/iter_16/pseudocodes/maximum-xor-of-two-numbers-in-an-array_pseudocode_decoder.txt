CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR index FROM thirty-one TO zero  
            DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO an empty set  
            FOR each number IN nums  
                SET prefix TO the integer number shifted right by index positions  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the bitwise exclusive OR of current_candidate AND prefix IS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK the loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
