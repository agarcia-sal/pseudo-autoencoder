CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left side of queue  
        IF i IS LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments element left of p and element left of q AND the result of is_same_tree with arguments element right of p and element right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_length AND left_value TO the result of dfs with argument element left of node  
            SET right_length AND right_value TO the result of dfs with argument element right of node  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF element left of node IS NOT none AND value of element left of node IS EQUAL TO value of node  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF element right of node IS NOT none AND value of element right of node IS EQUAL TO value of node  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the maximum of self.max_length AND left_arrow PLUS right_arrow  
            RETURN the maximum of left_arrow AND right_arrow AND value of node  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with argument root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
