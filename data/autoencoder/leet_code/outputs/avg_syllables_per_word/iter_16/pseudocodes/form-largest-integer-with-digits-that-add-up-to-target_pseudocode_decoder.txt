CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS cost AND target  
        SET cost_to_digit TO create an empty dictionary  
        FOR each index AND c IN the enumeration of cost  
            ASSIGN to cost_to_digit at key c the string representation of index plus one  
        END FOR  
        
        SET dp TO a list starting with an empty string followed by target number of elements each set to a string representing negative one  
        
        FOR each t FROM one TO target  
            FOR each c AND d IN the items of cost_to_digit  
                IF t is greater than or equal to c AND the element at position t minus c of dp is not equal to the string representing negative one  
                    SET candidate TO concatenation of d and the element at position t minus c of dp  
                    IF the element at position t of dp is equal to the string representing negative one OR the length of candidate is greater than the length of the element at position t of dp OR the length of candidate is equal to the length of the element at position t of dp AND candidate is lexicographically greater than the element at position t of dp  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF the element at position target of dp is not equal to the string representing negative one  
            RETURN the element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
