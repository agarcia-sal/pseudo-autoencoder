CLASS Solution  
    FUNCTION topKFrequent(nums AS List OF Integer, k AS Integer) RETURNS List OF Integer  
        SET frequency_map TO a mapping from elements TO zero integer values by default  
        FOR each num IN nums  
            SET frequency_map element at position num TO frequency_map element at position num PLUS one  
        END FOR  
        SET max_frequency TO the maximum value AMONG all values in the frequency_map  
        SET buckets TO a list containing empty lists for each integer FROM zero TO max_frequency INCLUSIVE  
        FOR each num AND freq IN the items of frequency_map  
            APPEND num TO the element at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length OF element at position freq of buckets GREATER THAN zero  
                EXTEND result BY all elements in element at position freq of buckets  
                IF the length OF result GREATER THAN OR EQUAL TO k  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
