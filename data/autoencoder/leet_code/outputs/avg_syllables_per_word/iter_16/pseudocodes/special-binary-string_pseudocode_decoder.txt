CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each position and character IN s  
                IF character EQUALS character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND concatenation of character one, CALL dfs WITH substring from position start PLUS one TO position, and character zero TO specials  
                    SET start TO position PLUS one  
                END IF  
            END FOR  
            SORT specials IN descending order by lexicographical comparison  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        RETURN CALL dfs WITH s  
    END FUNCTION  
END CLASS
