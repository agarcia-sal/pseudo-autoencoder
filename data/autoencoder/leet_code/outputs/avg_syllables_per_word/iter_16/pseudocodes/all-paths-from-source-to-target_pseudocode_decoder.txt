CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a new empty double ended queue initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE the queue q is not empty  
            REMOVE the element at the front of the queue q and ASSIGN it TO path  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each element v IN the element at position u of graph  
                APPEND a new path consisting of the current path FOLLOWED BY v TO the queue q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
