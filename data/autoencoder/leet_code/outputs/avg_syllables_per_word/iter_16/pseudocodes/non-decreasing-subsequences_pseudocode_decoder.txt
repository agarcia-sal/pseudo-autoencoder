CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a copy of t TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND the element at position u of nums TO t  
                CALL dfs WITH the integer u PLUS one, the element at position u of nums, and t  
                REMOVE the last element FROM t  
            END IF  
            IF the element at position u of nums NOT EQUALS last  
                CALL dfs WITH the integer u PLUS one, last, and t  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH zero, negative one thousand, and an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
