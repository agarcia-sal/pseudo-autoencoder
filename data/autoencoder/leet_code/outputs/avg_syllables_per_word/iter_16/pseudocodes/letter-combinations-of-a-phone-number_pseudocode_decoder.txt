CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        SET phone_map TO mapping FROM digit characters TO corresponding letter strings as follows  
            digit character two MAPS TO letters a b c  
            digit character three MAPS TO letters d e f  
            digit character four MAPS TO letters g h i  
            digit character five MAPS TO letters j k l  
            digit character six MAPS TO letters m n o  
            digit character seven MAPS TO letters p q r s  
            digit character eight MAPS TO letters t u v  
            digit character nine MAPS TO letters w x y z  
        FUNCTION backtrack(combination, next_digits)  
            IF length of next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the letter string that corresponds to the element at position zero of next_digits in phone_map  
                    CALL backtrack WITH concatenation of combination AND letter, and substring of next_digits starting from position one TO its end  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO empty list  
        CALL backtrack WITH empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
