CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i and j of grid IS NOT EQUAL TO five  
                RETURN False  
            END IF  
            
            SET numbers TO an empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x and y of grid IS LESS THAN one OR IS GREATER THAN nine OR IS IN numbers  
                        RETURN False  
                    END IF  
                    ADD element at position x and y of grid TO numbers  
                END FOR  
            END FOR  
            
            FOR start FROM i MINUS one TO i PLUS one  
                IF the sum of element at position start and j MINUS one of grid PLUS element at position start and j of grid PLUS element at position start and j PLUS one of grid IS NOT EQUAL TO fifteen OR the sum of element at position i MINUS one and start of grid PLUS element at position i and start of grid PLUS element at position i PLUS one and start of grid IS NOT EQUAL TO fifteen  
                    RETURN False  
                END IF  
            END FOR  
            
            IF the sum of element at position i MINUS one and j MINUS one of grid PLUS element at position i and j of grid PLUS element at position i PLUS one and j PLUS one of grid IS NOT EQUAL TO fifteen OR the sum of element at position i MINUS one and j PLUS one of grid PLUS element at position i and j of grid PLUS element at position i PLUS one and j MINUS one of grid IS NOT EQUAL TO fifteen  
                RETURN False  
            END IF  
            
            RETURN True  
        END FUNCTION  
        
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        
        FOR i FROM one TO rows MINUS one  
            FOR j FROM one TO cols MINUS one  
                IF is_magic(i, j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
