CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values of type list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH value at position zero of values  
    SET i TO one  
    SET queue TO a double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE element from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH value at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH value at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value at p IS NOT EQUAL TO value at q  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q dot left AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree WITH PARAMETERS self root of type Optional TreeNode RETURNS Optional TreeNode  
        IF root IS none  
            RETURN none  
        END IF  
        SET root dot left TO CALL pruneTree WITH PARAMETERS root dot left  
        SET root dot right TO CALL pruneTree WITH PARAMETERS root dot right  
        IF value at root EQUALS zero AND root dot left IS none AND root dot right IS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
