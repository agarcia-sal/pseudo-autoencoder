CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST containing the PAIRS one and zero, MINUS one and zero, zero and one, zero and MINUS one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two-dimensional LIST with m ROWS and n COLUMNS FILLED WITH infinite value  
            ASSIGN pointer TO a new double-ended queue named queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the TRIPLE i, j, zero TO queue  
                        SET element at position i and j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue is NOT empty  
                REMOVE and RETURN the FIRST element of queue INTO x, y, t  
                FOR each dx, dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx and ny of grid EQUALS zero AND element at position nx and ny of fire_times EQUALS infinite value  
                        SET element at position nx and ny of fire_times TO t PLUS one  
                        APPEND the TRIPLE nx, ny, t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            ASSIGN pointer TO a new double-ended queue containing the TRIPLE zero, zero, and wait NAMED queue  
            SET visited TO a new SET containing the PAIR zero and zero  
            
            WHILE queue is NOT empty  
                REMOVE and RETURN the FIRST element of queue INTO x, y, t  
                FOR each dx, dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the PAIR nx and ny IS NOT IN visited AND element at position nx and ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx and ny of fire_times IS GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN TRUE  
                            ELSE  
                                RETURN FALSE  
                            END IF  
                        END IF  
                        IF element at position nx and ny of fire_times IS GREATER THAN t PLUS one  
                            ADD the PAIR nx and ny TO visited  
                            APPEND the TRIPLE nx, ny, t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN FALSE  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF the SUM of left and right BY two  
            IF can_escape(mid)  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result NOT EQUALS m MULTIPLIED BY n  
            RETURN result  
        ELSE  
            RETURN ten RAISED TO THE POWER OF nine  
        END IF  
    END FUNCTION  
END CLASS
