CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s STRING, k INTEGER) RETURNS INTEGER  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO an empty collection mapping characters TO integers  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN the character at position right of s TO key  
            SET the value for key in char_map TO right  
            
            IF the NUMBER OF keys in char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map WITH the smallest value  
                SET left TO the value of leftmost_char in char_map PLUS one  
                REMOVE the entry for leftmost_char FROM char_map  
            END IF  
            
            SET current_window_length TO right MINUS left PLUS one  
            SET max_length TO the GREATER OF max_length AND current_window_length  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
