CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM i TO n MINUS one  
                SET temporary_value TO element at position i OF element at position j OF matrix  
                SET element at position i OF element at position j OF matrix TO element at position j OF element at position i OF matrix  
                SET element at position j OF element at position i OF matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR each i FROM zero TO n MINUS one  
            REVERSE the order of elements in element at position i OF matrix  
        END FOR  
    END FUNCTION  
END CLASS
