CLASS Solution  
    FUNCTION getMoneyAmount WITH parameters n  
        FUNCTION dp WITH parameters left right  
            IF left IS GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER OF dp WITH left AND pivot MINUS one AND dp WITH pivot PLUS one AND right  
                SET min_cost TO the LESSER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp CALLED WITH one AND n  
END CLASS
