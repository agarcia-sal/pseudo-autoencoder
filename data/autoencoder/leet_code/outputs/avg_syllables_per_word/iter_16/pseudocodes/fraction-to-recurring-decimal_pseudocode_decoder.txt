CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero EXCLUSIVE OR denominator LESS THAN zero  
            SET sign TO minus sign  
        ELSE  
            SET sign TO empty string  
        END IF  
        
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        
        SET quotient TO the integer quotient of numerator DIVIDED BY denominator  
        SET remainder TO the integer remainder of numerator DIVIDED BY denominator  
        SET integer_part TO string representation of quotient  
        
        IF remainder EQUALS zero  
            RETURN concatenation of sign and integer_part  
        END IF  
        
        SET seen_remainders TO an empty dictionary  
        SET fractional_part TO an empty list  
        
        WHILE remainder IS NOT equal to zero  
            IF remainder IS a key in seen_remainders  
                SET start TO the value associated with remainder in seen_remainders  
                SET non_repeating TO concatenation of elements from position zero TO position one LESS THAN start of fractional_part  
                SET repeating TO concatenation of elements from position start TO the end of fractional_part  
                RETURN concatenation of sign, integer_part, a period, non_repeating, an opening parenthesis, repeating, and a closing parenthesis  
            END IF  
            
            ASSIGN the value of the length of fractional_part TO the key remainder in seen_remainders  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer quotient of remainder DIVIDED BY denominator  
            SET remainder TO the integer remainder of remainder DIVIDED BY denominator  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN concatenation of sign, integer_part, a period, and concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
