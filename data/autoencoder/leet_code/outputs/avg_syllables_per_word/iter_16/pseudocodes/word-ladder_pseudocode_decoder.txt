CLASS Solution  
    FUNCTION ladderLength(beginWord string, endWord string, wordList list of string) RETURNS integer  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
          
        SET word_dict TO the DEFAULT DICTIONARY FOR lists  
        SET word_length TO the LENGTH OF beginWord  
          
        FOR each word IN wordList  
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_phrase TO the CONCATENATION OF the substring from position zero TO index MINUS one of word PLUS the ASTERISK SYMBOL PLUS the substring from position index PLUS one TO the end of word  
                APPEND word TO the element at intermediate_phrase of word_dict  
            END FOR  
        END FOR  
          
        SET queue TO a DOUBLE ENDED QUEUE initialized with the PAIR of beginWord and integer one  
        SET visited TO a SET initialized with beginWord  
          
        WHILE queue is not empty  
            REMOVE and RETURN the FIRST element PAIR OF current_word and current_length FROM queue  
              
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the CONCATENATION OF the substring from position zero TO index MINUS one of current_word PLUS the ASTERISK SYMBOL PLUS the substring from position index PLUS one TO the end of current_word  
                  
                FOR each word IN the element at intermediate_word of word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the PAIR of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
          
        RETURN zero  
    END FUNCTION  
END CLASS
