CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n INTEGER k INTEGER maxPts INTEGER  
        FUNCTION dfs WITH PARAMETER i INTEGER  
            IF i IS GREATER THAN OR EQUAL TO k  
                RETURN ONE IF i IS LESS THAN OR EQUAL TO n OTHERWISE RETURN ZERO  
            END IF  
            IF i IS EQUAL TO k MINUS ONE  
                SET difference TO n MINUS k PLUS ONE  
                SET minimum_value TO THE SMALLER OF difference AND maxPts  
                RETURN minimum_value DIVIDED BY maxPts  
            END IF  
            SET first_recursive_result TO dfs CALLED WITH i PLUS ONE  
            SET second_recursive_result TO dfs CALLED WITH i PLUS maxPts PLUS ONE  
            SET numerator TO first_recursive_result MINUS second_recursive_result  
            SET addition_result TO first_recursive_result PLUS numerator DIVIDED BY maxPts  
            RETURN addition_result  
        END FUNCTION  
        RETURN dfs CALLED WITH ZERO  
    END FUNCTION  
END CLASS
