CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold EQUALS zero  
            RETURN a list of Boolean True with the LENGTH OF queries  
        END IF  
        
        SET parent TO a list containing numbers FROM zero TO n INCLUSIVE  
        
        FUNCTION find(x)  
            IF the element at position x of parent NOT EQUALS x  
                SET the element at position x of parent TO the result of find applied to the element at position x of parent  
            END IF  
            RETURN the element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO the result of find applied to x  
            SET rootY TO the result of find applied to y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS one TO n INCLUSIVE  
            FOR j FROM two TIMES i TO n INCLUSIVE IN STEPS OF i  
                CALL union with i and j  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        FOR each pair a and b IN queries  
            APPEND the Boolean value of the result of find applied to a EQUALS the result of find applied to b TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
