CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        
        SET num_set TO the SET CONTAINING all elements IN nums  
        SET longest_streak TO zero  
        
        FOR each num IN num_set  
            IF the ELEMENT AT position one LESS THAN num IS NOT IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                
                WHILE the ELEMENT AT position one GREATER THAN current_num IS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                
                SET longest_streak TO THE GREATER OF longest_streak AND current_streak  
            END IF  
        END FOR  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
