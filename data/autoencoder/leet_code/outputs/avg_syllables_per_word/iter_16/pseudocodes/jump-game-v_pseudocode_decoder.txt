CLASS Solution  
    FUNCTION maxJumps(arr WITH TYPE list OF integers, d WITH TYPE integer) RETURNS integer  
        FUNCTION dp(i WITH TYPE integer) RETURNS integer  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x IS LESS THAN the LENGTH OF arr  
                    IF element at position i PLUS x of arr IS GREATER THAN OR EQUAL TO element at position i of arr  
                        BREAK  
                    END IF  
                    SET candidate_jumps TO one PLUS dp(i PLUS x)  
                    IF candidate_jumps IS GREATER THAN max_jumps  
                        SET max_jumps TO candidate_jumps  
                    END IF  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO zero  
                    IF element at position i MINUS x of arr IS GREATER THAN OR EQUAL TO element at position i of arr  
                        BREAK  
                    END IF  
                    SET candidate_jumps TO one PLUS dp(i MINUS x)  
                    IF candidate_jumps IS GREATER THAN max_jumps  
                        SET max_jumps TO candidate_jumps  
                    END IF  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET maximum_overall TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            SET current_jumps TO dp(i)  
            IF current_jumps IS GREATER THAN maximum_overall  
                SET maximum_overall TO current_jumps  
            END IF  
        END FOR  
        RETURN maximum_overall  
    END FUNCTION  
END CLASS
