CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills AS list OF strings, people AS list OF list OF strings) RETURNS list OF integers  
        INITIALIZE empty mapping d FROM string TO integer  
        FOR each index i AND skill s IN enumeration of req_skills  
            SET element at key s in mapping d TO i  
        END FOR  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        INITIALIZE list p WITH n elements all set to zero  
        FOR each index i AND skill_set ss IN enumeration of people  
            FOR each skill s IN ss  
                SET element at position i of list p TO element at position i of list p BITWISE OR WITH one LEFT SHIFTED BY element at key s in mapping d  
            END FOR  
        END FOR  
        INITIALIZE list f WITH size of two RAISED TO THE POWER OF m elements all set to infinite value  
        INITIALIZE list g WITH size of two RAISED TO THE POWER OF m elements all set to zero  
        INITIALIZE list h WITH size of two RAISED TO THE POWER OF m elements all set to zero  
        SET element at position zero of list f TO zero  
        FOR each integer i FROM zero TO two RAISED TO THE POWER OF m MINUS one  
            IF element at position i of list f EQUALS infinite value THEN  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            FOR each integer j FROM zero TO n MINUS one  
                IF element at position i of list f PLUS one IS LESS THAN element at position BITWISE OR OF i AND element at position j of list p of list f THEN  
                    SET element at position BITWISE OR OF i AND element at position j of list p of list f TO element at position i of list f PLUS one  
                    SET element at position BITWISE OR OF i AND element at position j of list p of list g TO j  
                    SET element at position BITWISE OR OF i AND element at position j of list p of list h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two RAISED TO THE POWER OF m MINUS one  
        INITIALIZE empty list ans  
        WHILE i IS NOT zero  
            APPEND element at position i of list g TO list ans  
            SET i TO element at position i of list h  
        END WHILE  
        RETURN list ans  
    END FUNCTION  
END CLASS
