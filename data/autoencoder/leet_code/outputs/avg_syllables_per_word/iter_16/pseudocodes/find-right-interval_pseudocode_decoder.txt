CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the list of pairs of the element at position zero of each interval AND the index of the interval IN intervals sorted by the element at position zero of the interval  
        SET result TO an empty list  
        FOR each pair of underscore AND end IN intervals  
            SET idx TO the smallest position found by binary search for a pair whose first element is greater than or equal to end IN start_times  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the element at position one of the pair at position idx of start_times TO result  
            ELSE  
                APPEND minus one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
