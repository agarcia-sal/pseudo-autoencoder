CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO the SET OF numbers such that for each pair of elements one from fronts and one from backs at the same position the elements are EQUAL  
        SET all_numbers TO the SET OF all elements IN fronts AND all elements IN backs  
        SET good_numbers TO the DIFFERENCE between all_numbers AND same  
        IF the LENGTH OF good_numbers is GREATER THAN zero  
            SET minimum_good_number TO the SMALLEST number IN good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
