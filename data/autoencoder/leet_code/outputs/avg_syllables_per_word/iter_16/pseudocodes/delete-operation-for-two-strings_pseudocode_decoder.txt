CLASS Solution  
    FUNCTION minDistance(word1 STRING, word2 STRING) RETURNS INTEGER  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        SET dp TO a two dimensional list with the number of rows equal to m PLUS one  
        AND the number of columns equal to n PLUS one  
        WHERE each element is initialized to zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position i and j of dp TO the element at position i MINUS one and j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i and j of dp TO the GREATER VALUE BETWEEN the element at position i MINUS one and j of dp AND the element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO the element at position m and n of dp  
        
        RETURN the value of m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
