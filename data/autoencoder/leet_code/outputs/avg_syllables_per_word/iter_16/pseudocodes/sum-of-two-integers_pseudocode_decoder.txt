CLASS Solution  
    FUNCTION getSum(a INTEGER, b INTEGER) RETURNS INTEGER  
        SET MAX TO the maximum thirty two bits integer value  
        SET mask TO the mask value corresponding to thirty two bits
        
        WHILE b IS NOT EQUALS TO zero HOLDING  
            SET carry TO the result of bitwise AND operation between a AND b THEN LEFT SHIFTED BY one position  
            SET a TO the result of bitwise XOR operation between a AND b THEN bitwise AND with mask  
            SET b TO the result of bitwise AND operation between carry AND mask  
        END WHILE
        
        IF a IS LESS THAN OR EQUAL TO MAX THEN  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the result of bitwise XOR operation between a AND mask  
        END IF  
    END FUNCTION  
END CLASS
