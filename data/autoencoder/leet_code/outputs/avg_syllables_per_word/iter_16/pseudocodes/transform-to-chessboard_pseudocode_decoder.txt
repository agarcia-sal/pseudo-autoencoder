CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR index i FROM zero TO N MINUS one  
            FOR index j FROM zero TO N MINUS one  
                IF the element at position i and j of board EXCLUSIVE OR the element at position i and zero of board EXCLUSIVE OR the element at position zero and j of board EXCLUSIVE OR the element at position zero and zero of board EQUALS true  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY the element at position i and zero of board  
            INCREMENT colSum BY the element at position zero and i of board  
            INCREMENT rowSwap BY the CONDITION whether the element at position i and zero of board EQUALS the remainder of i DIVIDED BY two  
            INCREMENT colSwap BY the CONDITION whether the element at position zero and i of board EQUALS the remainder of i DIVIDED BY two  
        END FOR  
        
        IF the integer division of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the integer division of the sum of N and one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the integer division of the sum of N and one BY two  
            RETURN negative one  
        END IF  
        
        IF the remainder of N DIVIDED BY two IS NOT EQUAL TO zero  
            IF the remainder of rowSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        
        RETURN the integer division of the sum of rowSwap and colSwap BY two  
    END FUNCTION  
END CLASS
