CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n  
        SET MOD TO one billion PLUS seven  
        SET a TO six  
        SET b TO six  
        FOR each number FROM two TO n  
            SET new_a TO the quantity of three MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MOD  
            SET new_b TO the quantity of two MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MOD  
            SET a TO new_a  
            SET b TO new_b  
        END FOR  
        RETURN the quantity of a PLUS b MODULO MOD  
    END FUNCTION  
END CLASS
