CLASS Solution  
    FUNCTION findBlackPixel(picture AS List OF List OF string, target AS integer) RETURNS integer  
        IF picture IS EMPTY OR element at position zero of picture IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        INITIALIZE row_count AS list of zeros with length m  
        INITIALIZE col_count AS list of zeros with length n  
        INITIALIZE row_patterns AS a mapping from string to integer with default zero  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at position index in element at position r of picture FOR index FROM zero TO n MINUS one  
            INCREMENT the count at key pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    SET pattern TO the concatenation of all elements at position index in element at position r of picture FOR index FROM zero TO n MINUS one  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the count at key pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
