CLASS Solution  
    FUNCTION minOperations(target AS List OF INTEGER, arr AS List OF INTEGER) RETURNS INTEGER  
        SET index_map TO a new empty mapping from INTEGER to INTEGER  
        SET position TO zero  
        FOR each value IN target  
            SET element at key value of index_map TO position  
            INCREMENT position BY one  
        END FOR  
        
        SET transformed_arr TO a new empty list of INTEGER  
        FOR each value IN arr  
            IF index_map CONTAINS key value THEN  
                APPEND element at key value of index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO a new empty list of INTEGER  
        FOR each num IN transformed_arr  
            SET pos TO the position where num would be inserted to maintain sorted order of lis using binary search for left insertion  
            IF pos EQUALS the LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO the LENGTH OF lis  
        
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
