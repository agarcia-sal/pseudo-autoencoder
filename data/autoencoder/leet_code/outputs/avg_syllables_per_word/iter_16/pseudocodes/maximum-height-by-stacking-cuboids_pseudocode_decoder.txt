CLASS Solution  
    FUNCTION maxHeight WITH parameters cuboids  
        FOR each cuboid IN cuboids  
            SORT dimensions of cuboid in non-decreasing order  
        END FOR  
        SORT cuboids lexicographically by their sorted dimensions  
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp as a list of length n with all values zero  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i of dp TO element at position three of cuboid at position i  
            FOR index j FROM zero TO i MINUS one  
                IF element at position one of cuboid at position j LESS THAN OR EQUAL TO element at position one of cuboid at position i AND  
                   element at position two of cuboid at position j LESS THAN OR EQUAL TO element at position two of cuboid at position i AND  
                   element at position three of cuboid at position j LESS THAN OR EQUAL TO element at position three of cuboid at position i  
                    SET element at position i of dp TO the maximum BETWEEN element at position i of dp AND the sum of element at position j of dp AND element at position three of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the maximum value in dp  
    END FUNCTION  
END CLASS
