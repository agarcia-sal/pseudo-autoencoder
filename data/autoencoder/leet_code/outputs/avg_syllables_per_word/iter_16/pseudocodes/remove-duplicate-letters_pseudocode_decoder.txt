CLASS Solution  
    FUNCTION removeDuplicateLetters WITH parameter s  
        SET last_occurrence TO a DICTIONARY mapping each character TO the LAST POSITION OF that character IN s  
        SET stack TO an EMPTY LIST  
        SET in_stack TO an EMPTY SET  
        FOR each index AND character IN s FROM START TO END  
            IF character IS IN in_stack  
                CONTINUE TO the next iteration  
            END IF  
            WHILE stack IS NOT EMPTY AND character IS LESS THAN the ELEMENT at the LAST POSITION OF stack AND index IS LESS THAN the LAST POSITION OF the ELEMENT at the LAST POSITION OF stack IN last_occurrence  
                SET removed_char TO THE ELEMENT REMOVED FROM the LAST POSITION OF stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        SET result_string TO THE CONCATENATION OF all ELEMENTS IN stack IN ORDER  
        RETURN result_string  
    END FUNCTION  
END CLASS
