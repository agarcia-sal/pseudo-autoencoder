CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the PAIR of x and y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the PAIR of x and y TO visited  
            FOR each pair dx and dy IN the list consisting of the PAIR negative one and zero, the PAIR one and zero, the PAIR zero and negative one, the PAIR zero and one  
                CALL dfs WITH x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO empty set  
            SET land_cells TO a list OF all PAIRS i and j WHERE i ranges FROM zero TO the LENGTH OF grid MINUS one AND j ranges FROM zero TO the LENGTH OF element at position zero of grid MINUS one AND element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN true  
            END IF  
            CALL dfs WITH the first element of land_cells at position zero, the first element of land_cells at position one, visited  
            RETURN whether the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF is_connected() EQUALS false  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF is_connected() EQUALS false  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
