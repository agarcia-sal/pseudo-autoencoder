CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO an empty set of unique sequences  
        SET sequence TO an empty list for the Eulerian path  

        FUNCTION dfs(node)  
            FOR each integer i FROM zero TO k MINUS one  
                SET edge TO the concatenation of node WITH the string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH the substring of edge FROM position one TO the end  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO the string of zero REPEATED n MINUS one times  
        CALL dfs WITH start_node  
        SET result TO the concatenation of all elements of sequence IN REVERSE ORDER WITH start_node APPENDED AT THE END  
        RETURN result  
    END FUNCTION  
END CLASS
