CLASS Solution  
    FUNCTION consecutiveNumbersSum(n)  
        SET count TO zero  
        FOR k FROM one TO n  
            IF the product of two AND n MODULO k EQUALS zero  
                SET m TO the quotient of the difference between the quotient of the product of two AND n DIVIDED BY k AND k MINUS one plus one DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUALS m CONVERTED TO INTEGER  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
