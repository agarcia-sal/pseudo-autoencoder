CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1 AS List of strings, sentence2 AS List of strings, similarPairs AS List of List of strings) RETURNS Boolean  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN False  
        END IF  

        SET parent TO an empty dictionary  

        FUNCTION find(x)  
            IF the element at key x of parent NOT EQUALS x  
                SET the element at key x of parent TO the result of calling find with the element at key x of parent  
            END IF  
            RETURN the element at key x of parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO the result of calling find with x  
            SET rootY TO the result of calling find with y  
            IF rootX NOT EQUALS rootY  
                SET the element at key rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN keys of parent  
                    SET the element at key word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x and y IN similarPairs  
            CALL union with x and y  
        END FOR  

        FOR each w1 and w2 IN elements at the same positions of sentence1 and sentence2 respectively  
            IF w1 NOT IN keys of parent  
                SET the element at key w1 of parent TO w1  
            END IF  
            IF w2 NOT IN keys of parent  
                SET the element at key w2 of parent TO w2  
            END IF  
            IF the result of calling find with w1 NOT EQUALS the result of calling find with w2  
                RETURN False  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
