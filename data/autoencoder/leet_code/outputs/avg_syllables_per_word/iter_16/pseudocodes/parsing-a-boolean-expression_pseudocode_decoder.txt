CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character IN expression  
            IF character EQUALS one of the values true false negation and or or  
                APPEND character TO stk  
            ELSE IF character EQUALS closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE element at last position of stk EQUALS one of the values true false  
                    INCREMENT count_true BY one IF element at last position of stk EQUALS true  
                    INCREMENT count_false BY one IF element at last position of stk EQUALS false  
                    REMOVE element at last position of stk  
                END WHILE  
                SET operator TO element at last position of stk  
                REMOVE element at last position of stk  
                IF operator EQUALS negation  
                    SET character TO true IF count_false IS zero OTHERWISE SET character TO false  
                ELSE IF operator EQUALS and  
                    SET character TO false IF count_false IS NOT zero OTHERWISE SET character TO true  
                ELSE IF operator EQUALS or  
                    SET character TO true IF count_true IS NOT zero OTHERWISE SET character TO false  
                END IF  
                APPEND character TO stk  
            END IF  
        END FOR  
        RETURN comparison WHETHER element at position zero of stk EQUALS true  
    END FUNCTION  
END CLASS
