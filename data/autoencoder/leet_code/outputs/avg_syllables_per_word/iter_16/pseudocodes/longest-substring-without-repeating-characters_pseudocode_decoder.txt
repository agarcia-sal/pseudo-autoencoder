CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH parameter s  
        SET char_index_map TO empty collection mapping characters to positions  
        SET max_length TO zero  
        SET start TO zero  
  
        FOR each position end AND corresponding character char IN enumeration of s  
            IF character char IS PRESENT IN char_index_map AND position mapped by char_index_map at character char IS GREATER THAN OR EQUAL TO start  
                SET start TO position mapped by char_index_map at character char PLUS one  
            END IF  
            SET the value mapped by char_index_map at character char TO end  
            SET max_length TO the GREATER OF max_length AND the difference BETWEEN end AND start PLUS one  
        END FOR  
  
        RETURN max_length  
    END FUNCTION  
END CLASS
