CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND height h IN enumerate heights  
            WHILE the element at the last position of stack DOES NOT EQUAL negative one AND height h IS LESS THAN the element at the position at the last position of stack in heights  
                SET height TO element at the position obtained by removing the last element from stack in heights  
                SET width TO index i MINUS the element at the last position of stack MINUS one  
                SET max_area TO the GREATER VALUE BETWEEN max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND index i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
