CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x GREATER THAN zero THEN  
                    INCREMENT cnt BY the POSITION TO THE RIGHT WHERE p DIVIDED BY x WOULD BE INSERTED IN nums2 TO MAINTAIN ORDER  
                ELSE IF x LESS THAN zero THEN  
                    INCREMENT cnt BY n MINUS the POSITION TO THE LEFT WHERE p DIVIDED BY x WOULD BE INSERTED IN nums2 TO MAINTAIN ORDER  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY THE VALUE ONE IF p GREATER THAN OR EQUAL TO zero OTHERWISE ZERO  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF THE ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND THE ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF THE ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND THE ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION TO THE LEFT WHERE k WOULD BE INSERTED IN THE SEQUENCE FROM NEGATIVE mx TO mx INCLUSIVE USING count AS THE KEY FUNCTION MINUS mx  
    END FUNCTION  
END CLASS
