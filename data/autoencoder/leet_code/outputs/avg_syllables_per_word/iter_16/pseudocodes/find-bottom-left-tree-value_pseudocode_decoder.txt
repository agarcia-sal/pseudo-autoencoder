CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values OF TYPE list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q dot left AND AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue WITH PARAMETERS self root  
        SET queue TO A NEW deque INITIALIZED WITH root  
        SET leftmost_value TO none  
        WHILE queue HOLDS ELEMENTS  
            SET level_size TO THE LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO node dot val  
                END IF  
                IF node dot left IS NOT none  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
