CLASS Solution  
    FUNCTION jobScheduling(startTime LIST OF INTEGER endTime LIST OF INTEGER profit LIST OF INTEGER) RETURNS INTEGER  
        SET jobs TO a list of tuples each consisting of element at position index of startTime paired with element at position index of endTime paired with element at position index of profit for each index in the range from zero TO the LENGTH OF startTime MINUS one SORTED BY the second element in each tuple in ascending order  
        SET end_times TO a list consisting of the second element of each tuple in jobs  
        SET dp TO a list consisting of zero repeated for the LENGTH OF jobs PLUS one times  
        FOR index FROM one TO the LENGTH OF jobs  
            SET j TO the position where the element at position index MINUS one in jobs evaluated at its first element would be inserted in end_times so that order is maintained using binary search  
            SET dp at position index TO the greater of dp at position index MINUS one AND the sum of dp at position j AND the third element of the tuple at position index MINUS one in jobs  
        END FOR  
        RETURN the last element in dp  
    END FUNCTION  
END CLASS
