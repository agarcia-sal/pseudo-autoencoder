CLASS Solution  
    FUNCTION subsetsWithDup(nums)  
        CALL sort ON nums  
        SET result TO empty list  
        SET subset TO empty list  
        
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF nums THEN  
                APPEND a copy of subset TO result  
                RETURN  
            END IF  
            
            APPEND element at position i of nums TO subset  
            CALL dfs WITH parameter i PLUS one  
            
            REMOVE the last element FROM subset  
            WHILE i PLUS one LESS THAN the LENGTH OF nums AND element at position i of nums EQUALS element at position i PLUS one of nums HOLD  
                INCREMENT i BY one  
            END WHILE  
            CALL dfs WITH parameter i PLUS one  
        END FUNCTION  
        
        CALL dfs WITH parameter zero  
        RETURN result  
    END FUNCTION  
END CLASS
