CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty mapping from state representations TO boolean values  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS AS A KEY IN memo  
                RETURN the value ASSOCIATED WITH state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO index PLUS one OF state EQUALS two PLUS symbols  
                    SET new_state TO the concatenation of the substring from start TO index OF state FOLLOWED BY two minus symbols FOLLOWED BY the substring from position index PLUS two TO end OF state  
                    IF the negation of can_win_helper APPLIED TO new_state HOLDS  
                        SET the value ASSOCIATED WITH state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value ASSOCIATED WITH state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win_helper APPLIED TO currentState  
    END FUNCTION  
END CLASS
