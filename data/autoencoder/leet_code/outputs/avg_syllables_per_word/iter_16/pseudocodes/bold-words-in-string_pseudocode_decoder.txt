CLASS Solution  
    FUNCTION boldWords(words AS List OF string, s AS string) RETURNS string  
        SET bold TO a list of boolean values all set to False with the LENGTH OF s  
        
        FOR each word IN words  
            SET start TO the position where word FIRST OCCURS in s or negative one if NOT FOUND  
            WHILE start IS NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of bold TO True  
                END FOR  
                SET start TO the position where word NEXT OCCURS in s AFTER position start PLUS one or negative one if NOT FOUND  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF s  
            IF the element at position i of bold IS True  
                APPEND the text consisting of the opening bold tag TO result  
                WHILE i IS LESS THAN the LENGTH OF s AND the element at position i of bold IS True  
                    APPEND the element at position i of string s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the text consisting of the closing bold tag TO result  
            ELSE  
                APPEND the element at position i of string s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
