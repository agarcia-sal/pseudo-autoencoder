CLASS Solution  
    FUNCTION subStrHash(s STRING power INTEGER modulo INTEGER k INTEGER hashValue INTEGER) RETURNS STRING  
        FUNCTION val(c CHARACTER) RETURNS INTEGER  
            RETURN THE NUMERICAL POSITION OF CHARACTER c IN THE ALPHABET STARTING FROM ONE FOR LETTER a  
        END FUNCTION  
      
        SET n TO THE LENGTH OF STRING s  
        SET current_hash TO zero  
        SET p_pow TO one  
      
        FOR index FROM n MINUS k TO n MINUS one  
            SET current_hash TO THE REMAINDER OF THE SUM OF current_hash PLUS THE VALUE OF CHARACTER AT POSITION index OF STRING s MULTIPLIED BY p_pow DIVIDED BY modulo  
            IF index IS LESS THAN n MINUS one  
                SET p_pow TO THE REMAINDER OF THE PRODUCT OF p_pow MULTIPLIED BY power DIVIDED BY modulo  
            END IF  
        END FOR  
      
        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN THE SUBSTRING FROM POSITION start_index TO THE END OF STRING s  
        END IF  
      
        SET p_pow_k_minus_1 TO p_pow  
      
        FOR index FROM n MINUS k MINUS one DOWN TO zero  
            SET current_hash TO THE REMAINDER OF current_hash MINUS THE PRODUCT OF THE VALUE OF CHARACTER AT POSITION index PLUS k OF STRING s MULTIPLIED BY p_pow_k_minus_1 DIVIDED BY modulo  
            SET current_hash TO THE REMAINDER OF THE SUM OF THE PRODUCT OF current_hash MULTIPLIED BY power PLUS THE VALUE OF CHARACTER AT POSITION index OF STRING s DIVIDED BY modulo  
            IF current_hash EQUALS hashValue  
                SET start_index TO index  
            END IF  
        END FOR  
      
        RETURN THE SUBSTRING FROM POSITION start_index TO POSITION start_index PLUS k OF STRING s  
    END FUNCTION  
END CLASS
