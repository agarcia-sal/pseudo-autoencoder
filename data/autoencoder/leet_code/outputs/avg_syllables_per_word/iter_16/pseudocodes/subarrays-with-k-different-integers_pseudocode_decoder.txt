CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums AS List OF integers, k AS integer) RETURNS integer  
        FUNCTION atMostKDistinct(t AS integer) RETURNS integer  
            SET count TO a new default dictionary mapping integers to integers with default zero  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the value in count associated with the element at position right of nums BY one  
                WHILE the NUMBER OF KEYS IN count IS GREATER THAN t  
                    DECREMENT the value in count associated with the element at position left of nums BY one  
                    IF the value in count associated with the element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY the value of right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the result of atMostKDistinct WITH k MINUS the result of atMostKDistinct WITH k MINUS one  
    END FUNCTION  
END CLASS
