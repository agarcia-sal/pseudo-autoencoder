CLASS Solution  
    FUNCTION insert(intervals AS List of List of Integer newInterval AS List of Integer) RETURNS List of List of Integer  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN a new list containing newInterval  
        END IF  
        
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        
        WHILE i LESS THAN n AND the element at position one of the element at position i of intervals LESS THAN the element at position zero of newInterval holds  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i LESS THAN n AND the element at position zero of the element at position i of intervals LESS THAN OR EQUALS the element at position one of newInterval holds  
            SET the element at position zero of newInterval TO the minimum value BETWEEN the element at position zero of newInterval AND the element at position zero of the element at position i of intervals  
            SET the element at position one of newInterval TO the maximum value BETWEEN the element at position one of newInterval AND the element at position one of the element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
