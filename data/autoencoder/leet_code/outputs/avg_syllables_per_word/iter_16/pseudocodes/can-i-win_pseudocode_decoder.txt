CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one divided by two is LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal IS LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        SET memo TO an empty dictionary  
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers EXISTS IN memo  
                RETURN the value at memo of the pair of current_total AND used_numbers  
            END IF  
            FOR each number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND one shifted left by the number IS EQUAL TO zero  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal OR the negation of the result of can_win with sum of current_total AND number AND bitwise OR of used_numbers AND one shifted left by number  
                        SET the value at memo of the pair of current_total AND used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET the value at memo of the pair of current_total AND used_numbers TO false  
            RETURN false  
        END FUNCTION  
        RETURN the result of can_win with zero AND zero  
    END FUNCTION  
END CLASS
