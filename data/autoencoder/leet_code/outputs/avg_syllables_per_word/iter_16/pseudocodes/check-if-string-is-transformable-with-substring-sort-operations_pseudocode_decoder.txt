CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AND t  
        CREATE a mapping from digits to lists of positions in s, named pos using default list structure  
        FOR each index position AND character c IN the string s  
            PUSH the index position INTO the list of positions associated with the digit INTERPRETED FROM character c IN pos  
        END FOR  
        FOR each index position AND character c IN the string t  
            SET digit TO the digit INTERPRETED FROM character c  
            IF the list of positions associated with digit IN pos IS empty  
                RETURN false  
            END IF  
            FOR each smaller digit FROM zero TO digit MINUS one  
                IF the list of positions associated with smaller digit IN pos IS NOT empty AND the first element at position zero in that list IS LESS THAN the first element at position zero in the list associated with digit  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest position from the list associated with digit IN pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
