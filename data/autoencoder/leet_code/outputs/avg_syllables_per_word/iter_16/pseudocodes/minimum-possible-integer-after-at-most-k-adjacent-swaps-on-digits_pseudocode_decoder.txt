CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the list formed by the sequence of characters in num  
        SET n TO the length of num  
        
        WHILE the value of k is GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF the element at position i of num IS GREATER THAN the element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the smallest value between i PLUS k PLUS one AND n MINUS one  
                        IF the element at position j of num IS LESS THAN min_digit  
                            SET min_digit TO the element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the sequence of elements from position i TO min_index of num TO the list constructed by placing the element at position min_index of num AT the beginning FOLLOWED BY the sequence of elements from position i TO min_index MINUS one of num  
                    DECREMENT k BY the difference between min_index AND i  
                    SET swapped TO true  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped IS false  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num as a single string  
    END FUNCTION  
END CLASS
