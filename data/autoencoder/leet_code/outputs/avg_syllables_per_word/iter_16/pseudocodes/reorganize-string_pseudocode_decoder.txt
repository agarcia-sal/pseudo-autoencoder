CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the frequency count of each character in s  
        SET max_heap TO a list where for each character and its frequency in count the element is negative frequency and character  
        PERFORM heapify operation on max_heap to organize it as a heap structure  
      
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO an empty list  
      
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN an empty string  
            END IF  
          
            REMOVE the element with smallest value from max_heap and ASSIGN first part to freq and second part to char  
            APPEND char TO result  
          
            INCREMENT freq BY one  
          
            IF prev_freq is less than zero  
                ADD the pair of prev_freq and prev_char back to max_heap  
            END IF  
          
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
      
        RETURN the concatenation of all elements in result as a string  
    END FUNCTION  
END CLASS
