CLASS Solution  
    FUNCTION largestOverlap(img1 AS List OF List OF Integer, img2 AS List OF List OF Integer)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of img1 AT position j EQUALS one  
                    APPEND the PAIR of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of img2 AT position j EQUALS one  
                    APPEND the PAIR of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a new mapping from PAIR OF Integer AND Integer TO Integer with default value zero  
        FOR each PAIR x1 AND y1 IN points1  
            FOR each PAIR x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key PAIR of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN the values of overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
