CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours AND minutes TO the result of converting the parts of time separated by the colon into integers  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        SET minutes_list TO a new list formed by applying to_minutes to each element in timePoints  
        SORT minutes_list IN ascending order  
        SET min_diff TO infinity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_diff TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            IF current_diff LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  
        SET circular_diff TO one thousand four hundred forty MINUS the last element of minutes_list PLUS the first element of minutes_list  
        IF circular_diff LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
