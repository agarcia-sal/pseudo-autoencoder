CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value IS LESS THAN lower_bound  
                RETURN false  
            END IF  
            WHILE the stack is not empty AND value IS GREATER THAN element at the last position of stack  
                SET lower_bound TO element popped from the stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
