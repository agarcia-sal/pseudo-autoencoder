CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero AND string one AND string eight  
            END IF  
            SET ans TO empty list  
            FOR each element v IN the RESULT OF calling dfs WITH PARAMETER u MINUS two  
                FOR each pair l AND r IN the set of string pairs eleven AND eighty eight AND sixty nine AND ninety six  
                    APPEND the concatenation of l FOLLOWED BY v FOLLOWED BY r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND the concatenation of string zero FOLLOWED BY v FOLLOWED BY string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN the RESULT OF calling dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
