CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        
        SET dp TO an empty list  
        
        FOR each num IN nums  
            SET idx TO the POSITION WHERE num SHOULD BE INSERTED IN dp TO KEEP dp IN NON-DECREASING ORDER USING BINARY SEARCH FROM THE BEGINNING OF dp  
            
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx OF dp TO num  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
