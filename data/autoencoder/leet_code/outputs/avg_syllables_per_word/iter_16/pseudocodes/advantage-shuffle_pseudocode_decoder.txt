CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        CALL sort nums1 INTO ascending order  
        SET nums2_with_indices TO the result of calling an auxiliary FUNCTION that creates a list of pairs each containing a value from nums2 and its original position  
        CALL sort nums2_with_indices INTO descending order based on the value in each pair  
        SET result TO a list initialized with zeros having the same length as nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair of value and index IN nums2_with_indices  
            IF the element at position right of nums1 is GREATER THAN value THEN  
                SET the element at position index of result TO the element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET the element at position index of result TO the element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
