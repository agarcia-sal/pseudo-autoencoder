CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO a default dictionary with empty list as default value  
        FOR each pair u AND v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
        
        SET queue TO a double ended queue containing the tuple of one current node one point zero current probability zero current time  
        SET visited TO a set containing one  
        
        WHILE there exists an element in queue  
            REMOVE the element from the front of queue and SET current_node current_probability current_time TO the unpacked elements  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the number of elements in the list at key current_node in graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration  
                END IF  
            END IF  
            
            SET possible_moves TO the number of elements in the list at key current_node in graph MINUS one if current_node NOT EQUALS one otherwise zero  
            
            FOR each neighbor IN the list at key current_node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor current_probability DIVIDED BY possible_moves current_time INCREMENTED BY one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
