CLASS Solution  
    FUNCTION maxNumber(nums1 AS List OF INTEGER, nums2 AS List OF INTEGER, k AS INTEGER) AS List OF INTEGER  
        FUNCTION maxSingleArray(nums AS List OF INTEGER, k AS INTEGER) AS List OF INTEGER  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND the LENGTH OF stack IS GREATER THAN zero AND the element at position the LENGTH OF stack MINUS one of stack IS LESS THAN num  
                    REMOVE the element at position the LENGTH OF stack MINUS one FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the SUBLIST of stack from position zero TO k MINUS one  
        END FUNCTION  
        
        FUNCTION merge(arr1 AS List OF INTEGER, arr2 AS List OF INTEGER) AS List OF INTEGER  
            SET result TO an empty list  
            WHILE the LENGTH OF arr1 PLUS the LENGTH OF arr2 IS GREATER THAN zero  
                IF arr1 IS GREATER THAN arr2 in lexicographical order THEN  
                    APPEND the element at position zero of arr1 TO result  
                    REMOVE the element at position zero FROM arr1  
                ELSE  
                    APPEND the element at position zero of arr2 TO result  
                    REMOVE the element at position zero FROM arr2  
                END IF  
            END WHILE  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2 THEN  
                SET candidate TO merge(maxSingleArray(nums1, i), maxSingleArray(nums2, k MINUS i))  
                IF max_number IS LESS THAN candidate in lexicographical order THEN  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
