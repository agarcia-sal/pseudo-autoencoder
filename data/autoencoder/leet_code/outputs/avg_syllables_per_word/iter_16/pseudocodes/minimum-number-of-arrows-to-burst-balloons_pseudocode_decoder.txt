CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY the end coordinate value of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the start coordinate value of the current balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the end coordinate value of the current balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
