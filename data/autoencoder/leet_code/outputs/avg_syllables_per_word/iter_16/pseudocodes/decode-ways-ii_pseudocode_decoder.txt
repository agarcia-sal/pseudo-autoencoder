CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO the power of nine PLUS seven  
        INITIALIZE dp AS a list of zeros with the LENGTH OF s PLUS one elements  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character representing asterisk  
                RETURN nine  
            ELSE IF char EQUALS the character representing zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character representing asterisk AND char2 EQUALS the character representing asterisk  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character representing asterisk  
                IF char2 LESS THAN OR EQUAL TO the character representing six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character representing asterisk  
                IF char1 EQUALS the character representing one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character representing two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer conversion of the concatenation of char1 followed by char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO the LENGTH OF s  
            SET element at position i of dp TO the sum of element at position i of dp PLUS element at position i MINUS one of dp MULTIPLIED BY ways_to_decode_single(the element at position i MINUS one of s)  
            SET element at position i of dp TO the remainder of element at position i of dp AFTER DIVISION BY MOD  
            
            IF i GREATER THAN one  
                SET element at position i of dp TO the sum of element at position i of dp PLUS element at position i MINUS two of dp MULTIPLIED BY ways_to_decode_pair(the element at position i MINUS two of s, the element at position i MINUS one of s)  
                SET element at position i of dp TO the remainder of element at position i of dp AFTER DIVISION BY MOD  
            END IF  
        END FOR  
        
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
