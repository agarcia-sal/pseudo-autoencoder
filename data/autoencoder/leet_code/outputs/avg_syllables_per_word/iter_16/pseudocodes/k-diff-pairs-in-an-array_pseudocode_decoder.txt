CLASS Solution  
    FUNCTION findPairs(nums AS List OF integer, k AS integer) RETURNS integer  
        IF k LESS THAN zero THEN  
            RETURN zero  
        END IF  
        
        SET num_counts TO an empty mapping from integers TO integers  
        FOR each num IN nums  
            IF num IS IN keys of num_counts THEN  
                INCREMENT the value at key num in num_counts BY one  
            ELSE  
                SET the value at key num in num_counts TO one  
            END IF  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN the keys of num_counts  
            IF k EQUALS zero THEN  
                IF the value at key num in num_counts GREATER THAN one THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF the key that equals the result of num PLUS k IS IN keys of num_counts THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
