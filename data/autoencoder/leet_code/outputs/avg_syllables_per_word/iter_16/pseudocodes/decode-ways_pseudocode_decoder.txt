CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS EMPTY OR the element at position zero of string s EQUALS character zero THEN  
            RETURN zero  
        END IF  
        
        SET dp TO a list of zeroes with the LENGTH OF string s PLUS one  
        SET the element at position zero of list dp TO one  
        SET the element at position one of list dp TO one  
        
        FOR index i FROM two TO the LENGTH OF string s  
            IF the element at position i MINUS one of string s NOT EQUALS character zero THEN  
                INCREMENT the element at position i of list dp BY the element at position i MINUS one of list dp  
            END IF  
            
            SET two_digit TO the integer value of the substring from position i MINUS two TO i of string s  
            IF two_digit IS GREATER THAN OR EQUAL TO ten AND two_digit IS LESS THAN OR EQUAL TO twenty six THEN  
                INCREMENT the element at position i of list dp BY the element at position i MINUS two of list dp  
            END IF  
        END FOR  
        
        RETURN the last element of list dp  
    END FUNCTION  
END CLASS
