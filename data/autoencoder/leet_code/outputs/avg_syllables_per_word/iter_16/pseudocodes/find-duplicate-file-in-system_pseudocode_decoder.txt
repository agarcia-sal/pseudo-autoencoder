CLASS Solution  
    FUNCTION findDuplicate WITH parameters paths  
        ASSIGN pointer TO a new dictionary that maps from content strings TO list of file path strings NAMED content_to_paths  
        FOR each path IN paths  
            SPLIT path INTO parts BY spaces  
            SET directory TO element at position one of parts  
            FOR each file_info IN elements from position two TO the end of parts  
                SPLIT file_info INTO file_name AND content BY the character for opening parenthesis  
                SET content TO substring from position one TO the length of content MINUS one  
                APPEND concatenation of directory PLUS forward slash PLUS file_name TO the list referenced by the content key in content_to_paths  
            END FOR  
        END FOR  
        RETURN list of elements named paths FROM the values of content_to_paths WHERE the LENGTH OF paths IS GREATER THAN one  
    END FUNCTION  
END CLASS
