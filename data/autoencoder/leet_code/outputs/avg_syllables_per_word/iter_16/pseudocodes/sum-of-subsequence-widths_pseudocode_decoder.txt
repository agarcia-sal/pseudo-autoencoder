CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO one billion PLUS seven  
        SORT the list nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of length n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the ELEMENT at position index MINUS one of power MULTIPLIED BY two MODULO MODULO  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET increment_value TO the ELEMENT at position index of nums MULTIPLIED BY the DIFFERENCE BETWEEN the ELEMENT at position index of power AND the ELEMENT at position n MINUS index MINUS one of power  
            SET total_sum TO the SUM OF total_sum AND increment_value MODULO MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
