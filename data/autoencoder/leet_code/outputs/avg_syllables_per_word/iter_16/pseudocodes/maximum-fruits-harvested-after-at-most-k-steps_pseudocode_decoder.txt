CLASS Solution  
    FUNCTION maxTotalFruits(fruits, startPos, k)  
        SET max_fruits TO zero  
        SET total_fruits TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF fruits MINUS one  
            SET position TO the element at position zero of the element at position right of fruits  
            SET amount TO the element at position one of the element at position right of fruits  
            INCREMENT total_fruits BY amount  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND NOT (startPos MINUS k IS LESS THAN OR EQUAL TO the element at position zero of the element at position left of fruits AND the element at position zero of the element at position left of fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND startPos MINUS k IS LESS THAN OR EQUAL TO the element at position zero of the element at position right of fruits AND the element at position zero of the element at position right of fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND THE MINIMUM OF THE ABSOLUTE VALUE OF THE difference BETWEEN the element at position zero of the element at position right of fruits AND startPos AND THE ABSOLUTE VALUE OF THE difference BETWEEN the element at position zero of the element at position left of fruits AND startPos PLUS the difference BETWEEN the element at position zero of the element at position right of fruits AND the element at position zero of the element at position left of fruits IS LESS THAN OR EQUAL TO k)  
                DECREMENT total_fruits BY the element at position one of the element at position left of fruits  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_fruits TO THE GREATER OF max_fruits AND total_fruits  
        END FOR  
        
        RETURN max_fruits  
    END FUNCTION  
END CLASS
