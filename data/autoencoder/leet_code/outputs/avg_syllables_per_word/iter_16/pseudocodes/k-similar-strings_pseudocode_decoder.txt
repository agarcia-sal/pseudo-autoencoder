CLASS Solution  
    FUNCTION kSimilarity WITH PARAMETERS s1 AND s2  
        DEFINE FUNCTION neighbors WITH PARAMETER s  
            SET index i TO zero  
            WHILE index i IS LESS THAN the LENGTH OF s AND the element at position index i of s EQUALS the element at position index i of s2  
                INCREMENT index i BY one  
            END WHILE  
            FOR index j FROM index i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position index j of s EQUALS the element at position index i of s2 AND the element at position index j of s DOES NOT EQUAL the element at position index j of s2  
                    YIELD a new string CONSISTING OF the substring from start TO index i MINUS one of s FOLLOWED BY the element at position index j of s FOLLOWED BY the substring from position index i PLUS one TO index j MINUS one of s FOLLOWED BY the element at position index i of s FOLLOWED BY the substring from position index j PLUS one TO end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a double-ended queue INITIALIZED WITH the pair CONSISTING OF s1 AND zero  
        SET visited TO a set CONTAINING s1  
        
        WHILE queue IS NOT empty  
            REMOVE and ASSIGN the first element of queue TO variables s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the result of applying neighbors TO s  
                IF neighbor IS NOT contained IN visited  
                    ADD neighbor TO visited  
                    APPEND the pair CONSISTING OF neighbor AND swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
