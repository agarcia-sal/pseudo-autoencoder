CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of zeros of length n  
        SET stack TO an empty list  
        FOR each log IN logs  
            SPLIT log INTO function_id AS integer action AS string timestamp AS integer BY the separator colon  
            IF action EQUALS the word start  
                IF stack is not empty  
                    SET previous_function_id TO the first element of the last tuple in stack  
                    SET previous_timestamp TO the second element of the last tuple in stack  
                    INCREMENT element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp  
                END IF  
                APPEND the tuple of function_id and timestamp TO stack  
            ELSE  
                POP the last tuple from stack INTO previous_function_id and previous_timestamp  
                INCREMENT element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp PLUS one  
                IF stack is not empty  
                    SET the last tuple of stack TO the tuple of the first element of the last tuple in stack and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
