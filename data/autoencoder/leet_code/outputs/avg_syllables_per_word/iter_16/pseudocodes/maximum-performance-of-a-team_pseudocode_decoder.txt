CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO A LIST THAT COMBINES EACH ELEMENT OF efficiency WITH THE CORRESPONDING ELEMENT OF speed AS PAIRS  
        SORT engineers IN ORDER OF THE FIRST ELEMENT OF EACH PAIR FROM HIGHEST TO LOWEST  
        SET speed_heap TO AN EMPTY COLLECTION THAT MAINTAINS ELEMENTS IN SORTED ORDER WITH THE SMALLEST ELEMENT ACCESSIBLE  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair OF efficiency AND speed IN engineers  
            ADD speed TO speed_heap  
            INCREMENT speed_sum BY speed  
            IF THE LENGTH OF speed_heap IS GREATER THAN k THEN  
                REMOVE THE SMALLEST speed FROM speed_heap  
                DECREMENT speed_sum BY THAT REMOVED speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY efficiency  
            IF current_performance IS GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance MODULO TEN TO THE POWER OF NINE PLUS seven  
    END FUNCTION  
END CLASS
