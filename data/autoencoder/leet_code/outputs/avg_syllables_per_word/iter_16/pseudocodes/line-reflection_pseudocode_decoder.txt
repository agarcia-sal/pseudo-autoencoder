CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the SET OF all elements obtained by CONVERTING each point IN points TO a tuple  
        SET minimum_x_coordinate TO the MINIMUM value of the element at position zero of each point IN unique_points  
        SET maximum_x_coordinate TO the MAXIMUM value of the element at position zero of each point IN unique_points  
        SET line_of_reflection TO minimum_x_coordinate PLUS maximum_x_coordinate DIVIDED BY two  
        FOR each point IN unique_points  
            SET x TO the element at position zero of point  
            SET y TO the element at position one of point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the pair of reflected_x and y is NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
