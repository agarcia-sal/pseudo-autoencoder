CLASS Solution  
    FUNCTION networkDelayTime(times AS List OF List OF Integer, n AS Integer, k AS Integer) RETURNS Integer  
        SET graph TO empty mapping from Integer to list  
        FOR each integer i FROM one TO n  
            SET list at position i of graph TO empty list  
        END FOR  
        FOR each triplet u v w IN times  
            APPEND tuple of v and w TO list at position u of graph  
        END FOR  
        SET dist TO empty mapping from Integer to distance value  
        FOR each integer i FROM one TO n  
            SET distance at position i of dist TO infinity  
        END FOR  
        SET distance at position k of dist TO zero  
        SET min_heap TO list containing tuple of zero and k  
        WHILE min_heap is not empty  
            EXTRACT tuple of current_dist and node FROM the heap structure min_heap with smallest first element  
            IF current_dist IS GREATER THAN distance at position node of dist  
                CONTINUE to next iteration of WHILE  
            END IF  
            FOR each tuple neighbor and weight IN list at position node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance IS LESS THAN distance at position neighbor of dist  
                    SET distance at position neighbor of dist TO distance  
                    INSERT tuple of distance and neighbor INTO min_heap maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO maximal value among all distance values in dist  
        IF max_time IS EQUAL TO infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
