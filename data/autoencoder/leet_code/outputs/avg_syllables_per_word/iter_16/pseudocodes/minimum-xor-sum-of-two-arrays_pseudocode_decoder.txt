CLASS Solution  
    FUNCTION minimumXORSum(nums1 AS List of Integer, nums2 AS List of Integer) RETURNS Integer  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of size TWO RAISED TO THE POWER OF n FILLED WITH positive infinity  
        SET the element at position zero of dp TO zero  
        
        FOR each mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the NUMBER OF ONES in the binary representation of mask  
            FOR each j FROM zero TO n MINUS one  
                IF the bit at position j in mask IS SET  
                    SET dp at position mask TO the minimum of dp at position mask AND the sum of dp at position mask WITH the bit at position j UNSET AND the result of the bitwise XOR of the element at position bit_count MINUS one of nums1 AND the element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position TWO RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
