CLASS Solution  
    FUNCTION leastInterval(tasks LIST OF string, n integer) RETURNS integer  
        SET task_counts TO the frequency count of each element in tasks  
        SET max_heap TO an empty list  
        FOR each count IN the collection of values in task_counts  
            APPEND the negative of count TO max_heap  
        END FOR  
        APPLY heapify operation to max_heap  
        
        SET time TO zero  
        SET queue TO an empty list representing pairs of time plus n plus one and task count  
        
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND the first element at position zero at position zero of queue EQUALS time  
                REMOVE the first element FROM queue and ASSIGN its elements TO discard placeholder and count  
                IF count LESS THAN zero  
                    APPLY heappush operation to add count into max_heap  
                END IF  
            END WHILE  
            
            IF max_heap is not empty  
                APPLY heappop operation to remove and ASSIGN the top element to count  
                INCREMENT count BY one  
                IF count LESS THAN zero  
                    APPEND the pair of time PLUS n PLUS one and count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
