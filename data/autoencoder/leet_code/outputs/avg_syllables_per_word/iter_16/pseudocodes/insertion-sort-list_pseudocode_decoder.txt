CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values OF TYPE list)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO an INSTANCE OF ListNode with val EQUALS element at position zero of values  
    SET p TO head  
    FOR each val IN the SUBLIST of values from position one TO the END  
        SET node TO an INSTANCE OF ListNode with val EQUALS val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the comparison of the value of p1 EQUALS the value of p2 AND the RESULT OF is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head OF TYPE Optional OF ListNode) RETURNS Optional OF ListNode  
        IF head IS none OR the next element of head IS none  
            RETURN head  
        END IF  
        SET dummy TO an INSTANCE OF ListNode with val EQUALS negative five thousand one  
        SET the next element of dummy TO head  
        SET current TO the next element of head  
        SET the next element of head TO none  
        WHILE current IS NOT none  
            SET next_node TO the next element of current  
            SET prev TO dummy  
            WHILE the next element of prev IS NOT none AND the value of the next element of prev IS LESS THAN the value of current  
                SET prev TO the next element of prev  
            END WHILE  
            SET the next element of current TO the next element of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
