CLASS Solution  
    FUNCTION fourSum(nums LIST OF INTEGER, target INTEGER) RETURNS LIST OF LIST OF INTEGER  
        SORT nums IN NON DECREASING ORDER  
        SET n TO the LENGTH OF nums  
        SET quadruplets TO an empty list  
        
        FOR i FROM zero TO n MINUS three MINUS one  
            IF i GREATER THAN zero AND element at position i of nums EQUALS element at position i MINUS one of nums  
                CONTINUE to the next iteration of the outer loop  
            END IF  
            
            FOR j FROM i PLUS one TO n MINUS two MINUS one  
                IF j GREATER THAN i PLUS one AND element at position j of nums EQUALS element at position j MINUS one of nums  
                    CONTINUE to the next iteration of the inner loop  
                END IF  
                
                SET left TO j PLUS one  
                SET right TO n MINUS one  
                
                WHILE left LESS THAN right  
                    SET total TO element at position i of nums PLUS element at position j of nums PLUS element at position left of nums PLUS element at position right of nums  
                    
                    IF total EQUALS target  
                        APPEND list consisting of element at position i of nums, element at position j of nums, element at position left of nums, and element at position right of nums TO quadruplets  
                        
                        WHILE left LESS THAN right AND element at position left of nums EQUALS element at position left PLUS one of nums  
                            INCREMENT left BY one  
                        END WHILE  
                        
                        WHILE left LESS THAN right AND element at position right of nums EQUALS element at position right MINUS one of nums  
                            DECREMENT right BY one  
                        END WHILE  
                        
                        INCREMENT left BY one  
                        DECREMENT right BY one  
                    
                    ELSE IF total LESS THAN target  
                        INCREMENT left BY one  
                    
                    ELSE  
                        DECREMENT right BY one  
                    END IF  
                END WHILE  
            END FOR  
        END FOR  
        
        RETURN quadruplets  
    END FUNCTION  
END CLASS
