CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze AS List Of List Of Integer, ball AS List Of Integer, hole AS List Of Integer  
        SET directions TO list containing tuples of zero AND one AND r THEN one AND zero AND d THEN zero AND negative one AND l THEN negative one AND zero AND u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO CONVERT ball TO tuple  
        SET hole TO CONVERT hole TO tuple  
        SET pq TO list containing a tuple of zero AND empty string AND start  
        SET visited TO an empty set  
        
        WHILE pq is not empty  
            REMOVE AND RETURN the smallest element FROM pq ASSIGN TO dist AND path AND position unpacked AS x AND y  
            
            IF position at x AND y EQUALS hole THEN  
                RETURN path  
            END IF  
            
            IF position at x AND y IS IN visited THEN  
                CONTINUE to next iteration of WHILE loop  
            END IF  
            
            ADD position at x AND y TO visited  
            
            FOR dx AND dy AND direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND element at position new_x plus dx of maze AT position new_y plus dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    IF position new_x AND new_y EQUALS hole THEN  
                        BREAK FROM WHILE loop  
                    END IF  
                END WHILE  
                
                IF position new_x AND new_y IS NOT IN visited THEN  
                    INSERT INTO pq a tuple of dist PLUS count AND path concatenated WITH direction AND tuple of new_x AND new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
