CLASS Solution  
    FUNCTION countSmaller(nums)  
        SET sorted_nums TO a sorted list of unique elements of nums  
        SET rank TO a mapping from each value val IN sorted_nums TO its index plus one  
        SET bit TO a list of zeros with the length of the LENGTH OF sorted_nums plus one  
        
        FUNCTION get_sum(idx)  
            SET res TO zero  
            WHILE idx IS GREATER THAN zero  
                INCREMENT res BY the element at position idx of bit  
                DECREMENT idx BY the bitwise AND of idx AND the negative of idx interpreted as a number  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        FUNCTION update(idx, delta)  
            WHILE idx IS LESS THAN the LENGTH OF bit  
                INCREMENT the element at position idx of bit BY delta  
                INCREMENT idx BY the bitwise AND of idx AND the negative of idx interpreted as a number  
            END WHILE  
        END FUNCTION  
        
        SET counts TO an empty list  
        FOR each num IN the reverse order of nums  
            SET r TO the value of num IN rank  
            APPEND the result of calling get_sum with r minus one TO counts  
            CALL update with arguments r and one  
        END FOR  
        
        RETURN the reverse order of counts  
    END FUNCTION  
END CLASS
