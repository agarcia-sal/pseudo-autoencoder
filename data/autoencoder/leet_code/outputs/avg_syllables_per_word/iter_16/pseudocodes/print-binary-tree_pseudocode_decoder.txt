CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node dot left TO NEW TreeNode WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node dot right TO NEW TreeNode WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH p dot left AND q dot left AND is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node IS none THEN  
                RETURN zero  
            END IF  
            RETURN one PLUS THE GREATER VALUE OF getHeight WITH node dot left AND getHeight WITH node dot right  
        END FUNCTION  

        SET height TO getHeight WITH root  
        SET width TO TWO RAISED TO THE POWER OF height MINUS one  
        SET res TO A LIST CONTAINING height ELEMENTS EACH OF WHICH IS A LIST OF width ELEMENTS EACH OF WHICH IS AN EMPTY STRING  

        FUNCTION placeNode(node, row, left, right)  
            IF node IS none THEN  
                RETURN  
            END IF  
            SET mid TO THE INTEGER DIVISION OF THE SUM OF left AND right BY two  
            SET the element at position mid of the element at position row of res TO THE STRING REPRESENTATION OF node dot val  
            CALL placeNode WITH node dot left, row PLUS one, left, mid  
            CALL placeNode WITH node dot right, row PLUS one, mid, right  
        END FUNCTION  

        CALL placeNode WITH root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
