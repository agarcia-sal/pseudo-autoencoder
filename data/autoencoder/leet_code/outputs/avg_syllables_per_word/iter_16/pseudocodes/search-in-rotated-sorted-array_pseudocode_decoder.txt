CLASS Solution  
    FUNCTION search(nums AS list of integers, target AS integer) RETURNS integer  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two  
            
            IF element at position mid of nums EQUALS target  
                RETURN mid  
            END IF  
            
            IF element at position left of nums IS LESS THAN OR EQUAL TO element at position mid of nums  
                IF element at position left of nums IS LESS THAN OR EQUAL TO target AND target IS LESS THAN element at position mid of nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF element at position mid of nums IS LESS THAN target AND target IS LESS THAN OR EQUAL TO element at position right of nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
