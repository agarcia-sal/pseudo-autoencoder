CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO an empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND element at the last position of stack IS GREATER THAN num  
                REMOVE the element at the last position FROM stack  
            END WHILE  
            APPEND num TO stack  
            SET count TO count PLUS the LENGTH OF stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
