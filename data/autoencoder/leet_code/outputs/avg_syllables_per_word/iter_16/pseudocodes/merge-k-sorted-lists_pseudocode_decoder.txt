CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO create new ListNode WITH THE ELEMENT at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting FROM position one TO the end  
        SET node TO create new ListNode WITH val  
        SET the element at position next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the element at position val of p1 EQUALS the element at position val of p2 AND is_same_list WITH PARAMETERS the element at position next of p1 the element at position next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS self lists  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists  
            IF lst IS NOT none  
                CALL heappush ON min_heap WITH a TUPLE of the element at position val of lst AND i AND lst  
            END IF  
        END FOR  
        SET dummy TO create new ListNode  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            CALL heappop ON min_heap AND DESTRUCTURE INTO val i node  
            SET the element at position next of current TO node  
            SET current TO the element at position next of current  
            IF the element at position next of node IS NOT none  
                CALL heappush ON min_heap WITH a TUPLE of the element at position val of the element at position next of node AND i AND the element at position next of node  
            END IF  
        END WHILE  
        RETURN the element at position next of dummy  
    END FUNCTION  
END CLASS
