CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO the list of integer values one thousand nine hundred zero five hundred four hundred one hundred ninety fifty forty ten nine five four one  
        SET syms TO the list of Roman numeral strings em see dee see cee ex see el ex ex eye ex vee eye vee eye  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero holds  
            FOR each iteration FROM one TO the integer division of num BY element at position i of val  
                APPEND element at position i of syms TO roman_numeral  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
