CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized with element at position zero of values  
    SET i TO one  
    ASSIGN queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater than zero HOLD  
        SET node TO the element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node dot left TO new TreeNode initialized with element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node dot right TO new TreeNode initialized with element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of is_same_tree with parameter p dot left and q dot left AND is_same_tree with parameter p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(self, root)  
        SET first TO None  
        SET second TO None  
        SET prev TO None  
        SET curr TO root  
        WHILE curr IS NOT None HOLD  
            IF curr dot left IS None THEN  
                IF prev IS NOT None AND prev dot val IS GREATER THAN curr dot val THEN  
                    IF first IS None THEN  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr dot right  
            ELSE  
                SET pred TO curr dot left  
                WHILE pred dot right IS NOT None AND pred dot right IS NOT EQUALS curr HOLD  
                    SET pred TO pred dot right  
                END WHILE  
                IF pred dot right IS None THEN  
                    SET pred dot right TO curr  
                    SET curr TO curr dot left  
                ELSE  
                    SET pred dot right TO None  
                    IF prev IS NOT None AND prev dot val IS GREATER THAN curr dot val THEN  
                        IF first IS None THEN  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr dot right  
                END IF  
            END IF  
        END WHILE  
        IF first IS NOT None AND second IS NOT None THEN  
            SWAP the values of first dot val and second dot val  
        END IF  
    END FUNCTION  
END CLASS
