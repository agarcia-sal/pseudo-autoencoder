CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO DEQUEUE OF empty  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE ELEMENT OBTAINED BY REMOVING THE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST WITH PARAMETERS self root low high  
        IF root IS none  
            RETURN none  
        END IF  
        IF root dot val IS LESS THAN low  
            RETURN self dot trimBST WITH PARAMETERS root dot right low high  
        END IF  
        IF root dot val IS GREATER THAN high  
            RETURN self dot trimBST WITH PARAMETERS root dot left low high  
        END IF  
        SET root dot left TO self dot trimBST WITH PARAMETERS root dot left low high  
        SET root dot right TO self dot trimBST WITH PARAMETERS root dot right low high  
        RETURN root  
    END FUNCTION  
END CLASS
