CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL helper_function to PUSH element at position zero of nums at position index INTO min_heap with index and zero  
            SET max_value TO the GREATER OF max_value AND element at position zero of nums at position index  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO a list containing negative one hundred thousand AND one hundred thousand  
        WHILE min_heap is not empty  
            REMOVE the smallest element from min_heap and ASSIGN its components TO min_value, list_index, element_index  
            IF max_value MINUS min_value is LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one is LESS THAN the LENGTH OF element at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of element at position list_index of nums  
                CALL helper_function to PUSH next_value, list_index, element_index PLUS one INTO min_heap  
                SET max_value TO the GREATER OF max_value AND next_value  
            ELSE  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
