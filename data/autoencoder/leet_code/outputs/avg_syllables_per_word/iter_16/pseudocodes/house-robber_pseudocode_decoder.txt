CLASS Solution  
    FUNCTION rob(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET element at position zero of dp TO the element at position zero of nums  
        SET element at position one of dp TO the GREATER value BETWEEN the element at position zero of nums AND the element at position one of nums  
        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET element at position index of dp TO the GREATER value BETWEEN the element at position index MINUS one of dp AND the SUM of the element at position index MINUS two of dp PLUS the element at position index of nums  
        END FOR  
        RETURN the element at the LAST position of dp  
    END FUNCTION  
END CLASS
