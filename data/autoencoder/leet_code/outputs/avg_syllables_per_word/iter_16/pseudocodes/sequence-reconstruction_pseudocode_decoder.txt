CLASS Solution  
    FUNCTION sequenceReconstruction(nums, sequences)  
        SET n TO the LENGTH OF nums  
        SET graph TO an empty mapping from keys to lists  
        SET indegree TO a list of zeroes with LENGTH n PLUS one  
        
        FOR each seq IN sequences  
            FOR index FROM zero TO the LENGTH OF seq MINUS one MINUS one  
                SET u TO element at position index of seq  
                SET v TO element at position index PLUS one of seq  
                APPEND v TO graph at key u  
                INCREMENT element at position v of indegree BY one  
            END FOR  
        END FOR  
        
        SET queue TO an empty double ended queue  
        FOR i FROM one TO n  
            IF element at position i of indegree EQUALS zero  
                APPEND i TO queue  
            END IF  
        END FOR  
        
        IF the LENGTH OF queue IS NOT EQUAL TO one  
            RETURN False  
        END IF  
        
        SET index TO zero  
        WHILE queue IS NOT empty  
            IF the LENGTH OF queue IS NOT EQUAL TO one  
                RETURN False  
            END IF  
            
            SET node TO the element removed from the front of queue  
            IF node IS NOT EQUAL TO element at position index of nums  
                RETURN False  
            END IF  
            INCREMENT index BY one  
            
            FOR each neighbor IN graph at key node  
                DECREMENT element at position neighbor of indegree BY one  
                IF element at position neighbor of indegree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN index IS EQUAL TO n  
    END FUNCTION  
END CLASS
