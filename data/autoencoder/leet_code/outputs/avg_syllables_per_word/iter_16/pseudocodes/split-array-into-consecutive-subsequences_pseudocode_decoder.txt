CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO the frequency count of each number in nums  
        SET end_with TO an empty frequency count of subsequences ending with particular numbers  
        
        FOR each num IN nums  
            IF the frequency count of num in num_count EQUALS zero  
                CONTINUE to next iteration  
            END IF  
            
            IF the frequency count of subsequences ending with number that is one less than num in end_with IS greater than zero  
                DECREMENT the frequency count of subsequences ending with number that is one less than num in end_with BY one  
                INCREMENT the frequency count of subsequences ending with num in end_with BY one  
            ELSE IF the frequency count of number that is one greater than num in num_count IS greater than zero AND the frequency count of number that is two greater than num in num_count IS greater than zero  
                DECREMENT the frequency count of number that is one greater than num in num_count BY one  
                DECREMENT the frequency count of number that is two greater than num in num_count BY one  
                INCREMENT the frequency count of subsequences ending with number that is two greater than num in end_with BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT the frequency count of num in num_count BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
