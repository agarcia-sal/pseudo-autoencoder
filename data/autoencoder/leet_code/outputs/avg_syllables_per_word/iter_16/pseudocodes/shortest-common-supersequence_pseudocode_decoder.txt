CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a two dimensional list WITH dimensions m PLUS one BY n PLUS one FILLED WITH zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                    SET element at row i column j of dp TO the element at row i MINUS one column j MINUS one of dp PLUS one  
                ELSE  
                    SET element at row i column j of dp TO the GREATER OF the element at row i MINUS one column j of dp AND the element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at row i MINUS one column j of dp GREATER THAN the element at row i column j MINUS one of dp THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements in scs  
        
        SET result_string TO the concatenation of all elements in scs in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
