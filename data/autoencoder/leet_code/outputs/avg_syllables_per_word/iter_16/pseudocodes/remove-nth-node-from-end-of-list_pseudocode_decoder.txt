CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val property of the current object TO val  
        SET the next property of the current object TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO A new ListNode OBJECT INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values FROM position one TO the end  
        SET node TO A new ListNode OBJECT INITIALIZED WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the expression that the val property of p1 EQUALS the val property of p2 AND the result of calling is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO A new ListNode OBJECT INITIALIZED WITH zero  
        SET the next property of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR index FROM zero TO n PLUS one  
            SET first TO the next property of first  
        END FOR  
        WHILE first IS NOT None  
            SET first TO the next property of first  
            SET second TO the next property of second  
        END WHILE  
        SET the next property of second TO the next property of the next property of second  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
