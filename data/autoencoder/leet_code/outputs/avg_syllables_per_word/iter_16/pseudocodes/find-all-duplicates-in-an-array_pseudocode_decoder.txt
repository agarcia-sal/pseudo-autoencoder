CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position index of nums IS NOT EQUAL TO element at position element at position index of nums MINUS one of nums  
                SET temporary_variable TO element at position index of nums  
                SET element at position index of nums TO element at position temporary_variable MINUS one of nums  
                SET element at position temporary_variable MINUS one of nums TO temporary_variable  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each position INDEX AND value VALUE IN nums WITH positions FROM zero TO the LENGTH OF nums MINUS one  
            IF VALUE IS NOT EQUAL TO INDEX PLUS one  
                APPEND VALUE TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
