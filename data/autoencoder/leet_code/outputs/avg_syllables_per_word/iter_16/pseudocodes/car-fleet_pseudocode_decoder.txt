CLASS Solution  
    FUNCTION carFleet(target INTEGER, position LIST OF INTEGER, speed LIST OF INTEGER) RETURNS INTEGER  
        SET list_of_position_speed_pairs TO sorted pairs of position and speed in descending order by position  
        SET list_of_time_to_target TO empty list  
        FOR each pair_of_position_and_speed IN list_of_position_speed_pairs  
            SET position_value TO position element of pair_of_position_and_speed  
            SET speed_value TO speed element of pair_of_position_and_speed  
            SET time_value TO target MINUS position_value DIVIDED BY speed_value  
            APPEND time_value TO list_of_time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time IN list_of_time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
