CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET head TO new ListNode with value equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO end of values  
        SET node TO new ListNode with value equal to val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN False  
    END IF  
    RETURN logical AND of element val at p1 EQUALS element val at p2 AND recursive call to is_same_list with element next at p1 AND element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node is not none  
                APPEND element val at node TO stack  
                SET node TO element next at node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO result of pushIntoStack with l1  
        SET stack2 TO result of pushIntoStack with l2  
        
        SET carry TO zero  
        SET result TO none  
        
        WHILE stack1 is not empty OR stack2 is not empty OR carry is not zero  
            IF stack1 is not empty THEN  
                SET val1 TO last element removed from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 is not empty THEN  
                SET val2 TO last element removed from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO new ListNode with value digit  
            SET next of new_node TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
