CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair of values u AND v IN edges  
            APPEND a tuple containing element at position v of scores AND v TO the list at position u of graph  
            APPEND a tuple containing element at position u of scores AND u TO the list at position v of graph  
        END FOR  
        FOR each element node IN graph  
            SET the list at position node of graph TO the three largest elements by first component in the current list at position node of graph  
        END FOR  
        SET max_score TO negative one  
        FOR each pair of values u AND v IN edges  
            FOR each pair of values score1 AND x IN the list at position u of graph  
                FOR each pair of values score2 AND y IN the list at position v of graph  
                    IF the quantity of distinct elements among x y u and v EQUALS four  
                        SET max_score TO the greater of max_score AND the sum of score1 PLUS score2 PLUS element at position u of scores PLUS element at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
