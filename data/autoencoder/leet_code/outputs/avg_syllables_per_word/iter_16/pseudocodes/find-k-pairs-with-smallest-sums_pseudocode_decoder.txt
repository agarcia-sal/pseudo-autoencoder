CLASS Solution  
    FUNCTION kSmallestPairs(nums1 AS List OF integers, nums2 AS List OF integers, k AS integer) RETURNS List OF List OF integers  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO zero THEN  
            RETURN an EMPTY list  
        END IF  
        
        SET min_heap TO an EMPTY list  
        
        FOR each index i FROM zero TO the LESSER OF k MINUS one AND the LENGTH OF nums1 MINUS one  
            INSERT INTO min_heap A tuple CONSISTING OF the SUM OF element at position i of nums1 AND element at position zero of nums2 FOLLOWED BY i FOLLOWED BY zero  
        END FOR  
        
        SET result TO an EMPTY list  
        
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result IS LESS THAN k DO  
            REMOVE the SMALLEST tuple FROM min_heap AND UNPACK AS ignore_value COMMA i COMMA j  
            APPEND A list CONSISTING OF element at position i of nums1 AND element at position j of nums2 TO result  
            
            IF j PLUS one IS LESS THAN the LENGTH OF nums2 THEN  
                INSERT INTO min_heap A tuple CONSISTING OF the SUM OF element at position i of nums1 AND element at position j PLUS one of nums2 FOLLOWED BY i FOLLOWED BY j PLUS one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
