CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values AS list)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element REMOVED FROM the FRONT of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION maxSumBST(root AS Optional TreeNode) RETURNS integer  
        SET self.max_sum TO zero  
          
        FUNCTION helper(node) RETURNS tuple of boolean integer integer integer  
            IF node IS none THEN  
                RETURN True comma zero comma positive infinity comma negative infinity  
            END IF  
            SET left_is_bst, left_sum, left_min, left_max TO helper(the left child of node)  
            SET right_is_bst, right_sum, right_min, right_max TO helper(the right child of node)  
            IF left_is_bst AND right_is_bst AND left_max IS LESS THAN the value of node AND the value of node IS LESS THAN right_min THEN  
                SET current_sum TO left_sum PLUS right_sum PLUS the value of node  
                SET current_min TO THE MINIMUM OF the value of node AND left_min  
                SET current_max TO THE MAXIMUM OF the value of node AND right_max  
                SET self.max_sum TO THE MAXIMUM OF self.max_sum AND current_sum  
                RETURN True comma current_sum comma current_min comma current_max  
            END IF  
            RETURN False comma zero comma positive infinity comma negative infinity  
        END FUNCTION  
          
        CALL helper(root)  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
