CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m INTEGER n INTEGER  
        SET dp TO a two dimensional array with dimensions m by n where each element is one  
        FOR variable i FROM one TO m minus one  
            FOR variable j FROM one TO n minus one  
                SET element at position i of dp and position j TO element at position i minus one of dp and position j PLUS element at position i of dp and position j minus one  
            END FOR  
        END FOR  
        RETURN element at position m minus one of dp and position n minus one  
    END FUNCTION  
END CLASS
