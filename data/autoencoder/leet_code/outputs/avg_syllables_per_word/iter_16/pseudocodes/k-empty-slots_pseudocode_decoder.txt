CLASS Solution  
    FUNCTION kEmptySlots WITH PARAMETERS bulbs LIST OF INTEGER AND k INTEGER  
        SET bloom_days TO an empty list  
        FOR each day AND position IN enumeration of bulbs STARTING FROM one  
            SET i TO the position at which position can be inserted in bloom_days to maintain sorted order  
            IF i is greater than zero AND position MINUS element at position i MINUS one of bloom_days EQUALS k  
                RETURN day  
            END IF  
            IF i is less than the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i to keep it sorted  
        END FOR  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
