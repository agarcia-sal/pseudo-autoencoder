CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each triple consisting of left right and height IN buildings  
            APPEND a tuple of left position negative height and right position TO events  
            APPEND a tuple of right position zero and zero TO events  
        END FOR  
        CALL sort on events to order all elements by their first position then by their second position then by their third position  
        SET result TO a list containing a single list of zero and zero as the initial key point  
        SET max_heap TO a list containing a single tuple of zero and positive infinity representing height zero and unlimited right edge  
        FOR each triple consisting of x position negative height and right edge r IN events  
            WHILE the right edge element at position one of the first tuple in max_heap is less than or equal to x position  
                REMOVE the first tuple from max_heap using heap removal procedure  
            END WHILE  
            IF negative height is not equal to zero  
                ADD the tuple of negative height and right edge r TO max_heap using heap addition procedure  
            END IF  
            IF the height element at position one of the last list in result is not equal to the negation of the height element at position zero of the first tuple in max_heap  
                APPEND a new list containing x position and the negation of the height element at position zero of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        IF the first list in result is equal to the list of zero and zero  
            REMOVE the first list from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
