CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF the RESULT of m PLUS n MINUS one MODULO two IS NOT EQUAL TO zero THEN  
            RETURN False  
        END IF  
        SET dp TO a three dimensional collection of BOOLEAN False values with dimensions m PLUS one BY n PLUS one BY m PLUS n PLUS one  
        SET the element at position zero zero zero of dp TO True  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF the element at position i j PLUS one k of dp IS True OR the element at position i PLUS one j k of dp IS True THEN  
                        IF the element at position i j of grid EQUALS opening parenthesis AND k PLUS one IS LESS THAN m PLUS n PLUS one THEN  
                            SET the element at position i PLUS one j PLUS one k PLUS one of dp TO True  
                        ELSE IF the element at position i j of grid EQUALS closing parenthesis AND k MINUS one IS GREATER THAN OR EQUAL TO zero THEN  
                            SET the element at position i PLUS one j PLUS one k MINUS one of dp TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position m n zero of dp  
    END FUNCTION  
END CLASS
