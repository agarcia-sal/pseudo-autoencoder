CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid IS EMPTY OR element at position zero of grid IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        CALL initialize_2d_list WITH parameters m AND n RETURNING dp  
        
        SET element at position zero at position zero of dp TO element at position zero at position zero of grid  
        
        FOR each j FROM one TO n MINUS one  
            SET element at position zero at position j of dp TO element at position zero at position j MINUS one of dp PLUS element at position zero at position j of grid  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            SET element at position i at position zero of dp TO element at position i MINUS one at position zero of dp PLUS element at position i at position zero of grid  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                SET element at position i at position j of dp TO the MINIMUM OF element at position i MINUS one at position j of dp AND element at position i at position j MINUS one of dp PLUS element at position i at position j of grid  
            END FOR  
        END FOR  
        
        RETURN element at position m MINUS one at position n MINUS one of dp  
    END FUNCTION  
    
    FUNCTION initialize_2d_list(m, n)  
        CREATE an empty list named result  
        FOR each row_index FROM zero TO m MINUS one  
            CREATE a list named row WITH n elements each SET TO zero  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
