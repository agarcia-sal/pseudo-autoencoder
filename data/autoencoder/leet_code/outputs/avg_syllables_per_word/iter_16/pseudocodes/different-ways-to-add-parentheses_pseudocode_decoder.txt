CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND the sum of l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONTAINS only digit characters  
                RETURN a list containing the integer value of sub_expr  
            END IF  

            SET results TO empty list  
            FOR each index i FROM zero TO the length of sub_expr MINUS one  
                IF the character at position i of sub_expr IS one of the symbols plus or minus or multiplication  
                    SET left_results TO the result of helper applied to the substring from position zero TO the position before i of sub_expr  
                    SET right_results TO the result of helper applied to the substring from position one after i TO the end of sub_expr  
                    EXTEND results BY the outcome of compute applied to left_results, right_results, and the character at position i of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN the result of helper applied to expression  
    END FUNCTION  
END CLASS
