CLASS Solution  
    FUNCTION minInterval(intervals WITH TYPE List of List of integers, queries WITH TYPE List of integers) RETURNS List of integers  
        SET intervals TO intervals SORTED BY the element at position zero of each interval in ascending order  
        SET sorted_queries TO a list containing pairs of each element of queries AND its position in queries sorted BY the first element of each pair in ascending order  
        SET min_heap TO an empty collection supporting retrieval of the minimum element  
        SET result TO a list filled with negative one and having the LENGTH OF queries  
        SET interval_index TO zero  
        FOR each query with its original_index IN sorted_queries  
            WHILE interval_index IS LESS THAN the LENGTH OF intervals AND the element at position zero of element at position interval_index of intervals IS LESS THAN OR EQUAL TO query  
                SET start TO the element at position zero of element at position interval_index of intervals  
                SET end TO the element at position one of element at position interval_index of intervals  
                SET interval_size TO end MINUS start PLUS one  
                INSERT the pair consisting of interval_size AND end INTO min_heap  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT EMPTY AND the second element of the minimum element in min_heap IS LESS THAN query  
                REMOVE the minimum element FROM min_heap  
            END WHILE  
            IF min_heap IS NOT EMPTY  
                SET the element at position original_index of result TO the first element of the minimum element in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
