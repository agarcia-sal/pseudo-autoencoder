CLASS Solution  
    FUNCTION addBoldTag(s string, words list of strings) RETURNS string  
        IF the LENGTH OF words EQUALS zero THEN  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        SET mask TO a list of BOOLEAN values of LENGTH n all set TO False  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s STARTING FROM start  
                IF start EQUALS negative one THEN  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of mask TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the element at position i of mask IS True THEN  
                APPEND the string labeled b tag TO result  
                WHILE i IS LESS THAN n AND the element at position i of mask IS True  
                    APPEND the character at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string labeled closing b tag TO result  
            ELSE  
                APPEND the character at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the CONCATENATION OF all elements of result AS a STRING  
    END FUNCTION  
END CLASS
