CLASS Solution  
    FUNCTION maximumScore(nums AND k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at position k of nums  
        SET max_score TO current_min  
        
        WHILE left IS GREATER THAN zero OR right IS LESS THAN the LENGTH OF nums MINUS one holds  
            IF left EQUALS zero THEN  
                INCREMENT right BY one  
            ELSE IF right EQUALS the LENGTH OF nums MINUS one THEN  
                DECREMENT left BY one  
            ELSE IF element at position left MINUS one of nums IS GREATER THAN OR EQUAL TO element at position right PLUS one of nums THEN  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            
            SET current_min TO the MINIMUM VALUE among current_min AND element at position left of nums AND element at position right of nums  
            SET score TO current_min MULTIPLIED BY the DIFFERENCE of right MINUS left PLUS one  
            SET max_score TO the MAXIMUM VALUE between max_score AND score  
        END WHILE  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
