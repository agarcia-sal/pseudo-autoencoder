CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        CREATE memo AS an empty dictionary  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            
            IF the tuple of index and prev_len IS a KEY IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the tuple of index and prev_len in memo  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index INCLUSIVE  
                IF the character at position index of num EQUALS the character zero THEN  
                    BREAK the current loop  
                END IF  
                
                SET current_num TO the substring from position index TO the position index PLUS length MINUS one of num  
                
                IF prev_len EQUALS zero OR the string current_num IS GREATER THAN OR EQUAL TO the substring from position index MINUS prev_len TO position index MINUS one of num THEN  
                    SET count TO count PLUS the RESULT OF dp of index PLUS length AND length  
                    SET count TO count MODULO MOD  
                END IF  
            END FOR  
            
            ASSIGN the VALUE of count TO memo AT the KEY tuple of index AND prev_len  
            RETURN count  
        END FUNCTION  
        
        RETURN the RESULT OF dp with zero AND zero  
    END FUNCTION  
END CLASS
