CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values DOES NOT EXIST OR values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val as element at position zero of values AND left as None AND right as None  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO NEW TreeNode WITH val as element at position i of values AND left as None AND right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO NEW TreeNode WITH val as element at position i of values AND left as None AND right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p DOES NOT EXIST AND q DOES NOT EXIST  
        RETURN True  
    ELSE IF p DOES NOT EXIST OR q DOES NOT EXIST  
        RETURN False  
    ELSE IF val of p IS NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN RESULT OF is_same_tree WITH p.left AND q.left AND RESULT OF is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root DOES NOT EXIST  
            RETURN zero  
        END IF  
        SET queue TO NEW double ended queue WITH element consisting of root AND zero  
        SET max_width TO zero  
        WHILE queue IS NOT EMPTY  
            SET level_length TO LENGTH OF queue  
            SET first_index TO second element OF first element OF queue  
            FOR each number FROM zero TO level_length MINUS one  
                SET node, index TO REMOVE first element FROM queue  
                IF number EQUALS level_length MINUS one  
                    IF max_width LESS THAN index MINUS first_index PLUS one  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left EXISTS  
                    APPEND element consisting of node.left AND index MULTIPLIED BY two TO queue  
                END IF  
                IF node.right EXISTS  
                    APPEND element consisting of node.right AND index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
