CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETERS n  
        SET teams TO list of strings representing numbers from one TO n inclusive  
        WHILE n GREATER THAN one holds  
            FOR index FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position index of teams TO a string formed by an opening parenthesis FOLLOWED BY element at position index of teams FOLLOWED BY a comma FOLLOWED BY element at position n MINUS one MINUS index of teams FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
