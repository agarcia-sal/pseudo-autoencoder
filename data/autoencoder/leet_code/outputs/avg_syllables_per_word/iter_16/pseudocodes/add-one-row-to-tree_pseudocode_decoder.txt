CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO TreeNode(element at position zero of values)  
    SET i TO one  
    SET queue TO a double ended queue initialized as empty  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO TreeNode(element at position i of values)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO TreeNode(element at position i of values)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree(the left child of p, the left child of q) AND the RESULT OF is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth IS EQUAL TO one THEN  
            RETURN TreeNode(val, left EQUALS root)  
        END IF  
        SET queue TO a double ended queue initialized with a tuple containing root and one  
        WHILE queue IS NOT empty  
            SET node, level TO the tuple removed from the front of queue  
            IF level IS EQUAL TO depth MINUS one THEN  
                SET node.left TO TreeNode(val, left EQUALS node.left)  
                SET node.right TO TreeNode(val, right EQUALS node.right)  
            ELSE IF level IS LESS THAN depth MINUS one THEN  
                IF node.left IS NOT none THEN  
                    APPEND a tuple of node.left and level PLUS one TO queue  
                END IF  
                IF node.right IS NOT none THEN  
                    APPEND a tuple of node.right and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
