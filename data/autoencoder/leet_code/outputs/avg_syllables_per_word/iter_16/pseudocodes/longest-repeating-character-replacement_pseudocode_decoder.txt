CLASS Solution  
    FUNCTION characterReplacement(s STRING, k INTEGER)  
        CREATE count AS a mapping from character to integer initialized to zero  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right of s IN count BY one  
            SET max_count TO the GREATER OF max_count AND the element at position right of s IN count  
            
            IF the value of right MINUS left PLUS one MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the element at position left of s IN count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND the value of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
