CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM value IN the SUBLIST of jobDifficulty FROM position i TO the END OF jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR each j FROM i TO n MINUS d PLUS one  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND the ELEMENT at position j of jobDifficulty  
                SET min_difficulty TO the MINIMUM value BETWEEN min_difficulty AND current_max PLUS the RESULT of dp(j PLUS one, d MINUS one)  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
END CLASS
