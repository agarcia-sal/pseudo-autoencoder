CLASS Solution  
    FUNCTION canSeePersonsCount WITH PARAMETER heights  
        SET n TO the LENGTH OF heights  
        SET answer TO list of zeroes of size n  
        SET stack TO empty list  
        FOR index FROM zero TO n MINUS one  
            WHILE stack is NOT empty AND height at position of element at position last element of stack of heights IS LESS THAN height at position index of heights  
                INCREMENT element at position popped element from stack of answer BY one  
            END WHILE  
            IF stack is NOT empty  
                INCREMENT element at position last element of stack of answer BY one  
            END IF  
            APPEND index TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
