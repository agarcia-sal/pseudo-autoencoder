CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR index i FROM zero TO n MINUS one MINUS one  
            PUSH INTO min_heap a tuple containing the element at position i of arr DIVIDED BY the element at position n MINUS one of arr AND the index i AND the index n MINUS one  
        END FOR  
        FOR each iteration FROM zero TO k MINUS one MINUS one  
            POP the smallest tuple FROM min_heap AND ASSIGN variable underscore TO the first element AND variable i TO the second element AND variable j TO the third element  
            IF the result of subtracting one FROM j IS GREATER THAN i THEN  
                PUSH INTO min_heap a tuple containing the element at position i of arr DIVIDED BY the element at position j MINUS one of arr AND the index i AND the index j MINUS one  
            END IF  
        END FOR  
        POP the smallest tuple FROM min_heap AND ASSIGN variable underscore TO the first element AND variable i TO the second element AND variable j TO the third element  
        RETURN a list consisting of the element at position i of arr AND the element at position j of arr  
    END FUNCTION  
END CLASS
