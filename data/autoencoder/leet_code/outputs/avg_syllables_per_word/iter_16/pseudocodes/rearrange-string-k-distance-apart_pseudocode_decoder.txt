CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of each character in s  
        
        SET max_heap TO a list where for each character and its count in freq the pair of negative count and character is appended  
        CONVERT max_heap into a heap structure maintaining the max heap based on frequency  
        
        SET wait_queue TO an empty list to store characters waiting to be placed  
        
        SET result TO an empty list to store the rearranged characters  
        
        WHILE max_heap is not empty  
            EXTRACT the character with the highest frequency and its count from max_heap  
            APPEND the extracted character to result  
            
            ADD the tuple of the extracted count incremented by one and the character to wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first element from wait_queue and call it count and char  
                IF count LESS THAN zero  
                    INSERT the tuple of count and char back into max_heap maintaining the heap property  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result is not equal to the LENGTH OF s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters in result as a single string  
    END FUNCTION  
END CLASS
