CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulted to zero left defaulted to none right defaulted to none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values list  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode constructed with element at position zero of values  
    SET i TO one  
    SET queue TO new double ended queue  
    ADD root TO the end of queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i is less than the length of values AND element at position i of values is not none  
            SET node dot left TO new TreeNode constructed with element at position i of values  
            ADD node dot left TO the end of queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND element at position i of values is not none  
            SET node dot right TO new TreeNode constructed with element at position i of values  
            ADD node dot right TO the end of queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF p dot val is not equal to q dot val  
        RETURN false  
    ELSE  
        RETURN logical AND of the result of calling is_same_tree with p dot left and q dot left AND the result of calling is_same_tree with p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView with parameters self and root optional of type TreeNode  
        IF root is none  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO new double ended queue containing root  
        WHILE queue is not empty  
            SET level_length TO the length of queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO element removed from the front of queue  
                IF i is equal to level_length MINUS one  
                    APPEND node dot val TO right_side_view  
                END IF  
                IF node dot left is not none  
                    ADD node dot left TO the end of queue  
                END IF  
                IF node dot right is not none  
                    ADD node dot right TO the end of queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
