CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF index AND current_sum ARE KEYS IN memo  
                RETURN the value associated with the key consiting of index AND current_sum IN memo  
            END IF  
            SET positive TO dfs with parameters index PLUS one AND current_sum PLUS element at position index of nums AND memo  
            SET negative TO dfs with parameters index PLUS one AND current_sum MINUS element at position index of nums AND memo  
            SET the value associated with the key consiting of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the value associated with the key consiting of index AND current_sum IN memo  
        END FUNCTION  
        RETURN dfs with parameters zero AND zero AND an empty dictionary  
    END FUNCTION  
END CLASS
