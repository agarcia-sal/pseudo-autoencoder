CLASS Solution  
    FUNCTION mincostToHireWorkers(quality AS List OF integer, wage AS List OF integer, k AS integer) RETURNS float  
        SET workers TO the SORTED list of pairs where each pair consists of the quotient of wage element divided by quality element AND the corresponding quality element FOR each corresponding pair of wage and quality elements  

        SET max_heap TO an empty list to represent a max-heap of qualities as negative values  
        SET total_quality TO zero representing the sum of qualities in the current group  
        SET min_cost TO positive infinity representing the minimal total cost found so far  

        FOR each ratio and quality_value IN workers  
            PUSH the NEGATIVE of quality_value INTO max_heap  
            INCREMENT total_quality BY quality_value  
            
            IF the LENGTH OF max_heap is GREATER THAN k  
                REMOVE the element with the HIGHEST quality from max_heap AND ADD its value to total_quality  
            END IF  
            
            IF the LENGTH OF max_heap is EQUAL TO k  
                SET min_cost TO the MINIMUM of min_cost AND the PRODUCT of total_quality AND ratio  
            END IF  
        END FOR  

        RETURN min_cost  
    END FUNCTION  
END CLASS
