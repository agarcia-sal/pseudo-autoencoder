CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO CREATE new TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO REMOVE the element from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO CREATE new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO CREATE new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the value at p IS NOT EQUAL TO the value at q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree CALLED WITH node left of p AND node left of q AND the result of is_same_tree CALLED WITH node right of p AND node right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS none THEN  
                RETURN  
            END IF  
            CALL inorder_traversal WITH node left of node  
            IF the LENGTH OF result IS LESS THAN k THEN  
                APPEND the value at node TO result  
            ELSE  
                IF the absolute value of the value at node MINUS target IS LESS THAN the absolute value of the element at position zero of result MINUS target THEN  
                    REMOVE the element from the front of result  
                    APPEND the value at node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH node right of node  
        END FUNCTION  
        SET result TO an empty double ended queue  
        CALL inorder_traversal WITH root  
        RETURN the conversion of result TO a list  
    END FUNCTION  
END CLASS
