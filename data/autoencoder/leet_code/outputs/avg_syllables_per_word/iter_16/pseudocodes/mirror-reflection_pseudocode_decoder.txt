CLASS Solution  
    FUNCTION mirrorReflection WITH PARAMETERS p q  
        CALL FUNCTION gcd WITH PARAMETERS p q AND ASSIGN RESULT TO gcd_value  
        SET lcm TO p MULTIPLIED BY q DIVIDED BY gcd_value  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        IF vertical_reflections MODULO two EQUALS zero THEN  
            RETURN zero  
        ELSE IF horizontal_reflections MODULO two EQUALS one THEN  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
