CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each element in nums  
        INITIALIZE dp AS a list of zeros with LENGTH equal to two RAISED TO the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two RAISED TO the power of the element at position one of count  
                SET the element at position zero of dp TO the element at position zero of dp MODULO MOD  
                CONTINUE to next iteration  
            END IF  
            IF num NOT IN the keys of prime_factors  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO the value corresponding to key num in prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO the bitwise OR of mask and two RAISED TO the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two RAISED TO the power of ten MINUS one DOWN TO zero  
                IF the bitwise AND of m and mask EQUALS zero  
                    SET dp at position bitwise OR of m and mask TO the sum of dp at position bitwise OR of m and mask and the product of dp at position m and the element at position num of count  
                    SET dp at position bitwise OR of m and mask TO dp at position bitwise OR of m and mask MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of all elements of dp starting from position one MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
