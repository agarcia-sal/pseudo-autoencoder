CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN false  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a two dimensional list of boolean values all set to false with m rows and n columns  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AND element at position y of that IS true OR element at position x of board AND element at position y of that DOES NOT EQUAL element at position index of word  
                RETURN false  
            END IF  

            SET element at position x of visited AND element at position y of that TO true  
            FOR each dx, dy IN the set consisting of negative one AND zero, one AND zero, zero AND negative one, zero AND one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one)  
                    RETURN true  
                END IF  
            END FOR  
            SET element at position x of visited AND element at position y of that TO false  
            RETURN false  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero)  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
