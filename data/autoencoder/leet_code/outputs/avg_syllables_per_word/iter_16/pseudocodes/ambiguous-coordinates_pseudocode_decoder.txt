CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR each d FROM one TO n  
                SET left TO the SUBSTRING from position one TO position d of frag  
                SET right TO the SUBSTRING from position d PLUS one TO the END of frag  
                IF left DOES NOT START WITH the CHARACTER zero OR left EQUALS zero AND right DOES NOT END WITH the CHARACTER zero  
                    YIELD the CONCATENATION of left PLUS the CHARACTER period IF d IS NOT EQUAL TO n OTHERWISE the EMPTY STRING PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING from position two TO the SECOND TO LAST CHARACTER of s  
        RETURN list CONSTRUCTED FROM the FOLLOWING  
            FOR each i FROM one TO the LENGTH OF s MINUS one  
                FOR each a IN the RESULT OF make WITH PARAMETER the SUBSTRING from position one TO position i of s  
                    FOR each b IN the RESULT OF make WITH PARAMETER the SUBSTRING from position i PLUS one TO the END of s  
                        YIELD the CONCATENATION of the LEFT PARENTHESIS PLUS a PLUS the STRING comma AND SPACE PLUS b PLUS the RIGHT PARENTHESIS  
                    END FOR  
                END FOR  
            END FOR  
    END FUNCTION  
END CLASS
