CLASS Solution  
    FUNCTION waysToDistribute(n INTEGER, k INTEGER) RETURNS INTEGER  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO A new two dimensional collection OF zero WITH dimensions n PLUS one BY k PLUS one  
        SET element at position zero AND position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET product TO j MULTIPLIED BY element at position i MINUS one AND position j of dp  
                SET sum_value TO product PLUS element at position i MINUS one AND position j MINUS one of dp  
                SET element at position i AND position j of dp TO sum_value MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position n AND position k of dp  
    END FUNCTION  
END CLASS
