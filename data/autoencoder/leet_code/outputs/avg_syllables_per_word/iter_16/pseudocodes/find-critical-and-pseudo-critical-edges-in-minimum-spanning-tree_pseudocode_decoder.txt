CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers FROM zero TO n MINUS one  
        SET self.rank TO list containing n elements all equal to one  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUAL TO u  
            SET element at position u of self.parent TO the result of calling find with element at position u of self.parent of self  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find with u of self  
        SET rootV TO the result of calling find with v of self  
          
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND edge IN enumeration of edges  
            APPEND i TO edge  
        END FOR  
          
        SORT edges ACCORDING TO the element at position two of each edge IN ascending order  
          
        FUNCTION mst(exclude IS none, include IS none)  
            CREATE uf AS new instance of UnionFind with n  
            SET weight TO zero  
              
            IF include IS NOT none  
                CALL union on uf WITH first element of include AND second element of include  
                INCREMENT weight BY third element of include  
            END IF  
              
            FOR each u, v, w, idx IN edges  
                IF idx IS EQUAL TO exclude  
                    CONTINUE to next iteration  
                END IF  
                IF calling union on uf WITH u AND v RETURNS true  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
              
            SET root TO result of calling find with zero of uf  
            FOR i FROM zero TO n MINUS one  
                IF result of calling find with i of uf IS NOT EQUAL TO root  
                    RETURN positive infinity  
                END IF  
            END FOR  
              
            RETURN weight  
        END FUNCTION  
          
        SET mst_weight TO the result of calling mst without parameters  
          
        SET critical TO empty list  
        SET pseudo_critical TO empty list  
          
        FOR each u, v, w, idx IN edges  
            IF result of calling mst with exclude equal to idx IS GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF result of calling mst with include equal to tuple of u, v, w IS EQUAL TO mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
          
        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
