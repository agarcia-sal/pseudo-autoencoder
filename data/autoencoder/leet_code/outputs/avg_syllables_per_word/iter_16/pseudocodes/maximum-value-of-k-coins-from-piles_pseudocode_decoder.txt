CLASS Solution  
    FUNCTION maxValueOfCoins WITH parameters piles AS list of list of integers AND k AS integer  
        SET n TO the LENGTH OF piles  
        SET dp TO a two dimensional list with n plus one rows and k plus one columns filled with zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i minus one of piles  
                SET current_value TO zero  
                SET max_l TO the MINIMUM OF j AND the LENGTH OF current_pile  
                FOR l FROM zero TO max_l  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY element at position l minus one of current_pile  
                    END IF  
                    SET dp at position i and j TO the MAXIMUM OF dp at position i and j AND the SUM OF dp at position i minus one and j minus l AND current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
