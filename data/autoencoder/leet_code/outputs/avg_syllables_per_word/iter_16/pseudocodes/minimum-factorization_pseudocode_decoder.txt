CLASS Solution  
    FUNCTION smallestFactorization WITH PARAMETER num  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO empty list  
        FOR digit FROM nine DOWN TO two  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        SET result TO the integer value formed by concatenating all elements of factors in order  
        
        IF result GREATER THAN two TO THE POWER OF thirty ONE MINUS one  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
