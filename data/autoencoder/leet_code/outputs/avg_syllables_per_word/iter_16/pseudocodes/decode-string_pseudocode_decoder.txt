CLASS Solution  
    FUNCTION decodeString WITH parameter s  
        SET stack TO an empty list  
        SET a TO an empty string  
        SET d TO an empty string  
        FOR each character IN s  
            IF the character represents a digit  
                SET d TO the concatenation of d and the character  
            ELSE IF the character represents a letter of the alphabet  
                SET a TO the concatenation of a and the character  
            ELSE IF the character EQUALS the left square bracket symbol  
                APPEND the tuple consisting of a and the integer value converted from d TO stack  
                SET a TO an empty string  
                SET d TO an empty string  
            ELSE IF the character EQUALS the right square bracket symbol  
                SET p and n TO the tuple value obtained by removing the last element from stack  
                SET a TO the concatenation of p and a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
