CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    CREATE an empty queue data structure named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE the first element FROM queue AND ASSIGN IT TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node dot left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node dot right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN None  
        END IF  
        SET max_val TO the maximum element IN nums  
        SET max_index TO the index OF max_val IN nums  
        SET root TO A NEW TreeNode WITH max_val  
        SET root dot left TO self dot constructMaximumBinaryTree(the elements from position zero TO max_index MINUS one of nums)  
        SET root dot right TO self dot constructMaximumBinaryTree(the elements from position max_index PLUS one TO the last position of nums)  
        RETURN root  
    END FUNCTION  
END CLASS
