CLASS TreeNode  
    FUNCTION __init__ WITH parameters val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element REMOVED FROM THE LEFT END OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p dot left AND q dot left AND is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers WITH parameters root  
        FUNCTION dfs WITH parameters node current_number  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS node dot val  
            IF node dot left IS none AND node dot right IS none  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs WITH node dot left AND current_number  
            SET right_sum TO dfs WITH node dot right AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs WITH root AND zero  
    END FUNCTION  
END CLASS
