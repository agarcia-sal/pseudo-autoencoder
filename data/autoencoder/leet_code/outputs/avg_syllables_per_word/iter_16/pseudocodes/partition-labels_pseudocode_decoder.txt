CLASS Solution  
    FUNCTION partitionLabels(s)  
        CREATE last_occurrence AS an EMPTY DICTIONARY  
        FOR each index AND character IN enumeration OF s  
            SET the value at key character IN last_occurrence TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        CREATE partition_sizes AS an EMPTY LIST  
        
        FOR each index AND character IN enumeration OF s  
            SET end TO the GREATER VALUE BETWEEN end AND the value at key character IN last_occurrence  
            
            IF index EQUALS end  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
