CLASS Solution  
    FUNCTION maxCandies(status LIST OF INTEGER, candies LIST OF INTEGER, keys LIST OF LIST OF INTEGER, containedBoxes LIST OF LIST OF INTEGER, initialBoxes LIST OF INTEGER) RETURNS INTEGER  
        SET boxes_we_have TO a new set containing elements of initialBoxes  
        SET keys_we_have TO an empty set  
        SET total_candies TO zero  
        SET progress_made TO true  
        WHILE progress_made holds  
            SET progress_made TO false  
            FOR each box IN the set converted from boxes_we_have to a list  
                IF the element at position box of status EQUALS one OR the element box is contained in keys_we_have  
                    INCREMENT total_candies BY the element at position box of candies  
                    ADD each element of the element at position box of keys TO keys_we_have  
                    ADD each element of the element at position box of containedBoxes TO boxes_we_have  
                    REMOVE the element box FROM boxes_we_have  
                    SET progress_made TO true  
                END IF  
            END FOR  
        END WHILE  
        RETURN total_candies  
    END FUNCTION  
END CLASS
