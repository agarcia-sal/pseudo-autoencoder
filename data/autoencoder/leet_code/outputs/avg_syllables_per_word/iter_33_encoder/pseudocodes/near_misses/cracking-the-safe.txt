CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO empty set  
        SET sequence TO empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string conversion of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH substring of edge FROM position two TO position length of edge  
                    APPEND string conversion of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string consisting of n MINUS one repetitions of character zero  
        CALL dfs WITH start_node  
        RETURN concatenation of all elements in sequence FOLLOWED BY start_node  
    END FUNCTION  
END CLASS
