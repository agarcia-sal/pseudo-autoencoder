CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO empty mapping FROM each integer FROM zero TO n MINUS one TO empty list  
        FOR each u, v, w IN flights  
            APPEND pair OF v AND w TO list AT element at position u of graph  
        END FOR  

        SET min_heap TO list CONTAINING one tuple of zero cost, minus one stops, and src city  
        SET min_cost TO mapping FROM src TO minus one  

        WHILE min_heap IS NOT empty  
            REMOVE and RETURN the smallest element FROM min_heap INTO cost, stops, city  

            IF city EQUALS dst  
                RETURN cost  
            END IF  

            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE to next iteration  
            END IF  

            FOR each neighbor, price IN list AT element at position city of graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  

                IF neighbor NOT IN keys of min_cost OR next_stops LESS THAN element at neighbor key of min_cost OR next_cost LESS THAN element at position zero of the first tuple in min_heap  
                    SET element at neighbor key of min_cost TO next_stops  
                    ADD tuple OF next_cost, next_stops, neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
