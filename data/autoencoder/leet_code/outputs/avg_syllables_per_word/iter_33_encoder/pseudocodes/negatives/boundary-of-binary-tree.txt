CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the property val TO val  
        SET the property left TO left  
        SET the property right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the collection values IS EMPTY  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode object initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE the collection queue IS NOT EMPTY  
        SET node TO the element obtained by removing the first element of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET the property left of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the property left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET the property right of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the property right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF the property val of p IS NOT EQUALS TO the property val of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with the property left of p and the property left of q AND the result of is_same_tree called with the property right of p and the property right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree with parameter root  
        IF root IS no value  
            RETURN an empty list  
        END IF  

        FUNCTION is_leaf with parameter node  
            RETURN node IS no value IS False AND the property left of node IS no value AND the property right of node IS no value  
        END FUNCTION  

        SET left_boundary TO an empty list  
        SET current TO the property left of root  
        WHILE current IS NOT no value  
            IF the result of is_leaf called with current IS False  
                APPEND the property val of current TO left_boundary  
            END IF  
            IF the property left of current IS NOT no value  
                SET current TO the property left of current  
            ELSE  
                SET current TO the property right of current  
            END IF  
        END WHILE  

        SET right_boundary TO an empty list  
        SET current TO the property right of root  
        WHILE current IS NOT no value  
            IF the result of is_leaf called with current IS False  
                APPEND the property val of current TO right_boundary  
            END IF  
            IF the property right of current IS NOT no value  
                SET current TO the property right of current  
            ELSE  
                SET current TO the property left of current  
            END IF  
        END WHILE  

        SET leaves TO an empty list  

        FUNCTION collect_leaves with parameter node  
            IF node IS no value  
                RETURN  
            END IF  
            IF the result of is_leaf called with node  
                APPEND the property val of node TO leaves  
            END IF  
            CALL collect_leaves with the property left of node  
            CALL collect_leaves with the property right of node  
        END FUNCTION  

        CALL collect_leaves with root  

        IF the result of is_leaf called with root  
            SET leaves TO an empty list  
        END IF  

        RETURN a new list formed by concatenating a list containing the property val of root followed by left_boundary followed by leaves followed by the reverse of right_boundary  
    END FUNCTION  
END CLASS
