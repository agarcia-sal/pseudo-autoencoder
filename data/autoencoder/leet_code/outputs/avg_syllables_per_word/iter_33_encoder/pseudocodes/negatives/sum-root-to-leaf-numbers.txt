CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN a null reference  
    END IF  
    SET root TO a new instance of TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        REMOVE the element from the front of queue and ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT a null reference THEN  
            SET the left child of node TO a new instance of TreeNode constructed with the element at position i of values  
            APPEND the left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT a null reference THEN  
            SET the right child of node TO a new instance of TreeNode constructed with the element at position i of values  
            APPEND the right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS a null reference AND q IS a null reference THEN  
        RETURN True  
    ELSE IF p IS a null reference OR q IS a null reference THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with the left child of p and the left child of q AND the result of is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node IS a null reference THEN  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS the value of node  
            IF the left child of node IS a null reference AND the right child of node IS a null reference THEN  
                RETURN current_number  
            END IF  
            SET left_sum TO the result of dfs with the left child of node and current_number  
            SET right_sum TO the result of dfs with the right child of node and current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN the result of dfs with root and zero  
    END FUNCTION  
END CLASS
