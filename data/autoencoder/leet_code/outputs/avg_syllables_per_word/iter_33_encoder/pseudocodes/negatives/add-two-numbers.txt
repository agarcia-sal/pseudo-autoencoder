CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the end  
        SET node TO a new ListNode with val  
        SET the element next to p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the evaluation of the element val at p1 EQUALS the element val at p2 AND is_same_list with the element next at p1 and the element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO a new ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry GREATER THAN zero  
            IF l1 IS NOT None  
                SET val1 TO the element val at l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the element val at l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY ten  
            SET the element next to current TO a new ListNode with the remainder of total DIVIDED BY ten  
            SET current TO the element next at current  
            
            IF l1 IS NOT None  
                SET l1 TO the element next at l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO the element next at l2  
            END IF  
        END WHILE  
        
        RETURN the element next at dummy_head  
    END FUNCTION  
END CLASS
