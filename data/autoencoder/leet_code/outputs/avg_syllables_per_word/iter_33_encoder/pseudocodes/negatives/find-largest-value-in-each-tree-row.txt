CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO None right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value AS element at position one of values  
    SET i TO one  
    CREATE an empty double-ended queue named queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        REMOVE and RETURN the element from the left of queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode with value AS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode with value AS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p THE left child of q) AND is_same_tree(the right child of p THE right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self root)  
        IF root IS None THEN  
            RETURN an empty list  
        END IF  
        CREATE an empty double-ended queue named queue initialized with root as its only element  
        CREATE an empty list named largest_values  
        WHILE the LENGTH OF queue IS GREATER THAN zero DO  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR index FROM one TO level_size DO  
                REMOVE and RETURN the element from the left of queue AND ASSIGN TO node  
                IF max_value IS LESS THAN the value of node THEN  
                    SET max_value TO the value of node  
                END IF  
                IF the left child of node IS NOT None THEN  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node IS NOT None THEN  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
