CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value of the instance property val TO val  
        SET the value of the instance property next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN values starting from position two TO the end  
        SET node TO new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list with parameters the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList with parameter head  
        IF head EQUALS None OR the next property of head EQUALS None  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with a value of negative five thousand AND one  
        SET the next property of dummy TO head  
        SET current TO the next property of head  
        SET the next property of head TO None  
        WHILE current is not None  
            SET next_node TO the next property of current  
            SET prev TO dummy  
            WHILE the next property of prev is not None AND the value property of the next property of prev LESS THAN the value property of current  
                SET prev TO the next property of prev  
            END WHILE  
            SET the next property of current TO the next property of prev  
            SET the next property of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
