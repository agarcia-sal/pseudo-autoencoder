CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with value element at position one of values  
    SET p TO head  
    FOR val IN the elements of values starting from position two to the end  
        SET node TO a new ListNode with value val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND the result of calling is_same_list with the next element of p1 and the next element of p2  
END FUNCTION

CLASS Solution  
    FUNCTION removeNthFromEnd(self head n)  
        SET dummy TO a new ListNode with value zero  
        SET the next element of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the next element of first  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO the next element of first  
            SET second TO the next element of second  
        END WHILE  
        SET the next element of second TO the next element of the next element of second  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
