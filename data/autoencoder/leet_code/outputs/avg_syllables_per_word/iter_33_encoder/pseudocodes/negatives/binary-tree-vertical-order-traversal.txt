CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and left equal to None and right equal to None  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE element from the left of queue and ASSIGN to node  
        IF i IS LESS THAN LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.left TO new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.right TO new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        SET column_table TO empty dictionary with default value empty list  
        SET queue TO empty double ended queue  
        APPEND tuple of root and zero TO queue  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            REMOVE element from the left of queue and ASSIGN to node and column_index  
            IF node IS NOT None THEN  
                APPEND node.val TO list at key column_index in column_table  
                IF column_index IS LESS THAN min_column THEN  
                    SET min_column TO column_index  
                END IF  
                IF column_index IS GREATER THAN max_column THEN  
                    SET max_column TO column_index  
                END IF  
                APPEND tuple of node.left and column_index MINUS one TO queue  
                APPEND tuple of node.right and column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO empty list  
        FOR x FROM min_column TO max_column  
            APPEND element at key x in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
