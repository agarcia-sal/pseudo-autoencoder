CLASS ListNode  
    FUNCTION __init__ with parameters val next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS NOT PRESENT  
        RETURN NONE  
    END IF  
    SET head TO A new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the end of values  
        SET node TO A new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 IS NONE AND p2 IS NONE  
        RETURN TRUE  
    END IF  
    IF p1 IS NOT PRESENT OR p2 IS NOT PRESENT  
        RETURN FALSE  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween with parameters head left right  
        IF head IS NOT PRESENT OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO A new ListNode with zero  
        SET the next property of dummy TO head  
        SET prev TO dummy  
        FOR variable from zero TO left MINUS one MINUS one  
            SET prev TO the next property of prev  
        END FOR  
        SET current TO the next property of prev  
        FOR variable from zero TO right MINUS left MINUS one  
            SET next_node TO the next property of current  
            SET the next property of current TO the next property of next_node  
            SET the next property of next_node TO the next property of prev  
            SET the next property of prev TO next_node  
        END FOR  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
