CLASS TreeNode  
    FUNCTION __init__(self, val equals zero, left equals none, right equals none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the first element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET the left property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET the right property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left property of p and the left property of q AND the result of calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN none  
        END IF  
        SET max_val TO the maximum element in nums  
        SET max_index TO the index of max_val in nums  
        SET root TO a new TreeNode initialized with max_val  
        SET the left property of root TO the result of calling constructMaximumBinaryTree on the sublist of nums from the first element up to the element before max_index  
        SET the right property of root TO the result of calling constructMaximumBinaryTree on the sublist of nums from the element after max_index to the last element  
        RETURN root  
    END FUNCTION  
END CLASS
