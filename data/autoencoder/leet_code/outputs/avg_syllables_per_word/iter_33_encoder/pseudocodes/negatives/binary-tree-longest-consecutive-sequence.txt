CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS NOT present  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the left end of queue  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS NOT present AND q IS NOT present  
        RETURN True  
    ELSE IF p IS NOT present OR q IS NOT present  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root IS NOT present  
            RETURN zero  
        END IF  
        
        FUNCTION dfs(node, parent, current_length)  
            IF node IS NOT present  
                RETURN current_length  
            END IF  
            
            IF parent IS present AND value of node EQUALS parent PLUS one  
                SET current_length TO current_length PLUS one  
            ELSE  
                SET current_length TO one  
            END IF  
            
            SET left_length TO dfs(left child of node, value of node, current_length)  
            SET right_length TO dfs(right child of node, value of node, current_length)  
            
            RETURN the maximum value among current_length, left_length, and right_length  
        END FUNCTION
        
        RETURN dfs(root, None, zero)  
    END FUNCTION  
END CLASS
