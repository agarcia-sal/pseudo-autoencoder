CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val defaulted to zero left defaulted to none right defaulted to none  
        SET the value attribute of self TO val  
        SET the left attribute of self TO left  
        SET the right attribute of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A new TreeNode CREATED WITH the element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        REMOVE and ASSIGN the first element of queue TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute of node TO A new TreeNode CREATED WITH the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute of node TO A new TreeNode CREATED WITH the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value attribute of p IS NOT EQUALS TO the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree WITH PARAMETERS the left attribute of p and the left attribute of q AND is_same_tree WITH PARAMETERS the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues WITH PARAMETERS self root target k  
        FUNCTION inorder_traversal WITH PARAMETER node  
            IF node IS none  
                RETURN  
            END IF  
            CALL inorder_traversal WITH the left attribute of node  
            IF the LENGTH OF result IS LESS THAN k  
                APPEND the value attribute of node TO result  
            ELSE  
                IF the ABSOLUTE VALUE OF the value attribute of node MINUS target IS LESS THAN the ABSOLUTE VALUE OF the element at position one of result MINUS target  
                    REMOVE the first element FROM result  
                    APPEND the value attribute of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH the right attribute of node  
        END FUNCTION  
        CREATE an empty double ended queue named result  
        CALL inorder_traversal WITH root  
        RETURN the list CONVERTED FROM result  
    END FUNCTION  
END CLASS
