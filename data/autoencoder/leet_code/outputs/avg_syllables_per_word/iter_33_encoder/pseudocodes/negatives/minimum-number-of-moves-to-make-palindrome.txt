CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s AS string  
        SET s TO list of characters from s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        WHILE the LENGTH OF s GREATER THAN one  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one DECREMENT BY one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP the element at position j of s WITH the element at position j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE the element at position zero FROM s  
                    REMOVE the element at the last position FROM s  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            IF no matching character was found in the FOR loop  
                REMOVE the element at position zero FROM s  
            END IF  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
