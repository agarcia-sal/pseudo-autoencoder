CLASS Solution  
    FUNCTION minEatingSpeed WITH PARAMETERS piles AND h  
        FUNCTION canFinish WITH PARAMETER k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the result of ceiling of pile DIVIDED BY k  
            END FOR  
            RETURN hours LESS THAN OR EQUAL TO h  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the maximum element of piles  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the result of left PLUS right DIVIDED BY two ignoring remainder  
            IF canFinish WITH PARAMETER mid  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
