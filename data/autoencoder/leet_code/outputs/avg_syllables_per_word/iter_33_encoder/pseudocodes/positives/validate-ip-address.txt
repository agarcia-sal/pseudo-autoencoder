CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF only digits  
                SET num TO the integer conversion of block  
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND the string conversion of num EQUALS block  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        FUNCTION is_valid_ipv6_block(block)  
            IF the length of block IS GREATER THAN OR EQUAL TO one AND the length of block IS LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block FROM base sixteen TO integer  
                    RETURN true  
                CATCH conversion failure  
                    RETURN false  
                END TRY  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        SET parts_ipv4 TO the result of splitting queryIP BY the period character  
        SET parts_ipv6 TO the result of splitting queryIP BY the colon character  
  
        IF the length of parts_ipv4 EQUALS four AND for every element part IN parts_ipv4 the function is_valid_ipv4_block returns true FOR part  
            RETURN the string "IPv4"  
        ELSE IF the length of parts_ipv6 EQUALS eight AND for every element part IN parts_ipv6 the function is_valid_ipv6_block returns true FOR part  
            RETURN the string "IPv6"  
        ELSE  
            RETURN the string "Neither"  
        END IF  
    END FUNCTION  
END CLASS
