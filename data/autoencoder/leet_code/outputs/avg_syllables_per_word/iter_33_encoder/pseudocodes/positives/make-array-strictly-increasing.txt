CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort arr2 in ascending order  
        SET dp TO a dictionary with key negative one mapped to value zero  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num and ops IN the items of dp  
                IF the current num is GREATER THAN last_num THEN  
                    SET the element at key num of new_dp TO the MINIMUM between the current value at key num of new_dp or positive infinity AND ops  
                END IF  
                
                SET idx TO the position after the last occurrence of last_num in arr2 using binary search  
                IF idx is LESS THAN the length of arr2 THEN  
                    SET the element at key element at position idx of arr2 in new_dp TO the MINIMUM between the current value at that key or positive infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp is empty THEN  
                RETURN negative one  
            END IF  
            
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM value among all values in dp  
    END FUNCTION  
END CLASS
