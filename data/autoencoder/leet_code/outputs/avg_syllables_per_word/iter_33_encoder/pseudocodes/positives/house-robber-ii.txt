CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the COLLECTION houses EQUALS the EMPTY COLLECTION  
                RETURN zero  
            END IF  
            IF the LENGTH OF the COLLECTION houses EQUALS one  
                RETURN the ELEMENT at POSITION one OF the COLLECTION houses  
            END IF  
            SET dp TO a COLLECTION OF zeroes WITH the LENGTH OF the COLLECTION houses  
            SET the ELEMENT at POSITION one OF dp TO the ELEMENT at POSITION one OF the COLLECTION houses  
            SET the ELEMENT at POSITION two OF dp TO the GREATER value BETWEEN the ELEMENT at POSITION one OF the COLLECTION houses AND the ELEMENT at POSITION two OF the COLLECTION houses  
            FOR i FROM three TO the LENGTH OF the COLLECTION houses  
                SET the ELEMENT at POSITION i OF dp TO the GREATER value BETWEEN the ELEMENT at POSITION i MINUS one OF dp AND the SUM of the ELEMENT at POSITION i MINUS two OF dp AND the ELEMENT at POSITION i OF the COLLECTION houses  
            END FOR  
            RETURN the ELEMENT at the LAST POSITION OF dp  
        END FUNCTION  
        IF the COLLECTION nums EQUALS the EMPTY COLLECTION  
            RETURN zero  
        END IF  
        IF the LENGTH OF the COLLECTION nums EQUALS one  
            RETURN the ELEMENT at POSITION one OF the COLLECTION nums  
        END IF  
        RETURN the GREATER value BETWEEN the RESULT OF rob_linear WITH the COLLECTION nums EXCLUDING the LAST ELEMENT AND the RESULT OF rob_linear WITH the COLLECTION nums EXCLUDING the FIRST ELEMENT  
    END FUNCTION  
END CLASS
