CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a double ended queue CONTAINING a tuple of zero zero zero one zero  
        SET visited TO a set CONTAINING a tuple of zero zero zero one  

        WHILE the queue is not empty  
            REMOVE from the left of queue and UNPACK INTO r1 c1 r2 c2 steps  
            
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND element at position r1 of grid AT position c1 PLUS one EQUALS zero AND element at position r2 of grid AT position c2 PLUS one EQUALS zero AND tuple of r1 c1 PLUS one r2 c2 PLUS one NOT IN visited  
                ADD tuple of r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND tuple of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one of grid AT position c1 EQUALS zero AND element at position r2 PLUS one of grid AT position c2 EQUALS zero AND tuple of r1 PLUS one c1 r2 PLUS one c2 NOT IN visited  
                ADD tuple of r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND tuple of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one of grid AT position c1 EQUALS zero AND element at position r1 PLUS one of grid AT position c2 EQUALS zero AND tuple of r1 c1 r1 PLUS one c1 NOT IN visited  
                ADD tuple of r1 c1 r1 PLUS one c1 TO visited  
                APPEND tuple of r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at position r1 of grid AT position c1 PLUS one EQUALS zero AND element at position r2 of grid AT position c2 PLUS one EQUALS zero AND tuple of r1 c1 r1 c1 PLUS one NOT IN visited  
                ADD tuple of r1 c1 r1 c1 PLUS one TO visited  
                APPEND tuple of r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
