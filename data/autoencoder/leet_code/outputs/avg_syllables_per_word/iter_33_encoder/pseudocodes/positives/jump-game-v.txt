CLASS Solution  
    FUNCTION maxJumps(arr parameter named arr OF TYPE List OF Integer, parameter named d OF TYPE Integer) RETURNS Integer  
        FUNCTION dp(parameter named i OF TYPE Integer) RETURNS Integer  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF the value of i PLUS x LESS THAN the length of arr THEN  
                    IF the element at position i PLUS x of arr GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    IF the sum of one PLUS the result of dp with argument i PLUS x GREATER THAN max_jumps THEN  
                        SET max_jumps TO the sum of one PLUS the result of dp with argument i PLUS x  
                    END IF  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF the value of i MINUS x GREATER THAN OR EQUAL TO zero THEN  
                    IF the element at position i MINUS x of arr GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    IF the sum of one PLUS the result of dp with argument i MINUS x GREATER THAN max_jumps THEN  
                        SET max_jumps TO the sum of one PLUS the result of dp with argument i MINUS x  
                    END IF  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET max_value TO zero  
        FOR i FROM zero TO the length of arr MINUS one  
            SET current TO dp with argument i  
            IF current GREATER THAN max_value THEN  
                SET max_value TO current  
            END IF  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
