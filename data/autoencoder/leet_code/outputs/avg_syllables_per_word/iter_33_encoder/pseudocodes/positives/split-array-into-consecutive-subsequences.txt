CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO a mapping of each number in nums to the count of its occurrences  
        SET end_with TO a mapping of each number to the count of subsequences ending with that number  
        
        FOR each num IN nums  
            IF the count of num in num_count EQUALS zero  
                CONTINUE to next iteration  
            END IF  
            
            IF the count of subsequences ending with the number which is one less than num in end_with GREATER THAN zero  
                DECREMENT the count of subsequences ending with the number which is one less than num in end_with BY one  
                INCREMENT the count of subsequences ending with num in end_with BY one  
            ELSE IF the count of num plus one in num_count GREATER THAN zero AND the count of num plus two in num_count GREATER THAN zero  
                DECREMENT the count of num plus one in num_count BY one  
                DECREMENT the count of num plus two in num_count BY one  
                INCREMENT the count of subsequences ending with num plus two in end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the count of num in num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
