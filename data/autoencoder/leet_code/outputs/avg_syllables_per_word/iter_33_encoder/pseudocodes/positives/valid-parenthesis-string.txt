CLASS Solution  
    FUNCTION checkValidString(s)  
        SET low TO zero  
        SET high TO zero  
        FOR each char IN s  
            IF char EQUALS the character representing an open parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF char EQUALS the character representing a closing parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            IF low LESS THAN zero  
                SET low TO zero  
            END IF  
        END FOR  
        RETURN the result of checking if low EQUALS zero  
    END FUNCTION  
END CLASS
