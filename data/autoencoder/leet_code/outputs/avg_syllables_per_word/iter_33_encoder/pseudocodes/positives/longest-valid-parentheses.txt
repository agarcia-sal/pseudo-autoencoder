CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO list containing the value negative one  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF character char EQUALS the opening parenthesis symbol  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND index i TO stack  
                ELSE  
                    SET difference TO index i MINUS the last element IN stack  
                    SET max_length TO the larger value BETWEEN max_length AND difference  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
