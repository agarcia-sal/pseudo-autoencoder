CLASS Solution  
    FUNCTION longestPath(parent, s)  
        SET tree TO new mapping with default as empty list  
        FOR index FROM zero TO the LENGTH OF parent MINUS one  
            SET p TO element at position index of parent  
            IF p NOT EQUALS negative one  
                APPEND index TO element at key p of tree  
            END IF  
        END FOR  
  
        SET self.result TO one  
  
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
  
            FOR each child IN element at key node of tree  
                SET child_length TO CALL dfs WITH argument child  
  
                IF character at position child of s NOT EQUALS character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
  
            SET self.result TO the greater value between self.result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
  
        CALL dfs WITH argument zero  
  
        RETURN self.result  
    END FUNCTION  
END CLASS
