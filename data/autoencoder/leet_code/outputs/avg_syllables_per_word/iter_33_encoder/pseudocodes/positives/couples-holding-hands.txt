CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        SET position TO an empty dictionary  
        FOR each index i FROM zero TO the LENGTH OF row MINUS one  
            SET person TO element at position i of row  
            SET element at key person IN position TO i  
        END FOR  
        SET swaps TO zero  
        FOR each index i FROM zero TO the LENGTH OF row MINUS one STEP two  
            SET current_person TO element at position i of row  
            SET correct_pair TO current_person PLUS one IF current_person IS EVEN ELSE current_person MINUS one  
            SET adjacent_person TO element at position i PLUS one of row  
            IF adjacent_person NOT EQUALS correct_pair  
                SET swap_idx TO the value at key correct_pair IN position  
                SET element at position swap_idx of row TO adjacent_person  
                SET element at position i PLUS one of row TO correct_pair  
                SET value at key adjacent_person IN position TO swap_idx  
                SET value at key correct_pair IN position TO i PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
