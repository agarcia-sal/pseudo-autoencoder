CLASS Solution  
    FUNCTION getSum(a b)  
        SET maximum TO the largest thirty two bits integer  
        SET mask TO a number representing thirty two bits all set to one  
        
        WHILE b NOT EQUALS zero holds  
            SET carry TO the result of the bitwise AND between a and b MULTIPLIED BY two  
            SET a TO the result of the bitwise XOR between a and b ANDed with mask  
            SET b TO the result of the bitwise AND between carry and mask  
        END WHILE  
        
        IF a LESS THAN OR EQUAL TO maximum holds  
            RETURN a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR between a and mask  
        END IF  
    END FUNCTION  
END CLASS
