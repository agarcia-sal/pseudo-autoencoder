CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO list of strings representing numbers from one TO n inclusive  
        WHILE n GREATER THAN one holds  
            FOR each i FROM zero TO n DIVIDED BY two MINUS one  
                SET teams at position i TO open parenthesis concatenated WITH element at position i of teams concatenated WITH comma concatenated WITH element at position n MINUS one MINUS i of teams concatenated WITH close parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
