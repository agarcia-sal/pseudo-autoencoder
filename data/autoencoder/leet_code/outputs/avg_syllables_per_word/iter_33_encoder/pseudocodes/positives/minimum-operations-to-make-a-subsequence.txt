CLASS Solution  
    FUNCTION minOperations(target, arr)  
        SET index_map TO empty mapping from element to position  
        FOR index FROM zero TO the LENGTH OF target MINUS one  
            SET the element at position index of target TO value  
            SET element at position value of index_map TO index  
        END FOR  
        
        SET transformed_arr TO empty list  
        FOR each element value IN arr  
            IF element value IN index_map  
                APPEND element at position value of index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO empty list  
        FOR each number num IN transformed_arr  
            SET position pos TO the insertion position in lis where num can be placed to keep lis sorted in ascending order  
            IF position pos EQUALS the LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO the LENGTH OF lis  
        SET minimum_operations TO the LENGTH OF target MINUS lcs_length  
        RETURN minimum_operations  
    END FUNCTION  
END CLASS
