CLASS Solution  
    FUNCTION hIndex(citations)  
        SET n TO the LENGTH OF citations  
        SET left TO zero  
        SET right TO n MINUS one  
        SET h_index TO zero  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS left MINUS right DIVIDED BY two  
            IF element at position mid of citations GREATER THAN OR EQUAL TO n MINUS mid  
                SET h_index TO n MINUS mid  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN h_index  
    END FUNCTION  
END CLASS
