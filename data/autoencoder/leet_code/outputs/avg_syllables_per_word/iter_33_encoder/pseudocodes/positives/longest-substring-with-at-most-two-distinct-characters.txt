CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        ASSIGN reference TO a new mapping that returns zero for any key  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each position right FROM zero TO the length of s MINUS one  
            INCREMENT the count of the element at position right of s in the mapping BY one  
            
            WHILE the number of keys in the mapping GREATER THAN two  
                DECREMENT the count of the element at position left of s in the mapping BY one  
                IF the count of the element at position left of s in the mapping EQUALS zero  
                    REMOVE the element at position left of s FROM the mapping  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the maximum BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
