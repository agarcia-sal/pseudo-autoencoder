CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO a two-dimensional collection with m plus one rows and n plus one columns filled with zero values  
        FOR each s IN strs  
            SET count_0 TO the number of times zero character appears in s  
            SET count_1 TO the number of times one character appears in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET value_with_current TO the element at position i minus count_0 of dp at position j minus count_1 PLUS one  
                    SET element at position i of dp at position j TO the greater value between the element at position i of dp at position j and value_with_current  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position m of dp at position n  
    END FUNCTION  
END CLASS
