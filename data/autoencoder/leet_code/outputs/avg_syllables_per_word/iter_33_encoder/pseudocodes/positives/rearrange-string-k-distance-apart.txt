CLASS Solution  
    FUNCTION rearrangeString(s string k integer) RETURNS string  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  

        SET freq TO the frequency mapping of each character in s  

        SET max_heap TO an empty list  
        FOR each character and count IN freq  
            APPEND a pair of negative count PLUS character TO max_heap  
        END FOR  
        CONVERT max_heap INTO a heap structure  

        SET wait_queue TO an empty list  

        SET result TO an empty list  

        WHILE max_heap is not empty  
            EXTRACT the top element from max_heap into count and character  
            APPEND character TO result  

            APPEND a pair of count PLUS one PLUS character TO wait_queue  

            IF the length of wait_queue EQUALS k  
                REMOVE the first element from wait_queue into count and character  
                IF count LESS THAN zero  
                    ADD the pair of count PLUS character TO max_heap as a heap element  
                END IF  
            END IF  
        END WHILE  

        IF the length of result NOT EQUALS the length of s  
            RETURN an empty string  
        END IF  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
