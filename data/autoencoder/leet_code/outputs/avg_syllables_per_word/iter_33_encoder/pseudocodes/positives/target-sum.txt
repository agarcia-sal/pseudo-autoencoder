CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the PAIR of index AND current_sum IS IN memo  
                RETURN the VALUE in memo ASSOCIATED WITH the PAIR of index AND current_sum  
            END IF  
            SET positive TO the RESULT OF dfs CALLED WITH index INCREMENTED BY one AND current_sum INCREMENTED BY the ELEMENT at position index OF nums AND memo  
            SET negative TO the RESULT OF dfs CALLED WITH index INCREMENTED BY one AND current_sum DECREMENTED BY the ELEMENT at position index OF nums AND memo  
            SET the VALUE in memo ASSOCIATED WITH the PAIR of index AND current_sum TO positive PLUS negative  
            RETURN the VALUE in memo ASSOCIATED WITH the PAIR of index AND current_sum  
        END FUNCTION  
        RETURN the RESULT OF dfs CALLED WITH zero AND zero AND an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
