CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s EQUALS reverse of s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num k  
            IF num EQUALS zero  
                RETURN the string zero  
            END IF  
            SET digits TO an empty list  
            WHILE num is greater than zero  
                APPEND the remainder of num DIVIDED BY k converted to integer TO digits  
                SET num TO the integer division of num BY k  
            END WHILE  
            RETURN concatenation of the string conversion of elements in the reverse order of digits  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD the string conversion of i  
                END FOR  
            ELSE  
                SET start TO ten raised to the power of integer division of (length MINUS one) BY two  
                SET end TO ten raised to the power of integer division of (length PLUS one) BY two  
                FOR half FROM start TO one LESS THAN end  
                    SET half_str TO the string conversion of half  
                    IF length MODULO two EQUALS zero  
                        SET full TO concatenation of half_str AND reverse of half_str  
                    ELSE  
                        SET full TO concatenation of half_str AND reverse of substring of half_str from the second last character TO the first character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers LESS THAN n  
            FOR each pal IN generate_palindromes with parameter length  
                SET num TO the integer conversion of pal  
                IF is_palindrome with the argument of to_base_k with parameters num AND k RETURNS true  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the sum of all elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
