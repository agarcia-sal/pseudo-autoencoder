CLASS Solution  
    FUNCTION topKFrequent(words AS List OF string, k AS integer) RETURNS List OF string  
        SET count TO the result of counting frequencies of each word in words  
        SET heap TO an empty list  
        FOR each word AND frequency IN the items of count  
            APPEND a tuple consisting of the negation of frequency AND the word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap structure organizing elements by the first then second element in ascending order  
        SET result TO an empty list  
        FOR index FROM one TO k  
            REMOVE the smallest element from heap AND APPEND the word part of that element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
