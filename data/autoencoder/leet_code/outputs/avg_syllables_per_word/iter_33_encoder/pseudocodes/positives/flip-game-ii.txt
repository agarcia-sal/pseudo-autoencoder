CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF the element state IS IN the dictionary memo THEN  
                RETURN the element at key state of the dictionary memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO position index PLUS one of state EQUALS two plus signs THEN  
                    SET new_state TO the concatenation of the substring from position zero TO position index of state PLUS two minus signs PLUS the substring from position index PLUS two TO the end of state  
                    
                    IF NOT the value returned by calling can_win_helper WITH new_state THEN  
                        SET the element at key state of the dictionary memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the element at key state of the dictionary memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the value returned by calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
