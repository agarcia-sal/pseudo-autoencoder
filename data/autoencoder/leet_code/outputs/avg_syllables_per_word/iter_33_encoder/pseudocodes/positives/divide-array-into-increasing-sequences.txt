CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums parameter, k parameter)  
        SET freq TO the result of counting the frequency of each element in nums parameter  
        SET max_freq TO the greatest value among all values in freq  
        IF the length of nums parameter DIVIDED BY k parameter IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
