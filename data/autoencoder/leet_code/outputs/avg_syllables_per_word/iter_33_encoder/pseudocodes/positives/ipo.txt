CLASS Solution  
    FUNCTION findMaximizedCapital(k INT w INT profits LIST OF INT capital LIST OF INT) RETURNS INT  
        SET projects TO list of pairs each containing element at position index of capital AND element at position index of profits FOR index FROM zero TO the LENGTH OF capital MINUS one  
        SORT projects BY the first element of each pair in ascending order  
        
        SET max_heap TO empty list  
        
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR loop_variable FROM one TO k  
            
            WHILE i LESS THAN n AND the first element of element at position i of projects LESS THAN OR EQUAL TO w  
                SET negative_profit TO negative of the second element of element at position i of projects  
                ADD negative_profit TO max_heap USING heap insertion  
                INCREMENT i BY one  
            END WHILE  
            
            IF the LENGTH OF max_heap EQUALS zero  
                BREAK the loop  
            END IF  
            
            SET most_profitable_project TO the result of removing the smallest element from max_heap using heap removal  
            INCREMENT w BY negative of most_profitable_project  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
