CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self root TO a list containing each integer from zero to size MINUS one  
        SET self rank TO a list containing size times the integer one  
    END FUNCTION  
  
    FUNCTION find(self, x)  
        IF element at position x of self root EQUALS x  
            RETURN x  
        END IF  
        SET element at position x of self root TO the result of calling self find with argument element at position x of self root  
        RETURN element at position x of self root  
    END FUNCTION  
  
    FUNCTION union(self, x, y)  
        SET rootX TO result of calling self find with argument x  
        SET rootY TO result of calling self find with argument y  
        IF rootX NOT EQUALS rootY  
            IF element at position rootX of self rank GREATER THAN element at position rootY of self rank  
                SET element at position rootY of self root TO rootX  
            ELSE IF element at position rootX of self rank LESS THAN element at position rootY of self rank  
                SET element at position rootX of self root TO rootY  
            ELSE  
                SET element at position rootY of self root TO rootX  
                INCREMENT element at position rootX of self rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION accountsMerge(self, accounts)  
        SET email_to_id TO an empty dictionary  
        SET uf TO a new instance of UnionFind initialized with the length of accounts  
  
        FOR index FROM zero TO length of accounts MINUS one  
            SET the tuple with first element ignored and second element captured as emails TO element at position index of accounts  
            FOR each email IN emails  
                IF email NOT IN email_to_id  
                    ASSIGN email_to_id at email TO index  
                ELSE  
                    CALL uf union with arguments index and value at key email of email_to_id  
                END IF  
            END FOR  
        END FOR  
  
        SET merged_emails TO an empty dictionary with default value as a list  
  
        FOR each email and account_id IN items of email_to_id  
            SET root_id TO result of calling uf find with argument account_id  
            APPEND email TO list at key root_id of merged_emails  
        END FOR  
  
        SET result TO an empty list  
  
        FOR each account_id and emails IN items of merged_emails  
            SET name TO element at position zero of element at position account_id of accounts  
            SET sorted_emails TO emails sorted in ascending order  
            APPEND a list constructed by concatenating name and sorted_emails TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
