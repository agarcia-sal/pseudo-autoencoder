CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS nothing OR the LENGTH OF board EQUALS zero OR the LENGTH OF element at position zero of board EQUALS zero  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        SET directions TO a list containing the pairs of values representing the eight neighbors each pair consisting of minus one AND minus one, minus one AND zero, minus one AND one, zero AND minus one, zero AND one, one AND minus one, one AND zero, AND one AND one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND (element at position ni of board AT position nj EQUALS one OR element at position ni of board AT position nj EQUALS three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board AT position j EQUALS one  
                    IF live_neighbors IS LESS THAN two OR live_neighbors IS GREATER THAN three  
                        SET element at position i of board AT position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three  
                        SET element at position i of board AT position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board AT position j EQUALS two  
                    SET element at position i of board AT position j TO one  
                ELSE IF element at position i of board AT position j EQUALS three  
                    SET element at position i of board AT position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
