CLASS Solution  
    FUNCTION maximumANDSum(nums numSlots)  
        FUNCTION dp(index slots)  
            IF index EQUALS the LENGTH OF nums THEN  
                RETURN zero  
            END IF  
            
            SET max_and_sum TO zero  
            SET slots_list TO a LIST containing the elements of slots  
            
            FOR i FROM zero TO numSlots MINUS one  
                IF the element at position i of slots_list LESS THAN two THEN  
                    INCREMENT the element at position i of slots_list BY one  
                    SET and_sum TO the element at position index of nums AND the value i PLUS one PLUS the RESULT OF dp with arguments index PLUS one and the TUPLE of slots_list  
                    SET max_and_sum TO the GREATER VALUE between max_and_sum AND and_sum  
                    DECREMENT the element at position i of slots_list BY one  
                END IF  
            END FOR  
            
            RETURN max_and_sum  
        END FUNCTION  
        
        SET initial_slots TO the TUPLE of zero REPEATED numSlots TIMES  
        RETURN the RESULT OF dp with arguments zero AND initial_slots  
    END FUNCTION  
END CLASS
