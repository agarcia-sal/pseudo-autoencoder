CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL push element at position zero of list at position index of nums index index zero TO min_heap  
            SET max_value TO greater value BETWEEN max_value AND element at position zero of list at position index of nums  
        END FOR  
        
        SET smallest_range TO positive infinity  
        SET result TO list containing negative one hundred thousand AND positive one hundred thousand  
        
        WHILE min_heap is not empty  
            CALL pop from min_heap AND ASSIGN the returned tuple TO min_value list_index element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            
            IF element_index PLUS one LESS THAN the LENGTH OF list at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums  
                CALL push tuple of next_value list_index element_index PLUS one TO min_heap  
                SET max_value TO greater value BETWEEN max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
