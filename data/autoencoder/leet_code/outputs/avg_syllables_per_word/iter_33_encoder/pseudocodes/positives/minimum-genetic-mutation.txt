CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO a set created from the elements of bank  
        SET q TO a double ended queue initialized with a pair containing start and zero  
        SET mp TO a mapping where the key A maps to the sequence T C G the key T maps to the sequence A C G the key C maps to the sequence A T G and the key G maps to the sequence A T C  
        WHILE the queue q is not empty  
            REMOVE the first element from the queue q and assign the first part to t and the second part to step  
            IF the string t EQUALS the string end  
                RETURN step  
            END IF  
            FOR each index i and character v at the position i in the string t  
                FOR each character j in the sequence mapped from v in mp  
                    SET next TO the concatenation of the substring from position zero TO the position before i of t PLUS the character j PLUS the substring from the position after i TO the last position of t  
                    IF the string next IS AN ELEMENT OF the set s  
                        APPEND the pair containing next and step PLUS one TO the queue q  
                        REMOVE the string next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
