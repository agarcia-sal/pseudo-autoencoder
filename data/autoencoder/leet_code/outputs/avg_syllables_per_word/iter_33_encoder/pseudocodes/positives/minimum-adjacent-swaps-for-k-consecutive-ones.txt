CLASS Solution  
    FUNCTION minMoves(nums k)  
        SET positions TO an empty list  
        FOR each index num IN nums with index i starting from zero  
            IF num EQUALS one  
                APPEND i TO positions  
            END IF  
        END FOR  

        FUNCTION calculate_cost(start end)  
            SET mid TO start PLUS end DIVIDED BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                INCREMENT cost BY the absolute value of element at position i of positions MINUS median PLUS mid MINUS i  
            END FOR  
            RETURN cost  
        END FUNCTION  

        SET min_cost TO infinity  
        SET limit TO the length of positions MINUS k PLUS one  
        FOR i FROM zero TO limit MINUS one  
            SET current_cost TO calculate_cost(i i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
