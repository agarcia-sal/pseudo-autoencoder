CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM the position AFTER i TO one LESS THAN the LENGTH OF s  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s NOT EQUALS the element at position j of s2  
                    YIELD the string formed by concatenating the substring from start TO position i MINUS one of s PLUS the element at position j of s PLUS the substring from position i PLUS one TO position j MINUS one of s PLUS the element at position i of s PLUS the substring from position j PLUS one TO the end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN reference TO a double-ended queue containing a pair consisting of s1 and zero  
        ASSIGN reference TO a set containing s1  
        
        WHILE the queue is not empty  
            SET s AND swaps TO the first element REMOVED FROM the queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the result of calling neighbors with s  
                IF neighbor is not in the set of visited elements  
                    ADD neighbor TO the set of visited elements  
                    APPEND a pair consisting of neighbor and swaps PLUS one TO the queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
