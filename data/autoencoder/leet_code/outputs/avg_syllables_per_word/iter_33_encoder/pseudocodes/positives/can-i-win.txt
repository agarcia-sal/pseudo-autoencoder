CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        SET memo TO an empty dictionary  
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers IS IN memo  
                RETURN the value mapped by the pair of current_total AND used_numbers from memo  
            END IF  
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the number one shifted LEFT BY number IS EQUALS zero  
                    IF the sum of current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(the sum of current_total PLUS number, the bitwise OR of used_numbers AND the number one shifted LEFT BY number)  
                        SET memo at key of the pair of current_total AND used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET memo at key of the pair of current_total AND used_numbers TO false  
            RETURN false  
        END FUNCTION  
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
