CLASS Solution  
    FUNCTION findKthNumber(m n k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                IF x DIVIDED BY i LESS THAN n  
                    INCREMENT count BY x DIVIDED BY i  
                ELSE  
                    INCREMENT count BY n  
                END IF  
            END FOR  
            IF count GREATER THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two  
            IF enough(mid)  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
