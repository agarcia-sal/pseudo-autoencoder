CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n  
        SET MOD TO one billion PLUS seven  
        SET a TO six  
        SET b TO six  
        FOR each integer index FROM two TO n  
            SET new_a TO three MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET new_b TO two MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET a TO new_a MODULO MOD  
            SET b TO new_b MODULO MOD  
        END FOR  
        SET result TO a PLUS b  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
