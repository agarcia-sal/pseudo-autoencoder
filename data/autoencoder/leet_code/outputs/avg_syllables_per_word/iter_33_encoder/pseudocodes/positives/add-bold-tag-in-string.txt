CLASS Solution  
    FUNCTION addBoldTag s words  
        IF the LENGTH OF words EQUALS zero  
            RETURN s  
        END IF  

        SET n TO the LENGTH OF s  
        SET mask TO a list of False with LENGTH equal to n  

        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF the FIRST occurrence of word IN s STARTING FROM start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of mask TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  

        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF the element at position i of mask EQUALS True  
                APPEND the string starting-bold-tag to result  
                WHILE i LESS THAN n AND the element at position i of mask EQUALS True  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string ending-bold-tag to result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
