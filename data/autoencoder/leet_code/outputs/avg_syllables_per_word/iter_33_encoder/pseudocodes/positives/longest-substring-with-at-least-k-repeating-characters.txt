CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        ASSIGN reference TO a collection that COUNTS each element IN s  
        FOR each character IN the counted collection  
            IF the FREQUENCY of character LESS THAN k  
                RETURN the MAXIMUM value AMONG the RESULTS of CALLING longestSubstring with each substring RESULTING from SPLITTING s BY character AND k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
