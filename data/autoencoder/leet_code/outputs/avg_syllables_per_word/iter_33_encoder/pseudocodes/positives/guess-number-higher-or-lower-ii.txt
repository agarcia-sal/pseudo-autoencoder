CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET minimum cost TO infinity  
            FOR pivot FROM left TO right  
                SET left cost TO dp of left AND pivot MINUS one  
                SET right cost TO dp of pivot PLUS one AND right  
                SET cost TO pivot PLUS maximum of left cost AND right cost  
                SET minimum cost TO minimum of minimum cost AND cost  
            END FOR  
            RETURN minimum cost  
        END FUNCTION  
        RETURN dp of one AND n  
    END FUNCTION  
END CLASS
