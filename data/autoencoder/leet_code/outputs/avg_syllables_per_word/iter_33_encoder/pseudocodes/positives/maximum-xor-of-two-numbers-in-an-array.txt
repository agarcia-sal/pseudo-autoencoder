CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR i FROM thirty one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO empty set  
            FOR each num IN nums  
                SET prefix TO the value of num DIVIDED BY two RAISED TO THE POWER OF i TAKE the INTEGER PART  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                SET required_value TO current_candidate EQUALS prefix XOR some unknown value  
                IF the value obtained by XOR of current_candidate and prefix IS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK out of the loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
