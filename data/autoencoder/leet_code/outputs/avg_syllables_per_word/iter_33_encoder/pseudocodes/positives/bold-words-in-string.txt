CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of Boolean false values with the length of s  
        FOR each word IN words  
            SET start TO the position in s where word first appears or negative one if not found  
            WHILE start is NOT EQUALS to negative one  
                FOR i FROM start TO start PLUS the length of word MINUS one  
                    SET element at position i of bold TO true  
                END FOR  
                SET start TO the position in s where word appears starting from position start PLUS one or negative one if not found  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the length of s  
            IF element at position i of bold EQUALS true  
                APPEND the string b tag start TO result  
                WHILE i LESS THAN the length of s AND element at position i of bold EQUALS true  
                    APPEND element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string b tag end TO result  
            ELSE  
                APPEND element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements of result  
    END FUNCTION  
END CLASS
