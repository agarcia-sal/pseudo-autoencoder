CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH onto min_heap the tuple of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr the value i and the value n MINUS one  
        END FOR  
        
        FOR temporary_variable FROM zero TO k MINUS one MINUS one  
            POP from min_heap into underscore_variable i j  
            IF the value j MINUS one GREATER THAN i THEN  
                PUSH onto min_heap the tuple of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr the value i and the value j MINUS one  
            END IF  
        END FOR  
        
        POP from min_heap into underscore_variable i j  
        RETURN a list containing the element at position i of arr and the element at position j of arr  
    END FUNCTION  
END CLASS
