CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO an empty list  
        FOR each number IN nums  
            APPEND the string representation of number TO nums_str  
        END FOR  

        FUNCTION compare(x, y)  
            IF the string x concatenated with the string y GREATER THAN the string y concatenated with the string x THEN  
                RETURN negative one  
            ELSE IF the string x concatenated with the string y LESS THAN the string y concatenated with the string x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  

        SORT nums_str with the sorting order determined by the compare function  

        IF the element at position zero of nums_str EQUALS the string zero THEN  
            RETURN the string zero  
        END IF  

        SET result TO an empty string  
        FOR each element IN nums_str  
            CONCATENATE element TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
