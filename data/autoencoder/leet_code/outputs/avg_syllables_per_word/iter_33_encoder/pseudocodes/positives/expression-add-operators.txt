CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS List of strings  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO LENGTH OF num  
                SET operand TO substring from position index TO position i OF num  
                IF LENGTH OF operand GREATER THAN one AND element at position zero OF operand EQUALS character zero  
                    CONTINUE  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, integer value of operand, integer value of operand  
                ELSE  
                    CALL dfs WITH arguments i, path PLUS the word PLUS PLUS operand, value PLUS integer value of operand, integer value of operand  
                    CALL dfs WITH arguments i, path PLUS the word MINUS PLUS operand, value MINUS integer value of operand, NEGATIVE integer value of operand  
                    CALL dfs WITH arguments i, path PLUS the word MULTIPLIED BY PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY integer value of operand, prev MULTIPLIED BY integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL dfs WITH arguments zero, empty string, zero, zero  
        RETURN result  
END CLASS
