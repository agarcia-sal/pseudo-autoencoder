CLASS Solution  
    FUNCTION coinChange(coins AS list of integers, amount AS integer) RETURNS integer  
        SET dp TO a list with the length of amount PLUS one where each element is amount PLUS one  
        SET element at position zero of dp TO zero  
        
        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO zero  
                    SET element at position a of dp TO the smaller value BETWEEN element at position a of dp AND element at position a MINUS coin of dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position amount of dp IS NOT EQUALS TO amount PLUS one  
            RETURN element at position amount of dp  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
