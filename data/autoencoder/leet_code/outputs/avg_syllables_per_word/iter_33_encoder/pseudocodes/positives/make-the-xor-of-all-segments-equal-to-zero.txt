CLASS Solution  
    FUNCTION minChanges(nums parameter, k parameter)  
        SET groups TO a new empty mapping from keys to empty lists  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            SET remainder TO index MODULO k  
            APPEND element at position index of nums TO the list stored at key remainder in groups  
        END FOR  
        
        SET freq TO a list of k elements where each element is a new empty mapping from keys to zero  
        SET min_size TO a list of k elements where each element is positive infinity  
        
        FOR index FROM zero TO k MINUS one  
            FOR each number IN the list stored at key index in groups  
                INCREMENT the mapping at key number in freq at position index BY one  
            END FOR  
            SET the element at position index in min_size TO the LENGTH OF the list stored at key index in groups  
        END FOR  
        
        SET dp TO a list where each element is positive infinity and the number of elements is two RAISED TO THE POWER OF ten  
        SET the element at position zero in dp TO zero  
        
        FOR index FROM zero TO k MINUS one  
            SET new_dp TO a list where each element is positive infinity and the number of elements is two RAISED TO THE POWER OF ten  
            SET min_dp TO the minimum value in dp  
            
            FOR j FROM zero TO two RAISED TO THE POWER OF ten MINUS one  
                SET the element at position j in new_dp TO min_dp PLUS the element at position index in min_size  
                
                FOR each key value number AND its frequency count IN the mapping freq at position index  
                    SET candidate TO the element at position j XOR number in dp PLUS the element at position index in min_size MINUS count  
                    IF candidate LESS THAN the element at position j in new_dp  
                        SET the element at position j in new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            ASSIGN dp reference TO new_dp  
        END FOR  
        
        RETURN the element at position zero in dp  
    END FUNCTION  
END CLASS
