CLASS Solution  
    FUNCTION maximizeXor(nums, queries)  
        SORT nums IN ASCENDING ORDER  
        SET sorted_queries TO the list of pairs of index and query FROM enumerating queries SORTED BY the second element at position one of the query  
        SET result TO a list of the LENGTH OF queries FILLED WITH the value MINUS one  
        SET trie TO an empty dictionary  
        SET num_index TO zero  
        FOR each query_index and pair xi mi IN sorted_queries  
            WHILE num_index LESS THAN the LENGTH OF nums AND element at position num_index of nums LESS THAN OR EQUAL TO mi  
                SET num TO element at position num_index of nums  
                SET node TO trie  
                FOR i FROM thirty-one DOWN TO zero  
                    SET bit TO the value of element at position i of the binary representation of num  
                    IF bit NOT IN node  
                        SET element at position bit of node TO an empty dictionary  
                    END IF  
                    SET node TO element at position bit of node  
                END FOR  
                INCREMENT num_index BY one  
            END WHILE  
            IF trie IS EMPTY  
                CONTINUE TO next iteration of the FOR loop  
            END IF  
            SET node TO trie  
            SET max_xor TO zero  
            FOR i FROM thirty-one DOWN TO zero  
                SET bit TO the value of element at position i of the binary representation of xi  
                SET toggle_bit TO one MINUS bit  
                IF toggle_bit IN node  
                    SET max_xor TO max_xor PLUS the value of one MULTIPLIED BY the value of two RAISED TO THE POWER OF i  
                    SET node TO element at position toggle_bit of node  
                ELSE  
                    SET node TO element at position bit of node  
                END IF  
            END FOR  
            SET element at position query_index of result TO max_xor  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
