CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty collection that maps characters to positions  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN reference of rightmost character in s at position right TO right IN char_map  
            
            IF the SIZE OF char_map GREATER THAN k THEN  
                SET leftmost_char TO the character in char_map WITH the minimum mapped position  
                SET left TO the position mapped to leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the greater value BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
