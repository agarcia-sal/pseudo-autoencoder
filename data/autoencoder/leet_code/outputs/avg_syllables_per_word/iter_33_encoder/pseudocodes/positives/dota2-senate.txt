CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        CREATE empty queue radiant  
        CREATE empty queue dire  
        FOR each index i FROM zero TO the LENGTH OF senate MINUS one  
            SET s TO element at position i of senate  
            IF s EQUALS the character R  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        WHILE radiant is not empty AND dire is not empty  
            SET r_index TO first element removed from radiant  
            SET d_index TO first element removed from dire  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        IF radiant is not empty  
            RETURN the string Radiant  
        ELSE  
            RETURN the string Dire  
        END IF  
    END FUNCTION  
END CLASS
