CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF start MINUS one  
            SET character_at_index TO element at position index of start  
            IF character_at_index NOT EQUALS the character "X"  
                APPEND the pair of character_at_index AND index TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF end MINUS one  
            SET character_at_index TO element at position index of end  
            IF character_at_index NOT EQUALS the character "X"  
                APPEND the pair of character_at_index AND index TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair_of_c1_i1 AND pair_of_c2_i2 IN the pairs FROM start_filtered AND end_filtered  
            SET c1 TO the first element of pair_of_c1_i1  
            SET i1 TO the second element of pair_of_c1_i1  
            SET c2 TO the first element of pair_of_c2_i2  
            SET i2 TO the second element of pair_of_c2_i2  
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character "L" AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character "R" AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
