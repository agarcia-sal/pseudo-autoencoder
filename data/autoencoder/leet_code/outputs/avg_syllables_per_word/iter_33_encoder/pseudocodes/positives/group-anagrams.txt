CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO an empty dictionary that returns an empty list for missing keys  
        FOR each string s IN strs  
            SET sorted_str TO the result of concatenating the characters of s arranged in ascending alphabetical order  
            APPEND s TO the list stored in anagrams with the key sorted_str  
        END FOR  
        RETURN the list of all values contained in anagrams  
    END FUNCTION  
END CLASS
