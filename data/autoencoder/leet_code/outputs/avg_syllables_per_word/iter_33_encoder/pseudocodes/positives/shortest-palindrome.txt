CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            IF check_str EQUALS the string formed by the characters of check_str in reverse order  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FOR i FROM the LENGTH OF s TO zero DECREMENT BY one  
            IF is_palindrome(the substring from position start zero TO position i MINUS one of s) IS true  
                SET to_add TO the string formed by the characters of the substring from position i TO the end of s in reverse order  
                RETURN the concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
