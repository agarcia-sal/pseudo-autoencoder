CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples where each tuple is each row of board  
        SET target TO tuple of tuples where first tuple is one two three and second tuple is four five zero  
        SET neighbors TO dictionary with keys zero mapping to set of one and three one mapping to set of zero two and four two mapping to set of one and five three mapping to set of zero and four four mapping to set of one three and five five mapping to set of two and four  
        
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO deque containing tuple of start zero_pos and zero  
        SET visited TO set containing start  
        
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and ASSIGN to state zero_pos and moves  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN neighbors at zero_pos  
                SET new_state TO list of lists converted from state  
                SET i TO integer division of zero_pos BY three  
                SET j TO remainder of zero_pos DIVIDED BY three  
                SET ni TO integer division of neighbor BY three  
                SET nj TO remainder of neighbor DIVIDED BY three  
                
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state and element at position nj of element at position ni of new_state TO element at position j of element at position i of new_state swapped  
                
                SET new_tuple_state TO tuple of tuples where each tuple is each row of new_state  
                
                IF new_tuple_state is not an element of visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
