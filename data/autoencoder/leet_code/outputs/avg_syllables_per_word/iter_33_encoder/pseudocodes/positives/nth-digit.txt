CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        SET start TO start PLUS the INTEGER RESULT OF the DIVISION of n MINUS one BY length  
        SET digit_index TO the REMAINDER of the DIVISION of n MINUS one BY length  
        SET number_string TO the STRING REPRESENTATION of start  
        SET digit_character TO the ELEMENT at position digit_index OF number_string  
        SET digit_value TO the INTEGER VALUE of digit_character  
        RETURN digit_value  
    END FUNCTION  
END CLASS
