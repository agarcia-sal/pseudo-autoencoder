CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN False  
        END IF  
        
        SET adj_list TO an empty mapping FROM integer TO empty list for each integer FROM zero TO n MINUS one  
        
        FOR each pair a AND b IN edges  
            APPEND b TO the list at key a IN adj_list  
            APPEND a TO the list at key b IN adj_list  
        END FOR  
        
        SET visited TO an empty set  
        
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at key node IN adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH zero  
        
        IF the LENGTH OF visited EQUALS n  
            RETURN True  
        ELSE  
            RETURN False  
        END IF  
    END FUNCTION  
END CLASS
