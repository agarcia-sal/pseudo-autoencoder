CLASS Solution  
    FUNCTION change(amount parameter, coins parameter)  
        SET dp TO a list of zeros with length of amount parameter PLUS one  
        SET element at position zero of dp TO one  
        FOR each coin IN coins parameter  
            FOR variable x FROM coin TO amount parameter  
                INCREMENT element at position x of dp BY element at position x MINUS coin of dp  
            END FOR  
        END FOR  
        RETURN element at position amount parameter of dp  
    END FUNCTION  
END CLASS
