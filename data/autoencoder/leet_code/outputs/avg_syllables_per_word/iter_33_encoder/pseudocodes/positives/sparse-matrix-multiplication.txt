CLASS Solution  
    FUNCTION multiply(mat1 parameter, mat2 parameter)  
        SET number_of_rows_of_mat1 TO the LENGTH OF mat1  
        SET number_of_columns_of_mat1 TO the LENGTH OF the element at position one of mat1  
        SET number_of_rows_of_mat2 TO the LENGTH OF mat2  
        SET number_of_columns_of_mat2 TO the LENGTH OF the element at position one of mat2  
        INITIALIZE result_matrix AS an empty list  
        FOR each row_index FROM one TO number_of_rows_of_mat1  
            INITIALIZE row_list AS an empty list  
            FOR each column_index FROM one TO number_of_columns_of_mat2  
                APPEND zero TO row_list  
            END FOR  
            APPEND row_list TO result_matrix  
        END FOR  
        FOR each row_index FROM one TO number_of_rows_of_mat1  
            FOR each column_index FROM one TO number_of_columns_of_mat2  
                FOR each inner_index FROM one TO number_of_rows_of_mat2  
                    SET current_value TO the element at position row_index of result_matrix at position column_index  
                    SET multiplicand_left TO the element at position row_index of mat1 at position inner_index  
                    SET multiplicand_right TO the element at position inner_index of mat2 at position column_index  
                    SET new_value TO current_value PLUS multiplicand_left MULTIPLIED BY multiplicand_right  
                    SET the element at position row_index of result_matrix at position column_index TO new_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN result_matrix  
    END FUNCTION  
END CLASS
