CLASS Solution  
    FUNCTION findLongestChain WITH parameter pairs  
        SORT pairs IN ORDER OF the second element of each element IN pairs  
        SET current_end TO the smallest possible value of negative infinity  
        SET chain_length TO zero  
        FOR each start AND end IN pairs  
            IF start GREATER THAN current_end THEN  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
