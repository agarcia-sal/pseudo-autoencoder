CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters representing the decimal digits of n  
        SET length TO the number of elements in digits  
        SET i TO length MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND element at position i of digits IS GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE element at position j of digits IS LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary variable TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temporary variable  
        SET digits TO concatenation of elements from position zero TO position i of digits PLUS elements from position i PLUS one TO the last position of digits in reverse order  
        SET result TO the integer value represented by concatenating elements of digits in order  
        IF result IS GREATER THAN the largest value representable by a signed thirty two bit integer  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
