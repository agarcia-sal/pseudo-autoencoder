CLASS Solution  
    FUNCTION kthSmallestPath(destination PARAMETER row_and_column AS List OF Integer, k AS Integer) RETURNS String  
        SET row TO element at position one of row_and_column  
        SET col TO element at position two of row_and_column  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero  
                SET paths_with_H TO combination of row PLUS col MINUS one AND col MINUS one  
                IF k LESS THAN OR EQUAL TO paths_with_H  
                    APPEND letter H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND letter V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND letter V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN concatenation of all elements in path as a single string  
    END FUNCTION  
END CLASS
