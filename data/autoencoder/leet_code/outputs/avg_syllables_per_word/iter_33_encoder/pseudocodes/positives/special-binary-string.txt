CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each position i AND character char IN the enumeration of s  
                IF character EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one WITH the result of calling dfs on the substring from position start PLUS one TO position i AND the character zero TO specials  
                    SET start TO position i PLUS one  
                END IF  
            END FOR  
            SORT the list specials in reverse lexicographical order  
            RETURN the concatenation of all elements of specials  
        END FUNCTION  
        RETURN the result of calling dfs on s  
    END FUNCTION  
END CLASS
