CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO whether binary CONTAINS zero  
        FOR each bit IN binary  
            IF bit EQUALS one  
                SET end_with_1 TO the sum of end_with_0 PLUS end_with_1 PLUS one DIVIDED BY MODULO MOD  
            ELSE  
                SET end_with_0 TO the sum of end_with_0 PLUS end_with_1 DIVIDED BY MODULO MOD  
            END IF  
        END FOR  
        RETURN the sum of end_with_0 PLUS end_with_1 PLUS has_zero DIVIDED BY MODULO MOD  
    END FUNCTION  
END CLASS
