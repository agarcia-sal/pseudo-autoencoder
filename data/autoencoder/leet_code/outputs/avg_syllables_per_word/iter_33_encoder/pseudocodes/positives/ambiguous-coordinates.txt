CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING from position one TO position d OF frag  
                SET right TO the SUBSTRING from position d PLUS one TO position n OF frag  
                IF (the FIRST CHARACTER OF left NOT EQUALS zero OR left EQUALS zero) AND (the LAST CHARACTER OF right NOT EQUALS zero)  
                    YIELD THE CONCATENATION OF left PLUS IF d NOT EQUALS n THEN a PERIOD ELSE an EMPTY STRING PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING from position two TO the LENGTH OF s MINUS one OF s  

        RETURN THE LIST OF STRINGS FOR i FROM one TO the LENGTH OF s MINUS one  
            FOR a AND b IN THE CARTESIAN PRODUCT OF make of the SUBSTRING from position one TO position i OF s AND make of the SUBSTRING from position i PLUS one TO the LENGTH OF s OF s  
                THE STRING OPENING PARENTHESIS PLUS a PLUS COMMA PLUS SPACE PLUS b PLUS CLOSING PARENTHESIS  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
