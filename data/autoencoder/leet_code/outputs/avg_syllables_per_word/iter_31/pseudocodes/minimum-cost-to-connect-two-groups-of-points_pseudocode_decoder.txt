CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a new two dimensional list with size size1 PLUS one BY the number of all subsets of size2 points, each element initialized to infinity  
        SET element at position zero at zero of dp TO zero  
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO the number of all subsets of size2 points MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET combined_mask TO the bitwise OR of mask and the bitwise value with one shifted left by j positions  
                    SET element at position i PLUS one at position combined_mask of dp TO the minimum of element at position i PLUS one at position combined_mask of dp AND element at position i at position mask of dp PLUS element at position i at position j of cost  
                END FOR  
                FOR j FROM zero TO size2 MINUS one  
                    IF the bitwise AND of mask and the bitwise value with one shifted left by j positions is NOT zero  
                        SET toggled_mask TO the bitwise XOR of mask and the bitwise value with one shifted left by j positions  
                        SET element at position i PLUS one at position mask of dp TO the minimum of element at position i PLUS one at position mask of dp AND element at position i PLUS one at position toggled_mask of dp PLUS element at position i at position j of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position size1 at position the number of all subsets of size2 points MINUS one of dp  
    END FUNCTION  
END CLASS
