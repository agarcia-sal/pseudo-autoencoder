CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each character IN s  
            IF character IS A DIGIT  
                SET length TO length MULTIPLIED BY the numeric value of character  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each character IN s IN REVERSE ORDER  
            SET k TO k MODULO length  
            IF k EQUALS zero AND character IS AN ALPHABETIC CHARACTER  
                RETURN character  
            END IF  
            IF character IS A DIGIT  
                SET length TO length DIVIDED BY the numeric value of character  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
