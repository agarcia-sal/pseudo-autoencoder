CLASS Solution  
    FUNCTION closestToTarget(arr target)  
        SET seen TO empty collection of unique values  
        SET min_diff TO infinity value  
        FOR each num IN arr  
            SET new_seen TO empty collection of unique values  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO new_seen  
            END FOR  
            SET new_seen TO union of new_seen AND collection containing num  
            SET seen TO new_seen  
            FOR each val IN seen  
                IF absolute value of val MINUS target LESS THAN min_diff  
                    SET min_diff TO absolute value of val MINUS target  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
