CLASS TreeNode  
    FUNCTION __init__(self, val SET TO zero, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO new double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i plus one of values NOT EQUALS None  
            SET node.left TO new TreeNode with element at position i plus one of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i plus one of values NOT EQUALS None  
            SET node.right TO new TreeNode with element at position i plus one of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF nums EQUALS empty  
            RETURN None  
        END IF  
        SET max_val TO the maximum element IN nums  
        SET max_index TO the position of max_val IN nums  
        SET root TO new TreeNode with max_val  
        SET root.left TO the result of calling constructMaximumBinaryTree ON the sublist of nums from position one TO position max_index minus one  
        SET root.right TO the result of calling constructMaximumBinaryTree ON the sublist of nums from position max_index plus two TO the end of nums  
        RETURN root  
    END FUNCTION  
END CLASS
