CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of characters of num  
        SET n TO the length of num  
        
        WHILE k GREATER THAN zero holds  
            SET swapped TO False  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO the character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the minimum of i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the slice of num from position i TO position min_index PLUS one TO a list consisting of element at position min_index of num followed by the slice of num from position i TO position min_index  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO True  
                    BREAK the for loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS False  
                BREAK the while loop  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num  
    END FUNCTION  
END CLASS
