CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO the frequency mapping of each character IN s  
        SET digits TO a list of ten zeros  
        SET the element at position zero of digits TO the frequency of the character z in count  
        SET the element at position two of digits TO the frequency of the character w in count  
        SET the element at position four of digits TO the frequency of the character u in count  
        SET the element at position six of digits TO the frequency of the character x in count  
        SET the element at position eight of digits TO the frequency of the character g in count  
        SET the element at position three of digits TO the frequency of the character h in count MINUS the element at position eight of digits  
        SET the element at position five of digits TO the frequency of the character f in count MINUS the element at position four of digits  
        SET the element at position seven of digits TO the frequency of the character s in count MINUS the element at position six of digits  
        SET the element at position nine of digits TO the frequency of the character i in count MINUS the element at position five of digits MINUS the element at position six of digits MINUS the element at position eight of digits  
        SET the element at position one of digits TO the frequency of the character o in count MINUS the element at position zero of digits MINUS the element at position two of digits MINUS the element at position four of digits  
        SET result TO an empty list  
        FOR each index i FROM zero TO nine  
            FOR each repetition FROM zero TO the element at position i of digits MINUS one  
                APPEND the string formation of i TO result  
            END FOR  
        END FOR  
        SET final_string TO the concatenation of all elements in result without separator  
        RETURN final_string  
    END FUNCTION  
END CLASS
