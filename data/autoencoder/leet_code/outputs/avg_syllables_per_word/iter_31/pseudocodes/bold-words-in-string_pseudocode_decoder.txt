CLASS Solution  
    FUNCTION boldWords(words AS list of string, s AS string) RETURNS string  
        SET bold TO a list of Boolean values all set to false with the length equal to the length of s  
        FOR each word IN words  
            SET start TO the position in s where word first appears or negative one if word does not appear in s  
            WHILE start DOES NOT EQUAL negative one  
                FOR i FROM start TO start PLUS the length of word MINUS one  
                    SET element at position i of bold TO true  
                END FOR  
                SET start TO the position in s where word next appears starting from position start PLUS one or negative one if word does not appear again  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the length of s  
            IF the element at position i of bold EQUALS true  
                APPEND the string the characters less than b greater than to result  
                WHILE i LESS THAN the length of s AND the element at position i of bold EQUALS true  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string the characters less than forward slash b greater than to result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of elements of result  
    END FUNCTION  
END CLASS
