CLASS UnionFind  
    FUNCTION __init__ with parameters self size  
        SET self parent TO list of numbers from zero TO size MINUS one  
        SET self rank TO list of zeros of length size  
        SET self size TO list of ones of length size  
    END FUNCTION  

    FUNCTION find with parameters self u  
        IF element at position u of self parent NOT EQUALS u THEN  
            SET element at position u of self parent TO self find of element at position u of self parent  
        END IF  
        RETURN element at position u of self parent  
    END FUNCTION  

    FUNCTION union with parameters self u v  
        SET rootU TO self find of u  
        SET rootV TO self find of v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self rank GREATER THAN element at position rootV of self rank THEN  
                SET element at position rootV of self parent TO rootU  
                INCREMENT element at position rootU of self size BY element at position rootV of self size  
            ELSE IF element at position rootU of self rank LESS THAN element at position rootV of self rank THEN  
                SET element at position rootU of self parent TO rootV  
                INCREMENT element at position rootV of self size BY element at position rootU of self size  
            ELSE  
                SET element at position rootV of self parent TO rootU  
                INCREMENT element at position rootU of self size BY element at position rootV of self size  
                INCREMENT element at position rootU of self rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings with parameters self words  
        FUNCTION bitmask with parameter word  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one MULTIPLIED BY two RAISED TO the power of the difference between the ASCII value of character AND the ASCII value of lowercase a  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a new default dictionary with list as default value type  
        SET n TO the length of words  
        SET uf TO a new UnionFind object initialized with n  

        FOR each index i and corresponding word IN enumeration of words  
            SET mask TO bitmask of word  
            APPEND i TO the list at key mask in word_map  
        END FOR  

        FOR each index i and corresponding word IN enumeration of words  
            SET mask TO bitmask of word  
            FOR j FROM zero TO twenty five inclusive  
                # Delete one letter scenario  
                SET new_mask TO mask XOR one MULTIPLIED BY two RAISED TO the power of j  
                IF new_mask is a key in word_map THEN  
                    FOR each element k IN the list at key new_mask in word_map  
                        CALL uf union with arguments i k  
                    END FOR  
                END IF  

                # Add one letter scenario  
                IF the bitwise AND of mask AND one MULTIPLIED BY two RAISED TO the power of j EQUALS zero THEN  
                    SET new_mask TO mask OR one MULTIPLIED BY two RAISED TO the power of j  
                    IF new_mask is a key in word_map THEN  
                        FOR each element k IN the list at key new_mask in word_map  
                            CALL uf union with arguments i k  
                        END FOR  
                    END IF  
                END IF  

                # Replace one letter scenario  
                IF the bitwise AND of mask AND one MULTIPLIED BY two RAISED TO the power of j NOT EQUALS zero THEN  
                    FOR k FROM zero TO twenty five inclusive  
                        IF the bitwise AND of mask AND one MULTIPLIED BY two RAISED TO the power of k EQUALS zero THEN  
                            SET new_mask TO the bitwise OR of the bitwise XOR of mask AND one MULTIPLIED BY two RAISED TO the power of j AND one MULTIPLIED BY two RAISED TO the power of k  
                            IF new_mask is a key in word_map THEN  
                                FOR each element l IN the list at key new_mask in word_map  
                                    CALL uf union with arguments i l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a new default dictionary with integer zero as default value  
        FOR i FROM zero TO n MINUS one inclusive  
            SET root TO uf find of i  
            INCREMENT element at key root in root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value among the values in root_sizes  
        SET num_groups TO the number of keys in root_sizes  

        RETURN a list containing num_groups and max_group_size  
    END FUNCTION  
END CLASS
