CLASS Solution  
    FUNCTION largestSumOfAverages(nums k)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeros with LENGTH n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FUNCTION average(i j)  
            SET numerator TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET dp TO a two dimensional list with n PLUS one rows and k PLUS one columns filled with zeroes  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at position one of dp TO average(i n MINUS one)  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET candidate TO element at position j at position partitions MINUS one of dp PLUS average(i j MINUS one)  
                    IF candidate GREATER THAN element at position i at position partitions of dp  
                        SET element at position i at position partitions of dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero at position k of dp  
    END FUNCTION  
END CLASS
