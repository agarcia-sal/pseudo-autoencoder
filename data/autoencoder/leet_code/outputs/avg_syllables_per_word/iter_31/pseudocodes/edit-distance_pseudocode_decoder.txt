CLASS Solution  
    FUNCTION minDistance WITH PARAMETERS word1 word2  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CREATE dp AS a two dimensional list OF zero WITH HEIGHT m PLUS one AND WIDTH n PLUS one  
        
        FOR i FROM zero TO m  
            SET element at position zero of element at position i of dp TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET element at position j of element at position zero of dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF word1 EQUALS element at position j MINUS one OF word2  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET element at position j of element at position i of dp TO one PLUS the MINIMUM OF  
                        element at position j of element at position i MINUS one of dp  
                        element at position j MINUS one of element at position i of dp  
                        element at position j MINUS one of element at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
