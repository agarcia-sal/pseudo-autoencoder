CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one THEN  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n DO  
                IF element at position i of used EQUALS False AND (element at position i of used MINUS element at position index of used EQUALS zero OR element at position index of used MINUS element at position i of used EQUALS zero) THEN  
                    SET element at position i of used TO True  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO list of False repeated n PLUS one times  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
