CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        CREATE a two-dimensional list dp of size n PLUS one by n PLUS one filled with zero  
        
        FOR j FROM zero TO n  
            SET element at position j of the first list in dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the letter I THEN  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of the list at position i MINUS one in dp MODULO MOD  
                    SET element at position j of the list at position i in dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of the list at position i MINUS one in dp MODULO MOD  
                    SET element at position j of the list at position i in dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the SUM OF elements in the list at position n in dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
