CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF LENGTH OF s LESS THAN k THEN  
            RETURN zero  
        END IF  
        SET count TO frequency mapping of each character in s  
        FOR each char IN keys of count  
            IF value at char in count LESS THAN k THEN  
                RETURN maximum value FROM applying longestSubstring to each substring FROM splitting s BY char WITH parameter k  
            END IF  
        END FOR  
        RETURN LENGTH OF s  
    END FUNCTION  
END CLASS
