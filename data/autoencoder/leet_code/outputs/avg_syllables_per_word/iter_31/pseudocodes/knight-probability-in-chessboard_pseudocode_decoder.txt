CLASS Solution  
    FUNCTION knightProbability(n INT, k INT, row INT, column INT) RETURNS FLOAT  
        SET moves TO the list of pairs consisting of the pair of negative two and negative one PLUS the pair of negative two and positive one PLUS the pair of negative one and negative two PLUS the pair of negative one and positive two PLUS the pair of positive one and negative two PLUS the pair of positive one and positive two PLUS the pair of positive two and negative one PLUS the pair of positive two and positive one  
        SET dp TO a two-dimensional list of size n by n filled with zero values  
        SET the element at position row and position column of dp TO one  
        FOR move_count FROM one TO k  
            SET new_dp TO a two-dimensional list of size n by n filled with zero values  
            FOR r FROM zero TO n MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF the element at position r and position c of dp EQUALS zero  
                        CONTINUE to the next iteration of the inner loop  
                    END IF  
                    FOR each pair dr_dc IN moves  
                        SET dr TO the first element of dr_dc  
                        SET dc TO the second element of dr_dc  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN n AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN n  
                            INCREMENT the element at position nr and position nc of new_dp BY the element at position r and position c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each value IN row_list  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
