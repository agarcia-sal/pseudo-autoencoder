CLASS Solution  
    FUNCTION partition(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS substring from the LENGTH OF sub MINUS one TO the FIRST element OF sub IN REVERSE ORDER  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF s THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF s PLUS one  
                IF is_palindrome(substring from position start TO position end MINUS one OF s) THEN  
                    CALL backtrack WITH end AND path PLUS list containing substring from position start TO position end MINUS one OF s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
