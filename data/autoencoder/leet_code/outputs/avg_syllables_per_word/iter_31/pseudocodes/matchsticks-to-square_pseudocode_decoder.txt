CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF all elements IN matchsticks  
        IF the remainder WHEN total_length IS DIVIDED BY four NOT EQUALS zero  
            RETURN False  
        END IF  
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN descending order  
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS the LENGTH OF matchsticks  
                FOR each side IN sides  
                    IF side NOT EQUALS side_length  
                        RETURN False  
                    END IF  
                END FOR  
                RETURN True  
            END IF  
            FOR i FROM zero TO three  
                IF the element at position i of sides PLUS the element at position index of matchsticks LESS THAN OR EQUAL TO side_length  
                    INCREMENT the element at position i of sides BY the element at position index of matchsticks  
                    IF can_form_sides(sides, index PLUS one) EQUALS True  
                        RETURN True  
                    END IF  
                    DECREMENT the element at position i of sides BY the element at position index of matchsticks  
                END IF  
                IF the element at position i of sides EQUALS zero  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN can_form_sides(a list of four zeros, zero)  
    END FUNCTION  
END CLASS
