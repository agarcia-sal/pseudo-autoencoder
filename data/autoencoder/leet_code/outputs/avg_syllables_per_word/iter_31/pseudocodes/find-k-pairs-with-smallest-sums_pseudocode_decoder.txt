CLASS Solution  
    FUNCTION kSmallestPairs(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  

        SET min_heap TO empty list  

        FOR i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            ADD to min_heap a tuple of the SUM of element at position i of nums1 AND element at position zero of nums2 AND i AND zero  
        END FOR  

        SET result TO empty list  

        WHILE min_heap NOT EQUALS empty AND the LENGTH OF result LESS THAN k  
            REMOVE the tuple of the smallest first element FROM min_heap AND UNPACK INTO discard_variable AND i AND j  
            APPEND the list of element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                ADD to min_heap a tuple of the SUM of element at position i of nums1 AND element at position j PLUS one of nums2 AND i AND j PLUS one  
            END IF  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
