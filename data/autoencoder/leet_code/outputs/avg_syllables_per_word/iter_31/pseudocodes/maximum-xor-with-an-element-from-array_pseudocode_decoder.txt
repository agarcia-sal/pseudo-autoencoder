CLASS Solution  
    FUNCTION maximizeXor(nums PARAMETER, queries PARAMETER)  
        CALL sort nums IN ASCENDING ORDER  
        SET sorted_queries TO the result of sorting the enumeration of queries by the second element of the second element of each enumerated item in ascending order  
        SET result TO a list where each element is negative one and the LENGTH OF queries number of elements  
        SET trie TO an empty dictionary  
        SET num_index TO zero  
        
        FOR each query_data IN sorted_queries  
            SET query_index TO the first element of query_data  
            SET xi_and_mi TO the second element of query_data  
            SET xi TO the first element of xi_and_mi  
            SET mi TO the second element of xi_and_mi  
            
            WHILE num_index LESS THAN the LENGTH OF nums AND the element at position num_index of nums LESS THAN OR EQUAL TO mi  
                SET num TO the element at position num_index of nums  
                SET node TO trie  
                
                FOR i FROM thirty one DOWN TO zero DECREMENT BY one  
                    SET bit TO the bit at position i in num described as the integer division of num by two raised to the power i MODULO two  
                    IF bit NOT IN node  
                        ASSIGN an empty dictionary TO the reference at bit in node  
                    END IF  
                    SET node TO the reference at bit in node  
                END FOR  
                
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS empty  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            
            FOR i FROM thirty one DOWN TO zero DECREMENT BY one  
                SET bit TO the bit at position i in xi described as the integer division of xi by two raised to the power i MODULO two  
                SET toggle_bit TO one MINUS bit  
                
                IF toggle_bit IN node  
                    ADD two raised to the power i TO max_xor  
                    SET node TO the reference at toggle_bit in node  
                ELSE  
                    SET node TO the reference at bit in node  
                END IF  
            END FOR  
            
            SET the element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
