CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        SET dp TO a two dimensional list with dimensions m PLUS one BY n PLUS one initialized with zeros  
        FOR i FROM zero TO m  
            SET the element at position zero of the list at position i of dp TO one  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of s EQUALS the element at position j MINUS one of t THEN  
                    SET the element at position j of the list at position i of dp TO the element at position j MINUS one of the list at position i MINUS one of dp PLUS the element at position j of the list at position i MINUS one of dp  
                ELSE  
                    SET the element at position j of the list at position i of dp TO the element at position j of the list at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of the list at position m of dp  
    END FUNCTION  
END CLASS
