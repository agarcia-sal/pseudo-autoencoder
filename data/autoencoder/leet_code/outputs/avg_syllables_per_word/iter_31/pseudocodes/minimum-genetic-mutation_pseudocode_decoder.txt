CLASS Solution  
    FUNCTION minMutation(start string, end string, bank List of string) RETURNS integer  
        SET s TO the set created from the elements of bank  
        SET q TO a double ended queue containing the tuple of start and zero  
        SET mp TO the mapping from character A TO characters T C G AND from character T TO characters A C G AND from character C TO characters A T G AND from character G TO characters A T C  
        WHILE the queue q is not empty  
            SET t AND step TO the removal of the element at the front of queue q  
            IF the string t is equal to the string end  
                RETURN step  
            END IF  
            FOR each index i and character v obtained by enumerating over the string t  
                FOR each character j IN the character sequence obtained from mapping mp at key v  
                    SET next TO the concatenation of the substring of t from the start UNTIL position before i PLUS character j PLUS the substring of t from position after i UNTIL the end  
                    IF the string next is an element of the set s  
                        APPEND the tuple of next and the value of step PLUS one TO the end of queue q  
                        REMOVE the element next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
