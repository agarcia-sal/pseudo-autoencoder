CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        ASSIGN reference TO the frequency count of digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO total_sum MODULO three  
        IF remainder EQUALS one  
            FOR each digit d IN the list containing one four seven  
                IF the frequency count of digit d IS GREATER THAN zero  
                    DECREMENT the frequency count of digit d BY one  
                    EXIT the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list containing two five eight  
                    IF the frequency count of digit d IS GREATER THAN one  
                        DECREMENT the frequency count of digit d BY two  
                        EXIT the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        IF remainder EQUALS two  
            FOR each digit d IN the list containing two five eight  
                IF the frequency count of digit d IS GREATER THAN zero  
                    DECREMENT the frequency count of digit d BY one  
                    EXIT the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list containing one four seven  
                    IF the frequency count of digit d IS GREATER THAN one  
                        DECREMENT the frequency count of digit d BY two  
                        EXIT the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        SET result TO an empty list  
        FOR digit d FROM nine DOWN TO zero  
            APPEND the string form of digit d repeated by the frequency count of digit d TO result  
        END FOR  
        SET final_number TO the joining of all elements in result forming a single string  
        IF final_number IS NOT empty AND the element at position zero of final_number EQUALS character zero  
            RETURN character zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
