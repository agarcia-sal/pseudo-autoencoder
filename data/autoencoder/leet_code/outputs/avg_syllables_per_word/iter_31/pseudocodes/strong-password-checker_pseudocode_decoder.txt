CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO false  
        FOR each character IN password  
            IF character IS a lowercase letter  
                SET has_lower TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET has_upper TO false  
        FOR each character IN password  
            IF character IS an uppercase letter  
                SET has_upper TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET has_digit TO false  
        FOR each character IN password  
            IF character IS a digit  
                SET has_digit TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS (number of true values among has_lower has_upper has_digit)  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        WHILE i LESS THAN n  
            IF character at position i of password EQUALS character at position i minus one of password AND character at position i of password EQUALS character at position i minus two of password  
                SET length TO two  
                WHILE i LESS THAN n AND character at position i of password EQUALS character at position i minus one of password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n LESS THAN six  
            RETURN the GREATER value BETWEEN missing_types AND six MINUS n  
        ELSE IF n LESS THAN OR EQUAL TO twenty  
            RETURN the GREATER value BETWEEN missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            SET replace TO replace MINUS the SMALLER value BETWEEN delete AND one  
            SET replace TO replace MINUS (the SMALLER value BETWEEN the GREATER value BETWEEN delete MINUS one AND zero AND two MULTIPLIED BY two) DIVIDED BY two  
            SET replace TO replace MINUS the GREATER value BETWEEN delete MINUS one MINUS two MULTIPLIED BY two AND zero DIVIDED BY three  
            RETURN delete PLUS the GREATER value BETWEEN missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
