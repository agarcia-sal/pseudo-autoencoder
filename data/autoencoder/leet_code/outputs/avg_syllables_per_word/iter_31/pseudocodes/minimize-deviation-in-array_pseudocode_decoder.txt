CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF the remainder when num is divided by two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            ADD the negative of num TO max_heap by pushing onto the heap  
        END FOR  
  
        SET min_deviation TO infinity  
  
        WHILE the element at position zero of max_heap is evenly divisible by two  
            SET max_value TO the negative of the element popped from max_heap by popping from the heap  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO the integer division of max_value BY two  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            ADD the negative of max_value TO max_heap by pushing onto the heap  
        END WHILE  
  
        IF min_deviation GREATER THAN the negative of the element at position zero of max_heap MINUS min_value  
            SET min_deviation TO the negative of the element at position zero of max_heap MINUS min_value  
        END IF  
  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
