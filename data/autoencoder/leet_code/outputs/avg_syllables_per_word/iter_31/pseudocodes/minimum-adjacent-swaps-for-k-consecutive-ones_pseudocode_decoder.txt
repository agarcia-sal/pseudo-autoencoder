CLASS Solution  
    FUNCTION minMoves(nums k)  
        SET positions TO empty list  
        FOR each index i AND element num IN nums with index tracking  
            IF num EQUALS one  
                APPEND index i TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start end)  
            SET mid TO start PLUS end DIVIDED BY two truncated to integer  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR each index i FROM start TO end MINUS one  
                SET distance TO absolute value of element at position i of positions MINUS median MINUS mid MINUS i  
                SET cost TO cost PLUS distance  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO infinite  
        FOR each index i FROM zero TO length of positions MINUS k  
            SET current_cost TO calculate_cost(i i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
