CLASS Solution  
    FUNCTION wordBreak(s string wordDict list of strings) RETURNS boolean  
        SET wordSet TO the set made from wordDict  
        SET dp TO a list of boolean values initialised to False repeated by the length of s PLUS one  
        SET the element at position zero of dp TO True  
        FOR i FROM one TO the length of s  
            FOR j FROM zero TO i MINUS one  
                IF the element at position j of dp EQUALS True AND the substring of s from position j PLUS one TO position i EQUALS an element of wordSet  
                    SET the element at position i of dp TO True  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the length of s of dp  
    END FUNCTION  
END CLASS
