CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing each plus sign in side with the character followed by plus sign and replacing each minus sign in side with the character followed by minus sign and then splitting the resulting string by the character  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token EQUALS empty string THEN  
                    CONTINUE to next iteration  
                END IF  
                IF token CONTAINS the character x THEN  
                    IF token EQUALS the character x OR token EQUALS the plus sign followed by the character x THEN  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS the minus sign followed by the character x THEN  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET number_value TO the integer obtained by removing the character x from token  
                        INCREMENT x_count BY number_value  
                    END IF  
                ELSE  
                    SET number_value TO the integer of token  
                    INCREMENT num_sum BY number_value  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  

        SET left_side AND right_side TO the two parts of equation separated by the equal sign  
        SET left_x AND left_num TO the result of calling parse_side with left_side  
        SET right_x AND right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero THEN  
            IF total_num EQUALS zero THEN  
                RETURN the string representing infinite solutions  
            ELSE  
                RETURN the string representing no solution  
            END IF  
        ELSE  
            SET solution_value TO total_num INTEGER DIVIDED BY total_x  
            RETURN the string constructed as the character x followed by the equal sign followed by the string representation of solution_value  
        END IF  
    END FUNCTION  
END CLASS
