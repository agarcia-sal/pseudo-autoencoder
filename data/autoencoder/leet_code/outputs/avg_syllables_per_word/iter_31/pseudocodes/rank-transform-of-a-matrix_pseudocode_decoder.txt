CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers FROM zero TO size MINUS one  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET root_u TO self.find(u)  
        SET root_v TO self.find(v)  
        IF root_u NOT EQUALS root_v  
            SET element at position root_u of self.parent TO root_v  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION matrixRankTransform(self, matrix)  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET value_to_indices TO a mapping from default key to empty list  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                APPEND the pair of r AND c TO the list at key element at position r of matrix AND position c of that element IN value_to_indices  
            END FOR  
        END FOR  
        
        SET row_rank TO a list of zero repeated m times  
        SET col_rank TO a list of zero repeated n times  
        SET result TO a list containing m elements each being a list containing n zeros  
        
        FOR each value IN the keys of value_to_indices sorted in ascending order  
            SET uf TO a new UnionFind instance INITIALIZED WITH m PLUS n  
            SET rank TO a mapping from default key TO integer zero  
            
            FOR each pair r AND c IN value_to_indices AT key value  
                CALL uf.union WITH r AND c PLUS m  
            END FOR  
            
            FOR each pair r AND c IN value_to_indices AT key value  
                SET root TO uf.find WITH r  
                SET rank AT key root TO THE maximum of current rank AT key root AND THE maximum of row_rank at position r AND col_rank at position c PLUS one  
            END FOR  
            
            FOR each pair r AND c IN value_to_indices AT key value  
                SET root TO uf.find WITH r  
                SET element at position r AND c of result TO rank AT key root  
                SET element at position r of row_rank TO rank AT key root  
                SET element at position c of col_rank TO rank AT key root  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
