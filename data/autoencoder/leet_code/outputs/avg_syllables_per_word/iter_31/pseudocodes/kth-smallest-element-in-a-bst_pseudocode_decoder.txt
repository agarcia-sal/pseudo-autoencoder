CLASS TreeNode  
    FUNCTION __init__ with parameters val MINUS zero left MINUS none right MINUS none  
        SET val OF self TO val  
        SET left OF self TO left  
        SET right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values as list  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET left OF node TO new TreeNode with value element at position i of values  
            APPEND left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET right OF node TO new TreeNode with value element at position i of values  
            APPEND right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF val OF p NOT EQUALS val OF q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with parameters left OF p and left OF q AND is_same_tree with parameters right OF p and right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest with parameters root k  
        SET stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE the LENGTH OF stack GREATER THAN zero OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO left OF current  
            END WHILE  
            SET current TO element removed from end of stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN val OF current  
            END IF  
            SET current TO right OF current  
        END WHILE  
    END FUNCTION  
END CLASS
