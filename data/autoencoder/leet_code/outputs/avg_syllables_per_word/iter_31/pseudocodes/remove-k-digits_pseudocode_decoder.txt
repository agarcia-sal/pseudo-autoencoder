CLASS Solution  
    FUNCTION removeKdigits(num parameter THAT IS a string, k parameter THAT IS an integer) THAT RETURNS a string  
        SET stack TO an empty list  
        FOR each digit IN the characters of num parameter  
            WHILE k parameter IS GREATER THAN zero AND stack IS NOT empty AND the last element in stack IS GREATER THAN digit  
                REMOVE the last element FROM stack  
                DECREMENT k parameter BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k parameter IS GREATER THAN zero  
            SET final_stack TO the sequence of elements in stack FROM the first element TO the element at the position of the length of stack MINUS k parameter MINUS one  
        ELSE  
            SET final_stack TO stack  
        END IF  
        SET result TO the concatenation of all elements in final_stack WITH all leading elements equal to the character zero REMOVED  
        IF result IS an empty string  
            RETURN the string zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
