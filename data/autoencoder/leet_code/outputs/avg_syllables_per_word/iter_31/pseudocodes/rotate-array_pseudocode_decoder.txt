CLASS Solution  
    FUNCTION rotate(nums k)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start end)  
            WHILE start LESS THAN end  
                SET temporary TO the element at position start of nums  
                SET the element at position start of nums TO the element at position end of nums  
                SET the element at position end of nums TO temporary  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH zero AND n MINUS one  
        CALL reverse WITH zero AND k MINUS one  
        CALL reverse WITH k AND n MINUS one  
    END FUNCTION  
END CLASS
