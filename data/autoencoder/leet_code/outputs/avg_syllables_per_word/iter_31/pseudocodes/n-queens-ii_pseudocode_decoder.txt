CLASS Solution  
    FUNCTION totalNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN the NEGATION OF the SUM of element at position col of cols PLUS element at position of row MINUS col of hills PLUS element at position of row PLUS col of dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen WITH PARAMETERS row AND col  
            SET element at position col of cols TO one  
            SET element at position of row MINUS col of hills TO one  
            SET element at position of row PLUS col of dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen WITH PARAMETERS row AND col  
            SET element at position col of cols TO zero  
            SET element at position of row MINUS col of hills TO zero  
            SET element at position of row PLUS col of dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS row DEFAULTED TO zero AND count DEFAULTED TO zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH row AND col EQUALS true  
                    place_queen WITH row AND col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO backtrack WITH row PLUS one AND count  
                    END IF  
                    remove_queen WITH row AND col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO a list of zero repeated n times  
        SET hills TO a list of zero repeated two MULTIPLIED BY n MINUS one times  
        SET dales TO a list of zero repeated two MULTIPLIED BY n MINUS one times  
        
        RETURN backtrack  
    END FUNCTION  
END CLASS
