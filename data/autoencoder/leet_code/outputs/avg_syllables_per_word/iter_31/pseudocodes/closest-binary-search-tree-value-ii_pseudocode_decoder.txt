CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty then  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS no value then  
            SET node.left TO new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS no value then  
            SET node.right TO new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value then  
        RETURN true  
    ELSE IF p EQUALS no value OR q EQUALS no value then  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val then  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node EQUALS no value then  
                RETURN no value  
            END IF  
            CALL inorder_traversal with argument node.left  
            IF the LENGTH OF result LESS THAN k then  
                APPEND node.val TO result  
            ELSE  
                IF the ABSOLUTE VALUE OF node.val MINUS target LESS THAN the ABSOLUTE VALUE OF element at position one of result MINUS target then  
                    REMOVE element from the front of result  
                    APPEND node.val TO result  
                ELSE  
                    RETURN no value  
                END IF  
            END IF  
            CALL inorder_traversal with argument node.right  
        END FUNCTION  
        SET result TO new empty deque  
        CALL inorder_traversal with argument root  
        RETURN the list converted from result  
    END FUNCTION  
END CLASS
