CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list containing the pairs of minus one and zero plus one and zero zero and minus one zero and one zero and zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n THEN  
                    SET element at position nx of matrix element at position ny TO one MINUS element at position nx of matrix element at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN tuple containing tuples each constructed from elements of each row of matrix  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple containing tuples each of zeros of length n repeated m times  
        SET initial_state TO matrix_to_tuple(mat)  
        SET queue TO list containing the pair initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE queue is not empty  
            REMOVE the first element from queue and SET it TO current_state and steps  
            IF current_state EQUALS target THEN  
                RETURN steps  
            END IF  

            SET current_matrix TO list of lists each constructed from elements of each row in current_state  

            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO copy of current_matrix using deep copy  
                    CALL flip with next_matrix and i and j  
                    SET next_state TO matrix_to_tuple(next_matrix)  
                    IF next_state NOT IN visited THEN  
                        ADD next_state TO visited  
                        APPEND the pair next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
