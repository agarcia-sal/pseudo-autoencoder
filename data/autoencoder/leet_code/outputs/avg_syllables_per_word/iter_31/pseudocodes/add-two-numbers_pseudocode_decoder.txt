CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO new ListNode with value as element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the end of values  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN self_val EQUALS other_val AND recursive call with next nodes  
        WHERE self_val IS element val of p1  
        AND other_val IS element val of p2  
        AND recursive call is_same_list with element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO new ListNode with value zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero  
            IF l1 IS NOT None  
                SET val1 TO element val of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO element val of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode with value total MODULO ten  
            SET current TO element next of current  
            IF l1 IS NOT None  
                SET l1 TO element next of l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO element next of l2  
            END IF  
        END WHILE  
        RETURN element next of dummy_head  
    END FUNCTION  
END CLASS
