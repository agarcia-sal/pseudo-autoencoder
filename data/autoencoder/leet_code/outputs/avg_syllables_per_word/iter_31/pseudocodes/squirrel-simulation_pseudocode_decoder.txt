CLASS Solution  
    FUNCTION minDistance(height parameter width parameter tree parameter squirrel parameter nuts parameter)  
        FUNCTION distance(p1 parameter p2 parameter)  
            SET horizontal_difference TO absolute value of element at position zero of p1 MINUS element at position zero of p2  
            SET vertical_difference TO absolute value of element at position one of p1 MINUS element at position one of p2  
            SET result TO horizontal_difference PLUS vertical_difference  
            RETURN result  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance(nut parameter tree)  
            SET first_nut_distance TO distance(squirrel parameter nut) PLUS distance(nut parameter tree)  
            SET later_nut_distance TO two MULTIPLIED BY distance(nut parameter tree)  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  
        
        SET result TO total_distance MINUS max_save  
        RETURN result  
    END FUNCTION  
END CLASS
