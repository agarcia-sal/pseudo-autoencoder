CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set of elements in wordDict  
        SET memo TO an empty dictionary mapping indices to lists of sentences  
        
        FUNCTION backtrack(start)  
            IF start IS a key in memo  
                RETURN the value at key start in memo  
            END IF  
            
            IF start EQUALS the LENGTH OF s  
                RETURN a list containing the empty string  
            END IF  
            
            SET res TO an empty list  
            FOR end FROM start PLUS one TO the LENGTH OF s  
                SET word TO the substring of s from position start TO position end MINUS one  
                IF word IS an element in wordSet  
                    FOR each sentence IN the result of backtrack called with end  
                        IF sentence IS the empty string  
                            APPEND word TO res  
                        ELSE  
                            APPEND the concatenation of word PLUS space PLUS sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET the value at key start in memo TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN the result of backtrack called with zero  
    END FUNCTION  
END CLASS
