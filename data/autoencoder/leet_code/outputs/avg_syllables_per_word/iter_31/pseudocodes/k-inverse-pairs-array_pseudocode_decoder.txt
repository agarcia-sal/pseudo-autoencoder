CLASS Solution  
    FUNCTION kInversePairs WITH PARAMETERS n k  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO a two dimensional list of zeros with n PLUS one rows and k PLUS one columns  
        SET element at position zero of list at position zero of dp TO one  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position j of list at position i MINUS one of dp  
                SET prefix_sum TO prefix_sum MINUS MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position j MINUS i of list at position i MINUS one of dp  
                    SET prefix_sum TO prefix_sum PLUS MODULO MOD  
                END IF  
                SET prefix_sum TO prefix_sum MODULO MOD  
                SET element at position j of list at position i of dp TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN element at position k of list at position n of dp  
    END FUNCTION  
END CLASS
