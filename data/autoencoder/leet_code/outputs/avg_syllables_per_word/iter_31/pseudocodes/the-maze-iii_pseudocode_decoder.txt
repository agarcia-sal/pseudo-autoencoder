CLASS Solution  
    FUNCTION findShortestWay(maze maze, ball ball, hole hole)  
        SET directions TO list of elements each containing zero and one and character r and one and zero and character d and zero and negative one and character l and negative one and zero and character u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO tuple CONSTRUCTED FROM ball  
        SET hole TO tuple CONSTRUCTED FROM hole  
        SET pq TO list containing a tuple of zero and empty string and start  
        SET visited TO empty set  
        
        WHILE pq holds  
            EXTRACT dist path x y FROM the result of removing the smallest element from pq  
            
            IF x EQUALS element at position zero of hole AND y EQUALS element at position one of hole THEN  
                RETURN path  
            END IF  
            
            IF the pair of x and y IS IN visited THEN  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            ADD the pair of x and y TO visited  
            
            FOR each dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AND element at position new_y PLUS dy OF that element EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    
                    IF new_x EQUALS element at position zero of hole AND new_y EQUALS element at position one of hole THEN  
                        BREAK out of the WHILE loop  
                    END IF  
                END WHILE  
                
                IF the pair of new_x and new_y IS NOT IN visited THEN  
                    ADD a tuple of dist PLUS count and path PLUS direction and the pair of new_x and new_y TO pq maintaining its order as a priority heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
