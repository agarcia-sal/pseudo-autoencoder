CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_string TO the string at position index of strs reversed in order  
            IF reversed_string GREATER THAN the string at position index of strs  
                SET the string at position index of strs TO reversed_string  
            END IF  
        END FOR  

        SET doubled_strs TO the sequence consisting of strs FOLLOWED BY strs  

        SET max_string TO the empty string  

        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the sequence consisting of false FOLLOWED BY true  
                IF is_reversed EQUALS true  
                    SET current TO the string at position index of strs reversed in order  
                ELSE  
                    SET current TO the string at position index of strs  
                END IF  
                SET remaining TO the concatenation of the strings at positions from index plus one TO index plus the LENGTH OF strs MINUS one of doubled_strs in order  

                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring of current from position cut_position TO the end FOLLOWED BY remaining FOLLOWED BY the substring of current from position zero TO position cut_position MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN max_string  
    END FUNCTION  
END CLASS
