CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER) RETURNS string  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string representing negative sign  
        ELSE  
            SET sign TO the empty string  
        END IF  

        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  

        SET quotient TO the result of integer division of numerator BY denominator  
        SET remainder TO the remainder of division of numerator BY denominator  
        SET integer_part TO the string representing quotient  

        IF remainder EQUALS zero  
            RETURN the concatenation of sign AND integer_part  
        END IF  

        CREATE an empty collection named seen_remainders to store keys and integer values  
        CREATE an empty list named fractional_part to hold string digits  

        WHILE remainder NOT EQUALS zero  
            IF remainder IS A key IN seen_remainders  
                SET start TO the integer value corresponding to remainder IN seen_remainders  
                SET non_repeating TO the concatenation of elements from the start of fractional_part TO the element before position start  
                SET repeating TO the concatenation of elements from position start TO the end of fractional_part  
                RETURN the concatenation of sign AND integer_part AND the string representing a decimal point AND non_repeating AND the string representing an opening parenthesis AND repeating AND the string representing a closing parenthesis  
            END IF  

            STORE the current length of fractional_part AS the value for the key remainder IN seen_remainders  

            MULTIPLY remainder BY ten  
            SET quotient TO the result of integer division of remainder BY denominator  
            SET remainder TO the remainder of division of remainder BY denominator  
            APPEND the string representing quotient TO fractional_part  
        END WHILE  

        RETURN the concatenation of sign AND integer_part AND the string representing a decimal point AND the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
