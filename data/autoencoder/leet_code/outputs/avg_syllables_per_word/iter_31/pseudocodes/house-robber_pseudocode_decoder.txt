CLASS Solution  
    FUNCTION rob WITH PARAMETER nums  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        SET dp TO a list of zeros WITH the LENGTH OF nums  
        SET the element at position zero of dp TO the element at position zero of nums  
        SET the element at position one of dp TO the GREATER OF the element at position zero of nums AND the element at position one of nums  
        FOR i FROM two TO the LENGTH OF nums MINUS one  
            SET the element at position i of dp TO the GREATER OF the element at position i MINUS one of dp AND the SUM OF the element at position i MINUS two of dp AND the element at position i of nums  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
