CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS None  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO A new ListNode with the first element of values  
    SET p TO head  
    FOR each val IN the elements of values starting from the second element TO the last element  
        SET node TO A new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the value attribute of p1 EQUALS the value attribute of p2 AND the result of is_same_list with the next attributes of p1 and p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS head  
        SET dummy TO A new ListNode with zero AND head as next  
        SET prev TO dummy  
        WHILE head HOLDS  
            IF the next attribute of head HOLDS AND the value attribute of head EQUALS the value attribute of the next attribute of head  
                WHILE the next attribute of head HOLDS AND the value attribute of head EQUALS the value attribute of the next attribute of head  
                    SET head TO the next attribute of head  
                END WHILE  
                SET the next attribute of prev TO the next attribute of head  
            ELSE  
                SET prev TO the next attribute of prev  
            END IF  
            SET head TO the next attribute of head  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
