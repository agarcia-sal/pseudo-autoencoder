CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a mapping with default integer zero  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count_part AND domain_part separated by whitespace  
            SET count TO the integer value of count_part  
            SPLIT domain_part by the character dot INTO subdomains  
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain TO the concatenation of elements from position index TO the last position of subdomains joined by the character dot  
                INCREMENT the value mapped by subdomain in domain_count BY count  
            END FOR  
        END FOR  
        SET result_list TO an empty list  
        FOR each pair of domain and count IN the items of domain_count  
            APPEND the string formed by the string of count followed by a whitespace followed by domain TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
