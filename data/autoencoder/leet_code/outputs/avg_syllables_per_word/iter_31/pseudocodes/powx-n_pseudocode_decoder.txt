CLASS Solution  
    FUNCTION myPow(x AS float, n AS int) AS float  
        FUNCTION qpow(a AS float, n AS int) AS float  
            SET ans TO one  
            WHILE n NOT EQUALS zero  
                IF n MODULO two EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two USING INTEGER DIVISION  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH PARAMETERS x AND n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH PARAMETERS x AND n MINUS n  
        END IF  
    END FUNCTION  
END CLASS
