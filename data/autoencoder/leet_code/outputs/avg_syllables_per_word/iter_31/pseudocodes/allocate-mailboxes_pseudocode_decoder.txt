CLASS Solution  
    FUNCTION minDistance(houses List INTEGER k INTEGER)  
        SORT houses IN ASCENDING ORDER  
        SET n TO the LENGTH OF houses  
        
        SET cost TO a two dimensional list of zeroes with n rows and n columns  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and position j of cost TO element at position i PLUS one and position j MINUS one of cost PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list filled with infinity with n PLUS one rows and k PLUS one columns  
        SET element at position zero and position zero of dp TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET element at position i and position j of dp TO the MINIMUM BETWEEN element at position i and position j of dp AND element at position p and position j MINUS one of dp PLUS element at position p and position i MINUS one of cost  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n and position k of dp  
    END FUNCTION  
END CLASS
