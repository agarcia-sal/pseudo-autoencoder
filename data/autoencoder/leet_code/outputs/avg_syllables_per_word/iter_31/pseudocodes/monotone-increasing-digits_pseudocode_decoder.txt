CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        ASSIGN reference TO the list of characters representing the digits of the number n converted to string  
        SET length TO the LENGTH OF the list of digits  
        SET marker TO length  
        FOR index i FROM length MINUS one TO one IN DECREASING ORDER  
            IF element at position i of digits LESS THAN element at position i MINUS one of digits  
                SET element at position i MINUS one of digits TO the character representation of the integer parsed from element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR index i FROM marker TO length MINUS one  
            SET element at position i of digits TO character nine  
        END FOR  
        RETURN the integer parsed from concatenating all elements in digits  
    END FUNCTION  
END CLASS
