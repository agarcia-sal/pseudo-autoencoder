CLASS Solution  
    FUNCTION findLongestChain WITH PARAMETER pairs  
        SORT pairs BY the second element of each pair in ascending order  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs WITH ELEMENTS start AND end  
            IF start GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
