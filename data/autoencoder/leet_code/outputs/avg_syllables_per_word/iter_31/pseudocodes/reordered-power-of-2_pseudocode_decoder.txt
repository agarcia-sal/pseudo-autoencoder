    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten  
            CONVERT power_of_2 TO string form  
            SORT characters of the string form in ascending order  
            CONVERT the sorted characters to a tuple of digits  
            ADD the tuple to power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        CONVERT n to string form  
        SORT characters of the string form in ascending order  
        CONVERT the sorted characters to a tuple named sorted_n_digits  
        IF sorted_n_digits IS AN ELEMENT OF power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
