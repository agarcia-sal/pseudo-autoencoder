CLASS Solution  
    FUNCTION largestNumber(cost parameter of type List of integer target parameter of type integer) RETURNS string  
        SET cost_to_digit TO an empty dictionary  
        FOR index FROM zero TO the LENGTH OF cost MINUS one  
            SET key_cost TO element at position index of cost  
            SET value_digit TO string representation of index PLUS one  
            ADD key_cost mapped TO value_digit TO cost_to_digit  
        END FOR  
        
        SET dp TO a list with one element as an empty string FOLLOWED BY target elements each set TO string indicating impossible to form number  
        
        FOR t FROM one TO target  
            FOR each key_cost AND value_digit IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO key_cost AND element at position t MINUS key_cost of dp NOT EQUALS string indicating impossible to form number THEN  
                    SET candidate TO value_digit CONCATENATED WITH element at position t MINUS key_cost of dp  
                    IF element at position t of dp EQUALS string indicating impossible to form number OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate LEXICOGRAPHICALLY GREATER THAN element at position t of dp) THEN  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp NOT EQUALS string indicating impossible to form number THEN  
            RETURN element at position target of dp  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
