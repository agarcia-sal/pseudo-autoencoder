CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node with parameter values  
    IF values EQUALS the empty collection  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the collection of elements of values starting from the second element to the last element  
        SET node TO a new ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN truth  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN falsity  
    END IF  
    RETURN the predicate that the value property of p1 EQUALS the value property of p2 AND the result of the function is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION

CLASS Solution  
    FUNCTION swapPairs with parameter head  
        SET dummy TO a new ListNode initialized with the natural number zero  
        SET the next property of dummy TO head  
        SET current TO dummy  
        WHILE the next property of current IS NOT nothing AND the next property of the next property of current IS NOT nothing  
            SET first TO the next property of current  
            SET second TO the next property of the next property of current  
            SET the next property of current TO second  
            SET the next property of first TO the next property of second  
            SET the next property of second TO first  
            SET current TO the next property of the next property of current  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
