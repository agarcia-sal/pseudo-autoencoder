CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the string representation of the element at position one of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the string representation of the element at position one of nums CONCATENATED WITH the character representing division CONCATENATED WITH the string representation of the element at position two of nums  
        END IF  
        RETURN the string representation of the element at position one of nums CONCATENATED WITH the character representing division CONCATENATED WITH the character representing opening parenthesis CONCATENATED WITH the string representations of elements from position two TO the last position of nums JOINED BY the character representing division CONCATENATED WITH the character representing closing parenthesis  
    END FUNCTION  
END CLASS
