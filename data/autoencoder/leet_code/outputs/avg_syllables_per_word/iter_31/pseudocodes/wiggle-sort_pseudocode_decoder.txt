CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true as the expectation that element at position zero of nums IS LESS THAN OR EQUAL TO element at position one of nums  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less EQUALS true  
                IF element at position i of nums IS GREATER THAN element at position i PLUS one of nums  
                    SET temporary_variable TO element at position i of nums  
                    SET element at position i of nums TO element at position i PLUS one of nums  
                    SET element at position i PLUS one of nums TO temporary_variable  
                END IF  
            ELSE  
                IF element at position i of nums IS LESS THAN element at position i PLUS one of nums  
                    SET temporary_variable TO element at position i of nums  
                    SET element at position i of nums TO element at position i PLUS one of nums  
                    SET element at position i PLUS one of nums TO temporary_variable  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
