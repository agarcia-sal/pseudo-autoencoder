CLASS Solution  
    FUNCTION getSum(a parameter of type integer b parameter of type integer) RETURNS integer  
        SET MAX TO the maximum value of a thirty two bits integer  
        SET mask TO the mask value for thirty two bits  
        WHILE the value of b is NOT EQUALS to zero  
            SET carry TO the bitwise AND operation of a AND b followed by shifting left by one position  
            SET a TO the bitwise XOR operation of a AND b followed by bitwise AND operation with mask  
            SET b TO the bitwise AND operation of carry AND mask  
        END WHILE  
        IF the value of a is LESS THAN OR EQUAL TO MAX  
            RETURN the value of a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR operation of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
