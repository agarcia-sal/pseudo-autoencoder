CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE deque named radiant  
        INITIALIZE deque named dire  
        
        FOR each index named i AND element named s IN the enumeration of senate  
            IF element at position i of senate EQUALS letter R  
                APPEND index i TO radiant  
            ELSE  
                APPEND index i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant is NOT EMPTY AND dire is NOT EMPTY  
            REMOVE and RETURN the element at the front of radiant INTO r_index  
            REMOVE and RETURN the element at the front of dire INTO d_index  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant is NOT EMPTY  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
