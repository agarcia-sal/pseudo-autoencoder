CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the list of tuples formed by combining corresponding elements of startTime, endTime, and profit  
        SORT jobs in order of the second element of each tuple representing endTime in ascending order  
        
        SET end_times TO an empty list  
        FOR each job IN jobs  
            APPEND the second element of job representing endTime TO end_times  
        END FOR  
        
        SET dp TO a list of zeros with the length equal to the count of jobs PLUS one  
        
        FOR index FROM one TO the count of jobs  
            SET current_start TO the first element of the job at position index MINUS one in jobs representing startTime  
            SET j TO the result of binary search for the position to insert current_start in end_times to keep it sorted on the right side  
            
            SET profit_without_current TO the element at position index MINUS one of dp  
            SET profit_with_current TO the element at position j of dp PLUS the third element of the job at position index MINUS one in jobs representing profit  
            
            IF profit_without_current GREATER THAN profit_with_current  
                SET dp element at position index TO profit_without_current  
            ELSE  
                SET dp element at position index TO profit_with_current  
            END IF  
        END FOR  
        
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
