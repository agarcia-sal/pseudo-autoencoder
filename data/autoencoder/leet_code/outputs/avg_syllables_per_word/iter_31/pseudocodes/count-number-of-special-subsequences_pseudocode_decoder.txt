CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO the number one followed by nine zeros PLUS one  
        SET dp TO a list containing three zeros  
        
        FOR each num IN nums  
            IF num EQUALS zero  
                SET element at position zero of dp TO the quantity two MULTIPLIED BY element at position zero of dp PLUS one  
                SET element at position zero of dp TO element at position zero of dp MODULO MOD  
            ELSE IF num EQUALS one  
                SET element at position one of dp TO the quantity two MULTIPLIED BY element at position one of dp PLUS element at position zero of dp  
                SET element at position one of dp TO element at position one of dp MODULO MOD  
            ELSE IF num EQUALS two  
                SET element at position two of dp TO the quantity two MULTIPLIED BY element at position two of dp PLUS element at position one of dp  
                SET element at position two of dp TO element at position two of dp MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
