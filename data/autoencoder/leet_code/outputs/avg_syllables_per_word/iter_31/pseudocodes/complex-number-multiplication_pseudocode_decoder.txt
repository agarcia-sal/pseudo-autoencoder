CLASS Solution  
    FUNCTION complexNumberMultiply(num1 num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the result of splitting num by the plus character  
            SET real TO the integer conversion of the element at position one of parts  
            SET imaginary TO the integer conversion of the substring of the element at position two of parts from the first position TO the second last position  
            RETURN real AND imaginary  
        END FUNCTION  

        SET real1 AND imag1 TO the result of calling parse_complex with num1  
        SET real2 AND imag2 TO the result of calling parse_complex with num2  

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  

        RETURN the string formed by concatenating real_part PLUS the plus character PLUS imaginary_part PLUS the letter i  
    END FUNCTION  
END CLASS
