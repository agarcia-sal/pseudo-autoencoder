CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF isOneEditDistance with parameters t AND s  
        END IF  

        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  

        IF diff GREATER THAN one  
            RETURN false  
        END IF  

        IF diff EQUALS zero  
            SET difference_count TO zero  
            FOR each position FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position of s NOT EQUALS the element at position of t  
                    INCREMENT difference_count BY one  
                END IF  
            END FOR  
            RETURN whether difference_count EQUALS one  
        ELSE  
            FOR each position FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position of s NOT EQUALS the element at position of t  
                    RETURN whether the substring from position to the END of s EQUALS the substring from position plus one to the END of t  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
