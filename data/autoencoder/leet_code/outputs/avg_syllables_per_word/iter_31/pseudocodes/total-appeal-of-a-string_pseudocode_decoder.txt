CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list with twenty-six elements each set to negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each index i AND character char IN s  
            SET index TO the numerical position of char within the alphabet starting from zero for the first letter  
            INCREMENT current_appeal BY the difference of i MINUS the element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET the element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
