CLASS Solution  
    FUNCTION maxCoins(nums)  
        SET nums TO one APPENDED WITH nums APPENDED WITH one  
        SET n TO the LENGTH OF nums  
        SET dp TO a two dimensional list of zeros with dimensions n BY n  
        FOR length FROM two TO n MINUS one  
            FOR i FROM zero TO n MINUS length MINUS one  
                SET j TO i PLUS length  
                FOR k FROM i PLUS one TO j MINUS one  
                    SET coins TO the element at position i of nums MULTIPLIED BY the element at position k of nums MULTIPLIED BY the element at position j of nums PLUS the element at position i at position k of dp PLUS the element at position k at position j of dp  
                    IF the element at position i at position j of dp LESS THAN coins  
                        SET the element at position i at position j of dp TO coins  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position zero at position n MINUS one of dp  
    END FUNCTION  
END CLASS
