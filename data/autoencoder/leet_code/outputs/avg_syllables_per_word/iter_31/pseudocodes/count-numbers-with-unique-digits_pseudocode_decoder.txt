CLASS Solution  
    FUNCTION countNumbersWithUniqueDigits WITH PARAMETER n  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        IF n EQUALS one  
            RETURN the number ten  
        END IF  
        SET count TO the number ten  
        SET available_digits TO the number nine  
        SET choices TO the number nine  
        FOR i FROM the number two TO n INCLUSIVE  
            SET choices TO choices MULTIPLIED BY available_digits  
            INCREMENT count BY choices  
            DECREMENT available_digits BY one  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
