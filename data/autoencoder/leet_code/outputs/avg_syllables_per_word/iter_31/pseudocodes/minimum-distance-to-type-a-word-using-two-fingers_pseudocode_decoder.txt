CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS no value  
                RETURN zero  
            END IF  
            SET x1 TO the integer division of the difference between the position of character c1 in alphabet and the position of character A in alphabet BY six  
            SET y1 TO the remainder of the division of the difference between the position of character c1 in alphabet and the position of character A in alphabet BY six  
            SET x2 TO the integer division of the difference between the position of character c2 in alphabet and the position of character A in alphabet BY six  
            SET y2 TO the remainder of the division of the difference between the position of character c2 in alphabet and the position of character A in alphabet BY six  
            RETURN the absolute value of the difference between x1 and x2 PLUS the absolute value of the difference between y1 and y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the character at position i of word  
            SET d1 TO the result of distance called with f1 and c PLUS the result of dp called with i incremented by one and c as f1 and f2 as f2  
            SET d2 TO the result of distance called with f2 and c PLUS the result of dp called with i incremented by one and f1 as f1 and c as f2  
            RETURN the minimum of d1 and d2  
        END FUNCTION  
        
        RETURN the result of dp called with zero and no value and no value  
    END FUNCTION  
END CLASS
