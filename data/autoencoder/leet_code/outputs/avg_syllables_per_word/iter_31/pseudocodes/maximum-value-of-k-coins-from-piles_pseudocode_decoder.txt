CLASS Solution  
    FUNCTION maxValueOfCoins WITH parameters piles AND k  
        SET n TO the LENGTH OF piles  
        SET dp TO a two dimensional list of zeroes with number of rows equal to n PLUS one AND number of columns equal to k PLUS one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one OF current_pile  
                    END IF  
                    SET dp at position i AND j TO the GREATER VALUE BETWEEN dp at position i AND j AND dp at position i MINUS one AND j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n AND k  
    END FUNCTION  
END CLASS
