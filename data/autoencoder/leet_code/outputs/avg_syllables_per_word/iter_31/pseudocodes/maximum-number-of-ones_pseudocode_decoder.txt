CLASS Solution  
    FUNCTION maximumNumberOfOnes(width parameter, height parameter, sideLength parameter, maxOnes parameter)  
        SET frequency list TO a list of zeros with length sideLength MULTIPLIED BY sideLength  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET position TO the remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of j DIVIDED BY sideLength  
                INCREMENT element at position position of frequency list BY one  
            END FOR  
        END FOR  
        SORT frequency list in descending order  
        SET total sum TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total sum BY element at position index of frequency list  
        END FOR  
        RETURN total sum  
    END FUNCTION  
END CLASS
