CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            SET pick TO element at position start of slices PLUS dp with parameters start INCREMENTED BY two end unchanged and k DECREMENTED BY one  
            SET skip TO dp with parameters start INCREMENTED BY one end unchanged and k unchanged  
            RETURN the GREATER VALUE BETWEEN pick AND skip  
        END FUNCTION  
        
        SET case1 TO dp with parameters one end EQUALS the LENGTH OF slices MINUS one and k EQUALS n  
        SET case2 TO dp with parameters zero end EQUALS the LENGTH OF slices MINUS two and k EQUALS n  
        
        RETURN the GREATER VALUE BETWEEN case1 AND case2  
    END FUNCTION  
END CLASS
