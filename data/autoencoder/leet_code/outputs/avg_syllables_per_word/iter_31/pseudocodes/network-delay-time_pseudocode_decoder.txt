CLASS Solution  
    FUNCTION networkDelayTime(times n k)  
        SET graph TO an empty mapping from each integer from one TO n TO an empty list  
        FOR each element IN times  
            SET u TO the first element of element  
            SET v TO the second element of element  
            SET w TO the third element of element  
            APPEND pair of v AND w TO the list at key u in graph  
        END FOR  
        SET dist TO a mapping from each integer from one TO n TO infinity  
        SET dist at key k TO zero  
        SET min_heap TO a list containing pair of zero AND k  
        WHILE min_heap is not empty  
            REMOVE and SET the pair from min_heap with the smallest first element TO current_dist AND node  
            IF current_dist GREATER THAN the value at key node in dist THEN  
                CONTINUE to the next iteration  
            END IF  
            FOR each pair neighbor AND weight IN the list at key node in graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN the value at key neighbor in dist THEN  
                    SET dist at key neighbor TO distance  
                    ADD pair of distance AND neighbor TO min_heap maintaining min_heap property  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value among all values in dist  
        IF max_time EQUALS infinity THEN  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
