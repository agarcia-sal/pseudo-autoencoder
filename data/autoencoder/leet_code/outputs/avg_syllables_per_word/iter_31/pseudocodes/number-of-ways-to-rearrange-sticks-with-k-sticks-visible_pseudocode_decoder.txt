CLASS Solution  
    FUNCTION rearrangeSticks(n integer, k integer) RETURNS integer  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        SET dp TO a two dimensional list with n PLUS one rows and k PLUS one columns filled with zero  
        
        SET the element at row zero and column zero of dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET value_visible TO the element at row i MINUS one and column j MINUS one of dp  
                SET value_not_visible TO the product of i MINUS one AND the element at row i MINUS one and column j of dp  
                SET the element at row i and column j of dp TO the remainder of the sum of value_visible AND value_not_visible when divided by MOD  
            END FOR  
        END FOR  
        
        RETURN the element at row n and column k of dp  
    END FUNCTION  
END CLASS
