CLASS Solution  
    FUNCTION pyramidTransition(bottom, allowed)  
        SET allowed_map TO a new default dictionary whose default value is a default dictionary whose default value is an empty list  
        FOR each rule IN allowed  
            SET left TO element at position zero of rule  
            SET right TO element at position one of rule  
            SET top TO element at position two of rule  
            APPEND top TO the list at allowed_map at left at right  
        END FOR  
        
        FUNCTION can_build_pyramid(current_bottom, current_top)  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN True  
            END IF  
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of calling can_build_pyramid with current_top and an empty list  
            END IF  
            
            SET left TO element at the position given by the LENGTH OF current_top of current_bottom  
            SET right TO element at the position given by the LENGTH OF current_top PLUS one of current_bottom  
            IF left IS IN allowed_map AND right IS IN allowed_map at left  
                FOR each top IN the list at allowed_map at left at right  
                    IF the result of calling can_build_pyramid with current_bottom and the concatenation of current_top and top EQUALS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
            RETURN False  
        END FUNCTION  
        
        RETURN the result of calling can_build_pyramid with bottom and an empty list  
    END FUNCTION  
END CLASS
