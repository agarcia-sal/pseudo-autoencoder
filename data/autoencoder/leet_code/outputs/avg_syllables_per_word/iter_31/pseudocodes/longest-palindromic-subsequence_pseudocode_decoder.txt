CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        CREATE a two dimensional array dp with dimensions n BY n filled with zeroes  
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND position i of dp TO one  
        END FOR  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i PLUS one AND position j MINUS one of dp PLUS two TO element at position i AND position j of dp  
                ELSE  
                    SET element at position i AND position j of dp TO the GREATER VALUE BETWEEN element at position i PLUS one AND position j of dp AND element at position i AND position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero AND position n MINUS one of dp  
    END FUNCTION  
END CLASS
