CLASS Solution  
    FUNCTION spiralMatrixIII(rows r_of_type_integer cols c_of_type_integer rStart cStart) RETURNS a list of lists of integers  
        SET directions TO a list of four pairs representing zero and one in the pattern zero one zero negative one negative one zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO an empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited is LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each element IN the range from zero TO steps MINUS one  
                IF x is GREATER THAN OR EQUAL TO zero AND x is LESS THAN rows AND y is GREATER THAN OR EQUAL TO zero AND y is LESS THAN cols  
                    APPEND a list containing x and y TO visited  
                END IF  
                SET dx TO the element at position direction_index of directions and first part of the pair  
                SET dy TO the element at position direction_index of directions and second part of the pair  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
