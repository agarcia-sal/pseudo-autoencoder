CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with val set to the element at position one of values and left set to None and right set to None  
    SET i TO one  
    INITIALIZE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE and ASSIGN the element from the front of queue TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None  
            SET node.left TO a new TreeNode with val set to the element at position i of values and left set to None and right set to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None  
            SET node.right TO a new TreeNode with val set to the element at position i of values and left set to None and right set to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN a new TreeNode with val set to val and left set to root and right set to None  
        END IF  
        INITIALIZE a double ended queue named queue with a tuple containing root and one as the first element  
        WHILE queue is not empty  
            REMOVE and ASSIGN the element from the front of queue TO node and level  
            IF level EQUALS depth MINUS one  
                SET node.left TO a new TreeNode with val set to val and left set to node.left and right set to None  
                SET node.right TO a new TreeNode with val set to val and left set to None and right set to node.right  
            ELSE IF level LESS THAN depth MINUS one  
                IF node.left IS NOT None  
                    APPEND a tuple containing node.left and level PLUS one TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND a tuple containing node.right and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
