CLASS Solution  
    FUNCTION visiblePoints(points PARAMETERS points AS list of list of integers angle AS integer location AS list of integers) RETURNS integer  
        SET angles TO empty list  
        SET extra_points TO zero  
        FOR each x COMMA y IN points  
            IF x EQUALS element at position zero of location AND y EQUALS element at position one of location  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_in_radians TO arctangent of the result of subtracting element at position one of location from y DIVIDED BY the result of subtracting element at position zero of location from x  
            SET angle_in_degrees TO conversion to degrees of angle_in_radians  
            APPEND angle_in_degrees TO angles  
        END FOR  
        SORT angles in ascending order  
        SET extended_angles TO angles concatenated with a new list formed by adding three hundred sixty to each element in angles  
        SET max_visible TO zero  
        SET left TO zero  
        FOR each right FROM zero TO length of extended_angles MINUS one  
            WHILE the difference between element at position right of extended_angles and element at position left of extended_angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET max_visible TO maximum of max_visible AND the result of subtracting left from right PLUS one  
        END FOR  
        RETURN the sum of max_visible AND extra_points  
    END FUNCTION  
END CLASS
