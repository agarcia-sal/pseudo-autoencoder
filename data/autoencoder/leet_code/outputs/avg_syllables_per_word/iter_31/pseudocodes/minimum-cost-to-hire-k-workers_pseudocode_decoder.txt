CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the sorted list of pairs consisting of the division of each element of wage by the corresponding element of quality and the corresponding element of quality for each pair of elements taken from wage and quality simultaneously  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO a value representing positive infinity  
        FOR each pair ratio and q IN workers  
            ADD the negative of q TO max_heap USING heap push operation  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap is GREATER THAN k THEN  
                REMOVE the smallest element FROM max_heap USING heap pop operation AND INCREMENT total_quality BY the removed element  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET min_cost TO the smaller value BETWEEN min_cost AND the MULTIPLICATION of total_quality AND ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
