CLASS Solution  
    FUNCTION minSkips(dist speed hoursBefore)  
        SET n TO the LENGTH OF dist  
        CREATE dp AS a two dimensional list with n rows AND n PLUS one columns EACH FILLED WITH infinity value  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at position zero at position zero OF dp TO element at position zero OF dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero at position j OF dp TO element at position zero OF dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i THEN  
                    SET element at position i at position j OF dp TO the CEILING OF the element at position i MINUS one at position j OF dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i OF dist  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET element at position i at position j OF dp TO the MINIMUM BETWEEN element at position i at position j OF dp AND the SUM OF element at position i MINUS one at position j MINUS one OF dp AND element at position i OF dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one at position j OF dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
