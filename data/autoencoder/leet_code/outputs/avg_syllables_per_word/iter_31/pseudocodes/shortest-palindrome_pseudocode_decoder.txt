CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS the REVERSE OF check_str  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(substring FROM position zero TO position i MINUS one OF s)  
                SET to_add TO the REVERSE OF substring FROM position i TO the END OF s  
                RETURN the concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
