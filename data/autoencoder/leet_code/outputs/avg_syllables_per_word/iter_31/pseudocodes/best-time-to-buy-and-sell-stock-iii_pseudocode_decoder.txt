CLASS Solution  
    FUNCTION maxProfit(prices)  
        IF the LENGTH OF prices EQUALS zero  
            RETURN zero  
        END IF  
        SET min_price1 TO infinity  
        SET min_price2 TO infinity  
        SET max_profit1 TO zero  
        SET max_profit2 TO zero  
        FOR each price IN prices  
            SET min_price1 TO the SMALLER value BETWEEN min_price1 AND price  
            SET max_profit1 TO the GREATER value BETWEEN max_profit1 AND price MINUS min_price1  
            SET min_price2 TO the SMALLER value BETWEEN min_price2 AND price MINUS max_profit1  
            SET max_profit2 TO the GREATER value BETWEEN max_profit2 AND price MINUS min_price2  
        END FOR  
        RETURN max_profit2  
    END FUNCTION  
END CLASS
