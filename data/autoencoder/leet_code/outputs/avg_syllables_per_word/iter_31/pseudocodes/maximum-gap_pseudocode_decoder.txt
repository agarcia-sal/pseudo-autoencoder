CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros with LENGTH n  
            SET count TO a list of zeros with LENGTH ten  
            
            FOR i FROM zero TO n MINUS one  
                SET index TO the INTEGER DIVISION OF element at position i of arr DIVIDED BY exp  
                INCREMENT element at position the REMAINDER OF index DIVIDED BY ten OF count BY one  
            END FOR  
            
            FOR i FROM one TO nine  
                INCREMENT element at position i of count BY element at position i MINUS one of count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET index TO the INTEGER DIVISION OF element at position i of arr DIVIDED BY exp  
                SET element at position element at position the REMAINDER OF index DIVIDED BY ten OF count MINUS one of output TO element at position i of arr  
                DECREMENT element at position the REMAINDER OF index DIVIDED BY ten OF count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum value in arr  
            
            SET exp TO one  
            WHILE the INTEGER DIVISION OF max1 BY exp GREATER THAN zero  
                CALL countingSort with arguments arr and exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort with argument nums  
        
        SET max_diff TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            SET difference TO element at position i of nums MINUS element at position i MINUS one of nums  
            IF difference GREATER THAN max_diff THEN  
                SET max_diff TO difference  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
