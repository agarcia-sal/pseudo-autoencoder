CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the collection values EQUALS no elements  
        RETURN no value  
    END IF  
    SET head TO a new ListNode created with the element at position one of values  
    SET p TO head  
    FOR each val IN the collection of elements starting from position two TO the end of values  
        SET node TO a new ListNode created with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the condition that the value attribute of p1 EQUALS the value attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne with parameter head  
        FUNCTION reverseList with parameter node  
            SET prev TO no value  
            WHILE node EXISTS  
                SET next_node TO the next attribute of node  
                SET the next attribute of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO the result of calling reverseList with head  
        
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current EXISTS AND carry EQUALS one  
            SET the value attribute of current TO the value attribute of current PLUS carry  
            SET carry TO the value attribute of current DIVIDED BY the number ten  
            SET the value attribute of current TO the value attribute of current MODULO the number ten  
            
            IF the next attribute of current DOES NOT EXIST AND carry EQUALS one  
                SET the next attribute of current TO a new ListNode created with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next attribute of current  
        END WHILE  
        
        RETURN the result of calling reverseList with reversed_head  
    END FUNCTION  
END CLASS
