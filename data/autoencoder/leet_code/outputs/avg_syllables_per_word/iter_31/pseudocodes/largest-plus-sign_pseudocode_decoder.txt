CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n  
                                   mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            CONVERT mine TO an ordered pair  
            ADD ordered pair TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of size n by n  
        EACH element of dp IS a list of four zeros representing counts in four directions  
        
        SET directions TO the list containing ordered pairs representing right down left and up directions  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the ordered pair consisting of i and j IS NOT IN mines_set  
                    IF j IS GREATER THAN zero  
                        SET element at position zero of element at position j MINUS one of element at position i in dp TO value  
                        SET element at position zero of element at position j of element at position i in dp TO value PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero  
                        SET element at position one of element at position j of element at position i MINUS one in dp TO value  
                        SET element at position one of element at position j of element at position i in dp TO value PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one TO zero stepping minus one  
            FOR j FROM n MINUS one TO zero stepping minus one  
                IF the ordered pair consisting of i and j IS NOT IN mines_set  
                    IF j IS LESS THAN n MINUS one  
                        SET element at position two of element at position j PLUS one of element at position i in dp TO value  
                        SET element at position two of element at position j of element at position i in dp TO value PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one  
                        SET element at position three of element at position j of element at position i PLUS one in dp TO value  
                        SET element at position three of element at position j of element at position i in dp TO value PLUS one  
                    END IF  
                    
                    SET plus_sign_order TO the minimum value among the four elements of dp at position i and j  
                    IF plus_sign_order IS GREATER THAN result  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
