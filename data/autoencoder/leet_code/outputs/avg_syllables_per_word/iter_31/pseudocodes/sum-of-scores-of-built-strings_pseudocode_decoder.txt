CLASS Solution  
    FUNCTION sumScores WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET lcp TO a list of length n with each element SET TO zero  
        SET l TO zero  
        SET r TO zero  
        FOR i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r THEN  
                SET lcp element at position i TO the MINIMUM OF r MINUS i PLUS one AND lcp element at position i MINUS l  
            END IF  
            WHILE i PLUS lcp element at position i LESS THAN n AND character at position lcp element at position i of s EQUALS character at position i PLUS lcp element at position i of s DO  
                INCREMENT lcp element at position i BY one  
            END WHILE  
            IF i PLUS lcp element at position i MINUS one GREATER THAN r THEN  
                SET l TO i  
                SET r TO i PLUS lcp element at position i MINUS one  
            END IF  
        END FOR  
        RETURN the SUM OF all elements IN lcp PLUS n  
    END FUNCTION  
END CLASS
