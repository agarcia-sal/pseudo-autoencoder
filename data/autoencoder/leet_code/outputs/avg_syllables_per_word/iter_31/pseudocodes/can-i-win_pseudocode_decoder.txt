CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the PRODUCT of maxChoosableInteger AND the SUM of maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  
        
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers IS A KEY IN memo  
                RETURN the value in memo at the key of current_total AND used_numbers  
            END IF  
            
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the left shift of one BY number IS EQUALS zero  
                    IF the SUM of current_total AND number GREATER THAN OR EQUAL TO desiredTotal OR the negation of can_win with the SUM of current_total AND number AND the bitwise OR of used_numbers AND the left shift of one BY number  
                        SET the value in memo at the key of current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the value in memo at the key of current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win with zero AND zero  
    END FUNCTION  
END CLASS
