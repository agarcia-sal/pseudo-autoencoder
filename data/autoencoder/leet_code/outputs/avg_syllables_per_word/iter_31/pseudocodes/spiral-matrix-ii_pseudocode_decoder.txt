CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO a list containing n elements where each element is a list containing n elements each set to zero  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE the condition top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right holds  
            FOR i FROM left TO right  
                SET the element at position top of matrix at position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                SET the element at position i of matrix at position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF the condition top LESS THAN OR EQUAL TO bottom holds  
                FOR i FROM right DOWN TO left  
                    SET the element at position bottom of matrix at position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF the condition left LESS THAN OR EQUAL TO right holds  
                FOR i FROM bottom DOWN TO top  
                    SET the element at position i of matrix at position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
