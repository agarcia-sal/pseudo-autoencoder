CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty collection of unique elements  
        SET stack TO collection containing element zero  
        WHILE the LENGTH OF stack IS GREATER THAN zero  
            SET current_room TO the last element of stack REMOVED from stack  
            ADD current_room TO visited  
            FOR each key IN the collection at position current_room of rooms  
                IF key IS NOT IN visited  
                    ADD key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF visited EQUALS the LENGTH OF rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
