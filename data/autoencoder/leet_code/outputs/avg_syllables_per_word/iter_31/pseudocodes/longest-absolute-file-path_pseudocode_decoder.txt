CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input by the newline character  
        SET stack TO an empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the count of the tab character within line  
            SET name TO the substring of line starting from the position after depth number of tab characters to the end  
            
            WHILE the LENGTH OF stack is GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the LENGTH OF stack is GREATER THAN zero  
                SET current_length TO the last element of stack PLUS the LENGTH OF name PLUS one  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF the character representing a dot EXISTS IN name  
                IF max_length is LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
