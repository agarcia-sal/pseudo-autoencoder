CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  

        SET below_twenty TO the list of words corresponding to numbers from zero to nineteen in order  
        SET tens TO the list of words corresponding to multiples of ten from zero to ninety in order  
        SET thousands TO the list of scale words empty string Thousand Million Billion in order  

        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a space  
            END IF  
            IF n LESS THAN one hundred  
                RETURN the element at position of the integer division of n by ten of tens concatenated with a space concatenated with the result of calling helper with the remainder of n divided by ten  
            END IF  
            RETURN the element at position of the integer division of n by one hundred of below_twenty concatenated with the word Hundred concatenated with a space concatenated with the result of calling helper with the remainder of n divided by one hundred  
        END FUNCTION  

        SET result TO empty string  
        SET i TO zero  

        WHILE num GREATER THAN zero holds  
            IF the remainder of num divided by one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num divided by one thousand concatenated with the element at position i of thousands concatenated with a space concatenated with result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  

        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
