CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs  
        SET anagrams TO an empty dictionary with default value of empty list  
        FOR each s IN strs  
            SET sorted_str TO the string formed by arranging characters of s in ascending alphabetical order  
            APPEND s TO the list at key sorted_str in anagrams  
        END FOR  
        SET result TO the list of all values contained in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
