CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val MINUS DEFAULT zero left MINUS DEFAULT None right MINUS DEFAULT None  
        SET the value property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element REMOVED from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the left property of node TO a new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the right property of node TO a new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left property of p AND the left property of q  
        AND the result of is_same_tree WITH PARAMETERS the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover WITH PARAMETERS root  
        FUNCTION dfs WITH PARAMETERS node  
            IF node EQUALS None  
                RETURN zero comma zero comma infinity  
            END IF  
            SET left_no_camera comma left_monitored comma left_camera TO the result of calling dfs WITH the left property of node  
            SET right_no_camera comma right_monitored comma right_camera TO the result of calling dfs WITH the right property of node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the minimum of the sum of left_camera PLUS the minimum of right_monitored AND right_camera  
            AND the sum of right_camera PLUS the minimum of left_monitored AND left_camera  
            AND the sum of left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum of left_no_camera AND left_monitored AND left_camera  
            PLUS the minimum of right_no_camera AND right_monitored AND right_camera  
            RETURN no_camera comma monitored comma camera  
        END FUNCTION  
        SET root_no_camera comma root_monitored comma root_camera TO the result of calling dfs WITH root  
        RETURN the minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
