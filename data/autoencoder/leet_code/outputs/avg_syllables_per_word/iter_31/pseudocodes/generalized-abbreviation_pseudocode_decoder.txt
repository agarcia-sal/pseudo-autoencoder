CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all elements IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH the STRING REPRESENTATION OF count AND the ELEMENT AT position index OF word AND count SET TO zero  
                ELSE  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH the ELEMENT AT position index OF word AND count SET TO zero  
                END IF  
                CALL backtrack WITH index INCREMENTED BY one AND path UNCHANGED AND count INCREMENTED BY one  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH zero AS index AND an empty list AS path AND zero AS count  
        RETURN result  
    END FUNCTION  
END CLASS
