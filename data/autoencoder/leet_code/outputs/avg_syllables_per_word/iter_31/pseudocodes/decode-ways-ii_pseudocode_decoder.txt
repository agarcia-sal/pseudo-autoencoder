CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        SET dp TO a list of zeros with size the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character star  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character star AND char2 EQUALS the character star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character star  
                IF the character at char2 is LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character star  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value formed by the concatenation of char1 and char2  
                IF num IS GREATER THAN OR EQUAL TO ten AND num IS LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO the LENGTH OF s  
            SET element at position i of dp TO the element at position i of dp PLUS the element at position i MINUS one of dp MULTIPLIED BY ways_to_decode_single(element at position i MINUS one of s)  
            SET element at position i of dp TO the remainder when the element at position i of dp IS DIVIDED BY MOD  
            
            IF i IS GREATER THAN one  
                SET element at position i of dp TO the element at position i of dp PLUS the element at position i MINUS two of dp MULTIPLIED BY ways_to_decode_pair(element at position i MINUS two of s, element at position i MINUS one of s)  
                SET element at position i of dp TO the remainder when the element at position i of dp IS DIVIDED BY MOD  
            END IF  
        END FOR  
        
        RETURN element at position the LENGTH OF dp MINUS one of dp  
    END FUNCTION  
END CLASS
