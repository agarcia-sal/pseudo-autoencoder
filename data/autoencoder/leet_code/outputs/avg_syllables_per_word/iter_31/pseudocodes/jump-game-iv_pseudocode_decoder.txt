CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new mapping from each value TO an empty list  
        FOR each position i AND element value IN arr with their positions  
            APPEND i TO the list mapped by value_indices for value  
        END FOR  
        
        SET queue TO a new double ended queue containing the pair of zero and zero  
        SET visited TO a new set containing zero  
        
        WHILE queue is not empty  
            REMOVE the first element FROM queue AND ASSIGN it TO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list consisting of current_index PLUS one, current_index MINUS one, AND all elements in the list mapped by value_indices for element at position current_index of arr  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list mapped by value_indices for element at position current_index of arr TO an empty list  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
