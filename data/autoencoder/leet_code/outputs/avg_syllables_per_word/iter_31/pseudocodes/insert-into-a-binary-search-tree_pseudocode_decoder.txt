CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to the element at position one of values and left equal to none and right equal to none  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO new TreeNode with value equal to the element at position i of values and left equal to none and right equal to none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO new TreeNode with value equal to the element at position i of values and left equal to none and right equal to none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS TO the value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments the left child of p and the left child of q AND is_same_tree with arguments the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root EQUALS none  
                RETURN new TreeNode with value equal to val and left equal to none and right equal to none  
            END IF  
            IF the value of root LESS THAN val  
                SET the right child of root TO dfs with argument the right child of root  
            ELSE  
                SET the left child of root TO dfs with argument the left child of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs with argument root  
    END FUNCTION  
END CLASS
