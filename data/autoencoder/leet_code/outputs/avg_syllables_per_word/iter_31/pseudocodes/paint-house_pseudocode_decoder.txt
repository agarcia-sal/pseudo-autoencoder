CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs EQUALS empty list  
            RETURN zero  
        END IF  
        
        SET previous red cost TO element at position zero of element at position zero of costs  
        SET previous blue cost TO element at position one of element at position zero of costs  
        SET previous green cost TO element at position two of element at position zero of costs  
        
        FOR index i FROM one TO the length of costs MINUS one  
            SET current red cost TO element at position zero of element at position i of costs PLUS the smaller value BETWEEN previous blue cost AND previous green cost  
            SET current blue cost TO element at position one of element at position i of costs PLUS the smaller value BETWEEN previous red cost AND previous green cost  
            SET current green cost TO element at position two of element at position i of costs PLUS the smaller value BETWEEN previous red cost AND previous blue cost  
            
            SET previous red cost TO current red cost  
            SET previous blue cost TO current blue cost  
            SET previous green cost TO current green cost  
        END FOR  
        
        RETURN the smaller value BETWEEN previous red cost AND the smaller value BETWEEN previous blue cost AND previous green cost  
    END FUNCTION  
END CLASS
