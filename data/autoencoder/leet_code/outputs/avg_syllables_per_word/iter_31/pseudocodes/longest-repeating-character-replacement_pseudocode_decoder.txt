CLASS Solution  
    FUNCTION characterReplacement(s k)  
        SET maximum length TO zero  
        SET maximum count TO zero  
        SET left pointer TO zero  
        ASSIGN count TO default dictionary mapping each character TO zero count  
        
        FOR each right pointer FROM zero TO the length of s MINUS one  
            INCREMENT the count of the element at position right pointer of s BY one  
            IF the maximum count is LESS THAN the count of the element at position right pointer of s  
                SET maximum count TO the count of the element at position right pointer of s  
            END IF  
            
            IF the length of the current window calculated as right pointer MINUS left pointer PLUS one MINUS maximum count is GREATER THAN k  
                DECREMENT the count of the element at position left pointer of s BY one  
                INCREMENT left pointer BY one  
            END IF  
            
            IF maximum length is LESS THAN the length of the current window calculated as right pointer MINUS left pointer PLUS one  
                SET maximum length TO the length of the current window calculated as right pointer MINUS left pointer PLUS one  
            END IF  
        END FOR  
        
        RETURN maximum length  
    END FUNCTION  
END CLASS
