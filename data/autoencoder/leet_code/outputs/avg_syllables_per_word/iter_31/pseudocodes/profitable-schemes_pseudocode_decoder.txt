CLASS Solution  
    FUNCTION profitableSchemes(n profit_minimum group profit_list) RETURNS integer  
        SET constant_modulus TO ten TO THE POWER OF nine PLUS one  
        CREATE two_dimensional_list dp WITH rows equal TO n PLUS one AND columns equal TO profit_minimum PLUS one FILLED WITH zeroes  
        FOR variable i FROM zero TO n INCLUSIVE  
            SET element at position zero of list at position i of dp TO one  
        END FOR  
        FOR each element p IN profit_list AND each element g IN group SIMULTANEOUSLY  
            FOR variable i FROM n DOWN TO g  
                FOR variable j FROM profit_minimum DOWN TO zero  
                    SET maximum_profit TO the GREATER VALUE BETWEEN zero AND j MINUS p  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position maximum_profit of list at position i MINUS g of dp  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO constant_modulus  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position profit_minimum of list at position n of dp  
    END FUNCTION  
END CLASS
