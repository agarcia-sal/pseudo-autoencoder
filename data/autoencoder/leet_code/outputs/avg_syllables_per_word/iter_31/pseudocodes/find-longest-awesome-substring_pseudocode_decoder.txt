CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping from zero TO minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN the enumeration of s  
            SET digit TO the integer value of char  
            TOGGLE the bit in prefix_xor corresponding to digit by applying prefix_xor XORED WITH one SHIFTED LEFT BY digit positions  
            
            IF prefix_xor EXISTS AS A KEY IN prefix_map THEN  
                SET max_length TO the maximum between max_length AND the difference of i MINUS the value at prefix_xor in prefix_map  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO the result of prefix_xor XORED WITH one SHIFTED LEFT BY j positions  
                IF flipped EXISTS AS A KEY IN prefix_map THEN  
                    SET max_length TO the maximum between max_length AND the difference of i MINUS the value at flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST AS A KEY IN prefix_map THEN  
                SET in prefix_map the key prefix_xor TO the value i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
