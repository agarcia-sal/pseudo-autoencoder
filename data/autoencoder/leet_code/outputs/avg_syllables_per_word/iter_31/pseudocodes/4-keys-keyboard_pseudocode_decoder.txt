CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  
          
        SET dp TO a list of zeros with the length of n PLUS one  
          
        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  
          
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET product TO the element at position j of dp MULTIPLIED BY i MINUS j MINUS one  
                SET element at position i of dp TO the maximum BETWEEN the element at position i of dp AND product  
            END FOR  
        END FOR  
          
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
