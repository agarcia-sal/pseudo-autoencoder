CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO the iterator over the characters of s  
            RETURN the result of checking for each character in word whether the character occurs in iterator in sequence  
        END FUNCTION  
        
        SORT dictionary in order such that longer words come before shorter words AND words with the same length are ordered in ascending lexicographical order  
        
        FOR each word IN dictionary  
            IF calling is_subsequence with word and s RETURNS true THEN  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN the empty string  
    END FUNCTION  
END CLASS
