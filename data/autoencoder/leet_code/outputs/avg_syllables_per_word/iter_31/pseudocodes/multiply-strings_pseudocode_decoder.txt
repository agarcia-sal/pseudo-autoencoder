CLASS Solution  
    FUNCTION multiply(num1  
                      num2)  
        IF num1 EQUALS string zero OR num2 EQUALS string zero  
            RETURN string zero  
        END IF  
        SET result TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the string formed by elements IN num1 from last position TO first position  
        SET num2 TO the string formed by elements IN num2 from last position TO first position  
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the difference of the character code of the element at position i of num1 MINUS character code of character zero MULTIPLIED BY the difference of the character code of the element at position j of num2 MINUS character code of character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                SET the element at position p1 of result TO the remainder of sum_ DIVIDED BY ten  
                INCREMENT the element at position p2 of result BY the quotient of sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation of string representations of elements of result in reversed order  
        RETURN the string formed by removing leading characters equal to string zero from result_str  
    END FUNCTION  
END CLASS
