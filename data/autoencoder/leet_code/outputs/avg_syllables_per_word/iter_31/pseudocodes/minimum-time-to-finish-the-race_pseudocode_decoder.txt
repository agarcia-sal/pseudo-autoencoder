CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of infinity values repeated fifteen times  
        FOR each pair of first_value AND rate IN tires  
            SET total_time TO zero  
            FOR i FROM zero TO fourteen  
                SET lap_time TO first_value MULTIPLIED BY rate RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS first_value  
                    BREAK the inner loop  
                END IF  
                INCREMENT total_time BY lap_time  
                IF min_times at position i GREATER THAN total_time  
                    SET min_times at position i TO total_time  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO a list of infinity values repeated numLaps PLUS one times  
        SET dp at position zero TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET candidate_time TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                IF dp at position i GREATER THAN candidate_time  
                    SET dp at position i TO candidate_time  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
