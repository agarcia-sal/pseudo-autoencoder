CLASS Solution  
    FUNCTION minOperations(target  
                           arr)  
        SET index_map TO empty list of pairs  
        FOR each index FROM zero TO the LENGTH OF target MINUS one  
            APPEND pair of element at position index of target AND index TO index_map  
        END FOR  
        SET mapping TO empty collection of pairs  
        FOR each pair of value AND index IN index_map  
            ASSIGN value AS key AND index AS value IN mapping  
        END FOR  
        SET transformed_arr TO empty list  
        FOR each value IN arr  
            IF value IS IN mapping  
                APPEND the value CORRESPONDING TO key value IN mapping TO transformed_arr  
            END IF  
        END FOR  
        SET lis TO empty list  
        FOR each num IN transformed_arr  
            SET pos TO position of insertion of num IN lis USING binary search for left insertion  
            IF pos EQUALS the LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
        SET lcs_length TO the LENGTH OF lis  
        SET result TO the LENGTH OF target MINUS lcs_length  
        RETURN result  
    END FUNCTION  
END CLASS
