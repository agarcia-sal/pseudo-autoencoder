CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the pair x AND y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the pair x AND y TO visited  
            FOR each pair dx AND dy IN the list of pairs consisting of minus one AND zero PLUS one AND zero PLUS zero AND minus one PLUS zero AND one  
                CALL dfs WITH x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO a list OF pairs i AND j FOR each i FROM zero TO the LENGTH OF grid MINUS one AND each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one WHERE the element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN true  
            END IF  
            CALL dfs WITH the element at position zero of land_cells at position zero AND the element at position zero of land_cells at position one AND visited  
            RETURN the CONDITION that the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT the result of calling is_connected()  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF the element at position i of grid at position j EQUALS one  
                    SET the element at position i of grid at position j TO zero  
                    IF NOT the result of calling is_connected()  
                        RETURN one  
                    END IF  
                    SET the element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
