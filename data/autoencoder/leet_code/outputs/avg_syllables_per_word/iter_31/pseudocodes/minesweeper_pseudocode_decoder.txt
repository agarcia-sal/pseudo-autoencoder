CLASS Solution  
    FUNCTION updateBoard(board parameter LIST OF LIST OF STRING click parameter LIST OF INTEGER) RETURNS LIST OF LIST OF STRING  
        FUNCTION count_adjacent_mines(r parameter INTEGER c parameter INTEGER) RETURNS INTEGER  
            SET count TO zero  
            FOR i FROM r MINUS one TO r PLUS one  
                FOR j FROM c MINUS one TO c PLUS one  
                    IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF board AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of board AND element at position i of board AT position j EQUALS the STRING representing a single M  
                        INCREMENT count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        FUNCTION dfs(r parameter INTEGER c parameter INTEGER)  
            IF NOT r GREATER THAN OR EQUAL TO zero AND r LESS THAN the LENGTH OF board AND c GREATER THAN OR EQUAL TO zero AND c LESS THAN the LENGTH OF element at position zero of board  
                RETURN  
            END IF  
            IF element at position r of board AT position c NOT EQUALS the STRING representing a single E  
                RETURN  
            END IF  
            
            SET mine_count TO the RESULT OF count_adjacent_mines WITH r AND c  
            IF mine_count GREATER THAN zero  
                SET element at position r of board AT position c TO the STRING representation of mine_count  
            ELSE  
                SET element at position r of board AT position c TO the STRING representing a single B  
                FOR i FROM r MINUS one TO r PLUS one  
                    FOR j FROM c MINUS one TO c PLUS one  
                        CALL dfs WITH i AND j  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET row TO element at position zero of click  
        SET col TO element at position one of click  
        IF element at position row of board AT position col EQUALS the STRING representing a single M  
            SET element at position row of board AT position col TO the STRING representing a single X  
            RETURN board  
        END IF  
        
        CALL dfs WITH row AND col  
        RETURN board  
    END FUNCTION  
END CLASS
