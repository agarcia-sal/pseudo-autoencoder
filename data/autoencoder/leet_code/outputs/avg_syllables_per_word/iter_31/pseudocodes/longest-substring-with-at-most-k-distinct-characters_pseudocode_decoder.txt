CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s k  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO an empty dictionary  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each right FROM zero TO the length of s MINUS one  
            SET the element at position right of s AS key in char_map TO right  
            
            IF the number of keys in char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map with the minimum associated value  
                REMOVE leftmost_char FROM char_map AND ADD one TO the removed value AND ASSIGN TO left  
            END IF  
            
            SET max_length TO the maximum of max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
