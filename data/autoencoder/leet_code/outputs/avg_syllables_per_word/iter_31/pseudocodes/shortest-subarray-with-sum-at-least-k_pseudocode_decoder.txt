CLASS Solution  
    FUNCTION shortestSubarray(nums k)  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            APPEND the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        ASSIGN reference TO an empty double ended queue dq  
        SET min_length TO infinity  
        
        FOR each index i AND element current_sum IN prefix_sum WITH indices  
            WHILE dq is not empty AND current_sum MINUS element at position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller of min_length AND i MINUS the element removed from the front of dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at the last position of dq in prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
