CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET is_palindrome TO a two dimensional collection of BOOLEAN values with dimensions n BY n where all are FALSE  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at position i of is_palindrome TO TRUE  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start at position end of is_palindrome TO the condition that element at position start of s EQUALS element at position end of s  
                ELSE  
                    SET element at position start at position end of is_palindrome TO the logical AND of  
                        the condition that element at position start of s EQUALS element at position end of s  
                        AND element at position start PLUS one at position end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        SET min_cuts TO a collection of length n where all elements are zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero at position i of is_palindrome  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO a representation of infinity  
                FOR j FROM one TO i  
                    IF logical AND of  
                        element at position j at position i of is_palindrome  
                        AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                    THEN  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
