CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative_sum_index TO mapping of zero TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND number num IN enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF cumulative_sum MINUS k EXISTS AS A KEY IN cumulative_sum_index  
                SET max_length TO maximum of max_length AND index i MINUS the element at key of cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            IF cumulative_sum DOES NOT EXIST AS A KEY IN cumulative_sum_index  
                SET the element at key cumulative_sum in cumulative_sum_index TO index i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
