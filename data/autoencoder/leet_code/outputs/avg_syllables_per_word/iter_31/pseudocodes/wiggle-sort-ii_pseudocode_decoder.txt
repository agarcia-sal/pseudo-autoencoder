CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort the list nums in ascending order  
        SET sorted_nums TO the full copy of the list nums  
        SET n TO the LENGTH OF the list nums  
        SET left TO the integer division of n MINUS one BY two  
        SET right TO n MINUS one  
        FOR each index i FROM zero TO n MINUS one  
            IF the remainder of i DIVIDED BY two EQUALS zero  
                SET element at position i of the list nums TO element at position left of the list sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i of the list nums TO element at position right of the list sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
