CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two-dimensional collection with n elements in each dimension where each element is set to True  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two TO zero DECREMENT BY one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET the element at position i and position j of f TO False  
                IF the element at position i of s EQUALS the element at position j of s  
                    SET the element at position i and position j of f TO the element at position i PLUS one and position j MINUS one of f  
                    IF the element at position i and position j of f AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k TO position k PLUS mx MINUS one  
    END FUNCTION  
END CLASS
