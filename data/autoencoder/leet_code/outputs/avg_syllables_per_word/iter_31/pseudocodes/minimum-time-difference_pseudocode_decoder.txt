CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET split_time TO the ELEMENTS of time SPLIT AT the OCCURRENCE OF colon  
            SET hours TO the ELEMENT at position zero of split_time CONVERTED TO an INTEGER  
            SET minutes TO the ELEMENT at position one of split_time CONVERTED TO an INTEGER  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO an EMPTY list  
        FOR each time IN timePoints  
            SET minute_value TO to_minutes WITH ARGUMENT time  
            APPEND minute_value TO minutes_list  
        END FOR  
        
        SORT minutes_list IN ASCENDING ORDER  
        
        SET min_diff TO positive infinity  
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET difference TO the ELEMENT at position index of minutes_list MINUS the ELEMENT at position index MINUS one of minutes_list  
            IF difference LESS THAN min_diff  
                SET min_diff TO difference  
            END IF  
        END FOR  
        
        SET circular_difference TO the NUMBER of minutes in a full day MINUS the ELEMENT at the LAST POSITION of minutes_list PLUS the ELEMENT at position zero of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
