CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO the number one followed by nine zeros PLUS seven  
        SET seats TO an empty list  
        FOR each index position AND corresponding character IN corridor  
            IF character EQUALS the letter S  
                APPEND index position TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index i FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO element at position index i of seats MINUS element at position index i MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
