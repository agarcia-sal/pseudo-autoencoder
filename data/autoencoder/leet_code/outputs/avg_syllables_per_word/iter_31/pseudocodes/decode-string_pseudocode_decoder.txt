CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO an empty list  
        SET a TO an empty string  
        SET d TO an empty string  
        FOR each character IN s  
            IF character represents a digit  
                SET d TO d concatenated with character  
            ELSE IF character represents a letter of the alphabet  
                SET a TO a concatenated with character  
            ELSE IF character EQUALS the character representing an opening bracket  
                APPEND the tuple consisting of a and the integer value of d TO stack  
                SET a TO an empty string  
                SET d TO an empty string  
            ELSE IF character EQUALS the character representing a closing bracket  
                SET p and n TO the result of removing the last element from stack  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
