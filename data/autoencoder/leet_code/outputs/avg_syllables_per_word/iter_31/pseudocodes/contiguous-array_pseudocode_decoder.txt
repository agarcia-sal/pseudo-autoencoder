CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary mapping zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND each element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS A KEY IN prefix_sum_index  
                SET subarray_length TO i MINUS the value associated with count in prefix_sum_index  
                IF subarray_length GREATER THAN max_length  
                    SET max_length TO subarray_length  
                END IF  
            ELSE  
                ASSIGN the value i TO the key count IN prefix_sum_index  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
