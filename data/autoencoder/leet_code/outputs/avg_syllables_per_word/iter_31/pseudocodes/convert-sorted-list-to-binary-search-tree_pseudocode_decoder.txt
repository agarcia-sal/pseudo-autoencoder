CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH the element at position one of values  
    SET p TO head  
    FOR each val IN values FROM the second element TO the last element  
        SET node TO A NEW ListNode WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the val of p1 EQUALS the val of p2 AND is_same_list(the next element of p1, the next element of p2)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i PLUS one of values NOT EQUALS None THEN  
            SET the left element of node TO A NEW TreeNode WITH the element at position i PLUS one of values  
            APPEND the left element of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i PLUS one of values NOT EQUALS None THEN  
            SET the right element of node TO A NEW TreeNode WITH the element at position i PLUS one of values  
            APPEND the right element of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val of p NOT EQUALS the val of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left element of p, the left element of q) AND is_same_tree(the right element of p, the right element of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head EQUALS None THEN  
            RETURN None  
        END IF  
        
        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next element of fast NOT EQUALS right  
                SET slow TO the next element of slow  
                SET fast TO the next element of the next element of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right THEN  
                RETURN None  
            END IF  
            
            SET mid TO findMiddle(left, right)  
            
            SET node TO A NEW TreeNode WITH the val of mid  
            SET the left element of node TO convertListToBST(left, mid)  
            SET the right element of node TO convertListToBST(the next element of mid, right)  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST(head, None)  
    END FUNCTION  
END CLASS
