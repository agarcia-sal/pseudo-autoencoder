CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end TO index FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR the element at position end of s EQUALS character I  
                SET segment TO the elements of result from position start TO position end PLUS one  
                SET reversed_segment TO the elements of segment in reverse order  
                REPLACE in result the elements from position start TO position end PLUS one WITH reversed_segment  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
