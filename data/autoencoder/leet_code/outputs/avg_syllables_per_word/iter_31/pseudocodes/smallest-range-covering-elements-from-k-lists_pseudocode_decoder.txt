CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            PUSH element at position zero of element at position index of nums AND index AND zero INTO min_heap USING heap push  
            SET max_value TO maximum OF max_value AND element at position zero of element at position index of nums  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO list containing negative one hundred thousand AND one hundred thousand  
        WHILE min_heap is not empty  
            POP the smallest element FROM min_heap USING heap pop AND ASSIGN TO min_value AND list_index AND element_index  
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one LESS THAN LENGTH OF element at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of element at position list_index of nums  
                PUSH next_value AND list_index AND element_index PLUS one INTO min_heap USING heap push  
                SET max_value TO maximum OF max_value AND next_value  
            ELSE  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
