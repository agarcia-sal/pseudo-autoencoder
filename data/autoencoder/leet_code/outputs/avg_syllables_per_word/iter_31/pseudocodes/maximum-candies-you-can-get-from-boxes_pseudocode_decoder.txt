CLASS Solution  
    FUNCTION maxCandies(status AS List of Integer keys candies AS List of Integer keys keys AS List of List of Integer keys containedBoxes AS List of List of Integer keys initialBoxes AS List of Integer) RETURNS Integer  
        SET boxes_we_have TO a set containing each element in initialBoxes  
        SET keys_we_have TO an empty set  
        SET total_candies TO zero  
        SET progress_made TO true  
        WHILE progress_made holds  
            SET progress_made TO false  
            FOR each box IN all elements in boxes_we_have converted to a list  
                IF element at position box of status EQUALS one OR box IS an element IN keys_we_have  
                    INCREMENT total_candies BY element at position box of candies  
                    ADD each element in element at position box of keys TO keys_we_have  
                    ADD each element in element at position box of containedBoxes TO boxes_we_have  
                    REMOVE box FROM boxes_we_have  
                    SET progress_made TO true  
                END IF  
            END FOR  
        END WHILE  
        RETURN total_candies  
    END FUNCTION  
END CLASS
