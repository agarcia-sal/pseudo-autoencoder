CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        FOR i FROM zero TO n MINUS one  
            WHILE element at position i of nums IS GREATER THAN OR EQUAL TO one AND element at position i of nums IS LESS THAN OR EQUAL TO n AND element at position at the element at position i of nums MINUS one of nums IS NOT EQUALS TO element at position i of nums  
                SET correct_index TO element at position i of nums MINUS one  
                SET temporary_value TO element at position i of nums  
                SET element at position i of nums TO element at position correct_index of nums  
                SET element at position correct_index of nums TO temporary_value  
            END WHILE  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of nums IS NOT EQUALS TO i PLUS one  
                RETURN i PLUS one  
            END IF  
        END FOR  
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
