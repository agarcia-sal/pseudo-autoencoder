CLASS Solution  
    FUNCTION kthSmallestPath(destination row_and_column LIST OF INTEGER k INTEGER) RETURNS STRING  
        SET row TO element at position one of row_and_column  
        SET col TO element at position two of row_and_column  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero  
                SET paths_with_H TO NUMBER OF combinations OF row PLUS col MINUS one CHOOSE col MINUS one  
                IF k LESS THAN OR EQUAL TO paths_with_H  
                    APPEND character H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND character V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND character V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN concatenation of all elements of path AS STRING  
    END FUNCTION  
END CLASS
