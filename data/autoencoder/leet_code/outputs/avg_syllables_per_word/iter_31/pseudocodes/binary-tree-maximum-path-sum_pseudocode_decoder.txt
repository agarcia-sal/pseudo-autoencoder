CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with the value of element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.left TO new TreeNode with the value of element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.right TO new TreeNode with the value of element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF the value of p.val NOT EQUALS the value of q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS no value THEN  
                RETURN negative infinity value, negative infinity value  
            END IF  
            SET left_max, left_gain TO the result of calling helper with the left child of node  
            SET right_max, right_gain TO the result of calling helper with the right child of node  
            SET current_max TO the value of node.val PLUS the greater value between left_gain and zero PLUS the greater value between right_gain and zero  
            SET current_gain TO the value of node.val PLUS the greatest value among left_gain right_gain and zero  
            SET global_max TO the greatest value among left_max right_max and current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of the result of calling helper with root  
    END FUNCTION  
END CLASS
