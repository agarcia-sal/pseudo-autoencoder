CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n PLUS one  
        SET self.rank TO list of zeros with length n PLUS one  
        SET self.count TO n  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO result of calling find with element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO result of calling find with u  
        SET rootV TO result of calling find with v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        SET uf_common TO new instance of UnionFind constructed with n  
        SET uf_alice TO new instance of UnionFind constructed with n  
        SET uf_bob TO new instance of UnionFind constructed with n  
  
        SET used_edges TO zero  
  
        FOR each t u v IN edges  
            IF t EQUALS three  
                IF result of calling union on uf_common with u and v IS True  
                    CALL union on uf_alice with u and v  
                    CALL union on uf_bob with u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
  
        FOR each t u v IN edges  
            IF t EQUALS one AND result of calling union on uf_alice with u and v IS True  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND result of calling union on uf_bob with u and v IS True  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
  
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one  
            RETURN LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
