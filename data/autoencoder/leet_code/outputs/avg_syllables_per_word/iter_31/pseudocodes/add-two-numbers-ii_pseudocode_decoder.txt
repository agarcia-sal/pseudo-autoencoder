CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO A new ListNode WITH val EQUALS the element at position zero of values AND next EQUALS None  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the END  
        SET node TO A new ListNode WITH val EQUALS val AND next EQUALS None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the logical AND of p1.val EQUALS p2.val AND the result of is_same_list called with p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO an empty list  
            WHILE node IS NOT None  
                APPEND the element val of node TO stack  
                SET node TO the element next of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO the result of pushIntoStack called with l1  
        SET stack2 TO the result of pushIntoStack called with l2  
        
        SET carry TO zero  
        SET result TO None  
        
        WHILE the LENGTH OF stack1 GREATER THAN zero OR the LENGTH OF stack2 GREATER THAN zero OR carry GREATER THAN zero  
            IF the LENGTH OF stack1 GREATER THAN zero THEN  
                REMOVE and SET val1 TO the last element of stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF the LENGTH OF stack2 GREATER THAN zero THEN  
                REMOVE and SET val2 TO the last element of stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO A new ListNode WITH val EQUALS digit AND next EQUALS result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
