CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND the negative of num TO max_heap  
        END FOR  
        CALL heapify TO convert max_heap INTO a heap structure  
        SET total_sum TO zero  
        FOR each element IN target  
            INCREMENT total_sum BY element  
        END FOR  
        WHILE the element at position zero of max_heap LESS THAN the negative of one  
            SET largest TO the negative of the result of popping the smallest element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN false  
            END IF  
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one  
                RETURN false  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            PUSH the negative of new_value INTO max_heap  
        END WHILE  
        RETURN true  
    END FUNCTION  
END CLASS
