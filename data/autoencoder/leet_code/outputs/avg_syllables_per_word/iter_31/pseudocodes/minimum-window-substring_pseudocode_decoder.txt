CLASS Solution  
    FUNCTION minWindow(s parameter s AS string, t parameter t AS string) RETURNS string  
        FROM the collection class IMPORT Counter  
        
        SET t_count TO Counter APPLIED TO t  
        SET required TO the LENGTH OF t_count  
        
        SET variable left TO zero  
        SET variable right TO zero  
        SET variable formed TO zero  
        SET window_counts TO Counter APPLIED TO an empty collection  
        
        SET min_length TO infinity  
        SET min_start TO zero  
        
        WHILE right LESS THAN the LENGTH OF s  
            SET current_character TO the element at position right of s  
            INCREMENT the count of current_character in window_counts BY one  
            
            IF current_character IS IN t_count AND the count of current_character in window_counts EQUALS the count of current_character in t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET character_at_left TO the element at position left of s  
                
                IF the difference between right and left PLUS one LESS THAN min_length  
                    SET min_length TO the difference between right and left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count of character_at_left in window_counts BY one  
                
                IF character_at_left IS IN t_count AND the count of character_at_left in window_counts LESS THAN the count of character_at_left in t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start TO the position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
