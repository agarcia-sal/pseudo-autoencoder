CLASS Solution  
    FUNCTION updateMatrix(mat)  
        IF mat EQUALS empty OR element at position zero of mat EQUALS empty  
            RETURN mat  
        END IF  
        
        SET rows TO the LENGTH OF mat  
        SET cols TO the LENGTH OF element at position zero of mat  
        SET dist TO an empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row_list TO an empty list  
            FOR each index_c FROM zero TO cols MINUS one  
                APPEND positive infinity TO row_list  
            END FOR  
            APPEND row_list TO dist  
        END FOR  
        
        SET queue TO an empty double ended queue  
        
        FOR each r FROM zero TO rows MINUS one  
            FOR each c FROM zero TO cols MINUS one  
                IF element at position r of mat at position c EQUALS zero  
                    SET element at position r of dist at position c TO zero  
                    APPEND the pair of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        
        SET directions TO a list containing the pairs of negative one and zero PLUS one and zero PLUS zero and negative one PLUS zero and one  
        
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE the element from the left of queue and ASSIGN to r and c  
            FOR each dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr of dist at position nc GREATER THAN element at position r of dist at position c PLUS one  
                    SET element at position nr of dist at position nc TO element at position r of dist at position c PLUS one  
                    APPEND the pair of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN dist  
    END FUNCTION  
END CLASS
