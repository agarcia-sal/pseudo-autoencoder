CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO a sorted list of pairs where each pair consists of the start time of an interval and the index of that interval in intervals  
        SET result TO an empty list  
        FOR each pair of any element and end_time IN intervals  
            SET position TO the smallest index in start_times WHERE the start time at that index IS GREATER THAN OR EQUAL TO end_time using binary search  
            IF position IS LESS THAN the length of start_times  
                APPEND the index from the pair at position in start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
