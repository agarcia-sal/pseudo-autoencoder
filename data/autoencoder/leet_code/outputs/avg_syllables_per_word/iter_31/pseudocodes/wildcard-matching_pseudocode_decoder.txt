CLASS Solution  
    FUNCTION isMatch(s string p string) RETURNS boolean  
        SET dp TO a two dimensional list of boolean values where each element is false with the number of rows equal to the length of string s PLUS one and the number of columns equal to the length of string p PLUS one  
        SET the element at position zero zero of dp TO true  
        
        FOR each integer j FROM one TO the length of string p  
            IF the character at position j MINUS one of string p EQUALS the character that is an asterisk  
                SET the element at position zero j of dp TO the element at position zero j MINUS one of dp  
            END IF  
        END FOR  
        
        FOR each integer i FROM one TO the length of string s  
            FOR each integer j FROM one TO the length of string p  
                IF the character at position j MINUS one of string p EQUALS the character that is an asterisk  
                    SET the element at position i j of dp TO the element at position i MINUS one j of dp OR the element at position i j MINUS one of dp  
                ELSE IF the character at position j MINUS one of string p EQUALS the character that is a question mark OR the character at position i MINUS one of string s EQUALS the character at position j MINUS one of string p  
                    SET the element at position i j of dp TO the element at position i MINUS one j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the length of string s the length of string p of dp  
    END FUNCTION  
END CLASS
