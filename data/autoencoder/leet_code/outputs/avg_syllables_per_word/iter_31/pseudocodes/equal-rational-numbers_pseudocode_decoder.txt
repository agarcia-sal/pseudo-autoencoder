CLASS Solution  
    FUNCTION isRationalEqual(s AS string, t AS string) RETURNS boolean  
        SET ratios TO list containing one FOLLOWED BY one DIVIDED BY nine FOLLOWED BY one DIVIDED BY ninety-nine FOLLOWED BY one DIVIDED BY nine hundred ninety-nine FOLLOWED BY one DIVIDED BY nine thousand nine hundred ninety-nine  

        FUNCTION valueOf(s AS string) RETURNS float  
            IF the position of the first occurrence of opening parenthesis in s EQUALS negative one THEN  
                RETURN the conversion of s TO float  
            END IF  

            SET leftParenIndex TO the position of the first occurrence of opening parenthesis in s  
            SET rightParenIndex TO the position of the first occurrence of closing parenthesis in s  
            SET dotIndex TO the position of the first occurrence of decimal point in s  

            SET integerAndNonRepeating TO the conversion TO float OF the substring of s from the start TO the position before leftParenIndex  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  

            SET repeating TO the conversion TO integer OF the substring of s from one position after leftParenIndex TO the position before rightParenIndex  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY zero point one RAISED TO THE POWER OF nonRepeatingLength MULTIPLIED BY the element at position repeatingLength of ratios  
        END FUNCTION  

        RETURN the absolute value of valueOf of s MINUS valueOf of t LESS THAN one times ten RAISED TO THE POWER OF negative nine  
    END FUNCTION  
END CLASS
