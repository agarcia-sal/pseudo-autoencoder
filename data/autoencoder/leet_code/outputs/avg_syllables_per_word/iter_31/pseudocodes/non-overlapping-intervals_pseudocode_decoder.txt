CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty list THEN  
            RETURN zero  
        END IF  
        SORT intervals BY the value at position two of each element IN ascending order  
        SET end TO the value at position two of the element at position one of intervals  
        SET count TO zero  
        FOR variable i FROM two TO the LENGTH OF intervals  
            IF the value at position one of the element at position i of intervals GREATER THAN OR EQUAL TO end THEN  
                SET end TO the value at position two of the element at position i of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
