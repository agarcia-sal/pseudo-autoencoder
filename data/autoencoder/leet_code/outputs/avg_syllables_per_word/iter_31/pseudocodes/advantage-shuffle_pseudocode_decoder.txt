CLASS Solution  
    FUNCTION advantageCount(nums1 list of integers nums2 list of integers) RETURNS list of integers  
        CALL sort nums1 in ascending order  
        SET nums2_with_indices TO list of pairs each containing element at position index of nums2 AND index for each index in the range from zero TO the length of nums2 MINUS one  
        CALL sort nums2_with_indices in descending order based on element at position zero of each pair  
        SET result TO list of zeros of length equal to the length of nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair in nums2_with_indices LET value BE element at position zero of pair AND index BE element at position one of pair  
            IF element at position right of nums1 GREATER THAN value  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
