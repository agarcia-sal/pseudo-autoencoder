CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        SET graph TO empty mapping with default empty list  
        FOR each city1, city2, toll IN highways  
            APPEND tuple of city2 and toll TO element at key city1 of graph  
            APPEND tuple of city1 and toll TO element at key city2 of graph  
        END FOR  
        
        SET max_cost TO negative one  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k  
                SET max_cost TO the greater value BETWEEN max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each neighbor, toll IN element at key city of graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    CALL dfs WITH arguments neighbor, visited, cost PLUS toll, highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        FOR start_city FROM zero TO n MINUS one  
            SET visited TO set containing start_city  
            CALL dfs WITH arguments start_city, visited, zero, zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
