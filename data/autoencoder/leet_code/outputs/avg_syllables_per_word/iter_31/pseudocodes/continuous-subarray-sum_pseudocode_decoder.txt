CLASS Solution  
    FUNCTION checkSubarraySum(nums list OF integer parameter, k integer parameter) RETURNS boolean  
        SET remainder_map TO a mapping FROM integer zero TO integer minus one  
        SET current_sum TO integer zero  
        FOR each index i AND element num IN enumeration OF nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS integer zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum EXISTS AS KEY IN remainder_map  
                IF i MINUS element at key current_sum OF remainder_map GREATER THAN integer one  
                    RETURN boolean True  
                END IF  
            ELSE  
                SET element at key current_sum OF remainder_map TO i  
            END ELSE  
        END FOR  
        RETURN boolean False  
    END FUNCTION  
END CLASS
