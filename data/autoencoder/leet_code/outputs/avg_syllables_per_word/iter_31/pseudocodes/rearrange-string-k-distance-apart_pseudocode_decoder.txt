CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO a mapping with counts of each character in s  
        
        SET max_heap TO an empty list  
        FOR each character char AND its count count IN freq  
            APPEND a tuple of negative count AND char TO max_heap  
        END FOR  
        
        APPLY heapify operation TO max_heap  
        
        SET wait_queue TO an empty list  
        SET result TO an empty list  
        
        WHILE the LENGTH OF max_heap GREATER THAN zero  
            EXTRACT the element with the largest frequency as count AND char FROM max_heap  
            APPEND char TO result  
            
            APPEND a tuple of count PLUS one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first element of wait_queue assigning its count AND char  
                IF count LESS THAN zero  
                    PUSH the tuple count AND char INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result NOT EQUALS the LENGTH OF s  
            RETURN the empty string  
        END IF  
        
        RETURN the concatenation of all characters in result as a single string  
    END FUNCTION  
END CLASS
