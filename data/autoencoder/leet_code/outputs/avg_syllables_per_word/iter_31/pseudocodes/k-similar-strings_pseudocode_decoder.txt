CLASS Solution  
    FUNCTION kSimilarity(s1 parameter s1 OF TYPE string s1 parameter s2 OF TYPE string s2) RETURNS integer  
        FUNCTION neighbors(s parameter s OF TYPE string s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s NOT EQUALS the element at position j of s2  
                    YIELD concatenate the substring from position zero TO position i MINUS one of s WITH the element at position j of s WITH the substring from position i PLUS one TO position j MINUS one of s WITH the element at position i of s WITH the substring from position j PLUS one TO the last position of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN reference TO a new double ended queue containing a pair of s1 and zero TO queue  
        ASSIGN reference TO a new set containing s1 TO visited  
        
        WHILE the LENGTH OF queue GREATER THAN zero  
            REMOVE and ASSIGN the first element of queue TO s and swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the result of neighbors called with s  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the pair of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
