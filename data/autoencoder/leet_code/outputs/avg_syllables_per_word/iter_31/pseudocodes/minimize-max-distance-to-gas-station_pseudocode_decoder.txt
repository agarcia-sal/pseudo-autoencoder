CLASS Solution  
    FUNCTION minmaxGasDist(stations AS List OF integers, k AS integer) RETURNS float  
        FUNCTION possible(max_dist AS float) RETURNS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index of stations MINUS element at position index MINUS one of stations  
                SET division_result TO distance_difference DIVIDED BY max_dist  
                INCREMENT additional_stations BY the INTEGER PART OF division_result  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) IS true  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
