CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO empty list  
        FOR each number FROM one TO n  
            APPEND the string representation of number TO teams  
        END FOR  
        WHILE n GREATER THAN one  
            FOR each index FROM zero TO the INTEGER DIVISION OF n BY two MINUS one  
                SET the element at position index of teams TO an opening parenthesis FOLLOWED BY the element at position index of teams FOLLOWED BY a comma FOLLOWED BY the element at position n MINUS one MINUS index of teams FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO the INTEGER DIVISION OF n BY two  
        END WHILE  
        RETURN the element at position zero of teams  
    END FUNCTION  
END CLASS
