CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty collection  
                RETURN zero  
            END IF  
            
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position one of houses  
            END IF  
            
            SET dp TO a list of zero values with the LENGTH OF houses  
            SET element at position one of dp TO element at position one of houses  
            SET element at position two of dp TO the GREATER value BETWEEN element at position one of houses AND element at position two of houses  
            
            FOR index FROM three TO the LENGTH OF houses  
                SET element at position index of dp TO the GREATER value BETWEEN element at position index MINUS one of dp AND the SUM OF element at position index MINUS two of dp PLUS element at position index of houses  
            END FOR  
            
            RETURN element at the LAST position of dp  
        END FUNCTION  
        
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        
        RETURN the GREATER value BETWEEN CALL rob_linear WITH the SUBCOLLECTION of nums from position one TO position the LENGTH OF nums MINUS one AND CALL rob_linear WITH the SUBCOLLECTION of nums from position two TO the LAST position  
    END FUNCTION  
END CLASS
