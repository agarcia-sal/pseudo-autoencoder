CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the difference BETWEEN the element at position zero of nums AND twenty four IS LESS THAN one times ten to the power of negative six  
            END IF  
            
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation op IN the list consisting of true division FUNCTION, multiplication FUNCTION, addition FUNCTION, subtraction FUNCTION  
                            IF op IS addition FUNCTION OR op IS multiplication FUNCTION  
                                IF j GREATER THAN i  
                                    CONTINUE to the next iteration of the loop over operations  
                                END IF  
                            END IF  
                            
                            IF op IS NOT true division FUNCTION OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying operation op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
