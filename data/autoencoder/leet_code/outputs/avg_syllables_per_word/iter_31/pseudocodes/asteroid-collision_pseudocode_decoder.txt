CLASS Solution  
    FUNCTION asteroidCollision WITH PARAMETER asteroids  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid IS LESS THAN zero AND element at the last position of stack IS GREATER THAN zero  
                IF element at the last position of stack IS LESS THAN the NEGATION of asteroid  
                    REMOVE element at the last position FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF element at the last position of stack EQUALS the NEGATION of asteroid  
                    REMOVE element at the last position FROM stack  
                END IF  
                BREAK from the WHILE loop  
            END WHILE  
            IF the WHILE loop did not BREAK  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
