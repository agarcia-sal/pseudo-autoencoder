CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the result of calling subsets function with the sublist of nums from position two TO the end  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            APPEND a new list consisting of the element at position one of nums FOLLOWED BY the elements of subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
