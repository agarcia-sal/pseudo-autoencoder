CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s AND left AND right  
            IF the LENGTH OF s EQUALS n MULTIPLIED BY two  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s FOLLOWED BY open parenthesis AND left PLUS one AND right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s FOLLOWED BY close parenthesis AND left AND right PLUS one  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
