CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET replaced_expression TO expression WITH every occurrence of the character minus REPLACED BY the character plus FOLLOWED BY the character minus  
        SET fractions_list TO the list OBTAINED BY SPLITTING replaced_expression BY the character plus  
        SET filtered_fractions TO the list of elements IN fractions_list WHERE the element IS NOT empty  
        SET result TO the sum OF fractions represented BY each element IN filtered_fractions  
        SET numerator_value TO the numerator PART OF result  
        SET denominator_value TO the denominator PART OF result  
        SET final_string TO the concatenation OF numerator_value, the character slash, and denominator_value  
        RETURN final_string  
    END FUNCTION  
END CLASS
