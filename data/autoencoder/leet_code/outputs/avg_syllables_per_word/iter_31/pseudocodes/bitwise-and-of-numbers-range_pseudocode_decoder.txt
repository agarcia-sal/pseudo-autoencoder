CLASS Solution  
    FUNCTION rangeBitwiseAnd(left PARAMETER int, right PARAMETER int) RETURNS int  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO the INTEGER VALUE OF left DIVIDED BY two  
            SET right TO the INTEGER VALUE OF right DIVIDED BY two  
            INCREMENT shift BY one  
        END WHILE  
        RETURN the INTEGER VALUE OF left MULTIPLIED BY the INTEGER VALUE OF two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
