CLASS Solution  
    FUNCTION minBuildTime(blocks LIST OF INTEGER split INTEGER) RETURNS INTEGER  
        CALL transform blocks INTO a min-heap structure  
        WHILE the LENGTH OF blocks GREATER THAN one  
            SET first TO the smallest element REMOVED FROM blocks  
            SET second TO the next smallest element REMOVED FROM blocks  
            SET combined_time TO the value of split PLUS the value of second  
            INSERT combined_time INTO blocks maintaining the min-heap structure  
        END WHILE  
        RETURN the element at position zero of blocks  
    END FUNCTION  
END CLASS
