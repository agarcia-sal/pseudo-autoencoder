CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of size n WITH all elements SET TO zero  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of dominoes EQUALS the character R  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS the character L  
                SET force TO zero  
            ELSE  
                IF force MINUS one IS GREATER THAN force zero  
                    SET force TO force MINUS one  
                ELSE  
                    SET force TO zero  
                END IF  
            END IF  
            INCREMENT the element at position i of forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one TO zero STEP minus one  
            IF the element at position i of dominoes EQUALS the character L  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS the character R  
                SET force TO zero  
            ELSE  
                IF force MINUS one IS GREATER THAN force zero  
                    SET force TO force MINUS one  
                ELSE  
                    SET force TO zero  
                END IF  
            END IF  
            DECREMENT the element at position i of forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each f IN forces  
            IF f IS GREATER THAN zero  
                APPEND the character R TO result  
            ELSE IF f IS LESS THAN zero  
                APPEND the character L TO result  
            ELSE  
                APPEND the character dot TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements of result AS a string  
    END FUNCTION  
END CLASS
