CLASS Solution  
    FUNCTION validTree(n number, edges list of lists of integers)  
        IF the LENGTH OF edges NOT EQUALS the VALUE OF n MINUS one THEN  
            RETURN false  
        END IF  
        SET adj_list TO a new empty mapping from integers TO lists  
        FOR index FROM zero TO n MINUS one  
            SET element at position index of adj_list TO an empty list  
        END FOR  
        FOR each pair element a and b IN edges  
            APPEND b TO the list element at key a of adj_list  
            APPEND a TO the list element at key b of adj_list  
        END FOR  
        SET visited TO a new empty set  
        FUNCTION dfs(node number)  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list element at key node of adj_list  
                CALL dfs WITH neighbor AS argument  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero AS argument  
        RETURN the LENGTH OF visited EQUALS the VALUE OF n  
    END FUNCTION  
END CLASS
