CLASS Solution  
    FUNCTION pourWater(heights PARAMETER, volume PARAMETER, k PARAMETER) RETURNS List of integers  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN zero AND element at position i MINUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights  
                DECREMENT i BY one  
            END WHILE  
            WHILE i LESS THAN k AND element at position i OF heights EQUALS element at position i PLUS one OF heights  
                INCREMENT i BY one  
            END WHILE  
            IF i LESS THAN k  
                SET element at position i OF heights TO element at position i OF heights PLUS one  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND element at position i PLUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights  
                INCREMENT i BY one  
            END WHILE  
            WHILE i GREATER THAN k AND element at position i OF heights EQUALS element at position i MINUS one OF heights  
                DECREMENT i BY one  
            END WHILE  
            IF i GREATER THAN k  
                SET element at position i OF heights TO element at position i OF heights PLUS one  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            
            SET element at position k OF heights TO element at position k OF heights PLUS one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
