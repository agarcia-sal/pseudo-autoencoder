CLASS Solution  
    FUNCTION shoppingOffers(price, special, needs)  
        SET special TO empty list  
        FOR each offer IN special  
            IF sum of elements from position one TO the position before last of offer GREATER THAN zero AND element at last position of offer LESS THAN sum of each element p MULTIPLIED BY each element n for elements p and n in price and elements from position one TO the position before last of offer respectively  
                APPEND offer TO special  
            END IF  
        END FOR  
        
        FUNCTION dfs(needs)  
            SET min_cost TO sum of each element p MULTIPLIED BY each element n for elements p and n in price and needs respectively  
            FOR each offer IN special  
                SET new_needs TO empty list  
                FOR each element n and element o IN needs and elements from position one TO the position before last of offer respectively  
                    APPEND n MINUS o TO new_needs  
                END FOR  
                IF for all elements n IN new_needs the condition n GREATER THAN OR EQUAL TO zero holds  
                    SET min_cost TO the smaller value BETWEEN min_cost AND element at last position of offer PLUS result of dfs called with the tuple form of new_needs  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        
        RETURN result of dfs called with the tuple form of needs  
    END FUNCTION  
END CLASS
