CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1, p2)  
            IF element at position zero of p1 EQUALS element at position zero of p2  
                RETURN infinity  
            END IF  
            RETURN the difference of element at position one of p1 MINUS element at position one of p2 DIVIDED BY the difference of element at position zero of p1 MINUS element at position zero of p2  
        END FUNCTION  
      
        SET max_points TO zero  
      
        FOR i FROM zero TO the LENGTH OF points MINUS one  
            ASSIGN reference TO a new empty mapping from default integer FOR variable slopes  
            FOR j FROM zero TO the LENGTH OF points MINUS one  
                IF i NOT EQUALS j  
                    SET s TO slope the element at position i of points AND the element at position j of points  
                    INCREMENT the value corresponding to key s in slopes BY one  
                    SET max_points TO the GREATER OF max_points AND the value corresponding to key s in slopes  
                END IF  
            END FOR  
        END FOR  
      
        IF the LENGTH OF points GREATER THAN zero  
            RETURN max_points PLUS one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
