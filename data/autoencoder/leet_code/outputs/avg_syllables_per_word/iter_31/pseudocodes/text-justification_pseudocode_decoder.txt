CLASS Solution  
    FUNCTION fullJustify(words maxWidth)  
        FUNCTION justify_line(line num_of_letters is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line EQUALS true  
                RETURN concatenation of all elements in line separated by single space FOLLOWED BY the repetition OF single space MULTIPLIED BY the difference BETWEEN maxWidth AND num_of_letters AND the LENGTH OF line MINUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND repetition OF single space MULTIPLIED BY spaces_between_words PLUS one TO the element at position index OF line  
                    ELSE  
                        APPEND repetition OF single space MULTIPLIED BY spaces_between_words TO the element at position index OF line  
                    END IF  
                END FOR  
                
                RETURN concatenation of all elements in line WITHOUT any separator  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result OF justify_line WITH arguments line num_of_letters false TO result  
                SET line TO empty list  
                SET num_of_letters TO zero  
            END IF  
            
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result OF justify_line WITH arguments line num_of_letters true TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
