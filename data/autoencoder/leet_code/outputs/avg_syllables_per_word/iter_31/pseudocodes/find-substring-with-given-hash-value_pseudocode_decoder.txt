CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the character code of c MINUS the character code of lowercase a PLUS one  
        END FUNCTION  
      
        SET n TO the LENGTH OF s  
        SET current_hash TO zero  
        SET p_pow TO one  
      
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO the remainder after dividing the sum of current_hash PLUS the value of element at position i of s MULTIPLIED BY p_pow BY modulo  
            IF i LESS THAN n MINUS one THEN  
                SET p_pow TO the remainder after dividing the product of p_pow MULTIPLIED BY power BY modulo  
            END IF  
        END FOR  
      
        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue THEN  
            RETURN the substring from position start_index TO position n MINUS one of s  
        END IF  
      
        SET p_pow_k_minus_1 TO p_pow  
        FOR i FROM n MINUS k MINUS one DOWN TO zero  
            SET current_hash TO the remainder after dividing the difference of current_hash MINUS the value of element at position i PLUS k of s MULTIPLIED BY p_pow_k_minus_1 BY modulo  
            SET current_hash TO the remainder after dividing the sum of current_hash MULTIPLIED BY power PLUS the value of element at position i of s BY modulo  
            IF current_hash EQUALS hashValue THEN  
                SET start_index TO i  
            END IF  
        END FOR  
      
        RETURN the substring from position start_index TO position start_index PLUS k MINUS one of s  
    END FUNCTION  
END CLASS
