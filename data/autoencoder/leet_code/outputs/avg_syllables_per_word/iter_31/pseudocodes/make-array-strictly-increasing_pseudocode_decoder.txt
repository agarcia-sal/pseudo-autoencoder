CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2 IN ASCENDING ORDER  
        SET dp TO a dictionary with key negative one and value zero  
        
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num AND ops IN dp  
                IF number GREATER THAN last_num  
                    SET new_dp at key number TO the minimum of value at key number in new_dp or positive infinity AND ops  
                END IF  
                
                SET idx TO the position in arr2 where last_num would be inserted to maintain ascending order read right  
                IF idx LESS THAN the LENGTH OF arr2  
                    SET new_dp at key element at position idx of arr2 TO the minimum of value at key element at position idx of arr2 in new_dp or positive infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp is empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
