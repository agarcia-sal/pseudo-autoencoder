CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new mapping from sequence string to integer count  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO position index PLUS ten MINUS one of s  
            SET the value at key sequence in sequence_count TO the current value at key sequence in sequence_count PLUS one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key_value_pair IN the items of sequence_count  
            IF the value part of key_value_pair IS GREATER THAN one  
                APPEND the key part of key_value_pair TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
