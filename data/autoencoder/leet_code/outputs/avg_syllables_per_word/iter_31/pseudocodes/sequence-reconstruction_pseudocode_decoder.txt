CLASS Solution  
    FUNCTION sequenceReconstruction(nums, sequences)  
        SET n TO the LENGTH OF nums  
        SET graph TO a new mapping from elements to empty lists  
        SET indegree TO a list of zeros with LENGTH n PLUS one  
        
        FOR each seq IN sequences  
            FOR i FROM zero TO the LENGTH OF seq MINUS one MINUS one  
                SET u TO element at position i of seq  
                SET v TO element at position i PLUS one of seq  
                APPEND v TO the list mapped from u in graph  
                INCREMENT the element at position v of indegree BY one  
            END FOR  
        END FOR  
        
        SET queue TO an empty double ended queue  
        FOR i FROM one TO n  
            IF the element at position i of indegree EQUALS zero  
                APPEND i TO queue  
            END IF  
        END FOR  
        
        IF the LENGTH OF queue NOT EQUALS one  
            RETURN False  
        END IF  
        
        SET index TO zero  
        WHILE queue is not empty  
            IF the LENGTH OF queue NOT EQUALS one  
                RETURN False  
            END IF  
            
            REMOVE the first element from queue and SET it TO node  
            IF node NOT EQUALS element at position index of nums  
                RETURN False  
            END IF  
            INCREMENT index BY one  
            
            FOR each neighbor IN the list mapped from node in graph  
                DECREMENT the element at position neighbor of indegree BY one  
                IF the element at position neighbor of indegree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN index EQUALS n  
    END FUNCTION  
END CLASS
