CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        INITIALIZE visited AS an empty set  
        INITIALIZE min_heap AS a list containing a tuple of the element at position zero zero of grid THEN zero THEN zero  
        ADD the tuple of zero THEN zero TO visited  
        
        SET directions TO a list containing the tuples of zero one THEN one zero THEN zero MINUS one THEN MINUS one zero  
        
        WHILE min_heap is not empty  
            REMOVE the smallest element FROM min_heap AND ASSIGN the first element of the tuple TO t THEN the second element TO i THEN the third element TO j  
            
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each tuple di THEN dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple of ni THEN nj NOT IN visited  
                    ADD the tuple of ni THEN nj TO visited  
                    INSERT INTO min_heap the tuple of the greater value BETWEEN t AND the element at position ni THEN nj of grid THEN ni THEN nj  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the integer MINUS one  
    END FUNCTION  
END CLASS
