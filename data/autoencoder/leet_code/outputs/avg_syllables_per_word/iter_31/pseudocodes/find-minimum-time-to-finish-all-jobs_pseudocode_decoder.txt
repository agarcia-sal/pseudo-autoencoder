CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        
        SET subset_sums TO a list of zeros with the LENGTH OF two RAISED TO THE POWER OF n  
        
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET element at position mask OR two RAISED TO THE POWER OF i of subset_sums TO element at position mask of subset_sums PLUS element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN element at position mask of subset_sums  
            END IF  
            SET res TO infinity  
            
            SET submask TO mask  
            WHILE submask GREATER THAN zero  
                SET first_value TO dp(k MINUS one, mask XOR submask)  
                SET second_value TO element at position submask of subset_sums  
                IF res GREATER THAN first_value GREATER THAN second_value OR res GREATER THAN second_value GREATER THAN first_value THEN  
                    SET res TO the value AMONG first_value AND second_value THAT IS GREATER AND LESS THAN res  
                ELSE IF res GREATER THAN first_value AND res GREATER THAN second_value THEN  
                    IF first_value GREATER THAN second_value THEN  
                        IF res GREATER THAN first_value THEN  
                            SET res TO first_value  
                        END IF  
                    ELSE  
                        IF res GREATER THAN second_value THEN  
                            SET res TO second_value  
                        END IF  
                    END IF  
                END IF  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, two RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
