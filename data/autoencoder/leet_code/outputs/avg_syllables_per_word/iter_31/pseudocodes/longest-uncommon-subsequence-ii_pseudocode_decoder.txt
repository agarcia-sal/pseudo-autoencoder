CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator TO AN ITERATOR OVER t  
            RETURN the evaluation that FOR EACH character IN s THE character IS FOUND IN iterator IN SEQUENCE  
        END FUNCTION  
        
        SORT strs BY descending LENGTH OF element AND THEN BY element IN LEXICOGRAPHICAL ORDER  
        
        FOR index i FROM the FIRST POSITION TO the LAST POSITION OF strs  
            SET unique TO TRUE  
            SET word TO element at position i of strs  
            FOR index j FROM the FIRST POSITION TO the LAST POSITION OF strs  
                SET other TO element at position j of strs  
                IF i NOT EQUALS j AND is_subsequence WITH PARAMETERS word AND other RETURNS TRUE  
                    SET unique TO FALSE  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique IS TRUE  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        RETURN the NEGATIVE ONE  
    END FUNCTION  
END CLASS
