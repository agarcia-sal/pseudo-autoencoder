CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid EQUALS no value OR element at position zero of obstacleGrid EQUALS no value OR element at position zero of element at position zero of obstacleGrid EQUALS one THEN  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero of obstacleGrid  
        SET dp TO a new list consisting of m elements each being a list consisting of n elements each set TO zero  
        SET element at position zero of element at position zero of dp TO one  
        FOR j FROM one TO n MINUS one  
            IF element at position j of element at position zero of obstacleGrid EQUALS zero THEN  
                SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            IF element at position i of element at position zero of obstacleGrid EQUALS zero THEN  
                SET element at position i of element at position zero of dp TO element at position i MINUS one of element at position zero of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i of element at position j of obstacleGrid EQUALS zero THEN  
                    SET element at position i of element at position j of dp TO element at position i MINUS one of element at position j of dp PLUS element at position i of element at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of element at position n MINUS one of dp  
    END FUNCTION  
END CLASS
