CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO empty dictionary  
        SET i TO zero  
        FOR each character IN order  
            SET element at key character of dictionary d TO i  
            INCREMENT i BY one  
        END FOR  
        SET sorted_list TO the list of characters of string s sorted by the key function that returns the value at key character of dictionary d if it exists otherwise zero  
        SET result_string TO the concatenation of all characters in sorted_list in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
