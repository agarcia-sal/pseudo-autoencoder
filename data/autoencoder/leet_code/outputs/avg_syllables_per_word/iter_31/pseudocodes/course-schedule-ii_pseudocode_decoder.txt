CLASS Solution  
    FUNCTION findOrder(numCourses numCourses comma prerequisites prerequisites)  
        SET adj_list TO an empty dictionary mapping each course from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros with the length of numCourses  
        FOR each pair of destination dest and source src IN prerequisites  
            APPEND destination TO the list at key source in adj_list  
            INCREMENT the element at position destination of in_degrees BY one  
        END FOR  
        SET queue TO a new empty double ended queue containing all courses from zero TO numCourses MINUS one WHERE the element at position course of in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        WHILE queue is not empty  
            REMOVE the element from the left side of queue and ASSIGN it TO current  
            APPEND current TO topological_order  
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor of in_degrees BY one  
                IF the element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        IF the length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
