CLASS Solution  
    FUNCTION waysToPartition(nums k)  
        SET total_sum TO the sum of all elements in nums  
        SET n TO the LENGTH OF nums  
        ASSIGN left_diff TO a new empty mapping from keys to integer defaulted to zero  
        ASSIGN right_diff TO a new empty mapping from keys to integer defaulted to zero  
        SET current_sum TO zero  
        FOR i FROM zero TO n MINUS one MINUS one  
            INCREMENT current_sum BY element at position i of nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT the value associated with key diff in right_diff BY one  
        END FOR  
        SET max_ways TO the value associated with key zero in right_diff  
        SET current_sum TO zero  
        FOR i FROM zero TO n MINUS one  
            INCREMENT current_sum BY element at position i of nums  
            SET diff_change TO k MINUS element at position i of nums  
            SET new_ways TO the value associated with key negative diff_change in left_diff PLUS the value associated with key diff_change in right_diff  
            IF new_ways GREATER THAN max_ways  
                SET max_ways TO new_ways  
            END IF  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT the value associated with key diff in right_diff BY one  
            INCREMENT the value associated with key diff in left_diff BY one  
        END FOR  
        RETURN max_ways  
    END FUNCTION  
END CLASS
