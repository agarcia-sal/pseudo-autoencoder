CLASS Solution  
    FUNCTION getCollisionTimes WITH PARAMETER cars  
        SET n TO the LENGTH OF cars  
        SET result TO a LIST OF size n WITH each element SET TO negative one point zero  
        SET stack TO an EMPTY LIST  
        
        FOR i FROM n MINUS one DOWN TO zero  
            SET pos TO the element at position zero of element at position i of cars  
            SET speed TO the element at position one of element at position i of cars  
            
            WHILE stack IS NOT EMPTY  
                SET next_index TO the last element of stack  
                SET next_pos TO the element at position zero of element at position next_index of cars  
                SET next_speed TO the element at position one of element at position next_index of cars  
                
                IF speed LESS THAN OR EQUAL TO next_speed  
                    REMOVE the last element FROM stack  
                    CONTINUE TO the next iteration of the WHILE loop  
                END IF  
                
                SET collision_time TO next_pos MINUS pos DIVIDED BY speed MINUS next_speed  
                
                IF element at position next_index of result GREATER THAN zero AND collision_time GREATER THAN OR EQUAL TO element at position next_index of result  
                    REMOVE the last element FROM stack  
                    CONTINUE TO the next iteration of the WHILE loop  
                END IF  
                
                SET element at position i of result TO collision_time  
                BREAK FROM the WHILE loop  
            END WHILE  
            
            APPEND i TO stack  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
