CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN reference TO value IN tree AT the key composed of depth and pos  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value in tree AT the key composed of depth and pos  
            IF current_value EQUALS None  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child_depth TO depth PLUS one  
            SET left_child_pos TO pos MULTIPLIED BY two MINUS one  
            SET right_child_depth TO depth PLUS one  
            SET right_child_pos TO pos MULTIPLIED BY two  
            IF the key composed of left_child_depth and left_child_pos NOT IN tree AND the key composed of right_child_depth and right_child_pos NOT IN tree  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs(left_child_depth, left_child_pos, new_path_sum)  
            SET right_sum TO dfs(right_child_depth, right_child_pos, new_path_sum)  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs(one, one, zero)  
    END FUNCTION  
END CLASS
