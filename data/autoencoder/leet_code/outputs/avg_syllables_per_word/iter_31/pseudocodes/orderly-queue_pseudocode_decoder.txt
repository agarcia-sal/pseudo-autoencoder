CLASS Solution  
    FUNCTION orderlyQueue(s string, k integer) RETURNS string  
        IF k GREATER THAN one THEN  
            RETURN the concatenation of all characters IN s sorted in ascending order  
        ELSE  
            SET n TO the length of s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the concatenation of the substring of s from position i TO the end PLUS the substring of s from position zero TO position i MINUS one  
                IF rotated LESS THAN smallest THEN  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
