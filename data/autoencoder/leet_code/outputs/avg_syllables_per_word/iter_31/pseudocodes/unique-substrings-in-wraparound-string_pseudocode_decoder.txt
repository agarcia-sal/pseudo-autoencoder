CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO dictionary with keys as each character from a to z and values set to zero  
        SET current_length TO zero  
        
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND ( the numeric code of the element at position index of s MINUS the numeric code of the element at position index MINUS one of s EQUALS one OR the element at position index of s EQUALS character a AND the element at position index MINUS one of s EQUALS character z )  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            
            IF the value in max_length for the key which is the element at position index of s LESS THAN current_length  
                SET the value in max_length for the key which is the element at position index of s TO current_length  
            END IF  
        END FOR  
        
        SET total_unique_substrings TO zero  
        FOR each value IN the values of max_length  
            INCREMENT total_unique_substrings BY value  
        END FOR  
        
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
