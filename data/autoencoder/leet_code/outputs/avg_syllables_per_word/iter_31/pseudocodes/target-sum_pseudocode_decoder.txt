CLASS Solution  
    FUNCTION findTargetSumWays(nums parameter, target parameter)  
        FUNCTION dfs(index parameter, current_sum parameter, memo parameter)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index parameter AND current_sum parameter EXISTS AS a KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE of index parameter AND current_sum parameter IN memo  
            END IF  
            SET positive TO the RESULT OF CALLING dfs WITH arguments index parameter PLUS one, current_sum parameter PLUS the element at position index parameter OF nums, AND memo parameter  
            SET negative TO the RESULT OF CALLING dfs WITH arguments index parameter PLUS one, current_sum parameter MINUS the element at position index parameter OF nums, AND memo parameter  
            SET the VALUE ASSOCIATED WITH the TUPLE of index parameter AND current_sum parameter IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the TUPLE of index parameter AND current_sum parameter IN memo  
        END FUNCTION  
        RETURN the RESULT OF CALLING dfs WITH arguments zero, zero, AND an empty COLLECTION  
    END FUNCTION  
END CLASS
