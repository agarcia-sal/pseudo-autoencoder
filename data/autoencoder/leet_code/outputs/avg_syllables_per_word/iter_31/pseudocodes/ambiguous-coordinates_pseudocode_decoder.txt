CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the substring from position one TO position d of frag  
                SET right TO the substring from position d PLUS one TO the LENGTH OF frag of frag  
                IF the first character of left IS NOT equal TO zero OR left EQUALS zero  
                AND the last character of right IS NOT equal TO zero  
                    YIELD concatenation of left PLUS a decimal point IF d IS NOT equal TO n OTHERWISE an empty string PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the substring from position two TO the position BEFORE the last position of s  
        RETURN a list constructed by  
            FOR i FROM one TO the LENGTH OF s MINUS one  
                FOR each a IN the iterable result of make applied TO the substring from position one TO position i of s  
                    FOR each b IN the iterable result of make applied TO the substring from position i PLUS one TO the LENGTH OF s of s  
                        FORM a string consisting of an opening parenthesis PLUS a PLUS a comma PLUS a space PLUS b PLUS a closing parenthesis  
                    END FOR  
                END FOR  
            END FOR  
    END FUNCTION  
END CLASS
