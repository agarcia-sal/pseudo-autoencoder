CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values AND left equal to nothing AND right equal to nothing  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue holds elements  
        REMOVE and ASSIGN node TO first element of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing THEN  
            SET node.left TO new TreeNode with value equal to element at position i of values AND left equal to nothing AND right equal to nothing  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing THEN  
            SET node.right TO new TreeNode with value equal to element at position i of values AND left equal to nothing AND right equal to nothing  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION countUnivalSubtrees(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS nothing THEN  
                RETURN True AND zero  
            END IF  
            SET is_left_unival, left_count TO helper(node.left)  
            SET is_right_unival, right_count TO helper(node.right)  
            IF is_left_unival AND is_right_unival THEN  
                IF (node.left EQUALS nothing OR node.left.val EQUALS node.val) AND (node.right EQUALS nothing OR node.right.val EQUALS node.val) THEN  
                    RETURN True AND left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False AND left_count PLUS right_count  
        END FUNCTION  
        SET _, count TO helper(root)  
        RETURN count  
    END FUNCTION  
END CLASS
