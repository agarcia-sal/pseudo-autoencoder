CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS nothing  
            SET node.left TO a new TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS nothing  
            SET node.right TO a new TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(self, root)  
        FUNCTION rob_from_node(node)  
            IF node EQUALS nothing  
                RETURN zero AND zero  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node with argument node.left  
            SET right_robbed, right_not_robbed TO rob_from_node with argument node.right  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the maximum of left_robbed and left_not_robbed PLUS the maximum of right_robbed and right_not_robbed  
            RETURN rob_this_node AND not_rob_this_node  
        END FUNCTION  
        RETURN the maximum of rob_from_node with argument root  
    END FUNCTION  
END CLASS
