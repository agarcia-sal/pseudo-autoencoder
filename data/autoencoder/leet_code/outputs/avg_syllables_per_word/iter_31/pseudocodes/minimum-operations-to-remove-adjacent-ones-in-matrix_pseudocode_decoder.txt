CLASS Solution  
    FUNCTION minimumOperations(grid)  
        FUNCTION bfs()  
            SET queue TO an empty double ended queue  
            FOR u FROM zero TO n MINUS one  
                IF element at position u of match EQUALS negative one  
                    SET element at position u of dist TO zero  
                    APPEND u TO queue  
                ELSE  
                    SET element at position u of dist TO infinity  
                END IF  
            END FOR  
            SET element at position negative one of dist TO infinity  
            WHILE queue IS NOT empty  
                REMOVE and RETURN element from the left end of queue INTO u  
                IF element at position u of dist IS LESS THAN element at position negative one of dist  
                    FOR each v IN element at position u of graph  
                        IF element at position element at position v of match of dist EQUALS infinity  
                            SET element at position element at position v of match of dist TO element at position u of dist PLUS one  
                            APPEND element at position v of match TO queue  
                        END IF  
                    END FOR  
                END IF  
            END WHILE  
            RETURN element at position negative one of dist IS LESS THAN infinity  
        END FUNCTION  
        
        FUNCTION dfs(u)  
            IF u IS NOT EQUALS negative one  
                FOR each v IN element at position u of graph  
                    IF element at position element at position v of match of dist EQUALS element at position u of dist PLUS one  
                        IF dfs(element at position v of match) RETURNS True  
                            SET element at position v of match TO u  
                            SET element at position u of match TO v  
                            RETURN True  
                        END IF  
                    END IF  
                END FOR  
                SET element at position u of dist TO infinity  
                RETURN False  
            END IF  
            RETURN True  
        END FUNCTION  
        
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        SET n TO rows MULTIPLIED BY cols  
        SET graph TO a list of empty sets of LENGTH n PLUS one  
        SET match TO a list of negative ones of LENGTH n PLUS one  
        SET dist TO a list of zeros of LENGTH n PLUS one  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of grid at position c EQUALS one  
                    SET u TO r MULTIPLIED BY cols PLUS c  
                    FOR each drdc IN list of pairs of negative one and zero and one and zero and zero and negative one and zero and one  
                        SET nr TO r PLUS first element of drdc  
                        SET nc TO c PLUS second element of drdc  
                        IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr of grid at position nc EQUALS one  
                            SET v TO nr MULTIPLIED BY cols PLUS nc  
                            ADD v TO element at position u of graph  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        SET matching TO zero  
        WHILE bfs() RETURNS True  
            FOR u FROM zero TO n MINUS one  
                IF element at position u of match EQUALS negative one AND dfs(u) RETURNS True  
                    INCREMENT matching BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN matching  
    END FUNCTION  
END CLASS
