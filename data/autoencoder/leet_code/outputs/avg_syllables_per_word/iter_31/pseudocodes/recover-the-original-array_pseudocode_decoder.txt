CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL method to sort nums in ascending order  
        SET n TO the length of nums  
        SET first TO element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE to next iteration  
            END IF  
            SET count TO a mapping recording the frequency of each element in nums  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF value associated with x in count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                IF value associated with x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT value associated with x in count BY one  
                DECREMENT value associated with x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the length of arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
