CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode with value as element at position one of values and left as nothing and right as nothing  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node.left TO new TreeNode with value as element at position i of values and left as nothing and right as nothing  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node.right TO new TreeNode with value as element at position i of values and left as nothing and right as nothing  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS nothing  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO dfs(node.left)  
            SET right_min, right_max, right_size TO dfs(node.right)  
            IF left_max LESS THAN node.val AND node.val LESS THAN right_min  
                RETURN minimum of node.val AND left_min, maximum of node.val AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, maximum of left_size AND right_size  
        END FUNCTION  
        RETURN element at position three of dfs(root)  
    END FUNCTION  
END CLASS
