CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF ONLY DIGITS THEN  
                SET num TO INTEGER CONVERSION OF block  
                IF num IS LESS THAN OR EQUAL TO two hundred and fifty five AND THE STRING CONVERSION OF num EQUALS block THEN  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF THE LENGTH OF block IS GREATER THAN OR EQUAL TO one AND THE LENGTH OF block IS LESS THAN OR EQUAL TO four THEN  
                TRY TO CONVERT block TO INTEGER WITH BASE sixteen  
                    RETURN True  
                IF CONVERSION FAILS THEN  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO THE LIST OF SUBSTRINGS OF queryIP SPLIT BY the character DOT  
        SET parts_ipv6 TO THE LIST OF SUBSTRINGS OF queryIP SPLIT BY the character COLON  

        IF THE LENGTH OF parts_ipv4 EQUALS four AND FOR EACH part IN parts_ipv4 THE RESULT OF is_valid_ipv4_block WITH part IS True THEN  
            RETURN THE STRING IPv4  
        ELSE IF THE LENGTH OF parts_ipv6 EQUALS eight AND FOR EACH part IN parts_ipv6 THE RESULT OF is_valid_ipv6_block WITH part IS True THEN  
            RETURN THE STRING IPv6  
        ELSE  
            RETURN THE STRING Neither  
        END IF  
    END FUNCTION  
END CLASS
