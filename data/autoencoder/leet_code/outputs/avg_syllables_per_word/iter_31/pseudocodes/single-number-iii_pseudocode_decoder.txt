CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO xor_all XOR num  
        END FOR  
        SET diff_bit TO the bitwise operation of xor_all AND the negative of xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the bitwise operation of num AND diff_bit NOT EQUALS zero  
                SET num1 TO num1 XOR num  
            ELSE  
                SET num2 TO num2 XOR num  
            END IF  
        END FOR  
        RETURN a list consisting of num1 and num2  
    END FUNCTION  
END CLASS
