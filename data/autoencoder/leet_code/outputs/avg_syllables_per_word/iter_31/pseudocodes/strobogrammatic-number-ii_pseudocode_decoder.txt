CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN the result of dfs WITH PARAMETER u MINUS two  
                FOR each pair OF strings eleven eighty-eight sixty-nine ninety-six  
                    APPEND concatenation of first character of pair plus v plus second character of pair TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of string zero plus v plus string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
