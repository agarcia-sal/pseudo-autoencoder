CLASS Solution  
    FUNCTION maximumRequests(n number_of_buildings requests list_of_lists_of_integers)  
        SET max_requests TO zero  
        SET num_requests TO the LENGTH OF requests  
        FOR mask FROM zero TO two RAISED TO THE POWER OF num_requests MINUS one  
            SET balance TO a list of zero repeated number_of_buildings times  
            SET count TO zero  
            FOR i FROM zero TO num_requests MINUS one  
                IF the bitwise AND of mask AND two RAISED TO THE POWER OF i NOT EQUALS zero  
                    SET from_building TO element at position i of requests THEN element at position zero  
                    SET to_building TO element at position i of requests THEN element at position one  
                    DECREMENT element at position from_building of balance BY one  
                    INCREMENT element at position to_building of balance BY one  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF for each element b IN balance the condition b EQUALS zero holds  
                SET max_requests TO the GREATER THAN OR EQUAL TO of max_requests AND count  
            END IF  
        END FOR  
        RETURN max_requests  
    END FUNCTION  
END CLASS
