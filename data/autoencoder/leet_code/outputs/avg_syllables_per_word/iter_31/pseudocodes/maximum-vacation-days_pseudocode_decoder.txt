CLASS Solution  
    FUNCTION maxVacationDays(flights AS list of list of integers days AS list of list of integers) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        SET dp TO a list of length n WHERE each element IS negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a mapping FROM integer TO list  
        FOR i FROM zero TO n MINUS one  
            ASSIGN an empty list TO the element at key i of graph  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of flights EQUALS one  
                    APPEND j TO the element at key i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n WHERE each element IS negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one  
                    CONTINUE to the next iteration  
                END IF  
                IF the element at position i of new_dp LESS THAN the element at position i of dp PLUS the element at position week of the element at position i of days  
                    SET the element at position i of new_dp TO the element at position i of dp PLUS the element at position week of the element at position i of days  
                END IF  
                FOR each j IN the element at key i of graph  
                    IF the element at position j of new_dp LESS THAN the element at position i of dp PLUS the element at position week of the element at position j of days  
                        SET the element at position j of new_dp TO the element at position i of dp PLUS the element at position week of the element at position j of days  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        SET max_value TO negative infinity  
        FOR each value IN dp  
            IF value GREATER THAN max_value  
                SET max_value TO value  
            END IF  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
