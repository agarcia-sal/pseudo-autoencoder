CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH value AT position one of values  
    SET p TO head  
    FOR each val IN elements of values FROM position two TO the end  
        SET node TO A NEW ListNode WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list(the next property of p1, the next property of p2)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH value AT position one of values  
    SET i TO one  
    SET queue TO AN EMPTY deque collection  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM THE LEFT OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET the left property of node TO A NEW TreeNode WITH value AT position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET the right property of node TO A NEW TreeNode WITH value AT position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    END IF  
    IF p EQUALS None OR q EQUALS None  
        RETURN False  
    END IF  
    IF the value property of p NOT EQUALS the value property of q  
        RETURN False  
    END IF  
    RETURN is_same_tree(the left property of p, the left property of q) AND is_same_tree(the right property of p, the right property of q)  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (element at position index of s IS A DIGIT OR element at position index of s EQUALS the minus sign)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO A NEW TreeNode WITH the integer PARSED FROM the substring of s FROM position start TO position index MINUS one  
            IF index LESS THAN the LENGTH OF s AND element at position index of s EQUALS an open parenthesis  
                SET the left property of node AND index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN the LENGTH OF s AND element at position index of s EQUALS an open parenthesis  
                SET the right property of node AND index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN the LENGTH OF s AND element at position index of s EQUALS a close parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND an unused variable TO parse_tree(zero)  
        RETURN root  
    END FUNCTION  
END CLASS
