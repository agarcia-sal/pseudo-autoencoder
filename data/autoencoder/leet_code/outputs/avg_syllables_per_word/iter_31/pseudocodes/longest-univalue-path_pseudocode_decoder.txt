CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position one of values and left equal to no value and right equal to no value  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values and left equal to no value and right equal to no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values and left equal to no value and right equal to no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN True  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left child of p and the left child of q AND the result of calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS no value THEN  
                RETURN zero, zero  
            END IF  
            SET left_length, left_value TO the result of calling dfs with the left child of node  
            SET right_length, right_value TO the result of calling dfs with the right child of node  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF the left child of node IS NOT no value AND the value of the left child of node EQUALS the value of node THEN  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF the right child of node IS NOT no value AND the value of the right child of node EQUALS the value of node THEN  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the greater of self.max_length AND the sum of left_arrow AND right_arrow  
            RETURN the greater of left_arrow AND right_arrow, and the value of node  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
