CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO an empty dictionary with default value zero  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET current_and TO element i BITWISE AND element j  
                INCREMENT the value in pair_and_count at key current_and BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and AND corresponding value count IN pair_and_count  
                IF pair_and BITWISE AND element k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
