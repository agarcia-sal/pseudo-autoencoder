CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  
        SET s TO list containing one followed by two followed by two  
        SET count TO one  
        SET index TO two  
        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the element at position LENGTH OF s MINUS one of s  
            FOR repeat FROM zero TO the element at position index of s MINUS one  
                APPEND next_num TO s  
            END FOR  
            IF next_num EQUALS one  
                INCREMENT count BY the element at position index of s  
            END IF  
            INCREMENT index BY one  
        END WHILE  
        IF the LENGTH OF s GREATER THAN n  
            SET extra_length TO the LENGTH OF s MINUS n  
            DECREMENT count BY the element at position LENGTH OF s MINUS one of s MULTIPLIED BY extra_length  
        END IF  
        RETURN count  
    END FUNCTION  
END CLASS
