CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each difference and y IN the items of dp  
                SET sum_when_added_to_taller_support TO difference PLUS rod  
                IF new_dp CONTAINS sum_when_added_to_taller_support  
                    SET current_value TO the value at sum_when_added_to_taller_support in new_dp  
                    IF current_value LESS THAN y  
                        SET new_dp at sum_when_added_to_taller_support TO y  
                    END IF  
                ELSE  
                    SET new_dp at sum_when_added_to_taller_support TO y  
                END IF  
                SET absolute_difference TO difference MINUS rod  
                IF absolute_difference LESS THAN zero  
                    SET absolute_difference TO rod MINUS difference  
                END IF  
                SET sum_when_added_to_shorter_support TO y PLUS the smaller of difference and rod  
                IF new_dp CONTAINS absolute_difference  
                    SET current_value TO the value at absolute_difference in new_dp  
                    IF current_value LESS THAN sum_when_added_to_shorter_support  
                        SET new_dp at absolute_difference TO sum_when_added_to_shorter_support  
                    END IF  
                ELSE  
                    SET new_dp at absolute_difference TO sum_when_added_to_shorter_support  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value at key zero in dp  
    END FUNCTION  
END CLASS
