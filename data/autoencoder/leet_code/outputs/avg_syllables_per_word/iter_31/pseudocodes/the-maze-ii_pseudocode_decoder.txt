CLASS Solution  
    FUNCTION shortestDistance with parameters maze start destination  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the LIST of pairs of one and zero MINUS one and zero zero and one zero and MINUS one  
        ASSIGN start TO the TUPLE of start  
        ASSIGN destination TO the TUPLE of destination  
        SET visited TO an empty SET  
        SET pq TO the LIST containing the PAIR of zero and start  

        WHILE pq is not empty  
            REMOVE the PAIR with the smallest first element FROM pq INTO dist and the PAIR x and y  
            
            IF the PAIR x and y EQUALS destination THEN  
                RETURN dist  
            END IF  
            
            IF the PAIR x and y IS IN visited THEN  
                CONTINUE to the next iteration  
            END IF  
            
            ADD the PAIR x and y TO visited  
            
            FOR each PAIR dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AND element at position new_y PLUS dy IS EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the PAIR new_x and new_y IS NOT IN visited THEN  
                    INSERT the PAIR of dist PLUS count and the PAIR new_x and new_y INTO pq maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
