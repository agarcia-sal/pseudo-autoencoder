CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the VALUE OF two RAISED TO THE POWER OF n MINUS one  
            SET good TO an empty list of Boolean values all set to false with the LENGTH OF n  
            SET bad TO an empty list of Boolean values all set to false with the LENGTH OF n  
            FOR j FROM zero TO n MINUS one  
                IF the result of the bitwise AND of i AND two RAISED TO THE POWER OF j EQUALS zero IS false  
                    SET the element at position j of good TO true  
                ELSE  
                    SET the element at position j of bad TO true  
                END IF  
            END FOR  
            SET valid TO true  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of good IS true  
                    FOR k FROM zero TO n MINUS one  
                        IF the element at position j of statements AT position k EQUALS zero AND the element at position k of good IS true OR the element at position j of statements AT position k EQUALS one AND the element at position k of bad IS true  
                            SET valid TO false  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                    IF valid IS false  
                        BREAK the outer FOR loop  
                    END IF  
                END IF  
            END FOR  
            IF valid IS true  
                SET max_good TO the GREATER OF max_good AND the SUM OF values in good where true counts as one  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
