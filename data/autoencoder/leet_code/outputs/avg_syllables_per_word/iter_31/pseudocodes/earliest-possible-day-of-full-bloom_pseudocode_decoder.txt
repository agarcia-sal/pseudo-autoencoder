CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET list_of_tasks TO the collection of pairs where each pair consists of the element at position index of plantTime AND the element at position index of growTime FOR each index FROM the start TO the end OF plantTime  
        SORT list_of_tasks BY the second element of each pair IN descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair IN list_of_tasks  
            SET plant TO the first element of pair  
            SET grow TO the second element of pair  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
