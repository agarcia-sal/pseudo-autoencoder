CLASS Solution  
    FUNCTION longestCommonSubpath(n, paths)  
        FUNCTION hash_subpaths(path, length)  
            SET P TO one hundred thirteen  
            SET MOD TO one billion PLUS seven  
            SET hash_value TO zero  
            SET power TO one  
            FOR index FROM zero TO length MINUS one  
                SET hash_value TO the value of hash_value MULTIPLIED BY P PLUS element at position index of path  
                SET hash_value TO the remainder of hash_value DIVIDED BY MOD  
                SET power TO the value of power MULTIPLIED BY P  
                SET power TO the remainder of power DIVIDED BY MOD  
            END FOR  
            SET hashes TO the set containing hash_value  
            FOR index FROM length TO the length of path MINUS one  
                SET hash_value TO the value of hash_value MULTIPLIED BY P PLUS element at position index of path MINUS element at position index MINUS length of path MULTIPLIED BY power  
                SET hash_value TO the remainder of hash_value DIVIDED BY MOD  
                ADD hash_value TO hashes  
            END FOR  
            RETURN hashes  
        END FUNCTION  
        FUNCTION check_common_subpath(length)  
            IF length EQUALS zero THEN  
                RETURN True  
            END IF  
            SET common_hashes TO the result of calling hash_subpaths with the first element of paths and length  
            FOR each path IN the elements of paths starting from the second element  
                SET current_hashes TO the result of calling hash_subpaths with path and length  
                SET common_hashes TO the intersection of common_hashes AND current_hashes  
                IF common_hashes IS empty THEN  
                    RETURN False  
                END IF  
            END FOR  
            RETURN whether common_hashes is not empty  
        END FUNCTION  
        SET left TO zero  
        SET right TO the minimum of the lengths of each path in paths  
        WHILE left LESS THAN right DO  
            SET mid TO the integer division of the sum of left AND right PLUS one BY two  
            IF the result of calling check_common_subpath with mid IS True THEN  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
