CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the member val TO the parameter val  
        SET the member left TO the parameter left  
        SET the member right TO the parameter right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the parameter values is empty  
        RETURN no value  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the length of queue is greater than zero  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the member left of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the member left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the member right of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the member right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is no value AND q is no value  
        RETURN True  
    ELSE IF p is no value OR q is no value  
        RETURN False  
    ELSE IF the member val of p NOT EQUALS the member val of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with the member left of p and the member left of q AND the result of is_same_tree called with the member right of p and the member right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder with parameters self root  
        IF root is no value  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue initialized with root  
        WHILE the length of queue is greater than zero  
            SET level_size TO the length of queue  
            SET current_level TO an empty list  
            FOR an unused variable from one TO level_size inclusive  
                SET node TO the element removed from the front of queue  
                APPEND the member val of node TO current_level  
                IF the member left of node is not no value  
                    APPEND the member left of node TO queue  
                END IF  
                IF the member right of node is not no value  
                    APPEND the member right of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
