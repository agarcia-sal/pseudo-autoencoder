CLASS UnionFind  
    FUNCTION __init__ with parameter size  
        SET self.parent TO list of numbers from zero TO size MINUS one  
        SET self.rank TO list of the number one repeated size times  
    END FUNCTION  
  
    FUNCTION find with parameter p  
        IF the element at position p of self.parent NOT EQUALS p THEN  
            SET the element at position p of self.parent TO the result of calling find with argument the element at position p of self.parent  
        END IF  
        RETURN the element at position p of self.parent  
    END FUNCTION  
  
    FUNCTION union with parameters p q  
        SET rootP TO the result of calling find with argument p  
        SET rootQ TO the result of calling find with argument q  
        IF rootP EQUALS rootQ THEN  
            RETURN False  
        END IF  
        IF the element at position rootP of self.rank GREATER THAN the element at position rootQ of self.rank THEN  
            SET the element at position rootQ of self.parent TO rootP  
        ELSE IF the element at position rootP of self.rank LESS THAN the element at position rootQ of self.rank THEN  
            SET the element at position rootP of self.parent TO rootQ  
        ELSE  
            SET the element at position rootQ of self.parent TO rootP  
            INCREMENT the element at position rootP of self.rank BY one  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION gcdSort with parameter nums  
        SET sorted_nums TO nums sorted in ascending order  
        SET max_val TO the maximum value in nums  
        SET uf TO a new instance of UnionFind with argument max_val PLUS one  
  
        FOR each num IN nums  
            FOR each factor FROM two TO the integer part of the square root of num PLUS one  
                IF num MODULO factor EQUALS zero THEN  
                    CALL union on uf with arguments num and factor  
                    CALL union on uf with arguments num and num DIVIDED BY factor  
                END IF  
            END FOR  
        END FOR  
  
        FOR each i FROM zero TO the length of nums MINUS one  
            IF result of calling find on uf with argument element at position i of nums NOT EQUALS result of calling find on uf with argument element at position i of sorted_nums THEN  
                RETURN False  
            END IF  
        END FOR  
  
        RETURN True  
    END FUNCTION  
END CLASS
