CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS list of strings  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one OF num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS the character zero  
                    CONTINUE TO next iteration  
                END IF  
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, the integer value of operand, and the integer value of operand  
                ELSE  
                    CALL dfs WITH arguments i, path concatenated with the character plus concatenated with operand, value PLUS the integer value of operand, and the integer value of operand  
                    CALL dfs WITH arguments i, path concatenated with the character minus concatenated with operand, value MINUS the integer value of operand, and the negative of the integer value of operand  
                    CALL dfs WITH arguments i, path concatenated with the character multiplied by concatenated with operand, value MINUS prev PLUS prev MULTIPLIED BY the integer value of operand, prev MULTIPLIED BY the integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL dfs WITH arguments zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
