CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s EQUALS REVERSE OF s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num EQUALS zero  
                RETURN the string representing zero  
            END IF  
            SET digits TO an empty list  
            WHILE num IS GREATER THAN zero  
                APPEND the remainder of num DIVIDED BY k TO digits  
                SET num TO the quotient of num DIVIDED BY k  
            END WHILE  
            RETURN concatenation of the string representations of elements in digits REVERSED  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the power of the integer division of length MINUS one BY two  
                SET end TO ten RAISED TO the power of the integer division of length PLUS one BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO string representation of half  
                    IF length MODULO two EQUALS zero  
                        SET full TO half_str CONCATENATED WITH REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH substring of half_str from second last element TO first element in reverse order  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes WITH PARAMETER length  
                SET num TO the integer value of pal  
                IF is_palindrome WITH PARAMETER to_base_k WITH PARAMETERS num AND k RETURNS true  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK the FOR loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the SUM OF all elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
