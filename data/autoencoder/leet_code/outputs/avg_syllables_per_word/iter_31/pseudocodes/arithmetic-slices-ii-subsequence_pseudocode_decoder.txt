CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        SET dp TO a list of default dictionaries mapping differences to integers for each position FROM zero TO n MINUS one  
        SET total_count TO zero  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET difference d TO the element at position i of nums MINUS the element at position j of nums  
                IF difference d EXISTS as a key IN the default dictionary at position j in dp  
                    INCREMENT the value for difference d in the default dictionary at position i in dp BY the value for difference d in the default dictionary at position j in dp PLUS one  
                    INCREMENT total_count BY the value for difference d in the default dictionary at position j in dp  
                ELSE  
                    INCREMENT the value for difference d in the default dictionary at position i in dp BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
