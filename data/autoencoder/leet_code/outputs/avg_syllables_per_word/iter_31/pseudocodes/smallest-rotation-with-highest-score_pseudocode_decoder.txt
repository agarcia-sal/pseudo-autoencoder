CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO the LENGTH OF nums  
        SET delta TO a list of zeroes of LENGTH n  
        
        FOR each index i AND element num IN nums  
            IF num LESS THAN OR EQUAL TO i  
                INCREMENT element at position zero of delta BY one  
                IF (i MINUS num PLUS one) LESS THAN n  
                    DECREMENT element at position (i MINUS num PLUS one) of delta BY one  
                END IF  
            END IF  
            
            IF (i PLUS one) LESS THAN n  
                INCREMENT element at position (i PLUS one) of delta BY one  
                IF ((i PLUS n MINUS num) PLUS one) LESS THAN n  
                    DECREMENT element at position ((i PLUS n MINUS num) PLUS one) of delta BY one  
                END IF  
            END IF  
        END FOR  
        
        SET score TO zero  
        SET max_score TO zero  
        SET best_k TO zero  
        
        FOR each index k FROM zero TO n MINUS one  
            INCREMENT score BY element at position k of delta  
            IF score GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
