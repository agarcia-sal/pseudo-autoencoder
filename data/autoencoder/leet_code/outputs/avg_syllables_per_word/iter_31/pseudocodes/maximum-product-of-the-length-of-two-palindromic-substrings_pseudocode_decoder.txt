CLASS Solution  
    FUNCTION maxProduct(s)  
        SET n TO the LENGTH OF s  
        SET max_len_end TO a list of size n FILLED WITH one  
        SET max_len_start TO a list of size n FILLED WITH one  
        
        FUNCTION expand_around_center(left, right)  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN n AND element at position left of s EQUALS element at position right of s  
                SET length TO right MINUS left PLUS one  
                IF element at position right of max_len_end IS LESS THAN length  
                    SET element at position right of max_len_end TO length  
                END IF  
                IF element at position left of max_len_start IS LESS THAN length  
                    SET element at position left of max_len_start TO length  
                END IF  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            CALL expand_around_center WITH i AND i  
        END FOR  
        
        SET prefix_max TO a list of size n FILLED WITH zero  
        SET suffix_max TO a list of size n FILLED WITH zero  
        
        SET element at position zero of prefix_max TO element at position zero of max_len_end  
        FOR i FROM one TO n MINUS one  
            IF element at position i MINUS one of prefix_max IS GREATER THAN element at position i of max_len_end  
                SET element at position i of prefix_max TO element at position i MINUS one of prefix_max  
            ELSE  
                SET element at position i of prefix_max TO element at position i of max_len_end  
            END IF  
        END FOR  
        
        SET element at position n MINUS one of suffix_max TO element at position n MINUS one of max_len_start  
        FOR i FROM n MINUS two DOWN TO zero  
            IF element at position i PLUS one of suffix_max IS GREATER THAN element at position i of max_len_start  
                SET element at position i of suffix_max TO element at position i PLUS one of suffix_max  
            ELSE  
                SET element at position i of suffix_max TO element at position i of max_len_start  
            END IF  
        END FOR  
        
        SET max_product TO zero  
        FOR i FROM zero TO n MINUS two  
            SET product TO element at position i of prefix_max MULTIPLIED BY element at position i PLUS one of suffix_max  
            IF max_product IS LESS THAN product  
                SET max_product TO product  
            END IF  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
