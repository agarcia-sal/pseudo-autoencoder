CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with the LENGTH OF two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the NEGATION OF the element at position i of nums INTO min_heap as a heap  
            IF the LENGTH OF min_heap GREATER THAN n  
                POP the smallest element FROM min_heap as a heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n  
                SET the element at position i of min_sums TO the NEGATION OF the SUM OF elements IN min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with the LENGTH OF two MULTIPLIED BY n PLUS one  
        
        FOR i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH the element at position i of nums INTO max_heap as a heap  
            IF the LENGTH OF max_heap GREATER THAN n  
                POP the smallest element FROM max_heap as a heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n  
                SET the element at position i of max_sums TO the SUM OF elements IN max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO infinity  
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET candidate_diff TO the element at position i of min_sums MINUS the element at position i PLUS one of max_sums  
            IF candidate_diff LESS THAN min_diff  
                SET min_diff TO candidate_diff  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
