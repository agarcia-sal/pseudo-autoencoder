CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SET n TO the LENGTH OF s  
        
        CREATE two dimensional list next_occurrence with four rows and n columns FILLED WITH negative one  
        CREATE two dimensional list prev_occurrence with four rows and n columns FILLED WITH negative one  
        
        FOR c FROM zero TO three  
            SET char TO the character at position corresponding to the character code of lowercase a PLUS c  
            SET last TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position i of row c of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR i FROM n MINUS one DOWN TO zero  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position i of row c of next_occurrence TO last  
            END FOR  
        END FOR  
        
        CREATE two dimensional list dp with n rows and n columns FILLED WITH zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at row i and column i of dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                FOR c FROM zero TO three  
                    SET char TO the character at position corresponding to the character code of lowercase a PLUS c  
                    SET left TO element at position i of row c of next_occurrence  
                    SET right TO element at position j of row c of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i THEN  
                        CONTINUE TO next iteration of c loop  
                    ELSE IF left EQUALS right THEN  
                        SET element at row i and column j of dp TO element at row i and column j of dp PLUS one MODULO MOD  
                    ELSE  
                        SET element at row i and column j of dp TO element at row i and column j of dp PLUS element at row left PLUS one and column right MINUS one of dp PLUS two MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at row zero and column n MINUS one of dp  
    END FUNCTION  
END CLASS
