CLASS Solution  
    FUNCTION isValidPalindrome(s k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional list of size n by n FILLED WITH zero  
            FOR i FROM zero TO n MINUS one  
                SET element at position i of dp at position i TO one  
            END FOR  
            FOR length FROM two TO n  
                FOR i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF element at position i of s EQUALS element at position j of s  
                        SET element at position i of dp at position j TO element at position i PLUS one of dp at position j MINUS one PLUS two  
                    ELSE  
                        IF element at position i PLUS one of dp at position j GREATER THAN element at position i of dp at position j MINUS one  
                            SET element at position i of dp at position j TO element at position i PLUS one of dp at position j  
                        ELSE  
                            SET element at position i of dp at position j TO element at position i of dp at position j MINUS one  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero of dp at position n MINUS one  
        END FUNCTION  
        SET longest_palindrome_length TO longest_palindromic_subseq(s)  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
