CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i j  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            RETURN the maximum BETWEEN element at position i of piles MINUS dfs OF i INCREMENTED BY one AND j AND element at position j of piles MINUS dfs OF i AND j DECREMENTED BY one  
        END FUNCTION  
        RETURN dfs OF zero AND LENGTH OF piles MINUS one GREATER THAN zero  
    END FUNCTION  
END CLASS
