CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO TreeNode(element at position one of values)  
    SET i TO one  
    INITIALIZE queue AS empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE element from the left of queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO TreeNode(element at position i of values)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO TreeNode(element at position i of values)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root EQUALS None  
            RETURN zero  
        END IF  
        FUNCTION dfs(node, parent, current_length)  
            IF node EQUALS None  
                RETURN current_length  
            END IF  
            IF parent NOT EQUALS None AND node.val EQUALS parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET left_length TO dfs(node.left, node.val, current_length)  
            SET right_length TO dfs(node.right, node.val, current_length)  
            RETURN the maximum of current_length AND left_length AND right_length  
        END FUNCTION  
        RETURN dfs(root, None, zero)  
    END FUNCTION  
END CLASS
