CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND element h IN heights  
            WHILE the element at the last position of stack NOT EQUALS negative one AND h LESS THAN the element at the element at the last position of stack of heights  
                SET height TO the element at the position obtained by removing the last element from stack of heights  
                SET width TO i MINUS the element at the last position of stack MINUS one  
                SET max_area TO the greater value between max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
