CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero and firstPerson  
        SORT meetings BY element at position three of each meeting in ascending order  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO element at position three of the meeting at position i in meetings  
            CREATE graph AS a mapping from keys to empty lists by default  
            WHILE i LESS THAN the LENGTH OF meetings AND element at position three of the meeting at position i in meetings EQUALS current_time  
                SET x TO element at position one of the meeting at position i in meetings  
                SET y TO element at position two of the meeting at position i in meetings  
                APPEND y TO the list associated with key x in graph  
                APPEND x TO the list associated with key y in graph  
                INCREMENT i BY one  
            END WHILE  
            CREATE queue AS a double ended queue containing each person FROM know_secret THAT is a key in graph  
            CREATE visited AS an empty set  
            WHILE queue is not empty  
                REMOVE and RETURN the element from the front of queue AND ASSIGN TO person  
                IF person is an element of visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list associated with key person in graph  
                    IF neighbor is not an element of visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list created from the elements of know_secret  
    END FUNCTION  
END CLASS
