CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of elements in nums  
        SET dp TO a list of zero elements with length equal to two raised to the power of ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two raised to the power of element at position one of count  
                CONTINUE to the next iteration  
            END IF  
            IF num NOT IN keys of prime_factors  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO value at key num in prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero STEP negative one  
                IF bitwise AND of m and mask EQUALS zero  
                    SET element at position bitwise OR of m and mask of dp TO (element at position bitwise OR of m and mask of dp PLUS element at position m of dp MULTIPLIED BY element at position num of count) MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements of dp from position one to the end MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
