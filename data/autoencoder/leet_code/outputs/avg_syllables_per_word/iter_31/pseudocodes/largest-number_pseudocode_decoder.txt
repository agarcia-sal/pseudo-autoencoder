CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a new list created by converting each element of nums INTO its string representation  
        
        FUNCTION compare(x, y)  
            IF the concatenation of x followed by y GREATER THAN the concatenation of y followed by x THEN  
                RETURN negative one  
            ELSE IF the concatenation of x followed by y LESS THAN the concatenation of y followed by x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING compare AS the comparator  
        
        IF element at position zero of nums_str EQUALS the string zero THEN  
            RETURN the string zero  
        END IF  
        
        SET result TO the concatenation of all elements in nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
