CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the element at the front of queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT a no value  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT a no value  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues WITH PARAMETERS self root  
        IF root IS no value  
            RETURN an empty list  
        END IF  
        CREATE a double ended queue named queue INITIATED WITH root AS its single element  
        CREATE an empty list named largest_values  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each position FROM one TO level_size  
                REMOVE the element at the front of queue AND ASSIGN TO node  
                IF max_value IS LESS THAN node.val  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT no value  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT no value  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
