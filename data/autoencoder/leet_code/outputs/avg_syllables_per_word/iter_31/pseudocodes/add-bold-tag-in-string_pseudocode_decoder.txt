CLASS Solution  
    FUNCTION addBoldTag(s string words List of string) string  
        IF the LENGTH OF words EQUALS zero THEN  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        SET mask TO a list of the value FALSE repeated n times  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the position of the first occurrence of word in s starting at position start  
                IF start EQUALS minus one THEN  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF the element at position i of mask IS TRUE THEN  
                APPEND the string representing start bold tag TO result  
                WHILE i LESS THAN n AND the element at position i of mask IS TRUE  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string representing end bold tag TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
