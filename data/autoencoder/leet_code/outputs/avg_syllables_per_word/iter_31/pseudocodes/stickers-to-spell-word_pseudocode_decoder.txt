CLASS Solution  
    FUNCTION minStickers(stickers AS List OF string, target AS string) RETURNS integer  
        SET stickers TO a new list created by for each sticker IN stickers create a mapping of character frequencies in the sticker  
        SORT stickers in descending order based on the highest frequency value among characters in each sticker  
        ASSIGN a double ended queue TO queue initialized with a pair consisting of target and zero  
        ASSIGN a mapping TO memo initialized with a key of target and value zero  
        WHILE the queue is not empty  
            REMOVE and ASSIGN the first pair from queue TO current_target and used_stickers  
            FOR each sticker IN stickers  
                IF the character at the first position of current_target is NOT found among the keys of sticker  
                    CONTINUE to the next iteration of the loop  
                END IF  
                CREATE a mapping called remaining as a frequency count of characters in current_target  
                FOR each character key in sticker  
                    DECREASE the count of character key in remaining by the count of character key in sticker  
                END FOR  
                CREATE a string new_target by concatenating all characters from remaining repeated according to their positive counts sorted in ascending order  
                IF the length of new_target EQUALS zero  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target is NOT a key in memo  
                    SET memo at key new_target TO used_stickers PLUS one  
                    APPEND the pair of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
