CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO an empty dictionary mapping each unique character in s TO an empty double ended queue  
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element of it  
            IF first_letter is a key in waiting  
                APPEND it TO the double ended queue AT key first_letter in waiting  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN the string s  
            IF c is a key in waiting  
                SET current_iterators TO the double ended queue AT key c in waiting  
                SET the double ended queue AT key c in waiting TO an empty double ended queue  
                FOR each it IN current_iterators  
                    SET next_char TO the next element of it OR none if iterator is exhausted  
                    IF next_char is none  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is a key in waiting  
                            APPEND it TO the double ended queue AT key next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
