CLASS Solution  
    FUNCTION minOperationsToFlip(expression)  
        SET stack TO an empty list  
        FOR each element e IN expression  
            IF element e EQUALS one of open parenthesis AND sign AND vertical bar  
                APPEND a pair of element e AND zero TO stack  
                CONTINUE to next iteration  
            ELSE IF element e EQUALS closing parenthesis  
                SET lastPair TO the last element REMOVED FROM stack  
                REMOVE the last element FROM stack  
            ELSE  
                SET lastPair TO a pair of element e AND one  
            END IF  
            IF stack is not empty AND the last element in stack AT position zero EQUALS one of sign AND vertical bar  
                SET op TO the last element AT position zero REMOVED FROM stack  
                SET a AND costA TO the last pair REMOVED FROM stack  
                SET b AND costB TO lastPair  
                IF op EQUALS sign  
                    IF a EQUALS zero AND b EQUALS zero  
                        SET lastPair TO a pair of zero AND one INCREMENTED BY the minimum of costA AND costB  
                    ELSE IF a EQUALS zero AND b EQUALS one  
                        SET lastPair TO a pair of zero AND one  
                    ELSE IF a EQUALS one AND b EQUALS zero  
                        SET lastPair TO a pair of zero AND one  
                    ELSE  
                        SET lastPair TO a pair of one AND the minimum of costA AND costB  
                    END IF  
                ELSE  
                    IF a EQUALS zero AND b EQUALS zero  
                        SET lastPair TO a pair of zero AND the minimum of costA AND costB  
                    ELSE IF a EQUALS zero AND b EQUALS one  
                        SET lastPair TO a pair of one AND one  
                    ELSE IF a EQUALS one AND b EQUALS zero  
                        SET lastPair TO a pair of one AND one  
                    ELSE  
                        SET lastPair TO a pair of one AND one INCREMENTED BY the minimum of costA AND costB  
                    END IF  
                END IF  
            END IF  
            APPEND lastPair TO stack  
        END FOR  
        RETURN the second element of the last element in stack  
    END FUNCTION  
END CLASS
