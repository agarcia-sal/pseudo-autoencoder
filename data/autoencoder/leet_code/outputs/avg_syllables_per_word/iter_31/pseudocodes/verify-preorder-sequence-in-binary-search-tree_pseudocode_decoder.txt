CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity value  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND value GREATER THAN element at position the LENGTH OF stack MINUS one of stack  
                SET lower_bound TO the element popped from the end of stack  
            END WHILE  
            
            APPEND value TO the end of stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
