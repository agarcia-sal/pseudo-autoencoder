CLASS Solution  
    FUNCTION canPartitionKSubsets(nums PARAMETER k)  
        SET total_sum TO the SUM OF nums  
        IF the REMAINDER OF total_sum DIVIDED BY k NOT EQUALS zero  
            RETURN false  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending order  

        FUNCTION can_partition(index PARAMETER k PARAMETER current_sum PARAMETER used)  
            IF k EQUALS zero  
                RETURN true  
            END IF  
            IF current_sum EQUALS target  
                RETURN the RESULT OF can_partition WITH zero AND k MINUS one AND zero AND used  
            END IF  

            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF the element at position i of used EQUALS false AND current_sum PLUS the element at position i of nums LESS THAN OR EQUAL TO target  
                    SET the element at position i of used TO true  
                    IF the RESULT OF can_partition WITH i PLUS one AND k AND current_sum PLUS the element at position i of nums AND used EQUALS true  
                        RETURN true  
                    END IF  
                    SET the element at position i of used TO false  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  

        SET used TO a LIST of false VALUES WITH the LENGTH OF nums  
        RETURN the RESULT OF can_partition WITH zero AND k AND zero AND used  
    END FUNCTION  
END CLASS
