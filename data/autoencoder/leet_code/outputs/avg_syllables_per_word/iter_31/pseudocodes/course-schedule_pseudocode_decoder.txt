CLASS Solution  
    FUNCTION canFinish(numCourses numCourses PARAMETERS, prerequisites prerequisites PARAMETERS)  
        ASSIGN reference TO a new collection mapping with default empty lists named graph  
        SET a list named indegree TO a list of zeros of length numCourses  
        FOR each pair of course and prereq IN prerequisites  
            APPEND the course TO the list at key prereq in graph  
            INCREMENT the value at position course in indegree BY one  
        END FOR  
        ASSIGN reference TO a new double ended queue named queue initialized with each course IN the range from zero TO numCourses MINUS one whose value at position course in indegree EQUALS zero  
        WHILE queue is not empty  
            REMOVE and ASSIGN the element from the front of queue TO current_course  
            DECREMENT numCourses BY one  
            FOR each next_course IN the list at key current_course in graph  
                DECREMENT the value at position next_course in indegree BY one  
                IF the value at position next_course in indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the condition WHETHER numCourses EQUALS zero  
    END FUNCTION  
END CLASS
