CLASS Solution  
    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min_heap TO empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            CALL heap push ON min_heap WITH THE TUPLE OF the element at position i of arr DIVIDED BY the element at position n MINUS one of arr AND i AND n MINUS one  
        END FOR  
        FOR each iteration FROM zero TO k MINUS one MINUS one  
            CALL heap pop ON min_heap AND ASSIGN THE RETURNED TUPLE TO discarded_fraction AND i AND j  
            IF j MINUS one GREATER THAN i THEN  
                CALL heap push ON min_heap WITH THE TUPLE OF the element at position i of arr DIVIDED BY the element at position j MINUS one of arr AND i AND j MINUS one  
            END IF  
        END FOR  
        CALL heap pop ON min_heap AND ASSIGN THE RETURNED TUPLE TO discarded_fraction AND i AND j  
        RETURN a list CONTAINING the element at position i of arr AND the element at position j of arr  
    END FUNCTION  
END CLASS
