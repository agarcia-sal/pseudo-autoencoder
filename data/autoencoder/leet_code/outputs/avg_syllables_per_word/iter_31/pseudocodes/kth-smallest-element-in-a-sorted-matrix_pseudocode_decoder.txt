CLASS Solution  
    FUNCTION kthSmallest(matrix WITH TYPE list of list of integer, k WITH TYPE integer) RETURNS integer  
        FUNCTION countLessEqual(mid WITH TYPE integer) RETURNS integer  
            SET count TO zero  
            SET row TO the LENGTH OF matrix MINUS one  
            SET col TO zero  
            WHILE row IS GREATER THAN OR EQUAL TO zero AND col IS LESS THAN the LENGTH OF matrix  
                IF element at position row of matrix AND position col of that element IS LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  

        SET n TO the LENGTH OF matrix  
        SET low TO element at position zero of matrix AND position zero of that element  
        SET high TO element at position n MINUS one of matrix AND position n MINUS one of that element  

        WHILE low IS LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two WITH INTEGER DIVISION  
            IF countLessEqual(mid) IS LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  

        RETURN low  
    END FUNCTION  
END CLASS
