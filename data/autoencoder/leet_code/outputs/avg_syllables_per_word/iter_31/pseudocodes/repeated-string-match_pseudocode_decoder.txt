CLASS Solution  
    FUNCTION repeatedStringMatch(a AS string, b AS string) RETURNS integer  
        SET minimum_repetitions TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF the string b is a substring of the string created by repeating the string a minimum_repetitions times  
            RETURN minimum_repetitions  
        END IF  
        IF the string b is a substring of the string created by repeating the string a the sum of minimum_repetitions PLUS one times  
            RETURN the sum of minimum_repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
