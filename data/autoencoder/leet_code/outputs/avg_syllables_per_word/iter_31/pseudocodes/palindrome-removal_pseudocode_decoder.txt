CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO a list consisting of n elements where each element is a list of n zeros  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at dp AT position i TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i of arr EQUALS element at position i PLUS one of arr  
                SET element at position i at dp AT position i PLUS one TO one  
            ELSE  
                SET element at position i at dp AT position i PLUS one TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of arr EQUALS element at position j of arr  
                    SET element at position i at dp AT position j TO element at position i PLUS one at dp AT position j MINUS one  
                ELSE  
                    SET element at position i at dp AT position j TO infinity  
                    FOR k FROM i TO j MINUS one  
                        IF element at position i at dp AT position j IS GREATER THAN element at position i at dp AT position k PLUS element at position k PLUS one at dp AT position j  
                            SET element at position i at dp AT position j TO element at position i at dp AT position k PLUS element at position k PLUS one at dp AT position j  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero at dp AT position n MINUS one  
    END FUNCTION  
END CLASS
