CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS nothing OR element at position zero of matrix EQUALS nothing THEN  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a new list containing m PLUS one elements each of which is a list containing n PLUS one zeros  
        SET max_side TO zero  
        FOR i FROM one TO m PLUS zero  
            FOR j FROM one TO n PLUS zero  
                IF element at position i MINUS one of matrix at position j MINUS one EQUALS string one THEN  
                    SET dp at position i at position j TO the minimum of dp at position i MINUS one at position j AND dp at position i at position j MINUS one AND dp at position i MINUS one at position j MINUS one PLUS one  
                    SET max_side TO the maximum of max_side AND dp at position i at position j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
