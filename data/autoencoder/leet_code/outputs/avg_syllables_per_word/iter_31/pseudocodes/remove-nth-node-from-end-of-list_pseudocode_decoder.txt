CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN values from position two TO the last position  
        SET node TO new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list called with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO new ListNode with zero  
        SET the next attribute of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the next attribute of first  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO the next attribute of first  
            SET second TO the next attribute of second  
        END WHILE  
        SET the next attribute of second TO the next attribute of the next attribute of second  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
