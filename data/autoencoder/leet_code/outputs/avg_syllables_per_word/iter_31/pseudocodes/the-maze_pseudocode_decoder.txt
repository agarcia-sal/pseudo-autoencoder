CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze start destination  
        SET directions TO the list of pairs representing zero and one zero and one zero and negative one and negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETERS x y  
            IF pair consisting of x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF the list consisting of x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD pair consisting of x and y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze AT position ny PLUS dy EQUALS zero  
                    SET nx TO nx PLUS dx  
                    SET ny TO ny PLUS dy  
                END WHILE  
                
                IF dfs WITH PARAMETERS nx ny RETURNS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs WITH PARAMETERS element at position zero of start element at position one of start  
    END FUNCTION  
END CLASS
