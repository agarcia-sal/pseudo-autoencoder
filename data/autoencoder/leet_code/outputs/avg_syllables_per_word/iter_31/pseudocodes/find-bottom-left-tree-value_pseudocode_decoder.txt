CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN nothing  
    END IF  
    SET root TO A new TreeNode INITIALIZED WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node.left TO A new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node.right TO A new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH p.left AND q.left AND the result of is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a double ended queue INITIALIZED WITH root  
        SET leftmost_value TO nothing  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO the first element REMOVED FROM queue  
                IF i EQUALS zero THEN  
                    SET leftmost_value TO the value of node  
                END IF  
                IF node.left IS NOT nothing THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT nothing THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
