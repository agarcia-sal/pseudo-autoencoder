CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
  
        SET is_prime TO a list of Boolean true values with the length n  
        SET element at position zero of is_prime TO Boolean false  
        SET element at position one of is_prime TO Boolean false  
  
        FOR start FROM two TO the integer part of n raised to the power of one divided by two PLUS one  
            IF the element at position start of is_prime EQUALS Boolean true  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one PLUS one STEP start  
                    SET element at position multiple of is_prime TO Boolean false  
                END FOR  
            END IF  
        END FOR  
  
        SET prime_count TO zero  
        FOR each value IN is_prime  
            IF value EQUALS Boolean true  
                INCREMENT prime_count BY one  
            END IF  
        END FOR  
  
        RETURN prime_count  
    END FUNCTION  
END CLASS
