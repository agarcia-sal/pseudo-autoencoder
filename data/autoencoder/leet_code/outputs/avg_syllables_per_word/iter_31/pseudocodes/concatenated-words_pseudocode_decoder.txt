CLASS Trie  
    FUNCTION __init__(self)  
        SET self.children TO a list of None elements with length twenty six  
        SET self.is_end TO False  
    END FUNCTION  

    FUNCTION insert(self, w)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the numerical order of character c MINUS the numerical order of character a  
            IF the element at position idx of node.children EQUALS None THEN  
                SET the element at position idx of node.children TO a new Trie object  
            END IF  
            SET node TO the element at position idx of node.children  
        END FOR  
        SET node.is_end TO True  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(self, words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the length of w THEN  
                RETURN True  
            END IF  
            SET node TO trie  
            FOR i FROM start TO the length of w MINUS one  
                SET idx TO the numerical order of the element at position i of w MINUS the numerical order of character a  
                IF the element at position idx of node.children EQUALS None THEN  
                    RETURN False  
                END IF  
                SET node TO the element at position idx of node.children  
                IF node.is_end EQUALS True AND dfs(w, i PLUS one) EQUALS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET trie TO a new Trie object  
        SET ans TO an empty list  
        SORT words in ascending order by the length of each element  
        FOR each element w IN words  
            IF w DOES NOT EQUAL the empty string THEN  
                IF dfs(w, zero) EQUALS True THEN  
                    APPEND w TO ans  
                ELSE  
                    CALL insert function of trie WITH w AS argument  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
