CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture EQUALS none OR the LENGTH OF picture EQUALS zero OR the LENGTH OF element at position zero of picture EQUALS zero  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a new list of size m with all elements set to zero  
        SET col_count TO a new list of size n with all elements set to zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of picture AT position j EQUALS the capital letter B  
                    INCREMENT element at position i of row_count BY one  
                    INCREMENT element at position j of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of picture AT position j EQUALS the capital letter B AND element at position i of row_count EQUALS one AND element at position j of col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
