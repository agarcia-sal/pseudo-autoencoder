CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with val equal to the element at position one of values and next equal to None  
    SET p TO head  
    FOR each val IN the list of elements from position two TO the last position of values  
        SET node TO a new ListNode with val equal to val and next equal to None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the result of the expression where the val attribute of p1 EQUALS the val attribute of p2 AND the recursive call of is_same_list with arguments p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS None OR the next attribute of head EQUALS None  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next attribute of head  
            WHILE fast IS NOT None AND the next attribute of fast IS NOT None  
                SET slow TO the next attribute of slow  
                SET fast TO the next attribute of the next attribute of fast  
            END WHILE  
            SET mid TO the next attribute of slow  
            SET the next attribute of slow TO None  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO a new ListNode with val equal to zero AND next equal to None  
            SET current TO dummy  
            WHILE l1 IS NOT None AND l2 IS NOT None  
                IF the val attribute of l1 LESS THAN the val attribute of l2  
                    SET the next attribute of current TO l1  
                    SET l1 TO the next attribute of l1  
                ELSE  
                    SET the next attribute of current TO l2  
                    SET l2 TO the next attribute of l2  
                END IF  
                SET current TO the next attribute of current  
            END WHILE  
            IF l1 IS NOT None  
                SET the next attribute of current TO l1  
            ELSE  
                SET the next attribute of current TO l2  
            END IF  
            RETURN the next attribute of dummy  
        END FUNCTION  

        SET left AND right TO the result of the call of split with head  
        SET left TO the result of the call of sortList with left  
        SET right TO the result of the call of sortList with right  

        RETURN the result of the call of merge with left AND right  
    END FUNCTION  
END CLASS
