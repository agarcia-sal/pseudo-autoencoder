CLASS Solution  
    FUNCTION shortestPathAllKeys(grid)  
        SET directions TO list of pairs zero and one zero and minus one one and zero minus one and zero  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start TO none  
        SET num_keys TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid EQUALS the character at sign at symbol THEN  
                    SET start TO pair of i and j  
                ELSE IF element at position i of grid at position j IS LOWERCASE LETTER  
                    INCREMENT num_keys BY one  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a double ended queue initialized with a single tuple consisting of element at position zero of start element and element at position one of start element zero zero  
        SET visited TO a set initialized with a single tuple consisting of element at position zero of start element element at position one of start element and zero  
        WHILE queue is not empty  
            REMOVE and RETURN the leftmost element of queue and ASSIGN it to x y keys steps  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET cell TO element at position nx of grid at position ny  
                    IF cell EQUALS the character at hash symbol THEN  
                        CONTINUE to next iteration of the loop  
                    END IF  
                    IF cell IS UPPERCASE LETTER  
                        SET key_needed TO one LEFT SHIFTED BY the difference between the Unicode code point of the lowercase version of cell MINUS the Unicode code point of letter a  
                        IF keys BITWISE AND key_needed EQUALS zero  
                            CONTINUE to next iteration of the loop  
                        END IF  
                    END IF  
                    IF cell IS LOWERCASE LETTER  
                        SET new_keys TO keys BITWISE OR one LEFT SHIFTED BY the difference between the Unicode code point of cell MINUS the Unicode code point of letter a  
                        IF new_keys EQUALS one LEFT SHIFTED BY num_keys MINUS one  
                            RETURN steps PLUS one  
                        END IF  
                    ELSE  
                        SET new_keys TO keys  
                    END IF  
                    IF the tuple of nx ny new_keys is not contained in visited  
                        ADD the tuple of nx ny new_keys to visited  
                        APPEND the tuple of nx ny new_keys steps PLUS one to queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
