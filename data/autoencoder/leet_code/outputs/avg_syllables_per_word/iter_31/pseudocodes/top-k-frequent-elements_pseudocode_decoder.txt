CLASS Solution  
    FUNCTION topKFrequent(nums k)  
        SET frequency_map TO a default dictionary with default integer zero  
        FOR each num IN nums  
            INCREMENT the element at position num of frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all the values in frequency_map  
        SET buckets TO a list consisting of empty lists repeated max_frequency PLUS one times  
        FOR each num frequency IN the pairs of items in frequency_map  
            APPEND num TO the element at position frequency of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one BY decrement of one  
            IF the length of the element at position freq of buckets GREATER THAN zero THEN  
                EXTEND result BY the element at position freq of buckets  
                IF the length of result GREATER THAN OR EQUAL TO k THEN  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
