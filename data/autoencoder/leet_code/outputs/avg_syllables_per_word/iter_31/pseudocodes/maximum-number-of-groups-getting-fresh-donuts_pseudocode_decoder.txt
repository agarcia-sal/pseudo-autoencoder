CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  

        SET remainder_count TO a list of zeros with the LENGTH EQUAL TO batchSize  

        FOR each group IN groups  
            INCREMENT element at position group MODULO batchSize of remainder_count BY one  
        END FOR  

        FUNCTION dp(remainder, variable number of remainder_count_elements)  
            SET remainder_count TO a list constructed from remainder_count_elements  
            SET max_happy TO zero  

            FOR index i FROM zero TO batchSize MINUS one  
                IF element at position i of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  

                DECREMENT element at position i of remainder_count BY one  
                SET new_remainder TO remainder MINUS i MODULO batchSize  

                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, unpack remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, unpack remainder_count)  
                END IF  

                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  

                INCREMENT element at position i of remainder_count BY one  
            END FOR  

            RETURN max_happy  
        END FUNCTION  

        RETURN dp(zero, unpack remainder_count)  
    END FUNCTION  
END CLASS
