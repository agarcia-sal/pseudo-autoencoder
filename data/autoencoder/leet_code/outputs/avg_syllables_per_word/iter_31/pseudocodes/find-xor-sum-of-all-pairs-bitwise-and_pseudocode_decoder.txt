CLASS Solution  
    FUNCTION getXORSum(arr1 arr OF integers arr2 arr OF integers)  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the bitwise XOR of xor_sum_arr1 AND num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the bitwise XOR of xor_sum_arr2 AND num  
        END FOR  
        RETURN the bitwise AND of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
