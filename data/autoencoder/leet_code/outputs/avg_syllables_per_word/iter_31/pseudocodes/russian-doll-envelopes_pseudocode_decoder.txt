CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        CALL sort envelopes BY the element at position zero of envelope IN ascending order AND BY the negative of the element at position one of envelope IN descending order  
        SET dp TO empty list  
        FOR each envelope_element UNDERSCORE h IN envelopes  
            SET idx TO the position found by binary search for h in dp where h can be inserted to maintain sorted order from left  
            IF idx EQUALS the length of dp THEN  
                APPEND h TO dp  
            ELSE  
                SET element at position idx of dp TO h  
            END IF  
        END FOR  
        RETURN the length of dp  
    END FUNCTION  
END CLASS
