CLASS Solution  
    FUNCTION minTrioDegree(n parameter ONE edges parameter TWO)  
        SET graph TO an empty mapping from each integer FROM one TO n parameter ONE mapped TO an empty set  
        FOR each pair u parameter THREE AND v parameter FOUR IN edges parameter TWO  
            ADD v parameter FOUR TO the set mapped FROM u parameter THREE IN graph  
            ADD u parameter THREE TO the set mapped FROM v parameter FOUR IN graph  
        END FOR  
        SET min_degree TO infinity  
        SET trio_found TO false  
        FOR integer i parameter FIVE FROM one TO n parameter ONE  
            FOR integer j parameter SIX FROM i parameter FIVE PLUS one TO n parameter ONE  
                FOR integer k parameter SEVEN FROM j parameter SIX PLUS one TO n parameter ONE  
                    IF the set mapped FROM i parameter FIVE IN graph CONTAINS j parameter SIX AND the set mapped FROM i parameter FIVE IN graph CONTAINS k parameter SEVEN AND the set mapped FROM j parameter SIX IN graph CONTAINS k parameter SEVEN  
                        SET trio_found TO true  
                        SET trio_degree TO the size OF the set mapped FROM i parameter FIVE IN graph PLUS the size OF the set mapped FROM j parameter SIX IN graph PLUS the size OF the set mapped FROM k parameter SEVEN IN graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found EQUALS true  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
