CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary with key zero assigned value one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            IF current_sum MINUS k EXISTS AS A KEY IN cumulative_sum_freq  
                INCREMENT count BY the value corresponding to the key current_sum MINUS k in cumulative_sum_freq  
            END IF  
            IF current_sum EXISTS AS A KEY IN cumulative_sum_freq  
                INCREMENT the value corresponding to the key current_sum in cumulative_sum_freq BY one  
            ELSE  
                SET the value corresponding to the key current_sum in cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
