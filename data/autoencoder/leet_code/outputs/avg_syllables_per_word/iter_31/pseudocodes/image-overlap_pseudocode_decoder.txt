CLASS Solution  
    FUNCTION largestOverlap(img1 parameter, img2 parameter)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img1 AND element at position j of that element EQUALS one  
                    APPEND pair of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img2 AND element at position j of that element EQUALS one  
                    APPEND pair of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a default dictionary with default integer zero  
        FOR each pair x1 AND y1 IN points1  
            FOR each pair x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key pair of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        IF overlap_count IS NOT empty  
            SET max_value TO the maximum value among all values in overlap_count  
        ELSE  
            SET max_value TO zero  
        END IF  
        RETURN max_value  
    END FUNCTION  
END CLASS
