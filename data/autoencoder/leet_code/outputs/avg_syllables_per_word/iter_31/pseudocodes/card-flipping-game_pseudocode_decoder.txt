CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO the set of each number x where for each pair of numbers x and y taken at the same position from fronts and backs the condition that x EQUALS y holds  
        SET all_numbers TO the set of all elements from fronts and all elements from backs  
        SET good_numbers TO all_numbers WITHOUT any element that is in same  
        IF the set good_numbers IS NOT empty  
            SET result TO the minimum element in good_numbers  
        ELSE  
            SET result TO zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
