CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO an empty mapping from characters to integers  
        SET guess_count TO an empty mapping from characters to integers  
        
        FOR each pair of elements s and g taken simultaneously from secret and guess  
            IF s EQUALS g THEN  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the count of s in secret_count BY one  
                INCREMENT the count of g in guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN the keys of secret_count  
            INCREMENT cows BY the smaller value BETWEEN the count of digit in secret_count AND the count of digit in guess_count  
        END FOR  
        
        SET result TO the concatenation of the textual form of bulls, the character A, the textual form of cows, and the character B  
        RETURN result  
    END FUNCTION  
END CLASS
