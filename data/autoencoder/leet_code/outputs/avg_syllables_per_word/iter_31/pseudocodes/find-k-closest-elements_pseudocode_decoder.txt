CLASS Solution  
    FUNCTION findClosestElements(arr k x)  
        SET left TO result of bisect_left_position_for x in arr MINUS one  
        SET right TO left PLUS one  
        WHILE the difference between right and left MINUS one IS LESS THAN k  
            IF left EQUALS minus one OR the position right IS LESS THAN the length of arr AND the absolute value of the element at position right of arr MINUS x IS LESS THAN the absolute value of the element at position left of arr MINUS x  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        RETURN the elements of arr from position left PLUS one TO position right MINUS one including both positions  
    END FUNCTION  
END CLASS
