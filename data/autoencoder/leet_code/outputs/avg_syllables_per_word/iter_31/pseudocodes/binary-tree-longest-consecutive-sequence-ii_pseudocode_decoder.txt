CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments the left child of p and the left child of q AND the result of is_same_tree with arguments the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN zero, zero  
            END IF  
            SET inc_len TO one  
            SET dec_len TO one  
            SET left_inc, left_dec TO the result of dfs with argument the left child of node  
            SET right_inc, right_dec TO the result of dfs with argument the right child of node  
            IF the left child of node IS NOT none  
                IF the value of node EQUALS the value of the left child of node PLUS one  
                    SET dec_len TO left_dec PLUS one  
                ELSE IF the value of node EQUALS the value of the left child of node MINUS one  
                    SET inc_len TO left_inc PLUS one  
                END IF  
            END IF  
            IF the right child of node IS NOT none  
                IF the value of node EQUALS the value of the right child of node PLUS one  
                    SET dec_len TO the maximum of dec_len AND right_dec PLUS one  
                ELSE IF the value of node EQUALS the value of the right child of node MINUS one  
                    SET inc_len TO the maximum of inc_len AND right_inc PLUS one  
                END IF  
            END IF  
            SET max_len TO the maximum of max_len AND inc_len PLUS dec_len MINUS one  
            RETURN inc_len, dec_len  
        END FUNCTION  
        SET max_len TO zero  
        CALL dfs with argument root  
        RETURN max_len  
    END FUNCTION  
END CLASS
