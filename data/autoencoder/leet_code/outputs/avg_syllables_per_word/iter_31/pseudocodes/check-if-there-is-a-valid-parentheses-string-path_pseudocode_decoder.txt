CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF the SUM of m AND n MINUS one MODULO two NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET dp TO a three dimensional list of dimensions m PLUS one BY n PLUS one BY m PLUS n PLUS one WHERE each element IS False  
        SET element at position zero OF element at position one OF element at position zero OF dp TO True  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF element at position j PLUS one OF element at position i OF dp IS True OR element at position j OF element at position i PLUS one OF dp IS True THEN  
                        IF element at position j OF element at position i OF grid EQUALS open parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one THEN  
                            SET element at position j PLUS one OF element at position i PLUS one OF element at position k PLUS one OF dp TO True  
                        ELSE IF element at position j OF element at position i OF grid EQUALS close parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero THEN  
                            SET element at position j PLUS one OF element at position i PLUS one OF element at position k MINUS one OF dp TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position n OF element at position m OF element at position zero OF dp  
    END FUNCTION  
END CLASS
