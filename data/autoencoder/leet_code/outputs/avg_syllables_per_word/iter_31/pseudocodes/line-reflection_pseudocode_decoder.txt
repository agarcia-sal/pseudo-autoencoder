CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of tuples corresponding to each point in points  
        SET min_x TO the minimum value of the first element of each point in unique_points  
        SET max_x TO the maximum value of the first element of each point in unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each pair of values x and y in unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple consisting of reflected_x and y is NOT in unique_points  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
