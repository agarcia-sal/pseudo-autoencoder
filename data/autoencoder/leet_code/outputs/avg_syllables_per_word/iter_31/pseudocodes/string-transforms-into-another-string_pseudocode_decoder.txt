CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each pair char1 char2 IN pairs formed by corresponding elements of str1 and str2  
            IF char1 IS A KEY IN conversion_map  
                IF value at key char1 in conversion_map DOES NOT EQUAL char2  
                    RETURN False  
                END IF  
            ELSE  
                SET conversion_map at key char1 TO char2  
            END IF  
        END FOR  
        IF the LENGTH OF the SET OF all unique elements of str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
