CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the list of cuts appended with zero and n sorted in ascending order  
        
        FUNCTION dp(left, right)  
            IF the difference between right and left is less than or equal to one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity value  
            FOR each index i FROM one more than left TO one less than right DO  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS dp called with left and i PLUS dp called with i and right  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp called with zero and the length of cuts MINUS one  
    END FUNCTION  
END CLASS
