CLASS Solution  
    FUNCTION stoneGameIII(stoneValue)  
        SET n TO the LENGTH OF stoneValue  
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        FOR i FROM n MINUS one TO zero INCLUSIVE DECREASING BY one  
            SET dp at position i TO the element at position i of stoneValue MINUS the element at position i PLUS one of dp  
            IF i PLUS two LESS THAN OR EQUAL TO n THEN  
                IF dp at position i GREATER THAN OR EQUAL TO the element at position i PLUS the element at position i PLUS one of stoneValue MINUS the element at position i PLUS two of dp THEN  
                    LEAVE THE CURRENT VALUE OF dp at position i AS IS  
                ELSE  
                    SET dp at position i TO the element at position i PLUS the element at position i PLUS one of stoneValue MINUS the element at position i PLUS two of dp  
                END IF  
            END IF  
            IF i PLUS three LESS THAN OR EQUAL TO n THEN  
                IF dp at position i GREATER THAN OR EQUAL TO the element at position i PLUS the element at position i PLUS one PLUS the element at position i PLUS two of stoneValue MINUS the element at position i PLUS three of dp THEN  
                    LEAVE THE CURRENT VALUE OF dp at position i AS IS  
                ELSE  
                    SET dp at position i TO the element at position i PLUS the element at position i PLUS one PLUS the element at position i PLUS two of stoneValue MINUS the element at position i PLUS three of dp  
                END IF  
            END IF  
        END FOR  
        IF the element at position zero of dp GREATER THAN zero THEN  
            RETURN Alice  
        ELSE IF the element at position zero of dp LESS THAN zero THEN  
            RETURN Bob  
        ELSE  
            RETURN Tie  
        END IF  
    END FUNCTION  
END CLASS
