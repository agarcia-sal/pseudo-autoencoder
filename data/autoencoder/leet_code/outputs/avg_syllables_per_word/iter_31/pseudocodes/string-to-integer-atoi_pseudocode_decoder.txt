CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two RAISED TO THE power of thirty ONE MINUS one  
        SET INT_MIN TO MINUS two RAISED TO THE power of thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index LESS THAN n AND the element at position index of s EQUALS the space character  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (the element at position index of s EQUALS the plus character OR the element at position index of s EQUALS the minus character)  
            IF the element at position index of s EQUALS the minus character  
                SET sign TO MINUS one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND the element at position index of s IS A DIGIT  
            SET digit TO the numerical value of the element at position index of s  
            IF result GREATER THAN INT_MAX MINUS digit DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
