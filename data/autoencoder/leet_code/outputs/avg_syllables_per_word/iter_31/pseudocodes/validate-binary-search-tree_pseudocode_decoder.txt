CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values with no left and no right  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue has elements  
        SET node TO element removed from left end of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS no value THEN  
            SET node.left TO new TreeNode with value equal to element at position i of values with no left and no right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS no value THEN  
            SET node.right TO new TreeNode with value equal to element at position i of values with no left and no right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p NOT EXISTS AND q NOT EXISTS THEN  
        RETURN true  
    ELSE IF p NOT EXISTS OR q NOT EXISTS THEN  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low equal to negative infinity, high equal to positive infinity)  
            IF node NOT EXISTS THEN  
                RETURN true  
            END IF  
            IF value of node LESS THAN OR EQUAL TO low OR value of node GREATER THAN OR EQUAL TO high THEN  
                RETURN false  
            END IF  
            RETURN validate(left child of node, low, value of node) AND validate(right child of node, value of node, high)  
        END FUNCTION  
        RETURN validate(root)  
    END FUNCTION  
END CLASS
