CLASS Solution  
    FUNCTION escapeGhosts(ghosts, target)  
        SET player_distance TO absolute value of element at position one of target PLUS absolute value of element at position two of target  
        FOR each ghost IN ghosts  
            SET ghost_distance TO absolute value of the difference between element at position one of ghost AND element at position one of target PLUS absolute value of the difference between element at position two of ghost AND element at position two of target  
            IF ghost_distance LESS THAN OR EQUAL TO player_distance  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
