CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the ceiling of pile DIVIDED BY k  
            END FOR  
            RETURN whether hours IS LESS THAN OR EQUAL TO h  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the maximum element IN piles  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the integer division result of left PLUS right DIVIDED BY two  
            IF canFinish(mid)  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
