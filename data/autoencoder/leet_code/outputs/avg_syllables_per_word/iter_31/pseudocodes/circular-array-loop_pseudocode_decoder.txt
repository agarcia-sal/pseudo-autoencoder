CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1 val2)  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero OR val1 LESS THAN zero AND val2 LESS THAN zero  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET n TO the LENGTH OF nums  

        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  

            IF element at position i OF nums EQUALS zero  
                CONTINUE TO next iteration of the loop  
            END IF  

            WHILE True  
                SET slow TO the result of element at position slow OF nums PLUS slow  
                SET slow TO slow MODULO n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  

                SET fast TO the result of element at position fast OF nums PLUS fast  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                SET fast TO the result of element at position fast OF nums PLUS fast  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                IF sameDirection(element at position slow OF nums element at position i OF nums) EQUALS False OR sameDirection(element at position fast OF nums element at position i OF nums) EQUALS False  
                    BREAK  
                END IF  

                IF slow EQUALS fast  
                    IF slow EQUALS the result of element at position slow OF nums PLUS slow MODULO n  
                        BREAK  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
