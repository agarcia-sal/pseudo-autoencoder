CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR THE LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE THE LENGTH OF queue GREATER THAN zero  
        SET node TO THE FIRST element REMOVED FROM queue  
        IF i LESS THAN THE LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder EQUALS none OR THE LENGTH OF inorder EQUALS zero OR postorder EQUALS none OR THE LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        SET root_val TO THE LAST element REMOVED FROM postorder  
        SET root TO A NEW TreeNode WITH root_val  
        SET root_index TO THE POSITION OF root_val IN inorder  
        SET root.right TO self.buildTree(the sublist of inorder FROM root_index PLUS one TO THE END, postorder)  
        SET root.left TO self.buildTree(the sublist of inorder FROM THE START TO root_index MINUS one PLUS one, postorder)  
        RETURN root  
    END FUNCTION  
END CLASS
