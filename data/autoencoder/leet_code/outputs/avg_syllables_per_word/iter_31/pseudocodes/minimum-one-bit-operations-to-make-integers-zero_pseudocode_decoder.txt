CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        
        SET k TO zero  
        WHILE the number one MULTIPLIED BY two RAISED TO the power of k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        
        SET most_significant_bit TO the number one MULTIPLIED BY two RAISED TO the power of k  
        
        RETURN the number one MULTIPLIED BY two RAISED TO the power of k PLUS one MINUS one MINUS the result of calling minimumOneBitOperations WITH the value of n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
