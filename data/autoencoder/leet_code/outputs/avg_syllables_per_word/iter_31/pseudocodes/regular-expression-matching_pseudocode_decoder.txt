CLASS Solution  
    FUNCTION isMatch(s string p string) RETURNS boolean  
        SET dp TO a two dimensional list of boolean values with the number of rows EQUALS the length of s PLUS one and the number of columns EQUALS the length of p PLUS one with all values SET TO false  
        SET element at position zero of list at position zero of dp TO true  
        FOR j FROM one TO the length of p  
            IF the element at position j MINUS one of string p EQUALS the character star  
                SET element at position zero of list at position j of dp TO the value of element at position zero of list at position j MINUS two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j MINUS one of string p EQUALS the element at position i MINUS one of string s OR the element at position j MINUS one of string p EQUALS the character dot  
                    SET element at position i of list at position j of dp TO the value of element at position i MINUS one of list at position j MINUS one of dp  
                ELSE IF the element at position j MINUS one of string p EQUALS the character star  
                    SET element at position i of list at position j of dp TO the value of element at position i of list at position j MINUS two of dp  
                    IF the element at position j MINUS two of string p EQUALS the element at position i MINUS one of string s OR the element at position j MINUS two of string p EQUALS the character dot  
                        SET element at position i of list at position j of dp TO the value of element at position i of list at position j of dp OR element at position i MINUS one of list at position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the value of element at position the length of s of list at position the length of p of dp  
    END FUNCTION  
END CLASS
