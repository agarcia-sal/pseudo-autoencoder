CLASS Solution  
    FUNCTION movesToStamp(stamp string TO stamp, target string TO target) RETURNS list of integers  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list containing m REPEATED the DIFFERENCE BETWEEN n AND m PLUS one TIMES  
        SET q TO an empty double ended queue  
        SET g TO a list containing n ELEMENTS each INITIALIZED TO an empty list  
        FOR i FROM zero TO the DIFFERENCE BETWEEN n AND m  
            FOR each index j AND character c IN the enumeration of stamp  
                IF the element at position i PLUS j OF target EQUALS c THEN  
                    DECREMENT the element at position i OF indeg BY one  
                    IF the element at position i OF indeg EQUALS zero THEN  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the element at position i PLUS j OF g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list containing n ELEMENTS each INITIALIZED TO False  
        WHILE q is not empty  
            SET i TO the leftmost element REMOVED FROM q  
            APPEND i TO ans  
            FOR j FROM zero TO m MINUS one  
                IF the element at position i PLUS j OF vis EQUALS False THEN  
                    SET the element at position i PLUS j OF vis TO True  
                    FOR each element k IN the element at position i PLUS j OF g  
                        DECREMENT the element at position k OF indeg BY one  
                        IF the element at position k OF indeg EQUALS zero THEN  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements of vis EQUAL True THEN  
            RETURN ans REVERSED in order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
