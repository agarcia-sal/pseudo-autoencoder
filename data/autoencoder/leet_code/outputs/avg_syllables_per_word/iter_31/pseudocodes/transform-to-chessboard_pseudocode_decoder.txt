CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i and j of board XOR the element at position i and zero of board XOR the element at position zero and j of board XOR the element at position zero and zero of board NOT EQUALS zero  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the element at position i and zero of board  
            INCREMENT colSum BY the element at position zero and i of board  
            INCREMENT rowSwap BY one IF the element at position i and zero of board EQUALS the remainder of i DIVIDED BY two OTHERWISE zero  
            INCREMENT colSwap BY one IF the element at position zero and i of board EQUALS the remainder of i DIVIDED BY two OTHERWISE zero  
        END FOR  
        IF the integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of the sum of N AND one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN the integer division of the sum of N AND one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two NOT EQUALS zero  
            IF the remainder of rowSwap DIVIDED BY two NOT EQUALS zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two NOT EQUALS zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum VALUE BETWEEN rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum VALUE BETWEEN colSwap AND N MINUS colSwap  
        END IF  
        RETURN the integer division of the sum of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
