CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS an empty collection  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and left equal to None and right equal to None  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node EQUALS None  
                RETURN True AND zero AND positive infinity AND negative infinity  
            END IF  
            SET left_is_bst TO the first returned value of calling helper with the left child of node  
            SET left_sum TO the second returned value of calling helper with the left child of node  
            SET left_min TO the third returned value of calling helper with the left child of node  
            SET left_max TO the fourth returned value of calling helper with the left child of node  
            SET right_is_bst TO the first returned value of calling helper with the right child of node  
            SET right_sum TO the second returned value of calling helper with the right child of node  
            SET right_min TO the third returned value of calling helper with the right child of node  
            SET right_max TO the fourth returned value of calling helper with the right child of node  
            IF left_is_bst EQUALS True AND right_is_bst EQUALS True AND left_max LESS THAN the value of node AND the value of node LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS the value of node  
                SET current_min TO the smaller value of the value of node AND left_min  
                SET current_max TO the larger value of the value of node AND right_max  
                SET self.max_sum TO the larger value of self.max_sum AND current_sum  
                RETURN True AND current_sum AND current_min AND current_max  
            END IF  
            RETURN False AND zero AND positive infinity AND negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
