CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        SET character count dictionary TO a new dictionary with default integer zero  
        SET left pointer TO zero  
        SET maximum length TO zero  
        FOR right pointer FROM zero TO the length of s MINUS one  
            INCREMENT the value in character count dictionary corresponding to element at position right pointer of s BY one  
            WHILE the number of keys in character count dictionary IS GREATER THAN two  
                DECREMENT the value in character count dictionary corresponding to element at position left pointer of s BY one  
                IF the value in character count dictionary corresponding to element at position left pointer of s EQUALS zero  
                    REMOVE the key corresponding to element at position left pointer of s FROM character count dictionary  
                END IF  
                INCREMENT left pointer BY one  
            END WHILE  
            SET maximum length TO the greater value BETWEEN maximum length AND right pointer MINUS left pointer PLUS one  
        END FOR  
        RETURN maximum length  
    END FUNCTION  
END CLASS
