CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET length of s1 TO len_s1  
        SET length of s2 TO len_s2  
        
        IF len_s1 GREATER THAN len_s2  
            RETURN False  
        END IF  
        
        SET count_s1 TO mapping of frequencies of each character in s1  
        SET count_s2 TO mapping of frequencies of each character in substring from start position zero TO position len_s1 MINUS one of s2  
        
        IF count_s1 EQUALS count_s2  
            RETURN True  
        END IF  
        
        FOR index i FROM len_s1 TO len_s2 MINUS one  
            INCREMENT count of character at position i of s2 BY one in count_s2  
            
            DECREMENT count of character at position i MINUS len_s1 of s2 BY one in count_s2  
            
            IF count of character at position i MINUS len_s1 of s2 EQUALS zero in count_s2  
                REMOVE the entry for character at position i MINUS len_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
