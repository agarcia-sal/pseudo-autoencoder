CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO the character plus  
            SET num TO zero  
            
            WHILE the LENGTH OF s GREATER THAN zero holds  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                
                IF char is a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                
                IF char EQUALS the character open parenthesis THEN  
                    SET num TO the result of calling helper with s  
                END IF  
                
                IF char is NOT a digit AND char is NOT a whitespace character OR the LENGTH OF s EQUALS zero THEN  
                    IF sign EQUALS the character plus THEN  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the character minus THEN  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the character asterisk THEN  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the character slash THEN  
                        SET the element at the last position of stack TO the integer division of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS the character close parenthesis THEN  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper with the list of characters of s  
    END FUNCTION  
END CLASS
