CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1 sentence1 AS List OF string sentence2 sentence2 AS List OF string similarPairs similarPairs AS List OF List OF string) RETURNS boolean  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN False  
        END IF  

        SET parent TO empty dictionary  

        FUNCTION find(x)  
            IF the element at key x of parent NOT EQUALS x  
                SET the element at key x of parent TO the result of invoking find WITH the element at key x of parent  
            END IF  
            RETURN the element at key x of parent  
        END FUNCTION  

        FUNCTION union(x y)  
            SET rootX TO the result of invoking find WITH x  
            SET rootY TO the result of invoking find WITH y  
            IF rootX NOT EQUALS rootY  
                SET the element at key rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN keys OF parent  
                    SET the element at key word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x y IN similarPairs  
            INVOKE union WITH x y  
        END FOR  

        FOR each w1 w2 IN parallel elements of sentence1 sentence2  
            IF w1 NOT IN keys OF parent  
                SET the element at key w1 of parent TO w1  
            END IF  
            IF w2 NOT IN keys OF parent  
                SET the element at key w2 of parent TO w2  
            END IF  
            IF the result of invoking find WITH w1 NOT EQUALS the result of invoking find WITH w2  
                RETURN False  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
