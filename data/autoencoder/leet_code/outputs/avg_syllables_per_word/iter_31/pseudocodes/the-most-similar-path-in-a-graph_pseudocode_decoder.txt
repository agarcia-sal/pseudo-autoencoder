CLASS Solution  
    FUNCTION mostSimilar(n, roads, names, targetPath)  
        SET graph TO a mapping of lists with default empty list  
        FOR each pair of elements u and v IN roads  
            APPEND element v TO the list at key u in graph  
            APPEND element u TO the list at key v in graph  
        END FOR  
        
        SET m TO the LENGTH OF targetPath  
        
        FUNCTION dp(i, j)  
            IF i EQUALS zero THEN  
                IF element at position j of names NOT EQUALS element at position i of targetPath THEN  
                    RETURN a pair consisting of one AND a list containing j  
                ELSE  
                    RETURN a pair consisting of zero AND a list containing j  
                END IF  
            END IF  
            
            SET min_cost TO positive infinity  
            SET min_path TO empty list  
            FOR each element k IN the list at key j in graph  
                SET cost_path_pair TO the result of calling dp with i MINUS one AND k  
                SET cost TO the first element of cost_path_pair  
                SET path TO the second element of cost_path_pair  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                    SET min_path TO path  
                END IF  
            END FOR  
            
            IF element at position j of names NOT EQUALS element at position i of targetPath THEN  
                SET current_cost TO min_cost PLUS one  
            ELSE  
                SET current_cost TO min_cost  
            END IF  
            
            RETURN a pair consisting of current_cost AND the concatenation of min_path AND a list containing j  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        SET min_path TO empty list  
        FOR index j FROM zero TO n MINUS one  
            SET cost_path_pair TO the result of calling dp with m MINUS one AND j  
            SET cost TO the first element of cost_path_pair  
            SET path TO the second element of cost_path_pair  
            IF cost LESS THAN min_cost THEN  
                SET min_cost TO cost  
                SET min_path TO path  
            END IF  
        END FOR  
        
        RETURN min_path  
    END FUNCTION  
END CLASS
