CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of length n with all elements SET TO negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position element at the last position of stack OF nums IS LESS THAN the element at position index OF nums  
                SET popped_index TO the removal of the last element FROM stack  
                SET the element at position popped_index OF result TO the element at position index OF nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
