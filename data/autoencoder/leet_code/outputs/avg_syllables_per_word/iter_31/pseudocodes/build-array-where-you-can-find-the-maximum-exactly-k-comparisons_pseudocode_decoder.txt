CLASS Solution  
    FUNCTION numOfArrays(n integer, m integer, k integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET dp TO a three dimensional list with dimensions n PLUS one by m PLUS one by k PLUS one initialized with zeros  
        FOR j FROM one TO m  
            SET element at position one of list at position j of list at position one of dp TO one  
        END FOR  
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position p of list at position j of list at position i of dp TO element at position p of list at position j of list at position i of dp PLUS element at position p MINUS one of list at position x of list at position i MINUS one of dp MODULO MOD  
                    END FOR  
                    SET element at position p of list at position j of list at position i of dp TO element at position p of list at position j of list at position i of dp PLUS element at position p of list at position j of list at position i MINUS one of dp MULTIPLIED BY j MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position k of list at position j of list at position n of dp MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
