CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN no value  
    END IF  
    SET root TO new instance of TreeNode with parameter val equal to element at position one of values and parameters left and right equal to no value  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.left TO new instance of TreeNode with parameter val equal to element at position i of values and parameters left and right equal to no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.right TO new instance of TreeNode with parameter val equal to element at position i of values and parameters left and right equal to no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value THEN  
        RETURN True  
    ELSE IF p EQUALS no value OR q EQUALS no value THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self, root)  
        IF root EQUALS no value THEN  
            RETURN no value  
        END IF  
        SET root.left TO the result of calling pruneTree of self with argument root.left  
        SET root.right TO the result of calling pruneTree of self with argument root.right  
        IF root.val EQUALS zero AND root.left EQUALS no value AND root.right EQUALS no value THEN  
            RETURN no value  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
