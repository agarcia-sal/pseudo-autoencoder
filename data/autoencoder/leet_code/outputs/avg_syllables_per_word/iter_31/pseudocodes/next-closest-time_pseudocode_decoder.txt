CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of all characters in time with all colon characters removed  
        SET hour AND minute TO the integer values obtained by splitting time at colon and converting the two parts respectively  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each first hour digit h1 IN digits  
                FOR each second hour digit h2 IN digits  
                    FOR each first minute digit m1 IN digits  
                        FOR each second minute digit m2 IN digits  
                            SET composed_hour TO the integer value obtained by concatenating h1 AND h2 in that order  
                            SET composed_minute TO the integer value obtained by concatenating m1 AND m2 in that order  
                            IF composed_hour LESS THAN twenty four AND composed_minute LESS THAN sixty THEN  
                                APPEND the pair composed_hour AND composed_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in increasing order  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the position of the pair hour AND minute in all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times THEN  
            SET next_hour AND next_minute TO the pair at the position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the first pair in all_times  
        END IF  
        
        RETURN a string representation of next_hour formatted as two digits followed by a colon followed by next_minute formatted as two digits  
    END FUNCTION  
END CLASS
