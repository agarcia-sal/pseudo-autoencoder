CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN the substring sub EQUALS the substring sub reversed  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two dimensional list of boolean FALSE values with dimensions n BY n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the element at position i of s EQUALS the element at position j of s AND the difference between j AND i IS LESS THAN OR EQUAL TO one OR the element at position i PLUS one AND position j MINUS one of is_pal IS TRUE  
                    SET the element at position i AND j of is_pal TO TRUE  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF the element at position zero AND position i MINUS one of is_pal IS TRUE  
                FOR j FROM i TO n MINUS two  
                    IF the element at position i AND j of is_pal IS TRUE AND the element at position j PLUS one AND position n MINUS one of is_pal IS TRUE  
                        RETURN TRUE  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN FALSE  
    END FUNCTION  
END CLASS
