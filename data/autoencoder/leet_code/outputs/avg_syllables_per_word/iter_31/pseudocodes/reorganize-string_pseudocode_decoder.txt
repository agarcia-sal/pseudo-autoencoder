CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO COUNT THE FREQUENCY OF EACH CHARACTER IN s  
        SET max_heap TO A LIST OF PAIRS WHERE EACH PAIR CONTAINS THE NEGATION OF FREQUENCY AND THE CHARACTER FROM count  
        CALL heapify ON max_heap TO ORGANIZE IT AS A HEAP  
        
        SET prev_char TO no value  
        SET prev_freq TO zero  
        SET result TO AN EMPTY LIST  
        
        WHILE max_heap IS NOT EMPTY OR prev_freq IS LESS THAN zero  
            IF max_heap IS EMPTY AND prev_freq IS LESS THAN zero  
                RETURN an empty string  
            END IF  
            
            SET freq AND char TO THE RESULT OF REMOVING THE SMALLEST ELEMENT FROM max_heap  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq IS LESS THAN zero  
                ADD THE PAIR OF prev_freq AND prev_char TO max_heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN THE CONCATENATION OF ALL ELEMENTS IN result  
    END FUNCTION  
END CLASS
