CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with parameter element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.left TO new TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.right TO new TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN True  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the left child of p and the left child of q AND the result of is_same_tree applied to the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS no value THEN  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO a double ended queue initialized with root as its single element  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element removed from the front of queue  
                IF i EQUALS level_length MINUS one THEN  
                    APPEND node.val TO right_side_view  
                END IF  
                IF node.left IS NOT no value THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT no value THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
