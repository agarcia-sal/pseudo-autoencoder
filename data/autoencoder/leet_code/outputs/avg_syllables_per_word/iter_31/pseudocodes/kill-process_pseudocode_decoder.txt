CLASS Solution  
    FUNCTION killProcess(pid parameter typed as list of integers ppid parameter typed as list of integers kill parameter typed as integer) RETURNING list of integers  
        SET tree TO a mapping from integer keys to empty list values  
        FOR each child PROCESS AND parent PROCESS IN simultaneous traversal of pid AND ppid  
            APPEND child PROCESS TO the list of children associated with parent PROCESS in tree  
        END FOR  
        SET queue TO a list containing only kill PROCESS  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE the first element FROM queue AND assign it TO current PROCESS  
            APPEND current PROCESS TO killed_processes  
            FOR each child PROCESS IN the list associated with current PROCESS in tree  
                APPEND child PROCESS TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
