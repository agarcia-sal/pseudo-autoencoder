CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the list of substrings obtained by splitting path by the slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS the empty string OR component EQUALS the dot character  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two consecutive dots  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the concatenation of the slash character AND the concatenation of elements of stack separated by the slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
