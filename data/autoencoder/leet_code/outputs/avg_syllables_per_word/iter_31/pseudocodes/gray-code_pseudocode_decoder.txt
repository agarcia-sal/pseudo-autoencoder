CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode with parameter n MINUS one  
        SET mask TO the number one MULTIPLIED BY two RAISED TO the power of n MINUS one  
        SET reversed_prev_gray TO the list prev_gray in reverse order  
        SET reflected_list TO an empty list  
        FOR each number IN reversed_prev_gray  
            SET reflected_number TO mask OR number  
            APPEND reflected_number TO reflected_list  
        END FOR  
        SET current_gray TO the concatenation of prev_gray FOLLOWED BY reflected_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
