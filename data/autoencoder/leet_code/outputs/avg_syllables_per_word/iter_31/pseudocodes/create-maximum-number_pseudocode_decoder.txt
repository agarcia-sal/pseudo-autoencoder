CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND stack IS NOT empty AND the last element of stack LESS THAN num  
                    REMOVE the last element FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  
        
        FUNCTION merge(arr1 arr2)  
            SET merged_list TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR iteration FROM one TO total_length  
                IF arr1 GREATER THAN arr2 when compared lexicographically THEN  
                    APPEND the first element of arr1 TO merged_list  
                    REMOVE the first element FROM arr1  
                ELSE  
                    APPEND the first element of arr2 TO merged_list  
                    REMOVE the first element FROM arr2  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2 THEN  
                SET candidate TO merge(maxSingleArray(nums1 i) maxSingleArray(nums2 k MINUS i))  
                IF max_number LESS THAN candidate WHEN COMPARED lexicographically THEN  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
