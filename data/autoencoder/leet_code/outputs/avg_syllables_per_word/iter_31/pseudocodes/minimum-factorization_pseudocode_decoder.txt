CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO empty list  
        FOR digit FROM nine TO two DECREMENT BY one  
            WHILE num MODULO digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        
        SET result_string TO concatenation of each element in factors converted to their natural language representation in order  
        
        SET int_result TO the integer value represented by result_string  
        
        IF int_result GREATER THAN two TO THE POWER OF thirty ONE MINUS one  
            RETURN zero  
        END IF  
        
        RETURN int_result  
    END FUNCTION  
END CLASS
