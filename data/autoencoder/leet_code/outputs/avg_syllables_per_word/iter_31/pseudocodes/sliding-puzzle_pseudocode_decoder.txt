CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples each constructed from each row in board  
        SET target TO tuple of tuples representing one two three and four five zero in two rows  
        SET neighbors TO a mapping from zero position to sets of adjacent positions as specified  
        
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO a new double ended queue initialized with tuple consisting of start zero_pos and zero  
        SET visited TO a new set initialized with start  
        
        WHILE queue is not empty  
            SET state zero_pos moves TO the first element removed from the left of queue  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN the collection associated with zero_pos in neighbors  
                SET new_state TO list of lists mapped from each tuple in state converted to list  
                SET i TO integer division of zero_pos by three  
                SET j TO remainder of zero_pos divided by three  
                SET ni TO integer division of neighbor by three  
                SET nj TO remainder of neighbor divided by three  
                
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state and  
                    SET element at position nj of element at position ni of new_state TO element at position j of element at position i of new_state  
                
                SET new_tuple_state TO tuple of tuples each constructed from each row in new_state  
                
                IF new_tuple_state is not in visited  
                    ADD new_tuple_state to visited  
                    APPEND tuple consisting of new_tuple_state neighbor moves PLUS one to queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
