CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS not present  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position zero of each interval  
        SET min_heap TO an empty list  
        FOR each meeting IN intervals  
            IF the LENGTH OF min_heap GREATER THAN zero AND the element at position zero of meeting GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element from min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap maintaining the min heap property  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
