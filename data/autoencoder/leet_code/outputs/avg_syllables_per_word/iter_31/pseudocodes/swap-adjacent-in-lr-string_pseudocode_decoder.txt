CLASS Solution  
    FUNCTION canTransform(start AS string, end AS string) RETURNS boolean  
        SET start_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF start MINUS one  
            SET character TO element at position index of start  
            IF character NOT EQUALS the character X  
                APPEND the pair of character and index TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF end MINUS one  
            SET character TO element at position index of end  
            IF character NOT EQUALS the character X  
                APPEND the pair of character and index TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair_of_character_and_index1 IN start_filtered AND each pair_of_character_and_index2 IN end_filtered IN PAIRS  
            SET character1 TO the first element of pair_of_character_and_index1  
            SET index1 TO the second element of pair_of_character_and_index1  
            SET character2 TO the first element of pair_of_character_and_index2  
            SET index2 TO the second element of pair_of_character_and_index2  
            
            IF character1 NOT EQUALS character2  
                RETURN false  
            END IF  
            
            IF character1 EQUALS the character L AND index1 LESS THAN index2  
                RETURN false  
            END IF  
            
            IF character1 EQUALS the character R AND index1 GREATER THAN index2  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
