CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a list containing n empty lists  
        FOR each element x y time IN edges  
            APPEND a pair consisting of y and time TO element at position x of graph  
            APPEND a pair consisting of x and time TO element at position y of graph  
        END FOR  

        SET pq TO a list containing a triple consisting of element at position zero of passingFees and zero and zero  
        SET visited_time TO a list containing n elements each initialized to infinite value  
        SET element at position zero of visited_time TO zero  

        WHILE pq is not empty  
            REMOVE the smallest element from pq and ASSIGN the three components TO current_cost current_city current_time  

            IF current_city EQUALS the LENGTH OF passingFees MINUS one  
                RETURN current_cost  
            END IF  

            FOR each pair neighbor travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  

                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET element at position neighbor of visited_time TO new_time  
                    ADD a triple consisting of new_cost neighbor new_time TO pq using heap push operation  
                END IF  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
