CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR index FROM zero TO the LENGTH OF row MINUS one  
                    SET element at position index of row TO one MINUS element at position index of row  
                END FOR  
            END IF  
        END FOR  
        FOR column_index FROM one TO the LENGTH OF element at position zero of grid MINUS one  
            SET count_of_ones TO zero  
            FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_of_ones BY element at position column_index of element at position row_index of grid  
            END FOR  
            IF count_of_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                    SET element at position column_index of element at position row_index of grid TO one MINUS element at position column_index of element at position row_index of grid  
                END FOR  
            END IF  
        END FOR  
        SET total_score TO zero  
        FOR each row IN grid  
            SET binary_representation TO an empty string  
            FOR each element IN row  
                APPEND the string representation of element TO binary_representation  
            END FOR  
            SET integer_value TO the integer value of binary_representation interpreted in base two  
            INCREMENT total_score BY integer_value  
        END FOR  
        RETURN total_score  
    END FUNCTION  
END CLASS
