CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair u AND v IN edges  
            APPEND v TO the list associated with u in graph  
            APPEND u TO the list associated with v in graph  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path FROM position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list associated with node in graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO the result of dfs with neighbor parent node path visited  
                    IF cycle IS NOT NONE THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN None  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO None  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO the result of dfs with node minus one an empty list visited  
                IF cycle IS NOT None THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zeros with length n  
        SET queue TO a double-ended queue initialized with the elements of cycle  
        SET visited TO a set initialized with the elements of cycle  
        
        SET level TO zero  
        WHILE queue is not empty  
            FOR counter FROM zero TO the length of queue MINUS one  
                REMOVE the element from the front of queue and ASSIGN it TO node  
                SET the element of distance at position node TO level  
                FOR each neighbor IN the list associated with node in graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
