CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN the elements of wordList THEN  
            RETURN zero  
        END IF  
      
        SET word_dict TO an empty dictionary that has lists as default values  
        SET word_length TO the LENGTH OF beginWord  
      
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of the substring from position zero TO position i MINUS one of word PLUS the literal word star PLUS the substring from position i PLUS one TO the last position of word  
                APPEND word TO the list at key intermediate_key in word_dict  
            END FOR  
        END FOR  
      
        SET queue TO a double ended queue initialized with the pair consisting of beginWord and the number one  
        SET visited TO a set initialized with the element beginWord  
      
        WHILE the queue is not empty DO  
            SET current_word TO the first element of the first pair in queue  
            SET current_length TO the second element of the first pair in queue  
            REMOVE the first pair FROM queue  
          
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of the substring from position zero TO position i MINUS one of current_word PLUS the literal word star PLUS the substring from position i PLUS one TO the last position of current_word  
              
                FOR each word IN the list obtained from word_dict at key intermediate_word  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the pair consisting of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
      
        RETURN zero  
    END FUNCTION  
END CLASS
