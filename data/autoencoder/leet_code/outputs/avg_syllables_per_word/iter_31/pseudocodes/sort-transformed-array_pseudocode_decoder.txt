CLASS Solution  
    FUNCTION sortTransformedArray(nums parameter, a parameter, b parameter, c parameter) RETURNS list  
        FUNCTION quadratic(x parameter) RETURNS number  
            RETURN the value of a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        SET result TO a new list of zeros with the LENGTH OF n  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a GREATER THAN OR EQUAL TO zero THEN  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                IF the result of quadratic applied to element at position left of nums GREATER THAN the result of quadratic applied to element at position right of nums THEN  
                    SET the element at position index of result TO the result of quadratic applied to element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET the element at position index of result TO the result of quadratic applied to element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right  
                IF the result of quadratic applied to element at position left of nums LESS THAN the result of quadratic applied to element at position right of nums THEN  
                    SET the element at position index of result TO the result of quadratic applied to element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET the element at position index of result TO the result of quadratic applied to element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
