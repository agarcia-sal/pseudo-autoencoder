CLASS Solution  
    FUNCTION multiply WITH PARAMETERS mat1 mat2  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position one of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position one of mat2  
        SET result TO a new list containing m elements each of which is a list containing n elements all set to zero  
        FOR each i FROM one TO m  
            FOR each j FROM one TO n  
                FOR each p FROM one TO k  
                    SET the element at position j of the element at position i of result TO the element at position j of the element at position i of result PLUS the element at position p of the element at position i of mat1 MULTIPLIED BY the element at position j of the element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
