CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO an empty mapping from keys to lists  
        FOR each pair composed of departure and arrival IN the reverse order of the sorted tickets  
            APPEND arrival TO the list associated with the departure in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph is not empty  
                SET next_airport TO the element at the last position of the list associated with airport in graph  
                REMOVE the element at the last position from the list associated with airport in graph  
                CALL visit with next_airport  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit with the literal airport named J F K  
        RETURN the elements of itinerary in reverse order  
    END FUNCTION  
END CLASS
