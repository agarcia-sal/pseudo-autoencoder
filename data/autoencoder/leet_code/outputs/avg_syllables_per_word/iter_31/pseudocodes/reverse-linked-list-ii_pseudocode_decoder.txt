CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO new ListNode with val as element at position zero of values and next as None  
    SET p TO head  
    FOR each val IN values starting from position one TO the end  
        SET node TO new ListNode with val as val and next as None  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val of p1 EQUALS the val of p2 AND is_same_list(the next of p1, the next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween(self, head, left, right)  
        IF head EQUALS None OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with val as zero and next as None  
        SET the next of dummy TO head  
        SET prev TO dummy  
        FOR an unused variable FROM zero TO left MINUS one MINUS one  
            SET prev TO the next of prev  
        END FOR  
        SET current TO the next of prev  
        FOR an unused variable FROM zero TO right MINUS left MINUS one  
            SET next_node TO the next of current  
            SET the next of current TO the next of next_node  
            SET the next of next_node TO the next of prev  
            SET the next of prev TO next_node  
        END FOR  
        RETURN the next of dummy  
    END FUNCTION  
END CLASS
