CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO a new mapping where each key maps to an empty set  
        SET indegree TO a new mapping where each key maps to zero  
        SET all_chars TO the set of all individual characters contained in all words combined  
        
        FOR index FROM zero TO the LENGTH OF words MINUS one MINUS one  
            SET word1 TO the element at position index of words  
            SET word2 TO the element at position index PLUS one of words  
            SET min_length TO the smaller value between the LENGTH OF word1 AND the LENGTH OF word2  
            SET found TO false  
            FOR position FROM zero TO min_length MINUS one  
                IF the character at position of word1 NOT EQUALS the character at position of word2  
                    IF the character at position of word2 is NOT CONTAINED IN the set mapped by the character at position of word1 in graph  
                        ADD the character at position of word2 TO the set mapped by the character at position of word1 in graph  
                        INCREMENT the value mapped by the character at position of word2 in indegree BY one  
                    END IF  
                    SET found TO true  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF found EQUALS false AND the LENGTH OF word1 GREATER THAN the LENGTH OF word2  
                RETURN an empty string  
            END IF  
        END FOR  
        
        SET queue TO a new double-ended collection initialized with all elements from all_chars for which the value mapped by the element in indegree EQUALS zero  
        SET result TO an empty list  
        
        WHILE queue is not empty  
            REMOVE the element from the front of queue AND ASSIGN it TO char  
            APPEND char TO result  
            FOR each neighbor IN the set mapped by char in graph  
                DECREMENT the value mapped by neighbor in indegree BY one  
                IF the value mapped by neighbor in indegree EQUALS zero  
                    ADD neighbor TO the end of queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF result NOT EQUALS the LENGTH OF all_chars  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters in result in order  
    END FUNCTION  
END CLASS
