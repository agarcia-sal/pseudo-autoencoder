CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        CREATE dp AS a two-dimensional list of zeros with the first dimension length m PLUS one and the second dimension length n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of str1 EQUALS element at position j MINUS one of str2  
                    SET element at position i and j of dp TO element at position i MINUS one and j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i and j of dp TO the GREATER OF element at position i MINUS one and j of dp AND element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET i TO m  
        SET j TO n  
        CREATE scs AS an empty list  
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF element at position i MINUS one of str1 EQUALS element at position j MINUS one of str2  
                APPEND element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF element at position i MINUS one and j of dp GREATER THAN element at position i and j MINUS one of dp  
                APPEND element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        WHILE i GREATER THAN zero  
            APPEND element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE j GREATER THAN zero  
            APPEND element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        REVERSE the order of elements in scs  
        RETURN the concatenation of all elements in scs into a string  
    END FUNCTION  
END CLASS
