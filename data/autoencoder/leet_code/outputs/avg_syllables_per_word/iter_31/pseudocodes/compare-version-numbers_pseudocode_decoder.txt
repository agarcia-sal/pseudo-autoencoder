CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO the list of substrings obtained by splitting version1 by the period character  
        SET v2_revisions TO the list of substrings obtained by splitting version2 by the period character  
        SET max_length TO the greater of the length of v1_revisions AND the length of v2_revisions  
        FOR i FROM zero TO max_length MINUS one  
            IF i LESS THAN the length of v1_revisions THEN  
                SET v1 TO the integer conversion of the element at position i of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF i LESS THAN the length of v2_revisions THEN  
                SET v2 TO the integer conversion of the element at position i of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2 THEN  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2 THEN  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
