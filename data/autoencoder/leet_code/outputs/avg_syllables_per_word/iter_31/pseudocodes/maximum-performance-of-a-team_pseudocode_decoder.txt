CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO empty list  
        FOR index FROM zero TO n MINUS one  
            APPEND pair of element at position index of efficiency AND element at position index of speed TO engineers  
        END FOR  
        SORT engineers IN ORDER FROM highest efficiency TO lowest efficiency  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff spd IN engineers  
            ADD spd TO speed_heap maintaining heap property  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap GREATER THAN k THEN  
                SET smallest_speed TO REMOVE smallest element FROM speed_heap maintaining heap property  
                DECREMENT speed_sum BY smallest_speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance MODULO one billion PLUS seven  
    END FUNCTION  
END CLASS
