CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN the LENGTH OF nums  
        END IF  
        SET up TO one  
        SET down TO one  
        FOR each index i FROM one TO LENGTH OF nums MINUS one  
            IF element at position i of nums GREATER THAN element at position i MINUS one of nums THEN  
                SET up TO down PLUS one  
            ELSE IF element at position i of nums LESS THAN element at position i MINUS one of nums THEN  
                SET down TO up PLUS one  
            END IF  
        END FOR  
        RETURN the GREATER VALUE BETWEEN up AND down  
    END FUNCTION  
END CLASS
