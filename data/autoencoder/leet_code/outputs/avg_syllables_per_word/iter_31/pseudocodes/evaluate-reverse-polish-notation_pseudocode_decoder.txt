CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS the character for addition OR token EQUALS the character for subtraction OR token EQUALS the character for multiplication OR token EQUALS the character for division  
                SET b TO the element at the last position removed from stack  
                SET a TO the element at the last position removed from stack  
                IF token EQUALS the character for addition  
                    APPEND the value of a PLUS b TO stack  
                ELSE IF token EQUALS the character for subtraction  
                    APPEND the value of a MINUS b TO stack  
                ELSE IF token EQUALS the character for multiplication  
                    APPEND the value of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS the character for division  
                    APPEND the value of the integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer value represented by token TO stack  
            END IF  
        END FOR  
        RETURN the element at position zero of stack  
    END FUNCTION  
END CLASS
