CLASS Solution  
    FUNCTION isNStraightHand(hand PARAMETER, groupSize PARAMETER) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        SET count TO the frequency count of each element in hand  
        SET unique_cards TO the sequence of unique elements in count arranged in ascending order  
        FOR each card IN unique_cards  
            IF the frequency of card in count GREATER THAN zero THEN  
                SET needed TO the frequency of card in count  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the frequency of the element at position card PLUS i in count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    SET the frequency of the element at position card PLUS i in count TO the frequency of the element at position card PLUS i in count MINUS needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
