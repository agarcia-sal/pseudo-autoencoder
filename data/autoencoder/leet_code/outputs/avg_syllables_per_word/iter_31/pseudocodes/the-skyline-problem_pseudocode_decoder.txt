CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO first element of building  
            SET right TO second element of building  
            SET height TO third element of building  
            APPEND a tuple of left AND negative height AND right TO events  
            APPEND a tuple of right AND zero AND zero TO events  
        END FOR  
        
        SORT events by the first element then by the second element then by the third element  
        
        SET result TO list containing a list of zero AND zero  
        SET max_heap TO list containing a tuple of zero AND positive infinity  
        
        FOR each event IN events  
            SET x TO the first element of event  
            SET negH TO the second element of event  
            SET r TO the third element of event  
            
            WHILE the right edge of the first tuple in max_heap is less than or equal to x holds  
                REMOVE the first element from max_heap using heap pop operation  
            END WHILE  
            
            IF negH does not equal zero THEN  
                ADD the tuple of negH AND r TO max_heap using heap push operation  
            END IF  
            
            SET current_height TO negative of the first element of the first tuple in max_heap  
            SET last_height TO the second element of the last list in result  
            IF last_height does not equal current_height THEN  
                APPEND a list of x AND current_height TO result  
            END IF  
        END FOR  
        
        IF the first list in result equals a list of zero AND zero THEN  
            REMOVE the first list from result  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
