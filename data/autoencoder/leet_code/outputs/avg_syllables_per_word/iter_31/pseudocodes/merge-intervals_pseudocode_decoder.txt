CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals IN ORDER OF element at position zero of interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR element at position one of element at position last of merged IS LESS THAN element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one of element at position last of merged TO THE GREATER OF element at position one of element at position last of merged AND element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
