CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET new_character_index TO the character code of element at position index of s MINUS the character code of letter a PLUS element at position index of shifts  
            SET wrapped_character_index TO new_character_index MODULO twenty six  
            SET new_character TO the character corresponding to wrapped_character_index PLUS the character code of letter a  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result into a single string  
    END FUNCTION  
END CLASS
