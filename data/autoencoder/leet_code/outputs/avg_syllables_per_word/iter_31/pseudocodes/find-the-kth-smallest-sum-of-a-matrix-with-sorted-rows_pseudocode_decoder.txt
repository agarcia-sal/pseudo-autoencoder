CLASS Solution  
    FUNCTION kthSmallest(mat AS list of list of integer k AS integer) RETURNS integer  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET initial_sum TO zero  
        FOR each row IN mat  
            INCREMENT initial_sum BY the element at position zero of row  
        END FOR  
        SET initial_indices TO a tuple containing zero repeated m times  
        SET min_heap TO a list containing a tuple of initial_sum and initial_indices  
        SET visited TO a set containing initial_indices  
        
        WHILE k GREATER THAN zero  
            SET current_tuple TO the removal of the smallest element from min_heap  
            SET current_sum TO the first element of current_tuple  
            SET indices TO the second element of current_tuple  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF the element at position i of indices PLUS one LESS THAN n  
                    SET new_indices TO a list copy of indices  
                    INCREMENT the element at position i of new_indices BY one  
                    SET new_indices_tuple TO a tuple of new_indices  
                    SET new_sum TO current_sum MINUS the element at position i of mat AT position of the element at position i of indices PLUS the element at position i of mat AT position of the element at position i of new_indices  
                    
                    IF new_indices_tuple NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        ADD the tuple of new_sum and new_indices_tuple TO min_heap and reorder min_heap to maintain heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
