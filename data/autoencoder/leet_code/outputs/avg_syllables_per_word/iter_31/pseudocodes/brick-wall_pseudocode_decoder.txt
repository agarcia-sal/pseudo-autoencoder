CLASS Solution  
    FUNCTION leastBricks(wall)  
        SET gap_counts TO a new mapping with default value zero for each key  
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN the elements of row except the last one  
                INCREMENT total BY brick  
                INCREMENT the value associated with total in gap_counts BY one  
            END FOR  
        END FOR  
        SET max_gaps TO the maximum value among all values in gap_counts if gap_counts is not empty otherwise zero  
        SET minimum_crossed_bricks TO the number of elements in wall MINUS max_gaps  
        RETURN minimum_crossed_bricks  
    END FUNCTION  
END CLASS
