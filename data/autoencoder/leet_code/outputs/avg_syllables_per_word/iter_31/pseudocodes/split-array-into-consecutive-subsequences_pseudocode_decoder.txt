CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO the frequency count of each element in nums  
        SET end_with TO an empty frequency count mapping for subsequences ending with a number  
        
        FOR each num IN nums  
            IF the frequency count of num in num_count EQUALS zero  
                CONTINUE to the next iteration  
            END IF  
            
            IF the frequency count of number one less than num in end_with GREATER THAN zero  
                DECREMENT the frequency count of number one less than num in end_with BY one  
                INCREMENT the frequency count of num in end_with BY one  
            ELSE IF the frequency count of number one greater than num in num_count GREATER THAN zero AND the frequency count of number two greater than num in num_count GREATER THAN zero  
                DECREMENT the frequency count of number one greater than num in num_count BY one  
                DECREMENT the frequency count of number two greater than num in num_count BY one  
                INCREMENT the frequency count of number two greater than num in end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the frequency count of num in num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
