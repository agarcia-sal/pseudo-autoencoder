CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the collection that counts occurrences of each element in nums  
        SET unique_nums TO list of keys in count arranged in ascending order  
        SET dp TO list of zeros with the length equal to the length of unique_nums PLUS one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the value associated to the element at position zero of unique_nums in count  
        FOR i FROM two TO the length of unique_nums PLUS one  
            SET num TO the element at position i MINUS one of unique_nums  
            IF num EQUALS the element at position i MINUS two of unique_nums PLUS one  
                SET dp at position i TO the maximum value BETWEEN dp at position i MINUS one AND dp at position i MINUS two PLUS num MULTIPLIED BY the value associated to num in count  
            ELSE  
                SET dp at position i TO dp at position i MINUS one PLUS num MULTIPLIED BY the value associated to num in count  
            END IF  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
