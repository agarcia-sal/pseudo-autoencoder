CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position one of values, no left child, no right child  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue holds any elements  
        SET node TO remove and return the leftmost element of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with element at position i of values, no left child, no right child  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with element at position i of values, no left child, no right child  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left child of p and the left child of q AND the result of is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS None  
            RETURN  
        END IF  
        SET stack TO a new list containing root  
        WHILE stack holds any elements  
            SET node TO remove and return the last element of stack  
            IF the right child of node NOT EQUALS None  
                APPEND the right child of node TO stack  
            END IF  
            IF the left child of node NOT EQUALS None  
                APPEND the left child of node TO stack  
            END IF  
            SET the left child of node TO None  
            IF stack holds any elements  
                SET the right child of node TO the last element of stack  
            ELSE  
                SET the right child of node TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
