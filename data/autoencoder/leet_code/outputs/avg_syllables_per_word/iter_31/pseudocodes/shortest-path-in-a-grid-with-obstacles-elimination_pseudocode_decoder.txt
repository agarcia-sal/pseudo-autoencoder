CLASS Solution  
    FUNCTION shortestPath(grid k)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN negative one  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF k GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        SET directions TO the collection of pairs representing zero and one zero and zero one zero and negative one zero and negative one zero  
        SET queue TO a double ended queue containing the tuple of zero zero k zero  
        SET visited TO a set containing the tuple of zero zero k  
        WHILE queue is not empty  
            SET x y remaining_k steps TO the FIRST element REMOVED FROM the front of queue  
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET new_k TO remaining_k MINUS the element at position nx of grid AT position ny  
                    IF new_k GREATER THAN OR EQUAL TO zero AND the tuple nx ny new_k NOT IN visited  
                        ADD the tuple nx ny new_k TO visited  
                        ADD the tuple nx ny new_k steps PLUS one TO the end of queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
