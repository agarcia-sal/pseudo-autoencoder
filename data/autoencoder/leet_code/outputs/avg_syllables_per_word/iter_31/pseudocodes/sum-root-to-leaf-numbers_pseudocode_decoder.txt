CLASS TreeNode  
    FUNCTION __init__ with parameters val LEFT EQUALS zero LEFT EQUALS null RIGHT EQUALS null  
        SET the val property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN null  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT null  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT null  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p AND q  
    IF p EQUALS null AND q EQUALS null  
        RETURN true  
    ELSE IF p EQUALS null OR q EQUALS null  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p AND the left property of q AND the result of is_same_tree with the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers with parameters root  
        FUNCTION dfs with parameters node AND current_number  
            IF node EQUALS null  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS the val property of node  
            IF the left property of node EQUALS null AND the right property of node EQUALS null  
                RETURN current_number  
            END IF  
            SET left_sum TO the result of dfs with the left property of node AND current_number  
            SET right_sum TO the result of dfs with the right property of node AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN the result of dfs with root AND zero  
    END FUNCTION  
END CLASS
