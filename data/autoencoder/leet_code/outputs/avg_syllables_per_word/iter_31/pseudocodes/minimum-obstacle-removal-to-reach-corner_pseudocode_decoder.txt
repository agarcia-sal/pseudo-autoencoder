CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST of tuples representing zero and one, one and zero, zero and negative one, negative one and zero  
        SET min_heap TO the LIST containing a tuple of zero zero zero representing the cost and coordinates  
        SET visited TO the SET containing the tuple of zero zero representing coordinates  
  
        WHILE min_heap is NOT empty  
            REMOVE the element with the smallest cost from min_heap ASSIGNING the elements TO cost x y  
  
            IF the tuple of x y EQUALS the tuple of m MINUS one n MINUS one  
                RETURN cost  
            END IF  
  
            FOR each dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple of nx ny NOT IN visited  
                    ADD the tuple of nx ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid AT position ny  
                    ADD the tuple of new_cost nx ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  
  
        RETURN negative one  
    END FUNCTION  
END CLASS
