CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the plus character  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                IF the character char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the numerical value of char  
                END IF  
                IF the character char EQUALS the left parenthesis character  
                    SET num TO the result of calling helper WITH s  
                END IF  
                IF the character char IS one of the following characters the plus character OR the minus character OR the multiplication character OR the division character OR the left parenthesis character OR the right parenthesis character OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS the plus character  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the multiplication character  
                        SET element at position the last index of stack TO element at position the last index of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character  
                        SET element at position the last index of stack TO the integer part of element at position the last index of stack DIVIDED BY num with truncation toward zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF the character char EQUALS the right parenthesis character  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        SET s TO the list of characters in s  
        RETURN the result of calling helper WITH s  
    END FUNCTION  
END CLASS
