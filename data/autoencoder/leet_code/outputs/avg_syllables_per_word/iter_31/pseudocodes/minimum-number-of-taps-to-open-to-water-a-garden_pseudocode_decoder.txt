CLASS Solution  
    FUNCTION minTaps(n ranges)  
        SET max_range TO a list of zeros with the length of n PLUS one  
        FOR index FROM zero TO n  
            IF element at position index of ranges GREATER THAN zero  
                SET left_bound TO the GREATER THAN OR EQUAL TO zero value BETWEEN zero AND index MINUS element at position index of ranges  
                SET right_bound TO the LESS THAN OR EQUAL TO value BETWEEN n AND index PLUS element at position index of ranges  
                SET element at position left_bound of max_range TO the GREATER THAN OR EQUAL TO value BETWEEN element at position left_bound of max_range AND right_bound  
            END IF  
        END FOR  
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR position FROM zero TO n  
            IF position GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF position GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the GREATER THAN OR EQUAL TO value BETWEEN farthest AND element at position position of max_range  
        END FOR  
        RETURN taps  
    END FUNCTION  
END CLASS
