CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero THEN  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_value_string TO the string representation of the absolute value of x  
        SET reversed_string TO the sequence of characters in absolute_value_string taken in reverse order  
        SET reversed_x TO the integer value corresponding to reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO THE power of thirty one OR result GREATER THAN two RAISED TO THE power of thirty one MINUS one THEN  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
