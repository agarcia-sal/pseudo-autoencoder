CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        
        SET start TO start PLUS the INTEGER DIVISION of n MINUS one BY length  
        
        SET number_string TO the STRING representation of start  
        SET digit_position TO the REMAINDER of n MINUS one DIVIDED BY length  
        SET digit_character TO the CHARACTER at position digit_position of number_string  
        
        RETURN the INTEGER value of digit_character  
    END FUNCTION  
END CLASS
