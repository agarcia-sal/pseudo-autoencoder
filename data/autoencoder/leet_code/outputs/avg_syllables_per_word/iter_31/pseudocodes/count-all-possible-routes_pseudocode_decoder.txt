CLASS Solution  
    FUNCTION countRoutes(locations parameter start parameter finish parameter fuel)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF locations  
        
        FUNCTION dp(city parameter remaining_fuel)  
            IF remaining_fuel LESS THAN zero THEN  
                RETURN zero  
            END IF  
            SET routes TO zero  
            IF city EQUALS finish THEN  
                SET routes TO one  
            END IF  
            FOR each next_city FROM zero TO n MINUS one  
                IF next_city NOT EQUALS city THEN  
                    SET fuel_cost TO the ABSOLUTE VALUE OF element at position city of locations MINUS element at position next_city of locations  
                    IF remaining_fuel GREATER THAN OR EQUAL TO fuel_cost THEN  
                        SET routes TO routes PLUS dp(next_city parameter remaining_fuel MINUS fuel_cost)  
                        SET routes TO routes MODULO MOD  
                    END IF  
                END IF  
            END FOR  
            RETURN routes  
        END FUNCTION  
        
        RETURN dp(start parameter fuel)  
    END FUNCTION  
END CLASS
