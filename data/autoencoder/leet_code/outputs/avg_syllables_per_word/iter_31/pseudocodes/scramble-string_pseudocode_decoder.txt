CLASS Solution  
    FUNCTION isScramble(s1 STRING_PARAMETER s2 STRING_PARAMETER) RETURNS BOOLEAN  
        IF s1 EQUALS s2  
            RETURN True  
        END IF  
        IF sorted characters of s1 NOT EQUALS sorted characters of s2  
            RETURN False  
        END IF  
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF isScramble(substring from start TO position i MINUS one of s1 STRING_PARAMETER substring from start TO position i MINUS one of s2) AND isScramble(substring from position i TO end of s1 STRING_PARAMETER substring from position i TO end of s2)  
                RETURN True  
            END IF  
            IF isScramble(substring from start TO position i MINUS one of s1 STRING_PARAMETER substring from position n MINUS i TO end of s2) AND isScramble(substring from position i TO end of s1 STRING_PARAMETER substring from start TO position n MINUS i MINUS one of s2)  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
