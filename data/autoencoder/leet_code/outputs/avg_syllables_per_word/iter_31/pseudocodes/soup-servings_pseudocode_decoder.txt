CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        
        SET n TO n PLUS twenty four DIVIDED BY twenty five rounded down to the nearest integer greater than or equal to the result  
        
        SET memo TO empty dictionary  
        
        FUNCTION dp(a, b)  
            IF the tuple of a and b is present in memo  
                RETURN the value corresponding to the tuple of a and b in memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN one HALF  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            
            SET prob TO zero POINT two five MULTIPLIED BY the sum of dp(a MINUS four, b) PLUS dp(a MINUS three, b MINUS one) PLUS dp(a MINUS two, b MINUS two) PLUS dp(a MINUS one, b MINUS three)  
            
            SET the value corresponding to the tuple of a and b in memo TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp(n, n)  
    END FUNCTION  
END CLASS
