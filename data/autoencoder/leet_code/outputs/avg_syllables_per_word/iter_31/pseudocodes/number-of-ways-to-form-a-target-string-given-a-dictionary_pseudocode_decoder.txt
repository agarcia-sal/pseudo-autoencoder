CLASS Solution  
    FUNCTION numWays(words target)  
        SET MOD TO ten RAISED TO THE power OF nine PLUS one  
        SET n TO the LENGTH OF element at position zero of words  
        SET m TO the LENGTH OF target  
      
        SET freq TO a list of default dictionaries of integer with size n  
        FOR each word IN words  
            FOR each position j AND character char IN the enumeration of word  
                INCREMENT the value for key char IN freq at position j BY one  
            END FOR  
        END FOR  
      
        SET dp TO a two dimensional list with m PLUS one rows AND n PLUS one columns FILLED WITH zero  
      
        FOR each index j FROM zero TO n  
            SET element at position zero OF dp at position j TO one  
        END FOR  
      
        FOR each index i FROM one TO m  
            FOR each index j FROM one TO n  
                SET element at position j OF dp at position i TO element at position j MINUS one OF dp at position i  
                IF value ASSOCIATED WITH key element at position i MINUS one OF target IN freq at position j MINUS one IS GREATER THAN zero  
                    SET element at position j OF dp at position i TO element at position j OF dp at position i PLUS element at position j MINUS one OF dp at position i MINUS one MULTIPLIED BY value ASSOCIATED WITH key element at position i MINUS one OF target IN freq at position j MINUS one  
                    SET element at position j OF dp at position i TO element at position j OF dp at position i MODULO MOD  
                END IF  
            END FOR  
        END FOR  
      
        RETURN element at position n OF dp at position m  
    END FUNCTION  
END CLASS
