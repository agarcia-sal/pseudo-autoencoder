CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO element at position j of element at position i of matrix  
                SET element at position j of element at position i of matrix TO element at position i of element at position j of matrix  
                SET element at position i of element at position j of matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            REVERSE the order of elements in element at position i of matrix  
        END FOR  
    END FUNCTION  
END CLASS
