CLASS Solution  
    FUNCTION insert(intervals AS List OF List OF Integer newInterval AS List OF Integer) AS List OF List OF Integer  
        IF the LENGTH OF intervals EQUALS zero THEN  
            RETURN a new list containing newInterval  
        END IF  

        SET merged TO a new empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  

        WHILE i LESS THAN n AND the element at position one of the element at position i of intervals LESS THAN the element at position zero of newInterval DO  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  

        WHILE i LESS THAN n AND the element at position zero of the element at position i of intervals LESS THAN OR EQUAL TO the element at position one of newInterval DO  
            SET the element at position zero of newInterval TO the smaller value BETWEEN the element at position zero of newInterval AND the element at position zero of the element at position i of intervals  
            SET the element at position one of newInterval TO the larger value BETWEEN the element at position one of newInterval AND the element at position one of the element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  

        APPEND newInterval TO merged  

        WHILE i LESS THAN n DO  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  

        RETURN merged  
    END FUNCTION  
END CLASS
