CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET sorted_pairs TO the list of pairs each consisting of element at position index of position AND element at position index of speed sorted by the first element of each pair in descending order  
        SET time_to_target TO an empty list  
        FOR each pair IN sorted_pairs  
            SET position_element TO the first element of pair  
            SET speed_element TO the second element of pair  
            SET time_element TO (target MINUS position_element) DIVIDED BY speed_element  
            APPEND time_element TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
