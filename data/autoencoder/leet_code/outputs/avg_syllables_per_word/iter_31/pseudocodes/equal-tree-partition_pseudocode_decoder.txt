CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with val AS the element at position zero of values AND left AS None AND right AS None  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with val AS the element at position i of values AND left AS None AND right AS None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with val AS the element at position i of values AND left AS None AND right AS None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the value of is_same_tree on p.left AND q.left AND is_same_tree on p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            RETURN node.val PLUS subtree_sum of node.left PLUS subtree_sum of node.right  
        END FUNCTION  

        SET total_sum TO subtree_sum of root  

        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False  

        FUNCTION can_split(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS can_split of node.left PLUS can_split of node.right  
            IF current_sum EQUALS target AND node NOT EQUALS root  
                SET the first element of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split with root as argument  
        RETURN the first element of found  
    END FUNCTION  
END CLASS
