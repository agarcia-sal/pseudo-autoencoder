CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty jobDifficulty LIST OF INTEGER d INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN the NEGATIVE OF one  
        END IF  
        
        FUNCTION dp(i INTEGER d INTEGER) RETURNS INTEGER  
            IF d EQUALS one  
                RETURN the MAXIMUM VALUE IN the SUBSEQUENCE OF jobDifficulty FROM position i TO THE END  
            END IF  
            
            SET min_difficulty TO INFINITY  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d PLUS one MINUS one  
                SET current_max TO the MAXIMUM VALUE BETWEEN current_max AND the ELEMENT AT position j OF jobDifficulty  
                SET candidate TO current_max PLUS the RESULT OF dp WITH PARAMETERS j PLUS one AND d MINUS one  
                SET min_difficulty TO the MINIMUM VALUE BETWEEN min_difficulty AND candidate  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN the RESULT OF dp WITH PARAMETERS zero AND d  
    END FUNCTION  
END CLASS
