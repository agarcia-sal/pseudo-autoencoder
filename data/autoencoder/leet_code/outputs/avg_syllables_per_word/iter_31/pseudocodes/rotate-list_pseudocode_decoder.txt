CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val EQUALS element at position one of values  
    SET p TO head  
    FOR each val IN elements starting from position two TO end of values  
        SET node TO new ListNode with val EQUALS val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that element val of p1 EQUALS element val of p2 AND call is_same_list with parameters element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(head, k)  
        IF head EQUALS none OR element next of head EQUALS none OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE element next of tail NOT EQUALS none  
            SET tail TO element next of tail  
            INCREMENT length BY one  
        END WHILE  
        SET element next of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR index FROM one TO steps_to_new_tail  
            SET new_tail TO element next of new_tail  
        END FOR  
        SET new_head TO element next of new_tail  
        SET element next of new_tail TO none  
        RETURN new_head  
    END FUNCTION  
END CLASS
