CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET the value of self DOT val TO val  
        SET the value of self DOT left TO left  
        SET the value of self DOT right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the negation of the truth value of values holds  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque structure  
    APPEND root TO queue  
    WHILE the truth value of queue holds  
        SET node TO the removal of the element from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not equal to None  
            SET the left attribute of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not equal to None  
            SET the right attribute of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF the negation of the truth value of p AND the negation of the truth value of q hold  
        RETURN True  
    ELSE IF the negation of the truth value of p OR the negation of the truth value of q hold  
        RETURN False  
    ELSE IF the value of the attribute val of p is not equal to the value of the attribute val of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left attributes of p and q AND the result of calling is_same_tree with the right attributes of p and q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF the negation of the truth value of preorder OR the negation of the truth value of inorder hold  
            RETURN None  
        END IF  
        SET root_val TO the element at position zero of preorder  
        SET root TO a new instance of TreeNode with root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET the left attribute of root TO the result of calling buildTree on self with the sublist of preorder from position one TO position root_index_inorder PLUS one AND the sublist of inorder from the start TO position root_index_inorder  
        SET the right attribute of root TO the result of calling buildTree on self with the sublist of preorder from position root_index_inorder PLUS one TO the end AND the sublist of inorder from position root_index_inorder PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
