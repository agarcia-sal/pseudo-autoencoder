CLASS Solution  
    FUNCTION findBlackPixel(picture AS list of list of string, target AS integer) RETURNS integer  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        
        SET variable m TO the LENGTH OF picture  
        SET variable n TO the LENGTH OF element at position zero of picture  
        SET variable row_count TO a list of zeros with size m  
        SET variable col_count TO a list of zeros with size n  
        SET variable row_patterns TO a mapping from string to integer with default zero  
        
        FOR variable r FROM zero TO m MINUS one  
            SET variable pattern TO the concatenation of all elements in element at position r of picture in order  
            INCREMENT value of row_patterns with key pattern BY one  
            FOR variable c FROM zero TO n MINUS one  
                IF element at position r of picture at position c EQUALS black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET variable lonely_pixels TO zero  
        
        FOR variable r FROM zero TO m MINUS one  
            FOR variable c FROM zero TO n MINUS one  
                IF element at position r of picture at position c EQUALS black pixel  
                    SET variable pattern TO the concatenation of all elements in element at position r of picture in order  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND value of row_patterns with key pattern EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
