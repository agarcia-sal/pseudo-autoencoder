CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            IF the LENGTH OF segment EQUALS one  
                RETURN true  
            ELSE IF the element at position zero of segment NOT EQUALS the character zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty five  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the concatenation of elements of path separated by the word period TO result  
                END IF  
                RETURN  
            END IF  
  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the substring of s from position start TO position start PLUS length MINUS one  
                    IF is_valid(segment)  
                        CALL backtrack WITH start PLUS length AND path WITH segment APPENDED  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET result TO empty list  
        CALL backtrack WITH zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
