CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the position found by searching for the first element in dp GREATER THAN OR EQUAL TO num  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
