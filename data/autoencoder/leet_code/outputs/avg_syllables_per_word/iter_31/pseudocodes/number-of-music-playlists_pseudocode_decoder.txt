CLASS Solution  
    FUNCTION numMusicPlaylists(n goal k)  
        SET MOD TO ten POWER nine PLUS one  
        INITIALIZE dp AS a list of lists with dimensions from zero TO goal PLUS one and zero TO n PLUS one filled with zero  
        SET element at position zero of dp AND element at position zero TO one  
        
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position i of dp AND element at position j TO element at position i MINUS one of dp AND element at position j MINUS one MULTIPLIED BY n MINUS j PLUS one PLUS element at position i of dp AND element at position j  
                SET element at position i of dp AND element at position j TO element at position i of dp AND element at position j MODULO MOD  
                
                IF j GREATER THAN k THEN  
                    SET element at position i of dp AND element at position j TO element at position i of dp AND element at position j PLUS element at position i MINUS one of dp AND element at position j MULTIPLIED BY j MINUS k  
                    SET element at position i of dp AND element at position j TO element at position i of dp AND element at position j MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position goal of dp AND element at position n  
    END FUNCTION  
END CLASS
