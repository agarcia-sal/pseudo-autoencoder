CLASS Solution  
    FUNCTION isTransformable(s, t)  
        DECLARE pos AS a mapping from digit to list of positions initialized to empty lists  
        
        FOR each index i AND character c IN s  
            SET digit CONVERTED from character c to its numerical value  
            ADD index i TO the list of positions for digit in pos using a procedure that maintains smallest element removal first  
        END FOR  
        
        FOR each index i AND character c IN t  
            SET digit CONVERTED from character c to its numerical value  
            
            IF the list of positions for digit in pos IS EMPTY  
                RETURN false  
            END IF  
            
            FOR each smaller FROM zero UP TO digit MINUS one  
                IF the list of positions for smaller in pos IS NOT EMPTY AND the smallest element of the list of positions for smaller IS LESS THAN the smallest element of the list of positions for digit  
                    RETURN false  
                END IF  
            END FOR  
            
            REMOVE the smallest element from the list of positions for digit in pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
