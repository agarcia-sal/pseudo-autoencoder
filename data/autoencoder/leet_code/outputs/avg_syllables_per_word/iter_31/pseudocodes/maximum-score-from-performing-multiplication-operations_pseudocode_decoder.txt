CLASS Solution  
    FUNCTION maximumScore(nums parameter multipliers parameter)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero values with the LENGTH OF m PLUS one  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zero values with the LENGTH OF m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i MINUS left  
                SET mul TO element at position i of multipliers  
                IF mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp GREATER THAN OR EQUAL TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                    SET element at position left of new_dp TO mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp  
                ELSE  
                    SET element at position left of new_dp TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
