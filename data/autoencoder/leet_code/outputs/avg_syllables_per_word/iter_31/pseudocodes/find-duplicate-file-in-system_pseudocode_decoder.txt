CLASS Solution  
    FUNCTION findDuplicate(paths)  
        ASSIGN reference TO a new empty mapping from file content TO list of file paths  
        FOR each path IN paths  
            SET parts TO the list of substrings obtained by splitting path on whitespace  
            SET directory TO element at position zero of parts  
            FOR each file_info IN elements from position one TO the last position of parts  
                SET file_name TO the substring of file_info before the first occurrence of the character representing open parenthesis  
                SET content TO the substring of file_info from the position after the first occurrence of the character representing open parenthesis TO the position before the last character of file_info  
                APPEND the concatenation of directory THEN the character representing forward slash THEN file_name TO the list associated with content in the mapping  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each group_of_paths IN the values of the mapping  
            IF the length of group_of_paths GREATER THAN one  
                APPEND group_of_paths TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
