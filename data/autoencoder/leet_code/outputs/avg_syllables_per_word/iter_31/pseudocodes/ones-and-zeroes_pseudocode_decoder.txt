CLASS Solution  
    FUNCTION findMaxForm(strs m n)  
        SET dp TO two dimensional list with m PLUS one rows and n PLUS one columns each element set to zero  
        FOR each s IN strs  
            SET count_0 TO the number of occurrences of character zero in s  
            SET count_1 TO the number of occurrences of character one in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i j TO the greater of value dp element at position i j AND value dp element at position i MINUS count_0 j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m n  
    END FUNCTION  
END CLASS
