CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start first second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
              
            SET expected TO the string representation of first PLUS second  
            SET length TO the LENGTH OF expected  
              
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring from position start TO position start PLUS length MINUS one of num NOT EQUALS expected  
                RETURN False  
            END IF  
              
            RETURN the result of calling is_valid_sequence with arguments start PLUS length second and the integer conversion of expected  
        END FUNCTION  
          
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the substring from position zero TO position i MINUS one of num  
                SET second TO the substring from position i TO position j MINUS one of num  
                  
                IF the LENGTH OF first GREATER THAN one AND the element at position zero of first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the element at position zero of second EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                  
                IF calling is_valid_sequence with arguments j the integer conversion of first and the integer conversion of second RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
          
        RETURN False  
    END FUNCTION  
END CLASS
