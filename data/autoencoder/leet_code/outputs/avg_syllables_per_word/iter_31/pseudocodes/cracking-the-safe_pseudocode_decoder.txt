CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO an empty set of sequences  
        SET sequence TO an empty list of elements  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO the concatenation of node and the string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH the substring of edge from position one TO the end  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO the string consisting of zero repeated n MINUS one times  
        CALL dfs WITH start_node  
        
        SET reversed_sequence TO the reverse of the list sequence  
        SET full_sequence TO the concatenation of all elements in reversed_sequence FOLLOWED BY start_node  
        RETURN full_sequence  
    END FUNCTION  
END CLASS
