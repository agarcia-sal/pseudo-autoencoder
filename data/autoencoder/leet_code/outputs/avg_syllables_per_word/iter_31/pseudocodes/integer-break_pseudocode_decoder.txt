CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        SET the element at position one of dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET candidate_one TO j MULTIPLIED BY the element at position i MINUS j of dp  
                SET candidate_two TO j MULTIPLIED BY i MINUS j  
                SET max_product TO the maximum value AMONG max_product candidate_one candidate_two  
            END FOR  
            SET the element at position i of dp TO max_product  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
