CLASS Solution  
    FUNCTION waysToDistribute(n parameter, k parameter) RETURNS integer  
        SET MOD TO ten RAISED TO THE power of nine PLUS one  
        SET dp TO a two dimensional list with n PLUS one rows and k PLUS one columns containing zeroes  
        SET element at position zero at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET value_one TO j MULTIPLIED BY element at position i MINUS one at position j of dp  
                SET value_two TO element at position i MINUS one at position j MINUS one of dp  
                SET element at position i at position j of dp TO the remainder of the division of the sum of value_one PLUS value_two by MOD  
            END FOR  
        END FOR  
        RETURN element at position n at position k of dp  
    END FUNCTION  
END CLASS
