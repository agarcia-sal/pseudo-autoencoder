CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        CREATE empty set named points  
        SET area TO zero  
        
        FOR each list named rectangle IN rectangles  
            SET x1 TO element at position zero of rectangle  
            SET y1 TO element at position one of rectangle  
            SET x2 TO element at position two of rectangle  
            SET y2 TO element at position three of rectangle  
            
            IF X1 GREATER THAN x1 THEN  
                SET X1 TO x1  
            END IF  
            IF Y1 GREATER THAN y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF X2 LESS THAN x2 THEN  
                SET X2 TO x2  
            END IF  
            IF Y2 LESS THAN y2 THEN  
                SET Y2 TO y2  
            END IF  
            
            SET area TO area PLUS the product of the difference of x2 MINUS x1 AND the difference of y2 MINUS y1  
            
            FOR each point p IN the list composed of tuples of x1 and y1 AND x1 and y2 AND x2 and y1 AND x2 and y2  
                IF p IS IN points THEN  
                    REMOVE p FROM points  
                ELSE  
                    ADD p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area NOT EQUALS the product of the difference of X2 MINUS X1 AND the difference of Y2 MINUS Y1 THEN  
            RETURN False  
        END IF  
        
        IF points NOT EQUALS the set composed of tuples of X1 and Y1 AND X1 and Y2 AND X2 and Y1 AND X2 and Y2 THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
