CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF nums  
        IF the REMAINDER OF total_sum DIVIDED BY two NOT EQUALS zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of False repeated target PLUS one times  
        SET the element at position zero of dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO the value of dp at position i OR the value of dp at the position i MINUS num  
            END FOR  
        END FOR  
        RETURN the value of dp at position target  
    END FUNCTION  
END CLASS
