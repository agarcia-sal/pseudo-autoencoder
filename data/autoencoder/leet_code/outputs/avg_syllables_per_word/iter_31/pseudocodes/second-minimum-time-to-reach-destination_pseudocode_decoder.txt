CLASS Solution  
    FUNCTION secondMinimum(n time change edges)  
        SET graph TO empty list of empty lists with size n PLUS one  
        FOR each pair u v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  

        SET queue TO double ended queue containing tuple of one zero zero  
        SET minTimes TO list of two element lists each containing positive infinity repeated two times with size n PLUS one  
        SET element at position one at position zero of minTimes TO zero  

        WHILE queue is not empty  
            REMOVE leftmost element from queue and ASSIGN to current currentTime count  

            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  

            IF integer division of currentTime by change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO integer division of currentTime by change PLUS one MULTIPLIED BY change PLUS time  
            END IF  

            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor at position zero of minTimes GREATER THAN nextTime  
                    SET element at position neighbor at position one of minTimes TO element at position neighbor at position zero of minTimes  
                    SET element at position neighbor at position zero of minTimes TO nextTime  
                    APPEND tuple of neighbor nextTime zero TO queue  
                ELSE IF element at position neighbor at position zero of minTimes LESS THAN nextTime AND nextTime LESS THAN element at position neighbor at position one of minTimes  
                    SET element at position neighbor at position one of minTimes TO nextTime  
                    APPEND tuple of neighbor nextTime one TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
