CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping from character to integer  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end and character IN enumeration of s  
            IF character IS IN char_index_map AND value at character in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO value at character in char_index_map PLUS one  
            END IF  
            SET value at character in char_index_map TO end  
            SET max_length TO the GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
