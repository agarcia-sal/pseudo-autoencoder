CLASS Solution  
    FUNCTION wordsAbbreviation(words)  
        FUNCTION abbreviate(word, prefix_length)  
            IF the LENGTH OF word MINUS prefix_length IS LESS THAN OR EQUAL TO two  
                RETURN word  
            END IF  
            RETURN substring from position one TO position prefix_length of word PLUS the LENGTH OF word MINUS prefix_length MINUS one PLUS the last element of word  
        END FUNCTION  
        
        FUNCTION get_unique_abbreviations(words_with_indices, prefix_length)  
            SET abbs TO a new mapping with default empty list  
            FOR each index AND word IN words_with_indices  
                APPEND the tuple of index AND word TO the element in abbs at key abbreviate(word, prefix_length)  
            END FOR  
            RETURN abbs  
        END FUNCTION  
        
        SET n TO the LENGTH OF words  
        SET ans TO a list of empty string repeated n times  
        SET words_with_indices TO a list of tuples where each tuple is index AND word for index AND word in enumeration of words  
        
        SET abbs TO get_unique_abbreviations(words_with_indices, one)  
        
        FOR each abb AND group IN abbs  
            IF the LENGTH OF group EQUALS one  
                SET index TO the first element of the first tuple in group  
                SET word TO the second element of the first tuple in group  
                SET the element at position index of ans TO abb  
            ELSE  
                SET new_group TO an empty list  
                FOR each index AND word IN group  
                    APPEND the tuple of index AND word TO new_group  
                END FOR  
                SET prefix_length TO two  
                WHILE the LENGTH OF new_group IS GREATER THAN zero  
                    SET new_abbs TO get_unique_abbreviations(new_group, prefix_length)  
                    SET new_group TO an empty list  
                    FOR each new_abb AND new_group_items IN new_abbs  
                        IF the LENGTH OF new_group_items EQUALS one  
                            SET index TO the first element of the first tuple in new_group_items  
                            SET word TO the second element of the first tuple in new_group_items  
                            SET the element at position index of ans TO new_abb  
                        ELSE  
                            FOR each item IN new_group_items  
                                APPEND item TO new_group  
                            END FOR  
                        END IF  
                    END FOR  
                    INCREMENT prefix_length BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN ans  
    END FUNCTION  
END CLASS
