CLASS Solution  
    FUNCTION nthUglyNumber(n)  
        SET ugly_numbers TO a list containing the number one  
        SET i2 TO zero  
        SET i3 TO zero  
        SET i5 TO zero  
        WHILE the LENGTH OF ugly_numbers LESS THAN n holds  
            SET next_multiple_of_2 TO element at position i2 of ugly_numbers MULTIPLIED BY two  
            SET next_multiple_of_3 TO element at position i3 of ugly_numbers MULTIPLIED BY three  
            SET next_multiple_of_5 TO element at position i5 of ugly_numbers MULTIPLIED BY five  
            SET next_ugly_number TO the smallest value among next_multiple_of_2 AND next_multiple_of_3 AND next_multiple_of_5  
            APPEND next_ugly_number TO ugly_numbers  
            IF next_ugly_number EQUALS next_multiple_of_2 THEN  
                INCREMENT i2 BY one  
            END IF  
            IF next_ugly_number EQUALS next_multiple_of_3 THEN  
                INCREMENT i3 BY one  
            END IF  
            IF next_ugly_number EQUALS next_multiple_of_5 THEN  
                INCREMENT i5 BY one  
            END IF  
        END WHILE  
        RETURN element at position n MINUS one of ugly_numbers  
    END FUNCTION  
END CLASS
