CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the set containing the symbols two five six nine  
        SET same_digits TO the set containing the symbols zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF for every digit in num_str the digit is an element of the union of change_digits and same_digits  
                IF there exists at least one digit in num_str such that the digit is an element of change_digits  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number(i) EQUALS true  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
