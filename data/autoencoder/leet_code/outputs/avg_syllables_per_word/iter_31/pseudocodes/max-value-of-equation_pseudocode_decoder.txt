CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        DECLARE deque_ AS an empty double ended queue  
        
        FOR each xj and yj IN points  
            WHILE the deque_ is not empty AND the subtraction of element at position one of the first element of deque_ FROM xj is GREATER THAN k  
                REMOVE the first element FROM deque_  
            END WHILE  
            
            IF the deque_ is not empty  
                SET max_value TO the maximum of max_value AND the addition of yj AND xj AND the first element at position zero of the first element of deque_  
            END IF  
            
            WHILE the deque_ is not empty AND the subtraction of yj MINUS xj is GREATER THAN OR EQUAL TO the first element at position zero of the last element of deque_  
                REMOVE the last element FROM deque_  
            END WHILE  
            
            APPEND the pair consisting of the subtraction of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
