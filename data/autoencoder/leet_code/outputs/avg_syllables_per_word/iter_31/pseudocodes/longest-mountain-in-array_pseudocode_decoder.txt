CLASS Solution  
    FUNCTION longestMountain(arr)  
        SET n TO the LENGTH OF arr  
        SET longest TO zero  
        SET i TO one  
        WHILE i LESS THAN n MINUS one  
            IF element at position i MINUS one of arr LESS THAN element at position i of arr AND element at position i of arr GREATER THAN element at position i PLUS one of arr  
                SET left TO i MINUS one  
                SET right TO i PLUS one  
                WHILE left GREATER THAN zero AND element at position left MINUS one of arr LESS THAN element at position left of arr  
                    DECREMENT left BY one  
                END WHILE  
                WHILE right LESS THAN n MINUS one AND element at position right of arr GREATER THAN element at position right PLUS one of arr  
                    INCREMENT right BY one  
                END WHILE  
                SET mountain_length TO right MINUS left PLUS one  
                IF mountain_length GREATER THAN longest  
                    SET longest TO mountain_length  
                END IF  
                SET i TO right  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN longest  
    END FUNCTION  
END CLASS
