CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SET houses TO houses sorted in ascending order  
        SET heaters TO heaters sorted in ascending order  
        SET min_radius TO zero  
        SET j TO zero  
        FOR each house IN houses  
            WHILE j LESS THAN the length of heaters MINUS one AND  
                  the absolute value of element at position j of heaters MINUS house GREATER THAN OR EQUAL TO the absolute value of element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            SET min_radius TO the greater of min_radius AND the absolute value of element at position j of heaters MINUS house  
        END FOR  
        RETURN min_radius  
    END FUNCTION  
END CLASS
