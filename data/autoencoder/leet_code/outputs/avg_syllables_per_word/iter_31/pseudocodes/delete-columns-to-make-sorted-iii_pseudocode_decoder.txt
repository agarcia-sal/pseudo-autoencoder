CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position one of strs  
        SET dp TO a list of n elements each SET TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i MINUS one  
                IF all elements at position k of element at position j of strs LESS THAN OR EQUAL TO element at position k of element at position i of strs FOR each k FROM one TO m  
                    SET dp at position i TO the GREATER VALUE BETWEEN dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence TO the GREATER VALUE of all elements in dp  
        RETURN n MINUS longest_non_decreasing_subsequence  
    END FUNCTION  
END CLASS
