CLASS Solution  
    FUNCTION intToRoman(num)  
        SET list_of_values TO list containing one thousand and nine hundred and five hundred and four hundred and one hundred and ninety and fifty and forty and ten and nine and five and four and one  
        SET list_of_symbols TO list containing the symbol representing one thousand and the symbol representing nine hundred and the symbol representing five hundred and the symbol representing four hundred and the symbol representing one hundred and the symbol representing ninety and the symbol representing fifty and the symbol representing forty and the symbol representing ten and the symbol representing nine and the symbol representing five and the symbol representing four and the symbol representing one  
        SET roman_numeral TO empty string  
        SET index TO zero  
        WHILE num GREATER THAN zero  
            SET count TO the integer quotient of num DIVIDED BY the element at position index of list_of_values  
            FOR each iteration FROM one TO count  
                SET roman_numeral TO roman_numeral PLUS the element at position index of list_of_symbols  
                DECREMENT num BY the element at position index of list_of_values  
            END FOR  
            INCREMENT index BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
