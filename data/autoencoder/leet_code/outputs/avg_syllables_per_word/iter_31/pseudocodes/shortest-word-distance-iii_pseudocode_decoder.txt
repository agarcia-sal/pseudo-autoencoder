CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict word_list OF STRINGS word1 word_one OF STRING word2 word_two) RETURNS INTEGER  
        SET min_distance TO infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word_one EQUALS word_two  
        FOR index i FROM zero TO the LENGTH OF word_list MINUS one  
            SET current_word TO element at position i of word_list  
            IF current_word EQUALS word_one  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            IF current_word EQUALS word_two AND NOT same_word  
                SET prev_word2 TO i  
            END IF  
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                IF the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2 LESS THAN min_distance  
                    SET min_distance TO the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2  
                END IF  
            END IF  
        END FOR  
        RETURN min_distance  
    END FUNCTION  
END CLASS
