CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF nums EQUALS no elements OR k EQUALS zero  
            RETURN no elements  
        END IF  
        
        SET deq TO empty double ended queue  
        SET result TO empty list  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq is not empty AND element at position zero of deq LESS THAN i MINUS k PLUS one  
                REMOVE the first element FROM deq  
            END IF  
            
            WHILE deq is not empty AND element at position element at last position of deq of nums LESS THAN element at position i of nums  
                REMOVE the last element FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND element at position element at first position of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
