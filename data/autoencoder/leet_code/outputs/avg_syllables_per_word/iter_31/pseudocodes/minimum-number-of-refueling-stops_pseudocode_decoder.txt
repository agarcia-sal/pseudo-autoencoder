CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO stations  

        FOR each position AND fuel IN stations  
            WHILE the LENGTH OF max_heap IS greater than zero AND current_fuel IS LESS THAN position  
                SET largest_fuel_amount TO the negative value of the element obtained by removing the element with the smallest value from max_heap  
                INCREMENT current_fuel BY largest_fuel_amount  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel IS LESS THAN position  
                RETURN negative one  
            END IF  
            ADD the negative value of fuel TO max_heap maintaining heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
