CLASS Solution  
    FUNCTION largestPathValue(colors  
                              edges)  
        SET n TO the LENGTH OF colors  
        ASSIGN reference TO an empty mapping from keys to lists AS graph  
        SET in_degree TO a list OF zero VALUES of LENGTH n  
        FOR each pair u AND v IN edges  
            APPEND v TO the list referenced BY graph AT key u  
            INCREMENT the element at position v OF in_degree BY one  
        END FOR  
        SET queue TO a double ended queue INITIALIZED WITH elements i FOR each i FROM zero TO n MINUS one WHERE the element at position i OF in_degree EQUALS zero  
        SET dp TO a list OF lists of zero VALUES OF LENGTH twenty six FOR each index FROM zero TO n MINUS one  
        SET visited TO zero  
        SET max_color_value TO zero  
        WHILE queue IS NOT empty  
            REMOVE the element from the LEFT end OF queue AND ASSIGN IT TO node  
            INCREMENT the element at position the difference BETWEEN the ASCII value of the element at position node OF colors AND the ASCII value of the lowercase letter a OF the list at position node OF dp BY one  
            SET max_color_value TO the GREATER THAN value BETWEEN max_color_value AND the element at position the difference BETWEEN the ASCII value of the element at position node OF colors AND the ASCII value of the lowercase letter a OF the list at position node OF dp  
            INCREMENT visited BY one  
            FOR each neighbor IN the list referenced BY graph AT key node  
                FOR each integer c FROM zero TO twenty six MINUS one  
                    SET the element at position c OF the list at position neighbor OF dp TO the GREATER THAN value BETWEEN the element at position c OF the list at position neighbor OF dp AND the element at position c OF the list at position node OF dp  
                END FOR  
                DECREMENT the element at position neighbor OF in_degree BY one  
                IF the element at position neighbor OF in_degree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        IF visited EQUALS n  
            RETURN max_color_value  
        ELSE  
            RETURN the negative of one  
        END IF  
    END FUNCTION  
END CLASS
