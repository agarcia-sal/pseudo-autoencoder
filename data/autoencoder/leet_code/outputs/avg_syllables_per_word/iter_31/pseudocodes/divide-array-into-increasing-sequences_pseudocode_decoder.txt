CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums k)  
        SET freq TO a mapping of element frequencies in nums  
        SET max_freq TO the greatest frequency value in freq  
        IF the length of nums DIVIDED BY k IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
