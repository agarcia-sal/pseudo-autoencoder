CLASS Solution  
    FUNCTION canWin currentState  
        SET memo TO an empty collection of key value pairs  
        
        FUNCTION can_win_helper state  
            IF state EXISTS AS a key IN memo  
                RETURN the value associated WITH state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO position index PLUS one OF state EQUALS two plus signs  
                    SET new_state TO the substring from the start TO position index MINUS one CONCATENATED WITH two minus signs CONCATENATED WITH the substring from position index PLUS two TO the end OF state  
                    IF NOT the result of can_win_helper WITH new_state  
                        SET the value associated WITH state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated WITH state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
