CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF the LENGTH OF digits EQUALS zero  
            RETURN an empty list  
        END IF  
        SET phone_map TO a mapping from digit character two TO the string of characters a b c and digit character three TO the string of characters d e f and digit character four TO the string of characters g h i and digit character five TO the string of characters j k l and digit character six TO the string of characters m n o and digit character seven TO the string of characters p q r s and digit character eight TO the string of characters t u v and digit character nine TO the string of characters w x y z  
        FUNCTION backtrack(combination next_digits)  
            IF the LENGTH OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string mapped by the element at position one of next_digits in phone_map  
                    CALL backtrack WITH the concatenation of combination and letter AND the substring from position two TO the end of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO an empty list  
        CALL backtrack WITH the empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
