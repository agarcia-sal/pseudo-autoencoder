CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a double ended queue initialized with a tuple containing zero as position one as speed and zero as steps  
        SET visited TO a set containing a tuple of zero as position and one as speed  
        
        WHILE the queue is not empty  
            REMOVE and RETURN the element from the front of the queue and ASSIGN position speed and steps from that element  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF a tuple of next_position and next_speed NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD a tuple of next_position and next_speed TO visited  
                APPEND a tuple of next_position next_speed and steps PLUS one TO queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF a tuple of position and next_speed NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD a tuple of position and next_speed TO visited  
                APPEND a tuple of position next_speed and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
