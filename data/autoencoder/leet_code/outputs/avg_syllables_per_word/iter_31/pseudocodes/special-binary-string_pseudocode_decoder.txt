CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each element i WITH value char IN s USING enumeration  
                IF char EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one PLUS the result of calling dfs WITH the substring of s from position start PLUS one TO position i PLUS the character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            SORT specials IN descending lexicographical order  
            RETURN the concatenation of all elements in specials  
        END FUNCTION  
        RETURN the result of calling dfs WITH s  
    END FUNCTION  
END CLASS
