CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            SET difference_in_first_coordinate TO element at position zero of p1 MINUS element at position zero of p2  
            SET difference_in_second_coordinate TO element at position one of p1 MINUS element at position one of p2  
            SET square_of_first_difference TO difference_in_first_coordinate MULTIPLIED BY difference_in_first_coordinate  
            SET square_of_second_difference TO difference_in_second_coordinate MULTIPLIED BY difference_in_second_coordinate  
            SET sum_of_squares TO square_of_first_difference PLUS square_of_second_difference  
            RETURN square root of sum_of_squares  
        END FUNCTION  
  
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance(p1, p2)  
            IF d GREATER THAN  two MULTIPLIED BY r  
                RETURN None  
            END IF  
            SET a TO square root of (r MULTIPLIED BY r MINUS (d DIVIDED BY two) MULTIPLIED BY (d DIVIDED BY two))  
            SET midpoint_first_coordinate TO element at position zero of p1 PLUS element at position zero of p2 DIVIDED BY two  
            SET midpoint_second_coordinate TO element at position one of p1 PLUS element at position one of p2 DIVIDED BY two  
            SET dx TO (element at position one of p2 MINUS element at position one of p1) MULTIPLIED BY a DIVIDED BY d  
            SET dy TO (element at position zero of p2 MINUS element at position zero of p1) MULTIPLIED BY a DIVIDED BY d  
            SET first_center TO midpoint_first_coordinate MINUS dx  
            SET second_center TO midpoint_second_coordinate PLUS dy  
            SET third_center TO midpoint_first_coordinate PLUS dx  
            SET fourth_center TO midpoint_second_coordinate MINUS dy  
            RETURN list containing the pair of first_center AND second_center and the pair of third_center AND fourth_center  
        END FUNCTION  
  
        FUNCTION is_inside(circle, point, r)  
            SET dist TO distance(circle, point)  
            IF dist LESS THAN OR EQUAL TO r  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
  
        SET n TO the LENGTH OF darts  
        IF n EQUALS one  
            RETURN one  
        END IF  
  
        SET max_count TO one  
  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center(element at position i of darts, element at position j of darts, r)  
                IF centers NOT EQUALS None  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside(center, dart, r)  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
  
        RETURN max_count  
    END FUNCTION  
END CLASS
