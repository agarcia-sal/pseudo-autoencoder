CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MODULO TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS seven  
        
        IF primeFactors LESS THAN OR EQUAL TO three THEN  
            RETURN primeFactors  
        END IF  
        
        SET number_of_threes TO primeFactors DIVIDED BY three  
        SET remainder_after_division TO primeFactors MODULO three  
        
        IF remainder_after_division EQUALS zero THEN  
            RETURN the result of raising three to the power of number_of_threes modulo MODULO  
        ELSE IF remainder_after_division EQUALS one THEN  
            RETURN the result of raising three to the power of number_of_threes MINUS one modulo MODULO MULTIPLIED BY four modulo MODULO  
        ELSE  
            RETURN the result of raising three to the power of number_of_threes modulo MODULO MULTIPLIED BY two modulo MODULO  
        END IF  
    END FUNCTION  
END CLASS
