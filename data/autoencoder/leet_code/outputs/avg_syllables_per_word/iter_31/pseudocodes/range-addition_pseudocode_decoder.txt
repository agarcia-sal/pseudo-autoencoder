CLASS Solution  
    FUNCTION getModifiedArray(length updates)  
        SET diff TO a list of zeros with the size equal to length PLUS one  
        FOR each triple startIdx endIdx inc IN updates  
            INCREMENT element at position startIdx of diff BY inc  
            IF element at position endIdx PLUS one LESS THAN length THEN  
                DECREMENT element at position endIdx PLUS one of diff BY inc  
            END IF  
        END FOR  
        SET arr TO a list of zeros with the size equal to length  
        SET element at position zero of arr TO element at position zero of diff  
        FOR index i FROM one TO length MINUS one  
            SET element at position i of arr TO element at position i MINUS one of arr PLUS element at position i of diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
