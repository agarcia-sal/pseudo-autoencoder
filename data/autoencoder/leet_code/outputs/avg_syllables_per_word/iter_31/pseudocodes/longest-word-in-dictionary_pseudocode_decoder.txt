CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY the LENGTH OF each element AND THEN BY each element IN alphabetical order  
        SET valid_words TO an empty set  
        SET longest_word TO an empty string  
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the substring of word from the first position TO the position BEFORE the last position IS IN valid_words  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
