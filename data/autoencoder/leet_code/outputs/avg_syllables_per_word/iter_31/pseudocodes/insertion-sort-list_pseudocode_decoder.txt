CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO the LENGTH OF values  
        SET node TO new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN false  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND the result of calling is_same_list with the element next of p1 AND the element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self head)  
        IF head EQUALS none OR the element next of head EQUALS none THEN  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with the value MINUS five thousand one  
        SET the next element of dummy TO head  
        SET current TO the element next of head  
        SET the next element of head TO none  
        WHILE current IS NOT none DO  
            SET next_node TO the element next of current  
            SET prev TO dummy  
            WHILE the element next of prev IS NOT none AND the element val of the element next of prev LESS THAN the element val of current DO  
                SET prev TO the element next of prev  
            END WHILE  
            SET the next element of current TO the element next of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the element next of dummy  
    END FUNCTION  
END CLASS
