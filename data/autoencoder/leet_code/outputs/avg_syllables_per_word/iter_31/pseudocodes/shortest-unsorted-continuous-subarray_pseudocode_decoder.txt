CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_list TO nums sorted in ascending order  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO length of nums MINUS one  
            IF element at position index of nums NOT EQUALS element at position index of sorted_list  
                IF start EQUALS negative one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        RETURN end MINUS start PLUS one  
    END FUNCTION  
END CLASS
