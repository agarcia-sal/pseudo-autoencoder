CLASS Solution  
    FUNCTION consecutiveNumbersSum WITH PARAMETER n  
        SET count TO zero  
        FOR k FROM one TO n PLUS one MINUS one  
            IF two MULTIPLIED BY n MODULO k EQUALS zero  
                SET m TO open parenthesis two MULTIPLIED BY n DIVIDED BY k MINUS k PLUS one close parenthesis DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUALS m CONVERTED TO integer  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
