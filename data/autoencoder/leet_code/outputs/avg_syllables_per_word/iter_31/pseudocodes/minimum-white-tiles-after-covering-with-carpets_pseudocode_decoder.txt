CLASS Solution  
    FUNCTION minimumWhiteTiles WITH PARAMETERS floor numCarpets carpetLen  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list with n PLUS one rows and numCarpets PLUS one columns FILLED WITH zero  
        
        FOR i FROM one TO n  
            SET element at position i AND zero OF dp TO element at position i MINUS one AND zero OF dp PLUS  
            IF element at position i MINUS one OF floor EQUALS character one THEN one ELSE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET element at position i AND j OF dp TO element at position i MINUS one AND j OF dp PLUS  
                IF element at position i MINUS one OF floor EQUALS character one THEN one ELSE zero  
                SET element at position i AND j OF dp TO THE MINIMUM OF element at position i AND j OF dp AND  
                element at position THE MAXIMUM OF zero AND i MINUS carpetLen AND j MINUS one OF dp  
            END FOR  
        END FOR  
        
        RETURN element at position n AND numCarpets OF dp  
    END FUNCTION  
END CLASS
