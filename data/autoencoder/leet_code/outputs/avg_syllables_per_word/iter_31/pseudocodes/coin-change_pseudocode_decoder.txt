CLASS Solution
    FUNCTION coinChange(coins, amount)
        SET dp TO list of the value amount PLUS one repeated the number of times amount PLUS one
        SET the element at position zero of dp TO zero
        FOR each a FROM one TO amount
            FOR each coin IN coins
                IF a MINUS coin GREATER THAN OR EQUAL TO zero
                    SET the element at position a of dp TO the smaller value BETWEEN the element at position a of dp AND the element at position a MINUS coin of dp PLUS one
                END IF
            END FOR
        END FOR
        IF the element at position amount of dp NOT EQUALS amount PLUS one
            RETURN the element at position amount of dp
        ELSE
            RETURN minus one
        END IF
    END FUNCTION
END CLASS
