CLASS Solution  
    FUNCTION exclusiveTime(n parameter, logs parameter)  
        SET exclusive_time TO a new list of zeros with length equal to n parameter  
        SET stack TO an empty list  
        
        FOR each log IN logs parameter  
            SET function_id text part TO the part of log text before the first colon  
            SET action text part TO the part of log text after the first colon and before the second colon  
            SET timestamp text part TO the part of log text after the second colon  
            CONVERT function_id text part TO integer and ASSIGN TO function_id  
            CONVERT timestamp text part TO integer and ASSIGN TO timestamp  
            
            IF action text part EQUALS the word start  
                IF stack is not empty  
                    SET previous_function_id and previous_timestamp TO the last element of stack  
                    INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp  
                END IF  
                APPEND a pair consisting of function_id and timestamp TO stack  
            ELSE  
                SET previous_function_id and previous_timestamp TO the element removed from the end of stack  
                INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp PLUS one  
                IF stack is not empty  
                    SET the last element of stack TO a pair consisting of the first part of the last element of stack and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
