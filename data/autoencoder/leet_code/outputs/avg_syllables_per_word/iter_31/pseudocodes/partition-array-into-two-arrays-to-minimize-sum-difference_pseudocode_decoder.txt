CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO THE LENGTH OF nums DIVIDED BY two  
        SET total_sum TO THE SUM OF nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO infinity  
        
        SET left_half TO THE ELEMENTS FROM THE BEGINNING TO POSITION n MINUS ONE OF nums  
        SET right_half TO THE ELEMENTS FROM POSITION n TO THE END OF nums  
        
        SET left_sums TO A LIST CONTAINING EMPTY DICTIONARIES REPEATED n PLUS one TIMES  
        SET right_sums TO A LIST CONTAINING EMPTY DICTIONARIES REPEATED n PLUS one TIMES  
        
        FOR index i FROM zero TO n INCLUSIVE  
            FOR EACH combination OF ELEMENTS OF left_half TAKEN i AT A TIME  
                SET subset_sum TO THE SUM OF combination  
                IF subset_sum IS NOT A KEY IN left_sums AT POSITION i THEN  
                    SET THE VALUE OF subset_sum IN left_sums AT POSITION i TO zero  
                END IF  
                INCREMENT THE VALUE OF subset_sum IN left_sums AT POSITION i BY one  
            END FOR  
            
            FOR EACH combination OF ELEMENTS OF right_half TAKEN i AT A TIME  
                SET subset_sum TO THE SUM OF combination  
                IF subset_sum IS NOT A KEY IN right_sums AT POSITION i THEN  
                    SET THE VALUE OF subset_sum IN right_sums AT POSITION i TO zero  
                END IF  
                INCREMENT THE VALUE OF subset_sum IN right_sums AT POSITION i BY one  
            END FOR  
        END FOR  
        
        FOR index i FROM zero TO n INCLUSIVE  
            SET left_values TO THE SORTED KEYS OF left_sums AT POSITION i  
            SET right_values TO THE SORTED KEYS OF right_sums AT POSITION n MINUS i  
            
            SET j TO zero  
            SET k TO THE LENGTH OF right_values MINUS one  
            WHILE j IS LESS THAN THE LENGTH OF left_values AND k IS GREATER THAN OR EQUAL TO zero  
                SET current_sum TO THE ELEMENT AT POSITION j OF left_values PLUS THE ELEMENT AT POSITION k OF right_values  
                SET min_diff TO THE MINIMUM OF min_diff AND THE ABSOLUTE VALUE OF total_sum MINUS two MULTIPLIED BY current_sum  
                
                IF current_sum IS LESS THAN target THEN  
                    INCREMENT j BY one  
                ELSE IF current_sum IS GREATER THAN target THEN  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
