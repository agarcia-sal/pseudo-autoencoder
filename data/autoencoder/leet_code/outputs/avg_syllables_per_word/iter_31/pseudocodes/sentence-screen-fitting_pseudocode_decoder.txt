CLASS Solution  
    FUNCTION wordsTyping(sentence SENTENCE OF STRINGS rows INTEGER cols INTEGER) RETURNS INTEGER  
        SET s TO the concatenation of all elements of sentence SEPARATED BY a single space PLUS a single space character  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        FOR each underscore IN the range FROM zero TO rows MINUS one  
            INCREMENT position BY cols  
            IF the element at position MODULO total_length OF s EQUALS a single space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at position MINUS one MODULO total_length OF s NOT EQUALS a single space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        SET result TO position DIVIDED BY total_length  
        RETURN result  
    END FUNCTION  
END CLASS
