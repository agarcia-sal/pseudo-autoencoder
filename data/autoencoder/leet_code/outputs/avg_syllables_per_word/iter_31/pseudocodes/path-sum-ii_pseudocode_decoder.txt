CLASS TreeNode  
    FUNCTION __init__ (val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO new deque  
    CALL append element root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO first element removed from queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value equal to element at position i of values  
            CALL append element node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value equal to element at position i of values  
            CALL append element node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with p.left and q.left AND the result of is_same_tree called with p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pathSum(root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node EQUALS none  
                RETURN  
            END IF  
            CALL append element node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left EQUALS none AND node.right EQUALS none AND current_sum EQUALS targetSum  
                CALL append a copy of path TO result  
            END IF  
            CALL dfs with node.left, current_sum, path  
            CALL dfs with node.right, current_sum, path  
            CALL remove last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs with root, zero, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
