CLASS Solution  
    FUNCTION findMaximizedCapital(k w profits capital)  
        SET projects TO a list of pairs consisting of element at position j of capital and element at position j of profits FOR index j FROM zero TO the LENGTH OF capital MINUS one  
        SORT projects BY the first element in each pair in ascending order  
        SET max_heap TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element at position i of projects LESS THAN OR EQUAL TO w  
                ADD the NEGATION OF the second element at position i of projects TO max_heap using a heap insertion operation  
                INCREMENT i BY one  
            END WHILE  
            IF the LENGTH OF max_heap EQUALS zero  
                BREAK the loop  
            END IF  
            REMOVE the smallest element from max_heap using a heap removal operation and NEGATE it and INCREMENT w BY that value  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
