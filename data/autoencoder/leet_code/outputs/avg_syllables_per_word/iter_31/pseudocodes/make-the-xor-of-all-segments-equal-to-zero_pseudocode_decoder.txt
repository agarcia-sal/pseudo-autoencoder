CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO an empty mapping from number TO an empty list  
        FOR each index i FROM zero TO the length of nums MINUS one  
            APPEND element at position i of nums TO the list corresponding to the remainder of i DIVIDED BY k in groups  
        END FOR  
        
        SET freq TO a list containing k mappings from number TO zero  
        SET min_size TO a list containing k instances of infinity  
        FOR each index i FROM zero TO k MINUS one  
            FOR each number num IN the list at key i in groups  
                INCREMENT the count of num in freq at position i BY one  
            END FOR  
            SET element at position i of min_size TO the length of the list at key i in groups  
        END FOR  
        
        SET dp TO a list of size two raised to the power of ten with all elements equal to infinity  
        SET element at position zero of dp TO zero  
        
        FOR each index i FROM zero TO k MINUS one  
            SET new_dp TO a list of size two raised to the power of ten with all elements equal to infinity  
            SET min_dp TO the minimum value in dp  
            FOR each index j FROM zero TO two raised to the power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each pair of number num and count in freq at position i  
                    SET candidate_value TO element at position of the bitwise exclusive OR of j and num in dp PLUS element at position i of min_size MINUS count  
                    IF candidate_value LESS THAN element at position j of new_dp THEN  
                        SET element at position j of new_dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
