CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing the value of negative one  
        SET max_length TO zero  
        FOR each position i and character char IN the enumeration of string s  
            IF character char EQUALS an opening parenthesis  
                APPEND position i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND position i TO stack  
                ELSE  
                    SET max_length TO the greater value BETWEEN max_length AND position i MINUS the last element AT the end OF stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
