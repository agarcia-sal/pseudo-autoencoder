CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count GREATER THAN x  
        END FUNCTION  
        RETURN the POSITION OF the FIRST element IN the RANGE FROM zero TO the LENGTH OF nums MINUS one WHERE f RETURNS True  
    END FUNCTION  
END CLASS
