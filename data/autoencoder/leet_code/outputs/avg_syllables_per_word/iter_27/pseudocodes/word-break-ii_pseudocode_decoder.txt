CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO a set containing all elements of wordDict  
        SET memo TO an empty dictionary  
        
        FUNCTION backtrack(start)  
            IF start EXISTS as a key in memo THEN  
                RETURN the value at key start in memo  
            END IF  
            IF start EQUALS the length of s THEN  
                RETURN a list containing an empty string element  
            END IF  
            
            SET res TO an empty list  
            FOR end FROM start PLUS one TO the length of s PLUS one  
                SET word TO the substring of s from position start TO position end MINUS one  
                IF word EXISTS in wordSet THEN  
                    FOR each sentence IN backtrack called with argument end  
                        IF sentence IS NOT an empty string THEN  
                            APPEND the concatenation of word PLUS a space PLUS sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            ASSIGN memo at key start TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack called with argument zero  
    END FUNCTION  
END CLASS
