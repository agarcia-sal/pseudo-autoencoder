CLASS Solution  
    FUNCTION findRadius(houses LIST OF INTEGER heaters LIST OF INTEGER)  
        CALL sort ON houses  
        CALL sort ON heaters  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN THE LENGTH OF heaters MINUS one AND THE ABSOLUTE VALUE OF element at position j of heaters MINUS house GREATER THAN OR EQUAL TO THE ABSOLUTE VALUE OF element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO THE GREATER OF min_radius AND THE ABSOLUTE VALUE OF element at position j of heaters MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
