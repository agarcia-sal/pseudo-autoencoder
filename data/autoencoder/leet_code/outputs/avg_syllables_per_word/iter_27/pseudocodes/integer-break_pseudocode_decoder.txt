CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO a list of zero repeated n PLUS one times  
        SET element at position one of dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET candidate_one TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET candidate_two TO j MULTIPLIED BY i MINUS j  
                IF max_product LESS THAN candidate_one OR max_product LESS THAN candidate_two  
                    IF candidate_one GREATER THAN OR EQUAL TO candidate_two  
                        SET max_product TO candidate_one  
                    ELSE  
                        SET max_product TO candidate_two  
                    END IF  
                END IF  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
