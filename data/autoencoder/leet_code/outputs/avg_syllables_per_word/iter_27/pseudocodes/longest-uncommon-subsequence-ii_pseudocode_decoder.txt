CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s t)  
            SET iterator TO the iterator of t  
            FOR each character IN s  
                IF the character is not found in the iterator continuing from current position THEN  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT the list strs BY the negative of the length of each element AND THEN BY lexicographical order of each element  
        
        FOR index i FROM zero TO the length of strs MINUS one  
            SET word TO the element at position i of strs  
            SET unique TO true  
            FOR index j FROM zero TO the length of strs MINUS one  
                SET other TO the element at position j of strs  
                IF i NOT EQUALS j AND is_subsequence of word IN other THEN  
                    SET unique TO false  
                    BREAK the loop  
                END IF  
            END FOR  
            IF unique EQUALS true THEN  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN the negative of one  
    END FUNCTION  
END CLASS
