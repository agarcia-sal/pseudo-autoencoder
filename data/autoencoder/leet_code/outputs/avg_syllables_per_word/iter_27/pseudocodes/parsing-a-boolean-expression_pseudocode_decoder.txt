CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character c IN expression  
            IF character c EQUALS one of the characters representing true or false or the logical not operator or logical and operator or logical or operator  
                APPEND character c TO stk  
            ELSE IF character c EQUALS the closing parenthesis character  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE the last element of stk EQUALS a character representing true or false  
                    IF the last element of stk EQUALS the character representing true  
                        INCREMENT count_true BY one  
                    END IF  
                    IF the last element of stk EQUALS the character representing false  
                        INCREMENT count_false BY one  
                    END IF  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET operator TO the last element of stk  
                REMOVE the last element FROM stk  
                IF operator EQUALS the logical not operator  
                    IF count_false GREATER THAN zero  
                        SET character c TO the character representing true  
                    ELSE  
                        SET character c TO the character representing false  
                    END IF  
                ELSE IF operator EQUALS the logical and operator  
                    IF count_false GREATER THAN zero  
                        SET character c TO the character representing false  
                    ELSE  
                        SET character c TO the character representing true  
                    END IF  
                ELSE IF operator EQUALS the logical or operator  
                    IF count_true GREATER THAN zero  
                        SET character c TO the character representing true  
                    ELSE  
                        SET character c TO the character representing false  
                    END IF  
                END IF  
                APPEND character c TO stk  
            END IF  
        END FOR  
        RETURN whether the first element of stk EQUALS the character representing true  
    END FUNCTION  
END CLASS
