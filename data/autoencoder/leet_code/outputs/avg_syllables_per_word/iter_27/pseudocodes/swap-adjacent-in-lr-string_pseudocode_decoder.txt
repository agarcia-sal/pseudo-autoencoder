CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO empty list  
        FOR each index AND character IN start  
            IF character NOT EQUALS the character representing X  
                APPEND the pair of character AND index TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO empty list  
        FOR each index AND character IN end  
            IF character NOT EQUALS the character representing X  
                APPEND the pair of character AND index TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair of character one AND index one IN start_filtered AND pair of character two AND index two IN end_filtered  
            IF character one NOT EQUALS character two  
                RETURN false  
            END IF  
            IF character one EQUALS the character representing L AND index one LESS THAN index two  
                RETURN false  
            END IF  
            IF character one EQUALS the character representing R AND index one GREATER THAN index two  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
