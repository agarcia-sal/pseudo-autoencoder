CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        CREATE empty mapping d FROM string TO integer  
        SET index i TO zero  
        FOR each skill s IN req_skills  
            SET d AT key s TO i  
            INCREMENT i BY one  
        END FOR  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        CREATE list p OF size n FILLED WITH zero  
        SET i TO zero  
        FOR each skill_list ss IN people  
            FOR each skill s IN ss  
                SET p AT position i TO the bitwise OR OF the current value of p AT position i AND the bitwise LEFT SHIFT OF one BY the value d AT key s  
            END FOR  
            INCREMENT i BY one  
        END FOR  
        CREATE list f OF size the power of two RAISED TO m FILLED WITH infinite value  
        CREATE list g OF size the power of two RAISED TO m FILLED WITH zero  
        CREATE list h OF size the power of two RAISED TO m FILLED WITH zero  
        SET f AT position zero TO zero  
        SET i TO zero  
        WHILE i LESS THAN the power of two RAISED TO m  
            IF f AT position i EQUALS infinite value  
                INCREMENT i BY one  
                CONTINUE TO next iteration of the loop  
            END IF  
            SET j TO zero  
            WHILE j LESS THAN n  
                IF the value f AT position i PLUS one LESS THAN the value f AT the bitwise OR OF i AND p AT position j  
                    SET f AT the bitwise OR OF i AND p AT position j TO the value f AT position i PLUS one  
                    SET g AT the bitwise OR OF i AND p AT position j TO j  
                    SET h AT the bitwise OR OF i AND p AT position j TO i  
                END IF  
                INCREMENT j BY one  
            END WHILE  
            INCREMENT i BY one  
        END WHILE  
        SET i TO the power of two RAISED TO m MINUS one  
        CREATE empty list ans  
        WHILE i NOT EQUALS zero  
            APPEND g AT position i TO ans  
            SET i TO h AT position i  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
