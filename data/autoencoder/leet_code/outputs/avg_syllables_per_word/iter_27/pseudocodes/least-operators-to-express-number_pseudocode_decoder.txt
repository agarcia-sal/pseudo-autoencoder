CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target_inner)  
            IF target_inner EQUALS zero  
                RETURN negative one  
            END IF  
            IF target_inner LESS THAN x  
                RETURN the minimum value BETWEEN two MULTIPLIED BY target_inner MINUS one AND two MULTIPLIED BY x MINUS target_inner  
            END IF  
            SET power TO one  
            WHILE x RAISED TO the power OF power LESS THAN OR EQUAL TO target_inner  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET result TO dp(target_inner MINUS x RAISED TO the power OF power) PLUS power  
            IF x RAISED TO the power OF power PLUS one MINUS target_inner LESS THAN target_inner  
                SET result TO the minimum value BETWEEN result AND dp(x RAISED TO the power OF power PLUS one MINUS target_inner) PLUS power PLUS one  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
