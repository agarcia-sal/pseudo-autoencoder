CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list of pairs of corresponding elements from difficulty and profit  
        SORT jobs by the first element in each pair in ascending order  
        SORT worker in ascending order  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND element at position zero of element at position job_index of jobs LESS THAN OR EQUAL TO ability  
                SET best_profit TO the GREATER THAN value BETWEEN best_profit AND element at position one of element at position job_index of jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
