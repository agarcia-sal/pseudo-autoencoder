CLASS Solution  
    FUNCTION minInterval(intervals WITH TYPE List of List of integer, queries WITH TYPE List of integer) RETURNS List of integer  
        SORT intervals IN ASCENDING ORDER BY the element at position zero of each element  
        SET sorted_queries TO the LIST of PAIRS composed of each element of queries WITH its corresponding position index in queries sorted by the first element of each pair in ASCENDING ORDER  
        SET min_heap TO an empty LIST  
        SET result TO a LIST of length equal to the LENGTH OF queries FILLED WITH the integer negative one  
        SET interval_index TO zero  
        FOR each pair query AND original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND the element at position zero of the element at position interval_index of intervals LESS THAN OR EQUAL TO query  
                SET start TO the element at position zero of the element at position interval_index of intervals  
                SET end TO the element at position one of the element at position interval_index of intervals  
                SET interval_size TO end MINUS start PLUS one  
                APPEND the PAIR composed of interval_size AND end TO min_heap using heap push operation  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE the LENGTH OF min_heap GREATER THAN zero AND the element at position one of the element at position zero of min_heap LESS THAN query  
                REMOVE the element at the top of min_heap using heap pop operation  
            END WHILE  
            IF the LENGTH OF min_heap GREATER THAN zero  
                SET the element at position original_index of result TO the element at position zero of the element at position zero of min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
