CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO the LENGTH OF temperatures  
        INITIALIZE answer AS a list of zeros with length n  
        INITIALIZE stack AS an empty list  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS NOT zero AND the element at position i of temperatures IS GREATER THAN the element at position the last element of stack of temperatures  
                SET prev_index TO the last element of stack  
                REMOVE the last element FROM stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
