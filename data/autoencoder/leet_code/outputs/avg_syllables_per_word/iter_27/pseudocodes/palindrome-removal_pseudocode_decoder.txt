CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO a list of lists of zeroes with n lists each containing n zeroes  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i of element at position i of dp TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i of arr EQUALS element at position i PLUS one of arr  
                SET element at position i of element at position i PLUS one of dp TO one  
            ELSE  
                SET element at position i of element at position i PLUS one of dp TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of arr EQUALS element at position j of arr  
                    SET element at position i of element at position j of dp TO element at position i PLUS one of element at position j MINUS one of dp  
                ELSE  
                    SET element at position i of element at position j of dp TO infinity  
                    FOR k FROM i TO j MINUS one  
                        SET element at position i of element at position j of dp TO the MINIMUM OF element at position i of element at position j of dp AND element at position i of element at position k of dp PLUS element at position k PLUS one of element at position j of dp  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero of element at position n MINUS one of dp  
    END FUNCTION  
END CLASS
