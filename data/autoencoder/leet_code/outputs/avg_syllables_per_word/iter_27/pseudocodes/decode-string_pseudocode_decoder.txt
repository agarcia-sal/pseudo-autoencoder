CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char IS a digit character  
                SET d TO the concatenation of d WITH char  
            ELSE IF char IS an alphabetic character  
                SET a TO the concatenation of a WITH char  
            ELSE IF char IS equal to the left square bracket character  
                APPEND a AND the integer value of d AS a pair TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char IS equal to the right square bracket character  
                SET p AND n TO the last pair REMOVED FROM stack  
                SET a TO the concatenation of p WITH a REPEATED n TIMES  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
