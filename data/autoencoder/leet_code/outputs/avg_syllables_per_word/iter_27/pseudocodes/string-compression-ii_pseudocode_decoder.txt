CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(s, k)  
        FUNCTION dp(i, k_value, prev_char, prev_count)  
            IF k_value LESS THAN zero  
                RETURN infinity value  
            END IF  
            
            IF i EQUALS the length of s  
                RETURN zero  
            END IF  
            
            IF element at position i of s EQUALS prev_char  
                IF prev_count EQUALS one OR prev_count EQUALS nine OR prev_count EQUALS ninety nine  
                    SET increment TO one  
                ELSE  
                    SET increment TO zero  
                END IF  
                RETURN increment PLUS dp(i PLUS one, k_value, prev_char, prev_count PLUS one)  
            ELSE  
                SET keep TO one PLUS dp(i PLUS one, k_value, element at position i of s, one)  
                SET delete TO dp(i PLUS one, k_value MINUS one, prev_char, prev_count)  
                IF keep LESS THAN delete  
                    RETURN keep  
                ELSE  
                    RETURN delete  
                END IF  
            END IF  
        END FUNCTION  
        
        RETURN dp(zero, k, empty string, zero)  
    END FUNCTION  
END CLASS
