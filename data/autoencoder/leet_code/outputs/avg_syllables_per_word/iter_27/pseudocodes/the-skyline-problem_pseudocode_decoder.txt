CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each element containing left right height IN buildings  
            APPEND a tuple containing left and the negative of height and right TO events  
            APPEND a tuple containing right and zero and zero TO events  
        END FOR  
        SORT events by the natural order of the tuples  
        SET result TO a list containing a list containing zero and zero  
        SET max_heap TO a list containing a tuple containing zero and positive infinity  
        FOR each element containing x negH r IN events  
            WHILE the second element of the first tuple in max_heap is LESS THAN OR EQUAL TO x  
                REMOVE the first tuple from max_heap using heap pop operation  
            END WHILE  
            IF negH NOT EQUALS zero  
                ADD the tuple containing negH and r TO max_heap using heap push operation  
            END IF  
            IF the second element of the last list in result NOT EQUALS the negative of the first element of the first tuple in max_heap  
                APPEND a list containing x and the negative of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        IF the first list in result EQUALS a list containing zero and zero  
            REMOVE the first list from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
