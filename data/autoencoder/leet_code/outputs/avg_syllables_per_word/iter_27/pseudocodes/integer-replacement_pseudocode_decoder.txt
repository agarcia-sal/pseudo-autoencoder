CLASS Solution  
    FUNCTION integerReplacement(n)  
        SET count TO zero  
        WHILE n NOT EQUALS one  
            IF n MODULO two EQUALS zero  
                SET n TO the integer result of n DIVIDED BY two  
            ELSE IF n EQUALS three  
                DECREMENT n BY one  
            ELSE IF (n PLUS one) MODULO four EQUALS zero  
                INCREMENT n BY one  
            ELSE  
                DECREMENT n BY one  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
