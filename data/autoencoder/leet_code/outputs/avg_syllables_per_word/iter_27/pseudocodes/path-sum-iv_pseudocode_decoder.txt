CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the integer part of num DIVIDED BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN value TO the element at key depth AND pos of tree  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at key depth AND pos of tree if it exists otherwise none  
            IF current_value EQUALS none  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            
            SET left_child_depth TO depth PLUS one  
            SET left_child_pos TO pos MULTIPLIED BY two MINUS one  
            SET right_child_depth TO depth PLUS one  
            SET right_child_pos TO pos MULTIPLIED BY two  
            
            IF left_child_depth AND left_child_pos NOT IN tree AND right_child_depth AND right_child_pos NOT IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO the result of dfs called with left_child_depth, left_child_pos, new_path_sum  
            SET right_sum TO the result of dfs called with right_child_depth, right_child_pos, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN the result of dfs called with one, one, zero  
    END FUNCTION  
END CLASS
