CLASS Solution  
    FUNCTION validArrangement(pairs)  
        SET graph TO a new mapping from element to list  
        SET out_degree TO a new mapping from element to count  
        SET in_degree TO a new mapping from element to count  
        
        FOR each pair consisting of u and v IN pairs  
            APPEND element v TO the list at key u IN graph  
            INCREMENT out_degree at key u BY one  
            INCREMENT in_degree at key v BY one  
        END FOR  
        
        SET start_node TO no value  
        FOR each node IN graph  
            IF out_degree at key node EQUALS in_degree at key node PLUS one  
                SET start_node TO node  
                BREAK the loop  
            END IF  
        END FOR  
        IF start_node HAS no value  
            SET start_node TO any element FROM graph keys  
        END IF  
        
        SET path TO an empty list  
        SET stack TO a list containing start_node  
        
        WHILE stack is not empty  
            SET u TO last element in stack  
            IF the list at key u IN graph is not empty  
                REMOVE the last element FROM the list at key u IN graph AND SET it TO v  
                APPEND v TO stack  
            ELSE  
                REMOVE the last element FROM stack AND APPEND it TO path  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO an empty list  
        FOR index FROM zero TO length of path MINUS one  
            SET next_index TO index PLUS one  
            APPEND a list containing element at position index of path AND element at position next_index of path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
