CLASS Solution  
    FUNCTION minmaxGasDist(stations LIST OF INTEGER k INTEGER)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR i FROM one TO the LENGTH OF stations MINUS one  
                SET distance_between_stations TO element at position i of stations MINUS element at position i MINUS one of stations  
                SET additional_needed TO distance_between_stations DIVIDED BY max_dist  
                INCREMENT additional_stations BY the INTEGER PART OF additional_needed  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) IS TRUE  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
