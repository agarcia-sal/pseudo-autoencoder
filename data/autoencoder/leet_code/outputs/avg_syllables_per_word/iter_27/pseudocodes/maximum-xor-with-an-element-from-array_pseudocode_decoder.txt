CLASS Solution  
    FUNCTION maximizeXor(nums AS List of integers, queries AS List of List of integers) RETURNS List of integers  
        PERFORM sort nums in ascending order  
        
        SET sorted_queries TO the list of pairs of index and query from enumerating queries  
        PERFORM sort sorted_queries by the second element of the query at position one in ascending order  
        
        SET result TO a new list filled with minus one repeated the LENGTH OF queries times  
        
        SET trie TO an empty collection of mappings  
        
        SET num_index TO zero  
        
        FOR each pair of query_index and query where query is a pair xi and mi IN sorted_queries  
            
            WHILE num_index is less than the LENGTH OF nums AND the element at position num_index of nums is less than or equal to mi  
                SET num TO the element at position num_index of nums  
                SET node TO trie  
                
                FOR i FROM thirty one DOWN TO zero  
                    SET bit TO the value of bitwise right shift of num by i positions MODULO two  
                    IF bit is not present as a key in node  
                        SET element at key bit in node TO an empty collection of mappings  
                    END IF  
                    SET node TO the element at key bit in node  
                END FOR  
                
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie is empty  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            
            FOR i FROM thirty one DOWN TO zero  
                SET bit TO the value of bitwise right shift of xi by i positions MODULO two  
                SET toggle_bit TO one MINUS bit  
                
                IF toggle_bit is present as a key in node  
                    SET max_xor TO max_xor OR bitwise left shift of one by i positions  
                    SET node TO the element at key toggle_bit in node  
                ELSE  
                    SET node TO the element at key bit in node  
                END IF  
            END FOR  
            
            SET the element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
