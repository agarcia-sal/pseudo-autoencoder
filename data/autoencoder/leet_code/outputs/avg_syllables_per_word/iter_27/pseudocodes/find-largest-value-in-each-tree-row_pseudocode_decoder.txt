CLASS TreeNode  
    FUNCTION __init__(self val zero left None right None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            ASSIGN a new instance of TreeNode with the element at position i of values TO node.left  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            ASSIGN a new instance of TreeNode with the element at position i of values TO node.right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments the left property of p and the left property of q AND the result of is_same_tree with arguments the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET queue TO a double ended queue containing root  
        SET largest_values TO an empty list  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each index FROM one TO level_size  
                SET node TO the element removed from the front of queue  
                IF max_value LESS THAN the val property of node  
                    SET max_value TO the val property of node  
                END IF  
                IF the left property of node NOT EQUALS none  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node NOT EQUALS none  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
