CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the PAIR of index AND current_sum IS IN memo  
                RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
            END IF  
            SET positive TO THE RESULT OF dfs WITH index PLUS one AND current_sum PLUS element at position index of nums AND memo  
            SET negative TO THE RESULT OF dfs WITH index PLUS one AND current_sum MINUS element at position index of nums AND memo  
            SET the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
        END FUNCTION  
        RETURN dfs WITH zero AND zero AND an EMPTY COLLECTION  
    END FUNCTION  
END CLASS
