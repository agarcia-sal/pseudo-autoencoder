CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO a new empty set  
        FOR each element mine IN mines  
            ADD tuple conversion of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list with n rows and n columns where each element is a list of four zeros representing counts in four directions  
        
        SET directions TO a list containing four pairs representing right zero one left and up minus one zero directions correspondingly  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair consisting of i and j DOES NOT EXIST IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at position zero of dp at position i of dp at position j TO element at position zero of dp at position i of dp at position j MINUS one PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at position one of dp at position i of dp at position j TO element at position one of dp at position i MINUS one of dp at position j PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair consisting of i and j DOES NOT EXIST IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at position two of dp at position i of dp at position j TO element at position two of dp at position i of dp at position j PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at position three of dp at position i of dp at position j TO element at position three of dp at position i PLUS one of dp at position j PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the four elements of dp at position i of dp at position j  
                    SET result TO the maximum value between result and plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
