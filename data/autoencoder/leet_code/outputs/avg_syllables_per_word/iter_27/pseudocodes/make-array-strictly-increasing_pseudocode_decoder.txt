CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort on arr2 in ascending order  
        
        SET dp TO a dictionary with key negative one and value zero  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num and ops IN all key value pairs of dp  
                IF num GREATER THAN last_num  
                    IF new_dp DOES NOT HAVE key num OR value at key num GREATER THAN ops  
                        SET value at key num in new_dp TO ops  
                    END IF  
                END IF  
                
                SET idx TO result of a binary search to find the position to insert last_num to the right in arr2  
                IF idx LESS THAN length of arr2  
                    SET candidate TO element at position idx of arr2  
                    IF new_dp DOES NOT HAVE key candidate OR value at key candidate GREATER THAN ops PLUS one  
                        SET value at key candidate in new_dp TO ops PLUS one  
                    END IF  
                END IF  
            END FOR  
            
            IF new_dp IS EMPTY  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
