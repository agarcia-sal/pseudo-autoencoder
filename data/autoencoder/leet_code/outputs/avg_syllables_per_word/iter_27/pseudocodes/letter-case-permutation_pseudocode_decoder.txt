CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND all elements concatenated as a single string of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH parameter the value of i PLUS one  
            IF element at position i of t IS an alphabetic character  
                SET element at position i of t TO the character obtained by toggling the case of the character at position i of t using ASCII case difference  
                CALL dfs WITH parameter the value of i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO the list of characters of s  
        SET ans TO an empty list  
        CALL dfs WITH parameter zero  
        RETURN ans  
    END FUNCTION  
END CLASS
