CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET list_of_start_and_index_pairs TO sorted list of pairs of element at position zero of interval AND index of interval FOR each index AND interval IN enumeration of intervals  
        SET result_list TO empty list  
        FOR each pair_of_ignored_and_end IN intervals  
            SET end_value TO element at position one of pair_of_ignored_and_end  
            SET search_index TO position found by binary search to find leftmost position to insert a pair with end_value in list_of_start_and_index_pairs  
            IF search_index LESS THAN the LENGTH OF list_of_start_and_index_pairs  
                APPEND element at position one of element at position search_index of list_of_start_and_index_pairs TO result_list  
            ELSE  
                APPEND negative one TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
