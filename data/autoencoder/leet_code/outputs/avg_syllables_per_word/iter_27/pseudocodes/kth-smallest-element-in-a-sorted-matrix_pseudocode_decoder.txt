CLASS Solution  
    FUNCTION kthSmallest(matrix WITH TYPE list of list of integer, k WITH TYPE integer) RETURNS integer  
        FUNCTION countLessEqual(mid WITH TYPE integer) RETURNS integer  
            SET count TO zero  
            SET row TO the length of matrix MINUS one  
            SET col TO zero  
            WHILE row GREATER THAN OR EQUAL TO zero AND col LESS THAN the length of matrix  
                IF the element at position row of matrix AT position col LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  
          
        SET n TO the length of matrix  
        SET low TO the element at position zero of matrix AT position zero  
        SET high TO the element at position n MINUS one of matrix AT position n MINUS one  
          
        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two  
            IF countLessEqual(mid) LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
          
        RETURN low  
    END FUNCTION  
END CLASS
