CLASS Solution  
    FUNCTION pyramidTransition(bottom, allowed)  
        SET allowed_map TO an empty map where each key maps to a map from key to a list of elements  
        FOR each rule IN allowed  
            SET left TO element at position one of rule  
            SET right TO element at position two of rule  
            SET top TO element at position three of rule  
            APPEND top TO the list at allowed_map at key left at key right  
        END FOR  
        
        FUNCTION can_build_pyramid(current_bottom, current_top)  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN True  
            END IF  
            
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of calling can_build_pyramid with current_top and an empty list  
            END IF  
            
            SET position TO the LENGTH OF current_top PLUS one  
            SET left TO element at position position of current_bottom  
            SET right TO element at position position PLUS one of current_bottom  
            IF left IS a key in allowed_map AND right IS a key in allowed_map at key left  
                FOR each top IN the list at allowed_map at key left at key right  
                    IF calling can_build_pyramid with current_bottom and current_top concatenated with top RETURNS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
            RETURN False  
        END FUNCTION  
        
        RETURN the result of calling can_build_pyramid with bottom and an empty list  
    END FUNCTION  
END CLASS
