CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum value in nums  
        SET num_set TO the set of distinct elements in nums  
        SET count TO zero  
        
        FOR g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            FOR m FROM g TO max_num INCLUSIVE STEP g  
                IF element at position m IN num_set EXISTS  
                    SET gcd_val TO the greatest common divisor of gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        BREAK out of the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
