CLASS Solution  
    FUNCTION combinationSum3(k parameter, n parameter)  
        FUNCTION backtrack(start parameter, path parameter, target parameter, k parameter)  
            IF target parameter EQUALS zero AND k parameter EQUALS zero  
                APPEND a copy of path parameter TO result  
                RETURN  
            END IF  
            IF target parameter LESS THAN zero OR k parameter LESS THAN zero  
                RETURN  
            END IF  
            FOR index i FROM start parameter TO nine  
                APPEND index i TO path parameter  
                CALL backtrack WITH start parameter EQUAL TO index i PLUS one AND path parameter AND target parameter MINUS index i AND k parameter MINUS one  
                REMOVE the last element FROM path parameter  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH start parameter EQUAL TO one AND an empty list AS path parameter AND target parameter EQUAL TO n parameter AND k parameter  
        RETURN result  
    END FUNCTION  
END CLASS
