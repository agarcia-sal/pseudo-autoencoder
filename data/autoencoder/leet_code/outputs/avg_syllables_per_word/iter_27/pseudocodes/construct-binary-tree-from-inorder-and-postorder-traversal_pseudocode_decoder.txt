CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO None right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO element popped from left of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value attribute of p NOT EQUALS value attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left attribute of p left attribute of q) AND is_same_tree(right attribute of p right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self inorder postorder)  
        IF inorder EQUALS empty OR postorder EQUALS empty  
            RETURN None  
        END IF  
        SET root_val TO element removed from end of postorder  
        SET root TO new TreeNode with root_val  
        SET root_index TO the position of root_val in inorder  
        SET root.right TO the result of calling buildTree with elements from position root_index PLUS one TO end of inorder AND postorder  
        SET root.left TO the result of calling buildTree with elements from start TO position root_index MINUS one of inorder AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
