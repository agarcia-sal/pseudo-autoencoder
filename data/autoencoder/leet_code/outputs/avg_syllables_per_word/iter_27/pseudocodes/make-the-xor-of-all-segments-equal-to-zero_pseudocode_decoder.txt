CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO new mapping with default value empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO element at position index MODULO k of groups  
        END FOR  
        
        SET freq TO list of new mappings with default value zero for positions FROM zero TO k MINUS one  
        SET min_size TO list of infinity repeated k times  
        FOR i FROM zero TO k MINUS one  
            FOR each number IN element at position i of groups  
                INCREMENT element at position number of mapping at position i of freq BY one  
            END FOR  
            SET element at position i of min_size TO the LENGTH OF element at position i of groups  
        END FOR  
        
        SET dp TO list of infinity repeated two RAISED TO the POWER OF ten times  
        SET element at position zero of dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO list of infinity repeated two RAISED TO the POWER OF ten times  
            SET min_dp TO the minimum VALUE IN dp  
            FOR j FROM zero TO two RAISED TO the POWER OF ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each key_value_pair IN mapping at position i of freq  
                    SET num TO the key of key_value_pair  
                    SET count TO the value of key_value_pair  
                    SET element at position j of new_dp TO the minimum VALUE BETWEEN element at position j of new_dp AND element at position j XOR num of dp PLUS element at position i of min_size MINUS count  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
