CLASS TreeNode  
    FUNCTION __init__(self val EQUALS zero left EQUALS none right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE the element at the left end FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p the left attribute of q) AND is_same_tree(the right attribute of p the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self root)  
        IF root EQUALS none  
            RETURN  
        END IF  
        SET stack TO a list containing root  
        WHILE the LENGTH OF stack is greater than zero  
            REMOVE the last element FROM stack AND ASSIGN TO node  
            IF the right attribute of node IS NOT none  
                APPEND the right attribute of node TO stack  
            END IF  
            IF the left attribute of node IS NOT none  
                APPEND the left attribute of node TO stack  
            END IF  
            SET the left attribute of node TO none  
            IF the LENGTH OF stack is greater than zero  
                SET the right attribute of node TO the last element of stack  
            ELSE  
                SET the right attribute of node TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
