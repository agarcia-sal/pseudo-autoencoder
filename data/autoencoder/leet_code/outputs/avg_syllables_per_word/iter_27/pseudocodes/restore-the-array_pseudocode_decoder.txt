CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros of size n PLUS one  
        SET the element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF the character at position j of s NOT EQUALS zero AND the integer value of the substring from position j TO position i MINUS one of s GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET the element at position i of dp TO the element at position i of dp PLUS the element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
