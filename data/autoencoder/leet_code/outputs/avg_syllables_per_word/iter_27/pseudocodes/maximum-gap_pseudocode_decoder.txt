    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two  
            RETURN zero  
        END IF  
          
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zero of size n  
            SET count TO a list of zero of size ten  
              
            FOR i FROM zero TO n MINUS one  
                SET index TO the element at position i of arr DIVIDED BY exp  
                INCREMENT the element at position index MODULO ten of count BY one  
            END FOR  
              
            FOR i FROM one TO nine  
                SET the element at position i of count TO the element at position i of count PLUS the element at position i MINUS one of count  
            END FOR  
              
            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET index TO the element at position i of arr DIVIDED BY exp  
                SET the element at position the element at position index MODULO ten of count MINUS one of output TO the element at position i of arr  
                DECREMENT the element at position index MODULO ten of count BY one  
                DECREMENT i BY one  
            END WHILE  
              
            FOR i FROM zero TO n MINUS one  
                SET the element at position i of arr TO the element at position i of output  
            END FOR  
        END FUNCTION  
          
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element of arr  
              
            SET exp TO one  
            WHILE max1 DIVIDED BY exp GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
          
        CALL radixSort WITH nums  
          
        SET max_diff TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            SET max_diff TO the maximum of max_diff AND the element at position i of nums MINUS the element at position i MINUS one of nums  
        END FOR  
          
        RETURN max_diff  
    END FUNCTION  
END CLASS
