CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO TreeNode of element at position one of values  
    SET i TO one plus zero  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element at the front removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO TreeNode of the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO TreeNode of the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left q.left) AND is_same_tree(p.right q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double ended queue containing root  
        SET left_to_right TO True  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO a new empty double ended queue  
            FOR each number in the range from zero TO level_size MINUS one  
                SET node TO the element at the front removed from queue  
                IF left_to_right EQUALS True  
                    APPEND node.val TO current_level  
                ELSE  
                    APPEND node.val TO the front of current_level  
                END IF  
                IF node.left NOT EQUALS none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND a list made from current_level TO result  
            SET left_to_right TO the logical NOT of left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
