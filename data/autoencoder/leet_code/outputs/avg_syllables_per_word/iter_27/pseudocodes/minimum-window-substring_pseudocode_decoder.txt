CLASS Solution  
    FUNCTION minWindow(s, t)  
        IMPORT a collection that counts occurrences of elements AS Counter  
        SET t_count TO a new Counter initialized with t  
        SET required TO the number of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO a new Counter with no elements  
        SET min_length TO an infinitely large value  
        SET min_start TO zero  
        
        WHILE right LESS THAN the LENGTH OF s  
            INCREMENT the count of the element at position right of s in window_counts BY one  
            IF the element at position right of s EXISTS IN t_count AND the count of that element in window_counts EQUALS the count of that element in t_count THEN  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the element at position left of s  
                IF right MINUS left PLUS one LESS THAN min_length THEN  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count of char in window_counts BY one  
                IF char EXISTS IN t_count AND the count of char in window_counts LESS THAN the count of char in t_count THEN  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS an infinitely large value THEN  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start TO the position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
