CLASS Solution  
    FUNCTION canVisitAllRooms WITH parameter rooms  
        SET visited TO an empty set  
        SET stack TO a list containing zero  
        WHILE the LENGTH OF stack GREATER THAN zero  
            SET current_room TO the element at the last position of stack  
            REMOVE the last element FROM stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room of rooms  
                IF key NOT IN visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF visited EQUALS the LENGTH OF rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
