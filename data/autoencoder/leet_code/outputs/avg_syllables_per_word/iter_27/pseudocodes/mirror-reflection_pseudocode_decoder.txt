CLASS Solution  
    FUNCTION mirrorReflection(p parameter, q parameter) RETURNS integer  
        SET greatest_common_divisor TO the greatest common divisor of p parameter AND q parameter  
        SET least_common_multiple TO p parameter MULTIPLIED BY q parameter DIVIDED BY greatest_common_divisor  
        SET vertical_reflections TO least_common_multiple DIVIDED BY p parameter  
        SET horizontal_reflections TO least_common_multiple DIVIDED BY q parameter  
        IF the remainder of vertical_reflections DIVIDED BY two EQUALS zero  
            RETURN zero  
        ELSE IF the remainder of horizontal_reflections DIVIDED BY two EQUALS one  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
