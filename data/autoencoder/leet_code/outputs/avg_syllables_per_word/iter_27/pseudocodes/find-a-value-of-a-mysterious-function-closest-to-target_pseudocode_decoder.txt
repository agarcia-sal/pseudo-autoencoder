CLASS Solution  
    FUNCTION closestToTarget(arr parameter, target parameter)  
        SET seen TO an empty set of unique values  
        SET min_diff TO an infinitely large value  
        FOR each num IN arr parameter  
            SET seen TO the set containing the bitwise AND of num AND each element x IN seen UNION the set containing num  
            FOR each val IN seen  
                SET min_diff TO the smaller value BETWEEN min_diff AND the absolute value of val MINUS target parameter  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
