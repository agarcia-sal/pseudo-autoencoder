CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO list of pairs where each pair consists of element at position i of plantTime and element at position i of growTime for each index i in the range from zero to the length of plantTime minus one  
        SORT tasks in descending order according to the second element of each pair  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair IN tasks  
            SET plant TO first element of pair  
            SET grow TO second element of pair  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
