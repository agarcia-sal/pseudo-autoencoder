CLASS Solution  
    FUNCTION maxProfit(prices)  
        SET max_profit TO zero  
        FOR i FROM one TO LENGTH OF prices MINUS one  
            IF element at position i of prices GREATER THAN element at position i MINUS one of prices  
                INCREMENT max_profit BY element at position i of prices MINUS element at position i MINUS one of prices  
            END IF  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
