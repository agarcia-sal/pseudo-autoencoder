CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO the list of each character in the string representation of num  
        SET last TO an empty dictionary  
        FOR each index i AND digit d IN the enumeration of digits  
            SET element at position i of last TO the integer value of digit d  
        END FOR  
        
        FOR each index i AND digit d IN the enumeration of digits  
            FOR digit d2 FROM nine DOWN TO the integer value of digit d PLUS one DECREMENT BY one  
                IF the element at position d2 of last EXISTS AND the element at position d2 of last GREATER THAN i  
                    SET temporary TO element at position i of digits  
                    SET element at position i of digits TO element at position d2 of last in digits  
                    SET element at position d2 of last in digits TO temporary  
                    SET result_string TO the concatenation of all elements in digits as a string  
                    SET result TO the integer value of result_string  
                    RETURN result  
                END IF  
            END FOR  
        END FOR  
        
        RETURN num  
    END FUNCTION  
END CLASS
