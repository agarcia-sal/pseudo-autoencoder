CLASS Solution  
    FUNCTION minWindow(s1, s2)  
        SET min_len TO infinite large value  
        SET min_start TO negative one  
        SET length_of_s1 TO the LENGTH OF s1  
        SET length_of_s2 TO the LENGTH OF s2  

        FOR i FROM zero TO length_of_s1 MINUS one  
            SET j TO zero  
            FOR k FROM i TO length_of_s1 MINUS one  
                IF element at position k of s1 EQUALS element at position j of s2  
                    INCREMENT j BY one  
                    IF j EQUALS length_of_s2  
                        IF k MINUS i PLUS one LESS THAN min_len  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        IF min_start EQUALS negative one  
            RETURN empty string  
        ELSE  
            RETURN substring of s1 from position min_start TO position min_start PLUS min_len MINUS one  
        END IF  
    END FUNCTION  
END CLASS
