CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO a set of tuples representing each point in points  
        SET min_x TO the minimum of the first element of each point in unique_points  
        SET max_x TO the maximum of the first element of each point in unique_points  
        SET line_of_reflection TO the sum of min_x AND max_x DIVIDED BY two  
        FOR each element IN unique_points  
            SET x TO the first element of element  
            SET y TO the second element of element  
            SET reflected_x TO the difference between two MULTIPLIED BY line_of_reflection AND x  
            IF a tuple consisting of reflected_x AND y NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
