CLASS Solution  
    FUNCTION validSquare(p1 LIST OF INTEGER p2 LIST OF INTEGER p3 LIST OF INTEGER p4 LIST OF INTEGER) RETURNS BOOLEAN  
        FUNCTION squared_distance(point1 LIST OF INTEGER point2 LIST OF INTEGER) RETURNS INTEGER  
            SET difference_in_first_coordinate TO element at position zero of point1 MINUS element at position zero of point2  
            SET difference_in_second_coordinate TO element at position one of point1 MINUS element at position one of point2  
            SET squared_distance_value TO difference_in_first_coordinate MULTIPLIED BY difference_in_first_coordinate PLUS difference_in_second_coordinate MULTIPLIED BY difference_in_second_coordinate  
            RETURN squared_distance_value  
        END FUNCTION  

        SET points TO list containing p1 PLUS p2 PLUS p3 PLUS p4  
        SET distances TO empty set  

        FOR i FROM zero TO three  
            FOR j FROM i PLUS one TO three  
                SET dist TO squared_distance(element at position i of points element at position j of points)  
                IF dist EQUALS zero  
                    RETURN false  
                END IF  
                ADD dist TO distances  
            END FOR  
        END FOR  

        IF the size of distances EQUALS two  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
