CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS the min of the sum of the remainder of remaining DIVIDED BY two AND dp of the integer division of remaining BY two AND the sum of the remainder of remaining DIVIDED BY three AND dp of the integer division of remaining BY three  
        END FUNCTION  
        RETURN dp of n  
    END FUNCTION  
END CLASS
