CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs EQUALS empty  
            RETURN zero  
        END IF  
        
        SET previous cost for red TO element at position zero of element at position zero of costs  
        SET previous cost for blue TO element at position one of element at position zero of costs  
        SET previous cost for green TO element at position two of element at position zero of costs  
        
        FOR index i FROM one TO the LENGTH OF costs MINUS one  
            SET current cost for red TO element at position zero of element at position i of costs PLUS the MINIMUM value BETWEEN previous cost for blue AND previous cost for green  
            SET current cost for blue TO element at position one of element at position i of costs PLUS the MINIMUM value BETWEEN previous cost for red AND previous cost for green  
            SET current cost for green TO element at position two of element at position i of costs PLUS the MINIMUM value BETWEEN previous cost for red AND previous cost for blue  
            
            SET previous cost for red TO current cost for red  
            SET previous cost for blue TO current cost for blue  
            SET previous cost for green TO current cost for green  
        END FOR  
        
        RETURN the MINIMUM value BETWEEN previous cost for red AND previous cost for blue AND previous cost for green  
    END FUNCTION  
END CLASS
