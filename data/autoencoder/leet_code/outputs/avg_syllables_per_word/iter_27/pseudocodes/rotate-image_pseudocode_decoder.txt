CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO element at position j of element at position i of matrix  
                SET element at position j of element at position i of matrix TO element at position i of element at position j of matrix  
                SET element at position i of element at position j of matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            SET start_index TO zero  
            SET end_index TO n MINUS one  
            WHILE start_index LESS THAN end_index  
                SET temporary_value TO element at position end_index of element at position i of matrix  
                SET element at position end_index of element at position i of matrix TO element at position start_index of element at position i of matrix  
                SET element at position start_index of element at position i of matrix TO temporary_value  
                INCREMENT start_index BY one  
                DECREMENT end_index BY one  
            END WHILE  
        END FOR  
    END FUNCTION  
END CLASS
