CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER OF the RESULT OF dp with left and pivot MINUS one AND the RESULT OF dp with pivot PLUS one and right  
                SET min_cost TO the SMALLER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN the RESULT OF dp with one and n  
    END FUNCTION  
END CLASS
