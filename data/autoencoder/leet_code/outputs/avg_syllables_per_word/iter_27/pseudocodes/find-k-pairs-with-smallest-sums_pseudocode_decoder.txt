CLASS Solution  
    FUNCTION kSmallestPairs(nums1 LIST OF INTEGER nums2 LIST OF INTEGER k INTEGER) RETURNS LIST OF LIST OF INTEGER  
        IF the LENGTH OF nums1 EQUALS zero OR the LENGTH OF nums2 EQUALS zero OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN an empty list  
        END IF  
        
        SET min_heap TO an empty list  
        FOR i FROM zero TO the minimum of k MINUS one AND the LENGTH OF nums1 MINUS one  
            PUSH onto min_heap A tuple CONTAINING the element at position i of nums1 PLUS the element at position zero of nums2 AND i AND zero  
        END FOR  
        
        SET result TO an empty list  
        WHILE min_heap IS NOT empty AND the LENGTH OF result LESS THAN k  
            POP the smallest element from min_heap INTO variables discard_value AND i AND j  
            APPEND a list CONTAINING the element at position i of nums1 AND the element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2 THEN  
                PUSH onto min_heap A tuple CONTAINING the element at position i of nums1 PLUS the element at position j PLUS one of nums2 AND i AND j PLUS one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
