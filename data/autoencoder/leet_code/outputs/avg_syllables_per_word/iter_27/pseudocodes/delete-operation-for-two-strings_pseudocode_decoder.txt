CLASS Solution  
    FUNCTION minDistance(word1  
                         word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  

        CREATE a two dimensional list dp with m PLUS one rows  
        FOR index_row FROM zero TO m  
            CREATE a list row with n PLUS one zeros  
            SET the element at position index_row of dp TO row  
        END FOR  

        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2 THEN  
                    SET the element at position i THEN j in dp TO the element at position i MINUS one THEN j MINUS one in dp PLUS one  
                ELSE  
                    IF the element at position i MINUS one THEN j in dp GREATER THAN the element at position i THEN j MINUS one in dp THEN  
                        SET the element at position i THEN j in dp TO the element at position i MINUS one THEN j in dp  
                    ELSE  
                        SET the element at position i THEN j in dp TO the element at position i THEN j MINUS one in dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        SET lcs_length TO the element at position m THEN n in dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
