CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS an empty list THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the sublist of values starting from position two TO the last position  
        SET node TO a new ListNode initialized with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the value field of p1 EQUALS the value field of p2 AND the result of is_same_list called with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(self head n)  
        SET dummy TO a new ListNode initialized with zero  
        SET the next element of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR index FROM one TO n PLUS one  
            SET first TO the next element of first  
        END FOR  
        WHILE first NOT EQUALS None DO  
            SET first TO the next element of first  
            SET second TO the next element of second  
        END WHILE  
        SET the next element of second TO the next element of the next element of second  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
