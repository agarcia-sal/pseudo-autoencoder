CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value equal to element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one TO the end  
        SET node TO new ListNode with value equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the predicate that element val of p1 EQUALS element val of p2 AND recursive call to is_same_list with element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(self, head, k)  
        FUNCTION reverseLinkedList(start, end)  
            SET prev TO None  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO element next of curr  
                SET element next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET element next of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO new ListNode with value equal to zero  
        SET element next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR each iteration FROM one TO k  
                SET kth_node TO element next of kth_node  
                IF kth_node EQUALS None  
                    RETURN element next of dummy  
                END IF  
            END FOR  
            SET next_group_start TO element next of kth_node  
            SET element next of kth_node TO None  
            CALL reverseLinkedList with element next of prev_group_end AND kth_node AND ASSIGN results TO new_group_start AND new_group_end  
            SET element next of prev_group_end TO new_group_start  
            SET element next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN element next of dummy  
    END FUNCTION  
END CLASS
