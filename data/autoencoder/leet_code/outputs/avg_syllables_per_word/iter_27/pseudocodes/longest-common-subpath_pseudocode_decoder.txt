CLASS Solution  
    FUNCTION longestCommonSubpath(n, paths)  
        FUNCTION hash_subpaths(path, length)  
            SET base_number TO one hundred thirteen  
            SET modulus TO one billion plus seven  
            SET hash_value TO zero  
            SET power TO one  
            FOR index FROM zero TO length MINUS one  
                SET hash_value TO hash_value MULTIPLIED BY base_number PLUS element at position index of path  
                SET hash_value TO the remainder of hash_value DIVIDED BY modulus  
                SET power TO power MULTIPLIED BY base_number  
                SET power TO the remainder of power DIVIDED BY modulus  
            END FOR  
            SET hashes TO set containing hash_value  
            FOR index FROM length TO the length of path MINUS one  
                SET hash_value TO hash_value MULTIPLIED BY base_number PLUS element at position index of path MINUS element at position index MINUS length of path MULTIPLIED BY power  
                SET hash_value TO the remainder of hash_value DIVIDED BY modulus  
                ADD hash_value TO hashes  
            END FOR  
            RETURN hashes  
        END FUNCTION  
        
        FUNCTION check_common_subpath(length)  
            IF length EQUALS zero THEN  
                RETURN true  
            END IF  
            SET common_hashes TO the result of hash_subpaths with first element of paths and length  
            FOR each path IN elements of paths from second element to last  
                SET current_hashes TO the result of hash_subpaths with path and length  
                SET common_hashes TO intersection of common_hashes AND current_hashes  
                IF common_hashes IS empty THEN  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true if common_hashes IS NOT empty otherwise false  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO minimum length among all elements in paths  
        WHILE left LESS THAN right DO  
            SET mid TO integer division of left PLUS right PLUS one BY two  
            IF check_common_subpath(mid) IS true THEN  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
