CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CALL heapify the list blocks to arrange it as a minimum heap  
        WHILE the LENGTH OF blocks is GREATER THAN one  
            SET first TO the result of removing the smallest element from blocks  
            SET second TO the result of removing the smallest element from blocks  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks maintaining heap property  
        END WHILE  
        RETURN the element at position zero of blocks  
    END FUNCTION  
END CLASS
