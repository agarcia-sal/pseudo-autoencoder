CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        SET n TO the LENGTH OF prevRoom  
        SET graph TO a mapping from each key TO an empty list by default  
        FOR room FROM one TO n MINUS one  
            APPEND room TO the list at key equal to the element at position room of prevRoom in graph  
        END FOR  
        
        FUNCTION dp(node)  
            IF the list at key node in graph EQUALS an empty list THEN  
                RETURN one AND one  
            END IF  
            
            SET total_ways TO one  
            SET total_size TO one  
            
            SET child_results TO an empty list  
            FOR each child IN the list at key node in graph  
                SET ways AND size TO the result of calling dp with argument child  
                APPEND the pair of ways AND size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            
            SET current_size TO one  
            FOR each pair of ways AND size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the number of combinations of current_size PLUS size MINUS one TAKEN size AT A TIME MULTIPLIED BY ways  
                SET total_ways TO total_ways MODULO the constant MOD  
                INCREMENT current_size BY size  
            END FOR  
            
            RETURN total_ways AND total_size  
        END FUNCTION  
        
        RETURN the first element of the result of calling dp with argument zero  
    END FUNCTION  
END CLASS
