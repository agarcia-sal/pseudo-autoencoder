CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list containing pairs one zero and minus one zero and zero one and zero minus one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a list of lists each containing m elements of a list each containing n elements of positive infinity  
            SET queue TO an empty double ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the tuple of i j zero TO queue  
                        SET element at position i then position j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE the LENGTH OF queue GREATER THAN zero  
                REMOVE the leftmost element from queue and ASSIGN it to x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx then position ny of grid EQUALS zero AND element at position nx then position ny of fire_times EQUALS positive infinity  
                        SET element at position nx then position ny of fire_times TO t PLUS one  
                        APPEND the tuple of nx ny t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a double ended queue containing a single tuple of zero zero wait  
            SET visited TO a set containing the tuple of zero zero  
            
            WHILE the LENGTH OF queue GREATER THAN zero  
                REMOVE the leftmost element from queue and ASSIGN it to x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple of nx ny NOT IN visited AND element at position nx then position ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx then position ny of fire_times GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN True  
                            ELSE  
                                RETURN False  
                            END IF  
                        END IF  
                        IF element at position nx then position ny of fire_times GREATER THAN t PLUS one  
                            ADD the tuple of nx ny TO visited  
                            APPEND the tuple of nx ny t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO minus one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of left PLUS right BY two  
            IF can_escape(mid) IS True  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result NOT EQUALS m MULTIPLIED BY n  
            RETURN result  
        ELSE  
            RETURN one followed by nine zeros  
        END IF  
    END FUNCTION  
END CLASS
