CLASS Solution  
    FUNCTION searchMatrix(matrix parameter LIST OF LIST OF INTEGERS  target parameter INTEGER) RETURNS BOOLEAN  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN FALSE  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS right DIVIDED BY two  
            SET row TO mid DIVIDED BY n  
            SET column TO mid MODULO n  
            SET mid_value TO element at position row of matrix at position column  
            IF mid_value EQUALS target  
                RETURN TRUE  
            ELSE IF mid_value LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN FALSE  
    END FUNCTION  
END CLASS
