CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SET points TO points SORTED BY the end coordinate of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the start coordinate of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the end coordinate of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
