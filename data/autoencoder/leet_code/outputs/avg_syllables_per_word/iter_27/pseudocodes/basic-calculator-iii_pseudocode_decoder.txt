CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus sign  
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO element at position zero of s  
                REMOVE element at position zero FROM s  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS open parenthesis  
                    SET num TO the result of calling helper with argument s  
                END IF  
                IF char EQUALS plus sign OR char EQUALS minus sign OR char EQUALS multiplication sign OR char EQUALS division sign OR char EQUALS open parenthesis OR char EQUALS close parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus sign  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus sign  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS multiplication sign  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division sign  
                        SET the element at the last position of stack TO the integer value of the element at the last position of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis  
                    BREAK the loop  
                END IF  
            END WHILE  
            RETURN the sum of elements in stack  
        END FUNCTION  

        SET s TO the list of characters in s  
        RETURN the result of calling helper with argument s  
    END FUNCTION  
END CLASS
