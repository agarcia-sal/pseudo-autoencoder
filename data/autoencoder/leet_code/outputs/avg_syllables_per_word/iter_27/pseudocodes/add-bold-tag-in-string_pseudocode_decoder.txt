CLASS Solution  
    FUNCTION addBoldTag(s   words)  
        IF the LENGTH OF words EQUALS zero  
            RETURN s  
        END IF  
          
        SET n TO the LENGTH OF s  
        SET mask TO a list of the BOOLEAN value false repeated n times  
          
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the position AT WHICH word APPEARS IN s STARTING FROM start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the BOOLEAN value AT position i OF mask TO true  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
          
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF the BOOLEAN value AT position i OF mask IS true  
                APPEND the string indicating the opening bold tag TO result  
                WHILE i LESS THAN n AND the BOOLEAN value AT position i OF mask IS true  
                    APPEND the character AT position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string indicating the closing bold tag TO result  
            ELSE  
                APPEND the character AT position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
          
        RETURN the concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
