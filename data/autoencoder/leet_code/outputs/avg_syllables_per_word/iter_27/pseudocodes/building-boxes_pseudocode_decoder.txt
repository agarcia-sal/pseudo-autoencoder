CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            RETURN the PRODUCT of k AND the SUM of k AND one AND the SUM of k AND two DIVIDED BY six  
        END FUNCTION  

        SET left TO zero  
        SET right TO n  
        WHILE the value of left IS LESS THAN the value of right  
            SET mid TO the SUM of left AND right AND one DIVIDED BY two  
            IF the result of triangular_pyramid with mid AS PARAMETER IS LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO the DIFFERENCE of mid AND one  
            END IF  
        END WHILE  

        SET boxes_used TO the result of triangular_pyramid with left AS PARAMETER  
        SET remaining TO the DIFFERENCE of n AND boxes_used  
        SET floor_boxes TO the PRODUCT of left AND the SUM of left AND one DIVIDED BY two  

        SET additional_boxes TO zero  
        WHILE the value of remaining IS GREATER THAN zero  
            INCREMENT additional_boxes BY one  
            DECREMENT remaining BY additional_boxes  
        END WHILE  

        RETURN the SUM of floor_boxes AND additional_boxes  
    END FUNCTION  
END CLASS
