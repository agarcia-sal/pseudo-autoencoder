CLASS Solution  
    FUNCTION maxDotProduct(nums1 AS List OF integers, nums2 AS List OF integers)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        INITIALIZE dp AS a list OF lists  
        FOR index_i FROM zero TO m  
            INITIALIZE row AS an empty list  
            FOR index_j FROM zero TO n  
                APPEND a very small number TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one OF nums1 MULTIPLIED BY element at position j MINUS one OF nums2  
                SET dp element at position i element at position j TO the maximum of  
                    current_product  
                    dp element at position i MINUS one element at position j  
                    dp element at position i element at position j MINUS one  
                    dp element at position i MINUS one element at position j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp element at position m element at position n  
    END FUNCTION  
END CLASS
