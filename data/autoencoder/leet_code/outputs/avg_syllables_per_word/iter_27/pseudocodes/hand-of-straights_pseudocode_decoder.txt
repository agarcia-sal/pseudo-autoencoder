CLASS Solution  
    FUNCTION isNStraightHand(hand AS List of INTEGER, groupSize AS INTEGER) RETURNS BOOLEAN  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        
        SET count TO a mapping of each distinct element in hand to the number of times it occurs  
        SET unique_cards TO the elements in count arranged in ascending order  
        
        FOR each card IN unique_cards  
            IF the number associated with card in count GREATER THAN zero THEN  
                SET needed TO the number associated with card in count  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF the number associated with the element obtained by the sum of card PLUS index in count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    DECREMENT the number associated with the element obtained by the sum of card PLUS index in count BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
