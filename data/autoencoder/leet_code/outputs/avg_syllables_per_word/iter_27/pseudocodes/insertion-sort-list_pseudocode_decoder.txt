CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode with value as element at position one of values  
    SET p TO head  
    FOR each val IN the collection of elements from position two TO the end of values  
        SET node TO a new ListNode with value as val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND the result of is_same_list with parameters as the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(head)  
        IF head EQUALS None OR the next element of head EQUALS None THEN  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode with value MINUS five thousand one  
        SET the next element of dummy TO head  
        SET current TO the next element of head  
        SET the next element of head TO None  
        WHILE current IS NOT None  
            SET next_node TO the next element of current  
            SET prev TO dummy  
            WHILE the next element of prev IS NOT None AND the value of the next element of prev LESS THAN the value of current  
                SET prev TO the next element of prev  
            END WHILE  
            SET the next element of current TO the next element of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
