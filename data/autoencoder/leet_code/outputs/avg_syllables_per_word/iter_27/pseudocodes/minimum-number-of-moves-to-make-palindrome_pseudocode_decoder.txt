CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s  
        SET s TO the LIST OF characters IN s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s IS GREATER THAN one  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF the ELEMENT at position i of s EQUALS the ELEMENT at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SET temporary TO the ELEMENT at position j PLUS one of s  
                        SET the ELEMENT at position j PLUS one of s TO the ELEMENT at position j of s  
                        SET the ELEMENT at position j of s TO temporary  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE the ELEMENT at position zero FROM s  
                    REMOVE the ELEMENT at the LAST position FROM s  
                    BREAK  
                END IF  
            END FOR  
            IF the FOR loop COMPLETED WITHOUT BREAKING  
                REMOVE the ELEMENT at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
