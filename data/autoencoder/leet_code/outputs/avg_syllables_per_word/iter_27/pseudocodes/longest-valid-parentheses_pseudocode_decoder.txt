CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a list containing negative one as the only element  
        SET max_length TO zero  
        FOR each index i AND character char IN the enumeration of s  
            IF character char EQUALS the opening parenthesis character  
                APPEND the index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the LENGTH OF stack EQUALS zero  
                    APPEND the index i TO stack  
                ELSE  
                    SET current_length TO the index i MINUS the element at position the last index of stack IN stack  
                    IF current_length GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
