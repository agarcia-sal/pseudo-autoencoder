CLASS ListNode  
    FUNCTION __init__ with parameters val defaulting to zero next defaulting to none  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node with parameter values  
    IF the values collection IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the subcollection of values starting from position one to the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 IS none AND p2 IS none  
        RETURN TRUE  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN FALSE  
    END IF  
    RETURN the logical expression that the value property of p1 EQUALS the value property of p2 AND the result of calling is_same_list with the next property of p1 and the next property of p2  
END FUNCTION

CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero left defaulting to none right defaulting to none  
        SET the value property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF the values collection IS EMPTY  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET index i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT EMPTY  
        SET node TO the element removed from the left side of queue  
        IF index i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET the left property of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT index i BY one  
        IF index i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET the right property of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT index i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p and q  
    IF p IS none AND q IS none  
        RETURN TRUE  
    ELSE IF p IS none OR q IS none  
        RETURN FALSE  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN FALSE  
    ELSE  
        RETURN the logical expression that the result of calling is_same_tree with the left property of p and the left property of q AND the result of calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION sortedListToBST with parameter head  
        IF head IS none  
            RETURN none  
        END IF  

        FUNCTION findMiddle with parameters left and right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next property of fast NOT EQUALS right  
                SET slow TO the next property of slow  
                SET fast TO the next property of the next property of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST with parameters left and right  
            IF left EQUALS right  
                RETURN none  
            END IF  

            SET mid TO the result of calling findMiddle with left and right  

            SET node TO a new instance of TreeNode initialized with the value property of mid  
            SET the left property of node TO the result of calling convertListToBST with left and mid  
            SET the right property of node TO the result of calling convertListToBST with the next property of mid and right  
            RETURN node  
        END FUNCTION  

        RETURN the result of calling convertListToBST with head and none  
    END FUNCTION  
END CLASS
