CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a three dimensional list with n PLUS one elements in the first dimension  
        EACH element in the first dimension is a list with two elements  
        EACH element in the second dimension is a list with three elements ALL initialized TO zero  
        SET element at position zero of dp at position zero at position zero TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position i of dp at position j at position zero TO element at position i of dp at position j at position zero PLUS element at position i MINUS one of dp at position j at position k  
                    IF j EQUALS zero  
                        SET element at position i of dp at position one at position zero TO element at position i of dp at position one at position zero PLUS element at position i MINUS one of dp at position zero at position k  
                    END IF  
                    IF k LESS THAN two  
                        SET element at position i of dp at position j at position k PLUS one TO element at position i of dp at position j at position k PLUS one PLUS element at position i MINUS one of dp at position j at position k  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS element at position n of dp at position j at position k  
            END FOR  
        END FOR  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
