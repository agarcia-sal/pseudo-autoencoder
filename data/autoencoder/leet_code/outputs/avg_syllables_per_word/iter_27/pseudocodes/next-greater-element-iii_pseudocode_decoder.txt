CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO the list of characters that form the string representation of n  
        SET length TO the number of elements in digits  
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of digits GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE element at position j of digits LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SET temp TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temp  
        SET digits TO the concatenation of the subsequence of digits from position zero TO position i AND the reversed subsequence of digits from position i PLUS one TO the last position  
        SET result_string TO the concatenation of all elements of digits in order  
        SET result TO the integer value of result_string  
        IF result GREATER THAN the integer represented by the number two raised to the thirty-first power MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
