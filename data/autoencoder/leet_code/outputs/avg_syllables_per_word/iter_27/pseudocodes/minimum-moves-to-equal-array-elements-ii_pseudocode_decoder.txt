CLASS Solution  
    FUNCTION minMoves2(nums)  
        CALL sort the list nums in ascending order  
        SET median TO element at position the integer division of the length of nums by two of nums  
        SET moves TO zero  
        FOR each num IN nums  
            SET difference TO num MINUS median  
            IF difference LESS THAN zero  
                SET absolute_difference TO difference MULTIPLIED BY negative one  
            ELSE  
                SET absolute_difference TO difference  
            END IF  
            INCREMENT moves BY absolute_difference  
        END FOR  
        RETURN moves  
    END FUNCTION  
END CLASS
