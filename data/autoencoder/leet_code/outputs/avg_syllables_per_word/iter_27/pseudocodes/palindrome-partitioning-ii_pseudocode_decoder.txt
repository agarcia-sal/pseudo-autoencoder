CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET is_palindrome TO a TWO DIMENSIONAL collection of BOOLEAN FALSE with dimensions n by n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND position i of is_palindrome TO TRUE  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start AND position end of is_palindrome TO the CONDITION that element at position start of s EQUALS element at position end of s  
                ELSE  
                    SET element at position start AND position end of is_palindrome TO the CONDITION that element at position start of s EQUALS element at position end of s AND element at position start PLUS one AND position end MINUS one of is_palindrome IS TRUE  
                END IF  
            END FOR  
        END FOR  
        
        SET min_cuts TO a collection of zeros of length n  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero AND position i of is_palindrome IS TRUE  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF element at position j AND position i of is_palindrome IS TRUE AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
