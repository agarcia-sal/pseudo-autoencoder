CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS no elements OR element at position zero of board EQUALS no elements  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        SET directions TO list of pairs consisting of minus one and minus one MINUS one and zero MINUS one and one ZERO and minus one ZERO and one ONE and minus one ONE and zero ONE and one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx and dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND (element at position ni of board at position nj EQUALS one OR element at position ni of board at position nj EQUALS three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board at position j EQUALS one  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three  
                        SET element at position i of board at position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three  
                        SET element at position i of board at position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS two  
                    SET element at position i of board at position j TO one  
                ELSE IF element at position i of board at position j EQUALS three  
                    SET element at position i of board at position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
