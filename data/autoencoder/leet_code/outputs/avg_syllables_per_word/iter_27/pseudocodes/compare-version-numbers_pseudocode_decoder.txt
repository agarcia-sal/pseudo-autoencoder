CLASS Solution  
    FUNCTION compareVersion(version1 version1 STRING version2 version2 STRING) RETURNS INTEGER  
        SET v1_revisions TO the result of splitting version1 by the character dot  
        SET v2_revisions TO the result of splitting version2 by the character dot  
        SET max_length TO the maximum value between the length of v1_revisions AND the length of v2_revisions  
        FOR i FROM zero TO max_length MINUS one  
            IF i LESS THAN the length of v1_revisions THEN  
                SET v1 TO the integer value of element at position i of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF i LESS THAN the length of v2_revisions THEN  
                SET v2 TO the integer value of element at position i of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2 THEN  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2 THEN  
                RETURN positive one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
