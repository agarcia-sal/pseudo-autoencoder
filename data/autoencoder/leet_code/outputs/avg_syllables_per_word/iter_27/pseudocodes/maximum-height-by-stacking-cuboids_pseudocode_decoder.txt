CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT the dimensions of cuboid in ascending order  
        END FOR  
        
        SORT cuboids lexicographically in ascending order  
        
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp as a list of zeroes with size n  
        
        FOR i FROM zero TO n MINUS one  
            SET dp at position i TO element at position three of cuboid at position i  
            
            FOR j FROM zero TO i MINUS one  
                IF element at position one of cuboid at position j LESS THAN OR EQUAL TO element at position one of cuboid at position i  
                AND element at position two of cuboid at position j LESS THAN OR EQUAL TO element at position two of cuboid at position i  
                AND element at position three of cuboid at position j LESS THAN OR EQUAL TO element at position three of cuboid at position i  
                THEN  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS element at position three of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the MAXIMUM VALUE IN dp  
    END FUNCTION  
END CLASS
