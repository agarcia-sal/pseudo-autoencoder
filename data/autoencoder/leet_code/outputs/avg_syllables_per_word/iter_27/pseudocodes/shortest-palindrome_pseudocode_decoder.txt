CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            SET reversed_str TO empty string  
            FOR index FROM the LENGTH OF check_str MINUS one DOWN TO zero  
                APPEND element at position index of check_str TO reversed_str  
            END FOR  
            IF check_str EQUALS reversed_str  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero  
            SET prefix_substring TO substring from position zero TO position i MINUS one of s  
            IF is_palindrome(prefix_substring)  
                SET suffix_substring TO substring from position i TO the END OF s  
                SET reversed_suffix TO empty string  
                FOR index FROM the LENGTH OF suffix_substring MINUS one DOWN TO zero  
                    APPEND element at position index of suffix_substring TO reversed_suffix  
                END FOR  
                RETURN reversed_suffix PLUS s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
