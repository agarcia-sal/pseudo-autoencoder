CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING representation of count TO path  
                END IF  
                APPEND the concatenation of all elements in path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH parameters the value of index PLUS one, the concatenation of path with the STRING representation of count AND the element at position index of word, and zero  
                ELSE  
                    CALL backtrack WITH parameters the value of index PLUS one, the concatenation of path with the element at position index of word, and zero  
                END IF  
                CALL backtrack WITH parameters the value of index PLUS one, path, and the value of count PLUS one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH parameters zero, an empty list, and zero  
        RETURN result  
    END FUNCTION  
END CLASS
