CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  
        
        SET dp TO list of zero repeated the quantity of n PLUS one times  
        
        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  
        
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET element at position i of dp TO the greater value BETWEEN element at position i of dp AND element at position j of dp MULTIPLIED BY i MINUS j MINUS one  
            END FOR  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
