CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO an empty list  
        FOR each element IN flowers  
            APPEND element at position zero of element TO start_times  
        END FOR  
        SORT start_times in ascending order  
        
        SET end_times TO an empty list  
        FOR each element IN flowers  
            APPEND element at position one of element TO end_times  
        END FOR  
        SORT end_times in ascending order  
        
        SET result TO an empty list  
        
        FOR each person IN people  
            SET started TO the count of elements in start_times which are LESS THAN OR EQUAL TO person  
            SET ended TO the count of elements in end_times which are LESS THAN person  
            SET flowers_in_bloom TO started MINUS ended  
            APPEND flowers_in_bloom TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
