CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN a list containing the empty string  
            END IF  
            IF n EQUALS one  
                RETURN a list containing the string zero and the string one and the string eight  
            END IF  
            SET middles TO the result of calling generate_strobogrammatic with n MINUS two and final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND the string zero concatenated with middle concatenated with string zero TO result  
                END IF  
                APPEND the string one concatenated with middle concatenated with string one TO result  
                APPEND the string six concatenated with middle concatenated with string nine TO result  
                APPEND the string eight concatenated with middle concatenated with string eight TO result  
                APPEND the string nine concatenated with middle concatenated with string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM the length of low TO the length of high  
                FOR each num IN generate_strobogrammatic with length and length  
                    IF (length EQUALS the length of low AND num LESS THAN low) OR (length EQUALS the length of high AND num GREATER THAN high)  
                        CONTINUE to the next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN the result of calling count_strobogrammatic with low and high  
    END FUNCTION  
END CLASS
