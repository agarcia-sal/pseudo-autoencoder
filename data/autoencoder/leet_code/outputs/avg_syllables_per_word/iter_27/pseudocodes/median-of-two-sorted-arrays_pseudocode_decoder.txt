CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1 nums1 PARAMETER OF LIST OF INTEGER nums2 nums2 PARAMETER OF LIST OF INTEGER) RETURNS FLOAT  
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2  
            SET temporary variable nums1 TO nums2  
            SET nums2 TO temporary variable nums1  
        END IF  
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        WHILE low IS LESS THAN OR EQUAL TO high  
            SET partitionX TO low PLUS high DIVIDED BY two  
            SET partitionY TO x PLUS y PLUS one DIVIDED BY two MINUS partitionX  
            IF partitionX EQUALS zero  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO the element at position partitionX MINUS one OF nums1  
            END IF  
            IF partitionX EQUALS x  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO the element at position partitionX OF nums1  
            END IF  
            IF partitionY EQUALS zero  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO the element at position partitionY MINUS one OF nums2  
            END IF  
            IF partitionY EQUALS y  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO the element at position partitionY OF nums2  
            END IF  
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX  
                IF (x PLUS y) MODULO two EQUALS zero  
                    SET maximum_left TO maxLeftX  
                    IF maxLeftY IS GREATER THAN maximum_left  
                        SET maximum_left TO maxLeftY  
                    END IF  
                    SET minimum_right TO minRightX  
                    IF minRightY IS LESS THAN minimum_right  
                        SET minimum_right TO minRightY  
                    END IF  
                    RETURN maximum_left PLUS minimum_right DIVIDED BY two  
                ELSE  
                    SET maximum_left TO maxLeftX  
                    IF maxLeftY IS GREATER THAN maximum_left  
                        SET maximum_left TO maxLeftY  
                    END IF  
                    RETURN maximum_left  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
