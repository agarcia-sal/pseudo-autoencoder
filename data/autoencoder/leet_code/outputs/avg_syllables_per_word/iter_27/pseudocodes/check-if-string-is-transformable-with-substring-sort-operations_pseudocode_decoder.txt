CLASS Solution  
    FUNCTION isTransformable(s stringParameter t stringParameter) RETURNS boolean  
        DECLARE pos AS mapping FROM digit TO list of positions  
        FOR each indexPosition AND character IN s  
            SET currentDigit TO the integer value of character  
            APPEND indexPosition TO the list in pos corresponding to currentDigit  
        END FOR  
        
        FOR each indexPosition AND character IN t  
            SET currentDigit TO the integer value of character  
            IF the list in pos corresponding to currentDigit IS EMPTY THEN  
                RETURN false  
            END IF  
            FOR each smallerDigit FROM zero TO currentDigit MINUS one  
                IF the list in pos corresponding to smallerDigit IS NOT EMPTY AND the first element of the list in pos corresponding to smallerDigit IS LESS THAN the first element of the list in pos corresponding to currentDigit THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the first element FROM the list in pos corresponding to currentDigit  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
