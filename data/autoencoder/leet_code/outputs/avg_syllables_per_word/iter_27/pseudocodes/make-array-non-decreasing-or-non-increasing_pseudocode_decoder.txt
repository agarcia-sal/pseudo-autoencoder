CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap IS NOT empty AND the NEGATIVE of element at position zero of max_heap IS GREATER THAN num  
                    INCREMENT operations BY the NEGATIVE of REMOVE element at position zero from max_heap MINUS num  
                    ADD the NEGATIVE of num TO max_heap  
                END IF  
                ADD the NEGATIVE of num TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
      
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap IS NOT empty AND element at position zero of min_heap IS LESS THAN num  
                    INCREMENT operations BY num MINUS REMOVE element at position zero from min_heap  
                    ADD num TO min_heap  
                END IF  
                ADD num TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
      
        RETURN the MINIMUM OF min_operations_to_non_decreasing(nums) AND min_operations_to_non_increasing(nums)  
    END FUNCTION  
END CLASS
