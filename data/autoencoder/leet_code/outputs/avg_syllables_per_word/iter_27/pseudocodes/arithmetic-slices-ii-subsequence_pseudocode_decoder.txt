CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three THEN  
            RETURN zero  
        END IF  
        
        SET dp TO a list containing for each position in the range from zero TO n MINUS one a mapping from difference TO count initialized to zero  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET difference d TO the element at position i of nums MINUS the element at position j of nums  
                
                IF difference d EXISTS AS a key IN the mapping at position j of dp THEN  
                    INCREMENT the value associated to key d in the mapping at position i of dp BY the value associated to key d in the mapping at position j of dp PLUS one  
                    INCREMENT total_count BY the value associated to key d in the mapping at position j of dp  
                ELSE  
                    INCREMENT the value associated to key d in the mapping at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
