CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO list of size amount PLUS one where each element IS amount PLUS one  
        SET element at position zero of dp TO zero  
        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin GREATER THAN OR EQUAL TO zero THEN  
                    SET element at position a of dp TO the smaller value BETWEEN element at position a of dp AND element at position a MINUS coin of dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF element at position amount of dp NOT EQUALS amount PLUS one THEN  
            RETURN element at position amount of dp  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
