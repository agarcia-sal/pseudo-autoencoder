CLASS Solution  
    FUNCTION digitsCount(digit, low, high)  
        FUNCTION count_digit(number, target_digit)  
            SET count TO zero  
            SET power_of_ten TO one  
            SET current TO number  
            WHILE current GREATER THAN zero  
                SET last TO element at position modulo of current BY ten  
                SET current TO element at position integer division of current BY ten  
                SET full_cycles TO current  
                IF target_digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_ten  
                IF last GREATER THAN target_digit  
                    INCREMENT count BY power_of_ten  
                ELSE IF last EQUALS target_digit  
                    INCREMENT count BY element at position modulo of number BY power_of_ten PLUS one  
                END IF  
                MULTIPLY power_of_ten BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN result OF count_digit ON high AND digit MINUS result OF count_digit ON low MINUS one AND digit  
    END FUNCTION  
END CLASS
