CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze start destination  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the list containing the pairs one and zero MINUS one and zero zero and one zero and MINUS one  
        SET start TO the tuple converted from start  
        SET destination TO the tuple converted from destination  
        SET visited TO an empty set  
        SET pq TO a list containing the tuple zero and start  

        WHILE pq is not empty  
            SET dist AND x AND y TO the result of removing the smallest element from pq  
            
            IF the tuple x and y EQUALS destination THEN  
                RETURN dist  
            END IF  
            
            IF the tuple x and y IS IN visited THEN  
                CONTINUE to the next iteration  
            END IF  
            
            ADD the tuple x and y TO visited  
            
            FOR each dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at new_x PLUS dx position and new_y PLUS dy position of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the tuple new_x and new_y IS NOT IN visited THEN  
                    INSERT the tuple dist PLUS count and the tuple new_x and new_y INTO pq maintaining the heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
