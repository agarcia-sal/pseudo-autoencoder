CLASS Solution  
    FUNCTION kthSmallest(mat parameter, k parameter)  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET initial_sum TO zero  
        FOR each row IN mat  
            INCREMENT initial_sum BY element at position zero of row  
        END FOR  
        SET initial_indices TO a sequence of zeros WITH LENGTH m  
        SET min_heap TO a list containing a pair of initial_sum AND initial_indices converted to an immutable sequence  
        SET visited TO a set containing the immutable sequence at position one of the first element of min_heap  
        
        WHILE k greater than zero  
            REMOVE AND RETURN the smallest element FROM min_heap INTO current_sum AND indices  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF element at position i of indices PLUS one LESS THAN n  
                    SET new_indices TO a copy of indices converted to a mutable sequence  
                    INCREMENT element at position i of new_indices BY one  
                    SET new_indices_tuple TO new_indices converted to an immutable sequence  
                    SET new_sum TO current_sum MINUS element at position i of mat at position element at position i of indices PLUS element at position i of mat at position element at position i of new_indices  
                    
                    IF new_indices_tuple NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        ADD the pair of new_sum AND new_indices_tuple TO min_heap AND reorganize min_heap to maintain its heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
