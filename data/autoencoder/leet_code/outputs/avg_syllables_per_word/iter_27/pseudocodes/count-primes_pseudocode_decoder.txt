CLASS Solution  
    FUNCTION countPrimes WITH parameter n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of boolean True repeated n times  
        SET element at position zero of is_prime TO False  
        SET element at position one of is_prime TO False  
        FOR start FROM two TO the integer part of the square root of n PLUS one  
            IF element at position start of is_prime EQUALS True  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one IN STEPS OF start  
                    SET element at position multiple of is_prime TO False  
                END FOR  
            END IF  
        END FOR  
        SET count_of_primes TO zero  
        FOR each element IN is_prime  
            IF element EQUALS True  
                INCREMENT count_of_primes BY one  
            END IF  
        END FOR  
        RETURN count_of_primes  
    END FUNCTION  
END CLASS
