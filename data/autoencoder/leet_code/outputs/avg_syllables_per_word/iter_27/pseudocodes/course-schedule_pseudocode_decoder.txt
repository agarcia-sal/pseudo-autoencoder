CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a mapping from keys to empty lists  
        SET indegree TO a list of zeros with length numCourses  
        
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a double ended queue containing each course in the range from zero TO numCourses minus one WHERE the element at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            REMOVE the element at the front of queue and SET it TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list associated with current_course in graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO the end of queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF numCourses EQUALS zero  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
