CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO an empty mapping with default value being an empty list  
        FOR each word IN words  
            FOR index FROM zero TO the length of word MINUS one  
                SET prefix TO substring from position zero TO position index MINUS one of word  
                APPEND word TO the list associated with prefix in prefix_to_words  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the length of square EQUALS the length of the first element of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO the concatenation of the element at position the length of square in each word of square  
            FOR each candidate IN the list associated with prefix in prefix_to_words  
                CALL backtrack with the list formed by the elements of square PLUS candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack with the list containing only word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
