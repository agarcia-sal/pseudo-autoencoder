CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a dictionary that returns an empty list for each new key  
        FOR each pair of departure and arrival airports IN the sorted list of tickets in reverse order  
            APPEND the arrival airport TO the list at the departure airport key in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list at the airport key in graph is not empty  
                CALL visit WITH the last element removed from the list at the airport key in graph  
            END WHILE  
            APPEND the airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH the airport named J F K  
        RETURN the itinerary in reverse order of its elements  
    END FUNCTION  
END CLASS
