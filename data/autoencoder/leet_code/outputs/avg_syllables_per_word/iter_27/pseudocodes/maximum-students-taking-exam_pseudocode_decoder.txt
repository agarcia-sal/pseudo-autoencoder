CLASS Solution  
    FUNCTION maxStudents(seats)  
        SET m TO the LENGTH OF seats  
        SET n TO the LENGTH OF element at position zero of seats  
        
        SET broken TO a list of zeroes with length m  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of element at position i of seats EQUALS the symbol representing a broken seat  
                    SET element at position i of broken TO element at position i of broken PLUS the number one MULTIPLIED BY two RAISED TO the power j  
                END IF  
            END FOR  
        END FOR  
        
        FUNCTION is_valid(arrangement)  
            RETURN the result of bitwise AND between arrangement AND arrangement shifted right by one EQUALS zero  
        END FUNCTION  
        
        SET dp TO a list containing m empty collections each representing a dictionary  
        
        FOR arrangement FROM zero TO two RAISED TO the power n MINUS one  
            IF is_valid(arrangement) AND the result of bitwise AND between arrangement AND element at position zero of broken EQUALS zero  
                SET key arrangement in element at position zero of dp TO the COUNT OF ones in binary representation of arrangement  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR arrangement FROM zero TO two RAISED TO the power n MINUS one  
                IF is_valid(arrangement) AND the result of bitwise AND between arrangement AND element at position i of broken EQUALS zero  
                    SET max_students TO zero  
                    FOR each prev_arrangement IN keys of element at position i MINUS one of dp  
                        IF the result of bitwise AND between arrangement AND prev_arrangement shifted right by one EQUALS zero AND the result of bitwise AND between arrangement AND prev_arrangement shifted left by one EQUALS zero  
                            IF value associated with key prev_arrangement in element at position i MINUS one of dp GREATER THAN max_students  
                                SET max_students TO value associated with key prev_arrangement in element at position i MINUS one of dp  
                            END IF  
                        END IF  
                    END FOR  
                    SET key arrangement in element at position i of dp TO max_students PLUS the COUNT OF ones in binary representation of arrangement  
                END IF  
            END FOR  
        END FOR  
        
        IF element at last position of dp is NOT empty  
            RETURN the maximum value among all values in element at last position of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
