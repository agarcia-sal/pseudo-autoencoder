CLASS Solution  
    FUNCTION minimumOneBitOperations WITH PARAMETER n  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        
        SET k TO zero  
        WHILE one LEFT SHIFTED BY the value of k PLUS one LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        
        SET most_significant_bit TO one LEFT SHIFTED BY the value of k  
        
        RETURN one LEFT SHIFTED BY the value of k PLUS one MINUS one MINUS the RESULT OF calling minimumOneBitOperations WITH PARAMETER n BITWISE XOR most_significant_bit  
    END FUNCTION  
END CLASS
