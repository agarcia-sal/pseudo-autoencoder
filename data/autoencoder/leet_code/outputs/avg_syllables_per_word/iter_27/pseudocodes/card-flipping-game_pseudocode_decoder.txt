CLASS Solution  
    FUNCTION flipgame WITH fronts AND backs  
        SET same TO an empty set  
        FOR each pair of number_x AND number_y TAKEN simultaneously from fronts AND backs  
            IF number_x EQUALS number_y  
                ADD number_x TO same  
            END IF  
        END FOR  
        SET all_numbers TO a set of all elements in fronts AND all elements in backs combined  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers is not empty  
            SET minimum_good_number TO the smallest element in good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
