CLASS Solution  
    FUNCTION replaceWords WITH PARAMETERS dictionary sentence  
        SORT dictionary BY LENGTH OF element  
        SET words TO list of elements obtained by splitting sentence at each space  
        FUNCTION replace WITH PARAMETER word  
            FOR each root IN dictionary  
                IF word STARTS WITH root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        SET replaced_words TO empty list  
        FOR each word IN words  
            APPEND result of replace WITH word TO replaced_words  
        END FOR  
        SET result_sentence TO concatenation of elements of replaced_words separated by single space  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
