CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position one of houses  
            END IF  
            SET dp TO a list of zeros with the LENGTH OF houses  
            SET element at position one of dp TO element at position one of houses  
            SET element at position two of dp TO the GREATER OF element at position one of houses AND element at position two of houses  
            FOR index FROM three TO the LENGTH OF houses  
                SET element at position index of dp TO the GREATER OF element at position index MINUS one of dp AND the SUM OF element at position index MINUS two of dp PLUS element at position index of houses  
            END FOR  
            RETURN element at the last position of dp  
        END FUNCTION  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        RETURN the GREATER OF rob_linear(the SUBLIST OF nums FROM position one TO the SECOND TO LAST position) AND rob_linear(the SUBLIST OF nums FROM position two TO the LAST position)  
    END FUNCTION  
END CLASS
