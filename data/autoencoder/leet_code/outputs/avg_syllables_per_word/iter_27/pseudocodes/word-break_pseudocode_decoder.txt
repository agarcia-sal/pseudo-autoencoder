CLASS Solution  
    FUNCTION wordBreak(s parameter, wordDict parameter)  
        SET wordSet TO the set of elements in wordDict parameter  
        SET dp TO the list of boolean false values with length equal to the length of s parameter PLUS one  
        SET the element at position zero of dp TO boolean true  
        
        FOR index i FROM one TO the length of s parameter  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position j of dp EQUALS boolean true AND the substring of s parameter from position j PLUS one TO position i of s parameter IS IN wordSet  
                    SET the element at position i of dp TO boolean true  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the length of s parameter of dp  
    END FUNCTION  
END CLASS
