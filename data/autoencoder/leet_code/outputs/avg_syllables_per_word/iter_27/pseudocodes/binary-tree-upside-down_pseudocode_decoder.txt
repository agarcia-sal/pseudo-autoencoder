CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values and left none and right none  
    SET i TO one  
    SET queue TO new empty deque collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not none  
            SET node.left TO new TreeNode with value element at position i of values and left none and right none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not none  
            SET node.right TO new TreeNode with value element at position i of values and left none and right none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root EQUALS none OR root.left EQUALS none  
            RETURN root  
        END IF  
        SET new_root TO the result of calling upsideDownBinaryTree on self with argument root.left  
        SET the left child of the left child of root TO the right child of root  
        SET the right child of the left child of root TO root  
        SET root.left TO none  
        SET root.right TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
