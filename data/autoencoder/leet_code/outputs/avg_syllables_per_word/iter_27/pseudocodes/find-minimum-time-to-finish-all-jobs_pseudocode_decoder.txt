CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeros with the LENGTH of TWO RAISED TO THE POWER OF n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF i MINUS one  
                SET element at position mask OR WITH TWO RAISED TO THE POWER OF i OF subset_sums TO element at position mask of subset_sums PLUS element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN element at position mask of subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS GREATER THAN zero  
                SET candidate TO the GREATER OF dp(k MINUS one, the BITWISE EXCLUSIVE OR of mask AND submask) AND element at position submask of subset_sums  
                IF candidate IS LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO the BITWISE AND of submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, TWO RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
