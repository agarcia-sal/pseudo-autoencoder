CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the number one PLUS MULTIPLIED BY n MINUS one  
            SET good TO an empty list of BOOLEAN false of LENGTH n  
            SET bad TO an empty list of BOOLEAN false of LENGTH n  
            FOR j FROM zero TO n MINUS one  
                IF the binary AND OF i AND the number one LEFT SHIFTED BY j IS NOT EQUALS zero  
                    SET element at position j of good TO true  
                ELSE  
                    SET element at position j of bad TO true  
                END IF  
            END FOR  
            SET valid TO true  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of good IS true  
                    FOR k FROM zero TO n MINUS one  
                        IF (the element at position j of statements AT position k EQUALS zero AND the element at position k of good IS true) OR (the element at position j of statements AT position k EQUALS one AND the element at position k of bad IS true)  
                            SET valid TO false  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS false  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS true  
                SET max_good TO the GREATER THAN OR EQUAL TO VALUE OF max_good AND the SUM OF all true ELEMENTS IN good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
