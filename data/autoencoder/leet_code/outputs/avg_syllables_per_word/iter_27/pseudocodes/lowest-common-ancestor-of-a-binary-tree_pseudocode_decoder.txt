CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO remove the first element from queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root EQUALS none OR root EQUALS p OR root EQUALS q THEN  
            RETURN root  
        END IF  
        SET left TO the result of calling lowestCommonAncestor on self with parameters root.left p and q  
        SET right TO the result of calling lowestCommonAncestor on self with parameters root.right p and q  
        IF left NOT EQUALS none AND right NOT EQUALS none THEN  
            RETURN root  
        END IF  
        IF left NOT EQUALS none THEN  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
