CLASS Solution  
    FUNCTION rob WITH parameter nums  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET element at position zero of dp TO element at position zero of nums  
        SET element at position one of dp TO the GREATER OF element at position zero of nums AND element at position one of nums  
        FOR index i FROM two TO the LENGTH OF nums MINUS one  
            SET element at position i of dp TO the GREATER OF element at position i MINUS one of dp AND the SUM OF element at position i MINUS two of dp AND element at position i of nums  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
