CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO bits that are set in bits that are set in ones XOR num AND bits that are NOT set in twos  
            SET twos TO bits that are set in bits that are set in twos XOR num AND bits that are NOT set in ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
