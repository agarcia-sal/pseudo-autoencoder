CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO an empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS the plus character  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS the minus character  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication character  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS  
                RETURN a list containing the integer conversion of sub_expr  
            END IF  

            SET results TO an empty list  
            FOR index FROM zero TO the length of sub_expr MINUS one  
                IF the element at position index of sub_expr EQUALS the plus character OR EQUALS the minus character OR EQUALS the multiplication character  
                    SET left_results TO the result of calling helper WITH the substring from position zero TO position index MINUS one of sub_expr  
                    SET right_results TO the result of calling helper WITH the substring from position index PLUS one TO the last position of sub_expr  
                    EXTEND results BY the result of calling compute WITH left_results RIGHT right_results AND the element at position index of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN the result of calling helper WITH expression  
    END FUNCTION  
END CLASS
