CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line EQUALS true  
                SET total_spaces_to_add TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
                RETURN the JOINED string of line with SINGLE SPACES BETWEEN each element PLUS total_spaces_to_add SPACES AT THE END  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND to element at position index of line SPACES with the LENGTH spaces_between_words PLUS one  
                    ELSE  
                        APPEND to element at position index of line SPACES with the LENGTH spaces_between_words  
                    END IF  
                END FOR  
                RETURN the CONCATENATION of all elements of line without SEPARATORS  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  

        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND justify_line(line, num_of_letters, false) TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  

        APPEND justify_line(line, num_of_letters, true) TO result  

        RETURN result  
    END FUNCTION  
END CLASS
