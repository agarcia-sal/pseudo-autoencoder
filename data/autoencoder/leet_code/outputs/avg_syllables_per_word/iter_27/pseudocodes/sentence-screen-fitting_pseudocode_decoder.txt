CLASS Solution  
    FUNCTION wordsTyping(sentence parameter IS list of strings rows parameter IS integer cols parameter IS integer) RETURNS integer  
        SET concatenated_sentence TO the concatenation of the elements of sentence separated by a single space PLUS a single trailing space  
        SET total_length TO the LENGTH OF concatenated_sentence  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the character at position MODULO total_length of concatenated_sentence EQUALS a space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character at position MINUS one MODULO total_length of concatenated_sentence NOT EQUALS a space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN the integer division of position BY total_length  
    END FUNCTION  
END CLASS
