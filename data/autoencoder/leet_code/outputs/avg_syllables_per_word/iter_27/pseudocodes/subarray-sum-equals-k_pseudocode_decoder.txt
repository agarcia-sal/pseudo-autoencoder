CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary mapping zero TO one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF the value of current_sum MINUS k EXISTS as a key IN cumulative_sum_freq  
                INCREMENT count BY the value associated with the key current_sum MINUS k IN cumulative_sum_freq  
            END IF  
            IF current_sum EXISTS as a key IN cumulative_sum_freq  
                INCREMENT the value associated with the key current_sum IN cumulative_sum_freq BY one  
            ELSE  
                SET the value associated with the key current_sum IN cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
