CLASS Solution  
    FUNCTION exclusiveTime(n parameter, logs parameter)  
        SET exclusive_time TO a list of zeros with length n parameter  
        SET stack TO an empty list  
        FOR each log IN logs parameter  
            SET function_id TO the element before the first occurrence of colon in log converted to an integer  
            SET action TO the element between the first and second occurrence of colon in log  
            SET timestamp TO the element after the second occurrence of colon in log converted to an integer  
            IF action EQUALS the word start  
                IF the LENGTH OF stack NOT EQUALS zero  
                    SET prev_function_id TO the first element of the last tuple in stack  
                    SET prev_timestamp TO the second element of the last tuple in stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple consisting of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id TO the first element of the last tuple popped from stack  
                SET prev_timestamp TO the second element of the last tuple popped from stack  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the LENGTH OF stack NOT EQUALS zero  
                    SET top_function_id TO the first element of the last tuple in stack  
                    SET stack at last position TO a tuple consisting of top_function_id and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
