CLASS Solution  
    FUNCTION checkWays(pairs)  
        SET graph TO a mapping from element to a set of elements  
        FOR each pair OF x AND y IN pairs  
            APPEND y TO the set associated with x in graph  
            APPEND x TO the set associated with y in graph  
        END FOR  
        
        SET max_degree TO the element in graph with the maximum count of associated elements  
        SET n TO the count of distinct elements in graph  
        
        IF the count of elements associated with max_degree NOT EQUALS n MINUS one THEN  
            RETURN zero  
        END IF  
        
        FUNCTION can_be_parent(u, v)  
            RETURN for each neighbor IN the set associated with v in graph that the neighbor is IN the set associated with u in graph OR the neighbor EQUALS u  
        END FUNCTION  
        
        SET ways TO one  
        
        FOR each u IN keys of graph  
            IF u EQUALS max_degree THEN  
                CONTINUE to next u  
            END IF  
            
            SET parent TO none  
            FOR each v IN the set associated with u in graph  
                IF v IN graph AND the count of elements associated with v in graph GREATER THAN OR EQUAL TO the count of elements associated with u in graph AND can_be_parent(v, u) THEN  
                    IF parent IS none OR the count of elements associated with v in graph LESS THAN the count of elements associated with parent in graph THEN  
                        SET parent TO v  
                    END IF  
                END IF  
            END FOR  
            
            IF parent IS none THEN  
                RETURN zero  
            END IF  
            
            IF the count of elements associated with parent in graph EQUALS the count of elements associated with u in graph THEN  
                SET ways TO two  
            END IF  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
