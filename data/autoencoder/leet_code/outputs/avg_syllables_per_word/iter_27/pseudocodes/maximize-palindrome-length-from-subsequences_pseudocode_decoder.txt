CLASS Solution  
    FUNCTION longestPalindrome(word1 word1 STRING word2 word2 STRING) RETURNS INTEGER  
        SET s TO concatenation of word1 PLUS word2  
        SET n TO length of s  
        SET dp TO list of lists of zero of size n by n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i of element at position i of dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i of element at position j of dp TO element at position i PLUS one of element at position j MINUS one of dp PLUS two  
                    IF i LESS THAN length of word1 AND j GREATER THAN OR EQUAL TO length of word1  
                        SET max_len TO maximum of max_len AND element at position i of element at position j of dp  
                    END IF  
                ELSE  
                    SET element at position i of element at position j of dp TO maximum of element at position i PLUS one of element at position j of dp AND element at position i of element at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
