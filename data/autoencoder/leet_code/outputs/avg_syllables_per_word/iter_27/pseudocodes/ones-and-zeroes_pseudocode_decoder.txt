CLASS Solution  
    FUNCTION findMaxForm(strs PARAMETERS m PARAMETERS n) RETURNS integer  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns filled with zero  
        FOR each string s IN strs  
            SET count_0 TO the number of occurrences of character zero in string s  
            SET count_1 TO the number of occurrences of character one in string s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET value_without_s TO the element at position i and j of dp  
                    SET value_with_s TO the element at position i MINUS count_0 and j MINUS count_1 of dp PLUS one  
                    SET the element at position i and j of dp TO the maximum of value_without_s AND value_with_s  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position m and n of dp  
    END FUNCTION  
END CLASS
