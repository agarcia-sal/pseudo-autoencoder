CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a mapping from each character from the character a to the character z TO zero  
        SET current_length TO zero  
        FOR index i FROM zero TO the length of s MINUS one  
            IF i GREATER THAN zero AND (the Unicode code of the element at position i of s MINUS the Unicode code of the element at position i MINUS one of s EQUALS one OR the element at position i of s EQUALS the character a AND the element at position i MINUS one of s EQUALS the character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at the element at position i of s TO the maximum VALUE BETWEEN max_length at the element at position i of s AND current_length  
        END FOR  
        SET total TO zero  
        FOR each value IN max_length  
            INCREMENT total BY value  
        END FOR  
        RETURN total  
    END FUNCTION  
END CLASS
