CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO the number one billion  
            CONVERT power_of_2 TO a string of digits  
            SORT the digits of the string in ascending order  
            CONVERT the sorted digits to a tuple  
            ADD the tuple of sorted digits TO power_of_2_sorted_digits  
            MULTIPLY power_of_2 BY two and SET power_of_2 TO the result  
        END WHILE  
        CONVERT the parameter n TO a string of digits  
        SORT the digits of the string in ascending order  
        CONVERT the sorted digits TO a tuple AND ASSIGN TO sorted_n_digits  
        IF sorted_n_digits IS IN power_of_2_sorted_digits THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
