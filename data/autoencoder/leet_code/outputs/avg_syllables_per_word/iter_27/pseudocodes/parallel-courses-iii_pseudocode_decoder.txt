CLASS Solution  
    FUNCTION minimumTime(n, relations, time)  
        SET graph TO a new default dictionary with list as default value  
        FOR each prevCourse AND nextCourse IN relations  
            APPEND prevCourse TO the list at key nextCourse in graph  
        END FOR  
        
        SET memo TO a list of length n PLUS one filled with negative one  
        
        FUNCTION dfs(course)  
            IF element at position course of memo NOT EQUALS negative one THEN  
                RETURN element at position course of memo  
            END IF  
            
            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN the list at key course in graph  
                SET current_time TO result of dfs called with argument prevCourse  
                IF current_time GREATER THAN max_prerequisite_time THEN  
                    SET max_prerequisite_time TO current_time  
                END IF  
            END FOR  
            
            SET element at position course of memo TO max_prerequisite_time PLUS element at position course MINUS one of time  
            RETURN element at position course of memo  
        END FUNCTION  
        
        SET total_minimum_time TO zero  
        FOR course FROM one TO n  
            SET current_dfs_time TO result of dfs called with argument course  
            IF current_dfs_time GREATER THAN total_minimum_time THEN  
                SET total_minimum_time TO current_dfs_time  
            END IF  
        END FOR  
        
        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
