CLASS Solution  
    FUNCTION orderlyQueue(s STRING_PARAMETER k INTEGER_PARAMETER) RETURNS STRING  
        IF k GREATER THAN one THEN  
            RETURN the concatenation of the characters of s sorted in ascending order  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the substring of s from position i TO the end FOLLOWED BY the substring of s from position one TO position i MINUS one  
                IF rotated LESS THAN smallest THEN  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
