CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the number represented by the element at position one of the time split by the character colon converted to integer  
            SET minutes TO the number represented by the element at position two of the time split by the character colon converted to integer  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  

        SET minutes_list TO a list of results of applying the function to_minutes to each element in timePoints  
        SORT minutes_list in ascending order  
        SET min_diff TO positive infinity  
        FOR index FROM two TO the LENGTH OF minutes_list  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO one thousand four hundred forty MINUS the last element of minutes_list PLUS the first element of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
