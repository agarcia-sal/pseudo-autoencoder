CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS the open parenthesis symbol  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET last_score TO the element REMOVED FROM the end of stack  
                IF score EQUALS zero  
                    SET score TO last_score PLUS one  
                ELSE  
                    SET score TO last_score PLUS two MULTIPLIED BY score  
                END IF  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
