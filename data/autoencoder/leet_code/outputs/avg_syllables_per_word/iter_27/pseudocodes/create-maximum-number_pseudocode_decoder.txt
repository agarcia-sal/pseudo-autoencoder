CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack AS a list  
        END FUNCTION  
        
        FUNCTION merge(arr1 arr2)  
            INITIALIZE merged_list TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR each iteration FROM one TO total_length  
                IF arr1 GREATER THAN arr2  
                    REMOVE the element at position one FROM arr1 AND APPEND it TO merged_list  
                ELSE  
                    REMOVE the element at position one FROM arr2 AND APPEND it TO merged_list  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k INCLUSIVE  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merge WITH maxSingleArray(nums1 i) AND maxSingleArray(nums2 k MINUS i)  
                IF max_number LESS THAN candidate  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
