CLASS Solution  
    FUNCTION getPermutation(n k)  
        SET numbers TO list of elements from one TO n  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n TO one DECREMENTING BY one  
            SET fact TO factorial of i MINUS one  
            SET index TO the integer division of k BY fact  
            APPEND element at position index of numbers converted to string TO result  
            REMOVE element at position index FROM numbers  
            SET k TO the remainder of k DIVIDED BY fact  
        END FOR  
        SET joined_result TO concatenation of all elements in result as a single string  
        RETURN joined_result  
    END FUNCTION  
END CLASS
