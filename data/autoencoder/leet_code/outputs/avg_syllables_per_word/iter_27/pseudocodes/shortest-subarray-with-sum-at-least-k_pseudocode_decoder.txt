CLASS Solution  
    FUNCTION shortestSubarray(nums k)  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            SET last_element TO element at the last position of prefix_sum  
            SET new_sum TO last_element PLUS num  
            APPEND new_sum TO prefix_sum  
        END FOR  
        
        ASSIGN reference TO an empty double-ended queue named dq  
        SET min_length TO positive infinity  
        
        FOR each pair of index named i AND element named current_sum IN prefix_sum WITH enumeration starting from zero  
            WHILE dq is not empty AND current_sum MINUS element at position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET candidate_length TO i MINUS element at the front position of dq  
                IF candidate_length IS LESS THAN min_length  
                    SET min_length TO candidate_length  
                END IF  
                REMOVE element at the front from dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position at the back of dq in prefix_sum  
                REMOVE element at the back from dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS EQUAL TO positive infinity  
            RETURN negative one  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
