CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO an empty list  
        FOR each token IN tokens  
            IF token EQUALS the plus symbol OR token EQUALS the minus symbol OR token EQUALS the multiplication symbol OR token EQUALS the division symbol  
                SET b TO the last element of stack AND REMOVE the last element from stack  
                SET a TO the last element of stack AND REMOVE the last element from stack  
                IF token EQUALS the plus symbol  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS the minus symbol  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS the multiplication symbol  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS the division symbol  
                    APPEND the integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer value of token TO stack  
            END IF  
        END FOR  
        RETURN the element at position one of stack  
    END FUNCTION  
END CLASS
