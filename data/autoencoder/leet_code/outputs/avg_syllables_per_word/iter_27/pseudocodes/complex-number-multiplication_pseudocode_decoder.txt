CLASS Solution  
    FUNCTION complexNumberMultiply(num1 num2)  
        FUNCTION parse_complex(num)  
            SET parts TO list of substrings of num at each occurrence of the plus sign  
            SET real TO integer converted from element at position zero of parts  
            SET imaginary_string TO substring of element at position one of parts from start TO position before the last character  
            SET imaginary TO integer converted from imaginary_string  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO parse_complex(num1)  
        SET real2 AND imag2 TO parse_complex(num2)  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN string formed by concatenating string representation of real_part plus plus sign plus string representation of imaginary_part plus character i  
    END FUNCTION  
END CLASS
