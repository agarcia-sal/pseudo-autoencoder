CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values EQUALS empty  
        RETURN no value  
    END IF  
    SET root TO new instance of TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the collection queue IS NOT empty  
        SET node TO the first element removed from the left of collection queue  
        IF the value i LESS THAN the length of collection values AND the element at position i of values IS NOT no value  
            SET the left property of node TO new instance of TreeNode with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF the value i LESS THAN the length of collection values AND the element at position i of values IS NOT no value  
            SET the right property of node TO new instance of TreeNode with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(root, val)  
        FUNCTION dfs(root)  
            IF root IS no value  
                RETURN new instance of TreeNode with val as the val parameter  
            END IF  
            IF the val property of root LESS THAN val  
                SET the right property of root TO the result of dfs called with the right property of root  
            ELSE  
                SET the left property of root TO the result of dfs called with the left property of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN the result of dfs called with root  
    END FUNCTION  
END CLASS
