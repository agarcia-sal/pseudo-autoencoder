CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        
        SET n TO n PLUS twenty four DIVIDED BY twenty five rounded down to nearest integer greater than or equal to n PLUS twenty four DIVIDED BY twenty five
        
        SET memo TO empty collection
        
        FUNCTION dp(a b)  
            IF pair consisting of a and b EXISTS IN memo  
                RETURN value associated with pair consisting of a and b IN memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF
            
            SET prob TO zero point two five MULTIPLIED BY sum of dp called with a MINUS four and b, dp called with a MINUS three and b MINUS one, dp called with a MINUS two and b MINUS two, and dp called with a MINUS one and b MINUS three  
            
            ASSIGN value of prob TO pair consisting of a and b IN memo  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp called with n and n  
    END FUNCTION  
END CLASS
