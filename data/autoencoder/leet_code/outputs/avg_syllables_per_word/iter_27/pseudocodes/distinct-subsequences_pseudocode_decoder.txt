CLASS Solution  
    FUNCTION numDistinct(s AS string, t AS string) RETURNS integer  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        CREATE a two dimensional list dp with the dimensions m PLUS one BY n PLUS one FILLED WITH zero  
        FOR i FROM zero TO m  
            SET the element at position i AND zero of dp TO one  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the character at position i MINUS one of s EQUALS the character at position j MINUS one of t THEN  
                    SET the element at position i AND j of dp TO the element at position i MINUS one AND j MINUS one of dp PLUS the element at position i MINUS one AND j of dp  
                ELSE  
                    SET the element at position i AND j of dp TO the element at position i MINUS one AND j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position m AND n of dp  
    END FUNCTION  
END CLASS
