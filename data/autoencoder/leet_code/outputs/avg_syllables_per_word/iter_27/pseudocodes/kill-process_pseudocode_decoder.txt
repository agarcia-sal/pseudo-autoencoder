CLASS Solution  
    FUNCTION killProcess(pid parameter, ppid parameter, kill parameter) RETURNS list  
        SET tree TO a new dictionary with default value as empty list  
        FOR each pair of child and parent obtained in parallel from pid and ppid  
            APPEND child TO the list corresponding to parent in tree  
        END FOR  
        SET queue TO a new list containing kill  
        SET killed_processes TO a new empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the element at position zero of queue  
            REMOVE the element at position zero FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY the list of children corresponding to current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
