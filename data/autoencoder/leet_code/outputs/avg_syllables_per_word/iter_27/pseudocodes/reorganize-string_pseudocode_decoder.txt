CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of each character in s  
        SET max_heap TO list of pairs where each pair is negative of frequency and character for each character and frequency in count  
        TRANSFORM max_heap INTO a heap data structure maintaining heap property  
        
        SET prev_char TO no value  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN empty string  
            END IF  
            
            REMOVE and GET pair of freq and char from top element of max_heap  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                ADD pair of prev_freq and prev_char TO max_heap maintaining heap property  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
