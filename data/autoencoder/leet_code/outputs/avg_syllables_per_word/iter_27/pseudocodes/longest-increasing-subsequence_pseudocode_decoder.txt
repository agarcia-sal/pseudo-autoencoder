CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS an empty list THEN  
            RETURN zero  
        END IF  
        
        SET dp TO an empty list  
        
        FOR each num IN nums  
            SET idx TO the result of finding the left insertion position of num IN dp using binary search  
            
            IF idx EQUALS the length of dp THEN  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
        
        RETURN the length of dp  
    END FUNCTION  
END CLASS
