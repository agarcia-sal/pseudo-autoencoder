CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO a mapping of digit frequencies in digits  
        SET total_sum TO the sum of all digits  
        SET remainder TO the remainder of total_sum DIVIDED BY three  
        
        IF remainder EQUALS one  
            FOR each digit d IN the list of digits one four seven  
                IF the frequency of digit d in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit d in count BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list of digits two five eight  
                    IF the frequency of digit d in count IS GREATER THAN one  
                        DECREMENT the frequency of digit d in count BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit d IN the list of digits two five eight  
                IF the frequency of digit d in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit d in count BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list of digits one four seven  
                    IF the frequency of digit d in count IS GREATER THAN one  
                        DECREMENT the frequency of digit d in count BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
        END IF  
        
        SET result TO an empty list  
        FOR each digit d FROM nine DOWN TO zero  
            SET string_of_digit TO the textual representation of digit d repeated the frequency of digit d in count times  
            APPEND string_of_digit TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF final_number IS empty OR the first element of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
