CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        
        SET num_counts TO empty mapping  
        FOR each num IN nums  
            IF element at position num IN num_counts EXISTS  
                SET element at position num IN num_counts TO element at position num IN num_counts PLUS one  
            ELSE  
                SET element at position num IN num_counts TO one  
            END IF  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF element at position num IN num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF element at position num PLUS k IN keys of num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
