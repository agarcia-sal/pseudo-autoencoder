CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET total_sum TO the SUM OF all elements IN nums  
        SET current_sum TO the SUM OF each index MULTIPLIED BY the element at position index OF nums FOR each index and element IN nums  
        SET max_value TO current_sum  
        FOR k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY the element at position the LENGTH OF nums MINUS k OF nums  
            IF max_value LESS THAN current_sum  
                SET max_value TO current_sum  
            END IF  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
