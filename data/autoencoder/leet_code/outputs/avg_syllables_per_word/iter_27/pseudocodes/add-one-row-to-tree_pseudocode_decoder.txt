CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    CREATE an empty deque called queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element obtained by removing the first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one THEN  
            RETURN a new TreeNode constructed with val and left assigned to root and right assigned to None  
        END IF  
        CREATE an empty deque called queue  
        APPEND the tuple consisting of root and one TO queue  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET node, level TO the elements obtained by removing the first element FROM queue  
            IF level EQUALS depth MINUS one THEN  
                SET node.left TO a new TreeNode constructed with val and left assigned to node.left and right assigned to None  
                SET node.right TO a new TreeNode constructed with val and right assigned to node.right and left assigned to None  
            ELSE IF level LESS THAN depth MINUS one THEN  
                IF node.left IS NOT None THEN  
                    APPEND the tuple consisting of node.left and level PLUS one TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND the tuple consisting of node.right and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
