CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS nothing OR the LENGTH OF board EQUALS zero  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a list containing m elements each being a list containing n elements each set to false  
        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x then position y of visited EQUALS true OR element at position x then position y of board NOT EQUALS element at position index of word  
                RETURN false  
            END IF  
            SET element at position x then position y of visited TO true  
            FOR each pair dx and dy IN the list containing pairs negative one and zero then one and zero then zero and negative one then zero and one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one)  
                    RETURN true  
                END IF  
            END FOR  
            SET element at position x then position y of visited TO false  
            RETURN false  
        END FUNCTION  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero)  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
