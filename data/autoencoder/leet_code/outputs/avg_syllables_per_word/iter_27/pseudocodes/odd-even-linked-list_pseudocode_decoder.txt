CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO new ListNode with val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN p1 dot val EQUALS p2 dot val AND is_same_list(p1 dot next p2 dot next)  
END FUNCTION

CLASS Solution  
    FUNCTION oddEvenList(self head)  
        IF head EQUALS None OR head dot next EQUALS None OR head dot next dot next EQUALS None THEN  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head dot next  
        SET even_head TO even  
        WHILE even IS NOT None AND even dot next IS NOT None  
            SET odd dot next TO even dot next  
            SET odd TO odd dot next  
            SET even dot next TO odd dot next  
            SET even TO even dot next  
        END WHILE  
        SET odd dot next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
