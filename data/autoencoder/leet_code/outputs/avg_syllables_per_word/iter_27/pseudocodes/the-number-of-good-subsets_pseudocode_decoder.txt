CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each element in nums  
        SET dp TO a list of zeroes of LENGTH two MULTIPLIED BY the power of ten  
        SET the element at position zero of dp TO one  
  
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY the power of two RAISED TO the count of one in count  
                CONTINUE to the next iteration  
            END IF  
            IF num NOT IN the keys of prime_factors  
                CONTINUE to the next iteration  
            END IF  
  
            SET factors TO the value in prime_factors at key num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR the power of two RAISED TO prime MINUS one  
            END FOR  
  
            FOR m FROM two power ten MINUS one DOWN TO zero  
                IF the bitwise AND of m AND mask EQUALS zero  
                    SET the element at position bitwise OR of m AND mask of dp TO the sum of the element at position bitwise OR of m AND mask of dp AND the product of the element at position m of dp AND the count of num in count  
                    SET the element at position bitwise OR of m AND mask of dp TO the remainder of the previous result DIVIDED BY MOD  
                END IF  
            END FOR  
        END FOR  
  
        SET result TO the sum of elements of dp from position one TO the end  
        SET result TO the remainder of result DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
