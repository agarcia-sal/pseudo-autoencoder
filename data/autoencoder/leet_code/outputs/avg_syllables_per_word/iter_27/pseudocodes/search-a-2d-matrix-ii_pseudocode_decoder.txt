CLASS Solution  
    FUNCTION searchMatrix(matrix parameter, target parameter) RETURNS boolean  
        IF the matrix parameter EQUALS nothing OR the element at position zero of the matrix parameter EQUALS nothing  
            RETURN false  
        END IF  
        SET row TO zero  
        SET col TO the length of the element at position zero of the matrix parameter MINUS one  
        WHILE row LESS THAN the length of the matrix parameter AND col GREATER THAN OR EQUAL TO zero  
            IF the element at position row of the matrix parameter at position col EQUALS target parameter  
                RETURN true  
            ELSE IF the element at position row of the matrix parameter at position col GREATER THAN target parameter  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
