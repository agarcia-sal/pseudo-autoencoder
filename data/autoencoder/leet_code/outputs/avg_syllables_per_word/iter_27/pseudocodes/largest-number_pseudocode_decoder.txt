CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a list of string representations of each element in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x FOLLOWED BY y IS GREATER THAN concatenation of y FOLLOWED BY x  
                RETURN negative one  
            ELSE IF concatenation of x FOLLOWED BY y IS LESS THAN concatenation of y FOLLOWED BY x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING compare AS the comparison function  
        
        IF element at position zero of nums_str EQUALS string representation of zero  
            RETURN string representation of zero  
        END IF  
        
        SET result TO concatenation of all elements in nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
