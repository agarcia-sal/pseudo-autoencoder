CLASS Solution  
    FUNCTION jump(nums)  
        IF the LENGTH OF nums LESS THAN OR EQUAL TO one THEN  
            RETURN zero  
        END IF  
        SET jumps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF farthest LESS THAN the SUM of i PLUS the element at position i of nums THEN  
                SET farthest TO the SUM of i PLUS the element at position i of nums  
            END IF  
            IF i EQUALS current_end THEN  
                INCREMENT jumps BY one  
                SET current_end TO farthest  
                IF current_end GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        RETURN jumps  
    END FUNCTION  
END CLASS
