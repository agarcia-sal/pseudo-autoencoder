CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SORT nums IN ASCENDING ORDER  
        SET sorted_nums TO A COPY OF nums  
        SET n TO THE LENGTH OF nums  
        SET left TO THE INTEGER DIVISION OF n MINUS one BY two  
        SET right TO n MINUS one  
        FOR index i FROM zero TO n MINUS one  
            IF the remainder when index i IS DIVIDED BY two EQUALS zero  
                SET element at position i of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
