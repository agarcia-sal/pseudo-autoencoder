CLASS Solution  
    FUNCTION findReplaceString(s string_parameter indices list_of_integers_parameter sources list_of_strings_parameter targets list_of_strings_parameter) RETURNS string  
        SET replacements TO a list of tuples each containing the element at the same position of indices sources and targets sorted by the first element of each tuple in ascending order  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each tuple composed of index source target IN replacements  
            APPEND the substring from position prev_end TO position index of string s TO result  
            IF the substring from position index TO position index PLUS the length of source of string s EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring from position index TO position index PLUS the length of source of string s TO result  
            END IF  
            SET prev_end TO index PLUS the length of source  
        END FOR  
        APPEND the substring from position prev_end TO the end of string s TO result  
        SET joined_result TO the concatenation of all elements in result  
        RETURN joined_result  
    END FUNCTION  
END CLASS
