CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left attribute of p AND the left attribute of q AND the result of calling is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree with parameters root  
        FUNCTION getHeight with parameter node  
            IF node IS no value THEN  
                RETURN zero  
            END IF  
            RETURN one PLUS the greater of the result of calling getHeight with the left attribute of node AND the result of calling getHeight with the right attribute of node  
        END FUNCTION  

        SET height TO the result of calling getHeight with root  
        SET width TO two RAISED TO THE power of height MINUS one  
        SET res TO a list composed of height elements WHERE each element IS a list composed of width elements EACH BEING the empty string  
        
        FUNCTION placeNode with parameters node row left right  
            IF node IS no value THEN  
                RETURN  
            END IF  
            SET mid TO the integer division of the sum of left AND right BY two  
            SET the element at position mid of the list at position row of res TO the string form of the value attribute of node  
            CALL placeNode with the left attribute of node AND the increment by one of row AND left AND mid  
            CALL placeNode with the right attribute of node AND the increment by one of row AND mid AND right  
        END FUNCTION  

        CALL placeNode with root AND zero AND zero AND width  
        RETURN res  
    END FUNCTION  
END CLASS
