CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeros with dimensions of n PLUS one by n PLUS one  
        
        FOR each j FROM zero TO n  
            SET element at position j of the first row of dp TO one  
        END FOR  
        
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the character indicating INCREASING sequence THEN  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one and j of dp MODULO MOD  
                    SET element at position i and j of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWNTO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one and j PLUS one of dp MODULO MOD  
                    SET element at position i and j of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the SUM of elements in the n-th row of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
