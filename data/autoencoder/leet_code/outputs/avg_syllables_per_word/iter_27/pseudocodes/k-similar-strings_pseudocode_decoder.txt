CLASS Solution  
    FUNCTION kSimilarity(s1 parameter s2 parameter) RETURNS integer  
        FUNCTION neighbors(s parameter)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD concatenation of substring from start TO position i MINUS one of s PLUS element at position j of s PLUS substring from position i PLUS one TO position j MINUS one of s PLUS element at position i of s PLUS substring from position j PLUS one TO end of s  
                END IF  
            END FOR  
        END FUNCTION  

        SET queue TO a new deque containing the tuple of s1 and zero  
        SET visited TO a new set containing s1  

        WHILE queue is not empty  
            REMOVE and ASSIGN the leftmost element of queue TO s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors called with s  
                IF neighbor is NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor AND swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
