CLASS Solution  
    FUNCTION escapeGhosts(ghosts target)  
        SET player_distance TO the absolute value of element at position zero of target PLUS the absolute value of element at position one of target  
        FOR each ghost IN ghosts  
            SET ghost_distance TO the absolute value of the difference between element at position zero of ghost AND element at position zero of target PLUS the absolute value of the difference between element at position one of ghost AND element at position one of target  
            IF ghost_distance IS LESS THAN OR EQUAL TO player_distance  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
