CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO an empty mapping with default value zero  
        FOR each cpdomain IN cpdomains  
            SET count and domain TO the result of splitting cpdomain by the space character into two parts  
            SET count TO the integer value of count  
            SET subdomains TO the result of splitting domain by the period character  
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain TO the concatenation of the elements of subdomains from position index TO the last element separated by the period character  
                INCREMENT the value at subdomain in domain_count BY count  
            END FOR  
        END FOR  
        RETURN the list constructed by FOR each domain and count IN the items of domain_count RETURNING the concatenation of count converted to string PLUS a space PLUS domain as a string  
    END FUNCTION  
END CLASS
