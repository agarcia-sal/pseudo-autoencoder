CLASS Solution  
    FUNCTION computeArea(ax1 ay1 ax2 ay2 bx1 by1 bx2 by2)  
        SET area1 TO the difference BETWEEN ax2 AND ax1 MULTIPLIED BY the difference BETWEEN ay2 AND ay1  
        SET area2 TO the difference BETWEEN bx2 AND bx1 MULTIPLIED BY the difference BETWEEN by2 AND by1  
        SET overlap_width TO the maximum value BETWEEN zero AND the difference BETWEEN the minimum value BETWEEN ax2 AND bx2 AND the maximum value BETWEEN ax1 AND bx1  
        SET overlap_height TO the maximum value BETWEEN zero AND the difference BETWEEN the minimum value BETWEEN ay2 AND by2 AND the maximum value BETWEEN ay1 AND by1  
        SET overlap_area TO the product OF overlap_width AND overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
