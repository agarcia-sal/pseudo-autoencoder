CLASS Solution  
    FUNCTION spiralMatrixIII(rows rRows cols rCols rStart rRStart cStart rCStart)  
        SET directions TO list of pairs direction East direction South direction West direction North  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rRStart  
        SET y TO rCStart  
          
        WHILE the LENGTH OF visited IS LESS THAN the MULTIPLICATION OF rRows AND rCols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each step FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rRows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN rCols  
                    APPEND list of x AND y TO visited  
                END IF  
                SET dx TO the element at position zero of the element at position direction_index of directions  
                SET dy TO the element at position one of the element at position direction_index of directions  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO (direction_index PLUS one) MODULO four  
            INCREMENT increment BY one  
        END WHILE  
          
        RETURN visited  
    END FUNCTION  
END CLASS
