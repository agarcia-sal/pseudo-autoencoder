CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list of characters representing the number n converted to a string  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one BY stepping downwards by one  
            IF the element at position i of digits LESS THAN the element at position i MINUS one of digits  
                SET the element at position i MINUS one of digits TO the character representing the integer value of the element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i of digits TO the character representing the digit nine  
        END FOR  
        RETURN the integer value of the concatenation of all elements in digits  
    END FUNCTION  
END CLASS
