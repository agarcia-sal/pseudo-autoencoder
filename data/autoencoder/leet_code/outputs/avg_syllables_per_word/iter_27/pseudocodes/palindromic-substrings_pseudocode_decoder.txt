CLASS Solution  
    FUNCTION countSubstrings WITH parameter s  
        FUNCTION expand_around_center WITH parameters left AND right  
            SET count TO zero  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN THE LENGTH OF s AND element at position left of s EQUALS element at position right of s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET total_palindromes TO zero  
        FOR i FROM zero TO THE LENGTH OF s MINUS one  
            SET total_palindromes TO total_palindromes PLUS expand_around_center WITH arguments i AND i  
            SET total_palindromes TO total_palindromes PLUS expand_around_center WITH arguments i AND i PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
