CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity value  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            PUSH element at position zero of element at position index of nums AND index AND zero INTO min_heap USING heap push  
            IF max_value LESS THAN element at position zero of element at position index of nums THEN  
                SET max_value TO element at position zero of element at position index of nums  
            END IF  
        END FOR  
        
        SET smallest_range TO positive infinity value  
        SET result TO list containing the large negative value AND the large positive value  
        
        WHILE min_heap is not empty  
            POP the smallest element FROM min_heap USING heap pop AND ASSIGN to min_value AND list_index AND element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            
            IF element_index PLUS one LESS THAN the LENGTH OF element at position list_index of nums THEN  
                SET next_value TO element at position element_index PLUS one of element at position list_index of nums  
                PUSH next_value AND list_index AND element_index PLUS one INTO min_heap USING heap push  
                IF max_value LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK the loop  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
