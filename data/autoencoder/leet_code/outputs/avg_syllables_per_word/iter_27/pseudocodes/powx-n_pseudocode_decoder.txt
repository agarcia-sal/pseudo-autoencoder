CLASS Solution  
    FUNCTION myPow(x FLOATING POINT, n INTEGER) RETURNS FLOATING POINT  
        FUNCTION qpow(a FLOATING POINT, n INTEGER) RETURNS FLOATING POINT  
            SET answer TO one  
            WHILE n GREATER THAN zero  
                IF n MODULO two EQUALS one  
                    SET answer TO answer MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two  
            END WHILE  
            RETURN answer  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH a EQUAL TO x AND n EQUAL TO n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH a EQUAL TO x AND n EQUAL TO negative n  
        END IF  
    END FUNCTION  
END CLASS
