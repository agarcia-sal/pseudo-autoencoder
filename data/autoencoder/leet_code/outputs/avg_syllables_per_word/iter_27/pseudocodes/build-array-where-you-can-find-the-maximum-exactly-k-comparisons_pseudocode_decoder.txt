CLASS Solution  
    FUNCTION numOfArrays(n INTEGER parameter, m INTEGER parameter, k INTEGER parameter) RETURNS INTEGER  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS three-dimensional list filled with zero  
        WHERE the first dimension length IS n PLUS one  
        AND the second dimension length IS m PLUS one  
        AND the third dimension length IS k PLUS one  
        
        FOR each j FROM one TO m DO  
            SET element at position one of element at position j of element at position one of dp TO one  
        END FOR  
        
        FOR each i FROM two TO n DO  
            FOR each j FROM one TO m DO  
                FOR each p FROM one TO k DO  
                    FOR each x FROM one TO j MINUS one DO  
                        SET element at position p of element at position j of element at position i of dp TO  
                            the value of element at position p of element at position j of element at position i of dp PLUS  
                            the value of element at position p MINUS one of element at position x of element at position i MINUS one of dp  
                            MODULO MOD  
                    END FOR  
                    SET element at position p of element at position j of element at position i of dp TO  
                        the value of element at position p of element at position j of element at position i of dp PLUS  
                        the product of the value of element at position p of element at position j of element at position i MINUS one of dp MULTIPLIED BY j  
                        MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each j FROM one TO m DO  
            SET result TO result PLUS the value of element at position k of element at position j of element at position n of dp  
            APPLY MODULO MOD TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
