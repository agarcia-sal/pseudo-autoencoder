CLASS Solution  
    FUNCTION longestPath(parent, s)  
        IMPORT collections default dictionary  
        SET tree TO a default dictionary with list as default value  
        FOR each index i WITH element p IN parent  
            IF p NOT EQUALS negative one  
                APPEND i TO the list at element p of tree  
            END IF  
        END FOR  
        SET self dot result TO one  
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN element at position node of tree  
                SET child_length TO dfs with argument child  
                IF character at position child of s NOT EQUALS character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET self dot result TO the maximum of self dot result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        CALL dfs with argument zero  
        RETURN self dot result  
    END FUNCTION  
END CLASS
