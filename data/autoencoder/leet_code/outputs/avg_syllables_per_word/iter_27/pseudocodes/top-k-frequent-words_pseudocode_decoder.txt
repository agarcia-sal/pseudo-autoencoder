CLASS Solution  
    FUNCTION topKFrequent(words LIST OF STRINGS k INTEGER) RETURNS LIST OF STRINGS  
        SET count TO result of counting frequency of each element IN words  
        SET heap TO empty list  
        FOR each element word AND its frequency freq IN count  
            APPEND tuple of negative of freq AND word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap structure using heap property  
        SET result TO empty list  
        FOR index FROM one TO k  
            REMOVE the element with the smallest value FROM heap  
            EXTRACT the second element of the removed tuple AS current_word  
            APPEND current_word TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
