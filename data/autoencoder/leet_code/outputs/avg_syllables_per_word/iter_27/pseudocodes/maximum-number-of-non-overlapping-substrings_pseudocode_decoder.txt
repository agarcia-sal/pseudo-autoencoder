CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO empty mapping  
        SET last TO empty mapping  
        FOR each index i AND character c IN enumeration of s  
            IF character c NOT IN first  
                SET first at character c TO index i  
            END IF  
            SET last at character c TO index i  
        END FOR  
        
        SET intervals TO empty list  
        FOR each character c IN keys of first  
            SET start TO first at character c  
            SET end TO last at character c  
            SET i TO start  
            WHILE i LESS THAN OR EQUAL TO end  
                IF first at character at position i of s LESS THAN start  
                    EXIT WHILE  
                END IF  
                SET end TO the greater value BETWEEN end AND last at character at position i of s  
                INCREMENT i BY one  
            END WHILE  
            IF i GREATER THAN end  
                APPEND pair of start AND end TO intervals  
            END IF  
        END FOR  
        
        SORT intervals BY the second element of each pair in ascending order  
        
        SET result TO empty list  
        SET prev_end TO a value less than any valid index  
        FOR each pair start AND end IN intervals  
            IF start GREATER THAN prev_end  
                APPEND substring from position start TO position end of s TO result  
                SET prev_end TO end  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
