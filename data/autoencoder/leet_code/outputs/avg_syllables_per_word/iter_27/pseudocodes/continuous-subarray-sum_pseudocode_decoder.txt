CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a dictionary with key zero and value negative one  
        SET current_sum TO zero  
        FOR each index i and element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero  
                SET current_sum TO the remainder of current_sum DIVIDED BY k  
            END IF  
            IF current_sum IS A KEY IN remainder_map  
                IF index i MINUS the value associated with current_sum in remainder_map IS GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the value i TO remainder_map at key current_sum  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
