CLASS Solution  
    FUNCTION findSubstring(s word_string words list_of_strings) RETURNS list_of_integers  
        IF s EQUALS empty OR words EQUALS empty OR element at position zero of words EQUALS empty  
            RETURN empty list  
        END IF  
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNT of each word in words  
        SET result TO empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO empty COUNT  
            FOR j FROM i TO i PLUS total_length MINUS one STEP BY word_length  
                SET word TO substring from position j TO position j PLUS word_length MINUS one of s  
                INCREMENT the count of word in seen BY one  
                IF the count of word in seen GREATER THAN the count of word in word_count  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
            IF the inner FOR loop completed without BREAK  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
