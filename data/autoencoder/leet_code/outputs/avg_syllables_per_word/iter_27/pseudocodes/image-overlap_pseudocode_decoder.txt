CLASS Solution  
    FUNCTION largestOverlap(img1 parameter, img2 parameter)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img1 EQUALS one  
                    APPEND a pair consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img2 EQUALS one  
                    APPEND a pair consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        ASSIGN overlap_count TO a new mapping with default integer zero  
        FOR each pair x1 AND y1 IN points1  
            FOR each pair x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value associated with the pair dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_value TO zero  
        FOR each value IN the values of overlap_count  
            IF value GREATER THAN maximum_value  
                SET maximum_value TO value  
            END IF  
        END FOR  
        RETURN maximum_value  
    END FUNCTION  
END CLASS
