CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a list of lists where the NUMBER OF OUTER ELEMENTS IS size1 PLUS one and the NUMBER OF INNER ELEMENTS IS two TO THE POWER OF size2; EACH INNER ELEMENT IS initialized TO positive infinity  
        SET element at position zero of element at position zero of dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two TO THE POWER OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET masked_position TO the result of the logical OR between mask and two TO THE POWER OF j  
                    SET element at position masked_position of element at position i PLUS one of dp TO the minimum of element at position masked_position of element at position i PLUS one of dp AND the sum of element at position mask of element at position i of dp PLUS element at position i of cost at position j  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF the result of the logical AND between mask and two TO THE POWER OF j HOLDS  
                        SET toggled_mask TO the result of the logical XOR between mask and two TO THE POWER OF j  
                        SET element at position mask of element at position i PLUS one of dp TO the minimum of element at position mask of element at position i PLUS one of dp AND the sum of element at position toggled_mask of element at position i PLUS one of dp PLUS element at position i of cost at position j  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position size1 of element at position two TO THE POWER OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
