CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode CREATED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new ListNode CREATED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the expression that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode CREATED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the left attribute of node TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the right attribute of node TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree CALLED WITH the left attribute of p AND the left attribute of q AND the result of is_same_tree CALLED WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s THEN  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (the element at position index of s IS a digit OR the element at position index of s EQUALS the character HYPHEN)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO a new TreeNode CREATED WITH the conversion TO integer OF the substring from position start TO position index of s  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character LEFT PARENTHESIS THEN  
                SET the left attribute of node AND index TO the values RETURNED BY parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character LEFT PARENTHESIS THEN  
                SET the right attribute of node AND index TO the values RETURNED BY parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character RIGHT PARENTHESIS THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND an unused variable TO the values RETURNED BY parse_tree CALLED WITH zero  
        RETURN root  
    END FUNCTION  
END CLASS
