CLASS Solution  
    FUNCTION maxJumps(arr parameter arr d parameter d)  
        FUNCTION dp(i parameter i)  
            SET max jumps TO one  
            FOR x parameter x FROM one TO d  
                IF i PLUS x LESS THAN the LENGTH OF arr AND element at position i PLUS x of arr GREATER THAN OR EQUAL TO element at position i of arr THEN  
                    BREAK loop  
                ELSE IF i PLUS x LESS THAN the LENGTH OF arr THEN  
                    SET candidate jumps TO one PLUS dp called with i PLUS x  
                    IF candidate jumps GREATER THAN max jumps THEN  
                        SET max jumps TO candidate jumps  
                    END IF  
                END IF  
            END FOR  
            FOR x parameter x FROM one TO d  
                IF i MINUS x GREATER THAN OR EQUAL TO zero AND element at position i MINUS x of arr GREATER THAN OR EQUAL TO element at position i of arr THEN  
                    BREAK loop  
                ELSE IF i MINUS x GREATER THAN OR EQUAL TO zero THEN  
                    SET candidate jumps TO one PLUS dp called with i MINUS x  
                    IF candidate jumps GREATER THAN max jumps THEN  
                        SET max jumps TO candidate jumps  
                    END IF  
                END IF  
            END FOR  
            RETURN max jumps  
        END FUNCTION  
        SET max result TO zero  
        FOR i parameter i FROM zero TO the LENGTH OF arr MINUS one  
            SET current result TO dp called with i  
            IF current result GREATER THAN max result THEN  
                SET max result TO current result  
            END IF  
        END FOR  
        RETURN max result  
    END FUNCTION  
END CLASS
