CLASS Solution  
    FUNCTION shortestDistance(grid)  
        IF grid EQUALS not OR element at position zero of grid EQUALS not  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET total_distance TO list of lists sized m by n filled with zero  
        SET reach_count TO list of lists sized m by n filled with zero  
        SET building_count TO zero  
        
        SET directions TO list containing elements representing one step down zero steps sideways one step up zero steps sideways zero steps down one step sideways zero steps down minus one step sideways  
        
        FUNCTION bfs(start_x start_y)  
            SET queue TO new deque containing tuple of start_x and start_y  
            SET visited TO list of lists sized m by n filled with false  
            SET distance TO zero  
            SET reachable TO zero  
            
            WHILE queue has elements  
                SET size TO the LENGTH OF queue  
                INCREMENT distance BY one  
                FOR each _ FROM zero TO size minus one  
                    SET x and y TO the elements removed from the left side of queue  
                    FOR each dx and dy IN directions  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of visited at position ny EQUALS false AND element at position nx of grid at position ny EQUALS zero  
                            SET element at position nx of visited at position ny TO true  
                            APPEND tuple of nx and ny TO queue  
                            SET element at position nx of total_distance at position ny TO element at position nx of total_distance at position ny PLUS distance  
                            SET element at position nx of reach_count at position ny TO element at position nx of reach_count at position ny PLUS one  
                            INCREMENT reachable BY one  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
            RETURN reachable  
        END FUNCTION  
        
        FOR i FROM zero TO m minus one  
            FOR j FROM zero TO n minus one  
                IF element at position i of grid at position j EQUALS one  
                    INCREMENT building_count BY one  
                    IF bfs(i j) LESS THAN building_count MINUS one  
                        RETURN negative one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET min_distance TO infinity  
        FOR i FROM zero TO m minus one  
            FOR j FROM zero TO n minus one  
                IF element at position i of grid at position j EQUALS zero AND element at position i of reach_count at position j EQUALS building_count  
                    IF element at position i of total_distance at position j LESS THAN min_distance  
                        SET min_distance TO element at position i of total_distance at position j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_distance NOT EQUALS infinity  
            RETURN min_distance  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
