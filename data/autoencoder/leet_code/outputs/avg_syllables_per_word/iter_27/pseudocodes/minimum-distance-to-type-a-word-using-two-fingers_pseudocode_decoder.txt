CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the integer division result of the character code of c1 MINUS the character code of the uppercase letter A divided BY six  
            SET y1 TO the remainder of the character code of c1 MINUS the character code of the uppercase letter A divided BY six  
            SET x2 TO the integer division result of the character code of c2 MINUS the character code of the uppercase letter A divided BY six  
            SET y2 TO the remainder of the character code of c2 MINUS the character code of the uppercase letter A divided BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO distance of f1 and c PLUS dp of i PLUS one and c and f2  
            SET d2 TO distance of f2 and c PLUS dp of i PLUS one and f1 and c  
            RETURN the lesser value of d1 and d2  
        END FUNCTION  
        
        RETURN dp of zero and None and None  
    END FUNCTION  
END CLASS
