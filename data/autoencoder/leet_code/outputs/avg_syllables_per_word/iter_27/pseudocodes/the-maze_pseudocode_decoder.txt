CLASS Solution  
    FUNCTION hasPath maze start destination  
        SET directions TO the list of pairs zero one one zero zero negative one negative one zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs x y  
            IF the pair x y IS IN visited  
                RETURN false  
            END IF  
            IF the list consisting of x y EQUALS destination  
                RETURN true  
            END IF  
            ADD the pair x y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND the element at position nx PLUS dx of maze AT position ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs nx ny  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs the element at position zero of start the element at position one of start  
    END FUNCTION  
END CLASS
