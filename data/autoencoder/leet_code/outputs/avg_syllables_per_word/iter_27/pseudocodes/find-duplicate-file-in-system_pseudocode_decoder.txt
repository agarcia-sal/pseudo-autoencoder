CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new default dictionary with list as default value  
        FOR each path IN paths  
            SET parts TO list of elements obtained by splitting path on whitespace  
            SET directory TO element at position zero of parts  
            FOR each file_info IN elements of parts from position one TO the end  
                SET file_name AND content TO elements obtained by splitting file_info at the first occurrence of the character representing opening parenthesis  
                SET content TO substring of content from position zero TO the position before the last character  
                APPEND the concatenation of directory PLUS the character representing forward slash PLUS file_name TO the list in content_to_paths associated with the key content  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each group IN the values of content_to_paths  
            IF the length of group IS GREATER THAN one  
                APPEND group TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
