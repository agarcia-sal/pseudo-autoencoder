CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            IF the PAIR of i AND j IS IN memo THEN  
                RETURN the VALUE of memo AT the PAIR of i AND j  
            END IF  
            SET result TO the VALUE of dp WITH the PARAMETERS i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF the ELEMENT at POSITION k of s EQUALS the ELEMENT at POSITION j of s THEN  
                    SET result TO the MINIMUM VALUE OF result AND the VALUE of dp WITH the PARAMETERS i AND k MINUS one AND memo PLUS the VALUE of dp WITH the PARAMETERS k AND j MINUS one AND memo  
                END IF  
            END FOR  
            SET the VALUE of memo AT the PAIR of i AND j TO result  
            RETURN result  
        END FUNCTION  
        RETURN dp WITH the PARAMETERS zero AND the LENGTH OF s MINUS one AND an EMPTY COLLECTION  
    END FUNCTION  
END CLASS
