CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters node left of p and node left of q AND the result of is_same_tree with parameters node right of p and node right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node IS none THEN  
                RETURN zero AND zero AND infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO the result of dfs called with node left of node  
            SET right_no_camera, right_monitored, right_camera TO the result of dfs called with node right of node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the minimum value AMONG left_camera PLUS the minimum value AMONG right_monitored AND right_camera TOGETHER AND right_camera PLUS the minimum value AMONG left_monitored AND left_camera TOGETHER AND left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum value AMONG left_no_camera AND left_monitored AND left_camera PLUS the minimum value AMONG right_no_camera AND right_monitored AND right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO the result of dfs called with root  
        RETURN the minimum value AMONG root_monitored AND root_camera  
    END FUNCTION  
END CLASS
