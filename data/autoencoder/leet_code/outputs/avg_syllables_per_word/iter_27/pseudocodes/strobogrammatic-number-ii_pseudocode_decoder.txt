CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN result of calling dfs with u MINUS two  
                FOR each pair l AND r IN list of pairs consisting of string one string one string eight string eight string six string nine string nine string six  
                    APPEND concatenation of l PLUS v PLUS r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of string zero PLUS v PLUS string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN result of calling dfs with n  
    END FUNCTION  
END CLASS
