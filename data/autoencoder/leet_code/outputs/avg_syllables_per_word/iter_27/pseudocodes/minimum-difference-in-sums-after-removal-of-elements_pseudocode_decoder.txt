CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with the LENGTH equal to two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            INSERT the NEGATION OF the element at position i of nums INTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN n  
                REMOVE the smallest element FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n  
                SET the element at position i of min_sums TO the NEGATION OF the SUM OF min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with the LENGTH equal to two MULTIPLIED BY n PLUS one  
        
        FOR i FROM the product of three and n MINUS one DOWN TO n  
            INSERT the element at position i of nums INTO max_heap  
            IF the LENGTH OF max_heap GREATER THAN n  
                REMOVE the smallest element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n  
                SET the element at position i of max_sums TO the SUM OF max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            IF min_diff GREATER THAN the element at position i of min_sums MINUS the element at position i PLUS one of max_sums  
                SET min_diff TO the element at position i of min_sums MINUS the element at position i PLUS one of max_sums  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
