CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET constructed from the elements of bank  
        SET q TO a double ended queue containing a PAIR composed of start and zero  
        SET mp TO a MAPPING where the key is character A and the value is the STRING of characters T C G and the key is character T and the value is the STRING of characters A C G and the key is character C and the value is the STRING of characters A T G and the key is character G and the value is the STRING of characters A T C  
        WHILE the QUEUE q is not empty  
            REMOVE the FIRST element from QUEUE q and ASSIGN its FIRST part to t and its SECOND part to step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR i FROM zero TO the LENGTH OF t MINUS one  
                SET v TO the ELEMENT at position i of t  
                FOR each character j IN the STRING mapped by key v in mp  
                    SET next TO the STRING composed of the SUBSTRING from position zero TO position i MINUS one of t PLUS character j PLUS the SUBSTRING from position i PLUS one TO the END of t  
                    IF next IS an ELEMENT OF s  
                        APPEND the PAIR composed of next and step PLUS one TO the QUEUE q  
                        REMOVE next FROM the SET s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
