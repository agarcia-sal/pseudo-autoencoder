CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val MINUS EQUAL TO zero AND next MINUS EQUAL TO None  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH parameter element at position one of values  
    SET p TO head  
    FOR each val IN the list of elements of values FROM position two TO the end  
        SET node TO A NEW ListNode WITH parameter val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND is_same_list WITH PARAMETERS the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION

CLASS Solution  
    FUNCTION reorderList WITH PARAMETER head  
        IF head EQUALS None OR the next ATTRIBUTE OF head EQUALS None  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT None AND the next ATTRIBUTE OF fast IS NOT None  
            SET slow TO the next ATTRIBUTE OF slow  
            SET fast TO the next ATTRIBUTE OF the next ATTRIBUTE OF fast  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr IS NOT None  
            SET next_temp TO the next ATTRIBUTE OF curr  
            SET the next ATTRIBUTE OF curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the next ATTRIBUTE OF second IS NOT None  
            SET temp1 TO the next ATTRIBUTE OF first  
            SET temp2 TO the next ATTRIBUTE OF second  
            SET the next ATTRIBUTE OF first TO second  
            SET the next ATTRIBUTE OF second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
