CLASS Solution  
    FUNCTION multiply(num1  num2)  
        IF num1 EQUALS zero OR num2 EQUALS zero  
            RETURN zero  
        END IF  
        SET result TO list of zeros with length EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the REVERSE of num1  
        SET num2 TO the REVERSE of num2  
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the NUMERIC VALUE OF element at position i of num1 MINUS the NUMERIC VALUE OF zero MULTIPLIED BY the NUMERIC VALUE OF element at position j of num2 MINUS the NUMERIC VALUE OF zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 of result  
                SET element at position p1 of result TO sum_ MODULO ten  
                INCREMENT element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_str TO the CONCATENATION OF string form of each element IN the REVERSE of result  
        RETURN result_str WITH ALL leading zeros REMOVED  
    END FUNCTION  
END CLASS
