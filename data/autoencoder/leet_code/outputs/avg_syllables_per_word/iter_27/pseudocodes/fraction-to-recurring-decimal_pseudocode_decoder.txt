CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero EXCLUSIVE OR denominator LESS THAN zero THEN  
            SET sign TO the string representing minus sign  
        ELSE  
            SET sign TO the empty string  
        END IF  
        
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        
        SET quotient TO the integer division result of numerator DIVIDED BY denominator  
        SET remainder TO the remainder after division of numerator BY denominator  
        SET integer_part TO the string representation of quotient  
        
        IF remainder EQUALS zero THEN  
            RETURN the concatenation of sign AND integer_part  
        END IF  
        
        SET seen_remainders TO an empty dictionary  
        SET fractional_part TO an empty list  
        
        WHILE remainder NOT EQUALS zero DO  
            IF remainder IS A KEY IN seen_remainders THEN  
                SET start TO the value mapped from remainder IN seen_remainders  
                SET non_repeating TO the concatenation of elements FROM position zero UP TO position start IN fractional_part  
                SET repeating TO the concatenation of elements FROM position start TO THE END IN fractional_part  
                RETURN the concatenation of sign, integer_part, the string representing a decimal point, non_repeating, the string representing an opening parenthesis, repeating, and the string representing a closing parenthesis  
            END IF  
            
            ASSIGN TO seen_remainders AT KEY remainder THE the current length of fractional_part  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer division result of remainder DIVIDED BY denominator  
            SET remainder TO the remainder after division of remainder BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN the concatenation of sign, integer_part, the string representing a decimal point, and the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
