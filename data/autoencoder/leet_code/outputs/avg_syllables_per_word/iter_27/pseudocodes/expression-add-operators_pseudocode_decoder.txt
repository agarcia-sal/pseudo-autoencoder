CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS list of string  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS the character zero  
                    CONTINUE  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH parameters i path operand CONVERTED TO integer operand CONVERTED TO integer  
                ELSE  
                    CALL dfs WITH parameters i path CONCATENATED WITH the character plus CONCATENATED WITH operand value PLUS operand CONVERTED TO integer operand CONVERTED TO integer  
                    CALL dfs WITH parameters i path CONCATENATED WITH the character minus CONCATENATED WITH operand value MINUS operand CONVERTED TO integer NEGATION OF operand CONVERTED TO integer  
                    CALL dfs WITH parameters i path CONCATENATED WITH the character asterisk CONCATENATED WITH operand value MINUS prev PLUS prev MULTIPLIED BY operand CONVERTED TO integer prev MULTIPLIED BY operand CONVERTED TO integer  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL dfs WITH parameters zero empty string zero zero  
        RETURN result  
    END FUNCTION  
END CLASS
