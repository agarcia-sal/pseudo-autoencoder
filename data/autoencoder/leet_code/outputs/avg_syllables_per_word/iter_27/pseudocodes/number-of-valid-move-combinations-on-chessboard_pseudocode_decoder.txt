CLASS Solution  
    FUNCTION countCombinations(pieces positions)  
        SET directions TO a mapping from piece type TO lists of direction pairs AS follows  
            SET rook TO the list of pairs each consisting of one or zero representing one step vertically or horizontally  
            SET queen TO the list including rook directions AND the diagonal directions each consisting of one or negative one steps vertically and horizontally  
            SET bishop TO the list of diagonal directions each consisting of one or negative one steps vertically and horizontally  
        
        SET positions TO a list obtained by SETTING each row and column in positions TO row MINUS one AND column MINUS one RESPECTIVELY  
        
        FUNCTION get_destinations(start piece_type)  
            SET r TO element at position one of start  
            SET c TO element at position two of start  
            SET dests TO a list containing start  
            FOR each dr dc IN directions at key piece_type  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                WHILE nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN eight AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN eight  
                    APPEND the pair consisting of nr AND nc TO dests  
                    INCREMENT nr BY dr  
                    INCREMENT nc BY dc  
                END WHILE  
            END FOR  
            RETURN dests  
        END FUNCTION  
        
        SET all_destinations TO a list obtained by FOR each pos piece IN the pairs formed by elements at the same positions of positions AND pieces OBTAIN get_destinations of pos AND piece  
        
        FUNCTION is_valid_combination(combination)  
            SET pos TO a copy of positions  
            SET n TO the length of pos  
            WHILE true  
                IF the length of the set obtained from pos IS LESS THAN n  
                    RETURN false  
                END IF  
                SET all_reached TO true  
                FOR i FROM zero TO n MINUS one  
                    IF element at position i of pos EQUALS element at position i of combination  
                        CONTINUE to the next iteration  
                    END IF  
                    SET all_reached TO false  
                    SET r TO element at position i of pos at position one  
                    SET c TO element at position i of pos at position two  
                    SET dr TO one IF element at position i of combination at position one IS GREATER THAN r ELSE negative one IF element at position i of combination at position one IS LESS THAN r ELSE zero  
                    SET dc TO one IF element at position i of combination at position two IS GREATER THAN c ELSE negative one IF element at position i of combination at position two IS LESS THAN c ELSE zero  
                    SET element at position i of pos TO the pair consisting of r PLUS dr AND c PLUS dc  
                END FOR  
                IF all_reached  
                    RETURN true  
                END IF  
            END WHILE  
        END FUNCTION  
        
        SET valid_count TO zero  
        FOR each combination IN the cartesian product of all_destinations  
            IF is_valid_combination(combination)  
                INCREMENT valid_count BY one  
            END IF  
        END FOR  
        
        RETURN valid_count  
    END FUNCTION  
END CLASS
