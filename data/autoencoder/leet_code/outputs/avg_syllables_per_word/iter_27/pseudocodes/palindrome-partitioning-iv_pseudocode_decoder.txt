CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            IF sub EQUALS the REVERSE of sub  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two dimensional list of boolean values initialized to false with dimensions n BY n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the element at position i of s EQUALS the element at position j of s AND  
                   (the difference between j and i IS LESS THAN OR EQUAL TO one OR the value at position i PLUS one AND j MINUS one in is_pal IS true)  
                    SET the value at position i AND j in is_pal TO true  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF the value at position zero AND i MINUS one in is_pal IS true  
                FOR j FROM i TO n MINUS two  
                    IF the value at position i AND j in is_pal IS true AND the value at position j PLUS one AND n MINUS one in is_pal IS true  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
