CLASS Solution  
    FUNCTION minDistance(height height PARAMETER width width PARAMETER tree tree PARAMETER squirrel squirrel PARAMETER nuts nuts)  
        FUNCTION distance(p1 p1 PARAMETER p2 p2)  
            SET horizontal_difference TO the absolute value of the subtraction of element at position zero of p1 MINUS element at position zero of p2  
            SET vertical_difference TO the absolute value of the subtraction of element at position one of p1 MINUS element at position one of p2  
            SET total_difference TO horizontal_difference PLUS vertical_difference  
            RETURN total_difference  
        END FUNCTION  

        SET total_distance TO zero  
        SET max_save TO negative infinity  

        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY the result of calling distance with nut and tree  
            SET first_nut_distance TO the sum of the result of calling distance with squirrel and nut PLUS the result of calling distance with nut and tree  
            SET later_nut_distance TO two MULTIPLIED BY the result of calling distance with nut and tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  

        SET result TO total_distance MINUS max_save  
        RETURN result  
    END FUNCTION  
END CLASS
