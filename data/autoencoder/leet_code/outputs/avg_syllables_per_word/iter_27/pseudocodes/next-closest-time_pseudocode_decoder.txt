CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time with the character colon replaced by an empty string  
        SET hour TO the integer value of the substring from position zero TO position one of time  
        SET minute TO the integer value of the substring from position three TO position four of time  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET combined_hour TO the integer value of the concatenation of h1 PLUS h2  
                            SET combined_minute TO the integer value of the concatenation of m1 PLUS m2  
                            IF combined_hour LESS THAN twenty four AND combined_minute LESS THAN sixty THEN  
                                APPEND the tuple of combined_hour AND combined_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        
        SET current_time_index TO the position of the tuple of hour AND minute IN the list all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times THEN  
            SET next_hour TO the element at position current_time_index PLUS one of all_times WITHIN the first element of the tuple  
            SET next_minute TO the element at position current_time_index PLUS one of all_times WITHIN the second element of the tuple  
        ELSE  
            SET next_hour TO the first element of the first tuple in all_times  
            SET next_minute TO the second element of the first tuple in all_times  
        END IF  
        
        RETURN the string representing next_hour with two digit width PLUS a colon PLUS the string representing next_minute with two digit width  
    END FUNCTION  
END CLASS
