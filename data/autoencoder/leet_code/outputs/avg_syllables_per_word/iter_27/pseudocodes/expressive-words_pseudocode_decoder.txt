CLASS Solution  
    FUNCTION expressiveWords(s, words)  
        
        FUNCTION compress(word)  
            IF word EQUALS empty  
                RETURN empty list  
            END IF  
            SET count TO one  
            SET compressed TO empty list  
            FOR index FROM two TO the LENGTH OF word  
                IF element at position index of word EQUALS element at position index MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the pair consisting of element at position index MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND the pair consisting of element at last position of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy(s_compressed, word_compressed)  
            IF the LENGTH OF s_compressed NOT EQUALS the LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair s_pair AND each pair w_pair IN s_compressed AND word_compressed  
                IF first element of s_pair NOT EQUALS first element of w_pair  
                    RETURN false  
                END IF  
                IF second element of s_pair NOT EQUALS second element of w_pair AND (second element of s_pair LESS THAN three OR second element of s_pair LESS THAN second element of w_pair)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET s_compressed TO the result of compress WITH argument s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO the result of compress WITH argument word  
            IF the result of is_stretchy WITH arguments s_compressed AND word_compressed  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
