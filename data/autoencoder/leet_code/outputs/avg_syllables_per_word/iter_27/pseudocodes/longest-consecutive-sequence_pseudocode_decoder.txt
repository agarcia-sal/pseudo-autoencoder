CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        
        SET num_set TO the UNIQUE ELEMENTS OF nums  
        SET longest_streak TO zero  
        
        FOR each num IN num_set  
            IF the ELEMENT AT THE POSITION REPRESENTED BY the VALUE OF num MINUS one IS NOT IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                
                WHILE the ELEMENT AT THE POSITION REPRESENTED BY the VALUE OF current_num PLUS one IS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                
                IF longest_streak LESS THAN current_streak  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
