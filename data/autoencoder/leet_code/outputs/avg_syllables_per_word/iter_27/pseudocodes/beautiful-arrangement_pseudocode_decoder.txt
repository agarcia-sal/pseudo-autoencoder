CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one THEN  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n DO  
                IF element at position i of used EQUALS False AND (the remainder of element at position i DIVIDED BY index EQUALS zero OR the remainder of index DIVIDED BY element at position i EQUALS zero) THEN  
                    SET element at position i of used TO True  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
      
        SET used TO a list of Boolean False values with length n PLUS one  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
