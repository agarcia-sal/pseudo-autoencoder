CLASS Solution  
    FUNCTION numberOfCombinations WITH parameter num  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an empty dictionary  
        
        FUNCTION dp WITH parameters index AND prev_len  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            IF the tuple of index AND prev_len IS a KEY IN memo THEN  
                RETURN the VALUE OF memo AT the tuple of index AND prev_len  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index PLUS one  
                IF the element at position index of num EQUALS character zero THEN  
                    BREAK the loop  
                END IF  
                
                SET current_num TO the substring of num FROM position index TO position index PLUS length MINUS one  
                
                IF prev_len EQUALS zero OR current_num IS GREATER THAN OR EQUAL TO the substring of num FROM position index MINUS prev_len TO position index MINUS one THEN  
                    SET count TO count PLUS THE RESULT OF dp WITH arguments index PLUS length AND length  
                    SET count TO count MODULO MOD  
                END IF  
            END FOR  
            
            SET memo AT the tuple of index AND prev_len TO count  
            RETURN count  
        END FUNCTION  
        
        RETURN the RESULT OF dp WITH arguments zero AND zero  
    END FUNCTION  
END CLASS
