CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode created with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue has elements  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode created with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode created with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF the value of p.val NOT EQUALS the value of q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with p.left AND q.left AND the result of is_same_tree called with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low, high)  
            IF node IS no value  
                RETURN true  
            END IF  
            IF the value of node.val LESS THAN OR EQUAL TO low OR the value of node.val GREATER THAN OR EQUAL TO high  
                RETURN false  
            END IF  
            RETURN the result of validate called with node.left AND low AND node.val AND the result of validate called with node.right AND node.val AND high  
        END FUNCTION  
        RETURN validate called with root  
    END FUNCTION  
END CLASS
