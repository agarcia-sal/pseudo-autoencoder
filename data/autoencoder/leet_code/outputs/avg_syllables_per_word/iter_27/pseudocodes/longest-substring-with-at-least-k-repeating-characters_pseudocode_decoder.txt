CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        
        SET count TO the frequency count of each character in s  
        
        FOR each character IN count  
            IF the frequency of character in count LESS THAN k  
                RETURN the maximum value among the results of calling longestSubstring with each substring in the collection obtained by splitting s by character and k  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
