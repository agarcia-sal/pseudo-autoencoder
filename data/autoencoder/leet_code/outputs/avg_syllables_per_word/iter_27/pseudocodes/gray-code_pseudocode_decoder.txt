CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        
        SET prev_gray TO CALL grayCode WITH n MINUS one  
        
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        
        SET current_gray TO concatenation of prev_gray WITH list constructed by FOR each number IN reversed prev_gray RETURNING mask OR number  
        
        RETURN current_gray  
    END FUNCTION  
END CLASS
