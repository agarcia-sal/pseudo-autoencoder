CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET time to target TO an empty list  
        SET zipped pairs TO the list of pairs each containing element at position index of position AND element at position index of speed  
        SORT zipped pairs in descending order based on element at position zero of each pair  
        FOR each pair IN zipped pairs  
            SET p TO element at position zero of pair  
            SET s TO element at position one of pair  
            SET time to target APPEND the difference between target AND p DIVIDED BY s  
        END FOR  
        
        SET fleets TO zero  
        SET current max time TO zero  
        
        FOR each time IN time to target  
            IF time GREATER THAN current max time THEN  
                INCREMENT fleets BY one  
                SET current max time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
