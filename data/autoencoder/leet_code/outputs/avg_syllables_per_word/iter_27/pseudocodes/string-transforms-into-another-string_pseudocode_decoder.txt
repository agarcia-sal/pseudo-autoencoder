CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN true  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each pair char1 char2 FROM corresponding elements of str1 AND str2  
            IF char1 IS A KEY IN conversion_map  
                IF value AT key char1 in conversion_map NOT EQUALS char2  
                    RETURN false  
                END IF  
            ELSE  
                SET value AT key char1 IN conversion_map TO char2  
            END IF  
        END FOR  
        IF the SIZE OF the SET OF characters IN str2 EQUALS twenty six  
            RETURN false  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS
