CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET modified_expression TO expression WITH all occurrences of minus replaced BY plus FOLLOWED BY minus  
        SET list_of_fractions TO the list of substrings obtained BY splitting modified_expression BY plus  
        SET list_of_valid_fractions TO the list of elements FROM list_of_fractions WHERE element IS NOT empty  
        SET result_fraction TO the sum OF converting each element IN list_of_valid_fractions TO a fraction object USING fractional parsing  
        SET numerator_part TO the numerator component OF result_fraction  
        SET denominator_part TO the denominator component OF result_fraction  
        SET result_string TO the concatenation OF numerator_part FOLLOWED BY slash FOLLOWED BY denominator_part AS a string  
        RETURN result_string  
    END FUNCTION  
END CLASS
