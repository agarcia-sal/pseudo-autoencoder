CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        CREATE a two dimensional list dp with dimensions m by n FILLED WITH zero  
        
        SET element at position zero of row zero of dp TO element at position zero of row zero of grid  
        
        FOR j FROM one TO n MINUS one  
            SET element at position zero of row j of dp TO element at position zero of row j MINUS one of dp PLUS element at position zero of row j of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            SET element at position i of row zero of dp TO element at position i MINUS one of row zero of dp PLUS element at position i of row zero of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i of row j of dp TO the minimum value BETWEEN element at position i MINUS one of row j of dp AND element at position i of row j MINUS one of dp PLUS element at position i of row j of grid  
            END FOR  
        END FOR  
        
        RETURN element at position m MINUS one of row n MINUS one of dp  
    END FUNCTION  
END CLASS
