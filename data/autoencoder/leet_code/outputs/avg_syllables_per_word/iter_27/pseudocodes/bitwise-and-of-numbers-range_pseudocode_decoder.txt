CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER PARAMETER right INTEGER PARAMETER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE value of left LESS THAN value of right  
            SET left TO integer division of left BY two  
            SET right TO integer division of right BY two  
            INCREMENT shift BY one  
        END WHILE  
        RETURN value of left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
