CLASS Solution  
    FUNCTION maximumScore(nums parameter, multipliers parameter)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeroes with the LENGTH OF m PLUS one  
        FOR i FROM m MINUS one TO zero DECREMENTING BY one  
            SET new_dp TO a list of zeroes with the LENGTH OF m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS the DIFFERENCE of i MINUS left  
                SET mul TO element at position i of multipliers  
                SET new_dp at position left TO the GREATER OF the SUM of mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp AND the SUM of mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
