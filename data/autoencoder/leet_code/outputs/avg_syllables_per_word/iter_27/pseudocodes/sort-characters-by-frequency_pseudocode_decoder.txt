CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO a mapping of each character in s TO the count of that character in s  
        SET sorted_characters TO the list of characters in frequency sorted in order of frequency of each character from highest TO lowest  
        SET result TO an empty string  
        FOR each character IN sorted_characters  
            APPEND the character repeated frequency of character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
