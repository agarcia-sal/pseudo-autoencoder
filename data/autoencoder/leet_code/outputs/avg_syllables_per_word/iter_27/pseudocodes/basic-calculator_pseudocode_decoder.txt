CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus sign  
            SET num TO zero  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                IF char IS digit character  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                IF char EQUALS open parenthesis  
                    SET num TO the result of calling helper WITH s  
                END IF  
                IF char IS NOT digit character AND char IS NOT whitespace character OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus sign  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus sign  
                        APPEND the negative of num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET the last element of stack TO the integer result of the last element of stack DIVIDED BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis  
                    BREAK the while loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        RETURN the result of calling helper WITH the list of characters in s  
    END FUNCTION  
END CLASS
