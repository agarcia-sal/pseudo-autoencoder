CLASS Solution  
    FUNCTION checkInclusion(s1 parameter TO string, s2 parameter TO string) RETURNS boolean  
        IMPORT Counter FROM collections  
          
        SET length_of_s1 TO the LENGTH OF s1  
        SET length_of_s2 TO the LENGTH OF s2  
          
        IF length_of_s1 GREATER THAN length_of_s2  
            RETURN false  
        END IF  
          
        SET count_s1 TO the mapping of character frequencies IN s1  
        SET count_s2 TO the mapping of character frequencies IN the substring of s2 from position zero TO position length_of_s1 MINUS one  
          
        IF the mapping count_s1 EQUALS the mapping count_s2  
            RETURN true  
        END IF  
          
        FOR index i FROM length_of_s1 TO length_of_s2 MINUS one  
            INCREMENT the frequency of the element at position i of s2 IN count_s2 BY one  
              
            DECREMENT the frequency of the element at position i MINUS length_of_s1 of s2 IN count_s2 BY one  
              
            IF the frequency of the element at position i MINUS length_of_s1 of s2 IN count_s2 EQUALS zero  
                REMOVE the element at position i MINUS length_of_s1 of s2 FROM count_s2  
            END IF  
              
            IF the mapping count_s1 EQUALS the mapping count_s2  
                RETURN true  
            END IF  
        END FOR  
          
        RETURN false  
    END FUNCTION  
END CLASS
