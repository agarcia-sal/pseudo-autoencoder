CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two dimensional list OF false WITH dimensions of the LENGTH OF s PLUS one BY the LENGTH OF p PLUS one  
        SET element at position zero of dp element at position zero TO true  
        FOR j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of p EQUALS the character asterisk  
                SET element at position zero of dp element at position j TO element at position zero of dp element at position j MINUS two  
            END IF  
        END FOR  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of p EQUALS element at position i MINUS one of s OR element at position j MINUS one of p EQUALS the character dot  
                    SET element at position i of dp element at position j TO element at position i MINUS one of dp element at position j MINUS one  
                ELSE IF element at position j MINUS one of p EQUALS the character asterisk  
                    SET element at position i of dp element at position j TO element at position i of dp element at position j MINUS two  
                    IF element at position j MINUS two of p EQUALS element at position i MINUS one of s OR element at position j MINUS two of p EQUALS the character dot  
                        SET element at position i of dp element at position j TO element at position i of dp element at position j OR element at position i MINUS one of dp element at position j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s of dp element at position LENGTH OF p  
    END FUNCTION  
END CLASS
