CLASS Solution  
    FUNCTION appealSum WITH PARAMETER s  
        SET last TO a list of twenty six times negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  

        FOR each i and char IN each index and element of s  
            SET index TO the character code of char MINUS the character code of lowercase a  
            INCREMENT current_appeal BY i MINUS the element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET the element at position index of last TO i  
        END FOR  

        RETURN total_appeal  
    END FUNCTION  
END CLASS
