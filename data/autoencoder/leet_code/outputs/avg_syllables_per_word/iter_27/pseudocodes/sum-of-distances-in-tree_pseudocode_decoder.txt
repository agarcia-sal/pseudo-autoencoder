CLASS Solution  
    FUNCTION sumOfDistancesInTree(n, edges)  
        SET graph TO a mapping from each node to an empty list  
        FOR each element consisting of a and b IN edges  
            APPEND b TO the list at position a in graph  
            APPEND a TO the list at position b in graph  
        END FOR  

        SET count TO a list of length n with each element set to one  
        SET dist TO a list of length n with each element set to zero  

        FUNCTION dfs1(u, p)  
            FOR each element v IN the list at position u in graph  
                IF v NOT EQUALS p THEN  
                    CALL dfs1 WITH arguments v and u  
                    INCREMENT count at position u BY count at position v  
                    INCREMENT dist at position u BY dist at position v PLUS count at position v  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs1 WITH arguments zero and negative one  

        FUNCTION dfs2(u, p)  
            FOR each element v IN the list at position u in graph  
                IF v NOT EQUALS p THEN  
                    SET dist at position v TO dist at position u MINUS count at position v PLUS n MINUS count at position v  
                    CALL dfs2 WITH arguments v and u  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs2 WITH arguments zero and negative one  

        RETURN dist  
    END FUNCTION  
END CLASS
