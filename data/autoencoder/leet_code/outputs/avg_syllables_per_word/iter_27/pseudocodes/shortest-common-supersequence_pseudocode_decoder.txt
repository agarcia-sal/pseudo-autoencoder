CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1 parameter, str2 parameter)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a two dimensional list where each element is zero with the number of rows equal TO m PLUS one AND the number of columns equal TO n PLUS one  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                    SET the element at row i column j of dp TO the element at row i MINUS one column j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at row i column j of dp TO the maximum BETWEEN the element at row i MINUS one column j of dp AND the element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at row i MINUS one column j of dp GREATER THAN the element at row i column j MINUS one of dp THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements in scs  
        
        RETURN the concatenation of all elements in scs as a single string  
    END FUNCTION  
END CLASS
