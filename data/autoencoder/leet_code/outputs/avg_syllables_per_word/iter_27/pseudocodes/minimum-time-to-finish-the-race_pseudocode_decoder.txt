CLASS Solution  
    FUNCTION minimumFinishTime(tires NUMERIC LIST OF LIST OF NUMERIC changeTime NUMERIC numLaps NUMERIC) RETURNS NUMERIC  
        SET min_times TO a list containing fifteen elements each set to infinity  
        FOR each tire_features IN tires  
            SET f TO the first element of tire_features  
            SET r TO the second element of tire_features  
            SET total_time TO zero  
            FOR i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK the inner loop  
                END IF  
                SET total_time TO total_time PLUS lap_time  
                IF total_time LESS THAN element at position i of min_times  
                    SET element at position i of min_times TO total_time  
                END IF  
            END FOR  
        END FOR  

        SET dp TO a list containing numLaps PLUS one elements each set to infinity  
        SET element at position zero of dp TO zero  

        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller of i MINUS one and fourteen  
                SET candidate_time TO element at position i MINUS j MINUS one of dp PLUS element at position j of min_times PLUS changeTime  
                IF candidate_time LESS THAN element at position i of dp  
                    SET element at position i of dp TO candidate_time  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position numLaps of dp MINUS changeTime  
    END FUNCTION  
END CLASS
