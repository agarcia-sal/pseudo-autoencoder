CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF the LENGTH OF digits EQUALS zero  
            RETURN an empty list  
        END IF  
        SET phone_map TO a mapping where digit two maps to letters a b c and digit three maps to letters d e f and digit four maps to letters g h i and digit five maps to letters j k l and digit six maps to letters m n o and digit seven maps to letters p q r s and digit eight maps to letters t u v and digit nine maps to letters w x y z  
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the letters mapped by the element at position zero of next_digits in phone_map  
                    CALL backtrack with the combination concatenated with letter and the substring from position one TO the END of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO an empty list  
        CALL backtrack with the empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
