CLASS Solution  
    FUNCTION getHint(secret STRING_PARAMETER guess STRING_PARAMETER) STRING_RETURN  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO new mapping with default zero for each key  
        SET guess_count TO new mapping with default zero for each key  
        
        FOR each pair of elements s AND g IN parallel iteration over secret AND guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT value at key s IN secret_count BY one  
                INCREMENT value at key g IN guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN keys of secret_count  
            SET minimum_value TO the smaller value BETWEEN value at digit IN secret_count AND value at digit IN guess_count  
            INCREMENT cows BY minimum_value  
        END FOR  
        
        RETURN string formed by concatenating bulls AS STRING AND the uppercase letter A AND cows AS STRING AND the uppercase letter B  
    END FUNCTION  
END CLASS
