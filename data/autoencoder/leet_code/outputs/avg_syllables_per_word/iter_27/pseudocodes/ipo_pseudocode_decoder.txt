CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER, w INTEGER, profits LIST OF INTEGER, capital LIST OF INTEGER) RETURNS INTEGER  
        SET projects TO a list of pairs where each pair consists of element at position index of capital AND element at position index of profits FOR each index from zero TO the length of capital MINUS one  
        SORT projects BY the first element of each pair in ascending order  
        
        SET max_heap TO an empty list  
        SET i TO zero  
        SET n TO the length of projects  
        
        FOR iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element of element at position i of projects LESS THAN OR EQUAL TO w  
                ADD the negative of the second element of element at position i of projects TO max_heap using heap insertion  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap is empty  
                BREAK the loop  
            END IF  
            
            REMOVE and retrieve the smallest element from max_heap using heap removal and ADD its negation TO w  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
