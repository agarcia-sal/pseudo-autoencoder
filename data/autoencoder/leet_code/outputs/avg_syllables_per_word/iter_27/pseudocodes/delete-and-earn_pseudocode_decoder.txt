CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a mapping from each element in nums to its number of occurrences  
        SET unique_nums TO the list of all keys in count sorted in ascending order  
        SET dp TO a list of zeros of length the LENGTH OF unique_nums PLUS one  
        SET element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the value at element at position zero of unique_nums in count  
        
        FOR i FROM two TO the LENGTH OF unique_nums PLUS one  
            SET num TO the element at position i MINUS one of unique_nums  
            IF num EQUALS the element at position i MINUS two of unique_nums PLUS one  
                SET element at position i of dp TO the maximum of element at position i MINUS one of dp AND the sum of element at position i MINUS two of dp PLUS num MULTIPLIED BY the value at num in count  
            ELSE  
                SET element at position i of dp TO element at position i MINUS one of dp PLUS num MULTIPLIED BY the value at num in count  
            END IF  
        END FOR  
        
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
