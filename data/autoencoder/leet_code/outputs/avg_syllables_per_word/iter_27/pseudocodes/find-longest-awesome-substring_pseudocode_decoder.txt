CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new collection with key zero and value negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each pair of index i AND character char IN the elements and positions of s  
            SET digit TO the integer value represented by char  
            TOGGLE the bit corresponding to digit in prefix_xor by applying exclusive OR with one shifted left by digit positions  
            
            IF prefix_xor exists as a key in prefix_map  
                SET max_length TO the larger value of max_length AND the difference between i AND the element at key prefix_xor in prefix_map  
            END IF  
            
            FOR each index j FROM zero TO nine  
                SET flipped TO prefix_xor exclusive OR with one shifted left by j positions  
                
                IF flipped exists as a key in prefix_map  
                    SET max_length TO the larger value of max_length AND the difference between i AND the element at key flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor does not exist as a key in prefix_map  
                SET the element at key prefix_xor in prefix_map TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
