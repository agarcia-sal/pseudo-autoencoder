CLASS Solution  
    FUNCTION new21Game(n k maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference LESS THAN maxPts  
                    SET minimum_value TO difference  
                ELSE  
                    SET minimum_value TO maxPts  
                END IF  
                RETURN minimum_value DIVIDED BY maxPts  
            END IF  
            SET value_to_return TO dfs(i PLUS one) PLUS (dfs(i PLUS one) MINUS dfs(i PLUS maxPts PLUS one)) DIVIDED BY maxPts  
            RETURN value_to_return  
        END FUNCTION  
        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
