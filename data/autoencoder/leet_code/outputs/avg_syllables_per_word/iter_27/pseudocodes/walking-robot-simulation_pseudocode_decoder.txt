CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of pairs zero and one zero and zero and one zero and minus one and minus one and zero  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO set of tuples converted from elements of obstacles  
        SET max_distance TO zero  
        
        FOR each command IN commands  
            IF command EQUALS minus two  
                SET dir_index TO dir_index MINUS one MODULO four  
            ELSE IF command EQUALS minus one  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO element at position dir_index of directions at position zero  
                SET dy TO element at position dir_index of directions at position one  
                
                FOR count FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    
                    IF tuple of next_x and next_y IS IN obstacles_set  
                        BREAK  
                    END IF  
                    
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                
                SET max_distance TO the GREATER OF max_distance AND x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
