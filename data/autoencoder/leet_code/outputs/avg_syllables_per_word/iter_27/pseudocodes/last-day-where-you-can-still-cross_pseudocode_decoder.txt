CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list of integers from zero TO n MINUS one  
        SET self.rank TO a list containing zero repeated n times  
    END FUNCTION  

    FUNCTION find(self, x)  
        IF element at position x of self.parent NOT EQUALS x THEN  
            SET element at position x of self.parent TO the result of calling find with parameter the element at position x of self.parent  
        END IF  
        RETURN element at position x of self.parent  
    END FUNCTION  

    FUNCTION union(self, x, y)  
        SET rootX TO the result of calling find with parameter x  
        SET rootY TO the result of calling find with parameter y  
        IF rootX NOT EQUALS rootY THEN  
            IF element at position rootX of self.rank GREATER THAN element at position rootY of self.rank THEN  
                SET element at position rootY of self.parent TO rootX  
            ELSE IF element at position rootX of self.rank LESS THAN element at position rootY of self.rank THEN  
                SET element at position rootX of self.parent TO rootY  
            ELSE  
                SET element at position rootY of self.parent TO rootX  
                INCREMENT element at position rootX of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION latestDayToCross(self, row, col, cells)  
        SET grid TO a list containing row elements each of which is a list containing col times the integer one  
        SET uf TO a new UnionFind object initialized with the value row MULTIPLIED BY col PLUS two  
        SET TOP TO row MULTIPLIED BY col  
        SET BOTTOM TO row MULTIPLIED BY col PLUS one  
        SET directions TO a list containing the tuples negative one and zero, one and zero, zero and negative one, zero and one  

        FOR day FROM the LENGTH OF cells MINUS one DOWN TO zero INCLUSIVE  
            SET r TO element at position day and position zero of cells  
            SET c TO element at position day and position one of cells  
            DECREMENT r BY one  
            DECREMENT c BY one  
            SET element at position r and position c of grid TO zero  

            FOR each pair dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN row AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN col AND element at position nr and position nc of grid EQUALS zero THEN  
                    CALL union of uf with parameters r MULTIPLIED BY col PLUS c and nr MULTIPLIED BY col PLUS nc  
                END IF  
            END FOR  

            IF r EQUALS zero THEN  
                CALL union of uf with parameters r MULTIPLIED BY col PLUS c and TOP  
            END IF  

            IF r EQUALS row MINUS one THEN  
                CALL union of uf with parameters r MULTIPLIED BY col PLUS c and BOTTOM  
            END IF  

            IF the result of calling find of uf with parameter TOP EQUALS the result of calling find of uf with parameter BOTTOM THEN  
                RETURN day  
            END IF  
        END FOR  

        RETURN zero  
    END FUNCTION  
END CLASS
