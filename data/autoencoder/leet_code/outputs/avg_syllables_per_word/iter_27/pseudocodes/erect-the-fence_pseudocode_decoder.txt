CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION cross(i, j, k)  
            SET a TO element at position i of trees  
            SET b TO element at position j of trees  
            SET c TO element at position k of trees  
            RETURN the quantity of the difference between the first element of b and the first element of a MULTIPLIED BY the difference between the second element of c and the second element of b MINUS the quantity of the difference between the second element of b and the second element of a MULTIPLIED BY the difference between the first element of c and the first element of b  
        END FUNCTION  

        SET n TO the LENGTH OF trees  
        IF n LESS THAN four THEN  
            RETURN trees  
        END IF  
        SORT trees in ascending order by default criteria  
        SET vis TO a list of boolean False repeated n times  
        SET stk TO a list containing zero as the first element  
        FOR i FROM one TO n MINUS one  
            WHILE the LENGTH OF stk GREATER THAN one AND cross of element at position LENGTH OF stk MINUS two of stk AND element at position LENGTH OF stk MINUS one of stk AND i LESS THAN zero holds  
                SET the element at position LENGTH OF stk MINUS one of stk TO be removed from stk  
                SET the element at position just removed of stk TO False in vis  
            END WHILE  
            SET vis at position i TO True  
            APPEND i TO stk  
        END FOR  
        SET m TO the LENGTH OF stk  
        FOR i FROM n MINUS two DOWNTO zero  
            IF vis at position i IS True THEN  
                CONTINUE to next iteration of the loop  
            END IF  
            WHILE the LENGTH OF stk GREATER THAN m AND cross of element at position LENGTH OF stk MINUS two of stk AND element at position LENGTH OF stk MINUS one of stk AND i LESS THAN zero holds  
                REMOVE the element at position LENGTH OF stk MINUS one of stk FROM stk  
            END WHILE  
            APPEND i TO stk  
        END FOR  
        REMOVE the last element FROM stk  
        RETURN a list containing the elements of trees at each index specified by each element in stk  
    END FUNCTION  
END CLASS
