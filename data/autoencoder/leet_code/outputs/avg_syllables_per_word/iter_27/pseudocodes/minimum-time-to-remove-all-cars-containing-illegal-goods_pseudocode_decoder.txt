CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zero repeated n times  
        SET right_costs TO a list of zero repeated n times  
        
        IF element at position zero of s EQUALS character one  
            SET element at position zero of left_costs TO one  
        ELSE  
            SET element at position zero of left_costs TO zero  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            IF element at position i of s EQUALS character zero  
                SET element at position i of left_costs TO element at position i MINUS one of left_costs  
            ELSE  
                IF i PLUS one LESS THAN OR EQUAL TO element at position i MINUS one of left_costs PLUS two  
                    SET element at position i of left_costs TO i PLUS one  
                ELSE  
                    SET element at position i of left_costs TO element at position i MINUS one of left_costs PLUS two  
                END IF  
            END IF  
        END FOR  
        
        IF element at position n MINUS one of s EQUALS character one  
            SET element at position n MINUS one of right_costs TO one  
        ELSE  
            SET element at position n MINUS one of right_costs TO zero  
        END IF  
        
        FOR i FROM n MINUS two DOWN TO zero  
            IF element at position i of s EQUALS character zero  
                SET element at position i of right_costs TO element at position i PLUS one of right_costs  
            ELSE  
                IF n MINUS i LESS THAN OR EQUAL TO element at position i PLUS one of right_costs PLUS two  
                    SET element at position i of right_costs TO n MINUS i  
                ELSE  
                    SET element at position i of right_costs TO element at position i PLUS one of right_costs PLUS two  
                END IF  
            END IF  
        END FOR  
        
        IF element at position n MINUS one of left_costs LESS THAN OR EQUAL TO element at position zero of right_costs  
            SET min_time TO element at position n MINUS one of left_costs  
        ELSE  
            SET min_time TO element at position zero of right_costs  
        END IF  
        
        FOR i FROM zero TO n MINUS two  
            IF min_time LESS THAN OR EQUAL TO element at position i of left_costs PLUS element at position i PLUS one of right_costs  
                CONTINUE  
            ELSE  
                SET min_time TO element at position i of left_costs PLUS element at position i PLUS one of right_costs  
            END IF  
        END FOR  
        
        RETURN min_time  
    END FUNCTION  
END CLASS
