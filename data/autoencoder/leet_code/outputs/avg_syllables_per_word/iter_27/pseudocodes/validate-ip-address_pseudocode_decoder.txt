CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF ONLY DIGITS THEN  
                SET num TO CONVERT block TO A NUMBER USING BASE TEN  
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND CONVERT num TO STRING EQUALS block THEN  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block IS GREATER THAN OR EQUAL TO one AND the LENGTH OF block IS LESS THAN OR EQUAL TO four THEN  
                TRY  
                    CONVERT block TO A NUMBER USING BASE sixteen  
                    RETURN True  
                CATCH conversion error  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO RESULT OF SPLITTING queryIP BY DOT  
        SET parts_ipv6 TO RESULT OF SPLITTING queryIP BY COLON  

        IF the LENGTH OF parts_ipv4 EQUALS four AND for EACH part IN parts_ipv4 is_valid_ipv4_block(part) IS True THEN  
            RETURN the STRING IPv4  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND for EACH part IN parts_ipv6 is_valid_ipv6_block(part) IS True THEN  
            RETURN the STRING IPv6  
        ELSE  
            RETURN the STRING Neither  
        END IF  
    END FUNCTION  
END CLASS
