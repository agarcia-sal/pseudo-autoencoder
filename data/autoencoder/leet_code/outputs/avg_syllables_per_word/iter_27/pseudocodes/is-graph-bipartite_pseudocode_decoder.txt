CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a list of zeroes with LENGTH n  
        
        FUNCTION dfs(node c)  
            SET element at position node of color TO c  
            FOR each neighbor IN element at position node of graph  
                IF element at position neighbor of color EQUALS c  
                    RETURN false  
                END IF  
                IF element at position neighbor of color EQUALS zero  
                    IF dfs(neighbor negative c) EQUALS false  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF element at position i of color EQUALS zero  
                IF dfs(i one) EQUALS false  
                    RETURN false  
                END IF  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
