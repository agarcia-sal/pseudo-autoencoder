CLASS Solution  
    FUNCTION topKFrequent(nums k) RETURNS List of integers  
        SET frequency_map TO an empty map with default value zero  
        FOR each num IN nums  
            INCREMENT the value at key num in frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a list of empty lists with the size equal to max_frequency PLUS one  
        FOR each key num and value freq IN frequency_map  
            APPEND num TO the list at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length of the list at position freq of buckets GREATER THAN zero THEN  
                APPEND all elements of the list at position freq of buckets TO result  
                IF the length of result GREATER THAN OR EQUAL TO k THEN  
                    RETURN the sublist of result from position one TO position k  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
