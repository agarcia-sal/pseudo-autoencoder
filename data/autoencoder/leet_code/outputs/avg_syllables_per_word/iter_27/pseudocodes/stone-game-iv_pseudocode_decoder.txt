CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of False values with length equal to n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF dp element at position i MINUS j MULTIPLIED BY j EQUALS False  
                    SET dp element at position i TO True  
                    BREAK out of the while loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN dp element at position n  
    END FUNCTION  
END CLASS
