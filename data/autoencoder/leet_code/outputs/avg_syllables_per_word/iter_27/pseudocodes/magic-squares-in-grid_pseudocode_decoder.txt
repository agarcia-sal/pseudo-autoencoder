CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i of grid at position j in grid NOT EQUALS five  
                RETURN false  
            END IF  
            SET numbers TO an empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x of grid at position y of grid LESS THAN one OR element at position x of grid at position y of grid GREATER THAN nine OR element at position x of grid at position y of grid IN numbers  
                        RETURN false  
                    END IF  
                    ADD element at position x of grid at position y of grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF sum of elements at positions start of grid at positions j MINUS one, j, and j PLUS one NOT EQUALS fifteen OR sum of elements at positions i MINUS one, i, and i PLUS one of grid at position start NOT EQUALS fifteen  
                    RETURN false  
                END IF  
            END FOR  
            IF sum of elements at positions i MINUS one, i, and i PLUS one of grid at positions j MINUS one, j, and j PLUS one NOT EQUALS fifteen OR sum of elements at positions i MINUS one, i, and i PLUS one of grid at positions j PLUS one, j, and j MINUS one NOT EQUALS fifteen  
                RETURN false  
            END IF  
            RETURN true  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the number of rows in grid  
        SET cols TO the number of columns in grid  
        FOR i FROM one TO rows MINUS two  
            FOR j FROM one TO cols MINUS two  
                IF is_magic(i, j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
