CLASS Solution  
    FUNCTION minDistance(houses parameter k)  
        SORT houses IN NON-DECREASING ORDER  
        SET n TO the LENGTH OF houses  
        
        SET cost TO a two dimensional list of zeros with dimensions n BY n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and position j of cost TO element at position i PLUS one and position j MINUS one of cost PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list filled with infinite values with dimensions n PLUS one BY k PLUS one  
        SET element at position zero and position zero of dp TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    IF element at position i and position j of dp IS GREATER THAN element at position p and position j MINUS one of dp PLUS element at position p and position i MINUS one of cost THEN  
                        SET element at position i and position j of dp TO element at position p and position j MINUS one of dp PLUS element at position p and position i MINUS one of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n and position k of dp  
    END FUNCTION  
END CLASS
