CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET attribute val TO val  
        SET attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode of element at position one of values  
    SET p TO head  
    FOR each val IN values starting from position two TO the end  
        SET node TO a new ListNode of val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN attribute val of p1 EQUALS attribute val of p2 AND call is_same_list with arguments attribute next of p1 AND attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition with parameters head AND x  
        SET less_head TO a new ListNode of zero  
        SET greater_head TO a new ListNode of zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT None  
            IF attribute val of current LESS THAN x  
                SET next of less TO current  
                SET less TO next of less  
            ELSE  
                SET next of greater TO current  
                SET greater TO next of greater  
            END IF  
            SET current TO next of current  
        END WHILE  
        SET next of less TO next of greater_head  
        SET next of greater TO None  
        RETURN next of less_head  
    END FUNCTION  
END CLASS
