CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS the smaller value BETWEEN the difference of n MINUS prefix PLUS one AND the difference of next_prefix MINUS prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION

        SET current TO one  
        SET k TO k MINUS one  
        WHILE k GREATER THAN zero  
            SET steps TO count_steps(current, n)  
            IF steps LESS THAN OR EQUAL TO k  
                SET current TO current PLUS one  
                SET k TO k MINUS steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                SET k TO k MINUS one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
