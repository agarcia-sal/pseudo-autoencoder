CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value property of this instance TO val  
        SET the next property of this instance TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the collection values EQUALS empty  
        RETURN no value  
    END IF  
    SET head TO new instance of ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the collection starting from position two TO the end of values  
        SET node TO new instance of ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates with parameter head  
        SET dummy TO new instance of ListNode with zero AND head  
        SET prev TO dummy  
        WHILE head HOLDS  
            IF the next property of head HOLDS AND the value property of head EQUALS the value property of the next property of head  
                WHILE the next property of head HOLDS AND the value property of head EQUALS the value property of the next property of head  
                    SET head TO the next property of head  
                END WHILE  
                SET the next property of prev TO the next property of head  
            ELSE  
                SET prev TO the next property of prev  
            END IF  
            SET head TO the next property of head  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
