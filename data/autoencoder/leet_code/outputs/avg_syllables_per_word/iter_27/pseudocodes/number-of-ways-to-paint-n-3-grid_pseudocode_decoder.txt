CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        SET a TO six  
        SET b TO six  
        FOR each integer from two TO n INCLUSIVE  
            SET new_a TO three MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET new_b TO two MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET a TO new_a MODULO MOD  
            SET b TO new_b MODULO MODULO MOD  
        END FOR  
        RETURN a PLUS b MODULO MOD  
    END FUNCTION  
END CLASS
