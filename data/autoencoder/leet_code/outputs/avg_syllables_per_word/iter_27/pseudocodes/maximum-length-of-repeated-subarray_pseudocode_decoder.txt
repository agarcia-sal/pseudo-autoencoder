CLASS Solution  
    FUNCTION findLength(nums1 nums2)  
        SET dp TO empty list  
        FOR i FROM zero TO the LENGTH OF nums1  
            SET row TO empty list  
            FOR j FROM zero TO the LENGTH OF nums2  
                APPEND zero TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  

        SET max_length TO zero  

        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF element at position i MINUS one of nums1 EQUALS element at position j MINUS one of nums2  
                    SET element at position i of dp at position j TO element at position i MINUS one of dp at position j MINUS one PLUS one  
                    IF element at position i of dp at position j GREATER THAN max_length  
                        SET max_length TO element at position i of dp at position j  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
