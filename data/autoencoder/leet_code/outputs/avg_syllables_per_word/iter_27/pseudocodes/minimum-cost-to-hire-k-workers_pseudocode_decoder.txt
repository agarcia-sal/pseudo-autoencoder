CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the LIST obtained by SORTING the SEQUENCE of PAIRS where each PAIR contains the RESULT of the DIVISION of element at position index of wage BY element at position index of quality AND element at position index of quality FOR each index IN the RANGE from the START TO the LENGTH OF wage  
        SET max_heap TO an empty LIST  
        SET total_quality TO zero  
        SET min_cost TO a very large number representing positive infinity  
        FOR each PAIR ratio and q IN workers  
            PUSH the NEGATION of q ONTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k THEN  
                SET removed_value TO the result of REMOVING the SMALLEST ELEMENT FROM max_heap  
                INCREMENT total_quality BY removed_value  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost LESS THAN min_cost THEN  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
