CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode object with value equal to element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue and ASSIGN IT TO queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the element from the front of queue and ASSIGN IT TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode object with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode object with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self root)  
        CREATE a double ended queue containing root and ASSIGN IT TO queue  
        SET leftmost_value TO none  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                REMOVE the element from the front of queue and ASSIGN IT TO node  
                IF i EQUALS zero  
                    SET leftmost_value TO the value of node  
                END IF  
                IF node.left NOT EQUALS none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
