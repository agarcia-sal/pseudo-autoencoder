CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values DOES NOT have any elements THEN  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue HAS elements  
        REMOVE the element from the left end of queue AND ASSIGN this element TO node  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT nothing THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT nothing THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS nothing AND q IS nothing THEN  
        RETURN true  
    ELSE IF p IS nothing OR q IS nothing THEN  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS TO the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left child of p AND the left child of q AND the result of calling is_same_tree with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS nothing THEN  
                RETURN the smallest possible floating point value, the smallest possible floating point value  
            END IF  
            SET left_max, left_gain TO the result of calling helper with the left child of node  
            SET right_max, right_gain TO the result of calling helper with the right child of node  
            SET current_max TO the value of node PLUS the greater of left_gain AND zero PLUS the greater of right_gain AND zero  
            SET current_gain TO the value of node PLUS the greatest of left_gain, right_gain, AND zero  
            SET global_max TO the greatest of left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of the result of calling helper with root  
    END FUNCTION  
END CLASS
