CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        
        SET start TO start PLUS the INTEGER DIVISION of n MINUS one BY length  
        
        SET string_representation TO CONVERT start TO string  
        SET digit_position TO the MODULO of n MINUS one BY length  
        
        RETURN the DIGIT AT POSITION digit_position OF string_representation CONVERTED TO integer  
    END FUNCTION  
END CLASS
