CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET value_indices TO an empty mapping from values to lists  
        FOR position FROM zero TO n MINUS one  
            SET value TO element at position position of arr  
            APPEND position TO the list at key value in value_indices  
        END FOR  
        SET queue TO an empty double ended queue  
        APPEND the pair consisting of zero and zero TO queue  
        SET visited TO an empty set  
        ADD zero TO visited  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE and RETURN the front element of queue AND ASSIGN TO current_index AND steps respectively  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            SET next_indices TO a new list consisting of current_index PLUS one, current_index MINUS one, concatenated with the list at key the element at position current_index of arr in value_indices  
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair consisting of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            SET the list at key the element at position current_index of arr in value_indices TO an empty list  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
