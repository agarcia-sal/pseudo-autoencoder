CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the sum of all elements in nums  
        IF total_sum MODULO two NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of False values with LENGTH target PLUS one  
        SET the element at position zero of dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp element at position i TO dp element at position i OR dp element at position i MINUS num  
            END FOR  
        END FOR  
        RETURN dp element at position target  
    END FUNCTION  
END CLASS
