CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF element at position one MODULO two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            ADD negative num TO max_heap as a heap element  
        END FOR  
        SET min_deviation TO infinity  
        WHILE element at position one of max_heap MODULO two EQUALS zero  
            SET max_value TO negative of the element popped from max_heap heap  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two using integer division  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            ADD negative max_value TO max_heap as a heap element  
        END WHILE  
        IF min_deviation GREATER THAN negative element at position one of max_heap MINUS min_value  
            SET min_deviation TO negative element at position one of max_heap MINUS min_value  
        END IF  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
