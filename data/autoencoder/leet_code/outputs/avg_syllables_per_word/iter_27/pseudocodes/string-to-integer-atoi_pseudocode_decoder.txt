CLASS Solution  
    FUNCTION myAtoi(s)  
        SET maximum_32_bit_signed_integer TO two RAISED TO THE power of thirty ONE MINUS one  
        SET minimum_32_bit_signed_integer TO NEGATIVE two RAISED TO THE power of thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET length_of_string TO the LENGTH OF s  
        SET sign TO one  
        
        WHILE index LESS THAN length_of_string AND element at position index of s EQUALS a single whitespace character  
            INCREMENT index BY one  
        END WHILE  
        
        IF index LESS THAN length_of_string AND (element at position index of s EQUALS the plus character OR element at position index of s EQUALS the minus character)  
            IF element at position index of s EQUALS the minus character  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        
        WHILE index LESS THAN length_of_string AND element at position index of s IS a digit character  
            SET digit TO the integer value of element at position index of s  
            IF result GREATER THAN (maximum_32_bit_signed_integer MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN maximum_32_bit_signed_integer  
                ELSE  
                    RETURN minimum_32_bit_signed_integer  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
