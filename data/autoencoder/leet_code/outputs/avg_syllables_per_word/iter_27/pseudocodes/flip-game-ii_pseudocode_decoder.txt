CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN the value ASSOCIATED WITH state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO position index PLUS one of state EQUALS two PLUS symbols THEN  
                    SET new_state TO the concatenation of the substring from the beginning TO position index MINUS one of state FOLLOWED BY two MINUS symbols FOLLOWED BY the substring from position index PLUS two TO the end of state  
                    IF it IS NOT the case THAT can_win_helper(new_state) HOLDS THEN  
                        SET the value ASSOCIATED WITH state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value ASSOCIATED WITH state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
