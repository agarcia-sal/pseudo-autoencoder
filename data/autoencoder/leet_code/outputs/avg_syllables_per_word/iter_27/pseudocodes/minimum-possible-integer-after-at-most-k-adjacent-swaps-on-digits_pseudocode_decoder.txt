CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of characters in num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero HOLD  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num THEN  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n  
                        IF element at position j of num LESS THAN min_digit THEN  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET a new list TO element at position min_index of num FOLLOWED BY the elements from position i TO position min_index MINUS one of num  
                    ASSIGN to the elements from position i TO position min_index of num this new list  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK  
                END IF  
            END FOR  
            
            IF swapped EQUALS false THEN  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num into a single string  
    END FUNCTION  
END CLASS
