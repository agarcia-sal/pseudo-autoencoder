CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  

        SET word_dict TO a dictionary that returns an empty list for missing keys  
        SET word_length TO the LENGTH OF beginWord  

        FOR each word IN wordList  
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of substring from position zero TO position index MINUS one of word PLUS the character star PLUS substring from position index PLUS one TO position word_length MINUS one of word  
                APPEND word TO the list at key intermediate_key in word_dict  
            END FOR  
        END FOR  

        SET queue TO a double-ended queue containing a pair of beginWord and one  
        SET visited TO a set containing beginWord  

        WHILE queue IS NOT empty  
            REMOVE from the left of queue AND ASSIGN TO current_word AND current_length  

            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of substring from position zero TO position index MINUS one of current_word PLUS the character star PLUS substring from position index PLUS one TO position word_length MINUS one of current_word  

                FOR each word IN the list at key intermediate_word in word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the pair of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN zero  
    END FUNCTION  
END CLASS
