CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO list of triples pairing elements at each matching position from startTime PLUS endTime PLUS profit sorted in ascending order according to the second element of each triple  
        SET end_times TO list of the second element of each triple in jobs  
        SET dp TO list of zeros with length equal to the length of jobs PLUS one  
        FOR i FROM one TO the length of jobs  
            SET current_job_start TO the first element of the triple at position i MINUS one of jobs  
            SET j TO the position that is the upper bound insertion point of current_job_start in end_times  
            SET dp at position i TO the greater value of the element at position i MINUS one of dp AND the sum of the element at position j of dp AND the third element of the triple at position i MINUS one of jobs  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
