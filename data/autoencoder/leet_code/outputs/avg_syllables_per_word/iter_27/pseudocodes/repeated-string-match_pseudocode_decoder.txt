CLASS Solution  
    FUNCTION repeatedStringMatch(a string_parameter b string_parameter) RETURNS integer  
        SET minimum repetitions TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF the string b IS contained WITHIN the string a REPEATED minimum repetitions TIMES  
            RETURN minimum repetitions  
        END IF  
        IF the string b IS contained WITHIN the string a REPEATED the sum of minimum repetitions PLUS one TIMES  
            RETURN the sum of minimum repetitions PLUS one  
        END IF  
        RETURN the negation of one  
    END FUNCTION  
END CLASS
