CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN list containing element zero  
        END IF  
        
        SET graph TO a mapping that provides an empty list for missing keys  
        SET degree TO list of zero values repeated n times  
        
        FOR each pair u v IN edges  
            APPEND v TO list at key u in graph  
            APPEND u TO list at key v in graph  
            INCREMENT degree at position u BY one  
            INCREMENT degree at position v BY one  
        END FOR  
        
        SET leaves TO a double-ended queue containing each index i FROM zero TO n MINUS one WHERE degree at position i EQUALS one  
        
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the LENGTH OF leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each iteration FROM one TO leaves_count  
                SET leaf TO the element at the front of leaves removed  
                FOR each neighbor IN list at key leaf in graph  
                    DECREMENT degree at position neighbor BY one  
                    IF degree at position neighbor EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN list formed from elements in leaves  
    END FUNCTION  
END CLASS
