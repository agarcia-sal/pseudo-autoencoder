CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO list of elements obtained by splitting input at each newline  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters contained in line  
            SET name TO the substring of line starting from position depth plus one TO the end of line  
            
            WHILE the length of stack GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the length of stack GREATER THAN zero  
                SET current_length TO the element at the last position of stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF name CONTAINS the character corresponding to a dot  
                IF current_length GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
