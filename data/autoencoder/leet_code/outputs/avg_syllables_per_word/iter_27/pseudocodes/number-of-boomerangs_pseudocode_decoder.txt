CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO zero  
        FOR each element p1 IN points  
            SET cnt TO an empty counter mapping  
            FOR each element p2 IN points  
                SET dx TO the element at position zero of p1 MINUS the element at position zero of p2  
                SET dy TO the element at position one of p1 MINUS the element at position one of p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the value in cnt at key d  
                INCREMENT the value in cnt at key d BY one  
            END FOR  
        END FOR  
        SET ans TO ans MULTIPLIED BY two  
        RETURN ans  
    END FUNCTION  
END CLASS
