CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS no elements  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN values starting from position two TO the last position  
        SET node TO a new instance of ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list(the element at position next of p1, the element at position next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS none OR the element at position next of head EQUALS none  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the element at position next of head  
            WHILE fast IS NOT none AND the element at position next of fast IS NOT none  
                SET slow TO the element at position next of slow  
                SET fast TO the element at position next of the element at position next of fast  
            END WHILE  
            SET mid TO the element at position next of slow  
            SET slow.next TO none  
            RETURN head, mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO a new instance of ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT none AND l2 IS NOT none  
                IF the value at l1 LESS THAN the value at l2  
                    SET current.next TO l1  
                    SET l1 TO the element at position next of l1  
                ELSE  
                    SET current.next TO l2  
                    SET l2 TO the element at position next of l2  
                END IF  
                SET current TO the element at position next of current  
            END WHILE  
            IF l1 IS NOT none  
                SET current.next TO l1  
            ELSE  
                SET current.next TO l2  
            END IF  
            RETURN the element at position next of dummy  
        END FUNCTION  

        SET left, right TO split(head)  
        SET left TO self.sortList(left)  
        SET right TO self.sortList(right)  

        RETURN merge(left, right)  
    END FUNCTION  
END CLASS
