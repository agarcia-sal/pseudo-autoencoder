CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the pair of x and y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the pair of x and y TO visited  
            FOR each dx dy IN the collection of pairs consisting of minus one and zero, one and zero, zero and minus one, zero and one  
                CALL dfs WITH x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  
  
        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO a list of pairs i j WHERE i ranges FROM zero TO the LENGTH OF grid MINUS one AND j ranges FROM zero TO the LENGTH OF element at position zero of grid MINUS one AND element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH the first element of the first pair in land_cells, the second element of the first pair in land_cells, visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  
  
        IF is_connected() EQUALS False  
            RETURN zero  
        END IF  
  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF is_connected() EQUALS False  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  
  
        RETURN two  
    END FUNCTION  
END CLASS
