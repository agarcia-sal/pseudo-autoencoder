CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF element at position zero of dungeon  
        SET dp TO a two dimensional list with dimensions m PLUS one BY n PLUS one where every element is assigned to infinity  
        SET element at position m row and element at position n MINUS one column of dp TO one  
        SET element at position m MINUS one row and element at position n column of dp TO one  
        FOR i FROM m MINUS one DOWNTO zero  
            FOR j FROM n MINUS one DOWNTO zero  
                SET min_health TO the MINIMUM OF element at position i PLUS one row and element at position j column of dp AND element at position i row and element at position j PLUS one column of dp MINUS element at position i row and element at position j column of dungeon  
                SET element at position i row and element at position j column of dp TO the MAXIMUM OF min_health AND one  
            END FOR  
        END FOR  
        RETURN element at position zero row and element at position zero column of dp  
    END FUNCTION  
END CLASS
