CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
          
        SET left_sums TO a LIST containing zero  
        FOR each num IN the SUBLIST of nums from position zero TO position mid MINUS one  
            SET new_sums TO an EMPTY LIST  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND left_sums BY new_sums  
        END FOR  
          
        SET right_sums TO a LIST containing zero  
        FOR each num IN the SUBLIST of nums from position mid TO the LAST ELEMENT  
            SET new_sums TO an EMPTY LIST  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND right_sums BY new_sums  
        END FOR  
          
        SORT right_sums IN ASCENDING ORDER  
          
        SET min_diff TO positive infinity  
          
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the RESULT OF THE BINARY SEARCH POSITION to INSERT target IN right_sums to MAINTAIN ORDER  
              
            IF pos IS LESS THAN the LENGTH OF right_sums  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS the ELEMENT AT POSITION pos OF right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            IF pos IS GREATER THAN zero  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS the ELEMENT AT POSITION pos MINUS one OF right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
          
        RETURN min_diff  
    END FUNCTION  
END CLASS
