CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference_value TO the ASCII value of character c MINUS the ASCII value of element at position one of string s  
                SET wrapped_difference TO difference_value MODULO the number of letters in the alphabet  
                APPEND the string representation of wrapped_difference TO canonical_form  
            END FOR  
            SET joined_canonical_form TO the concatenation of elements of canonical_form separated by the word comma  
            RETURN joined_canonical_form  
        END FUNCTION  
        
        SET groups TO a new dictionary with default value type list  
        
        FOR each string s IN strings  
            SET canonical_form TO the result of get_canonical_form with parameter s  
            APPEND string s TO the list in groups at key canonical_form  
        END FOR  
        
        SET result TO the list of all values in the dictionary groups  
        RETURN result  
    END FUNCTION  
END CLASS
