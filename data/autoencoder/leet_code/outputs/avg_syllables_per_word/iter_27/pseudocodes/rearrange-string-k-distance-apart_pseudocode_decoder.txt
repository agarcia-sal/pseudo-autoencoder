CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of each character in s  
        
        SET max_heap TO list of pairs of negative count and character for each character and its count in freq  
        TRANSFORM max_heap into a heap structure based on the first element of each pair  
        
        SET wait_queue TO empty list  
        
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            EXTRACT the pair count and character with the highest priority from max_heap  
            APPEND character TO result  
            
            APPEND the pair of count incremented by one and character TO wait_queue  
            
            IF the length of wait_queue EQUALS k  
                REMOVE the first pair count and character FROM wait_queue  
                IF count LESS THAN zero  
                    ADD the pair count and character BACK INTO max_heap maintaining heap property  
                END IF  
            END IF  
        END WHILE  
        
        IF the length of result NOT EQUALS the length of s  
            RETURN empty string  
        END IF  
        
        RETURN the concatenation of all characters in result as a single string  
    END FUNCTION  
END CLASS
