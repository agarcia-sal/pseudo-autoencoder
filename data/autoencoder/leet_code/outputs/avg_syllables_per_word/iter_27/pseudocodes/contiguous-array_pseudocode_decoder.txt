CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with key zero assigned to value negative one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index i AND each element num IN nums enumerated  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF prefix_sum_index CONTAINS key count  
                SET max_length TO the greater value BETWEEN max_length AND i MINUS the element at key count OF prefix_sum_index  
            ELSE  
                ASSIGN the element at key count OF prefix_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
