CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL sort_people_by_height_descending_and_k_ascending WITH people  
        SET queue TO an empty list  
        FOR each person IN people  
            CALL insert_person_into_queue_at_k_position WITH queue person  
        END FOR  
        RETURN queue  
    END FUNCTION  

    FUNCTION sort_people_by_height_descending_and_k_ascending WITH people  
        SORT people USING a comparison that orders by the height of the element at position zero in descending order AND the element at position one in ascending order  
    END FUNCTION  

    FUNCTION insert_person_into_queue_at_k_position WITH queue person  
        INSERT person INTO queue AT the position indicated by the element at position one of person  
    END FUNCTION  
END CLASS
