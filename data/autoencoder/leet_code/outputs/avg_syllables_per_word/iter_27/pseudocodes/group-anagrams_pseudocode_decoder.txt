CLASS Solution  
    FUNCTION groupAnagrams WITH parameter strs  
        SET anagrams TO a mapping from keys to empty lists  
        FOR each string s IN strs  
            SET sorted_str TO the string obtained by arranging the characters of s in ascending alphabetical order and combining them into one continuous sequence  
            APPEND string s TO the list corresponding to key sorted_str in anagrams  
        END FOR  
        SET result TO the collection of all lists stored as values in anagrams converted into a list of lists  
        RETURN result  
    END FUNCTION  
END CLASS
