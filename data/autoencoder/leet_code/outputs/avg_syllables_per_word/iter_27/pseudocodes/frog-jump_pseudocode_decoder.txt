CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS EQUALS empty OR the LENGTH OF stones IS LESS THAN two  
            RETURN false  
        END IF  
        SET stone_set TO the SET OF stones  
        SET memo TO empty dictionary  

        FUNCTION can_jump_to(position, jump)  
            IF position IS EQUALS element at position the LENGTH OF stones MINUS one of stones  
                RETURN true  
            END IF  
            IF the TUPLE OF position AND jump IS IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo  
            END IF  
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND position PLUS next_jump IS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET memo AT TUPLE OF position AND jump TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET memo AT TUPLE OF position AND jump TO false  
            RETURN false  
        END FUNCTION  

        RETURN can_jump_to(element at position zero of stones, zero)  
    END FUNCTION  
END CLASS
