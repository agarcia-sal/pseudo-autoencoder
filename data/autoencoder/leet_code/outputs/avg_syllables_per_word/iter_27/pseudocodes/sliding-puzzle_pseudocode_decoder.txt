CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple of tuples where each tuple is each row from board  
        SET target TO a tuple of tuples representing the rows containing one two three and four five zero in order  
        
        SET neighbors TO a mapping where  
            zero position zero maps to the set containing one and three  
            zero position one maps to the set containing zero two and four  
            zero position two maps to the set containing one and five  
            zero position three maps to the set containing zero and four  
            zero position four maps to the set containing one three and five  
            zero position five maps to the set containing two and four  
        
        SET zero_pos TO undefined  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        INITIALIZE queue AS a deque containing a tuple of start zero_pos and zero  
        INITIALIZE visited AS a set containing start  
        
        WHILE queue is not empty  
            REMOVE the first element of queue and ASSIGN to state zero_pos moves  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN the set mapped to zero_pos in neighbors  
                SET new_state TO a list of lists converted from state  
                SET i TO zero_pos DIVIDED BY three discarding remainder  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three discarding remainder  
                SET nj TO neighbor MODULO three  
                
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state  
                SET element at position nj of element at position ni of new_state TO zero  
                
                SET new_tuple_state TO a tuple of tuples where each tuple is each row of new_state  
                
                IF new_tuple_state is not in visited  
                    ADD new_tuple_state to visited  
                    APPEND a tuple of new_tuple_state neighbor and moves PLUS one to queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
