CLASS Solution  
    FUNCTION minimumWhiteTiles(floor NUMCarpets carpetLen)  
        SET n TO the LENGTH OF floor  
        INITIALIZE dp AS a two dimensional list WITH zero FOR each index FROM zero TO n PLUS one AND FOR each index FROM zero TO numCarpets PLUS one  
        FOR i FROM one TO n  
            SET current tile position TO i MINUS one  
            IF element at position current tile position of floor EQUALS one  
                SET dp at position i index zero TO dp at position i MINUS one index zero PLUS one  
            ELSE  
                SET dp at position i index zero TO dp at position i MINUS one index zero  
            END IF  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET current tile position TO i MINUS one  
                IF element at position current tile position of floor EQUALS one  
                    SET option one TO dp at position i MINUS one index j PLUS one  
                ELSE  
                    SET option one TO dp at position i MINUS one index j  
                END IF  
                SET carpet start TO i MINUS carpetLen  
                IF carpet start LESS THAN zero  
                    SET carpet start TO zero  
                END IF  
                SET option two TO dp at position carpet start index j MINUS one  
                IF option one LESS THAN option two  
                    SET dp at position i index j TO option one  
                ELSE  
                    SET dp at position i index j TO option two  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position n index numCarpets  
    END FUNCTION  
END CLASS
