CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping from character TO position  
        SET max_length TO zero  
        SET start TO zero  
        FOR end FROM zero TO one LESS THAN the LENGTH OF s  
            SET char TO element at position end of s  
            IF char EXISTS IN char_index_map AND the value ASSOCIATED WITH char IN char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value ASSOCIATED WITH char IN char_index_map PLUS one  
            END IF  
            SET the value ASSOCIATED WITH char IN char_index_map TO end  
            IF max_length IS LESS THAN the difference BETWEEN end AND start PLUS one  
                SET max_length TO the difference BETWEEN end AND start PLUS one  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
