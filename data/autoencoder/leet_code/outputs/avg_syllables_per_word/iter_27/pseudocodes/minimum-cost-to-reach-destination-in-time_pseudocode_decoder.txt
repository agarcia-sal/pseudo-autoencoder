CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a list of empty lists with size n  
        FOR each element consisting of x, y, and time IN edges  
            APPEND the pair of y and time TO the element at position x of graph  
            APPEND the pair of x and time TO the element at position y of graph  
        END FOR  
        SET pq TO a list containing a triplet consisting of the element at position zero of passingFees the number zero and the number zero  
        SET visited_time TO a list containing positive infinity repeated n times  
        SET the element at position zero of visited_time TO zero  
        WHILE the LENGTH OF pq is greater than zero  
            REMOVE and RETURN the element with the smallest first value from pq ASSIGN TO current_cost current_city current_time  
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            FOR each element consisting of neighbor and travel_time IN the element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS the element at position neighbor of passingFees  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN the element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET the element at position neighbor of visited_time TO new_time  
                    INSERT the triplet consisting of new_cost neighbor new_time INTO pq maintaining the order by the first value  
                END IF  
            END FOR  
        END WHILE  
        RETURN the number negative one  
    END FUNCTION  
END CLASS
