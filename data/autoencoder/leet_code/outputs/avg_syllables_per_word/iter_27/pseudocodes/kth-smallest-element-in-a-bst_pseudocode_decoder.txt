CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and left equal to none and right equal to none  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value equal to element at position i of values and left equal to none and right equal to none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value equal to element at position i of values and left equal to none and right equal to none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree with arguments the left child of p and the left child of q AND is_same_tree with arguments the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack is not empty OR current NOT EQUALS none  
            WHILE current NOT EQUALS none  
                APPEND current TO stack  
                SET current TO the left child of current  
            END WHILE  
            SET current TO element removed from the end of stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN the value of current  
            END IF  
            SET current TO the right child of current  
        END WHILE  
    END FUNCTION  
END CLASS
