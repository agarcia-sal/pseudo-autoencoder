CLASS Solution  
    FUNCTION findIntegers with parameter n  
        SET binary_representation TO the binary representation of n starting from the second character to the end  
        SET length TO the length of binary_representation  
        
        INITIALIZE list dp with length plus one elements all set to zero  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  
        
        FOR index i FROM two TO length inclusive  
            SET element at position i of dp TO the sum of element at position i minus one of dp AND element at position i minus two of dp  
        END FOR  
        
        SET result TO zero  
        SET previous_bit TO zero  
        
        FOR index i FROM zero TO length minus one inclusive  
            IF element at position i of binary_representation EQUALS one  
                INCREMENT result BY element at position length minus i minus one of dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    EXIT THE LOOP  
                END IF  
            END IF  
            SET previous_bit TO one IF element at position i of binary_representation EQUALS one OTHERWISE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
