CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO an empty list  
        FOR each number FROM one TO n  
            APPEND the string representation of number TO teams  
        END FOR  
        WHILE n GREATER THAN one  
            SET half TO n DIVIDED BY two  
            FOR each index i FROM zero TO half MINUS one  
                SET teams at position i TO an opening parenthesis concatenated WITH element at position i of teams concatenated WITH a comma concatenated WITH element at position n MINUS one MINUS i of teams concatenated WITH a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
