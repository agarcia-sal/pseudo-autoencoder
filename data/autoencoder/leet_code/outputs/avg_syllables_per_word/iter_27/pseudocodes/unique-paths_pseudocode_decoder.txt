CLASS Solution  
    FUNCTION uniquePaths(m n)  
        SET the two dimensional list named dp TO a list containing the value one repeated n times repeated m times  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET the element at position i of dp at position j TO the element at position i MINUS one of dp at position j PLUS the element at position i of dp at position j MINUS one  
            END FOR  
        END FOR  
        RETURN the element at position m MINUS one of dp at position n MINUS one  
    END FUNCTION  
END CLASS
