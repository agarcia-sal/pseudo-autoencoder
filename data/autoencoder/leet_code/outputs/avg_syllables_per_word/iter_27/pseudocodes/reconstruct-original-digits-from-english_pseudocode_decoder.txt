CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO frequency count of each character in s  
        SET digits TO list of ten zeroes representing counts of digits zero to nine  
        
        SET digits at position zero TO count of character z  
        SET digits at position two TO count of character w  
        SET digits at position four TO count of character u  
        SET digits at position six TO count of character x  
        SET digits at position eight TO count of character g  
        
        SET digits at position three TO count of character h MINUS digits at position eight  
        SET digits at position five TO count of character f MINUS digits at position four  
        SET digits at position seven TO count of character s MINUS digits at position six  
        SET digits at position nine TO count of character i MINUS digits at position five MINUS digits at position six MINUS digits at position eight  
        SET digits at position one TO count of character o MINUS digits at position zero MINUS digits at position two MINUS digits at position four  
        
        SET result TO empty list  
        FOR i FROM zero TO nine  
            FOR each occurrence FROM one TO digits at position i  
                APPEND string form of i TO result  
            END FOR  
        END FOR  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
