CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        ASSIGN reference TO a dictionary with default integer zero named char_count  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF string s MINUS one  
            INCREMENT the count associated with element at position right of string s in char_count BY one  
            
            WHILE the SIZE OF char_count IS GREATER THAN two  
                DECREMENT the count associated with element at position left of string s in char_count BY one  
                IF the count associated with element at position left of string s in char_count EQUALS zero  
                    REMOVE element at position left of string s FROM the dictionary char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
