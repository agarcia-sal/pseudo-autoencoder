CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of elements from zero TO n MINUS one  
        SET self.rank TO a list of ones with LENGTH n  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  

        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR variable i AND variable edge IN enumeration of edges  
            APPEND i TO edge  
        END FOR  

        SORT edges BY the element at position two of each edge in ascending order  

        FUNCTION mst(exclude, include)  
            SET uf TO new UnionFind object initialized with n  
            SET weight TO zero  

            IF include IS NOT None  
                CALL uf.union WITH the first element of include AND the second element of include  
                INCREMENT weight BY the third element of include  
            END IF  

            FOR variables u v w idx IN each element of edges  
                IF idx EQUALS exclude  
                    CONTINUE to the next iteration  
                END IF  
                IF uf.union(u, v) RETURNS True  
                    INCREMENT weight BY w  
                END IF  
            END FOR  

            SET root TO uf.find(zero)  
            FOR variable i FROM zero TO n MINUS one  
                IF uf.find(i) NOT EQUALS root  
                    RETURN positive infinity  
                END IF  
            END FOR  

            RETURN weight  
        END FUNCTION  

        SET mst_weight TO the result of calling mst with no arguments  

        SET critical TO empty list  
        SET pseudo_critical TO empty list  

        FOR variables u v w idx IN each element of edges  
            IF mst called with exclude equals idx RETURNS a value GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF mst called with include equals the tuple of u v w RETURNS a value EQUALS mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  

        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
