CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND a list consisting of zero THEN index PLUS one THEN element at position index of wells TO pipes  
        END FOR  
        SORT pipes BY the element at position two of each element in ascending order  
        SET parent TO a list of integers FROM zero TO n  
        FUNCTION find(x)  
            IF the element at position x of parent NOT EQUALS x  
                SET the element at position x of parent TO the result of calling find WITH the element at position x of parent  
            END IF  
            RETURN the element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO the result of calling find WITH x  
            SET rootY TO the result of calling find WITH y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each element consisting of house1, house2, cost IN pipes  
            IF the result of calling union WITH house1 AND house2 EQUALS True  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
