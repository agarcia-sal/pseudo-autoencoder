CLASS Solution  
    FUNCTION numMusicPlaylists(n, goal, k)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of length goal PLUS one each containing a list of length n PLUS one filled with zero  
        SET element at position zero of element at position zero of dp TO one  
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position i of element at position j of dp TO element at position i MINUS one of element at position j MINUS one of dp MULTIPLIED BY n MINUS j MINUS one PLUS one  
                SET element at position i of element at position j of dp TO element at position i of element at position j of dp MODULO MOD  
                IF j GREATER THAN k  
                    INCREMENT element at position i of element at position j of dp BY element at position i MINUS one of element at position j of dp MULTIPLIED BY j MINUS k  
                    SET element at position i of element at position j of dp TO element at position i of element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position goal of element at position n of dp  
    END FUNCTION  
END CLASS
