CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO empty collection of key value pairs  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET the element at position right of s AS key IN char_map TO right  
            
            IF the LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map with the MINIMUM value  
                REMOVE the key leftmost_char FROM char_map and SET left TO the removed value PLUS one  
            END IF  
            
            SET max_length TO the GREATER VALUE BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
