CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS current_string WITH DEFAULT empty string left_count WITH DEFAULT zero right_count WITH DEFAULT zero  
            IF the LENGTH OF current_string EQUALS two MULTIPLIED BY n  
                APPEND current_string TO result  
                RETURN  
            END IF  
            IF left_count LESS THAN n  
                CALL backtrack WITH current_string CONCATENATED WITH an opening parenthesis AND left_count INCREMENTED BY one AND right_count  
            END IF  
            IF right_count LESS THAN left_count  
                CALL backtrack WITH current_string CONCATENATED WITH a closing parenthesis AND left_count AND right_count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
