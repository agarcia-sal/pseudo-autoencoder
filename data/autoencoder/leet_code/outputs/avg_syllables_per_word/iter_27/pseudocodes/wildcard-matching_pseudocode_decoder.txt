CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two dimensional list of Boolean values with the number of rows equal to the length of string s PLUS one and the number of columns equal to the length of string p PLUS one and all values initialized to False  
        SET the element at position zero of dp and position zero of dp at zero TO True  
        FOR each index j FROM one TO the length of string p  
            IF the element at position j MINUS one of string p EQUALS the character star  
                SET the element at position zero of dp and position j of dp TO the value of the element at position zero of dp and position j MINUS one of dp  
            END IF  
        END FOR  
        FOR each index i FROM one TO the length of string s  
            FOR each index j FROM one TO the length of string p  
                IF the element at position j MINUS one of string p EQUALS the character star  
                    SET the element at position i of dp and position j of dp TO the logical OR of the element at position i MINUS one of dp and position j of dp AND the element at position i of dp and position j MINUS one of dp  
                ELSE IF the element at position j MINUS one of string p EQUALS the character question mark OR the element at position i MINUS one of string s EQUALS the element at position j MINUS one of string p  
                    SET the element at position i of dp and position j of dp TO the value of the element at position i MINUS one of dp and position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the value of the element at position the length of string s of dp and position the length of string p of dp  
    END FUNCTION  
END CLASS
