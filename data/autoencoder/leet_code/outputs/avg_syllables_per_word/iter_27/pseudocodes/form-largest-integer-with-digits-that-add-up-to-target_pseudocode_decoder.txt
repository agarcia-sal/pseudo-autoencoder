CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO empty dictionary  
        FOR each index AND each element c IN cost WITH index starting from zero  
            ASSIGN to cost_to_digit at key c the string representation of index PLUS one  
        END FOR  
        
        SET dp TO a list starting with empty string FOLLOWED BY a sequence of the string minus one repeated target times  
        
        FOR t FROM one TO target  
            FOR each cost_key AND digit_value IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO cost_key AND dp at position t MINUS cost_key IS NOT EQUAL TO string minus one  
                    SET candidate TO concatenation of digit_value and dp at position t MINUS cost_key  
                    IF dp at position t EQUALS string minus one OR LENGTH OF candidate GREATER THAN LENGTH OF dp at position t OR (LENGTH OF candidate EQUALS LENGTH OF dp at position t AND candidate GREATER THAN dp at position t)  
                        SET dp at position t TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF dp at position target NOT EQUAL TO string minus one  
            RETURN dp at position target  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
