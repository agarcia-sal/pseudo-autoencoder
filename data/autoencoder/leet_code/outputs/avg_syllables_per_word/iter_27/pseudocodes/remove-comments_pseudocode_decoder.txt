CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment EQUALS True  
                    IF element at position i of line EQUALS forward slash AND element at position i PLUS one of line EQUALS asterisk  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF element at position i of line EQUALS forward slash AND element at position i PLUS one of line EQUALS forward slash  
                        BREAK  
                    ELSE IF element at position i of line EQUALS forward slash AND element at position i PLUS one of line EQUALS asterisk  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND element at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line IS NOT empty AND in_block_comment EQUALS False  
                APPEND concatenation of elements of current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
