CLASS Solution  
    FUNCTION generatePalindromes with parameter s  
        FUNCTION dfs with parameter t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT cnt at character c BY two  
                    CALL dfs with concatenation of character c PLUS t PLUS character c  
                    INCREMENT cnt at character c BY two  
                END IF  
            END FOR  
        END FUNCTION  
        SET cnt TO a mapping of counts of each character in s  
        SET mid TO an empty string  
        FOR each character c AND its count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT an empty string  
                    RETURN an empty list  
                END IF  
                SET mid TO character c  
                DECREMENT cnt at character c BY one  
            END IF  
        END FOR  
        SET ans TO an empty list  
        CALL dfs with argument mid  
        RETURN ans  
    END FUNCTION  
END CLASS
