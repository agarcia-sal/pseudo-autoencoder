CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF s  
        SET last TO a list of negative one repeated twenty six times  
        SET last2 TO a list of negative one repeated twenty six times  
        SET result TO zero  
        
        FOR each position i AND character char IN s WITH their indices  
            SET idx TO the numerical code of char MINUS the numerical code of uppercase letter A  
            SET contribution TO the difference BETWEEN i AND the element at position idx of last MULTIPLIED BY the difference BETWEEN the element at position idx of last AND the element at position idx of last2  
            SET result TO the sum of result AND contribution MODULO MOD  
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        
        FOR each index i FROM zero TO twenty five  
            SET contribution TO the difference BETWEEN n AND the element at position i of last MULTIPLIED BY the difference BETWEEN the element at position i of last AND the element at position i of last2  
            SET result TO the sum of result AND contribution MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
