CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO stations  

        FOR each position fuel IN stations  
            WHILE the max_heap is not empty AND current_fuel LESS THAN position  
                REMOVE the smallest element FROM max_heap and assign its negation TO fuel_amount  
                INCREMENT current_fuel BY fuel_amount  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  
            INSERT the negation of fuel INTO max_heap  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
