CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty list  
            RETURN zero  
        END IF  
        SORT intervals BY the value at position zero of each element  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF LENGTH OF min_heap GREATER THAN zero AND the value at position zero of meeting GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the value at position one of meeting TO min_heap  
        END FOR  
        SET number_of_rooms TO LENGTH OF min_heap  
        RETURN number_of_rooms  
    END FUNCTION  
END CLASS
