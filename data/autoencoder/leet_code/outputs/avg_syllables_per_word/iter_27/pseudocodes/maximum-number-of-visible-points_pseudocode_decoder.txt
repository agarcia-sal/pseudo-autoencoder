CLASS Solution  
    FUNCTION visiblePoints(points pointsList OF LIST OF LIST OF INTEGER angle angleValue OF INTEGER location locationList OF LIST OF INTEGER) RETURNS INTEGER  
        SET angles TO empty list  
        SET extra_points TO zero  
        
        FOR each pair OF x AND y IN pointsList  
            IF x EQUALS element at position zero of locationList AND y EQUALS element at position one of locationList  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_rad TO the arctangent of the difference of y MINUS element at position one of locationList AND x MINUS element at position zero of locationList  
            SET angle_deg TO convert angle_rad FROM radians TO degrees  
            APPEND angle_deg TO angles  
        END FOR  
        
        SORT angles in ascending order  
        
        SET doubled_angles TO angles concatenated WITH each element of angles PLUS three hundred sixty  
        ASSIGN doubled_angles TO angles  
        
        SET max_visible TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the length of angles MINUS one  
            WHILE element at position right of angles MINUS element at position left of angles GREATER THAN angleValue  
                INCREMENT left BY one  
            END WHILE  
            SET max_visible TO the greater value BETWEEN max_visible AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
