CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO empty dictionary with default integer values  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET key TO the bitwise AND of i AND j  
                INCREMENT the value at key in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key and count value IN the elements and values of pair_and_count  
                IF the bitwise AND of key AND k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
