CLASS Solution  
    FUNCTION advantageCount(nums1 AS List OF integer, nums2 AS List OF integer) RETURNS List OF integer  
        SORT nums1 IN ascending order  
        SET nums2_with_indices TO a list of pairs where each pair consists of the element at position index of nums2 and index for each index and element in nums2  
        SORT nums2_with_indices IN descending order based on the element value  
        SET result TO a list of zero values with the LENGTH OF nums1  
        SET left TO zero  
        SET right TO LENGTH OF nums1 MINUS one  
        FOR each pair consisting of value and index IN nums2_with_indices  
            IF element at position right of nums1 GREATER THAN value THEN  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
