CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeros with the LENGTH equal to batchSize  
        FOR each group IN groups  
            INCREMENT element at position equal to the remainder of group DIVIDED BY batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, remainder_count_var)  
            SET remainder_count_var TO a list converted from remainder_count_var  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF element at position index of remainder_count_var EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT element at position index of remainder_count_var BY one  
                SET new_remainder TO remainder MINUS index MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, remainder_count_var as unpacked arguments)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, remainder_count_var as unpacked arguments)  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT element at position index of remainder_count_var BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, remainder_count as unpacked arguments)  
    END FUNCTION  
END CLASS
