CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a mapping from each index from zero TO n minus one TO an empty list  
        FOR each flight in flights  
            DECOMPOSE flight INTO u, v, w  
            APPEND a pair consisting of v and w TO the list at position u in graph  
        END FOR  
        
        SET min_heap TO a list containing a triple consisting of zero cost, stops as negative one, and src as current city  
        SET min_cost TO a mapping from src TO negative one  
        
        WHILE min_heap IS NOT empty  
            REMOVE and ASSIGN to cost stops city the element with the smallest cost from min_heap  
            
            IF city EQUALS dst  
                RETURN cost  
            END IF  
            
            IF stops IS GREATER THAN OR EQUAL TO k  
                CONTINUE TO the next iteration of the WHILE loop  
            END IF  
            
            FOR each neighbor price IN the list at position city in graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor NOT IN min_cost OR next_stops IS LESS THAN the value at neighbor in min_cost OR next_cost IS LESS THAN the cost value of the element at position zero in min_heap  
                    SET the value at neighbor in min_cost TO next_stops  
                    ADD a triple consisting of next_cost next_stops and neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
