CLASS Solution  
    FUNCTION sumScores WITH parameter s  
        SET n TO the LENGTH OF s  
        SET lcp TO a list of zeros with the LENGTH OF s  
        SET left_pointer TO zero  
        SET right_pointer TO zero  
        FOR index i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO right_pointer THEN  
                SET lcp at position i TO the SMALLER VALUE BETWEEN right_pointer MINUS i PLUS one AND lcp at position i MINUS left_pointer  
            END IF  
            WHILE i PLUS lcp at position i LESS THAN n AND element at position lcp at position i OF s EQUALS element at position i PLUS lcp at position i OF s  
                INCREMENT lcp at position i BY one  
            END WHILE  
            IF i PLUS lcp at position i MINUS one GREATER THAN right_pointer THEN  
                SET left_pointer TO i  
                SET right_pointer TO i PLUS lcp at position i MINUS one  
            END IF  
        END FOR  
        RETURN the SUM OF all elements IN lcp PLUS n  
    END FUNCTION  
END CLASS
