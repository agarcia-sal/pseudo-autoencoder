CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the element popped from the left side of queue  
        IF i LESS THAN the length of values AND element at position i of values is not None THEN  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values is not None THEN  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree on p.left and q.left AND the result of is_same_tree on p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left EQUALS None AND root.right EQUALS None THEN  
            RETURN root.val  
        END IF  

        SET graph TO a default dictionary that creates new empty lists for missing keys  

        FUNCTION dfs(node, parent EQUALS None)  
            IF parent is not None THEN  
                APPEND parent.val TO list at key node.val in graph  
                APPEND node.val TO list at key parent.val in graph  
            END IF  
            IF node.left is not None THEN  
                CALL dfs with node.left and node as arguments  
            END IF  
            IF node.right is not None THEN  
                CALL dfs with node.right and node as arguments  
            END IF  
        END FUNCTION  

        CALL dfs with root as argument  

        SET queue TO a double ended queue containing k  
        SET visited TO an empty set  

        WHILE queue is not empty DO  
            SET current TO the element popped from the left side of queue  
            ADD current TO visited  

            IF the length of list at key current in graph EQUALS one AND current NOT EQUALS root.val THEN  
                RETURN current  
            END IF  

            FOR each neighbor IN list at key current in graph DO  
                IF neighbor NOT IN visited THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN the integer negative one  
    END FUNCTION  
END CLASS
