CLASS Solution  
    FUNCTION shiftingLetters(s parameter of type string shifts parameter of type list of integers) RETURNS string  
        SET total_shifts TO zero  
        FOR index FROM length of shifts MINUS one TO zero DECREASING BY one  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO length of s MINUS one  
            SET numerical_value TO numerical position of element at position index of s MINUS numerical position of character representing letter a  
            SET shifted_value TO numerical_value PLUS element at position index of shifts  
            SET wrapped_value TO shifted_value MODULO twenty six  
            SET new_character TO character corresponding to wrapped_value PLUS numerical position of character representing letter a  
            APPEND new_character TO result  
        END FOR  
        RETURN concatenation of all elements of result  
    END FUNCTION  
END CLASS
