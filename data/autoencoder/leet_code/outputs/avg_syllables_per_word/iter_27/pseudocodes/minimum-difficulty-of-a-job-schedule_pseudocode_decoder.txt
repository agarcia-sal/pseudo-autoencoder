CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN the NEGATIVE one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM value of the elements from position i TO the END of jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR index j FROM i TO the value of n MINUS d PLUS one MINUS one  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND the element at position j of jobDifficulty  
                SET min_difficulty TO the MINIMUM value BETWEEN min_difficulty AND current_max PLUS the result of dp function CALLED with parameters j PLUS one AND d MINUS one  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN the result of dp function CALLED with zero AND d  
    END FUNCTION  
END CLASS
