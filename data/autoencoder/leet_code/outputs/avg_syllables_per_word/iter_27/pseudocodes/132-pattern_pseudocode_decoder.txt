CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums starting from the last element TO the first element  
            IF num LESS THAN second  
                RETURN True  
            END IF  
            WHILE the stack is not empty AND element at the last position of stack LESS THAN num  
                SET second TO remove and return the element at the last position of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
