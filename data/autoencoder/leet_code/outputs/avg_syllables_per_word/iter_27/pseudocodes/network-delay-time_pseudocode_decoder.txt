CLASS Solution  
    FUNCTION networkDelayTime(times list of list of integers n integer k integer) RETURNS integer  
        SET graph TO an empty collection that maps each integer from one TO n TO an empty list  
        FOR each element IN times  
            DESTRUCTURE element INTO u v w  
            APPEND the pair of v and w TO the list at key u in graph  
        END FOR  
        
        SET dist TO a collection that maps each integer from one TO n TO infinity  
        SET the value at key k in dist TO zero  
        
        SET min_heap TO a list containing the pair of zero and k  
        
        WHILE min_heap is not empty  
            EXTRACT the pair current_dist and node from the front element of min_heap by removing the element with the smallest first component  
            
            IF current_dist IS GREATER THAN the value at key node in dist  
                CONTINUE to the next iteration of the while loop  
            END IF  
            
            FOR each pair neighbor and weight IN the list at key node in graph  
                SET distance TO current_dist PLUS weight  
                IF distance IS LESS THAN the value at key neighbor in dist  
                    SET the value at key neighbor in dist TO distance  
                    ADD the pair of distance and neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time IS EQUAL TO infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
