CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN word representing zero  
        END IF  

        SET below_twenty TO list containing empty string and words representing numbers one through nineteen in order  
        SET tens TO list containing two empty strings followed by words representing multiples of ten from twenty to ninety in order  
        SET thousands TO list containing empty string and words representing thousand million and billion in order  

        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN element at position n of below_twenty CONCATENATED WITH space  
            ELSE IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten of tens CONCATENATED WITH space CONCATENATED WITH call to helper with argument n MODULO ten  
            ELSE  
                RETURN element at position n DIVIDED BY one hundred of below_twenty CONCATENATED WITH space CONCATENATED WITH word Hundred CONCATENATED WITH space CONCATENATED WITH call to helper with argument n MODULO one hundred  
            END IF  
        END FUNCTION  

        SET result TO empty string  
        SET i TO zero  

        WHILE num GREATER THAN zero  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO call to helper with argument num MODULO one thousand CONCATENATED WITH element at position i of thousands CONCATENATED WITH space CONCATENATED WITH result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  

        RETURN result with leading and trailing whitespace removed  
    END FUNCTION  
END CLASS
