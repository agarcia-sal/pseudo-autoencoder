CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    CREATE a deque called queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the leftmost element FROM queue AND ASSIGN it TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute OF node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute OF node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val attribute OF p NOT EQUALS the val attribute OF q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left attribute OF p and the left attribute OF q AND the result of calling is_same_tree with the right attribute OF p and the right attribute OF q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION tree2str(self root)  
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN empty string  
            END IF  
            SET result TO the string value of the val attribute OF node  
            IF the left attribute OF node EXISTS OR the right attribute OF node EXISTS  
                SET result TO result concatenated with an opening parenthesis concatenated with the result of calling helper with the left attribute OF node concatenated with a closing parenthesis  
            END IF  
            IF the right attribute OF node EXISTS  
                SET result TO result concatenated with an opening parenthesis concatenated with the result of calling helper with the right attribute OF node concatenated with a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN the result of calling helper with root  
    END FUNCTION  
END CLASS
