CLASS Solution  
    FUNCTION minOperations(target  
                           arr)  
        SET index_map TO empty mapping from value to index  
        FOR each index FROM zero TO the LENGTH OF target MINUS one  
            SET value TO element at position index of target  
            SET element at position value of index_map TO index  
        END FOR  
      
        SET transformed_arr TO empty list  
        FOR each value IN arr  
            IF value EXISTS AS a key IN index_map THEN  
                APPEND element at position value of index_map TO transformed_arr  
            END IF  
        END FOR  
      
        SET lis TO empty list  
        FOR each num IN transformed_arr  
            SET pos TO the smallest position in lis where num CAN BE inserted to keep lis sorted in ascending order  
            IF pos EQUALS the LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
      
        SET lcs_length TO the LENGTH OF lis  
      
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
