CLASS Solution  
    FUNCTION maxVacationDays(flights AS List OF List OF integer, days AS List OF List OF integer) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        INITIALIZE dp AS list OF length n WITH each element SET TO negative one  
        SET the element at position zero OF dp TO zero  
        
        INITIALIZE graph AS a mapping FROM integers TO list OF integers WITH keys FROM zero TO n MINUS one EACH associated WITH an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of flights at position j EQUALS one  
                    APPEND j TO the list associated WITH key i IN graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            INITIALIZE new_dp AS list OF length n WITH each element SET TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i OF dp EQUALS negative one  
                    CONTINUE TO the next iteration  
                END IF  
                SET the element at position i OF new_dp TO the maximum BETWEEN the current value at position i OF new_dp AND the element at position i OF dp PLUS the element at position i OF days at position week  
                FOR each j IN the list associated WITH key i IN graph  
                    SET the element at position j OF new_dp TO the maximum BETWEEN the current value at position j OF new_dp AND the element at position i OF dp PLUS the element at position j OF days at position week  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN the maximum value IN dp  
    END FUNCTION  
END CLASS
