CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO an empty dictionary  
        FOR each index AND character IN enumeration of s  
            SET character in last_occurrence TO index  
        END FOR  
        
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index AND character IN enumeration of s  
            IF character IS IN in_stack THEN  
                CONTINUE to next iteration  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND  
                  character IS LESS THAN the element at position LAST of stack AND  
                  index IS LESS THAN the value at key equal to element at position LAST of stack in last_occurrence  
                SET removed_char TO the element REMOVED FROM position LAST of stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result_string TO the concatenation of all elements IN stack in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
