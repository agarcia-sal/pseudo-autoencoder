CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        ASSIGN reference TO an empty deque named deque_  
        FOR each pair xj AND yj IN points  
            WHILE deque_ is not empty AND the difference between xj AND the second element of the first pair in deque_ is GREATER THAN k  
                REMOVE the first element FROM deque_  
            END WHILE  
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND the sum of yj PLUS xj PLUS the first element of the first pair in deque_  
            END IF  
            WHILE deque_ is not empty AND the difference between yj AND xj is GREATER THAN OR EQUAL TO the first element of the last pair in deque_  
                REMOVE the last element FROM deque_  
            END WHILE  
            APPEND the pair of the difference between yj AND xj PLUS xj TO deque_  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
