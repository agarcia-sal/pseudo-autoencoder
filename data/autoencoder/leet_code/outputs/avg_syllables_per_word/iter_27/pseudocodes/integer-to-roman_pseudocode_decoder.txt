CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO a list containing one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, and one in that order  
        SET syms TO a list containing M, CM, D, CD, C, XC, L, XL, X, IX, V, IV, and I in that order  
        SET roman_numeral TO an empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            SET count TO num DIVIDED BY element at position i of val  
            FOR each _ FROM one TO count  
                SET roman_numeral TO roman_numeral PLUS element at position i of syms  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
