CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO an empty list  
        FOR each num IN nums  
            ADD num TO min_heap USING heap push operation  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                REMOVE the smallest element FROM min_heap USING heap pop operation  
            END IF  
        END FOR  
        RETURN the first element AT position one OF min_heap  
    END FUNCTION  
END CLASS
