CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the length of values AND the element at position i of values DOES NOT EQUAL nothing  
            SET node dot left TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values DOES NOT EQUAL nothing  
            SET node dot right TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments p dot left AND q dot left AND the result of is_same_tree with arguments p dot right AND q dot right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION closestKValues with parameters root target k  
        FUNCTION inorder_traversal with parameter node  
            IF node EQUALS nothing  
                RETURN  
            END IF  
            CALL inorder_traversal with node dot left  
            IF the length of result LESS THAN k  
                APPEND the value attribute of node TO result  
            ELSE  
                IF the absolute value of the value attribute of node MINUS target LESS THAN the absolute value of the element at position one of result MINUS target  
                    REMOVE the element from the left end of result  
                    APPEND the value attribute of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with node dot right  
        END FUNCTION

        SET result TO an empty double ended queue  
        CALL inorder_traversal with root  
        RETURN the list converted from result  
    END FUNCTION  
END CLASS
