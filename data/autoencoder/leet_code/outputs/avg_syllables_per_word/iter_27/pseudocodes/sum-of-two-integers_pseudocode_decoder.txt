CLASS Solution  
    FUNCTION getSum(a integer parameter b integer parameter) RETURNS integer  
        SET MAX TO the greatest positive integer representable with thirty-two bits  
        SET mask TO the number representing thirty-two consecutive ones in binary  
        WHILE b NOT EQUALS zero  
            SET carry TO the bitwise AND of a and b MULTIPLIED BY two  
            SET a TO the bitwise exclusive OR of a and b AND the bitwise AND with mask  
            SET b TO the bitwise AND of carry and mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of a exclusive OR mask  
        END IF  
    END FUNCTION  
END CLASS
