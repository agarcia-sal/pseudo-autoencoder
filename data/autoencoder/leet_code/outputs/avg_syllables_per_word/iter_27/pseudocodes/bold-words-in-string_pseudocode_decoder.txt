CLASS Solution  
    FUNCTION boldWords(words PARAMETER, s PARAMETER)  
        SET bold TO a list of Boolean false values of the same length as s  
        FOR each word IN words  
            SET start TO the position of the first occurrence of word in s  
            WHILE start NOT EQUALS negative one  
                FOR i FROM start TO start PLUS the length of word MINUS one  
                    SET the element at position i of bold TO true  
                END FOR  
                SET start TO the position of the next occurrence of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the length of s  
            IF the element at position i of bold EQUALS true  
                APPEND the string consisting of less than symbol letter b greater than symbol TO result  
                WHILE i LESS THAN the length of s AND the element at position i of bold EQUALS true  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string consisting of less than symbol slash letter b greater than symbol TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of the elements of result into a single string  
    END FUNCTION  
END CLASS
