CLASS Solution  
    FUNCTION colorTheGrid(m parameter n parameter) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION generate_valid_columns(prev_col parameter) RETURNS list  
            SET colors TO list containing one two three  
            SET valid_cols TO empty list  
            
            FUNCTION backtrack(col parameter row parameter)  
                IF row EQUALS m THEN  
                    APPEND tuple constructed from col TO valid_cols  
                    RETURN  
                END IF  
                FOR each color IN colors  
                    IF col is empty OR element at last position of col NOT EQUALS color THEN  
                        IF prev_col IS NOT null AND element at position row of prev_col EQUALS color THEN  
                            CONTINUE to next iteration  
                        END IF  
                        APPEND color TO col  
                        CALL backtrack WITH col AND row PLUS one  
                        REMOVE last element FROM col  
                    END IF  
                END FOR  
            END FUNCTION  
            
            CALL backtrack WITH empty list AND zero  
            RETURN valid_cols  
        END FUNCTION  
        
        SET all_valid_cols TO CALL generate_valid_columns WITH null  
        SET num_valid_cols TO the LENGTH OF all_valid_cols  
        
        SET compatible_cols TO mapping FROM each integer i FROM zero TO num_valid_cols MINUS one TO empty list  
        FOR i FROM zero TO num_valid_cols MINUS one  
            FOR j FROM zero TO num_valid_cols MINUS one  
                SET all_elements_different TO true  
                FOR k FROM zero TO m MINUS one  
                    IF element at position k of element at position i of all_valid_cols EQUALS element at position k of element at position j of all_valid_cols THEN  
                        SET all_elements_different TO false  
                        BREAK out of inner loop  
                    END IF  
                END FOR  
                IF all_elements_different IS true THEN  
                    APPEND j TO compatible_cols at key i  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO list of integer one repeated num_valid_cols times  
        
        FOR iteration FROM one TO n MINUS one  
            SET next_dp TO list of integer zero repeated num_valid_cols times  
            FOR i FROM zero TO num_valid_cols MINUS one  
                FOR each j IN compatible_cols at key i  
                    SET next_dp element at position j TO the sum of element at position j of next_dp PLUS element at position i of dp  
                    SET next_dp element at position j TO next_dp element at position j MODULO MOD  
                END FOR  
            END FOR  
            SET dp TO next_dp  
        END FOR  
        
        SET total_ways TO zero  
        FOR each value IN dp  
            INCREMENT total_ways BY value  
        END FOR  
        SET total_ways TO total_ways MODULO MOD  
        
        RETURN total_ways  
    END FUNCTION  
END CLASS
