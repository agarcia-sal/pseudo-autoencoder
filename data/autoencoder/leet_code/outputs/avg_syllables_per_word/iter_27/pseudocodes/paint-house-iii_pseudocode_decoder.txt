CLASS Solution  
    FUNCTION minCost(houses LIST OF INTEGERS cost LIST OF LIST OF INTEGERS m INTEGER n INTEGER target INTEGER) RETURNS INTEGER  
        FUNCTION dp(i INTEGER prev_color INTEGER groups INTEGER) RETURNS INTEGER  
            IF i EQUALS m  
                IF groups EQUALS target  
                    RETURN zero  
                ELSE  
                    RETURN a value representing positive infinity  
                END IF  
            END IF  
            
            IF the element at position i of houses DOES NOT EQUAL zero  
                SET new_groups TO groups PLUS one IF the element at position i of houses NOT EQUALS prev_color OTHERWISE groups  
                RETURN the result of calling dp with i PLUS one prev_color set to the element at position i of houses and new_groups  
            END IF  
            
            SET min_cost TO a value representing positive infinity  
            FOR color FROM one TO n  
                SET new_groups TO groups PLUS one IF color NOT EQUALS prev_color OTHERWISE groups  
                SET current_cost TO the element at position i of cost at position color MINUS one PLUS the result of calling dp with i PLUS one color new_groups  
                IF current_cost LESS THAN min_cost  
                    SET min_cost TO current_cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        SET result TO the result of calling dp with zero zero zero  
        IF result NOT EQUALS a value representing positive infinity  
            RETURN result  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
