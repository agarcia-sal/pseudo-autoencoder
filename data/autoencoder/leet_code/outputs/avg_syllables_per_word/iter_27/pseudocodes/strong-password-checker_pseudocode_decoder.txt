CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO false  
        FOR each character IN password  
            IF character IS a lowercase letter  
                SET has_lower TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET has_upper TO false  
        FOR each character IN password  
            IF character IS an uppercase letter  
                SET has_upper TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET has_digit TO false  
        FOR each character IN password  
            IF character IS a digit  
                SET has_digit TO true  
                BREAK  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS the SUM of has_lower PLUS has_upper PLUS has_digit  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        
        WHILE i IS LESS THAN n  
            IF element at position i of password EQUALS element at position i MINUS one of password AND element at position i of password EQUALS element at position i MINUS two of password  
                SET length TO two  
                WHILE i IS LESS THAN n AND element at position i of password EQUALS element at position i MINUS one of password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN six  
            RETURN the GREATER VALUE BETWEEN missing_types AND six MINUS n  
        ELSE IF n IS LESS THAN OR EQUAL TO twenty  
            RETURN the GREATER VALUE BETWEEN missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            DECREMENT replace BY the SMALLER VALUE BETWEEN delete AND one  
            DECREMENT replace BY the INTEGER DIVISION BY two OF the SMALLER VALUE BETWEEN the GREATER VALUE BETWEEN delete MINUS one AND zero AND two MULTIPLIED BY two  
            DECREMENT replace BY the INTEGER DIVISION BY three OF the GREATER VALUE BETWEEN delete MINUS one MINUS two MULTIPLIED BY two AND zero  
            
            RETURN delete PLUS the GREATER VALUE BETWEEN missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
