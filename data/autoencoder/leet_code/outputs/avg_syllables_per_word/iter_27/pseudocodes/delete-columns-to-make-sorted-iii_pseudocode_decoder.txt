CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO an empty list  
        FOR position FROM zero TO n MINUS one  
            APPEND one TO dp  
        END FOR  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for every k FROM zero TO m MINUS one the element at position k of element at position j of strs IS LESS THAN OR EQUAL TO the element at position k of element at position i of strs  
                    IF the element at position i of dp IS LESS THAN the element at position j of dp PLUS one  
                        SET the element at position i of dp TO the element at position j of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the maximum value IN dp  
    END FUNCTION  
END CLASS
