CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO AN EMPTY doubly ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is GREATER THAN zero  
        SET node TO THE FIRST element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH p dot left AND q dot left AND is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO dfs WITH node dot left  
            SET right_min, right_max, right_size TO dfs WITH node dot right  
            IF left_max IS LESS THAN node dot val AND node dot val IS LESS THAN right_min  
                RETURN THE MINIMUM OF node dot val AND left_min, THE MAXIMUM OF node dot val AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, THE MAXIMUM OF left_size AND right_size  
        END FUNCTION  
        RETURN THE third element OF dfs WITH root  
    END FUNCTION  
END CLASS
