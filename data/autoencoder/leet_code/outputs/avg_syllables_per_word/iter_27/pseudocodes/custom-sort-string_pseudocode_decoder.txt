CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        FOR each index i and character c IN order by enumeration  
            SET element at key c in dictionary d TO i  
        END FOR  
        SET sorted_characters TO the list of characters in s sorted in ascending order according to the integer value corresponding to character at position in dictionary d if character exists otherwise zero  
        SET result_string TO the concatenation of all characters in sorted_characters in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
