CLASS Solution  
    FUNCTION numRescueBoats(people LIST OF INTEGER limit INTEGER) RETURNS INTEGER  
        SORT the list of people by their weights in ascending order  
        SET left TO zero  
        SET right TO the length of people MINUS one  
        SET boats TO zero  
        WHILE the value of left is less than or equal to the value of right  
            IF the element at position left of people PLUS the element at position right of people is LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
