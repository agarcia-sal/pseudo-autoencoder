CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO list of substrings obtained by dividing path at each forward slash  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS a single period  
                CONTINUE to next iteration  
            ELSE IF component EQUALS two periods  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO a single forward slash concatenated WITH all elements of stack joined by a single forward slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
