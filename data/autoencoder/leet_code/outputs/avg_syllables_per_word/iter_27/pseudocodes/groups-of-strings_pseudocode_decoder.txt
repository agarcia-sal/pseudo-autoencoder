CLASS UnionFind  
    FUNCTION __init__ (size)  
        SET parent TO list of each number from zero TO size MINUS one  
        SET rank TO a list of zero repeated size times  
        SET size TO a list of one repeated size times  
    END FUNCTION  

    FUNCTION find (u)  
        IF element at position u of parent NOT EQUALS u  
            SET element at position u of parent TO the result of calling find with the element at position u of parent  
        END IF  
        RETURN element at position u of parent  
    END FUNCTION  

    FUNCTION union (u, v)  
        SET rootU TO the result of calling find with u  
        SET rootV TO the result of calling find with v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of rank GREATER THAN element at position rootV of rank  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of size BY element at position rootV of size  
            ELSE IF element at position rootU of rank LESS THAN element at position rootV of rank  
                SET element at position rootU of parent TO rootV  
                INCREMENT element at position rootV of size BY element at position rootU of size  
            ELSE  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of size BY element at position rootV of size  
                INCREMENT element at position rootU of rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings (words)  
        FUNCTION bitmask (word)  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one SHIFTED LEFT BY the numerical order of character MINUS the numerical order of the letter a in the alphabet  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a new mapping from key to list  
        SET n TO the LENGTH OF words  
        SET uf TO a new UnionFind initialized with n  

        FOR each index i and word IN enumeration of words  
            SET mask TO result of calling bitmask with word  
            APPEND i TO the list at key mask in word_map  
        END FOR  

        FOR each index i and word IN enumeration of words  
            SET mask TO result of calling bitmask with word  
            FOR j FROM zero TO twenty five  
                SET new_mask TO mask XOR one SHIFTED LEFT BY j  
                IF word_map CONTAINS key new_mask  
                    FOR each k IN the list at key new_mask in word_map  
                        CALL uf.union with i and k  
                    END FOR  
                END IF  

                IF result of mask AND one SHIFTED LEFT BY j EQUALS zero  
                    SET new_mask TO mask OR one SHIFTED LEFT BY j  
                    IF word_map CONTAINS key new_mask  
                        FOR each k IN the list at key new_mask in word_map  
                            CALL uf.union with i and k  
                        END FOR  
                    END IF  
                END IF  

                IF result of mask AND one SHIFTED LEFT BY j NOT EQUALS zero  
                    FOR k FROM zero TO twenty five  
                        IF result of mask AND one SHIFTED LEFT BY k EQUALS zero  
                            SET new_mask TO mask XOR one SHIFTED LEFT BY j OR one SHIFTED LEFT BY k  
                            IF word_map CONTAINS key new_mask  
                                FOR each l IN the list at key new_mask in word_map  
                                    CALL uf.union with i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a new mapping from key to integer with default zero  
        FOR i FROM zero TO n MINUS one  
            SET root TO uf.find with i  
            INCREMENT element at key root in root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value of root_sizes  
        SET num_groups TO the number of keys in root_sizes  

        RETURN list containing num_groups FOLLOWED BY max_group_size  
    END FUNCTION  
END CLASS
