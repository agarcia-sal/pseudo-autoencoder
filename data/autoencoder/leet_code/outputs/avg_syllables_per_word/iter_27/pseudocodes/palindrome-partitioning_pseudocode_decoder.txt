CLASS Solution  
    FUNCTION partition WITH PARAMETER s  
        FUNCTION is_palindrome WITH PARAMETER sub  
            RETURN sub EQUALS reverse of sub  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS start path  
            IF start EQUALS the LENGTH OF s  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF s PLUS one  
                IF is_palindrome of substring from position start TO position end MINUS one of s  
                    CALL backtrack WITH PARAMETERS end path PLUS list containing substring from position start TO position end MINUS one of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH PARAMETERS zero empty list  
        RETURN result  
    END FUNCTION  
END CLASS
