CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t THEN  
            RETURN the RESULT of calling isOneEditDistance with t and s as arguments  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff GREATER THAN one THEN  
            RETURN False  
        END IF  
        IF diff EQUALS zero THEN  
            SET count_differences TO zero  
            FOR each pair of characters c1 and c2 FROM s and t respectively IN PARALLEL  
                IF c1 NOT EQUALS c2 THEN  
                    INCREMENT count_differences BY one  
                END IF  
            END FOR  
            RETURN whether count_differences EQUALS one  
        ELSE  
            FOR index i FROM zero TO the LENGTH OF s MINUS one  
                IF the character at position i of s NOT EQUALS the character at position i of t THEN  
                    RETURN whether the substring of s from position i TO the END EQUALS the substring of t from position i PLUS one TO the END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
