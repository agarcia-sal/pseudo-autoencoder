CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list containing the groups of numbers zero and one zero and one and zero and negative one and zero and negative one and zero  
        SET min_heap TO a list containing a group of numbers zero zero zero representing the initial cost and position  
        SET visited TO a set containing the group of numbers zero zero representing the initial position  
        
        WHILE min_heap has elements  
            REMOVE the smallest element from min_heap and ASSIGN cost x y to the removed values  
            
            IF group of numbers x and y EQUALS group of numbers m MINUS one and n MINUS one THEN  
                RETURN cost  
            END IF  
            
            FOR each group dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF zero LESS THAN OR EQUAL TO nx AND nx LESS THAN m AND zero LESS THAN OR EQUAL TO ny AND ny LESS THAN n AND group of numbers nx and ny NOT IN visited THEN  
                    ADD group of numbers nx and ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of grid at position ny  
                    INSERT the group of numbers new_cost nx ny INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
