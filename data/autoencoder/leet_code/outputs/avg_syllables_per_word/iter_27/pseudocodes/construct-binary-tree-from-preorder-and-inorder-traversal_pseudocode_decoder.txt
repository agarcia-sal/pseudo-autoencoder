CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to the first element of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE the element from the left end of queue and ASSIGN it TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder EQUALS None OR the LENGTH OF preorder EQUALS zero OR inorder EQUALS None OR the LENGTH OF inorder EQUALS zero  
            RETURN None  
        END IF  
        SET root_val TO element at position zero of preorder  
        SET root TO a new TreeNode with value root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root.left TO the result of calling self.buildTree with parameters the elements of preorder from position one TO position root_index_inorder plus one AND the elements of inorder from position zero TO position root_index_inorder  
        SET root.right TO the result of calling self.buildTree with parameters the elements of preorder from position root_index_inorder plus one TO the end AND the elements of inorder from position root_index_inorder plus one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
