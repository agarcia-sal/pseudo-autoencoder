CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the collection of characters two five six nine  
        SET same_digits TO the collection of characters zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF for each digit IN num_str every digit is contained in the union of change_digits and same_digits THEN  
                IF there exists a digit IN num_str such that the digit is contained in change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number applied to i THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
