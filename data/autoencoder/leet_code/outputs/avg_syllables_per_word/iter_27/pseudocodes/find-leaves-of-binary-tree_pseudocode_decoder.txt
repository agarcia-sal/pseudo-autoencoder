CLASS TreeNode  
    FUNCTION __init__(self, val DEFAULTS TO zero, left DEFAULTS TO none, right DEFAULTS TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO TreeNode of element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none THEN  
            SET node.left TO TreeNode of element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none THEN  
            SET node.right TO TreeNode of element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree of p.left AND q.left AND is_same_tree of p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node EQUALS none THEN  
                RETURN negative one  
            END IF  
            SET left_height TO collectLeaves of node.left  
            SET right_height TO collectLeaves of node.right  
            SET current_height TO the GREATER OF left_height AND right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO the LENGTH OF result THEN  
                APPEND an empty list TO result  
            END IF  
            APPEND node.val TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO an empty list  
        CALL collectLeaves with root  
        RETURN result  
    END FUNCTION  
END CLASS
