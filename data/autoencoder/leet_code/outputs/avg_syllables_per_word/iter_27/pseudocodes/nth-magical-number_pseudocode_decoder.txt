CLASS Solution  
    FUNCTION nthMagicalNumber(n, a, b)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        FUNCTION gcd(x, y)  
            WHILE value of y IS NOT equal to zero  
                SET temporary TO value of y  
                SET y TO remainder of x DIVIDED BY y  
                SET x TO value of temporary  
            END WHILE  
            RETURN value of x  
        END FUNCTION  
        
        FUNCTION lcm(x, y)  
            RETURN x MULTIPLIED BY y DIVIDED BY gcd(x, y)  
        END FUNCTION  
        
        SET L TO lcm(a, b)  
        
        SET left TO one  
        SET right TO n MULTIPLIED BY the smaller of a AND b  
        
        WHILE left IS LESS THAN right  
            SET mid TO the integer part of left PLUS right DIVIDED BY two  
            SET count TO integer part of mid DIVIDED BY a PLUS integer part of mid DIVIDED BY b MINUS integer part of mid DIVIDED BY L  
            
            IF count IS LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN remainder of left DIVIDED BY MOD  
    END FUNCTION  
END CLASS
