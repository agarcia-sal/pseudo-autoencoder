CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count GREATER THAN x  
        END FUNCTION  
        RETURN the position of the first element IN the sequence of numbers FROM zero TO the length of nums MINUS one WHERE function f returns true  
    END FUNCTION  
END CLASS
