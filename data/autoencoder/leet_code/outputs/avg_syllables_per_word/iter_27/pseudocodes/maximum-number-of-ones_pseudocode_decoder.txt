CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET frequency_list TO a list of zero values with the length of the product of sideLength and sideLength  
        FOR variable i FROM zero TO the value of width MINUS one  
            FOR variable j FROM zero TO the value of height MINUS one  
                SET position TO the remainder of variable i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of variable j DIVIDED BY sideLength  
                INCREMENT the element at position index position of frequency_list BY one  
            END FOR  
        END FOR  
        SORT frequency_list IN descending order  
        SET total_sum TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total_sum BY the element at position index of frequency_list  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
