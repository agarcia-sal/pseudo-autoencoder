CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list of pairs of integers representing all possible knight movements in chess equal to minus two and minus one, minus two and plus one, minus one and minus two, minus one and plus two, plus one and minus two, plus one and plus two, plus two and minus one, plus two and plus one  
        SET dp TO two dimensional list of size n by n filled with zeroes  
        SET element at position row and element at position column of dp TO one  
        FOR each move_count FROM one TO k  
            SET new_dp TO two dimensional list of size n by n filled with zeroes  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r and element at position c of dp EQUALS zero  
                        CONTINUE to next iteration  
                    END IF  
                    FOR each pair dr and dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr and element at position nc of new_dp BY element at position r and element at position c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_element IN dp  
            FOR each cell_probability IN row_element  
                INCREMENT total_probability BY cell_probability  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
