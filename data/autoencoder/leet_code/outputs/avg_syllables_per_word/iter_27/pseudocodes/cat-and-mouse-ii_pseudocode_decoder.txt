CLASS Solution  
    FUNCTION canMouseWin(grid, catJump, mouseJump)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET cat_start TO zero  
        SET mouse_start TO zero  
        SET food TO zero  
        SET dirs TO a sequence of integer values minus one zero one zero minus one  
        SET g_mouse TO a list of empty lists with quantity m MULTIPLIED BY n  
        SET g_cat TO a list of empty lists with quantity m MULTIPLIED BY n  

        FOR each index i AND element row IN enumeration of grid  
            FOR each index j AND element c IN enumeration of row  
                IF c EQUALS the character hash  
                    CONTINUE to the next iteration  
                END IF  
                SET v TO i MULTIPLIED BY n PLUS j  
                IF c EQUALS the character C  
                    SET cat_start TO v  
                ELSE IF c EQUALS the character M  
                    SET mouse_start TO v  
                ELSE IF c EQUALS the character F  
                    SET food TO v  
                END IF  
                FOR each pair a AND b IN consecutive pairs of dirs  
                    FOR k FROM zero TO mouseJump PLUS one  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid AT position y EQUALS the character hash  
                            BREAK the current loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v of g_mouse  
                    END FOR  
                    FOR k FROM zero TO catJump PLUS one  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid AT position y EQUALS the character hash  
                            BREAK the current loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v of g_cat  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN the result of calling the FUNCTION calc with parameters g_mouse, g_cat, mouse_start, cat_start, and food EQUALS one  
    END FUNCTION  

    FUNCTION calc(g_mouse, g_cat, mouse_start, cat_start, hole)  
        FUNCTION get_prev_states(state)  
            SET m TO element at position zero of state  
            SET c TO element at position one of state  
            SET t TO element at position two of state  
            SET pt TO t BITWISE XOR one  
            SET pre TO an empty list  
            IF pt EQUALS one  
                FOR each pc IN element at position c of g_cat  
                    IF element at indices m, pc, one in ans EQUALS zero  
                        APPEND a tuple consisting of m, pc, and pt TO pre  
                    END IF  
                END FOR  
            ELSE  
                FOR each pm IN element at position m of g_mouse  
                    IF element at indices pm, c, zero in ans EQUALS zero  
                        APPEND a tuple consisting of pm, c, and zero TO pre  
                    END IF  
                END FOR  
            END IF  
            RETURN pre  
        END FUNCTION  

        SET n TO the LENGTH OF g_mouse  
        SET degree TO a three-dimensional list with dimensions n BY n BY two filled with zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET element at indices i, j, zero in degree TO the LENGTH OF element at position i of g_mouse  
                SET element at indices i, j, one in degree TO the LENGTH OF element at position j of g_cat  
            END FOR  
        END FOR  

        SET ans TO a three-dimensional list with dimensions n BY n BY two filled with zero  
        SET q TO an empty double-ended queue  
        FOR i FROM zero TO n MINUS one  
            SET element at indices hole, i, one in ans TO one  
            SET element at indices i, hole, zero in ans TO two  
            SET element at indices i, i, one in ans TO two  
            SET element at indices i, i, zero in ans TO two  
            APPEND a tuple consisting of hole, i, and one TO q  
            APPEND a tuple consisting of i, hole, and zero TO q  
            APPEND a tuple consisting of i, i, and zero TO q  
            APPEND a tuple consisting of i, i, and one TO q  
        END FOR  

        WHILE q is not empty  
            REMOVE and assign the leftmost element of q TO state  
            SET t TO element at indices element at position zero of state, element at position one of state, element at position two of state in ans  
            FOR each prev_state IN the result of calling get_prev_states with state as argument  
                SET pm TO element at position zero of prev_state  
                SET pc TO element at position one of prev_state  
                SET pt TO element at position two of prev_state  
                IF pt EQUALS t MINUS one  
                    SET element at indices pm, pc, pt in ans TO t  
                    APPEND prev_state TO q  
                ELSE  
                    DECREMENT element at indices pm, pc, pt in degree BY one  
                    IF element at indices pm, pc, pt in degree EQUALS zero  
                        SET element at indices pm, pc, pt in ans TO t  
                        APPEND prev_state TO q  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN element at indices mouse_start, cat_start, zero in ans  
    END FUNCTION  
END CLASS
