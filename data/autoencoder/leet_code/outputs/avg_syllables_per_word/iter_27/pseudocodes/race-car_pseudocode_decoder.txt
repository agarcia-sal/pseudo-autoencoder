CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new empty double ended queue initialized with a single element composed of zero position PLUS one speed PLUS zero steps  
        ASSIGN reference TO a new empty set initialized with a single tuple composed of zero position PLUS one speed to record visited states  
        
        WHILE the queue holds elements  
            REMOVE the element from the left end of the queue and ASSIGN its first component TO position ASSIGN its second component TO speed ASSIGN its third component TO steps  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple composed of next_position AND next_speed is NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple composed of next_position AND next_speed TO visited  
                APPEND the tuple composed of next_position AND next_speed AND steps PLUS one TO the right end of the queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple composed of position AND next_speed is NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple composed of position AND next_speed TO visited  
                APPEND the tuple composed of position AND next_speed AND steps PLUS one TO the right end of the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
