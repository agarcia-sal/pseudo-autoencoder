CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the result of maxChoosableInteger MULTIPLIED BY the sum of maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  
  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  
  
        SET memo TO an empty mapping  
  
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers EXISTS IN memo  
                RETURN the value mapped from the pair of current_total AND used_numbers in memo  
            END IF  
  
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND one LEFT SHIFTED BY number EQUALS zero  
                    IF the sum of current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT the result of can_win with arguments the sum of current_total PLUS number AND the bitwise OR of used_numbers AND one LEFT SHIFTED BY number  
                        SET the value mapped from the pair of current_total AND used_numbers in memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
  
            SET the value mapped from the pair of current_total AND used_numbers in memo TO False  
            RETURN False  
        END FUNCTION  
  
        RETURN the result of calling can_win with zero AND zero  
    END FUNCTION  
END CLASS
