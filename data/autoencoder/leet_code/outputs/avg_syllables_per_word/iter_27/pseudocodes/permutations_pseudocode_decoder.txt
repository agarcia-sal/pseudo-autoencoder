CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO the concatenation of path PLUS the element at position i of remaining  
                SET new_remaining TO the concatenation of the elements from position zero TO position i MINUS one of remaining PLUS the elements from position i PLUS one TO the end of remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty list  
        CALL backtrack WITH empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
