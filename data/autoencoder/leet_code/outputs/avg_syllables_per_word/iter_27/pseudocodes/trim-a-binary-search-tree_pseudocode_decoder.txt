CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the property val OF self TO val  
        SET the property left OF self TO left  
        SET the property right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values collection DOES NOT CONTAIN any element  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode object initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty deque collection  
    APPEND root TO queue  
    WHILE the queue collection IS NOT EMPTY  
        SET node TO the first element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET the left property OF node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the left property OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET the right property OF node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the right property OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS no value AND q IS no value  
        RETURN truth value true  
    ELSE IF p IS no value OR q IS no value  
        RETURN truth value false  
    ELSE IF the val property OF p IS NOT EQUALS the val property OF q  
        RETURN truth value false  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left property OF p and the left property OF q AND the result of calling is_same_tree with the right property OF p and the right property OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST with parameters self root low high  
        IF root IS no value  
            RETURN no value  
        END IF  
        IF the val property OF root IS LESS THAN low  
            RETURN the result of calling trimBST OF self with parameters the right property OF root low high  
        END IF  
        IF the val property OF root IS GREATER THAN high  
            RETURN the result of calling trimBST OF self with parameters the left property OF root low high  
        END IF  
        SET the left property OF root TO the result of calling trimBST OF self with parameters the left property OF root low high  
        SET the right property OF root TO the result of calling trimBST OF self with parameters the right property OF root low high  
        RETURN root  
    END FUNCTION  
END CLASS
