CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO a new mapping from keys to lists  
        FOR each pair of nodes u AND v IN edges  
            APPEND v TO the list at key u IN graph  
            APPEND u TO the list at key v IN graph  
        END FOR  
        
        SET queue TO a new double ended queue containing the tuple of current node one current probability one point zero and current time zero  
        SET visited TO a new set containing element one  
        
        WHILE the length of queue is greater than zero  
            REMOVE the element at the front of queue AND ASSIGN the first element to current_node the second element to current_probability AND the third element to current_time  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the length of the list at key current_node IN graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
            END IF  
            
            SET possible_moves TO the length of the list at key current_node IN graph MINUS one IF current_node NOT EQUALS one OTHERWISE zero  
            
            FOR each neighbor IN the list at key current_node IN graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
