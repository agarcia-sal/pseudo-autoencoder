CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the LENGTH OF stack greater than zero AND element at the last position of stack GREATER THAN num  
                REMOVE the element at the last position FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
