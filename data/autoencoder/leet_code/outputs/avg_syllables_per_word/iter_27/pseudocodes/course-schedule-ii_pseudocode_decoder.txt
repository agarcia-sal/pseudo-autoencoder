CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping from each course FROM zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zero values with length numCourses  
        
        FOR each pair destination and source IN prerequisites  
            APPEND destination TO the list at key source IN adj_list  
            INCREMENT the element at position destination OF in_degrees BY one  
        END FOR  
        
        SET queue TO a double-ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at position course OF in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE the queue IS NOT empty  
            SET current TO the element removed from the front of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current IN adj_list  
                DECREMENT the element at position neighbor OF in_degrees BY one  
                IF the element at position neighbor OF in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
