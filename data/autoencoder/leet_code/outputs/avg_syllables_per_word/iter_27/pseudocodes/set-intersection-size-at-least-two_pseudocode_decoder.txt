CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL sort intervals BY the value of element at position two of each element AND BY the negation of the value of element at position one of each element IN descending order  
        SET first TO the value of element at position two of element at position one of intervals MINUS one  
        SET second TO the value of element at position two of element at position one of intervals  
        SET count TO two  
        FOR each element AFTER the first IN intervals  
            SET start TO the value of element at position one of element  
            SET end TO the value of element at position two of element  
            IF start IS GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
