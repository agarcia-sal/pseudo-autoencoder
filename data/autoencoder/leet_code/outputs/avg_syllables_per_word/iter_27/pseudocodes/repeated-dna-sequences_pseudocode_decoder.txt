CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a default dictionary with default value zero  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO position index PLUS ten MINUS one of s  
            INCREMENT the value at key sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence_key AND sequence_value IN the items of sequence_count  
            IF sequence_value GREATER THAN one THEN  
                APPEND sequence_key TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
