CLASS Solution  
    FUNCTION numWays(words  
                     target)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF element at position zero of words  
        SET m TO the LENGTH OF target  
        
        SET freq TO a list containing n dictionaries each mapping characters to integers initialized to zero  
        FOR each word IN words  
            FOR each position j AND character char IN word  
                INCREMENT the value at character char of dictionary at position j in freq BY one  
            END FOR  
        END FOR  
        
        SET dp TO a list containing m PLUS one elements each being a list containing n PLUS one integers initialized to zero  
        
        FOR each integer j FROM zero TO n  
            SET element at position zero of dp at position j TO one  
        END FOR  
        
        FOR each integer i FROM one TO m  
            FOR each integer j FROM one TO n  
                SET element at position j of dp at position i TO the element at position j MINUS one of dp at position i  
                IF the value at character at position i MINUS one of target in dictionary at position j MINUS one of freq IS GREATER THAN zero  
                    SET element at position j of dp at position i TO the element at position j of dp at position i PLUS the element at position j MINUS one of dp at position i MINUS one MULTIPLIED BY the value at character at position i MINUS one of target in dictionary at position j MINUS one of freq  
                    SET element at position j of dp at position i TO the element at position j of dp at position i MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of dp at position m  
    END FUNCTION  
END CLASS
