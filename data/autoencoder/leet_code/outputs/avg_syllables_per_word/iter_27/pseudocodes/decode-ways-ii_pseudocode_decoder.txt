CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO one billion PLUS seven  
        CREATE a list dp WITH length EQUAL TO the length OF s PLUS one FILLED WITH zeroes  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character star  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character star AND char2 EQUALS the character star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character star  
                IF character at char2 IS LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character star  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer REPRESENTATION OF the concatenation OF char1 AND char2  
                IF num IS GREATER THAN OR EQUAL TO ten AND num IS LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR index FROM one TO the length OF s  
            SET dp AT position index TO the remainder UPON division BY MOD OF the sum OF dp AT position index MINUS one MULTIPLIED BY the result OF ways_to_decode_single WITH argument element at position index MINUS one of s PLUS dp AT position index  
        END FOR  
        
        FOR index FROM one TO the length OF s  
            IF index IS GREATER THAN one  
                SET dp AT position index TO the remainder UPON division BY MOD OF the sum OF dp AT position index AND dp AT position index MINUS two MULTIPLIED BY the result OF ways_to_decode_pair WITH arguments element at position index MINUS two of s AND element at position index MINUS one of s  
            END IF  
        END FOR  
        
        RETURN element at the last position OF dp  
    END FUNCTION  
END CLASS
