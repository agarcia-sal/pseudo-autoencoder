CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF i GREATER THAN farthest  
                RETURN False  
            END IF  
            SET sum_index_and_value TO i PLUS element at position i of nums  
            IF sum_index_and_value GREATER THAN farthest  
                SET farthest TO sum_index_and_value  
            END IF  
        END FOR  
        IF farthest GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
            RETURN True  
        ELSE  
            RETURN False  
        END IF  
    END FUNCTION  
END CLASS
