CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_x TO the absolute value of x  
        SET string_representation TO the string representation of absolute_x  
        SET reversed_string TO the string representation from last character TO first character of string_representation  
        SET reversed_x TO the integer conversion of reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO THE POWER OF thirty one OR result GREATER THAN two RAISED TO THE POWER OF thirty one MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
