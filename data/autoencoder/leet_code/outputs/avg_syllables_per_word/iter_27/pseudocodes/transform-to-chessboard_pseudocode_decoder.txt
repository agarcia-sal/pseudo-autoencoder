CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i of board AT position j XOR the element at position i of board AT position zero XOR the element at position zero of board AT position j XOR the element at position zero of board AT position zero EQUALS true  
                    RETURN minus one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the element at position i of board AT position zero  
            INCREMENT colSum BY the element at position zero of board AT position i  
            INCREMENT rowSwap BY the result of the comparison the element at position i of board AT position zero EQUALS the remainder of i DIVIDED BY two  
            INCREMENT colSwap BY the result of the comparison the element at position zero of board AT position i EQUALS the remainder of i DIVIDED BY two  
        END FOR  
        IF the quotient of N DIVIDED BY two GREATER THAN rowSum OR rowSum GREATER THAN the quotient of N PLUS one DIVIDED BY two  
            RETURN minus one  
        END IF  
        IF the quotient of N DIVIDED BY two GREATER THAN colSum OR colSum GREATER THAN the quotient of N PLUS one DIVIDED BY two  
            RETURN minus one  
        END IF  
        IF the remainder of N DIVIDED BY two EQUALS one  
            IF the remainder of rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum of colSwap AND N MINUS colSwap  
        END IF  
        RETURN the quotient of the sum of rowSwap AND colSwap DIVIDED BY two  
    END FUNCTION  
END CLASS
