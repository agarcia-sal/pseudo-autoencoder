CLASS Solution  
    FUNCTION countVowelPermutation WITH PARAMETER n  
        SET MODULO TO one billion PLUS seven  
        SET count_a TO one  
        SET count_e TO one  
        SET count_i TO one  
        SET count_o TO one  
        SET count_u TO one  
        FOR each number FROM two TO n  
            SET next_a TO count_e PLUS count_i PLUS count_u  
            SET next_e TO count_a PLUS count_i  
            SET next_i TO count_e PLUS count_o  
            SET next_o TO count_i  
            SET next_u TO count_i PLUS count_o  
            SET count_a TO next_a MODULO MODULO  
            SET count_e TO next_e MODULO MODULO  
            SET count_i TO next_i MODULO MODULO  
            SET count_o TO next_o MODULO MODULO  
            SET count_u TO next_u MODULO MODULO  
        END FOR  
        SET total_count TO count_a PLUS count_e PLUS count_i PLUS count_o PLUS count_u  
        RETURN total_count MODULO MODULO  
    END FUNCTION  
END CLASS
