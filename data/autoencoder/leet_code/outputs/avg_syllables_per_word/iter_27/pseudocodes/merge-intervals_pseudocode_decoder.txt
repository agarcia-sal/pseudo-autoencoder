CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals IN ORDER OF element at position zero of each interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR element at position one of element at the last position of merged IS LESS THAN element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one of element at the last position of merged TO the maximum of element at position one of element at the last position of merged AND element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
