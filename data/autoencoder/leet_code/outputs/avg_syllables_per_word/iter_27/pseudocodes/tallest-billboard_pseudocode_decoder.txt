CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO dictionary WITH key zero AND value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d AND value y IN dp  
                SET candidate_key_one TO d PLUS rod  
                IF candidate_key_one IS NOT a key IN new_dp OR new_dp at candidate_key_one IS LESS THAN y  
                    SET new_dp at candidate_key_one TO y  
                END IF  
                SET absolute_difference TO d MINUS rod  
                IF absolute_difference IS LESS THAN zero  
                    SET absolute_difference TO rod MINUS d  
                END IF  
                SET candidate_key_two TO absolute_difference  
                SET candidate_value_two TO y PLUS the SMALLER OF d AND rod  
                IF candidate_key_two IS NOT a key IN new_dp OR new_dp at candidate_key_two IS LESS THAN candidate_value_two  
                    SET new_dp at candidate_key_two TO candidate_value_two  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN dp at zero  
    END FUNCTION  
END CLASS
