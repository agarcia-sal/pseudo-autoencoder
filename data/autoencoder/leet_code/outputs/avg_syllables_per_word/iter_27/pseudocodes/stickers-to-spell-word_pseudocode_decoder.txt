CLASS Solution  
    FUNCTION minStickers(stickers LIST OF STRINGS, target STRING) RETURNS INTEGER  
        SET stickers TO a list where each element is a count mapping of characters in each string of stickers  
        SORT stickers IN ORDER OF the negative value of the maximum frequency in each count mapping  
        SET queue TO a double-ended queue containing a tuple of target and zero  
        SET memo TO a mapping with key target and value zero  
        WHILE the queue is not empty  
            SET current_target AND used_stickers TO the first element removed from the queue  
            FOR each sticker IN stickers  
                IF the element at position zero of current_target IS NOT PRESENT in the keys of sticker  
                    CONTINUE to next iteration  
                END IF  
                SET remaining TO a count mapping of characters in current_target  
                DECREASE counts in remaining by the counts in sticker   
                SET new_target TO a string formed by concatenating the sorted list of elements in remaining repeated according to their counts  
                IF new_target IS an empty string  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target IS NOT A KEY in memo  
                    SET the value of memo at key new_target TO used_stickers PLUS one  
                    APPEND a tuple of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
