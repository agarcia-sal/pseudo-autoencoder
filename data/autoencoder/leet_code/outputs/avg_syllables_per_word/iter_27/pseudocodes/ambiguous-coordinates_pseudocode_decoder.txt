CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING from position one TO position d of frag  
                SET right TO the SUBSTRING from position d PLUS one TO the LENGTH OF frag of frag  
                IF NOT left STARTS WITH zero OR left EQUALS zero AND NOT right ENDS WITH zero  
                    IF d NOT EQUALS n  
                        YIELD left PLUS a decimal point PLUS right  
                    ELSE  
                        YIELD left PLUS right  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET s TO the SUBSTRING from position two TO the POSITION BEFORE the last position of s  
        
        RETURN a list of strings where each string is an opening parenthesis PLUS a PLUS a comma PLUS a space PLUS b PLUS a closing parenthesis  
        FOR i FROM one TO the LENGTH OF s MINUS one  
        FOR each a, b IN the CARTESIAN PRODUCT of make(the SUBSTRING from position one TO position i of s) AND make(the SUBSTRING from position i PLUS one TO the LENGTH OF s of s)  
    END FUNCTION  
END CLASS
