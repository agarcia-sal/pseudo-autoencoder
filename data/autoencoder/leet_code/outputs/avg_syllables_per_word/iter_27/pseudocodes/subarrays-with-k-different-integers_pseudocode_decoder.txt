CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO a default dictionary that maps each element to zero  
            SET left TO zero  
            SET result TO zero  
            FOR each right FROM zero TO the length of nums MINUS one  
                INCREMENT the count of the element at position right of nums BY one  
                WHILE the number of keys in count IS GREATER THAN t  
                    DECREMENT the count of the element at position left of nums BY one  
                    IF the count of the element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  

        RETURN the value returned by atMostKDistinct with argument k MINUS the value returned by atMostKDistinct with argument k MINUS one  
    END FUNCTION  
END CLASS
