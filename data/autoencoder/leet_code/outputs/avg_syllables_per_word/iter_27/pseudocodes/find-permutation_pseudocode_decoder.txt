CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end TO loop FROM zero TO the LENGTH OF s MINUS one PLUS one  
            IF end EQUALS the LENGTH OF s OR element at position end of s EQUALS the uppercase letter I  
                SET segment_to_reverse TO the elements from position start TO position end PLUS one of result  
                SET reversed_segment TO the elements of segment_to_reverse in reverse order  
                REPLACE the elements from position start TO position end PLUS one of result WITH reversed_segment  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
