CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant AS an empty double-ended queue  
        INITIALIZE dire AS an empty double-ended queue  
        FOR each index AND senator IN the enumeration of senate  
            IF senator EQUALS the character representing Radiant party  
                APPEND index TO radiant  
            ELSE  
                APPEND index TO dire  
            END IF  
        END FOR  
        WHILE radiant is not empty AND dire is not empty  
            REMOVE the element at the front of radiant AND ASSIGN to variable r_index  
            REMOVE the element at the front of dire AND ASSIGN to variable d_index  
            IF r_index LESS THAN d_index  
                APPEND the sum of r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND the sum of d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  
        IF radiant is not empty  
            RETURN the string representing Radiant party  
        ELSE  
            RETURN the string representing Dire party  
        END IF  
    END FUNCTION  
END CLASS
