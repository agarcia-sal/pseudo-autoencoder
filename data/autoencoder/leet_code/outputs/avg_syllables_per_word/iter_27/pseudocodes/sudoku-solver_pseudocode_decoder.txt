CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF element at position i of element at position row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position row of element at position col of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the integer division of row BY three  
            SET start_col TO three MULTIPLIED BY the integer division of col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_row PLUS i of element at position start_col PLUS j of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  

        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row of element at position col of board EQUALS the string representation of the empty cell  
                        FOR num IN the string representations of the numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position row of element at position col of board TO num  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position row of element at position col of board TO the string representation of the empty cell  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  

        CALL solve(board)  
    END FUNCTION  
END CLASS
