CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string element at position index of strs reversed  
            IF reversed_str GREATER THAN the string element at position index of strs  
                SET the string element at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs and strs  
        
        SET max_string TO the empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the collection of False and True  
                IF is_reversed EQUALS True  
                    SET current TO the string element at position index of strs reversed  
                ELSE  
                    SET current TO the string element at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of the string elements from position index plus one TO position index plus the LENGTH OF strs minus one of doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring from position cut_position TO the end of current the remaining and the substring from position zero TO position cut_position MINUS one of current  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
