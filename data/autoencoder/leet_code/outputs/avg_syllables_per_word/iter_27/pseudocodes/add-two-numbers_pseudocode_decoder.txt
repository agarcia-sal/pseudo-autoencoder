CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH THE element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the end  
        SET node TO A NEW ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN TRUE  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN FALSE  
    END IF  
    RETURN the expression that element val of p1 EQUALS element val of p2 AND the recursive call to is_same_list with the element next of p1 AND the element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO A NEW ListNode WITH zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry NOT EQUALS zero  
            IF l1 IS NOT none THEN  
                SET val1 TO element val of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT none THEN  
                SET val2 TO element val of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO A NEW ListNode WITH total MODULO ten  
            SET current TO element next of current  
            IF l1 IS NOT none THEN  
                SET l1 TO element next of l1  
            END IF  
            IF l2 IS NOT none THEN  
                SET l2 TO element next of l2  
            END IF  
        END WHILE  
        RETURN element next of dummy_head  
    END FUNCTION  
END CLASS
