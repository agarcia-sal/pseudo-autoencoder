CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN False  
        END IF  
        SET adj_list TO an empty mapping from integer to list  
        FOR index FROM zero TO n MINUS one  
            SET element at position index of adj_list TO empty list  
        END FOR  
        FOR each pair a AND b IN edges  
            APPEND b TO the list at position a of adj_list  
            APPEND a TO the list at position b of adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at position node of adj_list  
                CALL dfs with neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs with zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
