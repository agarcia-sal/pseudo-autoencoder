CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with parameters the left attribute of p and the left attribute of q AND the result of calling is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode WITH PARAMETERS self root key  
        IF root IS None  
            RETURN None  
        END IF  
        IF the val attribute of root IS GREATER THAN key  
            SET the left attribute of root TO the result of calling deleteNode with parameters self the left attribute of root and key  
            RETURN root  
        END IF  
        IF the val attribute of root IS LESS THAN key  
            SET the right attribute of root TO the result of calling deleteNode with parameters self the right attribute of root and key  
            RETURN root  
        END IF  
        IF the left attribute of root IS None  
            RETURN the right attribute of root  
        END IF  
        IF the right attribute of root IS None  
            RETURN the left attribute of root  
        END IF  
        SET node TO the right attribute of root  
        WHILE the left attribute of node IS NOT None  
            SET node TO the left attribute of node  
        END WHILE  
        SET the left attribute of node TO the left attribute of root  
        SET root TO the right attribute of root  
        RETURN root  
    END FUNCTION  
END CLASS
