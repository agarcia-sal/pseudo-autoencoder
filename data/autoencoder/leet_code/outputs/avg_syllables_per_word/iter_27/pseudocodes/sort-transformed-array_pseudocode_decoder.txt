CLASS Solution  
    FUNCTION sortTransformedArray(nums a b c)  
        FUNCTION quadratic(x)  
            RETURN the value of a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        SET result TO a list of zeros of LENGTH n  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a GREATER THAN OR EQUAL TO zero THEN  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right DO  
                IF the value of quadratic of element at position left of nums GREATER THAN the value of quadratic of element at position right of nums THEN  
                    SET element at position index of result TO the value of quadratic of element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO the value of quadratic of element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right DO  
                IF the value of quadratic of element at position left of nums LESS THAN the value of quadratic of element at position right of nums THEN  
                    SET element at position index of result TO the value of quadratic of element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO the value of quadratic of element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
