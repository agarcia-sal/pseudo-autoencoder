CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of tuples each containing zero and one representing right movement zero and one representing down movement zero and negative one representing left movement negative one and zero representing up movement each paired with corresponding character r d l u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET start TO the tuple conversion of ball  
        SET hole TO the tuple conversion of hole  
        SET pq TO list containing the tuple of zero the empty string and start  
        SET visited TO empty set  
        
        WHILE pq holds  
            SET dist path and position unpacked from heappop operation on pq  
            SET x and y to the elements of position  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF position IS IN visited  
                CONTINUE  
            END IF  
            
            ADD position TO visited  
            
            FOR each dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of maze the element at position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS first element of hole AND new_y EQUALS second element of hole  
                        BREAK  
                    END IF  
                END WHILE  
                
                IF the tuple of new_x and new_y IS NOT IN visited  
                    CALL heappush on pq with the tuple of dist PLUS count path concatenated with direction and the tuple of new_x and new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
