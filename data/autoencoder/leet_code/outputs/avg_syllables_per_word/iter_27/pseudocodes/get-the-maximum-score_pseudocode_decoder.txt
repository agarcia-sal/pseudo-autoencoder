CLASS Solution  
    FUNCTION maxSum(nums1 nums1_list OF integers nums2 nums2_list OF integers) RETURNS integer  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO one billion PLUS one  
        
        WHILE i LESS THAN the LENGTH OF nums1_list AND j LESS THAN the LENGTH OF nums2_list  
            IF element at position i of nums1_list LESS THAN element at position j of nums2_list  
                INCREMENT sum1 BY element at position i of nums1_list  
                INCREMENT i BY one  
            ELSE IF element at position i of nums1_list GREATER THAN element at position j of nums2_list  
                INCREMENT sum2 BY element at position j of nums2_list  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO the maximum of sum1 AND sum2 PLUS element at position i of nums1_list  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i LESS THAN the LENGTH OF nums1_list  
            INCREMENT sum1 BY element at position i of nums1_list  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j LESS THAN the LENGTH OF nums2_list  
            INCREMENT sum2 BY element at position j of nums2_list  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN the maximum of sum1 AND sum2 MODULO MOD  
    END FUNCTION  
END CLASS
