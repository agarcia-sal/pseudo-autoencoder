CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO list of pairs of element at position index of efficiency WITH element at position index of speed FOR each index from zero TO n minus one  
        SORT engineers in descending order based on the first element of each pair  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff spd IN engineers  
            ADD spd TO speed_heap maintaining heap property  
            INCREMENT speed_sum BY spd  
            IF the length of speed_heap GREATER THAN k  
                REMOVE smallest element FROM speed_heap maintaining heap property  
                DECREMENT speed_sum BY the removed element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN the remainder of max_performance DIVIDED BY one billion PLUS seven  
    END FUNCTION  
END CLASS
