CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the result of invoking subsets ON the elements of nums from position two TO the end  
        SET subsets_with_first TO a new list OF lists WHERE each element IS the element at position one of nums APPENDED TO each subset IN subsets_without_first  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
