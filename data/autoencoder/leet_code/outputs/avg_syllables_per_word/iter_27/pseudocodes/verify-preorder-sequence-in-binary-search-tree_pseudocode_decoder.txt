CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            WHILE stack IS NOT empty AND value GREATER THAN element at position LENGTH OF stack MINUS one of stack  
                SET lower_bound TO element removed from the end of stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
