CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty set  
        FOR length FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length INCLUSIVE  
                SET a TO the substring from position start TO position start PLUS length MINUS one OF text  
                SET b TO the substring from position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one OF text  
                IF a EQUALS b  
                    ADD the concatenation of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the SIZE OF substrings  
    END FUNCTION  
END CLASS
