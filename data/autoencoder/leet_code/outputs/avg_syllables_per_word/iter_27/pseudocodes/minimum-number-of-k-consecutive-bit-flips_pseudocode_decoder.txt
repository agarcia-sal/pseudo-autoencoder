CLASS Solution  
    FUNCTION minKBitFlips(nums k)  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a LIST OF zero REPEATED n PLUS one TIMES  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET flip_state TO flip_state NOT EQUALS element at position i of flipped  
            
            IF element at position i of nums EQUALS flip_state THEN  
                IF i PLUS k GREATER THAN n THEN  
                    RETURN negative one  
                END IF  
                SET flip_state TO flip_state NOT EQUALS one  
                SET element at position i PLUS k of flipped TO element at position i PLUS k of flipped NOT EQUALS one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
