CLASS Solution  
    FUNCTION minimumDeleteSum(s1 s1 AS string s2 s2 AS string) RETURNS integer  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        CREATE dp AS two dimensional list of zero with dimensions m PLUS one AND n PLUS one  
        
        FOR j FROM one TO n  
            SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position zero of element at position i of dp TO element at position zero of element at position i MINUS one of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s1 EQUALS element at position j MINUS one of s2 THEN  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET left_delete_cost TO element at position j of element at position i MINUS one of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1  
                    SET top_delete_cost TO element at position j MINUS one of element at position i of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
                    SET element at position j of element at position i of dp TO the smaller value BETWEEN left_delete_cost AND top_delete_cost  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
