CLASS TreeNode  
    FUNCTION __init__(self, val DEFAULTS TO zero, left DEFAULTS TO none, right DEFAULTS TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO first element removed from queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF value of p IS NOT EQUAL TO value of q THEN  
        RETURN false  
    ELSE  
        RETURN result of is_same_tree with parameters value of left child of p AND value of left child of q AND result of is_same_tree with parameters value of right child of p AND value of right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self, root)  
        IF root IS none THEN  
            RETURN none  
        END IF  
        SET root.left TO result of pruneTree called with parameter value of left child of root  
        SET root.right TO result of pruneTree called with parameter value of right child of root  
        IF value of root EQUALS zero AND root.left IS none AND root.right IS none THEN  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
