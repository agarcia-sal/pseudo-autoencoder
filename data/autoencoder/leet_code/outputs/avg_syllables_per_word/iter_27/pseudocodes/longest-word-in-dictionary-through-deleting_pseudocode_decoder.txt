CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET it TO an iterator over the elements of s  
            SET all_characters_in_it TO true if for each character in word the character appears in the iterator it in order  
            RETURN all_characters_in_it  
        END FUNCTION  
        
        SORT dictionary BY length of element in descending order AND BY lexicographical order in ascending order  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN an empty string  
    END FUNCTION  
END CLASS
