CLASS Solution  
    FUNCTION sequenceReconstruction(nums PARAMETER, sequences PARAMETER)  
        SET n TO the LENGTH OF nums  
        SET graph TO a mapping from keys to lists with default empty list  
        SET indegree TO a list of zeros with the LENGTH equal to n PLUS one  
        
        FOR each seq IN sequences  
            FOR each i FROM zero TO the LENGTH OF seq MINUS one MINUS one  
                SET u TO element at position i of seq  
                SET v TO element at position i PLUS one of seq  
                APPEND v TO the list mapped by u in graph  
                INCREMENT element at position v of indegree BY one  
            END FOR  
        END FOR  
        
        SET queue TO a double-ended queue containing elements i for each i FROM one TO n IF element at position i of indegree EQUALS zero  
        
        IF the LENGTH OF queue NOT EQUALS one  
            RETURN False  
        END IF  
        
        SET index TO zero  
        
        WHILE the LENGTH OF queue GREATER THAN zero  
            IF the LENGTH OF queue NOT EQUALS one  
                RETURN False  
            END IF  
            
            REMOVE element from the front of queue and ASSIGN it TO node  
            
            IF node NOT EQUALS element at position index of nums  
                RETURN False  
            END IF  
            
            INCREMENT index BY one  
            
            FOR each neighbor IN the list mapped by node in graph  
                DECREMENT element at position neighbor of indegree BY one  
                IF element at position neighbor of indegree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN index EQUALS n  
    END FUNCTION  
END CLASS
