CLASS Solution  
    FUNCTION multiply(mat1 matrix OF list OF integers mat2 matrix OF list OF integers) RETURNS matrix OF list OF integers  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position zero of mat2  
        SET result TO a new matrix of size m BY n FILLED WITH zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR p FROM zero TO k MINUS one  
                    INCREMENT the element at position j of the element at position i of result BY the element at position p of the element at position i of mat1 MULTIPLIED BY the element at position j of the element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
