CLASS TreeNode  
    FUNCTION __init__(self val ZERO left NONE right NONE)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values list)  
    IF values IS EMPTY  
        RETURN NONE  
    END IF  
    SET root TO A new TreeNode OBJECT CREATED WITH the element at position ONE of values  
    SET i TO ONE  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NONE  
            SET node.left TO A new TreeNode OBJECT CREATED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY ONE  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NONE  
            SET node.right TO A new TreeNode OBJECT CREATED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY ONE  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p IS NONE AND q IS NONE  
        RETURN TRUE  
    ELSE IF p IS NONE OR q IS NONE  
        RETURN FALSE  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN FALSE  
    ELSE  
        RETURN is_same_tree(p.left q.left) AND is_same_tree(p.right q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self nums List of integers)  
        IF nums IS EMPTY  
            RETURN NONE  
        END IF  
        SET max_val TO THE MAXIMUM VALUE IN nums  
        SET max_index TO THE POSITION OF max_val IN nums  
        SET root TO A new TreeNode OBJECT CREATED WITH max_val  
        SET root.left TO THE RESULT OF CALLING self.constructMaximumBinaryTree WITH THE SUBLIST OF nums FROM THE START TO THE POSITION BEFORE max_index  
        SET root.right TO THE RESULT OF CALLING self.constructMaximumBinaryTree WITH THE SUBLIST OF nums FROM THE POSITION AFTER max_index TO THE END  
        RETURN root  
    END FUNCTION  
END CLASS
