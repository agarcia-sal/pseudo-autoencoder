CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty list  
        RETURN no value  
    END IF  
    SET head TO a new instance of ListNode initialized with element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the last element of values  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND call is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs with parameter head  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        
        WHILE the next attribute of current DOES NOT EQUAL no value AND the next attribute of the next attribute of current DOES NOT EQUAL no value  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
