CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the Unicode code point of c MINUS the Unicode code point of lowercase a PLUS one  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        SET current_hash TO zero  
        SET p_pow TO one  
        
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO the remainder upon division of the sum of current_hash AND the product of val of the element at position i of s AND p_pow BY modulo  
            IF i LESS THAN n MINUS one  
                SET p_pow TO the remainder upon division of the product of p_pow AND power BY modulo  
            END IF  
        END FOR  
        
        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN the substring from position start_index TO the position one less than the sum of start_index AND k of s  
        END IF  
        
        SET p_pow_k_minus_1 TO p_pow  
        FOR i FROM n MINUS k MINUS one DOWN TO zero  
            SET current_hash TO the remainder upon division of the difference of current_hash AND the product of val of the element at position sum of i AND k of s AND p_pow_k_minus_1 BY modulo  
            SET current_hash TO the remainder upon division of the sum of the product of current_hash AND power AND val of the element at position i of s BY modulo  
            IF current_hash EQUALS hashValue  
                SET start_index TO i  
            END IF  
        END FOR  
        
        RETURN the substring from position start_index TO the position one less than the sum of start_index AND k of s  
    END FUNCTION  
END CLASS
