CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        SET directions TO list containing elements zero and one zero and one zero and negative one negative one and zero  
        SET min_heap TO list containing element zero zero zero representing the cost and the x and y positions  
        SET visited TO set containing the tuple of zero and zero representing the visited positions  
        
        WHILE min_heap is not empty  
            REMOVE the element with the smallest cost from min_heap and ASSIGN the values TO cost x and y  
            
            IF the tuple of x and y EQUALS the tuple of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each dx and dy in directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the tuple of nx and ny is NOT IN visited  
                    ADD the tuple of nx and ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid at position ny  
                    INSERT the tuple of new_cost nx and ny INTO min_heap maintaining the heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
