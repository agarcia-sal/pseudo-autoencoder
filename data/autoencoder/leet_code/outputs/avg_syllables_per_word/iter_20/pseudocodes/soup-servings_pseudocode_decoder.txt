CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one point zero  
        END IF  
        
        SET n TO n PLUS twenty four DIVIDED BY twenty five with integer division rounding down  
        
        SET memo TO an empty mapping  
        
        FUNCTION dp(a, b)  
            IF the pair of a and b EXISTS IN memo  
                RETURN the element at the pair of a and b IN memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            
            SET prob TO zero point two five MULTIPLIED BY the SUM of dp with a MINUS four AND b PLUS dp with a MINUS three AND b MINUS one PLUS dp with a MINUS two AND b MINUS two PLUS dp with a MINUS one AND b MINUS three  
            
            ASSIGN the pair of a and b IN memo TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp with n AND n  
    END FUNCTION  
END CLASS
