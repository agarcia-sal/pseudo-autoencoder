CLASS Solution  
    FUNCTION numTrees WITH PARAMETER n  
        SET dp TO a list containing zero repeated the quantity of n PLUS one times  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i  
                SET dp at position i TO dp at position i PLUS the element at position j MINUS one of dp MULTIPLIED BY the element at position i MINUS j of dp  
            END FOR  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
