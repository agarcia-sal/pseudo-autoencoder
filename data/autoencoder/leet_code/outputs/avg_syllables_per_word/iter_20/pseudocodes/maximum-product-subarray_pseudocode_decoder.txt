CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        SET max_product TO element at position zero of nums  
        SET min_product TO element at position zero of nums  
        SET result TO element at position zero of nums  
        FOR each num IN the elements of nums starting from position one TO the last position  
            IF num LESS THAN zero  
                SET temporary_variable TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary_variable  
            END IF  
            SET multiplication_max TO max_product MULTIPLIED BY num  
            IF num GREATER THAN multiplication_max  
                SET max_product TO num  
            ELSE  
                SET max_product TO multiplication_max  
            END IF  
            SET multiplication_min TO min_product MULTIPLIED BY num  
            IF num LESS THAN multiplication_min  
                SET min_product TO num  
            ELSE  
                SET min_product TO multiplication_min  
            END IF  
            IF result LESS THAN max_product  
                SET result TO max_product  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
