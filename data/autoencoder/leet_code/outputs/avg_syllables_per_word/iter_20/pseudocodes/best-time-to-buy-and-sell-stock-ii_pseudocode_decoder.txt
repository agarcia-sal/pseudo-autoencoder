CLASS Solution  
    FUNCTION maxProfit(prices)  
        SET max_profit TO zero  
        FOR index FROM one TO the LENGTH OF prices MINUS one  
            IF element at position index of prices IS GREATER THAN element at position index MINUS one of prices  
                INCREMENT max_profit BY element at position index of prices MINUS element at position index MINUS one of prices  
            END IF  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
