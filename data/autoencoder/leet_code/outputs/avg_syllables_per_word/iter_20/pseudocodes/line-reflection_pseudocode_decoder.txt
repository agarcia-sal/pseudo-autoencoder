CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO CONVERT each element of points INTO tuple AND COLLECT INTO a set  
        SET min_x TO the minimum value of the element at position zero of each point IN unique_points  
        SET max_x TO the maximum value of the element at position zero of each point IN unique_points  
        SET line_of_reflection TO the sum of min_x PLUS max_x DIVIDED BY two  
        FOR each point IN unique_points  
            SET x TO the element at position zero of point  
            SET y TO the element at position one of point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF a tuple composed of reflected_x AND y IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
