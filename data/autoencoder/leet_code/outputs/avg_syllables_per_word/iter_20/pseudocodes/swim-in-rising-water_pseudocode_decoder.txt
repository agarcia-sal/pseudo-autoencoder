CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an EMPTY set  
        SET min_heap TO a list containing a tuple of the element at position zero zero of grid THEN zero THEN zero  
        ADD the tuple of zero THEN zero TO visited  
        SET directions TO a list containing the tuple zero THEN one THEN the tuple one THEN zero THEN the tuple zero THEN negative one THEN the tuple negative one THEN zero  
        WHILE min_heap is NOT empty  
            REMOVE and ASSIGN the smallest element FROM min_heap TO t THEN i THEN j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each tuple di THEN dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple ni THEN nj NOT IN visited  
                    ADD the tuple ni THEN nj TO visited  
                    INSERT INTO min_heap the tuple of the maximum of t AND the element at position ni THEN nj of grid THEN ni THEN nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
