CLASS Solution  
    FUNCTION countRangeSum(nums WITH TYPE List OF INTEGER, lower WITH TYPE INTEGER, upper WITH TYPE INTEGER) RETURNS INTEGER  
        SET prefix_sums TO a list containing zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET increment_value TO the POSITION AFTER right IN prefix_sums MINUS the POSITION BEFORE left IN prefix_sums  
            INCREMENT count BY increment_value  
            
            INSERT current_sum INTO prefix_sums AT THE CORRECT SORTED POSITION  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
