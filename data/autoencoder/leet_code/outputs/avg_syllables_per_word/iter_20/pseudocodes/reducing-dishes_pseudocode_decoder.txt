CLASS Solution  
    FUNCTION maxSatisfaction WITH PARAMETER satisfaction AS list of integers  
        PERFORM sorting OF satisfaction IN ORDER from highest TO lowest  
        SET max_satisfaction TO zero  
        SET current_satisfaction TO zero  
        SET total_satisfaction TO zero  
        FOR each s IN satisfaction  
            INCREMENT current_satisfaction BY s  
            INCREMENT total_satisfaction BY current_satisfaction  
            IF total_satisfaction IS GREATER THAN max_satisfaction  
                SET max_satisfaction TO total_satisfaction  
            END IF  
        END FOR  
        RETURN max_satisfaction  
    END FUNCTION  
END CLASS
