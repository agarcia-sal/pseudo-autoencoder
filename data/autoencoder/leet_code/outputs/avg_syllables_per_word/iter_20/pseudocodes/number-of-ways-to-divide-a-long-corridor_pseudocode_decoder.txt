CLASS Solution  
    FUNCTION numberOfWays WITH corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO an empty list  
        FOR each index AND character IN corridor  
            IF character EQUALS uppercase letter S  
                APPEND index TO seats  
            END IF  
        END FOR  
        
        IF the LENGTH OF seats MODULO two IS NOT EQUAL TO zero OR the LENGTH OF seats IS LESS THAN two  
            RETURN zero  
        END IF  
        
        SET ways TO one  
        FOR each index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
