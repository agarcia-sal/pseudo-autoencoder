CLASS Solution  
    FUNCTION findRotateSteps(ring, key)  
        CREATE a dictionary named char_positions that assigns an empty list to each new character key  
        FOR each index i FROM zero TO the LENGTH OF ring MINUS one  
            SET char TO the character at position i of ring  
            APPEND i TO the list assigned to char_positions at key char  
        END FOR  
        
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i, prev_pos)  
            IF i EQUALS the LENGTH OF key THEN  
                RETURN zero  
            END IF  
            
            SET min_steps TO infinity  
            FOR each pos IN the list assigned to char_positions at key the character at position i of key  
                SET clockwise TO the remainder of pos MINUS prev_pos DIVIDED BY n  
                SET anticlockwise TO the remainder of prev_pos MINUS pos DIVIDED BY n  
                SET steps TO the smaller value BETWEEN clockwise AND anticlockwise PLUS one PLUS the result of dp with parameters i PLUS one AND pos  
                SET min_steps TO the smaller value BETWEEN min_steps AND steps  
            END FOR  
            
            RETURN min_steps  
        END FUNCTION  
        
        RETURN the result of dp with parameters zero AND zero  
    END FUNCTION  
END CLASS
