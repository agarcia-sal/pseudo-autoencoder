CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER, denominator INTEGER) RETURNS STRING  
        IF numerator LESS THAN zero XOR denominator LESS THAN zero  
            SET sign TO minus sign string  
        ELSE  
            SET sign TO empty string  
        END IF  

        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  

        SET quotient TO result of integer division of numerator BY denominator  
        SET remainder TO result of modulus of numerator BY denominator  
        SET integer_part TO string representation of quotient  

        IF remainder EQUALS zero  
            RETURN concatenation of sign AND integer_part  
        END IF  

        SET seen_remainders TO empty mapping from remainder TO position  
        SET fractional_part TO empty list  

        WHILE remainder NOT EQUAL zero  
            IF remainder IS IN seen_remainders  
                SET start TO the position associated with remainder in seen_remainders  
                SET non_repeating TO concatenation of elements from start zero UP TO start exclusive of fractional_part  
                SET repeating TO concatenation of elements from position start TO END of fractional_part  
                RETURN concatenation of sign, integer_part, decimal point string, non_repeating, opening parenthesis string, repeating, and closing parenthesis string  
            END IF  

            STORE the current length of fractional_part in seen_remainders under remainder as key  

            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO result of integer division of remainder BY denominator  
            SET remainder TO result of modulus of remainder BY denominator  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  

        RETURN concatenation of sign, integer_part, decimal point string, and concatenation of all elements of fractional_part  
    END FUNCTION  
END CLASS
