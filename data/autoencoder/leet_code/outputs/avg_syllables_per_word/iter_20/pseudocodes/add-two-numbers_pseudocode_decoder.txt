CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO the END  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN TRUE  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN FALSE  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND is_same_list CALLED WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS self l1 l2  
        SET dummy_head TO A NEW ListNode INITIALIZED WITH zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS NOT zero  
            IF l1 IS NOT none  
                SET val1 TO the val ATTRIBUTE OF l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT none  
                SET val2 TO the val ATTRIBUTE OF l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET the next ATTRIBUTE OF current TO A NEW ListNode INITIALIZED WITH total MODULO ten  
            SET current TO the next ATTRIBUTE OF current  
            
            IF l1 IS NOT none  
                SET l1 TO the next ATTRIBUTE OF l1  
            END IF  
            IF l2 IS NOT none  
                SET l2 TO the next ATTRIBUTE OF l2  
            END IF  
        END WHILE  
        
        RETURN the next ATTRIBUTE OF dummy_head  
    END FUNCTION  
END CLASS
