CLASS Solution  
    FUNCTION rotate(nums AS List of integers, k AS integer)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start AS integer, end AS integer)  
            WHILE start IS LESS THAN end  
                SET temporary TO element at position start of nums  
                SET element at position start of nums TO element at position end of nums  
                SET element at position end of nums TO temporary  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH start equal to zero AND end equal to n MINUS one  
        CALL reverse WITH start equal to zero AND end equal to k MINUS one  
        CALL reverse WITH start equal to k AND end equal to n MINUS one  
    END FUNCTION  
END CLASS
