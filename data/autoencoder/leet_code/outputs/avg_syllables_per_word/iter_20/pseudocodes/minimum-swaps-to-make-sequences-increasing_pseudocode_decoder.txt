CLASS Solution  
    FUNCTION minSwap(nums1 AS List OF integer, nums2 AS List OF integer) RETURNS integer  
        SET n TO the LENGTH OF nums1  
        SET dp TO a new list of n elements each containing a list of two zeros representing minimum swaps with no swap and with swap respectively  
        SET the element at position zero and zero of dp TO zero  
        SET the element at position zero and one of dp TO one  
        FOR index i FROM one TO n MINUS one  
            IF the element at position i of nums1 IS GREATER THAN the element at position i minus one of nums1 AND the element at position i of nums2 IS GREATER THAN the element at position i minus one of nums2  
                IF the element at position i of nums1 IS GREATER THAN the element at position i minus one of nums2 AND the element at position i of nums2 IS GREATER THAN the element at position i minus one of nums1  
                    SET the element at position i and zero of dp TO the MINIMUM OF the element at position i minus one and zero of dp AND the element at position i minus one and one of dp  
                    SET the element at position i and one of dp TO the element at position i and zero of dp PLUS one  
                ELSE  
                    SET the element at position i and zero of dp TO the element at position i minus one and zero of dp  
                    SET the element at position i and one of dp TO the element at position i minus one and one of dp PLUS one  
                END IF  
            ELSE  
                IF the element at position i of nums1 IS GREATER THAN the element at position i minus one of nums2 AND the element at position i of nums2 IS GREATER THAN the element at position i minus one of nums1  
                    SET the element at position i and zero of dp TO the element at position i minus one and one of dp  
                    SET the element at position i and one of dp TO the element at position i minus one and zero of dp PLUS one  
                ELSE  
                    SET the element at position i and zero of dp TO the element at position i minus one and one of dp  
                    SET the element at position i and one of dp TO the element at position i minus one and zero of dp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the MINIMUM OF the element at position n minus one and zero of dp AND the element at position n minus one and one of dp  
    END FUNCTION  
END CLASS
