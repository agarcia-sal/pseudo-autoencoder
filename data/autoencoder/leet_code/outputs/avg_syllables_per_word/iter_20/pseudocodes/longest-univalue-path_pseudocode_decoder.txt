CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode with parameter value at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO NEW TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO NEW TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value at p IS NOT EQUALS TO value at q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN zero, zero  
            END IF  
            SET left_length, left_value TO dfs(left child of node)  
            SET right_length, right_value TO dfs(right child of node)  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF left child of node IS NOT None AND value at left child of node EQUALS value at node  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF right child of node IS NOT None AND value at right child of node EQUALS value at node  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the maximum of self.max_length AND left_arrow PLUS right_arrow  
            RETURN the maximum of left_arrow AND right_arrow, value at node  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with parameter root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
