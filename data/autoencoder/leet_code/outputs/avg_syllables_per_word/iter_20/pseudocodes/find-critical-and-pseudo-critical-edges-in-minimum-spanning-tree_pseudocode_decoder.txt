CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list of numbers from zero TO n MINUS one  
        SET self.rank TO a list of ones WITH the LENGTH OF n  
    END FUNCTION

    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUALS TO u  
            SET element at position u of self.parent TO the result of calling find WITH element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION

    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find WITH u  
        SET rootV TO the result of calling find WITH v

        IF rootU IS NOT EQUALS TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN the enumeration of edges  
            APPEND i TO edge  
        END FOR

        CALL the sort METHOD on edges WITH a key FUNCTION that RETURNS element at position two of input x

        FUNCTION mst(exclude, include)  
            ASSIGN uf TO a new instance of UnionFind WITH n  
            SET weight TO zero

            IF include IS NOT NONE  
                CALL the union METHOD on uf WITH element at position zero of include AND element at position one of include  
                INCREMENT weight BY element at position two of include  
            END IF

            FOR each u, v, w, idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE to next iteration  
                END IF  
                IF calling union METHOD on uf WITH u AND v RETURNS True  
                    INCREMENT weight BY w  
                END IF  
            END FOR

            SET root TO the result of calling find WITH zero ON uf

            FOR i FROM zero TO n MINUS one  
                IF the result of calling find WITH i ON uf IS NOT EQUALS TO root  
                    RETURN infinity  
                END IF  
            END FOR

            RETURN weight  
        END FUNCTION

        SET mst_weight TO the result of calling mst WITH no arguments

        SET critical TO an empty list  
        SET pseudo_critical TO an empty list

        FOR each u, v, w, idx IN edges  
            IF the result of calling mst WITH exclude assigned TO idx IS GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of calling mst WITH include assigned TO u, v, w IS EQUALS TO mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR

        RETURN a list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
