CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree WITH PARAMETERS self root  
        IF root IS none  
            RETURN none  
        END IF  
        SET root dot left TO self dot pruneTree WITH PARAMETER root dot left  
        SET root dot right TO self dot pruneTree WITH PARAMETER root dot right  
        IF root dot val EQUALS zero AND root dot left IS none AND root dot right IS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
