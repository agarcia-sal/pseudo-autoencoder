CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO an empty list  
            SET sign TO the plus character  
            SET num TO zero  
            WHILE the LENGTH OF s IS GREATER THAN zero HOLDS  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                IF char is a digit HOLDS  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS the left parenthesis character HOLDS  
                    SET num TO the RESULT OF helper WITH PARAMETER s  
                END IF  
                IF char is NOT a digit AND char is NOT a whitespace character OR the LENGTH OF s EQUALS zero HOLDS  
                    IF sign EQUALS the plus character HOLDS  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character HOLDS  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the multiplication character HOLDS  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character HOLDS  
                        SET the element at the last position of stack TO the integer part of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS the right parenthesis character HOLDS  
                    BREAK out of the WHILE loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        RETURN the RESULT OF helper WITH PARAMETER the list formed by the characters of s  
    END FUNCTION  
END CLASS
