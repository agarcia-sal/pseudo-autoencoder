CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals IS empty  
            RETURN zero  
        END IF  
        SORT intervals IN PLACE BY the element at position one of each interval  
        SET end TO the element at position one of the interval at position zero of intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the element at position zero of the interval at position index of intervals IS GREATER THAN OR EQUAL TO end  
                SET end TO the element at position one of the interval at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
