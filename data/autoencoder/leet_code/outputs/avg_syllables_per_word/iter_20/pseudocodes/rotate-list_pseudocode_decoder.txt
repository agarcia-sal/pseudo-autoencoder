CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT None  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values DOES NOT EXIST OR EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new instance of ListNode WITH value at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the end  
        SET node TO new instance of ListNode WITH val  
        SET next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 DOES NOT EXIST OR p2 DOES NOT EXIST  
        RETURN False  
    END IF  
    RETURN the result of logical AND between the comparison of the val property of p1 EQUALS val property of p2 AND the recursive call of is_same_list WITH p1 next property AND p2 next property  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight WITH PARAMETERS head AND k  
        IF head DOES NOT EXIST OR next property of head DOES NOT EXIST OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE next property of tail EXISTS  
            SET tail TO next property of tail  
            INCREMENT length BY one  
        END WHILE  
        SET next property of tail TO head  
        SET k TO remainder of k DIVIDED BY length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each index FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO next property of new_tail  
        END FOR  
        SET new_head TO next property of new_tail  
        SET next property of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
