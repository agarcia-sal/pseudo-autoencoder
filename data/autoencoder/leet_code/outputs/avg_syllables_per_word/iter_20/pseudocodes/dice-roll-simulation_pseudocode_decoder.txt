CLASS Solution  
    FUNCTION dieSimulator(n, rollMax)  
        SET MOD TO one billion plus seven  
        
        SET dp TO a three-dimensional list of zeros with dimensions n plus one by six by the maximum value in rollMax plus one  
        
        FOR each index j FROM zero TO five  
            SET element at position one of element at position j of element at position one of dp TO one  
        END FOR  
        
        FOR each index i FROM two TO n  
            FOR each index j FROM zero TO five  
                FOR each index k FROM one TO element at position j of rollMax  
                    IF k is greater than one THEN  
                        SET element at position k of element at position j of element at position i of dp TO element at position k minus one of element at position j of element at position i minus one of dp  
                    ELSE  
                        SET element at position k of element at position j of element at position i of dp TO zero  
                        FOR each index x FROM zero TO five  
                            IF x is not equal to j THEN  
                                FOR each index y FROM one TO element at position x of rollMax  
                                    INCREMENT element at position k of element at position j of element at position i of dp BY element at position y of element at position x of element at position i minus one of dp  
                                END FOR  
                            END IF  
                        END FOR  
                        SET element at position k of element at position j of element at position i of dp TO element at position k of element at position j of element at position i of dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each index j FROM zero TO five  
            FOR each index k FROM one TO element at position j of rollMax  
                INCREMENT result BY element at position k of element at position j of element at position n of dp  
            END FOR  
        END FOR  
        
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
