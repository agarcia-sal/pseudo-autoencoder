CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the LIST houses is empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF the LIST houses EQUALS one  
                RETURN element at position one of the LIST houses  
            END IF  
            SET dp TO a LIST OF zeros OF LENGTH the LENGTH OF the LIST houses  
            SET element at position one of the LIST dp TO element at position one of the LIST houses  
            SET element at position two of the LIST dp TO the GREATER VALUE BETWEEN element at position one of the LIST houses AND element at position two of the LIST houses  
            FOR index FROM three TO the LENGTH OF the LIST houses  
                SET element at position index of the LIST dp TO the GREATER VALUE BETWEEN element at the POSITION index MINUS one of the LIST dp AND the SUM of element at the POSITION index MINUS two of the LIST dp AND element at position index of the LIST houses  
            END FOR  
            RETURN element at the LAST POSITION of the LIST dp  
        END FUNCTION  
        IF the LIST nums is empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF the LIST nums EQUALS one  
            RETURN element at position one of the LIST nums  
        END IF  
        RETURN the GREATER VALUE BETWEEN calling FUNCTION rob_linear WITH the LIST nums EXCLUDING the LAST element AND calling FUNCTION rob_linear WITH the LIST nums EXCLUDING the FIRST element  
    END FUNCTION  
END CLASS
