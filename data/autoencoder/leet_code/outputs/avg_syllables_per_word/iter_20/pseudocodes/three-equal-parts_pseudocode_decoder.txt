CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO the SUM of all elements in arr  
        
        IF total_ones EQUALS zero  
            RETURN a list containing zero AND two  
        END IF  
        
        IF the remainder of total_ones DIVIDED BY three IS NOT zero  
            RETURN a list containing negative one AND negative one  
        END IF  
        
        SET part_ones TO total_ones DIVIDED BY three  
        
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS one  
                    SET first_part_index TO i  
                END IF  
                IF count_ones EQUALS part_ones  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS part_ones PLUS one  
                    SET second_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS two MULTIPLIED BY part_ones PLUS one  
                    SET third_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET n TO the LENGTH OF arr MINUS third_part_index  
        
        IF the sum of first_part_index AND n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           the sum of second_part_index AND n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           the sequence of elements from position first_part_index TO the sum of first_part_index AND n MINUS one of arr IS EQUAL TO  
           the sequence of elements from position second_part_index TO the sum of second_part_index AND n MINUS one of arr AND  
           the sequence of elements from position third_part_index TO the end of arr  
            RETURN a list containing the sum of first_part_index AND n MINUS one AND the sum of second_part_index AND n  
        END IF  
        
        RETURN a list containing negative one AND negative one  
    END FUNCTION  
END CLASS
