CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO xor_all EXCLUSIVE OR num  
        END FOR  
        SET diff_bit TO the bitwise AND of xor_all AND the negative of xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the bitwise AND of num AND diff_bit is not equal to zero  
                SET num1 TO num1 EXCLUSIVE OR num  
            ELSE  
                SET num2 TO num2 EXCLUSIVE OR num  
            END IF  
        END FOR  
        RETURN a list containing num1 AND num2  
    END FUNCTION  
END CLASS
