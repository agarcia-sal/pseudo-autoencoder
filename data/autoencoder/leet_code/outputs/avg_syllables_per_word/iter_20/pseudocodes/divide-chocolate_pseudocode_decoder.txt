CLASS Solution  
    FUNCTION maximizeSweetness WITH PARAMETERS sweetness AND k  
        FUNCTION canDivide WITH PARAMETER min_sweet  
            SET pieces TO zero  
            SET current_sweet TO zero  
            FOR each sweet IN sweetness  
                INCREMENT current_sweet BY sweet  
                IF current_sweet IS GREATER THAN OR EQUAL TO min_sweet  
                    INCREMENT pieces BY one  
                    SET current_sweet TO zero  
                    IF pieces IS GREATER THAN k  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            RETURN pieces IS GREATER THAN k  
        END FUNCTION  
        
        SET left TO the MINIMUM ELEMENT IN sweetness  
        SET right TO the SUM OF ALL ELEMENTS IN sweetness DIVIDED BY the QUANTITY OF k INCREMENTED BY one  
        SET result TO zero  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the SUM OF left AND right DIVIDED BY two  
            IF canDivide WITH mid RETURNS true  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
