CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional table with m PLUS one ROWS and n PLUS one COLUMNS each initialized TO zero  
        FOR i FROM zero TO m  
            SET the element at position zero of the row at position i OF dp TO i  
        END FOR  
        FOR j FROM zero TO n  
            SET the element at position j of the row at position zero OF dp TO j  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position j of the row at position i OF dp TO the element at position j MINUS one of the row at position i MINUS one of dp  
                ELSE  
                    SET deletion_cost TO the element at position j of the row at position i MINUS one of dp  
                    SET insertion_cost TO the element at position j MINUS one of the row at position i OF dp  
                    SET replacement_cost TO the element at position j MINUS one of the row at position i MINUS one OF dp  
                    SET the element at position j of the row at position i OF dp TO one PLUS THE MINIMUM OF deletion_cost insertion_cost replacement_cost  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of the row at position m OF dp  
    END FUNCTION  
END CLASS
