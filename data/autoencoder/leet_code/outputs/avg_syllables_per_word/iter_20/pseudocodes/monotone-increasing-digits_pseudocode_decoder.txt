CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list of characters representing the decimal digits of the string representation of n  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR index FROM length MINUS one DOWN TO one  
            IF the character at position index of digits is less than the character at position index MINUS one of digits  
                SET the character at position index MINUS one of digits TO the string representation of the integer value of the character at position index MINUS one of digits MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one  
            SET the character at position index of digits TO the character nine  
        END FOR  
        RETURN the integer conversion of the concatenation of all characters in digits  
    END FUNCTION  
END CLASS
