CLASS Solution  
    FUNCTION hasValidPath(grid: list of list of string)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF the SUM of m PLUS n MINUS one MODULO two NOT EQUALS zero  
            RETURN false  
        END IF  
        
        INITIALIZE a three dimensional list named dp sized m plus one by n plus one by m plus n plus one with all elements SET TO false  
        
        SET element at position zero of list at position one of list at position zero of dp TO true  
        
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index k FROM zero TO m PLUS n  
                    IF element at position j plus one of list at position i of dp IS true OR element at position j of list at position i plus one of dp IS true  
                        IF element at position j of list at position i of grid EQUALS open parenthesis  
                            IF k PLUS one LESS THAN m PLUS n PLUS one  
                                SET element at position j plus one of list at position i plus one of dp AT position k PLUS one TO true  
                            END IF  
                        ELSE IF element at position j of list at position i of grid EQUALS close parenthesis  
                            IF k MINUS one GREATER THAN OR EQUAL TO zero  
                                SET element at position j plus one of list at position i plus one of dp AT position k MINUS one TO true  
                            END IF  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n of list at position m of dp AT position zero  
    END FUNCTION  
END CLASS
