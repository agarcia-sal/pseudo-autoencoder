CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list of pairs consisting of minus one and zero plus one and zero zero and minus one zero and plus one zero and zero  
            FOR each pair dx and dy IN directions  
                SET nx TO the sum of x and dx  
                SET ny TO the sum of y and dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET element at position nx and ny of matrix TO one MINUS element at position nx and ny of matrix  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN a tuple composed of tuples each being the rows of matrix converted to tuples  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO a tuple composed of tuples each consisting of zero repeated n times for m times  
        SET initial_state TO the RESULT OF matrix_to_tuple WHEN APPLIED TO mat  
        SET queue TO a list containing a pair of initial_state and zero  
        SET visited TO a set  
        ADD initial_state TO visited  

        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE from the FRONT element current_state and steps FROM queue  
            IF current_state IS EQUAL TO target  
                RETURN steps  
            END IF  

            SET current_matrix TO a list composed of lists each converted from rows of current_state  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a DEEP COPY OF current_matrix  
                    CALL flip WITH next_matrix i j  
                    SET next_state TO the RESULT OF matrix_to_tuple WHEN APPLIED TO next_matrix  
                    IF next_state IS NOT IN visited  
                        ADD next_state TO visited  
                        APPEND a pair of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
