CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the end of values  
        SET node TO new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN whether the value attribute of p1 EQUALS the value attribute of p2 AND the recursive call to is_same_list with parameters the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition WITH PARAMETERS head AND x  
        SET less_head TO new ListNode with zero  
        SET greater_head TO new ListNode with zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT none  
            IF the value attribute of current IS LESS THAN x  
                SET the next attribute of less TO current  
                SET less TO the next attribute of less  
            ELSE  
                SET the next attribute of greater TO current  
                SET greater TO the next attribute of greater  
            END IF  
            SET current TO the next attribute of current  
        END WHILE  
        SET the next attribute of less TO the next attribute of greater_head  
        SET the next attribute of greater TO none  
        RETURN the next attribute of less_head  
    END FUNCTION  
END CLASS
