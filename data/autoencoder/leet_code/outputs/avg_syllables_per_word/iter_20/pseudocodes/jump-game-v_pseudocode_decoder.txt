CLASS Solution  
    FUNCTION maxJumps(arr LIST OF INTEGERS, d INTEGER) RETURNS INTEGER  
        FUNCTION dp(i INTEGER) RETURNS INTEGER  
            SET max_jumps TO one  
            
            FOR x FROM one TO d  
                IF i PLUS x IS LESS THAN the LENGTH OF arr THEN  
                    IF element at position i PLUS x of arr IS GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET possible_jumps TO one PLUS dp(i PLUS x)  
                    IF possible_jumps IS GREATER THAN max_jumps THEN  
                        SET max_jumps TO possible_jumps  
                    END IF  
                END IF  
            END FOR  
            
            FOR x FROM one TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO zero THEN  
                    IF element at position i MINUS x of arr IS GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET possible_jumps TO one PLUS dp(i MINUS x)  
                    IF possible_jumps IS GREATER THAN max_jumps THEN  
                        SET max_jumps TO possible_jumps  
                    END IF  
                END IF  
            END FOR  
            
            RETURN max_jumps  
        END FUNCTION  
        
        SET max_overall TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            SET current_jumps TO dp(i)  
            IF current_jumps IS GREATER THAN max_overall THEN  
                SET max_overall TO current_jumps  
            END IF  
        END FOR  
        RETURN max_overall  
    END FUNCTION  
END CLASS
