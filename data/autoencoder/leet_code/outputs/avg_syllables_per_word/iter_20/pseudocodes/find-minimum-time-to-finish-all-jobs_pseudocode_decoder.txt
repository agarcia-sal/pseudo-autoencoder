CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeroes with LENGTH equal TO TWO RAISED TO THE POWER OF n  
        
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF i MINUS one  
                SET subset_sums at the POSITION OF the BITWISE OR of mask AND TWO RAISED TO THE POWER OF i TO subset_sums at the POSITION OF mask PLUS jobs at the POSITION OF i  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one  
                RETURN subset_sums at the POSITION OF mask  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET candidate_one TO dp with PARAMETERS k MINUS one AND bitwise exclusive OR of mask AND submask  
                SET candidate_two TO subset_sums at the POSITION OF submask  
                SET current_max TO the GREATER OF candidate_one AND candidate_two  
                SET res TO the SMALLER OF res AND current_max  
                SET submask TO the BITWISE AND of submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp with PARAMETERS k AND TWO RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
