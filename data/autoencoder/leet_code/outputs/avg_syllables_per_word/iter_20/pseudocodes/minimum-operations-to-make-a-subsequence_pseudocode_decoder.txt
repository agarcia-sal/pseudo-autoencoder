CLASS Solution  
    FUNCTION minOperations WITH PARAMETERS target AND arr  
        SET index_map TO AN EMPTY MAPPING FROM VALUES TO INDICES  
        FOR each index AND value IN target WITH THEIR POSITIONS  
            SET value IN index_map TO index  
        END FOR  
        
        SET transformed_arr TO AN EMPTY LIST  
        FOR each value IN arr  
            IF value EXISTS AS A KEY IN index_map  
                APPEND index_map AT value TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO AN EMPTY LIST  
        FOR each num IN transformed_arr  
            SET pos TO THE POSITION WHERE num SHOULD BE INSERTED IN lis TO KEEP IT NON-DECREASING USING BINARY SEARCH  
            IF pos EQUALS THE LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET ELEMENT AT POSITION pos OF lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO THE LENGTH OF lis  
        
        RETURN THE LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
