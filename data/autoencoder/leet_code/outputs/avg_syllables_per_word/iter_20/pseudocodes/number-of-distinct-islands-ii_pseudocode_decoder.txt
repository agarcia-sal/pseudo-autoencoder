CLASS Solution  
    FUNCTION numDistinctIslands2 WITH PARAMETERS grid  
        FUNCTION dfs WITH PARAMETERS x y island  
            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO the LENGTH OF grid OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x of element at position y of grid EQUALS zero  
                RETURN  
            END IF  
            SET element at position x of element at position y of grid TO zero  
            APPEND tuple of x AND y TO island  
            CALL dfs WITH PARAMETERS x PLUS one y island  
            CALL dfs WITH PARAMETERS x MINUS one y island  
            CALL dfs WITH PARAMETERS x y PLUS one island  
            CALL dfs WITH PARAMETERS x y MINUS one island  
        END FUNCTION  

        FUNCTION normalize WITH PARAMETER island  
            SET shapes TO an empty list  
            FOR each pair dx dy IN list of pairs consisting of one one AND one MINUS one AND MINUS one one AND MINUS one MINUS one  
                SET new_island TO empty list  
                FOR each pair x y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island IN ascending order  
                SET min_x TO minimum x value IN new_island  
                SET min_y TO minimum y value IN new_island  
                APPEND tuple of each pair of x MINUS min_x AND y MINUS min_y FOR each pair x y IN new_island TO shapes  
            END FOR  
            FOR each index i FROM zero TO LENGTH OF shapes MINUS one  
                APPEND tuple of each pair y AND NEGATIVE x FOR each pair x y IN element at position i of shapes TO shapes  
            END FOR  
            RETURN minimum element in shapes according to lexicographic order  
        END FUNCTION  

        SET islands TO an empty set  
        FOR each index i FROM zero TO LENGTH OF grid MINUS one  
            FOR each index j FROM zero TO LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of element at position j of grid EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH PARAMETERS i j island  
                    ADD normalize of island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
