CLASS Solution  
    FUNCTION isTransformable(s, t)  
        ASSIGN pointer REFERENCE TO a collection that maps each digit TO an empty list of positions  
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            ASSIGN current_character TO element at position index of s  
            APPEND index TO the list of positions corresponding TO the integer value of current_character in pointer  
        END FOR  
        FOR each index FROM zero TO LENGTH OF t MINUS one  
            ASSIGN current_character TO element at position index of t  
            ASSIGN digit TO the integer value of current_character  
            IF the list of positions corresponding TO digit in pointer IS empty  
                RETURN false  
            END IF  
            FOR each smaller_digit FROM zero TO digit MINUS one  
                IF the list of positions corresponding TO smaller_digit in pointer IS NOT empty AND the first element at position zero of the list of positions corresponding TO smaller_digit IS LESS THAN the first element at position zero of the list of positions corresponding TO digit  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the first element at position zero FROM the list of positions corresponding TO digit in pointer  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
