CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUALS u  
            SET element at position u of self.parent TO the result of calling find with element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET root_u TO result of calling find with u  
        SET root_v TO result of calling find with v  
        IF root_u IS NOT EQUALS root_v  
            SET element at position root_u of self.parent TO root_v  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION matrixRankTransform(self, matrix)  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET value_to_indices TO a mapping that returns an empty list for missing keys  
        
        FOR each row index r FROM zero TO m MINUS one  
            FOR each column index c FROM zero TO n MINUS one  
                APPEND pair of r AND c TO element at key of element at position r of matrix at position c in value_to_indices  
            END FOR  
        END FOR  
        
        SET row_rank TO a list of zero repeated m times  
        SET col_rank TO a list of zero repeated n times  
        SET result TO a list containing m elements each being a list of zero repeated n times  
        
        FOR each value IN the keys of value_to_indices sorted in ascending order  
            ASSIGN uf TO a new instance of UnionFind with the parameter m PLUS n  
            SET rank TO a mapping with default integer zero for missing keys  
            
            FOR each pair r, c IN element at key value of value_to_indices  
                CALL union method of uf with r AND c PLUS m  
            END FOR  
            
            FOR each pair r, c IN element at key value of value_to_indices  
                SET root TO result of calling find method of uf with r  
                SET element at key root in rank TO the maximum of element at key root in rank AND the maximum of element at position r of row_rank AND element at position c of col_rank PLUS one  
            END FOR  
            
            FOR each pair r, c IN element at key value of value_to_indices  
                SET root TO result of calling find method of uf with r  
                SET element at position r at position c of result TO element at key root in rank  
                SET element at position r of row_rank TO element at key root in rank  
                SET element at position c of col_rank TO element at key root in rank  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
