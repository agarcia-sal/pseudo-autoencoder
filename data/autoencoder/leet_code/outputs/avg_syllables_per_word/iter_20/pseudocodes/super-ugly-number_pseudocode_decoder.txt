CLASS Solution  
    FUNCTION nthSuperUglyNumber(n AS integer, primes AS list of integers) RETURNS integer  
        SET ugly_numbers TO a list initially containing the single integer one  
        SET heap TO a list formed by pairing each prime number WITH its corresponding index in primes  
        TRANSFORM heap INTO a min-heap structure  
        SET indices TO a list of zeroes with the length equal to the length of primes  
        WHILE the LENGTH OF ugly_numbers IS LESS THAN n  
            EXTRACT the element with the smallest first value from heap ASSIGNING its values TO next_ugly AND prime_index  
            IF next_ugly IS NOT EQUAL TO the element at the last position of ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            INCREMENT the element at position prime_index of indices BY one  
            SET next_multiple TO the element at position prime_index of primes MULTIPLIED BY the element at position at the element at position prime_index of indices of ugly_numbers  
            INSERT next_multiple and prime_index as a pair INTO the heap maintaining min-heap properties  
        END WHILE  
        RETURN the element at the last position of ugly_numbers  
    END FUNCTION  
END CLASS
