CLASS Solution  
    FUNCTION combine WITH n AND k  
        FUNCTION backtrack WITH start AND path  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one  
                APPEND i TO path  
                CALL backtrack WITH i PLUS one AND path  
                REMOVE the LAST element FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an EMPTY list  
        CALL backtrack WITH one AND an EMPTY list  
        RETURN result  
    END FUNCTION  
END CLASS
