CLASS Solution  
    FUNCTION secondMinimum(n, edges, time, change)  
        SET graph TO a list of empty lists with count n PLUS one  
        FOR each pair consisting of u and v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        ASSIGN queue TO an empty double ended queue initialized with tuple consisting of integer one current vertex integer zero current time and integer zero count of shortest or second shortest paths  
        SET minTimes TO a list of lists each containing two elements set to infinity with count n PLUS one  
        SET element at position one and position zero of minTimes TO zero  
        
        WHILE queue is not empty  
            REMOVE the element from front of queue and ASSIGN current vertex currentTime and count accordingly  
            
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
            
            IF the integer division of currentTime BY change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the sum of the increment by one of the integer division of currentTime BY change MULTIPLIED BY change PLUS time  
            END IF  
            
            FOR each neighbor IN element at position current of graph  
                IF element at position zero of element at position neighbor of minTimes GREATER THAN nextTime  
                    SET element at position one of element at position neighbor of minTimes TO element at position zero of element at position neighbor of minTimes  
                    SET element at position zero of element at position neighbor of minTimes TO nextTime  
                    APPEND tuple consisting of neighbor nextTime and zero TO queue  
                ELSE IF element at position zero of element at position neighbor of minTimes LESS THAN nextTime AND nextTime LESS THAN element at position one of element at position neighbor of minTimes  
                    SET element at position one of element at position neighbor of minTimes TO nextTime  
                    APPEND tuple consisting of neighbor nextTime and one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
