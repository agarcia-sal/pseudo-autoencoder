CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO the LIST obtained by PAIRING each element of plantTime WITH the corresponding element of growTime and SORTED by the SECOND element of each pair in DESCENDING order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair plant AND grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day IS GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
