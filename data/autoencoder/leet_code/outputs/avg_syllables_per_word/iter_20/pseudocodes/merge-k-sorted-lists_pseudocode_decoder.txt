CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULTING TO zero AND next DEFAULTING TO none  
        SET the attribute val OF self TO val  
        SET the attribute next OF self TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the attribute next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the expression that the attribute val OF p1 EQUALS the attribute val OF p2 AND the result of is_same_list WITH PARAMETERS the attribute next OF p1 AND the attribute next OF p2  
END FUNCTION

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS self AND lists AS list OF optional ListNode  
        SET min_heap TO an empty list  
        FOR each index i AND element lst IN lists USING enumeration  
            IF lst IS NOT none  
                CALL the function to push onto min_heap a tuple consisting of the attribute val OF lst, the index i, AND lst itself  
            END IF  
        END FOR  
        SET dummy TO A NEW ListNode WITH default initialization  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            CALL the function to pop the smallest element FROM min_heap AND ASSIGN the unpacked tuple TO val, i, AND node  
            SET the attribute next OF current TO node  
            SET current TO the attribute next OF current  
            IF the attribute next OF node IS NOT none  
                CALL the function to push onto min_heap a tuple consisting of the attribute val OF the attribute next OF node, the index i, AND the attribute next OF node  
            END IF  
        END WHILE  
        RETURN the attribute next OF dummy  
    END FUNCTION  
END CLASS
