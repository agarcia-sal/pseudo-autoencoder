CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF the character IS A digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of the character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND the character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF the element at position i of tokens IS the multiplication symbol THEN  
                    SET element at position i MINUS ONE of tokens TO element at position i MINUS ONE of tokens MULTIPLIED BY element at position i PLUS ONE of tokens  
                    REMOVE elements at positions from i TO i PLUS one of tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF the element at position i of tokens IS the addition symbol THEN  
                    SET result TO result PLUS element at position i PLUS ONE of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF the expression CONSISTS ONLY OF digits THEN  
                RETURN a set containing the integer value of expression  
            END IF  
            
            SET results TO empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS two IN STEPS OF two  
                SET left_results TO get_possible_answers(substring from position zero TO i MINUS one of expression)  
                SET right_results TO get_possible_answers(substring from position i PLUS one TO end of expression)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF the character at position i of expression IS the addition symbol THEN  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result IS LESS THAN OR EQUAL TO one thousand THEN  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer THEN  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers THEN  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
