CLASS Solution  
    FUNCTION minArea(image WITH TYPE List OF List OF string, x WITH TYPE integer, y WITH TYPE integer) RETURNS integer  
        IF image IS empty OR the LENGTH OF image AT position zero IS zero  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF image AT position zero  
        
        FUNCTION hasBlackPixelInRow(row WITH TYPE integer) RETURNS boolean  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS the character one  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col WITH TYPE integer) RETURNS boolean  
            FOR row FROM zero TO m MINUS one  
                IF element at position col of element at position row of image EQUALS the character one  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        FUNCTION findTop() RETURNS integer  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER RESULT OF (low PLUS high) DIVIDED BY two  
                IF hasBlackPixelInRow(mid) IS true  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom() RETURNS integer  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER RESULT OF (low PLUS high PLUS one) DIVIDED BY two  
                IF hasBlackPixelInRow(mid) IS true  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft() RETURNS integer  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER RESULT OF (low PLUS high) DIVIDED BY two  
                IF hasBlackPixelInColumn(mid) IS true  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight() RETURNS integer  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER RESULT OF (low PLUS high PLUS one) DIVIDED BY two  
                IF hasBlackPixelInColumn(mid) IS true  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  
        
        RETURN the QUANTITY obtained by the EXPRESSION bottom MINUS top PLUS one MULTIPLIED BY right MINUS left PLUS one  
    END FUNCTION  
END CLASS
