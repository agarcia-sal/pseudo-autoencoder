CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a list of zeroes with LENGTH equal TO n PLUS one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of s DOES NOT EQUAL character zero  
                    AND the NUMBER REPRESENTED BY substring from position j TO i MINUS one of s IS GREATER THAN OR EQUAL TO one  
                    AND the NUMBER REPRESENTED BY substring from position j TO i MINUS one of s IS LESS THAN OR EQUAL TO k  
                    THEN  
                        SET element at position i of dp TO the REMAINDER after DIVIDING the SUM of element at position i of dp PLUS element at position j of dp BY MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
