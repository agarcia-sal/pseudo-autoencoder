CLASS Solution  
    FUNCTION grayCode WITH PARAMETER n  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode WITH argument n MINUS one FROM this instance  
        SET mask TO one LEFT SHIFTED BY n MINUS one  
        SET current_gray TO prev_gray CONCATENATED WITH a list constructed by FOR each number IN the reversal of prev_gray TRANSFORMED BY mask BITWISE OR number  
        RETURN current_gray  
    END FUNCTION  
END CLASS
