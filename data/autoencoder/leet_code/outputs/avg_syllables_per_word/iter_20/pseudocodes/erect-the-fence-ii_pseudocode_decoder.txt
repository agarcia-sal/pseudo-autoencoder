CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            RETURN the SQUARE OF the DIFFERENCE BETWEEN element at position zero of p1 AND element at position zero of p2 PLUS the SQUARE OF the DIFFERENCE BETWEEN element at position one of p1 AND element at position one of p2  
        END FUNCTION  

        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO TWO MULTIPLIED BY the SUM of ax MULTIPLIED BY the DIFFERENCE BETWEEN by AND cy PLUS bx MULTIPLIED BY the DIFFERENCE BETWEEN cy AND ay PLUS cx MULTIPLIED BY the DIFFERENCE BETWEEN ay AND by  
            SET ux TO the QUOTIENT of the SUM of ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay MULTIPLIED BY the DIFFERENCE BETWEEN by AND cy PLUS bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by MULTIPLIED BY the DIFFERENCE BETWEEN cy AND ay PLUS cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy MULTIPLIED BY the DIFFERENCE BETWEEN ay AND by DIVIDED BY d  
            SET uy TO the QUOTIENT of the SUM of ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay MULTIPLIED BY the DIFFERENCE BETWEEN cx AND bx PLUS bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by MULTIPLIED BY the DIFFERENCE BETWEEN ax AND cx PLUS cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy MULTIPLIED BY the DIFFERENCE BETWEEN bx AND ax DIVIDED BY d  
            RETURN a list containing ux and uy, the SQUARE ROOT of the SUM of the SQUARE OF the DIFFERENCE BETWEEN ux AND ax PLUS the SQUARE OF the DIFFERENCE BETWEEN uy AND ay  
        END FUNCTION  

        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the LENGTH OF boundary EQUALS three THEN  
                IF the LENGTH OF boundary EQUALS zero THEN  
                    RETURN a list containing zero and zero, zero  
                ELSE IF the LENGTH OF boundary EQUALS one THEN  
                    RETURN element at position zero of boundary, zero  
                ELSE IF the LENGTH OF boundary EQUALS two THEN  
                    SET center TO a list containing the QUOTIENT of the SUM of element at position zero of element at position zero of boundary PLUS element at position zero of element at position one of boundary DIVIDED BY two AND the QUOTIENT of the SUM of element at position one of element at position zero of boundary PLUS element at position one of element at position one of boundary DIVIDED BY two  
                    SET radius TO the SQUARE ROOT of the DISTANCE SQUARED between element at position zero of boundary AND element at position one of boundary DIVIDED BY two  
                    RETURN center, radius  
                ELSE  
                    RETURN the RESULT of circumcenter with elements from boundary as parameters  
                END IF  
            END IF  
            SET p TO element at position n MINUS one of points  
            SET center, radius TO the RESULT of welzl with parameters points, boundary, n MINUS one  
            IF the DISTANCE SQUARED between center AND p LESS THAN OR EQUAL TO radius MULTIPLIED BY radius THEN  
                RETURN center, radius  
            END IF  
            RETURN the RESULT of welzl with parameters points, boundary with p appended, n MINUS one  
        END FUNCTION  

        CALL the shuffle FUNCTION from random MODULE on trees  
        SET center, radius TO the RESULT of welzl with parameters trees, an empty list, the LENGTH OF trees  
        RETURN a list containing element at position zero of center, element at position one of center, radius  
    END FUNCTION  
END CLASS
