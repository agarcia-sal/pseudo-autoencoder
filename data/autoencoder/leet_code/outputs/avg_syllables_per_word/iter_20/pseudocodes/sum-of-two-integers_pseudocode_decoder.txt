CLASS Solution  
    FUNCTION getSum(a INTEGER parameter, b INTEGER parameter) RETURNS INTEGER  
        SET MAX TO the largest thirty two bits integer value which equals seven followed by seven F characters in hexadecimal  
        SET mask TO the value that has all thirty two bits set to one  
        WHILE b IS NOT EQUAL TO zero  
            SET carry TO the bitwise AND of a and b then shifted left by one position  
            SET a TO the bitwise XOR of a and b then bitwise AND with mask  
            SET b TO the bitwise AND of carry and mask  
        END WHILE  
        IF a IS LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of the bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
