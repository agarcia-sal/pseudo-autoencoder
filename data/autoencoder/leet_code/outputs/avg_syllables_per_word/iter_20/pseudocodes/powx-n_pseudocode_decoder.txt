CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x AND n  
        FUNCTION qpow WITH PARAMETERS a AND n  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF the BITWISE AND of n AND one EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two DISCARDING THE FRACTIONAL PART  
            END WHILE  
            RETURN ans  
        END FUNCTION  

        IF n GREATER THAN OR EQUAL TO zero  
            RETURN the RESULT OF qpow WITH ARGUMENTS x AND n  
        ELSE  
            RETURN one DIVIDED BY the RESULT OF qpow WITH ARGUMENTS x AND negative n  
        END IF  
    END FUNCTION  
END CLASS
