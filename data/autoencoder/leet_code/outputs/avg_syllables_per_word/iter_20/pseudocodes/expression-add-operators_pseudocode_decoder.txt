CLASS Solution  
    FUNCTION addOperators(num, target)  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS zero  
                    CONTINUE TO the next iteration  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs with parameters i, operand, integer value of operand, integer value of operand  
                ELSE  
                    CALL dfs with parameters i, concatenation of path PLUS plus SIGN PLUS operand, value PLUS integer value of operand, integer value of operand  
                    CALL dfs with parameters i, concatenation of path PLUS minus SIGN PLUS operand, value MINUS integer value of operand, NEGATIVE integer value of operand  
                    CALL dfs with parameters i, concatenation of path PLUS multiplication SIGN PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY integer value of operand, prev MULTIPLIED BY integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL dfs with parameters zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
