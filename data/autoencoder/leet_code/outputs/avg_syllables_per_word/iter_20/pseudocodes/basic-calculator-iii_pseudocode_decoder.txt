CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO element at position zero of s  
                REMOVE element at position zero FROM s  
                IF char represents a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                IF char EQUALS left parenthesis  
                    SET num TO THE RESULT OF calling helper WITH s  
                END IF  
                IF char IS one of plus OR minus OR multiply OR divide OR left parenthesis OR right parenthesis OR s IS empty  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND the NEGATIVE of num TO stack  
                    ELSE IF sign EQUALS multiply  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS divide  
                        SET the element at the last position of stack TO the INTEGER PART of the element at the last position of stack DIVIDED BY num truncating towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS right parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            RETURN the SUM of all elements in stack  
        END FUNCTION  
        
        CONVERT s TO a list of characters  
        RETURN THE RESULT OF calling helper WITH s  
    END FUNCTION  
END CLASS
