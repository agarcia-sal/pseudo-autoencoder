CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinite quantity  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current_sum BY the element at position right of nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target holds  
                SET min_length TO the MINIMUM of min_length AND right MINUS left PLUS one  
                DECREMENT current_sum BY the element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length NOT EQUALS infinite quantity THEN  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
