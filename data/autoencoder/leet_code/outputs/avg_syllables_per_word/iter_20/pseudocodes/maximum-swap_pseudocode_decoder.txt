CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO the list conversion of the string conversion of num  
        SET last TO the dictionary mapping each digit converted to integer TO its last position index in digits  
        FOR each index i AND digit d IN the enumeration of digits  
            FOR each digit d2 FROM nine DOWN TO the digit d converted to integer PLUS one  
                IF the value corresponding to key d2 in last DEFAULTING TO zero IS GREATER THAN i  
                    SET temporary variable TO element at position i of digits  
                    SET element at position i of digits TO element at the position corresponding to key d2 in last of digits  
                    SET element at the position corresponding to key d2 in last of digits TO temporary variable  
                    RETURN the integer conversion of the concatenation of all elements in digits  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
