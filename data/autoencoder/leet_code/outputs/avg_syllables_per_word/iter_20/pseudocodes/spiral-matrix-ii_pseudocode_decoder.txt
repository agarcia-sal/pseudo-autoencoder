CLASS Solution  
    FUNCTION generateMatrix WITH PARAMETER n  
        SET matrix TO a two dimensional collection with n elements each containing n elements initialized to zero  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE the value of top is LESS THAN OR EQUAL TO the value of bottom AND the value of left is LESS THAN OR EQUAL TO the value of right holds  
            FOR each index i FROM the value of left TO the value of right  
                SET the element at position top of matrix at position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR each index i FROM the value of top TO the value of bottom  
                SET the element at position i of matrix at position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF the value of top is LESS THAN OR EQUAL TO the value of bottom holds  
                FOR each index i FROM the value of right DOWN TO the value of left  
                    SET the element at position bottom of matrix at position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF the value of left is LESS THAN OR EQUAL TO the value of right holds  
                FOR each index i FROM the value of bottom DOWN TO the value of top  
                    SET the element at position i of matrix at position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
