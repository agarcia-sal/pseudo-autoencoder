CLASS Solution  
    FUNCTION minWindow(s string, t string) RETURNS string  
        INITIALIZE frequency counter t_count TO the frequency of each character appearing in t  
        SET required TO the number of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        INITIALIZE frequency counter window_counts TO an empty frequency mapping  
        SET min_length TO positive infinity  
        SET min_start TO zero  
        
        WHILE right IS LESS THAN the LENGTH OF s  
            SET current_character TO element at position right of s  
            INCREMENT the count for current_character in window_counts BY one  
            
            IF current_character IS A KEY IN t_count AND the count for current_character in window_counts EQUALS the count for current_character in t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET character_at_left TO element at position left of s  
                
                IF right MINUS left PLUS one IS LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count for character_at_left in window_counts BY one  
                IF character_at_left IS A KEY IN t_count AND the count for character_at_left in window_counts IS LESS THAN the count for character_at_left in t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS positive infinity  
            RETURN an empty string  
        ELSE  
            RETURN substring from position min_start TO min_start PLUS min_length MINUS one of s  
        END IF  
    END FUNCTION  
END CLASS
