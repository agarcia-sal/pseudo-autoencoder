CLASS Solution  
    FUNCTION getHint(secret AS string, guess AS string) RETURNS string  
        IMPORT default dictionary FROM collections  
  
        SET bulls TO zero  
        SET cows TO zero  
  
        SET secret_count TO a default dictionary with default integer zero  
        SET guess_count TO a default dictionary with default integer zero  
  
        FOR each pair of character s and character g FROM pairing corresponding characters of secret and guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key s IN secret_count BY one  
                INCREMENT the value at key g IN guess_count BY one  
            END IF  
        END FOR  
  
        FOR each digit IN all keys of secret_count  
            INCREMENT cows BY the minimum of the value at key digit IN secret_count AND the value at key digit IN guess_count  
        END FOR  
  
        RETURN the string concatenation of bulls converted to string THEN the letter A THEN cows converted to string THEN the letter B  
    END FUNCTION  
END CLASS
