CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero AND next EQUALS none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode constructed with element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO end of values  
        SET node TO new ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the attribute val of p1 EQUALS the attribute val of p2 AND is_same_list with parameters the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween with parameters head AND left AND right  
        IF head EQUALS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode constructed with zero  
        SET the next attribute of dummy TO head  
        SET prev TO dummy  
        FOR count FROM one TO left MINUS one  
            SET prev TO the next attribute of prev  
        END FOR  
        SET current TO the next attribute of prev  
        FOR count FROM one TO right MINUS left  
            SET next_node TO the next attribute of current  
            SET the next attribute of current TO the next attribute of next_node  
            SET the next attribute of next_node TO the next attribute of prev  
            SET the next attribute of prev TO next_node  
        END FOR  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
