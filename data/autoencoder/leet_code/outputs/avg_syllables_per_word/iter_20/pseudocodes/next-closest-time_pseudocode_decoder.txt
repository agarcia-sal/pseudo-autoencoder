CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time after REPLACING colon WITH empty string  
        SET hour TO the integer CONVERTED FROM the substring from position zero TO one of time  
        SET minute TO the integer CONVERTED FROM the substring from position three TO four of time  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each first hour digit IN digits  
                FOR each second hour digit IN digits  
                    FOR each first minute digit IN digits  
                        FOR each second minute digit IN digits  
                            SET computed_hour TO the integer CONVERTED FROM the concatenation of first hour digit AND second hour digit  
                            SET computed_minute TO the integer CONVERTED FROM the concatenation of first minute digit AND second minute digit  
                            IF computed_hour LESS THAN twenty four AND computed_minute LESS THAN sixty  
                                APPEND the pair of computed_hour AND computed_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times SORTED IN ascending order  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        SET current_time_index TO the POSITION OF the pair of hour AND minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the LENGTH OF all_times  
            SET next_hour TO the first component of the element AT position current_time_index PLUS one OF all_times  
            SET next_minute TO the second component of the element AT position current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour TO the first component of the element AT position zero OF all_times  
            SET next_minute TO the second component of the element AT position zero OF all_times  
        END IF  
        
        RETURN the string consisting of next_hour FORMATTED AS two digit number FOLLOWED BY colon FOLLOWED BY next_minute FORMATTED AS two digit number  
    END FUNCTION  
END CLASS
