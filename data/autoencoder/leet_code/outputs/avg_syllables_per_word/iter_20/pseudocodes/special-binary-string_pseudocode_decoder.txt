CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index and character IN s with index starting from zero  
                IF character EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the character one CONCATENATED WITH dfs of the substring from position start PLUS one TO position index CONCATENATED WITH the character zero TO specials  
                    SET start TO index PLUS one  
                END IF  
            END FOR  
            SORT specials IN reverse LEXICOGRAPHICAL ORDER  
            RETURN the concatenation of all elements in specials  
        END FUNCTION  
        RETURN dfs of s  
    END FUNCTION  
END CLASS
