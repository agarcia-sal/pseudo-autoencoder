CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values list)  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.left TO new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.right TO new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value field of p IS NOT EQUALS the value field of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments the left child of p AND the left child of q AND the result of is_same_tree with arguments the right child of p AND the right child of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root optional TreeNode) RETURNS optional TreeNode  
        IF root IS none OR the left child of root IS none  
            RETURN root  
        END IF  
        SET new_root TO the result of calling upsideDownBinaryTree from self with the left child of root  
        SET the left child of the left child of root TO the right child of root  
        SET the right child of the left child of root TO root  
        SET the left child of root TO none  
        SET the right child of root TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
