CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_version TO the string at position index of strs REVERSED  
            IF reversed_version GREATER THAN the string at position index of strs  
                SET the string at position index of strs TO reversed_version  
            END IF  
        END FOR  
        
        SET doubled_strs TO strs FOLLOWED BY strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the collection containing false and true  
                IF is_reversed IS true  
                    SET current TO the string at position index of strs REVERSED  
                ELSE  
                    SET current TO the string at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of all strings from position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one in doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring from position cut_position TO the end of current FOLLOWED BY remaining FOLLOWED BY the substring from the start TO position cut_position MINUS one of current  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
