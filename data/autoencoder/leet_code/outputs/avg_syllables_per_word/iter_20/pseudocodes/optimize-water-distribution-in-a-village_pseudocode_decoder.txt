CLASS Solution  
    FUNCTION minCostToSupplyWater WITH PARAMETERS n AS integer wells AS list of integer pipes AS list of list of integer  
        FOR index FROM zero TO n MINUS one  
            APPEND list consisting of zero THEN index PLUS one THEN element at position index of wells TO pipes  
        END FOR  
        SORT pipes IN ORDER OF the element at position three of each sublist in ascending order  
        SET parent TO list of integers FROM zero TO n inclusive  
        FUNCTION find WITH PARAMETER x  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find WITH PARAMETER element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union WITH PARAMETERS x AND y  
            SET rootX TO find WITH PARAMETER x  
            SET rootY TO find WITH PARAMETER y  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each sublist consisting of house1 THEN house2 THEN cost IN pipes  
            IF union WITH PARAMETERS house1 AND house2  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
