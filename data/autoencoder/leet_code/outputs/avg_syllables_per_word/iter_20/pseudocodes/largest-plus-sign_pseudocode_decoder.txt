CLASS Solution  
    FUNCTION orderOfLargestPlusSign WITH PARAMETERS n AND mines  
        SET mines_set TO a set created by CONVERTING each element of mines TO a tuple  

        SET dp TO a two dimensional collection of size n by n WHERE each element IS a list of four zeroes representing four directions  

        SET directions TO a list containing four pairs representing right, down, left, and up directions respectively as zero and one increments  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair consisting of i AND j IS NOT IN mines_set  
                    IF j IS GREATER THAN zero  
                        SET element at position zero of the list at position j of the list at position i of dp TO element at position zero of the list at position j MINUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero  
                        SET element at position one of the list at position j of the list at position i of dp TO element at position one of the list at position j of the list at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        SET result TO zero  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair consisting of i AND j IS NOT IN mines_set  
                    IF j IS LESS THAN n MINUS one  
                        SET element at position two of the list at position j of the list at position i of dp TO element at position two of the list at position j PLUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one  
                        SET element at position three of the list at position j of the list at position i of dp TO element at position three of the list at position j of the list at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value of the list at position j of the list at position i of dp  
                    SET result TO the maximum of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
