CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping from each course in range from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros with LENGTH numCourses  
        FOR each pair of elements dest AND src IN prerequisites  
            APPEND element dest TO the list at key src in adj_list  
            INCREMENT element at position dest of in_degrees BY one  
        END FOR  
        SET queue TO a double ended queue initialized with all courses in range from zero TO numCourses MINUS one WHERE the element at position course of in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        WHILE queue IS NOT empty  
            SET current TO the element removed from the left end of queue  
            APPEND current TO topological_order  
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT element at position neighbor of in_degrees BY one  
                IF element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
