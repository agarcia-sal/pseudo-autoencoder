CLASS Solution  
    FUNCTION canPartitionKSubsets(nums AS List OF integers, k AS integer) RETURNS boolean  
        SET total_sum TO the SUM OF all elements IN nums  
        IF total_sum MODULO k IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending ORDER  
        
        FUNCTION can_partition(index AS integer, k AS integer, current_sum AS integer, used AS list of booleans) RETURNS boolean  
            IF k EQUALS zero  
                RETURN True  
            END IF  
            IF current_sum EQUALS target  
                RETURN the RESULT OF can_partition WITH index SET TO zero AND k DECREMENTED BY one AND current_sum SET TO zero AND used AS GIVEN  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF element at position i of used IS False AND current_sum PLUS element at position i of nums IS LESS THAN OR EQUAL TO target  
                    SET element at position i of used TO True  
                    IF the RESULT OF can_partition WITH i INCREMENTED BY one AND k AS GIVEN AND current_sum PLUS element at position i of nums AND used AS GIVEN IS True  
                        RETURN True  
                    END IF  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list of boolean False values with LENGTH EQUAL TO the LENGTH OF nums  
        RETURN the RESULT OF can_partition WITH index zero AND k AS GIVEN AND current_sum zero AND used AS GIVEN  
    END FUNCTION  
END CLASS
