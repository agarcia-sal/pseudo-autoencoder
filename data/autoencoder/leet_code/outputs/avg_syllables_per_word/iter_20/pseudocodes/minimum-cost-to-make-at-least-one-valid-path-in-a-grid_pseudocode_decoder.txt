CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list containing tuple zero and one tuple zero and negative one tuple one and zero tuple negative one and zero  
        SET costs TO a two dimensional list with m rows each containing n elements each element SET TO positive infinity  
        SET element at position zero row and position zero column of costs TO zero  
        
        SET pq TO list containing a tuple zero zero zero  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE the element with the smallest first element from pq ASSIGN TO cost and i and j  
            
            IF tuple i and j is IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            ADD tuple i and j TO visited  
            
            IF i EQUALS m MINUS one AND j EQUALS n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each k and tuple di and dj IN directions WITH INDEX  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                    SET new_cost TO cost PLUS one IF k PLUS one DOES NOT EQUAL element at position i row and position j column of grid ELSE zero  
                    IF new_cost IS LESS THAN element at position ni row and position nj column of costs  
                        SET element at position ni row and position nj column of costs TO new_cost  
                        INSERT the tuple new_cost and ni and nj INTO pq maintaining heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
