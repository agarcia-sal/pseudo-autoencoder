CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO create an iterator over s  
            RETURN whether every character in word EXISTS IN the sequential elements produced by iterator  
        END FUNCTION  
        
        SORT dictionary BY length of element in descending order AND lexicographical order in ascending order  
        
        FOR each word IN dictionary  
            IF the result of is_subsequence with word and s EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
