CLASS Solution  
    FUNCTION canTransform(start string, end string) RETURNS boolean  
        SET start_filtered TO an empty list  
        FOR each index i AND character c IN start by their sequential positions  
            IF character c is NOT EQUAL TO the letter X  
                APPEND the pair consisting of character c and index i TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO an empty list  
        FOR each index i AND character c IN end by their sequential positions  
            IF character c is NOT EQUAL TO the letter X  
                APPEND the pair consisting of character c and index i TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered is NOT EQUAL TO the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair (character c1 AND index i1) AND pair (character c2 AND index i2) IN both start_filtered AND end_filtered simultaneously  
            IF character c1 is NOT EQUAL TO character c2  
                RETURN false  
            END IF  
            IF character c1 is EQUAL TO the letter L AND index i1 is LESS THAN index i2  
                RETURN false  
            END IF  
            IF character c1 is EQUAL TO the letter R AND index i1 is GREATER THAN index i2  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
