CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO SORTED list composed of the first element of each element in flowers  
        SET end_times TO SORTED list composed of the second element of each element in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the number of elements in start_times that are less than OR EQUAL TO person using binary right insertion point  
            SET ended TO the number of elements in end_times that are less than person using binary left insertion point  
            SET count_of_flowers_in_bloom TO started MINUS ended  
            APPEND count_of_flowers_in_bloom TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
