CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional list with rows COUNT OF characters in s PLUS one and columns COUNT OF characters in p PLUS one initialized to False  
        SET element at position zero zero of dp TO True  
        FOR each index j FROM one TO COUNT OF characters in p  
            IF the character at position j MINUS one of p EQUALS the character represented by star  
                SET element at position zero j of dp TO element at position zero j MINUS one of dp  
            END IF  
        END FOR  
        FOR each index i FROM one TO COUNT OF characters in s  
            FOR each index j FROM one TO COUNT OF characters in p  
                IF the character at position j MINUS one of p EQUALS the character represented by star  
                    SET element at position i j of dp TO element at position i MINUS one j of dp OR element at position i j MINUS one of dp  
                ELSE IF the character at position j MINUS one of p EQUALS the character represented by question mark OR the character at position i MINUS one of s EQUALS the character at position j MINUS one of p  
                    SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position COUNT OF characters in s COUNT OF characters in p of dp  
    END FUNCTION  
END CLASS
