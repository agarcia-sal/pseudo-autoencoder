CLASS Solution  
    FUNCTION asteroidCollision WITH asteroids  
        SET stack TO an empty list  
        FOR each asteroid IN asteroids  
            WHILE stack is not empty AND asteroid is less than zero AND the element at position the LENGTH OF stack MINUS one of stack is greater than zero  
                IF the element at position the LENGTH OF stack MINUS one of stack is less than the negation of asteroid  
                    REMOVE the last element FROM stack  
                    CONTINUE to the next iteration of the while loop  
                ELSE IF the element at position the LENGTH OF stack MINUS one of stack EQUALS the negation of asteroid  
                    REMOVE the last element FROM stack  
                END IF  
                BREAK from the while loop  
            END WHILE  
            ELSE  
                APPEND asteroid TO stack  
            END ELSE  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
