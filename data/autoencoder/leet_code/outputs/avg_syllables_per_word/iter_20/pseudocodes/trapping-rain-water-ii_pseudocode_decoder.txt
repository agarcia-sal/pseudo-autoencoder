CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap IS EMPTY OR element at position zero of heightMap IS EMPTY THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a two dimensional list of FALSE with m rows and n columns  
        SET min_heap TO an empty list  
        
        FOR each index i FROM zero TO m MINUS one  
            PUSH onto min_heap the tuple consisting of element at position i and zero of heightMap heightMap value WITH i AND zero  
            PUSH onto min_heap the tuple consisting of element at position i and position n MINUS one of heightMap heightMap value WITH i AND position n MINUS one  
            SET element at position i and zero of visited TO TRUE  
            SET element at position i and position n MINUS one of visited TO TRUE  
        END FOR  
        
        FOR each index j FROM zero TO n MINUS one  
            PUSH onto min_heap the tuple consisting of element at position zero and position j of heightMap value WITH zero AND j  
            PUSH onto min_heap the tuple consisting of element at position m MINUS one and position j of heightMap value WITH position m MINUS one AND j  
            SET element at position zero and position j of visited TO TRUE  
            SET element at position m MINUS one and position j of visited TO TRUE  
        END FOR  
        
        SET directions TO the list containing four tuples of values negative one and zero, one and zero, zero and negative one, zero and one  
        SET water_trapped TO zero  
        
        WHILE min_heap IS NOT EMPTY  
            REMOVE the smallest element from min_heap ASSIGNING to height x and y  
            
            FOR each tuple dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx and ny of visited IS FALSE THEN  
                    SET element at position nx and ny of visited TO TRUE  
                    INCREMENT water_trapped BY the MAXIMUM of zero AND height MINUS element at position nx and ny of heightMap  
                    PUSH onto min_heap the tuple consisting of the MAXIMUM of height AND element at position nx and ny of heightMap WITH nx AND ny  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
