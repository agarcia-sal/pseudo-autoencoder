CLASS Solution  
    FUNCTION rotatedDigits WITH PARAMETER n  
        SET change_digits TO the set containing the elements two five six nine  
        SET same_digits TO the set containing the elements zero one eight  
        
        FUNCTION is_good_number WITH PARAMETER num  
            SET num_str TO the string representation of num  
            IF all digits of num_str are members of the union of change_digits and same_digits THEN  
                IF any digit of num_str is a member of change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO the sum of is_good_number of i FOR each i FROM one TO n  
        RETURN good_count  
    END FUNCTION  
END CLASS
