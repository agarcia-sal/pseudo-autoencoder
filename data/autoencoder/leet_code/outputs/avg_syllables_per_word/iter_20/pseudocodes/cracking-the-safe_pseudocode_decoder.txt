CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO empty collection of unique items  
        SET sequence TO empty list  

        FUNCTION dfs(node)  
            FOR each number i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge IS NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH the substring from position one TO the end of edge  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO string of zero REPEATED n MINUS one TIMES  
        CALL dfs WITH start_node  
        RETURN concatenation of the elements of sequence in reversed order AND start_node  
    END FUNCTION  
END CLASS
