CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS open parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                REMOVE last element FROM stack AND ASSIGN TO previous_score  
                IF score EQUALS zero  
                    SET score TO previous_score PLUS one  
                ELSE  
                    SET score TO previous_score PLUS two MULTIPLIED BY score  
                END IF  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
