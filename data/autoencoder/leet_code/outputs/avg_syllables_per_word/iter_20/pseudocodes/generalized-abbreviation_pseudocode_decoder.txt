CLASS Solution  
    FUNCTION generateAbbreviations WITH PARAMETERS word  
        FUNCTION backtrack WITH PARAMETERS index path count  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all ELEMENTS IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index PLUS one AS index  
                    AND path APPENDED WITH the STRING REPRESENTATION OF count AND the ELEMENT at position index OF word AS path  
                    AND zero AS count  
                ELSE  
                    CALL backtrack WITH index PLUS one AS index  
                    AND path APPENDED WITH the ELEMENT at position index OF word AS path  
                    AND zero AS count  
                END IF  
                CALL backtrack WITH index PLUS one AS index  
                AND path AS path  
                AND count PLUS one AS count  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero AS index AND an empty list AS path AND zero AS count  
        RETURN result  
    END FUNCTION  
END CLASS
