CLASS Solution  
    FUNCTION movesToChessboard WITH PARAMETERS board  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR index i FROM zero TO N MINUS one  
            FOR index j FROM zero TO N MINUS one  
                IF the element at position i of board AT position j XOR the element at position i of board AT position zero XOR the element at position zero of board AT position j XOR the element at position zero of board AT position zero DOES NOT EQUAL zero  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the element at position i of board AT position zero  
            INCREMENT colSum BY the element at position zero of board AT position i  
            INCREMENT rowSwap BY one IF the element at position i of board AT position zero EQUALS the remainder of i DIVIDED BY two ELSE increment by zero  
            INCREMENT colSwap BY one IF the element at position zero of board AT position i EQUALS the remainder of i DIVIDED BY two ELSE increment by zero  
        END FOR  
        IF the integer division of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two EQUALS one  
            IF the remainder of rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        SET totalSwaps TO the integer division of the SUM OF rowSwap AND colSwap BY two  
        RETURN totalSwaps  
    END FUNCTION  
END CLASS
