CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN true  
            END IF  
            SET expected TO the CONCATENATION OF first PLUS second CONVERTED TO STRING  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING of num from position start TO position start PLUS length MINUS one NOT EQUALS expected  
                RETURN false  
            END IF  
            RETURN the RESULT OF is_valid_sequence with parameters start PLUS length, second, AND the INTEGER CONVERSION OF expected  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the SUBSTRING of num from position zero TO position i MINUS one  
                SET second TO the SUBSTRING of num from position i TO position j MINUS one  
                IF the LENGTH OF first GREATER THAN one AND the FIRST CHARACTER of first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the FIRST CHARACTER of second EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                IF the RESULT OF is_valid_sequence with parameters j, the INTEGER CONVERSION OF first, AND the INTEGER CONVERSION OF second IS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
