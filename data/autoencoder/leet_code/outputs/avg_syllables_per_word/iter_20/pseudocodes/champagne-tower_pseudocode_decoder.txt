CLASS Solution  
    FUNCTION champagneTower(poured query_row query_glass)  
        SET current_row TO a list containing only poured  
        FOR each row FROM one TO query_row  
            SET next_row TO a list of zeros with the length equal to row plus one  
            FOR each index FROM zero TO the LENGTH OF current_row MINUS one  
                IF element at position index of current_row IS GREATER THAN one  
                    SET excess TO element at position index of current_row MINUS one DIVIDED BY two  
                    INCREMENT element at position index of next_row BY excess  
                    INCREMENT element at position index plus one of next_row BY excess  
                END IF  
            END FOR  
            SET current_row TO next_row  
        END FOR  
        IF element at position query_glass of current_row IS LESS THAN one  
            RETURN element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
