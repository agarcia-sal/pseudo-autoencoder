CLASS Solution  
    FUNCTION getPermutation(n, k)  
        SET numbers TO list containing each number FROM one TO n  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n TO one STEP minus one  
            SET fact TO factorial OF i MINUS one  
            SET index TO k DIVIDED BY fact  
            APPEND element at position index OF numbers TO result AS string  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        RETURN concatenation OF all elements IN result AS single string  
    END FUNCTION  
END CLASS
