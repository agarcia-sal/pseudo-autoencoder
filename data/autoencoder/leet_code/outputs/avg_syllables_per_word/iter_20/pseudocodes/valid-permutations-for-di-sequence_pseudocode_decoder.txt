CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeros sized n PLUS one BY n PLUS one  
        
        FOR each j FROM zero TO n  
            SET element at position j of element at position zero of dp TO one  
        END FOR  
        
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            IF character at position i MINUS one of s EQUALS the character I  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO the sum of prefix_sum PLUS element at position j of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO the sum of prefix_sum PLUS element at position j PLUS one of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the sum of element at position n of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
