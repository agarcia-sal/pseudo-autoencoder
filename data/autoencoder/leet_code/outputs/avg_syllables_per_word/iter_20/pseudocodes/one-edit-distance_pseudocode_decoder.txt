CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s IS GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF CALLING isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff IS GREATER THAN one  
            RETURN false  
        END IF  
        
        IF diff EQUALS zero  
            RETURN whether the TOTAL NUMBER OF POSITIONS WHERE the CHARACTER AT POSITION IN s IS NOT EQUAL TO the CHARACTER AT THE SAME POSITION IN t IS EQUAL TO one  
        ELSE  
            FOR each index FROM zero TO the LENGTH OF s MINUS one  
                IF the CHARACTER AT POSITION index OF s IS NOT EQUAL TO the CHARACTER AT POSITION index OF t  
                    RETURN whether the SUBSTRING FROM POSITION index TO THE END OF s IS EQUAL TO the SUBSTRING FROM POSITION index PLUS one TO THE END OF t  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
