CLASS Solution  
    FUNCTION repeatedStringMatch(a string, b string) RETURNS integer  
        SET minimum_repetitions TO the negative of the negative value of the LENGTH OF string b DIVIDED BY the LENGTH OF string a rounded down to the nearest integer higher than or equal to the division result  
        IF string b IS A SUBSTRING OF string a REPEATED minimum_repetitions TIMES  
            RETURN minimum_repetitions  
        END IF  
        IF string b IS A SUBSTRING OF string a REPEATED minimum_repetitions PLUS one TIMES  
            RETURN minimum_repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
