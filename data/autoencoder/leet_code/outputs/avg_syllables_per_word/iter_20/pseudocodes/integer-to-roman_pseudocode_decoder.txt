CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of integer values one thousand nine hundred and zero plus zero five hundred four hundred one hundred ninety fifty forty ten nine five four one  
        SET syms TO list of Roman numeral strings M CM D CD C XC L XL X IX V IV I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero holds  
            SET repetitions TO the quotient of num DIVIDED BY the element at position i of val  
            FOR count FROM one TO repetitions  
                APPEND the element at position i of syms TO roman_numeral  
                DECREMENT num BY the element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
