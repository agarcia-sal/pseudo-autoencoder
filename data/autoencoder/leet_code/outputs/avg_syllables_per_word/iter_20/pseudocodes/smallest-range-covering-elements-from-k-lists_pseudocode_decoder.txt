CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push FUNCTION WITH min_heap AND a tuple of element at position zero of list at position index of nums AND index AND zero  
            SET max_value TO maximum of max_value AND element at position zero of list at position index of nums  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO a list containing negative one hundred thousand AND one hundred thousand  
        WHILE min_heap is not empty  
            CALL heap pop FUNCTION WITH min_heap AND ASSIGN to min_value AND list_index AND element_index  
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF list at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums  
                CALL heap push FUNCTION WITH min_heap AND a tuple of next_value AND list_index AND element_index PLUS one  
                SET max_value TO maximum of max_value AND next_value  
            ELSE  
                BREAK the loop  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
