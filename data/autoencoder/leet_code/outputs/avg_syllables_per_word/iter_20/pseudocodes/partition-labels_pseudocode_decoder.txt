CLASS Solution  
    FUNCTION partitionLabels WITH PARAMETER s  
        SET last_occurrence TO A DICTIONARY MAPPING EACH character TO ITS LAST POSITION IN s  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO AN EMPTY LIST  
        FOR each position i AND character char IN s  
            SET end TO THE GREATER OF end AND last_occurrence AT character char  
            IF position i EQUALS end  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO position i PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
