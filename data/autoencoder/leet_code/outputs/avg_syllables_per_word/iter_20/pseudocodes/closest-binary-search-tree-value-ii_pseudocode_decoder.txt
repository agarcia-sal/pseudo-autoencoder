CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO new double ended queue container  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i is LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node dot left TO new TreeNode with value equal to element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node dot right TO new TreeNode with value equal to element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN false  
    ELSE  
        RETURN logical AND of is_same_tree of p dot left and q dot left AND is_same_tree of p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS none  
                RETURN  
            END IF  
            CALL inorder_traversal with node dot left  
            IF length of result IS LESS THAN k  
                APPEND node dot val TO result  
            ELSE  
                IF absolute value of node dot val MINUS target IS LESS THAN absolute value of element at position zero of result MINUS target  
                    REMOVE element from the front of result  
                    APPEND node dot val TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with node dot right  
        END FUNCTION  
        SET result TO new double ended queue container  
        CALL inorder_traversal with root  
        RETURN a list converted from result  
    END FUNCTION  
END CLASS
