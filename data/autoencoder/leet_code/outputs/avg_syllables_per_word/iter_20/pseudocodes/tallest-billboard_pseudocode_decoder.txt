CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a mapping from zero TO zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key named d AND value named y IN dp  
                SET sum_position TO d PLUS rod  
                IF value at key sum_position in new_dp IS NOT DEFINED OR value at key sum_position in new_dp IS LESS THAN y THEN  
                    SET value at key sum_position in new_dp TO y  
                END IF  
                SET difference_position TO the absolute value of d MINUS rod  
                SET minimum_height TO the smaller value BETWEEN d AND rod  
                IF value at key difference_position in new_dp IS NOT DEFINED OR value at key difference_position in new_dp IS LESS THAN y PLUS minimum_height THEN  
                    SET value at key difference_position in new_dp TO y PLUS minimum_height  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value at key zero in dp  
    END FUNCTION  
END CLASS
