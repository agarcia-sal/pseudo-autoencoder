CLASS Solution  
    FUNCTION findSubsequences WITH PARAMETERS nums  
        FUNCTION dfs WITH PARAMETERS u last t  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND the element at position u of nums TO t  
                CALL dfs WITH PARAMETERS the value of u PLUS one the element at position u of nums t  
                REMOVE the last element FROM t  
            END IF  
            IF the element at position u of nums NOT EQUAL TO last  
                CALL dfs WITH PARAMETERS the value of u PLUS one last t  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH PARAMETERS zero MINUS one thousand an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
