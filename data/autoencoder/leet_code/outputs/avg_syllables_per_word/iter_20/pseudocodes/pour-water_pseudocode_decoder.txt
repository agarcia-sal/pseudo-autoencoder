CLASS Solution  
    FUNCTION pourWater(heights AS List OF Integer, volume AS Integer, k AS Integer) RETURNS List OF Integer  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN zero AND the element at position i MINUS one of heights IS LESS THAN OR EQUAL TO the element at position i of heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i LESS THAN k AND the element at position i of heights EQUALS the element at position i PLUS one of heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i LESS THAN k  
                SET the element at position i of heights TO the element at position i of heights PLUS one  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND the element at position i PLUS one of heights IS LESS THAN OR EQUAL TO the element at position i of heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i GREATER THAN k AND the element at position i of heights EQUALS the element at position i MINUS one of heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i GREATER THAN k  
                SET the element at position i of heights TO the element at position i of heights PLUS one  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            
            SET the element at position k of heights TO the element at position k of heights PLUS one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
