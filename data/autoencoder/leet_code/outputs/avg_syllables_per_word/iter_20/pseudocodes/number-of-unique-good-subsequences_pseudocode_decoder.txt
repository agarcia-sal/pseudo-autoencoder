CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MODULO TO one billion PLUS seven  
        SET count_of_subsequences_ending_with_zero TO zero  
        SET count_of_subsequences_ending_with_one TO zero  
        SET binary_contains_zero TO whether the binary string CONTAINS the character zero  
        FOR each bit IN the binary string  
            IF the bit EQUALS the character one  
                SET count_of_subsequences_ending_with_one TO the sum of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS one MODULO the MODULO  
            ELSE  
                SET count_of_subsequences_ending_with_zero TO the sum of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one MODULO the MODULO  
            END IF  
        END FOR  
        RETURN the sum of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS binary_contains_zero MODULO the MODULO  
    END FUNCTION  
END CLASS
