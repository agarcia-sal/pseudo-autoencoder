CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode USING the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW DEQUE COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT None  
            SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT None  
            SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree OF the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND is_same_tree OF the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow WITH PARAMETERS self root val depth  
        IF depth IS EQUALS TO one  
            RETURN A NEW INSTANCE OF TreeNode USING val WITH left ATTRIBUTE ASSIGNED TO root  
        END IF  
        SET queue TO A NEW DEQUE COLLECTION CONTAINING A TUPLE OF root AND one  
        WHILE queue IS NOT empty  
            SET node AND level TO THE FIRST ELEMENT REMOVED FROM queue  
            IF level IS EQUALS TO depth MINUS one  
                SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING val WITH left ATTRIBUTE ASSIGNED TO the left ATTRIBUTE OF node  
                SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING val WITH right ATTRIBUTE ASSIGNED TO the right ATTRIBUTE OF node  
            ELSE IF level IS LESS THAN depth MINUS one  
                IF the left ATTRIBUTE OF node IS NOT None  
                    APPEND A TUPLE OF the left ATTRIBUTE OF node AND level PLUS one TO queue  
                END IF  
                IF the right ATTRIBUTE OF node IS NOT None  
                    APPEND A TUPLE OF the right ATTRIBUTE OF node AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
