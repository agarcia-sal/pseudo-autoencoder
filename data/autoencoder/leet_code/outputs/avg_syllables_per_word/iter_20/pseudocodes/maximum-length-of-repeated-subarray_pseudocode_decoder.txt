CLASS Solution  
    FUNCTION findLength(nums1 OF LIST OF INTEGER, nums2 OF LIST OF INTEGER) RETURNS INTEGER  
        SET dp TO A TWO DIMENSIONAL LIST OF INTEGERS WITH THE NUMBER OF ROWS EQUAL TO THE LENGTH OF nums1 PLUS one AND THE NUMBER OF COLUMNS EQUAL TO THE LENGTH OF nums2 PLUS one WHERE EACH ELEMENT IS ZERO  
        SET max_length TO zero  
        FOR i FROM one TO THE LENGTH OF nums1  
            FOR j FROM one TO THE LENGTH OF nums2  
                IF element at position i MINUS one OF nums1 EQUALS element at position j MINUS one OF nums2  
                    SET element at position i and j OF dp TO element at position i MINUS one and j MINUS one OF dp PLUS one  
                    IF element at position i and j OF dp GREATER THAN max_length  
                        SET max_length TO element at position i and j OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
