CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix IS LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS THE SMALLER VALUE BETWEEN the DIFFERENCE OF n AND prefix PLUS one AND the DIFFERENCE OF next_prefix AND prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        SET k TO k MINUS one  
        WHILE k IS GREATER THAN zero  
            SET steps TO count_steps WITH prefix current AND n  
            IF steps IS LESS THAN OR EQUAL TO k  
                SET current TO current PLUS one  
                SET k TO k MINUS steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                SET k TO k MINUS one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
