CLASS Solution  
    FUNCTION fullJustify WITH PARAMETERS words AND maxWidth  
        FUNCTION justify_line WITH PARAMETERS line AND num_of_letters AND is_last_line  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN the concatenation of all elements in line separated by single space FOLLOWED BY SPACES TO make the total LENGTH EQUAL TO maxWidth  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index IS LESS THAN extra_spaces  
                        APPEND SPACES EQUAL TO spaces_between_words PLUS one TO the element at position index of line  
                    ELSE  
                        APPEND SPACES EQUAL TO spaces_between_words TO the element at position index of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements in line WITHOUT ANY SEPARATOR  
            END IF  
        END FUNCTION  
          
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO zero  
          
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line IS GREATER THAN maxWidth  
                APPEND the result of calling justify_line WITH line AND num_of_letters AND false TO result  
                SET line TO empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
          
        APPEND the result of calling justify_line WITH line AND num_of_letters AND true TO result  
          
        RETURN result  
    END FUNCTION  
END CLASS
