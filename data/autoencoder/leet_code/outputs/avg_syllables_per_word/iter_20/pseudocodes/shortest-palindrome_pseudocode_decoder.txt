CLASS Solution  
    FUNCTION shortestPalindrome WITH PARAMETER s  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome WITH PARAMETER check_str  
            IF check_str EQUALS reverse of check_str  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FOR index FROM the LENGTH OF s DOWN TO zero BY decrementing one  
            IF is_palindrome WITH PARAMETER substring from position zero TO position index MINUS one of s  
                SET to_add TO reverse of substring from position index TO the end of s  
                RETURN concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
