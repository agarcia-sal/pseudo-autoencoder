CLASS Solution  
    FUNCTION multiply(num1 string, num2 string) RETURNS string  
        IF num1 EQUALS zero OR num2 EQUALS zero  
            RETURN zero  
        END IF  
          
        SET result TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
          
        SET num1 TO the string num1 in reverse order  
        SET num2 TO the string num2 in reverse order  
          
        FOR index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the result of converting the character at position i of num1 to its integer value MULTIPLIED BY the result of converting the character at position j of num2 to its integer value  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                  
                SET the element at position p1 of result TO sum_ MODULO ten  
                INCREMENT the element at position p2 of result BY the quotient of sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
          
        SET result_str TO the concatenation of all elements of result in reverse order converted to their string forms  
        RETURN the string result_str with all leading characters equal to zero removed  
    END FUNCTION  
END CLASS
