CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO CALL the FUNCTION subsets WITH the list of elements from position one TO the end of nums  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            APPEND a new list formed by element at position zero of nums FOLLOWED BY subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
