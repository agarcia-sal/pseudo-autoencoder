CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a default dictionary mapping each key to an empty list  
        FOR each pair of departure and destination IN the tickets sorted in reversed order  
            APPEND the destination TO the list associated with the departure in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph is not empty  
                CALL visit with the element removed from the end of the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit with the string letter J F K  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
