CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE TO next iteration  
                END IF  
                CALL backtrack WITH parameters i PLUS one, target MINUS element at position i of candidates, path WITH element at position i of candidates APPENDED  
            END FOR  
        END FUNCTION  
        SORT candidates IN NON-DECREASING order  
        SET result TO empty list  
        CALL backtrack WITH parameters zero, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
