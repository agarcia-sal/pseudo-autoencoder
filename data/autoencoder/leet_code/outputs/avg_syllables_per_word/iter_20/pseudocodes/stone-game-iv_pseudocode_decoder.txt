CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of False with length n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j IS LESS THAN OR EQUAL TO i  
                IF NOT element at position i MINUS j MULTIPLIED BY j of dp IS True  
                    SET element at position i of dp TO True  
                    BREAK the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
