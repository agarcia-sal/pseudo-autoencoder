CLASS Solution  
    FUNCTION findMaxForm(strs AS List OF string, m AS integer, n AS integer) RETURNS integer  
        SET dp TO a two dimensional list of size m plus one by n plus one filled with zeros  
        FOR each s IN strs  
            SET count_0 TO the number of occurrences of character zero IN string s  
            SET count_1 TO the number of occurrences of character one IN string s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET value_at_i_j TO element at position i of dp followed by element at position j  
                    SET value_at_i_minus_count_0_j_minus_count_1 TO element at position i minus count_0 of dp followed by element at position j minus count_1  
                    SET dp element at position i position j TO the greater of value_at_i_j AND value_at_i_minus_count_0_j_minus_count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position m of dp followed by element at position n  
    END FUNCTION  
END CLASS
