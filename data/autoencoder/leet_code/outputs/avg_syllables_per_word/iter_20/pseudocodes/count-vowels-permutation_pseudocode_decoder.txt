CLASS Solution  
    FUNCTION countVowelPermutation WITH PARAMETER n OF TYPE integer  
        SET MODULO TO ten RAISED TO THE POWER of nine PLUS seven  
        SET count_of_a TO one  
        SET count_of_e TO one  
        SET count_of_i TO one  
        SET count_of_o TO one  
        SET count_of_u TO one  
        FOR each number FROM two TO n  
            SET next_count_of_a TO sum of count_of_e PLUS count_of_i PLUS count_of_u  
            SET next_count_of_e TO sum of count_of_a PLUS count_of_i  
            SET next_count_of_i TO sum of count_of_e PLUS count_of_o  
            SET next_count_of_o TO count_of_i  
            SET next_count_of_u TO sum of count_of_i PLUS count_of_o  
            SET count_of_a TO remainder of next_count_of_a DIVIDED BY MODULO  
            SET count_of_e TO remainder of next_count_of_e DIVIDED BY MODULO  
            SET count_of_i TO remainder of next_count_of_i DIVIDED BY MODULO  
            SET count_of_o TO remainder of next_count_of_o DIVIDED BY MODULO  
            SET count_of_u TO remainder of next_count_of_u DIVIDED BY MODULO  
        END FOR  
        SET total_count TO sum of count_of_a PLUS count_of_e PLUS count_of_i PLUS count_of_o PLUS count_of_u  
        SET result TO remainder of total_count DIVIDED BY MODULO  
        RETURN result  
    END FUNCTION  
END CLASS
