CLASS Solution  
    FUNCTION minSkips with parameters dist and speed and hoursBefore  
        SET n TO the LENGTH OF dist  
        INITIALIZE dp as a list of lists where each inner list has n plus one elements all SET TO mathematical infinity, and there are n such inner lists  
        SET total_dist TO the SUM OF all elements of dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at position zero and position zero of dp TO the element at position zero of dist  
        FOR each integer j FROM one TO n MINUS one DO  
            SET element at position zero and position j of dp TO the element at position zero of dist  
        END FOR  
        
        FOR each integer i FROM one TO n MINUS one DO  
            FOR each integer j FROM zero TO i DO  
                IF j IS LESS THAN i THEN  
                    SET element at position i and position j of dp TO the CEILING OF element at position i MINUS one and position j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at position i and position j of dp TO the MINIMUM OF element at position i and position j of dp AND element at position i MINUS one and position j MINUS one of dp PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer j FROM zero TO n MINUS one DO  
            IF element at position n MINUS one and position j of dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
