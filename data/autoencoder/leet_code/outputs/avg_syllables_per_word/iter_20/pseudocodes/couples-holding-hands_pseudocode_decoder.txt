CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        CREATE a dictionary named position mapping each person TO their index IN row  
        SET swaps TO zero  
        FOR index FROM zero TO the LENGTH OF row MINUS one STEP two  
            SET correct_pair TO the value at position index of row EXCLUSIVE OR one  
            IF the value at position index PLUS one of row IS NOT EQUAL TO correct_pair  
                SET swap_idx TO position AT correct_pair  
                SWAP the value at position index PLUS one of row WITH the value at position swap_idx of row  
                SET position AT the value at position swap_idx of row TO swap_idx  
                SET position AT the value at position index PLUS one of row TO index PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
