CLASS ListNode  
    FUNCTION __init__ WITH parameters val defaulted TO zero AND next defaulted TO none  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one TO the end  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the val attribute of p1 EQUALS the val attribute of p2 AND is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH parameter head  
        FUNCTION reverseList WITH parameter node  
            SET prev TO none  
            WHILE node IS NOT none  
                SET next_node TO the next attribute of node  
                SET the next attribute of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList CALLED WITH head  
        
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT none AND carry EQUALS one  
            SET the val attribute of current TO the val attribute of current PLUS carry  
            SET carry TO the val attribute of current DIVIDED BY ten  
            SET the val attribute of current TO the val attribute of current MODULO ten  
            
            IF the next attribute of current IS none AND carry EQUALS one  
                SET the next attribute of current TO a new ListNode initialized with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next attribute of current  
        END WHILE  
        
        RETURN reverseList CALLED WITH reversed_head  
    END FUNCTION  
END CLASS
