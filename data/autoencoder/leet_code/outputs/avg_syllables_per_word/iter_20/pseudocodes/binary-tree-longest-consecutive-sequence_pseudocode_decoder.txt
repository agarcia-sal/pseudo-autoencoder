CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE and ASSIGN the leftmost element of queue TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the element val of p NOT EQUALS the element val of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments element left of p AND element left of q AND the result of is_same_tree with arguments element right of p AND element right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root IS none  
            RETURN zero  
        END IF  

        FUNCTION dfs(node, parent, current_length)  
            IF node IS none  
                RETURN current_length  
            END IF  

            IF parent IS NOT none AND the element val of node EQUALS parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  

            SET left_length TO the result of dfs with arguments element left of node, the element val of node, current_length  
            SET right_length TO the result of dfs with arguments element right of node, the element val of node, current_length  

            RETURN the maximum value among current_length, left_length, right_length  
        END FUNCTION  

        RETURN dfs with arguments root, none, zero  
    END FUNCTION  
END CLASS
