CLASS Solution  
    FUNCTION amountPainted WITH PARAMETER paint  
        SET max_end TO the maximum value of end FROM each start and end pair IN paint  
        SET painted TO a list of value minus one with length equal TO max_end PLUS one  
        SET worklog TO an empty list  
        
        FOR each start and end pair IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            
            WHILE current_start IS LESS THAN end  
                IF the element at position current_start of painted EQUALS minus one  
                    SET the element at position current_start of painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO the element at position current_start of painted  
                END IF  
            END WHILE  
            
            APPEND new_paint TO worklog  
        END FOR  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
