CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER n  
        SET parent TO list of integers FROM zero TO n inclusive  
        SET rank TO list of zero repeated n plus one times  
        SET count TO n  
    END FUNCTION  

    FUNCTION find WITH PARAMETER u  
        IF element at position u of parent NOT EQUALS u  
            SET element at position u of parent TO result of find FUNCTION with element at position u of parent  
        END IF  
        RETURN element at position u of parent  
    END FUNCTION  

    FUNCTION union WITH PARAMETERS u v  
        SET rootU TO result of find FUNCTION with u  
        SET rootV TO result of find FUNCTION with v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of rank GREATER THAN element at position rootV of rank  
                SET element at position rootV of parent TO rootU  
            ELSE IF element at position rootU of rank LESS THAN element at position rootV of rank  
                SET element at position rootU of parent TO rootV  
            ELSE  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of rank BY one  
            END IF  
            DECREMENT count BY one  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove WITH PARAMETERS n edges  
        SET uf_common TO new UnionFind initialized with n  
        SET uf_alice TO new UnionFind initialized with n  
        SET uf_bob TO new UnionFind initialized with n  

        SET used_edges TO zero  

        FOR each list containing three elements t u v IN edges  
            IF t EQUALS three  
                IF result of union FUNCTION of uf_common with parameters u v IS true  
                    CALL union FUNCTION of uf_alice with parameters u v  
                    CALL union FUNCTION of uf_bob with parameters u v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  

        FOR each list containing three elements t u v IN edges  
            IF t EQUALS one AND result of union FUNCTION of uf_alice with parameters u v IS true  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND result of union FUNCTION of uf_bob with parameters u v IS true  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  

        IF count property of uf_alice EQUALS one AND count property of uf_bob EQUALS one  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
