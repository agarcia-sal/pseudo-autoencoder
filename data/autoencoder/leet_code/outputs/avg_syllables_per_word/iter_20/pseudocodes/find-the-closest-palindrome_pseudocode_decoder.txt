CLASS Solution  
    FUNCTION nearestPalindromic WITH PARAMETER n  
        SET length TO the LENGTH of n  
        IF length EQUALS one  
            RETURN the STRING representation of the INTEGER value of n MINUS one  
        END IF  
        
        INITIALIZE candidates TO an EMPTY set  
        
        ADD the INTEGER value of ten RAISED TO the power of length PLUS one TO candidates  
        ADD the INTEGER value of ten RAISED TO the power of length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER value of the SUBSTRING from POSITION zero TO the result of the INTEGER division of the sum of length PLUS one BY two of n  
        
        FOR i FROM negative one TO positive one  
            SET new_prefix TO the STRING representation of the result of prefix PLUS i  
            IF length MODULO two EQUALS zero  
                SET candidate TO the concatenation of new_prefix AND the REVERSE of new_prefix  
            ELSE  
                SET candidate TO the concatenation of new_prefix AND the REVERSE of the SUBSTRING from the SECOND LAST position TO the FIRST position inclusive stepping BACKWARDS of new_prefix  
            END IF  
            ADD the INTEGER value of candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER value of n  
        REMOVE original_number FROM candidates IF it EXISTS  
        
        SET closest TO the element in candidates WITH the MINIMUM value according to the SORT KEY defined as the tuple of the ABSOLUTE value of the difference BETWEEN candidate AND original_number AND the candidate itself  
        
        RETURN the STRING representation of closest  
    END FUNCTION  
END CLASS
