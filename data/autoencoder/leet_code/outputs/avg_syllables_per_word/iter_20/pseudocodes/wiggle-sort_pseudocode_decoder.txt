CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true  
        FOR index FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less EQUALS true  
                IF element at position index of nums GREATER THAN element at position index PLUS one of nums  
                    SET temporary_variable TO element at position index of nums  
                    SET element at position index of nums TO element at position index PLUS one of nums  
                    SET element at position index PLUS one of nums TO temporary_variable  
                END IF  
            ELSE  
                IF element at position index of nums LESS THAN element at position index PLUS one of nums  
                    SET temporary_variable TO element at position index of nums  
                    SET element at position index of nums TO element at position index PLUS one of nums  
                    SET element at position index PLUS one of nums TO temporary_variable  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
