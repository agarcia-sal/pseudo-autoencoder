CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO a list of tuples each containing the element at position of indices the element at the same position of sources and the element at the same position of targets sorted in ascending order by the first element of each tuple  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each triple index source target IN replacements  
            APPEND the substring from position prev_end TO one less than position index of string s TO result  
            IF substring from position index TO position index PLUS the length of string source MINUS one of string s EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring from position index TO position index PLUS the length of string source MINUS one of string s TO result  
            END IF  
            SET prev_end TO index PLUS the length of string source  
        END FOR  
        APPEND the substring from position prev_end TO the end of string s TO result  
        RETURN the concatenation of all elements of result as a single string  
    END FUNCTION  
END CLASS
