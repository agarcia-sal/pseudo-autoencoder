CLASS Solution  
    FUNCTION decodeAtIndex(s AS string, k AS integer) RETURNS string  
        SET length TO zero  
        FOR each char IN s  
            IF char IS DIGIT  
                SET length TO length MULTIPLIED BY NUMERIC VALUE OF char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN s TRAVERSED IN REVERSE ORDER  
            SET k TO k MODULO length  
            IF k EQUALS zero AND char IS ALPHABETIC CHARACTER  
                RETURN char  
            END IF  
            IF char IS DIGIT  
                SET length TO length DIVIDED BY NUMERIC VALUE OF char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
