CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO TWO TO THE POWER OF n MINUS one  
            SET good TO a list of FALSE values with LENGTH n  
            SET bad TO a list of FALSE values with LENGTH n  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND TWO TO THE POWER OF j IS NOT EQUAL TO zero  
                    SET the ELEMENT at position j OF good TO TRUE  
                ELSE  
                    SET the ELEMENT at position j OF bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR j FROM zero TO n MINUS one  
                IF the ELEMENT at position j OF good IS TRUE  
                    FOR k FROM zero TO n MINUS one  
                        IF the ELEMENT at position j OF statements EQUALS zero AND the ELEMENT at position k OF good IS TRUE  
                            SET valid TO FALSE  
                            BREAK  
                        ELSE IF the ELEMENT at position j OF statements EQUALS one AND the ELEMENT at position k OF bad IS TRUE  
                            SET valid TO FALSE  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS TRUE  
                SET max_good TO the MAXIMUM OF max_good AND the SUM OF values in good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
