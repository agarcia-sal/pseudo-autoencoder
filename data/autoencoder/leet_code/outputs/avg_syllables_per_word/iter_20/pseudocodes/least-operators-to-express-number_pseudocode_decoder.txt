CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        DEFINE FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            
            IF target LESS THAN x  
                SET first_value TO two MULTIPLIED BY target MINUS one  
                SET second_value TO two MULTIPLIED BY x MINUS target  
                RETURN the MINIMUM OF first_value AND second_value  
            END IF  
            
            SET power TO one  
            WHILE x RAISED TO THE power IS LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            
            SET res TO dp OF target MINUS x RAISED TO THE power PLUS power  
            
            IF x RAISED TO THE power PLUS one MINUS target LESS THAN target  
                SET candidate TO dp OF x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
                SET res TO the MINIMUM OF res AND candidate  
            END IF  
            
            RETURN res  
        END FUNCTION  
        
        RETURN dp OF target  
    END FUNCTION  
END CLASS
