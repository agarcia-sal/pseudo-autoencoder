CLASS Solution  
    FUNCTION checkInclusion WITH PARAMETERS s1 AS string AND s2 AS string RETURNS boolean  
        IMPORT the class Counter FROM the collection utilities  
        
        SET len_s1 TO the LENGTH OF s1  
        SET len_s2 TO the LENGTH OF s2  
        
        IF the length of s1 IS GREATER THAN the length of s2  
            RETURN false  
        END IF  
        
        SET count_s1 TO the frequency count of each character in s1 using Counter  
        SET count_s2 TO the frequency count of each character at position from zero TO len_s1 MINUS one of s2 using Counter  
        
        IF count_s1 EQUALS count_s2  
            RETURN true  
        END IF  
        
        FOR index i FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of the character at position i of s2 in count_s2 BY one  
            
            DECREMENT the count of the character at position i MINUS len_s1 of s2 in count_s2 BY one  
            
            IF the count of the character at position i MINUS len_s1 of s2 in count_s2 EQUALS zero  
                REMOVE the character at position i MINUS len_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
