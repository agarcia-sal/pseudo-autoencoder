CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list OF string representations OF each element IN nums  
        
        FUNCTION compare(x, y)  
            IF concatenation OF x FOLLOWED BY y IS GREATER THAN concatenation OF y FOLLOWED BY x  
                RETURN minus one  
            ELSE IF concatenation OF x FOLLOWED BY y IS LESS THAN concatenation OF y FOLLOWED BY x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING comparison function compare  
        
        IF element at position zero of nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        RETURN concatenation OF all elements IN nums_str IN ORDER  
    END FUNCTION  
END CLASS
