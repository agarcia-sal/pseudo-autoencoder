CLASS Solution  
    FUNCTION fourSumCount(nums1 list of integer nums2 list of integer nums3 list of integer nums4 list of integer) RETURNS integer  
        SET sum_ab TO an empty dictionary mapping integer TO integer  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT the value at current_sum IN sum_ab BY one  
                ELSE  
                    SET the value at current_sum IN sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO the NEGATIVE OF c PLUS d  
                IF target EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT count BY the value at target IN sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
