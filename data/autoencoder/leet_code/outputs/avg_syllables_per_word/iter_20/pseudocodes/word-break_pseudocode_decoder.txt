CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the SET constructed from wordDict  
        SET dp TO a LIST of BOOLEAN values initialized to FALSE with the LENGTH OF s PLUS one  
        SET element at position zero of dp TO TRUE  
        
        FOR index i FROM one TO the LENGTH OF s  
            FOR index j FROM zero TO one LESS THAN i  
                IF element at position j of dp IS TRUE AND substring from position j TO i of s EXISTS IN wordSet  
                    SET element at position i of dp TO TRUE  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
