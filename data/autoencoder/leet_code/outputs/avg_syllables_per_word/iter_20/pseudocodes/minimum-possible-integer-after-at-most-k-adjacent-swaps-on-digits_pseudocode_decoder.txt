CLASS Solution  
    FUNCTION minInteger(num STRING, k INTEGER) RETURNS STRING  
        SET num TO list converted from input num string  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the MINIMUM of i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the sublist from position i TO min_index inclusive of num TO a list consisting of element at position min_index of num FOLLOWED BY the sublist from position i TO min_index exclusive of num  
                    DECREMENT k BY the result of min_index MINUS i  
                    SET swapped TO true  
                    BREAK out of the for loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS false  
                BREAK out of the while loop  
            END IF  
        END WHILE  
        
        RETURN concatenation of all elements in num  
    END FUNCTION  
END CLASS
