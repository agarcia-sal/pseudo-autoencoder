CLASS Solution  
    FUNCTION solveEquation WITH parameter equation  
        FUNCTION parse_side WITH parameter side  
            SET tokens TO side WITH each plus SIGN replaced by hash PLUS SIGN AND each minus SIGN replaced by hash MINUS SIGN THEN split by hash  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS empty STRING  
                    CONTINUE to next iteration  
                END IF  
                IF token CONTAINS letter x  
                    IF token EQUALS letter x OR token EQUALS PLUS SIGN letter x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS MINUS SIGN letter x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY integer CONVERSION OF token WITH letter x REMOVED  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY integer CONVERSION OF token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  

        ASSIGN left_side AND right_side TO THE TWO PARTS OF equation SPLIT BY equal SIGN  
        ASSIGN left_x AND left_num TO RESULT OF parse_side APPLIED TO left_side  
        ASSIGN right_x AND right_num TO RESULT OF parse_side APPLIED TO right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN string Infinite solutions  
            ELSE  
                RETURN string No solution  
            END IF  
        ELSE  
            SET division_result TO total_num DIVIDED BY total_x AS INTEGER DIVISION  
            RETURN string x EQUALS division_result  
        END IF  
    END FUNCTION  
END CLASS
