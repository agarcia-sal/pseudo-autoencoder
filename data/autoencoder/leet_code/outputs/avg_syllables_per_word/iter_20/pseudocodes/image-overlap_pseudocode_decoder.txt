CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO AN EMPTY LIST  
        FOR i FROM zero TO n MINUS ONE  
            FOR j FROM zero TO n MINUS ONE  
                IF the ELEMENT at position j of the ELEMENT at position i of img1 EQUALS one  
                    APPEND pair of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  

        SET points2 TO AN EMPTY LIST  
        FOR i FROM zero TO n MINUS ONE  
            FOR j FROM zero TO n MINUS ONE  
                IF the ELEMENT at position j of the ELEMENT at position i of img2 EQUALS one  
                    APPEND pair of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  

        SET overlap_count TO a DEFAULT DICTIONARY MAPPING TO zero  

        FOR each pair x1 AND y1 IN points1  
            FOR each pair x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the VALUE ASSOCIATED WITH the pair of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  

        SET maximum_overlap TO zero  
        FOR each value IN the VALUES of overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  

        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
