CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr AS List OF Integer, k AS Integer) RETURNS List OF Integer  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        
        FOR each i FROM zero TO n MINUS one MINUS one  
            ADD the TUPLE consisting of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr together with i and n MINUS one TO min_heap USING a HEAP PUSH operation  
        END FOR  
        
        FOR each _ FROM zero TO k MINUS one MINUS one  
            REMOVE the smallest element FROM min_heap USING a HEAP POP operation AND ASSIGN the returned TUPLE to _, i, j  
            IF the value of j MINUS one IS GREATER THAN i THEN  
                ADD the TUPLE consisting of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr together with i and j MINUS one TO min_heap USING a HEAP PUSH operation  
            END IF  
        END FOR  
        
        REMOVE the smallest element FROM min_heap USING a HEAP POP operation AND ASSIGN the returned TUPLE to _, i, j  
        RETURN a LIST consisting of the element at position i of arr AND the element at position j of arr  
    END FUNCTION  
END CLASS
