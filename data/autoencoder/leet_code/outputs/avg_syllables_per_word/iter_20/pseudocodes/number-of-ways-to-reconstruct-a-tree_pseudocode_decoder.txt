CLASS Solution  
    FUNCTION checkWays(pairs)  
        SET graph TO result of calling auxiliary function defaultdict with argument set  
        FOR each element x followed by element y IN pairs  
            APPEND y TO the set associated with key x in graph  
            APPEND x TO the set associated with key y in graph  
        END FOR  
        SET max_degree TO the element in graph with maximum size of the set associated with it  
        SET n TO the number of keys in graph  
        IF the size of the set associated with max_degree in graph IS NOT EQUAL TO n MINUS one  
            RETURN zero  
        END IF  
        FUNCTION can_be_parent(u, v)  
            RETURN the result of checking that for each neighbor IN the set associated with key v in graph each neighbor is IN the set associated with key u in graph OR neighbor EQUALS u  
        END FUNCTION  
        SET ways TO one  
        FOR each element u IN graph  
            IF u EQUALS max_degree  
                CONTINUE to next iteration  
            END IF  
            SET parent TO none  
            FOR each element v IN the set associated with key u in graph  
                IF v IS IN graph AND size of the set associated with key v in graph IS GREATER THAN OR EQUAL TO size of the set associated with key u in graph AND can_be_parent called with arguments v and u RETURNS true  
                    IF parent IS none OR size of the set associated with key v in graph IS LESS THAN size of the set associated with key parent in graph  
                        SET parent TO v  
                    END IF  
                END IF  
            END FOR  
            IF parent IS none  
                RETURN zero  
            END IF  
            IF size of the set associated with key parent in graph IS EQUAL TO size of the set associated with key u in graph  
                SET ways TO two  
            END IF  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
