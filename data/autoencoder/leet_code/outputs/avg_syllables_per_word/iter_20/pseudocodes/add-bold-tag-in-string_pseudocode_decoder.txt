CLASS Solution  
    FUNCTION addBoldTag with parameters s and words  
        IF words EQUALS empty  
            RETURN s  
        END IF  
        SET n TO the LENGTH OF s  
        SET mask TO list of FALSE for each position FROM zero TO n MINUS one  
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word FOUND IN s STARTING FROM start  
                IF start EQUALS minus one  
                    BREAK  
                END IF  
                FOR index FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET mask element at position index TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        SET result TO empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF mask element at position i IS TRUE  
                APPEND string TAG for bold start TO result  
                WHILE i IS LESS THAN n AND mask element at position i IS TRUE  
                    APPEND character at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND string TAG for bold end TO result  
            ELSE  
                APPEND character at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
