CLASS Solution  
    FUNCTION sequenceReconstruction(nums, sequences)  
        SET n TO the LENGTH OF nums  
        SET graph TO a new default dictionary with list as default value  
        SET indegree TO a list of zeros with LENGTH equal TO n PLUS one  
        
        FOR each sequence IN sequences  
            FOR index FROM zero TO the LENGTH OF sequence MINUS one MINUS one  
                SET u TO the element at position index of sequence  
                SET v TO the element at position index PLUS one of sequence  
                APPEND v TO the element at key u of graph  
                INCREMENT the element at position v of indegree BY one  
            END FOR  
        END FOR  
        
        SET queue TO a new deque with elements being each integer i FROM one TO n INCLUSIVE WHERE the element at position i of indegree EQUALS zero  
        
        IF the LENGTH OF queue NOT EQUALS one  
            RETURN False  
        END IF  
        
        SET index TO zero  
        WHILE the LENGTH OF queue GREATER THAN zero  
            IF the LENGTH OF queue NOT EQUALS one  
                RETURN False  
            END IF  
            
            SET node TO the result of removing the element from the left side of queue  
            IF node NOT EQUALS the element at position index of nums  
                RETURN False  
            END IF  
            INCREMENT index BY one  
            
            FOR each neighbor IN the element at key node of graph  
                DECREMENT the element at position neighbor of indegree BY one  
                IF the element at position neighbor of indegree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether index EQUALS n  
    END FUNCTION  
END CLASS
