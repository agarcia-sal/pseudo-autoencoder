CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty mapping from state to boolean value  
        
        FUNCTION can_win_helper(state)  
            IF state IS a key IN memo  
                RETURN the value CORRESPONDING TO state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                SET substring TO the substring from position index TO index PLUS one of state  
                IF substring EQUALS the two plus signs  
                    SET new_state TO the concatenation of the substring from position zero TO index of state PLUS the two minus signs PLUS the substring from position index PLUS two TO the end of state  
                    SET opponent_can_win TO the result of can_win_helper WITH new_state  
                    IF opponent_can_win IS false  
                        SET the value CORRESPONDING TO state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value CORRESPONDING TO state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
