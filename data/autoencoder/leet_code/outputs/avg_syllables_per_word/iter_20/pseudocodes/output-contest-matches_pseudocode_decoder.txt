CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO the LIST OF strings REPRESENTING numbers FROM one TO n inclusive  
        WHILE n IS GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO the STRING CONSISTING OF opening PARENTHESIS concatenated WITH element at position i of teams concatenated WITH COMMA concatenated WITH element at position n MINUS one MINUS i of teams concatenated WITH closing PARENTHESIS  
            END FOR  
            SET n TO n DIVIDED BY two INTEGER DIVISION  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
