CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each left edge coordinate left right edge coordinate right height value height IN buildings  
            APPEND a tuple consisting of left coordinate left negated height negated height and right coordinate right TO events  
            APPEND a tuple consisting of right coordinate right zero zero zero TO events  
        END FOR  
        SORT events by the natural order of their elements starting with the x coordinate followed by the height and then by the right coordinate  
        SET result TO a list containing a single element list of zero zero  
        SET max_heap TO a list containing a single tuple of zero and positive infinity  
        FOR each x coordinate x negative height negH right coordinate r IN events  
            WHILE the right coordinate of the element at position zero of max_heap is less than or equal to x coordinate x  
                REMOVE the top element from max_heap using heap pop operation  
            END WHILE  
            IF negative height negH is not equal to zero  
                ADD the tuple consisting of negative height negH and right coordinate r to max_heap using heap push operation  
            END IF  
            IF the height component of the last element in result is not equal to the negation of the height component of the element at position zero of max_heap  
                APPEND a list consisting of x coordinate x and the negation of the height component of the element at position zero of max_heap TO result  
            END IF  
        END FOR  
        IF the first element in result is a list equal to zero zero  
            REMOVE the first element from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
