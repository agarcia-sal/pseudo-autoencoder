CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a new list of zeros of size n  

        SET force TO zero  
        FOR index FROM zero TO n MINUS one  
            IF the element at position index of dominoes EQUALS the character R  
                SET force TO n  
            ELSE IF the element at position index of dominoes EQUALS the character L  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM OF force MINUS one AND zero  
            END IF  
            INCREMENT the element at position index of forces BY force  
        END FOR  

        SET force TO zero  
        FOR index FROM n MINUS one DOWN TO zero  
            IF the element at position index of dominoes EQUALS the character L  
                SET force TO n  
            ELSE IF the element at position index of dominoes EQUALS the character R  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM OF force MINUS one AND zero  
            END IF  
            DECREMENT the element at position index of forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each force_value IN forces  
            IF force_value GREATER THAN zero  
                APPEND the character R TO result  
            ELSE IF force_value LESS THAN zero  
                APPEND the character L TO result  
            ELSE  
                APPEND the character dot TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
