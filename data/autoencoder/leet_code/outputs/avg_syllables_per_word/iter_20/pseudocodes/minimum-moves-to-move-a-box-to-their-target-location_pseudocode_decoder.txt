CLASS Solution  
    FUNCTION minPushBox(grid)  
        SET directions TO list containing tuples of minus one and zero, one and zero, zero and minus one, and zero and one  
        FOR index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR index j FROM zero TO the LENGTH OF the element at position zero of grid MINUS one  
                IF the element at position i at position j of grid EQUALS the letter S  
                    SET player_start TO tuple of i and j  
                ELSE IF the element at position i at position j of grid EQUALS the letter B  
                    SET box_start TO tuple of i and j  
                ELSE IF the element at position i at position j of grid EQUALS the letter T  
                    SET target TO tuple of i and j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a double ended queue containing a tuple of player_start, box_start, and zero  
        SET visited TO an empty set  
        ADD the tuple of player_start and box_start TO visited  
        WHILE queue is not empty  
            REMOVE the element from the left end of queue and ASSIGN TO player_pos, box_pos, push_count  
            IF box_pos EQUALS target  
                RETURN push_count  
            END IF  
            FOR each tuple dx, dy IN directions  
                SET new_player_pos TO tuple of the element at position zero of player_pos PLUS dx and the element at position one of player_pos PLUS dy  
                IF the result of calling is_valid_position with new_player_pos and grid IS TRUE  
                    IF new_player_pos EQUALS box_pos  
                        SET new_box_pos TO tuple of the element at position zero of box_pos PLUS dx and the element at position one of box_pos PLUS dy  
                        IF the result of calling is_valid_position with new_box_pos and grid IS TRUE AND the tuple of new_player_pos and new_box_pos IS NOT IN visited  
                            ADD the tuple of new_player_pos and new_box_pos TO visited  
                            APPEND the tuple of new_player_pos, new_box_pos, and push_count PLUS one TO queue  
                        END IF  
                    ELSE  
                        IF the tuple of new_player_pos and box_pos IS NOT IN visited  
                            ADD the tuple of new_player_pos and box_pos TO visited  
                            APPEND the tuple of new_player_pos, box_pos, and push_count TO queue  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
  
    FUNCTION is_valid_position(position, grid)  
        SET x TO the element at position zero of position  
        SET y TO the element at position one of position  
        RETURN x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN the LENGTH OF grid AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN the LENGTH OF the element at position zero of grid AND the element at position x at position y of grid IS NOT EQUAL TO the character representing a wall  
    END FUNCTION  
END CLASS
