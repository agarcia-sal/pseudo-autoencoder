CLASS Solution  
    FUNCTION medianSlidingWindow(nums LIST OF INTEGER k INTEGER) RETURNS LIST OF FLOAT  
        SET lo TO empty LIST representing max heap of inverted values  
        SET hi TO empty LIST representing min heap  

        FUNCTION balance_heaps()  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi PLUS one  
                MOVE the element REMOVED FROM lo BY heappop inverted TO hi BY heappush  
            ELSE IF the LENGTH OF hi IS GREATER THAN the LENGTH OF lo  
                MOVE the element REMOVED FROM hi BY heappop inverted TO lo BY heappush  
            END IF  
        END FUNCTION  

        FUNCTION add_num(num INTEGER)  
            IF lo is empty OR num IS LESS THAN OR EQUAL TO the element at position zero of lo inverted  
                INSERT the inverted value of num INTO lo BY heappush  
            ELSE  
                INSERT num INTO hi BY heappush  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION remove_num(num INTEGER)  
            IF num IS LESS THAN OR EQUAL TO the element at position zero of lo inverted  
                REMOVE the inverted value of num FROM lo  
                IF lo is not empty  
                    REORDER lo TO maintain heap property BY heapify  
                END IF  
            ELSE  
                REMOVE num FROM hi  
                IF hi is not empty  
                    REORDER hi TO maintain heap property BY heapify  
                END IF  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION get_median() RETURNS FLOAT  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi  
                RETURN the inverted value of the element at position zero of lo  
            ELSE  
                RETURN the sum of the inverted value of the element at position zero of lo AND the element at position zero of hi DIVIDED BY two  
            END IF  
        END FUNCTION  

        FOR index FROM zero TO k MINUS one  
            CALL add_num(element at position index of nums)  
        END FOR  

        SET medians TO a new LIST initialized with the result of get_median()  

        FOR index FROM k TO the LENGTH OF nums MINUS one  
            CALL add_num(element at position index of nums)  
            CALL remove_num(element at position index MINUS k of nums)  
            APPEND the result of get_median() TO medians  
        END FOR  

        RETURN medians  
    END FUNCTION  
END CLASS
