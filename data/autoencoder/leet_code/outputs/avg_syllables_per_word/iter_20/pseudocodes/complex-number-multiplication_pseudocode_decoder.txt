CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO result of splitting num BY the character plus  
            SET real TO integer conversion of element at position zero of parts  
            SET imaginary TO integer conversion of substring from position zero TO position of the length of element at position one of parts MINUS one of element at position one of parts  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO result of calling parse_complex WITH num1  
        SET real2 AND imag2 TO result of calling parse_complex WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN string formed by concatenation of string representation of real_part PLUS the character plus PLUS string representation of imaginary_part PLUS the character i  
    END FUNCTION  
END CLASS
