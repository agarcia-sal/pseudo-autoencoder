CLASS Solution  
    FUNCTION findKthLargest(nums AS List OF integer, k AS integer) RETURNS integer  
        SET min_heap TO empty list  
        FOR each num IN nums  
            PERFORM heap push operation TO insert num INTO min_heap  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                PERFORM heap pop operation TO remove the smallest element FROM min_heap  
            END IF  
        END FOR  
        RETURN element at position zero of min_heap  
    END FUNCTION  
END CLASS
