CLASS Solution  
    FUNCTION nextGreaterElement WITH parameter n  
        SET digits TO a list of each character representing a digit in the string version of n  
        SET length TO the number of elements in digits  
        
        SET i TO length MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND the element at position i of digits IS GREATER THAN OR EQUAL TO the element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        
        SET j TO length MINUS one  
        WHILE the element at position j of digits IS LESS THAN OR EQUAL TO the element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        
        SWAP the element at position i of digits WITH the element at position j of digits  
        
        SET digits TO the concatenation of the subsequence of digits from the start TO position i PLUS one, AND the reverse of the subsequence of digits from position i PLUS one TO the end  
        
        SET result TO the integer representation of the concatenation of all elements in digits  
        
        IF result IS GREATER THAN two RAISED TO THE POWER OF thirty ONE MINUS one  
            RETURN negative one  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
