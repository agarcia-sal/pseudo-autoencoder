CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO list consisting of the value one repeated n times  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for every k FROM zero TO m MINUS one element at position k at position j of strs IS LESS THAN OR EQUAL TO element at position k at position i of strs  
                    SET dp element at position i TO the GREATER OF dp element at position i AND dp element at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM VALUE IN dp  
        RETURN n MINUS longest_non_decreasing_subsequence_length  
    END FUNCTION  
END CLASS
