CLASS Solution  
    FUNCTION subsetsWithDup(nums)  
        SORT nums IN ASCENDING ORDER  
        SET result TO empty list  
        SET subset TO empty list  
        
        FUNCTION dfs(i)  
            IF i IS GREATER THAN OR EQUAL TO the LENGTH OF nums  
                APPEND a COPY OF subset TO result  
                RETURN  
            END IF  
            
            APPEND element at position i of nums TO subset  
            CALL dfs WITH the value i PLUS one  
            
            REMOVE last element FROM subset  
            WHILE the value i PLUS one IS LESS THAN the LENGTH OF nums AND element at position i of nums EQUALS element at position i PLUS one of nums  
                INCREMENT i BY one  
            END WHILE  
            CALL dfs WITH the value i PLUS one  
        END FUNCTION  
        
        CALL dfs WITH zero  
        RETURN result  
    END FUNCTION  
END CLASS
