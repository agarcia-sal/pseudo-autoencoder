CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS cost AND target  
        SET cost_to_digit TO AN EMPTY DICTIONARY  
        FOR each index AND cost_value IN cost WITH THEIR POSITIONS  
            SET cost_value KEY IN cost_to_digit TO THE STRING REPRESENTATION OF index PLUS one  
        END FOR  
          
        SET dp TO A LIST STARTING WITH AN EMPTY STRING FOLLOWED BY target TIMES A STRING REPRESENTING NEGATIVE ONE  
          
        FOR t FROM one TO target  
            FOR each cost_value AND digit_string IN cost_to_digit ITEMS  
                IF t IS GREATER THAN OR EQUAL TO cost_value AND THE ELEMENT AT POSITION t MINUS cost_value OF dp IS NOT EQUAL TO STRING REPRESENTING NEGATIVE ONE  
                    SET candidate TO digit_string CONCATENATED WITH THE ELEMENT AT POSITION t MINUS cost_value OF dp  
                    IF ELEMENT AT POSITION t OF dp IS EQUAL TO STRING REPRESENTING NEGATIVE ONE OR THE LENGTH OF candidate IS GREATER THAN THE LENGTH OF ELEMENT AT POSITION t OF dp OR THE LENGTH OF candidate IS EQUAL TO THE LENGTH OF ELEMENT AT POSITION t OF dp AND candidate IS GREATER THAN THE ELEMENT AT POSITION t OF dp  
                        SET ELEMENT AT POSITION t OF dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        IF ELEMENT AT POSITION target OF dp IS NOT EQUAL TO STRING REPRESENTING NEGATIVE ONE  
            RETURN ELEMENT AT POSITION target OF dp  
        ELSE  
            RETURN STRING REPRESENTING ZERO  
        END IF  
    END FUNCTION  
END CLASS
