CLASS Solution  
    FUNCTION numFriendRequests(ages)  
        FUNCTION request_will_be_sent(age_x, age_y)  
            RETURN NOT (age_y LESS THAN OR EQUAL TO half of age_x PLUS seven OR age_y GREATER THAN age_x OR age_y GREATER THAN one hundred AND age_x LESS THAN one hundred)  
        END FUNCTION  

        SET count TO a list of zeroes with one hundred twenty one elements  
        FOR each age IN ages  
            INCREMENT element at position age of count BY one  
        END FOR  

        SET total_requests TO zero  
        FOR age_x FROM one TO one hundred twenty  
            FOR age_y FROM one TO one hundred twenty  
                IF request_will_be_sent(age_x, age_y)  
                    INCREMENT total_requests BY element at position age_x of count MULTIPLIED BY the difference between element at position age_y of count AND one when age_x EQUALS age_y otherwise zero  
                END IF  
            END FOR  
        END FOR  

        RETURN total_requests  
    END FUNCTION  
END CLASS
