CLASS Solution  
    FUNCTION findPermutation(parameter s)  
        SET variable result TO list of integers from one TO the length of s PLUS one inclusive  
        SET variable start TO zero  
        FOR variable end FROM zero TO the length of s inclusive  
            IF variable end EQUALS the length of s OR the element at position end of s EQUALS uppercase letter I  
                REVERSE the segment of variable result FROM position start TO position end PLUS one inclusive  
                SET variable start TO variable end PLUS one  
            END IF  
        END FOR  
        RETURN variable result  
    END FUNCTION  
END CLASS
