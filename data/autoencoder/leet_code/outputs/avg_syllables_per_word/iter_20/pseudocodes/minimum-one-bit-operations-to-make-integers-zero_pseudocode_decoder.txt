CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one SHIFTED LEFT BY the VALUE OF k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one SHIFTED LEFT BY the VALUE OF k  
        RETURN one SHIFTED LEFT BY the VALUE OF k PLUS one MINUS one MINUS minimumOneBitOperations RECEIVED n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
