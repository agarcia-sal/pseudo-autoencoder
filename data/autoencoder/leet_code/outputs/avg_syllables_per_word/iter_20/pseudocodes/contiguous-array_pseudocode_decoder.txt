CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary mapping zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index i AND element num IN the enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF count EXISTS AS A KEY IN prefix_sum_index  
                SET max_length TO THE GREATER VALUE BETWEEN max_length AND the RESULT OF i MINUS the VALUE ASSOCIATED WITH count IN prefix_sum_index  
            ELSE  
                SET THE VALUE ASSOCIATED WITH count IN prefix_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
