CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion  
            ADD the tuple of digits sorted in ascending order of the string representation of power_of_2 TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of digits sorted in ascending order of the string representation of n  
        RETURN whether sorted_n_digits IS an element of power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
