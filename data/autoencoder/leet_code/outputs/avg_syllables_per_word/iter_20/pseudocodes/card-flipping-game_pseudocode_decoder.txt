CLASS Solution  
    FUNCTION flipgame(fronts WITH TYPE List OF integer, backs WITH TYPE List OF integer) RETURNS integer  
        SET same TO an empty set  
        FOR each pair of elements front_number AND back_number IN fronts AND backs respectively  
            IF front_number EQUALS back_number  
                ADD front_number TO the set same  
            END IF  
        END FOR  
        SET all_numbers TO a set containing all elements from fronts and all elements from backs  
        SET good_numbers TO the set all_numbers EXCLUDING all elements of the set same  
        IF the set good_numbers IS NOT empty  
            SET minimum_good_number TO the smallest number in the set good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
