CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        ASSIGN dp TO the RESULT OF initialize_two_dimensional_list WITH parameters n PLUS one AND numCarpets PLUS one  
        FOR i FROM one TO n  
            SET dp at position i at zero carpets TO dp at position i MINUS one at zero carpets PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE ZERO  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i at j TO dp at position i MINUS one at j PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE ZERO  
                SET dp at position i at j TO the MINIMUM OF dp at position i at j AND dp at max of zero AND i MINUS carpetLen at j MINUS one  
            END FOR  
        END FOR  
        RETURN dp at position n at numCarpets  
    END FUNCTION

    FUNCTION initialize_two_dimensional_list(rows, columns)  
        CREATE a list containing rows elements where each element is a list containing columns zeros  
        RETURN this list  
    END FUNCTION  
END CLASS
