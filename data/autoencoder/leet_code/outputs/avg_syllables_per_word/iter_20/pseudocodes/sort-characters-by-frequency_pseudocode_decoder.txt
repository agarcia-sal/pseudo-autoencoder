CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO the mapping of each character in s to the number of times it appears  
        SET sorted_characters TO the list of characters in frequency sorted by the value of frequency for each character in descending order  
        SET result TO the concatenation of each character in sorted_characters repeated the number of times equal to frequency for that character  
        RETURN result  
    END FUNCTION  
END CLASS
