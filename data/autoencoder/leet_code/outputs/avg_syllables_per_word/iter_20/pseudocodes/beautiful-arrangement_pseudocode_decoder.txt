CLASS Solution  
    FUNCTION countArrangement n  
        FUNCTION backtrack used comma index  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR each i FROM one TO n  
                IF element at position i of used EQUALS False AND (the remainder of i DIVIDED BY index EQUALS zero OR the remainder of index DIVIDED BY i EQUALS zero)  
                    SET element at position i of used TO True  
                    INCREMENT count BY backtrack used comma index PLUS one  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list of False values of LENGTH n PLUS one  
        RETURN backtrack used comma one  
    END FUNCTION  
END CLASS
