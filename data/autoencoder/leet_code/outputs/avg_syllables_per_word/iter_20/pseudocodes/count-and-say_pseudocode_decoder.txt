CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS one  
            RETURN the string one  
        END IF  
        SET current_sequence TO the string one  
        FOR each integer index FROM two TO n  
            SET next_sequence TO empty string  
            SET count TO one  
            SET previous_char TO the element at position zero of current_sequence  
            FOR each integer position FROM one TO the LENGTH OF current_sequence MINUS one  
                IF the element at position position of current_sequence EQUALS previous_char  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the string representation of count FOLLOWED BY previous_char TO next_sequence  
                    SET previous_char TO the element at position position of current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND the string representation of count FOLLOWED BY previous_char TO next_sequence  
            SET current_sequence TO next_sequence  
        END FOR  
        RETURN current_sequence  
    END FUNCTION  
END CLASS
