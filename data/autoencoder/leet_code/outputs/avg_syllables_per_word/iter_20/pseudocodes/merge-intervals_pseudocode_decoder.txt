CLASS Solution  
    FUNCTION merge(intervals)  
        CALL sort ON intervals BASED ON the element at position zero of each interval  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the element at position one of the last element in merged IS LESS THAN the element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET the element at position one of the last element in merged TO the GREATER OF the element at position one of the last element in merged AND the element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
