CLASS Solution  
    FUNCTION maxSlidingWindow(nums AS List OF Integer, k AS Integer) RETURNS List OF Integer  
        IF nums IS EMPTY OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        SET deq TO a new empty double ended queue  
        SET result TO a new empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN index MINUS k PLUS one  
                REMOVE the element at position zero FROM deq  
            END IF  
            WHILE deq IS NOT empty AND the element at position element at position last of deq of nums IS LESS THAN the element at position index of nums  
                REMOVE the element at position last FROM deq  
            END WHILE  
            APPEND index TO deq  
            IF index IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
