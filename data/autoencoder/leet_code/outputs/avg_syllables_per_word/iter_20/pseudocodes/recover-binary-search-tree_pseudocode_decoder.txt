CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values list)  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET root TO TreeNode with element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(root Optional OF TreeNode)  
        SET first TO none  
        SET second TO none  
        SET prev TO none  
        SET curr TO root  
        
        WHILE curr IS NOT none  
            IF curr.left IS none  
                IF prev IS NOT none AND value of prev IS GREATER THAN value of curr  
                    IF first IS none  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr.right  
            ELSE  
                SET pred TO curr.left  
                WHILE pred.right IS NOT none AND pred.right IS NOT curr  
                    SET pred TO pred.right  
                END WHILE  
                
                IF pred.right IS none  
                    SET pred.right TO curr  
                    SET curr TO curr.left  
                ELSE  
                    SET pred.right TO none  
                    IF prev IS NOT none AND value of prev IS GREATER THAN value of curr  
                        IF first IS none  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr.right  
                END IF  
            END IF  
        END WHILE  
        
        IF first IS NOT none AND second IS NOT none  
            SET temporary TO value of first  
            SET first.val TO value of second  
            SET second.val TO temporary  
        END IF  
    END FUNCTION  
END CLASS
