CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL sort ON nums  
        SET n TO the LENGTH OF nums  
        SET first TO element at position zero of nums  
        FOR index FROM one TO n MINUS one  
            SET k TO element at position index of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position index of nums  
                CONTINUE TO the next iteration of the FOR loop  
            END IF  
            SET count TO CALL Counter FUNCTION WITH nums AS argument  
            SET arr TO empty list  
            FOR each x IN nums  
                IF the element at position x in count EQUALS zero  
                    CONTINUE TO the next iteration of the FOR loop  
                END IF  
                IF the element at position x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of the FOR loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the element at position x in count BY one  
                DECREMENT the element at position x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN empty list  
    END FUNCTION  
END CLASS
