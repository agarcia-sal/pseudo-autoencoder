CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
  
        SET freq TO the count of each character IN s  
          
        SET max_heap TO the list FOR each character AND its count IN freq items WITH count NEGATED AND the character  
        TRANSFORM max_heap INTO a heap structure  
          
        SET wait_queue TO an empty list  
          
        SET result TO an empty list  
          
        WHILE max_heap IS NOT empty  
            REMOVE the element with the highest priority FROM max_heap AND ASSIGN count AND char TO its components  
            APPEND char TO result  
              
            APPEND a tuple WITH count INCREMENTED BY one AND char TO wait_queue  
              
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first element FROM wait_queue AND ASSIGN count AND char TO its components  
                IF count IS LESS THAN zero  
                    ADD the tuple count AND char BACK INTO max_heap maintaining heap property  
                END IF  
            END IF  
        END WHILE  
          
        IF the LENGTH OF result IS NOT EQUAL TO the LENGTH OF s  
            RETURN an empty string  
        END IF  
          
        RETURN the concatenation OF all characters IN result  
    END FUNCTION  
END CLASS
