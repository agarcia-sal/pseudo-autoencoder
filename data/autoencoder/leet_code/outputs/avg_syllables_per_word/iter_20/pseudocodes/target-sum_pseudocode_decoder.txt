CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the PAIR of index AND current_sum EXISTS IN memo  
                RETURN the ELEMENT of memo at the PAIR of index AND current_sum  
            END IF  
            SET positive TO CALL dfs WITH index INCREMENTED BY one AND current_sum INCREMENTED BY the ELEMENT at position index of nums AND memo  
            SET negative TO CALL dfs WITH index INCREMENTED BY one AND current_sum DECREMENTED BY the ELEMENT at position index of nums AND memo  
            SET the ELEMENT of memo at the PAIR of index AND current_sum TO positive PLUS negative  
            RETURN the ELEMENT of memo at the PAIR of index AND current_sum  
        END FUNCTION  
        RETURN CALL dfs WITH zero AS index AND zero AS current_sum AND an EMPTY dictionary AS memo  
    END FUNCTION  
END CLASS
