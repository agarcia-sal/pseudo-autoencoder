CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node left TO a new TreeNode initialized with element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node right TO a new TreeNode initialized with element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p val NOT EQUALS q val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree applied to p left AND q left AND the result of is_same_tree applied to p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root IS none  
            RETURN  
        END IF  
        SET stack TO a list containing root  
        WHILE the LENGTH OF stack GREATER THAN zero  
            SET node TO the last element removed from stack  
            IF node right IS NOT none  
                APPEND node right TO stack  
            END IF  
            IF node left IS NOT none  
                APPEND node left TO stack  
            END IF  
            SET node left TO none  
            IF the LENGTH OF stack GREATER THAN zero  
                SET node right TO the last element of stack  
            ELSE  
                SET node right TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
