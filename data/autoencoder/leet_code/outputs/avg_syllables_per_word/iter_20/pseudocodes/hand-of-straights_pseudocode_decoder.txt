CLASS Solution  
    FUNCTION isNStraightHand(hand AS List of Integer, groupSize AS Integer) RETURNS Boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET count TO a mapping of each distinct element IN hand TO its frequency  
        SET unique_cards TO the list of distinct elements IN count sorted IN ascending order  
        FOR each card IN unique_cards  
            IF the frequency of card in count GREATER THAN zero THEN  
                SET needed TO the frequency of card in count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the frequency of the element at position card PLUS index in count LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the frequency of the element at position card PLUS index in count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
