CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        SET dp TO a two dimensional list of zeros with dimensions m plus one BY n plus one  
        
        FOR j FROM one TO n  
            SET element at position j of the first row of dp TO element at position j minus one of the first row of dp PLUS the ASCII CODE OF character at position j minus one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position i of the first column of dp TO element at position i minus one of the first column of dp PLUS the ASCII CODE OF character at position i minus one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i minus one of s1 EQUALS character at position j minus one of s2  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                ELSE  
                    SET element at position i j of dp TO the MINIMUM BETWEEN element at position i minus one j of dp PLUS the ASCII CODE OF character at position i minus one of s1 AND element at position i j minus one of dp PLUS the ASCII CODE OF character at position j minus one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m n of dp  
    END FUNCTION  
END CLASS
