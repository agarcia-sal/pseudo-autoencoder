CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET bin_representation TO the binary representation of n EXCLUDING the first position  
        SET length TO the LENGTH OF bin_representation  
        
        INITIALIZE dp AS a list of zeros with SIZE length PLUS one  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        
        FOR index FROM two TO length  
            SET the element at position index of dp TO the element at position index MINUS one of dp PLUS the element at position index MINUS two of dp  
        END FOR  
        
        SET result TO zero  
        SET previous_bit TO zero  
        
        FOR index FROM zero TO length MINUS one  
            IF the element at position index of bin_representation EQUALS the character representing one  
                INCREMENT result BY the element at position length MINUS index MINUS one of dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK OUT OF the LOOP  
                END IF  
            END IF  
            SET previous_bit TO one IF the element at position index of bin_representation EQUALS the character representing one OTHERWISE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
