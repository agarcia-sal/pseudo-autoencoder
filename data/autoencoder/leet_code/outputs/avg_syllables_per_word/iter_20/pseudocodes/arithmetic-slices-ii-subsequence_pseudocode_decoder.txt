CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list containing for each index from zero TO n MINUS one an associative collection that by default returns zero for any new key  
        SET total_count TO zero  
        
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO i MINUS one  
                SET d TO element at position i of nums MINUS element at position j of nums  
                
                IF associative collection at position j of dp CONTAINS key d  
                    INCREMENT the value at key d in associative collection at position i of dp BY the sum of the value at key d in associative collection at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key d in associative collection at position j of dp  
                ELSE  
                    INCREMENT the value at key d in associative collection at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
