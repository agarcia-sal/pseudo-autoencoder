CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left q.left AND is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves WITH PARAMETERS self root  
        FUNCTION collectLeaves WITH PARAMETER node  
            IF node IS none  
                RETURN minus one  
            END IF  
            SET left_height TO collectLeaves WITH PARAMETER node.left  
            SET right_height TO collectLeaves WITH PARAMETER node.right  
            SET current_height TO THE GREATER OF left_height AND right_height PLUS one  
            IF current_height IS GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND AN EMPTY list TO result  
            END IF  
            APPEND node.val TO element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO AN EMPTY list  
        CALL collectLeaves WITH PARAMETER root  
        RETURN result  
    END FUNCTION  
END CLASS
