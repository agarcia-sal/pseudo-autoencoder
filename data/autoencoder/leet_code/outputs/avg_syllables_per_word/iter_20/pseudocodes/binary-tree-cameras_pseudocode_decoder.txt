CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    ASSIGN root TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    ASSIGN queue TO AN EMPTY deque COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        ASSIGN node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUAL TO the val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p, the left ATTRIBUTE OF q AND is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p, the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover WITH PARAMETERS self, root AS Optional TreeNode  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN zero, zero, positive infinity  
            END IF  
            ASSIGN left_no_camera, left_monitored, left_camera TO THE RESULT OF dfs WITH PARAMETER the left ATTRIBUTE OF node  
            ASSIGN right_no_camera, right_monitored, right_camera TO THE RESULT OF dfs WITH PARAMETER the right ATTRIBUTE OF node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the MINIMUM OF the following THREE VALUES  
                left_camera PLUS the MINIMUM OF right_monitored AND right_camera  
                right_camera PLUS the MINIMUM OF left_monitored AND left_camera  
                left_camera PLUS right_camera  
            SET camera TO one PLUS THE MINIMUM OF left_no_camera, left_monitored, left_camera PLUS THE MINIMUM OF right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        ASSIGN root_no_camera, root_monitored, root_camera TO THE RESULT OF dfs WITH PARAMETER root  
        RETURN the MINIMUM OF root_monitored AND root_camera  
    END FUNCTION  
END CLASS
