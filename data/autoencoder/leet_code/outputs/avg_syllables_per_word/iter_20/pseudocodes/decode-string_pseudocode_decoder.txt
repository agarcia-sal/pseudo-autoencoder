CLASS Solution  
    FUNCTION decodeString with parameter s  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character IS a digit  
                SET d TO d CONCATENATED WITH character  
            ELSE IF character IS an alphabetic letter  
                SET a TO a CONCATENATED WITH character  
            ELSE IF character IS equal to opening square bracket  
                APPEND tuple of a AND the integer value converted from d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character IS equal to closing square bracket  
                SET p AND n TO the result of removing the last element FROM stack  
                SET a TO p CONCATENATED WITH a REPEATED n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
