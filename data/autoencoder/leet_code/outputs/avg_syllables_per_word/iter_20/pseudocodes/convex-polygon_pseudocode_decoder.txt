CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            SET term_one TO element at position zero of a MINUS element at position zero of o  
            SET term_two TO element at position one of b MINUS element at position one of o  
            SET first_product TO term_one MULTIPLIED BY term_two  
            SET term_three TO element at position one of a MINUS element at position one of o  
            SET term_four TO element at position zero of b MINUS element at position zero of o  
            SET second_product TO term_three MULTIPLIED BY term_four  
            SET result TO first_product MINUS second_product  
            RETURN result  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET o TO element at position index of points  
            SET a TO element at position of the remainder of index PLUS one DIVIDED BY n of points  
            SET b TO element at position of the remainder of index PLUS two DIVIDED BY n of points  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
