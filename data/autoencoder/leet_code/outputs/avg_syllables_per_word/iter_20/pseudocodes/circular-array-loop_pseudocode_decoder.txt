CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            RETURN (val1 GREATER THAN zero AND val2 GREATER THAN zero) OR (val1 LESS THAN zero AND val2 LESS THAN zero)  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
            
            IF element at position i of nums EQUALS zero  
                CONTINUE TO the next iteration  
            END IF  
            
            WHILE true HOLDS  
                SET slow TO the REMAINDER OF the SUM of slow PLUS element at position slow of nums DIVIDED BY n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  
                
                SET fast TO the REMAINDER OF the SUM of fast PLUS element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                SET fast TO the REMAINDER OF the SUM of fast PLUS element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                IF NOT sameDirection(element at position slow of nums, element at position i of nums) OR NOT sameDirection(element at position fast of nums, element at position i of nums)  
                    BREAK out of the WHILE loop  
                END IF  
                
                IF slow EQUALS fast  
                    IF slow EQUALS the REMAINDER OF the SUM of slow PLUS element at position slow of nums DIVIDED BY n  
                        BREAK out of the WHILE loop  
                    END IF  
                    RETURN true  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
