CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of occurrences of character X in the row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of occurrences of character O in the row  
        END FOR  
        
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN false  
        END IF  
        
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF the row EQUALS the string formed by player repeated three times  
                    RETURN true  
                END IF  
            END FOR  
            
            FOR index FROM zero TO two  
                IF the element at position index of the first row EQUALS player AND the element at position index of the second row EQUALS player AND the element at position index of the third row EQUALS player  
                    RETURN true  
                END IF  
            END FOR  
            
            IF the element at position zero of the first row EQUALS player AND the element at position one of the second row EQUALS player AND the element at position two of the third row EQUALS player  
                RETURN true  
            END IF  
            IF the element at position two of the first row EQUALS player AND the element at position one of the second row EQUALS player AND the element at position zero of the third row EQUALS player  
                RETURN true  
            END IF  
            
            RETURN false  
        END FUNCTION  
        
        SET winX TO the result of check_win applied to character X  
        SET winO TO the result of check_win applied to character O  
        
        IF winX AND winO  
            RETURN false  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN false  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
