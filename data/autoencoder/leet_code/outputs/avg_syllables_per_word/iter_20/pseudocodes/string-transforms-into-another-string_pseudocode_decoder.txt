CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2 THEN  
            RETURN true  
        END IF  
        
        SET conversion_map TO an empty mapping  
        
        FOR each pair of characters char1 and char2 TAKEN TOGETHER FROM str1 AND str2  
            IF char1 IS IN conversion_map THEN  
                IF the value mapped from char1 IN conversion_map NOT EQUALS char2 THEN  
                    RETURN false  
                END IF  
            ELSE  
                SET the mapping from char1 TO char2 IN conversion_map  
            END IF  
        END FOR  
        
        IF the number of UNIQUE characters in str2 EQUALS twenty six THEN  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
