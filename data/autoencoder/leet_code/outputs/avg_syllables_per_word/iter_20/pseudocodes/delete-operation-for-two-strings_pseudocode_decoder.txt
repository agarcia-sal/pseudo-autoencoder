CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional collection of zeroes with m PLUS one rows and n PLUS one columns  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at row i column j of dp TO the element at row i MINUS one column j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at row i column j of dp TO the GREATER OF the element at row i MINUS one column j of dp AND the element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at row m column n of dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
