CLASS Solution  
    FUNCTION findLongestChain WITH pairs AS list of list of integers  
        CALL sort ON pairs BASED ON the second element of each pair IN pairs  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each start AND end IN pairs  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
