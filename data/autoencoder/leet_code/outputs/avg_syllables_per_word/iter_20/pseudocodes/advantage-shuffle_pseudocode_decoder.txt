CLASS Solution  
    FUNCTION advantageCount(nums1 AS List OF integer, nums2 AS List OF integer) AS List OF integer  
        SORT nums1 IN ascending order  
        SET nums2_with_indices TO sorted list of tuples pairing each value from nums2 with its original index sorted BY value IN descending order  
        SET result TO a new list of zeros with the same length AS nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each tuple containing value and index IN nums2_with_indices  
            IF element at position right of nums1 IS GREATER THAN value  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
