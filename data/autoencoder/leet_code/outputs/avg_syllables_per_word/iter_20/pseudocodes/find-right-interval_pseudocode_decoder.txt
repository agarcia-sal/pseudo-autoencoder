CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the RESULT OF sorting the COLLECTION built by PAIRING the ELEMENT at position zero of interval WITH its INDEX for each INDEX and interval IN intervals  
        SET result TO an EMPTY list  
        FOR each pair underscore and end IN intervals  
            SET idx TO the POSITION FOUND by SEARCHING the LEFTMOST position where a PAIR of end and minimal value COULD BE INSERTED IN start_times maintaining ORDER  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the ELEMENT at position one of the ELEMENT at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
