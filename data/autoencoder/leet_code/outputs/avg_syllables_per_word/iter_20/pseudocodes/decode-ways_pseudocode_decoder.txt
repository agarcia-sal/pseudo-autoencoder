CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS empty OR element at position zero of s EQUALS character zero  
            RETURN zero  
        END IF  
        
        SET dp TO a list of zeros with the length of the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        
        FOR index FROM two TO the LENGTH OF s  
            IF element at position index MINUS one of s NOT EQUALS character zero  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of dp  
            END IF  
            
            SET two_digit TO the integer value of substring from position index MINUS two TO index of s  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS two of dp  
            END IF  
        END FOR  
        
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
