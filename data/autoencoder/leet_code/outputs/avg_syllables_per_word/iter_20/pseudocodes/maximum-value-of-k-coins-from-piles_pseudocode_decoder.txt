CLASS Solution  
    FUNCTION maxValueOfCoins WITH PARAMETERS piles AND k  
        SET n TO the LENGTH OF piles  
        INITIALIZE dp AS a two dimensional list of zeros with n PLUS one rows AND k PLUS one columns  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM value BETWEEN j AND the LENGTH OF current_pile  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp element at position i position j TO the MAXIMUM value BETWEEN dp element at position i position j AND dp element at position i MINUS one position j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position n position k  
    END FUNCTION  
END CLASS
