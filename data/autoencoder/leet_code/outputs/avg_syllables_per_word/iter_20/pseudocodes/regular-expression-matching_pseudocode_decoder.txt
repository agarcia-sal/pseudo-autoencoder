CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional list of False with dimension the LENGTH OF s PLUS one BY the LENGTH OF p PLUS one  
        SET element at position zero zero of dp TO True  
        FOR j FROM one TO the LENGTH OF p  
            IF the element at position j minus one of p EQUALS star character  
                SET element at position zero j of dp TO element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF the element at position j minus one of p EQUALS the element at position i minus one of s OR the element at position j minus one of p EQUALS dot character  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                ELSE IF the element at position j minus one of p EQUALS star character  
                    SET element at position i j of dp TO element at position i j minus two of dp  
                    IF the element at position j minus two of p EQUALS the element at position i minus one of s OR the element at position j minus two of p EQUALS dot character  
                        SET element at position i j of dp TO element at position i j of dp OR element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
