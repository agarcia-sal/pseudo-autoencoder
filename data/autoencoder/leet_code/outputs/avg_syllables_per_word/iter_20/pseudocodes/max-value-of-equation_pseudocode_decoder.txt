CLASS Solution  
    FUNCTION findMaxValueOfEquation(points AS List OF List OF integer, k AS integer) RETURNS integer  
        SET max_value TO negative infinity  
        INITIALIZE deque_ AS empty double ended queue  
        
        FOR each pair xj AND yj IN points  
            WHILE deque_ is not empty AND xj MINUS the element at position zero of deque_ at position one IS GREATER THAN k  
                REMOVE element from the front of deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the maximum of max_value AND yj PLUS xj PLUS the element at position zero of deque_ at position zero  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj IS GREATER THAN OR EQUAL TO the element at the last position of deque_ at position zero  
                REMOVE element from the back of deque_  
            END WHILE  
            
            APPEND the pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
