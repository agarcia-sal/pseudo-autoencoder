CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT dimensions of cuboid in nondecreasing order  
        END FOR  
        SORT cuboids in lexicographical order based on their dimensions  
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp AS a list of zeros with length n  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i of dp TO the element at position two of cuboid at position i  
            FOR index j FROM zero TO i MINUS one  
                IF element at position zero of cuboid at position j IS LESS THAN OR EQUAL TO element at position zero of cuboid at position i  
                AND element at position one of cuboid at position j IS LESS THAN OR EQUAL TO element at position one of cuboid at position i  
                AND element at position two of cuboid at position j IS LESS THAN OR EQUAL TO element at position two of cuboid at position i  
                    SET element at position i of dp TO the MAXIMUM OF element at position i of dp AND the SUM OF element at position j of dp PLUS element at position two of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM value in dp  
    END FUNCTION  
END CLASS
