CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze list of list of integers ball list of integers hole list of integers RETURNS string  
        SET directions TO list of tuples each containing zero, one, single character right then one, zero, single character down then zero, negative one, single character left then negative one, zero, single character up  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO tuple constructed from ball  
        SET hole TO tuple constructed from hole  
        SET pq TO list containing a tuple of zero, empty string, and start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            SET dist, path, tuple containing x and y TO the result of removing and returning the smallest element from pq  
            
            IF tuple containing x and y EQUALS hole  
                RETURN path  
            END IF  
            
            IF tuple containing x and y IS IN visited  
                CONTINUE with next iteration  
            END IF  
            
            ADD tuple containing x and y TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND element at position new_x plus dx of maze element at position new_y plus dy EQUALS zero  
                    SET new_x TO new_x plus dx  
                    SET new_y TO new_y plus dy  
                    INCREMENT count BY one  
                    IF tuple containing new_x and new_y EQUALS hole  
                        BREAK the WHILE loop  
                    END IF  
                END WHILE  
                
                IF tuple containing new_x and new_y IS NOT IN visited  
                    ADD tuple containing dist plus count, concatenation of path and direction, and tuple containing new_x and new_y TO the priority queue pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
