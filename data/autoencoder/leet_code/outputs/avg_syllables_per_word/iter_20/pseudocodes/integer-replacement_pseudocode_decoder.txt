CLASS Solution  
    FUNCTION integerReplacement(n)  
        SET count TO zero  
        WHILE n NOT EQUALS one  
            IF the remainder when n is divided by two EQUALS zero  
                SET n TO the quotient when n is divided by two  
            ELSE  
                IF n EQUALS three  
                    DECREMENT n BY one  
                ELSE IF the remainder when n PLUS one is divided by four EQUALS zero  
                    INCREMENT n BY one  
                ELSE  
                    DECREMENT n BY one  
                END IF  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
