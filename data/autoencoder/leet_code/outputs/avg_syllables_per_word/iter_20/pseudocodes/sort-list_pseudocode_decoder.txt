CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero AND next EQUAL TO none  
        SET the property val TO val  
        SET the property next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position one TO the end  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the result of the logical AND BETWEEN p1 val EQUALS p2 val AND the result of is_same_list INVOKED WITH PARAMETERS the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList WITH PARAMETER head  
        IF head IS none OR the next property of head IS none  
            RETURN head  
        END IF  
        
        FUNCTION split WITH PARAMETER head  
            SET slow TO head  
            SET fast TO the next property of head  
            WHILE fast IS NOT none AND the next property of fast IS NOT none  
                SET slow TO the next property of slow  
                SET fast TO the next property of the next property of fast  
            END WHILE  
            SET mid TO the next property of slow  
            SET the next property of slow TO none  
            RETURN head AND mid  
        END FUNCTION  
        
        FUNCTION merge WITH PARAMETERS l1 AND l2  
            SET dummy TO A NEW ListNode WITH DEFAULT PARAMETERS  
            SET current TO dummy  
            WHILE l1 IS NOT none AND l2 IS NOT none  
                IF the val property of l1 IS LESS THAN the val property of l2  
                    SET the next property of current TO l1  
                    SET l1 TO the next property of l1  
                ELSE  
                    SET the next property of current TO l2  
                    SET l2 TO the next property of l2  
                END IF  
                SET current TO the next property of current  
            END WHILE  
            IF l1 IS NOT none  
                SET the next property of current TO l1  
            ELSE  
                SET the next property of current TO l2  
            END IF  
            RETURN the next property of dummy  
        END FUNCTION  
        
        SET left AND right TO the result of split INVOKED WITH head  
        SET left TO the result of sortList INVOKED WITH left  
        SET right TO the result of sortList INVOKED WITH right  
        
        RETURN the result of merge INVOKED WITH left AND right  
    END FUNCTION  
END CLASS
