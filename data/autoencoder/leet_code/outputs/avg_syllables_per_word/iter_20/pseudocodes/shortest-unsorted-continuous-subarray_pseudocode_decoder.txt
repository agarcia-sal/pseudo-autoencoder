CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_nums TO the sorted version of nums  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index of nums DOES NOT EQUAL element at position index of sorted_nums  
                IF start EQUALS negative one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        SET length_of_subarray TO end MINUS start PLUS one  
        RETURN length_of_subarray  
    END FUNCTION  
END CLASS
