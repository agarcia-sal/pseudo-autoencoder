CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements of sentence separated by space PLUS a single space at the end  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each number FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the element at position MODULO total_length of s EQUALS the space character THEN  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at position MINUS one MODULO total_length of s DOES NOT EQUAL the space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN the INTEGER PART OF position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
