CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values of type list  
    IF the values list IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode created with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot left TO a new TreeNode created with element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot right TO a new TreeNode created with element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p dot left AND q dot left AND the result of is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView WITH PARAMETERS self root  
        IF root IS None  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a double-ended queue initialized with root  
        WHILE the queue is not empty  
            SET level_length TO the LENGTH OF queue  
            FOR index FROM zero TO level_length MINUS one  
                SET node TO the element removed from the front of queue  
                IF index EQUALS level_length MINUS one  
                    APPEND node dot val TO right_side_view  
                END IF  
                IF node dot left IS NOT None  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT None  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
