CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list of tuples representing the pairs of integers minus two and minus one, minus two and one, minus one and minus two, minus one and two, one and minus two, one and two, two and minus one, two and one  
        SET dp TO two dimensional list of zero REPEATED n TIMES for each dimension  
        SET element at position row of dp at position column TO one  
        FOR each iteration FROM one TO k  
            SET new_dp TO two dimensional list of zero REPEATED n TIMES for each dimension  
            FOR r FROM zero TO n MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF element at position r of dp at position c EQUALS zero  
                        CONTINUE to the next iteration of the inner loop  
                    END IF  
                    FOR each pair dr, dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr of new_dp at position nc BY element at position r of dp at position c DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each inner_row IN dp  
            FOR each probability_value IN inner_row  
                INCREMENT total_probability BY probability_value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
