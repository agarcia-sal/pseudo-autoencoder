CLASS Solution  
    FUNCTION maximumInvitations(favorite)  
        SET n TO the LENGTH OF favorite  
        SET graph TO a mapping from keys to empty lists  
        SET in_degree TO a list of zeros with length n  
        
        FOR index FROM zero TO n MINUS one  
            APPEND index TO the element at position element at position index of favorite in graph  
            INCREMENT element at position element at position index of favorite in in_degree BY one  
        END FOR  
        
        SET mutual_chains TO zero  
        SET visited TO a list of False values with length n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position element at position index of favorite in favorite EQUALS index AND index LESS THAN element at position index of favorite THEN  
                SET a TO index  
                SET b TO element at position index of favorite  
                SET chain_a TO the RESULT OF calling find_chain_length WITH parameters a graph visited  
                SET chain_b TO the RESULT OF calling find_chain_length WITH parameters b graph visited  
                INCREMENT mutual_chains BY chain_a PLUS chain_b  
            END IF  
        END FOR  
        
        SET longest_cycle TO zero  
        SET visited TO a list of False values with length n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position index in visited IS False THEN  
                SET cycle_length TO the RESULT OF calling find_cycle_length WITH parameters index graph visited favorite  
                IF cycle_length GREATER THAN longest_cycle THEN  
                    SET longest_cycle TO cycle_length  
                END IF  
            END IF  
        END FOR  
        
        RETURN the GREATER OF mutual_chains AND longest_cycle  
    END FUNCTION  
      
    FUNCTION find_chain_length(start graph visited)  
        SET length TO zero  
        SET queue TO a double ended queue initialized with start  
        SET element at position start in visited TO True  
        
        WHILE queue IS NOT empty  
            SET current TO the REMOVE FROM FRONT element of queue  
            FOR each neighbor IN element at position current in graph  
                IF element at position neighbor in visited IS False THEN  
                    SET element at position neighbor in visited TO True  
                    APPEND neighbor TO queue  
                    INCREMENT length BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN length  
    END FUNCTION  
      
    FUNCTION find_cycle_length(start graph visited favorite)  
        IF element at position start in visited IS True THEN  
            RETURN zero  
        END IF  
        
        SET stack TO an empty list  
        SET current TO start  
        
        WHILE element at position current in visited IS False  
            SET element at position current in visited TO True  
            APPEND current TO stack  
            SET current TO element at position current in favorite  
        END WHILE  
        
        IF current NOT IN stack THEN  
            RETURN zero  
        END IF  
        
        SET cycle_start_index TO the POSITION OF current IN stack  
        SET cycle_length TO the LENGTH OF stack MINUS cycle_start_index  
        
        RETURN cycle_length  
    END FUNCTION  
END CLASS
