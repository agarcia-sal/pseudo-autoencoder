CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values OF TYPE list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q AND AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree WITH PARAMETERS self root OF TYPE Optional OF TreeNode  
        IF root IS none  
            RETURN zero  
        END IF  
        SET queue TO NEW double ended queue WITH element consisting of root AND zero  
        SET max_width TO zero  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second component of the element at position zero of queue  
            FOR each iterator FROM zero TO level_length MINUS one  
                SET node TO the first component of element removed from the left side of queue  
                SET index TO the second component of element removed from the left side of queue  
                IF iterator EQUALS level_length MINUS one  
                    IF max_width IS LESS THAN index MINUS first_index PLUS one  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node dot left IS NOT none  
                    APPEND the pair consisting of node dot left AND index MULTIPLIED BY two TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND the pair consisting of node dot right AND index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
