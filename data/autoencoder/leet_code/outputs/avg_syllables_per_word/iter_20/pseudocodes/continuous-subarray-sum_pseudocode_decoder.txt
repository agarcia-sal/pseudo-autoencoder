CLASS Solution  
    FUNCTION checkSubarraySum(nums AS List of Integer, k AS Integer) RETURNS Boolean  
        SET remainder_map TO a dictionary WITH key zero AND value negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN enumeration of nums  
            INCREMENT current_sum BY num  
            IF k IS NOT EQUAL TO zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum IS A KEY IN remainder_map  
                IF the value at key current_sum in remainder_map SUBTRACTED BY i IS GREATER THAN one  
                    RETURN true  
                END IF  
            ELSE  
                SET the value at key current_sum in remainder_map TO i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
