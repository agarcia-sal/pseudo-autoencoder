CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new default dictionary with list as default value  
        FOR each path IN paths  
            SET parts TO the result of splitting path by spaces  
            SET directory TO the first element of parts  
            FOR each file_info IN the elements of parts starting from the second element  
                SET file_name TO the substring of file_info from the start TO the position before the first opening parenthesis  
                SET content TO the substring of file_info from the position after the first opening parenthesis TO the position before the last character  
                APPEND the concatenation of directory, a forward slash, and file_name TO the list in content_to_paths associated with content  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each paths_group IN the collection of values in content_to_paths  
            IF the LENGTH OF paths_group is greater than one  
                APPEND paths_group TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
