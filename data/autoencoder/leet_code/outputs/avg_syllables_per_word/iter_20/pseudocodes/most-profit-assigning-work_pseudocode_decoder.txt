CLASS Solution  
    FUNCTION maxProfitAssignment WITH PARAMETERS difficulty profit worker  
        SET jobs TO the result of creating a list by pairing elements of difficulty WITH elements of profit INTO tuples  
        SORT jobs BY the first element representing difficulty IN ascending order  
        SORT worker IN ascending order by their ability  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND the first element at position job_index of jobs IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO the GREATER VALUE BETWEEN best_profit AND the second element at position job_index of jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
