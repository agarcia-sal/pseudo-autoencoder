CLASS Solution  
    FUNCTION numOfWays WITH PARAMETERS n  
        SET MOD TO one billion PLUS seven  
        SET a TO six  
        SET b TO six  
        FOR each index FROM two TO n  
            SET new_a TO the quantity three MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MOD  
            SET new_b TO the quantity two MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MOD  
            SET a TO new_a  
            SET b TO new_b  
        END FOR  
        SET result TO a PLUS b MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
