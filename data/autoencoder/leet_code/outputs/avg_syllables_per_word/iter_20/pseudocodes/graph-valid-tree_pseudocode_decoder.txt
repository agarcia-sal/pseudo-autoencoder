CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN false  
        END IF  
        SET adjacency_list TO a new mapping FROM each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each pair a AND b IN edges  
            APPEND b TO the element at position a of adjacency_list  
            APPEND a TO the element at position b of adjacency_list  
        END FOR  
        SET visited TO a new empty set  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the element at position node of adjacency_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
