CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET first_row_has_zero TO the RESULT OF checking whether any element at position j of element at position zero of matrix EQUALS zero FOR each j FROM zero TO n MINUS one  
        SET first_col_has_zero TO the RESULT OF checking whether any element at position i of matrix at position zero EQUALS zero FOR each i FROM zero TO m MINUS one  
        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                IF element at position i of matrix at position j EQUALS zero  
                    SET element at position i of matrix at position zero TO zero  
                    SET element at position zero of matrix at position j TO zero  
                END IF  
            END FOR  
        END FOR  
        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                IF element at position i of matrix at position zero EQUALS zero OR element at position zero of matrix at position j EQUALS zero  
                    SET element at position i of matrix at position j TO zero  
                END IF  
            END FOR  
        END FOR  
        IF first_row_has_zero  
            FOR each j FROM zero TO n MINUS one  
                SET element at position zero of matrix at position j TO zero  
            END FOR  
        END IF  
        IF first_col_has_zero  
            FOR each i FROM zero TO m MINUS one  
                SET element at position i of matrix at position zero TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
