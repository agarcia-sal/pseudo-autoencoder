CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing zero AND firstPerson  
        SORT meetings BY the element at position three of each meeting in ascending order  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position three of the element at position i of meetings  
            SET graph TO an empty dictionary with default value an empty list  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position three of the element at position i of meetings EQUALS current_time  
                SET x TO the element at position one of the element at position i of meetings  
                SET y TO the element at position two of the element at position i of meetings  
                APPEND y TO the list associated with key x in graph  
                APPEND x TO the list associated with key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a double ended queue initialized with each person in know_secret who is a key in graph  
            SET visited TO an empty set  
            WHILE queue is not empty  
                REMOVE the leftmost element from queue AND ASSIGN it TO person  
                IF person IS IN visited  
                    CONTINUE to the next iteration of the while loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list associated with key person in graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list representation of know_secret  
    END FUNCTION  
END CLASS
