CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val defaulted to zero AND next defaulted to none  
        SET the property val TO val  
        SET the property next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the SUBSEQUENCE OF values FROM position one TO the end  
        SET node TO A NEW ListNode CREATED WITH val  
        SET the property next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the evaluation of the property val of p1 EQUALS the property val of p2 AND THE RESULT OF Calling is_same_list WITH PARAMETERS the property next of p1 AND the property next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS head  
        IF head IS none OR the property next of head IS none  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode CREATED WITH the value negative five thousand AND one  
        SET the property next of dummy TO head  
        SET current TO the property next of head  
        SET the property next of head TO none  
        WHILE current IS NOT none  
            SET next_node TO the property next of current  
            SET prev TO dummy  
            WHILE the property next of prev IS NOT none AND the property val of the property next of prev IS LESS THAN the property val of current  
                SET prev TO the property next of prev  
            END WHILE  
            SET the property next of current TO the property next of prev  
            SET the property next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the property next of dummy  
    END FUNCTION  
END CLASS
