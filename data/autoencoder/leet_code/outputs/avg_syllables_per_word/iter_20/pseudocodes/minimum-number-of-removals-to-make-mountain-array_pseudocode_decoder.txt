CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        SET lis TO a list of one repeated n times  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO i MINUS one  
                IF element at position i of nums IS GREATER THAN element at position j of nums  
                    SET lis at position i TO the GREATER OF value at position i of lis AND value at position j of lis PLUS one  
                END IF  
            END FOR  
        END FOR  

        SET lds TO a list of one repeated n times  
        FOR each index i FROM n MINUS one DOWN TO zero  
            FOR each index j FROM i PLUS one TO n MINUS one  
                IF element at position i of nums IS GREATER THAN element at position j of nums  
                    SET lds at position i TO the GREATER OF value at position i of lds AND value at position j of lds PLUS one  
                END IF  
            END FOR  
        END FOR  

        SET max_mountain_length TO zero  
        FOR each index i FROM one TO n MINUS two  
            IF value at position i of lis IS GREATER THAN one AND value at position i of lds IS GREATER THAN one  
                SET max_mountain_length TO the GREATER OF max_mountain_length AND value at position i of lis PLUS value at position i of lds MINUS one  
            END IF  
        END FOR  

        RETURN n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
