CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position index of nums IS NOT EQUAL TO the element at position the element at position index of nums MINUS one of nums  
                SET temporary_value TO the element at position index of nums  
                SET the element at position index of nums TO the element at position temporary_value MINUS one of nums  
                SET the element at position temporary_value MINUS one of nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each position_and_value IN enumeration of nums  
            SET current_index TO the position part of position_and_value  
            SET current_value TO the value part of position_and_value  
            IF current_value IS NOT EQUAL TO current_index PLUS one  
                APPEND current_value TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
