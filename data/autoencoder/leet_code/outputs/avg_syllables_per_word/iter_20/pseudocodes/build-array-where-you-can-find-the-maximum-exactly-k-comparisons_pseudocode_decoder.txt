CLASS Solution  
    FUNCTION numOfArrays WITH PARAMETERS n m k  
        SET MOD TO one billion PLUS seven  
        SET dp TO three dimensional list of zeros with dimensions n plus one by m plus one by k plus one  
        
        FOR each j FROM one TO m  
            SET element at position one position j position one of dp TO one  
        END FOR  
        
        FOR each i FROM two TO n  
            FOR each j FROM one TO m  
                FOR each p FROM one TO k  
                    FOR each x FROM one TO j MINUS one  
                        SET element at position i position j position p of dp TO element at position i position j position p of dp PLUS element at position i minus one position x position p minus one of dp  
                        SET element at position i position j position p of dp TO element at position i position j position p of dp MODULO MOD  
                    END FOR  
                    SET element at position i position j position p of dp TO element at position i position j position p of dp PLUS element at position i minus one position j position p of dp MULTIPLIED BY j  
                    SET element at position i position j position p of dp TO element at position i position j position p of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each j FROM one TO m  
            SET result TO result PLUS element at position n position j position k of dp  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
