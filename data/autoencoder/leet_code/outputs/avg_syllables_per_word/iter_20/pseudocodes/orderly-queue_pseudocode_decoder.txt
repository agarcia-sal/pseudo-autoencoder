CLASS Solution  
    FUNCTION orderlyQueue with parameters s and k  
        IF k GREATER THAN one  
            RETURN concatenation of characters OF s SORTED in ascending order  
        ELSE  
            SET n TO LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO substring of s FROM position i TO the END concatenated WITH substring of s FROM position zero TO position i MINUS one  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
