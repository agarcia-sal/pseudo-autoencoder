CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO a mapping from zero TO zero one TO one six TO nine eight TO eight nine TO six  
        SET valid_digits TO a list containing zero one six eight nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero  
                SET digit TO the remainder WHEN num IS DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the element at position digit OF rotate_map  
                SET num TO num DIVIDED BY ten rounded down to nearest integer  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit, current)  
            IF current GREATER THAN limit  
                RETURN zero  
            END IF  
            SET count TO one IF current NOT EQUALS zero AND is_confusing(current) HOLDS ELSE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero  
                    CONTINUE TO the next iteration  
                END IF  
                SET count TO count PLUS count_confusing_numbers(limit, new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n, zero)  
    END FUNCTION  
END CLASS
