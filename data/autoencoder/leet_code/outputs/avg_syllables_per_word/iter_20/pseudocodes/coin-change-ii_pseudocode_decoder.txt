CLASS Solution  
    FUNCTION change(amount, coins)  
        SET dp TO a list of zero repeated the amount plus one times  
        SET the element at position zero of dp TO one  
        FOR each coin IN coins  
            FOR x FROM coin TO amount  
                INCREMENT the element at position x of dp BY the element at position x MINUS coin of dp  
            END FOR  
        END FOR  
        RETURN the element at position amount of dp  
    END FUNCTION  
END CLASS
