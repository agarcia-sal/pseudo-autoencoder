CLASS Solution  
    FUNCTION kEmptySlots(bulbs AS List OF integer, k AS integer) RETURNS integer  
        SET bloom_days TO empty list  
        FOR each day AND position IN bulbs WITH day STARTING FROM one  
            SET i TO the position where position CAN BE inserted in bloom_days TO MAINTAIN SORT ORDER USING binary search TO FIND the LEFTMOST insertion point  
            IF i IS GREATER THAN zero AND the value of position MINUS the element at position i MINUS one of bloom_days EQUALS k  
                RETURN day  
            END IF  
            IF i IS LESS THAN the LENGTH OF bloom_days AND the element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i TO MAINTAIN SORT ORDER  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
