CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO the LENGTH OF grid OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the PAIR of x AND y IS IN visited OR element at position x of grid AT position y EQUALS zero  
                RETURN  
            END IF  
            ADD the PAIR of x AND y TO visited  
            FOR each PAIR dx AND dy IN the LIST of PAIRS consisting of MINUS one AND zero, one AND zero, zero AND MINUS one, zero AND one  
                CALL dfs WITH x PLUS dx, y PLUS dy, AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO the EMPTY SET  
            SET land_cells TO the LIST OF PAIRS i AND j WHERE i IS FROM zero TO the LENGTH OF grid MINUS one AND j IS FROM zero TO the LENGTH OF element at position zero of grid MINUS one AND element at position i of grid AT position j EQUALS one  
            IF land_cells IS EMPTY  
                RETURN True  
            END IF  
            CALL dfs WITH the FIRST element AT position zero OF land_cells, the SECOND element AT position one OF land_cells, AND visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF the NEGATION OF calling is_connected()  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid AT position j EQUALS one  
                    SET element at position i of grid AT position j TO zero  
                    IF the NEGATION OF calling is_connected()  
                        RETURN one  
                    END IF  
                    SET element at position i of grid AT position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
