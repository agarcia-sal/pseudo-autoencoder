CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
        
        SET left_half TO the SUBLIST of nums FROM position zero TO position n minus one  
        SET right_half TO the SUBLIST of nums FROM position n TO the END OF nums  
        
        SET left_sums TO an ARRAY of EMPTY DICTIONARIES HAVING the LENGTH OF n PLUS one  
        SET right_sums TO an ARRAY of EMPTY DICTIONARIES HAVING the LENGTH OF n PLUS one  
        
        FOR i FROM zero TO n INCLUSIVE  
            FOR each combo IN all COMBINATIONS OF left_half TAKING i elements  
                SET subset_sum TO the SUM OF elements IN combo  
                IF subset_sum IS NOT A KEY IN the DICTIONARY at position i OF left_sums  
                    SET the VALUE at key subset_sum IN left_sums AT position i TO zero  
                END IF  
                INCREMENT the VALUE at key subset_sum IN left_sums AT position i BY one  
            END FOR  
            
            FOR each combo IN all COMBINATIONS OF right_half TAKING i elements  
                SET subset_sum TO the SUM OF elements IN combo  
                IF subset_sum IS NOT A KEY IN the DICTIONARY at position i OF right_sums  
                    SET the VALUE at key subset_sum IN right_sums AT position i TO zero  
                END IF  
                INCREMENT the VALUE at key subset_sum IN right_sums AT position i BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n INCLUSIVE  
            SET left_values TO the SORTED list OF ALL KEYS IN the DICTIONARY at position i OF left_sums  
            SET right_values TO the SORTED list OF ALL KEYS IN the DICTIONARY at position n MINUS i OF right_sums  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element at position j OF left_values PLUS the element at position k OF right_values  
                SET diff_candidate TO the ABSOLUTE VALUE OF total_sum MINUS two TIMES current_sum  
                IF diff_candidate LESS THAN min_diff  
                    SET min_diff TO diff_candidate  
                END IF  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
