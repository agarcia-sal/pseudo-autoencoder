CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the value attribute TO parameter val  
        SET the next attribute TO parameter next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the parameter values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode OBJECT INITIALIZED WITH the ELEMENT AT POSITION zero OF the parameter values  
    SET p TO head  
    FOR each val IN THE SUBSEQUENCE OF values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode OBJECT INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN THE RESULT OF THE LOGICAL EXPRESSION THAT the value ATTRIBUTE OF p1 EQUALS the value ATTRIBUTE OF p2 AND THE RECURSIVE CALL TO is_same_list WITH THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETERS self AND head  
        IF head IS none OR the next ATTRIBUTE OF head IS none  
            RETURN  
        END IF  
        SET slow AND fast TO head  
        WHILE fast IS NOT none AND the next ATTRIBUTE OF fast IS NOT none  
            SET slow TO the next ATTRIBUTE OF slow  
            SET fast TO the next ATTRIBUTE OF the next ATTRIBUTE OF fast  
        END WHILE  
        SET prev TO none  
        SET curr TO slow  
        WHILE curr IS NOT none  
            SET next_temp TO the next ATTRIBUTE OF curr  
            SET the next ATTRIBUTE OF curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE the next ATTRIBUTE OF second IS NOT none  
            SET temp1 TO the next ATTRIBUTE OF first  
            SET temp2 TO the next ATTRIBUTE OF second  
            SET the next ATTRIBUTE OF first TO second  
            SET the next ATTRIBUTE OF second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
