CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND the element at position i of nums IS GREATER THAN OR EQUAL TO the element at position i PLUS one of nums HOLDS  
            DECREMENT i BY one  
        END WHILE  
        IF i IS GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE the element at position j of nums IS LESS THAN OR EQUAL TO the element at position i of nums HOLDS  
                DECREMENT j BY one  
            END WHILE  
            SET temporary_variable TO the element at position i of nums  
            SET the element at position i of nums TO the element at position j of nums  
            SET the element at position j of nums TO temporary_variable  
        END IF  
        SET left_index TO i PLUS one  
        SET right_index TO the LENGTH OF nums MINUS one  
        WHILE left_index IS LESS THAN right_index HOLDS  
            SET temporary_variable TO the element at position left_index of nums  
            SET the element at position left_index of nums TO the element at position right_index of nums  
            SET the element at position right_index of nums TO temporary_variable  
            INCREMENT left_index BY one  
            DECREMENT right_index BY one  
        END WHILE  
    END FUNCTION  
END CLASS
