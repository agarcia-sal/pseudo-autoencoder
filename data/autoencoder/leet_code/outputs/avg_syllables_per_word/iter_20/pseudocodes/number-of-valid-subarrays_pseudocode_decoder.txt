CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO an empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the stack is NOT empty AND the element at the last position of the stack IS GREATER THAN num  
                REMOVE the last element FROM the stack  
            END WHILE  
            APPEND num TO the stack  
            INCREMENT count BY the LENGTH OF the stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
