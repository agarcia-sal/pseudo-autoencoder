CLASS Solution  
    FUNCTION uniquePaths(m integer, n integer) RETURNS integer  
        SET dp TO a two dimensional array with m rows and n columns where each element is set TO one  
        FOR each integer i FROM one TO m MINUS one  
            FOR each integer j FROM one TO n MINUS one  
                SET element at position i of dp and position j TO element at position i MINUS one of dp and position j PLUS element at position i of dp and position j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of dp and position n MINUS one  
    END FUNCTION  
END CLASS
