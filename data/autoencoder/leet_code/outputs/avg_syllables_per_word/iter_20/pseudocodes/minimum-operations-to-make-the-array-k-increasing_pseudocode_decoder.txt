CLASS Solution  
    FUNCTION kIncreasing(arr AS List OF integers, k AS integer) RETURNS integer  
        FUNCTION longest_increasing_subsequence(subarr AS List OF integers) RETURNS integer  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis IS empty OR num IS GREATER THAN OR EQUAL TO element at position the LENGTH OF lis MINUS one of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position returned by bisect right function applied to lis and num  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  

        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  

        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one IN STEPS OF k  
                APPEND element at position i of arr TO subarr  
            END FOR  
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  

        RETURN total_operations  
    END FUNCTION  
END CLASS
