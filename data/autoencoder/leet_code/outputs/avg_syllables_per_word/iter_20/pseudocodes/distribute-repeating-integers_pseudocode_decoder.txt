CLASS Solution  
    FUNCTION canDistribute(nums AS List of Integer, quantity AS List of Integer) RETURNS Boolean  
        SET count TO the LIST of VALUES OF the FREQUENCY COUNTER of nums  
        SORT quantity IN DESCENDING ORDER  
        FUNCTION can_satisfy(index AS Integer) RETURNS Boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN True  
            END IF  
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF the ELEMENT at POSITION i of count IS GREATER THAN OR EQUAL TO the ELEMENT at POSITION index of quantity THEN  
                    DECREMENT the ELEMENT at POSITION i of count BY the ELEMENT at POSITION index of quantity  
                    IF the RESULT OF can_satisfy index PLUS one IS True THEN  
                        RETURN True  
                    END IF  
                    INCREMENT the ELEMENT at POSITION i of count BY the ELEMENT at POSITION index of quantity  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN the RESULT OF can_satisfy zero  
    END FUNCTION  
END CLASS
