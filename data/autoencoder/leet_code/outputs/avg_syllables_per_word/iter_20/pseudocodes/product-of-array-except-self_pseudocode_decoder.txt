CLASS Solution  
    FUNCTION productExceptSelf(nums)  
        SET n TO the LENGTH OF nums  
        SET answer TO a list with n elements each set TO one  
        SET left_product TO one  
        FOR index FROM zero TO n MINUS one  
            SET the element at position index of answer TO left_product  
            SET left_product TO left_product MULTIPLIED BY the element at position index of nums  
        END FOR  
        SET right_product TO one  
        FOR index FROM n MINUS one DOWN TO zero  
            SET the element at position index of answer TO the element at position index of answer MULTIPLIED BY right_product  
            SET right_product TO right_product MULTIPLIED BY the element at position index of nums  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
