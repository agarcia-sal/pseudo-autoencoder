CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO new empty collection of key value pairs  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            SET element at key consisting of depth and pos of tree TO value  
        END FOR  

        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at key consisting of depth and pos of tree  
            IF current_value EQUALS nothing  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair consisting of depth PLUS one and two MULTIPLIED BY pos MINUS one  
            SET right_child TO a pair consisting of depth PLUS one and two MULTIPLIED BY pos  
            IF the element at key left_child of tree EQUALS nothing AND the element at key right_child of tree EQUALS nothing  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs with arguments the first part of left_child, the second part of left_child and new_path_sum  
            SET right_sum TO dfs with arguments the first part of right_child, the second part of right_child and new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  

        RETURN dfs with arguments one, one and zero  
    END FUNCTION  
END CLASS
