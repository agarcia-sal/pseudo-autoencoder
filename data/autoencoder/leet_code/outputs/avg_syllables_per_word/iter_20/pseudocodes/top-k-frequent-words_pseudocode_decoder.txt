CLASS Solution  
    FUNCTION topKFrequent(words AS List OF strings, k AS integer) RETURNS List OF strings  
        SET count TO the frequency mapping of each word IN words  
        SET heap TO a list constructed by FOR each word AND frequency IN count DO create a tuple of negative frequency AND word  
        CONVERT heap INTO a heap structure for efficient extraction  
        SET result TO a list constructed by REPEATING k times DO extract the element with the highest priority from heap and take the word element of the tuple  
        RETURN result  
    END FUNCTION  
END CLASS
