CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i IS GREATER THAN OR EQUAL TO the LENGTH OF list t  
                APPEND the concatenation of all elements of list t TO list ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF element at position i of list t IS an alphabetical character  
                SET element at position i of list t TO the character obtained by toggling case of element at position i of list t  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        SET list t TO the list of characters converted from string s  
        SET list ans TO an empty list  
        CALL dfs WITH zero  
        RETURN list ans  
    END FUNCTION  
END CLASS
