CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO RESULT OF Counter APPLIED TO nums  
        SET end_with TO RESULT OF Counter APPLIED WITHOUT ARGUMENTS  
        
        FOR each num IN nums  
            IF the element at position num of num_count EQUALS zero  
                CONTINUE TO next iteration  
            END IF  
            
            IF the element at position num MINUS one of end_with IS GREATER THAN zero  
                DECREMENT the element at position num MINUS one of end_with BY one  
                INCREMENT the element at position num of end_with BY one  
            ELSE IF the element at position num PLUS one of num_count IS GREATER THAN zero AND the element at position num PLUS two of num_count IS GREATER THAN zero  
                DECREMENT the element at position num PLUS one of num_count BY one  
                DECREMENT the element at position num PLUS two of num_count BY one  
                INCREMENT the element at position num PLUS two of end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the element at position num of num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
