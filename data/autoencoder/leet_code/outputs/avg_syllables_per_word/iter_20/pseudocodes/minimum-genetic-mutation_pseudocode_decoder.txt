    FUNCTION minMutation(start, end, bank)  
        SET s TO THE SET OF ELEMENTS IN bank  
        INITIALIZE q AS A DOUBLE ENDED QUEUE CONTAINING THE PAIR OF start AND zero  
        SET mp TO A MAPPING OF LETTER A TO THE STRING TCG AND LETTER T TO THE STRING ACG AND LETTER C TO THE STRING ATG AND LETTER G TO THE STRING ATC  
        WHILE THE COLLECTION q IS NOT EMPTY  
            REMOVE THE FIRST PAIR FROM q AND ASSIGN THE FIRST ELEMENT TO t AND THE SECOND ELEMENT TO step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR EACH PAIR OF i AND v WHERE i IS THE INDEX AND v IS THE ELEMENT AT POSITION i OF t  
                FOR EACH LETTER j IN THE STRING THAT mp MAPS FROM THE LETTER v  
                    SET next TO THE CONCATENATION OF THE SUBSTRING FROM POSITION ZERO TO i OF t AND LETTER j AND THE SUBSTRING FROM POSITION i PLUS one TO THE LENGTH OF t OF t  
                    IF next IS AN ELEMENT OF s  
                        APPEND THE PAIR OF next AND step PLUS one TO THE COLLECTION q  
                        REMOVE next FROM THE SET s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN MINUS ONE  
    END FUNCTION  
END CLASS
