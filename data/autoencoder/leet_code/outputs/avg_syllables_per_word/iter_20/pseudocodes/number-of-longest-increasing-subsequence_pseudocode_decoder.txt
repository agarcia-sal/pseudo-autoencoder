CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF nums  
        SET lengths TO a list where each element is one and the total count of elements is n  
        SET counts TO a list where each element is one and the total count of elements is n  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position i of nums IS GREATER THAN the element at position j of nums THEN  
                    IF the element at position j of lengths PLUS one IS GREATER THAN the element at position i of lengths THEN  
                        SET the element at position i of lengths TO the element at position j of lengths PLUS one  
                        SET the element at position i of counts TO the element at position j of counts  
                    ELSE IF the element at position j of lengths PLUS one EQUALS the element at position i of lengths THEN  
                        INCREMENT the element at position i of counts BY the element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET longest TO the maximum VALUE in lengths  
        SET total_count TO zero  
        FOR index i FROM zero TO n MINUS one  
            IF the element at position i of lengths EQUALS longest THEN  
                INCREMENT total_count BY the element at position i of counts  
            END IF  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
