CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF the remainder of num DIVIDED BY two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            SET min_value TO the smaller value BETWEEN min_value AND num  
            ADD the negative value of num TO max_heap as a heap element  
        END FOR  
        
        SET min_deviation TO positive infinity  
        
        WHILE the element at position zero of max_heap MODULO two EQUALS zero  
            SET max_value TO the negative value of the heap element removed from max_heap  
            SET min_deviation TO the smaller value BETWEEN min_deviation AND the difference BETWEEN max_value AND min_value  
            SET max_value TO the integer division of max_value BY two  
            SET min_value TO the smaller value BETWEEN min_value AND max_value  
            ADD the negative value of max_value TO max_heap as a heap element  
        END WHILE  
        
        SET min_deviation TO the smaller value BETWEEN min_deviation AND the difference BETWEEN the negative value of the element at position zero of max_heap AND min_value  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
