CLASS Solution  
    FUNCTION numRescueBoats(people AS list of integers, limit AS integer) RETURNS integer  
        EXECUTE sorting of the list people in ascending order  
        SET left TO zero  
        SET right TO the length of the list people MINUS one  
        SET boats TO zero  
        WHILE the value of left IS LESS THAN OR EQUAL TO the value of right  
            IF the element at position left of the list people PLUS the element at position right of the list people IS LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
