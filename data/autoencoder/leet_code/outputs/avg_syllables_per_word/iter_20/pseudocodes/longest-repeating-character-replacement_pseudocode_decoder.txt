CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a new mapping from characters TO integer counts with default zero  
        
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count at the character at position right of s BY one  
            SET max_count TO the GREATER OF max_count AND the count at the character at position right of s  
            
            IF the LENGTH OF the current window defined by right MINUS left PLUS one MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the count at the character at position left of s BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND the LENGTH OF the current window defined by right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
