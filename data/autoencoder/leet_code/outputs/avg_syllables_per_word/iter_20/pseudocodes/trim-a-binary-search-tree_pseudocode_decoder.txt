CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the CONDITION values IS EMPTY holds  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DEQUE COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE ELEMENT REMOVED FROM THE LEFT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET THE left ATTRIBUTE OF node TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET THE right ATTRIBUTE OF node TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF THE val ATTRIBUTE OF p IS NOT EQUAL TO THE val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN CALL is_same_tree WITH THE left ATTRIBUTE OF p AND THE left ATTRIBUTE OF q AND AND CALL is_same_tree WITH THE right ATTRIBUTE OF p AND THE right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST WITH PARAMETERS self root low high  
        IF root IS none  
            RETURN none  
        END IF  
        IF THE val ATTRIBUTE OF root IS LESS THAN low  
            RETURN CALL self trimBST WITH THE right ATTRIBUTE OF root AND low AND high  
        END IF  
        IF THE val ATTRIBUTE OF root IS GREATER THAN high  
            RETURN CALL self trimBST WITH THE left ATTRIBUTE OF root AND low AND high  
        END IF  
        SET THE left ATTRIBUTE OF root TO CALL self trimBST WITH THE left ATTRIBUTE OF root AND low AND high  
        SET THE right ATTRIBUTE OF root TO CALL self trimBST WITH THE right ATTRIBUTE OF root AND low AND high  
        RETURN root  
    END FUNCTION  
END CLASS
