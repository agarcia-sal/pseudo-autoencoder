CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of size n filled with zero  
        SET stack TO an empty list  
        FOR each log IN logs  
            SPLIT log INTO function_id string action string AND timestamp string BY the colon delimiter  
            SET function_id TO the integer value of function_id string  
            SET timestamp TO the integer value of timestamp string  
            IF action EQUALS start  
                IF the stack is not empty  
                    SET prev_function_id TO the function identifier of the element at the last position of the stack  
                    SET prev_timestamp TO the timestamp of the element at the last position of the stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple of function_id AND timestamp TO stack  
            ELSE  
                REMOVE the last element FROM stack AND ASSIGN it TO prev_function_id AND prev_timestamp respectively  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the stack is not empty  
                    SET the function identifier of the element at the last position of the stack TO itself  
                    SET the timestamp of the element at the last position of the stack TO timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
