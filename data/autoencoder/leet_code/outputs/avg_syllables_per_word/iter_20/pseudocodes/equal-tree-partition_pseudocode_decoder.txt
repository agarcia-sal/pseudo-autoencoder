CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode created WITH element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode created WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode created WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value at p IS NOT EQUALS TO the value at q  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS node left of p AND node left of q AND is_same_tree WITH PARAMETERS node right of p AND node right of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION checkEqualTree WITH PARAMETERS self root  
        FUNCTION subtree_sum WITH PARAMETER node  
            IF node IS None  
                RETURN zero  
            END IF  
            RETURN the value at node PLUS subtree_sum WITH PARAMETER node left PLUS subtree_sum WITH PARAMETER node right  
        END FUNCTION  

        SET total_sum TO subtree_sum WITH PARAMETER root  

        IF total_sum MODULO two IS NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False  

        FUNCTION can_split WITH PARAMETER node  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO the value at node PLUS can_split WITH PARAMETER node left PLUS can_split WITH PARAMETER node right  
            IF current_sum EQUALS target AND node IS NOT root  
                SET the element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split WITH PARAMETER root  
        RETURN the element at position zero of found  
    END FUNCTION  
END CLASS
