CLASS Solution  
    FUNCTION palindromePartition(s, k)  
        FUNCTION min_changes_to_palindrome(sub)  
            SET position_i TO zero  
            SET position_j TO the LENGTH OF sub MINUS one  
            SET changes TO zero  
            WHILE position_i IS LESS THAN position_j  
                IF element at position position_i of sub IS NOT EQUALS element at position position_j of sub  
                    INCREMENT changes BY one  
                END IF  
                INCREMENT position_i BY one  
                DECREMENT position_j BY one  
            END WHILE  
            RETURN changes  
        END FUNCTION  
      
        SET length_n TO the LENGTH OF s  
        SET dp TO a list consisting of length_n PLUS one elements each being a list of k PLUS one elements each set to positive infinity value  
        SET element at position zero of element at position zero of dp TO zero  
      
        FOR index_i FROM one TO length_n  
            FOR index_j FROM one TO the smaller VALUE BETWEEN k AND index_i  
                FOR start FROM zero TO index_i MINUS one  
                    SET changes TO CALL min_changes_to_palindrome with the substring from position start TO index_i MINUS one of s  
                    SET element at position index_i of element at position index_j of dp TO the smaller VALUE BETWEEN element at position index_i of element at position index_j of dp AND element at position start of element at position index_j MINUS one of dp PLUS changes  
                END FOR  
            END FOR  
        END FOR  
      
        RETURN element at position length_n of element at position k of dp  
    END FUNCTION  
END CLASS
