CLASS Solution  
    FUNCTION minStickers(stickers AS List of strings, target AS string) RETURNS integer  
        SET stickers TO a list created by transforming each sticker IN stickers INTO a frequency counter of characters  
        SORT stickers IN ORDER OF descending BY maximum frequency value of characters in each sticker  
        SET queue TO a double ended queue initialized with a pair consisting of target and zero  
        SET memo TO a mapping from target TO zero  
        WHILE queue is not empty  
            REMOVE the leftmost pair FROM queue ASSIGNING current_target AND used_stickers  
            FOR each sticker IN stickers  
                IF the first character of current_target IS NOT PRESENT in the keys of sticker  
                    CONTINUE to the next iteration  
                END IF  
                SET remaining TO a frequency counter created from current_target  
                SUBTRACT the frequencies of sticker FROM remaining  
                SET new_target TO a string formed by concatenating all characters from remaining elements sorted in ascending order  
                IF new_target IS an empty string  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target IS NOT PRESENT in memo  
                    SET memo at new_target TO used_stickers PLUS one  
                    APPEND the pair of new_target AND used_stickers PLUS one TO the right end of queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
