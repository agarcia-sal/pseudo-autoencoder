CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO the LIST OF words EACH REVERSED AND THEN SORTED IN ASCENDING ORDER  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR the WORD at position index PLUS one OF reversed_words DOES NOT START WITH the WORD at position index OF reversed_words  
                INCREMENT total_length BY the LENGTH OF the WORD at position index OF reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
