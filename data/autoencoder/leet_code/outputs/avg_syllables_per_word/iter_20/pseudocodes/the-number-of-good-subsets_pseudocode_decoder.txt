CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of elements in nums  
        SET dp TO a list of zero values with length equal to two raised to the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the element at position one of count  
                APPLY modulo operation with MOD to the previous value  
                CONTINUE to next iteration  
            END IF  
            IF num is NOT a key in prime_factors  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO the value at key num in prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR with two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero  
                    SET the element at position bitwise OR of m and mask in dp TO the sum of the element at position bitwise OR of m and mask in dp PLUS the product of the element at position m in dp AND the element at position num in count  
                    APPLY modulo operation with MOD to the previous value  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of the elements of dp from position one to the end  
        APPLY modulo operation with MOD to result  
        RETURN result  
    END FUNCTION  
END CLASS
