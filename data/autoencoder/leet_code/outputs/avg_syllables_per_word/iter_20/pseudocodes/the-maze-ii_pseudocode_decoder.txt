CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze AS list of list of integers AND start AS list of integers AND destination AS list of integers  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the list containing pairs ONE AND ZERO MINUS ONE AND ZERO ZERO AND ONE ZERO AND MINUS ONE  
        CONVERT start TO a tuple  
        CONVERT destination TO a tuple  
        SET visited TO an empty set  
        SET priority queue pq TO a list containing the pair ZERO AND start  
        
        WHILE pq is not empty  
            REMOVE the smallest element FROM pq AND ASSIGN to dist AND position (x AND y)  
            
            IF position (x EQUALS x AND y EQUALS y) EQUALS destination  
                RETURN dist  
            END IF  
            
            IF position (x AND y) IS IN visited  
                CONTINUE TO the next iteration of the WHILE loop  
            END IF  
            
            ADD position (x AND y) TO visited  
            
            FOR each pair dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze at position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF position (new_x AND new_y) IS NOT IN visited  
                    ADD the pair dist PLUS count AND position (new_x AND new_y) TO pq USING heap push operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
