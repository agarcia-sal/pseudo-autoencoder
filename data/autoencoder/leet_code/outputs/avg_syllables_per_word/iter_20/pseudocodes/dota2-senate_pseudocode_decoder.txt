CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant AS an empty double-ended queue  
        INITIALIZE dire AS an empty double-ended queue  
  
        FOR each index AND senator_symbol IN the enumeration of senate  
            IF senator_symbol EQUALS the character representing Radiant  
                APPEND index TO radiant  
            ELSE  
                APPEND index TO dire  
            END IF  
        END FOR  
  
        WHILE the LENGTH OF radiant IS GREATER THAN zero AND the LENGTH OF dire IS GREATER THAN zero  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
  
            IF r_index IS LESS THAN d_index  
                APPEND the sum of r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND the sum of d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
  
        IF the LENGTH OF radiant IS GREATER THAN zero  
            RETURN the string representing Radiant  
        ELSE  
            RETURN the string representing Dire  
        END IF  
    END FUNCTION  
END CLASS
