CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each number IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero  
                SET g TO greatest common divisor OF element at position the LENGTH OF stack MINUS one of stack AND number  
                IF g EQUALS one  
                    BREAK the loop  
                END IF  
                SET number TO element at position the LENGTH OF stack MINUS one of stack MULTIPLIED BY number DIVIDED BY g  
                REMOVE element at position the LENGTH OF stack MINUS one FROM stack  
            END WHILE  
            APPEND number TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
