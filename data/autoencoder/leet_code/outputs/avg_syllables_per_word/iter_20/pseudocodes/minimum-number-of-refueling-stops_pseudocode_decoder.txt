CLASS Solution  
    FUNCTION minRefuelStops WITH PARAMETERS target startFuel stations  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list containing target AND zero TO stations  

        FOR position fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is LESS THAN position  
                SET largest_fuel_to_add TO the element obtained by removing the smallest element from max_heap  
                DECREMENT current_fuel BY negative of largest_fuel_to_add  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel is LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative of fuel TO max_heap maintaining heap order with auxiliary FUNCTION heap_push  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
