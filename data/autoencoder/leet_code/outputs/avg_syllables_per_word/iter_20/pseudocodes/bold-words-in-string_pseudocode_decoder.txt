CLASS Solution  
    FUNCTION boldWords(words AS List OF string, s AS string) RETURNS string  
        SET bold TO boolean list OF length EQUAL TO the LENGTH OF s WITH all values SET TO false  
        FOR each word IN words  
            SET start TO the POSITION OF word IN s  
            WHILE start IS NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i OF bold TO true  
                END FOR  
                SET start TO the POSITION OF word IN s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF s  
            IF the element at position i OF bold IS true  
                APPEND the string less than b greater than TO result  
                WHILE i IS LESS THAN the LENGTH OF s AND the element at position i OF bold IS true  
                    APPEND the character at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string less than slash b greater than TO result  
            ELSE  
                APPEND the character at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
