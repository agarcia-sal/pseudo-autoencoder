CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO false  
        SET result TO an empty list  
        SET current_line TO an empty list  

        FOR each line IN source  
            SET index TO zero  
            WHILE index LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the substring from position index TO position index PLUS one of line EQUALS the two characters star followed by slash  
                        SET in_block_comment TO false  
                        INCREMENT index BY one  
                    END IF  
                ELSE  
                    IF the substring from position index TO position index PLUS one of line EQUALS the two characters slash followed by slash  
                        BREAK out of the inner while loop  
                    ELSE IF the substring from position index TO position index PLUS one of line EQUALS the two characters slash followed by star  
                        SET in_block_comment TO true  
                        INCREMENT index BY one  
                    ELSE  
                        APPEND the element at position index of line TO current_line  
                    END IF  
                END IF  
                INCREMENT index BY one  
            END WHILE  

            IF current_line IS NOT empty AND in_block_comment IS false  
                APPEND the concatenation of all elements in current_line TO result  
                SET current_line TO an empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
