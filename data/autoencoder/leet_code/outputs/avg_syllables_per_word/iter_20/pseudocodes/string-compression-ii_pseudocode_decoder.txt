CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(s, k)  
        FUNCTION dp(i, k_value, prev_char, prev_count)  
            IF k_value LESS THAN zero  
                RETURN infinity value  
            END IF  
            IF i EQUALS length of s  
                RETURN zero  
            END IF  
            IF character at position i of s EQUALS prev_char  
                IF prev_count EQUALS one OR prev_count EQUALS nine OR prev_count EQUALS ninety nine  
                    SET increment TO one  
                ELSE  
                    SET increment TO zero  
                END IF  
                RETURN increment PLUS dp with parameters i PLUS one, k_value, prev_char, prev_count PLUS one  
            ELSE  
                SET keep_option TO one PLUS dp with parameters i PLUS one, k_value, character at position i of s, one  
                SET delete_option TO dp with parameters i PLUS one, k_value MINUS one, prev_char, prev_count  
                RETURN the smaller value BETWEEN keep_option AND delete_option  
            END IF  
        END FUNCTION  
        RETURN dp with parameters zero, k, empty string, zero  
    END FUNCTION  
END CLASS
