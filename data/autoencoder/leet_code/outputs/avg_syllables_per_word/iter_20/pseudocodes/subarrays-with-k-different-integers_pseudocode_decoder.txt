CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums LIST OF INTEGER, k INTEGER) RETURNS INTEGER  
        FUNCTION atMostKDistinct(t INTEGER) RETURNS INTEGER  
            SET count TO a new default dictionary mapping elements to zero  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the count for the element at position right of nums BY one  
                WHILE the NUMBER OF keys in count IS GREATER THAN t  
                    DECREMENT the count for the element at position left of nums BY one  
                    IF the count for the element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY the value of right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the value of atMostKDistinct applied to k MINUS the value of atMostKDistinct applied to k MINUS one  
    END FUNCTION  
END CLASS
