CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element at the front of queue removed from queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO NEW TreeNode initialized WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO NEW TreeNode initialized WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUALS the value property of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left property of p, the left property of q) AND is_same_tree(the right property of p, the right property of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a new double ended queue initialized WITH root as the sole element  
        SET leftmost_value TO none  
        WHILE the queue is not empty  
            SET level_size TO the LENGTH OF queue  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO the element at the front of queue removed from queue  
                IF index EQUALS zero  
                    SET leftmost_value TO the value property of node  
                END IF  
                IF the left property of node IS NOT none  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node IS NOT none  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
