CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EMPTY OR element at position zero of matrix IS EMPTY  
            RETURN false  
        END IF  
        
        SET row TO zero  
        SET col TO LENGTH OF element at position zero of matrix MINUS one  
        
        WHILE row IS LESS THAN LENGTH OF matrix AND col IS GREATER THAN OR EQUAL TO zero  
            IF element at position row of matrix AT position col EQUALS target  
                RETURN true  
            ELSE IF element at position row of matrix AT position col IS GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        
        RETURN false  
    END FUNCTION  
END CLASS
