CLASS Solution  
    FUNCTION maximumBeauty(flowers)  
        IMPORT default dictionary FROM collections  
        SET indices TO default dictionary WITH LIST AS DEFAULT VALUE  
        SET prefix_sum TO A LIST OF zero WITH THE LENGTH EQUAL TO THE LENGTH OF flowers PLUS one  
        FOR index FROM zero TO THE LENGTH OF flowers MINUS one  
            SET element AT POSITION index PLUS one OF prefix_sum TO element AT POSITION index OF prefix_sum PLUS THE GREATER OF element AT POSITION index OF flowers AND zero  
            APPEND index TO element AT POSITION flowers AT position index OF indices  
        END FOR  
        SET max_beauty TO NEGATIVE infinity  
        FOR each beauty IN keys OF indices  
            IF THE LENGTH OF element AT POSITION beauty OF indices IS GREATER THAN OR EQUAL TO two  
                SET first TO element AT POSITION zero OF element AT POSITION beauty OF indices  
                SET last TO element AT POSITION THE LENGTH OF element AT POSITION beauty OF indices MINUS one OF element AT POSITION beauty OF indices  
                SET current_beauty TO two MULTIPLIED BY beauty PLUS element AT POSITION last OF prefix_sum MINUS element AT POSITION first PLUS one OF prefix_sum  
                IF current_beauty IS GREATER THAN max_beauty  
                    SET max_beauty TO current_beauty  
                END IF  
            END IF  
        END FOR  
        RETURN max_beauty  
    END FUNCTION  
END CLASS
