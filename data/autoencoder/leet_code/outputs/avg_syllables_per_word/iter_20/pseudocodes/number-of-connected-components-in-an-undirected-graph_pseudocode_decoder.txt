CLASS Solution  
    FUNCTION countComponents(n, edges)  
        SET graph TO an empty mapping from integers TO list of integers  
        FOR each integer i FROM zero TO n MINUS one  
            SET element at position i of graph TO an empty list  
        END FOR  
        FOR each pair of integers a and b IN edges  
            APPEND b TO element at position a of graph  
            APPEND a TO element at position b of graph  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs(node)  
            IF node IS AN ELEMENT OF visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN element at position node of graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR each integer node FROM zero TO n MINUS one  
            IF node IS NOT AN ELEMENT OF visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
