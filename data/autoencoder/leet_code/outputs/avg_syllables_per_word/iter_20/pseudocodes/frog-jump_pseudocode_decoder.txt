CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the SET OF stones  
        SET memo TO an EMPTY MAP  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at THE LAST POSITION OF stones  
                RETURN True  
            END IF  
            
            IF the PAIR OF position AND jump EXISTS AS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo  
            END IF  
            
            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND position PLUS next_jump EXISTS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump) RETURNS True  
                        SET the VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the element at THE FIRST POSITION OF stones, zero)  
    END FUNCTION  
END CLASS
