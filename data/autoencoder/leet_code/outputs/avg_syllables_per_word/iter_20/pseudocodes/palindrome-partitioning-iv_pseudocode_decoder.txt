CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN the value of sub EQUALS the value of sub in reverse order  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO A TWO DIMENSIONAL list of FALSE values with dimensions n BY n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF element at position i of s EQUALS element at position j of s AND (the VALUE OF j MINUS i IS LESS THAN OR EQUAL TO one OR the VALUE of element at position i PLUS one AND j MINUS one in is_pal IS TRUE)  
                    SET element at position i AND j in is_pal TO TRUE  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF element at position zero AND i MINUS one in is_pal IS TRUE  
                FOR j FROM i TO n MINUS two  
                    IF element at position i AND j in is_pal IS TRUE AND element at position j PLUS one AND n MINUS one in is_pal IS TRUE  
                        RETURN TRUE  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
