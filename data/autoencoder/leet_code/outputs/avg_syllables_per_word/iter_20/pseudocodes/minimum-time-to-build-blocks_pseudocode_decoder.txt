CLASS Solution  
    FUNCTION minBuildTime(blocks AS List OF Integer, split AS Integer) RETURNS Integer  
        CALL heapify ON blocks TO ORGANIZE blocks AS A MINIMUM HEAP  
        WHILE the LENGTH OF blocks IS GREATER THAN one  
            SET first TO REMOVE the SMALLEST element FROM blocks USING heap pop operation  
            SET second TO REMOVE the NEXT SMALLEST element FROM blocks USING heap pop operation  
            SET combined_time TO split PLUS second  
            CALL heap push operation TO ADD combined_time INTO blocks  
        END WHILE  
        RETURN the ELEMENT AT position zero OF blocks  
    END FUNCTION  
END CLASS
