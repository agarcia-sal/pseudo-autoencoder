CLASS Solution  
    FUNCTION minMoves(nums LIST OF INTEGER, k INTEGER) RETURNS INTEGER  
        SET positions TO an empty list  
        FOR each index AND number IN nums with enumeration  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start INTEGER, end INTEGER) RETURNS INTEGER  
            SET middle TO start PLUS end DIVIDED BY two  
            SET median_value TO element at position middle of positions  
            SET total_cost TO zero  
            FOR index FROM start TO end MINUS one  
                SET offset_difference TO median_value MINUS middle MINUS index PLUS index  
                SET distance TO absolute value of element at position index of positions MINUS offset_difference  
                INCREMENT total_cost BY distance  
            END FOR  
            RETURN total_cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR i FROM zero TO the LENGTH OF positions MINUS k  
            SET candidate_cost TO result of calculate_cost WITH i AND i PLUS k  
            IF candidate_cost LESS THAN min_cost  
                SET min_cost TO candidate_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
