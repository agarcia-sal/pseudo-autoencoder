CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET string_to_process TO expression WITH each occurrence of hyphen REPLACED BY plus FOLLOWED BY hyphen  
        SET list_of_fraction_strings TO the SPLIT of string_to_process BY plus  
        SET list_of_non_empty_fraction_strings TO the FILTERING of list_of_fraction_strings RETAINING only elements that are NOT empty  
        SET list_of_fractions TO the CREATION of a list by CONVERTING each element of list_of_non_empty_fraction_strings TO a fraction object USING the standard fraction conversion  
        SET result_fraction TO the SUMMATION of all elements in list_of_fractions  
        SET numerator_value TO the numerator component of result_fraction  
        SET denominator_value TO the denominator component of result_fraction  
        SET result_string TO the CONCATENATION OF numerator_value FOLLOWED BY the symbol representing division FOLLOWED BY denominator_value AS textual components  
        RETURN result_string  
    END FUNCTION  
END CLASS
