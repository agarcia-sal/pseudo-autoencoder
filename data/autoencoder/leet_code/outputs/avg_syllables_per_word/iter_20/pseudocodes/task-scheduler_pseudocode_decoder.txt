CLASS Solution  
    FUNCTION leastInterval(tasks AS List OF strings, n AS integer) RETURNS integer  
        SET task_counts TO CALL Counter WITH tasks  
        SET max_heap TO empty list  
        FOR each count IN the VALUES OF task_counts  
            APPEND negative count TO max_heap  
        END FOR  
        CALL heapify ON max_heap  
        
        SET time TO zero  
        SET queue TO empty list  
        
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND the FIRST element at position zero of queue AT position zero EQUALS time  
                SET _, count TO REMOVE the FIRST element at position zero FROM queue  
                IF count LESS THAN zero  
                    CALL heappush WITH max_heap AND count  
                END IF  
            END WHILE  
            
            IF max_heap is not empty  
                SET count TO CALL heappop WITH max_heap  
                SET count TO count PLUS one  
                IF count LESS THAN zero  
                    APPEND a tuple WITH first element time PLUS n PLUS one AND second element count TO queue  
                END IF  
            END IF  
            
            SET time TO time PLUS one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
