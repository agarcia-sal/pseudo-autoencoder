CLASS Solution  
    FUNCTION exist(board AS List OF List OF string, word AS string) RETURNS boolean  
        IF board EQUALS empty THEN  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a new two dimensional list of boolean values WITH m rows and n columns each SET TO False  

        FUNCTION dfs(x AS integer, y AS integer, index AS integer) RETURNS boolean  
            IF index EQUALS the LENGTH OF word THEN  
                RETURN True  
            END IF  

            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the value at position x of visited at position y EQUALS True OR the character at position x of board at position y NOT EQUALS the character at position index of word THEN  
                RETURN False  
            END IF  

            SET the value at position x of visited at position y TO True  

            FOR each pair dx dy IN the list of pairs consisting of negative one and zero, one and zero, zero and negative one, zero and one  
                IF the result of calling dfs with arguments x PLUS dx, y PLUS dy, index PLUS one EQUALS True THEN  
                    RETURN True  
                END IF  
            END FOR  

            SET the value at position x of visited at position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the result of calling dfs with arguments i, j, zero EQUALS True THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
