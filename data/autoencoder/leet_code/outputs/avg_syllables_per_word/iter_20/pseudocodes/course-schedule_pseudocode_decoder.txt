CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a dictionary that returns an empty list for missing keys  
        SET indegree TO a list with the length of numCourses where each element is zero  
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO the list at the key prereq in graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        SET queue TO a double-ended queue initialized with each course FROM zero TO numCourses MINUS one WHERE the element at position course of indegree EQUALS zero  
        WHILE queue is not empty  
            REMOVE and RETURN the element from the left end of queue and ASSIGN to current_course  
            DECREMENT numCourses BY one  
            FOR each next_course IN the list at the key current_course in graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
