CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct WITH parameter s  
        DECLARE char_count AS a dictionary defaulting to zero for new keys  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the value at key character at position right of s IN char_count BY one  
            
            WHILE the NUMBER OF keys in char_count IS GREATER THAN two  
                DECREMENT the value at key character at position left of s IN char_count BY one  
                IF the value at key character at position left of s IN char_count EQUALS zero  
                    REMOVE the key character at position left of s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND the result of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
