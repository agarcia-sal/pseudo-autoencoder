CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums LIST OF INTEGER, k INTEGER) RETURNS BOOLEAN  
        SET freq TO the collection of element frequencies in nums  
        SET max_freq TO the maximum frequency value in freq  
        SET length_of_nums TO the LENGTH OF nums  
        IF length_of_nums DIVIDED BY k IS GREATER THAN OR EQUAL TO max_freq THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
