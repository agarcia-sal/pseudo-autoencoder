CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers FROM zero TO size MINUS one  
        SET self.rank TO list of zeros WITH LENGTH size  
        SET self.size TO list of ones WITH LENGTH size  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUAL TO u  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
                INCREMENT element at position rootV of self.size BY element at position rootU of self.size  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each char IN word  
                SET mask TO mask OR one SHIFTED LEFT BY ASCII VALUE OF char MINUS ASCII VALUE OF letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO new dictionary with default value as list  
        SET n TO LENGTH OF words  
        SET uf TO new UnionFind with n  

        FOR each index i and word IN enumeration of words  
            SET mask TO bitmask(word)  
            APPEND i TO element at position mask of word_map  
        END FOR  

        FOR each index i and word IN enumeration of words  
            SET mask TO bitmask(word)  
            FOR j FROM zero TO twenty five  
                SET new_mask TO mask XOR one SHIFTED LEFT BY j  
                IF new_mask IS IN word_map  
                    FOR each k IN element at position new_mask of word_map  
                        CALL uf.union(i, k)  
                    END FOR  
                END IF  

                IF mask AND one SHIFTED LEFT BY j EQUALS zero  
                    SET new_mask TO mask OR one SHIFTED LEFT BY j  
                    IF new_mask IS IN word_map  
                        FOR each k IN element at position new_mask of word_map  
                            CALL uf.union(i, k)  
                        END FOR  
                    END IF  
                END IF  

                IF mask AND one SHIFTED LEFT BY j IS NOT EQUAL TO zero  
                    FOR k FROM zero TO twenty five  
                        IF mask AND one SHIFTED LEFT BY k EQUALS zero  
                            SET new_mask TO mask XOR one SHIFTED LEFT BY j OR one SHIFTED LEFT BY k  
                            IF new_mask IS IN word_map  
                                FOR each l IN element at position new_mask of word_map  
                                    CALL uf.union(i, l)  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO new dictionary with default value as zero  
        FOR i FROM zero TO n MINUS one  
            SET root TO uf.find(i)  
            INCREMENT element at position root of root_sizes BY one  
        END FOR  

        SET max_group_size TO maximum value of root_sizes  
        SET num_groups TO LENGTH OF root_sizes  

        RETURN list consisting of num_groups followed by max_group_size  
    END FUNCTION  
END CLASS
