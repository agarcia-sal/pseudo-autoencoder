CLASS ListNode  
    FUNCTION __init__ WITH parameters val defaulted to zero AND next defaulted to none  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements from position one TO the end of values  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the logical AND of the condition that the value attribute of p1 EQUALS the value attribute of p2 AND the result of calling is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH parameter head  
        SET dummy TO a new ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        WHILE the next attribute of current IS NOT none AND the next attribute of the next attribute of current IS NOT none  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
