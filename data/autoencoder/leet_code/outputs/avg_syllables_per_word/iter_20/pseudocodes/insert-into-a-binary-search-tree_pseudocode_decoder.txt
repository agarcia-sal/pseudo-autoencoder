CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting TO zero left defaulting TO none right defaulting TO none  
        SET the property val OF self TO val  
        SET the property left OF self TO left  
        SET the property right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the property left OF node TO a new instance of TreeNode initialized WITH the element at position i of values  
            APPEND the property left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the property right OF node TO a new instance of TreeNode initialized WITH the element at position i of values  
            APPEND the property right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the property val OF p IS NOT EQUAL TO the property val OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters the property left OF p and the property left OF q AND is_same_tree with parameters the property right OF p and the property right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST with parameters root and val  
        FUNCTION dfs with parameter root  
            IF root IS none  
                RETURN a new instance of TreeNode initialized WITH val  
            END IF  
            IF the property val OF root IS LESS THAN val  
                SET the property right OF root TO dfs with parameter the property right OF root  
            ELSE  
                SET the property left OF root TO dfs with parameter the property left OF root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs with parameter root  
    END FUNCTION  
END CLASS
