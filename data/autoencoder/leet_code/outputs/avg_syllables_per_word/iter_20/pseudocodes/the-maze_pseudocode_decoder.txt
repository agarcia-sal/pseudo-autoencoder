CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze, start, destination  
        SET directions TO list of pairs of integers representing zero and one in the four cardinal directions right down left up as follows zero and one one and zero zero and negative one negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO empty set  
          
        FUNCTION dfs WITH PARAMETERS x, y  
            IF the PAIR of x and y IS IN visited  
                RETURN False  
            END IF  
            IF the LIST consisting of x and y EQUALS destination  
                RETURN True  
            END IF  
            ADD the PAIR of x and y TO visited  
              
            FOR each pair dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze at position ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs WITH PARAMETERS nx and ny RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
          
        RETURN dfs WITH PARAMETERS element at position zero of start and element at position one of start  
    END FUNCTION  
END CLASS
