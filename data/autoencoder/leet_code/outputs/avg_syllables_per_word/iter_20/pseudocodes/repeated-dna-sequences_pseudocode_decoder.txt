CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        ASSIGN pointer TO a new dictionary that returns zero for missing keys  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO index PLUS ten MINUS one of string s  
            INCREMENT the value at key sequence in the dictionary BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key value pair sequence AND count IN the dictionary  
            IF count IS GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
