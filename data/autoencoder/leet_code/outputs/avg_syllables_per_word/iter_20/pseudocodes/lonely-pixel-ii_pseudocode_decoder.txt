CLASS Solution  
    FUNCTION findBlackPixel(picture AS List OF List OF string, target AS Integer) RETURNS Integer  
        IF picture IS empty OR element at position zero of picture IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of m zeros  
        SET col_count TO a list of n zeros  
        SET row_patterns TO a mapping from strings to integers initially empty and defaulting to zero  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at each position in element at position r of picture  
            INCREMENT the value at key pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character black pixel  
                    INCREMENT the element at position r of row_count BY one  
                    INCREMENT the element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character black pixel  
                    SET pattern TO the concatenation of all elements at each position in element at position r of picture  
                    IF the element at position r of row_count EQUALS target AND the element at position c of col_count EQUALS target AND the value at key pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
