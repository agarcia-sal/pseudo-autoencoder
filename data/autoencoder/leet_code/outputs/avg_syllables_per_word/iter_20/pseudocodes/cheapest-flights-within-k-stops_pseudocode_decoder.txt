CLASS Solution  
    FUNCTION findCheapestPrice WITH PARAMETERS n flights src dst k  
        SET graph TO an empty mapping from each integer FROM zero TO n MINUS one TO an empty list  
        FOR each triple u v w IN flights  
            APPEND a pair consisting of v AND w TO the element at position u of graph  
        END FOR  
        SET min_heap TO a list containing a triple of zero MINUS one AND src  
        SET min_cost TO a mapping with src as key AND minus one as the value  
        WHILE min_heap is not empty  
            EXTRACT the smallest element from min_heap AND ASSIGN to cost stops city  
            IF city EQUALS dst  
                RETURN cost  
            END IF  
            IF stops IS GREATER THAN OR EQUAL TO k  
                CONTINUE to the next iteration of WHILE  
            END IF  
            FOR each pair neighbor price IN the element at position city of graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                IF neighbor is not a key in min_cost OR next_stops IS LESS THAN the value associated with neighbor in min_cost OR next_cost IS LESS THAN the first element of the smallest element in min_heap  
                    SET the value associated with neighbor in min_cost TO next_stops  
                    INSERT the triple consisting of next_cost next_stops AND neighbor INTO min_heap maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
