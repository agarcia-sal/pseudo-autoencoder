CLASS Solution  
    FUNCTION maxEqualFreq(nums)  
        SET count TO a new empty frequency counter  
        SET freq TO a new empty frequency counter for frequencies  
        SET max_len TO zero  
        
        FOR each index i AND element num IN nums  
            INCREMENT the frequency count of num BY one  
            
            IF the frequency count of num IS GREATER THAN one THEN  
                DECREMENT the frequency count of the frequency count of num MINUS one BY one  
                IF the frequency count of the frequency count of num MINUS one IS EQUAL TO zero THEN  
                    REMOVE the frequency count of num MINUS one FROM freq  
                END IF  
            END IF  
            
            INCREMENT the frequency count of the frequency count of num BY one  
            
            IF the LENGTH OF freq IS EQUAL TO one AND (freq CONTAINS the key one OR THE ONLY VALUE IN freq IS EQUAL TO one) THEN  
                SET max_len TO the current index i PLUS one  
            ELSE IF the LENGTH OF freq IS EQUAL TO two AND ((freq CONTAINS the key one AND THE frequency count for key one IS EQUAL TO one) OR (THE DIFFERENCE BETWEEN the MAXIMUM KEY of freq AND the MINIMUM KEY of freq IS EQUAL TO one AND THE frequency count for the MAXIMUM KEY of freq IS EQUAL TO one)) THEN  
                SET max_len TO the current index i PLUS one  
            END IF  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
