CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each number IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack IS LESS THAN number  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND number TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  
  
        FUNCTION merge(arr1, arr2)  
            SET result TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR each count FROM one TO total_length  
                IF arr1 IS GREATER THAN arr2 WHEN COMPARED LEXICOGRAPHICALLY  
                    APPEND the element at position zero of arr1 TO result  
                    REMOVE the element at position zero FROM arr1  
                ELSE  
                    APPEND the element at position zero of arr2 TO result  
                    REMOVE the element at position zero FROM arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
  
        SET max_number TO an empty list  
        FOR each i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merge WHEN CALLED WITH maxSingleArray OF nums1 AND i AND maxSingleArray OF nums2 AND k MINUS i  
                IF max_number IS LESS THAN candidate WHEN COMPARED LEXICOGRAPHICALLY  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
  
        RETURN max_number  
    END FUNCTION  
END CLASS
