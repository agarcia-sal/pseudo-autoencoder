CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the quotient of pile DIVIDED BY k  
            END FOR  
            RETURN whether hours IS LESS THAN OR EQUAL TO h  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the maximum element of piles  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the quotient of the sum of left AND right DIVIDED BY two  
            IF canFinish(mid) IS true  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
