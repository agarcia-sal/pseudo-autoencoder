CLASS Solution  
    FUNCTION minCost WITH PARAMETERS n AND cuts  
        SET cuts TO A LIST OF VALUES OF cuts COMBINED WITH zero AND n SORTED IN ASCENDING ORDER  
        
        FUNCTION dp WITH PARAMETERS left AND right  
            IF right MINUS left IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO A VALUE REPRESENTING POSITIVE INFINITY  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO VALUE OF element at position right OF cuts MINUS element at position left OF cuts PLUS CALL OF dp WITH left AND i PLUS CALL OF dp WITH i AND right  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN CALL OF dp WITH zero AND THE LENGTH OF cuts MINUS one  
END CLASS
