CLASS Solution  
    FUNCTION maxTotalFruits(fruits AS list of list of integers, startPos AS integer, k AS integer) RETURNS integer  
        SET max_fruits TO zero  
        SET total_fruits TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF fruits MINUS one  
            SET position TO element at position zero of element at position right of fruits  
            SET amount TO element at position one of element at position right of fruits  
            INCREMENT total_fruits BY amount  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND NOT (the element at position zero of element at position left of fruits IS GREATER THAN OR EQUAL TO startPos MINUS k AND the element at position zero of element at position left of fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND the element at position zero of element at position right of fruits IS GREATER THAN OR EQUAL TO startPos MINUS k AND the element at position zero of element at position right of fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND THE MINIMUM OF THE ABSOLUTE VALUE OF THE element at position zero of element at position right of fruits MINUS startPos AND THE ABSOLUTE VALUE OF THE element at position zero of element at position left of fruits MINUS startPos PLUS the element at position zero of element at position right of fruits MINUS the element at position zero of element at position left of fruits IS LESS THAN OR EQUAL TO k) HOLDS  
                DECREMENT total_fruits BY element at position one of element at position left of fruits  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_fruits TO THE GREATER OF max_fruits AND total_fruits  
        END FOR  
        
        RETURN max_fruits  
    END FUNCTION  
END CLASS
