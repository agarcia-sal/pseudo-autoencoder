CLASS Solution  
    FUNCTION getMoneyAmount WITH parameter n  
        DEFINE auxiliary FUNCTION dp WITH parameters left AND right  
            IF left IS GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET left_cost TO CALL dp WITH parameters left AND pivot MINUS one  
                SET right_cost TO CALL dp WITH parameters pivot PLUS one AND right  
                SET cost TO pivot PLUS THE GREATER OF left_cost AND right_cost  
                IF cost IS LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN CALL dp WITH parameters one AND n  
    END FUNCTION  
END CLASS
