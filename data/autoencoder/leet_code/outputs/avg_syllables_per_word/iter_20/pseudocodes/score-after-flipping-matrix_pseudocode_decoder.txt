CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR index FROM zero TO the LENGTH OF row MINUS one  
                    SET element at position index of row TO one MINUS element at position index of row  
                END FOR  
            END IF  
        END FOR  
        FOR column FROM one TO the LENGTH OF element at position zero of grid MINUS one  
            SET count_ones TO zero  
            FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_ones BY element at position column of element at position row_index of grid  
            END FOR  
            IF count_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                    SET element at position column of element at position row_index of grid TO one MINUS element at position column of element at position row_index of grid  
                END FOR  
            END IF  
        END FOR  
        SET score TO zero  
        FOR each row IN grid  
            SET binary_number TO the concatenation of converting each element of row TO its string representation  
            INCREMENT score BY converting binary_number from base two TO its integer value  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
