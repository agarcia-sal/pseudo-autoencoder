CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated m PLUS one times  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zero repeated m PLUS one times  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS the DIFFERENCE BETWEEN i AND left  
                SET mul TO the element at position i of multipliers  
                SET new_dp at position left TO the MAXIMUM BETWEEN the SUM OF mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp AND the SUM OF mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
