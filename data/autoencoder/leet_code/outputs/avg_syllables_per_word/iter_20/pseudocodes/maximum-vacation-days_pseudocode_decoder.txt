CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
        
        INITIALIZE dp AS a list of length n FILLED WITH negative one  
        SET element at position zero of dp TO zero  
        
        INITIALIZE graph AS a mapping FROM each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each integer i FROM zero TO n MINUS one  
            FOR each integer j FROM zero TO n MINUS one  
                IF element at position i of flights at position j EQUALS one  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer week FROM zero TO k MINUS one  
            INITIALIZE new_dp AS a list of length n FILLED WITH negative one  
            FOR each integer i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE TO next iteration  
                END IF  
                SET element at position i of new_dp TO the GREATER value BETWEEN element at position i of new_dp AND the SUM of element at position i of dp PLUS element at position i of days at position week  
                FOR each city j IN element at position i of graph  
                    SET element at position j of new_dp TO the GREATER value BETWEEN element at position j of new_dp AND the SUM of element at position i of dp PLUS element at position j of days at position week  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN the MAXIMUM value IN dp  
    END FUNCTION  
END CLASS
