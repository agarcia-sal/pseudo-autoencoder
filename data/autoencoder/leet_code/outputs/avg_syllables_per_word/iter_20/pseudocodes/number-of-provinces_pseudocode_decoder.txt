CLASS Solution  
    FUNCTION findCircleNum(isConnected)  
        FUNCTION dfs(city)  
            FOR neighbor FROM zero TO n MINUS one  
                IF element at position neighbor of element at position city of isConnected EQUALS one AND element at position neighbor of visited EQUALS False  
                    SET element at position neighbor of visited TO True  
                    CALL dfs WITH argument neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO the LENGTH OF isConnected  
        SET visited TO a list of boolean False elements with LENGTH n  
        SET province_count TO zero  

        FOR city FROM zero TO n MINUS one  
            IF element at position city of visited EQUALS False  
                CALL dfs WITH argument city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  

        RETURN province_count  
    END FUNCTION  
END CLASS
