CLASS Solution  
    FUNCTION smallestSubsequence(s, k, letter, repetition)  
        SET letter_count TO the NUMBER OF OCCURRENCES OF letter IN s  
        SET stack TO empty list  
        FOR each index FROM zero TO the LENGTH OF s MINUS one  
            SET char TO the ELEMENT at position index of s  
            WHILE the LENGTH OF stack IS NOT zero AND char IS LESS THAN the ELEMENT at the POSITION of the LAST ELEMENT of stack AND the SUM OF the LENGTH OF s MINUS index PLUS the LENGTH OF stack MINUS one IS GREATER THAN OR EQUAL TO k  
                IF the ELEMENT at the POSITION of the LAST ELEMENT of stack EQUALS letter  
                    IF letter_count IS GREATER THAN repetition  
                        INCREMENT repetition BY one  
                        REMOVE the LAST ELEMENT FROM stack  
                    ELSE  
                        BREAK the WHILE loop  
                    END IF  
                ELSE  
                    REMOVE the LAST ELEMENT FROM stack  
                END IF  
            END WHILE  
            IF the LENGTH OF stack IS LESS THAN k  
                IF char EQUALS letter  
                    APPEND char TO stack  
                    DECREMENT repetition BY one  
                ELSE IF the DIFFERENCE BETWEEN k AND the LENGTH OF stack IS GREATER THAN repetition  
                    APPEND char TO stack  
                END IF  
                IF char EQUALS letter  
                    DECREMENT letter_count BY one  
                END IF  
            END IF  
        END FOR  
        WHILE the LENGTH OF stack IS LESS THAN k  
            APPEND letter TO stack  
        END WHILE  
        RETURN the CONCATENATION OF ELEMENTS OF stack AS A STRING  
    END FUNCTION  
END CLASS
