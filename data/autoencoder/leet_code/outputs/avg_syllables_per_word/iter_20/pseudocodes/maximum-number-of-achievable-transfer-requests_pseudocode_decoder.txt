CLASS Solution  
    FUNCTION maximumRequests WITH PARAMETERS n AND requests  
        SET max_requests TO zero  
        SET num_requests TO the LENGTH OF requests  
        FOR each mask FROM zero TO the INTEGER REPRESENTED BY ONE SHIFTED LEFT BY num_requests MINUS one  
            SET balance TO a list of zeros WITH LENGTH n  
            SET count TO zero  
            FOR each i FROM zero TO num_requests MINUS one  
                IF the result of BITWISE AND of mask AND one SHIFTED LEFT BY i IS NOT EQUAL TO zero  
                    SET from_building TO the element at position zero of the element at position i of requests  
                    SET to_building TO the element at position one of the element at position i of requests  
                    DECREMENT the element at position from_building of balance BY one  
                    INCREMENT the element at position to_building of balance BY one  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF all elements b IN balance ARE EQUAL TO zero  
                SET max_requests TO the GREATER OF max_requests AND count  
            END IF  
        END FOR  
        RETURN max_requests  
    END FUNCTION  
END CLASS
