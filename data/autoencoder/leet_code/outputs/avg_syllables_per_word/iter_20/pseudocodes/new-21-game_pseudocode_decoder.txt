CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n INTEGER k INTEGER maxPts INTEGER  
        FUNCTION dfs WITH PARAMETER i INTEGER  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET minimum_value TO THE SMALLER OF the DIFFERENCE of n MINUS k PLUS one AND maxPts  
                SET probability TO minimum_value DIVIDED BY maxPts  
                RETURN probability  
            END IF  
            SET first_recursive_call TO dfs WITH PARAMETER i PLUS one  
            SET second_recursive_call TO dfs WITH PARAMETER i PLUS maxPts PLUS one  
            SET result TO first_recursive_call PLUS the DIFFERENCE of first_recursive_call MINUS second_recursive_call DIVIDED BY maxPts  
            RETURN result  
        END FUNCTION  
        RETURN dfs WITH PARAMETER zero  
    END FUNCTION  
END CLASS
