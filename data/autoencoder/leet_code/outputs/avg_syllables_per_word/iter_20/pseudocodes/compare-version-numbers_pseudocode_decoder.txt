CLASS Solution  
    FUNCTION compareVersion(version1 version1 AS string, version2 version2 AS string) RETURNS integer  
        SET v1_revisions TO the result of splitting version1 by the character dot  
        SET v2_revisions TO the result of splitting version2 by the character dot  
        SET max_length TO the greater value BETWEEN the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF index LESS THAN the LENGTH OF v1_revisions  
                SET v1 TO the integer conversion OF element at position index of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF index LESS THAN the LENGTH OF v2_revisions  
                SET v2 TO the integer conversion OF element at position index of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
