CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO one billion PLUS seven  
        SORT nums IN NON-DECREASING ORDER  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of size n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the element at position index MINUS one of power MULTIPLIED BY two MODULO MODULO  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO the REMAINDER AFTER DIVIDING the SUM OF total_sum AND the element at position index of nums MULTIPLIED BY the DIFFERENCE BETWEEN the element at position index of power AND the element at position n MINUS index MINUS one of power BY MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
