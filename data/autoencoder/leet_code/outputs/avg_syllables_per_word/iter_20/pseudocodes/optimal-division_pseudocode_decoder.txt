CLASS Solution  
    FUNCTION optimalDivision WITH nums  
        IF the LENGTH OF nums EQUALS one  
            RETURN convert element at position one of nums TO string  
        END IF  
        
        IF the LENGTH OF nums EQUALS two  
            RETURN convert element at position one of nums TO string CONCATENATED WITH the WORD division CONCATENATED WITH convert element at position two of nums TO string  
        END IF  
        
        RETURN convert element at position one of nums TO string CONCATENATED WITH the WORD division CONCATENATED WITH an OPENING PARENTHESIS CONCATENATED WITH the JOINING WITH the WORD division OF each element being the conversion TO string OF elements from position two TO the LAST POSITION of nums CONCATENATED WITH a CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
