CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO one billion PLUS seven  
        CREATE dp AS two dimensional list OF zeroes WITH n PLUS one ROWS AND k PLUS one COLUMNS  
        SET element at row zero column zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET dp at row i column j TO the SUM OF element at row i MINUS one column j MINUS one of dp AND the PRODUCT OF i MINUS one AND element at row i MINUS one column j of dp TAKEN MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at row n column k of dp  
    END FUNCTION  
END CLASS
