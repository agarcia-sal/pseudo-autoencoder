CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO a list that pairs each element of efficiency with the corresponding element of speed as a tuple consisting of efficiency element and speed element  
        SORT engineers so that their elements are ordered in descending order based on the efficiency component of each tuple  
        SET speed_heap TO an empty list to represent a collection that will maintain the smallest speeds  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each tuple consisting of eff and spd IN engineers  
            INSERT spd INTO speed_heap maintaining the heap property for minimum elements  
            INCREMENT speed_sum BY spd  
            IF the number of elements in speed_heap is greater than k  
                REMOVE the smallest element from speed_heap and call it removed_speed  
                DECREMENT speed_sum BY removed_speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the greater of max_performance and current_performance  
        END FOR  
        RETURN max_performance MODULO one billion PLUS seven  
    END FUNCTION  
END CLASS
