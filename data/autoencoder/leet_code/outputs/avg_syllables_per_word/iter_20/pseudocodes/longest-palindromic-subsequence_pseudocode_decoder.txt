CLASS Solution  
    FUNCTION longestPalindromeSubseq WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET dp TO A TWO DIMENSIONAL LIST OF SIZE n BY n FILLED WITH ZERO VALUES  
        FOR each i FROM zero TO n MINUS one  
            SET element at position i AND position i OF dp TO one  
        END FOR  
        FOR each length FROM two TO n  
            FOR each i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF the character at position i of s EQUALS the character at position j of s  
                    SET element at position i AND position j OF dp TO element at position i PLUS one AND position j MINUS one OF dp PLUS two  
                ELSE  
                    SET element at position i AND position j OF dp TO THE GREATER VALUE BETWEEN element at position i PLUS one AND position j OF dp AND element at position i AND position j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero AND position n MINUS one OF dp  
    END FUNCTION  
END CLASS
