CLASS Solution  
    FUNCTION numDistinctIslands(grid)  
        FUNCTION dfs(i, j, direction)  
            IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF grid AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of grid AND element at position i of grid AT position j EQUALS one  
                SET element at position i of grid AT position j TO zero  
                APPEND direction TO path  
                CALL dfs WITH i INCREMENTED BY one j direction AS lowercase dee  
                CALL dfs WITH i DECREMENTED BY one j direction AS lowercase you  
                CALL dfs WITH i j INCREMENTED BY one direction AS lowercase ar  
                CALL dfs WITH i j DECREMENTED BY one direction AS lowercase el  
                APPEND lowercase bee TO path  
            END IF  
            RETURN path  
        END FUNCTION  
        
        SET unique_islands TO a new empty set  
        
        FOR i FROM zero TO LENGTH OF grid MINUS one  
            FOR j FROM zero TO LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid AT position j EQUALS one  
                    SET path TO a new empty list  
                    ADD the tuple converted from the result of calling dfs WITH i j lowercase oh TO unique_islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the LENGTH OF unique_islands  
    END FUNCTION  
END CLASS
