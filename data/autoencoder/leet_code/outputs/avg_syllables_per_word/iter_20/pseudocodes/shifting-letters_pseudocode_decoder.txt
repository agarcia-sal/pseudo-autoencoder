CLASS Solution  
    FUNCTION shiftingLetters(s STRING, shifts LIST OF INTEGER)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character_code_difference TO the Unicode code of element at position index of s MINUS the Unicode code of lowercase a  
            SET shifted_code TO character_code_difference PLUS element at position index of shifts  
            SET wrapped_code TO shifted_code MODULO twenty six  
            SET new_character_code TO wrapped_code PLUS the Unicode code of lowercase a  
            SET new_character TO the character represented by new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result into a single string  
    END FUNCTION  
END CLASS
