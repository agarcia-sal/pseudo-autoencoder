CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with parameter value equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode with parameter value equal to element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode with parameter value equal to element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET node_dict TO a default dictionary mapping to empty lists  
        SET min_col TO zero  
        SET max_col TO zero  
        FUNCTION dfs(node, row, col)  
            IF node IS none  
                RETURN  
            END IF  
            APPEND a tuple of row and node dot val TO the list at key col in node_dict  
            SET min_col TO the minimum of min_col AND col  
            SET max_col TO the maximum of max_col AND col  
            CALL dfs WITH node dot left, row PLUS one, col MINUS one  
            CALL dfs WITH node dot right, row PLUS one, col PLUS one  
        END FUNCTION  
        CALL dfs WITH root, zero, zero  
        SET result TO an empty list  
        FOR col FROM min_col TO max_col  
            SET sorted_list TO the list at key col in node_dict sorted by the first element of each tuple then by the second element  
            APPEND to result a list of the second elements of each tuple in sorted_list  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
