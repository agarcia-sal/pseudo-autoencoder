CLASS Solution  
    FUNCTION maxSubArrayLen(nums OF TYPE List OF integer, k OF TYPE integer) RETURNS integer  
        SET cumulative_sum_index TO a dictionary mapping zero TO minus one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index AND number IN the enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS number  
            IF the difference of cumulative_sum MINUS k EXISTS AS a key IN cumulative_sum_index  
                SET current_length TO index MINUS the element at key cumulative_sum MINUS k of cumulative_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND current_length  
            END IF  
            IF cumulative_sum DOES NOT EXIST AS a key IN cumulative_sum_index  
                SET the element at key cumulative_sum of cumulative_sum_index TO index  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
