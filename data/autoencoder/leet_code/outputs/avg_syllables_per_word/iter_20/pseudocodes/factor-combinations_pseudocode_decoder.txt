CLASS Solution  
    FUNCTION getFactors WITH parameter n  
        FUNCTION backtrack WITH parameters start AND target  
            FOR each integer i FROM start TO the integer part of the square root of target PLUS one  
                IF the remainder of target DIVIDED BY i EQUALS zero  
                    APPEND the concatenation of path WITH a list containing i AND the result of dividing target BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND the result of dividing target BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
