CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns where each element IS set TO negative infinity  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET dp element at row i column j TO the MAXIMUM OF current_product AND element at position i MINUS one row column j of dp AND element at position i column j MINUS one of dp AND element at position i MINUS one row j MINUS one column of dp PLUS current_product  
            END FOR  
        END FOR  
        RETURN element at row m column n of dp  
    END FUNCTION  
END CLASS
