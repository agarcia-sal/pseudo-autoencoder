CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid DOES NOT EXIST OR element at position zero and zero of obstacleGrid EQUALS one  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero of obstacleGrid  
        SET dp TO a two dimensional list of zeros with m rows and n columns  
        SET element at position zero and zero of dp TO one  
        FOR j FROM one TO n MINUS one  
            IF element at position zero and j of obstacleGrid EQUALS zero  
                SET element at position zero and j of dp TO element at position zero and j MINUS one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            IF element at position i and zero of obstacleGrid EQUALS zero  
                SET element at position i and zero of dp TO element at position i MINUS one and zero of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i and j of obstacleGrid EQUALS zero  
                    SET element at position i and j of dp TO element at position i MINUS one and j of dp PLUS element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one and n MINUS one of dp  
    END FUNCTION  
END CLASS
