CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal) RETURNS boolean  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one, DIVIDED BY two, IS LESS THAN desiredTotal THEN  
            RETURN false  
        END IF  
        
        IF desiredTotal IS LESS THAN OR EQUAL TO zero THEN  
            RETURN true  
        END IF  
        
        SET memo TO empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers) RETURNS boolean  
            IF the pair of current_total AND used_numbers IS A KEY IN memo THEN  
                RETURN the value associated with the pair of current_total AND used_numbers in memo  
            END IF  
            
            FOR number FROM one TO maxChoosableInteger DO  
                IF the binary AND of used_numbers AND one LEFT SHIFTED BY number IS EQUAL TO zero THEN  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal OR NOT the result of calling can_win with the sum of current_total AND number AND the binary OR of used_numbers AND one LEFT SHIFTED BY number THEN  
                        SET the value associated with the pair of current_total AND used_numbers in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with the pair of current_total AND used_numbers in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_win with zero AND zero  
    END FUNCTION  
END CLASS
