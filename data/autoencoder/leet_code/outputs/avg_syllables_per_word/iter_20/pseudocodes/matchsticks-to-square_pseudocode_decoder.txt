CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF all elements IN matchsticks  
        IF the REMAINDER WHEN total_length IS DIVIDED BY four IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET side_length TO the INTEGER QUOTIENT OF total_length DIVIDED BY four  
        SORT matchsticks IN DESCENDING ORDER  
        FUNCTION can_form_sides(sides, index)  
            IF index IS EQUAL TO the LENGTH OF matchsticks  
                RETURN True IF ALL elements IN sides ARE EQUAL TO side_length ELSE False  
            END IF  
            FOR each integer i FROM zero TO three  
                IF the element at position i of sides PLUS the element at position index of matchsticks IS LESS THAN OR EQUAL TO side_length  
                    SET the element at position i of sides TO the element at position i of sides PLUS the element at position index of matchsticks  
                    IF can_form_sides(sides, index PLUS one) RETURNS True  
                        RETURN True  
                    END IF  
                    SET the element at position i of sides TO the element at position i of sides MINUS the element at position index of matchsticks  
                END IF  
                IF the element at position i of sides IS EQUAL TO zero  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN can_form_sides(a list of four zeros, zero)  
    END FUNCTION  
END CLASS
