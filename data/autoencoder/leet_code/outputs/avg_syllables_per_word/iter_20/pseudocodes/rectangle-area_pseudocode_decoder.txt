CLASS Solution  
    FUNCTION computeArea(ax1 ax1 AS INTEGER ay1 ay1 AS INTEGER ax2 ax2 AS INTEGER ay2 ay2 AS INTEGER bx1 bx1 AS INTEGER by1 by1 AS INTEGER bx2 bx2 AS INTEGER by2 by2 AS INTEGER) RETURNS INTEGER  
        SET area1 TO the difference between ax2 AND ax1 MULTIPLIED BY the difference between ay2 AND ay1  
        SET area2 TO the difference between bx2 AND bx1 MULTIPLIED BY the difference between by2 AND by1  
        SET overlap_width TO the maximum of zero AND the difference between the minimum of ax2 AND bx2 AND the maximum of ax1 AND bx1  
        SET overlap_height TO the maximum of zero AND the difference between the minimum of ay2 AND by2 AND the maximum of ay1 AND by1  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
