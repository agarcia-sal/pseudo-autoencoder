CLASS Solution  
    FUNCTION shortestSubarray(nums AS List OF integer, k AS integer) RETURNS integer  
        SET prefix_sum TO list containing single element zero  
        FOR each num IN nums  
            APPEND the sum of the last element of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        ASSIGN dq TO an empty double-ended queue  
        SET min_length TO infinite quantity representing a large number  
        
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND the difference between current_sum AND element at position at the front of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the lesser value of min_length AND the difference between i AND the element removed from the front of dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position at the back of dq of prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO infinite quantity representing a large number  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
