CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the frequency COUNTER of characters in s  
        SET max_heap TO an empty list  
        FOR each character AND its frequency IN count  
            APPEND a tuple of negative frequency AND character TO max_heap  
        END FOR  
        CALL heapify FUNCTION on max_heap TO ORGANIZE it as a heap  
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO an empty list  
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN an empty string  
            END IF  
            SET freq AND char TO the tuple obtained by removing the smallest element FROM max_heap  
            APPEND char TO result  
            INCREMENT freq BY one  
            IF prev_freq is less than zero  
                ADD the tuple of prev_freq AND prev_char BACK INTO max_heap as a heap element  
            END IF  
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        RETURN the concatenation of all elements IN result as a single string  
    END FUNCTION  
END CLASS
