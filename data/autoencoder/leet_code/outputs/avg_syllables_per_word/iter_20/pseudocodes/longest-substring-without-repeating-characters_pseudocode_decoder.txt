CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH PARAMETER s  
        SET char_index_map TO an empty mapping from characters to integer positions  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end position AND character AT position end IN s  
            IF character EXISTS AS A KEY IN char_index_map AND the value associated WITH character IN char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value associated WITH character IN char_index_map PLUS one  
            END IF  
            SET the value associated WITH character IN char_index_map TO end  
            SET max_length TO THE GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
