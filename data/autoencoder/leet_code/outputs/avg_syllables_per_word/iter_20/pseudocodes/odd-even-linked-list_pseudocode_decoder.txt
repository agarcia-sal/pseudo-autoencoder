CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET head TO A NEW ListNode WITH THE ELEMENT AT POSITION zero OF values  
    SET p TO head  
    FOR each val IN THE SUBSEQUENCE OF values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode WITH val  
        SET THE next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN false  
    END IF  
    RETURN THE RESULT OF THE LOGICAL EXPRESSION THAT THE val ATTRIBUTE OF p1 EQUALS THE val ATTRIBUTE OF p2 AND THE RESULT OF THE FUNCTION is_same_list WITH PARAMETERS THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self AND head OF TYPE OPTIONAL ListNode RETURNING OPTIONAL ListNode  
        IF head EQUALS nothing OR THE next ATTRIBUTE OF head EQUALS nothing OR THE next ATTRIBUTE OF THE next ATTRIBUTE OF head EQUALS nothing  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO THE next ATTRIBUTE OF head  
        SET even_head TO even  
        WHILE even IS NOT nothing AND THE next ATTRIBUTE OF even IS NOT nothing  
            SET THE next ATTRIBUTE OF odd TO THE next ATTRIBUTE OF even  
            SET odd TO THE next ATTRIBUTE OF odd  
            SET THE next ATTRIBUTE OF even TO THE next ATTRIBUTE OF odd  
            SET even TO THE next ATTRIBUTE OF even  
        END WHILE  
        SET THE next ATTRIBUTE OF odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
