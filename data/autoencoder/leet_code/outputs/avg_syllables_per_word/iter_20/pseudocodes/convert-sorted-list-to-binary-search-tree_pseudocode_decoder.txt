CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS None  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new ListNode initialized with val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical AND of p1 dot val EQUALS p2 dot val AND is_same_list WITH PARAMETERS p1 dot next AND p2 dot next  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS None AND right EQUALS None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element that is removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None  
            SET node dot left TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None  
            SET node dot right TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p dot val DOES NOT EQUAL q dot val  
        RETURN False  
    ELSE  
        RETURN the logical AND of is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS self AND head  
        IF head EQUALS None  
            RETURN None  
        END IF  

        FUNCTION findMiddle WITH PARAMETERS left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast DOES NOT EQUAL right AND fast dot next DOES NOT EQUAL right  
                SET slow TO slow dot next  
                SET fast TO fast dot next dot next  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST WITH PARAMETERS left AND right  
            IF left EQUALS right  
                RETURN None  
            END IF  
            SET mid TO findMiddle WITH PARAMETERS left AND right  
            SET node TO a new TreeNode initialized with mid dot val  
            SET node dot left TO convertListToBST WITH PARAMETERS left AND mid  
            SET node dot right TO convertListToBST WITH PARAMETERS mid dot next AND right  
            RETURN node  
        END FUNCTION  

        RETURN convertListToBST WITH PARAMETERS head AND None  
    END FUNCTION  
END CLASS
