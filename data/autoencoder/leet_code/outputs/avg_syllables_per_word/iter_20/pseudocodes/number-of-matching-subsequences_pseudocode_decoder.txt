CLASS Solution  
    FUNCTION numMatchingSubseq(s STRING, words LIST OF STRING) RETURNS INTEGER  
        SET waiting TO A NEW DICTIONARY THAT MAPS EACH UNIQUE CHARACTER IN s TO A NEW EMPTY DOUBLE ENDED QUEUE  
        
        FOR each word IN words  
            SET it TO AN ITERATOR OVER word  
            SET first_letter TO THE NEXT ELEMENT FROM it  
            IF first_letter IS A KEY IN waiting  
                APPEND it TO THE DOUBLE ENDED QUEUE ASSOCIATED WITH first_letter IN waiting  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN THE STRING s  
            IF c IS A KEY IN waiting  
                SET current_iterators TO THE DOUBLE ENDED QUEUE ASSOCIATED WITH c IN waiting  
                SET THE DOUBLE ENDED QUEUE ASSOCIATED WITH c IN waiting TO A NEW EMPTY DOUBLE ENDED QUEUE                
                FOR each it IN current_iterators  
                    SET next_char TO THE NEXT ELEMENT FROM it OR None IF IT IS EXHAUSTED  
                    IF next_char IS None  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char IS A KEY IN waiting  
                            APPEND it TO THE DOUBLE ENDED QUEUE ASSOCIATED WITH next_char IN waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
