CLASS Solution  
    FUNCTION jobScheduling(startTime LIST OF INTEGER endTime LIST OF INTEGER profit LIST OF INTEGER) RETURNS INTEGER  
        SET jobs TO a list of elements each composed of corresponding elements from startTime endTime and profit arranged in ascending order according to the second element which is end time of each element  
        SET end_times TO a list of the second element of each element in jobs representing the end time of each job  
        SET dp TO a list of zeros with a length one greater than the length of jobs representing the maximum profit achievable considering the first number of jobs equal to the index  
        FOR index FROM one TO the LENGTH OF jobs  
            SET j TO the position found by locating the insertion point to the right of the element at position index MINUS one in jobs start time within end_times  
            SET dp at position index TO the maximum of dp at position index MINUS one and dp at position j PLUS the third element of the job at position index MINUS one representing profit  
        END FOR  
        RETURN the last element of dp representing the maximum profit achievable considering all jobs  
    END FUNCTION  
END CLASS
