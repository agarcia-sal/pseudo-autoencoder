CLASS Solution  
    FUNCTION pathsWithMaxScore(board)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF board  
        SET dp TO a two dimensional list of size n BY n each element initialized TO a list where the first element is NEGATIVE_INFINITY and the second element is zero  
        SET element at position n MINUS one OF dp TO a list where the first element is zero and the second element is one  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the element at position i OF board AT position j IS EQUALS TO the letter X OR IS EQUALS TO the letter S  
                    CONTINUE to the next iteration  
                END IF  
                SET value TO zero  
                IF the element at position i OF board AT position j IS NOT EQUALS TO the letter E  
                    SET value TO the integer CONVERSION OF the element at position i OF board AT position j  
                END IF  
                FOR each pair (x y) IN the list containing the pairs one zero, zero one, and one one  
                    SET ni TO i PLUS x  
                    SET nj TO j PLUS y  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                        IF the first element at position ni OF dp AT position nj IS GREATER THAN the first element at position i OF dp AT position j  
                            SET element at position i OF dp AT position j TO a list where the first element is the sum of the first element at position ni OF dp AT position nj PLUS value and the second element is the second element at position ni OF dp AT position nj  
                        ELSE IF the first element at position ni OF dp AT position nj IS EQUALS TO the first element at position i OF dp AT position j  
                            SET the second element at position i OF dp AT position j TO the sum of the second element at position i OF dp AT position j PLUS the second element at position ni OF dp AT position nj MODULO MOD  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF the first element at position zero OF dp AT position zero IS EQUALS TO NEGATIVE_INFINITY  
            RETURN a list containing zero and zero  
        END IF  
        RETURN a list containing the first element at position zero OF dp AT position zero and the second element at position zero OF dp AT position zero  
    END FUNCTION  
END CLASS
