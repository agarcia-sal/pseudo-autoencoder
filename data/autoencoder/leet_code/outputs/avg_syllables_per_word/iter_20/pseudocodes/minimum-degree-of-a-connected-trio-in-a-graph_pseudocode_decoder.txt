CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO a mapping from each number from one TO n TO an empty set  
        FOR each pair of nodes u AND v IN edges  
            ADD v TO the set associated with u in graph  
            ADD u TO the set associated with v in graph  
        END FOR  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF the set associated with i in graph CONTAINS j AND the set associated with i in graph CONTAINS k AND the set associated with j in graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set associated with i in graph PLUS the LENGTH OF the set associated with j in graph PLUS the LENGTH OF the set associated with k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
