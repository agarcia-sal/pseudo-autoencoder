CLASS Solution  
    FUNCTION minmaxGasDist(stations AS List of integer, k AS integer) RETURNS floating point number  
        FUNCTION possible(max_dist AS floating point number) RETURNS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index of stations MINUS element at position index MINUS one of stations  
                SET stations_needed TO distance_difference DIVIDED BY max_dist CONVERTED TO integer value  
                INCREMENT additional_stations BY stations_needed  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero ZERO zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible WITH parameter mid HOLDS TRUE  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
