CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF the element endWord is NOT IN the collection wordList  
            RETURN zero  
        END IF  
        
        SET word_dict TO a new dictionary that returns an empty list for unseen keys  
        SET word_length TO the LENGTH OF the string beginWord  
        
        FOR each word IN the collection wordList  
            FOR each index i FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of the substring from position zero TO index i MINUS one of word, THEN the character asterisk, THEN the substring from position i PLUS one TO the END OF word  
                APPEND word TO the list found at key intermediate_key in word_dict  
            END FOR  
        END FOR  
        
        SET queue TO a new double-ended queue initialized with a pair consisting of the string beginWord and the integer one  
        SET visited TO a new set containing the string beginWord  
        
        WHILE the queue is not empty  
            SET current_pair TO the element removed from the left of queue  
            SET current_word TO the first element of current_pair  
            SET current_length TO the second element of current_pair  
            
            FOR each index i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of the substring from position zero TO index i MINUS one of current_word, THEN the character asterisk, THEN the substring from position i PLUS one TO the END OF current_word  
                
                FOR each word IN the list found at key intermediate_word in word_dict  
                    IF word EQUALS the string endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word is NOT IN the set visited  
                        ADD word TO the set visited  
                        APPEND the pair consisting of word and current_length PLUS one TO the queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
