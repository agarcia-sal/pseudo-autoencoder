CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF index GREATER THAN farthest  
                RETURN false  
            END IF  
            SET farthest TO the GREATER OF farthest AND the SUM OF index AND element at position index of nums  
        END FOR  
        RETURN farthest GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
    END FUNCTION  
END CLASS
