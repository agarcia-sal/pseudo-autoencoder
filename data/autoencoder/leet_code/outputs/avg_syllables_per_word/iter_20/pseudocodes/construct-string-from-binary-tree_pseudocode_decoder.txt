CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values OF TYPE list)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH value AT position one OF values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode WITH value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode WITH value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND the RESULT OF is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(root OF TYPE Optional TreeNode) RETURNS string  
        FUNCTION helper(node)  
            IF node IS none  
                RETURN empty string  
            END IF  
            SET result TO string REPRESENTATION OF node.val  
            IF node.left EXISTS OR node.right EXISTS  
                SET result TO result CONCATENATED WITH opening parenthesis CONCATENATED WITH helper WITH node.left CONCATENATED WITH closing parenthesis  
            END IF  
            IF node.right EXISTS  
                SET result TO result CONCATENATED WITH opening parenthesis CONCATENATED WITH helper WITH node.right CONCATENATED WITH closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper WITH root  
    END FUNCTION  
END CLASS
