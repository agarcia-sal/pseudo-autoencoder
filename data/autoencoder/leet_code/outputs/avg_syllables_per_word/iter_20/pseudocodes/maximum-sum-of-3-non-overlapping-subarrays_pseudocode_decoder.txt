CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums AS List OF integers, k AS integer) RETURNS List OF integers  
        SET n TO the LENGTH OF nums  
        SET sums TO a list of zeros of LENGTH n MINUS k PLUS one  
        SET current_sum TO the SUM OF elements from position zero TO k MINUS one of nums  
        SET element at position zero of sums TO current_sum  
        
        FOR i FROM one TO n MINUS k  
            INCREMENT current_sum BY element at position i PLUS k MINUS one of nums MINUS element at position i MINUS one of nums  
            SET element at position i of sums TO current_sum  
        END FOR  
        
        SET left TO a list of zeros of LENGTH n MINUS k PLUS one  
        SET right TO a list where each element is n MINUS k of LENGTH n MINUS k PLUS one  
        
        SET max_sum TO zero  
        FOR i FROM zero TO n MINUS k  
            IF element at position i of sums IS GREATER THAN element at position max_sum of sums  
                SET max_sum TO i  
            END IF  
            SET element at position i of left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWNTO zero  
            IF element at position i of sums IS GREATER THAN OR EQUAL TO element at position max_sum of sums  
                SET max_sum TO i  
            END IF  
            SET element at position i of right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a list of zeros of LENGTH three  
        FOR j FROM k TO n MINUS two TIMES k  
            SET i TO element at position j MINUS k of left  
            SET l TO element at position j PLUS k of right  
            SET total TO element at position i of sums PLUS element at position j of sums PLUS element at position l of sums  
            IF total IS GREATER THAN max_total  
                SET max_total TO total  
                SET result TO a list containing i THEN j THEN l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
