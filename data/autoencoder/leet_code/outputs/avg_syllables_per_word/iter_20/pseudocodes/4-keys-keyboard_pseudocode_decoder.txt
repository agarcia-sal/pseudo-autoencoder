CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three THEN  
            RETURN n  
        END IF  

        SET dp TO a list of zeros with the LENGTH OF n PLUS one  

        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  

        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET element at position i of dp TO the GREATER OF element at position i of dp AND the PRODUCT OF element at position j of dp AND the DIFFERENCE BETWEEN i AND j MINUS one  
            END FOR  
        END FOR  

        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
