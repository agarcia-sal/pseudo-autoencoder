CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty list THEN  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position zero of each interval from lowest TO highest  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF min_heap is not empty AND the element at position zero of meeting IS GREATER THAN OR EQUAL TO the element at position zero of min_heap THEN  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap maintaining it as a min heap  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
