CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO remainder of remaining DIVIDED BY two PLUS dp(quotient of remaining DIVIDED BY two)  
            SET option_two TO remainder of remaining DIVIDED BY three PLUS dp(quotient of remaining DIVIDED BY three)  
            RETURN one PLUS the minimum of option_one AND option_two  
        END FUNCTION  
        RETURN dp(n)  
    END FUNCTION  
END CLASS
