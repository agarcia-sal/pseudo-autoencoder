CLASS Solution  
    FUNCTION kSmallestPairs(nums1 AS List OF integer, nums2 AS List OF integer, k AS integer) RETURNS List OF List OF integer  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO zero THEN  
            RETURN an empty list  
        END IF  
          
        SET min_heap TO an empty list  
        FOR index i FROM zero TO the LESSER OF k MINUS one AND the LENGTH OF nums1 MINUS one  
            PUSH ONTO min_heap a tuple consisting of the SUM OF the element at position i of nums1 AND the element at position zero of nums2 AND the values i AND zero  
        END FOR  
          
        SET result TO an empty list  
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result IS LESS THAN k  
            POP the SMALLEST element FROM min_heap AND ASSIGN TO temporary variables ignore_value, i, j  
            APPEND a list consisting of the element at position i of nums1 AND the element at position j of nums2 TO result  
            IF j PLUS one IS LESS THAN the LENGTH OF nums2 THEN  
                PUSH ONTO min_heap a tuple consisting of the SUM OF the element at position i of nums1 AND the element at position j PLUS one of nums2 AND the values i AND j PLUS one  
            END IF  
        END WHILE  
          
        RETURN result  
    END FUNCTION  
END CLASS
