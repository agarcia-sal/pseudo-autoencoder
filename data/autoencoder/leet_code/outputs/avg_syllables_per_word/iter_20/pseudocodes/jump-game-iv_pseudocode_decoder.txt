CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO A DEFAULT DICTIONARY MAPPING EACH INTEGER TO A LIST OF INTEGER POSITIONS  
        FOR each i AND value IN THE ENUMERATION OF arr  
            APPEND i TO the LIST ASSOCIATED WITH value IN value_indices  
        END FOR  
        
        SET queue TO A DOUBLE ENDED QUEUE INITIALIZED WITH THE PAIR OF zero AND zero WHERE zero IS THE CURRENT INDEX AND zero IS THE CURRENT STEPS  
        SET visited TO A SET CONTAINING zero  
        
        WHILE queue IS NOT EMPTY  
            EXTRACT current_index AND steps BY REMOVING THE FIRST ELEMENT FROM queue  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO A LIST COMBINING current_index PLUS one, current_index MINUS one, AND THE LIST ASSOCIATED WITH THE VALUE AT POSITION current_index OF arr IN value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND THE PAIR OF next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET THE LIST ASSOCIATED WITH THE VALUE AT POSITION current_index OF arr IN value_indices TO AN EMPTY LIST  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
