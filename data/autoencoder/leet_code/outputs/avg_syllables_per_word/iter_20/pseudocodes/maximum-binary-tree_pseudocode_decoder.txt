CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position one of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p dot left AND q dot left AND the result of is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS self nums list of integers  
        IF nums IS EMPTY  
            RETURN none  
        END IF  
        SET max_val TO the maximum value IN nums  
        SET max_index TO the position index OF max_val IN nums  
        SET root TO NEW TreeNode WITH PARAMETER max_val  
        SET root dot left TO the result of calling constructMaximumBinaryTree WITH PARAMETERS self AND the sublist of nums FROM position one TO max_index minus one  
        SET root dot right TO the result of calling constructMaximumBinaryTree WITH PARAMETERS self AND the sublist of nums FROM position max_index plus one TO the end of nums  
        RETURN root  
    END FUNCTION  
END CLASS
