CLASS Solution  
    FUNCTION numBusesToDestination WITH PARAMETERS routes, source, target  
        IF source EQUALS target  
            RETURN zero  
        END IF  

        SET stop_to_buses TO an empty mapping from keys to lists  
        FOR each bus WITH INDEX bus_number IN routes  
            FOR each stop IN route at position bus_number of routes  
                APPEND bus_number TO the list at key stop IN stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO an empty double-ended queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus IN the list at key source IN stop_to_buses  
            APPEND a tuple of bus and one TO the right end of bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the tuple at the left end of bus_queue AND ASSIGN it TO current_bus and bus_count  

            FOR each stop IN the route at position current_bus of routes  
                IF stop EQUALS target  
                    RETURN bus_count  
                END IF  

                IF stop is not in visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list at key stop IN stop_to_buses  
                        IF next_bus is not in visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND the tuple of next_bus and bus_count PLUS one TO the right end of bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
