CLASS Solution  
    FUNCTION primePalindrome(n)  
        FUNCTION is_prime(num)  
            IF num LESS THAN two  
                RETURN false  
            END IF  
            IF num EQUALS two OR num EQUALS three  
                RETURN true  
            END IF  
            IF num MODULO two EQUALS zero OR num MODULO three EQUALS zero  
                RETURN false  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS zero OR num MODULO i PLUS two EQUALS zero  
                    RETURN false  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN true  
        END FUNCTION  

        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reversed string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO two  
            RETURN two  
        END IF  

        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven  
            RETURN eleven  
        END IF  

        WHILE true  
            IF n MODULO two EQUALS zero  
                INCREMENT n BY one  
                CONTINUE TO next iteration of WHILE  
            END IF  
            IF is_palindrome(n) IS true AND is_prime(n) IS true  
                RETURN n  
            END IF  
            IF LENGTH OF string representation of n MODULO two EQUALS zero  
                SET n TO ten RAISED TO THE POWER OF LENGTH OF string representation of n PLUS one  
            ELSE  
                INCREMENT n BY two  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
