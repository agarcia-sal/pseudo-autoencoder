CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort TO arr2  
        SET dp TO a dictionary with key negative one and value zero  
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num and ops IN the items of dp  
                IF num IS GREATER THAN last_num  
                    SET current_min_ops TO the value at key num in new_dp OR positive infinity IF key not present  
                    SET minimal_ops TO the minimum of current_min_ops AND ops  
                    SET the value at key num in new_dp TO minimal_ops  
                END IF  
                SET idx TO the position to insert last_num in arr2 such that arr2 remains sorted after insertion  
                IF idx IS LESS THAN the LENGTH OF arr2  
                    SET replacement_num TO the element at position idx of arr2  
                    SET current_min_ops TO the value at key replacement_num in new_dp OR positive infinity IF key not present  
                    SET minimal_ops TO the minimum of current_min_ops AND ops PLUS one  
                    SET the value at key replacement_num in new_dp TO minimal_ops  
                END IF  
            END FOR  
            IF new_dp IS empty  
                RETURN negative one  
            END IF  
            SET dp TO new_dp  
        END FOR  
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
