CLASS Solution  
    FUNCTION calcEquation(equations, values, queries)  
        SET graph TO an empty mapping from string to mapping from string to number  
        FOR each pair of variables a and b AND corresponding value IN the zipped pairs of equations and values  
            SET the element at position b of the mapping at position a of graph TO value  
            SET the element at position a of the mapping at position b of graph TO one DIVIDED BY value  
        END FOR  

        FUNCTION dfs(start, end, visited)  
            IF start is not a key in graph OR end is not a key in graph THEN  
                RETURN minus one point zero  
            END IF  
            IF start EQUALS end THEN  
                RETURN one point zero  
            END IF  
            ADD start TO visited  
            FOR each neighbor and weight IN the items of the mapping at position start of graph  
                IF neighbor is not an element of visited THEN  
                    SET result TO the invocation of dfs with neighbor, end, and visited  
                    IF result NOT EQUAL TO minus one point zero THEN  
                        RETURN weight MULTIPLIED BY result  
                    END IF  
                END IF  
            END FOR  
            REMOVE start FROM visited  
            RETURN minus one point zero  
        END FUNCTION  

        SET results TO an empty list  
        FOR each query IN queries  
            DESTRUCTURE query INTO start and end  
            SET visited TO an empty set  
            SET result TO the invocation of dfs with start, end, and visited  
            APPEND result TO results  
        END FOR  

        RETURN results  
    END FUNCTION  
END CLASS
