CLASS Solution  
    FUNCTION strobogrammaticInRange low STRING high STRING RETURNS INTEGER  
        FUNCTION generate_strobogrammatic n INTEGER final_length INTEGER RETURNS LIST OF STRINGS  
            IF n EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET middles TO CALL generate_strobogrammatic WITH n MINUS two AND final_length  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND concatenation of string zero AND middle AND string zero TO result  
                END IF  
                APPEND concatenation of string one AND middle AND string one TO result  
                APPEND concatenation of string six AND middle AND string nine TO result  
                APPEND concatenation of string eight AND middle AND string eight TO result  
                APPEND concatenation of string nine AND middle AND string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic low STRING high STRING RETURNS INTEGER  
            SET count TO zero  
            FOR length FROM LENGTH OF low TO LENGTH OF high  
                FOR each num IN CALL generate_strobogrammatic WITH length AND length  
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high)  
                        CONTINUE TO next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN CALL count_strobogrammatic WITH low AND high  
    END FUNCTION  
END CLASS
