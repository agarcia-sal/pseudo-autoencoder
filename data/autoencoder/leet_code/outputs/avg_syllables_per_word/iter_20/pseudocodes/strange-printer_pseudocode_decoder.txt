CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF tuple of i AND j IS IN memo  
                RETURN value AT key tuple of i AND j IN memo  
            END IF  
            SET result TO call FUNCTION dp WITH parameter i AND parameter j MINUS one AND parameter memo PLUS one  
            FOR each k FROM i TO j MINUS one  
                IF character at position k OF string s EQUALS character at position j OF string s  
                    SET result TO the MINIMUM OF result AND call FUNCTION dp WITH parameter i AND parameter k MINUS one AND parameter memo PLUS call FUNCTION dp WITH parameter k AND parameter j MINUS one AND parameter memo  
                END IF  
            END FOR  
            SET memo AT key tuple of i AND j TO result  
            RETURN result  
        END FUNCTION  
        RETURN call FUNCTION dp WITH zero AND the LENGTH OF string s MINUS one AND an empty dictionary  
    END FUNCTION  
END CLASS
