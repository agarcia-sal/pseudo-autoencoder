CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO substring from position one TO d of frag  
                SET right TO substring from position d PLUS one TO n of frag  
                IF left DOES NOT START WITH digit zero OR left EQUALS digit zero AND right DOES NOT END WITH digit zero  
                    OUTPUT left CONCATENATED WITH a decimal point IF d IS NOT EQUAL TO n OTHERWISE empty string CONCATENATED WITH right  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET s TO substring from position two TO the LENGTH OF s MINUS one of s  
        SET result TO an empty list  
        FOR i FROM one TO the LENGTH OF s MINUS one  
            FOR each pair (a, b) IN the Cartesian product OF make applied to substring from position one TO i of s AND make applied to substring from position i PLUS one TO the LENGTH OF s of s  
                APPEND an opening parenthesis CONCATENATED WITH a CONCATENATED WITH a comma AND space CONCATENATED WITH b CONCATENATED WITH a closing parenthesis TO result  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
