CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a default dictionary that creates an empty list when a new key is accessed  
        FOR each string s IN strs  
            SET sorted_str TO the concatenation of characters from s arranged in ascending alphabetical order  
            APPEND s TO the list mapped by sorted_str in anagrams  
        END FOR  
        SET result TO the list containing all the lists of strings stored as values in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
