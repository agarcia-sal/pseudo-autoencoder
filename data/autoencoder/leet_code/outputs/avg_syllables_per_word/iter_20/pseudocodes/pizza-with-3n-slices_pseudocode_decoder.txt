CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            SET pick TO element at position start of slices PLUS dp with start INCREMENTED BY two AND end AND k DECREMENTED BY one  
            SET skip TO dp with start INCREMENTED BY one AND end AND k  
            RETURN the GREATER VALUE BETWEEN pick AND skip  
        END FUNCTION  
        
        SET case1 TO dp with start SET TO one AND end SET TO the LENGTH OF slices MINUS one AND k SET TO n  
        SET case2 TO dp with start SET TO zero AND end SET TO the LENGTH OF slices MINUS two AND k SET TO n  
        
        RETURN the GREATER VALUE BETWEEN case1 AND case2  
    END FUNCTION  
END CLASS
