CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS nothing OR element at position zero of matrix EQUALS nothing  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO the RESULT OF calling auxiliary FUNCTION init_matrix WITH PARAMETERS m PLUS one AND n PLUS one  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AND element at position j MINUS one of that element EQUALS character one  
                    SET dp at position i AND position j TO the MINIMUM OF element at position i MINUS one AND position j of dp AND element at position i AND position j MINUS one of dp AND element at position i MINUS one AND position j MINUS one of dp PLUS one  
                    SET max_side TO the MAXIMUM OF max_side AND element at position i AND position j of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
      
    FUNCTION init_matrix(rows, columns)  
        SET matrix TO an EMPTY LIST  
        FOR each counter FROM one TO rows  
            SET row_list TO a LIST OF zeros with LENGTH EQUAL TO columns  
            APPEND row_list TO matrix  
        END FOR  
        RETURN matrix  
    END FUNCTION  
END CLASS
