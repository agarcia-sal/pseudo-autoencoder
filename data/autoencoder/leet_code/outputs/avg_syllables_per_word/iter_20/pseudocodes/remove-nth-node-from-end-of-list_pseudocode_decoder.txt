CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the values COLLECTION IS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode constructed WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO a new ListNode constructed WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the condition WHERE the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the recursive CALL TO is_same_list WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS self AND head AND n  
        SET dummy TO a new ListNode constructed WITH zero  
        SET the next ATTRIBUTE OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the next ATTRIBUTE OF first  
        END FOR  
        WHILE first IS NOT None  
            SET first TO the next ATTRIBUTE OF first  
            SET second TO the next ATTRIBUTE OF second  
        END WHILE  
        SET the next ATTRIBUTE OF second TO the next ATTRIBUTE OF the next ATTRIBUTE OF second  
        RETURN the next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
