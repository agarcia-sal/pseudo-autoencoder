CLASS Solution  
    FUNCTION superEggDrop WITH PARAMETERS k INTEGER AND n INTEGER  
        SET dynamic_programming_array TO list of zero repeated k PLUS one times  
        SET moves TO zero  
        WHILE element at position k of dynamic_programming_array IS LESS THAN n  
            INCREMENT moves BY one  
            FOR eggs FROM k DOWN TO one  
                SET element at position eggs of dynamic_programming_array TO one PLUS element at position eggs MINUS one of dynamic_programming_array PLUS element at position eggs of dynamic_programming_array  
            END FOR  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
