CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL the SORT method ON nums to arrange elements in ascending order  
        SET sorted_nums TO a new list containing each element at corresponding positions from nums starting at position one TO the LENGTH OF nums  
        SET n TO the LENGTH OF nums  
        SET left TO the result of subtracting one FROM n then DIVIDING BY two discarding remainder  
        SET right TO the result of subtracting one FROM n  
        FOR index FROM zero TO one LESS THAN n  
            IF the result of the remainder of index DIVIDED BY two EQUALS zero holds  
                SET element at position index of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position index of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
