CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO a new list that contains negative of each element at each position of target  
        CALL heapify on max_heap to transform it into a heap structure  
        SET total_sum TO the sum of all elements in target  
        WHILE the element at position zero of max_heap is LESS THAN negative one holds  
            SET largest TO the negative of the result of popping the smallest element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS EQUAL TO zero  
                RETURN False  
            END IF  
            SET new_value TO largest MODULO rest_sum  
            IF new_value IS EQUAL TO zero AND rest_sum IS NOT EQUAL TO one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            PUSH negative of new_value INTO max_heap  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
