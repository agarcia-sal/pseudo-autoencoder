CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF LENGTH OF t EQUALS LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND value v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT cnt at character c BY two  
                    CALL dfs WITH concatenation of character c PLUS t PLUS character c  
                    INCREMENT cnt at character c BY two  
                END IF  
            END FOR  
        END FUNCTION  

        ASSIGN cnt TO the frequency count of characters in s  
        SET mid TO empty string  
        FOR each character c AND value v IN cnt  
            IF value v MODULO two EQUALS one  
                IF mid IS NOT empty string  
                    RETURN empty list  
                END IF  
                SET mid TO character c  
                DECREMENT cnt at character c BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
