CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s IS EMPTY OR words IS EMPTY OR element at position zero of words IS EMPTY  
            RETURN empty list  
        END IF  
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNT OF each element IN words  
        SET result TO empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an empty count of elements  
            FOR j FROM i TO i PLUS total_length MINUS one STEP BY word_length  
                SET word TO substring from position j TO j PLUS word_length MINUS one of s  
                INCREMENT the count of word IN seen BY one  
                IF the count of word IN seen IS GREATER THAN the count of word IN word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop was NOT broken  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
