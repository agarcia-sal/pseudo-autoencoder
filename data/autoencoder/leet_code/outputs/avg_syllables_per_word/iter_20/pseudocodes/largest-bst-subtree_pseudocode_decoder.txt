CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET the val property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the values collection IS empty THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty DO  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the left property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the right property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the val property of p IS NOT EQUALS TO the val property of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left properties of p and q AND the result of calling is_same_tree with the right properties of p and q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(root)  
        FUNCTION dfs(node)  
            IF node IS none THEN  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO the result of calling dfs with the left property of node  
            SET right_min, right_max, right_size TO the result of calling dfs with the right property of node  
            IF left_max IS LESS THAN the val property of node AND the val property of node IS LESS THAN right_min THEN  
                RETURN the smaller value between the val property of node AND left_min, the greater value between the val property of node AND right_max, the sum of left_size AND right_size AND one  
            END IF  
            RETURN negative infinity, positive infinity, the greater value between left_size AND right_size  
        END FUNCTION  
        RETURN the third element of the tuple returned by calling dfs with root  
    END FUNCTION  
END CLASS
