CLASS Solution  
    FUNCTION combinationSum3(k, n)  
        FUNCTION backtrack(start, path, target, k)  
            IF target EQUALS zero AND k EQUALS zero  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero OR k LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO nine  
                APPEND i TO path  
                CALL backtrack WITH start SET TO i PLUS one path AS path target SET TO target MINUS i k SET TO k MINUS one  
                REMOVE the LAST ELEMENT FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH start SET TO one path AS an empty list target SET TO n k SET TO k  
        RETURN result  
    END FUNCTION  
END CLASS
