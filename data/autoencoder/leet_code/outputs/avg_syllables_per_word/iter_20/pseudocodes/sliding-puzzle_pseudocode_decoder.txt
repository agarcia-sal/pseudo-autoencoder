CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO the tuple of tuples formed by converting each row of board to a tuple  
        SET target TO the tuple consisting of the tuple containing one two three and the tuple containing four five zero  
        SET neighbors TO a mapping of zero position to a set of its possible move positions where zero position zero maps to positions one and three zero position one maps to positions zero two and four zero position two maps to positions one and five zero position three maps to positions zero and four zero position four maps to positions one three and five and zero position five maps to positions two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF the element at position j of the element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a double ended queue initialized with the tuple consisting of start zero_pos and zero  
        SET visited TO a set containing start  
        WHILE queue has elements  
            REMOVE and RETURN the leftmost element of queue ASSIGNING the first component to state the second component to zero_pos and the third component to moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set mapped by zero_pos in neighbors  
                SET new_state TO a new list of lists where each element is a new list converted from each element of state  
                SET i TO the integer division of zero_pos BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO the integer division of neighbor BY three  
                SET nj TO neighbor MODULO three  
                SET the element at position j of the element at position i of new_state TO the element at position nj of the element at position ni of new_state and the element at position nj of the element at position ni of new_state TO the previous value of the element at position j of the element at position i of new_state swapped simultaneously  
                SET new_tuple_state TO the tuple of tuples formed by converting each row of new_state to a tuple  
                IF new_tuple_state is not an element of visited  
                    ADD new_tuple_state to visited  
                    APPEND the tuple consisting of new_tuple_state neighbor and the result of moves PLUS one to queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
