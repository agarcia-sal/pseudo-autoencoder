CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF LENGTH OF nums LESS THAN 2 THEN  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO LENGTH OF arr  
            SET output TO list WITH n zeros  
            SET count TO list WITH ten zeros  
            
            FOR index FROM zero TO n MINUS one  
                SET digit_index TO INTEGER DIVISION OF element at position index of arr BY exp  
                INCREMENT element at position digit_index MODULO ten of count BY one  
            END FOR  
            
            FOR i FROM one TO nine  
                INCREMENT element at position i of count BY element at position i MINUS one of count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i IS GREATER THAN OR EQUAL TO zero  
                SET digit_index TO INTEGER DIVISION OF element at position i of arr BY exp  
                SET element at position element at position digit_index MODULO ten of count MINUS one of output TO element at position i of arr  
                DECREMENT element at position digit_index MODULO ten of count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO maximum element of arr  
            SET exp TO one  
            WHILE INTEGER DIVISION OF max1 BY exp IS GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                MULTIPLY exp BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort WITH nums  
        
        SET max_diff TO zero  
        FOR i FROM one TO LENGTH OF nums MINUS one  
            IF element at position i of nums MINUS element at position i MINUS one of nums IS GREATER THAN max_diff THEN  
                SET max_diff TO element at position i of nums MINUS element at position i MINUS one of nums  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
