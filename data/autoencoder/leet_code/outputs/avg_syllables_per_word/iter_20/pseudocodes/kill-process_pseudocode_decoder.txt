CLASS Solution  
    FUNCTION killProcess(pid AS List OF integers, ppid AS List OF integers, kill AS integer) RETURNS List OF integers  
        SET tree TO a new mapping with default value as an empty list  
        FOR each pair of child and parent taken simultaneously from pid and ppid  
            APPEND child TO the list associated with parent in tree  
        END FOR  
        SET queue TO a new list containing kill  
        SET killed_processes TO a new empty list  
        WHILE queue holds any elements  
            SET current TO the element at position zero of queue  
            REMOVE the element at position zero from queue  
            APPEND current TO killed_processes  
            EXTEND queue BY all children associated with current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
