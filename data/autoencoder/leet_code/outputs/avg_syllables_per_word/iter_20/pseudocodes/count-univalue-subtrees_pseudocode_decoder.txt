CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS A new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value field of p IS NOT EQUAL TO the value field of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree WITH parameters the left child field of p AND the left child field of q  
        AND the result of is_same_tree WITH parameters the right child field of p AND the right child field of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN True AND zero  
            END IF  
            SET is_left_unival, left_count TO THE TWO RESULTS RETURNED BY helper WITH the left child field of node  
            SET is_right_unival, right_count TO THE TWO RESULTS RETURNED BY helper WITH the right child field of node  
            IF is_left_unival AND is_right_unival  
                IF (node HAS NO left child OR the value field of the left child of node EQUALS the value field of node)  
                AND (node HAS NO right child OR the value field of the right child of node EQUALS the value field of node)  
                    RETURN True AND the SUM OF left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False AND the SUM OF left_count PLUS right_count  
        END FUNCTION  
        SET dummy, count TO THE TWO RESULTS RETURNED BY helper WITH root  
        RETURN count  
    END FUNCTION  
END CLASS
