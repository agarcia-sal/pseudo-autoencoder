CLASS Solution  
    FUNCTION reachableNodes(edges AS List of List of integers, maxMoves AS integer, n AS integer) RETURNS integer  
        SET graph TO a mapping from each integer from zero TO n MINUS one TO an empty list  
        FOR each triple consisting of u v cnt IN edges  
            APPEND the pair consisting of v and cnt TO the list at key u in graph  
            APPEND the pair consisting of u and cnt TO the list at key v in graph  
        END FOR  
        
        SET distances TO a list of length n where each element is positive infinity number  
        SET the element at position zero of distances TO zero  
        SET priority_queue TO a list containing the pair of number zero and number zero  
        
        WHILE priority_queue is not empty  
            REMOVE from priority_queue the pair with the minimum first element ASSIGNING to dist and node respectively  
            IF dist IS GREATER THAN the element at position node of distances  
                CONTINUE to next iteration of the loop  
            END IF  
            FOR each pair consisting of neighbor and cnt IN the list at key node in graph  
                SET new_dist TO dist PLUS cnt PLUS one  
                IF new_dist IS LESS THAN the element at position neighbor of distances  
                    SET the element at position neighbor of distances TO new_dist  
                    INSERT the pair new_dist and neighbor INTO priority_queue maintaining the heap property  
                END IF  
            END FOR  
        END WHILE  
        
        SET reachable_nodes TO zero  
        
        FOR each distance_value IN distances  
            IF distance_value IS LESS THAN OR EQUAL TO maxMoves  
                INCREMENT reachable_nodes BY one  
            END IF  
        END FOR  
        
        SET used_edges TO an empty mapping  
        FOR each triple consisting of u v cnt IN edges  
            SET moves_u TO the maximum of zero and the difference of maxMoves MINUS the element at position u of distances  
            SET moves_v TO the maximum of zero and the difference of maxMoves MINUS the element at position v of distances  
            SET reachable_new_nodes TO the minimum of cnt and the sum of moves_u PLUS moves_v  
            INCREMENT reachable_nodes BY reachable_new_nodes  
            ASSIGN to used_edges at key consisting of the pair u and v the value reachable_new_nodes  
        END FOR  
        
        RETURN reachable_nodes  
    END FUNCTION  
END CLASS
