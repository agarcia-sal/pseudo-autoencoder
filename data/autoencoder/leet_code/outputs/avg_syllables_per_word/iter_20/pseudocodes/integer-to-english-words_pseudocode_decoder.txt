CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the WORD Zero  
        END IF  
        
        SET below_twenty TO list of words representing numbers from empty string at position zero to Nineteen in increasing order  
        SET tens TO list of words representing multiples of ten from empty strings at positions zero and one up to Ninety at position nine  
        SET thousands TO list of scale words from empty string for units up to Billion at position three  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n by ten of tens concatenated with a space concatenated with the result of calling helper with n modulo ten  
            ELSE  
                RETURN the element at position the integer division of n by one hundred of below_twenty concatenated with the word Hundred concatenated with a space concatenated with the result of calling helper with n modulo one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF num modulo one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with num modulo one thousand concatenated with the element at position i of thousands concatenated with a space concatenated with result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN the result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
