CLASS Solution  
    FUNCTION poorPigs(buckets minutesToDie minutesToTest)  
        SET states TO the INTEGER DIVISION of minutesToTest BY minutesToDie PLUS one  
        SET logarithm_of_buckets TO the NATURAL LOGARITHM of buckets  
        SET logarithm_of_states TO the NATURAL LOGARITHM of states  
        SET pigs TO the CEILING VALUE of logarithm_of_buckets DIVIDED BY logarithm_of_states  
        RETURN pigs  
    END FUNCTION  
END CLASS
