CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL sort ON people BY sorting EACH person first BY NEGATIVE OF element at position zero OF person THEN BY element at position one OF person IN ascending order  
        SET queue TO empty list  
        FOR each person IN people  
            INSERT person AT position element at position one OF person INTO queue  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
