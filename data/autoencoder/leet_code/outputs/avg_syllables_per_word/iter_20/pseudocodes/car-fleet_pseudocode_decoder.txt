CLASS Solution  
    FUNCTION carFleet WITH PARAMETERS target position speed  
        SET list_of_pairs TO the ZIP of position AND speed sorted in reverse order  
        SET time_to_target TO an empty list  
        FOR each pair OF position_element AND speed_element IN list_of_pairs  
            SET time_to_target_element TO target MINUS position_element DIVIDED BY speed_element  
            APPEND time_to_target_element TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
