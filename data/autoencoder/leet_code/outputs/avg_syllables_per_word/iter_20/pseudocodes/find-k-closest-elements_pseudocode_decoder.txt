CLASS Solution  
    FUNCTION findClosestElements(arr AS List OF Integer, k AS Integer, x AS Integer) AS List OF Integer  
        SET left TO the POSITION WHICH WOULD INSERT x IN arr TO KEEP arr SORTED MINUS one  
        SET right TO left PLUS one  
        WHILE the DIFFERENCE of right MINUS left MINUS one IS LESS THAN k  
            IF left EQUALS negative one OR right IS LESS THAN the LENGTH OF arr AND the ABSOLUTE VALUE of the ELEMENT at position right of arr MINUS x IS LESS THAN the ABSOLUTE VALUE of the ELEMENT at position left of arr MINUS x  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        RETURN the SUBLIST of arr FROM the POSITION left PLUS one TO right  
    END FUNCTION  
END CLASS
