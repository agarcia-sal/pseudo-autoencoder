CLASS Solution  
    FUNCTION shortestPath(grid, k)  
        IF grid IS EMPTY OR element at position zero of grid IS EMPTY  
            RETURN minus one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k IS GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO a list of tuples representing zero and one, one and zero, zero and minus one, and minus one and zero  
        
        INITIALIZE queue AS a double ended queue containing a tuple of zero zero k and zero  
        
        INITIALIZE visited AS a set containing a tuple of zero zero k  
        
        WHILE queue is not empty  
            SET x, y, remaining_k, steps TO the result of removing the element from the front of queue  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx of grid at position ny  
                    IF new_k IS GREATER THAN OR EQUAL TO zero AND a tuple of nx ny and new_k IS NOT IN visited  
                        ADD a tuple of nx ny and new_k TO visited  
                        ADD a tuple of nx ny new_k and steps PLUS one TO the end of queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
