CLASS Solution  
    FUNCTION fourSum(nums AS List of Integers, target AS Integer) RETURNS List of Lists of Integers  
        PERFORM SORTING ON the list nums IN NONDECREASING ORDER  
        SET n TO the LENGTH OF nums  
        SET quadruplets TO an EMPTY list of lists  
        
        FOR index i FROM zero TO n MINUS three  
            IF i IS GREATER THAN zero AND element at position i of nums EQUALS element at position i MINUS one of nums  
                WITHHOLD the CURRENT iteration AND CONTINUE to the next iteration  
            END IF  
            FOR index j FROM i PLUS one TO n MINUS two  
                IF j IS GREATER THAN i PLUS one AND element at position j of nums EQUALS element at position j MINUS one of nums  
                    WITHHOLD the CURRENT iteration AND CONTINUE to the next iteration  
                END IF  
                SET left TO j PLUS one  
                SET right TO n MINUS one  
                WHILE left IS LESS THAN right  
                    SET total TO element at position i of nums PLUS element at position j of nums PLUS element at position left of nums PLUS element at position right of nums  
                    IF total EQUALS target  
                        APPEND the list consisting of element at position i of nums, element at position j of nums, element at position left of nums, element at position right of nums TO quadruplets  
                        WHILE left IS LESS THAN right AND element at position left of nums EQUALS element at position left PLUS one of nums  
                            INCREMENT left BY one  
                        END WHILE  
                        WHILE left IS LESS THAN right AND element at position right of nums EQUALS element at position right MINUS one of nums  
                            DECREMENT right BY one  
                        END WHILE  
                        INCREMENT left BY one  
                        DECREMENT right BY one  
                    ELSE IF total IS LESS THAN target  
                        INCREMENT left BY one  
                    ELSE  
                        DECREMENT right BY one  
                    END IF  
                END WHILE  
            END FOR  
        END FOR  
        RETURN quadruplets  
    END FUNCTION  
END CLASS
