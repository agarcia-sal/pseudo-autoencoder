CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new default dictionary of integer values  
        
        FOR each cpdomain IN cpdomains  
            SET count and domain TO the result of splitting cpdomain at the first space  
            SET count TO the integer conversion of count  
            SET subdomains TO the result of splitting domain at each period  
            
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain TO the concatenation of the elements from position index TO the end of subdomains joined by a period  
                INCREMENT the value at key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each domain and count IN the items of domain_count  
            APPEND the string consisting of count FOLLOWED BY a space FOLLOWED BY domain TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
