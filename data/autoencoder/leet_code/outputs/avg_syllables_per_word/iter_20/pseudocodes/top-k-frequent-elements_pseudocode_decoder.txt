CLASS Solution  
    FUNCTION topKFrequent(nums, k)  
        SET frequency_map TO a new mapping that returns zero for missing keys  
        FOR each num IN nums  
            INCREMENT the element at position num of frequency_map BY one  
        END FOR  
        SET max_frequency TO the greatest value among all values of frequency_map  
        SET buckets TO a new list containing empty lists repeated max_frequency PLUS one times  
        FOR each pair num and freq IN all pairs of keys and corresponding values of frequency_map  
            APPEND num TO the element at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the LENGTH OF the element at position freq of buckets IS GREATER THAN zero  
                APPEND all elements of the element at position freq of buckets TO result  
                IF the LENGTH OF result IS GREATER THAN OR EQUAL TO k  
                    RETURN the sublist of result from position one TO position k  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
