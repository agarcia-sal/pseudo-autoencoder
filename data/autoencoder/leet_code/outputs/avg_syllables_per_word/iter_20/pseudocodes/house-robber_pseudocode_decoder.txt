CLASS Solution  
    FUNCTION rob(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  

        SET dp TO a list of zeros of the LENGTH OF nums  
        SET element at position zero of dp TO element at position zero of nums  
        SET element at position one of dp TO the GREATER OF element at position zero of nums AND element at position one of nums  

        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET value_one TO element at position index MINUS one of dp  
            SET value_two TO element at position index MINUS two of dp PLUS element at position index of nums  
            SET element at position index of dp TO the GREATER OF value_one AND value_two  
        END FOR  

        RETURN element at the LAST POSITION of dp  
    END FUNCTION  
END CLASS
