CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a two dimensional list with size size1 PLUS one rows and each row having two TO THE power OF size2 columns filled with infinity value  
        SET element at row zero position zero of dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two TO THE power OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET current_mask TO bitwise OR of mask and two TO THE power OF j  
                    SET candidate_cost TO element at row i position mask of dp PLUS element at row i position j of cost  
                    SET element at row i PLUS one position current_mask of dp TO the MINIMUM OF element at row i PLUS one position current_mask of dp AND candidate_cost  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF bitwise AND of mask and two TO THE power OF j IS NOT EQUAL TO zero  
                        SET mask_without_j TO bitwise XOR of mask and two TO THE power OF j  
                        SET candidate_cost TO element at row i PLUS one position mask_without_j of dp PLUS element at row i position j of cost  
                        SET element at row i PLUS one position mask of dp TO the MINIMUM OF element at row i PLUS one position mask of dp AND candidate_cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at row size1 position two TO THE power OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
