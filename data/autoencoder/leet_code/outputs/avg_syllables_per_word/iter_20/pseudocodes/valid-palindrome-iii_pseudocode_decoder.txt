CLASS Solution  
    FUNCTION isValidPalindrome(s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional list of zeros with size n BY n  
            FOR i FROM zero TO n MINUS one  
                SET element at position i AND i of dp TO one  
            END FOR  
            FOR length FROM two TO n  
                FOR i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF element at position i of s EQUALS element at position j of s  
                        SET element at position i AND j of dp TO element at position i PLUS one AND j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i AND j of dp TO THE GREATER VALUE BETWEEN element at position i PLUS one AND j of dp AND element at position i AND j MINUS one of dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero AND n MINUS one of dp  
        END FUNCTION  
        SET longest_palindrome_length TO longest_palindromic_subseq WITH PARAMETER s  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
