CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty four LESS THAN one TIMES ten TO THE POWER OF MINUS six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUAL TO i AND k NOT EQUAL TO j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation IN the collection consisting of true division multiplication addition subtraction  
                            IF the operation IS addition OR the operation IS multiplication AND j GREATER THAN i  
                                CONTINUE to the next operation  
                            END IF  
                            
                            IF the operation IS NOT true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying operation to the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS True  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
