CLASS Solution  
    FUNCTION canVisitAllRooms WITH parameters rooms  
        SET visited TO empty set  
        SET stack TO new list containing zero  
        WHILE the LENGTH OF stack IS GREATER THAN zero  
            SET current_room TO last element removed FROM stack  
            ADD current_room TO visited  
            FOR each key IN element at position current_room of rooms  
                IF key IS NOT IN visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
