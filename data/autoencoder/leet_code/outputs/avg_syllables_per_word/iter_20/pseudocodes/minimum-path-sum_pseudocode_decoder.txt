CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        SET dp TO a two dimensional list of size m by n filled with zeros  
        
        SET element at position zero zero of dp TO element at position zero zero of grid  
        
        FOR each j FROM one TO n MINUS one  
            SET element at position zero j of dp TO element at position zero j MINUS one of dp PLUS element at position zero j of grid  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            SET element at position i zero of dp TO element at position i MINUS one zero of dp PLUS element at position i zero of grid  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                SET element at position i j of dp TO the minimum BETWEEN element at position i MINUS one j of dp AND element at position i j MINUS one of dp PLUS element at position i j of grid  
            END FOR  
        END FOR  
        
        RETURN element at position m MINUS one n MINUS one of dp  
    END FUNCTION  
END CLASS
