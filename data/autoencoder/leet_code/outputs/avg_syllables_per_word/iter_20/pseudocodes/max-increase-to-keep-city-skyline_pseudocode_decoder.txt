CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO the LENGTH OF grid  
        SET row_maxes TO a list CONTAINING the MAXIMUM value OF each row IN grid  
        SET col_maxes TO a list CONTAINING the MAXIMUM value OF each column IN grid OBTAINED BY transposing grid  
        SET total_increase TO zero  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                SET max_height TO the MINIMUM OF the element at position i of row_maxes AND the element at position j of col_maxes  
                IF max_height GREATER THAN the element at position i of grid AND the element at position j of grid  
                    INCREMENT total_increase BY max_height MINUS the element at position i of grid AND the element at position j of grid  
                END IF  
            END FOR  
        END FOR  
        RETURN total_increase  
    END FUNCTION  
END CLASS
