CLASS Solution  
    FUNCTION reverseWords(s)  
        CALL the reverse method OF list s TO reverse the entire list  
        SET start TO zero  
        FOR each end FROM zero TO the length of list s MINUS one  
            IF the element at position end of list s EQUALS a space character  
                ASSIGN to the segment from position start TO position end MINUS one of list s the reversed segment from position start TO position end MINUS one of list s  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        ASSIGN to the segment from position start TO the end of list s the reversed segment from position start TO the end of list s  
    END FUNCTION  
END CLASS
