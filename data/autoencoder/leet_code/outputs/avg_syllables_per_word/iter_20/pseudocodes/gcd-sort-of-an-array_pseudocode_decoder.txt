CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER size  
        SET parent TO a list containing integer values from zero TO size MINUS one  
        SET rank TO a list containing value one repeated size times  
    END FUNCTION  
  
    FUNCTION find WITH PARAMETER p  
        IF the element at position p of parent NOT EQUALS p THEN  
            SET the element at position p of parent TO the result of calling find WITH the element at position p of parent  
        END IF  
        RETURN the element at position p of parent  
    END FUNCTION  
  
    FUNCTION union WITH PARAMETERS p AND q  
        SET rootP TO the result of calling find WITH p  
        SET rootQ TO the result of calling find WITH q  
        IF rootP EQUALS rootQ THEN  
            RETURN false  
        END IF  
        IF the element at position rootP of rank GREATER THAN the element at position rootQ of rank THEN  
            SET the element at position rootQ of parent TO rootP  
        ELSE IF the element at position rootP of rank LESS THAN the element at position rootQ of rank THEN  
            SET the element at position rootP of parent TO rootQ  
        ELSE  
            SET the element at position rootQ of parent TO rootP  
            INCREMENT the element at position rootP of rank BY one  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION gcdSort WITH PARAMETER nums  
        SET sorted_nums TO nums arranged in ascending order  
        SET max_val TO the greatest value within nums  
        SET uf TO a new instance of UnionFind with parameter max_val PLUS one  
        FOR each num IN nums  
            FOR each factor FROM two TO the integer part of the square root of num PLUS one  
                IF num MODULO factor EQUALS zero THEN  
                    CALL union method of uf WITH num AND factor  
                    CALL union method of uf WITH num AND num DIVIDED BY factor  
                END IF  
            END FOR  
        END FOR  
        FOR each index FROM zero TO the length of nums MINUS one  
            IF the result of calling find method of uf WITH the element at position index of nums NOT EQUALS the result of calling find method of uf WITH the element at position index of sorted_nums THEN  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
