CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS an empty list THEN  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the position found by binary search for the leftmost position where num fits in dp  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
