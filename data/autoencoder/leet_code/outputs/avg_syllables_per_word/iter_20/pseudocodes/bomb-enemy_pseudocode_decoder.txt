CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS nothing OR the element at position zero of grid EQUALS nothing  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        SET max_kills TO zero  
        
        SET row_kills TO zero  
        SET col_kills TO a list of size n FILLED WITH zero  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF j EQUALS zero OR the element at position j MINUS one of row at position i of grid EQUALS the character WALL  
                    SET row_kills TO zero  
                    FOR k FROM j TO n MINUS one  
                        IF the element at position k of row at position i of grid EQUALS the character ENEMY  
                            INCREMENT row_kills BY one  
                        ELSE IF the element at position k of row at position i of grid EQUALS the character WALL  
                            BREAK the FOR LOOP  
                        END IF  
                    END FOR  
                END IF  
                
                IF i EQUALS zero OR the element at position j of row at position i MINUS one of grid EQUALS the character WALL  
                    SET the element at position j of col_kills TO zero  
                    FOR k FROM i TO m MINUS one  
                        IF the element at position j of row at position k of grid EQUALS the character ENEMY  
                            INCREMENT the element at position j of col_kills BY one  
                        ELSE IF the element at position j of row at position k of grid EQUALS the character WALL  
                            BREAK the FOR LOOP  
                        END IF  
                    END FOR  
                END IF  
                
                IF the element at position j of row at position i of grid EQUALS the character ZERO  
                    SET max_kills TO the LARGER OF max_kills AND row_kills PLUS the element at position j of col_kills  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
