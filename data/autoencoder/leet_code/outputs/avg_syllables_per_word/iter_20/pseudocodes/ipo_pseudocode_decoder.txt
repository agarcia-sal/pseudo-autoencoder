CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO a list of pairs each containing element at position index of capital and element at position index of profits for each index from zero TO the LENGTH OF capital MINUS one  
        SORT projects BY first element of each pair in ascending order  
        SET max_heap TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR count FROM zero TO k MINUS one  
            WHILE i LESS THAN n AND first element at position i of projects LESS THAN OR EQUAL TO w  
                ADD the negative value of second element at position i of projects TO max_heap maintaining heap property as a min-heap  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap is empty  
                BREAK the loop  
            END IF  
            SET w TO w PLUS the negative of the top element removed from max_heap maintaining heap property  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
