CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s k  
        IF the LENGTH OF s IS LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the FREQUENCY COUNTER OF characters IN s  
        FOR each char IN count  
            IF the FREQUENCY OF char IN count IS LESS THAN k  
                RETURN the MAXIMUM VALUE AMONG the RESULTS OF longestSubstring CALLED WITH PARAMETERS each substring OBTAINED BY SPLITTING s BY char AND k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
