CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO list OF zero WITH LENGTH EQUAL TO LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  

        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS a star symbol  
                RETURN nine  
            ELSE IF char EQUALS zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  

        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS a star symbol AND char2 EQUALS a star symbol  
                RETURN fifteen  
            ELSE IF char1 EQUALS a star symbol  
                IF char2 IS LESS THAN OR EQUAL TO character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS a star symbol  
                IF char1 EQUALS character one  
                    RETURN nine  
                ELSE IF char1 EQUALS character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value REPRESENTED BY the concatenation OF char1 AND char2  
                IF num IS GREATER THAN OR EQUAL TO ten AND num IS LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  

        FOR index FROM one TO LENGTH OF s  
            SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of dp MULTIPLIED BY ways_to_decode_single(element at position index MINUS one of s) MODULO MOD  
            IF index IS GREATER THAN one  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS two of dp MULTIPLIED BY ways_to_decode_pair(element at position index MINUS two of s, element at position index MINUS one of s) MODULO MOD  
            END IF  
        END FOR  

        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
