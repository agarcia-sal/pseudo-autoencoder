CLASS Solution  
    FUNCTION loudAndRich(richer AS List OF List OF Integer, quiet AS List OF Integer) RETURNS List OF Integer  
        SET graph TO result of calling defaultdict with an empty list as default value  
        FOR each pair a and b IN richer  
            APPEND a TO the list at element at position b of graph  
        END FOR  
        SET n TO the LENGTH OF quiet  
        SET answer TO a list containing value negative one repeated n times  
        FUNCTION dfs(person AS Integer) RETURNS Integer  
            IF the element at position person of answer EQUALS negative one  
                SET the element at position person of answer TO person  
                FOR each neighbor IN the list at element at position person of graph  
                    SET candidate TO the result of calling dfs with neighbor  
                    IF the element at position candidate of quiet LESS THAN the element at position at element at position person of answer of answer of quiet  
                        SET the element at position person of answer TO candidate  
                    END IF  
                END FOR  
            END IF  
            RETURN the element at position person of answer  
        END FUNCTION  
        FOR index i FROM zero TO n MINUS one  
            CALL dfs with i  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
