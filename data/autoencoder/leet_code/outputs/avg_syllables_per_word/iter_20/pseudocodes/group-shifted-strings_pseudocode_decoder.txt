CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET offset TO position of character c in alphabet MINUS position of character at position zero of s in alphabet  
                SET wrapped_offset TO offset MODULO twenty six  
                APPEND string representation of wrapped_offset TO canonical_form  
            END FOR  
            SET result_string TO concatenation of elements in canonical_form SEPARATED BY comma  
            RETURN result_string  
        END FUNCTION  
        
        SET groups TO dictionary with default value as empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form WITH parameter s  
            APPEND s TO list at key canonical_form in groups  
        END FOR  
        
        RETURN list of all values stored in groups  
    END FUNCTION  
END CLASS
