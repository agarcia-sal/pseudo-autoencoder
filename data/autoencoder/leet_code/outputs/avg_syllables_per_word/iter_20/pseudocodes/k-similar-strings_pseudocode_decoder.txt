CLASS Solution  
    FUNCTION kSimilarity(s1 AS string, s2 AS string) RETURNS integer  
        FUNCTION neighbors(s AS string)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR each j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUAL TO element at position j of s2  
                    YIELD a new string constructed by concatenating substring from position zero TO i MINUS one of s PLUS element at position j of s PLUS substring from position i PLUS one TO j MINUS one of s PLUS element at position i of s PLUS substring from position j PLUS one TO end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a double ended queue initialized with a tuple consisting of s1 and zero  
        SET visited TO a set initialized with s1  
        
        WHILE queue is not empty  
            SET s AND swaps TO the first element removed from the left of queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the sequence produced by neighbors called with s  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor AND swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
