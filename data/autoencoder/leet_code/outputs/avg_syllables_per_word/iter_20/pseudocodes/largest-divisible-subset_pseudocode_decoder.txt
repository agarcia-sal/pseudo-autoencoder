CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        PERFORM SORT ON nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET f TO a LIST OF LENGTH n WITH ALL ELEMENTS SET TO one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums MODULO the element at position j of nums EQUALS zero THEN  
                    SET f at position i TO the GREATER VALUE BETWEEN f at position i AND f at position j PLUS one  
                END IF  
            END FOR  
            IF f at position k LESS THAN f at position i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f at position k  
        SET i TO k  
        SET ans TO an empty LIST  
        WHILE m GREATER THAN zero  
            IF the element at position k of nums MODULO the element at position i of nums EQUALS zero AND f at position i EQUALS m THEN  
                APPEND the element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
