CLASS Solution  
    FUNCTION largestMultipleOfThree WITH PARAMETER digits AS List of integers  
        SET count TO Counter CREATED FROM digits  
        SET total_sum TO THE SUM OF ALL elements IN digits  
        SET remainder TO total_sum MODULO THREE  
        
        IF remainder EQUALS one  
            FOR each digit IN the list of digits one FOUR SEVEN  
                IF the element at position digit of count IS GREATER THAN zero  
                    DECREMENT the element at position digit of count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred IN the previous loop  
                FOR each digit IN the list of digits two FIVE EIGHT  
                    IF the element at position digit of count IS GREATER THAN one  
                        DECREMENT the element at position digit of count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit IN the list of digits two FIVE EIGHT  
                IF the element at position digit of count IS GREATER THAN zero  
                    DECREMENT the element at position digit of count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred IN the previous loop  
                FOR each digit IN the list of digits one FOUR SEVEN  
                    IF the element at position digit of count IS GREATER THAN one  
                        DECREMENT the element at position digit of count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        SET result TO an empty list  
        FOR each digit FROM nine DOWN TO zero  
            APPEND the string representation OF digit REPEATED the element at position digit of count times TO result  
        END FOR  
        
        SET final_number TO the concatenation OF all elements IN result  
        
        IF the length OF final_number IS GREATER THAN zero AND the element at position one of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
