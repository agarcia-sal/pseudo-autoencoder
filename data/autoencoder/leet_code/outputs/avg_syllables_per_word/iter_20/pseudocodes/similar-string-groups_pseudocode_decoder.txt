CLASS Solution  
    FUNCTION numSimilarGroups(strs)  
        FUNCTION are_similar(s1, s2)  
            SET diff TO an empty list  
            FOR each index FROM zero TO the LENGTH OF s1 MINUS one  
                IF the element at position index of s1 NOT EQUALS the element at position index of s2  
                    APPEND index TO diff  
                END IF  
            END FOR  
            IF the LENGTH OF diff EQUALS zero OR the LENGTH OF diff EQUALS two AND the element at position the first element of diff of s1 EQUALS the element at position the second element of diff of s2 AND the element at position the second element of diff of s1 EQUALS the element at position the first element of diff of s2  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FUNCTION dfs(node, visited)  
            ADD node TO visited  
            FOR each neighbor IN the element at position node of graph  
                IF neighbor NOT IN visited  
                    CALL dfs WITH neighbor AND visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET n TO the LENGTH OF strs  
        SET graph TO a mapping FROM each index FROM zero TO n MINUS one TO an empty list  
        
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM i PLUS one TO n MINUS one  
                IF CALL are_similar WITH the element at position i of strs AND the element at position j of strs RETURNS true  
                    APPEND j TO the element at position i of graph  
                    APPEND i TO the element at position j of graph  
                END IF  
            END FOR  
        END FOR  
        
        SET visited TO an empty set  
        SET num_groups TO zero  
        
        FOR each i FROM zero TO n MINUS one  
            IF i NOT IN visited  
                CALL dfs WITH i AND visited  
                INCREMENT num_groups BY one  
            END IF  
        END FOR  
        
        RETURN num_groups  
    END FUNCTION  
END CLASS
