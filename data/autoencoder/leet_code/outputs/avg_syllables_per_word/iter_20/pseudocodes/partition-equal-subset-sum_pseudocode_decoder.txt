CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF elements IN nums  
        IF the REMAINDER OF total_sum DIVIDED BY two IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        CREATE list dp OF BOOLEAN values WITH SIZE target PLUS one ALL SET TO False  
        SET element at position zero of dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET element at position i of dp TO the LOGICAL OR OF element at position i of dp AND element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
