CLASS Solution  
    FUNCTION countBattleships WITH board AS LIST OF LIST OF STRING  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF element at position i of board at position j EQUALS letter X  
                    IF i EQUALS zero OR element at position i MINUS one of board at position j EQUALS single character dot  
                        AND j EQUALS zero OR element at position i of board at position j MINUS one EQUALS single character dot  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
