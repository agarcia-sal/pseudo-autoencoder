CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of length n with each element SET TO negative one  
        SET stack TO an empty list  
        FOR each i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE stack is not empty AND the element at position the last element of stack of nums is LESS THAN the element at position index of nums holds  
                SET popped_index TO the last element of stack THEN REMOVE it FROM stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            IF i is LESS THAN n THEN  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
