CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s AS string  
        SET s TO a list conversion of s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s is GREATER THAN one  
            SET found_match TO false  
            FOR i FROM the LENGTH OF s MINUS one TO one IN STEPS OF minus one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SET temporary_variable TO the element at position j PLUS one of s  
                        SET the element at position j PLUS one of s TO the element at position j of s  
                        SET the element at position j of s TO temporary_variable  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE the element at position zero FROM s  
                    REMOVE the element at position negative one FROM s  
                    SET found_match TO true  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            
            IF found_match IS false  
                REMOVE the element at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
