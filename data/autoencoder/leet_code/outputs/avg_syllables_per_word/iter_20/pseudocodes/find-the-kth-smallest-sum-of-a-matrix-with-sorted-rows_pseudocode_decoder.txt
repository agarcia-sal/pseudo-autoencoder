CLASS Solution  
    FUNCTION kthSmallest(mat WITH TYPE LIST OF LIST OF INTEGER, k WITH TYPE INTEGER) RETURNS INTEGER  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET min_heap TO a LIST containing a SINGLE element consisting of  
            the SUM of element at position zero of each row in mat  
            AND a TUPLE of zeros of LENGTH m  
        SET visited TO a SET containing the TUPLE of zeros at position one of the SINGLE element of min_heap  
        
        WHILE k IS GREATER THAN zero HOLDS  
            REMOVE the SMALLEST element FROM min_heap AND ASSIGN TO current_sum AND indices  
            DECREMENT k BY one  
            
            IF k EQUALS zero THEN  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF element at position i of indices PLUS one IS LESS THAN n THEN  
                    SET new_indices TO a COPY OF indices CONVERTED TO LIST  
                    INCREMENT element at position i of new_indices BY one  
                    SET new_indices_tuple TO new_indices CONVERTED TO TUPLE  
                    SET new_sum TO current_sum MINUS element at position i AND at position element at position i of indices of mat PLUS element at position i AND at position element at position i of new_indices of mat  
                    
                    IF new_indices_tuple IS NOT IN visited THEN  
                        ADD new_indices_tuple TO visited  
                        ADD the PAIR consisting of new_sum AND new_indices_tuple TO min_heap USING HEAP PUSH  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
