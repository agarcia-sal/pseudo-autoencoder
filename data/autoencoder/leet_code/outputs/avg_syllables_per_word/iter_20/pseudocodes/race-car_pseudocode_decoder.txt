CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN pointer TO a new double ended queue initialized with a tuple of zero for position, one for speed, and zero for steps  
        ASSIGN visited TO a new set initialized with a tuple of zero for position and one for speed  
        WHILE the double ended queue is not empty  
            ASSIGN position, speed, steps TO the elements obtained by removing the element from the left end of the double ended queue  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed IS NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO the set visited  
                APPEND the tuple of next_position, next_speed, steps PLUS one TO the right end of the double ended queue  
            END IF  
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed IS NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO the set visited  
                APPEND the tuple of position, next_speed, steps PLUS one TO the right end of the double ended queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
