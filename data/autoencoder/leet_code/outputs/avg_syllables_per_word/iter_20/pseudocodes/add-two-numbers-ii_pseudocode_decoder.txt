CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values OF TYPE list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode OBJECT INITIALIZED WITH element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the END of values  
        SET node TO A NEW ListNode OBJECT INITIALIZED WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN THE RESULT OF THE LOGICAL AND OF THE EXPRESSION THAT element at position one of p1 dot val EQUALS element at position one of p2 dot val AND THE RECURSIVE CALL TO is_same_list WITH p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 OF TYPE optional ListNode AND l2 OF TYPE optional ListNode RETURNING optional ListNode  
        FUNCTION pushIntoStack WITH PARAMETER node  
            SET stack TO AN EMPTY LIST  
            WHILE node IS NOT none  
                APPEND element at position one of node dot val TO stack  
                SET node TO node dot next  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO THE RESULT OF pushIntoStack WITH l1  
        SET stack2 TO THE RESULT OF pushIntoStack WITH l2  
        
        SET carry TO zero  
        SET result TO none  
        
        WHILE THE LOGICAL OR OF stack1 IS NOT EMPTY OR stack2 IS NOT EMPTY OR carry IS NOT zero HOLDS  
            IF stack1 IS NOT EMPTY  
                SET val1 TO THE LAST ELEMENT REMOVED FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 IS NOT EMPTY  
                SET val2 TO THE LAST ELEMENT REMOVED FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO A NEW ListNode OBJECT INITIALIZED WITH digit  
            SET new_node dot next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
