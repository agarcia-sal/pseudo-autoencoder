CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a list of empty lists repeated n times  
        FOR each x y time IN edges  
            APPEND the pair consisting of y AND time TO element at position x of graph  
            APPEND the pair consisting of x AND time TO element at position y of graph  
        END FOR  
        SET pq TO a list containing a single triple consisting of element at position zero of passingFees THEN zero THEN zero  
        SET visited_time TO a list of infinite values repeated n times  
        SET element at position zero of visited_time TO zero  
        WHILE pq is not empty  
            REMOVE and RETURN the smallest element FROM pq INTO current_cost THEN current_city THEN current_time  
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            FOR each neighbor travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET element at position neighbor of visited_time TO new_time  
                    ADD the triple consisting of new_cost THEN neighbor THEN new_time TO pq in heap order  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
