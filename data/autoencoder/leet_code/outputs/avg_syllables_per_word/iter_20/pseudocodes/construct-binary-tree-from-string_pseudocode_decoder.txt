CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO ListNode with the element at position zero of values  
    SET p TO head  
    FOR each element val IN the sublist of values from position one TO the end  
        SET node TO ListNode with val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the comparison of the val attribute of p1 EQUALS to the val attribute of p2 AND the result of is_same_list called with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO a new deque object  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element removed from queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the left attribute of node TO TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET the right attribute of node TO TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree called with the left attribute of p AND the left attribute of q AND the result of is_same_tree called with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index IS GREATER THAN OR EQUAL TO the LENGTH OF s THEN  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index IS LESS THAN the LENGTH OF s AND (the element at position index of s IS a digit OR the element at position index of s EQUALS the minus sign)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO TreeNode with the integer value parsed from the substring from position start TO one LESS THAN index of s  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the opening parenthesis THEN  
                SET node.left AND index TO the two results returned by parse_tree called with index PLUS one  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the opening parenthesis THEN  
                SET node.right AND index TO the two results returned by parse_tree called with index PLUS one  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the closing parenthesis THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND discard_variable TO the two results returned by parse_tree called with zero  
        RETURN root  
    END FUNCTION  
END CLASS
