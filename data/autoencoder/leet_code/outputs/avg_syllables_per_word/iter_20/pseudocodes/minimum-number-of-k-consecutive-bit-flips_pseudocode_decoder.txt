CLASS Solution  
    FUNCTION minKBitFlips(nums WITH TYPE List OF Integer, k WITH TYPE Integer) RETURNS Integer  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a list of zeros with LENGTH of n PLUS one  
        SET flip_state TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            SET flip_state TO the EXCLUSIVE OR of flip_state AND the element at position i of flipped  
            
            IF the element at position i of nums EQUALS flip_state THEN  
                IF i PLUS k GREATER THAN n THEN  
                    RETURN negative one  
                END IF  
                SET flip_state TO the EXCLUSIVE OR of flip_state AND one  
                SET the element at position i PLUS k of flipped TO the EXCLUSIVE OR of the element at position i PLUS k of flipped AND one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
