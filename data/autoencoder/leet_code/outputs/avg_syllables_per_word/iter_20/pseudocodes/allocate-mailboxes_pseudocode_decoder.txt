CLASS Solution  
    FUNCTION minDistance WITH PARAMETERS houses AND k  
        SORT houses IN INCREASING ORDER  
        SET n TO the LENGTH OF houses  
          
        SET cost TO a two dimensional list of zeroes of size n BY n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i AND j of cost TO element at position i PLUS one AND j MINUS one of cost PLUS the element at position j of houses MINUS the element at position i of houses  
            END FOR  
        END FOR  
          
        SET dp TO a two dimensional list FILLED WITH infinity OF size n PLUS one BY k PLUS one  
        SET element at position zero AND zero of dp TO zero  
          
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET element at position i AND j of dp TO the minimum BETWEEN element at position i AND j of dp AND the sum of element at position p AND j MINUS one of dp AND element at position p AND i MINUS one of cost  
                END FOR  
            END FOR  
        END FOR  
          
        RETURN element at position n AND k of dp  
    END FUNCTION  
END CLASS
