CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zero repeated n times  
        SET right_costs TO a list of zero repeated n times  

        IF the element at position zero of s EQUALS character one  
            SET element at position zero of left_costs TO one  
        ELSE  
            SET element at position zero of left_costs TO zero  
        END IF  

        FOR index i FROM one TO n MINUS one  
            IF the element at position i of s EQUALS character zero  
                SET element at position i of left_costs TO element at position i MINUS one of left_costs  
            ELSE  
                SET element at position i of left_costs TO the MINIMUM BETWEEN the sum of index i PLUS one AND the sum of element at position i MINUS one of left_costs PLUS two  
            END IF  
        END FOR  

        IF the element at position n MINUS one of s EQUALS character one  
            SET element at position n MINUS one of right_costs TO one  
        ELSE  
            SET element at position n MINUS one of right_costs TO zero  
        END IF  

        FOR index i FROM n MINUS two DOWN TO zero  
            IF the element at position i of s EQUALS character zero  
                SET element at position i of right_costs TO element at position i PLUS one of right_costs  
            ELSE  
                SET element at position i of right_costs TO the MINIMUM BETWEEN the difference of n MINUS index i AND the sum of element at position i PLUS one of right_costs PLUS two  
            END IF  
        END FOR  

        SET min_time TO the MINIMUM BETWEEN element at position n MINUS one of left_costs AND element at position zero of right_costs  

        FOR index i FROM zero TO n MINUS two  
            SET min_time TO the MINIMUM BETWEEN min_time AND the sum of element at position i of left_costs PLUS element at position i PLUS one of right_costs  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
