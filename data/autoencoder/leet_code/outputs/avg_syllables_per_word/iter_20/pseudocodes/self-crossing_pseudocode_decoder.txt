CLASS Solution  
    FUNCTION isSelfCrossing(distance)  
        SET n TO the LENGTH OF distance  
        IF n LESS THAN four  
            RETURN False  
        END IF  
        FOR i FROM three TO n MINUS one  
            IF the element at position i of distance GREATER THAN OR EQUAL TO the element at position i MINUS two of distance AND the element at position i MINUS one of distance LESS THAN OR EQUAL TO the element at position i MINUS three of distance  
                RETURN True  
            END IF  
            IF i GREATER THAN OR EQUAL TO four AND the element at position i MINUS one of distance EQUALS the element at position i MINUS three of distance AND the SUM of the element at position i of distance AND the element at position i MINUS four of distance GREATER THAN OR EQUAL TO the element at position i MINUS two of distance  
                RETURN True  
            END IF  
            IF i GREATER THAN OR EQUAL TO five AND the element at position i MINUS two of distance GREATER THAN OR EQUAL TO the element at position i MINUS four of distance AND the element at position i MINUS three of distance GREATER THAN OR EQUAL TO the element at position i MINUS one of distance AND the SUM of the element at position i MINUS one of distance AND the element at position i MINUS five of distance GREATER THAN OR EQUAL TO the element at position i MINUS three of distance AND the SUM of the element at position i of distance AND the element at position i MINUS four of distance GREATER THAN OR EQUAL TO the element at position i MINUS two of distance  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
