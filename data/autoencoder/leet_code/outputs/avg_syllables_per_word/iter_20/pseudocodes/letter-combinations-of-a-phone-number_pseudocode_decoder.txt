CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF the LENGTH OF digits EQUALS zero  
            RETURN an empty list  
        END IF  
        
        SET phone_map TO a mapping from digit two TO letters abc and digit three TO letters def and digit four TO letters ghi and digit five TO letters jkl and digit six TO letters mno and digit seven TO letters pqrs and digit eight TO letters tuv and digit nine TO letters wxyz  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the letters MAPPED BY the element at position zero of next_digits IN phone_map  
                    CALL backtrack WITH the concatenation of combination AND letter AND the substring from position one TO the end of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO an empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
