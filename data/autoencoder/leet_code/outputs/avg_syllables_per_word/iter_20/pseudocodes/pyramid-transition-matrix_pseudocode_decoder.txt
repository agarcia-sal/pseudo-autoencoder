CLASS Solution  
    FUNCTION pyramidTransition(bottom, allowed)  
        SET allowed_map TO a default dictionary where each key maps to another default dictionary mapping keys to lists  
        FOR each rule IN allowed  
            SET left TO element at position zero of rule  
            SET right TO element at position one of rule  
            SET top TO element at position two of rule  
            APPEND top TO the list at key right in the dictionary at key left in allowed_map  
        END FOR  
        
        FUNCTION can_build_pyramid(current_bottom, current_top)  
            IF the length of current_bottom EQUALS one  
                RETURN True  
            END IF  
            IF the length of current_top EQUALS the length of current_bottom MINUS one  
                RETURN the result of can_build_pyramid with current_top and an empty list as arguments  
            END IF  
            
            SET left TO element at position the length of current_top of current_bottom  
            SET right TO element at position the length of current_top PLUS one of current_bottom  
            IF allowed_map contains the key left AND the dictionary at key left in allowed_map contains the key right  
                FOR each top IN the list at key right in the dictionary at key left in allowed_map  
                    IF can_build_pyramid called with current_bottom and current_top concatenated with top returns True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
            RETURN False  
        END FUNCTION  
        
        RETURN the result of calling can_build_pyramid with bottom and an empty list as arguments  
    END FUNCTION  
END CLASS
