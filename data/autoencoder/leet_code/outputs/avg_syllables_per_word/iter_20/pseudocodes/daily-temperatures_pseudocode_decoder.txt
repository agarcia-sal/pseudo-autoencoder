CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a list of zeroes of length n  
        SET stack TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            WHILE the stack is NOT empty AND the element at position i of temperatures IS GREATER THAN the element at the element at the last position of stack of temperatures  
                SET prev_index TO the removal of the last element of stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
