CLASS Solution  
    FUNCTION mincostToHireWorkers(quality AS List of integers, wage AS List of integers, k AS integer) RETURNS float  
        SET workers TO the SORTED list formed by each element being a pair where the first element is the division of element at the current position of wage BY element at the same position of quality AND the second element is the element at the same position of quality  
        SET max_heap TO an empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair consisting of ratio AND q IN workers  
            PUSH the NEGATIVE of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap IS GREATER THAN k  
                INCREMENT total_quality BY the value obtained by REMOVING the SMALLEST element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS k  
                SET min_cost TO the MINIMUM of min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
