CLASS Solution  
    FUNCTION minChanges(nums AS list OF integers, k AS integer) RETURNS integer  
        SET groups TO a new mapping from integer keys to lists of integers with default empty lists  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO the list of elements at key index MODULO k of groups  
        END FOR  

        SET freq TO a new list of length k where each element is a new mapping from integers to integers with default zero  
        SET min_size TO a new list of length k filled with positive infinity value  

        FOR index FROM zero TO k MINUS one  
            FOR each number IN element at position index of groups  
                INCREMENT value at key number of element at position index of freq BY one  
            END FOR  
            SET element at position index of min_size TO the LENGTH OF element at position index of groups  
        END FOR  

        SET dp TO a new list of length two RAISED TO THE power of ten filled with positive infinity value  
        SET element at position zero of dp TO zero  

        FOR index FROM zero TO k MINUS one  
            SET new_dp TO a new list of length two RAISED TO_THE power of ten filled with positive infinity value  
            SET min_dp TO the minimum value contained in dp  
            FOR j FROM zero TO two RAISED TO_THE power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position index of min_size  
                FOR each pair number and count IN the mapping at position index of freq  
                    SET candidate_value TO element at position j XOR number of dp PLUS element at position index of min_size MINUS count  
                    SET element at position j of new_dp TO the smaller value BETWEEN element at position j of new_dp AND candidate_value  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
