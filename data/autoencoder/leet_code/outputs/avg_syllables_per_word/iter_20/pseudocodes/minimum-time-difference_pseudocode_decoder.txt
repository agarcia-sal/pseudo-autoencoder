CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the integer conversion of the substring from position zero TO one of time split by colon  
            SET minutes TO the integer conversion of the substring from position one TO end of time split by colon  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        SET minutes_list TO a list formed by applying to_minutes to each element in timePoints  
        SORT minutes_list IN ascending order  
        SET min_diff TO infinity  
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO one thousand four hundred forty MINUS the element at the last position of minutes_list PLUS the element at position zero of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
