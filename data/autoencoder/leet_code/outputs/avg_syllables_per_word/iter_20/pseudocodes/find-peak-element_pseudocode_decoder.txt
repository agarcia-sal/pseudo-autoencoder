CLASS Solution  
    FUNCTION findPeakElement(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left IS LESS THAN right  
            SET mid TO the RESULT OF the INTEGER DIVISION of the SUM of left AND right BY two  
            IF the element at position mid of nums IS GREATER THAN the element at position mid PLUS one of nums  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
