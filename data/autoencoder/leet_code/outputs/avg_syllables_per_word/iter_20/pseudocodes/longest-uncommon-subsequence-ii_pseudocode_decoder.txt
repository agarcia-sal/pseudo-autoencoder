CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator TO the ITERATOR OF t  
            RETURN whether for EACH character IN s the character IS CONTAINED IN iterator IN SEQUENCE  
        END FUNCTION  
        
        SORT strs BY the NEGATION OF the LENGTH OF each element THEN BY the element ITSELF IN ASCENDING LEXICOGRAPHICAL ORDER  
        
        FOR each index i AND element word IN strs  
            SET unique TO TRUE  
            FOR each index j AND element other IN strs  
                IF i IS NOT EQUAL TO j AND is_subsequence OF word AND other IS TRUE  
                    SET unique TO FALSE  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique IS TRUE  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN NEGATIVE ONE  
    END FUNCTION  
END CLASS
