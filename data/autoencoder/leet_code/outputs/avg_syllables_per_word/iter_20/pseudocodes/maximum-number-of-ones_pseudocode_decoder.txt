CLASS Solution  
    FUNCTION maximumNumberOfOnes(width INTEGER PARAMETER height INTEGER PARAMETER sideLength INTEGER PARAMETER maxOnes INTEGER PARAMETER) RETURNS INTEGER  
        SET frequency_list TO a list containing zeros with the length equal to sideLength MULTIPLIED BY sideLength  
        FOR index_i FROM zero TO width MINUS one  
            FOR index_j FROM zero TO height MINUS one  
                SET position TO the remainder of index_i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of index_j DIVIDED BY sideLength  
                INCREMENT the element at position of frequency_list BY one  
            END FOR  
        END FOR  
        SORT frequency_list IN descending order  
        SET sum_of_top_frequencies TO the sum of the elements from position zero TO maxOnes MINUS one of frequency_list  
        RETURN sum_of_top_frequencies  
    END FUNCTION  
END CLASS
