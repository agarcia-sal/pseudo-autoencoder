CLASS Solution  
    FUNCTION maxRunTime WITH PARAMETERS n AND batteries  
        FUNCTION canRunFor WITH PARAMETER time  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the SMALLER VALUE BETWEEN battery AND time  
            END FOR  
            IF total IS GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO the SUM OF all elements in batteries DIVIDED BY n  
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER VALUE OF left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid) IS true  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
