CLASS TreeNode  
    FUNCTION __init__ with parameters val EQUALS zero left EQUALS none right EQUALS none  
        SET the attribute val TO parameter val  
        SET the attribute left TO parameter left  
        SET the attribute right TO parameter right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values which is a list  
    IF the parameter values IS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode constructed with element at position zero of values  
    SET i TO one  
    INITIALIZE queue as a new deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE the leftmost element from queue and SET it TO node  
        IF integer i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET the attribute left of node TO a new instance of TreeNode constructed with element at position i of values  
            APPEND the attribute left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF integer i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET the attribute right of node TO a new instance of TreeNode constructed with element at position i of values  
            APPEND the attribute right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF parameter p IS none AND parameter q IS none  
        RETURN True  
    ELSE IF parameter p IS none OR parameter q IS none  
        RETURN False  
    ELSE IF the attribute val of p IS NOT EQUAL TO the attribute val of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with attributes left of p and left of q AND the result of calling is_same_tree with attributes right of p and right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom with parameters self and root which is optional TreeNode  
        IF parameter root IS none  
            RETURN an empty list  
        END IF  
        INITIALIZE result as an empty list  
        INITIALIZE queue as a new deque collection containing root  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            INITIALIZE current_level as an empty list  
            FOR each element ignoring index FROM one TO level_size  
                REMOVE the leftmost element from queue and SET it TO node  
                APPEND the attribute val of node TO current_level  
                IF the attribute left of node IS NOT none  
                    APPEND the attribute left of node TO queue  
                END IF  
                IF the attribute right of node IS NOT none  
                    APPEND the attribute right of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN the list result in reversed order  
    END FUNCTION  
END CLASS
