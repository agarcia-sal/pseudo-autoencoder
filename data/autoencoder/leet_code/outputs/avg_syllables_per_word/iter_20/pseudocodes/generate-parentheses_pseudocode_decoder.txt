CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s defaulted to empty string left defaulted to zero right defaulted to zero)  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis AND left INCREMENTED BY one AND right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis AND left AND right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack  
        RETURN result  
    END FUNCTION  
END CLASS
