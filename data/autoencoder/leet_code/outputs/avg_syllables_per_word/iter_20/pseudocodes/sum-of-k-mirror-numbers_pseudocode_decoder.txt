CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s IS EQUAL TO THE REVERSE OF s  
        END FUNCTION  
          
        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num IS EQUAL TO zero  
                RETURN STRING REPRESENTING zero  
            END IF  
            SET digits TO AN EMPTY LIST  
            WHILE num IS GREATER THAN zero  
                APPEND THE REMAINDER OF num DIVIDED BY k TO digits  
                SET num TO THE INTEGER DIVISION OF num BY k  
            END WHILE  
            RETURN STRING FOR EACH ELEMENT IN digits IN REVERSE ORDER CONCATENATED TOGETHER  
        END FUNCTION  
          
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length IS EQUAL TO one  
                FOR i FROM one TO nine  
                    YIELD STRING REPRESENTATION OF i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF THE INTEGER DIVISION OF length MINUS one BY two  
                SET end TO ten RAISED TO THE POWER OF THE INTEGER DIVISION OF length PLUS one BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO STRING REPRESENTATION OF half  
                    IF length MODULO two IS EQUAL TO zero  
                        SET full TO half_str CONCATENATED WITH THE REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH half_str WITHOUT LAST CHARACTER IN REVERSE ORDER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
          
        SET k_mirror_numbers TO AN EMPTY LIST  
        SET length TO one  
        WHILE LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR pal IN generate_palindromes WITH PARAMETER length  
                SET num TO INTEGER CONVERSION OF pal  
                IF is_palindrome OF to_base_k WITH PARAMETERS num AND k IS TRUE  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers IS EQUAL TO n  
                        BREAK OUT OF FOR LOOP  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
          
        RETURN THE SUM OF k_mirror_numbers  
    END FUNCTION  
END CLASS
