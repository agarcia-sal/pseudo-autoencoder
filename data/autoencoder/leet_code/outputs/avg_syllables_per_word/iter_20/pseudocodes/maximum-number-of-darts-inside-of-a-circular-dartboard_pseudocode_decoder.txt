CLASS Solution  
    FUNCTION numPoints WITH PARAMETERS darts AND r  
        FUNCTION distance WITH PARAMETERS p1 AND p2  
            SET difference of first element of p1 MINUS first element of p2 TO difference_in_x  
            SET difference of second element of p1 MINUS second element of p2 TO difference_in_y  
            SET square of difference_in_x TO square_x  
            SET square of difference_in_y TO square_y  
            SET sum_of_squares TO square_x PLUS square_y  
            RETURN square root of sum_of_squares  
        END FUNCTION  
      
        FUNCTION circle_center WITH PARAMETERS p1 AND p2 AND r  
            SET d TO result of distance with arguments p1 and p2  
            IF d IS GREATER THAN two MULTIPLIED BY r  
                RETURN nothing  
            END IF  
            SET square of r TO square_r  
            SET half_of_d TO d DIVIDED BY two  
            SET square of half_of_d TO square_half_d  
            SET a TO square root of square_r MINUS square_half_d  
            SET h_first_element TO first element of p1 PLUS first element of p2 DIVIDED BY two  
            SET h_second_element TO second element of p1 PLUS second element of p2 DIVIDED BY two  
            SET dx TO difference of second element of p2 MINUS second element of p1 MULTIPLIED BY a DIVIDED BY d  
            SET dy TO difference of first element of p2 MINUS first element of p1 MULTIPLIED BY a DIVIDED BY d  
            RETURN list consisting of pair of h_first_element MINUS dx AND h_second_element PLUS dy AND pair of h_first_element PLUS dx AND h_second_element MINUS dy  
        END FUNCTION  
      
        FUNCTION is_inside WITH PARAMETERS circle AND point AND r  
            SET dist_to_point TO result of distance with arguments circle and point  
            IF dist_to_point IS LESS THAN OR EQUAL TO r  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
      
        SET n TO length of darts  
        IF n IS EQUAL TO one  
            RETURN one  
        END IF  
      
        SET max_count TO one  
      
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM i PLUS one TO n MINUS one  
                SET centers TO result of circle_center with arguments element at position i of darts AND element at position j of darts AND r  
                IF centers IS NOT nothing  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside with arguments center AND dart AND r RETURNS true  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count IS GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
      
        RETURN max_count  
    END FUNCTION  
END CLASS
