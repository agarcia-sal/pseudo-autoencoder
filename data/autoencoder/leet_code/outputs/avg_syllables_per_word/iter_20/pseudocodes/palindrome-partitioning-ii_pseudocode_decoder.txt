CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  

        SET is_palindrome TO a two dimensional list with n ROWS and n COLUMNS FILLED WITH FALSE values  

        FOR index i FROM zero TO n MINUS one  
            SET element at position i at position i of is_palindrome TO TRUE  
        END FOR  

        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start at position end of is_palindrome TO the condition that the element at position start of s EQUALS the element at position end of s  
                ELSE  
                    SET element at position start at position end of is_palindrome TO the condition that the element at position start of s EQUALS the element at position end of s AND the value at position start PLUS one at position end MINUS one of is_palindrome IS TRUE  
                END IF  
            END FOR  
        END FOR  

        SET min_cuts TO a list of length n FILLED WITH zeroes  

        FOR index i FROM zero TO n MINUS one  
            IF element at position zero at position i of is_palindrome IS TRUE  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity value  
                FOR index j FROM one TO i  
                    IF element at position j at position i of is_palindrome IS TRUE AND the sum of element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO the sum of element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
