CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n IS LESS THAN d  
            RETURN negative one  
        END IF  
  
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM OF the SUBLIST of jobDifficulty FROM position i TO the END  
            END IF  
  
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR each integer j FROM i TO n MINUS d PLUS one  
                SET current_max TO the MAXIMUM OF current_max AND the element at position j of jobDifficulty  
                SET min_difficulty TO the MINIMUM OF min_difficulty AND current_max PLUS the RESULT OF dp WITH parameters j PLUS one AND d MINUS one  
            END FOR  
  
            RETURN min_difficulty  
        END FUNCTION  
  
        RETURN the RESULT OF dp WITH parameters zero AND d  
    END FUNCTION  
END CLASS
