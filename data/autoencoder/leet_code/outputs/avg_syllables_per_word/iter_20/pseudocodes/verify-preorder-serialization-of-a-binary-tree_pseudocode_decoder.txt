CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the list obtained by splitting preorder by commas  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the symbol hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN slots EQUALS zero  
    END FUNCTION  
END CLASS
