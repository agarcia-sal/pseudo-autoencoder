CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        CALL sort points IN ORDER OF element at position two of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF element at position one of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO element at position two of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
