CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character IN expression  
            IF character EQUALS one of the characters representing true false NOT AND OR  
                APPEND character TO stk  
            ELSE IF character EQUALS the closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE the last element of stk IS one of the characters representing true or false  
                    INCREMENT count_true BY one IF the last element of stk EQUALS the character representing true  
                    INCREMENT count_false BY one IF the last element of stk EQUALS the character representing false  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET operator TO the last element of stk  
                REMOVE the last element FROM stk  
                IF operator EQUALS the character representing NOT  
                    SET character TO the character representing true IF count_false IS GREATER THAN zero OTHERWISE the character representing false  
                ELSE IF operator EQUALS the character representing AND  
                    SET character TO the character representing false IF count_false IS GREATER THAN zero OTHERWISE the character representing true  
                ELSE IF operator EQUALS the character representing OR  
                    SET character TO the character representing true IF count_true IS GREATER THAN zero OTHERWISE the character representing false  
                END IF  
                APPEND character TO stk  
            END IF  
        END FOR  
        RETURN whether the first element of stk EQUALS the character representing true  
    END FUNCTION  
END CLASS
