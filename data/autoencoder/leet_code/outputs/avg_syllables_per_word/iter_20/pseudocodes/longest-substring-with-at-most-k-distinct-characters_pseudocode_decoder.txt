CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET the element at position right of s IN char_map TO right  
            IF the SIZE OF char_map GREATER THAN k  
                SET leftmost_char TO the key IN char_map THAT HAS the MINIMUM value  
                SET left TO the value AT leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            SET max_length TO THE MAXIMUM OF max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
