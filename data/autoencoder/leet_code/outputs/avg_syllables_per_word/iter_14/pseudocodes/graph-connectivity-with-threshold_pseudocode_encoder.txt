CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold IS EQUAL TO zero  
            RETURN a list of True values with the LENGTH EQUAL TO the LENGTH OF queries  
        END IF  
        
        SET parent TO a list of integers FROM zero TO n INCLUSIVE  
        
        FUNCTION find(x)  
            IF the element at position x OF parent IS NOT EQUAL TO x  
                SET the element at position x OF parent TO find(the element at position x OF parent)  
            END IF  
            RETURN the element at position x OF parent  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX IS NOT EQUAL TO rootY  
                SET the element at position rootY OF parent TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS one TO n INCLUSIVE  
            FOR j FROM i MULTIPLIED BY two TO n INCLUSIVE IN STEPS OF i  
                CALL union WITH i AND j  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        
        FOR each pair a and b IN queries  
            APPEND to result the VALUE OF find(a) IS EQUAL TO find(b)  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
