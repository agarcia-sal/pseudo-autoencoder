CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each value IN nums  
                IF value IS LESS THAN OR EQUAL TO x THEN  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count IS GREATER THAN x THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN the RESULT OF bisecting left OVER the RANGE FROM zero TO the LENGTH OF nums WITH the KEY FUNCTION f AND TARGET true  
    END FUNCTION  
END CLASS
