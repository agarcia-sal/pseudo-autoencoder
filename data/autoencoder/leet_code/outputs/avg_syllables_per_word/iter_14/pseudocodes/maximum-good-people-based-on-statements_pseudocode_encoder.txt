CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  

        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a LIST OF n TIMES False  
            SET bad TO a LIST OF n TIMES False  

            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND OF i AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero  
                    SET the ELEMENT AT POSITION j OF good TO True  
                ELSE  
                    SET the ELEMENT AT POSITION j OF bad TO True  
                END IF  
            END FOR  

            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF the ELEMENT AT POSITION j OF good IS True  
                    FOR k FROM zero TO n MINUS one  
                        IF (the ELEMENT AT POSITION k OF the ELEMENT AT POSITION j OF statements IS EQUAL TO zero AND the ELEMENT AT POSITION k OF good IS True)  
                           OR (the ELEMENT AT POSITION k OF the ELEMENT AT POSITION j OF statements IS EQUAL TO one AND the ELEMENT AT POSITION k OF bad IS True)  
                            SET valid TO False  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS False  
                        BREAK  
                    END IF  
                END IF  
            END FOR  

            IF valid IS True  
                SET max_good TO the GREATER OF max_good AND the SUM OF the BOOLEAN ELEMENTS OF good CONVERTED TO INTEGERS  
            END IF  
        END FOR  

        RETURN max_good  
    END FUNCTION  
END CLASS
