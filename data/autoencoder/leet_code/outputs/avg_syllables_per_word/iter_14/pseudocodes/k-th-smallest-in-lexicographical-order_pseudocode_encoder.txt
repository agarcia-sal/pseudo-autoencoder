CLASS Solution  
    FUNCTION findKthNumber WITH PARAMETERS n AND k  
        FUNCTION count_steps WITH PARAMETERS prefix AND n  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix IS LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS THE MINIMUM OF the DIFFERENCE OF n MINUS prefix PLUS one AND the DIFFERENCE OF next_prefix MINUS prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        DECREMENT k BY one  
        WHILE k IS GREATER THAN zero  
            SET steps TO count_steps WITH ARGUMENTS current AND n  
            IF steps IS LESS THAN OR EQUAL TO k  
                INCREMENT current BY one  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                DECREMENT k BY one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
