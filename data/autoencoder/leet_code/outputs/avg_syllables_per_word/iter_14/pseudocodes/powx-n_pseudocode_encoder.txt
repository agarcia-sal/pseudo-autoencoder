CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x AS float AND n AS integer  
        FUNCTION qpow WITH PARAMETERS a AS float AND n AS integer  
            SET ans TO one  
            WHILE n IS GREATER THAN zero  
                IF the LEAST SIGNIFICANT BIT OF n IS EQUAL TO one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n INTEGER DIVIDED BY two USING RIGHT BIT SHIFT  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH ARGUMENTS x AND n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH ARGUMENTS x AND NEGATIVE n  
        END IF  
    END FUNCTION  
END CLASS
