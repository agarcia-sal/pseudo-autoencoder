CLASS TreeNode  
    FUNCTION __init__ WITH parameters self val SET TO zero left SET TO none right SET TO none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE element from the left OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node left TO NEW TreeNode WITH element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node right TO NEW TreeNode WITH element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p val IS NOT EQUAL TO q val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH p left AND q left AND is_same_tree WITH p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree WITH parameters self inorder postorder  
        IF inorder IS EMPTY OR postorder IS EMPTY  
            RETURN none  
        END IF  
        SET root_val TO REMOVE last element FROM postorder  
        SET root TO NEW TreeNode WITH root_val  
        SET root_index TO POSITION OF root_val IN inorder  
        SET root right TO self buildTree WITH SUBLIST OF inorder FROM root_index PLUS one TO END AND postorder  
        SET root left TO self buildTree WITH SUBLIST OF inorder FROM START TO root_index AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
