CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO an empty list  
            SET num TO zero  
            SET sign TO plus sign  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the element at position zero OF s and REMOVE the element at position zero OF s  
                IF char IS A DIGIT  
                    SET num TO num MULTIPLIED BY ten PLUS the integer VALUE OF char  
                END IF  
                IF char IS AN OPENING PARENTHESIS  
                    SET num TO the RESULT OF helper WITH PARAMETER s  
                END IF  
                IF char IS IN THE LIST consisting of plus sign, minus sign, multiplication sign, division sign, opening parenthesis, CLOSING parenthesis OR the LENGTH OF s IS EQUAL TO zero  
                    IF sign IS EQUAL TO plus sign  
                        APPEND num TO stack  
                    ELSE IF sign IS EQUAL TO minus sign  
                        APPEND NEGATIVE num TO stack  
                    ELSE IF sign IS EQUAL TO multiplication sign  
                        SET the element at position LENGTH OF stack MINUS one OF stack TO the element at position LENGTH OF stack MINUS one OF stack MULTIPLIED BY num  
                    ELSE IF sign IS EQUAL TO division sign  
                        SET the element at position LENGTH OF stack MINUS one OF stack TO the INTEGER CONVERSION OF the element at position LENGTH OF stack MINUS one OF stack DIVIDED BY num WITH TRUNCATION TOWARDS zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char IS A CLOSING parenthesis  
                    BREAK THE LOOP  
                END IF  
            END WHILE  
            RETURN the SUM OF all elements IN stack  
        END FUNCTION  
      
        SET s TO the LIST OF CHARACTERS OF s  
        RETURN the RESULT OF helper WITH PARAMETER s  
    END FUNCTION  
END CLASS
