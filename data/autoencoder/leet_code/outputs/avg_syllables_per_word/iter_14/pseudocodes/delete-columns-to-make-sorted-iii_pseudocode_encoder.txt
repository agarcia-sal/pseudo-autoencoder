CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero OF strs  
        
        CREATE dp AS a list OF LENGTH n WITH every element SET TO one  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR every k FROM zero TO m MINUS one element at position k AT position j OF strs IS LESS THAN OR EQUAL TO element at position k AT position i OF strs IS TRUE  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM element OF dp  
        SET result TO n MINUS longest_non_decreasing_subsequence_length  
        RETURN result  
    END FUNCTION  
END CLASS
