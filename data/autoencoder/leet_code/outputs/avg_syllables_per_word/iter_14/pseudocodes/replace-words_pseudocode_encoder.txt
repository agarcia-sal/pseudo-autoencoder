CLASS Solution  
    FUNCTION replaceWords WITH PARAMETERS dictionary AND sentence  
        CALL sort dictionary BY length OF each element IN dictionary  
        
        SET words TO SPLIT sentence INTO words BY spaces  
        
        FUNCTION replace WITH PARAMETER word  
            FOR each root IN dictionary  
                IF word STARTS WITH root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        
        SET replaced_words TO EMPTY LIST  
        FOR each word IN words  
            APPEND RESULT OF replace WITH word TO replaced_words  
        END FOR  
        
        SET result_sentence TO JOIN elements OF replaced_words WITH spaces  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
