CLASS Solution  
    FUNCTION isInterleave WITH PARAMETERS s1 s2 s3  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 IS NOT EQUAL TO the LENGTH OF s3  
            RETURN FALSE  
        END IF  

        SET dp TO a list OF FALSE repeated the LENGTH OF s2 PLUS one TIMES  
        SET the element at position zero OF dp TO TRUE  

        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i IS GREATER THAN zero  
                    SET the element at position j OF dp TO the element at position j OF dp AND the element at position i MINUS one OF s1 IS EQUAL TO the element at position i PLUS j MINUS one OF s3  
                END IF  
                IF j IS GREATER THAN zero  
                    SET the element at position j OF dp TO the element at position j OF dp OR the element at position j MINUS one OF dp AND the element at position j MINUS one OF s2 IS EQUAL TO the element at position i PLUS j MINUS one OF s3  
                END IF  
            END FOR  
        END FOR  

        RETURN the element at position the LENGTH OF s2 OF dp  
    END FUNCTION  
END CLASS
