CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        FOR i FROM zero TO n MINUS one  
            WHILE the ELEMENT at position i OF nums IS GREATER THAN OR EQUAL TO one AND the ELEMENT at position i OF nums IS LESS THAN OR EQUAL TO n AND the ELEMENT at position THE ELEMENT at position i OF nums MINUS one OF nums IS NOT EQUAL TO the ELEMENT at position i OF nums  
                SET correct_index TO the ELEMENT at position i OF nums MINUS one  
                SET temporary_value TO the ELEMENT at position i OF nums  
                SET the ELEMENT at position i OF nums TO the ELEMENT at position correct_index OF nums  
                SET the ELEMENT at position correct_index OF nums TO temporary_value  
            END WHILE  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            IF the ELEMENT at position i OF nums IS NOT EQUAL TO i PLUS one  
                RETURN i PLUS one  
            END IF  
        END FOR  
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
