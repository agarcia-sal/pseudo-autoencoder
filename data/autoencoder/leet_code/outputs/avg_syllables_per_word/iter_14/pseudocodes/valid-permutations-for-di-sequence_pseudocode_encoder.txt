CLASS Solution  
    FUNCTION numPermsDISequence WITH PARAMETER s  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        CREATE dp AS a two-dimensional list OF size n PLUS one BY n PLUS one FILLED WITH zero  
        
        FOR j FROM zero TO n  
            SET element at position j OF dp AT position zero TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one OF s IS EQUAL TO the character representing INCREASE  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j OF dp AT position i MINUS one MODULO MOD  
                    SET element at position j OF dp AT position i TO prefix_sum  
                END FOR  
            ELSE IF element at position i MINUS one OF s IS EQUAL TO the character representing DECREASE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one OF dp AT position i MINUS one MODULO MOD  
                    SET element at position j OF dp AT position i TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each number IN element at position n OF dp  
            SET result TO result PLUS number MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
