CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO the largest thirty-two bits signed integer value  
        SET mask TO the value representing thirty-two bits all set to one  

        WHILE b IS NOT EQUAL TO zero  
            SET carry TO the bitwise AND of a AND b shifted left BY one position  
            SET a TO the bitwise XOR of a AND b ANDed WITH mask  
            SET b TO carry ANDed WITH mask  
        END WHILE  

        IF a IS LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
