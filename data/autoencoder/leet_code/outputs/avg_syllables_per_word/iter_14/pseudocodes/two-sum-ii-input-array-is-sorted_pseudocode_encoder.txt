CLASS Solution  
    FUNCTION twoSum(numbers list of integers, target integer) RETURNS list of integers  
        SET left TO zero  
        SET right TO the LENGTH OF numbers MINUS one  
        
        WHILE left IS LESS THAN right  
            SET current_sum TO the element at position left OF numbers PLUS the element at position right OF numbers  
            IF current_sum IS EQUAL TO target  
                RETURN a list consisting of left PLUS one AND right PLUS one  
            ELSE IF current_sum IS LESS THAN target  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
