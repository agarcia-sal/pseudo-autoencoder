CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        SET directions TO the LIST containing elements representing zero AND one then one AND zero then zero AND negative one then negative one AND zero  
        SET min_heap TO the LIST containing the TRIPLE containing zero THEN zero THEN zero  
        SET visited TO a SET containing the PAIR containing zero THEN zero  
        
        WHILE min_heap IS NOT EMPTY  
            REMOVE and OBTAIN the element with the minimum FIRST value from min_heap and ASSIGN its values respectively TO cost THEN x THEN y  
            
            IF the PAIR containing x THEN y IS EQUAL TO the PAIR containing m MINUS one THEN n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each PAIR containing dx THEN dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the PAIR containing nx THEN ny IS NOT IN visited  
                    ADD the PAIR containing nx THEN ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx OF grid THEN the element at position ny OF that element  
                    ADD the TRIPLE containing new_cost THEN nx THEN ny TO min_heap preserving min_heap order by the FIRST element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
