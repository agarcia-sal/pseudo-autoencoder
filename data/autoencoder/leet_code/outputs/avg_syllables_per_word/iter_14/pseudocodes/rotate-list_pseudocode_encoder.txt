CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero next SET TO None  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values EQUAL TO None OR the LENGTH OF values EQUAL TO zero  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH PARAMETER element at position zero OF values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO THE END  
        SET node TO NEW ListNode WITH PARAMETER val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUAL TO None AND p2 EQUAL TO None  
        RETURN True  
    END IF  
    IF p1 EQUAL TO None OR p2 EQUAL TO None  
        RETURN False  
    END IF  
    RETURN (p1.val EQUAL TO p2.val) AND is_same_list WITH PARAMETERS p1.next p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight WITH PARAMETERS self head k  
        IF head EQUAL TO None OR head.next EQUAL TO None OR k EQUAL TO zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE tail.next IS NOT None  
            SET tail TO tail.next  
            INCREMENT length BY one  
        END WHILE  
        SET tail.next TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each index FROM zero TO steps_to_new_tail  
            SET new_tail TO new_tail.next  
        END FOR  
        SET new_head TO new_tail.next  
        SET new_tail.next TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
