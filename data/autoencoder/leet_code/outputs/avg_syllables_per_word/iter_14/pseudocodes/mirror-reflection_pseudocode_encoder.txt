CLASS Solution  
    FUNCTION mirrorReflection WITH p AND q  
        SET lcm TO p MULTIPLIED BY q DIVIDED BY gcd OF p AND q  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        IF vertical_reflections MODULO two EQUAL TO zero  
            RETURN zero  
        ELSE IF horizontal_reflections MODULO two EQUAL TO one  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
