CLASS Solution  
    FUNCTION computeArea WITH PARAMETERS ax1 ay1 ax2 ay2 bx1 by1 bx2 by2  
        SET area1 TO the DIFFERENCE BETWEEN ax2 AND ax1 MULTIPLIED BY the DIFFERENCE BETWEEN ay2 AND ay1  
        SET area2 TO the DIFFERENCE BETWEEN bx2 AND bx1 MULTIPLIED BY the DIFFERENCE BETWEEN by2 AND by1  
        
        IF the MINIMUM OF ax2 AND bx2 IS GREATER THAN the MAXIMUM OF ax1 AND bx1 THEN  
            SET overlap_width TO the MINIMUM OF ax2 AND bx2 MINUS the MAXIMUM OF ax1 AND bx1  
        ELSE  
            SET overlap_width TO zero  
        END IF  
        
        IF the MINIMUM OF ay2 AND by2 IS GREATER THAN the MAXIMUM OF ay1 AND by1 THEN  
            SET overlap_height TO the MINIMUM OF ay2 AND by2 MINUS the MAXIMUM OF ay1 AND by1  
        ELSE  
            SET overlap_height TO zero  
        END IF  
        
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
