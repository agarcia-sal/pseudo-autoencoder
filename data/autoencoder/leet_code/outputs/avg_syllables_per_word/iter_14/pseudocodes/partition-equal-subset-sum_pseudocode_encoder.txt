CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF all elements IN nums  
        
        IF total_sum MODULO two IS NOT EQUAL TO zero  
            RETURN false  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        
        INITIALIZE dp AS a list OF boolean values ALL set TO false WITH the LENGTH OF target PLUS one  
        SET the element at position zero OF dp TO true  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO dp at position i OR dp at position i MINUS num  
            END FOR  
        END FOR  
        
        RETURN dp at position target  
    END FUNCTION  
END CLASS
