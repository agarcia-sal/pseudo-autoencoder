CLASS Solution  
    FUNCTION reverseWords(s)  
        REVERSE the ENTIRE list s  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF the element at position end OF s IS EQUAL TO a single space character  
                REVERSE the sublist FROM position start TO position one LESS THAN end OF s  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        REVERSE the sublist FROM position start TO the END OF s  
    END FUNCTION  
END CLASS
