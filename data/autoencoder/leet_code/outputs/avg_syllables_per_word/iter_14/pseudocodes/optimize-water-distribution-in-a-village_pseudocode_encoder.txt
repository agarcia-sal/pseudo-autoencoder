CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list OF zero FOLLOWED BY index PLUS one FOLLOWED BY element at position index OF wells TO pipes  
        END FOR  
        SORT pipes IN ORDER OF element at position two OF each pipe IN ASCENDING ORDER  
        SET parent TO list OF numbers FROM zero TO n INCLUSIVE  
        FUNCTION find(x)  
            IF element at position x OF parent IS NOT EQUAL TO x  
                SET element at position x OF parent TO find(element at position x OF parent)  
            END IF  
            RETURN element at position x OF parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX IS NOT EQUAL TO rootY  
                SET element at position rootX OF parent TO rootY  
                RETURN TRUE  
            END IF  
            RETURN FALSE  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each pipe IN pipes  
            SET house1 TO element at position zero OF pipe  
            SET house2 TO element at position one OF pipe  
            SET cost TO element at position two OF pipe  
            IF union(house1, house2) IS TRUE  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
