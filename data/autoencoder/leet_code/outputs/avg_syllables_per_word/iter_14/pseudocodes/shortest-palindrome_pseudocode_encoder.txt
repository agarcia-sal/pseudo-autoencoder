CLASS Solution  
    FUNCTION shortestPalindrome WITH PARAMETER s  
        IF s IS EMPTY STRING  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome WITH PARAMETER check_str  
            IF check_str IS EQUAL TO THE REVERSED ORDER OF check_str  
                RETURN TRUE  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  
        
        FOR i FROM THE LENGTH OF s DOWN TO zero INCLUSIVE  
            SET substring_to_check TO THE ELEMENTS FROM THE START UP TO POSITION i MINUS one OF s  
            IF is_palindrome WITH PARAMETER substring_to_check RETURNS TRUE  
                SET to_add TO THE ELEMENTS FROM POSITION i TO THE END OF s IN REVERSED ORDER  
                RETURN to_add CONCATENATED WITH s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
