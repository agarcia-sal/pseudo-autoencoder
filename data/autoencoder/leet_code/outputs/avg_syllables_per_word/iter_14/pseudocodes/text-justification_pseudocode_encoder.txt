CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line IS EQUAL TO one OR is_last_line IS TRUE  
                SET joined_line TO CONCATENATE all elements IN line SEPARATED BY a SINGLE SPACE  
                SET number_of_spaces_to_add TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
                SET justified_line TO joined_line CONCATENATED WITH number_of_spaces_to_add NUMBER OF SINGLE SPACES  
                RETURN justified_line  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET number_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY number_of_gaps  
                SET extra_spaces TO total_spaces MODULO number_of_gaps  
                
                FOR i FROM zero TO number_of_gaps MINUS one  
                    IF i IS LESS THAN extra_spaces  
                        APPEND spaces_between_words PLUS one NUMBER OF SINGLE SPACES TO element at position i OF line  
                    ELSE  
                        APPEND spaces_between_words NUMBER OF SINGLE SPACES TO element at position i OF line  
                    END IF  
                END FOR  
                
                SET justified_line TO THE CONCATENATION OF ALL elements IN line WITH NO SEPARATORS  
                RETURN justified_line  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line IS GREATER THAN maxWidth  
                CALL justify_line WITH line, num_of_letters, and is_last_line EQUAL TO FALSE AND APPEND RESULT TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        CALL justify_line WITH line, num_of_letters, and is_last_line EQUAL TO TRUE AND APPEND RESULT TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
