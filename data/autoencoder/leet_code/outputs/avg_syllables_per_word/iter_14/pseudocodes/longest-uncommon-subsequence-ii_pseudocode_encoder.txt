CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_of_t TO the iterator of t  
            RETURN whether every character c in s IS FOUND IN iterator_of_t IN SEQUENCE  
        END FUNCTION  
        
        SORT strs BY length OF element IN DESCENDING ORDER THEN BY the element IN LEXICOGRAPHICAL ORDER ASCENDING  
        
        FOR index_i FROM zero TO the LENGTH OF strs MINUS one  
            SET word TO element at position index_i OF strs  
            SET unique TO TRUE  
            FOR index_j FROM zero TO the LENGTH OF strs MINUS one  
                IF index_i IS NOT EQUAL TO index_j AND is_subsequence(word, element at position index_j OF strs) IS TRUE  
                    SET unique TO FALSE  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique IS TRUE  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
