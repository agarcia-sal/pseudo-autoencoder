CLASS Solution  
    FUNCTION frogPosition WITH PARAMETERS n edges t target  
        IF n EQUAL TO one  
            RETURN one point zero  
        END IF  
        
        SET graph TO a default dictionary that generates an empty list for new keys  
        FOR each pair u and v IN edges  
            APPEND v TO the list at key u IN graph  
            APPEND u TO the list at key v IN graph  
        END FOR  
        
        SET queue TO a double ended queue containing a tuple of one current node one point zero current probability and zero current time  
        SET visited TO a set containing the integer one  
        
        WHILE the queue is not empty  
            REMOVE and RETURN the element at the left end of queue INTO current_node current_probability current_time  
            
            IF current_time EQUAL TO t OR current_node NOT EQUAL TO one AND the length of the list at key current_node IN graph EQUAL TO one  
                IF current_node EQUAL TO target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration of the while loop  
                END IF  
            END IF  
            
            SET possible_moves TO the length of the list at key current_node IN graph MINUS one IF current_node NOT EQUAL TO one OTHERWISE zero  
            
            FOR each neighbor IN the list at key current_node IN graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
