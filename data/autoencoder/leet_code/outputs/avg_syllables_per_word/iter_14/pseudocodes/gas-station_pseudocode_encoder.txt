CLASS Solution  
    FUNCTION canCompleteCircuit WITH gas AND cost  
        IF the SUM OF gas IS LESS THAN the SUM OF cost  
            RETURN negative one  
        END IF  
        
        SET start TO zero  
        SET total_tank TO zero  
        
        FOR index FROM zero TO the LENGTH OF gas MINUS one  
            SET total_tank TO total_tank PLUS the element at position index OF gas MINUS the element at position index OF cost  
            
            IF total_tank IS LESS THAN zero  
                SET start TO index PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        
        RETURN start  
    END FUNCTION  
END CLASS
