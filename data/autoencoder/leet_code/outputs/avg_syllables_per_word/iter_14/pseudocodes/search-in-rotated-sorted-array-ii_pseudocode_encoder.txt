CLASS Solution  
    FUNCTION search(nums, target)  
        IF nums IS EQUAL TO empty  
            RETURN false  
        END IF  
          
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
          
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS left MINUS right DIVIDED BY two  // left PLUS the QUOTIENT of left MINUS right DIVIDED BY two  
              
            IF element at position mid OF nums IS EQUAL TO target  
                RETURN true  
            END IF  
              
            IF element at position left OF nums IS EQUAL TO element at position mid OF nums AND element at position mid OF nums IS EQUAL TO element at position right OF nums  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF element at position left OF nums IS LESS THAN OR EQUAL TO element at position mid OF nums  
                IF element at position left OF nums IS LESS THAN OR EQUAL TO target AND target IS LESS THAN element at position mid OF nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF element at position mid OF nums IS LESS THAN target AND target IS LESS THAN OR EQUAL TO element at position right OF nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
          
        RETURN false  
    END FUNCTION  
END CLASS
