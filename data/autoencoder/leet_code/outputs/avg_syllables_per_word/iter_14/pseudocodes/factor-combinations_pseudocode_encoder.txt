CLASS Solution  
    FUNCTION getFactors WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS start AND target  
            FOR i FROM start TO the INTEGER VALUE OF the SQUARE ROOT OF target PLUS one  
                IF target MODULO i IS EQUAL TO zero  
                    APPEND a NEW LIST CONSISTING OF path FOLLOWED BY i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the LAST ELEMENT FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an EMPTY LIST  
        SET path TO an EMPTY LIST  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
