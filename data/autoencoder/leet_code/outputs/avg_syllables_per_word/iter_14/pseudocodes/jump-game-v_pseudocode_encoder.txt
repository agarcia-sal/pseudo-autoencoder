CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x IS LESS THAN THE LENGTH OF arr  
                    IF element at position i PLUS x OF arr IS GREATER THAN OR EQUAL TO element at position i OF arr  
                        BREAK  
                    END IF  
                    SET max_jumps TO THE GREATER VALUE BETWEEN max_jumps AND one PLUS dp(i PLUS x)  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO zero  
                    IF element at position i MINUS x OF arr IS GREATER THAN OR EQUAL TO element at position i OF arr  
                        BREAK  
                    END IF  
                    SET max_jumps TO THE GREATER VALUE BETWEEN max_jumps AND one PLUS dp(i MINUS x)  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET maximum_result TO zero  
        FOR i FROM zero TO THE LENGTH OF arr MINUS one  
            SET current_value TO dp(i)  
            IF current_value IS GREATER THAN maximum_result  
                SET maximum_result TO current_value  
            END IF  
        END FOR  
        RETURN maximum_result  
    END FUNCTION  
END CLASS
