CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR the PAIR of x AND y IS IN visited OR element at position x OF grid AT position y EQUAL TO zero  
                RETURN  
            END IF  
            ADD the PAIR of x AND y TO visited  
            FOR each PAIR of dx AND dy IN the LIST of PAIRS consisting of negative one AND zero, one AND zero, zero AND negative one, zero AND one  
                CALL dfs WITH PARAMETERS x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an EMPTY SET  
            SET land_cells TO an EMPTY LIST  
            FOR i FROM zero TO the LENGTH OF grid MINUS one  
                FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                    IF element at position i OF grid AT position j EQUAL TO one  
                        APPEND the PAIR of i AND j TO land_cells  
                    END IF  
                END FOR  
            END FOR  
            IF land_cells IS EMPTY  
                RETURN TRUE  
            END IF  
            CALL dfs WITH PARAMETERS the FIRST element at position zero OF land_cells, the FIRST element at position one OF land_cells, visited  
            RETURN the LENGTH OF visited EQUAL TO the LENGTH OF land_cells  
        END FUNCTION  

        IF is_connected() EQUAL TO FALSE  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i OF grid AT position j EQUAL TO one  
                    SET element at position i OF grid AT position j TO zero  
                    IF is_connected() EQUAL TO FALSE  
                        RETURN one  
                    END IF  
                    SET element at position i OF grid AT position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
