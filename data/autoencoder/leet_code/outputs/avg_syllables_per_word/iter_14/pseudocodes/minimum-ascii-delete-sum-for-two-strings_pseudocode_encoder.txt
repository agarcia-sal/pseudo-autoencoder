CLASS Solution  
    FUNCTION minimumDeleteSum(s1 STRING, s2 STRING)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        SET dp TO the RESULT OF auxiliary FUNCTION createTwoDimensionalArray WITH PARAMETERS m PLUS one AND n PLUS one OF zero  
        
        FOR j FROM one TO n  
            SET dp AT zero AT j TO dp AT zero AT j MINUS one PLUS the ASCII CODE OF the CHARACTER AT POSITION j MINUS one OF s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET dp AT i AT zero TO dp AT i MINUS one AT zero PLUS the ASCII CODE OF the CHARACTER AT POSITION i MINUS one OF s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the CHARACTER AT POSITION i MINUS one OF s1 IS EQUAL TO the CHARACTER AT POSITION j MINUS one OF s2  
                    SET dp AT i AT j TO dp AT i MINUS one AT j MINUS one  
                ELSE  
                    SET dp AT i AT j TO the MINIMUM OF dp AT i MINUS one AT j PLUS the ASCII CODE OF the CHARACTER AT POSITION i MINUS one OF s1 AND dp AT i AT j MINUS one PLUS the ASCII CODE OF the CHARACTER AT POSITION j MINUS one OF s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp AT m AT n  
    END FUNCTION  
  
    FUNCTION createTwoDimensionalArray(rows INTEGER, columns INTEGER, initialValue INTEGER)  
        DESCRIBE this FUNCTION AS PRODUCING a two dimensional ARRAY WITH rows NUMBER OF ROWS AND columns NUMBER OF COLUMNS WHERE EACH ELEMENT IS initialValue  
    END FUNCTION  
END CLASS
