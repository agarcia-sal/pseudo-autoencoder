CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO the LIST OF each word IN words TRANSFORMED BY reversing all characters OF word THEN SORTED IN ASCENDING ORDER  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF index PLUS one EQUAL TO the LENGTH OF reversed_words OR the STRING starting WITH the element at position index PLUS one OF reversed_words IS NOT EQUAL TO the element at position index OF reversed_words  
                INCREMENT total_length BY the LENGTH OF the element at position index OF reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
