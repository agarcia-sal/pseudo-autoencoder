CLASS Solution  
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean  
        IMPORT the counter data structure to COUNT elements  
  
        SET length_of_s1 TO the LENGTH OF s1  
        SET length_of_s2 TO the LENGTH OF s2  
  
        IF length_of_s1 IS GREATER THAN length_of_s2  
            RETURN false  
        END IF  
  
        SET count_s1 TO the COUNTER OF characters IN s1  
        SET count_s2 TO the COUNTER OF characters IN the SUBSTRING FROM zero TO length_of_s1 OF s2  
  
        IF count_s1 IS EQUAL TO count_s2  
            RETURN true  
        END IF  
  
        FOR index FROM length_of_s1 TO length_of_s2 MINUS one  
            INCREMENT the count OF the character AT position index OF s2 IN count_s2 BY one  
  
            DECREMENT the count OF the character AT position index MINUS length_of_s1 OF s2 IN count_s2 BY one  
  
            IF the count OF the character AT position index MINUS length_of_s1 OF s2 IN count_s2 IS EQUAL TO zero  
                REMOVE the character AT position index MINUS length_of_s1 OF s2 FROM count_s2  
            END IF  
  
            IF count_s1 IS EQUAL TO count_s2  
                RETURN true  
            END IF  
        END FOR  
  
        RETURN false  
    END FUNCTION  
END CLASS
