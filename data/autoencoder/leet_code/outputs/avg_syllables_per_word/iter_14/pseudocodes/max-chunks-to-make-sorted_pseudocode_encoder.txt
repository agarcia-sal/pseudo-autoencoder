CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO zero  
        SET chunks TO zero  
        
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            IF max_so_far GREATER THAN OR EQUAL TO element at position index OF arr  
                SET max_so_far TO max_so_far  
            ELSE IF element at position index OF arr GREATER THAN max_so_far  
                SET max_so_far TO element at position index OF arr  
            END IF  
            
            IF max_so_far EQUAL TO index  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
