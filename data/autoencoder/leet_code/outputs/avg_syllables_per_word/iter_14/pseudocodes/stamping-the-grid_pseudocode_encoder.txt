CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        
        FUNCTION prefix_sum(matrix)  
            INITIALIZE p AS a LIST of zeroes with LENGTH m PLUS one, each element being a LIST of zeroes with LENGTH n PLUS one  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET element at position r OF p AT position c TO the element at position r MINUS one OF matrix AT position c MINUS one PLUS element at position r MINUS one OF p AT position c PLUS element at position r OF p AT position c MINUS one MINUS element at position r MINUS one OF p AT position c MINUS one  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            RETURN element at position r2 PLUS one OF p AT position c2 PLUS one MINUS element at position r2 PLUS one OF p AT position c1 MINUS element at position r1 OF p AT position c2 PLUS one PLUS element at position r1 OF p AT position c1  
        END FUNCTION  
        
        SET p_grid TO prefix_sum(grid)  
        
        INITIALIZE p_stamps AS a LIST of zeroes with LENGTH m PLUS one, each element being a LIST of zeroes with LENGTH n PLUS one  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r OF grid AT position c IS EQUAL TO zero THEN  
                    IF r PLUS stampHeight IS LESS THAN OR EQUAL TO m AND c PLUS stampWidth IS LESS THAN OR EQUAL TO n THEN  
                        IF submatrix_sum(p_grid, r, c, r PLUS stampHeight MINUS one, c PLUS stampWidth MINUS one) IS EQUAL TO zero THEN  
                            INCREMENT element at position r OF p_stamps AT position c BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO prefix_sum(p_stamps)  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r OF grid AT position c IS EQUAL TO zero THEN  
                    IF submatrix_sum(p_stamps, the GREATER OF zero AND r MINUS stampHeight PLUS one, the GREATER OF zero AND c MINUS stampWidth PLUS one, r, c) IS EQUAL TO zero THEN  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
