CLASS Solution  
    FUNCTION updateMatrix WITH PARAMETER mat  
        IF mat IS EMPTY OR element at position zero OF mat IS EMPTY  
            RETURN mat  
        END IF  

        SET rows TO the LENGTH OF mat  
        SET cols TO the LENGTH OF element at position zero OF mat  
        SET dist TO A TWO-DIMENSIONAL LIST OF SIZE rows BY cols WHERE EACH ELEMENT IS SET TO infinity  
        SET queue TO AN EMPTY DOUBLE ENDED QUEUE  

        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r OF mat AT position c IS EQUAL TO zero  
                    SET element at position r OF dist AT position c TO zero  
                    ADD the TUPLE CONSISTING OF r AND c TO THE END OF queue  
                END IF  
            END FOR  
        END FOR  

        SET directions TO THE LIST OF TUPLES CONSISTING OF minus one AND zero, one AND zero, zero AND minus one, zero AND one  

        WHILE queue IS NOT EMPTY  
            REMOVE THE FIRST ELEMENT OF queue AND ASSIGN TO r AND c  
            FOR EACH dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr OF dist AT position nc IS GREATER THAN element at position r OF dist AT position c PLUS one  
                    SET element at position nr OF dist AT position nc TO element at position r OF dist AT position c PLUS one  
                    ADD THE TUPLE CONSISTING OF nr AND nc TO THE END OF queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN dist  
    END FUNCTION  
END CLASS
