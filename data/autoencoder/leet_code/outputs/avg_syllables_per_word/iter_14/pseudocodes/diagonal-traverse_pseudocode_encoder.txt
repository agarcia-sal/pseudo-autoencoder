CLASS Solution  
    FUNCTION findDiagonalOrder(mat)  
        IF mat IS EQUAL TO nothing OR the first element at position zero OF mat IS EQUAL TO nothing  
            RETURN empty list  
        END IF  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the first element at position zero OF mat  
        SET result TO empty list  
        SET r TO zero  
        SET c TO zero  
        SET direction TO one  # one REPRESENTS up-right AND negative one REPRESENTS down-left  
        
        FOR each step FROM zero TO the PRODUCT OF m AND n MINUS one  
            APPEND the element at position r OF mat THEN element at position c TO result  
            IF direction IS EQUAL TO one  
                SET new_r TO r MINUS one  
                SET new_c TO c PLUS one  
            ELSE  
                SET new_r TO r PLUS one  
                SET new_c TO c MINUS one  
            END IF  
            
            IF new_r IS LESS THAN zero OR new_r IS EQUAL TO m OR new_c IS LESS THAN zero OR new_c IS EQUAL TO n  
                IF direction IS EQUAL TO one  
                    IF new_c IS EQUAL TO n  
                        SET new_r TO r PLUS one  
                        SET new_c TO c  
                    ELSE  
                        SET new_r TO zero  
                        SET new_c TO c PLUS one  
                    END IF  
                ELSE  
                    IF new_r IS EQUAL TO m  
                        SET new_r TO r  
                        SET new_c TO c PLUS one  
                    ELSE  
                        SET new_r TO r PLUS one  
                        SET new_c TO zero  
                    END IF  
                END IF  
                SET direction TO direction MULTIPLIED BY negative one  
            END IF  
            SET r TO new_r  
            SET c TO new_c  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
