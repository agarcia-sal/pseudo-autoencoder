CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num IS EQUAL TO one  
            RETURN one  
        END IF  

        SET factors TO an empty list  
        FOR digit FROM nine DOWN TO two  
            WHILE num MODULO digit IS EQUAL TO zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  

        IF num IS NOT EQUAL TO one  
            RETURN zero  
        END IF  

        SORT factors IN ascending order  
        SET result TO an integer formed by concatenating each element AS string IN factors in order  

        IF result IS GREATER THAN two RAISED TO THE THIRTY FIRST MINUS one  
            RETURN zero  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
