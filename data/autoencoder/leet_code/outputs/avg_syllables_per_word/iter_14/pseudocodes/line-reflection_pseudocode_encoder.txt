CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the SET OF the RESULT OF converting each point IN points TO a TUPLE  
        
        SET min_x TO the SMALLEST value OF the element at position zero OF each point IN unique_points  
        SET max_x TO the LARGEST value OF the element at position zero OF each point IN unique_points  
        
        SET line_of_reflection TO the SUM of min_x AND max_x DIVIDED BY two  
        
        FOR each point IN unique_points  
            SET x TO the element at position zero OF point  
            SET y TO the element at position one OF point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            
            IF the TUPLE consisting of reflected_x AND y IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
