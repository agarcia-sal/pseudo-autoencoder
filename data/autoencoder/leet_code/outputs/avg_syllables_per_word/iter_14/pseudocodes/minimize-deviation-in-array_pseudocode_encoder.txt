CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO an empty list  
        SET min_value TO a value representing positive infinity  
        FOR each num IN nums  
            IF the remainder WHEN num IS DIVIDED BY two IS EQUAL TO one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value IS GREATER THAN num  
                SET min_value TO num  
            END IF  
            PUSH the negative of num INTO max_heap USING heap push operation  
        END FOR  

        SET min_deviation TO a value representing positive infinity  

        WHILE the remainder WHEN the element at position zero OF max_heap IS DIVIDED BY two IS EQUAL TO zero  
            POP the maximum value FROM max_heap USING heap pop operation AND MULTIPLY BY negative one TO GET max_value  
            IF min_deviation IS GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two  
            IF min_value IS GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            PUSH the negative of max_value INTO max_heap USING heap push operation  
        END WHILE  

        IF min_deviation IS GREATER THAN negative of the element at position zero OF max_heap MINUS min_value  
            SET min_deviation TO negative of the element at position zero OF max_heap MINUS min_value  
        END IF  

        RETURN min_deviation  
    END FUNCTION  
END CLASS
