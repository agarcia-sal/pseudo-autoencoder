CLASS Solution  
    FUNCTION gameOfLife WITH PARAMETER board  
        IF board IS EMPTY OR the first element OF board IS EMPTY  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the first element OF board  
        
        SET directions TO a list containing the ordered pairs of numbers as follows:  
            first element IS number minus one and number minus one  
            second element IS number minus one and zero  
            third element IS number minus one and number plus one  
            fourth element IS zero and number minus one  
            fifth element IS zero and number plus one  
            sixth element IS number plus one and number minus one  
            seventh element IS number plus one and zero  
            eighth element IS number plus one and number plus one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND (the element at position ni OF board AT position nj IS EQUAL TO one OR the element at position ni OF board AT position nj IS EQUAL TO three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF the element at position i OF board AT position j IS EQUAL TO one  
                    IF live_neighbors IS LESS THAN two OR live_neighbors IS GREATER THAN three  
                        SET the element at position i OF board AT position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors IS EQUAL TO three  
                        SET the element at position i OF board AT position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i OF board AT position j IS EQUAL TO two  
                    SET the element at position i OF board AT position j TO one  
                ELSE IF the element at position i OF board AT position j IS EQUAL TO three  
                    SET the element at position i OF board AT position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
