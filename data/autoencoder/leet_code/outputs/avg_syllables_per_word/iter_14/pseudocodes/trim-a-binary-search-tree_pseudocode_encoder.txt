CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EQUAL TO empty or values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode with value equal to element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO a new instance of TreeNode with value equal to element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO a new instance of TreeNode with value equal to element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left, q.left AND the result of is_same_tree with arguments p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS None  
            RETURN None  
        END IF  
        IF root.val IS LESS THAN low  
            RETURN the result of calling trimBST on self with arguments root.right, low, high  
        END IF  
        IF root.val IS GREATER THAN high  
            RETURN the result of calling trimBST on self with arguments root.left, low, high  
        END IF  
        SET root.left TO the result of calling trimBST on self with arguments root.left, low, high  
        SET root.right TO the result of calling trimBST on self with arguments root.right, low, high  
        RETURN root  
    END FUNCTION  
END CLASS
