CLASS Solution  
    FUNCTION numMusicPlaylists WITH PARAMETERS n goal k  
        SET MOD TO ten RAISED TO THE power OF nine PLUS one  
        
        INITIALIZE dp AS a two dimensional list WITH zero VALUES  
        dp HAS goal PLUS one ROWS AND n PLUS one COLUMNS  
        
        SET element at row zero AND column zero OF dp TO one  
        
        FOR i FROM one TO goal INCLUSIVE  
            FOR j FROM one TO n INCLUSIVE  
                ADD TO element at row i AND column j OF dp THE PRODUCT OF element at row i MINUS one AND column j MINUS one OF dp AND n MINUS j MINUS one PLUS one  
                CALCULATE element at row i AND column j OF dp MODULO MOD AND ASSIGN RESULT BACK TO element at row i AND column j OF dp  
                
                IF j GREATER THAN k  
                    ADD TO element at row i AND column j OF dp THE PRODUCT OF element at row i MINUS one AND column j OF dp AND j MINUS k  
                    CALCULATE element at row i AND column j OF dp MODULO MOD AND ASSIGN RESULT BACK TO element at row i AND column j OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at row goal AND column n OF dp  
    END FUNCTION  
END CLASS
