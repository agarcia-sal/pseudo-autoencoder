CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO the list of all positions i WHERE the element x AT position i OF corridor IS EQUAL TO the letter S  
        IF the LENGTH OF seats MODULO two IS NOT EQUAL TO zero OR the LENGTH OF seats IS LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO the element at position index OF seats MINUS the element at position index MINUS one OF seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
