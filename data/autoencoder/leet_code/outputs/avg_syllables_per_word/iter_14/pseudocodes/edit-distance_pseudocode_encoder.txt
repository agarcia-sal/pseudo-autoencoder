CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CREATE two dimensional list dp with m PLUS one rows and n PLUS one columns FILLED WITH zero  
        
        FOR i FROM zero TO m  
            SET element at position zero OF row at position i OF dp TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET element at position j OF row at position zero OF dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF word1 EQUAL TO element at position j MINUS one OF word2  
                    SET element at position j OF row at position i OF dp TO element at position j MINUS one OF row at position i MINUS one OF dp  
                ELSE  
                    SET element at position j OF row at position i OF dp TO one PLUS the MINIMUM OF element at position j OF row at position i MINUS one OF dp  
                                                                                            AND element at position j MINUS one OF row at position i OF dp  
                                                                                            AND element at position j MINUS one OF row at position i MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n OF row at position m OF dp  
    END FUNCTION  
END CLASS
