CLASS Solution  
    FUNCTION minFlips WITH PARAMETERS mat  
        FUNCTION flip WITH PARAMETERS matrix x y  
            SET directions TO a list of pairs consisting of minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one PLUS zero and zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN the number of rows m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN the number of columns n  
                    SET element at position nx OF matrix element at position ny TO one MINUS element at position nx OF matrix element at position ny  
                END IF  
            END FOR  
        END FUNCTION  
  
        FUNCTION matrix_to_tuple WITH PARAMETER matrix  
            RETURN a tuple formed by applying a tuple conversion to each row IN matrix  
        END FUNCTION  
  
        SET m TO the number of rows IN mat  
        SET n TO the number of columns IN the first row of mat  
        SET target TO a tuple formed by applying a tuple conversion to a sequence of zeros of length n FOR each row up to m  
        SET initial_state TO matrix_to_tuple APPLIED TO mat  
        SET queue TO a list containing a pair composed of initial_state AND zero  
        SET visited TO a new empty set  
        ADD initial_state TO visited  
  
        WHILE queue IS NOT empty  
            REMOVE the first pair FROM queue AND ASSIGN TO current_state AND steps  
            IF current_state IS EQUAL TO target  
                RETURN steps  
            END IF  
  
            SET current_matrix TO a new list formed by converting each row IN current_state TO a list  
  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy OF current_matrix  
                    CALL flip WITH PARAMETERS next_matrix i j  
                    SET next_state TO matrix_to_tuple APPLIED TO next_matrix  
                    IF next_state IS NOT IN visited  
                        ADD next_state TO visited  
                        APPEND the pair next_state AND steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
  
        RETURN minus one  
    END FUNCTION  
END CLASS
