CLASS Solution  
    FUNCTION threeSumSmaller(nums AS list of integers, target AS integer) RETURNS integer  
        CALL the METHOD sort ON nums TO ARRANGE elements IN ASCENDING ORDER  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS two MINUS one  
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            
            WHILE left IS LESS THAN right  
                SET current_sum TO the element at position i OF nums PLUS the element at position left OF nums PLUS the element at position right OF nums  
                
                IF current_sum IS LESS THAN target  
                    INCREMENT count BY right MINUS left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
