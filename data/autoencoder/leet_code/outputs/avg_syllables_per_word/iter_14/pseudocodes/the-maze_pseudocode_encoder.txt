CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO a list containing the pairs zero and one, one and zero, zero and negative one, negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero OF maze  
        SET visited TO an empty set  
        
        FUNCTION dfs(x, y)  
            IF the pair x and y IS IN visited THEN  
                RETURN False  
            END IF  
            IF a list containing x and y IS EQUAL TO destination THEN  
                RETURN True  
            END IF  
            ADD the pair x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND the element at position nx PLUS dx OF maze IS A list AND the element at position ny PLUS dy OF the element at position nx PLUS dx OF maze IS EQUAL TO zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs(nx, ny) IS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs(the element at position zero OF start, the element at position one OF start)  
    END FUNCTION  
END CLASS
