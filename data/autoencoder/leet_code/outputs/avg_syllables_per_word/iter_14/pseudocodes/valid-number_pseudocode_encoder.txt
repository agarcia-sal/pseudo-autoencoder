CLASS Solution  
    FUNCTION isNumber WITH parameter s  
        IMPORT regular_expression_module AS re  
        SET number_pattern TO a pattern THAT MATCHES start_of_string FOLLOWED BY an optional plus_or_minus_sign FOLLOWED BY either one OR the other OF the following: one OR more digits FOLLOWED BY an optional decimal_point FOLLOWED BY zero OR more digits OR a decimal_point FOLLOWED BY one OR more digits FOLLOWED BY an optional exponent_part WHICH IS the letter e FOLLOWED BY an optional plus_or_minus_sign FOLLOWED BY one OR more digits AND the end_of_string  
        IF the entire string s FULLY MATCHES number_pattern USING re THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
