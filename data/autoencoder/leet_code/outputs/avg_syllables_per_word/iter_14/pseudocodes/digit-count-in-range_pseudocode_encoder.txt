CLASS Solution  
    FUNCTION digitsCount WITH PARAMETERS d low high  
        FUNCTION count_digit WITH PARAMETERS n digit  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero IS TRUE  
                SET last TO the remainder of current DIVIDED BY ten  
                SET current TO the quotient of current DIVIDED BY ten  
                SET full_cycles TO current  
                IF digit EQUAL TO zero THEN  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit THEN  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUAL TO digit THEN  
                    INCREMENT count BY the remainder of n DIVIDED BY power_of_10 PLUS one  
                END IF  
                MULTIPLY power_of_10 BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  
        RETURN count_digit WITH ARGUMENTS high d MINUS count_digit WITH ARGUMENTS low MINUS one d  
    END FUNCTION  
END CLASS
