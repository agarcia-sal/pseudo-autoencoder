CLASS Solution  
    FUNCTION canCross WITH PARAMETER stones  
        IF stones IS empty OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the CONVERSION OF stones INTO a SET FOR constant time LOOKUPS  
        SET memo TO an EMPTY DICTIONARY FOR storing visited states  
        
        FUNCTION can_jump_to WITH PARAMETERS position AND jump  
            IF position IS EQUAL TO the LAST element OF stones  
                RETURN True  
            END IF  
            
            IF the TUPLE OF position AND jump IS IN memo  
                RETURN the VALUE STORED IN memo FOR the TUPLE OF position AND jump  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND position PLUS next_jump IS IN stone_set  
                    IF can_jump_to WITH position PLUS next_jump AND next_jump RETURNS True  
                        SET memo AT the TUPLE OF position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo AT the TUPLE OF position AND jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to WITH the FIRST element OF stones AND zero  
    END FUNCTION  
END CLASS
