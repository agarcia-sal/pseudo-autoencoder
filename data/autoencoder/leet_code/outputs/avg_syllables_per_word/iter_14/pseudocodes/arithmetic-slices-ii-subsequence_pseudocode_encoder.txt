CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list OF empty mappings FROM difference TO integer WITH n elements  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO element at position i OF nums MINUS element at position j OF nums  
                
                IF dp[j] HAS KEY d  
                    INCREMENT the VALUE ASSOCIATED WITH d IN dp[i] BY the VALUE ASSOCIATED WITH d IN dp[j] PLUS one  
                    INCREMENT total_count BY the VALUE ASSOCIATED WITH d IN dp[j]  
                ELSE  
                    INCREMENT the VALUE ASSOCIATED WITH d IN dp[i] BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
