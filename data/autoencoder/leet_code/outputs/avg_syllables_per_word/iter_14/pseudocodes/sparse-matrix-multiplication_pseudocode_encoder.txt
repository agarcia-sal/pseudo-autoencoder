CLASS Solution  
    FUNCTION multiply WITH PARAMETERS mat1 AND mat2  
        SET the number_of_rows_of_mat1 TO the LENGTH OF mat1  
        SET the number_of_columns_of_mat1 TO the LENGTH OF element at position zero OF mat1  
        SET the number_of_rows_of_mat2 TO the LENGTH OF mat2  
        SET the number_of_columns_of_mat2 TO the LENGTH OF element at position zero OF mat2  
        
        FUNCTION initialize_result_matrix WITH PARAMETERS rows AND columns  
            SET matrix TO empty list  
            FOR each index FROM zero TO rows MINUS one  
                SET row_list TO empty list  
                FOR each index FROM zero TO columns MINUS one  
                    APPEND zero TO row_list  
                END FOR  
                APPEND row_list TO matrix  
            END FOR  
            RETURN matrix  
        END FUNCTION
        
        SET result TO CALL initialize_result_matrix WITH the number_of_rows_of_mat1 AND the number_of_columns_of_mat2  
        
        FOR each row_index FROM zero TO the number_of_rows_of_mat1 MINUS one  
            FOR each column_index FROM zero TO the number_of_columns_of_mat2 MINUS one  
                FOR each middle_index FROM zero TO the number_of_columns_of_mat1 MINUS one  
                    SET current_value TO element at position row_index OF result AT position column_index  
                    SET multiplication_value TO element at position row_index OF mat1 AT position middle_index MULTIPLIED BY element at position middle_index OF mat2 AT position column_index  
                    SET new_value TO current_value PLUS multiplication_value  
                    SET element at position row_index OF result AT position column_index TO new_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
