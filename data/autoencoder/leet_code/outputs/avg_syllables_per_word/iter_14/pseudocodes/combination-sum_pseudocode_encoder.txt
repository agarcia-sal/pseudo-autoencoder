CLASS Solution  
    FUNCTION combinationSum WITH PARAMETERS candidates AND target  
        FUNCTION backtrack WITH PARAMETERS start AND target AND path  
            IF target IS EQUAL TO zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target IS LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH PARAMETERS i AND target MINUS element at position i OF candidates AND path CONCATENATED WITH element at position i OF candidates  
            END FOR  
        END FUNCTION  
  
        SET result TO empty list  
        CALL backtrack WITH PARAMETERS zero AND target AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
