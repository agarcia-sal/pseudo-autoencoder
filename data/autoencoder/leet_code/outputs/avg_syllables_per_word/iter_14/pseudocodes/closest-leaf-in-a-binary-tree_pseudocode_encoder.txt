CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH parameters element at position zero OF values  
    SET i TO one  
    SET queue TO NEW double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH parameter element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH parameter element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf WITH PARAMETERS self, root, k  
        IF root.left IS none AND root.right IS none  
            RETURN root.val  
        END IF  

        SET graph TO A map WITH DEFAULT VALUE AS empty list  

        FUNCTION dfs WITH PARAMETERS node, parent WITH DEFAULT none  
            IF parent IS NOT none  
                APPEND parent.val TO the list ASSOCIATED WITH key node.val IN graph  
                APPEND node.val TO the list ASSOCIATED WITH key parent.val IN graph  
            END IF  
            IF node.left IS NOT none  
                CALL dfs WITH PARAMETERS node.left AND node  
            END IF  
            IF node.right IS NOT none  
                CALL dfs WITH PARAMETERS node.right AND node  
            END IF  
        END FUNCTION  

        CALL dfs WITH PARAMETERS root  

        SET queue TO A new double-ended queue INITIALIZED WITH element k  
        SET visited TO AN empty set  

        WHILE queue IS NOT empty  
            SET current TO REMOVE first element FROM queue  
            ADD current TO visited  

            IF the number of elements IN the list ASSOCIATED WITH key current IN graph IS EQUAL TO one AND current IS NOT EQUAL TO root.val  
                RETURN current  
            END IF  

            FOR each neighbor IN the list ASSOCIATED WITH key current IN graph  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
