CLASS TreeNode  
    FUNCTION __init__(self, val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS NONE OR THE LENGTH OF values EQUAL TO zero THEN  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUAL TO element at position zero OF values  
    SET i TO one  
    SET queue TO A NEW double-ended queue  
    APPEND root TO queue  
    WHILE THE LENGTH OF queue IS GREATER THAN zero  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none THEN  
            SET node.left TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none THEN  
            SET node.right TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p.val NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node IS none THEN  
                RETURN zero  
            END IF  
            RETURN one PLUS THE GREATER OF getHeight(node.left) AND getHeight(node.right)  
        END FUNCTION  
        
        SET height TO getHeight(root)  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO A LIST OF height ELEMENTS EACH BEING A LIST OF width ELEMENTS EACH SET TO empty string  
        
        FUNCTION placeNode(node, row, left, right)  
            IF node IS none THEN  
                RETURN  
            END IF  
            SET mid TO THE INTEGER RESULT OF left PLUS right DIVIDED BY two  
            SET element at position mid OF element at position row OF res TO THE STRING REPRESENTATION OF node.val  
            CALL placeNode WITH PARAMETERS node.left, row PLUS one, left, mid  
            CALL placeNode WITH PARAMETERS node.right, row PLUS one, mid, right  
        END FUNCTION  
        
        CALL placeNode WITH PARAMETERS root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
