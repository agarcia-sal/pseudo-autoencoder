CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n AS integer  
        SET MODULO TO ten RAISED TO THE POWER OF nine PLUS one  
        SET variable a TO six  
        SET variable b TO six  
        FOR each index FROM two TO n INCLUSIVE  
            SET new_a TO three MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET new_b TO two MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET a TO new_a MODULO MODULO  
            SET b TO new_b MODULO MODULO  
        END FOR  
        SET result TO a PLUS b  
        SET result TO result MODULO MODULO  
        RETURN result  
    END FUNCTION  
END CLASS
