CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        PERFORM sorting OF nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET f TO a list of length n WHERE each element IS set TO one  
        SET k TO zero  
        FOR variable i FROM zero TO n MINUS one  
            FOR variable j FROM zero TO i MINUS one  
                IF the element at position i OF nums MODULO the element at position j OF nums IS EQUAL TO zero  
                    SET the element at position i OF f TO the GREATER VALUE BETWEEN the element at position i OF f AND the element at position j OF f PLUS one  
                END IF  
            END FOR  
            IF the element at position k OF f IS LESS THAN the element at position i OF f  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO the element at position k OF f  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m IS GREATER THAN zero  
            IF the element at position k OF nums MODULO the element at position i OF nums IS EQUAL TO zero AND the element at position i OF f IS EQUAL TO m  
                APPEND the element at position i OF nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
