CLASS Solution  
    FUNCTION findLonelyPixel WITH PARAMETER picture  
        IF picture IS empty OR element at position zero OF picture IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero OF picture  
        
        CREATE a list named row_count WITH m ELEMENTS EACH SET TO zero  
        CREATE a list named col_count WITH n ELEMENTS EACH SET TO zero  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j OF element at position i OF picture IS EQUAL TO the character B  
                    INCREMENT element at position i OF row_count BY one  
                    INCREMENT element at position j OF col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixel_count TO zero  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j OF element at position i OF picture IS EQUAL TO the character B AND element at position i OF row_count IS EQUAL TO one AND element at position j OF col_count IS EQUAL TO one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
