CLASS Solution  
    FUNCTION jobScheduling(startTime LIST OF INTEGER endTime LIST OF INTEGER profit LIST OF INTEGER) RETURNS INTEGER  
        SET jobs TO SORTED LIST OBTAINED BY COMBINING startTime WITH endTime AND profit ORDERED BY end time OF EACH JOB  
        SET end_times TO EMPTY LIST  
        FOR each job IN jobs  
            APPEND end time ELEMENT OF job TO end_times  
        END FOR  
        SET dp TO LIST OF zero INTEGER ELEMENTS WITH LENGTH EQUAL TO LENGTH OF jobs PLUS one  
        FOR index FROM one TO LENGTH OF jobs  
            SET j TO POSITION FOUND BY BINARY SEARCHING end_times FOR VALUE LESS THAN OR EQUAL TO start time ELEMENT OF job AT POSITION index MINUS one  
            IF j IS NOT DEFINED  
                SET j TO zero  
            END IF  
            SET dp AT POSITION index TO THE GREATER OF dp AT POSITION index MINUS one AND SUM OF dp AT POSITION j AND profit ELEMENT OF job AT POSITION index MINUS one  
        END FOR  
        RETURN dp AT LAST POSITION  
    END FUNCTION  
END CLASS
