CLASS Solution  
    FUNCTION canFinish WITH PARAMETERS numCourses AND prerequisites  
        CREATE graph AS A DEFAULT MAPPING FROM ELEMENTS TO LISTS  
        CREATE indegree AS A LIST OF SIZE numCourses WITH ALL ELEMENTS SET TO zero  

        FOR each pair course AND prereq IN prerequisites  
            APPEND course TO THE LIST AT KEY prereq IN graph  
            INCREMENT indegree ELEMENT AT POSITION course BY one  
        END FOR  

        CREATE queue AS A DOUBLE ENDED QUEUE INITIALIZED WITH ALL course ELEMENTS FROM zero TO numCourses MINUS one WHERE indegree ELEMENT AT POSITION course IS EQUAL TO zero  

        WHILE queue IS NOT EMPTY  
            REMOVE AND ASSIGN THE FIRST ELEMENT OF queue TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN THE LIST AT KEY current_course IN graph  
                DECREMENT indegree ELEMENT AT POSITION next_course BY one  
                IF indegree ELEMENT AT POSITION next_course IS EQUAL TO zero  
                    ADD next_course TO THE END OF queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN TRUE IF numCourses IS EQUAL TO zero OTHERWISE RETURN FALSE  
    END FUNCTION  
END CLASS
