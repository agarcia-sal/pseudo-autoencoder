CLASS Solution  
    FUNCTION minCost WITH PARAMETERS n AND cuts  
        SET cuts TO sorted list RESULTING FROM the concatenation OF cuts AND zero AND n  
        
        FUNCTION dp WITH PARAMETERS left AND right  
            IF right MINUS left IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR index FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right OF cuts MINUS element at position left OF cuts PLUS dp CALLED WITH left AND index PLUS dp CALLED WITH index AND right  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp CALLED WITH zero AND the LENGTH OF cuts MINUS one  
END CLASS
