CLASS Solution  
    FUNCTION sortColors WITH PARAMETER nums  
        SET low TO zero  
        SET mid TO zero  
        SET high TO the LENGTH OF nums MINUS one  
        WHILE mid IS LESS THAN OR EQUAL TO high  
            IF element at position mid OF nums IS EQUAL TO zero  
                SET temporary TO element at position mid OF nums  
                SET element at position mid OF nums TO element at position low OF nums  
                SET element at position low OF nums TO temporary  
                INCREMENT low BY one  
                INCREMENT mid BY one  
            ELSE IF element at position mid OF nums IS EQUAL TO one  
                INCREMENT mid BY one  
            ELSE  
                SET temporary TO element at position mid OF nums  
                SET element at position mid OF nums TO element at position high OF nums  
                SET element at position high OF nums TO temporary  
                DECREMENT high BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
