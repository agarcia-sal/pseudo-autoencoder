CLASS Solution  
    FUNCTION stoneGame WITH PARAMETER piles  
        FUNCTION dfs WITH PARAMETERS i AND j  
            IF i IS GREATER THAN j  
                RETURN zero  
            END IF  
            SET left_choice TO element at position i OF piles MINUS dfs WITH PARAMETERS i PLUS one AND j  
            SET right_choice TO element at position j OF piles MINUS dfs WITH PARAMETERS i AND j MINUS one  
            IF left_choice IS GREATER THAN right_choice  
                RETURN left_choice  
            ELSE  
                RETURN right_choice  
            END IF  
        END FUNCTION  
        SET result TO dfs WITH PARAMETERS zero AND LENGTH OF piles MINUS one  
        IF result IS GREATER THAN zero  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
