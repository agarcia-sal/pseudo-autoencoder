CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 AND s2  
        IF s1 IS EQUAL TO s2  
            RETURN TRUE  
        END IF  
        IF THE CHARACTERS OF s1 SORTED ARE NOT EQUAL TO THE CHARACTERS OF s2 SORTED  
            RETURN FALSE  
        END IF  
        SET n TO THE LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF THE RESULT OF isScramble CALLED WITH THE SUBSTRING FROM zero TO i MINUS one OF s1 AND THE SUBSTRING FROM zero TO i MINUS one OF s2 IS TRUE  
            AND THE RESULT OF isScramble CALLED WITH THE SUBSTRING FROM i TO THE END OF s1 AND THE SUBSTRING FROM i TO THE END OF s2 IS TRUE  
                RETURN TRUE  
            END IF  
            IF THE RESULT OF isScramble CALLED WITH THE SUBSTRING FROM zero TO i MINUS one OF s1 AND THE SUBSTRING FROM THE POSITION LENGTH OF s2 MINUS i TO THE END OF s2 IS TRUE  
            AND THE RESULT OF isScramble CALLED WITH THE SUBSTRING FROM i TO THE END OF s1 AND THE SUBSTRING FROM zero TO THE POSITION LENGTH OF s2 MINUS i MINUS one OF s2 IS TRUE  
                RETURN TRUE  
            END IF  
        END FOR  
        RETURN FALSE  
    END FUNCTION  
END CLASS
