CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums IS LESS THAN two  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zero repeated n times  
            SET count TO a list of zero repeated ten times  
            
            FOR i FROM zero TO n MINUS one  
                SET index TO the element at position i OF arr DIVIDED BY exp  
                INCREMENT the element at position index MODULO ten OF count BY one  
            END FOR  
            
            FOR i FROM one TO nine  
                INCREMENT the element at position i OF count BY the element at position i MINUS one OF count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i IS GREATER THAN OR EQUAL TO zero  
                SET index TO the element at position i OF arr DIVIDED BY exp  
                SET the element at position element at position index MODULO ten OF count MINUS one OF output TO the element at position i OF arr  
                DECREMENT the element at position index MODULO ten OF count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR i FROM zero TO n MINUS one  
                SET the element at position i OF arr TO the element at position i OF output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element IN arr  
            SET exp TO one  
            WHILE max1 DIVIDED BY exp IS GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort WITH nums  
        
        SET max_diff TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            IF the element at position i OF nums MINUS the element at position i MINUS one OF nums IS GREATER THAN max_diff  
                SET max_diff TO the element at position i OF nums MINUS the element at position i MINUS one OF nums  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
