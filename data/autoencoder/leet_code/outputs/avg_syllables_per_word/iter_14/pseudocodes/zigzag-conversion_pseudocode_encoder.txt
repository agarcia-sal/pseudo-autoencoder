CLASS Solution  
    FUNCTION convert WITH PARAMETERS s AND numRows  
        IF numRows IS EQUAL TO one  
            RETURN s  
        END IF  
        
        SET rows TO a list containing numRows number of empty strings  
        SET current_row TO zero  
        SET going_down TO false  
        
        FOR each character IN s  
            APPEND character TO element at position current_row OF rows  
            IF current_row IS EQUAL TO zero OR current_row IS EQUAL TO numRows MINUS one  
                SET going_down TO the logical negation of going_down  
            END IF  
            IF going_down IS TRUE  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        
        SET result TO the concatenation of all elements OF rows in order  
        RETURN result  
    END FUNCTION  
END CLASS
