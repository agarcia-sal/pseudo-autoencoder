CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n IS LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        
        SET is_prime TO A LIST OF BOOLEAN TRUE VALUES WITH LENGTH n  
        SET element at position zero OF is_prime TO BOOLEAN FALSE  
        SET element at position one OF is_prime TO BOOLEAN FALSE  
        
        SET limit TO INTEGER PART OF THE SQUARE ROOT OF n PLUS one  
        
        FOR start FROM two TO limit MINUS one  
            IF element at position start OF is_prime IS BOOLEAN TRUE  
                SET multiple TO start MULTIPLIED BY start  
                WHILE multiple IS LESS THAN n  
                    SET element at position multiple OF is_prime TO BOOLEAN FALSE  
                    INCREMENT multiple BY start  
                END WHILE  
            END IF  
        END FOR  
        
        SET prime_count TO zero  
        FOR each boolean_value IN is_prime  
            IF boolean_value IS BOOLEAN TRUE  
                INCREMENT prime_count BY one  
            END IF  
        END FOR  
        
        RETURN prime_count  
    END FUNCTION  
END CLASS
