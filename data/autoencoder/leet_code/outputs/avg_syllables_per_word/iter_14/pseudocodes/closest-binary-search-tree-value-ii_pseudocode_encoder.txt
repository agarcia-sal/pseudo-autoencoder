CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left side OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node left TO a new TreeNode initialized with element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node right TO a new TreeNode initialized with element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute OF p IS NOT EQUAL TO the val attribute OF q  
        RETURN False  
    ELSE  
        RETURN the result OF calling is_same_tree WITH PARAMETERS the left attribute OF p AND the left attribute OF q ANDED WITH the result OF calling is_same_tree WITH PARAMETERS the right attribute OF p AND the right attribute OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues WITH PARAMETERS self root target k  
        FUNCTION inorder_traversal WITH PARAMETER node  
            IF node IS None  
                RETURN  
            END IF  
            CALL inorder_traversal WITH PARAMETER the left attribute OF node  
            IF the LENGTH OF result IS LESS THAN k  
                APPEND the val attribute OF node TO result  
            ELSE  
                IF the ABSOLUTE VALUE OF the val attribute OF node MINUS target IS LESS THAN the ABSOLUTE VALUE OF the element at position zero OF result MINUS target  
                    REMOVE element FROM the left side OF result  
                    APPEND the val attribute OF node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH PARAMETER the right attribute OF node  
        END FUNCTION  
        SET result TO an empty double ended queue  
        CALL inorder_traversal WITH PARAMETER root  
        RETURN a list converted FROM result  
    END FUNCTION  
END CLASS
