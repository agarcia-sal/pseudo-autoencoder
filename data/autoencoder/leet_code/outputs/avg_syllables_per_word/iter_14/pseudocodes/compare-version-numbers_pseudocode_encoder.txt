CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO the list of substrings obtained by SPLITTING version1 BY the character dot  
        SET v2_revisions TO the list of substrings obtained by SPLITTING version2 BY the character dot  
        SET max_length TO the GREATER OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions  
        FOR i FROM zero TO max_length MINUS one  
            IF i IS LESS THAN the LENGTH OF v1_revisions  
                SET v1 TO the integer VALUE OF the element at position i OF v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF i IS LESS THAN the LENGTH OF v2_revisions  
                SET v2 TO the integer VALUE OF the element at position i OF v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 IS LESS THAN v2  
                RETURN minus one  
            ELSE IF v1 IS GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
