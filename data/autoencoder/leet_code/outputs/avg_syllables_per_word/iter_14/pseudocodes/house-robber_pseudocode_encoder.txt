CLASS Solution  
    FUNCTION rob(nums)  
        IF nums IS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums IS equal to one  
            RETURN element at position zero OF nums  
        END IF  
        
        SET dp TO a list of zero values with the LENGTH OF nums elements  
        SET element at position zero OF dp TO element at position zero OF nums  
        
        IF element at position zero OF nums IS GREATER THAN OR EQUAL TO element at position one OF nums  
            SET element at position one OF dp TO element at position zero OF nums  
        ELSE  
            SET element at position one OF dp TO element at position one OF nums  
        END IF  
        
        FOR index i FROM two TO the LENGTH OF nums MINUS one  
            IF element at position i MINUS one OF dp IS GREATER THAN OR EQUAL TO the SUM OF element at position i MINUS two OF dp AND element at position i OF nums  
                SET element at position i OF dp TO element at position i MINUS one OF dp  
            ELSE  
                SET element at position i OF dp TO the SUM OF element at position i MINUS two OF dp AND element at position i OF nums  
            END IF  
        END FOR  
        
        RETURN element at the last position OF dp  
    END FUNCTION  
END CLASS
