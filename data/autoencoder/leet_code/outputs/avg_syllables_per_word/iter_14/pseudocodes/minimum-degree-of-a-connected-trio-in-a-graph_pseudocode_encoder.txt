CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO an empty mapping from integer TO a set of integers  
        FOR each integer i FROM one TO n  
            ASSIGN reference TO an empty set in graph for key i  
        END FOR  
        
        FOR each pair of integers u and v IN edges  
            ADD v TO the set associated with key u IN graph  
            ADD u TO the set associated with key v IN graph  
        END FOR  
        
        SET min_degree TO a very large positive number representing infinity  
        SET trio_found TO false  
        
        FOR each integer i FROM one TO n  
            FOR each integer j FROM i PLUS one TO n  
                FOR each integer k FROM j PLUS one TO n  
                    IF the set associated with key i IN graph CONTAINS j AND the set associated with key i IN graph CONTAINS k AND the set associated with key j IN graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the size of the set associated with key i IN graph PLUS the size of the set associated with key j IN graph PLUS the size of the set associated with key k IN graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found IS true  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
