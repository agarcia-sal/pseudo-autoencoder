CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n k maxPts  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE IF i EQUAL TO k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference LESS THAN maxPts  
                    SET numerator TO difference  
                ELSE  
                    SET numerator TO maxPts  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            ELSE  
                SET value1 TO dfs WITH PARAMETER i PLUS one  
                SET value2 TO dfs WITH PARAMETER i PLUS maxPts PLUS one  
                SET numerator2 TO value1 MINUS value2  
                SET part2 TO numerator2 DIVIDED BY maxPts  
                RETURN value1 PLUS part2  
            END IF  
        END FUNCTION  
        RETURN dfs WITH PARAMETER zero  
    END FUNCTION  
END CLASS
