CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET list_of_position_speed_pairs TO the result of pairing each element from position WITH each element from speed  
        SET sorted_pairs TO list_of_position_speed_pairs SORTED IN order from greatest TO least based on the first element of each pair  
        SET list_of_time_to_target TO an empty list  
        FOR each pair IN sorted_pairs  
            SET position_element TO the first element OF pair  
            SET speed_element TO the second element OF pair  
            SET time FOR this pair TO the difference BETWEEN target AND position_element DIVIDED BY speed_element  
            APPEND time TO list_of_time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN list_of_time_to_target  
            IF time_element IS GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
