CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the member val TO val  
        SET the member next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values IS EQUAL TO zero  
        RETURN None  
    END IF  
    SET head TO A NEW INSTANCE OF ListNode WITH THE element at position zero OF values  
    SET p TO head  
    FOR each val IN the SUBLIST OF values FROM position one TO THE END  
        SET node TO A NEW INSTANCE OF ListNode WITH val  
        SET the member next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the member val OF p1 IS EQUAL TO the member val OF p2 AND is_same_list WITH THE MEMBER next OF p1 AND THE MEMBER next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETER head  
        FUNCTION reverseList WITH PARAMETER node  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO the member next OF node  
                SET the member next OF node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  

        SET reversed_head TO reverseList WITH PARAMETER head  
        SET carry TO one  
        SET current TO reversed_head  

        WHILE current IS NOT None AND carry IS NOT zero  
            SET the member val OF current TO the member val OF current PLUS carry  
            SET carry TO the member val OF current DIVIDED BY ten  
            SET the member val OF current TO the member val OF current MODULO ten  

            IF the member next OF current IS None AND carry IS NOT zero  
                SET the member next OF current TO A NEW INSTANCE OF ListNode WITH carry  
                SET carry TO zero  
            END IF  

            SET prev TO current  
            SET current TO the member next OF current  
        END WHILE  

        RETURN reverseList WITH PARAMETER reversed_head  
    END FUNCTION  
END CLASS
