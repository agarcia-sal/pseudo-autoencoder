CLASS Solution  
    FUNCTION isOneEditDistance WITH PARAMETERS s AND t  
        IF the LENGTH OF s IS GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling the FUNCTION isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff IS GREATER THAN one  
            RETURN false  
        END IF  
        
        IF diff IS EQUAL TO zero  
            SET difference_count TO zero  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF element at position index OF s IS NOT EQUAL TO element at position index OF t  
                    INCREMENT difference_count BY one  
                END IF  
            END FOR  
            RETURN whether difference_count IS EQUAL TO one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF element at position index OF s IS NOT EQUAL TO element at position index OF t  
                    RETURN whether the SUBSTRING OF s FROM position index TO END IS EQUAL TO the SUBSTRING OF t FROM position index PLUS one TO END  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
