CLASS Solution  
    FUNCTION longestDecomposition WITH PARAMETER text  
        SET left TO zero  
        SET right TO the LENGTH OF text MINUS one  
        SET left_part TO the empty string  
        SET right_part TO the empty string  
        SET k TO zero  
        
        WHILE left IS LESS THAN right  
            SET left_part TO left_part CONCATENATED WITH element at position left OF text  
            SET right_part TO element at position right OF text CONCATENATED WITH right_part  
            
            IF left_part IS EQUAL TO right_part  
                INCREMENT k BY two  
                SET left_part TO the empty string  
                SET right_part TO the empty string  
            END IF  
            
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        
        IF left_part IS NOT EQUAL TO the empty string OR left IS EQUAL TO right  
            INCREMENT k BY one  
        END IF  
        
        RETURN k  
    END FUNCTION  
END CLASS
