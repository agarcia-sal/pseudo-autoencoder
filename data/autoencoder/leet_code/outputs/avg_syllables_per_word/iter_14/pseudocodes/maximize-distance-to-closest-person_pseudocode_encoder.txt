CLASS Solution  
    FUNCTION maxDistToClosest(seats)  
        SET max_distance TO zero  
        SET start TO zero  
        SET end TO zero  
        SET n TO the LENGTH OF seats  
        
        WHILE start IS LESS THAN n AND element at position start OF seats IS EQUAL TO zero  
            INCREMENT start BY one  
        END WHILE  
        
        SET end TO n MINUS one  
        WHILE end IS GREATER THAN OR EQUAL TO zero AND element at position end OF seats IS EQUAL TO zero  
            DECREMENT end BY one  
        END WHILE  
        
        IF max_distance IS LESS THAN start  
            SET max_distance TO start  
        END IF  
        
        IF max_distance IS LESS THAN n MINUS end MINUS one  
            SET max_distance TO n MINUS end MINUS one  
        END IF  
        
        SET current_distance TO zero  
        FOR i FROM start TO end  
            IF element at position i OF seats IS EQUAL TO zero  
                INCREMENT current_distance BY one  
                SET half_distance TO current_distance PLUS one DIVIDED BY two  
                IF max_distance IS LESS THAN half_distance  
                    SET max_distance TO half_distance  
                END IF  
            ELSE  
                SET current_distance TO zero  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
