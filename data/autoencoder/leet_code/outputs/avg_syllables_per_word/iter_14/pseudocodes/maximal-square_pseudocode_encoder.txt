CLASS Solution  
    FUNCTION maximalSquare WITH PARAMETER matrix  
        IF matrix IS EMPTY OR element at position zero OF matrix IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero OF matrix  
        SET dp TO a two-dimensional list of zero VALUES with m PLUS one rows and n PLUS one columns  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF matrix AT position j MINUS one EQUAL TO the character one  
                    SET dp AT position i AT position j TO the MINIMUM OF dp AT position i MINUS one AT position j AND dp AT position i AT position j MINUS one AND dp AT position i MINUS one AT position j MINUS one PLUS one  
                    SET max_side TO the GREATER OF max_side AND dp AT position i AT position j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
