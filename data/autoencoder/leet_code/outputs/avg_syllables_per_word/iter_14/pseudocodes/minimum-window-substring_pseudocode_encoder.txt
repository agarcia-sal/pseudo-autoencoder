CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s AND t  
        CREATE t_count AS A COLLECTION THAT COUNTS FREQUENCY OF EACH CHARACTER IN t  
        SET required TO THE NUMBER OF UNIQUE CHARACTERS IN t_count  
        
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        CREATE window_counts AS A COLLECTION THAT COUNTS FREQUENCY OF EACH CHARACTER IN THE CURRENT WINDOW  
        
        SET min_length TO A VALUE REPRESENTING INFINITY  
        SET min_start TO zero  
        
        WHILE right IS LESS THAN THE LENGTH OF s  
            INCREMENT THE COUNT OF THE CHARACTER AT POSITION right OF s IN window_counts BY one  
            
            IF THE CHARACTER AT POSITION right OF s IS PRESENT IN t_count AND THE COUNT OF THAT CHARACTER IN window_counts IS EQUAL TO THE COUNT IN t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND formed IS EQUAL TO required  
                SET char TO THE CHARACTER AT POSITION left OF s  
                
                IF THE VALUE OF right MINUS left PLUS one IS LESS THAN min_length  
                    SET min_length TO THE VALUE OF right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT THE COUNT OF char IN window_counts BY one  
                
                IF char IS PRESENT IN t_count AND THE COUNT OF char IN window_counts IS LESS THAN THE COUNT IN t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length IS EQUAL TO THE VALUE REPRESENTING INFINITY  
            RETURN AN EMPTY STRING  
        ELSE  
            RETURN THE SUBSTRING OF s FROM min_start TO min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
