CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each m IN nums  
            IF m EQUAL TO m1  
                INCREMENT n1 BY one  
            ELSE IF m EQUAL TO m2  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUAL TO zero  
                SET m1 TO m  
                SET n1 TO one  
            ELSE IF n2 EQUAL TO zero  
                SET m2 TO m  
                SET n2 TO one  
            ELSE  
                DECREMENT n1 BY one  
                DECREMENT n2 BY one  
            END IF  
        END FOR  
        SET result TO an empty list  
        FOR each candidate IN list containing m1 and m2  
            SET count_of_candidate TO the number of times candidate appears in nums  
            IF count_of_candidate GREATER THAN the LENGTH OF nums DIVIDED BY three  
                APPEND candidate TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
