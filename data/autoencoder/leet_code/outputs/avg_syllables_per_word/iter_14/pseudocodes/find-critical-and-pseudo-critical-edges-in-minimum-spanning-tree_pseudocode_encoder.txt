CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n MINUS one  
        SET self.rank TO list containing the integer one repeated n times  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u OF self.parent IS NOT EQUAL TO u  
            SET element at position u OF self.parent TO self.find(element at position u OF self.parent)  
        END IF  
        RETURN element at position u OF self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
  
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU OF self.rank IS GREATER THAN element at position rootV OF self.rank  
                SET element at position rootV OF self.parent TO rootU  
            ELSE IF element at position rootU OF self.rank IS LESS THAN element at position rootV OF self.rank  
                SET element at position rootU OF self.parent TO rootV  
            ELSE  
                SET element at position rootV OF self.parent TO rootU  
                INCREMENT element at position rootU OF self.rank BY one  
            END IF  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND edge IN edges WITH enumeration  
            APPEND i TO edge  
        END FOR  
  
        SORT edges BY the element at position two OF each edge IN ascending order  
  
        FUNCTION mst(exclude, include)  
            SET uf TO new UnionFind WITH parameter n  
            SET weight TO zero  
  
            IF include IS NOT EQUAL TO null  
                CALL uf.union WITH parameters element at position zero OF include AND element at position one OF include  
                INCREMENT weight BY element at position two OF include  
            END IF  
  
            FOR each u, v, w, idx IN edges  
                IF idx IS EQUAL TO exclude  
                    CONTINUE to next iteration  
                END IF  
                IF uf.union(u, v) IS true  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
  
            SET root TO uf.find(zero)  
  
            FOR i FROM zero TO n MINUS one  
                IF uf.find(i) IS NOT EQUAL TO root  
                    RETURN positive infinity  
                END IF  
            END FOR  
  
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO CALL mst WITH no parameters for exclude AND include  
  
        SET critical TO empty list  
        SET pseudo_critical TO empty list  
  
        FOR each u, v, w, idx IN edges  
            IF CALL mst WITH exclude equal TO idx IS GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF CALL mst WITH include equal TO the tuple of u, v, w IS EQUAL TO mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
  
        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
