CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
  
        SET dp TO an empty list  
  
        FOR each num IN nums  
            SET idx TO the position where num CAN BE INSERTED to KEEP dp sorted USING BINARY SEARCH ON dp AND num  
  
            IF idx IS EQUAL TO the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx OF dp TO num  
            END IF  
        END FOR  
  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
