CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the list obtained by splitting num USING the character plus  
            SET real TO the integer CONVERTED FROM the element at position zero OF parts  
            SET imaginary_substring TO the substring FROM position zero TO the length OF the element at position one OF parts MINUS one OF the element at position one OF parts  
            SET imaginary TO the integer CONVERTED FROM imaginary_substring  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result OF parse_complex WITH num1  
        SET real2 AND imag2 TO the result OF parse_complex WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        SET result_string TO the string REPRESENTATION OF real_part PLUS the character plus PLUS the string REPRESENTATION OF imaginary_part PLUS the character i  
        RETURN result_string  
    END FUNCTION  
END CLASS
