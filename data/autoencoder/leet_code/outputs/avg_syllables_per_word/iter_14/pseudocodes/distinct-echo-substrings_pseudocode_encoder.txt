CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty collection of unique elements  
        FOR length FROM one TO the INTEGER DIVISION OF n BY two PLUS one MINUS one  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one MINUS one  
                SET a TO the SUBSTRING FROM position start TO position start PLUS length MINUS one OF text  
                SET b TO the SUBSTRING FROM position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one OF text  
                IF a IS EQUAL TO b  
                    APPEND the CONCATENATION OF a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the NUMBER OF ELEMENTS IN substrings  
    END FUNCTION  
END CLASS
