CLASS Solution  
    FUNCTION isNStraightHand WITH PARAMETERS hand AND groupSize  
        IF the LENGTH OF hand MODULO groupSize IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        
        SET count TO the frequency mapping OF elements IN hand  
        SET unique_cards TO the ascending sorted sequence OF unique elements IN count  
        
        FOR each card IN unique_cards  
            IF the count OF card IS GREATER THAN zero  
                SET needed TO the count OF card  
                FOR index FROM zero TO groupSize MINUS one  
                    SET current_card TO card PLUS index  
                    IF the count OF current_card IS LESS THAN needed  
                        RETURN False  
                    END IF  
                    DECREMENT the count OF current_card BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
