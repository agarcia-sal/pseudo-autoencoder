CLASS Solution  
    FUNCTION maxProduct WITH PARAMETER words  
        SET word_sets TO CALL helper FUNCTION create_character_sets WITH PARAMETER words  
        SET max_product TO zero  
        FOR i FROM zero TO LENGTH OF words MINUS one  
            FOR j FROM i PLUS one TO LENGTH OF words MINUS one  
                IF intersection of element at position i OF word_sets AND element at position j OF word_sets IS empty  
                    SET product TO LENGTH OF element at position i OF words MULTIPLIED BY LENGTH OF element at position j OF words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  
  
    FUNCTION create_character_sets WITH PARAMETER words  
        INITIALIZE character_sets AS empty list  
        FOR each word IN words  
            CONVERT word TO set of characters  
            APPEND this set TO character_sets  
        END FOR  
        RETURN character_sets  
    END FUNCTION  
END CLASS
