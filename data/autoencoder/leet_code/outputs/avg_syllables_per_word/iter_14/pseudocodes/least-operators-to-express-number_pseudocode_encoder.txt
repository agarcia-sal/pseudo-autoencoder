CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target IS EQUAL TO zero  
                RETURN negative one  
            END IF  
            IF target IS LESS THAN x  
                RETURN the MINIMUM BETWEEN the EXPRESSION two MULTIPLIED BY target MINUS one AND the EXPRESSION two MULTIPLIED BY the DIFFERENCE BETWEEN x AND target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO the power power IS LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO the RESULT OF dp WITH PARAMETER the DIFFERENCE BETWEEN target AND x RAISED TO the power power PLUS power  
            IF the DIFFERENCE BETWEEN x RAISED TO the power power PLUS one AND target IS LESS THAN target  
                SET res TO the MINIMUM BETWEEN res AND the RESULT OF dp WITH PARAMETER the DIFFERENCE BETWEEN x RAISED TO the power power PLUS one AND target PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH PARAMETER target  
    END FUNCTION  
END CLASS
