CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list OF pairs consisting of minus two AND minus one, minus two AND one, minus one AND minus two, minus one AND two, one AND minus two, one AND two, two AND minus one, two AND one  
        
        CREATE a two dimensional list named dp consisting OF n rows and n columns FILLED WITH zero  
        SET element at position row of dp AT position column TO one  
        
        FOR counter FROM one TO k  
            CREATE a two dimensional list named new_dp consisting OF n rows and n columns FILLED WITH zero  
            
            FOR r FROM zero TO n MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF element at position r of dp AT position c EQUAL TO zero  
                        CONTINUE TO next iteration OF inner loop  
                    END IF  
                    
                    FOR each pair dr AND dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN n AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN n  
                            INCREMENT element at position nr of new_dp AT position nc BY element at position r of dp AT position c DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            
            ASSIGN dp TO new_dp  
        END FOR  
        
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each value IN row_list  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
        
        RETURN total_probability  
    END FUNCTION  
END CLASS
