CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO CONVERT num TO SEQUENCE OF CHARACTERS REPRESENTING EACH DIGIT  
        SET last TO CREATE MAPPING FROM EACH INTEGER DIGIT TO ITS LAST POSITION IN digits  
        
        FOR each index i AND character d IN digits  
            FOR each integer d2 FROM nine DOWN TO INTEGER VALUE OF d PLUS one  
                IF d2 EXISTS IN last AND VALUE AT last FOR d2 IS GREATER THAN i  
                    SET temporary TO element at position i OF digits  
                    SET element at position i OF digits TO element at position VALUE AT last FOR d2 OF digits  
                    SET element at position VALUE AT last FOR d2 OF digits TO temporary  
                    RETURN CONVERT SEQUENCE digits TO INTEGER  
                END IF  
            END FOR  
        END FOR  
        
        RETURN num  
    END FUNCTION  
END CLASS
