CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n INCREMENT BY one  
                SET left TO SUBSTRING FROM position zero TO d OF frag  
                SET right TO SUBSTRING FROM position d TO n OF frag  
                IF left DOES NOT START WITH character zero OR left IS EQUAL TO character zero THEN  
                    IF right DOES NOT END WITH character zero THEN  
                        IF d IS NOT EQUAL TO n THEN  
                            RETURN CONCATENATION OF left WITH character period AND right  
                        ELSE  
                            RETURN left CONCATENATED WITH right  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET s TO SUBSTRING FROM position one TO LENGTH OF s MINUS one OF s  
        
        SET result_list TO an EMPTY list  
        FOR i FROM one TO LENGTH OF s MINUS one INCREMENT BY one  
            FOR each a IN THE RESULT OF make WITH PARAMETER THE SUBSTRING FROM position zero TO i OF s  
                FOR each b IN THE RESULT OF make WITH PARAMETER THE SUBSTRING FROM position i TO LENGTH OF s OF s  
                    APPEND CONCATENATION OF character open parenthesis WITH a WITH character comma WITH character space WITH b WITH character close parenthesis TO result_list  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
