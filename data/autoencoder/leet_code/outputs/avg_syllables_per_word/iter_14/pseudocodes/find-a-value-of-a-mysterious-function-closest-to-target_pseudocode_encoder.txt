CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO infinity  
        FOR each num IN arr  
            SET temporary_set TO empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO temporary_set  
            END FOR  
            SET seen TO union of temporary_set AND set containing only num  
            FOR each val IN seen  
                SET current_difference TO absolute value of val MINUS target  
                IF current_difference LESS THAN min_diff  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
