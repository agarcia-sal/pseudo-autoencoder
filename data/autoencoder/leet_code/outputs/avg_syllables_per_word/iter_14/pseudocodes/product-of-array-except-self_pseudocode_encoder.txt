CLASS Solution  
    FUNCTION productExceptSelf(nums)  
        SET n TO the LENGTH OF nums  
        SET answer TO a list of n elements each SET TO one  
        
        SET left_product TO one  
        FOR i FROM zero TO n MINUS one  
            SET the element at position i OF answer TO left_product  
            SET left_product TO left_product MULTIPLIED BY the element at position i OF nums  
        END FOR  
        
        SET right_product TO one  
        FOR i FROM n MINUS one DOWNTO zero  
            SET the element at position i OF answer TO the element at position i OF answer MULTIPLIED BY right_product  
            SET right_product TO right_product MULTIPLIED BY the element at position i OF nums  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
