CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze, start, destination  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero OF maze  
        SET directions TO the LIST OF the TUPLES OF  
            one AND zero  
            minus one AND zero  
            zero AND one  
            zero AND minus one  
        END LIST  
        SET start TO the TUPLE OF start  
        SET destination TO the TUPLE OF destination  
        SET visited TO the EMPTY SET  
        SET pq TO the LIST CONTAINING the TUPLE OF zero AND start  
        
        WHILE pq IS NOT EMPTY  
            SET dist AND position TO heapq POP the SMALLEST ELEMENT FROM pq  
            SET x TO the FIRST ELEMENT OF position  
            SET y TO the SECOND ELEMENT OF position  
            
            IF x AND y ARE EQUAL TO the FIRST ELEMENT AND SECOND ELEMENT OF destination RESPECTIVELY  
                RETURN dist  
            END IF  
            
            IF the TUPLE OF x AND y IS IN visited  
                CONTINUE TO the NEXT ITERATION  
            END IF  
            
            ADD the TUPLE OF x AND y TO visited  
            
            FOR each dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the ELEMENT at position new_x PLUS dx OF maze AT position new_y PLUS dy IS EQUAL TO zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the TUPLE OF new_x AND new_y IS NOT IN visited  
                    INSERT the TUPLE OF dist PLUS count AND the TUPLE OF new_x AND new_y INTO pq USING heapq PUSH  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
