CLASS Solution  
    FUNCTION largestNumber WITH PARAMETER nums  
        SET nums_str TO A NEW LIST CONTAINING THE STRING CONVERSION OF EACH ELEMENT IN nums  
        
        FUNCTION compare WITH PARAMETERS x AND y  
            IF THE CONCATENATION OF x AND y IS GREATER THAN THE CONCATENATION OF y AND x  
                RETURN negative one  
            ELSE IF THE CONCATENATION OF x AND y IS LESS THAN THE CONCATENATION OF y AND x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING THE compare FUNCTION AS THE SORTING KEY  
        
        IF THE ELEMENT AT POSITION zero OF nums_str IS EQUAL TO THE STRING REPRESENTATION OF zero  
            RETURN THE STRING REPRESENTATION OF zero  
        END IF  
        
        RETURN THE RESULT OF JOINING ALL ELEMENTS OF nums_str INTO A SINGLE STRING  
    END FUNCTION  
END CLASS
