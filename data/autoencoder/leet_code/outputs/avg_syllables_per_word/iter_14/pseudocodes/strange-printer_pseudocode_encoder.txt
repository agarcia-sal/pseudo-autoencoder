CLASS Solution  
    FUNCTION strangePrinter WITH PARAMETER s  
        FUNCTION dp WITH PARAMETERS i j memo  
            IF i IS GREATER THAN j  
                RETURN zero  
            END IF  
            IF memo CONTAINS KEY i AND j  
                RETURN VALUE OF memo AT i AND j  
            END IF  
            SET result TO CALL dp WITH ARGUMENTS i j MINUS one memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF element at position k OF s IS EQUAL TO element at position j OF s  
                    SET result TO THE MINIMUM OF result AND THE SUM OF CALL dp WITH ARGUMENTS i k MINUS one memo AND CALL dp WITH ARGUMENTS k j MINUS one memo  
                END IF  
            END FOR  
            SET VALUE OF memo AT i AND j TO result  
            RETURN result  
        END FUNCTION  
        RETURN CALL dp WITH ARGUMENTS zero LENGTH OF s MINUS one AN EMPTY COLLECTION  
    END FUNCTION  
END CLASS
