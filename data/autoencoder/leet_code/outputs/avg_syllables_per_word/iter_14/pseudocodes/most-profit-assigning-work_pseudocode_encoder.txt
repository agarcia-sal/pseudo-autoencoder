CLASS Solution  
    FUNCTION maxProfitAssignment WITH PARAMETERS difficulty LIST OF integer profit LIST OF integer worker LIST OF integer  
        SET jobs TO A LIST THAT COMBINES EACH ELEMENT OF difficulty WITH THE CORRESPONDING ELEMENT OF profit INTO PAIRS  
        SORT jobs ACCORDING TO THE FIRST ELEMENT OF EACH PAIR IN ASCENDING ORDER  
        SORT worker IN ASCENDING ORDER  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO THE LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND THE FIRST ELEMENT OF THE job_index TERM OF jobs IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO THE GREATER OF best_profit AND THE SECOND ELEMENT OF THE job_index TERM OF jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
