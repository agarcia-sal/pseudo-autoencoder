CLASS Solution  
    FUNCTION minTotalDistance(grid)  
        SET rows TO an empty list  
        SET cols TO an empty list  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i OF grid MINUS one  
                IF element at position i OF grid AT position j IS EQUAL TO one  
                    APPEND i TO rows  
                    APPEND j TO cols  
                END IF  
            END FOR  
        END FOR  
        
        CALL sort ON rows  
        CALL sort ON cols  
        
        SET median_row TO element at position the LENGTH OF rows DIVIDED BY two OF rows  
        SET median_col TO element at position the LENGTH OF cols DIVIDED BY two OF cols  
        
        SET min_distance TO zero  
        FOR each row IN rows  
            IF row IS GREATER THAN median_row  
                INCREMENT min_distance BY row MINUS median_row  
            ELSE  
                INCREMENT min_distance BY median_row MINUS row  
            END IF  
        END FOR  
        FOR each col IN cols  
            IF col IS GREATER THAN median_col  
                INCREMENT min_distance BY col MINUS median_col  
            ELSE  
                INCREMENT min_distance BY median_col MINUS col  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
