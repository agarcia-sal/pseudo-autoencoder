CLASS Solution  
    FUNCTION combinationSum4 WITH parameters nums AND target  
        SET dp TO an empty list OF length target PLUS one FILLED WITH zero  
        SET the element at position zero OF dp TO one  
        FOR amount FROM one TO target  
            FOR each num IN nums  
                IF amount IS GREATER THAN OR EQUAL TO num  
                    INCREMENT the element at position amount OF dp BY the element at position amount MINUS num OF dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position target OF dp  
    END FUNCTION  
END CLASS
