CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH ARGUMENT element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY DO  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node left TO NEW TreeNode WITH ARGUMENT element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node right TO NEW TreeNode WITH ARGUMENT element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p val IS NOT EQUAL TO q val THEN  
        RETURN False  
    ELSE  
        RETURN RESULT OF is_same_tree WITH ARGUMENTS p left AND q left AND RESULT OF is_same_tree WITH ARGUMENTS p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder WITH PARAMETERS self root  
        IF root IS None THEN  
            RETURN EMPTY LIST  
        END IF  
        SET result TO EMPTY LIST  
        SET queue TO NEW double ended queue CONTAINING root  
        WHILE queue IS NOT EMPTY DO  
            SET level_size TO THE LENGTH OF queue  
            SET current_level TO EMPTY LIST  
            FOR index FROM zero TO level_size MINUS one DO  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                APPEND node val TO current_level  
                IF node left IS NOT None THEN  
                    APPEND node left TO queue  
                END IF  
                IF node right IS NOT None THEN  
                    APPEND node right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
