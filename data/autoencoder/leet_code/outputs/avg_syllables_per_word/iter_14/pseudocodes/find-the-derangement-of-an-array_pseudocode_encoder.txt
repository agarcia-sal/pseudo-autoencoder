CLASS Solution  
    FUNCTION findDerangement WITH PARAMETER n  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        
        IF n EQUAL TO one  
            RETURN zero  
        END IF  
        
        IF n EQUAL TO two  
            RETURN one  
        END IF  
        
        SET prev2 TO one  
        SET prev1 TO zero  
        
        FOR i FROM three TO n  
            SET current TO (i MINUS one) MULTIPLIED BY (prev1 PLUS prev2) MODULO MOD  
            SET prev2 TO prev1  
            SET prev1 TO current  
        END FOR  
        
        RETURN prev1  
    END FUNCTION  
END CLASS
