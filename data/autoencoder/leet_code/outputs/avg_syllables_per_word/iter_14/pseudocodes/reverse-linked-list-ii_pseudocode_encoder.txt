CLASS ListNode  
    FUNCTION __init__(val EQUAL TO zero COMMA next EQUAL TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN the SUBLIST OF values FROM position one TO THE END  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 COMMA p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN boolean RESULT OF (p1.val EQUAL TO p2.val) AND is_same_list(p1.next COMMA p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween(head COMMA left COMMA right)  
        IF head IS none OR left EQUAL TO right  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode INITIALIZED WITH zero  
        SET dummy.next TO head  
        SET prev TO dummy  
        FOR index FROM one TO left MINUS one  
            SET prev TO prev.next  
        END FOR  
        SET current TO prev.next  
        FOR index FROM one TO right MINUS left  
            SET next_node TO current.next  
            SET current.next TO next_node.next  
            SET next_node.next TO prev.next  
            SET prev.next TO next_node  
        END FOR  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
