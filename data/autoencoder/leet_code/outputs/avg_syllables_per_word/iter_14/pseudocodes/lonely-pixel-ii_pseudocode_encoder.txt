CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS empty OR element at position zero OF picture IS empty  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero OF picture  
        SET row_count TO a new list OF size m WITH all elements SET TO zero  
        SET col_count TO a new list OF size n WITH all elements SET TO zero  
        SET row_patterns TO a new dictionary WITH default value zero  

        FOR r FROM zero TO m MINUS one  
            SET pattern TO concatenation OF all elements at each position OF element at position r OF picture IN their order  
            INCREMENT the value at key pattern IN row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture IS EQUAL TO black pixel  
                    INCREMENT element at position r OF row_count BY one  
                    INCREMENT element at position c OF col_count BY one  
                END IF  
            END FOR  
        END FOR  

        SET lonely_pixels TO zero  

        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture IS EQUAL TO black pixel  
                    SET pattern TO concatenation OF all elements at each position OF element at position r OF picture IN their order  
                    IF element at position r OF row_count IS EQUAL TO target  
                       AND element at position c OF col_count IS EQUAL TO target  
                       AND the value at key pattern IN row_patterns IS EQUAL TO target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
