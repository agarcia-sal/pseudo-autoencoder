CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO false  
        SET result TO an empty list  
        SET current_line TO an empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF line  
                IF in_block_comment IS TRUE  
                    IF the element at position i AND the element at position i PLUS one OF line ARE EQUAL TO the sequence indicating the end of a block comment  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the element at position i AND the element at position i PLUS one OF line ARE EQUAL TO the sequence indicating the start of a single line comment  
                        BREAK the WHILE loop  
                    ELSE IF the element at position i AND the element at position i PLUS one OF line ARE EQUAL TO the sequence indicating the start of a block comment  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i OF line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line IS NOT EMPTY AND in_block_comment IS FALSE  
                APPEND the concatenation of all elements IN current_line TO result  
                SET current_line TO an empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
