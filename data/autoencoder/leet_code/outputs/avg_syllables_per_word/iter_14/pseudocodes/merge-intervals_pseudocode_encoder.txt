CLASS Solution  
    FUNCTION merge WITH parameter intervals AS a list of lists of integers  
        CALL sort_intervals BASED ON the first element of each interval IN intervals  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the second element at last position OF merged IS LESS THAN the first element of interval  
                APPEND interval TO merged  
            ELSE  
                SET the second element at last position OF merged TO the maximum value BETWEEN the second element at last position OF merged AND the second element OF interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
