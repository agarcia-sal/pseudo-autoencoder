CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO a new empty list  
        FOR each number FROM one TO n  
            APPEND the string representation of number TO teams  
        END FOR  
        
        WHILE n IS GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET team_at_position i IN teams TO a string formed by an opening parenthesis FOLLOWED BY the element at position i OF teams FOLLOWED BY a comma FOLLOWED BY the element at position n MINUS one MINUS i OF teams FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        
        RETURN the element at position zero OF teams  
    END FUNCTION  
END CLASS
