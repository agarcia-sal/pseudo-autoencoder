CLASS Solution  
    FUNCTION minInsertions WITH PARAMETER s  
        FUNCTION longest_palindromic_subsequence WITH PARAMETER s  
            SET n TO the LENGTH OF s  
            SET dp TO a TWO-DIMENSIONAL ARRAY of size n BY n FILLED WITH zero  
            
            FOR i FROM zero TO n MINUS one  
                SET the ELEMENT at position i AND position i OF dp TO one  
            END FOR  
            
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF the CHARACTER at position i OF s IS EQUAL TO the CHARACTER at position j OF s AND cl IS EQUAL TO two THEN  
                        SET the ELEMENT at position i AND position j OF dp TO two  
                    ELSE IF the CHARACTER at position i OF s IS EQUAL TO the CHARACTER at position j OF s THEN  
                        SET the ELEMENT at position i AND position j OF dp TO the ELEMENT at position i PLUS one AND position j MINUS one OF dp PLUS two  
                    ELSE  
                        SET the ELEMENT at position i AND position j OF dp TO the GREATER VALUE BETWEEN the ELEMENT at position i AND position j MINUS one OF dp AND the ELEMENT at position i PLUS one AND position j OF dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN the ELEMENT at position zero AND position n MINUS one OF dp  
        END FUNCTION  
        
        SET result TO the LENGTH OF s MINUS longest_palindromic_subsequence WITH PARAMETER s  
        RETURN result  
    END FUNCTION  
END CLASS
