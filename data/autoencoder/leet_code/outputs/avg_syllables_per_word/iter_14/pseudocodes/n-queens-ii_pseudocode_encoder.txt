CLASS Solution  
    FUNCTION totalNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN NOT THE SUM OF element at position col OF cols PLUS element at position row MINUS col OF hills PLUS element at position row PLUS col OF dales EQUAL TO zero  
        END FUNCTION  
        
        FUNCTION place_queen WITH PARAMETERS row AND col  
            SET element at position col OF cols TO one  
            SET element at position row MINUS col OF hills TO one  
            SET element at position row PLUS col OF dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen WITH PARAMETERS row AND col  
            SET element at position col OF cols TO zero  
            SET element at position row MINUS col OF hills TO zero  
            SET element at position row PLUS col OF dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS row EQUAL TO zero AND count EQUAL TO zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH row AND col IS TRUE  
                    CALL place_queen WITH row AND col  
                    IF row PLUS one EQUAL TO n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO backtrack WITH row PLUS one AND count  
                    END IF  
                    CALL remove_queen WITH row AND col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO a list OF zero REPEATED n TIMES  
        SET hills TO a list OF zero REPEATED two MULTIPLIED BY n MINUS one TIMES  
        SET dales TO a list OF zero REPEATED two MULTIPLIED BY n MINUS one TIMES  
        
        RETURN backtrack WITH no PARAMETERS  
    END FUNCTION  
END CLASS
