CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL sort_the_pairs_by_second_element WITH pairs AND ASSIGN RESULT TO pairs  
        SET current_end TO negative_infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            SET start TO first element OF pair  
            SET end TO second element OF pair  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  

    FUNCTION sort_the_pairs_by_second_element WITH input_pairs  
        RETURN input_pairs SORTED IN ASCENDING ORDER ACCORDING TO the second element OF each pair  
    END FUNCTION  
END CLASS
