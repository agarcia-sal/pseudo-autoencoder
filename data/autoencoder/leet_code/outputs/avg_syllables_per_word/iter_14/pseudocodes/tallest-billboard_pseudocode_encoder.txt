CLASS Solution  
    FUNCTION tallestBillboard WITH PARAMETER rods  
        SET dp TO a dictionary with key zero mapped to zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d AND corresponding value y IN dp  
                SET key_for_taller_support TO d PLUS rod  
                IF key_for_taller_support IS NOT IN new_dp OR value at key_for_taller_support IN new_dp IS LESS THAN y  
                    SET value at key_for_taller_support IN new_dp TO y  
                END IF  
                SET absolute_difference TO the absolute value of d MINUS rod  
                SET candidate_value TO y PLUS the minimum value between d AND rod  
                IF absolute_difference IS NOT IN new_dp OR value at absolute_difference IN new_dp IS LESS THAN candidate_value  
                    SET value at absolute_difference IN new_dp TO candidate_value  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value at key zero IN dp  
    END FUNCTION  
END CLASS
