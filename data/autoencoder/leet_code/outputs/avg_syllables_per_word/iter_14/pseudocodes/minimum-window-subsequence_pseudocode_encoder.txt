CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s1 AND s2  
        SET min_len TO positive infinity  
        SET min_start TO negative one  
        SET len1 TO the LENGTH OF s1  
        SET len2 TO the LENGTH OF s2  
        
        FOR each index i FROM zero TO len1 MINUS one  
            SET j TO zero  
            FOR each index k FROM i TO len1 MINUS one  
                IF the character AT position k OF s1 IS EQUAL TO the character AT position j OF s2  
                    INCREMENT j BY one  
                    IF j IS EQUAL TO len2  
                        IF the result OF k MINUS i PLUS one IS LESS THAN min_len  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK out of the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start IS EQUAL TO negative one  
            RETURN an empty string  
        ELSE  
            SET result_substring TO the substring OF s1 FROM position min_start TO the position min_start PLUS min_len MINUS one  
            RETURN result_substring  
        END IF  
    END FUNCTION  
END CLASS
