CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s IS EQUAL TO the REVERSE OF s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num IS EQUAL TO zero  
                RETURN the STRING REPRESENTATION OF zero  
            END IF  
            SET digits TO an EMPTY list  
            WHILE num IS GREATER THAN zero  
                APPEND the INTEGER VALUE OF num MODULO k TO digits  
                SET num TO num DIVIDED BY k USING INTEGER DIVISION  
            END WHILE  
            RETURN the CONCATENATION OF the STRING REPRESENTATION OF each ELEMENT in digits IN REVERSE ORDER  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length IS EQUAL TO one  
                FOR i FROM one TO nine  
                    YIELD the STRING REPRESENTATION OF i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF the INTEGER VALUE OF (length MINUS one) DIVIDED BY two USING INTEGER DIVISION  
                SET end TO ten RAISED TO THE POWER OF the INTEGER VALUE OF (length PLUS one) DIVIDED BY two USING INTEGER DIVISION  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO the STRING REPRESENTATION OF half  
                    IF length MODULO two IS EQUAL TO zero  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH the SUBSTRING OF half_str FROM the SECOND TO LAST ELEMENT DOWN TO the FIRST ELEMENT IN REVERSE ORDER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an EMPTY list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes WITH PARAMETER length  
                SET num TO the INTEGER VALUE OF pal  
                IF is_palindrome WITH PARAMETER to_base_k WITH PARAMETERS num AND k IS TRUE  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers IS EQUAL TO n  
                        BREAK the FOR LOOP  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the SUM OF k_mirror_numbers  
    END FUNCTION  
END CLASS
