CLASS Solution  
    FUNCTION findMaximizedCapital WITH PARAMETERS k AND w AND profits AND capital  
        SET projects TO the RESULT OF combining each element of capital WITH the corresponding element of profits INTO pairs  
        SORT projects BY the FIRST element OF each pair IN ascending order  
        
        SET max_heap TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR index FROM one TO k  
            WHILE i IS LESS THAN n AND the FIRST element OF the element at position i OF projects IS LESS THAN OR EQUAL TO w  
                PUSH the NEGATIVE of the SECOND element OF the element at position i OF projects INTO max_heap  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS EMPTY  
                BREAK out OF the FOR loop  
            END IF  
            
            SET w TO w PLUS the NEGATIVE of the REMOVED top element FROM max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
