CLASS Solution  
    FUNCTION maxDotProduct WITH PARAMETERS nums1 nums2  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        CALL initializeTwoDimensionalList WITH PARAMETERS m PLUS one n PLUS one NEGATIVE infinity ASSIGN TO dp  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one OF nums1 MULTIPLIED BY element at position j MINUS one OF nums2  
                
                SET candidate_one TO current_product  
                SET candidate_two TO element at position i MINUS one ROW AND position j OF dp  
                SET candidate_three TO element at position i ROW AND position j MINUS one OF dp  
                SET candidate_four TO element at position i MINUS one ROW AND position j MINUS one OF dp PLUS current_product  
                
                SET dp at position i ROW AND position j TO the MAXIMUM OF candidate_one candidate_two candidate_three candidate_four  
            END FOR  
        END FOR  
        
        RETURN element at position m ROW AND position n OF dp  
    END FUNCTION  

    FUNCTION initializeTwoDimensionalList WITH PARAMETERS rows cols initial_value  
        CREATE empty list named table  
        FOR row_index FROM zero TO rows MINUS one  
            CREATE empty list named current_row  
            FOR col_index FROM zero TO cols MINUS one  
                APPEND initial_value TO current_row  
            END FOR  
            APPEND current_row TO table  
        END FOR  
        RETURN table  
    END FUNCTION  
END CLASS
