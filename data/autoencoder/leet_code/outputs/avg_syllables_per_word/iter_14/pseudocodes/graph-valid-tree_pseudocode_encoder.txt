CLASS Solution  
    FUNCTION validTree WITH PARAMETERS n AND edges  
        IF LENGTH OF edges IS NOT EQUAL TO n MINUS one  
            RETURN false  
        END IF  
        
        SET adj_list TO EMPTY MAPPING FROM INTEGER TO EMPTY LIST FOR EACH INTEGER INDEX FROM zero TO n MINUS one  
        FOR EACH pair OF a AND b IN edges  
            APPEND b TO THE LIST AT KEY a IN adj_list  
            APPEND a TO THE LIST AT KEY b IN adj_list  
        END FOR  
        
        SET visited TO EMPTY SET  
        
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR EACH neighbor IN THE LIST AT KEY node IN adj_list  
                CALL dfs WITH ARGUMENT neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH ARGUMENT zero  
        
        RETURN LENGTH OF visited IS EQUAL TO n  
    END FUNCTION  
END CLASS
