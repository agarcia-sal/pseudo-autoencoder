CLASS Solution  
    FUNCTION minimumEffort(tasks)  
        CALL helper FUNCTION sortTasksDescendingByDifference WITH tasks  
        SET current_energy TO zero  
        SET required_initial_energy TO zero  
        FOR each task IN tasks  
            SET actual TO element at position one OF task  
            SET minimum TO element at position two OF task  
            IF current_energy IS LESS THAN minimum  
                INCREMENT required_initial_energy BY minimum MINUS current_energy  
                SET current_energy TO minimum  
            END IF  
            DECREMENT current_energy BY actual  
        END FOR  
        RETURN required_initial_energy  
    END FUNCTION  
  
    FUNCTION sortTasksDescendingByDifference(tasks)  
        SORT tasks IN PLACE BY descending order OF the difference BETWEEN element at position two AND element at position one OF each task  
    END FUNCTION  
END CLASS
