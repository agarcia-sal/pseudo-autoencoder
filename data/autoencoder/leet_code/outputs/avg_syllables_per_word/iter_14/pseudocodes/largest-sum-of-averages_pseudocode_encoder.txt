CLASS Solution  
    FUNCTION largestSumOfAverages WITH PARAMETERS nums AND k  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list OF zeroes WITH SIZE n PLUS one  
        FOR index i FROM zero TO n MINUS one  
            SET the ELEMENT at position i PLUS one OF prefix_sum TO the ELEMENT at position i OF prefix_sum PLUS the ELEMENT at position i OF nums  
        END FOR  

        FUNCTION average WITH PARAMETERS i AND j  
            SET numerator TO the ELEMENT at position j PLUS one OF prefix_sum MINUS the ELEMENT at position i OF prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  

        SET dp TO a two-dimensional list FILLED WITH zeroes WITH DIMENSIONS n PLUS one BY k PLUS one  

        FOR index i FROM zero TO n MINUS one  
            SET the ELEMENT at position i AND k ONE OF dp TO CALL average WITH PARAMETERS i AND n MINUS one  
        END FOR  

        FOR index i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR index j FROM i PLUS one TO n MINUS one  
                    SET candidate TO the ELEMENT at position j AND partitions MINUS one OF dp PLUS CALL average WITH PARAMETERS i AND j MINUS one  
                    IF candidate GREATER THAN the ELEMENT at position i AND partitions OF dp  
                        SET the ELEMENT at position i AND partitions OF dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN the ELEMENT at position zero AND k OF dp  
    END FUNCTION  
END CLASS
