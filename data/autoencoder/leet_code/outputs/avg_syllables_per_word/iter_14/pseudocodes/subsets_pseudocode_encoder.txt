CLASS Solution  
    FUNCTION subsets WITH PARAMETER nums  
        IF nums IS EMPTY  
            RETURN A LIST CONTAINING AN EMPTY LIST  
        END IF  
        SET subsets_without_first TO THE RESULT OF CALLING subsets WITH PARAMETER THE LIST OF ELEMENTS FROM POSITION one TO THE END OF nums  
        SET subsets_with_first TO A NEW LIST FOR EACH subset IN subsets_without_first WHERE EACH ELEMENT IS A NEW LIST CONSISTING OF THE ELEMENT AT POSITION zero OF nums FOLLOWED BY subset  
        RETURN A NEW LIST WHICH IS THE CONCATENATION OF subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
