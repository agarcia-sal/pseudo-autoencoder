CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list OF zero WITH the LENGTH n  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i OF dominoes IS EQUAL TO character R  
                SET force TO n  
            ELSE IF the element at position i OF dominoes IS EQUAL TO character L  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            INCREMENT the element at position i OF forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF the element at position i OF dominoes IS EQUAL TO character L  
                SET force TO n  
            ELSE IF the element at position i OF dominoes IS EQUAL TO character R  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            DECREMENT the element at position i OF forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each f IN forces  
            IF f IS GREATER THAN zero  
                APPEND character R TO result  
            ELSE IF f IS LESS THAN zero  
                APPEND character L TO result  
            ELSE  
                APPEND character full stop TO result  
            END IF  
        END FOR  

        RETURN the concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
