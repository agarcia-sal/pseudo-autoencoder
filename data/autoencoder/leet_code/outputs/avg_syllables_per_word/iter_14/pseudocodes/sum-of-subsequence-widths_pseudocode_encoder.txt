CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO ten RAISED TO THE power OF nine PLUS one  
        SORT nums IN NONDECREASING ORDER  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO A LIST OF size n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET power at position index TO (power at position index MINUS one MULTIPLIED BY two) MODULO MODULO  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO (total_sum PLUS nums at position index MULTIPLIED BY (power at position index MINUS power at position n MINUS index MINUS one)) MODULO MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
