CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            IF sub IS EQUAL TO THE REVERSE OF sub THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        SET n TO THE LENGTH OF s  
  
        CREATE a two-dimensional list is_pal OF size n BY n FILLED WITH false  
  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF THE CHARACTER at position i OF s IS EQUAL TO THE CHARACTER at position j OF s AND (j MINUS i IS LESS THAN OR EQUAL TO one OR THE VALUE at position i PLUS one AND j MINUS one OF is_pal IS true) THEN  
                    SET THE VALUE at position i AND j OF is_pal TO true  
                END IF  
            END FOR  
        END FOR  
  
        FOR i FROM one TO n MINUS two  
            IF THE VALUE at position zero AND i MINUS one OF is_pal IS true THEN  
                FOR j FROM i TO n MINUS two  
                    IF THE VALUE at position i AND j OF is_pal IS true AND THE VALUE at position j PLUS one AND n MINUS one OF is_pal IS true THEN  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  
  
        RETURN false  
    END FUNCTION  
END CLASS
