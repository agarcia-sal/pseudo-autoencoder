CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF nums IS empty  
            RETURN zero  
        END IF  
        SET num_set TO the SET containing each element IN nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the ELEMENT WHICH IS num MINUS one IS NOT IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the ELEMENT WHICH IS current_num PLUS one IS IN num_set  
                    SET current_num TO current_num PLUS one  
                    INCREMENT current_streak BY one  
                END WHILE  
                IF current_streak IS GREATER THAN longest_streak  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
