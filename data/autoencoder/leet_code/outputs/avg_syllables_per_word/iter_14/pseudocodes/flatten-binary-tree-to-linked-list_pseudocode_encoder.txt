CLASS TreeNode  
    FUNCTION __init__(val EQUAL TO zero, left EQUAL TO None, right EQUAL TO None)  
        SET the instance variable val TO val  
        SET the instance variable left TO left  
        SET the instance variable right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero OF values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT EMPTY  
        SET node TO the element removed from the left of the queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET the left child OF node TO a new TreeNode initialized with the element at position i OF values  
            APPEND the left child OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET the right child OF node TO a new TreeNode initialized with the element at position i OF values  
            APPEND the right child OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute OF p IS NOT EQUAL TO the val attribute OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH the left child OF p AND the left child OF q AND the result of is_same_tree WITH the right child OF p AND the right child OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(root)  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO a new list containing root as the only element  
        WHILE stack IS NOT EMPTY  
            REMOVE the last element FROM stack AND SET it TO node  
            IF the right child OF node IS NOT None  
                APPEND the right child OF node TO stack  
            END IF  
            IF the left child OF node IS NOT None  
                APPEND the left child OF node TO stack  
            END IF  
            SET the left child OF node TO None  
            IF stack IS NOT EMPTY  
                SET the right child OF node TO the last element OF stack  
            ELSE  
                SET the right child OF node TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
