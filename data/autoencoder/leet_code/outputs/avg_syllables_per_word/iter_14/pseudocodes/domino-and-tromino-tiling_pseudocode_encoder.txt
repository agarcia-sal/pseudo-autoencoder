CLASS Solution  
    FUNCTION numTilings WITH PARAMETER n  
        SET MOD TO ten raised TO the power of nine PLUS one  
        
        DEFINE dp AS a list of zeroes with LENGTH n PLUS one  
        DEFINE dp1 AS a list of zeroes with LENGTH n PLUS one  
        
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        
        IF n IS GREATER THAN OR EQUAL TO two  
            SET element at position two OF dp TO two  
            SET element at position two OF dp1 TO one  
        END IF  
        
        FOR i FROM three TO n  
            SET element at position i OF dp TO the SUM of element at position i MINUS one OF dp PLUS element at position i MINUS two OF dp PLUS two TIMES element at position i MINUS one OF dp1  
            SET element at position i OF dp TO element at position i OF dp MODULO MOD  
            SET element at position i OF dp1 TO the SUM of element at position i MINUS two OF dp PLUS element at position i MINUS one OF dp1  
            SET element at position i OF dp1 TO element at position i OF dp1 MODULO MOD  
        END FOR  
        
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
