CLASS Solution  
    FUNCTION rearrangeString WITH PARAMETERS s AND k  
        IF k LESS THAN OR EQUAL TO one THEN  
            RETURN s  
        END IF  
  
        SET freq TO the COUNTER of characters IN s  
  
        SET max_heap TO a LIST CREATED BY FOR each character AND count IN freq ITEMS CREATING a TUPLE OF the NEGATION OF count AND character  
        PERFORM HEAPIFY ON max_heap  
  
        SET wait_queue TO an EMPTY LIST  
  
        SET result TO an EMPTY LIST  
  
        WHILE max_heap IS NOT EMPTY DO  
            PERFORM HEAPPOP ON max_heap AND ASSIGN THE RESULT TO count AND char  
            APPEND char TO result  
  
            APPEND A TUPLE OF count PLUS one AND char TO wait_queue  
  
            IF the LENGTH OF wait_queue IS EQUAL TO k THEN  
                REMOVE THE FIRST ELEMENT FROM wait_queue AND ASSIGN TO count AND char  
                IF count IS LESS THAN zero THEN  
                    PERFORM HEAPPUSH OF the TUPLE count AND char ONTO max_heap  
                END IF  
            END IF  
        END WHILE  
  
        IF the LENGTH OF result IS NOT EQUAL TO the LENGTH OF s THEN  
            RETURN an EMPTY STRING  
        END IF  
  
        RETURN the CONCATENATION OF all ELEMENTS IN result INTO a SINGLE STRING  
    END FUNCTION  
END CLASS
