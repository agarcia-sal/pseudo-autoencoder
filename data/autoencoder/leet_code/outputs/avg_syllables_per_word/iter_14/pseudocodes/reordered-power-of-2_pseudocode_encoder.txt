CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion  
            SET current_power_string TO the string representation OF power_of_2  
            SET sorted_digits TO the tuple of characters sorted IN ascending order OF current_power_string  
            ADD sorted_digits TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET n_string TO the string representation OF n  
        SET sorted_n_digits TO the tuple of characters sorted IN ascending order OF n_string  
        IF sorted_n_digits IS IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
