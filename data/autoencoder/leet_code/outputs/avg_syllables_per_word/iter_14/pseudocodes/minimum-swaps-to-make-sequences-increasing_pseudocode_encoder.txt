CLASS Solution  
    FUNCTION minSwap(nums1 LIST OF INTEGER, nums2 LIST OF INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF nums1  
        SET dp TO a new list of n elements each being a list of two zeros representing minimum swaps for no swap and swap states  
        
        SET element at zero AT position zero OF dp TO zero  
        SET element at zero AT position one OF dp TO one  
        
        FOR index i FROM one TO n MINUS one  
            IF element at position i OF nums1 IS GREATER THAN element at position i MINUS one OF nums1  
               AND element at position i OF nums2 IS GREATER THAN element at position i MINUS one OF nums2 THEN  
                IF element at position i OF nums1 IS GREATER THAN element at position i MINUS one OF nums2  
                   AND element at position i OF nums2 IS GREATER THAN element at position i MINUS one OF nums1 THEN  
                    SET element at position i AT position zero OF dp TO the MINIMUM OF element at position i MINUS one AT position zero OF dp AND element at position i MINUS one AT position one OF dp  
                    SET element at position i AT position one OF dp TO element at position i AT position zero OF dp PLUS one  
                ELSE  
                    SET element at position i AT position zero OF dp TO element at position i MINUS one AT position zero OF dp  
                    SET element at position i AT position one OF dp TO element at position i MINUS one AT position one OF dp PLUS one  
                END IF  
            ELSE  
                IF element at position i OF nums1 IS GREATER THAN element at position i MINUS one OF nums2  
                   AND element at position i OF nums2 IS GREATER THAN element at position i MINUS one OF nums1 THEN  
                    SET element at position i AT position zero OF dp TO element at position i MINUS one AT position one OF dp  
                    SET element at position i AT position one OF dp TO element at position i MINUS one AT position zero OF dp PLUS one  
                ELSE  
                    SET element at position i AT position zero OF dp TO element at position i MINUS one AT position one OF dp  
                    SET element at position i AT position one OF dp TO element at position i MINUS one AT position zero OF dp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN the MINIMUM OF element at position n MINUS one AT position zero OF dp AND element at position n MINUS one AT position one OF dp  
    END FUNCTION  
END CLASS
