CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the NUMBER OF TIMES character EQUAL TO letter X APPEARS IN row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the NUMBER OF TIMES character EQUAL TO letter O APPEARS IN row  
        END FOR  
        
        IF countX IS LESS THAN countO OR countX IS GREATER THAN countO PLUS one  
            RETURN false  
        END IF  
        
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row IS EQUAL TO player REPEATED three TIMES  
                    RETURN true  
                END IF  
            END FOR  
            FOR col FROM zero TO two  
                IF character at position col OF element at position zero OF board IS EQUAL TO player AND  
                   character at position col OF element at position one OF board IS EQUAL TO player AND  
                   character at position col OF element at position two OF board IS EQUAL TO player  
                    RETURN true  
                END IF  
            END FOR  
            IF character at position zero OF element at position zero OF board IS EQUAL TO player AND  
               character at position one OF element at position one OF board IS EQUAL TO player AND  
               character at position two OF element at position two OF board IS EQUAL TO player  
                RETURN true  
            END IF  
            IF character at position two OF element at position zero OF board IS EQUAL TO player AND  
               character at position one OF element at position one OF board IS EQUAL TO player AND  
               character at position zero OF element at position two OF board IS EQUAL TO player  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET winX TO check_win(letter X)  
        SET winO TO check_win(letter O)  
        
        IF winX IS true AND winO IS true  
            RETURN false  
        END IF  
        IF winX IS true AND countX IS EQUAL TO countO  
            RETURN false  
        END IF  
        IF winO IS true AND countX IS NOT EQUAL TO countO  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
