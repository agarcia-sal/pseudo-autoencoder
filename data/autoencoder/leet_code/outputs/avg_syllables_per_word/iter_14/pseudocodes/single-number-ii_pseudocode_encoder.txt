CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO (bitwise exclusive OR of ones WITH num) AND bitwise negation of twos  
            SET twos TO (bitwise exclusive OR of twos WITH num) AND bitwise negation of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
