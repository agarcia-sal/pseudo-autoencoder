CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits IS EMPTY  
            RETURN empty list  
        END IF  

        SET phone_map TO a mapping from digit characters to corresponding letter strings in the order two maps to abc three maps to def four maps to ghi five maps to jkl six maps to mno seven maps to pqrs eight maps to tuv nine maps to wxyz  

        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits IS EQUAL TO zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string mapped from the element at position zero OF next_digits IN phone_map  
                    CALL backtrack WITH the string CONCATENATION OF combination AND letter AND the substring FROM position one TO end OF next_digits  
                END FOR  
            END IF  
        END FUNCTION  

        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
