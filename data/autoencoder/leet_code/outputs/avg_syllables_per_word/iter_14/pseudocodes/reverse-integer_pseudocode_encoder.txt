CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_x TO the absolute value OF x  
        SET string_x TO the conversion of absolute_x TO a string  
        SET reversed_string_x TO the reverse ORDER OF characters IN string_x  
        SET reversed_x TO the conversion of reversed_string_x TO an integer  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative one MULTIPLIED BY two RAISED TO THE power OF thirty ONE OR result GREATER THAN two RAISED TO THE power OF thirty ONE MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
