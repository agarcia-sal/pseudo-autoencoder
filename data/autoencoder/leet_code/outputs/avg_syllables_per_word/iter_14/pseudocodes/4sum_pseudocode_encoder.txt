CLASS Solution  
    FUNCTION fourSum WITH PARAMETERS nums AND target  
        CALL sort ON nums TO ARRANGE ELEMENTS IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET quadruplets TO an empty list  
        
        FOR i FROM zero TO n MINUS three MINUS one  
            IF i GREATER THAN zero AND element at position i OF nums EQUAL TO element at position i MINUS one OF nums  
                CONTINUE TO the next iteration of the FOR loop  
            END IF  
            
            FOR j FROM i PLUS one TO n MINUS two MINUS one  
                IF j GREATER THAN i PLUS one AND element at position j OF nums EQUAL TO element at position j MINUS one OF nums  
                    CONTINUE TO the next iteration of the FOR loop  
                END IF  
                
                SET left TO j PLUS one  
                SET right TO n MINUS one  
                
                WHILE left LESS THAN right  
                    SET total TO element at position i OF nums PLUS element at position j OF nums PLUS element at position left OF nums PLUS element at position right OF nums  
                    
                    IF total EQUAL TO target  
                        APPEND a list containing element at position i OF nums, element at position j OF nums, element at position left OF nums, element at position right OF nums TO quadruplets  
                        
                        WHILE left LESS THAN right AND element at position left OF nums EQUAL TO element at position left PLUS one OF nums  
                            INCREMENT left BY one  
                        END WHILE  
                        
                        WHILE left LESS THAN right AND element at position right OF nums EQUAL TO element at position right MINUS one OF nums  
                            DECREMENT right BY one  
                        END WHILE  
                        
                        INCREMENT left BY one  
                        DECREMENT right BY one  
                        
                    ELSE IF total LESS THAN target  
                        INCREMENT left BY one  
                        
                    ELSE  
                        DECREMENT right BY one  
                        
                    END IF  
                END WHILE  
            END FOR  
        END FOR  
        
        RETURN quadruplets  
    END FUNCTION  
END CLASS
