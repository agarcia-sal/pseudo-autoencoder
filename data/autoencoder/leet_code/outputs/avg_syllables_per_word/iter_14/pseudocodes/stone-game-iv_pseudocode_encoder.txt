CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of LENGTH n PLUS one FILLED WITH False  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j IS LESS THAN OR EQUAL TO i  
                IF dp AT position i MINUS j MULTIPLIED BY j IS EQUAL TO False  
                    SET dp AT position i TO True  
                    BREAK out of the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN dp AT position n  
    END FUNCTION  
END CLASS
