CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s AS empty string, left AS zero, right AS zero  
            IF LENGTH OF s IS EQUAL TO two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left IS LESS THAN n  
                CALL backtrack WITH ARGUMENTS s CONCATENATED WITH opening parenthesis CHARACTER, left INCREMENTED BY one, right  
            END IF  
            IF right IS LESS THAN left  
                CALL backtrack WITH ARGUMENTS s CONCATENATED WITH closing parenthesis CHARACTER, left, right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH NO ARGUMENTS  
        RETURN result  
    END FUNCTION  
END CLASS
