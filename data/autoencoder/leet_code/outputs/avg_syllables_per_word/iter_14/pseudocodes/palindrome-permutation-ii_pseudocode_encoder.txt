CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF LENGTH OF t EQUAL TO LENGTH OF s THEN  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND count v IN the ITEMS OF cnt  
                IF v GREATER THAN one THEN  
                    DECREMENT cnt AT c BY two  
                    CALL dfs WITH PARAMETER CONCATENATION OF c PLUS t PLUS c  
                    INCREMENT cnt AT c BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO THE COUNTER OF s  
        SET mid TO an EMPTY string  
        FOR each character c AND count v IN the ITEMS OF cnt  
            IF v MODULO two EQUAL TO one THEN  
                IF mid IS NOT an EMPTY string THEN  
                    RETURN an EMPTY list  
                END IF  
                SET mid TO c  
                DECREMENT cnt AT c BY one  
            END IF  
        END FOR  
        SET ans TO an EMPTY list  
        CALL dfs WITH PARAMETER mid  
        RETURN ans  
    END FUNCTION  
END CLASS
