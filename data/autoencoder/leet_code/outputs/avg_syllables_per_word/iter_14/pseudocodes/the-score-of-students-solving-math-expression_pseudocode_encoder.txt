CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character IS a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS numeric value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF the element at position i OF tokens IS the multiplication symbol THEN  
                    SET the element at position i minus one OF tokens TO the element at position i minus one MULTIPLIED BY the element at position i plus one OF tokens  
                    REMOVE elements from position i TO position i plus two EXCLUSIVE FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO the element at position zero OF tokens  
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF the element at position i OF tokens IS the addition symbol THEN  
                    SET result TO result PLUS the element at position i plus one OF tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF expression CONSISTS ONLY OF digits THEN  
                RETURN a set containing the numeric value of expression  
            END IF  
            
            SET results TO empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS one MINUS one STEP two  
                SET left_results TO get_possible_answers(substring FROM position zero TO i EXCLUSIVE OF expression)  
                SET right_results TO get_possible_answers(substring FROM position i plus one TO end OF expression)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF the character at position i OF expression IS the addition symbol THEN  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result IS LESS THAN OR EQUAL TO one thousand THEN  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer IS EQUAL TO correct_answer THEN  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers THEN  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
