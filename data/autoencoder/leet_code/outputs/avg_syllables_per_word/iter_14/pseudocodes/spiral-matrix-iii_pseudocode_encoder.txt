CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET directions TO auxiliary FUNCTION that RETURNS a list of four pairs where each pair represents a movement in vertical and horizontal directions respectively ordered as east zero one, south one zero, west zero negative one, and north negative one zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO an empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols  
            IF the remainder when increment IS DIVIDED BY two IS EQUAL TO zero  
                INCREMENT steps BY one  
            END IF  
            FOR each count FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND the pair consisting of x AND y TO visited  
                END IF  
                SET dx TO the element at position direction_index FIRST element OF directions  
                SET dy TO the element at position direction_index SECOND element OF directions  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO the remainder when direction_index PLUS one IS DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
