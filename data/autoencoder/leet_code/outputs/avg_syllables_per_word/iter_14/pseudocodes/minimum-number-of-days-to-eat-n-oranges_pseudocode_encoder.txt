CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        DEFINE FUNCTION dp WITH PARAMETER remaining  
            IF remaining IS LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET result TO one PLUS MINIMUM OF the SUM OF the remainder of remaining DIVIDED BY two AND dp OF remaining INTEGER DIVIDED BY two AND the SUM OF the remainder of remaining DIVIDED BY three AND dp OF remaining INTEGER DIVIDED BY three  
            RETURN result  
        END FUNCTION  
        RETURN dp OF n  
END CLASS
