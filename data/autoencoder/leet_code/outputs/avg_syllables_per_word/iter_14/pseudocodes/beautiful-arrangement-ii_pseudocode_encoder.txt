CLASS Solution  
    FUNCTION constructArray(n integer, k integer) RETURNS list of integer  
        SET result TO a list containing the sequence from the integer one TO the integer n MINUS k MINUS one  
        SET left TO the integer n MINUS k  
        SET right TO the integer n  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF left IS EQUAL TO right  
                APPEND left TO result  
            ELSE  
                APPEND left TO result  
                APPEND right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
