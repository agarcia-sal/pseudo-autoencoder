CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of LENGTH n FILLED WITH negative one  
        SET stack TO an empty list  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position the last element OF stack OF nums IS LESS THAN the element at position index OF nums  
                SET position TO the REMOVED last element FROM stack  
                SET the element at position position OF result TO the element at position index OF nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
