CLASS Solution  
    FUNCTION rotate(nums LIST OF INTEGER k INTEGER)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start INTEGER end INTEGER)  
            WHILE start IS LESS THAN end  
                SET temporary_value TO element at position start OF nums  
                SET element at position start OF nums TO element at position end OF nums  
                SET element at position end OF nums TO temporary_value  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH arguments zero AND n MINUS one  
        CALL reverse WITH arguments zero AND k MINUS one  
        CALL reverse WITH arguments k AND n MINUS one  
    END FUNCTION  
END CLASS
