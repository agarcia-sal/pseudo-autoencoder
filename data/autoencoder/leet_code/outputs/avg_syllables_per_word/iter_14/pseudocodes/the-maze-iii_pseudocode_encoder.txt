CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze LIST OF LIST OF integer ball LIST OF integer hole LIST OF integer RETURNS string  
        SET directions TO a list containing tuples of zero AND one AND letter r, one AND zero AND letter d, zero AND negative one AND letter l, negative one AND zero AND letter u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero OF maze  
        SET start TO a tuple made from elements of ball  
        SET hole TO a tuple made from elements of hole  
        SET pq TO a list containing a tuple of zero, empty string, start  
        SET visited TO an empty set  
        
        WHILE pq IS NOT empty  
            REMOVE the element with the smallest distance FROM pq AND ASSIGN its contents TO dist, path, and a tuple x, y  
            
            IF the tuple x, y IS EQUAL TO hole  
                RETURN path  
            END IF  
            
            IF the tuple x, y IS IN visited  
                CONTINUE to the next iteration  
            END IF  
            
            ADD the tuple x, y TO visited  
            
            FOR each tuple dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE condition new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND the element at position new_x plus dx of maze AND the element at position new_y plus dy OF that element IS EQUAL TO zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    IF the tuple new_x, new_y IS EQUAL TO hole  
                        EXIT the WHILE loop  
                    END IF  
                END WHILE  
                
                IF the tuple new_x, new_y IS NOT IN visited  
                    ADD a tuple of dist PLUS count, path CONCATENATED WITH direction, AND the tuple new_x, new_y TO pq AS new heap element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
