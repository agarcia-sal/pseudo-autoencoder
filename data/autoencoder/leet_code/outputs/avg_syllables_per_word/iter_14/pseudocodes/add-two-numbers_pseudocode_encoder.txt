CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val defaulted to zero next defaulted to none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN (element val OF p1) EQUAL TO (element val OF p2) AND is_same_list(element next OF p1 element next OF p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS self l1 l2  
        SET dummy_head TO A NEW ListNode CREATED WITH zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS GREATER THAN zero  
            IF l1 IS NOT none  
                SET val1 TO element val OF l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT none  
                SET val2 TO element val OF l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO A NEW ListNode CREATED WITH total MODULO ten  
            SET current TO element next OF current  
            
            IF l1 IS NOT none  
                SET l1 TO element next OF l1  
            END IF  
            IF l2 IS NOT none  
                SET l2 TO element next OF l2  
            END IF  
        END WHILE  
        
        RETURN element next OF dummy_head  
    END FUNCTION  
END CLASS
