CLASS Solution  
    FUNCTION boxDelivering WITH PARAMETERS boxes portsCount maxBoxes maxWeight  
        SET n TO the LENGTH OF boxes  
        SET dp TO a list OF length n PLUS one WHERE each element IS infinity  
        SET element at POSITION zero OF dp TO zero  
        
        SET left TO zero  
        SET current_weight TO zero  
        SET extra_trips TO zero  
        
        FOR each right FROM zero TO n MINUS one  
            INCREMENT current_weight BY element at POSITION one OF element at POSITION right OF boxes  
            
            IF right IS GREATER THAN zero AND element at POSITION zero OF element at POSITION right OF boxes IS NOT EQUAL TO element at POSITION zero OF element at POSITION right MINUS one OF boxes  
                INCREMENT extra_trips BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND (right MINUS left PLUS one IS GREATER THAN maxBoxes OR current_weight IS GREATER THAN maxWeight)  
                DECREMENT current_weight BY element at POSITION one OF element at POSITION left OF boxes  
                
                IF left IS GREATER THAN zero AND element at POSITION zero OF element at POSITION left OF boxes IS NOT EQUAL TO element at POSITION zero OF element at POSITION left MINUS one OF boxes  
                    DECREMENT extra_trips BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            SET element at POSITION right PLUS one OF dp TO element at POSITION left OF dp PLUS two PLUS extra_trips  
        END FOR  
        
        RETURN element at POSITION n OF dp  
    END FUNCTION  
END CLASS
