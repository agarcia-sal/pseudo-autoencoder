CLASS Solution  
    FUNCTION characterReplacement(s AS string, k AS integer) RETURNS integer  
        INITIALIZE a new dictionary count THAT RETURNS zero FOR any new key  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  

        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the value IN count ASSOCIATED WITH the element at position right OF s BY one  
            SET max_count TO the GREATER OF max_count AND the value IN count ASSOCIATED WITH the element at position right OF s  

            IF the value OF right MINUS the value OF left PLUS one MINUS max_count IS GREATER THAN k  
                DECREMENT the value IN count ASSOCIATED WITH the element at position left OF s BY one  
                INCREMENT left BY one  
            END IF  

            SET max_length TO the GREATER OF max_length AND the value OF right MINUS the value OF left PLUS one  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
