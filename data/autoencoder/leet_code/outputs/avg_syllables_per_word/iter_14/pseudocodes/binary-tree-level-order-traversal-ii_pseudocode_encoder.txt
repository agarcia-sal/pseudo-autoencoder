CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val SET TO zero LEFT SET TO none RIGHT SET TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS NOT PRESENT  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero OF values  
    SET i TO one  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS NOT PRESENT AND q IS NOT PRESENT  
        RETURN true  
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        SET left_check TO CALL is_same_tree WITH PARAMETERS p.left q.left  
        SET right_check TO CALL is_same_tree WITH PARAMETERS p.right q.right  
        RETURN left_check AND right_check  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom WITH PARAMETER root  
        IF root IS NOT PRESENT  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO NEW deque INITIALIZED WITH root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO LENGTH OF queue  
            SET current_level TO empty list  
            FOR count FROM one TO level_size  
                SET node TO REMOVE FIRST ELEMENT FROM queue  
                APPEND node.val TO current_level  
                IF node.left IS PRESENT  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS PRESENT  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result REVERSED IN ORDER  
    END FUNCTION  
END CLASS
