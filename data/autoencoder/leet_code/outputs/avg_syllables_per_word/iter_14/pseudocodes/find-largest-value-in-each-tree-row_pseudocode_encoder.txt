CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUAL TO element at position zero OF values AND left AS None AND right AS None  
    SET i TO one  
    SET queue TO NEW empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.left TO NEW TreeNode WITH val EQUAL TO element at position i OF values AND left AS None AND right AS None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.right TO NEW TreeNode WITH val EQUAL TO element at position i OF values AND left AS None AND right AS None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree OF p.left AND q.left AND is_same_tree OF p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        SET queue TO NEW double-ended queue WITH root AS INITIAL element  
        SET largest_values TO NEW empty list  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR count FROM zero TO level_size MINUS one  
                SET node TO REMOVE first element FROM queue  
                IF max_value IS LESS THAN node.val THEN  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
