CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE FIRST ELEMENT OF values  
    SET p TO head  
    FOR each val IN THE SUBSEQUENCE OF values STARTING FROM THE SECOND ELEMENT TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN THE CONDITION THAT the val ATTRIBUTE OF p1 IS EQUAL TO the val ATTRIBUTE OF p2 AND THE RESULT OF is_same_list CALLED WITH THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH PARAMETERS self head  
        SET dummy TO A NEW ListNode CREATED WITH zero  
        SET the next ATTRIBUTE OF dummy TO head  
        SET current TO dummy  
        
        WHILE the next ATTRIBUTE OF current IS NOT none AND THE next ATTRIBUTE OF the next ATTRIBUTE OF current IS NOT none  
            SET first TO the next ATTRIBUTE OF current  
            SET second TO the next ATTRIBUTE OF the next ATTRIBUTE OF current  
            
            SET the next ATTRIBUTE OF current TO second  
            SET the next ATTRIBUTE OF first TO the next ATTRIBUTE OF second  
            SET the next ATTRIBUTE OF second TO first  
            
            SET current TO the next ATTRIBUTE OF the next ATTRIBUTE OF current  
        END WHILE  
        
        RETURN the next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
