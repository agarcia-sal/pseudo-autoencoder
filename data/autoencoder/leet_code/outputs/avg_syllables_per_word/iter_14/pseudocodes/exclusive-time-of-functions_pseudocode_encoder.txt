CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO list of zero repeated n times  
        SET stack TO empty list  
        
        FOR each log IN logs  
            SET function_id, action, timestamp TO elements extracted by splitting log by colon  
            SET function_id TO integer value of function_id  
            SET timestamp TO integer value of timestamp  
            
            IF action EQUAL TO start  
                IF stack IS NOT empty  
                    SET prev_function_id, prev_timestamp TO last element of stack  
                    INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND tuple of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id, prev_timestamp TO last element removed from stack  
                INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp PLUS one  
                IF stack IS NOT empty  
                    SET last_function_id, last_timestamp TO last element of stack  
                    SET stack last element TO tuple of last_function_id and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
