CLASS Solution  
    FUNCTION flipLights(n, presses)  
        SET n TO the MINIMUM OF n AND six  
        
        IF presses EQUAL TO zero  
            RETURN one  
        END IF  
        
        IF n EQUAL TO one  
            RETURN two  
        END IF  
        
        IF n EQUAL TO two  
            IF presses EQUAL TO one  
                RETURN three  
            ELSE  
                RETURN four  
            END IF  
        END IF  
        
        IF presses EQUAL TO one  
            RETURN four  
        END IF  
        
        IF presses EQUAL TO two  
            RETURN seven  
        END IF  
        
        RETURN eight  
    END FUNCTION  
END CLASS
