CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a mapping that returns empty list for missing keys  
        FOR each pair u and v IN edges  
            APPEND v TO the list corresponding to key u IN graph  
            APPEND u TO the list corresponding to key v IN graph  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node IN path  
                RETURN the sublist of path FROM cycle_start TO the END  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list corresponding to node IN graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs with arguments neighbor, node, path, visited  
                    IF cycle IS NOT null THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN null  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO null  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs with arguments node, negative one, empty list, visited  
                IF cycle IS NOT null THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zeros of length n  
        SET queue TO a double-ended queue initialized with elements of cycle  
        SET visited TO a set initialized with elements of cycle  
        
        SET level TO zero  
        WHILE queue IS NOT empty  
            SET iteration_count TO the number of elements in queue  
            FOR counter FROM one TO iteration_count  
                REMOVE the first element FROM queue AND ASSIGN it TO node  
                SET distance at position node TO level  
                FOR each neighbor IN the list corresponding to node IN graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
