CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF board MINUS one  
                IF element at position i OF board at position j IS EQUAL TO character X  
                    IF i IS EQUAL TO zero OR element at position i MINUS one OF board at position j IS EQUAL TO character dot  
                        AND j IS EQUAL TO zero OR element at position i OF board at position j MINUS one IS EQUAL TO character dot  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
