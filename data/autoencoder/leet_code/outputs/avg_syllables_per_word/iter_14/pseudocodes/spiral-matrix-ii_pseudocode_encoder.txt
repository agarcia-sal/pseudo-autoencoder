CLASS Solution  
    FUNCTION generateMatrix WITH PARAMETER n  
        SET matrix TO a new list of lists where each inner list is of length n filled with zeroes and the total number of inner lists is n  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                SET the element at position top OF matrix AT position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                SET the element at position i OF matrix AT position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left  
                    SET the element at position bottom OF matrix AT position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top  
                    SET the element at position i OF matrix AT position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
