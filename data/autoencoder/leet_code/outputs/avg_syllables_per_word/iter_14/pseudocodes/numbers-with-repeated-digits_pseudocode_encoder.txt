CLASS Solution  
    FUNCTION numDupDigitsAtMostN WITH PARAMETER n  
        FUNCTION count_unique_digits WITH PARAMETER x  
            SET s TO THE STRING REPRESENTATION OF x  
            SET length TO THE LENGTH OF s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm WITH ARGUMENTS nine MINUS one AND i MINUS one  
            END FOR  
            SET seen TO an EMPTY SET  
            FOR each index i AND digit_character digit IN s WITH ENUMERATION  
                SET start_range TO zero  
                IF i IS EQUAL TO zero THEN  
                    SET start_range TO one  
                END IF  
                SET digit_value TO THE NUMERIC VALUE OF digit  
                FOR d FROM start_range TO digit_value MINUS one  
                    IF d IS NOT AN ELEMENT OF seen THEN  
                        INCREMENT count BY perm WITH ARGUMENTS nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF digit_value IS AN ELEMENT OF seen THEN  
                    BREAK THE LOOP  
                END IF  
                ADD digit_value TO seen  
            END FOR  
            IF THE LOOP COMPLETED WITHOUT BREAKING THEN  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        SET total_unique_digit_numbers TO count_unique_digits WITH ARGUMENT n  
        SET result TO n MINUS total_unique_digit_numbers  
        RETURN result  
    END FUNCTION  

    FUNCTION perm WITH PARAMETERS m AND n  
        IF n IS EQUAL TO zero THEN  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
