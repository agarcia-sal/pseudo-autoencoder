CLASS Solution  
    FUNCTION getPermutation WITH PARAMETERS n AND k  
        SET numbers TO a list of integers FROM one TO n  
        DECREMENT k BY one  
        SET result TO an empty list  
        FOR integer i FROM n DOWN TO one  
            SET fact TO the factorial of i MINUS one  
            SET index TO the integer quotient of k DIVIDED BY fact  
            APPEND the string representation of the element at position index OF numbers TO result  
            REMOVE the element at position index FROM numbers  
            SET k TO the remainder of k DIVIDED BY fact  
        END FOR  
        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
