CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUAL TO word2  
        
        FOR index FROM zero TO the LENGTH OF wordsDict MINUS one  
            SET word TO element at position index OF wordsDict  
            
            IF word EQUAL TO word1  
                IF same_word IS TRUE  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO index  
            END IF  
            
            IF word EQUAL TO word2  
                IF same_word IS FALSE  
                    SET prev_word2 TO index  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUAL TO negative one AND prev_word2 NOT EQUAL TO negative one  
                SET current_distance TO prev_word1 MINUS prev_word2  
                IF current_distance LESS THAN zero  
                    SET current_distance TO prev_word2 MINUS prev_word1  
                END IF  
                IF current_distance LESS THAN min_distance  
                    SET min_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
