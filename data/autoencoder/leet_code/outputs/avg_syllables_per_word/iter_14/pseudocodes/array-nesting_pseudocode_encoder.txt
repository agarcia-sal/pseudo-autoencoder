CLASS Solution  
    FUNCTION arrayNesting WITH PARAMETER nums  
        SET visited TO a list of Boolean false values with the LENGTH OF nums elements  
        SET max_length TO zero  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF visited element at position index IS FALSE  
                SET current_length TO zero  
                SET k TO index  
                WHILE visited element at position k IS FALSE  
                    SET visited element at position k TO true  
                    SET k TO element at position k OF nums  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO the GREATER OF max_length AND current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
