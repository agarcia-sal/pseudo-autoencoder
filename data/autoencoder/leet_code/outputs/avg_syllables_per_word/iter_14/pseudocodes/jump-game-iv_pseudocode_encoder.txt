CLASS Solution  
    FUNCTION minJumps WITH PARAMETER arr AS list of integers  
        SET n TO the LENGTH OF arr  
        IF n EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new dictionary WITH DEFAULT VALUE AS empty list  
        FOR each pair OF i and value WHERE i is the INDEX and value is the ELEMENT in arr  
            APPEND i TO the list AT key value IN value_indices  
        END FOR  
        
        SET queue TO a new double ended queue CONTAINING a tuple OF zero AND zero representing current index AND current steps  
        SET visited TO a new set CONTAINING zero  
        
        WHILE queue IS NOT empty  
            SET current_index AND steps TO the FIRST element REMOVED FROM queue  
            
            IF current_index EQUAL TO n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list CONTAINING current_index PLUS one AND current_index MINUS one FOLLOWED BY the list AT key arr AT position current_index IN value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the tuple OF next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list AT key arr AT position current_index IN value_indices TO an empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
