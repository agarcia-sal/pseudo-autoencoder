CLASS Solution  
    FUNCTION checkRecord WITH PARAMETER n  
        SET MOD TO one billion PLUS seven  
        
        SET dp TO the result of creating a three dimensional list of zeroes with dimensions n PLUS one BY two BY three  
        
        SET element at position zero OF dp AT position zero OF dp AT position zero OF dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position i OF dp AT position j OF dp AT position zero OF dp TO the sum of element at position i OF dp AT position j OF dp AT position zero OF dp PLUS element at position i MINUS one OF dp AT position j OF dp AT position k OF dp MODULO MOD  
                    
                    IF j IS EQUAL TO zero  
                        SET element at position i OF dp AT position one OF dp AT position zero OF dp TO the sum of element at position i OF dp AT position one OF dp AT position zero OF dp PLUS element at position i MINUS one OF dp AT position zero OF dp AT position k OF dp MODULO MOD  
                    END IF  
                    
                    IF k IS LESS THAN two  
                        SET element at position i OF dp AT position j OF dp AT position k PLUS one OF dp TO the sum of element at position i OF dp AT position j OF dp AT position k PLUS one OF dp PLUS element at position i MINUS one OF dp AT position j OF dp AT position k OF dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO the sum of result PLUS element at position n OF dp AT position j OF dp AT position k OF dp MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
