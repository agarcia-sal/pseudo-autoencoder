CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 IS EQUAL TO None  
                RETURN zero  
            END IF  
            SET numeric_value1 TO the Unicode code point of c1 MINUS the Unicode code point of uppercase A  
            SET numeric_value2 TO the Unicode code point of c2 MINUS the Unicode code point of uppercase A  
            SET x1 TO numeric_value1 DIVIDED BY six  
            SET y1 TO numeric_value1 MODULO six  
            SET x2 TO numeric_value2 DIVIDED BY six  
            SET y2 TO numeric_value2 MODULO six  
            SET horizontal_difference TO the absolute value of x1 MINUS x2  
            SET vertical_difference TO the absolute value of y1 MINUS y2  
            RETURN horizontal_difference PLUS vertical_difference  
        END FUNCTION  

        FUNCTION dp(i, f1, f2)  
            IF i IS EQUAL TO the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i OF word  
            SET d1 TO distance BETWEEN f1 AND c PLUS dp CALLED WITH i PLUS one AND c AND f2  
            SET d2 TO distance BETWEEN f2 AND c PLUS dp CALLED WITH i PLUS one AND f1 AND c  
            IF d1 IS LESS THAN d2  
                RETURN d1  
            ELSE  
                RETURN d2  
            END IF  
        END FUNCTION  

        RETURN dp CALLED WITH zero AND None AND None  
    END FUNCTION  
END CLASS
