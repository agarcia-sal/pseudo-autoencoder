CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a dictionary that returns an empty list for any new key  
        
        FOR each path IN paths  
            SET parts TO result of splitting path by whitespace  
            SET directory TO element at position zero OF parts  
            
            FOR each file_info IN elements from position one TO the end OF parts  
                SET file_name_and_content TO result of splitting file_info by the character for opening parenthesis  
                SET file_name TO element at position zero OF file_name_and_content  
                SET content_with_closing TO element at position one OF file_name_and_content  
                SET content TO substring FROM position zero TO the length OF content_with_closing MINUS one OF content_with_closing  
                
                APPEND concatenation of directory, the forward slash character, and file_name TO the list at content in content_to_paths  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        FOR each paths_list IN the values OF content_to_paths  
            IF the length OF paths_list IS GREATER THAN one  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
