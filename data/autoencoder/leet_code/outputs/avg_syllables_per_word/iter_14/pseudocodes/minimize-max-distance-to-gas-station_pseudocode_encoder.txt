CLASS Solution  
    FUNCTION minmaxGasDist WITH PARAMETERS stations AND k  
        FUNCTION possible WITH PARAMETER max_dist  
            SET additional_stations TO zero  
            FOR index FROM one TO LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index OF stations MINUS element at position index MINUS one OF stations  
                SET count_additional TO distance_difference DIVIDED BY max_dist  
                INCREMENT additional_stations BY INTEGER PART OF count_additional  
            END FOR  
            IF additional_stations IS LESS THAN OR EQUAL TO k THEN  
                RETURN TRUE  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left IS GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO (left PLUS right) DIVIDED BY two  
            IF possible OF mid IS TRUE THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
