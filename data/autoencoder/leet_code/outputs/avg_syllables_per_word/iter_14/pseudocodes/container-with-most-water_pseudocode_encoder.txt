CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        
        WHILE left IS LESS THAN right  
            SET current_min_height TO the MINIMUM OF element at position left OF height AND element at position right OF height  
            SET current_width TO right MINUS left  
            SET current_area TO current_min_height MULTIPLIED BY current_width  
            SET max_area TO the MAXIMUM OF max_area AND current_area  
            
            IF element at position left OF height IS LESS THAN element at position right OF height  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
