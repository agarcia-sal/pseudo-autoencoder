CLASS Solution  
    FUNCTION killProcess WITH PARAMETERS pid list OF integers ppid list OF integers kill integer RETURNS list OF integers  
        SET tree TO a new default dictionary with list as default value  
        FOR each pair child and parent IN parallel traversal of pid and ppid  
            APPEND child TO the value at key parent IN tree  
        END FOR  
        SET queue TO a new list containing the single element kill  
        SET killed_processes TO a new empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE the element at position zero FROM queue AND ASSIGN it TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY all elements in the value at key current IN tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
