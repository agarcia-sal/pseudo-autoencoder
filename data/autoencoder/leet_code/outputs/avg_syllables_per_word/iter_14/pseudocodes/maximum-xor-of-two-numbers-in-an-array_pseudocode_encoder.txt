CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR bit_position FROM thirty one DOWN TO zero  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO an empty set  
            FOR each number IN nums  
                SET prefix TO number DIVIDED BY two TO THE POWER OF bit_position  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix_element IN prefixes  
                SET required_prefix TO current_candidate EXCLUSIVE OR prefix_element  
                IF required_prefix IS IN prefixes THEN  
                    SET max_xor TO current_candidate  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
