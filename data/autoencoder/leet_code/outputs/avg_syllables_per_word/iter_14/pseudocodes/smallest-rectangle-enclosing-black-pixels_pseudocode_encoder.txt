CLASS Solution  
    FUNCTION minArea(image AS List of List of string, x AS integer, y AS integer) AS integer  
        IF image IS EMPTY OR element at position zero OF image IS EMPTY THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero OF image  
        
        FUNCTION hasBlackPixelInRow(row AS integer) AS boolean  
            FOR each pixel IN element at position row OF image  
                IF pixel IS EQUAL TO string one THEN  
                    RETURN TRUE  
                END IF  
            END FOR  
            RETURN FALSE  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col AS integer) AS boolean  
            FOR row FROM zero TO m MINUS one  
                IF element at position col OF element at position row OF image IS EQUAL TO string one THEN  
                    RETURN TRUE  
                END IF  
            END FOR  
            RETURN FALSE  
        END FUNCTION  
        
        FUNCTION findTop() AS integer  
            SET low TO zero  
            SET high TO x  
            WHILE low IS LESS THAN high  
                SET mid TO the INTEGER DIVISION of the SUM of low AND high BY two  
                IF hasBlackPixelInRow(mid) IS TRUE THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom() AS integer  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low IS LESS THAN high  
                SET mid TO the INTEGER DIVISION of the SUM of low, high, AND one BY two  
                IF hasBlackPixelInRow(mid) IS TRUE THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft() AS integer  
            SET low TO zero  
            SET high TO y  
            WHILE low IS LESS THAN high  
                SET mid TO the INTEGER DIVISION of the SUM of low AND high BY two  
                IF hasBlackPixelInColumn(mid) IS TRUE THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight() AS integer  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low IS LESS THAN high  
                SET mid TO the INTEGER DIVISION of the SUM of low, high, AND one BY two  
                IF hasBlackPixelInColumn(mid) IS TRUE THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  
        
        RETURN the PRODUCT of the SUM of bottom MINUS top PLUS one AND the SUM of right MINUS left PLUS one  
    END FUNCTION  
END CLASS
