CLASS Solution  
    FUNCTION maxPerformance WITH PARAMETERS n speed efficiency k  
        SET engineers TO A NEW LIST OF PAIRS WHERE EACH PAIR CONSISTS OF THE CORRESPONDING ELEMENT OF efficiency AND THE CORRESPONDING ELEMENT OF speed  
        SORT engineers IN DESCENDING ORDER BASED ON THE FIRST ELEMENT OF EACH PAIR  
        SET speed_heap TO AN EMPTY MINIMUM PRIORITY COLLECTION  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair IN engineers  
            SET eff TO the FIRST ELEMENT OF pair  
            SET spd TO the SECOND ELEMENT OF pair  
            ADD spd TO speed_heap  
            INCREMENT speed_sum BY spd  
            IF THE NUMBER OF ELEMENTS IN speed_heap IS GREATER THAN k THEN  
                REMOVE THE SMALLEST ELEMENT FROM speed_heap AND DECREMENT speed_sum BY THAT ELEMENT  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance IS GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        SET MODULUS TO the NUMBER ONE FOLLOWED BY nine ZEROES PLUS one  
        SET max_performance TO max_performance MODULO MODULUS  
        RETURN max_performance  
    END FUNCTION  
END CLASS
