CLASS Solution  
    FUNCTION maxSlidingWindow WITH parameters nums AND k  
        IF nums IS empty OR k IS EQUAL TO zero  
            RETURN an empty list  
        END IF  
        
        SET deq TO an empty double ended queue  
        SET result TO an empty list  
        
        FOR i FROM zero TO the length of nums MINUS one  
            IF deq IS NOT empty AND the first element of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the first element FROM deq  
            END IF  
            
            WHILE deq IS NOT empty AND the element at position the last element of deq OF nums IS LESS THAN the element at position i OF nums  
                REMOVE the last element FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position the first element of deq OF nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
