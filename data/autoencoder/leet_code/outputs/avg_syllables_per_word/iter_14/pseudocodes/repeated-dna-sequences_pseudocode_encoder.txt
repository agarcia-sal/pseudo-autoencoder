CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new default dictionary that returns zero for missing keys  
        SET length_of_s TO the LENGTH OF s  
        SET limit TO length_of_s MINUS nine  
        FOR index FROM zero TO limit MINUS one  
            SET sequence TO substring FROM position index TO position index PLUS ten MINUS one OF s  
            SET current_count TO the value at sequence IN sequence_count  
            INCREMENT current_count BY one  
            SET sequence_count AT sequence TO current_count  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each pair OF sequence AND count IN sequence_count  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
