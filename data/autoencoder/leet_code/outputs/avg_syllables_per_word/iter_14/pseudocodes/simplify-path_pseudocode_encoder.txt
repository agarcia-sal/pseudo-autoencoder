CLASS Solution  
    FUNCTION simplifyPath WITH parameter path  
        SET components TO SPLIT path BY the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component IS EQUAL TO the empty string OR component IS EQUAL TO a single period  
                CONTINUE to the next iteration  
            ELSE IF component IS EQUAL TO two periods  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character slash CONCATENATED WITH all elements of stack JOINED BY the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
