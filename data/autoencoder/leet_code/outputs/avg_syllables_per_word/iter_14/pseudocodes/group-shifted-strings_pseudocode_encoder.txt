CLASS Solution  
    FUNCTION groupStrings WITH PARAMETER strings  
        FUNCTION get_canonical_form WITH PARAMETER s  
            IF s IS empty  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            SET first_character TO the element at position zero OF s  
            FOR each character c IN s  
                SET difference TO the ASCII code of c MINUS the ASCII code of first_character  
                IF difference IS LESS THAN zero  
                    SET difference TO difference PLUS twenty six  
                END IF  
                APPEND the string representation of difference TO canonical_form  
            END FOR  
            SET result TO the concatenation of elements in canonical_form SEPARATED BY commas  
            RETURN result  
        END FUNCTION  
        
        SET groups TO a new dictionary with default value as empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form WITH PARAMETER s  
            APPEND s TO the list at key canonical_form IN groups  
        END FOR  
        
        SET grouped_lists TO the list of all values IN groups  
        RETURN grouped_lists  
    END FUNCTION  
END CLASS
