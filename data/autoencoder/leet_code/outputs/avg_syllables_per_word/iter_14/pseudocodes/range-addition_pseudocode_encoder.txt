CLASS Solution  
    FUNCTION getModifiedArray WITH PARAMETERS length updates  
        SET diff TO an array of zeroes WITH size length PLUS one  
        FOR each element consisting of startIdx endIdx inc IN updates  
            INCREMENT element at position startIdx OF diff BY inc  
            IF startIdx PLUS one LESS THAN OR EQUAL TO endIdx AND endIdx PLUS one LESS THAN length THEN  
                DECREMENT element at position endIdx PLUS one OF diff BY inc  
            END IF  
        END FOR  
        SET arr TO an array of zeroes WITH size length  
        SET element at position zero OF arr TO element at position zero OF diff  
        FOR index FROM one TO length MINUS one  
            SET element at position index OF arr TO element at position index MINUS one OF arr PLUS element at position index OF diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
