CLASS Solution  
    FUNCTION solveEquation WITH PARAMETER equation  
        FUNCTION parse_side WITH PARAMETER side  
            SET tokens TO the result of replacing plus signs in side WITH hash plus sign and replacing minus signs in side WITH hash minus sign then splitting the resulting string BY hash character  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS empty string  
                    CONTINUE to next iteration  
                END IF  
                IF token CONTAINS the character x  
                    IF token IS EQUAL TO the character x OR token IS EQUAL TO plus sign concatenated WITH character x  
                        INCREMENT x_count BY one  
                    ELSE IF token IS EQUAL TO minus sign concatenated WITH character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the integer conversion of token WITH character x REMOVED  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer conversion of token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  
              
        SET left_side AND right_side TO the result of splitting equation BY the equal sign  
        SET left_x AND left_num TO the result of calling parse_side WITH left_side  
        SET right_x AND right_num TO the result of calling parse_side WITH right_side  
              
        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  
              
        IF total_x IS EQUAL TO zero  
            IF total_num IS EQUAL TO zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string composed of character x concatenated WITH equal sign concatenated WITH the integer division of total_num BY total_x  
        END IF  
    END FUNCTION  
END CLASS
