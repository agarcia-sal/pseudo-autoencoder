CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
  
        SET max_product TO element at position zero OF nums  
        SET min_product TO element at position zero OF nums  
        SET result TO element at position zero OF nums  
  
        FOR each num IN nums starting from position one TO the end  
            IF num IS LESS THAN zero  
                SET temporary TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary  
            END IF  
  
            IF num IS GREATER THAN OR EQUAL TO max_product MULTIPLIED BY num  
                SET max_product TO num  
            ELSE  
                SET max_product TO max_product MULTIPLIED BY num  
            END IF  
  
            IF num IS LESS THAN OR EQUAL TO min_product MULTIPLIED BY num  
                SET min_product TO num  
            ELSE  
                SET min_product TO min_product MULTIPLIED BY num  
            END IF  
  
            IF result IS LESS THAN max_product  
                SET result TO max_product  
            END IF  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
