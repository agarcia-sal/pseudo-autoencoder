CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a new deque containing one tuple of zero zero zero one zero representing tail row tail column head row head column steps  
        SET visited TO a new set containing one tuple of zero zero zero one representing visited positions  
        
        WHILE queue IS NOT EMPTY  
            REMOVE the element from the LEFT END OF queue AND UNPACK INTO r1 c1 r2 c2 steps  
            
            IF r1 EQUAL TO n MINUS one AND c1 EQUAL TO n MINUS two AND r2 EQUAL TO n MINUS one AND c2 EQUAL TO n MINUS one  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND element at position r1 and c1 PLUS one OF grid EQUAL TO zero AND element at position r2 and c2 PLUS one OF grid EQUAL TO zero AND tuple of r1 c1 PLUS one r2 c2 PLUS one NOT IN visited  
                ADD tuple of r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND tuple of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one and c1 OF grid EQUAL TO zero AND element at position r2 PLUS one and c2 OF grid EQUAL TO zero AND tuple of r1 PLUS one c1 r2 PLUS one c2 NOT IN visited  
                ADD tuple of r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND tuple of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUAL TO r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one and c1 OF grid EQUAL TO zero AND element at position r1 PLUS one and c2 OF grid EQUAL TO zero AND tuple of r1 c1 r1 PLUS one c1 NOT IN visited  
                ADD tuple of r1 c1 r1 PLUS one c1 TO visited  
                APPEND tuple of r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUAL TO c2 AND c1 PLUS one LESS THAN n AND element at position r1 and c1 PLUS one OF grid EQUAL TO zero AND element at position r2 and c2 PLUS one OF grid EQUAL TO zero AND tuple of r1 c1 r1 c1 PLUS one NOT IN visited  
                ADD tuple of r1 c1 r1 c1 PLUS one TO visited  
                APPEND tuple of r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
