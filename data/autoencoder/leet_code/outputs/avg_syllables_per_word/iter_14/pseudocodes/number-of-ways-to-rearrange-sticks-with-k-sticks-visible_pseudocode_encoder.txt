CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        CREATE dp AS a two dimensional list of zeros WITH dimensions n PLUS one BY k PLUS one  
        
        SET element at position zero and zero OF dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position i and j OF dp TO the remainder WHEN the sum OF element at position i MINUS one and j MINUS one OF dp PLUS the product OF i MINUS one AND element at position i MINUS one and j OF dp IS divided BY MOD  
            END FOR  
        END FOR  
        
        RETURN element at position n and k OF dp  
    END FUNCTION  
END CLASS
