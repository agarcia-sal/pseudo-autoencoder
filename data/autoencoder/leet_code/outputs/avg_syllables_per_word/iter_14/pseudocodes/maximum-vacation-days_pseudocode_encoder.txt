CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero OF days  
        
        SET dp TO a list of length n FILLED WITH negative one  
        SET the element at position zero OF dp TO zero  
        
        SET graph TO an empty mapping FROM integer TO list  
        FOR i FROM zero TO n MINUS one  
            ASSIGN an empty list TO the element at key i OF graph  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j OF the element at position i OF flights EQUAL TO one THEN  
                    APPEND j TO the element at key i OF graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n FILLED WITH negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i OF dp EQUAL TO negative one THEN  
                    CONTINUE TO the next iteration  
                END IF  
                
                IF the element at position i OF new_dp LESS THAN dp at position i PLUS the element at position week OF the element at position i OF days THEN  
                    SET the element at position i OF new_dp TO dp at position i PLUS the element at position week OF the element at position i OF days  
                END IF  
                
                FOR each j IN the element at key i OF graph  
                    IF the element at position j OF new_dp LESS THAN dp at position i PLUS the element at position week OF the element at position j OF days THEN  
                        SET the element at position j OF new_dp TO dp at position i PLUS the element at position week OF the element at position j OF days  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        SET result TO the maximum value IN dp  
        RETURN result  
    END FUNCTION  
END CLASS
