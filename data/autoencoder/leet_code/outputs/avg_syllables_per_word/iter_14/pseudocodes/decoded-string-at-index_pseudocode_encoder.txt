CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each char IN s  
            IF char IS A DIGIT  
                SET length TO length MULTIPLIED BY THE NUMERICAL VALUE OF char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN THE REVERSE OF s  
            SET k TO k MODULO length  
            IF k EQUAL TO zero AND char IS AN ALPHABET CHARACTER  
                RETURN char  
            END IF  
            IF char IS A DIGIT  
                SET length TO length DIVIDED BY THE NUMERICAL VALUE OF char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
