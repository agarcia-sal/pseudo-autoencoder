CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 IS EQUAL TO str2  
            RETURN true  
        END IF  
        
        SET conversion_map TO an empty mapping  
        
        FOR each paired element char1 AND char2 IN corresponding positions FROM str1 AND str2  
            IF char1 IS a key IN conversion_map  
                IF the value FOR char1 IN conversion_map IS NOT EQUAL TO char2  
                    RETURN false  
                END IF  
            ELSE  
                ASSIGN char2 AS the value FOR char1 IN conversion_map  
            END IF  
        END FOR  
        
        IF the count OF unique characters IN str2 IS EQUAL TO twenty six  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
