CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EQUAL TO no object OR element at position zero OF matrix IS EQUAL TO no object  
            RETURN false  
        END IF  
        
        SET row TO zero  
        SET col TO the LENGTH OF element at position zero OF matrix MINUS one  
        
        WHILE row IS LESS THAN the LENGTH OF matrix AND col IS GREATER THAN OR EQUAL TO zero  
            IF element at position row AND position col OF matrix IS EQUAL TO target  
                RETURN true  
            ELSE IF element at position row AND position col OF matrix IS GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        
        RETURN false  
    END FUNCTION  
END CLASS
