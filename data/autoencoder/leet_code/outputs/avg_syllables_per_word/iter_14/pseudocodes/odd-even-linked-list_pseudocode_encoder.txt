CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the property val OF self TO val  
        SET the property next OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO A NEW instance OF ListNode INITIALIZED WITH the element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO A NEW instance OF ListNode INITIALIZED WITH val  
        SET the next property OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the expression that the val property OF p1 IS EQUAL TO the val property OF p2 AND the result OF is_same_list INVOKED WITH the next property OF p1 AND the next property OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETER head  
        IF head IS None OR the next property OF head IS None OR the next property OF the next property OF head IS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next property OF head  
        SET even_head TO even  
        WHILE even IS NOT None AND the next property OF even IS NOT None  
            SET the next property OF odd TO the next property OF even  
            SET odd TO the next property OF odd  
            SET the next property OF even TO the next property OF odd  
            SET even TO the next property OF even  
        END WHILE  
        SET the next property OF odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
