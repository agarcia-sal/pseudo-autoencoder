CLASS Solution  
    FUNCTION waysToPartition(nums LIST OF INTEGER, k INTEGER) RETURNS INTEGER  
        SET total_sum TO the SUM OF all elements IN nums  
        SET n TO the LENGTH OF nums  
        
        SET left_diff TO a new dictionary WITH DEFAULT INTEGER zero  
        SET right_diff TO a new dictionary WITH DEFAULT INTEGER zero  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one MINUS one  
            INCREMENT current_sum BY the element at position index OF nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT the value AT diff IN right_diff BY one  
        END FOR  
        
        SET max_ways TO the value AT zero IN right_diff  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one  
            INCREMENT current_sum BY the element at position index OF nums  
            
            SET diff_change TO k MINUS the element at position index OF nums  
            
            SET new_ways TO the value AT NEGATIVE diff_change IN left_diff PLUS the value AT diff_change IN right_diff  
            IF new_ways GREATER THAN max_ways THEN  
                SET max_ways TO new_ways  
            END IF  
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT the value AT diff IN right_diff BY one  
            INCREMENT the value AT diff IN left_diff BY one  
        END FOR  
        
        RETURN max_ways  
    END FUNCTION  
END CLASS
