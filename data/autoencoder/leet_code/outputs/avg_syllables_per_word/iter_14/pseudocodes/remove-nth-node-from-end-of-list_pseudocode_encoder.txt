CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET val ATTRIBUTE OF SELF TO val  
        SET next ATTRIBUTE OF SELF TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EQUAL TO empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH val SET TO element at position zero OF values  
    SET p TO head  
    FOR each val IN values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode WITH val SET TO val  
        SET next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN val ATTRIBUTE OF p1 IS EQUAL TO val ATTRIBUTE OF p2 AND is_same_list WITH PARAMETERS next ATTRIBUTE OF p1 AND next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO A NEW ListNode WITH val SET TO zero  
        SET next ATTRIBUTE OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        
        FOR i FROM zero TO n  
            SET first TO next ATTRIBUTE OF first  
        END FOR  
        
        WHILE first IS NOT None  
            SET first TO next ATTRIBUTE OF first  
            SET second TO next ATTRIBUTE OF second  
        END WHILE  
        
        SET next ATTRIBUTE OF second TO next ATTRIBUTE OF next ATTRIBUTE OF second  
        
        RETURN next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
