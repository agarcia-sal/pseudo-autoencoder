CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH val EQUAL TO element at position zero OF values  
    SET p TO head  
    FOR each val IN values starting from position one TO the end  
        SET node TO NEW ListNode WITH val EQUAL TO val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN (p1 dot val IS EQUAL TO p2 dot val) AND is_same_list WITH PARAMETERS p1 dot next p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS self head  
        SET dummy TO NEW ListNode WITH val EQUAL TO zero next EQUAL TO head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF head dot next IS NOT none AND head dot val IS EQUAL TO head dot next dot val  
                WHILE head dot next IS NOT none AND head dot val IS EQUAL TO head dot next dot val  
                    SET head TO head dot next  
                END WHILE  
                SET prev dot next TO head dot next  
            ELSE  
                SET prev TO prev dot next  
            END IF  
            SET head TO head dot next  
        END WHILE  
        RETURN dummy dot next  
    END FUNCTION  
END CLASS
