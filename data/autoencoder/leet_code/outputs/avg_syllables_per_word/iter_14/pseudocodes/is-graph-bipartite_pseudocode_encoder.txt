CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a LIST OF zero WITH THE LENGTH OF n  
        
        FUNCTION dfs(node, c)  
            SET the element at position node OF color TO c  
            FOR each neighbor IN the element at position node OF graph  
                IF the element at position neighbor OF color IS EQUAL TO c  
                    RETURN False  
                END IF  
                IF the element at position neighbor OF color IS EQUAL TO zero  
                    IF dfs(neighbor, c MULTIPLIED BY negative one) IS EQUAL TO False  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i OF color IS EQUAL TO zero  
                IF dfs(i, one) IS EQUAL TO False  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
