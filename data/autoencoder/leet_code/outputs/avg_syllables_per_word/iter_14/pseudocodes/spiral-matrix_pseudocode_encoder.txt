CLASS Solution  
    FUNCTION spiralOrder WITH PARAMETER matrix  
        IF matrix IS EMPTY OR the LENGTH OF matrix AT position zero IS zero  
            RETURN an EMPTY list  
        END IF  
        
        SET result TO an EMPTY list  
        SET top TO zero  
        SET bottom TO the LENGTH OF matrix MINUS one  
        SET left TO zero  
        SET right TO the LENGTH OF matrix AT position zero MINUS one  
        
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR index i FROM left TO right  
                APPEND the element AT position top OF matrix AT position i TO result  
            END FOR  
            INCREMENT top BY one  
            
            FOR index i FROM top TO bottom  
                APPEND the element AT position i OF matrix AT position right TO result  
            END FOR  
            DECREMENT right BY one  
            
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR index i FROM right DOWN TO left  
                    APPEND the element AT position bottom OF matrix AT position i TO result  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            
            IF left IS LESS THAN OR EQUAL TO right  
                FOR index i FROM bottom DOWN TO top  
                    APPEND the element AT position i OF matrix AT position left TO result  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
