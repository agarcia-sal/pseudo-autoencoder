CLASS Solution  
    FUNCTION optimalDivision WITH parameter nums  
        IF the LENGTH OF nums EQUAL TO one  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums  
        END IF  
        IF the LENGTH OF nums EQUAL TO two  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums CONCATENATED WITH the character REPRESENTING DIVISION CONCATENATED WITH the STRING REPRESENTATION OF element at position one OF nums  
        END IF  
        RETURN the STRING REPRESENTATION OF element at position zero OF nums CONCATENATED WITH the character REPRESENTING DIVISION CONCATENATED WITH an OPENING PARENTHESIS CONCATENATED WITH the STRING REPRESENTATIONS OF elements FROM position one TO the END OF nums JOINED TOGETHER BY the character REPRESENTING DIVISION CONCATENATED WITH a CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
