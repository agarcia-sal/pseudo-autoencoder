CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS EQUAL TO nothing OR board IS EMPTY  
            RETURN false  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero OF board  

        FUNCTION dfs(x, y, index)  
            IF index IS EQUAL TO the LENGTH OF word  
                RETURN true  
            END IF  

            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO m OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO n OR the element at position x OF visited AT position y IS true OR the element at position x OF board AT position y IS NOT EQUAL TO the element at position index OF word  
                RETURN false  
            END IF  

            SET the element at position x OF visited AT position y TO true  

            FOR each pair dx, dy IN the list containing pairs minus one and zero, one and zero, zero and minus one, zero and one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one) IS true  
                    RETURN true  
                END IF  
            END FOR  

            SET the element at position x OF visited AT position y TO false  
            RETURN false  
        END FUNCTION  

        SET visited TO a two-dimensional list with m rows and n columns where each element IS false  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero) IS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  

        RETURN false  
    END FUNCTION  
END CLASS
