CLASS Solution  
    FUNCTION letterCasePermutation WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER i  
            IF i IS GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND concatenation OF all elements IN t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH PARAMETER i PLUS one  
            IF element at position i OF t IS AN ALPHABETICAL CHARACTER  
                SET element at position i OF t TO the character REPRESENTED BY the NUMERIC VALUE OF element at position i OF t EXCLUSIVE OR with thirty two  
                CALL dfs WITH PARAMETER i PLUS one  
            END IF  
        END FUNCTION  
  
        SET t TO a list REPRESENTING all characters OF s  
        SET ans TO an empty list  
        CALL dfs WITH PARAMETER zero  
        RETURN ans  
    END FUNCTION  
END CLASS
