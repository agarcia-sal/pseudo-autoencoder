CLASS Solution
    FUNCTION splitArraySameAverage(nums)
        SET n TO the LENGTH OF nums
        SET total_sum TO the SUM OF nums
        
        IF n EQUAL TO one
            RETURN False
        END IF
        
        SET dp TO a LIST OF EMPTY SETS WITH the LENGTH OF n DIVIDED BY two PLUS one
        ADD zero TO the ELEMENT AT position zero OF dp
        
        FOR each num IN nums
            FOR size FROM n DIVIDED BY two DOWN TO one
                FOR each s IN the ELEMENT AT position size MINUS one OF dp
                    SET new_sum TO s PLUS num
                    IF new_sum MULTIPLIED BY n EQUAL TO size MULTIPLIED BY total_sum
                        RETURN True
                    END IF
                    ADD new_sum TO the ELEMENT AT position size OF dp
                END FOR
            END FOR
        END FOR
        
        RETURN False
    END FUNCTION
END CLASS
