CLASS Solution  
    FUNCTION findMaxValueOfEquation WITH PARAMETERS points AND k  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        
        FOR each element consisting of xj AND yj IN points  
            WHILE deque_ is not empty AND the difference between xj AND the element at front position one of deque_ IS GREATER THAN k  
                REMOVE element from the front of deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND the sum of yj PLUS xj PLUS the element at front position zero of deque_  
            END IF  
            
            WHILE deque_ is not empty AND the difference between yj AND xj IS GREATER THAN OR EQUAL TO the element at last position zero of deque_  
                REMOVE element from the back of deque_  
            END WHILE  
            
            APPEND the pair consisting of the difference between yj AND xj AND xj TO the back of deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
