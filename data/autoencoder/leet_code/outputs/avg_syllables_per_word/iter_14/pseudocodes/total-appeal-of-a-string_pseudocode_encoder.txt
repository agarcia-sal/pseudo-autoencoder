CLASS Solution  
    FUNCTION appealSum WITH PARAMETER s  
        SET list last TO a new list of length twenty six WITH EACH element SET TO minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        SET position_counter TO zero  
        FOR each character IN s  
            SET index TO the Unicode code value of character MINUS the Unicode code value of the lowercase letter a  
            SET difference TO position_counter MINUS element at position index OF list last  
            INCREMENT current_appeal BY difference  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index OF list last TO position_counter  
            INCREMENT position_counter BY one  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
