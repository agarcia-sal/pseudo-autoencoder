CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix IS EQUAL TO null OR the LENGTH OF matrix IS EQUAL TO zero OR the LENGTH OF element at position zero OF matrix IS EQUAL TO zero  
            RETURN  
        END IF  

        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero OF matrix  

        SET first_row_has_zero TO false  
        FOR each j FROM zero TO n MINUS one  
            IF element at position j OF element at position zero OF matrix IS EQUAL TO zero  
                SET first_row_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        SET first_col_has_zero TO false  
        FOR each i FROM zero TO m MINUS one  
            IF element at position zero OF element at position i OF matrix IS EQUAL TO zero  
                SET first_col_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                IF element at position j OF element at position i OF matrix IS EQUAL TO zero  
                    SET element at position zero OF element at position i OF matrix TO zero  
                    SET element at position j OF element at position zero OF matrix TO zero  
                END IF  
            END FOR  
        END FOR  

        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                IF element at position zero OF element at position i OF matrix IS EQUAL TO zero OR element at position j OF element at position zero OF matrix IS EQUAL TO zero  
                    SET element at position j OF element at position i OF matrix TO zero  
                END IF  
            END FOR  
        END FOR  

        IF first_row_has_zero IS true  
            FOR each j FROM zero TO n MINUS one  
                SET element at position j OF element at position zero OF matrix TO zero  
            END FOR  
        END IF  

        IF first_col_has_zero IS true  
            FOR each i FROM zero TO m MINUS one  
                SET element at position zero OF element at position i OF matrix TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
