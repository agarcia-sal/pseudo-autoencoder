CLASS Solution  
    FUNCTION addOperators(num, target)  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUAL TO the LENGTH OF num  
                IF value EQUAL TO target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO the SUBSTRING FROM index TO i MINUS one OF num  
                IF the LENGTH OF operand GREATER THAN one AND the FIRST CHARACTER OF operand EQUAL TO zero  
                    CONTINUE TO next iteration  
                END IF  
                
                IF index EQUAL TO zero  
                    CALL dfs WITH ARGUMENTS i, operand, INTEGER OF operand, INTEGER OF operand  
                ELSE  
                    CALL dfs WITH ARGUMENTS i, CONCATENATION OF path PLUS the SINGLE CHARACTER plus PLUS operand, value PLUS INTEGER OF operand, INTEGER OF operand  
                    CALL dfs WITH ARGUMENTS i, CONCATENATION OF path PLUS the SINGLE CHARACTER minus PLUS operand, value MINUS INTEGER OF operand, NEGATIVE INTEGER OF operand  
                    CALL dfs WITH ARGUMENTS i, CONCATENATION OF path PLUS the SINGLE CHARACTER multiplied BY PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY INTEGER OF operand, prev MULTIPLIED BY INTEGER OF operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH ARGUMENTS zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
