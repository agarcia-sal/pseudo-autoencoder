CLASS Solution  
    FUNCTION asteroidCollision WITH PARAMETER asteroids  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND element at position the last index OF stack IS GREATER THAN zero  
                IF element at position the last index OF stack IS LESS THAN the NEGATION OF asteroid  
                    REMOVE the last element FROM stack  
                    CONTINUE TO the next iteration OF the WHILE loop  
                ELSE IF element at position the last index OF stack IS EQUAL TO the NEGATION OF asteroid  
                    REMOVE the last element FROM stack  
                END IF  
                BREAK OUT OF the WHILE loop  
            END WHILE  
            IF the WHILE loop was not broken due to a BREAK statement  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
