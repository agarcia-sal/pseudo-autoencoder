CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUAL TO element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE ELEMENT FROM FRONT OF queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS self root  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO RESULT OF CALLING dfs WITH PARAMETER node.left  
            SET right_node AND right_depth TO RESULT OF CALLING dfs WITH PARAMETER node.right  
            IF left_depth IS GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth IS GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN THE FIRST ELEMENT OF THE RESULT OF CALLING dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
