CLASS Solution  
    FUNCTION movesToChessboard WITH PARAMETER board  
        SET N TO the LENGTH OF board  
          
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
          
        FOR each i FROM zero TO N MINUS one  
            FOR each j FROM zero TO N MINUS one  
                IF the VALUE AT position i OF board AND xor WITH the VALUE AT position zero OF row i OF board AND xor WITH the VALUE AT position j OF row zero OF board AND xor WITH the VALUE AT position zero OF row zero OF board IS NOT EQUAL TO zero  
                    RETURN negative one  
                END IF  
            END FOR  
            SET rowSum TO rowSum PLUS the VALUE AT position zero OF row i OF board  
            SET colSum TO colSum PLUS the VALUE AT position i OF row zero OF board  
            IF the VALUE AT position zero OF row i OF board IS EQUAL TO the REMAINDER OF i DIVIDED BY two  
                INCREMENT rowSwap BY one  
            END IF  
            IF the VALUE AT position i OF row zero OF board IS EQUAL TO the REMAINDER OF i DIVIDED BY two  
                INCREMENT colSwap BY one  
            END IF  
        END FOR  
          
        IF the INTEGER DIVISION OF N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the INTEGER DIVISION OF N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the INTEGER DIVISION OF N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the INTEGER DIVISION OF N PLUS one BY two  
            RETURN negative one  
        END IF  
          
        IF the REMAINDER OF N DIVIDED BY two IS EQUAL TO one  
            IF the REMAINDER OF rowSwap DIVIDED BY two IS EQUAL TO one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the REMAINDER OF colSwap DIVIDED BY two IS EQUAL TO one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
          
        RETURN the INTEGER DIVISION OF rowSwap PLUS colSwap BY two  
    END FUNCTION  
END CLASS
