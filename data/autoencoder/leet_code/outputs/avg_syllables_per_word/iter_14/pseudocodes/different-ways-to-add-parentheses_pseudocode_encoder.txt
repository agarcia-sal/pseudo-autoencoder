CLASS Solution  
    FUNCTION diffWaysToCompute WITH PARAMETER expression  
        FUNCTION compute WITH PARAMETERS left right operator  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator IS EQUAL TO the plus symbol  
                        APPEND l PLUS r TO results  
                    ELSE IF operator IS EQUAL TO the minus symbol  
                        APPEND l MINUS r TO results  
                    ELSE IF operator IS EQUAL TO the multiplication symbol  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper WITH PARAMETER sub_expr  
            IF sub_expr IS composed only of digits  
                RETURN list containing integer conversion OF sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR index FROM zero TO the length OF sub_expr MINUS one  
                IF character at position index OF sub_expr IS EQUAL TO any of the symbols plus OR minus OR multiplication  
                    SET left_results TO helper WITH parameter substring FROM start TO index MINUS one OF sub_expr  
                    SET right_results TO helper WITH parameter substring FROM index PLUS one TO end OF sub_expr  
                    EXTEND results BY compute WITH PARAMETERS left_results right_results character at position index OF sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        RETURN helper WITH PARAMETER expression  
    END FUNCTION  
END CLASS
