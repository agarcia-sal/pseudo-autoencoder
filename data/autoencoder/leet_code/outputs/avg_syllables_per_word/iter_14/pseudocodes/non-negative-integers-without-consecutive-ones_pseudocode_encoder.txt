CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO the BINARY REPRESENTATION OF n STARTING FROM the SECOND CHARACTER  
        SET length TO the LENGTH OF bin_rep  
        
        INITIALIZE dp AS A LIST OF LENGTH length PLUS one WITH ALL ELEMENTS SET TO zero  
        SET the ELEMENT AT POSITION zero OF dp TO one  
        SET the ELEMENT AT POSITION one OF dp TO two  
        
        FOR i FROM two TO length INCLUSIVE  
            SET the ELEMENT AT POSITION i OF dp TO the ELEMENT AT POSITION i MINUS one OF dp PLUS the ELEMENT AT POSITION i MINUS two OF dp  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR i FROM zero TO length MINUS one  
            IF the ELEMENT AT POSITION i OF bin_rep IS EQUAL TO one AS A CHARACTER  
                INCREMENT result BY the ELEMENT AT POSITION length MINUS i MINUS one OF dp  
                IF prev_bit IS EQUAL TO one  
                    DECREMENT result BY one  
                    BREAK OUT OF the LOOP  
                END IF  
            END IF  
            SET prev_bit TO one IF the ELEMENT AT POSITION i OF bin_rep IS EQUAL TO one AS A CHARACTER ELSE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
