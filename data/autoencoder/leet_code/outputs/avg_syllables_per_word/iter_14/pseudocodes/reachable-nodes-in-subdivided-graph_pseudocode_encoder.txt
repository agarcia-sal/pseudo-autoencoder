CLASS Solution  
    FUNCTION reachableNodes(edges, maxMoves, n)  
        SET graph TO an empty mapping from each integer from zero TO n MINUS one TO an empty list  
        FOR each element consisting of u, v, and cnt IN edges  
            APPEND the pair consisting of v and cnt TO the list associated WITH key u IN graph  
            APPEND the pair consisting of u and cnt TO the list associated WITH key v IN graph  
        END FOR  

        SET distances TO a list of length n FILLED WITH positive infinity  
        SET the element at position zero OF distances TO zero  
        SET priority_queue TO a list CONTAINING a single pair consisting of zero and zero  

        WHILE priority_queue IS NOT empty  
            REMOVE the element with the smallest first value FROM priority_queue AND ASSIGN its first value TO dist AND its second value TO node  
            IF dist IS GREATER THAN the element at position node OF distances  
                CONTINUE TO the next iteration  
            END IF  
            FOR each element consisting of neighbor and cnt IN the list associated WITH key node IN graph  
                SET new_dist TO dist PLUS cnt PLUS one  
                IF new_dist IS LESS THAN the element at position neighbor OF distances  
                    SET the element at position neighbor OF distances TO new_dist  
                    ADD the pair consisting of new_dist and neighbor INTO priority_queue MAINTAINING priority order BY first value  
                END IF  
            END FOR  
        END WHILE  

        SET reachable_nodes TO zero  
        FOR each distance_value IN distances  
            IF distance_value IS LESS THAN OR EQUAL TO maxMoves  
                INCREMENT reachable_nodes BY one  
            END IF  
        END FOR  

        SET used_edges TO an empty mapping  
        FOR each element consisting of u, v, and cnt IN edges  
            SET moves_u TO the maximum BETWEEN zero AND maxMoves MINUS the element at position u OF distances  
            SET moves_v TO the maximum BETWEEN zero AND maxMoves MINUS the element at position v OF distances  
            SET reachable_new_nodes TO the minimum BETWEEN cnt AND moves_u PLUS moves_v  
            INCREMENT reachable_nodes BY reachable_new_nodes  
            ASSIGN reachable_new_nodes TO the mapping key consisting of the pair u and v IN used_edges  
        END FOR  

        RETURN reachable_nodes  
    END FUNCTION  
END CLASS
