CLASS Solution  
    FUNCTION coinChange(coins AS list OF integer, amount AS integer) AS integer  
        SET dp TO a list of length amount PLUS one where each element IS amount PLUS one  
        SET the element at position zero OF dp TO zero  
        FOR a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO zero  
                    SET the element at position a OF dp TO the smaller BETWEEN the element at position a OF dp AND the element at position a MINUS coin OF dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF the element at position amount OF dp IS EQUAL TO amount PLUS one  
            RETURN minus one  
        ELSE  
            RETURN the element at position amount OF dp  
        END IF  
    END FUNCTION  
END CLASS
