CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(nums, k)  
        IF k IS LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
          
        SET product TO one  
        SET left TO zero  
        SET count TO zero  
          
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET product TO product MULTIPLIED BY element at position right OF nums  
              
            WHILE product IS GREATER THAN OR EQUAL TO k AND left IS LESS THAN OR EQUAL TO right  
                SET product TO product DIVIDED BY element at position left OF nums  
                INCREMENT left BY one  
            END WHILE  
              
            INCREMENT count BY right MINUS left PLUS one  
        END FOR  
          
        RETURN count  
    END FUNCTION  
END CLASS
