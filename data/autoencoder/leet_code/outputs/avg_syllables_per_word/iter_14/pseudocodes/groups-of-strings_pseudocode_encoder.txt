CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO a list containing values FROM zero TO size MINUS one  
        SET self.rank TO a list containing zero repeated size times  
        SET self.size TO a list containing one repeated size times  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u OF self.parent IS NOT EQUAL TO u  
            SET element at position u OF self.parent TO self.find(element at position u OF self.parent)  
        END IF  
        RETURN element at position u OF self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU OF self.rank IS GREATER THAN element at position rootV OF self.rank  
                SET element at position rootV OF self.parent TO rootU  
                INCREMENT element at position rootU OF self.size BY element at position rootV OF self.size  
            ELSE IF element at position rootU OF self.rank IS LESS THAN element at position rootV OF self.rank  
                SET element at position rootU OF self.parent TO rootV  
                INCREMENT element at position rootV OF self.size BY element at position rootU OF self.size  
            ELSE  
                SET element at position rootV OF self.parent TO rootU  
                INCREMENT element at position rootU OF self.size BY element at position rootV OF self.size  
                INCREMENT element at position rootU OF self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each character IN word  
                SET character_code TO the Unicode code of character  
                SET base_code TO the Unicode code of the letter a  
                SET position TO character_code MINUS base_code  
                SET mask TO mask OR one shifted left BY position  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO an empty dictionary with default value being an empty list  
        SET n TO the length of words  
        SET uf TO a new UnionFind object initialized with n  

        FOR each index i and word IN words indexed  
            SET mask TO bitmask(word)  
            APPEND i TO the value associated with key mask IN word_map  
        END FOR  

        FOR each index i and word IN words indexed  
            SET mask TO bitmask(word)  

            FOR position j FROM zero TO twenty five  
                // Delete one letter  
                SET new_mask TO mask XOR one shifted left BY j  
                IF new_mask IS a key in word_map  
                    FOR each k IN the list associated with new_mask IN word_map  
                        CALL uf.union(i, k)  
                    END FOR  
                END IF  

                // Add one letter  
                IF mask AND one shifted left BY j IS EQUAL TO zero  
                    SET new_mask TO mask OR one shifted left BY j  
                    IF new_mask IS a key in word_map  
                        FOR each k IN the list associated with new_mask IN word_map  
                            CALL uf.union(i, k)  
                        END FOR  
                    END IF  
                END IF  

                // Replace one letter  
                IF mask AND one shifted left BY j IS NOT EQUAL TO zero  
                    FOR position k FROM zero TO twenty five  
                        IF mask AND one shifted left BY k IS EQUAL TO zero  
                            SET new_mask TO (mask XOR one shifted left BY j) OR one shifted left BY k  
                            IF new_mask IS a key in word_map  
                                FOR each l IN the list associated with new_mask IN word_map  
                                    CALL uf.union(i, l)  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO an empty dictionary with default value being zero  
        FOR index i FROM zero TO n MINUS one  
            SET root TO uf.find(i)  
            INCREMENT element at key root in root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value in root_sizes  
        SET num_groups TO the number of keys in root_sizes  

        RETURN a list containing num_groups FOLLOWED BY max_group_size  
    END FUNCTION  
END CLASS
