CLASS Solution  
    FUNCTION evalRPN WITH PARAMETER tokens  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUAL TO PLUS OR token EQUAL TO MINUS OR token EQUAL TO MULTIPLICATION SYMBOL OR token EQUAL TO DIVISION SYMBOL  
                SET b TO the last element REMOVED FROM stack  
                SET a TO the last element REMOVED FROM stack  
                IF token EQUAL TO PLUS  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUAL TO MINUS  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUAL TO MULTIPLICATION SYMBOL  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUAL TO DIVISION SYMBOL  
                    SET division_result TO a DIVIDED BY b  
                    SET truncated_result TO integer part OF division_result TOWARDS zero  
                    APPEND truncated_result TO stack  
                END IF  
            ELSE  
                CONVERT token TO integer  
                APPEND converted integer TO stack  
            END IF  
        END FOR  
        RETURN element at position zero OF stack  
    END FUNCTION  
END CLASS
