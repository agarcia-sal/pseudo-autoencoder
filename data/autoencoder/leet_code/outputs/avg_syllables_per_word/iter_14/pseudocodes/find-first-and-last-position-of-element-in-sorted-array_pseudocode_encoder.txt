CLASS Solution  
    FUNCTION searchRange(nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY two using integer division  
                IF element at position mid OF nums IS LESS THAN target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY two using integer division  
                IF element at position mid OF nums IS LESS THAN OR EQUAL TO target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN right  
        END FUNCTION  

        SET left_index TO the RESULT OF find_left WITH nums AND target  
        SET right_index TO the RESULT OF find_right WITH nums AND target  

        IF left_index IS LESS THAN OR EQUAL TO right_index AND zero IS LESS THAN OR EQUAL TO right_index AND right_index IS LESS THAN the LENGTH OF nums AND element at position left_index OF nums IS EQUAL TO target  
            RETURN list CONTAINING left_index AND right_index  
        ELSE  
            RETURN list CONTAINING negative one AND negative one  
        END IF  
    END FUNCTION  
END CLASS
