CLASS Solution  
    FUNCTION findMin WITH PARAMETER nums  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF the SUM of left AND right BY two  
            
            IF the element at position mid OF nums IS GREATER THAN the element at position right OF nums  
                SET left TO mid PLUS one  
            ELSE IF the element at position mid OF nums IS LESS THAN the element at position right OF nums  
                SET right TO mid  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN the element at position left OF nums  
    END FUNCTION  
END CLASS
