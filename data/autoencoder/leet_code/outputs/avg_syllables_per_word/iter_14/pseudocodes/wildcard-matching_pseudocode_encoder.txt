CLASS Solution  
    FUNCTION isMatch WITH PARAMETERS s p  
        INITIALIZE dp TO a two-dimensional list of boolean values FALSE with dimensions LENGTH OF s PLUS one BY LENGTH OF p PLUS one  
        SET element at position zero ZERO OF dp TO a list with element at position zero ZERO set to TRUE  
        FOR each position j FROM one TO LENGTH OF p  
            IF element at position j MINUS one OF p IS EQUAL TO the character represented by asterisk  
                SET element at position zero j OF dp TO element at position zero j MINUS one OF dp  
            END IF  
        END FOR  
        FOR each position i FROM one TO LENGTH OF s  
            FOR each position j FROM one TO LENGTH OF p  
                IF element at position j MINUS one OF p IS EQUAL TO the character represented by asterisk  
                    SET element at position i j OF dp TO element at position i MINUS one j OF dp OR element at position i j MINUS one OF dp  
                ELSE IF element at position j MINUS one OF p IS EQUAL TO the character represented by question mark OR element at position i MINUS one OF s IS EQUAL TO element at position j MINUS one OF p  
                    SET element at position i j OF dp TO element at position i MINUS one j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s LENGTH OF p OF dp  
    END FUNCTION  
END CLASS
