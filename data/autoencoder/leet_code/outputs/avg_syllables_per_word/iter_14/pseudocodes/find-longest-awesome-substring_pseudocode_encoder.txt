CLASS Solution  
    FUNCTION longestAwesome WITH PARAMETER s  
        SET prefix_map TO a new mapping WITH key zero MAPPED TO value negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN s  
            SET digit TO integer value OF character char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR WITH one SHIFTED LEFT BY digit positions  
            
            IF prefix_xor IS IN prefix_map  
                SET max_length TO the GREATER VALUE OF max_length AND index i MINUS the VALUE MAPPED BY prefix_xor IN prefix_map  
            END IF  
            
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR WITH one SHIFTED LEFT BY j positions  
                IF flipped IS IN prefix_map  
                    SET max_length TO the GREATER VALUE OF max_length AND index i MINUS the VALUE MAPPED BY flipped IN prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor IS NOT IN prefix_map  
                SET the VALUE MAPPED BY prefix_xor IN prefix_map TO index i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
