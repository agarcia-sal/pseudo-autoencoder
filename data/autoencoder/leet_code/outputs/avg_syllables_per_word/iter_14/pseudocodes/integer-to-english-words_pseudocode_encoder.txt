CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUAL TO zero  
            RETURN the word zero  
        END IF  
  
        SET below_twenty TO the list of words empty string, one, two, three, four, five, six, seven, eight, nine, ten, eleven, twelve, thirteen, fourteen, fifteen, sixteen, seventeen, eighteen, nineteen IN ORDER  
        SET tens TO the list of words empty string, empty string, twenty, thirty, forty, fifty, sixty, seventy, eighty, ninety IN ORDER  
        SET thousands TO the list of words empty string, thousand, million, billion IN ORDER  
  
        FUNCTION helper(n)  
            IF n EQUAL TO zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the word at position n OF below_twenty PLUS a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN the word at position n DIVIDED BY ten OF tens PLUS a single space PLUS the RESULT OF helper with the remainder of n DIVIDED BY ten  
            ELSE  
                RETURN the word at position n DIVIDED BY one hundred OF below_twenty PLUS the word hundred PLUS a single space PLUS the RESULT OF helper with the remainder of n DIVIDED BY one hundred  
            END IF  
        END FUNCTION  
  
        SET result TO empty string  
        SET i TO zero  
  
        WHILE num GREATER THAN zero  
            IF the remainder of num DIVIDED BY one thousand NOT EQUAL TO zero  
                SET result TO the RESULT OF helper with the remainder of num DIVIDED BY one thousand PLUS the word at position i OF thousands PLUS a single space PLUS result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
  
        RETURN the string result WITH LEADING AND TRAILING SPACES REMOVED  
    END FUNCTION  
END CLASS
