CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO THE RESULT OF SORTING THE TUPLES COMPOSED OF elements FROM indices SOURCES AND targets ORDERED BY THE FIRST ELEMENT OF EACH TUPLE  
        SET result TO AN EMPTY LIST  
        SET prev_end TO zero  
        FOR each index_source_target IN replacements  
            SET index TO the FIRST element OF index_source_target  
            SET source TO the SECOND element OF index_source_target  
            SET target TO the THIRD element OF index_source_target  
            APPEND the SUBSTRING OF s FROM prev_end TO index TO result  
            IF the SUBSTRING OF s FROM index TO index PLUS the LENGTH OF source IS EQUAL TO source THEN  
                APPEND target TO result  
            ELSE  
                APPEND the SUBSTRING OF s FROM index TO index PLUS the LENGTH OF source TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the SUBSTRING OF s FROM prev_end TO THE END OF s TO result  
        RETURN THE CONCATENATION OF ALL ELEMENTS OF result INTO A SINGLE STRING  
    END FUNCTION  
END CLASS
