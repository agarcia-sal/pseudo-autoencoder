CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO the SUM OF all elements IN arr  
        
        IF total_ones IS EQUAL TO zero  
            RETURN a list WITH elements zero AND two  
        END IF  
        
        IF total_ones MODULO three IS NOT EQUAL TO zero  
            RETURN a list WITH elements minus one AND minus one  
        END IF  
        
        SET part_ones TO total_ones DIVIDED BY three  
        
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR each index FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position index OF arr IS EQUAL TO one  
                INCREMENT count_ones BY one  
                IF count_ones IS EQUAL TO one  
                    SET first_part_index TO index  
                END IF  
                IF count_ones IS EQUAL TO part_ones  
                    BREAK the FOR loop  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR each index FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position index OF arr IS EQUAL TO one  
                INCREMENT count_ones BY one  
                IF count_ones IS EQUAL TO part_ones PLUS one  
                    SET second_part_index TO index  
                    BREAK the FOR loop  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR each index FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position index OF arr IS EQUAL TO one  
                INCREMENT count_ones BY one  
                IF count_ones IS EQUAL TO part_ones MULTIPLIED BY two PLUS one  
                    SET third_part_index TO index  
                    BREAK the FOR loop  
                END IF  
            END IF  
        END FOR  
        
        SET n TO the LENGTH OF arr MINUS third_part_index  
        IF first_part_index PLUS n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           second_part_index PLUS n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           the SUBLIST OF arr FROM first_part_index TO first_part_index PLUS n MINUS one IS EQUAL TO the SUBLIST OF arr FROM second_part_index TO second_part_index PLUS n MINUS one AND  
           the SUBLIST OF arr FROM second_part_index TO second_part_index PLUS n MINUS one IS EQUAL TO the SUBLIST OF arr FROM third_part_index TO THE END  
            RETURN a list WITH elements first_part_index PLUS n MINUS one AND second_part_index PLUS n  
        END IF  
        
        RETURN a list WITH elements minus one AND minus one  
    END FUNCTION  
END CLASS
