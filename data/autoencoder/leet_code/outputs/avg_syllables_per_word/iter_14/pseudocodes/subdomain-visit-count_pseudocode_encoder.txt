CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a dictionary that returns zero for missing keys  
        
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count_part AND domain_part BY the first space  
            CONVERT count_part TO an integer AND SET TO count  
            SPLIT domain_part INTO subdomains BY the character dot  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                JOIN the elements FROM index TO the end OF subdomains USING the character dot AND SET TO subdomain  
                INCREMENT the VALUE at key subdomain IN domain_count BY count  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        FOR each domain, count IN the ITEMS OF domain_count  
            CONCATENATE the string representation of count, one space, AND domain AND APPEND TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
