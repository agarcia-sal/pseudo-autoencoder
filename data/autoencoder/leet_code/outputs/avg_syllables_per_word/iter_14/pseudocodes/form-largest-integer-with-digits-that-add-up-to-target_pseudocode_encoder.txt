CLASS Solution  
    FUNCTION largestNumber(cost AS list OF integers, target AS integer)  
        CREATE cost_to_digit AS dictionary MAPPING each element c FROM cost TO STRING OF its position index PLUS one IN cost  
        CREATE dp AS list WITH element at position zero AS empty string AND element at positions from one TO target AS string minus one  
        FOR t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c OF dp IS NOT string minus one  
                    SET candidate TO concatenation of d AND element at position t MINUS c OF dp  
                    IF element at position t OF dp IS string minus one OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t OF dp OR (LENGTH OF candidate EQUAL TO LENGTH OF element at position t OF dp AND candidate GREATER THAN element at position t OF dp)  
                        SET element at position t OF dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position target OF dp IS NOT string minus one  
            RETURN element at position target OF dp  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
