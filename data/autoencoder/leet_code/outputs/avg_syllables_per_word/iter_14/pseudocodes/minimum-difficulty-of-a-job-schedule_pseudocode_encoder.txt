CLASS Solution  
    FUNCTION minDifficulty WITH jobDifficulty AND d  
        SET n TO the LENGTH OF jobDifficulty  
  
        IF n IS LESS THAN d  
            RETURN the NEGATIVE of one  
        END IF  
  
        FUNCTION dp WITH i AND d  
            IF d IS EQUAL TO one  
                RETURN the MAXIMUM value IN the SUBLIST OF jobDifficulty FROM position i TO the END  
            END IF  
  
            SET min_difficulty TO infinity  
            SET current_max TO zero  
  
            FOR each j FROM i TO n MINUS d PLUS one  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND the ELEMENT AT position j OF jobDifficulty  
                SET candidate TO current_max PLUS THE RESULT OF dp WITH j PLUS one AND d MINUS one  
                IF candidate IS LESS THAN min_difficulty  
                    SET min_difficulty TO candidate  
                END IF  
            END FOR  
  
            RETURN min_difficulty  
        END FUNCTION  
  
        RETURN the RESULT OF dp WITH zero AND d  
    END FUNCTION  
END CLASS
