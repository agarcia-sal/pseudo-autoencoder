CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the JOINING OF the SORTED ELEMENTS OF s  
        END IF  
        SET n TO the LENGTH OF s  
        SET smallest TO s  
        FOR i FROM one TO n MINUS one  
            SET rotated TO the CONCATENATION OF the SUBSTRING OF s FROM position i TO the END AND the SUBSTRING OF s FROM position zero TO i MINUS one  
            IF rotated LESS THAN smallest  
                SET smallest TO rotated  
            END IF  
        END FOR  
        RETURN smallest  
    END FUNCTION  
END CLASS
