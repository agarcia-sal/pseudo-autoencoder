CLASS Solution  
    FUNCTION parseTernary WITH PARAMETER expression  
        FUNCTION evaluate WITH PARAMETER index  
            IF index IS EQUAL TO the LENGTH OF expression MINUS one OR the CHARACTER AT POSITION index PLUS one OF expression IS EQUAL TO colon  
                RETURN the CHARACTER AT POSITION index OF expression AND index PLUS two  
            END IF  
            
            SET true_part AND next_index TO THE RESULT OF CALLING evaluate WITH ARGUMENT index PLUS two  
            SET false_part AND final_index TO THE RESULT OF CALLING evaluate WITH ARGUMENT next_index  
            
            IF the CHARACTER AT POSITION index OF expression IS EQUAL TO uppercase letter T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        
        SET result TO the FIRST VALUE OF THE RESULT OF CALLING evaluate WITH ARGUMENT zero  
        RETURN result  
    END FUNCTION  
END CLASS
