CLASS Solution  
    FUNCTION findNthDigit WITH PARAMETER n  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n IS GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        
        SET start TO start PLUS THE INTEGER DIVISION OF n MINUS one BY length  
        
        SET index TO THE REMAINDER OF n MINUS one DIVIDED BY length  
        SET start_string TO THE STRING REPRESENTATION OF start  
        SET digit_char TO THE CHARACTER AT POSITION index OF start_string  
        SET digit TO THE INTEGER VALUE OF digit_char  
        RETURN digit  
    END FUNCTION  
END CLASS
