CLASS Solution  
    FUNCTION threeSum WITH PARAMETER nums  
        CALL sort METHOD ON nums  
        SET result TO empty list  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS two MINUS one  
            IF i GREATER THAN zero AND element at position i OF nums EQUAL TO element at position i MINUS one OF nums  
                CONTINUE TO next iteration of the loop  
            END IF  
            
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            WHILE left LESS THAN right  
                SET total TO element at position i OF nums PLUS element at position left OF nums PLUS element at position right OF nums  
                IF total EQUAL TO zero  
                    APPEND list OF element at position i OF nums, element at position left OF nums, element at position right OF nums TO result  
                    
                    WHILE left LESS THAN right AND element at position left OF nums EQUAL TO element at position left PLUS one OF nums  
                        INCREMENT left BY one  
                    END WHILE  
                    WHILE left LESS THAN right AND element at position right OF nums EQUAL TO element at position right MINUS one OF nums  
                        DECREMENT right BY one  
                    END WHILE  
                    
                    INCREMENT left BY one  
                    DECREMENT right BY one  
                ELSE IF total LESS THAN zero  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
