CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two RAISED TO THE power of thirty ONE MINUS one  
        SET INT_MIN TO NEGATIVE two RAISED TO THE power of thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index LESS THAN n AND the ELEMENT at POSITION index OF s IS A SPACE CHARACTER  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (the ELEMENT at POSITION index OF s EQUAL TO the PLUS SIGN OR the ELEMENT at POSITION index OF s EQUAL TO the MINUS SIGN)  
            IF the ELEMENT at POSITION index OF s EQUAL TO the MINUS SIGN  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND the ELEMENT at POSITION index OF s IS A DIGIT  
            SET digit TO the NUMERIC VALUE OF the ELEMENT at POSITION index OF s  
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUAL TO one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
