CLASS Solution  
    FUNCTION numRabbits WITH parameter answers AS list of integers  
        SET answer_counts TO the result of counting occurrences of each element IN answers  
        SET total_rabbits TO zero  
        FOR each answer AND corresponding count IN the items OF answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the integer division OF count PLUS answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
