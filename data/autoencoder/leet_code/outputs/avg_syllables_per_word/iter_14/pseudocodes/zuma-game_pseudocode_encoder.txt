CLASS Solution  
    FUNCTION findMinStep(board, hand)  
  
        FUNCTION clean(board)  
            SET stack TO an empty list  
            FOR each ball IN board  
                IF stack IS NOT empty AND the first element of the last element of stack IS NOT EQUAL TO ball AND the second element of the last element of stack IS GREATER THAN OR EQUAL TO three  
                    REMOVE the last element FROM stack  
                END IF  
                IF stack IS empty OR the first element of the last element of stack IS NOT EQUAL TO ball  
                    APPEND a list containing ball AND one TO stack  
                ELSE  
                    INCREMENT the second element of the last element of stack BY one  
                END IF  
            END FOR  
            IF stack IS NOT empty AND the second element of the last element of stack IS GREATER THAN OR EQUAL TO three  
                REMOVE the last element FROM stack  
            END IF  
            SET result TO an empty string  
            FOR each element consisting of ball AND count IN stack  
                APPEND ball REPEATED count TIMES TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
  
        FUNCTION dfs(board, hand_counter)  
            IF board IS empty  
                RETURN zero  
            END IF  
            SET min_steps TO a very large number representing infinity  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF board  
                SET j TO i PLUS one  
                WHILE j IS LESS THAN the LENGTH OF board AND the element at position i OF board IS EQUAL TO the element at position j OF board  
                    INCREMENT j BY one  
                END WHILE  
                SET increment TO three MINUS j MINUS i  
                IF the value at key the element at position i OF board IN hand_counter IS GREATER THAN OR EQUAL TO increment  
                    SET new_hand_counter TO a copy of hand_counter  
                    DECREMENT the value at key the element at position i OF board IN new_hand_counter BY increment  
                    SET new_board TO clean(the substring FROM start zero TO i OF board CONCATENATED WITH the substring FROM start j TO end OF board)  
                    SET steps TO dfs(new_board, new_hand_counter)  
                    IF steps IS GREATER THAN OR EQUAL TO zero  
                        IF min_steps IS GREATER THAN steps PLUS increment  
                            SET min_steps TO steps PLUS increment  
                        END IF  
                    END IF  
                END IF  
                SET i TO j  
            END WHILE  
            IF min_steps IS EQUAL TO a very large number representing infinity  
                RETURN negative one  
            ELSE  
                RETURN min_steps  
            END IF  
        END FUNCTION  
  
        SET hand_counter TO a mapping from each unique character in hand TO its count  
        RETURN dfs(board, hand_counter)  
    END FUNCTION  
END CLASS
