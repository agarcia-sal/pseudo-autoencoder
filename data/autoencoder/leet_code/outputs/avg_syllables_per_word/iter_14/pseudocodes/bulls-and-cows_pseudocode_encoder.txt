CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        
        SET secret_count TO A DEFAULT DICTIONARY THAT RETURNS ZERO FOR MISSING KEYS  
        SET guess_count TO A DEFAULT DICTIONARY THAT RETURNS ZERO FOR MISSING KEYS  
        
        FOR each pair OF characters s AND g IN secret AND guess SIMULTANEOUSLY  
            IF s IS EQUAL TO g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value AT key s OF secret_count BY one  
                INCREMENT the value AT key g OF guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN secret_count KEYS  
            INCREMENT cows BY THE MINIMUM OF the value AT key digit OF secret_count AND the value AT key digit OF guess_count  
        END FOR  
        
        SET result TO THE STRING REPRESENTATION OF bulls CONCATENATED WITH THE LETTER A CONCATENATED WITH THE STRING REPRESENTATION OF cows CONCATENATED WITH THE LETTER B  
        RETURN result  
    END FUNCTION  
END CLASS
