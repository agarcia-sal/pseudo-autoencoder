CLASS Solution  
    FUNCTION minInteger WITH PARAMETERS num AND k  
        SET num TO the list of characters in num  
        SET n TO the length of num  
        
        WHILE k IS GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i OF num IS GREATER THAN element at position i PLUS one OF num  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the smaller value BETWEEN i PLUS k PLUS one AND n MINUS one  
                        IF element at position j OF num IS LESS THAN min_digit  
                            SET min_digit TO element at position j OF num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET num FROM position i TO min_index INCLUSIVE TO the list consisting of element at position min_index OF num FOLLOWED BY the slice of num FROM position i TO min_index MINUS one  
                    
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    EXIT FOR  
                END IF  
            END FOR  
            
            IF swapped IS false  
                EXIT WHILE  
            END IF  
        END WHILE  
        
        RETURN the concatenation of elements in num  
    END FUNCTION  
END CLASS
