CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO the LENGTH OF dist  
        INITIALIZE dp AS a two-dimensional list with n rows  
        EACH row CONTAINS n PLUS one elements SET TO infinity representing unachievable time  
        
        SET total_dist TO zero  
        FOR each element IN dist  
            INCREMENT total_dist BY element  
        END FOR  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN minus one  
        END IF  
        
        SET element at position zero OF row zero OF dp TO element at position zero OF dist  
        FOR j FROM one TO n MINUS one  
            SET element at position j OF row zero OF dp TO element at position zero OF dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET element at position j OF row i OF dp TO the CEILING of the element at position j OF row i MINUS one OF dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i OF dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at position j OF row i OF dp TO the MINIMUM OF the current value of element at position j OF row i OF dp AND the element at position j MINUS one OF row i MINUS one OF dp PLUS element at position i OF dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position j OF row n MINUS one OF dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
