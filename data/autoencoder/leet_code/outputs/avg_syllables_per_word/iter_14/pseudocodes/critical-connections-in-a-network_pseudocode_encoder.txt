CLASS Solution  
    FUNCTION criticalConnections WITH PARAMETERS n, connections  
        FUNCTION dfs WITH PARAMETERS node, parent, disc, low, time, graph, critical  
            SET element at position node OF disc TO element at position zero OF time  
            SET element at position node OF low TO element at position zero OF time  
            INCREMENT element at position zero OF time BY one  
            
            FOR each neighbor IN element at position node OF graph  
                IF element at position neighbor OF disc IS EQUAL TO negative one  
                    CALL dfs WITH neighbor, node, disc, low, time, graph, critical  
                    SET element at position node OF low TO THE MINIMUM OF element at position node OF low AND element at position neighbor OF low  
                    IF element at position neighbor OF low IS GREATER THAN element at position node OF disc  
                        APPEND list CONTAINING node AND neighbor TO critical  
                    END IF  
                ELSE IF neighbor IS NOT EQUAL TO parent  
                    SET element at position node OF low TO THE MINIMUM OF element at position node OF low AND element at position neighbor OF disc  
                END IF  
            END FOR  
        END FUNCTION  
        
        CALL initializeGraph WITH n AND connections RETURNING graph  
        
        CALL initializeListWithValue WITH value negative one AND length n RETURNING disc  
        CALL initializeListWithValue WITH value negative one AND length n RETURNING low  
        SET time TO list CONTAINING zero  
        
        SET critical TO empty list  
        
        FOR node FROM zero TO n MINUS one  
            IF element at position node OF disc IS EQUAL TO negative one  
                CALL dfs WITH node, negative one, disc, low, time, graph, critical  
            END IF  
        END FOR  
        
        RETURN critical  
    END FUNCTION  
END CLASS  

FUNCTION initializeGraph WITH PARAMETERS n, connections  
    SET graph TO empty list  
    FOR index FROM zero TO n MINUS one  
        APPEND empty list TO graph  
    END FOR  
    FOR each pair u AND v IN connections  
        APPEND v TO element at position u OF graph  
        APPEND u TO element at position v OF graph  
    END FOR  
    RETURN graph  
END FUNCTION  

FUNCTION initializeListWithValue WITH PARAMETERS value, length  
    SET result TO empty list  
    FOR index FROM zero TO length MINUS one  
        APPEND value TO result  
    END FOR  
    RETURN result  
END FUNCTION
