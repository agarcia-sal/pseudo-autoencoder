CLASS Solution  
    FUNCTION minCost WITH PARAMETER costs  
        IF costs IS EQUAL TO empty  
            RETURN zero  
        END IF  
        
        SET prev_red TO element at position zero OF element at position zero OF costs  
        SET prev_blue TO element at position one OF element at position zero OF costs  
        SET prev_green TO element at position two OF element at position zero OF costs  
        
        FOR i FROM one TO the LENGTH OF costs MINUS one  
            SET current_red TO element at position zero OF element at position i OF costs PLUS the MINIMUM VALUE BETWEEN prev_blue AND prev_green  
            SET current_blue TO element at position one OF element at position i OF costs PLUS the MINIMUM VALUE BETWEEN prev_red AND prev_green  
            SET current_green TO element at position two OF element at position i OF costs PLUS the MINIMUM VALUE BETWEEN prev_red AND prev_blue  
            
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        
        RETURN the MINIMUM VALUE BETWEEN prev_red AND prev_blue AND prev_green  
    END FUNCTION  
END CLASS
