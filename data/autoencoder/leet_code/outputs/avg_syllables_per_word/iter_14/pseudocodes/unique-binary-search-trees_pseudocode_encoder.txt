CLASS Solution  
    FUNCTION numTrees WITH PARAMETER n  
        SET dp TO a list of zeroes with the LENGTH OF n PLUS one  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i  
                SET element at position i OF dp TO element at position i OF dp PLUS element at position j MINUS one OF dp MULTIPLIED BY element at position i MINUS j OF dp  
            END FOR  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
