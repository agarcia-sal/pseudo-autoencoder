CLASS Solution  
    FUNCTION longestValidParentheses WITH PARAMETER s  
        SET stack TO a new list containing a single element negative one  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF char IS EQUAL TO open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the LENGTH OF stack IS EQUAL TO zero  
                    APPEND i TO stack  
                ELSE  
                    SET current_length TO i MINUS the last element OF stack  
                    IF current_length IS GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
