CLASS Solution  
    FUNCTION reorganizeString WITH PARAMETER s  
        SET count TO the frequency count of each character in s  
        SET max_heap TO the list of pairs composed of the NEGATION OF frequency PLUS the character FROM each item IN count  
        TRANSFORM max_heap INTO a heap structure  
        
        SET prev_char TO null  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap IS NOT empty OR prev_freq IS LESS THAN zero  
            IF max_heap IS empty AND prev_freq IS LESS THAN zero  
                RETURN the empty string  
            END IF  
            
            REMOVE the smallest element FROM max_heap AND ASSIGN the first element TO freq AND the second element TO char  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq IS LESS THAN zero  
                INSERT prev_freq AND prev_char INTO max_heap AS a pair  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all characters IN result  
    END FUNCTION  
END CLASS
