CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF less IS true  
                IF the element at position i OF nums IS GREATER THAN the element at position i PLUS one OF nums  
                    SET temporary TO the element at position i OF nums  
                    SET the element at position i OF nums TO the element at position i PLUS one OF nums  
                    SET the element at position i PLUS one OF nums TO temporary  
                END IF  
            ELSE  
                IF the element at position i OF nums IS LESS THAN the element at position i PLUS one OF nums  
                    SET temporary TO the element at position i OF nums  
                    SET the element at position i OF nums TO the element at position i PLUS one OF nums  
                    SET the element at position i PLUS one OF nums TO temporary  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
