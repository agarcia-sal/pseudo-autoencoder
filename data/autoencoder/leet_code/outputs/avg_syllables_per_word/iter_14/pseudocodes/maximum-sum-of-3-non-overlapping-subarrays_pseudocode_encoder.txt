CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums, k)  
        SET n TO the LENGTH OF nums  
        SET sums TO a LIST OF zero VALUES WITH LENGTH n MINUS k PLUS one  
        SET current_sum TO the SUM OF the ELEMENTS FROM position zero TO position k MINUS one OF nums  
        SET element at position zero OF sums TO current_sum  
        
        FOR i FROM one TO n MINUS k  
            INCREMENT current_sum BY the ELEMENT at position i PLUS k MINUS one OF nums MINUS the ELEMENT at position i MINUS one OF nums  
            SET element at position i OF sums TO current_sum  
        END FOR  
        
        SET left TO a LIST OF zero VALUES WITH LENGTH n MINUS k PLUS one  
        SET right TO a LIST OF the VALUE n MINUS k REPEATED n MINUS k PLUS one TIMES  
        
        SET max_sum TO zero  
        FOR i FROM zero TO n MINUS k  
            IF element at position i OF sums IS GREATER THAN element at position max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET element at position i OF left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWN TO zero  
            IF element at position i OF sums IS GREATER THAN OR EQUAL TO element at position max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET element at position i OF right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a LIST OF three zero VALUES  
        FOR j FROM k TO n MINUS two TIMES k  
            SET i TO element at position j MINUS k OF left  
            SET l TO element at position j PLUS k OF right  
            SET total TO element at position i OF sums PLUS element at position j OF sums PLUS element at position l OF sums  
            IF total IS GREATER THAN max_total  
                SET max_total TO total  
                SET result TO a LIST CONTAINING i, j, l IN THAT ORDER  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
