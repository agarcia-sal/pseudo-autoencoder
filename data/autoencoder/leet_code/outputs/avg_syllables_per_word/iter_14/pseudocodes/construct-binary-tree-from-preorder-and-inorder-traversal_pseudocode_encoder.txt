CLASS TreeNode
    FUNCTION __init__(self, val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO NEW TreeNode WITH val EQUAL TO element at position zero OF values
    SET i TO one
    SET queue TO NEW double ended queue
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO the FIRST element REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO NEW TreeNode WITH val EQUAL TO element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO NEW TreeNode WITH val EQUAL TO element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION buildTree(self, preorder, inorder)
        IF preorder IS empty OR inorder IS empty
            RETURN none
        END IF
        SET root_val TO element at position zero OF preorder
        SET root TO NEW TreeNode WITH val EQUAL TO root_val
        SET root_index_inorder TO the POSITION OF root_val IN inorder
        SET root.left TO self.buildTree(sublist of preorder FROM position one TO root_index_inorder PLUS one, sublist of inorder FROM position zero TO root_index_inorder)
        SET root.right TO self.buildTree(sublist of preorder FROM position root_index_inorder PLUS one TO END, sublist of inorder FROM position root_index_inorder PLUS one TO END)
        RETURN root
    END FUNCTION
END CLASS
