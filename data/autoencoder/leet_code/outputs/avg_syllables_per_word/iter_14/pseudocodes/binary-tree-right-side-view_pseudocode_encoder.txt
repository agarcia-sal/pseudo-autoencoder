CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val LEFT right  
        SET self.val TO val  
        SET self.left TO LEFT  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN null  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT null  
            SET node.left TO NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT null  
            SET node.right TO NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS null AND q IS null  
        RETURN True  
    ELSE IF p IS null OR q IS null  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView WITH PARAMETERS self root  
        IF root IS null  
            RETURN AN EMPTY LIST  
        END IF  
        SET right_side_view TO AN EMPTY LIST  
        SET queue TO A DOUBLE ENDED QUEUE INITIALIZED WITH root  
        WHILE queue IS NOT EMPTY  
            SET level_length TO THE LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                IF i IS EQUAL TO level_length MINUS one  
                    APPEND node.val TO right_side_view  
                END IF  
                IF node.left IS NOT null  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT null  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
