CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO a dictionary mapping zero TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each pair of index i AND number num IN enumeration OF nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF cumulative_sum MINUS k EXISTS AS a key IN cumulative_sum_index  
                SET potential_length TO i MINUS the value AT key cumulative_sum MINUS k IN cumulative_sum_index  
                IF potential_length GREATER THAN max_length  
                    SET max_length TO potential_length  
                END IF  
            END IF  
            IF cumulative_sum DOES NOT EXIST AS a key IN cumulative_sum_index  
                SET the value AT key cumulative_sum IN cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
