CLASS Solution  
    FUNCTION maximumScore(nums, k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at position k OF nums  
        SET max_score TO current_min  
        
        WHILE left IS GREATER THAN zero OR right IS LESS THAN the LENGTH OF nums MINUS one IS TRUE  
            IF left IS EQUAL TO zero THEN  
                INCREMENT right BY one  
            ELSE IF right IS EQUAL TO the LENGTH OF nums MINUS one THEN  
                DECREMENT left BY one  
            ELSE IF element at position left MINUS one OF nums IS GREATER THAN OR EQUAL TO element at position right PLUS one OF nums THEN  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            
            SET current_min TO the MINIMUM OF current_min AND element at position left OF nums AND element at position right OF nums  
            SET score TO current_min MULTIPLIED BY right MINUS left PLUS one  
            SET max_score TO the MAXIMUM OF max_score AND score  
        END WHILE  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
