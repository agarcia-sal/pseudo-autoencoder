CLASS Solution  
    FUNCTION findUnsortedSubarray WITH PARAMETER nums  
        SET sorted_nums TO a version of nums arranged IN ascending order  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index OF nums IS NOT EQUAL TO element at position index OF sorted_nums  
                IF start IS EQUAL TO negative one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        SET result TO end MINUS start PLUS one  
        RETURN result  
    END FUNCTION  
END CLASS
