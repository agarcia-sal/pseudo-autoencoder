CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUAL TO the LENGTH OF nums  
                IF current_sum EQUAL TO target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF memo CONTAINS the PAIR of index AND current_sum  
                RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
            END IF  

            SET positive TO dfs with index INCREMENTED BY one AND current_sum INCREMENTED BY the element at position index OF nums AND memo  
            SET negative TO dfs with index INCREMENTED BY one AND current_sum DECREMENTED BY the element at position index OF nums AND memo  

            SET the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the PAIR of index AND current_sum IN memo  
        END FUNCTION  

        RETURN dfs with zero AS index AND zero AS current_sum AND an EMPTY COLLECTION AS memo  
    END FUNCTION  
END CLASS
