CLASS Solution  
    FUNCTION minStickers WITH PARAMETERS stickers AND target  
        SET stickers TO A LIST OF THE RESULT OF CALLING Counter FUNCTION ON EACH sticker IN stickers  
        SORT stickers BY THE NEGATION OF THE LARGEST VALUE IN THE FREQUENCIES OF EACH sticker  
        SET queue TO A NEW deque CONTAINING A TUPLE OF target AND zero  
        SET memo TO A DICTIONARY WITH KEY target AND VALUE zero  
        WHILE queue IS NOT EMPTY  
            REMOVE AND RETURN THE LEFTMOST ELEMENT OF queue INTO current_target AND used_stickers  
            FOR EACH sticker IN stickers  
                IF THE FIRST CHARACTER OF current_target IS NOT PRESENT IN sticker  
                    CONTINUE TO NEXT ITERATION OF THE LOOP  
                END IF  
                SET remaining TO A NEW Counter CREATED FROM current_target  
                DECREMENT remaining BY THE FREQUENCY COUNTS OF sticker  
                SET new_target TO A STRING CREATED BY JOINING ALL ELEMENTS OF remaining SORTED IN ORDER  
                IF new_target IS AN EMPTY STRING  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target IS NOT A KEY IN memo  
                    SET memo[new_target] TO used_stickers PLUS one  
                    APPEND A TUPLE OF new_target AND used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
