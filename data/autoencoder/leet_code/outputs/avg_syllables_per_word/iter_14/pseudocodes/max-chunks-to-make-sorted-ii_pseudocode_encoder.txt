CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO element at position zero OF arr  
        FOR each num IN arr  
            SET current_max TO the GREATER OF current_max AND num  
            APPEND current_max TO max_so_far  
        END FOR  
        
        SET sorted_arr TO arr IN sorted ORDER  
        SET chunks TO zero  
        SET max_chunk TO zero  
        
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            SET max_chunk TO the GREATER OF max_chunk AND element at position index OF arr  
            IF max_chunk EQUAL TO element at position index OF sorted_arr  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
