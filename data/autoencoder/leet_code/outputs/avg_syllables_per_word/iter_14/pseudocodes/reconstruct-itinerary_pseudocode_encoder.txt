CLASS Solution  
    FUNCTION findItinerary WITH PARAMETER tickets AS list of list of string  
        SET graph TO a new mapping with default value of an empty list  
        FOR each pair of from_location AND to_location IN the reverse order traversal of the sorted tickets  
            APPEND to_location TO the list at key from_location IN graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit WITH PARAMETER airport AS string  
            WHILE the list at key airport IN graph IS NOT empty  
                SET next_destination TO the last element of the list at key airport IN graph  
                REMOVE the last element FROM the list at key airport IN graph  
                CALL visit WITH next_destination  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string J F K  
        SET reversed_itinerary TO the reverse order of itinerary  
        RETURN reversed_itinerary  
    END FUNCTION  
END CLASS
