CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO minus sign  
        ELSE  
            SET sign TO empty string  
        END IF  
        
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
        
        SET quotient TO numerator DIVIDED BY denominator  
        SET remainder TO numerator MODULO denominator  
        SET integer_part TO string representation OF quotient  
        
        IF remainder EQUAL TO zero  
            RETURN concatenation OF sign AND integer_part  
        END IF  
        
        INITIALIZE seen_remainders AS empty mapping  
        INITIALIZE fractional_part AS empty list  
        
        WHILE remainder NOT EQUAL TO zero  
            IF remainder EXISTS AS KEY IN seen_remainders  
                SET start TO value ASSOCIATED WITH remainder IN seen_remainders  
                SET non_repeating TO concatenation OF elements FROM fractional_part FROM index zero TO start minus one  
                SET repeating TO concatenation OF elements FROM fractional_part FROM index start TO THE END  
                RETURN concatenation OF sign, integer_part, dot CHARACTER, non_repeating, opening parenthesis CHARACTER, repeating, closing parenthesis CHARACTER  
            END IF  
            
            SET the value ASSOCIATED WITH remainder IN seen_remainders TO LENGTH OF fractional_part  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO remainder DIVIDED BY denominator  
            SET remainder TO remainder MODULO denominator  
            APPEND string representation OF quotient TO fractional_part  
        END WHILE  
        
        RETURN concatenation OF sign, integer_part, dot CHARACTER, concatenation OF elements IN fractional_part  
    END FUNCTION  
END CLASS
