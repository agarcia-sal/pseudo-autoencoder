CLASS Solution  
    FUNCTION findCircleNum WITH PARAMETER isConnected  
        FUNCTION dfs WITH PARAMETER city  
            FOR each neighbor FROM zero TO n MINUS one  
                IF the element at position city OF isConnected AT position neighbor IS EQUAL TO one AND the element at position neighbor OF visited IS EQUAL TO false  
                    SET the element at position neighbor OF visited TO true  
                    CALL dfs WITH neighbor  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET n TO the LENGTH OF isConnected  
        SET visited TO A LIST OF false VALUES WITH LENGTH n  
        SET province_count TO zero  
        
        FOR each city FROM zero TO n MINUS one  
            IF the element at position city OF visited IS EQUAL TO false  
                CALL dfs WITH city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  
        
        RETURN province_count  
    END FUNCTION  
END CLASS
