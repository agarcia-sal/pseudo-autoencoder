CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET skill_to_index_mapping TO A MAPPING FROM each skill in req_skills TO its POSITION index IN req_skills  
        SET total_required_skills TO the LENGTH OF req_skills  
        SET total_people TO the LENGTH OF people  
        SET people_skills_bitmask_list TO A LIST OF zero WITH LENGTH total_people  
        FOR each index i AND skill_list ss IN people WITH POSITION index i  
            FOR each skill s IN skill_list ss  
                SET the element at position i OF people_skills_bitmask_list TO the current value BITWISE OR WITH a value computed as one LEFT SHIFTED BY the index corresponding to skill s FROM skill_to_index_mapping  
            END FOR  
        END FOR  
        SET minimum_team_size_list TO A LIST OF infinite WITH LENGTH EQUAL TO two RAISED TO THE POWER OF total_required_skills  
        SET last_added_person_index_list TO A LIST OF zero WITH LENGTH EQUAL TO two RAISED TO THE POWER OF total_required_skills  
        SET previous_skill_set_list TO A LIST OF zero WITH LENGTH EQUAL TO two RAISED TO THE POWER OF total_required_skills  
        SET the element at position zero OF minimum_team_size_list TO zero  
        FOR each skill_set_index i FROM zero TO two RAISED TO THE POWER OF total_required_skills MINUS one  
            IF the element at position i OF minimum_team_size_list IS infinite  
                CONTINUE TO next iteration  
            END IF  
            FOR each person_index j FROM zero TO total_people MINUS one  
                SET new_skill_set TO the BITWISE OR OF i AND the element at position j OF people_skills_bitmask_list  
                IF the element at position i OF minimum_team_size_list PLUS one IS LESS THAN the element at position new_skill_set OF minimum_team_size_list  
                    SET the element at position new_skill_set OF minimum_team_size_list TO the element at position i OF minimum_team_size_list PLUS one  
                    SET the element at position new_skill_set OF last_added_person_index_list TO j  
                    SET the element at position new_skill_set OF previous_skill_set_list TO i  
                END IF  
            END FOR  
        END FOR  
        SET current_skill_set TO two RAISED TO THE POWER OF total_required_skills MINUS one  
        SET answer_list TO an empty list  
        WHILE current_skill_set IS NOT zero  
            APPEND the element at position current_skill_set OF last_added_person_index_list TO answer_list  
            SET current_skill_set TO the element at position current_skill_set OF previous_skill_set_list  
        END WHILE  
        RETURN answer_list  
    END FUNCTION  
END CLASS
