CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO positive infinity  
        
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET current_sum TO current_sum PLUS element at position right OF nums  
            
            WHILE current_sum IS GREATER THAN OR EQUAL TO target  
                IF min_length IS GREATER THAN right MINUS left PLUS one  
                    SET min_length TO right MINUS left PLUS one  
                END IF  
                SET current_sum TO current_sum MINUS element at position left OF nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        
        IF min_length IS EQUAL TO positive infinity  
            RETURN zero  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
