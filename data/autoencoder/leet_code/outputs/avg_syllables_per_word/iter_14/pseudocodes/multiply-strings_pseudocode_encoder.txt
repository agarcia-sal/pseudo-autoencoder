CLASS Solution  
    FUNCTION multiply(num1 AS string, num2 AS string) RETURNS string  
        IF num1 IS EQUAL TO zero STRING OR num2 IS EQUAL TO zero STRING  
            RETURN zero STRING  
        END IF  
        
        SET result TO a list of zero integers WITH THE LENGTH OF the LENGTH OF num1 PLUS the LENGTH OF num2  
        
        SET num1 TO the REVERSE OF num1  
        SET num2 TO the REVERSE OF num2  
        
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the DIFFERENCE BETWEEN the CHARACTER CODE OF the ELEMENT AT position i OF num1 AND the CHARACTER CODE OF zero STRING MULTIPLIED BY the DIFFERENCE BETWEEN the CHARACTER CODE OF the ELEMENT AT position j OF num2 AND the CHARACTER CODE OF zero STRING  
                SET p1 TO the SUM OF i AND j  
                SET p2 TO the SUM OF i AND j PLUS one  
                SET sum_ TO the SUM OF mul AND the ELEMENT AT position p1 OF result  
                
                SET the ELEMENT AT position p1 OF result TO sum_ MODULO ten  
                INCREMENT the ELEMENT AT position p2 OF result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the CONCATENATION OF the ELEMENTS OF result IN REVERSE ORDER CONVERTED TO STRINGS  
        RETURN result_str WITH ALL LEADING zero CHARACTERS REMOVED  
    END FUNCTION  
END CLASS
