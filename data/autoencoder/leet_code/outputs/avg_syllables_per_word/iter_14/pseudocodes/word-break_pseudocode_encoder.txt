CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the CONVERSION OF wordDict INTO a SET  
        SET dp TO a LIST OF Boolean FALSE VALUES WITH LENGTH EQUAL TO the LENGTH OF s PLUS one  
        SET the element at position zero OF dp TO Boolean TRUE  
        
        FOR i FROM one TO the LENGTH OF s PLUS one MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position j OF dp IS TRUE AND the SUBSTRING FROM position j TO position i MINUS one OF s IS IN wordSet  
                    SET the element at position i OF dp TO Boolean TRUE  
                    BREAK the INNER loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position LENGTH OF s OF dp  
    END FUNCTION  
END CLASS
