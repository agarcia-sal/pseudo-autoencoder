CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET dp TO a list that for each number from zero TO n INCLUSIVE CONTAINS a list that for each number from zero TO k INCLUSIVE CONTAINS zero  
        SET element at position zero OF list at position zero OF dp TO one  
        FOR i FROM one TO n INCLUSIVE  
            FOR j FROM one TO k INCLUSIVE  
                SET element at position j OF list at position i OF dp TO the remainder of the division of the result of j MULTIPLIED BY element at position j OF list at position i MINUS one OF dp PLUS element at position j MINUS one OF list at position i MINUS one OF dp BY MOD  
            END FOR  
        END FOR  
        RETURN element at position k OF list at position n OF dp  
    END FUNCTION  
END CLASS
