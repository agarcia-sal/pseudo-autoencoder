CLASS TreeNode  
    FUNCTION __init__(self, val EQUAL TO zero, left EQUAL TO None, right EQUAL TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUAL TO element at position zero OF values  
    SET i TO one  
    SET queue TO A NEW double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element REMOVED FROM the FRONT OF queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO A NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low EQUAL TO negative infinity, high EQUAL TO infinity)  
            IF node IS None  
                RETURN True  
            END IF  
            IF node.val IS LESS THAN OR EQUAL TO low OR node.val IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN validate(node.left, low, node.val) AND validate(node.right, node.val, high)  
        END FUNCTION  
        RETURN validate(root)  
    END FUNCTION  
END CLASS
