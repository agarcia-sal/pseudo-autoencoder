CLASS Solution  
    FUNCTION kInversePairs WITH PARAMETERS n AND k  
        SET MOD TO ten TO THE POWER OF nine PLUS one  
        SET dp TO the LIST OF LISTS OF zeros WITH DIMENSIONS n PLUS one BY k PLUS one  
        SET the element AT position zero OF dp AT position zero TO one  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS the element AT position j OF dp AT position i MINUS one  
                SET prefix_sum TO prefix_sum MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS the element AT position j MINUS i OF dp AT position i MINUS one PLUS MOD  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                END IF  
                SET the element AT position j OF dp AT position i TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN the element AT position k OF dp AT position n  
    END FUNCTION  
END CLASS
