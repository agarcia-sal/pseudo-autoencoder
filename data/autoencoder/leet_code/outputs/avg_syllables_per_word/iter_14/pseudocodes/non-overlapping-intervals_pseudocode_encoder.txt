CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals IS EMPTY  
            RETURN zero  
        END IF  
        CALL the SORT method ON intervals ORDERING BY the element at position one OF each interval  
        SET end TO the element at position one OF the element at position zero OF intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the element at position zero OF the element at position index OF intervals IS GREATER THAN OR EQUAL TO end  
                SET end TO the element at position one OF the element at position index OF intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
