CLASS Solution  
    FUNCTION maxProfit(prices)  
        IF prices IS EQUAL TO empty  
            RETURN zero  
        END IF  
        
        SET min_price1 TO positive infinity  
        SET min_price2 TO positive infinity  
        SET max_profit1 TO zero  
        SET max_profit2 TO zero  
        
        FOR each price IN prices  
            IF price IS LESS THAN min_price1  
                SET min_price1 TO price  
            END IF  
            IF price MINUS min_price1 IS GREATER THAN max_profit1  
                SET max_profit1 TO price MINUS min_price1  
            END IF  
            
            IF price MINUS max_profit1 IS LESS THAN min_price2  
                SET min_price2 TO price MINUS max_profit1  
            END IF  
            IF price MINUS min_price2 IS GREATER THAN max_profit2  
                SET max_profit2 TO price MINUS min_price2  
            END IF  
        END FOR  
        
        RETURN max_profit2  
    END FUNCTION  
END CLASS
