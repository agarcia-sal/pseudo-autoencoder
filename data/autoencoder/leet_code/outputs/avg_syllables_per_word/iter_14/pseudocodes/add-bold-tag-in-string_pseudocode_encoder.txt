CLASS Solution  
    FUNCTION addBoldTag WITH PARAMETERS s AND words  
        IF words IS EMPTY  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        SET mask TO a list of BOOLEAN FALSE values with LENGTH n  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s STARTING FROM start  
                IF start IS EQUAL TO negative one  
                    BREAK THE WHILE LOOP  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the ELEMENT at POSITION i OF mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the ELEMENT at POSITION i OF mask IS TRUE  
                APPEND the string BOLD OPEN TAG TO result  
                WHILE i IS LESS THAN n AND the ELEMENT at POSITION i OF mask IS TRUE  
                    APPEND the ELEMENT at POSITION i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string BOLD CLOSE TAG TO result  
            ELSE  
                APPEND the ELEMENT at POSITION i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
