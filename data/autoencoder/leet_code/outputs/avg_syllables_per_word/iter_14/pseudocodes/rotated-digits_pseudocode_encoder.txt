CLASS Solution  
    FUNCTION rotatedDigits WITH PARAMETER n  
        SET change_digits TO the set containing elements two five six nine  
        SET same_digits TO the set containing elements zero one eight  
        
        FUNCTION is_good_number WITH PARAMETER num  
            SET num_str TO the conversion of num TO string  
            IF all digits in num_str ARE elements OF the union of change_digits and same_digits THEN  
                IF any digit in num_str IS an element OF change_digits THEN  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number WITH ARGUMENT i RETURNS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
