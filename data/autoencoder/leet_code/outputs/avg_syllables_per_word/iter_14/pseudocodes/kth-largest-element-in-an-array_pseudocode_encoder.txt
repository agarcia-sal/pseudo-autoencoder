CLASS Solution  
    FUNCTION findKthLargest WITH PARAMETERS nums AND k  
        SET min_heap TO an empty list  
        FOR each num IN nums  
            INSERT num INTO min_heap USING a heap data structure push operation  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                REMOVE the smallest element FROM min_heap USING a heap data structure pop operation  
            END IF  
        END FOR  
        RETURN the element at position zero OF min_heap  
    END FUNCTION  
END CLASS
