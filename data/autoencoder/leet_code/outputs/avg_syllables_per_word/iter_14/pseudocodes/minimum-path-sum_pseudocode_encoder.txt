CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid IS EQUAL TO None OR the LENGTH OF grid IS EQUAL TO zero OR the LENGTH OF element at position zero OF grid IS EQUAL TO zero  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  

        CALL createTwoDimensionalList WITH m AND n RETURNING dp

        SET element at position zero OF element at position zero OF dp TO element at position zero OF element at position zero OF grid  

        FOR j FROM one TO n MINUS one  
            SET element at position j OF element at position zero OF dp TO element at position j MINUS one OF element at position zero OF dp PLUS element at position j OF element at position zero OF grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            SET element at position zero OF element at position i OF dp TO element at position zero OF element at position i MINUS one OF dp PLUS element at position zero OF element at position i OF grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET minimum_predecessor TO the MINIMUM OF element at position j OF element at position i MINUS one OF dp AND element at position j MINUS one OF element at position i OF dp  
                SET element at position j OF element at position i OF dp TO minimum_predecessor PLUS element at position j OF element at position i OF grid  
            END FOR  
        END FOR  

        RETURN element at position n MINUS one OF element at position m MINUS one OF dp  
    END FUNCTION  

    FUNCTION createTwoDimensionalList(row_count, column_count)  
        CREATE a two-dimensional list with row_count rows and column_count columns filled with zeroes and RETURN it  
    END FUNCTION  
END CLASS
