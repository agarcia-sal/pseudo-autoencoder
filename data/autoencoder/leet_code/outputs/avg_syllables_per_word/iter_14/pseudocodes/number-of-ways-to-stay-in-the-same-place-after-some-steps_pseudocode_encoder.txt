CLASS Solution  
    FUNCTION numWays WITH PARAMETER steps AND PARAMETER arrLen  
        SET MOD TO one raised to the power of nine PLUS seven  
        SET max_index TO the minimum OF the integer division of steps BY two AND arrLen MINUS one  
        SET dp TO a list of zeros WITH LENGTH max_index PLUS one  
        SET the element at position zero OF dp TO one  
        FOR each index FROM one TO steps  
            SET prev_dp TO a new list copied FROM dp  
            FOR each position i FROM zero TO max_index  
                SET the element at position i OF dp TO the element at position i OF prev_dp  
                IF i IS GREATER THAN zero  
                    SET the element at position i OF dp TO the sum OF the element at position i OF dp AND the element at position i MINUS one OF prev_dp  
                    THEN TAKE the result MODULO MOD  
                END IF  
                IF i IS LESS THAN max_index  
                    SET the element at position i OF dp TO the sum OF the element at position i OF dp AND the element at position i PLUS one OF prev_dp  
                    THEN TAKE the result MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position zero OF dp  
    END FUNCTION  
END CLASS
