CLASS Solution  
    FUNCTION strobogrammaticInRange WITH PARAMETERS low high  
        FUNCTION generate_strobogrammatic WITH PARAMETERS n final_length  
            IF n EQUAL TO zero  
                RETURN a list containing the empty string  
            END IF  
            IF n EQUAL TO one  
                RETURN a list containing the strings zero one eight  
            END IF  
            SET middles TO the result of generate_strobogrammatic WITH PARAMETERS n MINUS two final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUAL TO final_length  
                    APPEND the string formed by concatenating zero middle zero TO result  
                END IF  
                APPEND the string formed by concatenating one middle one TO result  
                APPEND the string formed by concatenating six middle nine TO result  
                APPEND the string formed by concatenating eight middle eight TO result  
                APPEND the string formed by concatenating nine middle six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic WITH PARAMETERS low high  
            SET count TO zero  
            FOR length FROM the length of low TO the length of high  
                FOR each num IN the result of generate_strobogrammatic WITH PARAMETERS length length  
                    IF length EQUAL TO the length of low AND num IS LESS THAN low OR length EQUAL TO the length of high AND num IS GREATER THAN high  
                        CONTINUE TO next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN the result of count_strobogrammatic WITH PARAMETERS low high  
END CLASS
