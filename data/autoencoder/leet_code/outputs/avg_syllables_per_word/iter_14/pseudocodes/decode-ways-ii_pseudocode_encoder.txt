CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
          
        INITIALIZE dp AS A list WITH LENGTH OF THE LENGTH OF s PLUS one FILLED WITH zero  
        SET element at position zero OF dp TO one  
          
        FUNCTION ways_to_decode_single(char)  
            IF char EQUAL TO the symbol star THEN  
                RETURN nine  
            ELSE IF char EQUAL TO the character zero THEN  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
          
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUAL TO the symbol star AND char2 EQUAL TO the symbol star THEN  
                RETURN fifteen  
            ELSE IF char1 EQUAL TO the symbol star THEN  
                IF char2 LESS THAN OR EQUAL TO the character six THEN  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUAL TO the symbol star THEN  
                IF char1 EQUAL TO the character one THEN  
                    RETURN nine  
                ELSE IF char1 EQUAL TO the character two THEN  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the INTEGER VALUE OF the concatenation OF char1 FOLLOWED BY char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
          
        FOR i FROM one TO the LENGTH OF s INCLUSIVE  
            SET dp AT position i TO THE REMAINDER OF THE SUM OF dp AT position i PLUS THE PRODUCT OF dp AT position i MINUS one MULTIPLIED BY ways_to_decode_single OF the ELEMENT AT position i MINUS one OF s DIVIDED BY MOD  
              
            IF i GREATER THAN one THEN  
                SET dp AT position i TO THE REMAINDER OF THE SUM OF dp AT position i PLUS THE PRODUCT OF dp AT position i MINUS two MULTIPLIED BY ways_to_decode_pair OF the ELEMENTS AT positions i MINUS two AND i MINUS one OF s DIVIDED BY MOD  
            END IF  
        END FOR  
          
        RETURN element AT THE LAST position OF dp  
    END FUNCTION  
END CLASS
