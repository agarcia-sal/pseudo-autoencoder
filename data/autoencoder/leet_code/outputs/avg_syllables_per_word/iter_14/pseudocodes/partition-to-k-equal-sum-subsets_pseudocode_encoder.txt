CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the SUM OF all elements IN nums  
        IF the total_sum MODULO k IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO the INTEGER DIVISION OF total_sum BY k  
        SORT nums IN descending ORDER  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k IS EQUAL TO zero  
                RETURN True  
            END IF  
            IF current_sum IS EQUAL TO target  
                RETURN the RESULT OF can_partition WITH index SET TO zero k DECREMENTED BY one current_sum SET TO zero AND used UNCHANGED  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF the element at position i OF used IS False AND current_sum PLUS the element at position i OF nums IS LESS THAN OR EQUAL TO target  
                    SET the element at position i OF used TO True  
                    IF the RESULT OF can_partition WITH index SET TO i PLUS one k UNCHANGED current_sum INCREMENTED BY the element at position i OF nums AND used UNCHANGED IS True  
                        RETURN True  
                    END IF  
                    SET the element at position i OF used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a LIST OF the LENGTH OF nums WITH all elements SET TO False  
        RETURN the RESULT OF can_partition WITH index SET TO zero k UNCHANGED current_sum SET TO zero AND used  
    END FUNCTION  
END CLASS
