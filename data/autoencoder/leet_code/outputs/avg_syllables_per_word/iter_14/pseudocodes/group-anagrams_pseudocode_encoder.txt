CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs  
        SET anagrams TO a new default dictionary that creates an empty list for new keys  
        FOR each string s IN strs  
            SET sorted_characters TO the characters OF s SORTED in ascending alphabetical order  
            SET sorted_str TO the concatenation OF all elements IN sorted_characters  
            APPEND s TO the list ASSOCIATED WITH sorted_str IN anagrams  
        END FOR  
        SET result TO the list OF all values IN anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
