CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary mapping zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN enumerated nums  
            IF num EQUAL TO one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS A KEY IN prefix_sum_index  
                SET current_length TO i MINUS the value ASSOCIATED WITH count IN prefix_sum_index  
                IF current_length GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                SET the value ASSOCIATED WITH count IN prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
