CLASS Solution  
    FUNCTION splitIntoFibonacci WITH PARAMETERS num  
        FUNCTION backtrack WITH PARAMETERS start path  
            IF start EQUAL TO the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start OF num EQUAL TO character ZERO AND end GREATER THAN start PLUS one  
                    CONTINUE TO next iteration of the loop  
                END IF  
                SET substring_string TO the substring FROM position start TO end MINUS one OF num  
                SET next_num TO the integer CONVERSION OF substring_string  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE THIRTY FIRST POWER  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUAL TO the element at position the LENGTH OF path MINUS one OF path PLUS the element at position the LENGTH OF path MINUS two OF path  
                    SET result TO backtrack WITH ARGUMENTS end AND the concatenation OF path WITH next_num  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN empty list  
        END FUNCTION  
        RETURN backtrack WITH ARGUMENTS zero AND empty list  
    END FUNCTION  
END CLASS
