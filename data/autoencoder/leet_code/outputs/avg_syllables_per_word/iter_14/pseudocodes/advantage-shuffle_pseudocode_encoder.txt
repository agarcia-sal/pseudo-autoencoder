CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        CALL SORT nums1 IN ascending order  
        
        SET nums2_with_indices TO A new list obtained by PAIRING each element of nums2 with its corresponding index SUCH THAT each item is a pair of the value and its original index  
        CALL SORT nums2_with_indices IN descending order BASED ON the value component OF each pair  
        
        SET result TO A new list OF length EQUAL TO the length OF nums1 FILLED WITH zeroes  
        
        SET left TO zero  
        SET right TO the length OF nums1 MINUS one  
        
        FOR each pair OF value AND index IN nums2_with_indices  
            IF element at position right OF nums1 IS GREATER THAN value  
                SET element at position index OF result TO element at position right OF nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index OF result TO element at position left OF nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
