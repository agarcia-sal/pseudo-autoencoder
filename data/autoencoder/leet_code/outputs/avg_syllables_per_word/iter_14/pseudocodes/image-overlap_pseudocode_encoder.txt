CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an EMPTY list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j OF the element at position i OF img1 IS EQUAL TO one  
                    APPEND the pair consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an EMPTY list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j OF the element at position i OF img2 IS EQUAL TO one  
                    APPEND the pair consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        CREATE overlap_count AS a NEW mapping FROM PAIRS TO integers WITH default integer zero  
        FOR each pair consisting of x1 AND y1 IN points1  
            FOR each pair consisting of x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value AT key consisting of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN the collection OF values OF overlap_count  
            IF value IS GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
