CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs consisting of the first element OF each interval AND its index in intervals  
        
        SET result TO empty list  
        FOR each pair of underscore AND end IN intervals  
            SET idx TO the position found by binary searching start_times FOR the pair where the first element IS GREATER THAN OR EQUAL TO end  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the second element at position idx OF start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
