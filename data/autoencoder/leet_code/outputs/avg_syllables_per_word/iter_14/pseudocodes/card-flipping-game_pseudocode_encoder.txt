CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO the SET OF each element x FROM fronts AND element y FROM backs WHERE x IS EQUAL TO y  
        SET all_numbers TO the SET OBTAINED BY COMBINING all elements FROM fronts AND all elements FROM backs  
        SET good_numbers TO all_numbers MINUS same  
        IF the SIZE OF good_numbers IS GREATER THAN zero  
            SET result TO the MINIMUM ELEMENT IN good_numbers  
            RETURN result  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
