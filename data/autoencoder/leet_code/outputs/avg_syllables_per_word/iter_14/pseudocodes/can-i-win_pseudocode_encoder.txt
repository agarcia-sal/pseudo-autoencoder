CLASS Solution  
    FUNCTION canIWin WITH PARAMETERS maxChoosableInteger AND desiredTotal  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS one) DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  
        
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_win WITH PARAMETERS current_total AND used_numbers  
            IF the pair of current_total AND used_numbers EXISTS AS a key IN memo  
                RETURN the value ASSOCIATED WITH the pair of current_total AND used_numbers IN memo  
            END IF  
            
            FOR number FROM one TO maxChoosableInteger  
                IF used_numbers AND the binary representation of one SHIFTED LEFT BY number IS EQUAL TO zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win WITH current_total PLUS number AND used_numbers OR binary representation of one SHIFTED LEFT BY number  
                        SET the value ASSOCIATED WITH the pair of current_total AND used_numbers IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the value ASSOCIATED WITH the pair of current_total AND used_numbers IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win WITH zero AND zero  
    END FUNCTION  
END CLASS
