CLASS Solution  
    FUNCTION topKFrequent(nums LIST OF integer, k integer) RETURNS LIST OF integer  
        SET frequency_map TO a new default dictionary mapping integer TO integer WITH default value zero  
        FOR each num IN nums  
            INCREMENT the value at key num IN frequency_map BY one  
        END FOR  
        
        SET max_frequency TO the maximum value AMONG all values OF frequency_map  
        
        SET buckets TO a new list OF empty lists WITH length equal TO max_frequency PLUS one  
        
        FOR each num, freq IN the items OF frequency_map  
            APPEND num TO the list AT position freq OF buckets  
        END FOR  
        
        SET result TO an empty list  
        
        FOR freq FROM max_frequency DOWN TO one BY decrementing one  
            IF the length OF the list AT position freq OF buckets IS GREATER THAN zero  
                APPEND all elements OF the list AT position freq OF buckets TO result  
                IF the length OF result IS GREATER THAN OR EQUAL TO k  
                    RETURN the first k elements OF result  
                END IF  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
