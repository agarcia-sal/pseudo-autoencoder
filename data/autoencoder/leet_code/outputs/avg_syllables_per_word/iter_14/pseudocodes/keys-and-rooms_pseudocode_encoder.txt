CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO an empty set  
        SET stack TO a list containing zero  
        
        WHILE the LENGTH OF stack IS GREATER THAN zero  
            SET current_room TO the element REMOVED FROM the END OF stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room OF rooms  
                IF key IS NOT IN visited  
                    ADD key TO stack  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF visited IS EQUAL TO the LENGTH OF rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
