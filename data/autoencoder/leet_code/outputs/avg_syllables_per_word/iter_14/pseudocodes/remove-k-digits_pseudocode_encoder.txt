CLASS Solution  
    FUNCTION removeKdigits(num AS string, k AS integer) AS string  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at position the LENGTH OF stack MINUS one OF stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero  
            SET final_stack TO all elements EXCEPT the last k elements OF stack  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET concatenated_string TO concatenate all elements OF final_stack INTO one string  
        SET result TO concatenated_string WITH all leading zero characters REMOVED  
        
        IF result IS empty  
            RETURN string zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
