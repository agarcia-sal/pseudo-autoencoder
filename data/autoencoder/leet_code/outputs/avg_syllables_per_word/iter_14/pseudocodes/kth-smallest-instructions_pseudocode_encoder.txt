CLASS Solution  
    FUNCTION kthSmallestPath WITH PARAMETERS destination AND k  
        SET row TO the element at position zero OF destination  
        SET col TO the element at position one OF destination  
        SET path TO an empty list  
        SET total_moves TO col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col IS GREATER THAN zero  
                SET paths_with_H TO the number of combinations OF row PLUS col MINUS one CHOOSE col MINUS one  
                IF k IS LESS THAN OR EQUAL TO paths_with_H  
                    APPEND the character H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND the character V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND the character V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN the concatenation OF all elements OF path INTO a single string  
    END FUNCTION  
END CLASS
