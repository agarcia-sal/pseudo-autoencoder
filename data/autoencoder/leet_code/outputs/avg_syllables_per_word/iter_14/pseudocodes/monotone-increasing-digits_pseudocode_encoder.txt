CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list CREATED FROM the string representation OF n  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR i FROM length MINUS one DOWN TO one  
            IF the element at position i OF digits IS LESS THAN the element at position i MINUS one OF digits  
                SET the element at position i MINUS one OF digits TO the string representation OF the integer value OF the element at position i MINUS one OF digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i OF digits TO the string representation OF nine  
        END FOR  
        SET result_string TO the concatenation OF all elements IN digits IN ORDER  
        SET result_integer TO the integer value OF result_string  
        RETURN result_integer  
    END FUNCTION  
END CLASS
