CLASS Solution  
    FUNCTION pourWater WITH PARAMETERS heights AS list of integers volume AS integer k AS integer RETURNS list of integers  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i IS GREATER THAN zero AND element at position i MINUS one OF heights IS LESS THAN OR EQUAL TO element at position i OF heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i IS LESS THAN k AND element at position i OF heights IS EQUAL TO element at position i PLUS one OF heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i IS LESS THAN k  
                INCREMENT element at position i OF heights BY one  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i IS LESS THAN n MINUS one AND element at position i PLUS one OF heights IS LESS THAN OR EQUAL TO element at position i OF heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i IS GREATER THAN k AND element at position i OF heights IS EQUAL TO element at position i MINUS one OF heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i IS GREATER THAN k  
                INCREMENT element at position i OF heights BY one  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            
            INCREMENT element at position k OF heights BY one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
