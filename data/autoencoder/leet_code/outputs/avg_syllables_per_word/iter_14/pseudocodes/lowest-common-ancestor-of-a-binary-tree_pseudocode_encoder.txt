CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode USING THE element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND THE element at position i OF values IS NOT EQUAL TO none  
            SET THE left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING THE element at position i OF values  
            APPEND THE left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND THE element at position i OF values IS NOT EQUAL TO none  
            SET THE right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode USING THE element at position i OF values  
            APPEND THE right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF THE val ATTRIBUTE OF p IS NOT EQUAL TO THE val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN THE LOGICAL AND OF is_same_tree WITH THE left ATTRIBUTE OF p AND THE left ATTRIBUTE OF q AND is_same_tree WITH THE right ATTRIBUTE OF p AND THE right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor WITH PARAMETERS self root p q  
        IF root IS none OR root IS EQUAL TO p OR root IS EQUAL TO q  
            RETURN root  
        END IF  
        SET left TO self.lowestCommonAncestor WITH THE left ATTRIBUTE OF root AND p AND q  
        SET right TO self.lowestCommonAncestor WITH THE right ATTRIBUTE OF root AND p AND q  
        IF left IS NOT none AND right IS NOT none  
            RETURN root  
        END IF  
        IF left IS NOT none  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
