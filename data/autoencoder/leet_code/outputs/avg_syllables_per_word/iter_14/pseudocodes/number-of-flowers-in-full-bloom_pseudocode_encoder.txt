CLASS Solution  
    FUNCTION fullBloomFlowers(flowers AS list OF list OF integer, people AS list OF integer) RETURNS list OF integer  
        SET start_times TO A LIST OF start times EXTRACTED FROM each flower IN flowers  
        SORT start_times IN ascending order  
        SET end_times TO A LIST OF end times EXTRACTED FROM each flower IN flowers  
        SORT end_times IN ascending order  
        
        SET result TO an empty list  
        
        FOR each person IN people  
            SET started TO THE position FOUND BY binary search FOR the first element GREATER THAN person IN start_times  
            SET ended TO THE position FOUND BY binary search FOR the first element GREATER THAN OR EQUAL TO person IN end_times  
            SET bloom_count TO started MINUS ended  
            APPEND bloom_count TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
