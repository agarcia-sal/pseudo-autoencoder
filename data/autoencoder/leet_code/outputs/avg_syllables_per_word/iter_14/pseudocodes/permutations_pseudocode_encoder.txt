CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining EQUAL TO nothing  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO path PLUS element at position index OF remaining  
                SET new_remaining TO elements FROM position zero TO index MINUS one OF remaining PLUS elements FROM position index PLUS one TO the END OF remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  

        SET permutations TO an empty list  
        CALL backtrack WITH an empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
