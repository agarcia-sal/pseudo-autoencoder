CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO the sequence s converted to an iterator  
            FOR each character IN word  
                IF character IS NOT FOUND IN iterator SEQUENTIALLY  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
  
        SORT dictionary BY length of word IN DESCENDING ORDER AND BY lexicographical order IN ASCENDING ORDER  
  
        FOR each word IN dictionary  
            IF is_subsequence(word, s) IS true  
                RETURN word  
            END IF  
        END FOR  
  
        RETURN empty string  
    END FUNCTION  
END CLASS
