CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AND t  
        USE a dictionary that returns an empty list for missing keys CALLED pos  
        FOR each index i AND character c IN s  
            PUSH index i INTO the heap LIST OF the INTEGER VALUE OF character c IN pos  
        END FOR  
        
        FOR each index i AND character c IN t  
            SET digit TO the INTEGER VALUE OF character c  
            IF the heap LIST FOR digit IN pos IS empty  
                RETURN false  
            END IF  
            
            FOR each smaller FROM zero TO digit MINUS one  
                IF the heap LIST FOR smaller IN pos IS NOT empty AND the FIRST element OF the heap LIST FOR smaller IN pos IS LESS THAN the FIRST element OF the heap LIST FOR digit IN pos  
                    RETURN false  
                END IF  
            END FOR  
            
            REMOVE the FIRST element FROM the heap LIST FOR digit IN pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
