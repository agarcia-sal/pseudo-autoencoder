CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s AND k  
        IF k IS EQUAL TO zero  
            RETURN zero  
        END IF  
        
        SET char_map TO an empty mapping of characters to positions  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN the element at position right OF the string s AS key IN char_map WITH value right  
            
            IF the SIZE OF char_map IS GREATER THAN k  
                SET leftmost_char TO the key WITH the MINIMUM value IN char_map  
                SET left TO the value CORRESPONDING TO leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET current_length TO right MINUS left PLUS one  
            IF current_length IS GREATER THAN max_length  
                SET max_length TO current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
