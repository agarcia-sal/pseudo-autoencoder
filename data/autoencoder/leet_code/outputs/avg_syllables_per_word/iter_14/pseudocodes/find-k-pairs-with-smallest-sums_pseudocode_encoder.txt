CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF the LENGTH OF nums1 IS EQUAL TO zero OR the LENGTH OF nums2 IS EQUAL TO zero OR k IS LESS THAN OR EQUAL TO zero  
            RETURN an empty list  
        END IF  
        
        SET min_heap TO an empty list  
        
        FOR index i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            CALL heap_push ON min_heap WITH the element CONSISTING OF THE SUM OF element at position i OF nums1 PLUS element at position zero OF nums2 FOLLOWED BY i FOLLOWED BY zero  
        END FOR  
        
        SET result TO an empty list  
        
        WHILE the LENGTH OF min_heap IS GREATER THAN zero AND the LENGTH OF result IS LESS THAN k  
            CALL heap_pop ON min_heap AND SET THE RETURNED SECOND AND THIRD ELEMENTS TO variables i AND j RESPECTIVELY  
            APPEND a list CONSISTING OF element at position i OF nums1 AND element at position j OF nums2 TO result  
            
            IF j PLUS one IS LESS THAN the LENGTH OF nums2  
                CALL heap_push ON min_heap WITH the element CONSISTING OF THE SUM OF element at position i OF nums1 PLUS element at position j PLUS one OF nums2 FOLLOWED BY i FOLLOWED BY j PLUS one  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
