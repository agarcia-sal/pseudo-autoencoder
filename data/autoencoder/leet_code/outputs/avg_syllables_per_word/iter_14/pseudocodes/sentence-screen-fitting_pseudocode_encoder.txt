CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the RESULT OF joining all elements in sentence WITH a single space PLUS a single space character  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the ELEMENT at POSITION position MODULO total_length OF s IS EQUAL TO a single space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position IS GREATER THAN zero AND the ELEMENT at POSITION position MINUS one MODULO total_length OF s IS NOT EQUAL TO a single space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
