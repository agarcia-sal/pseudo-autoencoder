CLASS Solution  
    FUNCTION kIncreasing(arr AS List OF integer, k AS integer) AS integer  
        FUNCTION longest_increasing_subsequence(subarr AS List OF integer) AS integer  
            SET lis TO an empty list  
            FOR each num IN subarr  
                IF lis IS empty OR num IS GREATER THAN OR EQUAL TO the last element OF lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position obtained by finding the first element GREATER THAN num IN lis using binary search from the right  
                    SET the element at position pos OF lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
      
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
      
        FOR each start FROM zero TO k MINUS one  
            SET subarr TO an empty list  
            FOR each i STARTING FROM start TO less than n IN STEPS OF k  
                APPEND the element at position i OF arr TO subarr  
            END FOR  
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  
      
        RETURN total_operations  
    END FUNCTION  
END CLASS
