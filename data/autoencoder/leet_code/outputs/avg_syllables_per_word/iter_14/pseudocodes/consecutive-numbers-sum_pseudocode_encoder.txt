CLASS Solution  
    FUNCTION consecutiveNumbersSum WITH PARAMETER n  
        SET count TO zero  
        FOR k FROM one TO n INCLUSIVE  
            IF the product of two AND n MODULO k EQUAL TO zero  
                SET m TO the quotient of the difference between the quotient of the product of two AND n DIVIDED BY k AND k PLUS one DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUAL TO the integer part of m  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
