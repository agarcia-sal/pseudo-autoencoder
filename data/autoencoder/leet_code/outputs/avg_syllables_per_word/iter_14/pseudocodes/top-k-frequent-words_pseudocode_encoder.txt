CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO a mapping of word frequencies calculated from words  
        SET heap TO an empty list  
        FOR each word AND frequency IN count  
            APPEND a tuple of negative frequency AND word TO heap  
        END FOR  
        TRANSFORM heap INTO a valid heap structure to maintain heap properties  
        SET result TO an empty list  
        FOR iteration FROM one TO k  
            EXTRACT the smallest element from heap and GET the word from the tuple  
            APPEND the extracted word TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
