CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EQUAL TO empty OR element at position zero OF matrix IS EQUAL TO empty  
            RETURN false  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero OF matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the QUOTIENT OF left PLUS right DIVIDED BY two  
            SET mid_value TO element at position the QUOTIENT OF mid DIVIDED BY n OF matrix AT position the REMAINDER OF mid DIVIDED BY n  
            
            IF mid_value IS EQUAL TO target  
                RETURN true  
            ELSE IF mid_value IS LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        RETURN false  
    END FUNCTION  
END CLASS
