CLASS Solution  
    FUNCTION maximumBeauty(flowers AS list OF integers, newFlowers AS integer, target AS integer, full AS integer, partial AS integer) RETURNS integer  
        CALL a METHOD TO arrange elements OF flowers IN ascending ORDER  
        SET n TO the LENGTH OF flowers  
        SET max_beauty TO zero  
        
        CREATE a list prefix_sum OF size n PLUS one WITH all elements SET TO zero  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF flowers  
        END FOR  
        
        SET complete_gardens TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF element at position i OF flowers IS GREATER THAN OR EQUAL TO target THEN  
                INCREMENT complete_gardens BY one  
            ELSE  
                BREAK the FOR loop  
            END IF  
        END FOR  
        
        SET remaining_flowers TO newFlowers  
        SET max_incomplete_flowers TO zero  
        
        FOR i FROM complete_gardens TO n  
            IF i IS GREATER THAN zero THEN  
                SET remaining_flowers TO remaining_flowers MINUS the DIFFERENCE BETWEEN target AND element at position n MINUS i OF flowers  
                IF remaining_flowers IS LESS THAN zero THEN  
                    BREAK the FOR loop  
                END IF  
            END IF  
            
            SET left TO zero  
            SET right TO n MINUS i MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO the INTEGER DIVISION of the SUM of left AND right BY two  
                SET cost TO element at position mid OF flowers MULTIPLIED BY mid PLUS one MINUS element at position mid PLUS one OF prefix_sum  
                IF cost IS GREATER THAN remaining_flowers THEN  
                    SET right TO right MINUS one  
                ELSE  
                    SET left TO left PLUS one  
                END IF  
            END WHILE  
            
            IF right IS GREATER THAN OR EQUAL TO zero THEN  
                SET max_incomplete_flowers TO element at position right OF flowers PLUS the INTEGER DIVISION OF the DIFFERENCE BETWEEN remaining_flowers AND the DIFFERENCE BETWEEN element at position right OF flowers MULTIPLIED BY right PLUS one AND element at position right PLUS one OF prefix_sum BY right PLUS one  
            ELSE  
                SET max_incomplete_flowers TO zero  
            END IF  
            
            SET total_beauty TO i MULTIPLIED BY full PLUS the PRODUCT OF the MINIMUM OF max_incomplete_flowers AND target MINUS one AND partial  
            IF total_beauty IS GREATER THAN max_beauty THEN  
                SET max_beauty TO total_beauty  
            END IF  
        END FOR  
        
        RETURN max_beauty  
    END FUNCTION  
END CLASS
