CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of length fifteen FILLED WITH infinity  
        FOR each tire IN tires  
            ASSIGN f TO the first element OF tire  
            ASSIGN r TO the second element OF tire  
            SET total_time TO zero  
            FOR i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE power OF i  
                IF lap_time IS GREATER THAN changeTime PLUS f  
                    BREAK the loop  
                END IF  
                INCREMENT total_time BY lap_time  
                IF total_time IS LESS THAN min_times element at position i  
                    SET min_times element at position i TO total_time  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO a list of length numLaps PLUS one FILLED WITH infinity  
        SET dp element at position zero TO zero
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET candidate_time TO dp element at position i MINUS j MINUS one PLUS min_times element at position j PLUS changeTime  
                IF candidate_time IS LESS THAN dp element at position i  
                    SET dp element at position i TO candidate_time  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO dp element at position numLaps MINUS changeTime  
        RETURN result  
    END FUNCTION  
END CLASS
