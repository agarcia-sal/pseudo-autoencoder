CLASS Solution  
    FUNCTION numberOfArrays WITH PARAMETERS s AND k  
        SET MOD TO ten TO THE POWER OF nine PLUS one  
        SET n TO the LENGTH OF s  
        
        SET dp TO a LIST OF zeroES OF LENGTH n PLUS one  
        SET element at position zero OF dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF the CHARACTER at position j OF s IS NOT EQUAL TO zero AND  
                   the NUMBER FORMED BY the SUBSTRING FROM position j TO i MINUS one OF s IS GREATER THAN OR EQUAL TO one AND  
                   the SAME NUMBER IS LESS THAN OR EQUAL TO k  
                    SET element at position i OF dp TO the REMAINDER OF the SUM of element at position i OF dp PLUS element at position j OF dp DIVIDED BY MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
