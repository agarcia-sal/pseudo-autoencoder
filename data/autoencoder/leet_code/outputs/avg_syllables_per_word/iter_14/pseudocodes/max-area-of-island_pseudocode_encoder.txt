CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid IS empty  
            RETURN zero  
        END IF  
        
        SET max_area TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero OF grid  
        
        FUNCTION dfs(r, c)  
            IF r LESS THAN zero OR r GREATER THAN OR EQUAL TO rows OR c LESS THAN zero OR c GREATER THAN OR EQUAL TO cols OR element at position r OF grid AT position c EQUAL TO zero  
                RETURN zero  
            END IF  
            SET element at position r OF grid AT position c TO zero  
            SET area TO one  
            INCREMENT area BY dfs(r PLUS one, c)  
            INCREMENT area BY dfs(r MINUS one, c)  
            INCREMENT area BY dfs(r, c PLUS one)  
            INCREMENT area BY dfs(r, c MINUS one)  
            RETURN area  
        END FUNCTION  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r OF grid AT position c EQUAL TO one  
                    SET max_area TO the GREATER OF max_area AND dfs(r, c)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
