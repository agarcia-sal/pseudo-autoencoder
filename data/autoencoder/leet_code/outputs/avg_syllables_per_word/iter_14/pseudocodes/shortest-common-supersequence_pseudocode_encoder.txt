CLASS Solution  
    FUNCTION shortestCommonSupersequence WITH PARAMETERS str1 AS STRING AND str2 AS STRING  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a two dimensional list OF zero WITH m PLUS one ROWS AND n PLUS one COLUMNS
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the CHARACTER AT position i MINUS one OF str1 IS EQUAL TO the CHARACTER AT position j MINUS one OF str2  
                    SET the ELEMENT AT position i AND j OF dp TO the ELEMENT AT position i MINUS one AND j MINUS one OF dp PLUS one  
                ELSE  
                    IF the ELEMENT AT position i MINUS one AND j OF dp IS GREATER THAN the ELEMENT AT position i AND j MINUS one OF dp  
                        SET the ELEMENT AT position i AND j OF dp TO the ELEMENT AT position i MINUS one AND j OF dp  
                    ELSE  
                        SET the ELEMENT AT position i AND j OF dp TO the ELEMENT AT position i AND j MINUS one OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i IS GREATER THAN zero AND j IS GREATER THAN zero  
            IF the CHARACTER AT position i MINUS one OF str1 IS EQUAL TO the CHARACTER AT position j MINUS one OF str2  
                APPEND the CHARACTER AT position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the ELEMENT AT position i MINUS one AND j OF dp IS GREATER THAN the ELEMENT AT position i AND j MINUS one OF dp  
                APPEND the CHARACTER AT position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the CHARACTER AT position j MINUS one OF str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i IS GREATER THAN zero  
            APPEND the CHARACTER AT position i MINUS one OF str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j IS GREATER THAN zero  
            APPEND the CHARACTER AT position j MINUS one OF str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE THE ORDER OF ELEMENTS IN scs  
        
        SET result_string TO the CONCATENATION OF ALL ELEMENTS IN scs INTO A SINGLE STRING  
        
        RETURN result_string  
    END FUNCTION  
END CLASS
