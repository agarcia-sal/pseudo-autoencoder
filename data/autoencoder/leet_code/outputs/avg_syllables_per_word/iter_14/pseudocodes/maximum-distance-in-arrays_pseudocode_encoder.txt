CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO element at position zero OF element at position zero OF arrays  
        SET max_val TO element at position zero OF element at position last OF arrays  
        SET max_distance TO zero  
        FOR index i FROM one TO the LENGTH OF arrays MINUS one  
            SET current_array TO element at position i OF arrays  
            SET current_min TO element at position zero OF current_array  
            SET current_max TO element at position last OF current_array  
            IF max_distance IS LESS THAN absolute value of current_max MINUS min_val  
                SET max_distance TO absolute value of current_max MINUS min_val  
            END IF  
            IF max_distance IS LESS THAN absolute value of max_val MINUS current_min  
                SET max_distance TO absolute value of max_val MINUS current_min  
            END IF  
            IF min_val IS GREATER THAN current_min  
                SET min_val TO current_min  
            END IF  
            IF max_val IS LESS THAN current_max  
                SET max_val TO current_max  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
