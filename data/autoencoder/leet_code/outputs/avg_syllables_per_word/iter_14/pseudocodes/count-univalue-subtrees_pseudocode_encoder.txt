CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val EQUAL TO element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node dot left TO NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node dot right TO NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN TRUE  
    ELSE IF p IS none OR q IS none  
        RETURN FALSE  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN FALSE  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left  
               AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees WITH PARAMETERS self root  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN TRUE COMMA zero  
            END IF  
            SET is_left_unival COMMA left_count TO helper WITH PARAMETER node dot left  
            SET is_right_unival COMMA right_count TO helper WITH PARAMETER node dot right  
            IF is_left_unival AND is_right_unival  
                IF (node dot left IS none OR node dot left dot val IS EQUAL TO node dot val) AND  
                   (node dot right IS none OR node dot right dot val IS EQUAL TO node dot val)  
                    RETURN TRUE COMMA left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN FALSE COMMA left_count PLUS right_count  
        END FUNCTION  
        SET underscore COMMA count TO helper WITH PARAMETER root  
        RETURN count  
    END FUNCTION  
END CLASS
