CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUAL TO zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO the RESULT OF calling grayCode WITH n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one  
        SET reversed_prev_gray TO the REVERSED ORDER OF prev_gray  
        SET extended_list TO an EMPTY list  
        FOR each num IN reversed_prev_gray  
            SET combined_num TO mask OR num  
            APPEND combined_num TO extended_list  
        END FOR  
        SET current_gray TO prev_gray CONCATENATED WITH extended_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
