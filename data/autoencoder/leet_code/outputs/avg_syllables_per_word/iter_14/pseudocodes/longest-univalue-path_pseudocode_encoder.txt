CLASS TreeNode  
    FUNCTION __init__(self, val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS NOT PRESENT  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position zero OF values  
    SET i TO one  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO element removed from front OF queue  
        IF i IS LESS THAN length OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value equal to element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN length OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value equal to element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_length AND left_value TO result of dfs with argument node.left  
            SET right_length AND right_value TO result of dfs with argument node.right  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node.left IS PRESENT AND node.left.val IS EQUAL TO node.val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node.right IS PRESENT AND node.right.val IS EQUAL TO node.val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the greater of self.max_length AND left_arrow PLUS right_arrow  
            RETURN the greater of left_arrow AND right_arrow AND node.val  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with argument root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
