CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR element at position x of grid at position y EQUAL TO zero  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND tuple of x AND y TO island  
            CALL dfs WITH x PLUS one AND y AND island  
            CALL dfs WITH x MINUS one AND y AND island  
            CALL dfs WITH x AND y PLUS one AND island  
            CALL dfs WITH x AND y MINUS one AND island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx AND dy IN the list of pairs consisting of one AND one, one AND minus one, minus one AND one, minus one AND minus one  
                SET new_island TO empty list  
                FOR each pair x AND y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island IN ascending order  
                SET min_x TO the minimum value of x FOR each pair x AND y IN new_island  
                SET min_y TO the minimum value of y FOR each pair x AND y IN new_island  
                APPEND tuple of pairs consisting of x MINUS min_x AND y MINUS min_y FOR each pair x AND y IN new_island TO shapes  
            END FOR  
            FOR index FROM zero TO the LENGTH OF shapes MINUS one  
                SET rotated_shape TO tuple of pairs consisting of y AND negative x FOR each pair x AND y IN element at position index OF shapes  
                APPEND rotated_shape TO shapes  
            END FOR  
            RETURN the minimum element IN shapes  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i of grid at position j EQUAL TO one  
                    SET island TO empty list  
                    CALL dfs WITH i AND j AND island  
                    ADD normalize(island) TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
