CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  
  
FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO AN INSTANCE OF TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT OF queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT None  
            SET node left TO AN INSTANCE OF TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT None  
            SET node right TO AN INSTANCE OF TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  
  
FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p val IS NOT EQUAL TO q val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p left AND q left AND is_same_tree WITH PARAMETERS p right AND q right  
    END IF  
END FUNCTION  
  
CLASS Solution  
    FUNCTION largestBSTSubtree WITH PARAMETERS self root  
          
        FUNCTION dfs WITH PARAMETER node  
            IF node IS None  
                RETURN positive infinity FOR MINIMUM VALUE positive infinity NEGATED FOR MAXIMUM VALUE AND zero FOR SIZE  
            END IF  
            SET left_min left_max left_size TO dfs WITH PARAMETER node left  
            SET right_min right_max right_size TO dfs WITH PARAMETER node right  
            IF left_max IS LESS THAN node val AND node val IS LESS THAN right_min  
                RETURN MINIMUM OF node val AND left_min FOR MINIMUM VALUE MAXIMUM OF node val AND right_max FOR MAXIMUM VALUE AND left_size PLUS right_size PLUS one FOR SIZE  
            END IF  
            RETURN negative infinity FOR MINIMUM VALUE positive infinity FOR MAXIMUM VALUE AND MAXIMUM OF left_size AND right_size FOR SIZE  
        END FUNCTION  
          
        RETURN THE THIRD ELEMENT OF dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
