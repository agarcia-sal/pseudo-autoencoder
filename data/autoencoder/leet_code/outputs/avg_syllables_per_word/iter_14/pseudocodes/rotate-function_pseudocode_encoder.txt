CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET total_sum TO the SUM OF all elements IN nums  
        SET current_sum TO zero  
        FOR each index i FROM zero TO n MINUS one  
            INCREMENT current_sum BY i MULTIPLIED BY the element at position i OF nums  
        END FOR  
        
        SET max_value TO current_sum  
        
        FOR k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY the element at position n MINUS k OF nums  
            IF current_sum GREATER THAN max_value  
                SET max_value TO current_sum  
            END IF  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
