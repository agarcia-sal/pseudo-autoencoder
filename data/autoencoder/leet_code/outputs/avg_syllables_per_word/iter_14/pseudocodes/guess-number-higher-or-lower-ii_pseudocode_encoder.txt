CLASS Solution  
    FUNCTION getMoneyAmount WITH PARAMETER n  
        FUNCTION dp WITH PARAMETERS left right  
            IF left IS GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS THE GREATER OF dp WITH PARAMETERS left AND pivot MINUS one AND dp WITH PARAMETERS pivot PLUS one AND right  
                IF cost IS LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp WITH PARAMETERS one AND n  
    END FUNCTION  
END CLASS
