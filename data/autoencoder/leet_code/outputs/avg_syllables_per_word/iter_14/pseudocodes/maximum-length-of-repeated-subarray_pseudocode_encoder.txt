CLASS Solution  
    FUNCTION findLength(nums1 AS List of Integer, nums2 AS List of Integer) RETURNS Integer  
        SET dp TO a two dimensional list of zero values with rows equal to the LENGTH OF nums1 PLUS one and columns equal to the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF element at position i MINUS one OF nums1 IS EQUAL TO element at position j MINUS one OF nums2  
                    SET element at row i column j OF dp TO element at row i MINUS one column j MINUS one OF dp PLUS one  
                    IF element at row i column j OF dp IS GREATER THAN max_length  
                        SET max_length TO element at row i column j OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
