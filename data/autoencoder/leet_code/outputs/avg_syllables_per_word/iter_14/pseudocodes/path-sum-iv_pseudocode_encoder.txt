CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the result of num DIVIDED BY ten THEN MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN value TO the entry at key depth AND pos IN tree  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value in tree AT key depth AND pos  
            IF current_value IS EQUAL TO no value THEN  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            
            SET left_child_depth TO depth PLUS one  
            SET left_child_pos TO two MULTIPLIED BY pos MINUS one  
            SET right_child_depth TO depth PLUS one  
            SET right_child_pos TO two MULTIPLIED BY pos  
            
            IF the entry at key left_child_depth AND left_child_pos NOT IN tree  
            AND the entry at key right_child_depth AND right_child_pos NOT IN tree THEN  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO dfs WITH parameters left_child_depth, left_child_pos, new_path_sum  
            SET right_sum TO dfs WITH parameters right_child_depth, right_child_pos, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs WITH parameters one, one, zero  
    END FUNCTION  
END CLASS
