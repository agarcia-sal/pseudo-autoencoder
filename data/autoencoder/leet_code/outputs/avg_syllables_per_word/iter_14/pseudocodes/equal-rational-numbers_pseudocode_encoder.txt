    FUNCTION isRationalEqual WITH PARAMETERS s AND t  
        SET ratios TO list CONTAINING one FOLLOWED BY one DIVIDED BY nine FOLLOWED BY one DIVIDED BY ninety nine FOLLOWED BY one DIVIDED BY nine hundred ninety nine FOLLOWED BY one DIVIDED BY nine thousand nine hundred ninety nine  

        FUNCTION valueOf WITH PARAMETER s  
            SET positionOfLeftParenthesis TO the POSITION OF the FIRST OCCURRENCE OF the character left parenthesis IN s  
            IF positionOfLeftParenthesis EQUAL TO negative one  
                RETURN the CONVERSION OF s TO floating point number  
            END IF  
            
            SET positionOfRightParenthesis TO the POSITION OF the FIRST OCCURRENCE OF the character right parenthesis IN s  
            SET positionOfDecimalPoint TO the POSITION OF the FIRST OCCURRENCE OF the character decimal point IN s  

            SET integerAndNonRepeating TO the CONVERSION TO floating point number OF the SUBSTRING OF s FROM start position ZERO UP TO positionOfLeftParenthesis MINUS one  
            SET lengthOfNonRepeatingPart TO positionOfLeftParenthesis MINUS positionOfDecimalPoint MINUS one  

            SET repeatingPart TO the CONVERSION TO integer OF the SUBSTRING OF s FROM positionOfLeftParenthesis PLUS one UP TO positionOfRightParenthesis MINUS one  
            SET lengthOfRepeatingPart TO positionOfRightParenthesis MINUS positionOfLeftParenthesis MINUS one  

            SET value TO integerAndNonRepeating PLUS repeatingPart MULTIPLIED BY ten RAISED TO THE POWER OF negative lengthOfNonRepeatingPart MULTIPLIED BY the ELEMENT AT position lengthOfRepeatingPart OF ratios  
            RETURN value  
        END FUNCTION  

        SET difference TO the ABSOLUTE VALUE OF valueOf WITH ARGUMENT s MINUS valueOf WITH ARGUMENT t  
        IF difference LESS THAN one TIMES ten RAISED TO THE POWER OF negative nine  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
