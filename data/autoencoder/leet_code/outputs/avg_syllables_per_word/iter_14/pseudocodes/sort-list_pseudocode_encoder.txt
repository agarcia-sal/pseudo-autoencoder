CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN elements FROM position one TO THE END OF values  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN self.val EQUAL TO p2.val AND is_same_list WITH PARAMETERS p1.next p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList WITH PARAMETERS self head  
        IF head IS none OR head.next IS none  
            RETURN head  
        END IF  

        FUNCTION split WITH PARAMETER head  
            SET slow TO head  
            SET fast TO head.next  
            WHILE fast IS NOT none AND fast.next IS NOT none  
                SET slow TO slow.next  
                SET fast TO fast.next.next  
            END WHILE  
            SET mid TO slow.next  
            SET slow.next TO none  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge WITH PARAMETERS l1 l2  
            SET dummy TO A NEW ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT none AND l2 IS NOT none  
                IF l1.val LESS THAN l2.val  
                    SET current.next TO l1  
                    SET l1 TO l1.next  
                ELSE  
                    SET current.next TO l2  
                    SET l2 TO l2.next  
                END IF  
                SET current TO current.next  
            END WHILE  
            IF l1 IS NOT none  
                SET current.next TO l1  
            ELSE  
                SET current.next TO l2  
            END IF  
            RETURN dummy.next  
        END FUNCTION  

        SET left AND right TO split WITH PARAMETER head  
        SET left TO self.sortList WITH PARAMETER left  
        SET right TO self.sortList WITH PARAMETER right  
        RETURN merge WITH PARAMETERS left right  
    END FUNCTION  
END CLASS
