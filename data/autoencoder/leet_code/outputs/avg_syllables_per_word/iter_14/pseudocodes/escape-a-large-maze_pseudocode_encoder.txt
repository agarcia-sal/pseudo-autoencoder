CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked IS EMPTY  
            RETURN true  
        END IF  
        
        SET blocked_set TO a set containing each element CONVERTED TO a tuple FROM blocked  
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY n MINUS one THEN DIVIDED BY two  
        
        FUNCTION bfs(start, end)  
            SET queue TO a double ended queue INITIALIZED WITH a tuple CONVERTED FROM start  
            SET visited TO a set INITIALIZED WITH queue  
            
            WHILE queue IS NOT EMPTY  
                REMOVE and RETURN the first element FROM queue ASSIGN TO x and y  
                
                IF the LENGTH OF visited IS GREATER THAN max_steps OR x IS EQUAL TO the first element OF end AND y IS EQUAL TO the second element OF end  
                    RETURN true  
                END IF  
                
                FOR each pair dx and dy IN the list of four pairs zero and one zero and minus one one and zero minus one and zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN ten RAISED TO THE POWER OF six AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN ten RAISED TO THE POWER OF six AND the tuple nx and ny IS NOT IN blocked_set AND the tuple nx and ny IS NOT IN visited  
                        ADD the tuple nx and ny TO visited  
                        APPEND the tuple nx and ny TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN false  
        END FUNCTION  
        
        RETURN the result of bfs CALLED WITH source AND target AND the result of bfs CALLED WITH target AND source  
    END FUNCTION  
END CLASS
