CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences WITH PARAMETER binary  
        SET MODULO_VALUE TO ten RAISED TO THE power OF nine PLUS one  
        SET count_of_subsequences_ending_with_zero TO zero  
        SET count_of_subsequences_ending_with_one TO zero  
        SET presence_of_zero TO boolean value WHETHER zero CHARACTER IS PRESENT IN binary  
        FOR each bit_character IN binary  
            IF bit_character IS EQUAL TO one CHARACTER  
                SET count_of_subsequences_ending_with_one TO count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS one MODULO MODULO_VALUE  
            ELSE IF bit_character IS EQUAL TO zero CHARACTER  
                SET count_of_subsequences_ending_with_zero TO count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one MODULO MODULO_VALUE  
            END IF  
        END FOR  
        SET total_unique_good_subsequences TO count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS presence_of_zero MODULO MODULO_VALUE  
        RETURN total_unique_good_subsequences  
    END FUNCTION  
END CLASS
