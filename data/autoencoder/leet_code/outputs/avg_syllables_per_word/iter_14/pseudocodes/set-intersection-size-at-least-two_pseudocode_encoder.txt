CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals SORTED BY the second element OF each interval IN ASCENDING ORDER, AND IN CASE of TIE BY the negative of the first element OF each interval IN ASCENDING ORDER  
        
        SET first TO the second element OF the first interval MINUS one  
        SET second TO the second element OF the first interval  
        SET count TO two  
        
        FOR each interval FROM the second interval TO the last interval IN intervals  
            SET start TO the first element OF interval  
            SET end TO the second element OF interval  
            
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
