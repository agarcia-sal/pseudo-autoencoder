CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x IS GREATER THAN zero  
                    INCREMENT cnt BY the POSITION THAT THE VALUE p DIVIDED BY x WOULD BE INSERTED TO THE RIGHT IN nums2  
                ELSE IF x IS LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION THAT THE VALUE p DIVIDED BY x WOULD BE INSERTED TO THE LEFT IN nums2  
                ELSE  
                    IF p IS GREATER THAN OR EQUAL TO zero  
                        INCREMENT cnt BY n  
                    END IF  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1  
        MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  

        SET result TO the POSITION THAT k WOULD BE INSERTED TO THE LEFT IN THE RANGE FROM negative mx TO mx PLUS one WITH THE KEY FUNCTION count  
        RETURN result MINUS mx  
    END FUNCTION  
END CLASS
