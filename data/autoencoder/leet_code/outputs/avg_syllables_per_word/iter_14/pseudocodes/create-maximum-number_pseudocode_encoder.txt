CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND the LAST ELEMENT OF stack IS LESS THAN num  
                    REMOVE the LAST ELEMENT FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the FIRST k ELEMENTS OF stack  
        END FUNCTION  
        
        FUNCTION merge(arr1, arr2)  
            SET merged_list TO an empty list  
            WHILE the LENGTH OF arr1 PLUS the LENGTH OF arr2 IS GREATER THAN zero  
                IF arr1 IS GREATER THAN arr2 IN LEXICOGRAPHICAL ORDER  
                    APPEND the FIRST ELEMENT OF arr1 TO merged_list  
                    REMOVE the FIRST ELEMENT FROM arr1  
                ELSE  
                    APPEND the FIRST ELEMENT OF arr2 TO merged_list  
                    REMOVE the FIRST ELEMENT FROM arr2  
                END IF  
            END WHILE  
            RETURN merged_list  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k INCLUSIVE  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(maxSingleArray(nums1, i), maxSingleArray(nums2, k MINUS i))  
                IF max_number IS LESS THAN candidate  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
