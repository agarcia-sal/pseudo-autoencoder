CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF matchsticks  
        IF the REMAINDER OF total_length DIVIDED BY four IS NOT EQUAL TO zero  
            RETURN FALSE  
        END IF  
        
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN DESCENDING ORDER  
        
        FUNCTION can_form_sides(sides, index)  
            IF index IS EQUAL TO the LENGTH OF matchsticks  
                FOR each side IN sides  
                    IF side IS NOT EQUAL TO side_length  
                        RETURN FALSE  
                    END IF  
                END FOR  
                RETURN TRUE  
            END IF  
            
            FOR i FROM zero TO three  
                IF element at position i OF sides PLUS element at position index OF matchsticks IS LESS THAN OR EQUAL TO side_length  
                    INCREMENT element at position i OF sides BY element at position index OF matchsticks  
                    IF can_form_sides(sides, index PLUS one) IS TRUE  
                        RETURN TRUE  
                    END IF  
                    DECREMENT element at position i OF sides BY element at position index OF matchsticks  
                END IF  
                
                IF element at position i OF sides IS EQUAL TO zero  
                    BREAK  
                END IF  
            END FOR  
            
            RETURN FALSE  
        END FUNCTION  
        
        RETURN can_form_sides(A LIST OF four zeros, zero)  
    END FUNCTION  
END CLASS
