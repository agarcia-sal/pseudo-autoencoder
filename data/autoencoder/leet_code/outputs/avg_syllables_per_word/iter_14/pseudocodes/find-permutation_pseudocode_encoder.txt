CLASS Solution  
    FUNCTION findPermutation WITH PARAMETER s  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        
        SET start TO zero  
        FOR end_index FROM zero TO the LENGTH OF s  
            IF end_index EQUAL TO the LENGTH OF s OR the character at position end_index OF s EQUAL TO the letter I  
                SET segment_length TO end_index MINUS start PLUS one  
                SET reversed_segment TO an empty list  
                FOR position FROM zero TO segment_length MINUS one  
                    APPEND the element at position start PLUS segment_length MINUS one MINUS position OF result TO reversed_segment  
                END FOR  
                
                FOR position FROM zero TO segment_length MINUS one  
                    SET the element at position start PLUS position OF result TO the element at position position OF reversed_segment  
                END FOR  
                
                SET start TO end_index PLUS one  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
