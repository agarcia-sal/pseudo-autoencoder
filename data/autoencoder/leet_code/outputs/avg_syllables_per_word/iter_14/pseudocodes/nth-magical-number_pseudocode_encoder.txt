CLASS Solution  
    FUNCTION nthMagicalNumber WITH PARAMETERS n AND a AND b  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        
        FUNCTION gcd WITH PARAMETERS x AND y  
            WHILE y IS NOT EQUAL TO zero  
                SET temporary TO y  
                SET y TO x MODULO y  
                SET x TO temporary  
            END WHILE  
            RETURN x  
        END FUNCTION  
        
        FUNCTION lcm WITH PARAMETERS x AND y  
            RETURN x MULTIPLIED BY y DIVIDED BY gcd OF x AND y  
        END FUNCTION  
        
        SET L TO lcm OF a AND b  
        
        SET left TO one  
        SET right TO n MULTIPLIED BY the MINIMUM OF a AND b  
        
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY two TAKING THE FLOOR VALUE  
            SET count TO mid DIVIDED BY a PLUS mid DIVIDED BY b MINUS mid DIVIDED BY L  
            
            IF count IS LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left MODULO MOD  
    END FUNCTION  
END CLASS
