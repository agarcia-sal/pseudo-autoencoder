CLASS Solution  
    FUNCTION minimumTime(n, relations, time)  
        SET graph TO result of calling a function that creates a new default dictionary with list as default value  
        FOR each pair prevCourse and nextCourse IN relations  
            APPEND prevCourse TO the value at key nextCourse IN graph  
        END FOR  
        
        SET memo TO a list of length n PLUS one FILLED with negative one  
        
        FUNCTION dfs(course)  
            IF element at position course OF memo IS NOT EQUAL TO negative one  
                RETURN element at position course OF memo  
            END IF  
            
            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN the value at key course OF graph  
                SET dfs_result TO result of calling dfs WITH prevCourse  
                IF dfs_result IS GREATER THAN max_prerequisite_time  
                    SET max_prerequisite_time TO dfs_result  
                END IF  
            END FOR  
            
            SET element at position course OF memo TO max_prerequisite_time PLUS element at position course MINUS one OF time  
            RETURN element at position course OF memo  
        END FUNCTION  
        
        SET total_minimum_time TO zero  
        FOR course FROM one TO n  
            SET course_time TO result of calling dfs WITH course  
            IF course_time IS GREATER THAN total_minimum_time  
                SET total_minimum_time TO course_time  
            END IF  
        END FOR  
        
        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
