CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF the element at position zero OF dungeon  
        SET dp TO a two-dimensional list of dimensions m PLUS one BY n PLUS one filled with infinity values  
        SET the element at position m OF dp AND position n MINUS one TO one  
        SET the element at position m MINUS one OF dp AND position n TO one  
        
        FOR i FROM m MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                SET min_health TO the MINIMUM OF the element at position i PLUS one OF dp AND position j AND the element at position i OF dp AND position j PLUS one MINUS the element at position i OF dungeon AND position j  
                IF min_health IS LESS THAN one  
                    SET the element at position i OF dp AND position j TO one  
                ELSE  
                    SET the element at position i OF dp AND position j TO min_health  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position zero OF dp AND position zero  
    END FUNCTION  
END CLASS
