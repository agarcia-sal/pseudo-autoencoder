CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH PARAMETER s  
        SET char_index_map TO an empty mapping from character TO integer  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end FROM zero TO the LENGTH OF s MINUS one AND corresponding character char AT position end OF s  
            IF char EXISTS AS A KEY IN char_index_map AND the VALUE MAPPED BY char IN char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the VALUE MAPPED BY char IN char_index_map PLUS one  
            END IF  
            ASSIGN the VALUE end TO the ENTRY AT KEY char IN char_index_map  
            SET max_length TO THE GREATER OF max_length AND the DIFFERENCE BETWEEN end AND start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
