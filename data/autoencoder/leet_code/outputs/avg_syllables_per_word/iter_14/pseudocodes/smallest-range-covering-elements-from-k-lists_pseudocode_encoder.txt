CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity value  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push ON min_heap WITH element CONSTRUCTED FROM element at position zero OF nums at position index AND index AND zero  
            IF max_value IS LESS THAN element at position zero OF nums at position index THEN  
                SET max_value TO element at position zero OF nums at position index  
            END IF  
        END FOR  
        
        SET smallest_range TO positive infinity value  
        SET result TO list containing negative one hundred thousand AND positive one hundred thousand  
        
        WHILE min_heap IS NOT empty  
            CALL heap pop ON min_heap AND DESTRUCTURE result INTO min_value list_index element_index  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            
            IF element_index PLUS one IS LESS THAN the LENGTH OF nums at position list_index THEN  
                SET next_value TO element at position element_index PLUS one OF nums at position list_index  
                CALL heap push ON min_heap WITH element CONSTRUCTED FROM next_value AND list_index AND element_index PLUS one  
                IF max_value IS LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
