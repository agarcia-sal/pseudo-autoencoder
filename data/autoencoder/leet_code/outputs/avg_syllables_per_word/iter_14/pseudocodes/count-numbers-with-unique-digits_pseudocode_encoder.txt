CLASS Solution  
    FUNCTION countNumbersWithUniqueDigits WITH PARAMETER n  
        IF n EQUAL TO zero  
            RETURN one  
        END IF  
        IF n EQUAL TO one  
            RETURN ten  
        END IF  
        SET count TO ten  
        SET available_digits TO nine  
        SET choices TO nine  
        FOR i FROM two TO n  
            SET choices TO choices MULTIPLIED BY available_digits  
            INCREMENT count BY choices  
            DECREMENT available_digits BY one  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
