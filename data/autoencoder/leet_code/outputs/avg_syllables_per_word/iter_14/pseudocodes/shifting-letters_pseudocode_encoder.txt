CLASS Solution  
    FUNCTION shiftingLetters WITH PARAMETERS s AND shifts  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index OF shifts  
            SET element at position index OF shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character_code TO the character code OF element at position index OF s  
            SET base_code TO the character code OF the letter a  
            SET shifted_value TO character_code MINUS base_code PLUS element at position index OF shifts  
            SET wrapped_value TO shifted_value MODULO twenty six  
            SET new_character_code TO wrapped_value PLUS base_code  
            SET new_character TO the character REPRESENTED BY new_character_code  
            APPEND new_character TO result  
        END FOR  
        SET final_string TO the concatenation OF all elements IN result  
        RETURN final_string  
    END FUNCTION  
END CLASS
