CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO auxiliary FUNCTION initialize two-dimensional list with dimensions LENGTH OF s PLUS one BY LENGTH OF p PLUS one filled WITH boolean false values  
        SET value at position zero OF dp at position zero TO true  
        FOR j FROM one TO LENGTH OF p  
            IF element at position j MINUS one OF p IS EQUAL TO character star  
                SET value at position zero OF dp at position j TO value at position zero OF dp at position j MINUS two  
            END IF  
        END FOR  
        FOR i FROM one TO LENGTH OF s  
            FOR j FROM one TO LENGTH OF p  
                IF element at position j MINUS one OF p IS EQUAL TO element at position i MINUS one OF s OR element at position j MINUS one OF p IS EQUAL TO character dot  
                    SET value at position i OF dp at position j TO value at position i MINUS one OF dp at position j MINUS one  
                ELSE IF element at position j MINUS one OF p IS EQUAL TO character star  
                    SET value at position i OF dp at position j TO value at position i OF dp at position j MINUS two  
                    IF element at position j MINUS two OF p IS EQUAL TO element at position i MINUS one OF s OR element at position j MINUS two OF p IS EQUAL TO character dot  
                        SET value at position i OF dp at position j TO value at position i OF dp at position j OR value at position i MINUS one OF dp at position j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN value at position LENGTH OF s OF dp at position LENGTH OF p  
    END FUNCTION  
END CLASS
