CLASS Solution  
    FUNCTION canTransform WITH PARAMETERS start STRING end STRING  
        SET start_filtered TO the RESULT OF calling a helper FUNCTION that RETURNS a list of pairs of character and its index FOR each character AND index IN start WHERE character IS NOT EQUAL TO letter X  
        SET end_filtered TO the RESULT OF calling a helper FUNCTION that RETURNS a list of pairs of character and its index FOR each character AND index IN end WHERE character IS NOT EQUAL TO letter X  
        IF the LENGTH OF start_filtered IS NOT EQUAL TO the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair c1 AND i1 FROM start_filtered AND each pair c2 AND i2 FROM end_filtered SIMULTANEOUSLY  
            IF c1 IS NOT EQUAL TO c2  
                RETURN false  
            END IF  
            IF c1 IS EQUAL TO letter L AND i1 IS LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 IS EQUAL TO letter R AND i1 IS GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
