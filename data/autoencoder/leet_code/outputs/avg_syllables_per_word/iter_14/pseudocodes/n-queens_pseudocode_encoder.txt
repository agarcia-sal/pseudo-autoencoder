CLASS Solution  
    FUNCTION solveNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row col  
            RETURN NOT (element at position col OF cols PLUS element at position row MINUS col OF hills PLUS element at position row PLUS col OF dales)  
        END FUNCTION  
        
        FUNCTION place_queen WITH PARAMETERS row col  
            ADD the pair row col TO queens  
            SET element at position col OF cols TO one  
            SET element at position row MINUS col OF hills TO one  
            SET element at position row PLUS col OF dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen WITH PARAMETERS row col  
            REMOVE the pair row col FROM queens  
            SET element at position col OF cols TO zero  
            SET element at position row MINUS col OF hills TO zero  
            SET element at position row PLUS col OF dales TO zero  
        END FUNCTION  
        
        FUNCTION add_solution  
            SET solution TO an empty list  
            FOR each pair underscore col IN the pairs of queens SORTED BY row  
                SET row_string TO a string of period characters REPEATED col times  
                APPEND letter Q TO row_string  
                APPEND a string of period characters REPEATED n MINUS col MINUS one times TO row_string  
                APPEND row_string TO solution  
            END FOR  
            APPEND solution TO output  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETER row EQUAL TO zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH row col RETURNS TRUE  
                    CALL place_queen WITH row col  
                    IF row PLUS one EQUAL TO n  
                        CALL add_solution  
                    ELSE  
                        CALL backtrack WITH row PLUS one  
                    END IF  
                    CALL remove_queen WITH row col  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET cols TO a list of zero repeated n times  
        SET hills TO a list of zero repeated two MULTIPLIED BY n MINUS one times  
        SET dales TO a list of zero repeated two MULTIPLIED BY n MINUS one times  
        SET queens TO an empty set  
        SET output TO an empty list  
        CALL backtrack WITH default parameter  
        RETURN output  
    END FUNCTION  
END CLASS
