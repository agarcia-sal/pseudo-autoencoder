CLASS Solution  
    FUNCTION crackSafe WITH PARAMETERS n AND k  
        SET visited TO an empty set  
        SET sequence TO an empty list  
      
        FUNCTION dfs WITH PARAMETER node  
            FOR each i FROM zero TO k MINUS one  
                SET edge TO node CONCATENATED WITH string representation of i  
                IF edge NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs WITH argument AS substring OF edge FROM position one TO end  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET start_node TO string consisting of zero REPEATED n MINUS one TIMES  
        CALL dfs WITH argument start_node  
        SET reversed_sequence TO sequence REVERSED  
        SET result TO concatenation OF all elements OF reversed_sequence FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
