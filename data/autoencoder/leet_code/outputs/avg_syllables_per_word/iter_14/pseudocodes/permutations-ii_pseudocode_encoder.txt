CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path IS EQUAL TO the LENGTH OF nums THEN  
                APPEND a COPY OF path TO res  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF nums MINUS one  
                IF element at position index OF used IS TRUE OR  
                   index IS GREATER THAN zero AND element at position index OF nums IS EQUAL TO element at position index MINUS one OF nums AND element at position index MINUS one OF used IS FALSE THEN  
                    CONTINUE TO the NEXT iteration  
                END IF  
                SET element at position index OF used TO TRUE  
                APPEND element at position index OF nums TO path  
                CALL backtrack WITH path, used, res  
                REMOVE the LAST element FROM path  
                SET element at position index OF used TO FALSE  
            END FOR  
        END FUNCTION  
  
        SORT nums IN ASCENDING ORDER  
        SET res TO an empty list  
        SET used TO a list of FALSE values WITH length EQUAL TO the LENGTH OF nums  
        CALL backtrack WITH an empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
