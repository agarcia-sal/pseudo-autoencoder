CLASS Solution  
    FUNCTION expressiveWords WITH PARAMETERS s AND words  
        FUNCTION compress WITH PARAMETER word  
            IF word IS EQUAL TO empty  
                RETURN empty list  
            END IF  
            SET count TO one  
            SET compressed TO empty list  
            FOR index FROM one TO LENGTH OF word MINUS one  
                IF element at position index OF word IS EQUAL TO element at position index MINUS one OF word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND pair OF element at position index MINUS one OF word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND pair OF last element OF word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy WITH PARAMETERS s_compressed AND word_compressed  
            IF LENGTH OF s_compressed IS NOT EQUAL TO LENGTH OF word_compressed  
                RETURN False  
            END IF  
            FOR each pair sc AND sc_count AND each pair wc AND wc_count IN CORRESPONDING POSITIONS OF s_compressed AND word_compressed  
                IF sc IS NOT EQUAL TO wc  
                    RETURN False  
                END IF  
                IF sc_count IS NOT EQUAL TO wc_count AND (sc_count IS LESS THAN three OR sc_count IS LESS THAN wc_count)  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        SET s_compressed TO RESULT OF compress WITH PARAMETER s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO RESULT OF compress WITH PARAMETER word  
            IF RESULT OF is_stretchy WITH PARAMETERS s_compressed AND word_compressed IS True  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
