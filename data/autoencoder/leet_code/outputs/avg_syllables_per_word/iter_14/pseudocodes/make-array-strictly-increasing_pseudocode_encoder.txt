CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort arr2 IN ASCENDING ORDER  
        
        SET dp TO a DICTIONARY with a KEY of negative one MAPPED TO zero  
        
        FOR each num IN arr1  
            SET new_dp TO an EMPTY DICTIONARY  
            FOR each last_num AND ops IN ITEMS OF dp  
                IF num IS GREATER THAN last_num THEN  
                    IF num IS NOT A KEY IN new_dp OR the VALUE OF new_dp AT num IS GREATER THAN ops THEN  
                        SET the VALUE OF new_dp AT num TO ops  
                    END IF  
                END IF  
                
                SET idx TO the POSITION TO INSERT last_num INTO arr2 TO MAINTAIN ASCENDING ORDER USING BINARY SEARCH TO THE RIGHT  
                IF idx IS LESS THAN the LENGTH OF arr2 THEN  
                    SET replace_num TO the ELEMENT AT POSITION idx OF arr2  
                    IF replace_num IS NOT A KEY IN new_dp OR the VALUE OF new_dp AT replace_num IS GREATER THAN ops PLUS one THEN  
                        SET the VALUE OF new_dp AT replace_num TO ops PLUS one  
                    END IF  
                END IF  
            END FOR  
            
            IF the LENGTH OF new_dp IS EQUAL TO zero THEN  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM VALUE AMONG ALL VALUES IN dp  
    END FUNCTION  
END CLASS
