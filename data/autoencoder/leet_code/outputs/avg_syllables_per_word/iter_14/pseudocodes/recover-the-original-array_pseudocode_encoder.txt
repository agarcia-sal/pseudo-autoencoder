CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL nums SORT IN INCREASING ORDER  
        SET n TO the LENGTH OF nums  
        
        SET first TO element at position zero OF nums  
        
        FOR i FROM one TO n MINUS one  
            SET k TO element at position i OF nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUAL TO element at position i OF nums  
                CONTINUE TO next iteration of the loop  
            END IF  
            
            SET count TO a new COUNTER initialized with nums  
            SET arr TO a new empty list  
            
            FOR each x IN nums  
                IF the value ASSOCIATED WITH x IN count EQUAL TO zero  
                    CONTINUE TO next iteration of the loop  
                END IF  
                IF the value ASSOCIATED WITH x PLUS two MULTIPLIED BY k IN count EQUAL TO zero  
                    BREAK out of the loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the value ASSOCIATED WITH x IN count BY one  
                DECREMENT the value ASSOCIATED WITH x PLUS two MULTIPLIED BY k IN count BY one  
            END FOR  
            
            IF the LENGTH OF arr EQUAL TO n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        
        RETURN an empty list  
    END FUNCTION  
END CLASS
