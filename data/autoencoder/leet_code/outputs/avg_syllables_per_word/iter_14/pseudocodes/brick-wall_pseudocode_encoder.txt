CLASS Solution  
    FUNCTION leastBricks WITH PARAMETERS wall  
        SET gap_counts TO a new dictionary that returns zero for any new key  
        
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row EXCEPT the last element  
                INCREMENT total BY brick  
                INCREMENT the value at key total IN gap_counts BY one  
            END FOR  
        END FOR  
        
        SET max_gaps TO the maximum value among all values IN gap_counts OR zero if gap_counts is empty  
        
        SET number_of_rows TO the number of elements IN wall  
        SET minimum_crossed_bricks TO number_of_rows MINUS max_gaps  
        RETURN minimum_crossed_bricks  
    END FUNCTION  
END CLASS
