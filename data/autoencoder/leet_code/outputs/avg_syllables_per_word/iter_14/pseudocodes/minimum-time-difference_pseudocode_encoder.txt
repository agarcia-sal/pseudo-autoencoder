CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the integer value of the substring before the colon IN time  
            SET minutes TO the integer value of the substring after the colon IN time  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
  
        SET minutes_list TO a list formed by applying to_minutes TO each element IN timePoints  
  
        SORT minutes_list IN ascending order  
  
        SET min_diff TO positive infinity  
  
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET current_diff TO element at position index OF minutes_list MINUS element at position index MINUS one OF minutes_list  
            IF current_diff IS LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  
  
        SET circular_diff TO the quantity of one thousand four hundred forty MINUS element at the last position OF minutes_list PLUS element at position zero OF minutes_list  
        IF circular_diff IS LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  
  
        RETURN min_diff  
    END FUNCTION  
END CLASS
