CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET CONSTRUCTED FROM the ELEMENTS OF bank  
        SET q TO an EMPTY DOUBLE ENDED QUEUE  
        APPEND the TUPLE OF start AND zero TO q  
        SET mp TO a MAPPING FROM A TO THE STRING CONTAINING T C AND G  
                      AND FROM T TO THE STRING CONTAINING A C AND G  
                      AND FROM C TO THE STRING CONTAINING A T AND G  
                      AND FROM G TO THE STRING CONTAINING A T AND C  
        WHILE q IS NOT EMPTY  
            SET t AND step TO THE FIRST TUPLE REMOVED FROM q  
            IF t IS EQUAL TO end  
                RETURN step  
            END IF  
            FOR EACH position i AND CHARACTER v IN t  
                FOR EACH CHARACTER j IN GET THE MAPPING OF v FROM mp  
                    SET next TO THE STRING FORMED BY THE SUBSTRING OF t FROM THE START TO THE POSITION BEFORE i  
                                      CONCATENATED WITH j  
                                      CONCATENATED WITH THE SUBSTRING OF t FROM THE POSITION AFTER i TO THE END  
                    IF next IS IN s  
                        APPEND THE TUPLE OF next AND step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
