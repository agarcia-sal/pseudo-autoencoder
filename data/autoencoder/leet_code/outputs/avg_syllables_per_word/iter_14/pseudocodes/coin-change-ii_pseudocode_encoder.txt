CLASS Solution
    FUNCTION change(amount, coins)
        SET dp TO a list of zero repeated for the quantity of amount PLUS one
        SET element at position zero OF dp TO one
        
        FOR each coin IN coins
            FOR index FROM coin TO amount
                INCREMENT element at position index OF dp BY element at position index MINUS coin OF dp
            END FOR
        END FOR
        
        RETURN element at position amount OF dp
    END FUNCTION
END CLASS
