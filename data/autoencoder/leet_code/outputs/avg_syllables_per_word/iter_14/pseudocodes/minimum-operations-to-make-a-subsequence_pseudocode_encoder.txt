CLASS Solution  
    FUNCTION minOperations WITH PARAMETERS target AND arr  
        SET index_map TO A NEW EMPTY MAP  
        SET index_counter TO zero  
        FOR each element value IN target  
            ASSIGN value AS KEY AND index_counter AS VALUE IN index_map  
            INCREMENT index_counter BY one  
        END FOR  
        
        SET transformed_arr TO A NEW EMPTY LIST  
        FOR each element value IN arr  
            IF value IS A KEY IN index_map THEN  
                APPEND element at value OF index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO A NEW EMPTY LIST  
        FOR each number num IN transformed_arr  
            SET pos TO POSITION FOUND BY FINDING THE LEFTMOST POSITION TO INSERT num IN lis TO KEEP lis IN NON-DECREASING ORDER USING BINARY SEARCH  
            IF pos IS EQUAL TO THE LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos OF lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO THE LENGTH OF lis  
        
        SET result TO THE LENGTH OF target MINUS lcs_length  
        RETURN result  
    END FUNCTION  
END CLASS
