CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS A DIGIT  
                CONCATENATE char TO d  
            ELSE IF char IS AN ALPHABETIC CHARACTER  
                CONCATENATE char TO a  
            ELSE IF char IS EQUAL TO THE CHARACTER OPENING SQUARE BRACKET  
                APPEND the PAIR OF a AND THE INTEGER VALUE OF d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char IS EQUAL TO THE CHARACTER CLOSING SQUARE BRACKET  
                REMOVE the LAST ELEMENT FROM stack AND ASSIGN TO p AND n  
                SET a TO p CONCATENATED WITH a REPEATED n TIMES  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
