CLASS Solution  
    FUNCTION minimumTimeRequired WITH PARAMETERS jobs AND k  
        SET n TO the LENGTH OF jobs  
        
        SET subset_sums TO a LIST of LENGTH two RAISED TO THE POWER OF n WITH ALL ELEMENTS SET TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET new_mask TO the BITWISE OR of mask AND two RAISED TO THE POWER OF i  
                SET element AT POSITION new_mask OF subset_sums TO element AT POSITION mask OF subset_sums PLUS element AT POSITION i OF jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp WITH PARAMETERS k AND mask  
            IF k EQUAL TO one THEN  
                RETURN element AT POSITION mask OF subset_sums  
            END IF  
            
            SET res TO infinity  
            SET submask TO mask  
            
            WHILE submask IS GREATER THAN zero  
                SET temp1 TO dp WITH PARAMETERS k MINUS one AND the BITWISE XOR of mask AND submask  
                SET temp2 TO element AT POSITION submask OF subset_sums  
                SET candidate TO the GREATER OF temp1 AND temp2  
                IF candidate IS LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO the BITWISE AND of submask MINUS one AND mask  
            END WHILE  
            
            RETURN res  
        END FUNCTION  
        
        SET full_mask TO two RAISED TO THE POWER OF n MINUS one  
        RETURN dp WITH PARAMETERS k AND full_mask  
    END FUNCTION  
END CLASS
