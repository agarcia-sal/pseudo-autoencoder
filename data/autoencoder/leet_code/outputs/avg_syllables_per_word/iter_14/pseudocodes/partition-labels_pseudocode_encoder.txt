CLASS Solution  
    FUNCTION partitionLabels WITH PARAMETERS s  
        SET last_occurrence TO A NEW EMPTY MAPPING FROM CHARACTER TO INTEGER  
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element AT position index OF s  
            ASSIGN last_occurrence AT key character TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO A NEW EMPTY LIST  
        
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element AT position index OF s  
            IF end IS LESS THAN last_occurrence AT key character THEN  
                SET end TO last_occurrence AT key character  
            END IF  
            
            IF index IS EQUAL TO end THEN  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
