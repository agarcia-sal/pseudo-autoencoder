CLASS TreeNode  
    FUNCTION __init__(self, val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values IS EQUAL TO zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUAL TO the element at position zero OF values  
    SET i TO one  
    SET queue TO A NEW empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT EQUAL TO none  
            SET node.left TO A NEW TreeNode WITH val EQUAL TO the element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT EQUAL TO none  
            SET node.right TO A NEW TreeNode WITH val EQUAL TO the element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root IS none  
            RETURN none  
        END IF  
        IF root.val IS GREATER THAN key  
            SET root.left TO self.deleteNode(root.left, key)  
            RETURN root  
        END IF  
        IF root.val IS LESS THAN key  
            SET root.right TO self.deleteNode(root.right, key)  
            RETURN root  
        END IF  
        IF root.left IS none  
            RETURN root.right  
        END IF  
        IF root.right IS none  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left IS NOT none  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
