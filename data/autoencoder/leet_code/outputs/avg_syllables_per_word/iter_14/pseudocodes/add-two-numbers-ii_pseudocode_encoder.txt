CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF the LENGTH OF values IS EQUAL TO zero  
        RETURN none  
    END IF  
    SET head TO a new instance OF ListNode WITH PARAMETERS element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the last position  
        SET node TO a new instance OF ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value of p1.val IS EQUAL TO the value of p2.val AND is_same_list WITH PARAMETERS p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
    
        FUNCTION pushIntoStack WITH PARAMETERS node  
            SET stack TO an empty list  
            WHILE node IS NOT none  
                APPEND the value of node.val TO stack  
                SET node TO node.next  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO pushIntoStack WITH PARAMETERS l1  
        SET stack2 TO pushIntoStack WITH PARAMETERS l2  
        
        SET carry TO zero  
        SET result TO none  
        
        WHILE the LENGTH OF stack1 IS GREATER THAN zero OR the LENGTH OF stack2 IS GREATER THAN zero OR carry IS NOT EQUAL TO zero  
            IF the LENGTH OF stack1 IS GREATER THAN zero  
                SET val1 TO the element REMOVED FROM the END OF stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF the LENGTH OF stack2 IS GREATER THAN zero  
                SET val2 TO the element REMOVED FROM the END OF stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten USING INTEGER DIVISION  
            SET digit TO total MODULO ten  
            
            SET new_node TO a new instance OF ListNode WITH PARAMETERS digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
