CLASS Solution  
    FUNCTION countVowelPermutation WITH PARAMETER n  
        SET MODULO TO one billion PLUS seven  
        
        SET variable a TO one  
        SET variable e TO one  
        SET variable i TO one  
        SET variable o TO one  
        SET variable u TO one  
        
        FOR each number FROM two TO n  
            SET variable a_next TO variable e PLUS variable i PLUS variable u  
            SET variable e_next TO variable a PLUS variable i  
            SET variable i_next TO variable e PLUS variable o  
            SET variable o_next TO variable i  
            SET variable u_next TO variable i PLUS variable o  
            
            SET variable a TO variable a_next MODULO MODULO  
            SET variable e TO variable e_next MODULO MODULO  
            SET variable i TO variable i_next MODULO MODULO  
            SET variable o TO variable o_next MODULO MODULO  
            SET variable u TO variable u_next MODULO MODULO  
        END FOR  
        
        SET variable result TO variable a PLUS variable e PLUS variable i PLUS variable o PLUS variable u  
        SET variable result TO variable result MODULO MODULO  
        RETURN variable result  
    END FUNCTION  
END CLASS
