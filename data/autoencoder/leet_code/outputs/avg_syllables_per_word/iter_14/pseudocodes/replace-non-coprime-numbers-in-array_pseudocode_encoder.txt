CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero  
                SET g TO greatest common divisor OF the element at position length of stack MINUS one OF stack AND num  
                IF g IS EQUAL TO one  
                    BREAK  
                END IF  
                SET num TO the element at position length of stack MINUS one OF stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE the element at position length of stack MINUS one FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
