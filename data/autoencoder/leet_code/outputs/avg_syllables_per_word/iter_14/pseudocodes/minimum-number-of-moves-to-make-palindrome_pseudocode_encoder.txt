CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s AS string  
        SET s TO list OF characters FROM s  
        SET n TO LENGTH OF s  
        SET moves TO zero  
        
        WHILE LENGTH OF s GREATER THAN one IS TRUE  
            SET found_match TO FALSE  
            
            FOR i FROM LENGTH OF s MINUS one DOWN TO one  
                IF element at position i OF s EQUAL TO element at position zero OF s THEN  
                    SET found_match TO TRUE  
                    
                    FOR j FROM i TO LENGTH OF s MINUS two  
                        SWAP element at position j OF s WITH element at position j PLUS one OF s  
                        INCREMENT moves BY one  
                    END FOR  
                    
                    REMOVE element at position zero FROM s  
                    REMOVE element at position LENGTH OF s MINUS one FROM s  
                    BREAK  
                END IF  
            END FOR  
            
            IF found_match EQUAL TO FALSE THEN  
                REMOVE element at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
