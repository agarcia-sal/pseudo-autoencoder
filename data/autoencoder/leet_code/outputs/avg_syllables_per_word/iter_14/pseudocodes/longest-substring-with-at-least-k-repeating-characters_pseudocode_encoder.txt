CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s AND k  
        IF the LENGTH OF s IS LESS THAN k  
            RETURN zero  
        END IF  
        
        SET count TO CALL to Counter WITH PARAMETER s TO COMPUTE frequency OF each character IN s  
        
        FOR each character IN count  
            IF the frequency OF character IN count IS LESS THAN k  
                RETURN the MAXIMUM OF CALL to longestSubstring WITH PARAMETERS each substring OBTAINED BY SPLITTING s BY character AND k  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
