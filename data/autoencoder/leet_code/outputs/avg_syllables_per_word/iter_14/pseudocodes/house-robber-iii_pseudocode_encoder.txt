CLASS TreeNode  
    FUNCTION __init__ TO take val EQUAL TO zero AND left EQUAL TO none AND right EQUAL TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node TO take values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH element at position zero OF values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE element from the beginning OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree TO take p AND q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree OF p.left AND q.left AND is_same_tree OF p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob TO take root AS Optional TreeNode  
        FUNCTION rob_from_node TO take node  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node OF node.left  
            SET right_robbed, right_not_robbed TO rob_from_node OF node.right  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO maximum OF left_robbed AND left_not_robbed PLUS maximum OF right_robbed AND right_not_robbed  
            RETURN rob_this_node AND not_rob_this_node  
        END FUNCTION  
        RETURN maximum OF rob_from_node OF root  
    END FUNCTION  
END CLASS
