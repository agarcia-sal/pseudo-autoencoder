CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a new dictionary that provides zero for any missing key  
        
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET bitwise_and_result TO bitwise AND of i AND j  
                INCREMENT the value associated with bitwise_and_result IN pair_and_count BY one  
            END FOR  
        END FOR  
        
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key-value pair pair_and AND count IN pair_and_count  
                IF bitwise AND of pair_and AND k IS EQUAL TO zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        
        RETURN triplet_count  
    END FUNCTION  
END CLASS
