CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start IS EQUAL TO the LENGTH OF num  
                RETURN true  
            END IF  
            
            SET expected TO the string representation OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length IS GREATER THAN the LENGTH OF num OR the SUBSTRING FROM start TO start PLUS length OF num IS NOT EQUAL TO expected  
                RETURN false  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH start PLUS length, second, AND the integer conversion OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the SUBSTRING FROM zero TO i OF num  
                SET second TO the SUBSTRING FROM i TO j OF num  
                
                IF the LENGTH OF first IS GREATER THAN one AND the FIRST CHARACTER OF first IS EQUAL TO the CHARACTER zero OR the LENGTH OF second IS GREATER THAN one AND the FIRST CHARACTER OF second IS EQUAL TO the CHARACTER zero  
                    CONTINUE TO the next iteration  
                END IF  
                
                IF is_valid_sequence(j, the integer conversion OF first, the integer conversion OF second) RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
