CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val LEFT right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN NONE  
    END IF  
    SET root TO A NEW TreeNode WITH PARAMETER element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DEQUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM THE LEFT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT NONE  
            SET node left TO A NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT NONE  
            SET node right TO A NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS NONE AND q IS NONE  
        RETURN TRUE  
    ELSE IF p IS NONE OR q IS NONE  
        RETURN FALSE  
    ELSE IF p val IS NOT EQUAL TO q val  
        RETURN FALSE  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p left q left AND is_same_tree WITH PARAMETERS p right q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST WITH PARAMETERS self root val  
        FUNCTION dfs WITH PARAMETER root  
            IF root IS NONE  
                RETURN A NEW TreeNode WITH PARAMETER val  
            END IF  
            IF root val IS LESS THAN val  
                SET root right TO dfs WITH PARAMETER root right  
            ELSE  
                SET root left TO dfs WITH PARAMETER root left  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
