CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap IS empty OR the LENGTH OF element at position zero OF heightMap IS zero  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero OF heightMap  
        SET visited TO a two dimensional list with m rows and n columns FILLED WITH False values  
        SET min_heap TO an empty priority queue structure  
        
        FOR index i FROM zero TO m MINUS one  
            INSERT INTO min_heap a tuple CONTAINING element at position i AT zero OF heightMap AND i AND zero  
            INSERT INTO min_heap a tuple CONTAINING element at position i AT position n MINUS one OF heightMap AND i AND position n MINUS one  
            SET element at position i AT zero OF visited TO True  
            SET element at position i AT position n MINUS one OF visited TO True  
        END FOR  
        
        FOR index j FROM zero TO n MINUS one  
            INSERT INTO min_heap a tuple CONTAINING element at position zero AT position j OF heightMap AND zero AND j  
            INSERT INTO min_heap a tuple CONTAINING element at position m MINUS one AT position j OF heightMap AND position m MINUS one AND j  
            SET element at position zero AT position j OF visited TO True  
            SET element at position m MINUS one AT position j OF visited TO True  
        END FOR  
        
        SET directions TO a list CONTAINING tuples OF values: negative one AND zero, one AND zero, zero AND negative one, zero AND one  
        SET water_trapped TO zero  
        
        WHILE min_heap IS NOT empty  
            EXTRACT the tuple WITH the smallest first element FROM min_heap AND ASSIGN its values TO height AND x AND y  
            
            FOR each pair dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx AT position ny OF visited IS False  
                    SET element at position nx AT position ny OF visited TO True  
                    IF height IS GREATER THAN element at position nx AT position ny OF heightMap  
                        INCREMENT water_trapped BY height MINUS element at position nx AT position ny OF heightMap  
                    ELSE  
                        INCREMENT water_trapped BY zero  
                    END IF  
                    INSERT INTO min_heap a tuple CONTAINING the GREATER VALUE BETWEEN height AND element at position nx AT position ny OF heightMap AND nx AND ny  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
