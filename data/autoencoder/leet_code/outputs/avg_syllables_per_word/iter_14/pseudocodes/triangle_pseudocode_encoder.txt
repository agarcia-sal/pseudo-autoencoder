CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two DOWN TO zero  
            FOR col FROM zero TO the LENGTH OF element at position row OF triangle MINUS one  
                SET element at position col OF element at position row OF triangle TO element at position col OF element at position row OF triangle PLUS THE MINIMUM BETWEEN element at position col OF element at position row PLUS one OF triangle AND element at position col PLUS one OF element at position row PLUS one OF triangle  
            END FOR  
        END FOR  
        RETURN element at position zero OF element at position zero OF triangle  
    END FUNCTION  
END CLASS
