CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters obtained by replacing the colon character in time with an empty string  
        SET hour AND minute TO the integer values obtained by splitting time by the colon character and converting each part to integer  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET hour_value TO the integer value obtained by concatenating character h1 followed by character h2  
                            SET minute_value TO the integer value obtained by concatenating character m1 followed by character m2  
                            IF hour_value IS LESS THAN twenty four AND minute_value IS LESS THAN sixty  
                                APPEND the tuple consisting of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of calling generate_times with digits as argument  
        SET current_time_index TO the position of the tuple consisting of hour and minute in all_times  

        IF current_time_index PLUS one IS LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the elements at position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the elements at position zero in all_times  
        END IF  

        RETURN the string obtained by formatting next_hour as two digit number followed by colon character followed by next_minute as two digit number  
    END FUNCTION  
END CLASS
