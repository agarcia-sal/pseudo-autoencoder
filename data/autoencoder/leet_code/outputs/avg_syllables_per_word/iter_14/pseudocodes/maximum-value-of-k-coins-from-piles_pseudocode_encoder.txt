CLASS Solution  
    FUNCTION maxValueOfCoins WITH PARAMETERS piles AND k  
        SET n TO the LENGTH OF piles  
        SET dp TO a list WITH n PLUS one ELEMENTS WHERE EACH ELEMENT IS a list WITH k PLUS one ELEMENTS ALL SET TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one OF piles  
                SET current_value TO zero  
                SET limit TO the SMALLER OF j AND the LENGTH OF current_pile  
                
                FOR l FROM zero TO limit  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one OF current_pile  
                    END IF  
                    SET dp at position i THEN j TO the GREATER OF dp at position i THEN j AND dp at position i MINUS one THEN j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n THEN k  
    END FUNCTION  
END CLASS
