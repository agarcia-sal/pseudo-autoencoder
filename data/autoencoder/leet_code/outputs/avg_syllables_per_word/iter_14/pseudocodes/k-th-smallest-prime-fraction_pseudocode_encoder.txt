    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min heap TO an EMPTY list  
        
        FOR index i FROM zero TO n MINUS one MINUS one  
            SET fraction TO element at position i OF arr DIVIDED BY element at position n MINUS one OF arr  
            ADD the TUPLE of fraction AND i AND n MINUS one TO min heap USING a MINIMUM priority procedure  
        END FOR  
        
        FOR counter FROM zero TO k MINUS one MINUS one  
            REMOVE the SMALLEST element FROM min heap AND ASSIGN the FRACTION PART TO an IGNORE variable underscore  
            ASSIGN the SECOND element TO i  
            ASSIGN the THIRD element TO j  
            IF j MINUS one IS GREATER THAN i THEN  
                SET new fraction TO element at position i OF arr DIVIDED BY element at position j MINUS one OF arr  
                ADD the TUPLE of new fraction AND i AND j MINUS one TO min heap USING a MINIMUM priority procedure  
            END IF  
        END FOR  
        
        REMOVE the SMALLEST element FROM min heap AND ASSIGN fraction TO an IGNORE variable underscore  
        ASSIGN the SECOND element TO i  
        ASSIGN the THIRD element TO j  
        RETURN a LIST containing element at position i OF arr AND element at position j OF arr  
    END FUNCTION  
END CLASS
