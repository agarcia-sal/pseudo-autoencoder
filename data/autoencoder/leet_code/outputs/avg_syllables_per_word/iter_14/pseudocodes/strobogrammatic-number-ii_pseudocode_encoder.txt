CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUAL TO zero  
                RETURN list CONTAINING empty string  
            END IF  
            IF u EQUAL TO one  
                RETURN list CONTAINING string zero string one string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN dfs WITH PARAMETER u MINUS two  
                FOR each pair IN list OF string one string one AND string eight string eight AND string six string nine AND string nine string six  
                    SET l TO first character OF pair  
                    SET r TO second character OF pair  
                    APPEND concatenation OF l THEN v THEN r TO ans  
                END FOR  
                IF u NOT EQUAL TO n  
                    APPEND concatenation OF string zero THEN v THEN string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
