CLASS Solution  
    FUNCTION jump WITH PARAMETER nums  
        IF the LENGTH OF nums IS LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET jumps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR index i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            SET farthest TO THE GREATER OF farthest AND i PLUS the ELEMENT AT POSITION i OF nums  
            
            IF i IS EQUAL TO current_end  
                INCREMENT jumps BY one  
                SET current_end TO farthest
                
                IF current_end IS GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
                    BREAK OUT OF THE LOOP  
                END IF  
            END IF  
        END FOR  
        
        RETURN jumps  
    END FUNCTION  
END CLASS
