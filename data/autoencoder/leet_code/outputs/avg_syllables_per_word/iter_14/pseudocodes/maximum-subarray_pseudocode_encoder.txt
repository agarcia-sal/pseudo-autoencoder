CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero OF nums  
        SET current_sum TO element at position zero OF nums  
        FOR each num IN elements of nums FROM position one TO the end  
            SET current_sum TO the greater of num AND current_sum PLUS num  
            SET max_sum TO the greater of max_sum AND current_sum  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
