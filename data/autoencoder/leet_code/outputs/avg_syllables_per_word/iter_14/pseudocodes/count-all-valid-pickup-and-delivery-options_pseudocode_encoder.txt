CLASS Solution  
    FUNCTION countOrders(n)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        SET count TO one  
        FOR i FROM two TO n  
            SET count TO count MULTIPLIED BY i MULTIPLIED BY (two MULTIPLIED BY i MINUS one) COMPUTED MODULO MOD  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
