CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with LENGTH equal TO two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the NEGATIVE OF the element at position i OF nums INTO min_heap USING heap push operation  
            IF the LENGTH OF min_heap IS GREATER THAN n  
                POP the smallest element FROM min_heap USING heap pop operation  
            END IF  
            IF the LENGTH OF min_heap IS EQUAL TO n  
                SET the element at position i OF min_sums TO the NEGATIVE OF the SUM OF all elements IN min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with LENGTH equal TO two MULTIPLIED BY n PLUS one  
        
        FOR i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH the element at position i OF nums INTO max_heap USING heap push operation  
            IF the LENGTH OF max_heap IS GREATER THAN n  
                POP the smallest element FROM max_heap USING heap pop operation  
            END IF  
            IF the LENGTH OF max_heap IS EQUAL TO n  
                SET the element at position i OF max_sums TO the SUM OF all elements IN max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            IF the element at position i OF min_sums MINUS the element at position i PLUS one OF max_sums IS LESS THAN min_diff  
                SET min_diff TO the element at position i OF min_sums MINUS the element at position i PLUS one OF max_sums  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
