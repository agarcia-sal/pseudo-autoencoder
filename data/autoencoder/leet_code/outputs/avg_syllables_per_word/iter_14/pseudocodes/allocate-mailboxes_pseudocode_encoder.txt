CLASS Solution  
    FUNCTION minDistance WITH PARAMETERS houses AND k  
        SORT houses IN ASCENDING ORDER  
        SET n TO the LENGTH OF houses  
        
        FUNCTION initialize_cost_matrix WITH PARAMETERS size  
            CREATE a two-dimensional list WITH size BY size FILLED WITH zero  
            RETURN this two-dimensional list  
        END FUNCTION
        
        SET cost TO initialize_cost_matrix WITH PARAMETER n  
        
        FOR i FROM n MINUS one DOWNTO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost at position i AND j TO cost at position i PLUS one AND j MINUS one PLUS element at position j OF houses MINUS element at position i OF houses  
            END FOR  
        END FOR  
        
        FUNCTION initialize_dp_matrix WITH PARAMETERS rows AND columns  
            CREATE a two-dimensional list WITH rows BY columns FILLED WITH infinity  
            RETURN this two-dimensional list  
        END FUNCTION
        
        SET dp TO initialize_dp_matrix WITH PARAMETERS n PLUS one AND k PLUS one  
        
        SET dp at position zero AND zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET current_value TO dp at position p AND j MINUS one PLUS cost at position p AND i MINUS one  
                    IF current_value IS LESS THAN dp at position i AND j  
                        SET dp at position i AND j TO current_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n AND k  
    END FUNCTION  
END CLASS
