CLASS Solution  
    FUNCTION kSimilarity WITH PARAMETERS s1 AND s2  
        FUNCTION neighbors WITH PARAMETER s  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF s AND the element AT position i OF s IS EQUAL TO the element AT position i OF s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element AT position j OF s IS EQUAL TO the element AT position i OF s2 AND the element AT position j OF s IS NOT EQUAL TO the element AT position j OF s2  
                    YIELD concatenation OF substring OF s FROM zero TO i MINUS one PLUS element AT position j OF s PLUS substring OF s FROM i PLUS one TO j MINUS one PLUS element AT position i OF s PLUS substring OF s FROM j PLUS one TO end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a new double-ended queue CONTAINING a tuple OF s1 AND zero  
        SET visited TO a new set CONTAINING s1  
        
        WHILE queue IS NOT EMPTY  
            SET s AND swaps TO the first element REMOVED FROM queue  
            IF s IS EQUAL TO s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors WITH PARAMETER s  
                IF neighbor IS NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple OF neighbor AND swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
