CLASS Solution  
    FUNCTION judgePoint24 WITH PARAMETER cards  
        FUNCTION solve WITH PARAMETER nums  
            IF the LENGTH OF nums IS EQUAL TO one  
                IF the ABSOLUTE VALUE OF the element at position zero OF nums MINUS twenty four IS LESS THAN zero point zero zero zero zero zero one  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            END IF  
            
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i IS NOT EQUAL TO j  
                        SET new_nums TO an empty list  
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k IS NOT EQUAL TO i AND k IS NOT EQUAL TO j  
                                APPEND the element at position k OF nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each op IN the list of operations true division multiplication addition subtraction  
                            IF (op IS EQUAL TO addition OR op IS EQUAL TO multiplication) AND j IS GREATER THAN i  
                                CONTINUE to the next iteration of the loop  
                            END IF  
                            
                            IF op IS NOT EQUAL TO true division OR the element at position j OF nums IS NOT EQUAL TO zero  
                                APPEND the result of applying operation op TO the element at position i OF nums AND the element at position j OF nums TO new_nums  
                                IF solve WITH PARAMETER new_nums RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        IMPORT true division, multiplication, addition, subtraction FROM operator  
        RETURN solve WITH PARAMETER cards  
    END FUNCTION  
END CLASS
