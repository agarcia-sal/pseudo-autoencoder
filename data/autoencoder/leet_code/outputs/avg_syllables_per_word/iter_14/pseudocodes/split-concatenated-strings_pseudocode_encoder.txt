CLASS Solution  
    FUNCTION splitLoopedString WITH PARAMETER strs AS List OF String  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the REVERSE OF element at position index OF strs  
            IF reversed_str GREATER THAN element at position index OF strs  
                SET element at position index OF strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the CONCATENATION OF strs WITH strs  
        SET max_string TO the empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the collection of False and True  
                IF is_reversed IS TRUE  
                    SET current TO the REVERSE OF element at position index OF strs  
                ELSE  
                    SET current TO element at position index OF strs  
                END IF  
                
                SET remaining TO the CONCATENATION OF elements FROM position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one OF doubled_strs  
                
                FOR j FROM zero TO the LENGTH OF current  
                    SET candidate TO the CONCATENATION OF the substring FROM position j TO THE END OF current WITH remaining WITH the substring FROM position zero TO position j MINUS one OF current  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
