CLASS Solution  
    FUNCTION numDistinct with parameters s and t  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
          
        CREATE dp AS a two-dimensional array of size m PLUS one BY n PLUS one FILLED WITH zero  
          
        FOR i FROM zero TO m  
            SET element at position zero OF row at position i OF dp TO one  
        END FOR  
          
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF s EQUAL TO element at position j MINUS one OF t THEN  
                    SET element at position j OF row at position i OF dp TO the SUM OF element at position j MINUS one OF row at position i MINUS one OF dp AND element at position j OF row at position i MINUS one OF dp  
                ELSE  
                    SET element at position j OF row at position i OF dp TO element at position j OF row at position i MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
          
        RETURN element at position n OF row at position m OF dp  
    END FUNCTION  
END CLASS
