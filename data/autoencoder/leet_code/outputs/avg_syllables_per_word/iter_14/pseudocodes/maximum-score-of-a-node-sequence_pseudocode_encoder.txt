CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO a new mapping from element TO list of pairs of score AND index  
        FOR each pair u, v IN edges  
            APPEND a pair consisting of the element at position v OF scores AND v TO the list AT key u OF graph  
            APPEND a pair consisting of the element at position u OF scores AND u TO the list AT key v OF graph  
        END FOR  
        
        FOR each node IN the keys OF graph  
            SET the list AT key node OF graph TO the three pairs WITH the highest first element FROM the list AT key node OF graph  
        END FOR  
        
        SET max_score TO negative one  
        
        FOR each pair u, v IN edges  
            FOR each pair score1, x IN the list AT key u OF graph  
                FOR each pair score2, y IN the list AT key v OF graph  
                    IF the count OF unique elements IN the set consisting OF x, y, u, v IS EQUAL TO four  
                        IF max_score IS LESS THAN score1 PLUS score2 PLUS the element at position u OF scores PLUS the element at position v OF scores  
                            SET max_score TO score1 PLUS score2 PLUS the element at position u OF scores PLUS the element at position v OF scores  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
