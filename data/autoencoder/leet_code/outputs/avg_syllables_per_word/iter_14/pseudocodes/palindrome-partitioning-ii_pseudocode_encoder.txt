CLASS Solution  
    FUNCTION minCut WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        CALL create_palindrome_table WITH PARAMETERS s AND n AND ASSIGN RESULT TO is_palindrome
        
        CALL create_min_cuts_table WITH PARAMETERS is_palindrome AND n AND ASSIGN RESULT TO min_cuts
        
        RETURN element at position n MINUS one OF min_cuts  
    END FUNCTION
    
    FUNCTION create_palindrome_table WITH PARAMETERS s AND n  
        CREATE two DIMENSIONAL list is_palindrome OF SIZE n BY n FILLED WITH False  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AT row i OF is_palindrome TO True  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUAL TO two  
                    SET element at position start AT row end OF is_palindrome TO TRUE IF element at position start OF s EQUAL TO element at position end OF s OTHERWISE False  
                ELSE  
                    SET element at position start AT row end OF is_palindrome TO TRUE IF element at position start OF s EQUAL TO element at position end OF s AND element at position start PLUS one AT row end MINUS one OF is_palindrome IS True OTHERWISE False  
                END IF  
            END FOR  
        END FOR  
        
        RETURN is_palindrome  
    END FUNCTION
    
    FUNCTION create_min_cuts_table WITH PARAMETERS is_palindrome AND n  
        CREATE list min_cuts OF SIZE n FILLED WITH zero  
        
        FOR i FROM zero TO n MINUS one  
            IF element at position zero AT row i OF is_palindrome IS True  
                SET element at position i OF min_cuts TO zero  
            ELSE  
                SET element at position i OF min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF element at position j AT row i OF is_palindrome IS True AND element at position j MINUS one OF min_cuts PLUS one IS LESS THAN element at position i OF min_cuts  
                        SET element at position i OF min_cuts TO element at position j MINUS one OF min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN min_cuts  
    END FUNCTION  
END CLASS
