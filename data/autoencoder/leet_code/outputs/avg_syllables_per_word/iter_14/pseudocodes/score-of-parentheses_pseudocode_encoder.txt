CLASS Solution  
    FUNCTION scoreOfParentheses WITH PARAMETER s  
        SET stack TO new empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character IS EQUAL TO the character representing an opening parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO the element removed FROM the end OF stack PLUS  
                    IF score IS EQUAL TO zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
