CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple composed of element at position zero zero OF grid, zero, zero  
        ADD tuple composed of zero and zero TO visited  
        SET directions TO a list containing tuples composed of zero and one, one and zero, zero and negative one, negative one and zero  
        
        WHILE min_heap IS NOT empty  
            REMOVE and RETURN the smallest element FROM min_heap INTO variables t, i, j  
            IF i EQUAL TO n MINUS one AND j EQUAL TO n MINUS one  
                RETURN t  
            END IF  
            
            FOR each tuple composed of di and dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND tuple composed of ni and nj NOT IN visited  
                    ADD tuple composed of ni and nj TO visited  
                    INSERT tuple composed of the MAXIMUM OF t AND element at position ni nj OF grid, ni, nj INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
