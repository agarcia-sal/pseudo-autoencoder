CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid IS EQUAL TO nothing OR element at position zero OF grid IS EQUAL TO nothing  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        SET max_kills TO zero  
        
        SET row_kills TO zero  
        SET col_kills TO a list of zeros of LENGTH n  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF j IS EQUAL TO zero OR element at position j MINUS one OF element at position i OF grid IS EQUAL TO wall  
                    SET row_kills TO zero  
                    FOR k FROM j TO n MINUS one  
                        IF element at position k OF element at position i OF grid IS EQUAL TO enemy  
                            INCREMENT row_kills BY one  
                        ELSE IF element at position k OF element at position i OF grid IS EQUAL TO wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF i IS EQUAL TO zero OR element at position i MINUS one OF element at position j OF grid IS EQUAL TO wall  
                    SET element at position j OF col_kills TO zero  
                    FOR k FROM i TO m MINUS one  
                        IF element at position k OF element at position j OF grid IS EQUAL TO enemy  
                            INCREMENT element at position j OF col_kills BY one  
                        ELSE IF element at position k OF element at position j OF grid IS EQUAL TO wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF element at position j OF element at position i OF grid IS EQUAL TO empty  
                    SET max_kills TO the GREATER OF max_kills AND row_kills PLUS element at position j OF col_kills  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
