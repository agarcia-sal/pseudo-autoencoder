CLASS Solution  
    FUNCTION generateAbbreviations WITH PARAMETER word  
        FUNCTION backtrack WITH PARAMETERS index path count  
            IF index IS EQUAL TO the LENGTH OF word  
                IF count IS GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all elements IN path TO result  
            ELSE  
                IF count IS GREATER THAN zero  
                    CALL backtrack WITH index INCREMENTED BY one  
                        path CONCATENATED WITH the STRING REPRESENTATION OF count AND the ELEMENT AT POSITION index OF word  
                        AND count SET TO zero  
                ELSE  
                    CALL backtrack WITH index INCREMENTED BY one  
                        path CONCATENATED WITH the ELEMENT AT POSITION index OF word  
                        AND count SET TO zero  
                END IF  
                CALL backtrack WITH index INCREMENTED BY one  
                    path UNCHANGED  
                    AND count INCREMENTED BY one  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH zero AND an empty list AND zero  
        RETURN result  
    END FUNCTION  
END CLASS
