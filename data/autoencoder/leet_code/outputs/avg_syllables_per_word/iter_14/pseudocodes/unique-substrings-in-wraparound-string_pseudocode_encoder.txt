CLASS Solution  
    FUNCTION findSubstringInWraproundString WITH PARAMETER s  
        SET max_length TO a DICTIONARY MAPPING each character IN the sequence FROM character a TO character z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND ((the Unicode code point OF character AT position index OF s MINUS the Unicode code point OF character AT position index MINUS one OF s) EQUAL TO one OR (character AT position index OF s EQUAL TO character a AND character AT position index MINUS one OF s EQUAL TO character z)) THEN  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length AT character AT position index OF s TO the GREATER VALUE BETWEEN max_length AT character AT position index OF s AND current_length  
        END FOR  
        SET total_unique_substrings TO the SUM OF all values IN max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
