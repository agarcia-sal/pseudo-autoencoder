CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the Unicode code point of character c MINUS the Unicode code point of character a PLUS one  
        END FUNCTION  
        
        SET n TO the LENGTH OF string s  
        SET current_hash TO zero  
        SET p_pow TO one  
        
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO current_hash PLUS val of element at position i OF string s MULTIPLIED BY p_pow  
            SET current_hash TO current_hash MODULO modulo  
            IF i IS LESS THAN n MINUS one  
                SET p_pow TO p_pow MULTIPLIED BY power  
                SET p_pow TO p_pow MODULO modulo  
            END IF  
        END FOR  
        
        SET start_index TO n MINUS k  
        IF current_hash IS EQUAL TO hashValue  
            RETURN substring FROM position start_index TO start_index PLUS k OF string s  
        END IF  
        
        SET p_pow_k_minus_1 TO p_pow  
        
        FOR i FROM n MINUS k MINUS one DOWN TO zero  
            SET current_hash TO current_hash MINUS val of element at position i PLUS k OF string s MULTIPLIED BY p_pow_k_minus_1  
            SET current_hash TO current_hash MODULO modulo  
            SET current_hash TO current_hash MULTIPLIED BY power  
            SET current_hash TO current_hash PLUS val of element at position i OF string s  
            SET current_hash TO current_hash MODULO modulo  
            
            IF current_hash IS EQUAL TO hashValue  
                SET start_index TO i  
            END IF  
        END FOR  
        
        RETURN substring FROM position start_index TO start_index PLUS k OF string s  
    END FUNCTION  
END CLASS
