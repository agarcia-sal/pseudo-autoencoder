CLASS Solution  
    FUNCTION encode WITH PARAMETER s  
        FUNCTION get_encoded_length WITH PARAMETERS count AND substring  
            RETURN the LENGTH OF the STRING REPRESENTATION OF count PLUS the LENGTH OF substring PLUS two  
        END FUNCTION  
  
        FUNCTION dp WITH PARAMETERS i AND j  
            SET substring TO the SUBSTRING FROM position i TO j OF s  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five  
                RETURN substring  
            END IF  
  
            SET shortest TO substring  
            FOR k FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
                IF n MODULO k EQUAL TO zero  
                    SET repeat_count TO the INTEGER DIVISION OF n BY k  
                    IF substring EQUAL TO the SUBSTRING FROM position zero TO k OF substring REPEATED repeat_count TIMES  
                        SET encoded_candidate TO the STRING REPRESENTATION OF repeat_count CONCATENATED WITH an opening bracket CONCATENATED WITH dp WITH ARGUMENTS i AND i PLUS k MINUS one CONCATENATED WITH a closing bracket  
                        IF the LENGTH OF encoded_candidate LESS THAN the LENGTH OF shortest  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  
  
            FOR k FROM i TO j MINUS one  
                SET left_encoded TO dp WITH ARGUMENTS i AND k  
                SET right_encoded TO dp WITH ARGUMENTS k PLUS one AND j  
                SET combined TO left_encoded CONCATENATED WITH right_encoded  
                IF the LENGTH OF combined LESS THAN the LENGTH OF shortest  
                    SET shortest TO combined  
                END IF  
            END FOR  
  
            RETURN shortest  
        END FUNCTION  
  
        RETURN dp WITH ARGUMENTS zero AND the LENGTH OF s MINUS one  
END CLASS
