CLASS Solution  
    FUNCTION combinationSum2 WITH parameters candidates WITH target  
        FUNCTION backtrack WITH parameters start WITH target WITH path  
            IF target EQUAL TO zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i OF candidates EQUAL TO element at position i MINUS one OF candidates  
                    CONTINUE TO next iteration OF loop  
                END IF  
                CALL backtrack WITH parameters i PLUS one WITH target MINUS element at position i OF candidates WITH path CONCATENATED WITH element at position i OF candidates  
            END FOR  
        END FUNCTION  
        SORT candidates IN non-decreasing order  
        SET result TO empty list  
        CALL backtrack WITH parameters zero WITH target WITH empty list  
        RETURN result  
    END FUNCTION  
END CLASS
