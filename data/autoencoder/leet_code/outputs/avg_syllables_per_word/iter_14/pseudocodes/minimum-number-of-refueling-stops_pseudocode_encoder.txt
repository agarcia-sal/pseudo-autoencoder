CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a new list consisting of target and zero TO stations  
          
        FOR each element consisting of position and fuel IN stations  
            WHILE the LENGTH OF max_heap IS GREATER THAN zero AND current_fuel IS LESS THAN position  
                REMOVE the smallest element FROM max_heap and ASSIGN its negated value TO refuel_amount  
                INCREMENT current_fuel BY refuel_amount  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel IS LESS THAN position  
                RETURN minus one  
            END IF  
            INSERT the negation of fuel INTO max_heap maintaining heap property  
        END FOR  
          
        RETURN stops  
    END FUNCTION  
END CLASS
