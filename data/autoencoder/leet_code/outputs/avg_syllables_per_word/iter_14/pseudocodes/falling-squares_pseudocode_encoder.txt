CLASS Solution  
    FUNCTION fallingSquares WITH PARAMETER positions AS list of list of integer  
        SET max_heights TO an empty list  
        SET intervals TO an empty list  
        SET current_max_height TO zero  
        FOR each position IN positions  
            SET left TO element at position zero OF position  
            SET side_length TO element at position one OF position  
            SET right TO left PLUS side_length MINUS one  
            SET height TO side_length  
            FOR each interval IN intervals  
                SET interval_left TO element at position zero OF interval  
                SET interval_right TO element at position one OF interval  
                SET interval_height TO element at position two OF interval  
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left THEN  
                    IF height LESS THAN side_length PLUS interval_height THEN  
                        SET height TO side_length PLUS interval_height  
                    END IF  
                END IF  
            END FOR  
            APPEND a new list containing left, right, and height TO intervals  
            IF current_max_height LESS THAN height THEN  
                SET current_max_height TO height  
            END IF  
            APPEND current_max_height TO max_heights  
        END FOR  
        RETURN max_heights  
    END FUNCTION  
END CLASS
