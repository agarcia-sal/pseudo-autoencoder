CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element IN nums  
        SET num_set TO the unique elements OF nums AS a collection  
        SET count TO zero  
        
        FOR each g FROM one TO max_num  
            SET gcd_val TO zero  
            FOR each m FROM g TO max_num IN STEPS OF g  
                IF m IS AN ELEMENT OF num_set THEN  
                    SET gcd_val TO the greatest common divisor OF gcd_val AND m  
                    IF gcd_val IS EQUAL TO g THEN  
                        INCREMENT count BY one  
                        BREAK FROM the inner FOR loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
