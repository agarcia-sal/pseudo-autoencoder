CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET val TO val  
        SET next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH val AS THE ELEMENT AT POSITION zero OF values  
    SET p TO head  
    FOR each val IN SUBLIST OF values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode WITH val AS val  
        SET next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN BOOLEAN OF THE EXPRESSION val OF p1 EQUAL TO val OF p2 AND RESULT OF FUNCTION is_same_list WITH PARAMETERS next OF p1 AND next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETER head  
        IF head IS None OR next OF head IS None  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode WITH val EQUAL TO negative five thousand AND one  
        SET next OF dummy TO head  
        SET current TO next OF head  
        SET next OF head TO None  
        WHILE current IS NOT None  
            SET next_node TO next OF current  
            SET prev TO dummy  
            WHILE next OF prev IS NOT None AND val OF next OF prev LESS THAN val OF current  
                SET prev TO next OF prev  
            END WHILE  
            SET next OF current TO next OF prev  
            SET next OF prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN next OF dummy  
    END FUNCTION  
END CLASS
