CLASS Solution  
    FUNCTION gridIllumination WITH PARAMETERS n lamps queries  
        INITIALIZE row_count AS an empty dictionary that returns zero for unknown keys  
        INITIALIZE col_count AS an empty dictionary that returns zero for unknown keys  
        INITIALIZE diag_count AS an empty dictionary that returns zero for unknown keys  
        INITIALIZE anti_diag_count AS an empty dictionary that returns zero for unknown keys  
        
        INITIALIZE lamp_positions AS an empty set  
        
        FOR each pair r c IN lamps  
            IF the pair r c IS NOT IN lamp_positions  
                ADD the pair r c TO lamp_positions  
                INCREMENT the value at key r IN row_count BY one  
                INCREMENT the value at key c IN col_count BY one  
                INCREMENT the value at key r MINUS c IN diag_count BY one  
                INCREMENT the value at key r PLUS c IN anti_diag_count BY one  
            END IF  
        END FOR  
        
        INITIALIZE directions AS a list containing the pairs negative one negative one negative one zero negative one one zero negative one zero zero one one negative one one zero one one one  
        
        INITIALIZE result AS an empty list  
        
        FOR each pair r c IN queries  
            IF the value at key r IN row_count IS GREATER THAN zero OR the value at key c IN col_count IS GREATER THAN zero OR the value at key r MINUS c IN diag_count IS GREATER THAN zero OR the value at key r PLUS c IN anti_diag_count IS GREATER THAN zero  
                APPEND one TO result  
            ELSE  
                APPEND zero TO result  
            END IF  
            
            FOR each pair dr dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF the pair nr nc IS IN lamp_positions  
                    REMOVE the pair nr nc FROM lamp_positions  
                    DECREMENT the value at key nr IN row_count BY one  
                    DECREMENT the value at key nc IN col_count BY one  
                    DECREMENT the value at key nr MINUS nc IN diag_count BY one  
                    DECREMENT the value at key nr PLUS nc IN anti_diag_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
