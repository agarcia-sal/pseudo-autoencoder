CLASS Solution  
    FUNCTION integerReplacement WITH PARAMETER n  
        SET count TO zero  
        WHILE n IS NOT EQUAL TO one  
            IF n MODULO two IS EQUAL TO zero  
                SET n TO n DIVIDED BY two  
            ELSE  
                IF n IS EQUAL TO three  
                    SET n TO n MINUS one  
                ELSE IF (n PLUS one) MODULO four IS EQUAL TO zero  
                    SET n TO n PLUS one  
                ELSE  
                    SET n TO n MINUS one  
                END IF  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
