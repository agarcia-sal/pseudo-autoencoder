CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET dp TO a list containing zero, zero, zero  
        FOR each num IN nums  
            IF num IS EQUAL TO zero  
                SET dp element at position zero TO the value of two MULTIPLIED BY dp element at position zero PLUS one THEN MODULO MOD  
            ELSE IF num IS EQUAL TO one  
                SET dp element at position one TO the value of two MULTIPLIED BY dp element at position one PLUS dp element at position zero THEN MODULO MOD  
            ELSE IF num IS EQUAL TO two  
                SET dp element at position two TO the value of two MULTIPLIED BY dp element at position two PLUS dp element at position one THEN MODULO MOD  
            END IF  
        END FOR  
        RETURN dp element at position two  
    END FUNCTION  
END CLASS
