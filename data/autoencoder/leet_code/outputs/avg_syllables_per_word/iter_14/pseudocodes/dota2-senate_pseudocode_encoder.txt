CLASS Solution  
    FUNCTION predictPartyVictory WITH PARAMETER senate  
        FROM collections IMPORT deque  
        
        SET radiant TO an empty deque  
        SET dire TO an empty deque  
        
        FOR each index i AND character s IN senate  
            IF s EQUAL TO the character representing Radiant PARTY  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO the element REMOVED FROM the left side OF radiant  
            SET d_index TO the element REMOVED FROM the left side OF dire  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant IS NOT empty  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
