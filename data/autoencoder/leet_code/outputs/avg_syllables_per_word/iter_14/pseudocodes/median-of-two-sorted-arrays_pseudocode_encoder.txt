CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2  
            SET temporary_first TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary_first  
        END IF  
        
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low IS LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION OF the SUM of low AND high BY two  
            SET partitionY TO the INTEGER DIVISION OF the SUM of x AND y PLUS one BY two MINUS partitionX  
            
            IF partitionX IS EQUAL TO zero  
                SET maxLeftX TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftX TO the element at position partitionX MINUS one OF nums1  
            END IF  
            
            IF partitionX IS EQUAL TO x  
                SET minRightX TO POSITIVE_INFINITY  
            ELSE  
                SET minRightX TO the element at position partitionX OF nums1  
            END IF  
            
            IF partitionY IS EQUAL TO zero  
                SET maxLeftY TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftY TO the element at position partitionY MINUS one OF nums2  
            END IF  
            
            IF partitionY IS EQUAL TO y  
                SET minRightY TO POSITIVE_INFINITY  
            ELSE  
                SET minRightY TO the element at position partitionY OF nums2  
            END IF  
            
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX  
                IF the MODULO OF the SUM of x AND y BY two IS EQUAL TO zero  
                    SET maximum_left TO maxLeftX  
                    IF maxLeftY IS GREATER THAN maximum_left  
                        SET maximum_left TO maxLeftY  
                    END IF  
                    SET minimum_right TO minRightX  
                    IF minRightY IS LESS THAN minimum_right  
                        SET minimum_right TO minRightY  
                    END IF  
                    RETURN the SUM of maximum_left AND minimum_right DIVIDED BY two  
                ELSE  
                    IF maxLeftX IS GREATER THAN maxLeftY  
                        RETURN maxLeftX  
                    ELSE  
                        RETURN maxLeftY  
                    END IF  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
