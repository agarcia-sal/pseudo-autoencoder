CLASS Solution  
    FUNCTION boldWords(words AS list of strings, s AS string) RETURNS string  
        SET bold TO a list of boolean values all set to false with the LENGTH OF s  
        FOR each word IN words  
            SET start TO the POSITION OF word IN s  
            WHILE start IS NOT EQUAL TO negative one  
                FOR index FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position index OF bold TO true  
                END FOR  
                SET start TO the POSITION OF word IN s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF s  
            IF element at position i OF bold IS true  
                APPEND string consisting of character less than b greater than TO result  
                WHILE i IS LESS THAN the LENGTH OF s AND element at position i OF bold IS true  
                    APPEND element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND string consisting of character less than slash b greater than TO result  
            ELSE  
                APPEND element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements OF result  
    END FUNCTION  
END CLASS
