CLASS Solution  
    FUNCTION judgeSquareSum WITH PARAMETER c  
        SET left TO zero  
        SET right TO the INTEGER VALUE OF THE SQUARE ROOT OF c  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right  
            IF current_sum IS EQUAL TO c  
                RETURN TRUE  
            ELSE IF current_sum IS LESS THAN c  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN FALSE  
    END FUNCTION  
END CLASS
