CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            SET first_difference_x TO element at position zero OF a MINUS element at position zero OF o  
            SET second_difference_y TO element at position one OF b MINUS element at position one OF o  
            SET first_term TO first_difference_x MULTIPLIED BY second_difference_y  
            SET third_difference_y TO element at position one OF a MINUS element at position one OF o  
            SET fourth_difference_x TO element at position zero OF b MINUS element at position zero OF o  
            SET second_term TO third_difference_y MULTIPLIED BY fourth_difference_x  
            SET result TO first_term MINUS second_term  
            RETURN result  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET o TO element at position index OF points  
            SET a TO element at position index PLUS one MODULO n OF points  
            SET b TO element at position index PLUS two MODULO n OF points  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUAL TO zero  
                IF orientation EQUAL TO zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUAL TO one AND cross LESS THAN zero OR orientation EQUAL TO negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
