CLASS Solution  
    FUNCTION makeLargestSpecial WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER s  
            SET count TO zero  
            SET start TO zero  
            SET specials TO an empty list  
            
            FOR each index i ALONG WITH each character char IN the string s  
                IF char IS EQUAL TO the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count IS EQUAL TO zero  
                    APPEND the concatenation of the character one PLUS the result of calling dfs WITH the substring of s FROM the position start PLUS one TO the position i PLUS the character zero TO the list specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            
            SORT the list specials IN descending lexicographical order  
            SET result TO the concatenation of all elements IN specials  
            RETURN result  
        END FUNCTION  
        
        RETURN the result of calling dfs WITH s  
    END FUNCTION  
END CLASS
