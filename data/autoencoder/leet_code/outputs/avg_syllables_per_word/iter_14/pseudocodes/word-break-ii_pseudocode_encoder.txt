CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the SET CONTAINING all ELEMENTS OF wordDict  
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION backtrack WITH PARAMETER start  
            IF start IS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH start IN memo  
            END IF  
            IF start IS EQUAL TO the LENGTH OF s  
                RETURN a LIST CONTAINING an EMPTY STRING  
            END IF  
            
            SET res TO an EMPTY LIST  
            FOR end FROM start PLUS one TO the LENGTH OF s PLUS one  
                SET word TO the SUBSTRING OF s FROM POSITION start TO end MINUS one  
                IF word IS A MEMBER OF wordSet  
                    FOR each sentence IN the RESULT OF backtrack WITH ARGUMENT end  
                        IF sentence IS NOT AN EMPTY STRING  
                            APPEND the CONCATENATION OF word WITH A SPACE AND sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            ASSIGN TO memo AT KEY start THE LIST res  
            RETURN res  
        END FUNCTION  
        
        RETURN the RESULT OF backtrack WITH ARGUMENT zero  
    END FUNCTION  
END CLASS
