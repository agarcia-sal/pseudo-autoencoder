CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with val equal to element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left side OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node left TO a new TreeNode with val equal to element at position i OF values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node right TO a new TreeNode with val equal to element at position i OF values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p val IS NOT EQUAL TO q val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p left AND q left AND is_same_tree with arguments p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree WITH PARAMETERS self root  
        IF root IS None  
            RETURN zero  
        END IF  
        SET queue TO a new double ended queue containing a tuple of root and zero  
        SET max_width TO zero  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second element of the first tuple in queue  
            FOR each number FROM zero TO level_length MINUS one  
                SET a tuple of node and index TO the element removed from the left side OF queue  
                IF number IS EQUAL TO level_length MINUS one  
                    IF index MINUS first_index PLUS one IS GREATER THAN max_width  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node left IS NOT None  
                    APPEND a tuple of node left and index MULTIPLIED BY two TO queue  
                END IF  
                IF node right IS NOT None  
                    APPEND a tuple of node right and index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
