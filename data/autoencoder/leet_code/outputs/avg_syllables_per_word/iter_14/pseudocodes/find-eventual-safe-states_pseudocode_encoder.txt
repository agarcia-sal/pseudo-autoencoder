CLASS Solution  
    FUNCTION eventualSafeNodes(graph)  
        FUNCTION is_safe(node)  
            IF visited element at position node IS EQUAL TO one  
                RETURN False  
            END IF  
            IF visited element at position node IS EQUAL TO two  
                RETURN True  
            END IF  
            SET element at position node OF visited TO one  
            FOR each neighbor IN element at position node OF graph  
                IF is_safe(neighbor) IS EQUAL TO False  
                    RETURN False  
                END IF  
            END FOR  
            SET element at position node OF visited TO two  
            RETURN True  
        END FUNCTION  
        SET n TO the LENGTH OF graph  
        SET visited TO a list of length n with all elements SET TO zero  
        SET safe_nodes TO an empty list  
        FOR node FROM zero TO n MINUS one  
            IF is_safe(node) IS EQUAL TO True  
                APPEND node TO safe_nodes  
            END IF  
        END FOR  
        RETURN safe_nodes  
    END FUNCTION  
END CLASS
