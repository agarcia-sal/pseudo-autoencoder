CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO the list of integer values ONE thousand  
        APPEND nine hundred TO val  
        APPEND five hundred TO val  
        APPEND four hundred TO val  
        APPEND one hundred TO val  
        APPEND ninety TO val  
        APPEND fifty TO val  
        APPEND forty TO val  
        APPEND ten TO val  
        APPEND nine TO val  
        APPEND five TO val  
        APPEND four TO val  
        APPEND one TO val  
              
        SET syms TO the list of Roman numeral symbols LETTER M  
        APPEND sequence LETTER C LETTER M TO syms  
        APPEND letter D TO syms  
        APPEND sequence LETTER C LETTER D TO syms  
        APPEND letter C TO syms  
        APPEND sequence LETTER X LETTER C TO syms  
        APPEND letter L TO syms  
        APPEND sequence LETTER X LETTER L TO syms  
        APPEND letter X TO syms  
        APPEND sequence LETTER I LETTER X TO syms  
        APPEND letter V TO syms  
        APPEND sequence LETTER I LETTER V TO syms  
        APPEND letter I TO syms  
          
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num IS GREATER THAN zero  
            SET times TO num DIVIDED BY element at position i OF val  
            FOR count FROM one TO times  
                SET roman_numeral TO roman_numeral PLUS element at position i OF syms  
                SET num TO num MINUS element at position i OF val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
