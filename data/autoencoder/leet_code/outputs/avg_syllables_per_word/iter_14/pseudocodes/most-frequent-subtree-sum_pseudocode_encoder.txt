CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val EQUAL TO element at position zero OF values  
    SET i TO one  
    SET queue TO an empty deque structure  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH val EQUAL TO element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(element at attribute left OF p, element at attribute left OF q) AND is_same_tree(element at attribute right OF p, element at attribute right OF q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO element at attribute val OF node PLUS subtree_sum(element at attribute left OF node) PLUS subtree_sum(element at attribute right OF node)  
            INCREMENT count AT KEY current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
      
        SET count TO a mapping from integer keys TO integer values defaulting to zero  
        CALL subtree_sum WITH root AS argument  
      
        SET max_freq TO the maximum value AMONG all values IN count OR zero IF count IS empty  
      
        SET result TO a list OF all keys IN count WHERE the corresponding value IS EQUAL TO max_freq  
      
        RETURN result  
    END FUNCTION  
END CLASS
