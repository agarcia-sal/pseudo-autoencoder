CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO the LENGTH OF nums  
        SET delta TO a list of zeroes WITH the LENGTH OF n  
        
        FOR i FROM zero TO n MINUS one  
            SET num TO element at position i OF nums  
            IF num IS LESS THAN OR EQUAL TO i  
                INCREMENT element at position zero OF delta BY one  
                IF i MINUS num PLUS one IS LESS THAN n  
                    DECREMENT element at position i MINUS num PLUS one OF delta BY one  
                END IF  
            END IF  
            IF i PLUS one IS LESS THAN n  
                INCREMENT element at position i PLUS one OF delta BY one  
                IF i PLUS n MINUS num PLUS one IS LESS THAN n  
                    DECREMENT element at position i PLUS n MINUS num PLUS one OF delta BY one  
                END IF  
            END IF  
        END FOR  
        
        SET score TO zero  
        SET max_score TO zero  
        SET best_k TO zero  
        
        FOR k FROM zero TO n MINUS one  
            INCREMENT score BY element at position k OF delta  
            IF score IS GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
