CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  

        FOR each value IN preorder  
            IF value IS LESS THAN lower_bound  
                RETURN false  
            END IF  

            WHILE the LENGTH OF stack IS GREATER THAN zero AND value IS GREATER THAN the element at the last position OF stack  
                SET lower_bound TO the element removed FROM the last position OF stack  
            END WHILE  

            APPEND value TO stack  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
