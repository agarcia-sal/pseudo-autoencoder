CLASS Solution  
    FUNCTION checkSubarraySum WITH PARAMETERS nums AND k  
        SET remainder_map TO a dictionary mapping zero TO negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN enumeration of nums  
            INCREMENT current_sum BY num  
            IF k IS NOT EQUAL TO zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF remainder_map CONTAINS a key current_sum  
                IF i MINUS the element AT key current_sum OF remainder_map IS GREATER THAN one  
                    RETURN true  
                END IF  
            ELSE  
                ASSIGN the element AT key current_sum OF remainder_map TO i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
