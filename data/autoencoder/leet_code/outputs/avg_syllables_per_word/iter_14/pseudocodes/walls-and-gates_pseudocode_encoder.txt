CLASS Solution  
    FUNCTION wallsAndGates WITH PARAMETER rooms  
        IF rooms IS EMPTY OR element at position zero OF rooms IS EMPTY  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero OF rooms  
        ASSIGN reference TO a new empty double-ended queue NAMED queue  

        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r OF rooms AT position c IS EQUAL TO zero  
                    APPEND the tuple of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  

        SET directions TO a list containing four tuples representing zero AND one, one AND zero, zero AND negative one, negative one AND zero  

        WHILE queue IS NOT EMPTY  
            REMOVE and RETURN the element at the front OF queue AND ASSIGN its first value TO r AND its second value TO c  
            FOR each tuple dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr OF rooms AT position nc IS EQUAL TO the integer value two billion one hundred forty-seven million four hundred eighty-three thousand six hundred forty-seven  
                    SET element at position nr OF rooms AT position nc TO element at position r OF rooms AT position c PLUS one  
                    APPEND the tuple of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
