CLASS Solution  
    FUNCTION longestDupSubstring(s)  
        FUNCTION search(length)  
            SET seen TO empty set  
            FOR i FROM zero TO the LENGTH OF s MINUS length PLUS one MINUS one  
                SET substring TO substring FROM position i TO i PLUS length MINUS one OF s  
                IF substring IS IN seen  
                    RETURN substring  
                END IF  
                ADD substring TO seen  
            END FOR  
            RETURN empty string  
        END FUNCTION  

        SET left TO one  
        SET right TO the LENGTH OF s MINUS one  
        SET longest TO empty string  

        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS right DIVIDED BY two USING INTEGER DIVISION  
            SET candidate TO search(mid)  
            IF candidate IS NOT empty string  
                SET longest TO candidate  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  

        RETURN longest  
    END FUNCTION  
END CLASS
