CLASS Solution  
    FUNCTION shortestSubarray WITH PARAMETERS nums AND k  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            APPEND the element at the last position OF prefix_sum PLUS num TO prefix_sum  
        END FOR  
        CREATE dq AS a new empty double ended queue  
        SET min_length TO positive infinity  
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS element at the position at the front OF dq GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller of min_length AND i MINUS the element at the position at the front OF dq  
                REMOVE the element at the front OF dq  
            END WHILE  
            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO element at the position at the last OF dq in prefix_sum  
                REMOVE the element at the last OF dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
