CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid IS EMPTY OR element at position zero OF element at position zero OF obstacleGrid IS EQUAL TO one  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero OF obstacleGrid  
        
        SET dp TO a new two-dimensional list with m rows and n columns FILLED WITH zero  
        
        SET element at position zero OF element at position zero OF dp TO one  
        
        FOR j FROM one TO n MINUS one  
            IF element at position j OF element at position zero OF obstacleGrid IS EQUAL TO zero  
                SET element at position j OF element at position zero OF dp TO element at position j MINUS one OF element at position zero OF dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            IF element at position i OF element at position zero OF obstacleGrid IS EQUAL TO zero  
                SET element at position zero OF element at position i OF dp TO element at position zero OF element at position i MINUS one OF dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position j OF element at position i OF obstacleGrid IS EQUAL TO zero  
                    SET element at position j OF element at position i OF dp TO element at position j OF element at position i MINUS one OF dp PLUS element at position j MINUS one OF element at position i OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n MINUS one OF element at position m MINUS one OF dp  
    END FUNCTION  
END CLASS
