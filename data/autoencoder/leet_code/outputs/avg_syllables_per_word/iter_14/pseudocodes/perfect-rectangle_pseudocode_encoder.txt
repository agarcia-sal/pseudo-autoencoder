CLASS Solution  
    FUNCTION isRectangleCover WITH PARAMETER rectangles  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO empty set  
        SET area TO zero  
        
        FOR each rectangle IN rectangles  
            SET x1 TO element at position zero OF rectangle  
            SET y1 TO element at position one OF rectangle  
            SET x2 TO element at position two OF rectangle  
            SET y2 TO element at position three OF rectangle  
            
            IF X1 IS GREATER THAN x1  
                SET X1 TO x1  
            END IF  
            
            IF Y1 IS GREATER THAN y1  
                SET Y1 TO y1  
            END IF  
            
            IF X2 IS LESS THAN x2  
                SET X2 TO x2  
            END IF  
            
            IF Y2 IS LESS THAN y2  
                SET Y2 TO y2  
            END IF  
            
            SET width TO x2 MINUS x1  
            SET height TO y2 MINUS y1  
            INCREMENT area BY width MULTIPLIED BY height  
            
            SET corner_points TO list of tuples: element one as x1 and y1, element two as x1 and y2, element three as x2 and y1, element four as x2 and y2  
            
            FOR each point IN corner_points  
                IF point IS IN points  
                    REMOVE point FROM points  
                ELSE  
                    ADD point TO points  
                END IF  
            END FOR  
        END FOR  
        
        SET expected_area_width TO X2 MINUS X1  
        SET expected_area_height TO Y2 MINUS Y1  
        
        IF area IS NOT EQUAL TO expected_area_width MULTIPLIED BY expected_area_height  
            RETURN False  
        END IF  
        
        SET expected_corners TO set containing tuples X1 and Y1, X1 and Y2, X2 and Y1, X2 and Y2  
        
        IF points IS NOT EQUAL TO expected_corners  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
