CLASS Solution  
    FUNCTION mincostToHireWorkers WITH PARAMETERS quality, wage, k  
        SET workers TO SORTED LIST OF PAIRS WHERE EACH PAIR CONSISTS OF THE WAGE AT THE SAME POSITION DIVIDED BY QUALITY AT THE SAME POSITION AND THE QUALITY AT THE SAME POSITION FROM THE CORRESPONDING ELEMENTS OF wage AND quality  
        
        SET max_heap TO AN EMPTY LIST  
        SET total_quality TO zero  
        SET min_cost TO POSITIVE INFINITY  
        
        FOR each pair OF ratio AND q IN workers  
            PUSH the NEGATIVE OF q INTO max_heap USING HEAP PUSH OPERATION  
            INCREMENT total_quality BY q  
            
            IF the LENGTH OF max_heap IS GREATER THAN k  
                SET total_quality TO total_quality PLUS THE RESULT OF HEAP POP FROM max_heap  
            END IF  
            
            IF the LENGTH OF max_heap IS EQUAL TO k  
                SET min_cost TO THE SMALLER VALUE BETWEEN min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
