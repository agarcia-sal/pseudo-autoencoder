CLASS Solution  
    FUNCTION numWays WITH PARAMETERS n AND k  
        IF n IS EQUAL TO zero  
            RETURN zero  
        END IF  
        IF n IS EQUAL TO one  
            RETURN k  
        END IF  
        IF n IS EQUAL TO two  
            RETURN k MULTIPLIED BY k  
        END IF  
        
        SET same TO k  
        SET diff TO k MULTIPLIED BY the DIFFERENCE BETWEEN k AND one  
        SET total TO the SUM OF same AND diff  
        
        FOR i FROM three TO n  
            SET new_same TO diff  
            SET new_diff TO total MULTIPLIED BY the DIFFERENCE BETWEEN k AND one  
            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO the SUM OF same AND diff  
        END FOR  
        
        RETURN total  
    END FUNCTION  
END CLASS
