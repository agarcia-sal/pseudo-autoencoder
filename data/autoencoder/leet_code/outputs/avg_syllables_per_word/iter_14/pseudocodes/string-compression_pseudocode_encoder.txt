CLASS Solution  
    FUNCTION compress(chars)  
        IF chars IS EMPTY  
            RETURN zero  
        END IF  
        
        SET write_index TO zero  
        SET read_index TO zero  
        
        WHILE read_index IS LESS THAN the LENGTH OF chars  
            SET char TO the element at position read_index OF chars  
            SET count TO zero  
            
            WHILE read_index IS LESS THAN the LENGTH OF chars AND the element at position read_index OF chars IS EQUAL TO char  
                INCREMENT read_index BY one  
                INCREMENT count BY one  
            END WHILE  
            
            SET the element at position write_index OF chars TO char  
            INCREMENT write_index BY one  
            
            IF count IS GREATER THAN one  
                FOR each digit IN the string representation OF count  
                    SET the element at position write_index OF chars TO digit  
                    INCREMENT write_index BY one  
                END FOR  
            END IF  
        END WHILE  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
