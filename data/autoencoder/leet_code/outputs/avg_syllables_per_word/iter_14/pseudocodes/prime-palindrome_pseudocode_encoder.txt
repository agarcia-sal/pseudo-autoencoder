CLASS Solution  
    FUNCTION primePalindrome WITH PARAMETER n  
        FUNCTION is_prime WITH PARAMETER num  
            IF num LESS THAN two  
                RETURN False  
            END IF  
            IF num EQUAL TO two OR num EQUAL TO three  
                RETURN True  
            END IF  
            IF num MODULO two EQUAL TO zero OR num MODULO three EQUAL TO zero  
                RETURN False  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUAL TO zero OR num MODULO i PLUS two EQUAL TO zero  
                    RETURN False  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN True  
        END FUNCTION  
  
        FUNCTION is_palindrome WITH PARAMETER num  
            RETURN string representation of num EQUAL TO string representation of num REVERSED  
        END FUNCTION  
  
        IF n LESS THAN OR EQUAL TO two  
            RETURN two  
        END IF  
  
        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven  
            RETURN eleven  
        END IF  
  
        WHILE True  
            IF n MODULO two EQUAL TO zero  
                INCREMENT n BY one  
                CONTINUE TO next iteration of loop  
            END IF  
  
            IF is_palindrome WITH ARGUMENT n IS True AND is_prime WITH ARGUMENT n IS True  
                RETURN n  
            END IF  
  
            IF LENGTH OF string representation of n MODULO two EQUAL TO zero  
                SET n TO ten RAISED TO THE POWER OF LENGTH OF string representation of n PLUS one  
            ELSE  
                INCREMENT n BY two  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
