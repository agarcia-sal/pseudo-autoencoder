CLASS Solution  
    FUNCTION removeDuplicates WITH parameter nums  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
        
        SET write_index TO zero  
        
        FOR i FROM zero TO the LESSER OF two AND the LENGTH OF nums MINUS one  
            SET the element at position write_index OF nums TO the element at position i OF nums  
            INCREMENT write_index BY one  
        END FOR  
        
        FOR i FROM two TO the LENGTH OF nums MINUS one  
            IF the element at position i OF nums IS NOT EQUAL TO the element at position write_index MINUS two OF nums  
                SET the element at position write_index OF nums TO the element at position i OF nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
