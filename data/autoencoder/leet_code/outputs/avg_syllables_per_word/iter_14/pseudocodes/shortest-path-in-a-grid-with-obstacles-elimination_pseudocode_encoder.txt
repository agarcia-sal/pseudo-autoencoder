CLASS Solution  
    FUNCTION shortestPath(grid, k)  
        IF grid IS empty OR element at position zero OF grid IS empty  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        
        IF k IS GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO a list of the pairs zero and one, one and zero, zero and negative one, negative one and zero  
        
        INITIALIZE queue WITH a double ended queue containing a single tuple of zero, zero, k, zero  
        
        INITIALIZE visited WITH a set containing a single tuple of zero, zero, k  
        
        WHILE queue IS NOT empty  
            REMOVE and RETURN the leftmost element FROM queue AND ASSIGN TO x, y, remaining_k, steps  
            
            IF x IS EQUAL TO m MINUS one AND y IS EQUAL TO n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each pair dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx OF grid AT position ny  
                    IF new_k IS GREATER THAN OR EQUAL TO zero AND tuple of nx, ny, new_k IS NOT IN visited  
                        ADD tuple of nx, ny, new_k TO visited  
                        APPEND tuple of nx, ny, new_k, steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
