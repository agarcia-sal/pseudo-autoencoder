CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO FALSE  
        FOR each character IN password  
            IF character IS A LOWERCASE LETTER  
                SET has_lower TO TRUE  
                BREAK  
            END IF  
        END FOR  
        
        SET has_upper TO FALSE  
        FOR each character IN password  
            IF character IS AN UPPERCASE LETTER  
                SET has_upper TO TRUE  
                BREAK  
            END IF  
        END FOR  
        
        SET has_digit TO FALSE  
        FOR each character IN password  
            IF character IS A DIGIT  
                SET has_digit TO TRUE  
                BREAK  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS (the BOOLEAN VALUE OF has_lower PLUS has_upper PLUS has_digit)  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        WHILE i IS LESS THAN n  
            IF the character at position i OF password IS EQUAL TO the character at position i MINUS one OF password AND the character at position i MINUS one OF password IS EQUAL TO the character at position i MINUS two OF password  
                SET length TO two  
                WHILE i IS LESS THAN n AND the character at position i OF password IS EQUAL TO the character at position i MINUS one OF password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three IS EQUAL TO zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three IS EQUAL TO one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN six  
            RETURN the GREATER OF missing_types AND six MINUS n  
        ELSE IF n IS LESS THAN OR EQUAL TO twenty  
            RETURN the GREATER OF missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            SET replace TO replace MINUS the SMALLER OF delete AND one  
            SET replace TO replace MINUS (the SMALLER OF (the GREATER OF delete MINUS one AND zero) AND two MULTIPLIED BY two) DIVIDED BY two  
            SET replace TO replace MINUS (the GREATER OF delete MINUS one MINUS two MULTIPLIED BY two AND zero) DIVIDED BY three  
            
            RETURN delete PLUS the GREATER OF missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
