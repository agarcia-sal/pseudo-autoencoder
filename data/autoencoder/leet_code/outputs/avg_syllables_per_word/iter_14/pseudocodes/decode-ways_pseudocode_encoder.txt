CLASS Solution  
    FUNCTION numDecodings WITH PARAMETER s  
        IF s IS EMPTY OR the element at position zero OF s IS EQUAL TO character zero  
            RETURN zero  
        END IF  
        
        SET dp TO a list of zeros with length EQUAL TO the LENGTH OF s PLUS one  
        SET the element at position zero OF dp TO one  
        SET the element at position one OF dp TO one  
        
        FOR index i FROM two TO the LENGTH OF s PLUS one  
            IF the element at position index i MINUS one OF s IS NOT EQUAL TO character zero  
                INCREMENT the element at position index i OF dp BY the element at position index i MINUS one OF dp  
            END IF  
            
            SET two_digit TO the VALUE OF the two characters FROM position index i MINUS two TO index i OF s INTERPRETED AS integer  
            IF two_digit IS GREATER THAN OR EQUAL TO ten AND two_digit IS LESS THAN OR EQUAL TO twenty six  
                INCREMENT the element at position index i OF dp BY the element at position index i MINUS two OF dp  
            END IF  
        END FOR  
        
        RETURN the last element OF dp  
    END FUNCTION  
END CLASS
