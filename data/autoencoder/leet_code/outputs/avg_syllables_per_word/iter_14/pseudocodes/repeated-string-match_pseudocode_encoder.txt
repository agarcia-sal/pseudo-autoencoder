CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET length_of_b TO the LENGTH OF b  
        SET length_of_a TO the LENGTH OF a  
        SET minimum_repeats TO the NEGATIVE OF the INTEGER DIVISION RESULT FROM NEGATIVE length_of_b DIVIDED BY length_of_a  
        
        IF b IS CONTAINED WITHIN the STRING RESULTING FROM REPEATING a minimum_repeats TIMES  
            RETURN minimum_repeats  
        END IF  
        
        IF b IS CONTAINED WITHIN the STRING RESULTING FROM REPEATING a minimum_repeats PLUS one TIMES  
            RETURN minimum_repeats PLUS one  
        END IF  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
