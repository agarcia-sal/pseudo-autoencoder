CLASS Solution  
    FUNCTION findSubstring WITH PARAMETERS s AND words  
        IF s IS EMPTY OR words IS EMPTY OR element at position zero OF words IS EMPTY  
            RETURN an EMPTY list  
        END IF  
        
        SET word_length TO the LENGTH OF element at position zero OF words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the FREQUENCY COUNTER OF words  
        SET result TO an EMPTY list  
        
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an EMPTY FREQUENCY COUNTER  
            FOR j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO the SUBSTRING FROM position j TO j PLUS word_length MINUS one OF s  
                INCREMENT the FREQUENCY OF word IN seen BY one  
                IF the FREQUENCY OF word IN seen IS GREATER THAN the FREQUENCY OF word IN word_count  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
            IF the inner FOR loop was NOT broken  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
