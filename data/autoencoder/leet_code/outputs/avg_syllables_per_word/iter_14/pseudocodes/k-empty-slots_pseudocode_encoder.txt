CLASS Solution  
    FUNCTION kEmptySlots WITH PARAMETERS bulbs AND k  
        SET bloom_days TO an empty list  
        SET day TO one  
        FOR each position IN bulbs  
            SET i TO the position to insert position in bloom_days to keep it sorted in ascending order using bisect left  
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one OF bloom_days MINUS one EQUAL TO k  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i OF bloom_days MINUS position MINUS one EQUAL TO k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT index i  
            INCREMENT day BY one  
        END FOR  
        RETURN minus one  
    END FUNCTION  
END CLASS
