CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1 val2)  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero THEN  
                RETURN True  
            ELSE IF val1 LESS THAN zero AND val2 LESS THAN zero THEN  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
          
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
            
            IF element at position i OF nums EQUAL TO zero THEN  
                CONTINUE TO next iteration of FOR loop  
            END IF  
            
            WHILE True  
                SET slow TO slow PLUS element at position slow OF nums  
                IF slow LESS THAN zero THEN  
                    SET slow TO slow PLUS n  
                END IF  
                SET slow TO slow MODULO n  
                
                SET fast TO fast PLUS element at position fast OF nums  
                IF fast LESS THAN zero THEN  
                    SET fast TO fast PLUS n  
                END IF  
                SET fast TO fast MODULO n  
                
                SET fast TO fast PLUS element at position fast OF nums  
                IF fast LESS THAN zero THEN  
                    SET fast TO fast PLUS n  
                END IF  
                SET fast TO fast MODULO n  
                
                IF sameDirection(element at position slow OF nums element at position i OF nums) IS False OR sameDirection(element at position fast OF nums element at position i OF nums) IS False THEN  
                    BREAK the WHILE loop  
                END IF  
                
                IF slow EQUAL TO fast THEN  
                    IF slow EQUAL TO (slow PLUS element at position slow OF nums) MODULO n THEN  
                        BREAK the WHILE loop  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
