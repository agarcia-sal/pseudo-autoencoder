CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional array of size n BY n FILLED WITH zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND position i OF dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF character at position i OF s EQUAL TO character at position j OF s THEN  
                    SET element at position i PLUS one AND position j MINUS one OF dp TO be value at element at position i PLUS one AND position j MINUS one OF dp PLUS two  
                    SET element at position i AND position j OF dp TO that value  
                ELSE  
                    IF value at element at position i PLUS one AND position j OF dp GREATER THAN OR EQUAL TO value at element at position i AND position j MINUS one OF dp THEN  
                        SET element at position i AND position j OF dp TO value at element at position i PLUS one AND position j OF dp  
                    ELSE  
                        SET element at position i AND position j OF dp TO value at element at position i AND position j MINUS one OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN value at element at position zero AND position n MINUS one OF dp  
    END FUNCTION  
END CLASS
