CLASS Solution  
    FUNCTION minDistance WITH PARAMETERS word1 AND word2  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL create_two_dimensional_list WITH PARAMETERS m PLUS one AND n PLUS one AND ASSIGN RESULT TO dp
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the ELEMENT at position i MINUS one OF word1 IS EQUAL TO the ELEMENT at position j MINUS one OF word2  
                    SET the ELEMENT at row i AND column j OF dp TO the ELEMENT at row i MINUS one AND column j MINUS one OF dp PLUS one  
                ELSE  
                    SET the ELEMENT at row i AND column j OF dp TO the GREATER VALUE BETWEEN the ELEMENT at row i MINUS one AND column j OF dp AND the ELEMENT at row i AND column j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO the ELEMENT at row m AND column n OF dp  
        
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
    
    FUNCTION create_two_dimensional_list WITH PARAMETERS rows AND columns  
        CREATE empty list named result_list  
        FOR each index_row FROM zero TO rows MINUS one  
            CREATE empty list named row_list  
            FOR each index_column FROM zero TO columns MINUS one  
                APPEND zero TO row_list  
            END FOR  
            APPEND row_list TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
