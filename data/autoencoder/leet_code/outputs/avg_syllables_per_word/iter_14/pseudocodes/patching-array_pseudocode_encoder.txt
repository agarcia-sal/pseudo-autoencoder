CLASS Solution  
    FUNCTION minPatches(nums AS list OF integer, n AS integer) RETURNS integer  
        SET missing TO one  
        SET patches TO zero  
        SET i TO zero  
        WHILE missing IS LESS THAN OR EQUAL TO n IS TRUE  
            IF i IS LESS THAN the LENGTH OF nums AND element at position i OF nums IS LESS THAN OR EQUAL TO missing IS TRUE  
                SET missing TO missing PLUS element at position i OF nums  
                INCREMENT i BY one  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        RETURN patches  
    END FUNCTION  
END CLASS
