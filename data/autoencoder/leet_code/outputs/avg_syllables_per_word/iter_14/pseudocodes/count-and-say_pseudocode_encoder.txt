CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n IS EQUAL TO one  
            RETURN the string one  
        END IF  
        
        SET current_sequence TO the string one  
        
        FOR each iteration FROM two TO n  
            SET next_sequence TO empty string  
            SET count TO one  
            SET previous_char TO the element at position zero OF current_sequence  
            
            FOR each index FROM one TO the LENGTH OF current_sequence MINUS one  
                IF the element at position index OF current_sequence IS EQUAL TO previous_char  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO the element at position index OF current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            
            APPEND the string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            
            SET current_sequence TO next_sequence  
        END FOR  
        
        RETURN current_sequence  
    END FUNCTION  
END CLASS
