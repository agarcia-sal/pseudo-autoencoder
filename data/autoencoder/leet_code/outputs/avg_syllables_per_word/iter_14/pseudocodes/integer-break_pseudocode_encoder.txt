CLASS Solution  
    FUNCTION integerBreak WITH PARAMETER n  
        SET dp TO auxiliary FUNCTION create_list_of_zeroes WITH PARAMETER n PLUS one  
        SET element at position one OF dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET candidate_one TO max_product  
                SET candidate_two TO j MULTIPLIED BY element at position i MINUS j OF dp  
                SET candidate_three TO j MULTIPLIED BY i MINUS j  
                IF candidate_two GREATER THAN candidate_one AND candidate_two GREATER THAN max_product  
                    SET max_product TO candidate_two  
                END IF  
                IF candidate_three GREATER THAN candidate_one AND candidate_three GREATER THAN max_product  
                    SET max_product TO candidate_three  
                END IF  
            END FOR  
            SET element at position i OF dp TO max_product  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
