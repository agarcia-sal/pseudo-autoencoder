CLASS Solution  
    FUNCTION singleNonDuplicate(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS the DIFFERENCE OF right MINUS left DIVIDED BY two  
            
            IF the REMAINDER OF mid DIVIDED BY two IS EQUAL TO one  
                DECREMENT mid BY one  
            END IF  
            
            IF the ELEMENT AT POSITION mid OF nums IS EQUAL TO the ELEMENT AT POSITION mid PLUS one OF nums  
                SET left TO mid PLUS two  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN the ELEMENT AT POSITION left OF nums  
    END FUNCTION  
END CLASS
