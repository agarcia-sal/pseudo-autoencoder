CLASS Solution  
    FUNCTION solveSudoku WITH PARAMETERS board  
        FUNCTION is_valid WITH PARAMETERS board row col num  
            FOR i FROM zero TO eight  
                IF element at position col OF element at position row OF board IS EQUAL TO num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position col OF element at position i OF board IS EQUAL TO num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_col PLUS j OF element at position start_row PLUS i OF board IS EQUAL TO num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        FUNCTION solve WITH PARAMETER board  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position col OF element at position row OF board IS EQUAL TO character dot  
                        FOR num IN string representations of numbers FROM one TO nine  
                            IF function is_valid WITH ARGUMENTS board row col num RETURNS True  
                                SET element at position col OF element at position row OF board TO num  
                                IF function solve WITH ARGUMENT board RETURNS True  
                                    RETURN True  
                                END IF  
                                SET element at position col OF element at position row OF board TO character dot  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        CALL function solve WITH ARGUMENT board  
    END FUNCTION  
END CLASS
