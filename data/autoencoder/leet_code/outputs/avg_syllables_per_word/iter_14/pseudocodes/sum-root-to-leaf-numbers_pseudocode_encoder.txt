CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EQUAL TO empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH PARAMETER element at position zero OF values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT EQUAL TO none  
            SET node.left TO A NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT EQUAL TO none  
            SET node.right TO A NEW TreeNode WITH PARAMETER element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS EQUAL TO none AND q IS EQUAL TO none  
        RETURN true  
    ELSE IF p IS EQUAL TO none OR q IS EQUAL TO none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left q.left AND is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers WITH PARAMETERS self root  
        FUNCTION dfs WITH PARAMETERS node current_number  
            IF node IS EQUAL TO none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS node.val  
            IF node.left IS EQUAL TO none AND node.right IS EQUAL TO none  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs WITH PARAMETERS node.left current_number  
            SET right_sum TO dfs WITH PARAMETERS node.right current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs WITH PARAMETERS root zero  
    END FUNCTION  
END CLASS
