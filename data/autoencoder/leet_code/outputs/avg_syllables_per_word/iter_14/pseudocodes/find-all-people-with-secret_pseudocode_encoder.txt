CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero AND firstPerson  
        SORT meetings BY the value at position two OF each element IN meetings IN ascending order  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the value at position two OF the element at position i OF meetings  
            SET graph TO a new mapping from keys TO empty lists by default  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the value at position two OF the element at position i OF meetings IS EQUAL TO current_time  
                SET x TO the value at position zero OF the element at position i OF meetings  
                SET y TO the value at position one OF the element at position i OF meetings  
                APPEND y TO the list mapped by x IN graph  
                APPEND x TO the list mapped by y IN graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new deque containing each person FROM know_secret WHERE the person IS a key IN graph  
            SET visited TO an empty set  
            WHILE the LENGTH OF queue IS GREATER THAN zero  
                REMOVE AND RETURN the element at the front OF queue AND ASSIGN it TO person  
                IF person IS IN visited  
                    CONTINUE TO the next iteration OF the WHILE loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list mapped by person IN graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list converted FROM know_secret  
    END FUNCTION  
END CLASS
