CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO NEW TreeNode WITH values[0]
    SET i TO 1
    CREATE queue AS EMPTY DEQUE
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH values[i]
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH values[i]
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION trimBST(root: Optional[TreeNode], low: int, high: int) -> Optional[TreeNode]
        IF root IS None
            RETURN None
        IF root.val < low
            RETURN self.trimBST(root.right, low, high)
        IF root.val > high
            RETURN self.trimBST(root.left, low, high)
        SET root.left TO self.trimBST(root.left, low, high)
        SET root.right TO self.trimBST(root.right, low, high)
        RETURN root
    END FUNCTION
END CLASS
