CLASS Solution
    FUNCTION combinationSum2(candidates, target)
        FUNCTION backtrack(start, target, path)
            IF target EQUALS 0 THEN
                APPEND path TO result
                RETURN
            END IF
            IF target LESS THAN 0 THEN
                RETURN
            END IF
            FOR i FROM start TO LENGTH OF candidates MINUS 1
                IF i GREATER THAN start AND candidates[i] EQUALS candidates[i - 1] THEN
                    CONTINUE
                END IF
                CALL backtrack(i + 1, target MINUS candidates[i], path APPENDED WITH candidates[i])
            END FOR
        END FUNCTION

        SORT candidates IN NON-DECREASING ORDER
        SET result TO EMPTY LIST
        CALL backtrack(0, target, EMPTY LIST)
        RETURN result
    END FUNCTION
END CLASS
