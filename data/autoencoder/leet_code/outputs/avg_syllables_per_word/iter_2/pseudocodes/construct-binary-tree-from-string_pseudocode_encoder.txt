CLASS ListNode
    FUNCTION __init__(val = 0, next = None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET head TO ListNode(values[0])
    SET p TO head
    FOR val IN values FROM SECOND ELEMENT TO END
        SET node TO ListNode(val)
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    CREATE queue AS deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION str2tree(s)
        FUNCTION parse_tree(index)
            IF index GREATER THAN OR EQUAL TO LENGTH OF s
                RETURN None, index
            END IF

            SET start TO index
            WHILE index LESS THAN LENGTH OF s AND (s[index] IS DIGIT OR s[index] EQUALS '-')
                INCREMENT index BY 1
            END WHILE

            SET node TO TreeNode(CONVERT s FROM start TO index TO INTEGER)

            IF index LESS THAN LENGTH OF s AND s[index] EQUALS '('
                SET node.left, index TO parse_tree(index + 1)
            END IF

            IF index LESS THAN LENGTH OF s AND s[index] EQUALS '('
                SET node.right, index TO parse_tree(index + 1)
            END IF

            IF index LESS THAN LENGTH OF s AND s[index] EQUALS ')'
                INCREMENT index BY 1
            END IF

            RETURN node, index
        END FUNCTION

        SET root, _ TO parse_tree(0)
        RETURN root
    END FUNCTION
END CLASS
