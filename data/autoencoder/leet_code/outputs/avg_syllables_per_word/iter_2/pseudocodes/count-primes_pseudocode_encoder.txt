CLASS Solution
    FUNCTION countPrimes(n)
        IF n <= 2 THEN
            RETURN 0
        END IF

        SET is_prime TO list of True repeated n times
        SET is_prime[0] TO False
        SET is_prime[1] TO False

        FOR start FROM 2 TO integer part of square root of n INCLUSIVE
            IF is_prime[start] IS True THEN
                FOR multiple FROM start * start TO n - 1 STEPPING BY start
                    SET is_prime[multiple] TO False
                END FOR
            END IF
        END FOR

        RETURN count of True values in is_prime
    END FUNCTION
END CLASS
