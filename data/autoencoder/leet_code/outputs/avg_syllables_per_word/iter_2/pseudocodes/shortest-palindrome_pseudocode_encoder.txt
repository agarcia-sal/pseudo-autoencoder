CLASS Solution
    FUNCTION shortestPalindrome(s)
        IF s IS empty
            RETURN s
        END IF

        FUNCTION is_palindrome(check_str)
            RETURN check_str IS EQUAL TO reverse of check_str
        END FUNCTION

        FOR i FROM LENGTH OF s DOWNTO 0
            IF is_palindrome(substring of s FROM 0 TO i)
                SET to_add TO reverse of substring of s FROM i TO end
                RETURN to_add CONCATENATED WITH s
            END IF
        END FOR
    END FUNCTION
END CLASS
