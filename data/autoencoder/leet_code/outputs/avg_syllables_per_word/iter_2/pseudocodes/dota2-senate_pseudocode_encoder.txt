CLASS Solution
    FUNCTION predictPartyVictory(senate)
        INITIALIZE radiant AS empty queue
        INITIALIZE dire AS empty queue
        SET length TO LENGTH OF senate

        FOR each index, senator IN senate with indices
            IF senator EQUALS 'R' THEN
                APPEND index TO radiant
            ELSE
                APPEND index TO dire
        END FOR

        WHILE radiant IS NOT EMPTY AND dire IS NOT EMPTY
            SET r_index TO REMOVE FIRST ELEMENT FROM radiant
            SET d_index TO REMOVE FIRST ELEMENT FROM dire

            IF r_index LESS THAN d_index THEN
                APPEND r_index PLUS length TO radiant
            ELSE
                APPEND d_index PLUS length TO dire
        END WHILE

        IF radiant IS NOT EMPTY THEN
            RETURN "Radiant"
        ELSE
            RETURN "Dire"
    END FUNCTION
END CLASS
