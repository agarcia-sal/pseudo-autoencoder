CLASS Solution
    FUNCTION productExceptSelf(nums)
        SET n TO LENGTH OF nums
        SET answer TO LIST of 1 REPEATED n TIMES
        
        SET left_product TO 1
        FOR i FROM 0 TO n - 1
            SET answer[i] TO left_product
            MULTIPLY left_product BY nums[i]
        
        SET right_product TO 1
        FOR i FROM n - 1 DOWN TO 0
            MULTIPLY answer[i] BY right_product
            MULTIPLY right_product BY nums[i]
        
        RETURN answer
    END FUNCTION
END CLASS
