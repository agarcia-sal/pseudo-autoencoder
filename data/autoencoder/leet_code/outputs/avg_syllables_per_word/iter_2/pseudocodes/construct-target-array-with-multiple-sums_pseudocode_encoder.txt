CLASS Solution
    FUNCTION isPossible(target: List of Integer) RETURNS Boolean
        SET max_heap TO a list of negative values from target
        TRANSFORM max_heap INTO a heap structure
        
        SET total_sum TO sum of values in target
        
        WHILE the top value in max_heap IS LESS THAN -1
            SET largest TO negative of removing top value from max_heap
            SET rest_sum TO total_sum MINUS largest
            
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS 0
                RETURN False
            END IF
            
            SET new_value TO largest MODULO rest_sum
            IF new_value IS 0 AND rest_sum IS NOT 1
                RETURN False
            END IF
            
            SET total_sum TO rest_sum PLUS new_value
            ADD negative of new_value TO max_heap
        END WHILE
        
        RETURN True
    END FUNCTION
END CLASS
