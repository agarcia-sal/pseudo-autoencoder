CLASS Solution
    FUNCTION isOneEditDistance(s, t)
        IF LENGTH OF s > LENGTH OF t THEN
            RETURN self.isOneEditDistance(t, s)
        END IF

        SET diff TO LENGTH OF t MINUS LENGTH OF s

        IF diff > 1 THEN
            RETURN False
        END IF

        IF diff == 0 THEN
            SET mismatch_count TO 0
            FOR i FROM 0 TO LENGTH OF s MINUS 1
                IF s[i] NOT EQUAL t[i] THEN
                    INCREMENT mismatch_count BY 1
                END IF
            END FOR
            RETURN mismatch_count EQUALS 1
        ELSE
            FOR i FROM 0 TO LENGTH OF s MINUS 1
                IF s[i] NOT EQUAL t[i] THEN
                    RETURN SUBSTRING OF s FROM i TO END EQUALS SUBSTRING OF t FROM i+1 TO END
                END IF
            END FOR
            RETURN True
        END IF
    END FUNCTION
END CLASS
