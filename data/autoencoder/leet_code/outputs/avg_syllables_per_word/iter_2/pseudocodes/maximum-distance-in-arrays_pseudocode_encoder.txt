CLASS Solution
    FUNCTION maxDistance(arrays)
        SET min_val TO first element of first array in arrays
        SET max_val TO last element of first array in arrays
        SET max_distance TO 0
        FOR i FROM 1 TO LENGTH OF arrays MINUS 1
            SET current_array TO arrays AT INDEX i
            SET current_min TO first element of current_array
            SET current_max TO last element of current_array
            SET max_distance TO maximum OF max_distance, absolute value OF (current_max MINUS min_val), absolute value OF (max_val MINUS current_min)
            SET min_val TO minimum OF min_val, current_min
            SET max_val TO maximum OF max_val, current_max
        RETURN max_distance
    END FUNCTION
END CLASS
