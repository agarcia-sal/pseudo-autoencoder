CLASS Solution
    FUNCTION uniqueLetterString(s)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7
        SET n TO LENGTH OF s

        SET last TO LIST OF SIZE 26 WITH ALL VALUES -1
        SET last2 TO LIST OF SIZE 26 WITH ALL VALUES -1

        SET result TO 0

        FOR i, char IN ENUMERATE s
            SET idx TO ORDINAL VALUE OF char MINUS ORDINAL VALUE OF 'A'

            SET result TO (result PLUS (i MINUS last[idx]) TIMES (last[idx] MINUS last2[idx])) MODULO MOD

            SET last2[idx] TO last[idx]
            SET last[idx] TO i

        FOR i IN RANGE 0 TO 25
            SET result TO (result PLUS (n MINUS last[i]) TIMES (last[i] MINUS last2[i])) MODULO MOD

        RETURN result
    END FUNCTION
END CLASS
