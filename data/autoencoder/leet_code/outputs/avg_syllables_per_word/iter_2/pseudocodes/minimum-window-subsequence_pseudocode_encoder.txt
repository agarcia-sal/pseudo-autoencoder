CLASS Solution
    FUNCTION minWindow(s1: string, s2: string) RETURNS string
        SET min_len TO infinity
        SET min_start TO -1
        SET len1 TO LENGTH OF s1
        SET len2 TO LENGTH OF s2
        
        FOR i FROM 0 TO len1 - 1
            SET j TO 0
            FOR k FROM i TO len1 - 1
                IF s1[k] EQUALS s2[j] THEN
                    INCREMENT j BY 1
                    IF j EQUALS len2 THEN
                        IF (k - i + 1) LESS THAN min_len THEN
                            SET min_len TO (k - i + 1)
                            SET min_start TO i
                        END IF
                        BREAK
                    END IF
                END IF
            END FOR
        END FOR
        
        IF min_start EQUALS -1 THEN
            RETURN empty string
        ELSE
            RETURN SUBSTRING OF s1 FROM min_start TO (min_start + min_len - 1)
        END IF
    END FUNCTION
END CLASS
