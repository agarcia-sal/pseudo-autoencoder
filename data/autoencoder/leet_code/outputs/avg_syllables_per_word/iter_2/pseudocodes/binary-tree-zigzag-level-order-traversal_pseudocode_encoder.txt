CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO new TreeNode(values[0])
    SET i TO 1
    SET queue TO new deque containing root
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO new TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO new TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION zigzagLevelOrder(root)
        IF root IS None
            RETURN empty list
        SET result TO empty list
        SET queue TO new deque containing root
        SET left_to_right TO True
        WHILE queue IS NOT EMPTY
            SET level_size TO LENGTH OF queue
            SET current_level TO new empty deque
            FOR EACH _ FROM 1 TO level_size
                SET node TO queue.popleft()
                IF left_to_right IS True
                    APPEND node.val TO current_level
                ELSE
                    APPENDLEFT node.val TO current_level
                IF node.left EXISTS
                    APPEND node.left TO queue
                IF node.right EXISTS
                    APPEND node.right TO queue
            END FOR
            APPEND list formed from current_level TO result
            SET left_to_right TO NOT left_to_right
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
