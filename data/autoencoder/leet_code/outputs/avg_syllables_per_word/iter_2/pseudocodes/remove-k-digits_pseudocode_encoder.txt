CLASS Solution
    FUNCTION removeKdigits(num, k)
        SET stack TO empty list
        
        FOR each digit IN num
            WHILE k > 0 AND stack is not empty AND last element in stack > digit
                REMOVE last element from stack
                DECREMENT k BY 1
            END WHILE
            APPEND digit TO stack
        END FOR
        
        IF k > 0 THEN
            SET final_stack TO stack EXCLUDING last k elements
        ELSE
            SET final_stack TO stack
        END IF
        
        SET result TO concatenation of elements in final_stack
        REMOVE leading zeros FROM result
        
        IF result is empty THEN
            RETURN "0"
        ELSE
            RETURN result
        END IF
    END FUNCTION
END CLASS
