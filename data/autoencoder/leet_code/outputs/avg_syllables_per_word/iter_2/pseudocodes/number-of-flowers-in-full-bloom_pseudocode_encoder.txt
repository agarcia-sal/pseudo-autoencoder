CLASS Solution
    FUNCTION fullBloomFlowers(flowers: List of List of Integer, people: List of Integer) RETURNS List of Integer
        SET start_times TO sorted list of all start times from flowers
        SET end_times TO sorted list of all end times from flowers
        SET result TO empty list
        
        FOR each person IN people
            SET started TO count of values in start_times less than or equal to person
            SET ended TO count of values in end_times less than person
            APPEND started minus ended TO result
        
        RETURN result
    END FUNCTION
END CLASS
