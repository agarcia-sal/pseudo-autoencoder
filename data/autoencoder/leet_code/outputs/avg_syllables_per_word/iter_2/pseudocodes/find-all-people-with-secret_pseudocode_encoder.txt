CLASS Solution
    FUNCTION findAllPeople(n, meetings, firstPerson)
        SET know_secret TO SET containing 0 and firstPerson
        SORT meetings BY the third element (time)
        SET i TO 0
        WHILE i LESS THAN LENGTH OF meetings
            SET current_time TO meetings[i][2]
            SET graph TO empty mapping from person to list of connected persons
            WHILE i LESS THAN LENGTH OF meetings AND meetings[i][2] EQUALS current_time
                SET x TO meetings[i][0]
                SET y TO meetings[i][1]
                APPEND y TO graph[x]
                APPEND x TO graph[y]
                INCREMENT i BY 1
            SET queue TO list of persons IN know_secret THAT ARE KEYS IN graph
            SET queue TO deque initialized with that list
            SET visited TO empty set
            WHILE queue IS NOT empty
                SET person TO REMOVE first element FROM queue
                IF person IN visited THEN
                    CONTINUE
                ADD person TO visited
                ADD person TO know_secret
                FOR EACH neighbor IN graph[person]
                    IF neighbor NOT IN visited THEN
                        ADD neighbor TO queue
        RETURN list converted from know_secret
    END FUNCTION
END CLASS
