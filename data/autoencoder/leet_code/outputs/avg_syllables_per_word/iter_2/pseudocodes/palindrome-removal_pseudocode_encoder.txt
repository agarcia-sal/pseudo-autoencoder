CLASS Solution
    FUNCTION minimumMoves(arr)
        SET n TO LENGTH OF arr
        SET dp TO a two-dimensional list of size n by n with all values 0

        FOR i FROM 0 TO n - 1
            SET dp[i][i] TO 1

        FOR i FROM 0 TO n - 2
            IF arr[i] EQUALS arr[i + 1]
                SET dp[i][i + 1] TO 1
            ELSE
                SET dp[i][i + 1] TO 2

        FOR length FROM 3 TO n
            FOR i FROM 0 TO n - length
                SET j TO i + length - 1
                IF arr[i] EQUALS arr[j]
                    SET dp[i][j] TO dp[i + 1][j - 1]
                ELSE
                    SET dp[i][j] TO infinity
                    FOR k FROM i TO j - 1
                        SET dp[i][j] TO MINIMUM OF dp[i][j] AND (dp[i][k] + dp[k + 1][j])

        RETURN dp[0][n - 1]
    END FUNCTION
END CLASS
