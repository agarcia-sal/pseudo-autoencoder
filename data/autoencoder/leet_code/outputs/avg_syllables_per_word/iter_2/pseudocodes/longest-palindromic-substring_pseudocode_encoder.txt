CLASS Solution
    FUNCTION longestPalindrome(s)
        SET n TO LENGTH OF s
        SET f TO two-dimensional list of size n by n with all values TRUE
        SET k TO 0
        SET mx TO 1
        FOR i FROM n - 2 DOWNTO 0
            FOR j FROM i + 1 TO n - 1
                SET f[i][j] TO FALSE
                IF s[i] EQUALS s[j]
                    SET f[i][j] TO f[i + 1][j - 1]
                    IF f[i][j] IS TRUE AND mx LESS THAN j MINUS i PLUS 1
                        SET k TO i
                        SET mx TO j MINUS i PLUS 1
        RETURN SUBSTRING OF s FROM k TO k PLUS mx
    END FUNCTION
END CLASS
