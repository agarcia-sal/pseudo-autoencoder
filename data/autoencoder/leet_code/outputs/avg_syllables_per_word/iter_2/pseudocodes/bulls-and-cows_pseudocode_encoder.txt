CLASS Solution
    FUNCTION getHint(secret, guess)
        SET bulls TO 0
        SET cows TO 0
        SET secret_count TO empty dictionary with default integer values
        SET guess_count TO empty dictionary with default integer values

        FOR each pair (s, g) IN corresponding characters of secret and guess
            IF s EQUALS g THEN
                INCREMENT bulls BY 1
            ELSE
                INCREMENT secret_count[s] BY 1
                INCREMENT guess_count[g] BY 1

        FOR each digit IN keys of secret_count
            INCREMENT cows BY minimum of secret_count[digit] and guess_count[digit]

        RETURN string concatenation of bulls, "A", cows, "B"
    END FUNCTION
END CLASS
