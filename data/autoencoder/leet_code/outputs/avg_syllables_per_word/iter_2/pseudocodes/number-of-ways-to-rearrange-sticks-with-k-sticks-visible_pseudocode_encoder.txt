CLASS Solution
    FUNCTION rearrangeSticks(n, k)
        SET MOD TO 10^9 + 7
        SET dp TO a 2D list of size (n + 1) by (k + 1) filled with 0
        SET dp[0][0] TO 1
        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                SET dp[i][j] TO (dp[i - 1][j - 1] + (i - 1) * dp[i - 1][j]) MODULO MOD
        RETURN dp[n][k]
    END FUNCTION
END CLASS
