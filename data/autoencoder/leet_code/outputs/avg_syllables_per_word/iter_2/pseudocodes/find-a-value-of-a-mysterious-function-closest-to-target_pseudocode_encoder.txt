CLASS Solution
    FUNCTION closestToTarget(arr, target)
        SET seen TO empty set
        SET min_diff TO infinity
        FOR each num IN arr
            SET seen TO set of (num AND x) FOR each x IN seen UNION {num}
            FOR each val IN seen
                IF absolute value of (val MINUS target) < min_diff THEN
                    SET min_diff TO absolute value of (val MINUS target)
        RETURN min_diff
    END FUNCTION
END CLASS
