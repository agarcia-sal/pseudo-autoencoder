CLASS Solution
    FUNCTION topKFrequent(nums: List of Integer, k: Integer) RETURNS List of Integer
        SET frequency_map TO a map with default value 0
        FOR each num IN nums
            INCREMENT frequency_map[num] BY 1
        SET max_frequency TO the maximum value in frequency_map
        SET buckets TO a list of empty lists, with size max_frequency plus 1
        FOR each num, freq IN frequency_map items
            APPEND num TO buckets[freq]
        SET result TO an empty list
        FOR freq FROM max_frequency DOWN TO 1
            IF LENGTH OF buckets[freq] > 0 THEN
                APPEND all elements of buckets[freq] TO result
                IF LENGTH OF result >= k THEN
                    RETURN the first k elements of result
        RETURN result
    END FUNCTION
END CLASS
