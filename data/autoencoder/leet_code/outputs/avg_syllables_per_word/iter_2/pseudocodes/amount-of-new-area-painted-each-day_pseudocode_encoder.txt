CLASS Solution
    FUNCTION amountPainted(paint: List of List of Integer)
        SET max_end TO maximum end value IN paint
        SET painted TO list of size (max_end + 1) FILLED with -1
        SET worklog TO empty list
        
        FOR each start, end IN paint
            SET current_start TO start
            SET new_paint TO 0
            
            WHILE current_start < end
                IF painted[current_start] EQUALS -1
                    SET painted[current_start] TO end
                    INCREMENT new_paint BY 1
                    INCREMENT current_start BY 1
                ELSE
                    SET current_start TO painted[current_start]
            
            APPEND new_paint TO worklog
        
        RETURN worklog
    END FUNCTION
END CLASS
