CLASS Solution
    FUNCTION stoneGameVIII(stones)
        SET prefix_sum TO list of zeros with length of stones
        SET prefix_sum[0] TO stones[0]
        FOR i FROM 1 TO LENGTH OF stones MINUS 1
            SET prefix_sum[i] TO prefix_sum[i - 1] PLUS stones[i]
        SET dp TO prefix_sum at last index
        FOR i FROM LENGTH OF stones MINUS 2 DOWNTO 1 STEP -1
            SET dp TO MAXIMUM of dp AND (prefix_sum[i] MINUS dp)
        RETURN dp
    END FUNCTION
END CLASS
