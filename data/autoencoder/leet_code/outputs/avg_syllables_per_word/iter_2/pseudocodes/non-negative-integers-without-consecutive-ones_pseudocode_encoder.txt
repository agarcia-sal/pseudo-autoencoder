CLASS Solution
    FUNCTION findIntegers(n)
        SET bin_rep TO binary representation of n as string without '0b' prefix
        SET length TO length of bin_rep
        
        SET dp TO list of zeros with size length plus 1
        SET dp[0] TO 1
        SET dp[1] TO 2
        
        FOR i FROM 2 TO length
            SET dp[i] TO dp[i - 1] PLUS dp[i - 2]
        
        SET result TO 0
        SET prev_bit TO 0
        
        FOR i FROM 0 TO length MINUS 1
            IF bin_rep[i] EQUALS '1' THEN
                INCREMENT result BY dp[length MINUS i MINUS 1]
                IF prev_bit EQUALS 1 THEN
                    DECREMENT result BY 1
                    BREAK LOOP
            END IF
            SET prev_bit TO 1 IF bin_rep[i] EQUALS '1' OTHERWISE 0
        
        RETURN result PLUS 1
    END FUNCTION
END CLASS
