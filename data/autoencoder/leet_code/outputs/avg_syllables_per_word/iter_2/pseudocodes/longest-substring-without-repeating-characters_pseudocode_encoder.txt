CLASS Solution
    FUNCTION lengthOfLongestSubstring(s)
        SET char_index_map TO empty map
        SET max_length TO 0
        SET start TO 0
        
        FOR each end, char IN s WITH index
            IF char IN char_index_map AND char_index_map[char] >= start THEN
                SET start TO char_index_map[char] + 1
            END IF
            SET char_index_map[char] TO end
            SET max_length TO maximum OF max_length AND (end - start + 1)
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
