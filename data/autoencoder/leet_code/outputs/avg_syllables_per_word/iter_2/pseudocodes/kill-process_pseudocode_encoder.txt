CLASS Solution
    FUNCTION killProcess(pid: List[int], ppid: List[int], kill: int) RETURNS List[int]
        SET tree TO a mapping from integer to list of integers with default empty list
        FOR each child, parent IN zip of pid and ppid
            APPEND child TO tree[parent]
        END FOR

        SET queue TO list containing kill
        SET killed_processes TO empty list

        WHILE queue is not empty
            REMOVE first element FROM queue AND SET TO current
            APPEND current TO killed_processes
            EXTEND queue BY tree[current]
        END WHILE

        RETURN killed_processes
    END FUNCTION
END CLASS
