CLASS Solution
    FUNCTION nextClosestTime(time)
        SET digits TO SET OF CHARACTERS IN time WITHOUT ":"
        SET hour, minute TO INTEGERS FROM SPLITTING time BY ":"
        
        FUNCTION generate_times(digits)
            SET times TO EMPTY LIST
            FOR each h1 IN digits
                FOR each h2 IN digits
                    FOR each m1 IN digits
                        FOR each m2 IN digits
                            SET hour_val TO INTEGER FROM CONCATENATING h1 AND h2
                            SET minute_val TO INTEGER FROM CONCATENATING m1 AND m2
                            IF hour_val LESS THAN 24 AND minute_val LESS THAN 60 THEN
                                APPEND (hour_val, minute_val) TO times
            RETURN times SORTED IN ASCENDING ORDER
        END FUNCTION
        
        SET all_times TO generate_times(digits)
        SET current_time_index TO POSITION OF (hour, minute) IN all_times
        
        IF current_time_index + 1 LESS THAN LENGTH OF all_times THEN
            SET next_hour, next_minute TO all_times AT current_time_index + 1
        ELSE
            SET next_hour, next_minute TO all_times AT 0
        
        RETURN next_hour FORMATTED TO TWO DIGITS CONCATENATED WITH ":" AND next_minute FORMATTED TO TWO DIGITS
    END FUNCTION
END CLASS
