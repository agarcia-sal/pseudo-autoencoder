CLASS Solution
    FUNCTION findShortestWay(maze: List of List of int, ball: List of int, hole: List of int) RETURNS string
        SET directions TO [(0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')]
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze[0]
        SET start TO tuple of ball
        SET hole TO tuple of hole
        SET priority_queue TO list containing (0, empty string, start)
        SET visited TO empty set

        WHILE priority_queue IS NOT empty
            REMOVE (dist, path, (x, y)) WITH smallest dist FROM priority_queue

            IF (x, y) EQUALS hole
                RETURN path

            IF (x, y) IN visited
                CONTINUE to next iteration

            ADD (x, y) TO visited

            FOR EACH (dx, dy, direction) IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO 0

                WHILE new_x + dx IS BETWEEN 0 AND m - 1 AND new_y + dy IS BETWEEN 0 AND n - 1 AND maze[new_x + dx][new_y + dy] EQUALS 0
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY 1

                    IF (new_x, new_y) EQUALS hole
                        BREAK out of WHILE loop

                IF (new_x, new_y) NOT IN visited
                    ADD (dist + count, path CONCATENATED WITH direction, (new_x, new_y)) TO priority_queue

        RETURN "impossible"
    END FUNCTION
END CLASS
