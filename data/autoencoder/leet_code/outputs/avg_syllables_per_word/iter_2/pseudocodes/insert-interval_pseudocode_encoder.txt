CLASS Solution
    FUNCTION insert(intervals, newInterval)
        IF intervals IS EMPTY THEN
            RETURN list containing newInterval
        END IF

        SET merged TO empty list
        SET i TO 0
        SET n TO LENGTH OF intervals

        WHILE i LESS THAN n AND intervals[i][1] LESS THAN newInterval[0]
            APPEND intervals[i] TO merged
            INCREMENT i BY 1
        END WHILE

        WHILE i LESS THAN n AND intervals[i][0] LESS THAN OR EQUAL TO newInterval[1]
            SET newInterval[0] TO MINIMUM OF newInterval[0] AND intervals[i][0]
            SET newInterval[1] TO MAXIMUM OF newInterval[1] AND intervals[i][1]
            INCREMENT i BY 1
        END WHILE

        APPEND newInterval TO merged

        WHILE i LESS THAN n
            APPEND intervals[i] TO merged
            INCREMENT i BY 1
        END WHILE

        RETURN merged
    END FUNCTION
END CLASS
