CLASS Solution
    FUNCTION kMirror(k, n)
        FUNCTION is_palindrome(s)
            RETURN s EQUALS s REVERSED
        END FUNCTION

        FUNCTION to_base_k(num, k)
            IF num EQUALS 0
                RETURN "0"
            SET digits TO empty list
            WHILE num IS NOT 0
                APPEND (num MODULO k) TO digits
                SET num TO num DIVIDED BY k
            RETURN concatenation of string versions of digits REVERSED
        END FUNCTION

        FUNCTION generate_palindromes(length)
            IF length EQUALS 1
                FOR i FROM 1 TO 9
                    YIELD string version of i
            ELSE
                SET start TO 10 RAISED TO THE POWER OF ((length MINUS 1) DIVIDED BY 2)
                SET end TO 10 RAISED TO THE POWER OF ((length PLUS 1) DIVIDED BY 2)
                FOR half FROM start TO end MINUS 1
                    SET half_str TO string version of half
                    IF length MODULO 2 EQUALS 0
                        SET full TO half_str CONCATENATED WITH half_str REVERSED
                    ELSE
                        SET full TO half_str CONCATENATED WITH half_str FROM SECOND LAST CHARACTER TO FIRST CHARACTER REVERSED
                    YIELD full
        END FUNCTION

        SET k_mirror_numbers TO empty list
        SET length TO 1
        WHILE LENGTH OF k_mirror_numbers LESS THAN n
            FOR pal IN generate_palindromes(length)
                SET num TO integer conversion of pal
                IF is_palindrome(to_base_k(num, k))
                    APPEND num TO k_mirror_numbers
                    IF LENGTH OF k_mirror_numbers EQUALS n
                        BREAK
            INCREMENT length BY 1

        RETURN SUM OF k_mirror_numbers
    END FUNCTION
END CLASS
