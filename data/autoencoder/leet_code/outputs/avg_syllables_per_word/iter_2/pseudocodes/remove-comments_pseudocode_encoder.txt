CLASS Solution
    FUNCTION removeComments(source)
        SET in_block_comment TO False
        SET result TO empty list
        SET current_line TO empty list

        FOR each line IN source
            SET i TO 0
            WHILE i LESS THAN LENGTH OF line
                IF in_block_comment
                    IF SUBSTRING of line FROM i TO i+2 EQUALS "*/"
                        SET in_block_comment TO False
                        INCREMENT i BY 1
                ELSE
                    IF SUBSTRING of line FROM i TO i+2 EQUALS "//"
                        BREAK loop
                    ELSE IF SUBSTRING of line FROM i TO i+2 EQUALS "/*"
                        SET in_block_comment TO True
                        INCREMENT i BY 1
                    ELSE
                        APPEND character line[i] TO current_line
                INCREMENT i BY 1

            IF current_line IS NOT EMPTY AND in_block_comment IS False
                APPEND CONCATENATION of current_line TO result
                SET current_line TO empty list

        RETURN result
    END FUNCTION
END CLASS
