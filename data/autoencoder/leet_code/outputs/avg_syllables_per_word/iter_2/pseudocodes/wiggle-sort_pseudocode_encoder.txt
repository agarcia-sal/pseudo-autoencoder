CLASS Solution
    FUNCTION wiggleSort(nums)
        SET less TO true
        FOR i FROM 0 TO LENGTH OF nums MINUS 2
            IF less
                IF nums[i] GREATER THAN nums[i + 1]
                    SWAP nums[i] AND nums[i + 1]
            ELSE
                IF nums[i] LESS THAN nums[i + 1]
                    SWAP nums[i] AND nums[i + 1]
            SET less TO NOT less
    END FUNCTION
END CLASS
