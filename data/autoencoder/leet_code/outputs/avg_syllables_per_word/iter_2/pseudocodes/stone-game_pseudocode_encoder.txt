CLASS Solution
    FUNCTION stoneGame(piles)
        FUNCTION dfs(i, j)
            IF i > j THEN
                RETURN 0
            END IF
            SET left_choice TO piles[i] MINUS dfs(i + 1, j)
            SET right_choice TO piles[j] MINUS dfs(i, j - 1)
            RETURN MAXIMUM OF left_choice AND right_choice
        END FUNCTION
        RETURN dfs(0, LENGTH OF piles MINUS 1) GREATER THAN 0
    END FUNCTION
END CLASS
