CLASS Solution
    FUNCTION minimumObstacles(grid)
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid AT index 0
        SET directions TO list of (0, 1), (1, 0), (0, -1), (-1, 0)
        SET min_heap TO list containing tuple (0, 0, 0)  # cost, x, y
        SET visited TO set containing tuple (0, 0)
        
        WHILE min_heap IS NOT EMPTY
            REMOVE the smallest element FROM min_heap INTO cost, x, y
            
            IF (x, y) EQUALS (m - 1, n - 1)
                RETURN cost
            
            FOR each dx, dy IN directions
                SET nx TO x PLUS dx
                SET ny TO y PLUS dy
                
                IF nx IS BETWEEN 0 AND m MINUS 1 INCLUSIVE AND ny IS BETWEEN 0 AND n MINUS 1 INCLUSIVE AND (nx, ny) NOT IN visited
                    ADD (nx, ny) TO visited
                    SET new_cost TO cost PLUS grid AT nx AND ny
                    ADD (new_cost, nx, ny) TO min_heap
        
        RETURN -1
END CLASS
