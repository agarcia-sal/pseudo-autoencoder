CLASS Solution
    FUNCTION minInterval(intervals: List of List of Integer, queries: List of Integer) RETURNS List of Integer
        SORT intervals BY start time
        SET sorted_queries TO list of pairs (query, original_index) sorted BY query value
        
        SET min_heap TO empty list
        SET result TO list of -1 repeated LENGTH OF queries times
        
        SET interval_index TO 0
        FOR EACH (query, original_index) IN sorted_queries
            WHILE interval_index LESS THAN LENGTH OF intervals AND intervals[interval_index][0] LESS THAN OR EQUAL TO query
                SET start TO intervals[interval_index][0]
                SET end TO intervals[interval_index][1]
                SET interval_size TO end MINUS start PLUS 1
                ADD (interval_size, end) TO min_heap
                INCREMENT interval_index BY 1
            
            WHILE min_heap IS NOT EMPTY AND min_heap's first element's end LESS THAN query
                REMOVE the top element FROM min_heap
            
            IF min_heap IS NOT EMPTY
                SET result[original_index] TO the first element's interval_size in min_heap
        
        RETURN result
    END FUNCTION
END CLASS
