CLASS Solution
    FUNCTION numMatchingSubseq(s, words)
        SET waiting TO dictionary mapping each unique character in s TO empty queue
        FOR each word IN words
            SET it TO iterator over word
            SET first_letter TO next character from it
            IF first_letter IN waiting THEN
                APPEND it TO waiting[first_letter]
        SET match_count TO 0
        FOR each character c IN s
            IF c IN waiting THEN
                SET current_iterators TO waiting[c]
                SET waiting[c] TO empty queue
                FOR each it IN current_iterators
                    SET next_char TO next character from it OR None if exhausted
                    IF next_char IS None THEN
                        INCREMENT match_count BY 1
                    ELSE
                        IF next_char IN waiting THEN
                            APPEND it TO waiting[next_char]
        RETURN match_count
    END FUNCTION
END CLASS
