CLASS Solution
    FUNCTION countComponents(n, edges)
        SET graph TO dictionary mapping each integer from 0 TO n-1 TO empty list
        FOR each pair (a, b) IN edges
            APPEND b TO graph[a]
            APPEND a TO graph[b]
        SET visited TO empty set

        FUNCTION dfs(node)
            IF node IN visited THEN
                RETURN
            ADD node TO visited
            FOR each neighbor IN graph[node]
                CALL dfs(neighbor)
            END FOR
        END FUNCTION

        SET components TO 0
        FOR node FROM 0 TO n-1
            IF node NOT IN visited THEN
                CALL dfs(node)
                INCREMENT components BY 1
            END IF
        END FOR

        RETURN components
    END FUNCTION
END CLASS
