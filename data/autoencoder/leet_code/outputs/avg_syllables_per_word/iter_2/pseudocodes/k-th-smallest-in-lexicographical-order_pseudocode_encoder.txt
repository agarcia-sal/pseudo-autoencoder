CLASS Solution
    FUNCTION findKthNumber(n, k)
        FUNCTION count_steps(prefix, n)
            SET next_prefix TO prefix + 1
            SET steps TO 0
            WHILE prefix <= n
                SET steps TO steps + MINIMUM OF (n - prefix + 1) AND (next_prefix - prefix)
                SET prefix TO prefix * 10
                SET next_prefix TO next_prefix * 10
            END WHILE
            RETURN steps
        END FUNCTION

        SET current TO 1
        SET k TO k - 1

        WHILE k > 0
            SET steps TO count_steps(current, n)
            IF steps <= k
                SET current TO current + 1
                SET k TO k - steps
            ELSE
                SET current TO current * 10
                SET k TO k - 1
            END IF
        END WHILE

        RETURN current
    END FUNCTION
END CLASS
