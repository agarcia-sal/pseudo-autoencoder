CLASS Solution
    FUNCTION generateMatrix(n)
        SET matrix TO a new n by n matrix filled with zeros
        SET top TO 0
        SET bottom TO n MINUS 1
        SET left TO 0
        SET right TO n MINUS 1
        SET num TO 1
        WHILE top <= bottom AND left <= right
            FOR i FROM left TO right
                SET matrix[top][i] TO num
                INCREMENT num BY 1
            INCREMENT top BY 1
            FOR i FROM top TO bottom
                SET matrix[i][right] TO num
                INCREMENT num BY 1
            DECREMENT right BY 1
            IF top <= bottom THEN
                FOR i FROM right DOWN TO left
                    SET matrix[bottom][i] TO num
                    INCREMENT num BY 1
                DECREMENT bottom BY 1
            IF left <= right THEN
                FOR i FROM bottom DOWN TO top
                    SET matrix[i][left] TO num
                    INCREMENT num BY 1
                INCREMENT left BY 1
        RETURN matrix
    END FUNCTION
END CLASS
