CLASS Solution
    FUNCTION canCross(stones)
        IF stones IS EMPTY OR LENGTH OF stones < 2 THEN
            RETURN False
        END IF

        SET stone_set TO SET OF stones
        SET memo TO EMPTY MAP

        FUNCTION can_jump_to(position, jump)
            IF position EQUALS LAST ELEMENT OF stones THEN
                RETURN True
            END IF

            IF (position, jump) EXISTS IN memo THEN
                RETURN memo[(position, jump)]
            END IF

            FOR next_jump FROM jump - 1 TO jump + 1
                IF next_jump > 0 AND position + next_jump IN stone_set THEN
                    IF can_jump_to(position + next_jump, next_jump) THEN
                        SET memo[(position, jump)] TO True
                        RETURN True
                    END IF
                END IF
            END FOR

            SET memo[(position, jump)] TO False
            RETURN False
        END FUNCTION

        RETURN can_jump_to(stones[0], 0)
    END FUNCTION
END CLASS
