CLASS Solution
    FUNCTION minMeetingRooms(intervals)
        IF intervals IS EMPTY
            RETURN 0
        END IF

        SORT intervals BY start time

        SET min_heap TO empty list

        FOR each meeting IN intervals
            IF min_heap IS NOT EMPTY AND meeting start time IS GREATER THAN OR EQUAL TO min_heap minimum value
                REMOVE min_heap minimum value
            END IF

            ADD meeting end time TO min_heap
        END FOR

        RETURN LENGTH OF min_heap
    END FUNCTION
END CLASS
