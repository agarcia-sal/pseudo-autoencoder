CLASS Solution
    FUNCTION longestAwesome(s)
        SET prefix_map TO dictionary with key 0 mapped to -1
        SET prefix_xor TO 0
        SET max_length TO 0

        FOR i, char IN enumeration of s
            SET digit TO integer value of char
            TOGGLE bit digit in prefix_xor by XOR with (1 shifted left by digit)

            IF prefix_xor EXISTS in prefix_map
                SET max_length TO maximum of max_length AND difference of i AND prefix_map[prefix_xor]

            FOR j FROM 0 TO 9
                SET flipped TO prefix_xor XOR (1 shifted left by j)
                IF flipped EXISTS in prefix_map
                    SET max_length TO maximum of max_length AND difference of i AND prefix_map[flipped]

            IF prefix_xor NOT EXISTS in prefix_map
                SET prefix_map[prefix_xor] TO i

        RETURN max_length
    END FUNCTION
END CLASS
