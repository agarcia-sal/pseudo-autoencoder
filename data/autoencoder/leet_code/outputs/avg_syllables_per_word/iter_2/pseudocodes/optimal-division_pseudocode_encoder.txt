CLASS Solution
    FUNCTION optimalDivision(nums)
        IF LENGTH OF nums EQUALS 1
            RETURN STRING OF nums[0]
        IF LENGTH OF nums EQUALS 2
            RETURN STRING OF nums[0] CONCATENATED WITH "/" CONCATENATED WITH STRING OF nums[1]
        RETURN STRING OF nums[0] CONCATENATED WITH "/(" CONCATENATED WITH JOIN STRINGS OF nums FROM INDEX 1 TO END USING "/" CONCATENATED WITH ")" 
    END FUNCTION
END CLASS
