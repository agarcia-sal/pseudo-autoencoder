CLASS Solution
    FUNCTION findMaxValueOfEquation(points: List[List[int]], k: int) RETURNS int
        SET max_value TO negative infinity
        CREATE empty deque
        
        FOR each (xj, yj) IN points
            WHILE deque is not empty AND xj MINUS first element's second value GREATER THAN k
                REMOVE first element from deque
            
            IF deque is not empty
                SET max_value TO maximum of max_value AND (yj PLUS xj PLUS first element's first value)
            
            WHILE deque is not empty AND (yj MINUS xj) GREATER THAN OR EQUAL TO last element's first value
                REMOVE last element from deque
            
            ADD (yj MINUS xj, xj) to the back of deque
        
        RETURN max_value
    END FUNCTION
END CLASS
