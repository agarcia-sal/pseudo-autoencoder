CLASS Solution
    FUNCTION frogPosition(n, edges, t, target)
        IF n EQUALS 1 THEN
            RETURN 1.0
        END IF

        SET graph TO empty map with default as empty list
        FOR each pair u, v IN edges DO
            APPEND v TO graph[u]
            APPEND u TO graph[v]
        END FOR

        SET queue TO new deque containing tuple (1, 1.0, 0)
        SET visited TO set containing 1

        WHILE queue is not empty DO
            SET current_node, current_probability, current_time TO front element FROM queue
            REMOVE front element FROM queue

            IF current_time EQUALS t OR (current_node NOT EQUAL 1 AND LENGTH OF graph[current_node] EQUALS 1) THEN
                IF current_node EQUALS target THEN
                    RETURN current_probability
                ELSE
                    CONTINUE
                END IF
            END IF

            SET possible_moves TO LENGTH OF graph[current_node]
            IF current_node NOT EQUAL 1 THEN
                DECREMENT possible_moves BY 1
            END IF

            FOR each neighbor IN graph[current_node] DO
                IF neighbor NOT IN visited THEN
                    ADD neighbor TO visited
                    APPEND (neighbor, current_probability DIVIDED BY possible_moves, current_time PLUS 1) TO queue
                END IF
            END FOR
        END WHILE

        RETURN 0.0
    END FUNCTION
END CLASS
