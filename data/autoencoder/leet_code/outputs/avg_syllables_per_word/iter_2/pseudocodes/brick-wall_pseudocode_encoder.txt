CLASS Solution
    FUNCTION leastBricks(wall)
        SET gap_counts TO empty map with default 0

        FOR each row IN wall
            SET total TO 0
            FOR each brick IN all bricks of row except the last one
                INCREMENT total BY brick
                INCREMENT gap_counts[total] BY 1

        SET max_gaps TO maximum value in gap_counts, or 0 if empty
        RETURN length of wall MINUS max_gaps
    END FUNCTION
END CLASS
