CLASS Solution
    FUNCTION matrixScore(grid)
        FOR each row IN grid
            IF row[0] EQUALS 0
                FOR i FROM 0 TO LENGTH OF row MINUS 1
                    SET row[i] TO 1 MINUS row[i]
        FOR col FROM 1 TO LENGTH OF grid[0] MINUS 1
            SET count_ones TO 0
            FOR row_index FROM 0 TO LENGTH OF grid MINUS 1
                INCREMENT count_ones BY grid[row_index][col]
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY 2
                FOR row_index FROM 0 TO LENGTH OF grid MINUS 1
                    SET grid[row_index][col] TO 1 MINUS grid[row_index][col]
        SET score TO 0
        FOR each row IN grid
            SET binary_number TO concatenate elements of row AS strings
            INCREMENT score BY integer value of binary_number IN base 2
        RETURN score
    END FUNCTION
END CLASS
