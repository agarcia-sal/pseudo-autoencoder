CLASS Solution
    FUNCTION findCheapestPrice(n, flights, src, dst, k)
        SET graph TO dictionary mapping each city index from 0 to n-1 TO empty list
        FOR each u, v, w IN flights
            APPEND tuple (v, w) TO graph[u]

        SET min_heap TO list containing tuple (0, -1, src)
        SET min_cost TO dictionary with key src and value -1

        WHILE min_heap is not empty
            EXTRACT smallest element (cost, stops, city) FROM min_heap

            IF city EQUALS dst
                RETURN cost

            IF stops IS GREATER THAN OR EQUAL TO k
                CONTINUE to next iteration

            FOR each neighbor, price IN graph[city]
                SET next_cost TO cost PLUS price
                SET next_stops TO stops PLUS 1

                IF neighbor NOT IN min_cost OR next_stops LESS THAN min_cost[neighbor] OR next_cost LESS THAN min_heap smallest cost
                    SET min_cost[neighbor] TO next_stops
                    ADD tuple (next_cost, next_stops, neighbor) TO min_heap

        RETURN -1
END CLASS
