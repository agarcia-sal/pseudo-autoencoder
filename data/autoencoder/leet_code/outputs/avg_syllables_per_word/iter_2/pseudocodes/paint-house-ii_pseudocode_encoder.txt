CLASS Solution
    FUNCTION minCostII(costs)
        IF costs IS EMPTY
            RETURN 0
        SET n TO LENGTH OF costs
        SET k TO LENGTH OF costs[0]
        IF k IS 1
            IF n IS 1
                RETURN costs[0][0]
            ELSE
                RETURN infinity
        FOR i FROM 1 TO n - 1
            SET min_cost TO infinity
            SET min_cost_index TO -1
            SET second_min_cost TO infinity
            FOR j FROM 0 TO k - 1
                IF costs[i - 1][j] < min_cost
                    SET second_min_cost TO min_cost
                    SET min_cost TO costs[i - 1][j]
                    SET min_cost_index TO j
                ELSE IF costs[i - 1][j] < second_min_cost
                    SET second_min_cost TO costs[i - 1][j]
            FOR j FROM 0 TO k - 1
                IF j IS min_cost_index
                    INCREMENT costs[i][j] BY second_min_cost
                ELSE
                    INCREMENT costs[i][j] BY min_cost
        RETURN MINIMUM VALUE IN costs[n - 1]
    END FUNCTION
END CLASS
