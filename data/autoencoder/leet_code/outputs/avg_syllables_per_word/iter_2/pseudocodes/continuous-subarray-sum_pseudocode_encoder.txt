CLASS Solution
    FUNCTION checkSubarraySum(nums, k)
        SET remainder_map TO dictionary with key 0 and value -1
        SET current_sum TO 0
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            SET num TO nums[i]
            INCREMENT current_sum BY num
            IF k IS NOT 0
                SET current_sum TO current_sum MODULO k
            END IF
            IF current_sum IS IN remainder_map
                IF i MINUS remainder_map[current_sum] GREATER THAN 1
                    RETURN True
                END IF
            ELSE
                SET remainder_map[current_sum] TO i
            END IF
        END FOR
        RETURN False
    END FUNCTION
END CLASS
