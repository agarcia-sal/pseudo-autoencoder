CLASS Solution
    FUNCTION triangleNumber(nums)
        SORT nums
        SET count TO 0
        SET n TO LENGTH OF nums
        
        FOR k FROM n - 1 DOWN TO 2
            SET i TO 0
            SET j TO k - 1
            WHILE i < j
                IF nums[i] + nums[j] > nums[k] THEN
                    INCREMENT count BY j - i
                    DECREMENT j BY 1
                ELSE
                    INCREMENT i BY 1
        RETURN count
    END FUNCTION
END CLASS
