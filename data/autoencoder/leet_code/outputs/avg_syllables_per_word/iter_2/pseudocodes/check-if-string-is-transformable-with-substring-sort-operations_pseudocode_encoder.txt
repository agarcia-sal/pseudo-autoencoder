CLASS Solution
    FUNCTION isTransformable(s, t)
        SET pos TO a default dictionary of empty lists
        FOR each index i and character c IN s
            ADD i TO the heap corresponding to integer value of c in pos
        FOR each index i and character c IN t
            SET digit TO integer value of c
            IF the heap for digit in pos is empty
                RETURN False
            FOR each smaller FROM 0 TO digit MINUS 1
                IF heap for smaller in pos is not empty AND smallest element in that heap LESS THAN smallest element in heap for digit
                    RETURN False
            REMOVE the smallest element from heap for digit in pos
        RETURN True
    END FUNCTION
END CLASS
