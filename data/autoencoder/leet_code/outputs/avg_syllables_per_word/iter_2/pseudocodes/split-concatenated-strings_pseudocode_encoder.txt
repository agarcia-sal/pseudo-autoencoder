CLASS Solution
    FUNCTION splitLoopedString(strs)
        FOR i FROM 0 TO LENGTH OF strs MINUS 1
            SET reversed_str TO REVERSE OF strs[i]
            IF reversed_str IS LEXICOGRAPHICALLY GREATER THAN strs[i]
                SET strs[i] TO reversed_str
        SET doubled_strs TO CONCATENATION OF strs WITH strs
        SET max_string TO EMPTY STRING
        FOR i FROM 0 TO LENGTH OF strs MINUS 1
            FOR is_reversed IN [FALSE, TRUE]
                IF is_reversed IS TRUE
                    SET current TO REVERSE OF strs[i]
                ELSE
                    SET current TO strs[i]
                SET remaining TO CONCATENATION OF doubled_strs FROM INDEX i+1 TO i+LENGTH OF strs MINUS 1
                FOR j FROM 0 TO LENGTH OF current
                    SET candidate TO CONCATENATION OF current FROM INDEX j TO END, remaining, AND current FROM START TO INDEX j MINUS 1
                    IF candidate IS LEXICOGRAPHICALLY GREATER THAN max_string
                        SET max_string TO candidate
        RETURN max_string
    END FUNCTION
END CLASS
