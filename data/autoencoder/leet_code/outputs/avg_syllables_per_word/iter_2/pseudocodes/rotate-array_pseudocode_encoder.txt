CLASS Solution
    FUNCTION rotate(nums, k)
        SET n TO LENGTH OF nums
        SET k TO k MODULO n

        FUNCTION reverse(start, end)
            WHILE start < end
                SET temp TO nums[start]
                SET nums[start] TO nums[end]
                SET nums[end] TO temp
                INCREMENT start BY 1
                DECREMENT end BY 1
            END WHILE
        END FUNCTION

        CALL reverse(0, n - 1)
        CALL reverse(0, k - 1)
        CALL reverse(k, n - 1)
    END FUNCTION
END CLASS
