CLASS Solution
    FUNCTION maxChunksToSorted(arr)
        SET max_so_far TO empty list
        SET current_max TO first element of arr
        FOR each num IN arr
            IF num GREATER THAN current_max THEN
                SET current_max TO num
            END IF
            APPEND current_max TO max_so_far
        END FOR

        SET sorted_arr TO arr sorted in ascending order
        SET chunks TO 0
        SET max_chunk TO 0

        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] GREATER THAN max_chunk THEN
                SET max_chunk TO arr[i]
            END IF
            IF max_chunk EQUALS sorted_arr[i] THEN
                INCREMENT chunks BY 1
            END IF
        END FOR

        RETURN chunks
    END FUNCTION
END CLASS
