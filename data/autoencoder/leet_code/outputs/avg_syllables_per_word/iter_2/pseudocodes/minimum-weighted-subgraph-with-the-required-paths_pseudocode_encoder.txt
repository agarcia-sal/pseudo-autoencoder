CLASS Solution
    FUNCTION minimumWeight(n, edges, src1, src2, dest)
        FUNCTION dijkstra(graph, start)
            SET distances TO a list of size n with all values as infinity
            SET distances[start] TO 0
            SET heap TO a list containing tuple (0, start)
            
            WHILE heap is not empty
                SET current_distance, current_node TO removal of smallest item from heap
                
                IF current_distance is greater than distances[current_node]
                    CONTINUE to next iteration
                
                FOR each neighbor, weight IN graph[current_node]
                    SET distance TO current_distance plus weight
                    
                    IF distance is less than distances[neighbor]
                        SET distances[neighbor] TO distance
                        ADD tuple (distance, neighbor) TO heap
            END WHILE
            
            RETURN distances
        END FUNCTION
        
        SET graph TO a list of n empty lists
        SET reverse_graph TO a list of n empty lists
        
        FOR each u, v, w IN edges
            APPEND tuple (v, w) TO graph[u]
            APPEND tuple (u, w) TO reverse_graph[v]
        
        SET dist_from_src1 TO result of dijkstra(graph, src1)
        SET dist_from_src2 TO result of dijkstra(graph, src2)
        SET dist_to_dest TO result of dijkstra(reverse_graph, dest)
        
        SET min_weight TO infinity
        
        FOR i FROM 0 TO n minus 1
            IF dist_from_src1[i] is not infinity AND dist_from_src2[i] is not infinity AND dist_to_dest[i] is not infinity
                SET min_weight TO the smaller of min_weight and sum of dist_from_src1[i], dist_from_src2[i], and dist_to_dest[i]
        
        IF min_weight is infinity
            RETURN -1
        ELSE
            RETURN min_weight
    END FUNCTION
END CLASS
