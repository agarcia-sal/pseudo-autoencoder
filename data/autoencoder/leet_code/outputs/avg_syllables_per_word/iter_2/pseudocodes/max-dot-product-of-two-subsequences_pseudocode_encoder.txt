CLASS Solution
    FUNCTION maxDotProduct(nums1, nums2)
        SET m TO LENGTH OF nums1
        SET n TO LENGTH OF nums2
        SET dp TO 2D LIST of size (m + 1) by (n + 1) with all values SET TO negative infinity

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                SET current_product TO nums1 AT (i - 1) MULTIPLIED BY nums2 AT (j - 1)
                SET dp[i][j] TO MAXIMUM OF
                    current_product,
                    dp[i - 1][j],
                    dp[i][j - 1],
                    dp[i - 1][j - 1] PLUS current_product
        RETURN dp[m][n]
    END FUNCTION
END CLASS
