CLASS Solution
    FUNCTION strangePrinter(s)
        FUNCTION dp(i, j, memo)
            IF i > j THEN
                RETURN 0
            IF (i, j) IN memo THEN
                RETURN memo[(i, j)]
            SET result TO dp(i, j - 1, memo) + 1
            FOR k FROM i TO j - 1
                IF s[k] EQUALS s[j] THEN
                    SET result TO MINIMUM OF result AND dp(i, k - 1, memo) + dp(k, j - 1, memo)
            SET memo[(i, j)] TO result
            RETURN result
        END FUNCTION
        RETURN dp(0, LENGTH OF s MINUS 1, EMPTY_DICTIONARY)
    END FUNCTION
END CLASS
