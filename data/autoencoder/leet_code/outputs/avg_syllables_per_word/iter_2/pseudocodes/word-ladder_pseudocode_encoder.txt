CLASS Solution
    FUNCTION ladderLength(beginWord, endWord, wordList)
        IF endWord NOT IN wordList THEN
            RETURN 0
        END IF

        SET word_dict TO empty dictionary mapping to list of words
        SET word_length TO LENGTH OF beginWord

        FOR each word IN wordList
            FOR i FROM 0 TO word_length MINUS 1
                SET intermediate_word TO CONCATENATION OF SUBSTRING OF word FROM START TO i, "*" , AND SUBSTRING OF word FROM i+1 TO END
                APPEND word TO word_dict[intermediate_word]
            END FOR
        END FOR

        SET queue TO a double-ended queue with element (beginWord, 1)
        SET visited TO a set containing beginWord

        WHILE queue IS NOT EMPTY
            SET current_word, current_length TO REMOVE FRONT ELEMENT FROM queue

            FOR i FROM 0 TO word_length MINUS 1
                SET intermediate_word TO CONCATENATION OF SUBSTRING OF current_word FROM START TO i, "*", AND SUBSTRING OF current_word FROM i+1 TO END

                FOR each word IN word_dict[intermediate_word]
                    IF word EQUALS endWord THEN
                        RETURN current_length PLUS 1
                    END IF

                    IF word NOT IN visited THEN
                        ADD word TO visited
                        ADD (word, current_length PLUS 1) TO queue
                    END IF
                END FOR
            END FOR
        END WHILE

        RETURN 0
    END FUNCTION
END CLASS
