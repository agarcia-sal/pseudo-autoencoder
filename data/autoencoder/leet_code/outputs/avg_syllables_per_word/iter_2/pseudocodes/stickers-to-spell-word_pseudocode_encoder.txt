CLASS Solution
    FUNCTION minStickers(stickers: List of String, target: String) RETURNS Integer
        SET stickers TO list of frequency counters for each sticker
        SORT stickers by descending order of their maximum character frequency
        
        INITIALIZE queue WITH tuple (target, 0)
        INITIALIZE memo WITH key target and value 0
        
        WHILE queue is not empty
            SET current_target, used_stickers TO dequeue from queue
            
            FOR each sticker IN stickers
                IF first character of current_target NOT IN sticker
                    CONTINUE to next sticker
                
                SET remaining TO frequency counter of current_target
                SUBTRACT frequencies of sticker FROM remaining
                
                SET new_target TO string formed by sorting elements of remaining
                
                IF new_target is empty
                    RETURN used_stickers + 1
                
                IF new_target NOT IN memo
                    SET memo[new_target] TO used_stickers + 1
                    ENQUEUE (new_target, used_stickers + 1) INTO queue
        
        RETURN -1
    END FUNCTION
END CLASS
