CLASS Solution
    FUNCTION minDistance(word1, word2)
        SET m TO LENGTH OF word1
        SET n TO LENGTH OF word2

        CREATE dp AS TABLE WITH DIMENSIONS (m + 1) BY (n + 1) FILLED WITH 0

        FOR i FROM 0 TO m
            SET dp[i][0] TO i
        FOR j FROM 0 TO n
            SET dp[0][j] TO j

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                IF CHARACTER AT word1 AT POSITION (i - 1) EQUALS CHARACTER AT word2 AT POSITION (j - 1)
                    SET dp[i][j] TO dp[i - 1][j - 1]
                ELSE
                    SET dp[i][j] TO 1 PLUS MINIMUM OF dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1]

        RETURN dp[m][n]
    END FUNCTION
END CLASS
