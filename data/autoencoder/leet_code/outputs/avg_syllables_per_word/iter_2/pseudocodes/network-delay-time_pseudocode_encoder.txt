CLASS Solution
    FUNCTION networkDelayTime(times, n, k)
        SET graph TO dictionary mapping each integer from 1 to n TO empty list
        FOR each u, v, w IN times
            APPEND (v, w) TO graph[u]
        SET dist TO dictionary mapping each integer from 1 to n TO infinity
        SET dist[k] TO 0
        SET min_heap TO list containing tuple (0, k)
        WHILE min_heap is not empty
            SET current_dist, node TO remove smallest element from min_heap
            IF current_dist > dist[node]
                CONTINUE
            FOR each neighbor, weight IN graph[node]
                SET distance TO current_dist + weight
                IF distance < dist[neighbor]
                    SET dist[neighbor] TO distance
                    ADD (distance, neighbor) TO min_heap
        SET max_time TO maximum value in dist
        IF max_time EQUALS infinity
            RETURN -1
        ELSE
            RETURN max_time
    END FUNCTION
END CLASS
