CLASS Solution
    FUNCTION findMaximizedCapital(k, w, profits, capital)
        SET projects TO list of tuples pairing each capital with corresponding profit
        SORT projects by capital

        SET max_heap TO empty list
        SET i TO 0
        SET n TO length of projects

        FOR each iteration FROM 1 TO k
            WHILE i < n AND projects[i]'s capital <= w
                ADD negative of projects[i]'s profit TO max_heap
                INCREMENT i BY 1

            IF max_heap is empty
                BREAK loop

            REMOVE largest profit FROM max_heap (by popping smallest negative)
            INCREMENT w BY that profit

        RETURN w
    END FUNCTION
END CLASS
