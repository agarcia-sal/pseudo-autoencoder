CLASS Solution
    FUNCTION integerBreak(n)
        SET dp TO list of zeros with length n plus 1
        SET dp[1] TO 1
        
        FOR i FROM 2 TO n
            SET max_product TO 0
            FOR j FROM 1 TO i minus 1
                SET product_option1 TO j multiplied by dp[i minus j]
                SET product_option2 TO j multiplied by (i minus j)
                SET max_product TO maximum of max_product, product_option1, product_option2
            SET dp[i] TO max_product
        
        RETURN dp[n]
    END FUNCTION
END CLASS
