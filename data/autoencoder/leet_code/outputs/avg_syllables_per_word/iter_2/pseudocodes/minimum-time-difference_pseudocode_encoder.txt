CLASS Solution
    FUNCTION findMinDifference(timePoints: List of String) RETURNS Integer
        FUNCTION to_minutes(time: String) RETURNS Integer
            SET hours, minutes TO SPLIT time BY ':' CONVERTED TO INTEGERS
            RETURN hours MULTIPLIED BY 60 PLUS minutes
        END FUNCTION

        SET minutes_list TO APPLY to_minutes TO EACH item IN timePoints
        SORT minutes_list

        SET min_diff TO infinity
        FOR i FROM 1 TO LENGTH OF minutes_list MINUS 1
            SET current_diff TO minutes_list[i] MINUS minutes_list[i - 1]
            IF current_diff LESS THAN min_diff THEN
                SET min_diff TO current_diff
            END IF
        END FOR

        SET circular_diff TO (1440 MINUS minutes_list[LENGTH OF minutes_list MINUS 1]) PLUS minutes_list[0]
        IF circular_diff LESS THAN min_diff THEN
            SET min_diff TO circular_diff
        END IF

        RETURN min_diff
    END FUNCTION
END CLASS
