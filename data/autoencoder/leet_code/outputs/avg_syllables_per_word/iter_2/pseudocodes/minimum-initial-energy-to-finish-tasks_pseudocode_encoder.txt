CLASS Solution
    FUNCTION minimumEffort(tasks)
        SORT tasks BY (minimum energy required MINUS actual energy spent) IN DESCENDING ORDER
        SET current_energy TO 0
        SET required_initial_energy TO 0
        FOR EACH actual, minimum IN tasks
            IF current_energy IS LESS THAN minimum THEN
                INCREMENT required_initial_energy BY (minimum MINUS current_energy)
                SET current_energy TO minimum
            END IF
            DECREMENT current_energy BY actual
        END FOR
        RETURN required_initial_energy
    END FUNCTION
END CLASS
