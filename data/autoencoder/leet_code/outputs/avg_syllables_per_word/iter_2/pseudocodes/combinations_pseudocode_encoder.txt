CLASS Solution
    FUNCTION combine(n, k)
        FUNCTION backtrack(start, path)
            IF LENGTH OF path EQUALS k
                APPEND COPY OF path TO result
                RETURN
            END IF
            FOR i FROM start TO n
                APPEND i TO path
                CALL backtrack(i + 1, path)
                REMOVE LAST ELEMENT FROM path
            END FOR
        END FUNCTION

        SET result TO EMPTY LIST
        CALL backtrack(1, EMPTY LIST)
        RETURN result
    END FUNCTION
END CLASS
