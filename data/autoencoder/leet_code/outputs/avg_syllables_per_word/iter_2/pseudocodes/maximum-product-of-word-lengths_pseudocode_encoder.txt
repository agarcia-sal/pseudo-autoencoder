CLASS Solution
    FUNCTION maxProduct(words: List of String) RETURNS Integer
        SET word_sets TO empty list
        FOR each word IN words
            APPEND set of characters in word TO word_sets
        SET max_product TO 0
        FOR i FROM 0 TO LENGTH OF words MINUS 1
            FOR j FROM i PLUS 1 TO LENGTH OF words MINUS 1
                IF intersection of word_sets[i] AND word_sets[j] IS empty
                    SET product TO LENGTH OF words[i] MULTIPLIED BY LENGTH OF words[j]
                    IF product GREATER THAN max_product
                        SET max_product TO product
        RETURN max_product
    END FUNCTION
END CLASS
