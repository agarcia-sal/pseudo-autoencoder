CLASS Solution
    FUNCTION countAndSay(n)
        IF n EQUALS 1
            RETURN "1"
        END IF

        SET current_sequence TO "1"

        FOR _ FROM 1 TO n MINUS 1
            SET next_sequence TO empty string
            SET count TO 1
            SET previous_char TO first character OF current_sequence

            FOR i FROM 1 TO LENGTH OF current_sequence MINUS 1
                IF character AT position i OF current_sequence EQUALS previous_char
                    INCREMENT count BY 1
                ELSE
                    APPEND string representation OF count TO next_sequence
                    APPEND previous_char TO next_sequence
                    SET previous_char TO character AT position i OF current_sequence
                    SET count TO 1
                END IF
            END FOR

            APPEND string representation OF count TO next_sequence
            APPEND previous_char TO next_sequence

            SET current_sequence TO next_sequence
        END FOR

        RETURN current_sequence
    END FUNCTION
END CLASS
