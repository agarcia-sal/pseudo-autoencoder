CLASS Solution
    FUNCTION smallestGoodBase(n: string) RETURNS string
        SET n TO integer value of n
        SET max_length TO integer part of log base 2 of n PLUS 1
        
        FOR m FROM max_length DOWN TO 2
            SET k TO integer part of n raised to power (1 DIVIDED BY (m MINUS 1))
            IF k LESS THAN 2
                CONTINUE TO next iteration
            
            SET num TO (k raised to power m MINUS 1) DIVIDED BY (k MINUS 1)
            
            IF num EQUALS n
                RETURN string representation of k
        
        RETURN string representation of (n MINUS 1)
    END FUNCTION
END CLASS
