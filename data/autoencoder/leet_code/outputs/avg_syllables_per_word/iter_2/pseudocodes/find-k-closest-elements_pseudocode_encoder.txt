CLASS Solution
    FUNCTION findClosestElements(arr, k, x)
        SET left TO POSITION WHERE x WOULD BE INSERTED IN arr MINUS 1
        SET right TO left PLUS 1
        WHILE right MINUS left MINUS 1 IS LESS THAN k
            IF left IS -1 OR (right IS LESS THAN LENGTH OF arr AND ABSOLUTE VALUE OF arr[right] MINUS x IS LESS THAN ABSOLUTE VALUE OF arr[left] MINUS x)
                INCREMENT right BY 1
            ELSE
                DECREMENT left BY 1
        RETURN SUBLIST OF arr FROM left PLUS 1 TO right
    END FUNCTION
END CLASS
