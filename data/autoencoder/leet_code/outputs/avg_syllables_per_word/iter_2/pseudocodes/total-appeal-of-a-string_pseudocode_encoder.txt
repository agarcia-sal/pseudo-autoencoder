CLASS Solution
    FUNCTION appealSum(s)
        SET last TO list of length 26 filled with -1
        SET total_appeal TO 0
        SET current_appeal TO 0
        FOR each index i AND character char IN s
            SET index TO position of char in alphabet (0 for 'a', ..., 25 for 'z')
            INCREMENT current_appeal BY (i MINUS last[index])
            INCREMENT total_appeal BY current_appeal
            SET last[index] TO i
        RETURN total_appeal
    END FUNCTION
END CLASS
