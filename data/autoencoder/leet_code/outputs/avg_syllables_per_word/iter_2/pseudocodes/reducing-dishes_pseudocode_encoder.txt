CLASS Solution
    FUNCTION maxSatisfaction(satisfaction: List of Integer) RETURNS Integer
        SORT satisfaction IN DESCENDING ORDER
        SET max_satisfaction TO 0
        SET current_satisfaction TO 0
        SET total_satisfaction TO 0
        FOR each s IN satisfaction
            INCREMENT current_satisfaction BY s
            INCREMENT total_satisfaction BY current_satisfaction
            IF total_satisfaction GREATER THAN max_satisfaction THEN
                SET max_satisfaction TO total_satisfaction
        RETURN max_satisfaction
    END FUNCTION
END CLASS
