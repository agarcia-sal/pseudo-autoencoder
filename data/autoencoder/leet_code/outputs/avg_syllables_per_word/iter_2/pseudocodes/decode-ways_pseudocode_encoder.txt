CLASS Solution
    FUNCTION numDecodings(s: string) RETURNS int
        IF s IS EMPTY OR s[0] IS '0' THEN
            RETURN 0
        END IF

        SET dp TO LIST OF ZEROS WITH LENGTH LENGTH OF s PLUS 1
        SET dp[0] TO 1
        SET dp[1] TO 1

        FOR i FROM 2 TO LENGTH OF s PLUS 1 DO
            IF s[i - 1] IS NOT '0' THEN
                INCREMENT dp[i] BY dp[i - 1]
            END IF

            SET two_digit TO INTEGER VALUE OF s FROM INDEX i - 2 TO i
            IF two_digit IS GREATER THAN OR EQUAL TO 10 AND two_digit IS LESS THAN OR EQUAL TO 26 THEN
                INCREMENT dp[i] BY dp[i - 2]
            END IF
        END FOR

        RETURN dp[LENGTH OF dp MINUS 1]
    END FUNCTION
END CLASS
