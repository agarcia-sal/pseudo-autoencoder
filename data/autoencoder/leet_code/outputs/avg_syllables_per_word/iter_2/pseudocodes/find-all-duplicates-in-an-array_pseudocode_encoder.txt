CLASS Solution
    FUNCTION findDuplicates(nums)
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            WHILE nums[i] IS NOT EQUAL TO nums[nums[i] MINUS 1]
                SET temp TO nums[i]
                SET nums[i] TO nums[temp MINUS 1]
                SET nums[temp MINUS 1] TO temp
        SET result TO EMPTY LIST
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            IF nums[i] IS NOT EQUAL TO i PLUS 1
                APPEND nums[i] TO result
        RETURN result
    END FUNCTION
END CLASS
