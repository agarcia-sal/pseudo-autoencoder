CLASS Solution
    FUNCTION minArea(image: List of List of String, x: Integer, y: Integer) RETURNS Integer
        IF image IS EMPTY OR image[0] IS EMPTY THEN
            RETURN 0
        END IF

        SET m TO LENGTH OF image
        SET n TO LENGTH OF image[0]

        FUNCTION hasBlackPixelInRow(row: Integer) RETURNS Boolean
            FOR EACH pixel IN image[row]
                IF pixel EQUALS '1' THEN
                    RETURN True
                END IF
            END FOR
            RETURN False
        END FUNCTION

        FUNCTION hasBlackPixelInColumn(col: Integer) RETURNS Boolean
            FOR row FROM 0 TO m - 1
                IF image[row][col] EQUALS '1' THEN
                    RETURN True
                END IF
            END FOR
            RETURN False
        END FUNCTION

        FUNCTION findTop() RETURNS Integer
            SET low TO 0
            SET high TO x
            WHILE low < high
                SET mid TO (low + high) DIVIDED BY 2
                IF hasBlackPixelInRow(mid) THEN
                    SET high TO mid
                ELSE
                    SET low TO mid + 1
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findBottom() RETURNS Integer
            SET low TO x
            SET high TO m - 1
            WHILE low < high
                SET mid TO (low + high + 1) DIVIDED BY 2
                IF hasBlackPixelInRow(mid) THEN
                    SET low TO mid
                ELSE
                    SET high TO mid - 1
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findLeft() RETURNS Integer
            SET low TO 0
            SET high TO y
            WHILE low < high
                SET mid TO (low + high) DIVIDED BY 2
                IF hasBlackPixelInColumn(mid) THEN
                    SET high TO mid
                ELSE
                    SET low TO mid + 1
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findRight() RETURNS Integer
            SET low TO y
            SET high TO n - 1
            WHILE low < high
                SET mid TO (low + high + 1) DIVIDED BY 2
                IF hasBlackPixelInColumn(mid) THEN
                    SET low TO mid
                ELSE
                    SET high TO mid - 1
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        SET top TO findTop()
        SET bottom TO findBottom()
        SET left TO findLeft()
        SET right TO findRight()

        RETURN (bottom - top + 1) TIMES (right - left + 1)
    END FUNCTION
END CLASS
