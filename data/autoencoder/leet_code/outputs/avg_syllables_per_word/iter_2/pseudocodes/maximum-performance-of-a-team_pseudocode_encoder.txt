CLASS Solution
    FUNCTION maxPerformance(n, speed, efficiency, k)
        SET engineers TO list of pairs (efficiency, speed)
        SORT engineers BY efficiency in descending order
        SET speed_heap TO empty min-heap
        SET speed_sum TO 0
        SET max_performance TO 0
        FOR each (eff, spd) IN engineers
            ADD spd TO speed_heap
            INCREMENT speed_sum BY spd
            IF size of speed_heap > k THEN
                REMOVE smallest speed FROM speed_heap AND SUBTRACT it FROM speed_sum
            SET current_performance TO speed_sum MULTIPLIED BY eff
            SET max_performance TO maximum of max_performance AND current_performance
        RETURN max_performance MODULO (10^9 + 7)
    END FUNCTION
END CLASS
