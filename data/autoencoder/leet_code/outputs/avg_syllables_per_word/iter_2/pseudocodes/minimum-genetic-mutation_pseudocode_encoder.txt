CLASS Solution
    FUNCTION minMutation(start, end, bank)
        SET s TO SET OF bank
        SET q TO DEQUE CONTAINING TUPLE (start, 0)
        SET mp TO DICTIONARY MAPPING 'A' TO 'TCG', 'T' TO 'ACG', 'C' TO 'ATG', 'G' TO 'ATC'
        WHILE q IS NOT EMPTY
            SET t, step TO REMOVE FIRST ELEMENT FROM q
            IF t EQUALS end
                RETURN step
            FOR i, v IN ENUMERATE CHARACTERS OF t
                FOR j IN mp[v]
                    SET next TO CONCATENATION OF SUBSTRING OF t FROM START TO i, j, AND SUBSTRING OF t FROM i+1 TO END
                    IF next IS IN s
                        APPEND (next, step + 1) TO q
                        REMOVE next FROM s
        RETURN -1
END CLASS
