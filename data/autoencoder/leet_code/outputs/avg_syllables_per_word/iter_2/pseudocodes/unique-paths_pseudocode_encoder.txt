CLASS Solution
    FUNCTION uniquePaths(m, n)
        SET dp TO 2D array of size m by n with all elements set to 1
        FOR i FROM 1 TO m MINUS 1
            FOR j FROM 1 TO n MINUS 1
                SET dp[i][j] TO dp[i MINUS 1][j] PLUS dp[i][j MINUS 1]
        RETURN dp[m MINUS 1][n MINUS 1]
    END FUNCTION
END CLASS
