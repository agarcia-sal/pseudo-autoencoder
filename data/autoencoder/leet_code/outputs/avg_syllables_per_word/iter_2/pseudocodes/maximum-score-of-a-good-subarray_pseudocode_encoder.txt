CLASS Solution
    FUNCTION maximumScore(nums: List[int], k: int) RETURNS int
        SET left TO k
        SET right TO k
        SET current_min TO nums[k]
        SET max_score TO current_min

        WHILE left > 0 OR right < LENGTH OF nums MINUS 1
            IF left EQUALS 0
                INCREMENT right BY 1
            ELSE IF right EQUALS LENGTH OF nums MINUS 1
                DECREMENT left BY 1
            ELSE IF nums[left MINUS 1] GREATER THAN OR EQUAL TO nums[right PLUS 1]
                DECREMENT left BY 1
            ELSE
                INCREMENT right BY 1
            END IF

            SET current_min TO MINIMUM OF current_min, nums[left], nums[right]
            SET score TO current_min MULTIPLIED BY (right MINUS left PLUS 1)
            SET max_score TO MAXIMUM OF max_score AND score
        END WHILE

        RETURN max_score
    END FUNCTION
END CLASS
