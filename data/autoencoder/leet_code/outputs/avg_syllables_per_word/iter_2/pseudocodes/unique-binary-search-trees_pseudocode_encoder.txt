CLASS Solution
    FUNCTION numTrees(n)
        SET dp TO list of zeros with length n plus 1
        SET dp[0] TO 1
        SET dp[1] TO 1
        FOR i FROM 2 TO n
            FOR j FROM 1 TO i
                SET dp[i] TO dp[i] PLUS dp[j minus 1] TIMES dp[i minus j]
        RETURN dp[n]
    END FUNCTION
END CLASS
