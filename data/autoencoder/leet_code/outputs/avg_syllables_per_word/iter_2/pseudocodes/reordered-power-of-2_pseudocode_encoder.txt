CLASS Solution
    FUNCTION reorderedPowerOf2(n)
        SET power_of_2_sorted_digits TO empty set
        SET power_of_2 TO 1
        WHILE power_of_2 LESS THAN OR EQUAL TO 10 TO THE POWER OF 9
            ADD sorted digits of power_of_2 AS tuple TO power_of_2_sorted_digits
            MULTIPLY power_of_2 BY 2
        SET sorted_n_digits TO sorted digits of n AS tuple
        RETURN WHETHER sorted_n_digits IS IN power_of_2_sorted_digits
    END FUNCTION
END CLASS
