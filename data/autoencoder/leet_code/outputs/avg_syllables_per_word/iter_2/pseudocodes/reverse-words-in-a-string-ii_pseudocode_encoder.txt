CLASS Solution
    FUNCTION reverseWords(s: List of characters)
        CALL s.reverse()
        SET start TO 0
        FOR end FROM 0 TO LENGTH OF s MINUS 1
            IF s[end] EQUALS ' '
                CALL reverse_sublist FROM s[start] TO s[end - 1]
                SET start TO end + 1
        CALL reverse_sublist FROM s[start] TO END OF s
    END FUNCTION
    
    FUNCTION reverse_sublist FROM s[start] TO s[end]
        SET left TO start
        SET right TO end
        WHILE left LESS THAN right
            SWAP s[left] AND s[right]
            INCREMENT left BY 1
            DECREMENT right BY 1
    END FUNCTION
END CLASS
