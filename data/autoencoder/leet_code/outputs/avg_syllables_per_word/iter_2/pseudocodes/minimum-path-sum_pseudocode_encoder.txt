CLASS Solution
    FUNCTION minPathSum(grid)
        IF grid IS empty OR grid[0] IS empty THEN
            RETURN 0
        END IF

        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]

        SET dp TO 2D list of size m by n filled with 0

        SET dp[0][0] TO grid[0][0]

        FOR j FROM 1 TO n - 1
            SET dp[0][j] TO dp[0][j - 1] PLUS grid[0][j]
        END FOR

        FOR i FROM 1 TO m - 1
            SET dp[i][0] TO dp[i - 1][0] PLUS grid[i][0]
        END FOR

        FOR i FROM 1 TO m - 1
            FOR j FROM 1 TO n - 1
                SET dp[i][j] TO MINIMUM OF dp[i - 1][j] AND dp[i][j - 1] PLUS grid[i][j]
            END FOR
        END FOR

        RETURN dp[m - 1][n - 1]
    END FUNCTION
END CLASS
