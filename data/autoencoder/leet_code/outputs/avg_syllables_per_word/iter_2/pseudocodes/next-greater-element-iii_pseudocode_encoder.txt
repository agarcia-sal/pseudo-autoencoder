CLASS Solution
    FUNCTION nextGreaterElement(n)
        SET digits TO list of characters representing n
        SET length TO size of digits

        SET i TO length MINUS 2
        WHILE i >= 0 AND digits[i] IS GREATER THAN OR EQUAL TO digits[i + 1]
            DECREMENT i BY 1

        IF i EQUALS -1
            RETURN -1

        SET j TO length MINUS 1
        WHILE digits[j] IS LESS THAN OR EQUAL TO digits[i]
            DECREMENT j BY 1

        SWAP digits[i] WITH digits[j]

        SET digits TO concatenation of digits from start to i PLUS reverse of digits from i+1 to end

        SET result TO integer formed by joining digits

        IF result IS GREATER THAN 2 TO THE POWER 31 MINUS 1
            RETURN -1

        RETURN result
    END FUNCTION
END CLASS
