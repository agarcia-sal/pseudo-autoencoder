CLASS Solution
    FUNCTION getFactors(n)
        FUNCTION backtrack(start, target)
            FOR i FROM start TO INTEGER PART OF SQUARE ROOT OF target + 1
                IF target MODULO i EQUALS 0
                    APPEND path PLUS [i, target DIVIDED BY i] TO result
                    APPEND i TO path
                    CALL backtrack(i, target DIVIDED BY i)
                    REMOVE LAST ELEMENT FROM path
        SET result TO empty list
        SET path TO empty list
        CALL backtrack(2, n)
        RETURN result
    END FUNCTION
END CLASS
