CLASS Solution
    FUNCTION maxRotateFunction(nums)
        SET n TO LENGTH OF nums
        IF n EQUALS 1 THEN
            RETURN 0
        END IF
        
        SET total_sum TO SUM OF nums
        SET current_sum TO 0
        FOR i FROM 0 TO n MINUS 1
            INCREMENT current_sum BY i TIMES nums[i]
        
        SET max_value TO current_sum
        
        FOR k FROM 1 TO n MINUS 1
            SET current_sum TO current_sum PLUS total_sum MINUS n TIMES nums[n MINUS k]
            IF current_sum GREATER THAN max_value THEN
                SET max_value TO current_sum
            END IF
        
        RETURN max_value
    END FUNCTION
END CLASS
