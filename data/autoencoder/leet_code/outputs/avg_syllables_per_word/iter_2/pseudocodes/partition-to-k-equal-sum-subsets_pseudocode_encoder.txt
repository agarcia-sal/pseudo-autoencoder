CLASS Solution
    FUNCTION canPartitionKSubsets(nums, k)
        SET total_sum TO sum of nums
        IF total_sum MODULO k IS NOT 0 THEN
            RETURN False
        SET target TO total_sum DIVIDED BY k
        SORT nums in descending order
        
        FUNCTION can_partition(index, k, current_sum, used)
            IF k IS 0 THEN
                RETURN True
            IF current_sum EQUALS target THEN
                RETURN can_partition(0, k MINUS 1, 0, used)
            
            FOR i FROM index TO LENGTH OF nums MINUS 1
                IF used[i] IS False AND current_sum PLUS nums[i] LESS THAN OR EQUAL TO target THEN
                    SET used[i] TO True
                    IF can_partition(i PLUS 1, k, current_sum PLUS nums[i], used) THEN
                        RETURN True
                    SET used[i] TO False
            RETURN False
        
        SET used TO list of False with length LENGTH OF nums
        RETURN can_partition(0, k, 0, used)
    END FUNCTION
END CLASS
