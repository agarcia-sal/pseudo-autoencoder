CLASS Solution
    FUNCTION maxProfitAssignment(difficulty, profit, worker)
        SET jobs TO list of pairs from difficulty and profit
        SORT jobs by difficulty
        SORT worker by ability
        
        SET max_profit TO 0
        SET best_profit TO 0
        SET job_index TO 0
        SET n TO LENGTH OF jobs
        
        FOR each ability IN worker
            WHILE job_index < n AND jobs[job_index].difficulty <= ability
                SET best_profit TO maximum of best_profit AND jobs[job_index].profit
                INCREMENT job_index BY 1
            INCREMENT max_profit BY best_profit
        
        RETURN max_profit
    END FUNCTION
END CLASS
