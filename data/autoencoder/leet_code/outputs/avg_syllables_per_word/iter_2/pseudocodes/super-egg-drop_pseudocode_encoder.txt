CLASS Solution
    FUNCTION superEggDrop(k, n)
        SET dp TO list of zeros with length k + 1
        SET moves TO 0
        WHILE dp[k] LESS THAN n
            INCREMENT moves BY 1
            FOR eggs FROM k DOWNTO 1
                SET dp[eggs] TO 1 PLUS dp[eggs - 1] PLUS dp[eggs]
        RETURN moves
    END FUNCTION
END CLASS
