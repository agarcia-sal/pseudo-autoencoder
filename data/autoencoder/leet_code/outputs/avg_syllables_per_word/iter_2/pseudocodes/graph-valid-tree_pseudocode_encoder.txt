CLASS Solution
    FUNCTION validTree(n, edges)
        IF LENGTH OF edges NOT EQUAL TO n MINUS 1
            RETURN False

        SET adj_list TO DICTIONARY MAPPING EACH INTEGER FROM 0 TO n MINUS 1 TO EMPTY LIST
        FOR EACH a, b IN edges
            APPEND b TO adj_list[a]
            APPEND a TO adj_list[b]

        SET visited TO EMPTY SET

        FUNCTION dfs(node)
            IF node IN visited
                RETURN
            ADD node TO visited
            FOR EACH neighbor IN adj_list[node]
                CALL dfs(neighbor)
        END FUNCTION

        CALL dfs(0)

        RETURN LENGTH OF visited EQUAL TO n
    END FUNCTION
END CLASS
