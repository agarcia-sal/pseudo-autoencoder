CLASS Solution
    FUNCTION strongPasswordChecker(password)
        SET n TO LENGTH OF password
        
        SET has_lower TO FALSE
        SET has_upper TO FALSE
        SET has_digit TO FALSE
        FOR each character IN password
            IF character IS lowercase
                SET has_lower TO TRUE
            IF character IS uppercase
                SET has_upper TO TRUE
            IF character IS digit
                SET has_digit TO TRUE
        
        SET missing_types TO 3 - (has_lower + has_upper + has_digit)
        
        SET replace TO 0
        SET one TO 0
        SET two TO 0
        SET i TO 2
        WHILE i < n
            IF password[i] EQUALS password[i - 1] AND password[i] EQUALS password[i - 2]
                SET length TO 2
                WHILE i < n AND password[i] EQUALS password[i - 1]
                    INCREMENT length BY 1
                    INCREMENT i BY 1
                SET replace TO replace + (length DIVIDED BY 3)
                IF length MODULO 3 EQUALS 0
                    INCREMENT one BY 1
                ELSE IF length MODULO 3 EQUALS 1
                    INCREMENT two BY 1
            ELSE
                INCREMENT i BY 1
        
        IF n < 6
            RETURN MAXIMUM OF missing_types AND (6 - n)
        ELSE IF n <= 20
            RETURN MAXIMUM OF missing_types AND replace
        ELSE
            SET delete TO n - 20

            SET replace TO replace - MINIMUM OF delete AND one
            SET replace TO replace - (MINIMUM OF MAXIMUM OF (delete - one) AND 0, two * 2) DIVIDED BY 2
            SET replace TO replace - MAXIMUM OF (delete - one - 2 * two) AND 0 DIVIDED BY 3

            RETURN delete + MAXIMUM OF missing_types AND replace
    END FUNCTION
END CLASS
