CLASS Solution
    FUNCTION canTransform(start, end)
        SET start_filtered TO EMPTY LIST
        FOR each index i AND character c IN start
            IF c IS NOT 'X'
                APPEND (c, i) TO start_filtered

        SET end_filtered TO EMPTY LIST
        FOR each index i AND character c IN end
            IF c IS NOT 'X'
                APPEND (c, i) TO end_filtered

        IF LENGTH OF start_filtered IS NOT EQUAL TO LENGTH OF end_filtered
            RETURN False

        FOR each pair (c1, i1), (c2, i2) IN ZIP OF start_filtered AND end_filtered
            IF c1 IS NOT EQUAL TO c2
                RETURN False
            IF c1 IS 'L' AND i1 IS LESS THAN i2
                RETURN False
            IF c1 IS 'R' AND i1 IS GREATER THAN i2
                RETURN False

        RETURN True
    END FUNCTION
END CLASS
