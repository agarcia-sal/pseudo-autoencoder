CLASS Solution
    FUNCTION intersectionSizeTwo(intervals)
        SORT intervals BY end ASCENDING, start DESCENDING
        SET first TO intervals[0][1] MINUS 1
        SET second TO intervals[0][1]
        SET count TO 2
        FOR EACH interval FROM intervals[1] TO END
            SET start TO interval[0]
            SET end TO interval[1]
            IF start GREATER THAN second THEN
                SET first TO end MINUS 1
                SET second TO end
                INCREMENT count BY 2
            ELSE IF start GREATER THAN first THEN
                SET first TO second
                SET second TO end
                INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
