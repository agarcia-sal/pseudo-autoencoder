CLASS Solution
    FUNCTION findLUSlength(strs)
        FUNCTION is_subsequence(s, t)
            SET iterator TO ITERATOR OVER t
            FOR each character c IN s
                IF c NOT IN iterator THEN
                    RETURN False
            RETURN True
        END FUNCTION

        SORT strs BY descending LENGTH THEN lexicographically

        FOR each index i, word IN strs
            SET unique TO True
            FOR each index j, other IN strs
                IF i NOT EQUAL TO j AND is_subsequence(word, other) THEN
                    SET unique TO False
                    BREAK inner loop
            IF unique THEN
                RETURN LENGTH OF word

        RETURN -1
    END FUNCTION
END CLASS
