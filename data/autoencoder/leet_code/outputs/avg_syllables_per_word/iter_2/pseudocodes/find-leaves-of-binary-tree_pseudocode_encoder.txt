CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY
        RETURN None
    SET root TO NEW TreeNode WITH values[0]
    SET i TO 1
    CREATE queue AS DEQUE
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO POP LEFT FROM queue
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH values[i]
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH values[i]
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val DOES NOT EQUAL q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION findLeaves(root)
        FUNCTION collectLeaves(node)
            IF node IS None
                RETURN -1
            SET left_height TO collectLeaves(node.left)
            SET right_height TO collectLeaves(node.right)
            SET current_height TO MAXIMUM OF left_height AND right_height PLUS 1
            IF current_height GREATER THAN OR EQUAL TO LENGTH OF result
                APPEND EMPTY LIST TO result
            APPEND node.val TO result[current_height]
            RETURN current_height
        END FUNCTION

        SET result TO EMPTY LIST
        CALL collectLeaves WITH root
        RETURN result
    END FUNCTION
END CLASS
