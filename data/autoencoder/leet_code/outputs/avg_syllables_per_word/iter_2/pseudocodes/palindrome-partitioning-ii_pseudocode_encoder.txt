CLASS Solution
    FUNCTION minCut(s)
        SET n TO LENGTH OF s
        IF n <= 1 THEN
            RETURN 0
        END IF

        INITIALIZE is_palindrome AS 2D LIST OF False WITH DIMENSION n BY n

        FOR i FROM 0 TO n - 1
            SET is_palindrome[i][i] TO True
        END FOR

        FOR length FROM 2 TO n
            FOR start FROM 0 TO n - length
                SET end TO start + length - 1
                IF length == 2 THEN
                    SET is_palindrome[start][end] TO (s[start] EQUALS s[end])
                ELSE
                    SET is_palindrome[start][end] TO (s[start] EQUALS s[end]) AND is_palindrome[start + 1][end - 1]
                END IF
            END FOR
        END FOR

        INITIALIZE min_cuts AS LIST OF 0 WITH LENGTH n

        FOR i FROM 0 TO n - 1
            IF is_palindrome[0][i] THEN
                SET min_cuts[i] TO 0
            ELSE
                SET min_cuts[i] TO INFINITY
                FOR j FROM 1 TO i
                    IF is_palindrome[j][i] AND (min_cuts[j - 1] + 1) < min_cuts[i] THEN
                        SET min_cuts[i] TO min_cuts[j - 1] + 1
                    END IF
                END FOR
            END IF
        END FOR

        RETURN min_cuts[n - 1]
    END FUNCTION
END CLASS
