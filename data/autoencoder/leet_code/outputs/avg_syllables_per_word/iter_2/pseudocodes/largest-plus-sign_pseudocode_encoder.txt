CLASS Solution
    FUNCTION orderOfLargestPlusSign(n, mines)
        SET mines_set TO empty set
        FOR each mine IN mines
            ADD tuple of mine TO mines_set
        
        SET dp TO 2D array of size n by n where each element is a list [0, 0, 0, 0]
        
        FOR i FROM 0 TO n - 1
            FOR j FROM 0 TO n - 1
                IF (i, j) NOT IN mines_set THEN
                    IF j > 0 THEN
                        SET dp[i][j][0] TO dp[i][j - 1][0] + 1
                    IF i > 0 THEN
                        SET dp[i][j][1] TO dp[i - 1][j][1] + 1
        
        SET result TO 0
        FOR i FROM n - 1 DOWNTO 0
            FOR j FROM n - 1 DOWNTO 0
                IF (i, j) NOT IN mines_set THEN
                    IF j < n - 1 THEN
                        SET dp[i][j][2] TO dp[i][j + 1][2] + 1
                    IF i < n - 1 THEN
                        SET dp[i][j][3] TO dp[i + 1][j][3] + 1
                    SET plus_sign_order TO MINIMUM OF dp[i][j]
                    SET result TO MAXIMUM OF result AND plus_sign_order
        
        RETURN result
    END FUNCTION
END CLASS
