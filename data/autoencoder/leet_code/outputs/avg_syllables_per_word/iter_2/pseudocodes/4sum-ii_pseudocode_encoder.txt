CLASS Solution
    FUNCTION fourSumCount(nums1, nums2, nums3, nums4)
        SET sum_ab TO empty dictionary
        FOR each a IN nums1
            FOR each b IN nums2
                SET current_sum TO a PLUS b
                IF current_sum IN sum_ab
                    INCREMENT sum_ab[current_sum] BY 1
                ELSE
                    SET sum_ab[current_sum] TO 1
        SET count TO 0
        FOR each c IN nums3
            FOR each d IN nums4
                SET target TO NEGATION OF (c PLUS d)
                IF target IN sum_ab
                    INCREMENT count BY sum_ab[target]
        RETURN count
    END FUNCTION
END CLASS
