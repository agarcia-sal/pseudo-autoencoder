CLASS Solution
    FUNCTION letterCombinations(digits)
        IF digits IS EMPTY
            RETURN empty list
        END IF

        SET phone_map TO mapping of digits to letters:
            '2' TO 'abc'
            '3' TO 'def'
            '4' TO 'ghi'
            '5' TO 'jkl'
            '6' TO 'mno'
            '7' TO 'pqrs'
            '8' TO 'tuv'
            '9' TO 'wxyz'

        FUNCTION backtrack(combination, next_digits)
            IF LENGTH OF next_digits IS 0
                APPEND combination TO output
            ELSE
                FOR EACH letter IN phone_map[next_digits AT POSITION 0]
                    CALL backtrack WITH (combination + letter, next_digits WITHOUT FIRST CHARACTER)
                END FOR
            END IF
        END FUNCTION

        SET output TO empty list
        CALL backtrack WITH empty string AND digits
        RETURN output
    END FUNCTION
END CLASS
