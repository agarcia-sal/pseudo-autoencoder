CLASS Solution
    FUNCTION partitionLabels(s)
        SET last_occurrence TO EMPTY DICTIONARY
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            SET last_occurrence[s[i]] TO i
        SET start TO 0
        SET end TO 0
        SET partition_sizes TO EMPTY LIST
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            SET char TO s[i]
            SET end TO MAXIMUM OF end AND last_occurrence[char]
            IF i EQUALS end THEN
                APPEND (end MINUS start PLUS 1) TO partition_sizes
                SET start TO i PLUS 1
        RETURN partition_sizes
    END FUNCTION
END CLASS
