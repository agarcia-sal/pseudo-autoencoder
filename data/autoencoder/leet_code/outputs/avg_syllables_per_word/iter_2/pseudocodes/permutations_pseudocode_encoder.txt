CLASS Solution
    FUNCTION permute(nums)
        FUNCTION backtrack(path, remaining)
            IF remaining IS EMPTY
                APPEND path TO permutations
                RETURN
            END IF
            FOR i FROM 0 TO LENGTH OF remaining MINUS 1
                SET new_path TO path WITH remaining[i] APPENDED
                SET new_remaining TO CONCATENATION OF remaining UP TO i AND remaining AFTER i
                CALL backtrack(new_path, new_remaining)
            END FOR
        END FUNCTION

        SET permutations TO EMPTY LIST
        CALL backtrack WITH EMPTY LIST AND nums
        RETURN permutations
    END FUNCTION
END CLASS
