CLASS Solution
    FUNCTION uniquePathsWithObstacles(obstacleGrid)
        IF obstacleGrid IS EMPTY OR obstacleGrid[0][0] EQUALS 1
            RETURN 0
        SET m TO NUMBER OF ROWS IN obstacleGrid
        SET n TO NUMBER OF COLUMNS IN obstacleGrid
        SET dp TO 2D LIST OF SIZE m BY n FILLED WITH 0
        SET dp[0][0] TO 1

        FOR j FROM 1 TO n - 1
            IF obstacleGrid[0][j] EQUALS 0
                SET dp[0][j] TO dp[0][j - 1]

        FOR i FROM 1 TO m - 1
            IF obstacleGrid[i][0] EQUALS 0
                SET dp[i][0] TO dp[i - 1][0]

        FOR i FROM 1 TO m - 1
            FOR j FROM 1 TO n - 1
                IF obstacleGrid[i][j] EQUALS 0
                    SET dp[i][j] TO dp[i - 1][j] PLUS dp[i][j - 1]

        RETURN dp[m - 1][n - 1]
    END FUNCTION
END CLASS
