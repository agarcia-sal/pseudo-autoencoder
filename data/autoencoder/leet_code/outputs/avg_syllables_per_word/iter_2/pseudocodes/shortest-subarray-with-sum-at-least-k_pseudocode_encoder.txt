CLASS Solution
    FUNCTION shortestSubarray(nums, k)
        SET prefix_sum TO list containing 0
        FOR each num IN nums
            APPEND (last element of prefix_sum + num) TO prefix_sum

        SET dq TO empty deque
        SET min_length TO infinity

        FOR i, current_sum IN prefix_sum with indices
            WHILE dq is not empty AND current_sum MINUS prefix_sum at dq front IS GREATER THAN OR EQUAL TO k
                SET min_length TO minimum of min_length AND (i MINUS dq front element)
                REMOVE element from front of dq

            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO prefix_sum at dq back
                REMOVE element from back of dq

            APPEND i TO dq

        IF min_length IS NOT infinity
            RETURN min_length
        ELSE
            RETURN -1
    END FUNCTION
END CLASS
