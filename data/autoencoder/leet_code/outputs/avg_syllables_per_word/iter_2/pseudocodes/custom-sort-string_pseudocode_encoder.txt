CLASS Solution
    FUNCTION customSortString(order, s)
        SET d TO EMPTY DICTIONARY
        FOR i FROM 0 TO LENGTH OF order MINUS 1
            SET c TO CHARACTER AT POSITION i IN order
            SET d[c] TO i
        SET sorted_list TO SORTED s BY GETTING d VALUE FOR EACH CHARACTER, DEFAULT 0 IF NOT FOUND
        SET result TO JOIN sorted_list INTO STRING
        RETURN result
    END FUNCTION
END CLASS
