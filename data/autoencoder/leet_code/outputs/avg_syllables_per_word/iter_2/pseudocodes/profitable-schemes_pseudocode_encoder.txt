CLASS Solution
    FUNCTION profitableSchemes(n, minProfit, group, profit)
        SET MOD TO 10 EXPONENT 9 PLUS 7
        INITIALIZE dp AS 2D LIST OF SIZE (n + 1) BY (minProfit + 1) WITH ALL VALUES 0
        FOR i FROM 0 TO n
            SET dp[i][0] TO 1
        
        FOR EACH p, g IN profit AND group
            FOR i FROM n DOWNTO g
                FOR j FROM minProfit DOWNTO 0
                    SET dp[i][j] TO (dp[i][j] + dp[i - g][MAXIMUM OF 0 AND (j - p)]) MODULO MOD
        
        RETURN dp[n][minProfit]
    END FUNCTION
END CLASS
