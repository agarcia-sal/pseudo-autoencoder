CLASS Solution
    FUNCTION isReflected(points)
        SET unique_points TO set of points converted to tuples
        SET min_x TO minimum x-coordinate in unique_points
        SET max_x TO maximum x-coordinate in unique_points
        SET line_of_reflection TO (min_x + max_x) DIVIDED BY 2
        FOR each (x, y) IN unique_points
            SET reflected_x TO 2 TIMES line_of_reflection MINUS x
            IF (reflected_x, y) NOT IN unique_points THEN
                RETURN False
        RETURN True
    END FUNCTION
END CLASS
