CLASS Solution
    FUNCTION exist(board, word)
        IF board IS EMPTY
            RETURN False
        SET m TO LENGTH OF board
        SET n TO LENGTH OF board[0]
        SET visited TO 2D LIST OF False WITH SIZE m BY n

        FUNCTION dfs(x, y, index)
            IF index EQUALS LENGTH OF word
                RETURN True
            IF x IS LESS THAN 0 OR x IS GREATER THAN OR EQUAL TO m OR y IS LESS THAN 0 OR y IS GREATER THAN OR EQUAL TO n OR visited[x][y] IS True OR board[x][y] DOES NOT EQUAL word[index]
                RETURN False

            SET visited[x][y] TO True
            FOR EACH (dx, dy) IN [(-1, 0), (1, 0), (0, -1), (0, 1)]
                IF dfs(x + dx, y + dy, index + 1) IS True
                    RETURN True
            SET visited[x][y] TO False
            RETURN False
        END FUNCTION

        FOR i FROM 0 TO m - 1
            FOR j FROM 0 TO n - 1
                IF dfs(i, j, 0) IS True
                    RETURN True
        RETURN False
    END FUNCTION
END CLASS
