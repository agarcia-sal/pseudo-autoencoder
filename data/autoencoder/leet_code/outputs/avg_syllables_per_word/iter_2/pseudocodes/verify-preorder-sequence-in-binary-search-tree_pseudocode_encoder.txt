CLASS Solution
    FUNCTION verifyPreorder(preorder: List of integers) RETURNS Boolean
        SET stack TO empty list
        SET lower_bound TO negative infinity
        
        FOR each value IN preorder
            IF value < lower_bound THEN
                RETURN False
            END IF
            
            WHILE stack is not empty AND value > last element in stack DO
                SET lower_bound TO pop last element from stack
            END WHILE
            
            APPEND value TO stack
        END FOR
        
        RETURN True
    END FUNCTION
END CLASS
