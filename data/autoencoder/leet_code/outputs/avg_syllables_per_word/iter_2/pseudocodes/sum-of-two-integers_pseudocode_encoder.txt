CLASS Solution
    FUNCTION getSum(a: integer, b: integer) RETURNS integer
        SET MAX TO 2147483647
        SET mask TO 4294967295
        
        WHILE b IS NOT 0
            SET carry TO (a BITWISE AND b) SHIFT LEFT BY 1
            SET a TO (a BITWISE XOR b) BITWISE AND mask
            SET b TO carry BITWISE AND mask
        
        IF a IS LESS THAN OR EQUAL TO MAX
            RETURN a
        ELSE
            RETURN BITWISE COMPLEMENT OF (a BITWISE XOR mask)
    END FUNCTION
END CLASS
