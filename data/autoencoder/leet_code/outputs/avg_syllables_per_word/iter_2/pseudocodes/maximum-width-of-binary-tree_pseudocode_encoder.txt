CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO deque()
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION widthOfBinaryTree(root)
        IF root IS None
            RETURN 0
        END IF
        SET queue TO deque CONTAINING (root, 0)
        SET max_width TO 0
        WHILE queue IS NOT EMPTY
            SET level_length TO LENGTH OF queue
            SET _, first_index TO queue[0]
            FOR count FROM 0 TO level_length MINUS 1
                SET node, index TO queue.popleft()
                IF count EQUALS level_length MINUS 1
                    SET max_width TO MAXIMUM OF max_width AND (index MINUS first_index PLUS 1)
                END IF
                IF node.left EXISTS
                    APPEND (node.left, 2 TIMES index) TO queue
                END IF
                IF node.right EXISTS
                    APPEND (node.right, 2 TIMES index PLUS 1) TO queue
                END IF
            END FOR
        END WHILE
        RETURN max_width
    END FUNCTION
END CLASS
