CLASS Solution
    FUNCTION findLonelyPixel(picture)
        IF picture IS EMPTY OR picture[0] IS EMPTY THEN
            RETURN 0
        SET m TO LENGTH OF picture
        SET n TO LENGTH OF picture[0]
        SET row_count TO LIST OF ZEROES WITH LENGTH m
        SET col_count TO LIST OF ZEROES WITH LENGTH n
        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                IF picture[i][j] EQUALS 'B' THEN
                    INCREMENT row_count[i] BY 1
                    INCREMENT col_count[j] BY 1
        SET lonely_pixel_count TO 0
        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                IF picture[i][j] EQUALS 'B' AND row_count[i] EQUALS 1 AND col_count[j] EQUALS 1 THEN
                    INCREMENT lonely_pixel_count BY 1
        RETURN lonely_pixel_count
    END FUNCTION
END CLASS
