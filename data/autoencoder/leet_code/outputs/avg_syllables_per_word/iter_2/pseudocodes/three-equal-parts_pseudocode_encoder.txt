CLASS Solution
    FUNCTION threeEqualParts(arr)
        SET total_ones TO sum of all elements in arr

        IF total_ones IS 0 THEN
            RETURN [0, 2]
        END IF

        IF total_ones MODULO 3 IS NOT 0 THEN
            RETURN [-1, -1]
        END IF

        SET part_ones TO total_ones DIVIDED BY 3

        SET first_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] IS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones IS 1 THEN
                    SET first_part_index TO i
                END IF
                IF count_ones IS part_ones THEN
                    BREAK LOOP
                END IF
            END IF
        END FOR

        SET second_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] IS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones IS part_ones PLUS 1 THEN
                    SET second_part_index TO i
                    BREAK LOOP
                END IF
            END IF
        END FOR

        SET third_part_index TO 0
        SET count_ones TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            IF arr[i] IS 1 THEN
                INCREMENT count_ones BY 1
                IF count_ones IS (2 TIMES part_ones) PLUS 1 THEN
                    SET third_part_index TO i
                    BREAK LOOP
                END IF
            END IF
        END FOR

        SET n TO LENGTH OF arr MINUS third_part_index

        IF (first_part_index PLUS n) LESS THAN OR EQUAL TO LENGTH OF arr AND
           (second_part_index PLUS n) LESS THAN OR EQUAL TO LENGTH OF arr AND
           SUBARRAY of arr FROM first_part_index TO first_part_index PLUS n MINUS 1 EQUALS
           SUBARRAY of arr FROM second_part_index TO second_part_index PLUS n MINUS 1 AND
           SUBARRAY of arr FROM third_part_index TO END THEN
            RETURN [first_part_index PLUS n MINUS 1, second_part_index PLUS n]
        END IF

        RETURN [-1, -1]
    END FUNCTION
END CLASS
