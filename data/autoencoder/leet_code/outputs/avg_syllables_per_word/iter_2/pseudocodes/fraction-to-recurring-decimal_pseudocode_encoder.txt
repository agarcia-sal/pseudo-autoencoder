CLASS Solution
    FUNCTION fractionToDecimal(numerator, denominator)
        IF (numerator < 0) XOR (denominator < 0) THEN
            SET sign TO "-"
        ELSE
            SET sign TO ""
        
        SET numerator TO absolute value of numerator
        SET denominator TO absolute value of denominator
        
        SET quotient, remainder TO integer division and remainder of numerator divided by denominator
        SET integer_part TO string representation of quotient
        
        IF remainder EQUALS 0 THEN
            RETURN concatenation of sign and integer_part
        
        SET seen_remainders TO empty dictionary
        SET fractional_part TO empty list
        
        WHILE remainder IS NOT 0 DO
            IF remainder IS IN seen_remainders THEN
                SET start TO value from seen_remainders for remainder
                SET non_repeating TO concatenation of fractional_part from start 0 up to start
                SET repeating TO concatenation of fractional_part from start to end
                RETURN concatenation of sign, integer_part, ".", non_repeating, "(", repeating, ")"
            
            STORE position of remainder in seen_remainders as length of fractional_part
            SET remainder TO remainder multiplied by 10
            SET quotient, remainder TO integer division and remainder of remainder divided by denominator
            APPEND string representation of quotient TO fractional_part
        
        RETURN concatenation of sign, integer_part, ".", concatenation of fractional_part
    END FUNCTION
END CLASS
