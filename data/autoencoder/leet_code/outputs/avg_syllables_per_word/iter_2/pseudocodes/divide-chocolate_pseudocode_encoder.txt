CLASS Solution
    FUNCTION maximizeSweetness(sweetness, k)
        FUNCTION canDivide(min_sweet)
            SET pieces TO 0
            SET current_sweet TO 0
            FOR each sweet IN sweetness
                INCREMENT current_sweet BY sweet
                IF current_sweet >= min_sweet THEN
                    INCREMENT pieces BY 1
                    SET current_sweet TO 0
                    IF pieces > k THEN
                        RETURN True
            RETURN pieces > k
        END FUNCTION

        SET left TO minimum value IN sweetness
        SET right TO sum of sweetness DIVIDED BY (k + 1)
        SET result TO 0
        WHILE left <= right DO
            SET mid TO left + right DIVIDED BY 2
            IF canDivide(mid) THEN
                SET result TO mid
                SET left TO mid + 1
            ELSE
                SET right TO mid - 1
        RETURN result
    END FUNCTION
END CLASS
