CLASS Solution
    FUNCTION minimumTimeRequired(jobs, k)
        SET n TO LENGTH OF jobs
        SET subset_sums TO LIST OF SIZE (2 RAISED TO n) FILLED WITH 0
        
        FOR i FROM 0 TO n - 1
            FOR mask FROM 0 TO (2 RAISED TO i) - 1
                SET subset_sums OF (mask BITWISE OR (1 SHIFT LEFT BY i)) TO subset_sums AT mask PLUS jobs AT i
        
        FUNCTION dp(k, mask) CACHE RESULTS
            IF k EQUALS 1 THEN
                RETURN subset_sums AT mask
            SET res TO INFINITY
            SET submask TO mask
            WHILE submask IS NOT 0
                SET res TO MINIMUM OF res AND MAXIMUM OF dp(k - 1, mask BITWISE XOR submask) AND subset_sums AT submask
                SET submask TO (submask MINUS 1) BITWISE AND mask
            RETURN res
        
        RETURN dp(k, (1 SHIFT LEFT BY n) MINUS 1)
    END FUNCTION
END CLASS
