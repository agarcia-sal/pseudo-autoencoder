CLASS Solution
    FUNCTION numRescueBoats(people, limit)
        SORT people IN ascending order
        SET left TO 0
        SET right TO LENGTH OF people MINUS 1
        SET boats TO 0
        WHILE left <= right
            IF people[left] PLUS people[right] <= limit
                INCREMENT left BY 1
            DECREMENT right BY 1
            INCREMENT boats BY 1
        RETURN boats
    END FUNCTION
END CLASS
