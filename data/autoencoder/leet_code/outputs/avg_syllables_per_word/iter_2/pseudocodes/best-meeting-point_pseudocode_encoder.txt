CLASS Solution
    FUNCTION minTotalDistance(grid: List[List[int]]) RETURNS int
        SET rows TO empty list
        SET cols TO empty list
        FOR i FROM 0 TO LENGTH OF grid MINUS 1
            FOR j FROM 0 TO LENGTH OF grid[0] MINUS 1
                IF grid[i][j] EQUALS 1
                    APPEND i TO rows
                    APPEND j TO cols
        CALL rows SORT
        CALL cols SORT
        SET median_row TO element AT INDEX (LENGTH OF rows DIVIDED BY 2) IN rows
        SET median_col TO element AT INDEX (LENGTH OF cols DIVIDED BY 2) IN cols
        SET min_distance TO 0
        FOR each row IN rows
            INCREMENT min_distance BY ABSOLUTE VALUE OF (row MINUS median_row)
        FOR each col IN cols
            INCREMENT min_distance BY ABSOLUTE VALUE OF (col MINUS median_col)
        RETURN min_distance
    END FUNCTION
END CLASS
