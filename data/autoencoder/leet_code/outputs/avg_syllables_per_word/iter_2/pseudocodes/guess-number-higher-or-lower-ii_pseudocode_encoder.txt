CLASS Solution
    FUNCTION getMoneyAmount(n)
        DEFINE dp(left, right) WITH CACHE
            IF left GREATER OR EQUAL TO right THEN
                RETURN 0
            SET min_cost TO infinity
            FOR pivot FROM left TO right
                SET cost TO pivot PLUS MAXIMUM OF dp(left, pivot MINUS 1) AND dp(pivot PLUS 1, right)
                IF cost LESS THAN min_cost THEN
                    SET min_cost TO cost
            RETURN min_cost
        RETURN dp(1, n)
    END FUNCTION
END CLASS
