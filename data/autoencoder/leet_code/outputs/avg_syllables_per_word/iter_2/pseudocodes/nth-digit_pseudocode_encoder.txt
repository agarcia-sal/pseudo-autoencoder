CLASS Solution
    FUNCTION findNthDigit(n)
        SET length TO 1
        SET count TO 9
        SET start TO 1
        WHILE n > length * count
            SET n TO n - (length * count)
            INCREMENT length BY 1
            SET count TO count * 10
            SET start TO start * 10
        END WHILE
        SET start TO start + (n - 1) DIVIDED BY length
        SET digit_index TO (n - 1) MODULO length
        SET digit_string TO CONVERT start TO STRING
        SET digit_character TO CHARACTER AT digit_index IN digit_string
        RETURN CONVERT digit_character TO INTEGER
    END FUNCTION
END CLASS
