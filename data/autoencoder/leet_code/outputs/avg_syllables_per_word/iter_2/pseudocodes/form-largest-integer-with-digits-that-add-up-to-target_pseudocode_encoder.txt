CLASS Solution
    FUNCTION largestNumber(cost: List[int], target: int) RETURNS string
        SET cost_to_digit TO DICTIONARY mapping each cost c TO string of digit (index of c in cost plus 1)
        SET dp TO LIST with first element as empty string and rest as "-1" repeated target times
        
        FOR t FROM 1 TO target
            FOR each (c, d) IN cost_to_digit
                IF t >= c AND dp[t - c] IS NOT "-1"
                    SET candidate TO d CONCATENATED WITH dp[t - c]
                    IF dp[t] IS "-1" OR LENGTH OF candidate > LENGTH OF dp[t] OR 
                       (LENGTH OF candidate = LENGTH OF dp[t] AND candidate > dp[t])
                        SET dp[t] TO candidate
        
        IF dp[target] IS NOT "-1"
            RETURN dp[target]
        ELSE
            RETURN "0"
    END FUNCTION
END CLASS
