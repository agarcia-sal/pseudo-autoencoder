CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO NEW TreeNode WITH values[0]
    SET i TO 1
    SET queue TO NEW deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT OF queue
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION insertIntoBST(root, val)
        FUNCTION dfs(root)
            IF root IS None
                RETURN NEW TreeNode WITH val
            END IF
            IF root.val LESS THAN val
                SET root.right TO dfs(root.right)
            ELSE
                SET root.left TO dfs(root.left)
            END IF
            RETURN root
        END FUNCTION
        RETURN dfs(root)
    END FUNCTION
END CLASS
