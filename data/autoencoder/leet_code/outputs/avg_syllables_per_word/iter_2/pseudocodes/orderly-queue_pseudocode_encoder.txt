CLASS Solution
    FUNCTION orderlyQueue(s, k)
        IF k > 1 THEN
            SET sorted_list TO SORT s
            SET result TO JOIN sorted_list INTO STRING
            RETURN result
        ELSE
            SET n TO LENGTH OF s
            SET smallest TO s
            FOR i FROM 1 TO n MINUS 1 DO
                SET rotated TO CONCATENATE SUBSTRING OF s FROM i TO END WITH SUBSTRING OF s FROM START TO i
                IF rotated IS LESS THAN smallest THEN
                    SET smallest TO rotated
            RETURN smallest
    END FUNCTION
END CLASS
