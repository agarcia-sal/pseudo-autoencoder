CLASS Solution
    FUNCTION escapeGhosts(ghosts, target)
        SET player_distance TO absolute value of target[0] PLUS absolute value of target[1]
        FOR each ghost IN ghosts
            SET ghost_distance TO absolute value of ghost[0] MINUS target[0] PLUS absolute value of ghost[1] MINUS target[1]
            IF ghost_distance LESS THAN OR EQUAL TO player_distance
                RETURN False
        RETURN True
    END FUNCTION
END CLASS
