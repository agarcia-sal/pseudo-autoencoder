CLASS Solution
    FUNCTION threeSum(nums)
        SORT nums
        SET result TO empty list
        SET n TO LENGTH OF nums
        
        FOR i FROM 0 TO n MINUS 3
            IF i GREATER THAN 0 AND nums[i] EQUALS nums[i MINUS 1]
                CONTINUE
            
            SET left TO i PLUS 1
            SET right TO n MINUS 1
            
            WHILE left LESS THAN right
                SET total TO nums[i] PLUS nums[left] PLUS nums[right]
                
                IF total EQUALS 0
                    APPEND [nums[i], nums[left], nums[right]] TO result
                    
                    WHILE left LESS THAN right AND nums[left] EQUALS nums[left PLUS 1]
                        INCREMENT left BY 1
                    
                    WHILE left LESS THAN right AND nums[right] EQUALS nums[right MINUS 1]
                        DECREMENT right BY 1
                    
                    INCREMENT left BY 1
                    DECREMENT right BY 1
                
                ELSE IF total LESS THAN 0
                    INCREMENT left BY 1
                
                ELSE
                    DECREMENT right BY 1
        
        RETURN result
    END FUNCTION
END CLASS
