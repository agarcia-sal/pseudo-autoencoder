CLASS Solution
    FUNCTION canFinish(numCourses, prerequisites)
        SET graph TO empty mapping of lists
        SET indegree TO list of zeros with length numCourses
        
        FOR each pair (course, prereq) IN prerequisites
            APPEND course TO graph[prereq]
            INCREMENT indegree[course] BY 1
        
        SET queue TO list of courses with indegree equal to 0
        
        WHILE queue is not empty
            SET current_course TO first element removed from queue
            DECREMENT numCourses BY 1
            
            FOR each next_course IN graph[current_course]
                DECREMENT indegree[next_course] BY 1
                IF indegree[next_course] EQUALS 0
                    APPEND next_course TO queue
        
        RETURN True IF numCourses EQUALS 0 OTHERWISE False
    END FUNCTION
END CLASS
