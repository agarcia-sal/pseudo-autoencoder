CLASS Solution
    FUNCTION largestNumber(nums: List of integers) RETURNS string
        SET nums_str TO list of strings converted from nums
        
        FUNCTION compare(x, y)
            IF concatenation of x and y IS GREATER THAN concatenation of y and x
                RETURN -1
            ELSE IF concatenation of x and y IS LESS THAN concatenation of y and x
                RETURN 1
            ELSE
                RETURN 0
            END IF
        END FUNCTION
        
        SORT nums_str USING compare AS comparator
        
        IF first element of nums_str IS EQUAL TO "0"
            RETURN "0"
        END IF
        
        SET result TO concatenation of all elements in nums_str
        RETURN result
    END FUNCTION
END CLASS
