CLASS Solution
    FUNCTION rotate(matrix)
        SET n TO LENGTH OF matrix
        FOR i FROM 0 TO n MINUS 1
            FOR j FROM i TO n MINUS 1
                SET temp TO matrix[i][j]
                SET matrix[i][j] TO matrix[j][i]
                SET matrix[j][i] TO temp
        FOR i FROM 0 TO n MINUS 1
            REVERSE matrix[i]
    END FUNCTION
END CLASS
