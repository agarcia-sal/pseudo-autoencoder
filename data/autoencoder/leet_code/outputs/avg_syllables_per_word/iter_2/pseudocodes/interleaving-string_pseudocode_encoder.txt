CLASS Solution
    FUNCTION isInterleave(s1, s2, s3)
        IF LENGTH OF s1 PLUS LENGTH OF s2 IS NOT EQUAL TO LENGTH OF s3
            RETURN False
        END IF

        SET dp TO LIST OF Booleans OF SIZE LENGTH OF s2 PLUS 1 WITH ALL False
        SET dp[0] TO True

        FOR i FROM 0 TO LENGTH OF s1
            FOR j FROM 0 TO LENGTH OF s2
                IF i GREATER THAN 0
                    SET dp[j] TO dp[j] AND (CHARACTER AT s1 AT POSITION i MINUS 1 EQUALS CHARACTER AT s3 AT POSITION i PLUS j MINUS 1)
                END IF
                IF j GREATER THAN 0
                    SET dp[j] TO dp[j] OR (dp[j MINUS 1] AND (CHARACTER AT s2 AT POSITION j MINUS 1 EQUALS CHARACTER AT s3 AT POSITION i PLUS j MINUS 1))
                END IF
            END FOR
        END FOR

        RETURN dp[LENGTH OF s2]
    END FUNCTION
END CLASS
