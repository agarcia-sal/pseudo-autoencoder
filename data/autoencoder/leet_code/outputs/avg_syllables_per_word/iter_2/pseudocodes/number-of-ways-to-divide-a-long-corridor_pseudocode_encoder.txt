CLASS Solution
    FUNCTION numberOfWays(corridor)
        SET MOD TO 1000000007
        SET seats TO LIST OF INDICES i WHERE corridor AT i IS 'S'

        IF LENGTH OF seats MODULO 2 IS NOT 0 OR LENGTH OF seats LESS THAN 2
            RETURN 0
        END IF

        SET ways TO 1
        FOR i FROM 2 TO LENGTH OF seats STEP 2
            SET ways TO ways MULTIPLIED BY (seats AT i MINUS seats AT (i MINUS 1)) MODULO MOD
        END FOR

        RETURN ways
    END FUNCTION
END CLASS
