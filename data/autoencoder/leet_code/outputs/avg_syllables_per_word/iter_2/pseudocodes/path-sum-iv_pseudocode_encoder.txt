CLASS Solution
    FUNCTION pathSum(nums)
        SET tree TO empty dictionary
        FOR each num IN nums
            SET depth TO num DIVIDED BY 100
            SET pos TO (num DIVIDED BY 10) MODULO 10
            SET value TO num MODULO 10
            SET tree[(depth, pos)] TO value
        END FOR

        FUNCTION dfs(depth, pos, path_sum)
            SET current_value TO tree.get((depth, pos))
            IF current_value IS NONE
                RETURN 0
            END IF

            SET new_path_sum TO path_sum PLUS current_value

            SET left_child TO (depth PLUS 1, 2 MULTIPLIED BY pos MINUS 1)
            SET right_child TO (depth PLUS 1, 2 MULTIPLIED BY pos)

            IF left_child NOT IN tree AND right_child NOT IN tree
                RETURN new_path_sum
            END IF

            SET left_sum TO dfs(left_child[0], left_child[1], new_path_sum)
            SET right_sum TO dfs(right_child[0], right_child[1], new_path_sum)

            RETURN left_sum PLUS right_sum
        END FUNCTION

        RETURN dfs(1, 1, 0)
    END FUNCTION
END CLASS
