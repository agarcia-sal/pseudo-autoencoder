CLASS Solution
    FUNCTION hasPath(maze, start, destination)
        SET directions TO list of (0, 1), (1, 0), (0, -1), (-1, 0)
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze at index 0
        SET visited TO empty set
        
        FUNCTION dfs(x, y)
            IF (x, y) IN visited THEN
                RETURN False
            IF [x, y] EQUALS destination THEN
                RETURN True
            ADD (x, y) TO visited
            
            FOR each (dx, dy) IN directions
                SET nx TO x
                SET ny TO y
                WHILE nx + dx IS WITHIN 0 AND m - 1 AND ny + dy IS WITHIN 0 AND n - 1 AND maze VALUE AT (nx + dx, ny + dy) IS 0
                    INCREMENT nx BY dx
                    INCREMENT ny BY dy
                IF dfs(nx, ny) IS True THEN
                    RETURN True
            RETURN False
        END FUNCTION
        
        RETURN dfs(start AT index 0, start AT index 1)
    END FUNCTION
END CLASS
