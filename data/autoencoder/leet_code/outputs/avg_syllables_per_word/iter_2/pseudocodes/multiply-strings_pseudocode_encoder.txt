CLASS Solution
    FUNCTION multiply(num1, num2)
        IF num1 EQUALS "0" OR num2 EQUALS "0"
            RETURN "0"
        END IF

        SET result TO list of zeros with length LENGTH OF num1 PLUS LENGTH OF num2
        SET num1 TO reverse of num1
        SET num2 TO reverse of num2

        FOR i FROM 0 TO LENGTH OF num1 MINUS 1
            FOR j FROM 0 TO LENGTH OF num2 MINUS 1
                SET mul TO (numeric value of num1 at i) MULTIPLIED BY (numeric value of num2 at j)
                SET p1 TO i PLUS j
                SET p2 TO i PLUS j PLUS 1
                SET sum_ TO mul PLUS result at p1

                SET result at p1 TO sum_ MODULO 10
                INCREMENT result at p2 BY sum_ DIVIDED BY 10
            END FOR
        END FOR

        SET result_str TO concatenation of elements in reverse of result as strings
        REMOVE leading zeros from result_str
        RETURN result_str
    END FUNCTION
END CLASS
