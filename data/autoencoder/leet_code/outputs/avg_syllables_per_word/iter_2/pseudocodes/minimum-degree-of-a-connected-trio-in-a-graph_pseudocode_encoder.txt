CLASS Solution
    FUNCTION minTrioDegree(n, edges)
        SET graph TO dictionary with keys from 1 to n and empty sets as values

        FOR each (u, v) IN edges
            ADD v TO graph[u]
            ADD u TO graph[v]

        SET min_degree TO infinity
        SET trio_found TO false

        FOR i FROM 1 TO n
            FOR j FROM i + 1 TO n
                FOR k FROM j + 1 TO n
                    IF j IN graph[i] AND k IN graph[i] AND k IN graph[j]
                        SET trio_found TO true
                        SET trio_degree TO LENGTH OF graph[i] PLUS LENGTH OF graph[j] PLUS LENGTH OF graph[k] MINUS 6
                        SET min_degree TO MINIMUM OF min_degree AND trio_degree

        IF trio_found
            RETURN min_degree
        ELSE
            RETURN -1
    END FUNCTION
END CLASS
