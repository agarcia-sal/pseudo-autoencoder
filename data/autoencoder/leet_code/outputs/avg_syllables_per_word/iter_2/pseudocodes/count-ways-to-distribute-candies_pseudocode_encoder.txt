CLASS Solution
    FUNCTION waysToDistribute(n, k)
        SET MOD TO 1000000007
        INITIALIZE dp AS 2D ARRAY OF SIZE (n + 1) BY (k + 1) FILLED WITH 0
        SET dp[0][0] TO 1
        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                SET dp[i][j] TO (j TIMES dp[i - 1][j] PLUS dp[i - 1][j - 1]) MODULO MOD
        RETURN dp[n][k]
    END FUNCTION
END CLASS
