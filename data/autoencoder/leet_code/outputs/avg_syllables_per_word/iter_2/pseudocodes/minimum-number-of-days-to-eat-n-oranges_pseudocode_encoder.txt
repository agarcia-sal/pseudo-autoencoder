CLASS Solution
    FUNCTION minDays(n)
        DEFINE dp(remaining) WITH memoization
            IF remaining <= 1
                RETURN remaining
            END IF
            RETURN 1 + MINIMUM OF (remaining MODULO 2 + dp(remaining DIVIDED BY 2), remaining MODULO 3 + dp(remaining DIVIDED BY 3))
        END FUNCTION
        RETURN dp(n)
    END FUNCTION
END CLASS
