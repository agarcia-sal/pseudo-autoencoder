CLASS Solution
    FUNCTION findSubstring(s: string, words: list of string) RETURNS list of integer
        IF s IS EMPTY OR words IS EMPTY OR words[0] IS EMPTY THEN
            RETURN empty list
        END IF

        SET word_length TO LENGTH OF words[0]
        SET num_words TO LENGTH OF words
        SET total_length TO word_length MULTIPLIED BY num_words
        SET word_count TO COUNT OF EACH word IN words
        SET result TO empty list

        FOR i FROM 0 TO LENGTH OF s MINUS total_length INCLUSIVE
            SET seen TO EMPTY COUNTER
            FOR j FROM i TO i PLUS total_length EXCLUSIVE STEP word_length
                SET word TO SUBSTRING OF s FROM j TO j PLUS word_length
                INCREMENT seen[word] BY 1
                IF seen[word] IS GREATER THAN word_count[word] THEN
                    BREAK INNER LOOP
                END IF
            END FOR
            IF INNER LOOP COMPLETED WITHOUT BREAK THEN
                APPEND i TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
