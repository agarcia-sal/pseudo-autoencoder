CLASS Solution
    FUNCTION jump(nums)
        IF LENGTH OF nums <= 1 THEN
            RETURN 0
        END IF

        SET jumps TO 0
        SET current_end TO 0
        SET farthest TO 0

        FOR i FROM 0 TO LENGTH OF nums MINUS 2
            SET farthest TO MAXIMUM OF farthest AND (i PLUS nums[i])

            IF i EQUALS current_end THEN
                INCREMENT jumps BY 1
                SET current_end TO farthest

                IF current_end GREATER THAN OR EQUAL TO LENGTH OF nums MINUS 1 THEN
                    BREAK
                END IF
            END IF
        END FOR

        RETURN jumps
    END FUNCTION
END CLASS
