CLASS Solution
    FUNCTION kSmallestPairs(nums1, nums2, k)
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k <= 0 THEN
            RETURN empty list
        END IF

        SET min_heap TO empty list
        FOR i FROM 0 TO MIN(k, LENGTH OF nums1) - 1 DO
            PUSH (nums1[i] + nums2[0], i, 0) INTO min_heap
        END FOR

        SET result TO empty list
        WHILE min_heap IS NOT EMPTY AND LENGTH OF result < k DO
            POP smallest_sum, i, j FROM min_heap
            APPEND [nums1[i], nums2[j]] TO result
            IF j + 1 < LENGTH OF nums2 THEN
                PUSH (nums1[i] + nums2[j + 1], i, j + 1) INTO min_heap
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
