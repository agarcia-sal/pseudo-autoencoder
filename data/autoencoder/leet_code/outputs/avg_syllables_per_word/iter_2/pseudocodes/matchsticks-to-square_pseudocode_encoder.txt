CLASS Solution
    FUNCTION makesquare(matchsticks)
        SET total_length TO sum of matchsticks
        IF total_length MODULO 4 IS NOT 0
            RETURN False
        END IF

        SET side_length TO total_length DIVIDED BY 4
        SORT matchsticks IN descending order

        FUNCTION can_form_sides(sides, index)
            IF index EQUALS LENGTH OF matchsticks
                RETURN True IF all sides EQUAL side_length ELSE False
            END IF

            FOR i FROM 0 TO 3
                IF sides[i] PLUS matchsticks[index] IS LESS THAN OR EQUAL TO side_length
                    INCREMENT sides[i] BY matchsticks[index]
                    IF can_form_sides(sides, index PLUS 1) IS True
                        RETURN True
                    END IF
                    DECREMENT sides[i] BY matchsticks[index]
                END IF

                IF sides[i] EQUALS 0
                    BREAK
                END IF
            END FOR

            RETURN False
        END FUNCTION

        RETURN can_form_sides([0, 0, 0, 0], 0)
    END FUNCTION
END CLASS
