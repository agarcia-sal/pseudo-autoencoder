CLASS Solution
    FUNCTION asteroidCollision(asteroids)
        SET stack TO empty list
        FOR each asteroid IN asteroids
            WHILE stack is not empty AND asteroid is less than 0 AND top of stack is greater than 0
                IF top of stack is less than negative asteroid
                    REMOVE top element from stack
                    CONTINUE to next iteration of WHILE loop
                ELSE IF top of stack equals negative asteroid
                    REMOVE top element from stack
                END IF
                BREAK from WHILE loop
            END WHILE
            IF WHILE loop did not BREAK prematurely
                APPEND asteroid TO stack
        RETURN stack
    END FUNCTION
END CLASS
