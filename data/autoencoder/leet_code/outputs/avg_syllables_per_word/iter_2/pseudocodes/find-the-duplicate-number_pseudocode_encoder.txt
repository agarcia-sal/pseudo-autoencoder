CLASS Solution
    FUNCTION findDuplicate(nums)
        FUNCTION f(x)
            SET count TO 0
            FOR each value IN nums
                IF value LESS THAN OR EQUAL TO x THEN
                    INCREMENT count BY 1
            RETURN count GREATER THAN x
        END FUNCTION

        RETURN bisect_left of range from 0 to LENGTH OF nums with target True and key function f
    END FUNCTION
END CLASS
