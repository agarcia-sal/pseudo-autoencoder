CLASS Solution
    FUNCTION isRectangleCover(rectangles)
        SET X1 TO infinity
        SET Y1 TO infinity
        SET X2 TO negative infinity
        SET Y2 TO negative infinity

        INITIALIZE points AS empty set
        SET area TO 0

        FOR each rectangle IN rectangles
            DESTRUCTURE rectangle INTO x1, y1, x2, y2
            SET X1 TO minimum of X1 and x1
            SET Y1 TO minimum of Y1 and y1
            SET X2 TO maximum of X2 and x2
            SET Y2 TO maximum of Y2 and y2

            INCREMENT area BY (x2 MINUS x1) TIMES (y2 MINUS y1)

            FOR each point p IN list of points [(x1, y1), (x1, y2), (x2, y1), (x2, y2)]
                IF p IS IN points
                    REMOVE p FROM points
                ELSE
                    ADD p TO points

        IF area IS NOT EQUAL TO (X2 MINUS X1) TIMES (Y2 MINUS Y1)
            RETURN False

        IF points IS NOT EQUAL TO set containing (X1, Y1), (X1, Y2), (X2, Y1), (X2, Y2)
            RETURN False

        RETURN True
    END FUNCTION
END CLASS
