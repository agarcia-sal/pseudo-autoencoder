CLASS Solution
    FUNCTION groupStrings(strings)
        FUNCTION get_canonical_form(s)
            IF s IS empty
                RETURN empty string
            SET canonical_form TO empty list
            FOR each character c IN s
                SET diff TO (ordinal value of c MINUS ordinal value of first character in s) MODULO 26
                APPEND string representation of diff TO canonical_form
            RETURN concatenation of canonical_form items separated by comma
        END FUNCTION

        SET groups TO empty dictionary with default value as empty list

        FOR each string s IN strings
            SET canonical_form TO get_canonical_form(s)
            APPEND s TO groups[canonical_form]

        RETURN list of all values in groups
    END FUNCTION
END CLASS
