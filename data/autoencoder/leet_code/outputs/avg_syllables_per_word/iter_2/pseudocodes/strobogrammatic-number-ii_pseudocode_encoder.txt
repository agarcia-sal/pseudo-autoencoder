CLASS Solution
    FUNCTION findStrobogrammatic(n)
        FUNCTION dfs(u)
            IF u EQUALS 0 THEN
                RETURN list containing empty string
            IF u EQUALS 1 THEN
                RETURN list containing '0', '1', '8'
            SET ans TO empty list
            FOR each v IN dfs(u MINUS 2)
                FOR each pair IN list of pairs ('1', '1'), ('8', '8'), ('6', '9'), ('9', '6')
                    SET l TO first element of pair
                    SET r TO second element of pair
                    APPEND concatenation of l, v, and r TO ans
                IF u NOT EQUAL TO n THEN
                    APPEND concatenation of '0', v, and '0' TO ans
            RETURN ans
        RETURN dfs(n)
END CLASS
