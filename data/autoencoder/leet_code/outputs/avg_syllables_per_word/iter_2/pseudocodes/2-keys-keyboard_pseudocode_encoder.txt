CLASS Solution
    FUNCTION minSteps(n)
        IF n EQUALS 1 THEN
            RETURN 0
        END IF

        SET operations TO 0
        SET factor TO 2

        WHILE n GREATER THAN 1
            WHILE n MODULO factor EQUALS 0
                INCREMENT operations BY factor
                SET n TO n DIVIDED BY factor
            END WHILE
            INCREMENT factor BY 1
        END WHILE

        RETURN operations
    END FUNCTION
END CLASS
