CLASS Solution
    FUNCTION wordBreak(s, wordDict)
        SET wordSet TO set of wordDict
        SET dp TO list of False with length LENGTH OF s PLUS 1
        SET dp[0] TO True
        FOR i FROM 1 TO LENGTH OF s
            FOR j FROM 0 TO i MINUS 1
                IF dp[j] IS True AND substring of s FROM j TO i IS IN wordSet
                    SET dp[i] TO True
                    BREAK inner loop
        RETURN dp[LENGTH OF s]
    END FUNCTION
END CLASS
