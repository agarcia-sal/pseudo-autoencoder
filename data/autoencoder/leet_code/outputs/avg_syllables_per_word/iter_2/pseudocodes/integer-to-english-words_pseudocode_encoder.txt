CLASS Solution
    FUNCTION numberToWords(num)
        IF num == 0
            RETURN "Zero"
        
        SET below_twenty TO ["", "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten",
                            "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"]
        SET tens TO ["", "", "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"]
        SET thousands TO ["", "Thousand", "Million", "Billion"]
        
        FUNCTION helper(n)
            IF n == 0
                RETURN ""
            ELSE IF n < 20
                RETURN below_twenty[n] + " "
            ELSE IF n < 100
                RETURN tens[n DIVIDED BY 10] + " " + helper(n MODULO 10)
            ELSE
                RETURN below_twenty[n DIVIDED BY 100] + " Hundred " + helper(n MODULO 100)
        END FUNCTION
        
        SET result TO ""
        SET i TO 0
        
        WHILE num > 0
            IF num MODULO 1000 != 0
                SET result TO helper(num MODULO 1000) + thousands[i] + " " + result
            SET num TO num DIVIDED BY 1000
            INCREMENT i BY 1
        
        RETURN result STRIPPED OF LEADING AND TRAILING SPACES
    END FUNCTION
END CLASS
