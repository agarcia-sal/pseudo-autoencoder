CLASS Solution
    FUNCTION findOrder(numCourses, prerequisites)
        SET adj_list TO empty dictionary mapping each course from 0 TO numCourses - 1 TO empty list
        SET in_degrees TO list of zeros with length numCourses

        FOR each pair (dest, src) IN prerequisites
            APPEND dest TO adj_list[src]
            INCREMENT in_degrees[dest] BY 1

        SET queue TO deque of all courses WHERE in_degrees[course] EQUALS 0
        SET topological_order TO empty list

        WHILE queue is not empty
            SET current TO REMOVE first element FROM queue
            APPEND current TO topological_order

            FOR each neighbor IN adj_list[current]
                DECREMENT in_degrees[neighbor] BY 1
                IF in_degrees[neighbor] EQUALS 0
                    APPEND neighbor TO queue

        IF LENGTH OF topological_order EQUALS numCourses
            RETURN topological_order
        ELSE
            RETURN empty list
    END FUNCTION
END CLASS
