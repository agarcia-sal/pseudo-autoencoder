CLASS Solution
    FUNCTION countRangeSum(nums, lower, upper)
        SET prefix_sums TO [0]
        SET current_sum TO 0
        SET count TO 0
        FOR each num IN nums
            INCREMENT current_sum BY num
            SET left TO current_sum MINUS upper
            SET right TO current_sum MINUS lower
            INCREMENT count BY (NUMBER OF VALUES IN prefix_sums LESS THAN OR EQUAL TO right) MINUS (NUMBER OF VALUES IN prefix_sums LESS THAN left)
            INSERT current_sum INTO prefix_sums MAINTAINING SORTED ORDER
        RETURN count
    END FUNCTION
END CLASS
