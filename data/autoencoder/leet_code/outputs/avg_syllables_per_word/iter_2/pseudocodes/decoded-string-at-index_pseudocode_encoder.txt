CLASS Solution
    FUNCTION decodeAtIndex(s, k)
        SET length TO 0

        FOR each char IN s
            IF char is digit THEN
                SET length TO length MULTIPLIED BY integer value of char
            ELSE
                INCREMENT length BY 1

        FOR each char IN reverse order of s
            SET k TO k MODULO length

            IF k EQUALS 0 AND char is alphabetic THEN
                RETURN char

            IF char is digit THEN
                SET length TO length DIVIDED BY integer value of char
            ELSE
                DECREMENT length BY 1
    END FUNCTION
END CLASS
