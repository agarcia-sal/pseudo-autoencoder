CLASS Solution
    FUNCTION makeLargestSpecial(s)
        FUNCTION dfs(s)
            SET count TO 0
            SET start TO 0
            SET specials TO empty list

            FOR i, char IN ENUMERATE s
                IF char EQUALS '1'
                    INCREMENT count BY 1
                ELSE
                    DECREMENT count BY 1
                
                IF count EQUALS 0
                    APPEND '1' CONCATENATED WITH dfs(SUBSTRING s FROM start + 1 TO i) CONCATENATED WITH '0' TO specials
                    SET start TO i + 1
            
            SORT specials IN descending lexicographical order
            RETURN CONCATENATION OF specials
        END FUNCTION

        RETURN dfs(s)
    END FUNCTION
END CLASS
