CLASS Solution
    FUNCTION maxJumps(arr, d)
        FUNCTION dp(i)  WITH MEMOIZATION
            SET max_jumps TO 1
            FOR x FROM 1 TO d
                IF i + x IS LESS THAN LENGTH OF arr THEN
                    IF arr AT (i + x) IS GREATER THAN OR EQUAL TO arr AT i THEN
                        BREAK LOOP
                    END IF
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 PLUS dp(i + x))
                END IF
            END FOR
            FOR x FROM 1 TO d
                IF i - x IS GREATER THAN OR EQUAL TO 0 THEN
                    IF arr AT (i - x) IS GREATER THAN OR EQUAL TO arr AT i THEN
                        BREAK LOOP
                    END IF
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 PLUS dp(i - x))
                END IF
            END FOR
            RETURN max_jumps
        END FUNCTION
        SET result TO MAXIMUM OF dp(i) FOR EACH i FROM 0 TO LENGTH OF arr MINUS 1
        RETURN result
    END FUNCTION
END CLASS
