CLASS Solution
    FUNCTION scoreOfParentheses(s)
        SET stack TO empty list
        SET score TO 0
        FOR each char IN s
            IF char EQUALS '('
                APPEND score TO stack
                SET score TO 0
            ELSE
                SET last_score TO POP last element FROM stack
                IF score EQUALS 0
                    SET score TO last_score PLUS 1
                ELSE
                    SET score TO last_score PLUS 2 TIMES score
        RETURN score
    END FUNCTION
END CLASS
