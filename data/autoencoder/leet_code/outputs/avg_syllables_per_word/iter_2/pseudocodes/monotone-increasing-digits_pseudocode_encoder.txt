CLASS Solution
    FUNCTION monotoneIncreasingDigits(n)
        SET digits TO list of characters representing n
        SET length TO size of digits
        SET marker TO length
        FOR i FROM length MINUS 1 DOWNTO 1
            IF digit at index i LESS THAN digit at index i MINUS 1
                SET digit at index i MINUS 1 TO string of (integer value of digit at index i MINUS 1 MINUS 1)
                SET marker TO i
        FOR i FROM marker TO length MINUS 1
            SET digit at index i TO '9'
        RETURN integer value formed by concatenating digits
    END FUNCTION
END CLASS
