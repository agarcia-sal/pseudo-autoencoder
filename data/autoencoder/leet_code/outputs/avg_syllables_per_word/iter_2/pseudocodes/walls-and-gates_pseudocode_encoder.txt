CLASS Solution
    FUNCTION wallsAndGates(rooms)
        IF rooms IS EMPTY OR rooms[0] IS EMPTY THEN
            RETURN
        SET rows TO LENGTH OF rooms
        SET cols TO LENGTH OF rooms[0]
        SET queue TO EMPTY DEQUE

        FOR r FROM 0 TO rows MINUS 1
            FOR c FROM 0 TO cols MINUS 1
                IF rooms[r][c] EQUALS 0 THEN
                    APPEND (r, c) TO queue

        SET directions TO LIST OF (0, 1), (1, 0), (0, -1), (-1, 0)

        WHILE queue IS NOT EMPTY
            SET (r, c) TO POP LEFT FROM queue
            FOR each (dr, dc) IN directions
                SET nr TO r PLUS dr
                SET nc TO c PLUS dc
                IF nr IS GREATER OR EQUAL TO 0 AND nr IS LESS THAN rows AND nc IS GREATER OR EQUAL TO 0 AND nc IS LESS THAN cols AND rooms[nr][nc] EQUALS 2147483647 THEN
                    SET rooms[nr][nc] TO rooms[r][c] PLUS 1
                    APPEND (nr, nc) TO queue
    END FUNCTION
END CLASS
