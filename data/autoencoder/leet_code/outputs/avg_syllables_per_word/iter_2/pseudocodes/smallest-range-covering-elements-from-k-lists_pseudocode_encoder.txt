CLASS Solution
    FUNCTION smallestRange(nums)
        SET min_heap TO empty list
        SET max_value TO negative infinity
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            PUSH (nums[i][0], i, 0) INTO min_heap
            SET max_value TO maximum of max_value AND nums[i][0]
        
        SET smallest_range TO positive infinity
        SET result TO list containing -100000 AND 100000
        
        WHILE min_heap is not empty
            POP (min_value, list_index, element_index) FROM min_heap
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN
                SET smallest_range TO max_value MINUS min_value
                SET result TO list containing min_value AND max_value
            
            IF element_index PLUS 1 LESS THAN LENGTH OF nums[list_index] THEN
                SET next_value TO nums[list_index][element_index PLUS 1]
                PUSH (next_value, list_index, element_index PLUS 1) INTO min_heap
                SET max_value TO maximum of max_value AND next_value
            ELSE
                BREAK
        
        RETURN result
    END FUNCTION
END CLASS
