CLASS Solution
    FUNCTION tallestBillboard(rods: List of Integer)
        SET dp TO dictionary with key 0 and value 0
        
        FOR each rod IN rods
            SET new_dp TO a copy of dp
            FOR each key d and value y IN dp
                SET sum1 TO new_dp value at key d plus rod or 0 if not present
                SET new_dp at key d plus rod TO maximum of sum1 and y
                
                SET difference TO absolute value of d minus rod
                SET sum2 TO new_dp value at key difference or 0 if not present
                SET new_dp at key difference TO maximum of sum2 and y plus minimum of d and rod
            SET dp TO new_dp
        
        RETURN dp value at key 0
    END FUNCTION
END CLASS
