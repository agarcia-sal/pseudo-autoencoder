CLASS Solution
    FUNCTION findMedianSortedArrays(nums1, nums2)
        IF LENGTH OF nums1 > LENGTH OF nums2 THEN
            SET temp TO nums1
            SET nums1 TO nums2
            SET nums2 TO temp
        END IF

        SET x TO LENGTH OF nums1
        SET y TO LENGTH OF nums2
        SET low TO 0
        SET high TO x

        WHILE low <= high
            SET partitionX TO (low + high) DIVIDED BY 2
            SET partitionY TO (x + y + 1) DIVIDED BY 2 MINUS partitionX

            IF partitionX == 0 THEN
                SET maxLeftX TO NEGATIVE_INFINITY
            ELSE
                SET maxLeftX TO nums1 AT INDEX (partitionX - 1)
            END IF

            IF partitionX == x THEN
                SET minRightX TO POSITIVE_INFINITY
            ELSE
                SET minRightX TO nums1 AT INDEX partitionX
            END IF

            IF partitionY == 0 THEN
                SET maxLeftY TO NEGATIVE_INFINITY
            ELSE
                SET maxLeftY TO nums2 AT INDEX (partitionY - 1)
            END IF

            IF partitionY == y THEN
                SET minRightY TO POSITIVE_INFINITY
            ELSE
                SET minRightY TO nums2 AT INDEX partitionY
            END IF

            IF maxLeftX <= minRightY AND maxLeftY <= minRightX THEN
                IF (x + y) MODULO 2 == 0 THEN
                    RETURN (MAXIMUM OF maxLeftX AND maxLeftY PLUS MINIMUM OF minRightX AND minRightY) DIVIDED BY 2
                ELSE
                    RETURN MAXIMUM OF maxLeftX AND maxLeftY
                END IF
            ELSE IF maxLeftX > minRightY THEN
                SET high TO partitionX MINUS 1
            ELSE
                SET low TO partitionX PLUS 1
            END IF
        END WHILE
    END FUNCTION
END CLASS
