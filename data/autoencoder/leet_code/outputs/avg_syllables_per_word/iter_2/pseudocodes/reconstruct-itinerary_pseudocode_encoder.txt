CLASS Solution
    FUNCTION findItinerary(tickets)
        SET graph TO a default dictionary mapping to empty lists
        FOR each pair (from_airport, to_airport) IN tickets sorted in reverse order
            APPEND to_airport TO graph[from_airport]
        SET itinerary TO empty list
        FUNCTION visit(airport)
            WHILE graph[airport] is not empty
                CALL visit WITH graph[airport].pop() as argument
            APPEND airport TO itinerary
        CALL visit WITH "JFK" as argument
        RETURN itinerary reversed
    END FUNCTION
END CLASS
