CLASS Solution
    FUNCTION maximumGood(statements)
        SET n TO LENGTH OF statements
        SET max_good TO 0
        FOR i FROM 0 TO (2 RAISED TO n) MINUS 1
            SET good TO LIST OF n FALSE VALUES
            SET bad TO LIST OF n FALSE VALUES
            FOR j FROM 0 TO n MINUS 1
                IF i BITWISE-AND (1 RAISED TO j) IS NOT 0 THEN
                    SET good[j] TO TRUE
                ELSE
                    SET bad[j] TO TRUE
            SET valid TO TRUE
            FOR j FROM 0 TO n MINUS 1
                IF good[j] IS TRUE THEN
                    FOR k FROM 0 TO n MINUS 1
                        IF (statements[j][k] IS 0 AND good[k] IS TRUE) OR (statements[j][k] IS 1 AND bad[k] IS TRUE) THEN
                            SET valid TO FALSE
                            BREAK INNER LOOP
                    IF valid IS FALSE THEN
                        BREAK OUTER LOOP
            IF valid IS TRUE THEN
                SET max_good TO MAXIMUM OF max_good AND SUM OF good
        RETURN max_good
    END FUNCTION
END CLASS
