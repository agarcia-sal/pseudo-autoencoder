CLASS Solution
    FUNCTION searchRange(nums, target)
        FUNCTION find_left(nums, target)
            SET left TO 0
            SET right TO LENGTH OF nums MINUS 1
            WHILE left <= right
                SET mid TO (left + right) DIVIDED BY 2
                IF nums[mid] < target
                    SET left TO mid + 1
                ELSE
                    SET right TO mid - 1
            RETURN left
        END FUNCTION

        FUNCTION find_right(nums, target)
            SET left TO 0
            SET right TO LENGTH OF nums MINUS 1
            WHILE left <= right
                SET mid TO (left + right) DIVIDED BY 2
                IF nums[mid] <= target
                    SET left TO mid + 1
                ELSE
                    SET right TO mid - 1
            RETURN right
        END FUNCTION

        SET left_index TO find_left(nums, target)
        SET right_index TO find_right(nums, target)

        IF left_index <= right_index AND right_index >= 0 AND right_index < LENGTH OF nums AND nums[left_index] == target
            RETURN LIST OF left_index, right_index
        ELSE
            RETURN LIST OF -1, -1
    END FUNCTION
END CLASS
