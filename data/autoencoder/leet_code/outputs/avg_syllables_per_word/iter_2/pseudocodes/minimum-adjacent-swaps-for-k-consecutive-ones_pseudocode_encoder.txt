CLASS Solution
    FUNCTION minMoves(nums, k)
        SET positions TO empty list
        FOR each index i AND value num IN nums
            IF num EQUALS 1
                APPEND i TO positions
        END FOR

        FUNCTION calculate_cost(start, end)
            SET mid TO (start + end) DIVIDED BY 2
            SET median TO positions[mid]
            SET cost TO 0
            FOR i FROM start TO end MINUS 1
                SET cost TO cost PLUS ABSOLUTE VALUE OF (positions[i] MINUS (median MINUS (mid MINUS i)))
            END FOR
            RETURN cost
        END FUNCTION

        SET min_cost TO infinity
        FOR i FROM 0 TO LENGTH OF positions MINUS k
            SET current_cost TO calculate_cost(i, i PLUS k)
            IF current_cost LESS THAN min_cost
                SET min_cost TO current_cost
            END IF
        END FOR

        RETURN min_cost
    END FUNCTION
END CLASS
