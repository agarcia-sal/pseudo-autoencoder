CLASS Solution
    FUNCTION isBipartite(graph: List of List of int) RETURNS bool
        SET n TO LENGTH OF graph
        SET color TO LIST OF 0s WITH LENGTH n

        FUNCTION dfs(node, c) RETURNS bool
            SET color[node] TO c
            FOR each neighbor IN graph[node]
                IF color[neighbor] IS c THEN
                    RETURN False
                IF color[neighbor] IS 0 THEN
                    IF NOT dfs(neighbor, NEGATE c) THEN
                        RETURN False
            RETURN True
        END FUNCTION

        FOR i FROM 0 TO n MINUS 1
            IF color[i] IS 0 THEN
                IF NOT dfs(i, 1) THEN
                    RETURN False
        RETURN True
    END FUNCTION
END CLASS
