CLASS Solution
    FUNCTION minBuildTime(blocks, split)
        TRANSFORM blocks INTO min-heap
        
        WHILE LENGTH OF blocks GREATER THAN 1
            SET first TO EXTRACT minimum FROM blocks
            SET second TO EXTRACT minimum FROM blocks
            
            SET combined_time TO split PLUS second
            
            INSERT combined_time INTO blocks
        END WHILE
        
        RETURN THE ONLY ELEMENT IN blocks
    END FUNCTION
END CLASS
