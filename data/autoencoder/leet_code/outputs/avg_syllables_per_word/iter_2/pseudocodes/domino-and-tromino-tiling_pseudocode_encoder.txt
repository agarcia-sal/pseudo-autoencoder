CLASS Solution
    FUNCTION numTilings(n)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7

        SET dp TO LIST OF LENGTH n+1 FILLED WITH 0
        SET dp1 TO LIST OF LENGTH n+1 FILLED WITH 0

        SET dp[0] TO 1
        SET dp[1] TO 1

        IF n IS GREATER OR EQUAL TO 2
            SET dp[2] TO 2
            SET dp1[2] TO 1
        END IF

        FOR i FROM 3 TO n
            SET dp[i] TO (dp[i - 1] + dp[i - 2] + 2 TIMES dp1[i - 1]) MODULO MOD
            SET dp1[i] TO (dp[i - 2] + dp1[i - 1]) MODULO MOD
        END FOR

        RETURN dp[n]
    END FUNCTION
END CLASS
