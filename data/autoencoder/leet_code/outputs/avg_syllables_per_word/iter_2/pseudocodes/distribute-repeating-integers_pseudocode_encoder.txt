CLASS Solution
    FUNCTION canDistribute(nums, quantity)
        SET count TO list of frequencies of each number in nums
        SORT quantity in descending order
        
        FUNCTION can_satisfy(index)
            IF index EQUALS LENGTH OF quantity THEN
                RETURN True
            END IF
            FOR i FROM 0 TO LENGTH OF count MINUS 1
                IF count[i] GREATER THAN OR EQUAL TO quantity[index] THEN
                    DECREMENT count[i] BY quantity[index]
                    IF can_satisfy(index PLUS 1) THEN
                        RETURN True
                    END IF
                    INCREMENT count[i] BY quantity[index]
                END IF
            END FOR
            RETURN False
        END FUNCTION
        
        RETURN can_satisfy(0)
    END FUNCTION
END CLASS
