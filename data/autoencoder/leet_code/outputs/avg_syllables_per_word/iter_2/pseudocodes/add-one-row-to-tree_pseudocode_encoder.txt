CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO NEW TreeNode WITH val = values[0]
    SET i TO 1
    INITIALIZE queue AS new empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH val = values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH val = values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION addOneRow(root: Optional TreeNode, val: Integer, depth: Integer) RETURNS Optional TreeNode
        IF depth IS EQUAL TO 1
            RETURN NEW TreeNode WITH val = val, left = root
        END IF
        INITIALIZE queue AS new deque WITH ELEMENT (root, 1)
        WHILE queue IS NOT EMPTY
            SET node, level TO queue.popleft()
            IF level IS EQUAL TO depth MINUS 1
                SET node.left TO NEW TreeNode WITH val = val, left = node.left
                SET node.right TO NEW TreeNode WITH val = val, right = node.right
            ELSE IF level IS LESS THAN depth MINUS 1
                IF node.left IS NOT None
                    APPEND (node.left, level PLUS 1) TO queue
                END IF
                IF node.right IS NOT None
                    APPEND (node.right, level PLUS 1) TO queue
                END IF
            END IF
        END WHILE
        RETURN root
    END FUNCTION
END CLASS
