CLASS Solution
    FUNCTION minCost(n, cuts)
        SET cuts TO sorted list of cuts appended with 0 and n

        FUNCTION dp(left, right) 
            IF right MINUS left LESS THAN OR EQUAL TO 1
                RETURN 0
            END IF

            SET min_cost TO infinity

            FOR i FROM left PLUS 1 TO right MINUS 1
                SET cost TO cuts[right] MINUS cuts[left] PLUS dp(left, i) PLUS dp(i, right)
                IF cost LESS THAN min_cost
                    SET min_cost TO cost
                END IF
            END FOR

            RETURN min_cost
        END FUNCTION

        RETURN dp(0, LENGTH OF cuts MINUS 1)
    END FUNCTION
END CLASS
