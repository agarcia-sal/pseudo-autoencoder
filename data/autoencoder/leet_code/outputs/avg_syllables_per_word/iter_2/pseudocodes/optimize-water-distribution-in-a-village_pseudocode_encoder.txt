CLASS Solution
    FUNCTION minCostToSupplyWater(n, wells, pipes)
        FOR i FROM 0 TO n MINUS 1
            APPEND [0, i PLUS 1, wells[i]] TO pipes
        SORT pipes BY cost
        
        SET parent TO list of integers from 0 TO n
        
        FUNCTION find(x)
            IF parent[x] IS NOT EQUAL TO x
                SET parent[x] TO find(parent[x])
            RETURN parent[x]
        END FUNCTION
        
        FUNCTION union(x, y)
            SET rootX TO find(x)
            SET rootY TO find(y)
            IF rootX IS NOT EQUAL TO rootY
                SET parent[rootX] TO rootY
                RETURN True
            RETURN False
        END FUNCTION
        
        SET min_cost TO 0
        FOR EACH house1, house2, cost IN pipes
            IF union(house1, house2) IS True
                INCREMENT min_cost BY cost
        RETURN min_cost
    END FUNCTION
END CLASS
