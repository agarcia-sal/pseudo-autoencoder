CLASS Solution
    FUNCTION ambiguousCoordinates(s: string) RETURNS list of string
        FUNCTION make(frag: string) RETURNS generator of string
            SET n TO LENGTH OF frag
            FOR d FROM 1 TO n
                SET left TO SUBSTRING OF frag FROM START TO d
                SET right TO SUBSTRING OF frag FROM d TO END
                IF (left DOES NOT START WITH '0' OR left EQUALS '0') AND (right DOES NOT END WITH '0')
                    YIELD left + ('.' IF d IS NOT n ELSE '') + right
        END FUNCTION

        SET s TO SUBSTRING OF s FROM SECOND CHARACTER TO SECOND LAST CHARACTER
        RETURN LIST OF STRINGS FORMATTED AS "(" + a + ", " + b + ")"
            FOR i FROM 1 TO LENGTH OF s MINUS 1
            FOR EACH a IN make(SUBSTRING OF s FROM START TO i)
            FOR EACH b IN make(SUBSTRING OF s FROM i TO END)
END CLASS
