CLASS Solution
    FUNCTION findNumberOfLIS(nums)
        IF nums IS EMPTY
            RETURN 0
        SET n TO LENGTH OF nums
        SET lengths TO LIST OF n ONES
        SET counts TO LIST OF n ONES
        FOR i FROM 0 TO n MINUS 1
            FOR j FROM 0 TO i MINUS 1
                IF nums[i] IS GREATER THAN nums[j]
                    IF lengths[j] PLUS 1 IS GREATER THAN lengths[i]
                        SET lengths[i] TO lengths[j] PLUS 1
                        SET counts[i] TO counts[j]
                    ELSE IF lengths[j] PLUS 1 EQUALS lengths[i]
                        INCREMENT counts[i] BY counts[j]
        SET longest TO MAXIMUM VALUE IN lengths
        SET total_count TO 0
        FOR i FROM 0 TO n MINUS 1
            IF lengths[i] EQUALS longest
                INCREMENT total_count BY counts[i]
        RETURN total_count
    END FUNCTION
END CLASS
