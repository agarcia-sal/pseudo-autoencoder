CLASS Solution
    FUNCTION minimumTime(s)
        SET n TO LENGTH OF s
        SET left_costs TO LIST OF SIZE n FILLED WITH 0
        SET right_costs TO LIST OF SIZE n FILLED WITH 0

        IF s[0] EQUALS '1'
            SET left_costs[0] TO 1
        ELSE
            SET left_costs[0] TO 0
        FOR i FROM 1 TO n - 1
            IF s[i] EQUALS '0'
                SET left_costs[i] TO left_costs[i - 1]
            ELSE
                SET left_costs[i] TO MINIMUM OF (i + 1) AND (left_costs[i - 1] + 2)

        IF s[n - 1] EQUALS '1'
            SET right_costs[n - 1] TO 1
        ELSE
            SET right_costs[n - 1] TO 0
        FOR i FROM n - 2 DOWN TO 0
            IF s[i] EQUALS '0'
                SET right_costs[i] TO right_costs[i + 1]
            ELSE
                SET right_costs[i] TO MINIMUM OF (n - i) AND (right_costs[i + 1] + 2)

        SET min_time TO MINIMUM OF left_costs[n - 1] AND right_costs[0]
        FOR i FROM 0 TO n - 2
            SET min_time TO MINIMUM OF min_time AND (left_costs[i] + right_costs[i + 1])

        RETURN min_time
    END FUNCTION
END CLASS
