CLASS Solution
    FUNCTION searchMatrix(matrix, target)
        IF matrix IS EMPTY OR matrix[0] IS EMPTY
            RETURN False
        END IF
        SET m TO LENGTH OF matrix
        SET n TO LENGTH OF matrix[0]
        SET left TO 0
        SET right TO (m TIMES n) MINUS 1
        WHILE left <= right
            SET mid TO (left PLUS right) DIVIDED BY 2
            SET mid_value TO matrix AT (mid DIVIDED BY n) AND (mid MODULO n)
            IF mid_value EQUALS target
                RETURN True
            ELSE IF mid_value LESS THAN target
                SET left TO mid PLUS 1
            ELSE
                SET right TO mid MINUS 1
            END IF
        END WHILE
        RETURN False
    END FUNCTION
END CLASS
