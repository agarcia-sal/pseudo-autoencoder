CLASS Solution
    FUNCTION minInteger(num: string, k: integer) RETURNS string
        SET num TO list of characters in num
        SET n TO LENGTH OF num
        
        WHILE k > 0
            SET swapped TO False
            FOR i FROM 0 TO n - 2
                IF num[i] > num[i + 1]
                    SET min_digit TO '9'
                    SET min_index TO -1
                    FOR j FROM i TO MINIMUM OF (i + k) AND (n - 1)
                        IF num[j] < min_digit
                            SET min_digit TO num[j]
                            SET min_index TO j
                    SET num[i TO min_index] TO [num[min_index]] CONCATENATED WITH num[i TO min_index - 1]
                    DECREMENT k BY (min_index - i)
                    SET swapped TO True
                    BREAK FOR
            IF swapped IS False
                BREAK WHILE
        
        RETURN concatenation of characters in num
    END FUNCTION
END CLASS
