CLASS Solution
    FUNCTION longestSubstring(s, k)
        IF LENGTH OF s LESS THAN k
            RETURN 0
        SET count TO FREQUENCY OF CHARACTERS IN s
        FOR EACH char IN count
            IF count[char] LESS THAN k
                SET results TO EMPTY LIST
                FOR EACH substring IN s SPLIT BY char
                    APPEND longestSubstring(substring, k) TO results
                RETURN MAXIMUM VALUE IN results
        RETURN LENGTH OF s
    END FUNCTION
END CLASS
