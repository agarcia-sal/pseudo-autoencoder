CLASS Solution
    FUNCTION countVowelPermutation(n)
        SET MOD TO 1000000007
        SET a, e, i, o, u TO 1, 1, 1, 1, 1
        FOR index FROM 1 TO n - 1
            SET a_next TO e + i + u
            SET e_next TO a + i
            SET i_next TO e + o
            SET o_next TO i
            SET u_next TO i + o
            SET a TO a_next MODULO MOD
            SET e TO e_next MODULO MOD
            SET i TO i_next MODULO MOD
            SET o TO o_next MODULO MOD
            SET u TO u_next MODULO MOD
        RETURN (a + e + i + o + u) MODULO MOD
    END FUNCTION
END CLASS
