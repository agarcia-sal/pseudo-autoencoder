CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO queue.pop left element
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION verticalTraversal(root)
        IF root IS None
            RETURN empty list
        END IF
        SET node_dict TO empty dictionary with list as default value
        SET min_col TO 0
        SET max_col TO 0

        FUNCTION dfs(node, row, col)
            IF node IS None
                RETURN
            END IF
            APPEND (row, node.val) TO node_dict[col]
            SET min_col TO minimum of min_col and col
            SET max_col TO maximum of max_col and col
            CALL dfs(node.left, row + 1, col - 1)
            CALL dfs(node.right, row + 1, col + 1)
        END FUNCTION

        CALL dfs(root, 0, 0)

        SET result TO empty list
        FOR col FROM min_col TO max_col
            SET sorted_nodes TO node_dict[col] sorted by row ascending THEN by value ascending
            APPEND list of values FROM sorted_nodes TO result
        END FOR

        RETURN result
    END FUNCTION
END CLASS
