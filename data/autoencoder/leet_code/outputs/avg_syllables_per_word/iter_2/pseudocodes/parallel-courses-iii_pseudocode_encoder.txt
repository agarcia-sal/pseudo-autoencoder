CLASS Solution
    FUNCTION minimumTime(n, relations, time)
        SET graph TO new mapping with default empty list
        FOR each pair (prevCourse, nextCourse) IN relations
            APPEND prevCourse TO graph[nextCourse]
        SET memo TO list of length (n + 1) filled with -1

        FUNCTION dfs(course)
            IF memo[course] NOT EQUAL TO -1 THEN
                RETURN memo[course]
            SET max_prerequisite_time TO 0
            FOR each prevCourse IN graph[course]
                SET current_time TO dfs(prevCourse)
                IF current_time GREATER THAN max_prerequisite_time THEN
                    SET max_prerequisite_time TO current_time
            SET memo[course] TO max_prerequisite_time PLUS time[course MINUS 1]
            RETURN memo[course]
        END FUNCTION

        SET total_minimum_time TO 0
        FOR course FROM 1 TO n
            SET current_course_time TO dfs(course)
            IF current_course_time GREATER THAN total_minimum_time THEN
                SET total_minimum_time TO current_course_time
        RETURN total_minimum_time
    END FUNCTION
END CLASS
