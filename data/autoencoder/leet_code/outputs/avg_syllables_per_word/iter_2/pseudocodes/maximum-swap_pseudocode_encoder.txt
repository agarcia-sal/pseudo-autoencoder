CLASS Solution
    FUNCTION maximumSwap(num)
        SET digits TO list of characters in string form of num
        INITIALIZE last AS empty dictionary
        FOR each index i and character d IN digits
            SET last[integer value of d] TO i
        FOR each index i and character d IN digits
            FOR d2 FROM 9 DOWN TO integer value of d PLUS 1
                IF last CONTAINS d2 AND last[d2] > i
                    SWAP digits[i] WITH digits[last[d2]]
                    RETURN integer value of concatenation of digits
        RETURN num
    END FUNCTION
END CLASS
