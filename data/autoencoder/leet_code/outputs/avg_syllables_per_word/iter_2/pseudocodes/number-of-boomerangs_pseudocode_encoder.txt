CLASS Solution
    FUNCTION numberOfBoomerangs(points)
        SET ans TO 0
        FOR each p1 IN points
            SET cnt TO empty counter
            FOR each p2 IN points
                SET dx TO p1 at index 0 MINUS p2 at index 0
                SET dy TO p1 at index 1 MINUS p2 at index 1
                SET d TO dx SQUARED PLUS dy SQUARED
                INCREMENT ans BY cnt at key d
                INCREMENT cnt at key d BY 1
        RETURN ans MULTIPLIED BY 2
    END FUNCTION
END CLASS
