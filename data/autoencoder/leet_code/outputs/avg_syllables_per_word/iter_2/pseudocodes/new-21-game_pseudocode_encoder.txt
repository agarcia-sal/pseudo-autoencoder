CLASS Solution
    FUNCTION new21Game(n, k, maxPts)
        FUNCTION dfs(i)
            IF i IS GREATER THAN OR EQUAL TO k THEN
                RETURN 1 IF i IS LESS THAN OR EQUAL TO n ELSE 0
            END IF
            IF i EQUALS k MINUS 1 THEN
                SET value TO MINIMUM OF (n MINUS k PLUS 1) AND maxPts DIVIDED BY maxPts
                RETURN value
            END IF
            RETURN dfs(i PLUS 1) PLUS (dfs(i PLUS 1) MINUS dfs(i PLUS maxPts PLUS 1)) DIVIDED BY maxPts
        END FUNCTION
        RETURN dfs(0)
    END FUNCTION
END CLASS
