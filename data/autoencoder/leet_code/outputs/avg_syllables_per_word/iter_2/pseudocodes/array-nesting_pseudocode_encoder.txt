CLASS Solution
    FUNCTION arrayNesting(nums)
        SET visited TO list of False with length of nums
        SET max_length TO 0
        FOR i FROM 0 TO length of nums MINUS 1
            IF visited at i IS False THEN
                SET current_length TO 0
                SET k TO i
                WHILE visited at k IS False
                    SET visited at k TO True
                    SET k TO nums at k
                    INCREMENT current_length BY 1
                END WHILE
                SET max_length TO maximum of max_length and current_length
            END IF
        END FOR
        RETURN max_length
    END FUNCTION
END CLASS
