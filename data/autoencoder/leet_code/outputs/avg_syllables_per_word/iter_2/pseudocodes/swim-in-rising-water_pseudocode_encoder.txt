CLASS Solution
    FUNCTION swimInWater(grid: List of List of integers) RETURNS integer
        SET n TO LENGTH OF grid
        SET visited TO empty set
        SET min_heap TO list containing tuple (grid[0][0], 0, 0)
        ADD (0, 0) TO visited

        SET directions TO list of tuples [(0, 1), (1, 0), (0, -1), (-1, 0)]

        WHILE min_heap is not empty
            REMOVE smallest element FROM min_heap AND UNPACK INTO t, i, j
            IF i EQUALS n MINUS 1 AND j EQUALS n MINUS 1
                RETURN t
            
            FOR each di, dj IN directions
                SET ni TO i PLUS di
                SET nj TO j PLUS dj
                IF ni GREATER THAN OR EQUAL TO 0 AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO 0 AND nj LESS THAN n AND (ni, nj) NOT IN visited
                    ADD (ni, nj) TO visited
                    ADD tuple (MAXIMUM OF t AND grid[ni][nj], ni, nj) TO min_heap
        
        RETURN -1
    END FUNCTION
END CLASS
