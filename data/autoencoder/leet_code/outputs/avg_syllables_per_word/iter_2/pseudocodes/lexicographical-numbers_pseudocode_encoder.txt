CLASS Solution
    FUNCTION lexicalOrder(n)
        SET result TO empty list

        FUNCTION dfs(current)
            IF current > n THEN
                RETURN
            END IF
            APPEND current TO result
            FOR i FROM 0 TO 9
                SET next_num TO current MULTIPLIED BY 10 PLUS i
                IF next_num > n THEN
                    RETURN
                END IF
                CALL dfs(next_num)
            END FOR
        END FUNCTION

        FOR i FROM 1 TO 9
            CALL dfs(i)
        END FOR

        RETURN result
    END FUNCTION
END CLASS
