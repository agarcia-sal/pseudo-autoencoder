CLASS Solution
    FUNCTION letterCasePermutation(s)
        FUNCTION dfs(i)
            IF i >= LENGTH OF t THEN
                APPEND CONCATENATION OF t TO ans
                RETURN
            END IF
            CALL dfs(i + 1)
            IF t[i] IS ALPHABETIC THEN
                TOGGLE CASE OF t[i]
                CALL dfs(i + 1)
            END IF
        END FUNCTION

        SET t TO LIST OF CHARACTERS IN s
        SET ans TO EMPTY LIST
        CALL dfs(0)
        RETURN ans
    END FUNCTION
END CLASS
