CLASS Solution
    FUNCTION racecar(target)
        SET queue TO deque containing tuple (0, 1, 0)  # position, speed, steps
        SET visited TO set containing tuple (0, 1)
        
        WHILE queue is not empty
            SET position, speed, steps TO dequeue from queue
            
            IF position EQUALS target
                RETURN steps
            
            SET next_position TO position PLUS speed
            SET next_speed TO speed TIMES 2
            IF tuple (next_position, next_speed) NOT IN visited AND next_position GREATER OR EQUAL TO 0 AND next_position LESS THAN 2 TIMES target
                ADD tuple (next_position, next_speed) TO visited
                ENQUEUE tuple (next_position, next_speed, steps PLUS 1) TO queue
            
            IF speed GREATER THAN 0
                SET next_speed TO -1
            ELSE
                SET next_speed TO 1
            IF tuple (position, next_speed) NOT IN visited AND position GREATER OR EQUAL TO 0 AND position LESS THAN 2 TIMES target
                ADD tuple (position, next_speed) TO visited
                ENQUEUE tuple (position, next_speed, steps PLUS 1) TO queue
    END FUNCTION
END CLASS
