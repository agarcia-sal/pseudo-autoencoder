CLASS Solution
    FUNCTION majorityElement(nums)
        SET n1 TO 0
        SET n2 TO 0
        SET m1 TO 0
        SET m2 TO 1
        FOR each m IN nums
            IF m EQUALS m1
                INCREMENT n1 BY 1
            ELSE IF m EQUALS m2
                INCREMENT n2 BY 1
            ELSE IF n1 EQUALS 0
                SET m1 TO m
                SET n1 TO 1
            ELSE IF n2 EQUALS 0
                SET m2 TO m
                SET n2 TO 1
            ELSE
                DECREMENT n1 BY 1
                DECREMENT n2 BY 1
        SET result TO EMPTY LIST
        FOR each m IN [m1, m2]
            IF count of m IN nums GREATER THAN LENGTH OF nums DIVIDED BY 3
                APPEND m TO result
        RETURN result
    END FUNCTION
END CLASS
