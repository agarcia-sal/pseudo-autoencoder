CLASS Solution
    FUNCTION findDuplicate(paths)
        SET content_to_paths TO empty dictionary mapping to empty lists
        FOR each path IN paths
            SET parts TO SPLIT path BY spaces
            SET directory TO parts AT index 0
            FOR each file_info IN parts FROM index 1 TO end
                SPLIT file_info BY '(' INTO file_name AND content_with_paren
                SET content TO REMOVE last character FROM content_with_paren
                APPEND CONCATENATE directory, "/", file_name TO content_to_paths AT key content
        RETURN LIST OF path lists IN content_to_paths WHERE LENGTH OF path list IS GREATER THAN 1
    END FUNCTION
END CLASS
