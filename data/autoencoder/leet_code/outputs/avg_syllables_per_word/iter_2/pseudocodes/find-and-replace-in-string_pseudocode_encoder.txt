CLASS Solution
    FUNCTION findReplaceString(s, indices, sources, targets)
        SET replacements TO sorted list of tuples (index, source, target) from indices, sources, targets
        SET result TO empty list
        SET prev_end TO 0
        FOR each (index, source, target) IN replacements
            APPEND substring of s from prev_end TO index TO result
            IF substring of s from index TO index plus length of source EQUALS source
                APPEND target TO result
            ELSE
                APPEND substring of s from index TO index plus length of source TO result
            SET prev_end TO index plus length of source
        APPEND substring of s from prev_end TO end TO result
        RETURN concatenation of all strings in result
    END FUNCTION
END CLASS
