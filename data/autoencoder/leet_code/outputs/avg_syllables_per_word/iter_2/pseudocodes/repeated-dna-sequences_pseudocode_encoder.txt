CLASS Solution
    FUNCTION findRepeatedDnaSequences(s)
        SET sequence_count TO a dictionary with default integer values
        FOR i FROM 0 TO LENGTH OF s MINUS 10
            SET sequence TO substring of s FROM i TO i PLUS 10
            INCREMENT sequence_count[sequence] BY 1
        SET repeated_sequences TO an empty list
        FOR each seq, count IN sequence_count items
            IF count GREATER THAN 1
                APPEND seq TO repeated_sequences
        RETURN repeated_sequences
    END FUNCTION
END CLASS
