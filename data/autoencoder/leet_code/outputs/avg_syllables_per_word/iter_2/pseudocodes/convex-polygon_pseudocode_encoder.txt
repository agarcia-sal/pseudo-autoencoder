CLASS Solution
    FUNCTION isConvex(points)
        FUNCTION cross_product(o, a, b)
            RETURN (a[0] MINUS o[0]) TIMES (b[1] MINUS o[1]) MINUS (a[1] MINUS o[1]) TIMES (b[0] MINUS o[0])
        END FUNCTION

        SET n TO LENGTH OF points
        IF n LESS THAN OR EQUAL TO 3 THEN
            RETURN True
        END IF

        SET orientation TO 0

        FOR i FROM 0 TO n MINUS 1
            SET o TO points[i]
            SET a TO points[(i PLUS 1) MODULO n]
            SET b TO points[(i PLUS 2) MODULO n]
            SET cross TO cross_product(o, a, b)

            IF cross NOT EQUAL TO 0 THEN
                IF orientation EQUAL TO 0 THEN
                    IF cross GREATER THAN 0 THEN
                        SET orientation TO 1
                    ELSE
                        SET orientation TO -1
                    END IF
                ELSE IF (orientation EQUAL TO 1 AND cross LESS THAN 0) OR (orientation EQUAL TO -1 AND cross GREATER THAN 0) THEN
                    RETURN False
                END IF
            END IF
        END FOR

        RETURN True
    END FUNCTION
END CLASS
