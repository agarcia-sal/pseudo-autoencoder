CLASS Solution
    FUNCTION search(nums, target)
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        WHILE left LESS THAN OR EQUAL TO right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2
            IF nums AT mid EQUALS target
                RETURN mid
            END IF
            IF nums AT left LESS THAN OR EQUAL TO nums AT mid
                IF nums AT left LESS THAN OR EQUAL TO target AND target LESS THAN nums AT mid
                    SET right TO mid MINUS 1
                ELSE
                    SET left TO mid PLUS 1
                END IF
            ELSE
                IF nums AT mid LESS THAN target AND target LESS THAN OR EQUAL TO nums AT right
                    SET left TO mid PLUS 1
                ELSE
                    SET right TO mid MINUS 1
                END IF
            END IF
        END WHILE
        RETURN -1
    END FUNCTION
END CLASS
