CLASS Solution
    FUNCTION findContestMatch(n)
        SET teams TO list of strings from "1" to string of n
        WHILE n GREATER THAN 1
            FOR i FROM 0 TO n DIVIDED BY 2 MINUS 1
                SET teams[i] TO "(" CONCATENATED WITH teams[i] CONCATENATED WITH "," CONCATENATED WITH teams[n MINUS 1 MINUS i] CONCATENATED WITH ")"
            SET n TO n DIVIDED BY 2
        RETURN teams[0]
    END FUNCTION
END CLASS
