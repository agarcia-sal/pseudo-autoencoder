CLASS Solution
    FUNCTION judgePoint24(cards: List of Integer) RETURNS Boolean
        FUNCTION solve(nums: List of Number) RETURNS Boolean
            IF LENGTH OF nums EQUALS 1 THEN
                RETURN absolute value of (nums[0] MINUS 24) LESS THAN 0.000001
            END IF

            FOR i FROM 0 TO LENGTH OF nums MINUS 1 DO
                FOR j FROM 0 TO LENGTH OF nums MINUS 1 DO
                    IF i NOT EQUAL TO j THEN
                        SET new_nums TO EMPTY LIST
                        FOR k FROM 0 TO LENGTH OF nums MINUS 1 DO
                            IF k NOT EQUAL TO i AND k NOT EQUAL TO j THEN
                                APPEND nums[k] TO new_nums
                            END IF
                        END FOR

                        FOR op IN (division, multiplication, addition, subtraction) DO
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i THEN
                                CONTINUE TO NEXT op
                            END IF

                            IF op IS NOT division OR nums[j] NOT EQUAL TO 0 THEN
                                APPEND op(nums[i], nums[j]) TO new_nums
                                IF solve(new_nums) RETURNS TRUE THEN
                                    RETURN TRUE
                                END IF
                                REMOVE LAST ELEMENT FROM new_nums
                            END IF
                        END FOR
                    END IF
                END FOR
            END FOR

            RETURN FALSE
        END FUNCTION

        IMPORT division, multiplication, addition, subtraction OPERATORS
        RETURN solve(cards)
    END FUNCTION
END CLASS
