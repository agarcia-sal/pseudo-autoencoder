CLASS Solution
    FUNCTION splitIntoFibonacci(num)
        FUNCTION backtrack(start, path)
            IF start EQUALS LENGTH OF num AND LENGTH OF path GREATER THAN OR EQUAL TO 3
                RETURN path
            FOR end FROM start + 1 TO LENGTH OF num + 1
                IF num AT start EQUALS '0' AND end GREATER THAN start + 1
                    CONTINUE
                SET next_num TO INTEGER OF SUBSTRING OF num FROM start TO end
                IF next_num GREATER THAN OR EQUAL TO 2 TO THE POWER OF 31
                    BREAK
                IF LENGTH OF path LESS THAN 2 OR next_num EQUALS path AT -1 PLUS path AT -2
                    SET result TO backtrack(end, path APPENDED WITH next_num)
                    IF result IS NOT EMPTY
                        RETURN result
            RETURN EMPTY LIST
        END FUNCTION
        RETURN backtrack(0, EMPTY LIST)
    END FUNCTION
END CLASS
