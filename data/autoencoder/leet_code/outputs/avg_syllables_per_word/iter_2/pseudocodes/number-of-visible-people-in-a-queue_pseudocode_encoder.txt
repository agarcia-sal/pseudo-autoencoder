CLASS Solution
    FUNCTION canSeePersonsCount(heights)
        SET n TO LENGTH OF heights
        SET answer TO list of zeros with length n
        SET stack TO empty list

        FOR i FROM 0 TO n - 1
            WHILE stack IS NOT empty AND heights AT top of stack IS LESS THAN heights AT i
                SET index TO POP from stack
                INCREMENT answer AT index BY 1
            IF stack IS NOT empty
                INCREMENT answer AT top of stack BY 1
            APPEND i TO stack

        RETURN answer
