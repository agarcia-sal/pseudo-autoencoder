CLASS ListNode
    FUNCTION __init__(val = 0, next = None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET head TO ListNode(values[0])
    SET p TO head
    FOR each val IN values starting from second element
        SET node TO ListNode(val)
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS Solution
    FUNCTION removeNthFromEnd(head, n)
        SET dummy TO ListNode(0)
        SET dummy.next TO head
        SET first TO dummy
        SET second TO dummy
        FOR i FROM 0 TO n
            SET first TO first.next
        END FOR
        WHILE first IS NOT None
            SET first TO first.next
            SET second TO second.next
        END WHILE
        SET second.next TO second.next.next
        RETURN dummy.next
    END FUNCTION
END CLASS
