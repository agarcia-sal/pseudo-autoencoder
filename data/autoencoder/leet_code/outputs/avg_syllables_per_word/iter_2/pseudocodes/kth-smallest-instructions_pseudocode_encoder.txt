CLASS Solution
    FUNCTION kthSmallestPath(destination, k)
        SET row TO destination[0]
        SET col TO destination[1]
        SET path TO empty list
        SET total_moves TO col + row
        FOR i FROM 0 TO total_moves - 1
            IF col > 0 THEN
                SET paths_with_H TO combination(row + col - 1, col - 1)
                IF k <= paths_with_H THEN
                    APPEND 'H' TO path
                    DECREMENT col BY 1
                ELSE
                    APPEND 'V' TO path
                    DECREMENT row BY 1
                    DECREMENT k BY paths_with_H
            ELSE
                APPEND 'V' TO path
                DECREMENT row BY 1
        RETURN concatenation of elements in path
    END FUNCTION
END CLASS
