CLASS Solution
    FUNCTION findPaths(m, n, maxMove, startRow, startColumn)
        SET MOD TO 10^9 + 7
        SET directions TO LIST OF (0, 1), (0, -1), (1, 0), (-1, 0)
        INITIALIZE dp AS 3D LIST WITH DIMENSIONS m x n x (maxMove + 1) FILLED WITH 0
        
        FOR k FROM 1 TO maxMove
            FOR i FROM 0 TO m - 1
                FOR j FROM 0 TO n - 1
                    FOR EACH (di, dj) IN directions
                        SET ni TO i + di
                        SET nj TO j + dj
                        IF ni LESS THAN 0 OR ni GREATER OR EQUAL TO m OR nj LESS THAN 0 OR nj GREATER OR EQUAL TO n
                            SET dp[i][j][k] TO (dp[i][j][k] + 1) MODULO MOD
                        ELSE
                            SET dp[i][j][k] TO (dp[i][j][k] + dp[ni][nj][k - 1]) MODULO MOD
        
        RETURN dp[startRow][startColumn][maxMove]
    END FUNCTION
END CLASS
