CLASS Solution
    FUNCTION maxSubArray(nums)
        SET max_sum TO nums AT INDEX 0
        SET current_sum TO nums AT INDEX 0
        FOR each num IN nums FROM INDEX 1 TO END
            SET current_sum TO MAXIMUM OF num AND (current_sum PLUS num)
            SET max_sum TO MAXIMUM OF max_sum AND current_sum
        RETURN max_sum
    END FUNCTION
END CLASS
