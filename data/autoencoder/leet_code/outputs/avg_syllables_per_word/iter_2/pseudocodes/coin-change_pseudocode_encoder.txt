CLASS Solution
    FUNCTION coinChange(coins: List[int], amount: int) RETURNS int
        SET dp TO LIST OF size (amount + 1) WITH EACH ELEMENT SET TO (amount + 1)
        SET dp[0] TO 0
        FOR a FROM 1 TO amount
            FOR coin IN coins
                IF a MINUS coin GREATER OR EQUAL TO 0
                    SET dp[a] TO MINIMUM OF dp[a] AND (dp[a MINUS coin] PLUS 1)
        IF dp[amount] EQUALS amount PLUS 1
            RETURN -1
        ELSE
            RETURN dp[amount]
    END FUNCTION
END CLASS
