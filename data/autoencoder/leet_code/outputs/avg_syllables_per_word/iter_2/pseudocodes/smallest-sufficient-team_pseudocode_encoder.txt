CLASS Solution
    FUNCTION smallestSufficientTeam(req_skills, people)
        SET d TO dictionary mapping each skill in req_skills to its index
        SET m TO length of req_skills
        SET n TO length of people
        SET p TO list of length n filled with 0

        FOR i FROM 0 TO n - 1
            FOR each skill s IN people[i]
                SET p[i] TO p[i] BITWISE OR (1 LEFT SHIFTED BY d[s])

        SET f TO list of length (1 LEFT SHIFTED BY m) filled with infinity
        SET g TO list of length (1 LEFT SHIFTED BY m) filled with 0
        SET h TO list of length (1 LEFT SHIFTED BY m) filled with 0
        SET f[0] TO 0

        FOR i FROM 0 TO (1 LEFT SHIFTED BY m) - 1
            IF f[i] EQUALS infinity
                CONTINUE to next iteration
            FOR j FROM 0 TO n - 1
                IF f[i] + 1 LESS THAN f[i BITWISE OR p[j]]
                    SET f[i BITWISE OR p[j]] TO f[i] + 1
                    SET g[i BITWISE OR p[j]] TO j
                    SET h[i BITWISE OR p[j]] TO i

        SET i TO (1 LEFT SHIFTED BY m) - 1
        SET ans TO empty list
        WHILE i IS NOT 0
            APPEND g[i] TO ans
            SET i TO h[i]
        RETURN ans
    END FUNCTION
END CLASS
