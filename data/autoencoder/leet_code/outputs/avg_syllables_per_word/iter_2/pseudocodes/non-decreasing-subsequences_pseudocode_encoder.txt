CLASS Solution
    FUNCTION findSubsequences(nums)
        FUNCTION dfs(u, last, t)
            IF u EQUALS LENGTH OF nums THEN
                IF LENGTH OF t GREATER THAN 1 THEN
                    APPEND COPY OF t TO ans
                END IF
                RETURN
            END IF
            IF nums[u] GREATER THAN OR EQUAL TO last THEN
                APPEND nums[u] TO t
                CALL dfs(u + 1, nums[u], t)
                REMOVE LAST ELEMENT FROM t
            END IF
            IF nums[u] NOT EQUAL TO last THEN
                CALL dfs(u + 1, last, t)
            END IF
        END FUNCTION
        SET ans TO EMPTY LIST
        CALL dfs(0, -1000, EMPTY LIST)
        RETURN ans
    END FUNCTION
END CLASS
