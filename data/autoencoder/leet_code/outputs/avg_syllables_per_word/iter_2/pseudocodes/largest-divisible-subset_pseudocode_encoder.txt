CLASS Solution
    FUNCTION largestDivisibleSubset(nums)
        SORT nums
        SET n TO LENGTH OF nums
        SET f TO LIST OF 1 REPEATED n TIMES
        SET k TO 0
        FOR i FROM 0 TO n - 1
            FOR j FROM 0 TO i - 1
                IF nums[i] MODULO nums[j] EQUALS 0
                    SET f[i] TO MAXIMUM OF f[i] AND f[j] + 1
            IF f[k] LESS THAN f[i]
                SET k TO i
        SET m TO f[k]
        SET i TO k
        SET ans TO EMPTY LIST
        WHILE m GREATER THAN 0
            IF nums[k] MODULO nums[i] EQUALS 0 AND f[i] EQUALS m
                APPEND nums[i] TO ans
                SET k TO i
                DECREMENT m BY 1
            DECREMENT i BY 1
        RETURN ans
    END FUNCTION
END CLASS
