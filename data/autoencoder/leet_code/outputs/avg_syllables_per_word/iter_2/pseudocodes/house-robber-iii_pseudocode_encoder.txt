CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    IF p IS None OR q IS None
        RETURN False
    IF p.val != q.val
        RETURN False
    RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION rob(self, root: Optional[TreeNode]) RETURNS int
        FUNCTION rob_from_node(node)
            IF node IS None
                RETURN 0, 0
            SET left_robbed, left_not_robbed TO rob_from_node(node.left)
            SET right_robbed, right_not_robbed TO rob_from_node(node.right)
            SET rob_this_node TO node.val + left_not_robbed + right_not_robbed
            SET not_rob_this_node TO MAXIMUM OF left_robbed, left_not_robbed PLUS MAXIMUM OF right_robbed, right_not_robbed
            RETURN rob_this_node, not_rob_this_node
        END FUNCTION
        RETURN MAXIMUM OF rob_from_node(root)
    END FUNCTION
END CLASS
