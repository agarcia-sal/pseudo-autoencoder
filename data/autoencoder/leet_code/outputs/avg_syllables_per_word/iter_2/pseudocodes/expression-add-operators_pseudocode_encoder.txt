CLASS Solution
    FUNCTION addOperators(num, target)
        FUNCTION dfs(index, path, value, prev)
            IF index EQUALS LENGTH OF num THEN
                IF value EQUALS target THEN
                    APPEND path TO result
                RETURN
            END IF
            
            FOR i FROM index + 1 TO LENGTH OF num + 1
                SET operand TO SUBSTRING OF num FROM index TO i
                IF LENGTH OF operand > 1 AND FIRST CHARACTER OF operand EQUALS '0' THEN
                    CONTINUE TO NEXT i
                END IF
                
                IF index EQUALS 0 THEN
                    CALL dfs(i, operand, CONVERT operand TO INTEGER, CONVERT operand TO INTEGER)
                ELSE
                    CALL dfs(i, path + '+' + operand, value + CONVERT operand TO INTEGER, CONVERT operand TO INTEGER)
                    CALL dfs(i, path + '-' + operand, value - CONVERT operand TO INTEGER, NEGATIVE OF CONVERT operand TO INTEGER)
                    CALL dfs(i, path + '*' + operand, value - prev + prev * CONVERT operand TO INTEGER, prev * CONVERT operand TO INTEGER)
                END IF
            END FOR
        END FUNCTION
        
        SET result TO EMPTY LIST
        CALL dfs(0, EMPTY STRING, 0, 0)
        RETURN result
    END FUNCTION
END CLASS
