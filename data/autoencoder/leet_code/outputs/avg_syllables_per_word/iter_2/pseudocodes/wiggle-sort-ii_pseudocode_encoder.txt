CLASS Solution
    FUNCTION wiggleSort(nums: List of integers) RETURNS None
        CALL sort ON nums
        SET sorted_nums TO COPY of nums
        SET n TO LENGTH of nums
        SET left TO (n MINUS 1) DIVIDED BY 2
        SET right TO n MINUS 1
        FOR i FROM 0 TO n MINUS 1
            IF i MODULO 2 EQUALS 0 THEN
                SET nums[i] TO sorted_nums[left]
                DECREMENT left BY 1
            ELSE
                SET nums[i] TO sorted_nums[right]
                DECREMENT right BY 1
            END IF
        END FOR
    END FUNCTION
END CLASS
