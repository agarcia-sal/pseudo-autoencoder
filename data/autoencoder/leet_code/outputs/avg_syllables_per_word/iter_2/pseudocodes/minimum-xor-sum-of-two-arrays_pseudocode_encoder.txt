CLASS Solution
    FUNCTION minimumXORSum(nums1, nums2)
        SET n TO LENGTH OF nums1
        SET dp TO LIST OF SIZE (2 TO THE POWER n) FILLED WITH infinity
        SET dp[0] TO 0

        FOR mask FROM 0 TO (2 TO THE POWER n) MINUS 1 INCLUSIVE
            SET bit_count TO NUMBER OF ONES IN BINARY REPRESENTATION OF mask

            FOR j FROM 0 TO n MINUS 1
                IF mask HAS BIT j SET
                    SET dp[mask] TO MINIMUM OF dp[mask] AND (dp[mask WITH BIT j CLEARED] PLUS (nums1[bit_count MINUS 1] XOR nums2[j]))

        RETURN dp[(2 TO THE POWER n) MINUS 1]
    END FUNCTION
END CLASS
