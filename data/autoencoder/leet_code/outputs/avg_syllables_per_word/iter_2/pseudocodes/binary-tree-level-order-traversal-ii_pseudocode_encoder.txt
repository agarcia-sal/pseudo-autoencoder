CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO new TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO new TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO new TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION levelOrderBottom(root)
        IF root IS None
            RETURN empty list
        END IF
        SET result TO empty list
        SET queue TO deque containing root
        WHILE queue IS NOT empty
            SET level_size TO LENGTH OF queue
            SET current_level TO empty list
            FOR index FROM 0 TO level_size MINUS 1
                SET node TO queue.popleft()
                APPEND node.val TO current_level
                IF node.left IS NOT None
                    APPEND node.left TO queue
                END IF
                IF node.right IS NOT None
                    APPEND node.right TO queue
                END IF
            END FOR
            APPEND current_level TO result
        END WHILE
        RETURN result REVERSED
    END FUNCTION
END CLASS
