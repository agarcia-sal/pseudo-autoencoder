CLASS Solution
    FUNCTION pourWater(heights, volume, k)
        SET n TO LENGTH OF heights
        
        FOR each unit FROM 1 TO volume
            SET i TO k
            WHILE i > 0 AND heights[i - 1] <= heights[i]
                DECREMENT i BY 1
            WHILE i < k AND heights[i] == heights[i + 1]
                INCREMENT i BY 1
            IF i < k
                INCREMENT heights[i] BY 1
                CONTINUE to next unit
            
            SET i TO k
            WHILE i < n - 1 AND heights[i + 1] <= heights[i]
                INCREMENT i BY 1
            WHILE i > k AND heights[i] == heights[i - 1]
                DECREMENT i BY 1
            IF i > k
                INCREMENT heights[i] BY 1
                CONTINUE to next unit
            
            INCREMENT heights[k] BY 1
        
        RETURN heights
    END FUNCTION
END CLASS
