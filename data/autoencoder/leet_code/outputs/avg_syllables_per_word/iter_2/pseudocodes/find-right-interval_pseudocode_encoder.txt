CLASS Solution
    FUNCTION findRightInterval(intervals)
        SET start_times TO SORTED LIST OF (interval[0], index) FOR EACH index, interval IN ENUMERATE(intervals)
        SET result TO EMPTY LIST
        FOR EACH interval IN intervals
            SET end TO interval[1]
            SET idx TO POSITION FOUND BY BINARY SEARCH LEFT IN start_times FOR (end)
            IF idx IS LESS THAN LENGTH OF start_times
                APPEND start_times[idx][1] TO result
            ELSE
                APPEND -1 TO result
        RETURN result
    END FUNCTION
END CLASS
