CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO new TreeNode(values[0])
    SET i TO 1
    SET queue TO new empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue POPLEFT
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO new TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO new TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION isValidBST(self, root: Optional[TreeNode])
        FUNCTION validate(node, low = negative infinity, high = positive infinity)
            IF node IS None
                RETURN True
            IF node.val LESS THAN OR EQUAL TO low OR node.val GREATER THAN OR EQUAL TO high
                RETURN False
            RETURN validate(node.left, low, node.val) AND validate(node.right, node.val, high)
        END FUNCTION
        RETURN validate(root)
    END FUNCTION
END CLASS
