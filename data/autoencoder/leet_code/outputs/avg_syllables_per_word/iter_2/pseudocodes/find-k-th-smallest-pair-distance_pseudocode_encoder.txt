CLASS Solution
    FUNCTION smallestDistancePair(nums, k)
        SORT nums
        SET n TO LENGTH OF nums

        FUNCTION count(mid)
            SET count TO 0
            SET left TO 0
            FOR right FROM 0 TO n - 1
                WHILE nums[right] MINUS nums[left] GREATER THAN mid
                    INCREMENT left BY 1
                INCREMENT count BY right MINUS left
            RETURN count
        END FUNCTION

        SET left TO 0
        SET right TO nums[n - 1] MINUS nums[0]
        WHILE left LESS THAN right
            SET mid TO (left PLUS right) DIVIDED BY 2
            IF count(mid) LESS THAN k
                SET left TO mid PLUS 1
            ELSE
                SET right TO mid
        RETURN left
END CLASS
