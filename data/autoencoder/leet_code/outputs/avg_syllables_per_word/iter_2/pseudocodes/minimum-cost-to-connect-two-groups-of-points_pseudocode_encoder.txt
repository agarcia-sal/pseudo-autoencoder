CLASS Solution
    FUNCTION connectTwoGroups(cost)
        SET size1 TO LENGTH OF cost
        SET size2 TO LENGTH OF cost[0]

        INITIALIZE dp AS 2D ARRAY WITH size1 + 1 ROWS AND 2^size2 COLUMNS FILLED WITH infinity
        SET dp[0][0] TO 0

        FOR i FROM 0 TO size1 - 1
            FOR mask FROM 0 TO 2^size2 - 1
                FOR j FROM 0 TO size2 - 1
                    SET new_mask TO mask BITWISE OR (1 SHIFT LEFT BY j)
                    SET dp[i + 1][new_mask] TO MINIMUM OF dp[i + 1][new_mask] AND dp[i][mask] PLUS cost[i][j]

                FOR j FROM 0 TO size2 - 1
                    IF mask BITWISE AND (1 SHIFT LEFT BY j) IS NOT 0
                        SET dp[i + 1][mask] TO MINIMUM OF dp[i + 1][mask] AND dp[i + 1][mask BITWISE XOR (1 SHIFT LEFT BY j)] PLUS cost[i][j]

        RETURN dp[size1][2^size2 - 1]
    END FUNCTION
END CLASS
