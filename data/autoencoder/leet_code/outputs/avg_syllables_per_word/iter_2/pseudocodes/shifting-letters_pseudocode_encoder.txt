CLASS Solution
    FUNCTION shiftingLetters(s, shifts)
        SET total_shifts TO 0
        FOR i FROM LENGTH OF shifts MINUS 1 DOWNTO 0
            INCREMENT total_shifts BY shifts[i]
            SET shifts[i] TO total_shifts
        SET result TO empty list
        FOR i FROM 0 TO LENGTH OF s MINUS 1
            SET new_char_code TO (ORDINAL VALUE OF s[i] MINUS ORDINAL VALUE OF 'a' PLUS shifts[i]) MODULO 26 PLUS ORDINAL VALUE OF 'a'
            SET new_char TO CHARACTER REPRESENTED BY new_char_code
            APPEND new_char TO result
        RETURN CONCATENATION OF CHARACTERS IN result
    END FUNCTION
END CLASS
