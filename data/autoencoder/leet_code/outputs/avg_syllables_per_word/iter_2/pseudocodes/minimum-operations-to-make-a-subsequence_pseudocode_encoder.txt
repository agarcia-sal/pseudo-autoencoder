CLASS Solution
    FUNCTION minOperations(target: List of Integer, arr: List of Integer) RETURNS Integer
        SET index_map TO an empty map
        FOR index FROM 0 TO LENGTH OF target MINUS 1
            SET index_map[target[index]] TO index
        END FOR

        SET transformed_arr TO an empty list
        FOR each value IN arr
            IF value IS IN index_map
                APPEND index_map[value] TO transformed_arr
            END IF
        END FOR

        SET lis TO an empty list
        FOR each num IN transformed_arr
            SET pos TO the leftmost position in lis WHERE num CAN BE INSERTED to maintain sorting order
            IF pos IS EQUAL TO LENGTH OF lis
                APPEND num TO lis
            ELSE
                SET lis[pos] TO num
            END IF
        END FOR

        SET lcs_length TO LENGTH OF lis
        RETURN LENGTH OF target MINUS lcs_length
    END FUNCTION
END CLASS
