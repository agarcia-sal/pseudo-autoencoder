CLASS Solution
    FUNCTION lengthLongestPath(input: string) RETURNS int
        SET lines TO SPLIT input BY newline
        SET stack TO empty list
        SET max_length TO 0
        
        FOR each line IN lines
            SET depth TO COUNT number of tab characters in line
            SET name TO substring of line starting after depth tab characters
            
            WHILE LENGTH of stack GREATER THAN depth
                REMOVE last element from stack
            
            IF stack is not empty THEN
                SET current_length TO last element of stack PLUS LENGTH of name PLUS 1
            ELSE
                SET current_length TO LENGTH of name
            
            IF name CONTAINS '.' THEN
                SET max_length TO maximum of max_length and current_length
            ELSE
                APPEND current_length TO stack
        
        RETURN max_length
    END FUNCTION
END CLASS
