CLASS Solution
    FUNCTION jobScheduling(startTime, endTime, profit)
        SET jobs TO list of tuples combining startTime, endTime, and profit sorted by each job's end time
        SET end_times TO list of end times extracted from jobs
        SET dp TO list of zeros with length equal to number of jobs plus one

        FOR i FROM 1 TO length of jobs
            SET j TO position found by binary searching right in end_times for jobs[i-1]'s start time
            SET dp[i] TO maximum of dp[i-1] and dp[j] plus jobs[i-1]'s profit

        RETURN last element in dp
    END FUNCTION
END CLASS
