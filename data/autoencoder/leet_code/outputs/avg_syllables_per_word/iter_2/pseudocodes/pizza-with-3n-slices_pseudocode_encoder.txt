CLASS Solution
    FUNCTION maxSizeSlices(slices)
        SET n TO LENGTH OF slices DIVIDED BY 3

        FUNCTION dp(start, end, k)
            IF k EQUALS 0 OR start GREATER THAN end THEN
                RETURN 0
            END IF
            SET pick TO slices[start] PLUS dp(start PLUS 2, end, k MINUS 1)
            SET skip TO dp(start PLUS 1, end, k)
            RETURN MAXIMUM OF pick AND skip
        END FUNCTION

        SET case1 TO dp(1, LENGTH OF slices MINUS 1, n)
        SET case2 TO dp(0, LENGTH OF slices MINUS 2, n)

        RETURN MAXIMUM OF case1 AND case2
    END FUNCTION
END CLASS
