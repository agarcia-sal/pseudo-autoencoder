CLASS Solution
    FUNCTION nearestPalindromic(n: string)
        SET length TO LENGTH OF n
        IF length EQUALS 1
            RETURN STRING REPRESENTATION OF (INTEGER VALUE OF n MINUS 1)
        
        SET candidates TO EMPTY SET
        
        ADD (10 RAISED TO length PLUS 1) TO candidates
        ADD (10 RAISED TO (length MINUS 1) MINUS 1) TO candidates
        
        SET prefix TO INTEGER VALUE OF SUBSTRING OF n FROM START TO (length PLUS 1) DIVIDED BY 2
        
        FOR i FROM -1 TO 1
            SET new_prefix TO STRING REPRESENTATION OF (prefix PLUS i)
            IF length MODULO 2 EQUALS 0
                SET candidate TO CONCATENATION OF new_prefix AND REVERSE OF new_prefix
            ELSE
                SET candidate TO CONCATENATION OF new_prefix AND REVERSE OF SUBSTRING OF new_prefix EXCLUDING LAST CHARACTER
            ADD INTEGER VALUE OF candidate TO candidates
        
        SET original_number TO INTEGER VALUE OF n
        REMOVE original_number FROM candidates IF IT EXISTS
        
        SET closest TO ELEMENT IN candidates WITH MINIMUM VALUE OF (ABSOLUTE DIFFERENCE FROM original_number, THEN BY NUMERICAL VALUE)
        
        RETURN STRING REPRESENTATION OF closest
    END FUNCTION
END CLASS
