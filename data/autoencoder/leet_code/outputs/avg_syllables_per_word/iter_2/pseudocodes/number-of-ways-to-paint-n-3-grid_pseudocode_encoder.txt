CLASS Solution
    FUNCTION numOfWays(n)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7
        SET a TO 6
        SET b TO 6
        FOR index FROM 2 TO n INCLUSIVE
            SET new_a TO (3 TIMES a PLUS 2 TIMES b) MODULO MOD
            SET new_b TO (2 TIMES a PLUS 2 TIMES b) MODULO MOD
            SET a TO new_a
            SET b TO new_b
        RETURN (a PLUS b) MODULO MOD
    END FUNCTION
END CLASS
