CLASS Solution
    FUNCTION characterReplacement(s, k)
        SET max_length TO 0
        SET max_count TO 0
        SET left TO 0
        SET count TO empty map with default integer value
        
        FOR right FROM 0 TO LENGTH OF s MINUS 1
            INCREMENT count[s[right]] BY 1
            SET max_count TO MAXIMUM OF max_count AND count[s[right]]
            
            IF (right MINUS left PLUS 1) MINUS max_count GREATER THAN k THEN
                DECREMENT count[s[left]] BY 1
                INCREMENT left BY 1
            END IF
            
            SET max_length TO MAXIMUM OF max_length AND (right MINUS left PLUS 1)
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
