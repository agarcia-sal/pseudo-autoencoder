CLASS Solution
    FUNCTION nextGreaterElements(nums)
        SET n TO LENGTH OF nums
        SET result TO LIST WITH n ELEMENTS ALL SET TO -1
        SET stack TO EMPTY LIST

        FOR i FROM 0 TO 2 * n MINUS 1
            SET index TO i MODULO n
            WHILE stack IS NOT EMPTY AND nums AT POSITION stack LAST ELEMENT IS LESS THAN nums AT POSITION index
                SET popped_index TO REMOVE LAST ELEMENT FROM stack
                SET result AT POSITION popped_index TO nums AT POSITION index
            IF i IS LESS THAN n
                APPEND index TO stack

        RETURN result
    END FUNCTION
END CLASS
