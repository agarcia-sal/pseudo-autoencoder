CLASS Solution
    FUNCTION numSubarrayProductLessThanK(nums, k)
        IF k <= 1 THEN
            RETURN 0
        END IF

        SET product TO 1
        SET left TO 0
        SET count TO 0

        FOR right FROM 0 TO LENGTH OF nums MINUS 1
            SET product TO product MULTIPLIED BY nums[right]

            WHILE product >= k AND left <= right
                SET product TO product DIVIDED BY nums[left]
                INCREMENT left BY 1
            END WHILE

            INCREMENT count BY right MINUS left PLUS 1
        END FOR

        RETURN count
    END FUNCTION
END CLASS
