CLASS Solution
    FUNCTION mincostToHireWorkers(quality, wage, k)
        SET workers TO sorted list of pairs (wage divided by quality, quality) for each pair in wage and quality
        
        SET max_heap TO empty list
        SET total_quality TO 0
        SET min_cost TO infinity
        
        FOR each ratio, q IN workers
            PUSH negative q INTO max_heap
            INCREMENT total_quality BY q
            
            IF LENGTH OF max_heap > k THEN
                SET removed_quality TO POP from max_heap
                INCREMENT total_quality BY removed_quality
            
            IF LENGTH OF max_heap == k THEN
                SET cost TO total_quality multiplied by ratio
                IF cost < min_cost THEN
                    SET min_cost TO cost
        
        RETURN min_cost
    END FUNCTION
END CLASS
