CLASS Solution
    FUNCTION minimumDeviation(nums)
        SET max_heap TO empty list
        SET min_value TO infinity
        FOR each num IN nums
            IF num MODULO 2 IS 1 THEN
                SET num TO num MULTIPLIED BY 2
            END IF
            SET min_value TO the smaller of min_value and num
            ADD negative num TO max_heap
        END FOR

        SET min_deviation TO infinity

        WHILE top element of max_heap MODULO 2 IS 0
            SET max_value TO negative of removing top element from max_heap
            SET min_deviation TO the smaller of min_deviation and max_value MINUS min_value
            SET max_value TO max_value DIVIDED BY 2
            SET min_value TO the smaller of min_value and max_value
            ADD negative max_value TO max_heap
        END WHILE

        SET min_deviation TO the smaller of min_deviation and negative top element of max_heap MINUS min_value

        RETURN min_deviation
    END FUNCTION
END CLASS
