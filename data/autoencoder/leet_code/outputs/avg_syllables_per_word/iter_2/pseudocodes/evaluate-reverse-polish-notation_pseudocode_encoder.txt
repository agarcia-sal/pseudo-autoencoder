CLASS Solution
    FUNCTION evalRPN(tokens)
        SET stack TO empty list
        FOR each token IN tokens
            IF token IS one of "+", "-", "*", "/"
                SET b TO last element removed from stack
                SET a TO last element removed from stack
                IF token IS "+"
                    APPEND a plus b TO stack
                ELSE IF token IS "-"
                    APPEND a minus b TO stack
                ELSE IF token IS "*"
                    APPEND a multiplied by b TO stack
                ELSE IF token IS "/"
                    APPEND integer division of a divided by b TO stack
            ELSE
                APPEND integer conversion of token TO stack
        RETURN first element of stack
    END FUNCTION
END CLASS
