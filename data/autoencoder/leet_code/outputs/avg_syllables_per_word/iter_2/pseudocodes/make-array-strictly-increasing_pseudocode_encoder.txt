CLASS Solution
    FUNCTION makeArrayIncreasing(arr1, arr2)
        SORT arr2 IN ASCENDING ORDER
        SET dp TO DICTIONARY WITH KEY -1 AND VALUE 0
        
        FOR each num IN arr1
            SET new_dp TO EMPTY DICTIONARY
            FOR each last_num, ops IN dp
                IF num > last_num THEN
                    IF num NOT IN new_dp OR ops < new_dp[num] THEN
                        SET new_dp[num] TO ops
                END IF
                
                SET idx TO POSITION OF SMALLEST ELEMENT IN arr2 GREATER THAN last_num
                IF idx IS LESS THAN LENGTH OF arr2 THEN
                    SET replacement TO arr2 AT idx
                    IF replacement NOT IN new_dp OR ops + 1 < new_dp[replacement] THEN
                        SET new_dp[replacement] TO ops + 1
                    END IF
                END IF
            END FOR
            
            IF new_dp IS EMPTY THEN
                RETURN -1
            END IF
            
            SET dp TO new_dp
        END FOR
        
        RETURN MINIMUM VALUE AMONG dp VALUES
    END FUNCTION
END CLASS
