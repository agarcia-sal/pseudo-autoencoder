CLASS Solution
    FUNCTION search(nums, target)
        IF nums IS EMPTY
            RETURN False
        END IF

        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1

        WHILE left <= right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2

            IF nums[mid] EQUALS target
                RETURN True
            END IF

            IF nums[left] EQUALS nums[mid] AND nums[mid] EQUALS nums[right]
                INCREMENT left BY 1
                DECREMENT right BY 1
            ELSE IF nums[left] LESS THAN OR EQUAL TO nums[mid]
                IF nums[left] LESS THAN OR EQUAL TO target AND target LESS THAN nums[mid]
                    SET right TO mid MINUS 1
                ELSE
                    SET left TO mid PLUS 1
                END IF
            ELSE
                IF nums[mid] LESS THAN target AND target LESS THAN OR EQUAL TO nums[right]
                    SET left TO mid PLUS 1
                ELSE
                    SET right TO mid MINUS 1
                END IF
            END IF
        END WHILE

        RETURN False
    END FUNCTION
END CLASS
