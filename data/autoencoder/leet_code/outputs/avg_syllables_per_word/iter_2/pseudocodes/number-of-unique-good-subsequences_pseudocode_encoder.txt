CLASS Solution
    FUNCTION numberOfUniqueGoodSubsequences(binary)
        SET MOD TO 10 POWER 9 PLUS 7
        SET end_with_0 TO 0
        SET end_with_1 TO 0
        SET has_zero TO TRUE IF '0' IS IN binary OTHERWISE FALSE

        FOR EACH bit IN binary
            IF bit IS '1'
                SET end_with_1 TO (end_with_0 PLUS end_with_1 PLUS 1) MODULO MOD
            ELSE
                SET end_with_0 TO (end_with_0 PLUS end_with_1) MODULO MOD

        RETURN (end_with_0 PLUS end_with_1 PLUS has_zero) MODULO MOD
    END FUNCTION
END CLASS
