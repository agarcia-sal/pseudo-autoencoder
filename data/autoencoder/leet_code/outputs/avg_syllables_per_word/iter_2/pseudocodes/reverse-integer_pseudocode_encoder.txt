CLASS Solution
    FUNCTION reverse(x)
        IF x < 0 THEN
            SET sign TO -1
        ELSE
            SET sign TO 1
        SET reversed_x TO integer value of string representation of absolute value of x reversed
        SET result TO sign MULTIPLIED BY reversed_x
        IF result LESS THAN -2 TO THE POWER OF 31 OR result GREATER THAN 2 TO THE POWER OF 31 MINUS 1 THEN
            RETURN 0
        RETURN result
    END FUNCTION
END CLASS
