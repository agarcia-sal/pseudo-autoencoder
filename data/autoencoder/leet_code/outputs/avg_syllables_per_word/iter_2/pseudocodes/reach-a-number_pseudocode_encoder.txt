CLASS Solution
    FUNCTION reachNumber(target)
        SET target TO absolute value of target
        SET k TO 0
        SET sum_steps TO 0
        WHILE sum_steps < target
            INCREMENT k BY 1
            INCREMENT sum_steps BY k
        IF (sum_steps MINUS target) MODULO 2 EQUALS 0
            RETURN k
        ELSE
            IF k MODULO 2 EQUALS 0
                RETURN k PLUS 1
            ELSE
                RETURN k PLUS 2
            END IF
        END IF
    END FUNCTION
END CLASS
