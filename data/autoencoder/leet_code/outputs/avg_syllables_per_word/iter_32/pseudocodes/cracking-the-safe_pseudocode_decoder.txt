CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO empty set  
        SET sequence TO empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO node CONCATENATED WITH the string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH substring of edge FROM position one TO the end  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO a string consisting of the character zero REPEATED n MINUS one times  
        CALL dfs WITH start_node  
        RETURN concatenation of sequence in reversed order FOLLOWED BY start_node  
    END FUNCTION  
END CLASS
