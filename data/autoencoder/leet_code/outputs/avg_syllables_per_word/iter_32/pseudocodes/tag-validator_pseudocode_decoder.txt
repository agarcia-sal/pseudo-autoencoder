CLASS Solution  
    FUNCTION isValid(code)  
        IMPORT regular expression module as re  
        
        SET cdata_pattern TO string representing the pattern for CDATA sections  
        
        SET code TO the result of replacing all occurrences matching cdata_pattern in code with an empty string using regular expression substitution  
        
        INITIALIZE stack AS an empty list  
        
        SET i TO zero  
        WHILE i LESS THAN the length of code  
            IF the element at position i of code EQUALS the character less than  
                SET j TO the position of the character greater than in code starting the search from position i incremented by one  
                IF j EQUALS negative one  
                    RETURN False  
                END IF  
                
                SET tag_content TO the substring of code from position i incremented by one TO position j minus one  
                
                IF tag_content IS an empty string  
                    RETURN False  
                END IF  
                
                IF the element at position zero of tag_content EQUALS the character forward slash  
                    SET tag_name TO the substring of tag_content from position one TO the end  
                    IF stack IS empty OR the element at the last position of stack NOT EQUALS tag_name  
                        RETURN False  
                    END IF  
                    REMOVE the last element FROM stack  
                ELSE IF the element at position zero of tag_content EQUALS the character exclamation mark  
                    RETURN False  
                ELSE  
                    IF the length of tag_content LESS THAN one OR length of tag_content GREATER THAN nine OR tag_content IS NOT composed of only uppercase letters  
                        RETURN False  
                    END IF  
                    APPEND tag_content TO stack  
                END IF  
                
                SET i TO j PLUS one  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the logical negation of the condition that stack IS NOT empty AND the count of all matches in code of the pattern representing an uppercase tag surrounded by less than and greater than characters followed by any content and the corresponding closing tag IS GREATER THAN zero  
    END FUNCTION  
END CLASS
