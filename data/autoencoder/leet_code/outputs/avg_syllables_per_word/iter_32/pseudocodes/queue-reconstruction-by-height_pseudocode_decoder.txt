CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL sort_people_by_height_descending_and_k_ascending WITH people  
        SET queue TO empty list  
        FOR each person IN people  
            CALL insert_person_at_position_specified_by_k WITH queue AND person  
        END FOR  
        RETURN queue  
    END FUNCTION  

    FUNCTION sort_people_by_height_descending_and_k_ascending(people)  
        SORT people BY the value of element at position zero of each person MULTIPLIED BY negative one THEN BY the value of element at position one of each person IN ascending order  
    END FUNCTION  

    FUNCTION insert_person_at_position_specified_by_k(queue, person)  
        INSERT person INTO queue AT the position of element at position one of person  
    END FUNCTION  
END CLASS
