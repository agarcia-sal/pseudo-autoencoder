CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF DIGITS ONLY THEN  
                SET num TO CONVERSION OF block TO INTEGER  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND CONVERSION OF num TO STRING EQUALS block THEN  
                    RETURN TRUE  
                ELSE  
                    RETURN FALSE  
                END IF  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF LENGTH OF block GREATER THAN OR EQUAL TO one AND LENGTH OF block LESS THAN OR EQUAL TO four THEN  
                TRY  
                    ATTEMPT CONVERSION OF block FROM HEXADECIMAL TO INTEGER  
                    RETURN TRUE  
                CATCH CONVERSION FAILURE  
                    RETURN FALSE  
                END TRY  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO SPLIT queryIP BY the full stop character  
        SET parts_ipv6 TO SPLIT queryIP BY the colon character  

        IF LENGTH OF parts_ipv4 EQUALS four AND FOR EACH element IN parts_ipv4 IS_VALID_IPV4_BLOCK(element) IS TRUE THEN  
            RETURN String REPRESENTING Internet Protocol version four  
        ELSE IF LENGTH OF parts_ipv6 EQUALS eight AND FOR EACH element IN parts_ipv6 IS_VALID_IPV6_BLOCK(element) IS TRUE THEN  
            RETURN String REPRESENTING Internet Protocol version six  
        ELSE  
            RETURN String REPRESENTING Neither protocol  
        END IF  
    END FUNCTION  
END CLASS
