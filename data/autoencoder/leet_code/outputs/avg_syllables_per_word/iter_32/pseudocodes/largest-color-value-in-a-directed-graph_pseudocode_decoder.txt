CLASS Solution  
    FUNCTION largestPathValue(colors, edges)  
        SET n TO the LENGTH OF colors  
        ASSIGN reference TO an empty mapping with default EMPTY LIST FOR each key AS graph  
        SET in_degree TO a LIST OF zero OF LENGTH n  
        
        FOR each pair u AND v IN edges  
            APPEND v TO the LIST AT key u IN graph  
            INCREMENT the element at position v OF in_degree BY one  
        END FOR  
        
        SET queue TO a deque containing each integer i FROM zero TO n MINUS one WHERE the element at position i OF in_degree EQUALS zero  
        SET dp TO a LIST OF n ELEMENTS WHERE each ELEMENT IS a LIST OF twenty six zero VALUES  
        SET visited TO zero  
        SET max_color_value TO zero  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND ASSIGN TO node  
            INCREMENT the element at position (the Unicode code point of the element at position node OF colors MINUS the Unicode code point of lowercase a) OF dp[node] BY one  
            SET max_color_value TO the GREATER value BETWEEN max_color_value AND the element at position (the Unicode code point of the element at position node OF colors MINUS the Unicode code point of lowercase a) OF dp[node]  
            INCREMENT visited BY one  
            
            FOR each neighbor IN the LIST AT key node IN graph  
                FOR each integer c FROM zero TO twenty five  
                    SET the element at position c OF dp[neighbor] TO the GREATER value BETWEEN the element at position c OF dp[neighbor] AND the element at position c OF dp[node]  
                END FOR  
                DECREMENT the element at position neighbor OF in_degree BY one  
                IF the element at position neighbor OF in_degree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF visited EQUALS n  
            RETURN max_color_value  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
