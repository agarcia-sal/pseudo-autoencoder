CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        ASSIGN reference TO a new default dictionary with default integer value called count  
        
        FOR right FROM zero TO the length of s MINUS one  
            INCREMENT the element at position right of s in count BY one  
            SET max_count TO the greater value between max_count AND the element at position right of s in count  
            
            IF the difference between the value of right MINUS the value of left PLUS one MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the element at position left of s in count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the greater value between max_length AND the difference between right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
