CLASS Solution  
    FUNCTION dieSimulator(n, rollMax)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a three dimensional list with dimensions n PLUS one BY six BY the maximum value in rollMax PLUS one FILLED with zeros  
        
        FOR j FROM zero TO five  
            SET element at position one of the list at position j of the list at position one of dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM zero TO five  
                FOR k FROM one TO element at position j of rollMax  
                    IF k GREATER THAN one  
                        SET element at position k of the list at position j of the list at position i of dp TO the element at position k MINUS one of the list at position j of the list at position i MINUS one of dp  
                    ELSE  
                        SET sum_value TO zero  
                        FOR x FROM zero TO five  
                            IF x NOT EQUALS j  
                                FOR y FROM one TO element at position x of rollMax  
                                    INCREMENT sum_value BY the element at position y of the list at position x of the list at position i MINUS one of dp  
                                END FOR  
                            END IF  
                        END FOR  
                        SET element at position k of the list at position j of the list at position i of dp TO sum_value MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result_sum TO zero  
        FOR j FROM zero TO five  
            FOR k FROM one TO element at position j of rollMax  
                INCREMENT result_sum BY the element at position k of the list at position j of the list at position n of dp  
            END FOR  
        END FOR  
        RETURN result_sum MODULO MOD  
    END FUNCTION  
END CLASS
