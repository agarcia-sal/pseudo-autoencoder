CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET num_set TO the SET of elements IN nums  
        SET longest_streak TO zero  
        
        FOR each num IN num_set  
            IF the element obtained by the value of num MINUS one IS NOT IN num_set THEN  
                SET current_num TO num  
                SET current_streak TO one  
                
                WHILE the element obtained by the value of current_num PLUS one IS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                
                IF current_streak GREATER THAN longest_streak THEN  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
