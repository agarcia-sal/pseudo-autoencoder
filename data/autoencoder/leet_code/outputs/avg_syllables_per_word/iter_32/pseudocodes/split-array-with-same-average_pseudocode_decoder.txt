CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF nums  
        IF n EQUALS one  
            RETURN False  
        END IF  
        SET dp TO a list of empty sets with the size OF the INTEGER DIVISION of n BY two PLUS one  
        ADD zero TO the element at position zero OF dp  
        FOR each num IN nums  
            FOR size FROM the INTEGER DIVISION of n BY two DOWN TO one  
                FOR each s IN the element at position size MINUS one OF dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    ADD new_sum TO the element at position size OF dp  
                END FOR  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
