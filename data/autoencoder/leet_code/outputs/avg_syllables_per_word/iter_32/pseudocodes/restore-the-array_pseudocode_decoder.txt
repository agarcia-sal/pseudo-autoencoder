CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of s NOT EQUALS zero AND the INTEGER VALUE of substring from position j TO position i of s GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET element at position i of dp TO the REMAINDER AFTER DIVIDING the SUM of element at position i of dp PLUS element at position j of dp BY MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
