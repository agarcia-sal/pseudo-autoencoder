CLASS Solution  
    FUNCTION spiralMatrixIII(rows parameter TO integer, cols parameter TO integer, rStart parameter TO integer, cStart parameter TO integer) RETURNS list of list of integers  
        SET directions TO a list containing pairs representing moving zero steps vertically and one step horizontally, one step vertically and zero steps horizontally, zero steps vertically and negative one step horizontally, negative one step vertically and zero steps horizontally in this order  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO an empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols  
            IF the remainder of increment DIVIDED BY two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each underscore FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND a list containing x and y TO visited  
                END IF  
                SET dx TO the first element at position direction_index of directions  
                SET dy TO the second element at position direction_index of directions  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of the sum of direction_index AND one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
