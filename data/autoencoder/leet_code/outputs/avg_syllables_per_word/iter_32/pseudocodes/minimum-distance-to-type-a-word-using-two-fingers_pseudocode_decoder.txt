CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1 c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET first_index_x TO the INTEGER RESULT OF THE POSITION OF the character c1 IN THE ALPHABET MINUS the POSITION OF the character A DIVIDED BY six  
            SET first_index_y TO the REMAINDER OF THE POSITION OF the character c1 IN THE ALPHABET MINUS the POSITION OF the character A DIVIDED BY six  
            SET second_index_x TO the INTEGER RESULT OF THE POSITION OF the character c2 IN THE ALPHABET MINUS the POSITION OF the character A DIVIDED BY six  
            SET second_index_y TO the REMAINDER OF THE POSITION OF the character c2 IN THE ALPHABET MINUS the POSITION OF the character A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF first_index_x MINUS second_index_x PLUS the ABSOLUTE VALUE OF first_index_y MINUS second_index_y  
        END FUNCTION  
        
        FUNCTION dp(i f1 f2)  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET current_character TO the element at position i of word  
            SET distance_one TO the result of distance applied to parameters f1 and current_character PLUS the result of dp applied to parameters i PLUS one current_character f2  
            SET distance_two TO the result of distance applied to parameters f2 and current_character PLUS the result of dp applied to parameters i PLUS one f1 current_character  
            RETURN the MINIMUM OF distance_one AND distance_two  
        END FUNCTION  
        
        RETURN the result of dp applied to parameters zero None None  
    END FUNCTION  
END CLASS
