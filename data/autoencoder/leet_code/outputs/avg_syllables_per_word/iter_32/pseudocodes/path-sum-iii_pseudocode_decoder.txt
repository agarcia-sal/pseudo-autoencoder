CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A new TreeNode WITH val SET TO the element at position one of values AND left SET TO None AND right SET TO None  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element REMOVED FROM the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO A new TreeNode WITH val SET TO the element at position i of values AND left SET TO None AND right SET TO None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO A new TreeNode WITH val SET TO the element at position i of values AND left SET TO None AND right SET TO None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val property OF p NOT EQUALS the val property OF q THEN  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH parameters p.left AND q.left AND the RESULT OF is_same_tree WITH parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node IS None THEN  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS the val property OF node  
            SET paths_to_here TO THE VALUE ASSOCIATED WITH the KEY current_sum MINUS targetSum IN prefix_sums OR zero IF THE KEY DOES NOT EXIST  
            SET THE VALUE ASSOCIATED WITH the KEY current_sum IN prefix_sums TO THE CURRENT VALUE ASSOCIATED WITH current_sum IN prefix_sums OR zero IF THE KEY DOES NOT EXIST PLUS one  
            SET paths_in_left TO THE RESULT OF helper WITH PARAMETERS node.left, current_sum, prefix_sums  
            SET paths_in_right TO THE RESULT OF helper WITH PARAMETERS node.right, current_sum, prefix_sums  
            DECREMENT THE VALUE ASSOCIATED WITH the KEY current_sum IN prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        INITIALIZE prefix_sums AS A MAPPING WITH KEY zero ASSOCIATED WITH VALUE one  
        RETURN THE RESULT OF helper WITH PARAMETERS root, zero, prefix_sums  
    END FUNCTION  
END CLASS
