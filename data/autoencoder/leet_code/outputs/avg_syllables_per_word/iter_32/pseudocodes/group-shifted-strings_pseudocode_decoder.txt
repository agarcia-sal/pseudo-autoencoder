CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string THEN  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO numerical position of character c MINUS numerical position of character at position zero of string s  
                SET wrapped_difference TO difference MODULO twenty six  
                APPEND string representation of wrapped_difference TO canonical_form  
            END FOR  
            SET joined_form TO concatenation of elements of canonical_form SEPARATED BY the string consisting of a comma  
            RETURN joined_form  
        END FUNCTION  
        
        SET groups TO a mapping from keys to lists with default empty list for new keys  
        
        FOR each string s IN strings  
            SET canonical_form TO result of get_canonical_form with parameter s  
            APPEND s TO the list at key canonical_form in groups  
        END FOR  
        
        SET result TO list of all values in groups  
        RETURN result  
    END FUNCTION  
END CLASS
