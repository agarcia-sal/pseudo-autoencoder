CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND corresponding count v IN the items OF cnt  
                IF v GREATER THAN one  
                    DECREMENT the count OF c IN cnt BY two  
                    CALL dfs WITH the concatenation OF c PLUS t PLUS c AS argument  
                    INCREMENT the count OF c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  

        ASSIGN cnt TO the frequency count OF characters IN s  
        SET mid TO empty string  
        FOR each character c AND corresponding count v IN the items OF cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty string  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT the count OF c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH mid AS argument  
        RETURN ans  
    END FUNCTION  
END CLASS
