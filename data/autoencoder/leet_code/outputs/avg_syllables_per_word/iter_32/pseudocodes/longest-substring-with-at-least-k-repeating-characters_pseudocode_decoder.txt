CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the result of counting the frequency of each character in s  
        FOR each character IN the set of keys in count  
            IF the frequency of character IN count LESS THAN k  
                RETURN the maximum value among the results of calling longestSubstring with each substring obtained by splitting s at the character and with parameter k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
