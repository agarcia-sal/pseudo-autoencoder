CLASS Solution  
    FUNCTION convert(s string numRows integer) RETURNS string  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list consisting of numRows empty strings  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            SET element at position current_row of rows TO element at position current_row of rows CONCATENATED WITH character  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down EQUALS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result_string TO the empty string  
        FOR each row_string IN rows  
            SET result_string TO result_string CONCATENATED WITH row_string  
        END FOR  
        RETURN result_string  
    END FUNCTION  
END CLASS
