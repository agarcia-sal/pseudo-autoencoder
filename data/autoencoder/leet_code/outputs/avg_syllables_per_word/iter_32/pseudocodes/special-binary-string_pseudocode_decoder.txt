CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN elements of s  
                IF character char EQUALS character one THEN  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero THEN  
                    APPEND character one CONCATENATED WITH result of dfs called with substring from position start PLUS one TO position i CONCATENATED WITH character zero TO specials  
                    SET start TO index i PLUS one  
                END IF  
            END FOR  
            SORT specials IN reverse lexicographical order  
            RETURN concatenation of all elements in specials as a single sequence  
        END FUNCTION  
        RETURN result of dfs called with s  
    END FUNCTION  
END CLASS
