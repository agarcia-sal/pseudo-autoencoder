CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  
        
        SET s TO list containing one two two  
        SET count TO one  
        SET index TO two  
        
        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the element at position LENGTH OF s MINUS one of s  
            FOR counter FROM one TO the element at position index of s  
                APPEND next_num TO s  
            END FOR  
            IF next_num EQUALS one  
                INCREMENT count BY the element at position index of s  
            END IF  
            INCREMENT index BY one  
        END WHILE  
        
        IF the LENGTH OF s GREATER THAN n  
            DECREMENT count BY the element at position LENGTH OF s MINUS one of s MULTIPLIED BY the DIFFERENCE of the LENGTH OF s MINUS n  
        END IF  
        
        RETURN count  
    END FUNCTION  
END CLASS
