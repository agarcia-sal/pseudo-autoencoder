CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        SET heap TO blocks  
        PERFORM heapify OPERATION ON heap  
        WHILE the LENGTH OF heap GREATER THAN one  
            SET first TO the removal of the SMALLEST element FROM heap  
            SET second TO the removal of the SMALLEST element FROM heap  
            SET combined_time TO the value of split PLUS the value of second  
            ADD combined_time TO heap using heap push operation  
        END WHILE  
        RETURN element AT position one OF heap  
    END FUNCTION  
END CLASS
