CLASS Solution  
    FUNCTION exist(board word)  
        IF board EQUALS an empty collection THEN  
            RETURN false  
        END IF  
          
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a new collection of size m where each element is a collection of size n with all values set to false  
          
        FUNCTION dfs(x y index)  
            IF index EQUALS the LENGTH OF word THEN  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AND element at position y of that EQUALS true OR element at position x of board AND element at position y of that NOT EQUALS element at position index of word THEN  
                RETURN false  
            END IF  
              
            SET element at position x of visited AND element at position y of that TO true  
            FOR each pair dx dy IN the collection of pairs consisting of negative one and zero PLUS one and zero PLUS zero and negative one PLUS zero and one  
                IF dfs(x PLUS dx y PLUS dy index PLUS one) EQUALS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            SET element at position x of visited AND element at position y of that TO false  
            RETURN false  
        END FUNCTION  
          
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i j zero) EQUALS true THEN  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
