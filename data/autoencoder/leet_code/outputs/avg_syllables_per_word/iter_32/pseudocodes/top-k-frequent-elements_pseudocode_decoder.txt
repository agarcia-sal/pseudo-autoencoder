CLASS Solution  
    FUNCTION topKFrequent(nums k) RETURNS list  
        SET frequency_map TO a new default dictionary with default value zero  
        FOR each num IN nums  
            INCREMENT the value at key num in frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a list of empty lists with length max_frequency PLUS one  
        FOR each entry num freq IN frequency_map  
            APPEND num TO the element at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the LENGTH OF the element at position freq of buckets GREATER THAN zero THEN  
                EXTEND result BY all elements in the element at position freq of buckets  
                IF the LENGTH OF result GREATER THAN OR EQUAL TO k THEN  
                    RETURN a list consisting of elements from position one TO position k of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
