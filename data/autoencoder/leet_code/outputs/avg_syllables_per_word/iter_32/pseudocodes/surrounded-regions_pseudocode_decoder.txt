CLASS Solution  
    FUNCTION solve(board)  
        IF board EQUALS empty OR element at position zero of board EQUALS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        FUNCTION dfs(x, y)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of board AT position y NOT EQUALS character O  
                RETURN  
            END IF  
            SET element at position x of board AT position y TO character S  
            CALL dfs WITH parameters x PLUS one AND y  
            CALL dfs WITH parameters x MINUS one AND y  
            CALL dfs WITH parameters x AND y PLUS one  
            CALL dfs WITH parameters x AND y MINUS one  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            CALL dfs WITH parameters i AND zero  
            CALL dfs WITH parameters i AND n MINUS one  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            CALL dfs WITH parameters zero AND j  
            CALL dfs WITH parameters m MINUS one AND j  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board AT position j EQUALS character O  
                    SET element at position i of board AT position j TO character X  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board AT position j EQUALS character S  
                    SET element at position i of board AT position j TO character O  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
