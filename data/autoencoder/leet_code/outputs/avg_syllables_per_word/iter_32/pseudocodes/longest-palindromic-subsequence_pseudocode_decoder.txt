CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional array OF size n BY n WITH all elements SET TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND position i OF dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i AND position j OF dp TO element at position i PLUS one AND position j MINUS one OF dp PLUS two  
                ELSE  
                    IF element at position i PLUS one AND position j OF dp GREATER THAN element at position i AND position j MINUS one OF dp  
                        SET element at position i AND position j OF dp TO element at position i PLUS one AND position j OF dp  
                    ELSE  
                        SET element at position i AND position j OF dp TO element at position i AND position j MINUS one OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero AND position n MINUS one OF dp  
    END FUNCTION  
END CLASS
