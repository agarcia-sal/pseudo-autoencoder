CLASS Solution  
    FUNCTION myAtoi WITH PARAMETER s  
        SET maximum integer value TO two raised to the power of thirty ONE MINUS one  
        SET minimum integer value TO negative one MULTIPLIED BY two raised to the power of thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET length of s TO n  
        SET sign TO one  
        WHILE index LESS THAN n AND element at position index of s EQUALS a space character  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (element at position index of s EQUALS the plus character OR element at position index of s EQUALS the minus character)  
            IF element at position index of s EQUALS the minus character  
                SET sign TO negative one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND element at position index of s IS a digit character  
            SET digit TO numeric value of element at position index of s  
            IF result GREATER THAN (maximum integer value MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN maximum integer value  
                ELSE  
                    RETURN minimum integer value  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
