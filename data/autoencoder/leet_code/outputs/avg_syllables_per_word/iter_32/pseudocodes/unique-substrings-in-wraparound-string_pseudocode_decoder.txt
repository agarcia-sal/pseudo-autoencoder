CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a new dictionary with each lowercase letter of the English alphabet mapped TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the value of the character at position index of s MINUS the value of the character at position index MINUS one of s EQUALS one OR the character at position index of s EQUALS the character a AND the character at position index MINUS one of s EQUALS the character z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET the value at key of the character at position index of s in max_length TO the GREATER VALUE BETWEEN its current value AND current_length  
        END FOR  
        SET total_count TO zero  
        FOR each key IN max_length  
            INCREMENT total_count BY the value at key in max_length  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
