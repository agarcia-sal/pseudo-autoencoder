CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO empty dictionary mapping characters to integers  
        SET guess_count TO empty dictionary mapping characters to integers  
        
        FOR each pair s and g IN corresponding elements of secret and guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key s in secret_count BY one  
                INCREMENT the value at key g in guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN keys of secret_count  
            SET minimum_count TO the smaller value BETWEEN the value at digit in secret_count AND the value at digit in guess_count  
            INCREMENT cows BY minimum_count  
        END FOR  
        
        RETURN a string expressing bulls followed by letter A followed by cows followed by letter B  
    END FUNCTION  
END CLASS
