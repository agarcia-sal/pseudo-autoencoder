CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty set  
        FOR length FROM one TO the INTEGER PART OF n DIVIDED BY two INCLUSIVE  
            FOR start FROM zero TO the INTEGER PART OF n MINUS two MULTIPLIED BY length INCLUSIVE  
                SET a TO the substring of text from position start TO position start PLUS length MINUS one  
                SET b TO the substring of text from position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one  
                IF a EQUALS b  
                    APPEND the concatenation of a PLUS b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF substrings  
    END FUNCTION  
END CLASS
