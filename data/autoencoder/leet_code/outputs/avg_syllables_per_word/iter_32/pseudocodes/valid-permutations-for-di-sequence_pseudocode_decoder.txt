CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF s  
        SET dp TO a list WITH the LENGTH of n PLUS one WHERE each element IS a list WITH the LENGTH of n PLUS one FILLED WITH zeroes  
        
        FOR j FROM zero TO n  
            SET element at position j of element at position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the uppercase character I THEN  
                FOR j FROM zero TO n MINUS i  
                    SET prefix_sum TO prefix_sum PLUS element at position j of element at position i MINUS one of dp THEN TAKE THE REMAINDER AFTER DIVISION BY MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of element at position i MINUS one of dp THEN TAKE THE REMAINDER AFTER DIVISION BY MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each number IN element at position n of dp  
            SET result TO result PLUS number THEN TAKE THE REMAINDER AFTER DIVISION BY MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
