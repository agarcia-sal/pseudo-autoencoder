CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO an iterator over the characters of s  
            FOR each character IN word  
                IF character is NOT found in iterator in order  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
          
        SORT dictionary by the length of element in descending order AND by element in ascending lexicographical order  
          
        FOR each word IN dictionary  
            IF the result of calling is_subsequence with word AND s EQUALS true  
                RETURN word  
            END IF  
        END FOR  
          
        RETURN an empty string  
    END FUNCTION  
END CLASS
