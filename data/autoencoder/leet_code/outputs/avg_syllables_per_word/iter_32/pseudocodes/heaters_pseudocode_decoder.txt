CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT the list houses in ascending order  
        SORT the list heaters in ascending order  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN the LENGTH OF heaters MINUS one AND  
                  the ABSOLUTE VALUE of element at position j of heaters MINUS house GREATER THAN OR EQUAL TO  
                  the ABSOLUTE VALUE of element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO the GREATER VALUE BETWEEN min_radius AND the ABSOLUTE VALUE of element at position j of heaters MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
