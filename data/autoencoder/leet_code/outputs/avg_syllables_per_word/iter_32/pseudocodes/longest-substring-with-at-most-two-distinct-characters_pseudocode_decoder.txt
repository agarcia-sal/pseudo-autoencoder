CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        ASSIGN reference TO a new default dictionary of integers named char_count  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the value associated with the element at position right of s IN char_count BY one  
            WHILE the NUMBER OF keys IN char_count GREATER THAN two  
                DECREMENT the value associated with the element at position left of s IN char_count BY one  
                IF the value associated with the element at position left of s IN char_count EQUALS zero  
                    REMOVE the element at position left of s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
