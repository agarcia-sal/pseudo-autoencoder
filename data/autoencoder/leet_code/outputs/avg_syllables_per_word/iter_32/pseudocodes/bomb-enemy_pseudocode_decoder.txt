CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET max_kills TO zero  
        SET row_kills TO zero  
        SET col_kills TO a list of zero repeated n times  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF j EQUALS zero OR element at position j MINUS one of element at position i of grid EQUALS the character representing wall  
                    SET row_kills TO zero  
                    FOR k FROM j TO n MINUS one  
                        IF element at position k of element at position i of grid EQUALS the character representing enemy  
                            INCREMENT row_kills BY one  
                        ELSE IF element at position k of element at position i of grid EQUALS the character representing wall  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                IF i EQUALS zero OR element at position i MINUS one of element at position j of grid EQUALS the character representing wall  
                    SET element at position j of col_kills TO zero  
                    FOR k FROM i TO m MINUS one  
                        IF element at position k of element at position j of grid EQUALS the character representing enemy  
                            INCREMENT element at position j of col_kills BY one  
                        ELSE IF element at position k of element at position j of grid EQUALS the character representing wall  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                IF element at position j of element at position i of grid EQUALS the character representing empty  
                    SET max_kills TO the GREATER OF max_kills AND the SUM of row_kills AND element at position j of col_kills  
                END IF  
            END FOR  
        END FOR  
        RETURN max_kills  
    END FUNCTION  
END CLASS
