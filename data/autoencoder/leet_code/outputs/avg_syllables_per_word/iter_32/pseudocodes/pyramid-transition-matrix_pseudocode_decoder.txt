CLASS Solution  
    FUNCTION pyramidTransition(bottom, allowed)  
        SET allowed_map TO an empty mapping from single character TO a mapping from single character TO a list of single characters  
        FOR each rule IN allowed  
            SET left TO element at position zero of rule  
            SET right TO element at position one of rule  
            SET top TO element at position two of rule  
            APPEND top TO the list at allowed_map at key left at key right  
        END FOR  
        
        FUNCTION can_build_pyramid(current_bottom, current_top)  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN True  
            END IF  
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of can_build_pyramid called with current_top and an empty list  
            END IF  
            
            SET left TO element at position the LENGTH OF current_top of current_bottom  
            SET right TO element at position the LENGTH OF current_top PLUS one of current_bottom  
            IF allowed_map CONTAINS key left AND allowed_map at key left CONTAINS key right  
                FOR each top IN the list at allowed_map at key left at key right  
                    IF the result of can_build_pyramid called with current_bottom and the concatenation of current_top and top IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
            RETURN False  
        END FUNCTION  
        
        RETURN result of can_build_pyramid called with bottom and an empty list  
    END FUNCTION  
END CLASS
