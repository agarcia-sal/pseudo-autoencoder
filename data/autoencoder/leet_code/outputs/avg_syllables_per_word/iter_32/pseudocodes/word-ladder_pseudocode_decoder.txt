CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        SET word_dict TO an empty dictionary with default value as empty list  
        SET word_length TO the LENGTH OF beginWord  
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring from position zero TO position i MINUS one of word PLUS the string representing a special character PLUS substring from position i PLUS one TO position word_length MINUS one of word  
                APPEND word TO the list at key intermediate_word in word_dict  
            END FOR  
        END FOR  
        SET queue TO a double ended queue containing a tuple of beginWord and integer one  
        SET visited TO a set containing beginWord  
        WHILE queue is not empty  
            REMOVE the element from the left end of queue and unpack into current_word and current_length  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring from position zero TO position i MINUS one of current_word PLUS the string representing a special character PLUS substring from position i PLUS one TO position word_length MINUS one of current_word  
                FOR each word IN the list at key intermediate_word in word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
