CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO a list of pairs formed by combining each element of plantTime with the corresponding element of growTime  
        SORT tasks in order so that elements with the larger second item come before elements with the smaller second item  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair in tasks as plant and grow  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF earliest_bloom LESS THAN bloom_day  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
