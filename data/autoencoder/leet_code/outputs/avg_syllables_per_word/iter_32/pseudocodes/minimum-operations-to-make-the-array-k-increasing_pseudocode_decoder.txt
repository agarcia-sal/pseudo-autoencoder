CLASS Solution  
    FUNCTION kIncreasing(arr parameter_of_list_of_integer k parameter_of_integer) RETURNS integer  
        FUNCTION longest_increasing_subsequence(subarr parameter_of_list_of_integer) RETURNS integer  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis EQUALS empty list OR num GREATER THAN OR EQUAL TO element at position the last index of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position at which to insert num to maintain sorted order from the right in lis  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one STEP k  
                APPEND element at position i of arr TO subarr  
            END FOR  
            SET lis_length TO result of FUNCTION longest_increasing_subsequence WITH subarr AS ARGUMENT  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
