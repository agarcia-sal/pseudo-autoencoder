CLASS TreeNode  
    FUNCTION __init__(self, val SET TO zero, left SET TO none, right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF the LENGTH OF preorder EQUALS zero OR the LENGTH OF inorder EQUALS zero THEN  
            RETURN none  
        END IF  
        SET root_val TO element at position one of preorder  
        SET root TO new TreeNode with value root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root.left TO the result of calling self.buildTree with the sublist of preorder from position two TO position root_index_inorder PLUS one and the sublist of inorder from position one TO position root_index_inorder  
        SET root.right TO the result of calling self.buildTree with the sublist of preorder from position root_index_inorder PLUS two TO the end and the sublist of inorder from position root_index_inorder PLUS two TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
