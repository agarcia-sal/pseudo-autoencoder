CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 LIST OF INTEGER nums2 LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        FUNCTION count(p INTEGER) RETURNS INTEGER  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x GREATER THAN zero THEN  
                    INCREMENT cnt BY the POSITION AFTER THE LAST ELEMENT LESS THAN OR EQUAL TO the QUOTIENT OF p DIVIDED BY x IN nums2  
                ELSE IF x LESS THAN zero THEN  
                    INCREMENT cnt BY n MINUS THE POSITION OF THE FIRST ELEMENT NOT LESS THAN the QUOTIENT OF p DIVIDED BY x IN nums2  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY ONE IF p GREATER THAN OR EQUAL TO zero OTHERWISE zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION OF the FIRST ELEMENT NOT LESS THAN k IN the RANGE FROM the NEGATIVE OF mx TO mx INCLUSIVE USING count AS A KEY MINUS mx  
    END FUNCTION  
END CLASS
