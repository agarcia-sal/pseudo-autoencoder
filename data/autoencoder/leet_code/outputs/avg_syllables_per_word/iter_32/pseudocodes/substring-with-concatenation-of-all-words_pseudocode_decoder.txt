CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s EQUALS empty OR words EQUALS empty OR element at position zero of words EQUALS empty  
            RETURN empty list  
        END IF  
        
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO a new Counter initialized with words  
        SET result TO empty list  
        
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO a new Counter initialized empty  
            FOR j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO the substring from position j TO position j PLUS word_length MINUS one of s  
                INCREMENT count of word in seen BY one  
                IF count of word in seen GREATER THAN count of word in word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop was not broken  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
