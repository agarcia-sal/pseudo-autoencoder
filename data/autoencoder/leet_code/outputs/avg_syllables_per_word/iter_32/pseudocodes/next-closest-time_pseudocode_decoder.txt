CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time with the character colon REMOVED  
        SET hour TO the integer value of the substring from position zero TO position one of time MULTIPLIED BY ten PLUS the integer value of the substring at position three TO position four of time  
        SET minute TO the integer value of the substring from position three TO position four of time  

        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET hour_value TO the integer value of character h1 MULTIPLIED BY ten PLUS the integer value of character h2  
                            SET minute_value TO the integer value of character m1 MULTIPLIED BY ten PLUS the integer value of character m2  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty  
                                APPEND the tuple of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of generate_times with digits as argument  
        SET current_time_index TO the position in all_times where the tuple of hour and minute is found  

        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour TO the first element of the tuple at the position current_time_index PLUS one in all_times  
            SET next_minute TO the second element of the tuple at the position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour TO the first element of the tuple at position zero in all_times  
            SET next_minute TO the second element of the tuple at position zero in all_times  
        END IF  

        RETURN the string formed by the two digit representation of next_hour followed by the character colon followed by the two digit representation of next_minute  
    END FUNCTION  
END CLASS
