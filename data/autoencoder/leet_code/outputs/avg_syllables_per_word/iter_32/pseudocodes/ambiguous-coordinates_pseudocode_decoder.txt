CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the substring of frag from position one TO position d  
                SET right TO the substring of frag from position d PLUS one TO the LENGTH OF frag  
                IF the first character of left NOT EQUALS zero OR left EQUALS zero AND the last character of right NOT EQUALS zero THEN  
                    YIELD the concatenation of left PLUS if d NOT EQUALS n THEN a period PLUS right ELSE right  
                END IF  
            END FOR  
        END FUNCTION  
        SET s TO the substring of s from position two TO the LENGTH OF s MINUS one  
        RETURN a list composed of strings formed by an opening parenthesis PLUS a PLUS a comma PLUS a space PLUS b PLUS a closing parenthesis FOR i FROM one TO the LENGTH OF s MINUS one FOR each a, b IN the cartesian product of make of the substring of s from position one TO position i AND make of the substring of s from position i PLUS one TO the LENGTH OF s  
    END FUNCTION  
END CLASS
