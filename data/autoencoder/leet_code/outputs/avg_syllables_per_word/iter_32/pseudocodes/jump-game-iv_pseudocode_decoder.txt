CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO an empty mapping from values TO list of positions  
        FOR each position i FROM zero TO n MINUS one  
            SET value TO element at position i of arr  
            APPEND i TO the list associated with value in value_indices  
        END FOR  
        
        SET queue TO a double ended queue initialized with the pair of zero and zero representing current position zero and current steps zero  
        SET visited TO a set containing zero  
        
        WHILE the queue is not empty  
            REMOVE the leftmost element from queue AND ASSIGN TO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a list consisting of current_index PLUS one AND current_index MINUS one FOLLOWED BY the list of indices associated with the value at element at position current_index of arr in value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list associated with the value at element at position current_index of arr in value_indices TO an empty list  
        END WHILE  
        
        RETURN the negation of one  
    END FUNCTION  
END CLASS
