CLASS TrieNode  
    FUNCTION __init__()  
        SET children TO an empty collection of key value pairs  
        SET count TO zero  
    END FUNCTION  
END CLASS  

CLASS Trie  
    FUNCTION __init__(bit_length)  
        ASSIGN reference TO a new TrieNode TO root  
        SET bit_length TO bit_length  
    END FUNCTION  

    FUNCTION insert(num)  
        ASSIGN reference TO root TO node  
        FOR i FROM bit_length TO zero DECREMENTING BY one  
            SET bit TO the result of shifting num to the right by i positions AND the value one  
            IF bit NOT IN the keys of children in node  
                ASSIGN reference TO a new TrieNode TO the element at key bit in children of node  
            END IF  
            ASSIGN reference TO the element at key bit in children of node TO node  
            INCREMENT count in node BY one  
        END FOR  
    END FUNCTION  

    FUNCTION count_less_than(num, threshold)  
        SET count TO zero  
        ASSIGN reference TO root TO node  
        FOR i FROM bit_length TO zero DECREMENTING BY one  
            IF node IS null  
                BREAK the loop  
            END IF  
            SET bit TO the result of shifting num to the right by i positions AND the value one  
            SET threshold_bit TO the result of shifting threshold to the right by i positions AND the value one  
            IF threshold_bit EQUALS one  
                INCREMENT count BY the count value of the element at key bit in children of node OR zero if absent  
                ASSIGN reference TO the element at key bit XOR one in children of node TO node  
            ELSE  
                ASSIGN reference TO the element at key bit in children of node TO node  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION countPairs(nums, low, high)  
        ASSIGN reference TO a new Trie TO trie  
        SET nice_pairs TO zero  
        FOR each num IN nums  
            INCREMENT nice_pairs BY the result of calling count_less_than on trie with num and high PLUS one MINUS the result of calling count_less_than on trie with num and low  
            CALL insert on trie with num  
        END FOR  
        RETURN nice_pairs  
    END FUNCTION  
END CLASS
