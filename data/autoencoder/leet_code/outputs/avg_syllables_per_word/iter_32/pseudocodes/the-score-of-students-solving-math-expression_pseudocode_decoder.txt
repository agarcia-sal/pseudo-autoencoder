CLASS Solution  
    FUNCTION scoreOfStudents(s parameter s STRING answers parameter answers LIST OF INTEGER) RETURNS INTEGER  
        FUNCTION evaluate_expression(parameter expression STRING) RETURNS INTEGER  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character IS A DIGIT  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the multiplication symbol  
                    SET element at position i MINUS ONE of tokens TO element at position i MINUS ONE of tokens MULTIPLIED BY element at position i PLUS ONE of tokens  
                    REMOVE elements from position i TO position i PLUS ONE from tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i IS LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the addition symbol  
                    SET result TO result PLUS element at position i PLUS ONE of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(parameter expression STRING) RETURNS SET OF INTEGER  
            IF expression IS A DIGIT STRING  
                RETURN a set containing the integer value of expression  
            END IF  
            
            SET results TO empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS two STEP two  
                SET left_results TO get_possible_answers(substring from position zero TO position i MINUS ONE of expression)  
                SET right_results TO get_possible_answers(substring from position i PLUS ONE TO the END of expression)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF element at position i of expression EQUALS the addition symbol  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result IS LESS THAN OR EQUAL TO one thousand  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
