CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of occurrences of the character X in the current row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of occurrences of the character O in the current row  
        END FOR  
        
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN False  
        END IF  
        
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF the current row EQUALS the player character REPEATED three times  
                    RETURN True  
                END IF  
            END FOR  
            FOR col FROM zero TO two  
                IF the element at position col of the first row EQUALS player AND the element at position col of the second row EQUALS player AND the element at position col of the third row EQUALS player  
                    RETURN True  
                END IF  
            END FOR  
            IF the element at position zero of the first row EQUALS player AND the element at position one of the second row EQUALS player AND the element at position two of the third row EQUALS player  
                RETURN True  
            END IF  
            IF the element at position two of the first row EQUALS player AND the element at position one of the second row EQUALS player AND the element at position zero of the third row EQUALS player  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET winX TO the result of calling check_win with parameter character X  
        SET winO TO the result of calling check_win with parameter character O  
        
        IF winX AND winO  
            RETURN False  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN False  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
