CLASS Solution  
    FUNCTION wordBreak(s parameter, wordDict parameter)  
        SET wordSet TO set containing each element of wordDict parameter  
        SET memo TO empty dictionary  
        
        FUNCTION backtrack(start parameter)  
            IF start parameter EQUALS a key in memo  
                RETURN the value at key start parameter in memo  
            END IF  
            
            IF start parameter EQUALS the LENGTH OF s parameter  
                RETURN list containing the empty string  
            END IF  
            
            SET res TO empty list  
            FOR end parameter FROM start parameter PLUS one TO the LENGTH OF s parameter  
                SET word TO substring of s parameter from position start parameter TO position end parameter MINUS one  
                IF word EQUALS an element in wordSet  
                    FOR each sentence IN the result of backtrack called with end parameter  
                        IF sentence EQUALS the empty string  
                            APPEND word TO res  
                        ELSE  
                            APPEND concatenation of word PLUS space PLUS sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET the value at key start parameter in memo TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN the result of backtrack called with zero  
    END FUNCTION  
END CLASS
