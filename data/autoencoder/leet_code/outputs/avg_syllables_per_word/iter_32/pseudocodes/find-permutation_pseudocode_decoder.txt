CLASS Solution  
    FUNCTION findPermutation WITH PARAMETER s  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR the element at position end of s EQUALS the character I  
                SET segment TO the elements from position start TO position end PLUS one of result  
                SET reversed_segment TO the elements of segment IN REVERSED ORDER  
                FOR offset FROM zero TO the LENGTH OF reversed_segment MINUS one  
                    SET the element at position start PLUS offset of result TO the element at position offset of reversed_segment  
                END FOR  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
