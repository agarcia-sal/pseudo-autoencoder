CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each char IN s  
            IF char represents a digit  
                SET length TO length MULTIPLIED BY the numeric value represented by char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN s iterated from the last element to the first element  
            SET k TO k MODULO length  
            IF k EQUALS zero AND char represents an alphabetic character  
                RETURN char  
            END IF  
            IF char represents a digit  
                SET length TO length DIVIDED BY the numeric value represented by char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
