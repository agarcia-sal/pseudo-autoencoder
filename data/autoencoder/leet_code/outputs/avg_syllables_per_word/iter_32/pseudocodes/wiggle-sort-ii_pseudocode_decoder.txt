CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SORT the collection nums in ascending order  
        SET sorted_nums TO a copy of the collection nums  
        SET n TO the LENGTH OF the collection nums  
        SET left TO the integer division of the expression n MINUS one BY two  
        SET right TO the value of n MINUS one  
        FOR each index i FROM zero TO n MINUS one  
            IF the remainder of the division of i BY two EQUALS zero  
                SET the element at position i of the collection nums TO the element at position left of the collection sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET the element at position i of the collection nums TO the element at position right of the collection sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
