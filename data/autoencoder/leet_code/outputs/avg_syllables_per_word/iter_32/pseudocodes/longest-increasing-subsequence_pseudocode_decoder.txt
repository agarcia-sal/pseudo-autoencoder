CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS a list with no elements THEN  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the position found by the binary search procedure that finds the leftmost position to insert num into dp so that dp remains sorted in ascending order  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
