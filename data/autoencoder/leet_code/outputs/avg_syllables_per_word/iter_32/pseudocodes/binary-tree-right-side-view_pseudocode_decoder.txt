CLASS TreeNode  
    FUNCTION __init__(self val assigned zero left assigned none right assigned none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one plus one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the length of values AND the element at position i of values is not none  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values is not none  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the value of p is NOT EQUALS to the value of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree with p.left and q.left AND calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self root)  
        IF root is none  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a double ended queue initialized with root as the single element  
        WHILE the queue is not empty  
            SET level_length TO the length of queue  
            FOR i FROM one TO level_length  
                SET node TO the element removed from the left end of queue  
                IF i EQUALS level_length  
                    APPEND the value of node TO right_side_view  
                END IF  
                IF node.left is not none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right is not none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
