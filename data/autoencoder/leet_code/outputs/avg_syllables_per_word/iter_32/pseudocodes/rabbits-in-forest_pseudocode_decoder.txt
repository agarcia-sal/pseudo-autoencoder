CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO result of counting occurrences of each element in answers  
        SET total_rabbits TO zero  
        FOR each unique answer AND its corresponding count IN answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the integer division of count PLUS answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
