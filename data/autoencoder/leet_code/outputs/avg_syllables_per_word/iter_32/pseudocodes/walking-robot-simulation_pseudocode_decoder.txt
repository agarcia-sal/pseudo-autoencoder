CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of pairs where first pair is zero PLUS zero ONE second pair is one PLUS zero zero third pair is zero MINUS one zero fourth pair is zero MINUS one  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO an empty set  
        FOR each position_pair IN obstacles  
            CONVERT position_pair TO tuple form  
            ADD tuple form TO obstacles_set  
        END FOR  
        SET max_distance TO zero  
        FOR each command IN commands  
            IF command EQUALS minus two  
                SET dir_index TO dir_index MINUS one  
                SET dir_index TO the remainder of dir_index DIVIDED BY four  
            ELSE IF command EQUALS minus one  
                SET dir_index TO dir_index PLUS one  
                SET dir_index TO the remainder of dir_index DIVIDED BY four  
            ELSE  
                SET dx TO first element of element at position dir_index of directions  
                SET dy TO second element of element at position dir_index of directions  
                FOR step FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF tuple formed by next_x AND next_y IS IN obstacles_set  
                        BREAK out of inner loop  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance  
                    SET max_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
