CLASS Solution  
    FUNCTION rearrangeSticks(n k)  
        SET MOD TO ten RAISED TO THE power OF nine PLUS one  
        CREATE dp AS a list of lists where each inner list contains zeros of length k PLUS one AND there are n PLUS one such inner lists  
        SET element at position zero of element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET dp at position i at position j TO the REMAINDER AFTER DIVIDING the SUM OF dp at position i MINUS one at position j MINUS one AND the PRODUCT OF i MINUS one AND dp at position i MINUS one at position j BY MOD  
            END FOR  
        END FOR  
        RETURN dp at position n at position k  
    END FUNCTION  
END CLASS
