CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the list of elements obtained by splitting path by the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS the empty string OR component EQUALS a single dot character  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two dots characters  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character slash concatenated with the concatenation of elements of stack separated by the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
