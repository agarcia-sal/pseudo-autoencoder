CLASS Solution  
    FUNCTION minimumWhiteTiles(floor numCarpets carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list of zeroes with the number of rows EQUAL TO n PLUS one and the number of columns EQUAL TO numCarpets PLUS one  
        
        FOR i FROM one TO n  
            SET dp element at position i and zero TO dp element at position i MINUS one and zero PLUS one IF element at position i MINUS one of floor EQUALS character one OTHERWISE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at position i and j TO dp element at position i MINUS one and j PLUS one IF element at position i MINUS one of floor EQUALS character one OTHERWISE zero  
                SET dp element at position i and j TO the MINIMUM OF dp element at position i and j AND dp element at position MAXIMUM OF zero AND i MINUS carpetLen and j MINUS one  
            END FOR  
        END FOR  
        
        RETURN dp element at position n and numCarpets  
    END FUNCTION  
END CLASS
