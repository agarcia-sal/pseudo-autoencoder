CLASS Solution  
    FUNCTION canFinish(numCourses LIST OF INTEGERS, prerequisites LIST OF LIST OF INTEGERS) RETURNS BOOLEAN  
        ASSIGN reference TO a new mapping with default value as empty list NAMED graph  
        SET indegree TO a list of zeroes with the length equal to numCourses  
        
        FOR each pair course AND prereq IN prerequisites  
            APPEND course TO the list mapped by prereq in graph  
            INCREMENT the value at position course in indegree BY one  
        END FOR  
        
        ASSIGN reference TO a new double ended queue initialized with all courses FROM zero TO numCourses MINUS one WHERE the value at position course in indegree EQUALS zero NAMED queue  
        
        WHILE the queue is not empty  
            REMOVE the element from the front of queue and ASSIGN it TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list mapped by current_course in graph  
                DECREMENT the value at position next_course in indegree BY one  
                IF the value at position next_course in indegree EQUALS zero THEN  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the result of numCourses EQUALS zero  
    END FUNCTION  
END CLASS
