CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(root, val)  
        FUNCTION dfs(root)  
            IF root EQUALS none  
                RETURN a new TreeNode with value val  
            END IF  
            IF the value of root LESS THAN val  
                SET root.right TO dfs with parameter root.right  
            ELSE  
                SET root.left TO dfs with parameter root.left  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs with parameter root  
    END FUNCTION  
END CLASS
