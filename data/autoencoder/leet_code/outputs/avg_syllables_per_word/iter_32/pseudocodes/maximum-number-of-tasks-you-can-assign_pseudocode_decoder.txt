CLASS Solution  
    FUNCTION maxTaskAssign(tasks tasks_parameter, workers workers_parameter, pills pills_parameter, strength strength_parameter) RETURNS integer  
        CALL sort the tasks_parameter in ascending order  
        CALL sort the workers_parameter in ascending order  
        
        FUNCTION canAssignTasks(k k_parameter) RETURNS boolean  
            SET available_workers TO the last k_parameter elements of workers_parameter  
            SET pills_used TO zero  
            
            FOR index i FROM k_parameter MINUS one TO zero DECREMENT BY one  
                SET task TO the element at position i of tasks_parameter  
                SET worker_idx TO the position of the first element in available_workers that is greater than or equals to task  
                IF worker_idx IS LESS THAN the length of available_workers THEN  
                    REMOVE element at position worker_idx FROM available_workers  
                ELSE IF pills_used IS LESS THAN pills_parameter THEN  
                    SET worker_idx TO the position of the first element in available_workers that is greater than or equals to task MINUS strength_parameter  
                    IF worker_idx IS LESS THAN the length of available_workers THEN  
                        REMOVE element at position worker_idx FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN false  
                    END IF  
                ELSE  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the minimum of the length of tasks_parameter AND the length of workers_parameter PLUS one  
        WHILE left IS LESS THAN right DO  
            SET mid TO the integer division of the sum of left PLUS right BY two  
            IF canAssignTasks(mid) RETURNS true THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
