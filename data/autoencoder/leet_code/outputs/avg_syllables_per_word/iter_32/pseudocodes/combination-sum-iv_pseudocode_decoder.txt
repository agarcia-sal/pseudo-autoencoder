CLASS Solution  
    FUNCTION combinationSum4(nums target)  
        SET dp TO list of zero repeated the LENGTH OF target PLUS one times  
        SET element at position zero of dp TO one  
        FOR amount FROM one TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    SET element at position amount of dp TO element at position amount of dp PLUS element at position amount MINUS num of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
