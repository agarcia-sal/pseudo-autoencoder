CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH value equal to the element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the end of values  
        SET node TO new ListNode WITH value equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND the result of calling is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node NOT EQUALS None  
                APPEND the value of node TO stack  
                SET node TO the next element of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO the result of calling pushIntoStack with l1  
        SET stack2 TO the result of calling pushIntoStack with l2  
        
        SET carry TO zero  
        SET result TO None  
        
        WHILE stack1 NOT EQUALS empty OR stack2 NOT EQUALS empty OR carry NOT EQUALS zero  
            IF stack1 NOT EQUALS empty  
                SET val1 TO the last element REMOVED FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            
            IF stack2 NOT EQUALS empty  
                SET val2 TO the last element REMOVED FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY ten  
            SET digit TO the remainder of total DIVIDED BY ten  
            
            SET new_node TO new ListNode WITH value equal to digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
