CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET number_of_people TO the LENGTH OF statements  
        SET maximum_number_of_good_people TO zero  
        FOR integer i FROM zero TO the number_of_people RAISED TO THE POWER OF two MINUS one  
            INITIALIZE list_of_good_people AS a list of FALSE with the LENGTH OF number_of_people  
            INITIALIZE list_of_bad_people AS a list of FALSE with the LENGTH OF number_of_people  
            FOR integer j FROM zero TO number_of_people MINUS one  
                IF the bitwise AND of i AND the number one LEFT SHIFTED BY j EQUALS the number one LEFT SHIFTED BY j  
                    SET the element at position j of list_of_good_people TO TRUE  
                ELSE  
                    SET the element at position j of list_of_bad_people TO TRUE  
                END IF  
            END FOR  
            SET valid_combination TO TRUE  
            FOR integer j FROM zero TO number_of_people MINUS one  
                IF the element at position j of list_of_good_people EQUALS TRUE  
                    FOR integer k FROM zero TO number_of_people MINUS one  
                        IF (the element at position j of statements AT position k EQUALS zero AND the element at position k of list_of_good_people EQUALS TRUE) OR (the element at position j of statements AT position k EQUALS one AND the element at position k of list_of_bad_people EQUALS TRUE)  
                            SET valid_combination TO FALSE  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid_combination EQUALS FALSE  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid_combination EQUALS TRUE  
                SET count_of_good_people TO zero  
                FOR each person_status IN list_of_good_people  
                    IF person_status EQUALS TRUE  
                        INCREMENT count_of_good_people BY one  
                    END IF  
                END FOR  
                IF count_of_good_people GREATER THAN maximum_number_of_good_people  
                    SET maximum_number_of_good_people TO count_of_good_people  
                END IF  
            END IF  
        END FOR  
        RETURN maximum_number_of_good_people  
    END FUNCTION  
END CLASS
