CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the plus character  
            WHILE the LENGTH OF s GREATER THAN zero holds  
                SET char TO element at first position of s  
                REMOVE element at first position FROM s  
                IF char IS a digit ONE BY ONE  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric VALUE OF char  
                END IF  
                IF char EQUALS the left parenthesis character  
                    SET num TO the result OF calling helper WITH s  
                END IF  
                IF char EQUALS the plus character OR char EQUALS the minus character OR char EQUALS the multiplication character OR char EQUALS the division character OR char EQUALS the left parenthesis character OR char EQUALS the right parenthesis character OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS the plus character  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character  
                        APPEND the negative OF num TO stack  
                    ELSE IF sign EQUALS the multiplication character  
                        SET element at last position of stack TO element at last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character  
                        SET element at last position of stack TO the integer part OF element at last position of stack DIVIDED BY num truncated towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS the right parenthesis character  
                    BREAK  
                END IF  
            END WHILE  
            RETURN the sum OF all elements IN stack  
        END FUNCTION  
        SET s TO the list OF characters IN s  
        RETURN the result OF calling helper WITH s  
    END FUNCTION  
END CLASS
