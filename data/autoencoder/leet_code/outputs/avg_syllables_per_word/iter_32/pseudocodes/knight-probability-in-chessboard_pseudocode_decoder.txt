CLASS Solution  
    FUNCTION knightProbability(n k row column)  
        SET moves TO list of pairs of integers representing two steps in one direction and one step perpendicular in all eight possible ways for a knight on a chessboard  
        SET dp TO two dimensional list of size n by n with all elements set to zero  
        SET element at position row and column of dp TO one  
        FOR each iteration FROM one TO k  
            SET new_dp TO two dimensional list of size n by n with all elements set to zero  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r and c of dp GREATER THAN zero  
                        FOR each pair dr dc IN moves  
                            SET nr TO r PLUS dr  
                            SET nc TO c PLUS dc  
                            IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                                INCREMENT element at position nr and nc of new_dp BY element at position r and c of dp DIVIDED BY eight  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each probability_value IN row_list  
                INCREMENT total_probability BY probability_value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
