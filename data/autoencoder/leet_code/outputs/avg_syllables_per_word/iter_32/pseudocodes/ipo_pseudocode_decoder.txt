CLASS Solution  
    FUNCTION findMaximizedCapital(k parameter, w parameter, profits parameter, capital parameter)  
        SET projects TO list of pairs of elements from capital parameter and profits parameter where each pair consists of element at position in capital parameter and element at corresponding position in profits parameter  
        SORT projects BY the first element of each pair in ascending order  
          
        SET max_heap TO empty list  
          
        SET i TO zero  
        SET n TO the LENGTH OF projects  
          
        FOR loop_index FROM one TO k parameter  
            WHILE i LESS THAN n AND the first element of element at position i of projects LESS THAN OR EQUAL TO w parameter  
                APPEND the negative value of the second element of element at position i of projects TO max_heap maintaining heap property  
                INCREMENT i BY one  
            END WHILE  
              
            IF the LENGTH OF max_heap EQUALS zero  
                BREAK the loop  
            END IF  
              
            SET w parameter TO w parameter PLUS the negative of the element removed from max_heap which is the smallest element in the heap  
        END FOR  
          
        RETURN w parameter  
    END FUNCTION  
END CLASS
