CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums parameter LIST OF INTEGERS k parameter INTEGER)  
        SET freq TO the frequency count of each element in nums  
        SET max_freq TO the greatest frequency value in freq  
        SET condition TO the integer division of the length of nums BY k  
        RETURN whether condition IS GREATER THAN OR EQUAL TO max_freq  
    END FUNCTION  
END CLASS
