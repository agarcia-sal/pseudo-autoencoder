CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN True  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN False  
    ELSE IF the val property of p IS NOT EQUAL TO the val property of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left property of p, the left property of q) AND is_same_tree(the right property of p, the right property of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS no value THEN  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max TO the first element of helper(the left property of node)  
            SET left_gain TO the second element of helper(the left property of node)  
            SET right_max TO the first element of helper(the right property of node)  
            SET right_gain TO the second element of helper(the right property of node)  
            SET current_max TO the val property of node PLUS the greater value BETWEEN left_gain AND zero PLUS the greater value BETWEEN right_gain AND zero  
            SET current_gain TO the val property of node PLUS the greatest value AMONG left_gain, right_gain, AND zero  
            SET global_max TO the greatest value AMONG left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of helper(root)  
    END FUNCTION  
END CLASS
