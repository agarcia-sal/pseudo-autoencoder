CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character is a digit  
                SET d TO d PLUS character  
            ELSE IF character is an alphabetic letter  
                SET a TO a PLUS character  
            ELSE IF character EQUALS the opening bracket character  
                APPEND the tuple of a AND the integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character EQUALS the closing bracket character  
                SET p AND n TO the last element removed FROM stack  
                SET a TO p PLUS a MULTIPLIED BY n  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
