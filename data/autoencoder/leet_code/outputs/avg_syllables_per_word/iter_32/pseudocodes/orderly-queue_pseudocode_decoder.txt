CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the RESULT OF concatenating all elements IN the LIST of characters of s SORTED in ascending order  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the SUBSTRING OF s from position i TO the END CONCATENATED WITH the SUBSTRING OF s from position zero TO position i MINUS one  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
