CLASS Solution  
    FUNCTION totalStrength(strength)  
        SET MODULO TO one billion PLUS seven  
        SET n TO the LENGTH OF strength  
          
        SET prefix TO a list of zero repeated n PLUS one TIMES  
        FOR index FROM zero TO n MINUS one  
            SET element at position index PLUS one of prefix TO the SUM of element at position index of prefix PLUS element at position index of strength MODULO MODULO  
        END FOR  
          
        SET prefix_prefix TO a list of zero repeated n PLUS two TIMES  
        FOR index FROM zero TO n  
            SET element at position index PLUS one of prefix_prefix TO the SUM of element at position index of prefix_prefix PLUS element at position index of prefix MODULO MODULO  
        END FOR  
          
        SET prev_smaller TO a list of negative one repeated n TIMES  
        SET next_smaller_or_equal TO a list of n repeated n TIMES  
          
        SET stack TO an empty list  
        FOR index FROM zero TO n MINUS one  
            WHILE stack is not empty AND element at position element at the last position of stack of strength IS GREATER THAN OR EQUAL TO element at position index of strength  
                REMOVE the last element FROM stack  
            END WHILE  
            IF stack is not empty  
                SET element at position index of prev_smaller TO the element at the last position of stack  
            END IF  
            APPEND index TO stack  
        END FOR  
          
        CLEAR stack  
        FOR index FROM n MINUS one DOWN TO zero  
            WHILE stack is not empty AND element at position element at the last position of stack of strength IS GREATER THAN element at position index of strength  
                REMOVE the last element FROM stack  
            END WHILE  
            IF stack is not empty  
                SET element at position index of next_smaller_or_equal TO the element at the last position of stack  
            END IF  
            APPEND index TO stack  
        END FOR  
          
        SET total_strength TO zero  
        FOR index FROM zero TO n MINUS one  
            SET left_boundary TO element at position index of prev_smaller PLUS one  
            SET right_boundary TO element at position index of next_smaller_or_equal  
              
            SET sum_left TO the PRODUCT of (index MINUS left_boundary PLUS one) AND (element at position right_boundary PLUS one of prefix_prefix MINUS element at position index PLUS one of prefix_prefix) MODULO MODULO  
            SET sum_right TO the PRODUCT of (right_boundary MINUS index) AND (element at position index PLUS one of prefix_prefix MINUS element at position left_boundary of prefix_prefix) MODULO MODULO  
              
            SET contribution TO the PRODUCT of element at position index of strength AND (sum_left MINUS sum_right) MODULO MODULO  
            SET total_strength TO the SUM of total_strength AND contribution MODULO MODULO  
        END FOR  
          
        RETURN total_strength  
    END FUNCTION  
END CLASS
