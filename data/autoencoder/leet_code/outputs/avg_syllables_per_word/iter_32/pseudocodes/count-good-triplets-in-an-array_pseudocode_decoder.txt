CLASS BIT  
    FUNCTION __init__(self, n)  
        SET self.size TO n  
        SET self.tree TO a list of zero values of length n PLUS one  
    END FUNCTION  
      
    FUNCTION update(self, idx, delta)  
        WHILE idx LESS THAN OR EQUAL TO self.size  
            INCREMENT element at position idx of self.tree BY delta  
            SET idx TO idx PLUS idx AND the NEGATION of idx  
        END WHILE  
    END FUNCTION  
      
    FUNCTION query(self, idx)  
        SET result TO zero  
        WHILE idx GREATER THAN zero  
            INCREMENT result BY element at position idx of self.tree  
            SET idx TO idx MINUS idx AND the NEGATION of idx  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION goodTriplets(self, nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        
        SET pos TO a list of zero values of length n  
        FOR each index i AND each number num IN nums1  
            SET element at position num of pos TO i  
        END FOR  
        
        SET compressed TO an empty list  
        FOR each number num IN nums2  
            APPEND element at position num of pos TO compressed  
        END FOR  
        
        SET bit TO a new instance of BIT with n as parameter  
        
        SET smaller_count TO a list of zero values of length n  
        SET larger_count TO a list of zero values of length n  
        
        FOR each index i FROM zero TO n MINUS one  
            SET element at position i of smaller_count TO the result of bit.query at element at position i of compressed  
            CALL bit.update with element at position i of compressed PLUS one and one as parameters  
        END FOR  
        
        SET bit TO a new instance of BIT with n as parameter  
        
        FOR each index i FROM n MINUS one DOWN TO zero  
            SET element at position i of larger_count TO the result of bit.query at n MINUS the result of bit.query at element at position i of compressed PLUS one  
            CALL bit.update with element at position i of compressed PLUS one and one as parameters  
        END FOR  
        
        SET good_triplets TO zero  
        FOR each index i FROM zero TO n MINUS one  
            INCREMENT good_triplets BY element at position i of smaller_count MULTIPLIED BY element at position i of larger_count  
        END FOR  
        
        RETURN good_triplets  
    END FUNCTION  
END CLASS
