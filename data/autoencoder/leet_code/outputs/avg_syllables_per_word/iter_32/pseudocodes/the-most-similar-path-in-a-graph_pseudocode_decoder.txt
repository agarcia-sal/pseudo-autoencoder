CLASS Solution  
    FUNCTION mostSimilar(n targetPath roads names)  
        SET graph TO a new dictionary where each key maps to an empty list  
        FOR each pair u and v IN roads  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
        
        SET m TO the length of targetPath  
        
        FUNCTION dp(i j)  
            IF i EQUALS zero THEN  
                IF element at position j of names NOT EQUALS element at position i of targetPath THEN  
                    RETURN one AND a list containing j  
                ELSE  
                    RETURN zero AND a list containing j  
                END IF  
            END IF  
            
            SET min_cost TO infinity  
            SET min_path TO an empty list  
            FOR each element k IN the list at key j in graph  
                SET cost_path TO dp(i MINUS one j k)  
                SET cost TO the first component of cost_path  
                SET path TO the second component of cost_path  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                    SET min_path TO path  
                END IF  
            END FOR  
            
            SET current_cost TO min_cost PLUS  
                one IF element at position j of names NOT EQUALS element at position i of targetPath OTHERWISE zero  
            RETURN current_cost AND the list obtained by appending j to min_path  
        END FUNCTION  
        
        SET min_cost TO infinity  
        SET min_path TO an empty list  
        FOR each index j FROM zero TO n MINUS one  
            SET cost_path TO dp(m MINUS one j)  
            SET cost TO the first component of cost_path  
            SET path TO the second component of cost_path  
            IF cost LESS THAN min_cost THEN  
                SET min_cost TO cost  
                SET min_path TO path  
            END IF  
        END FOR  
        
        RETURN min_path  
    END FUNCTION  
END CLASS
