CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the list of tuples consisting of zero and one zero and one zero and negative one and negative one and zero  
        SET min_heap TO the list containing a tuple of zero zero zero representing cost and horizontal position and vertical position  
        SET visited TO the set containing a tuple of zero zero representing positions already visited  
        
        WHILE the LENGTH OF min_heap is GREATER THAN zero  
            REMOVE and RETURN the tuple containing cost and horizontal position x and vertical position y with the smallest cost from min_heap  
            
            IF the tuple containing x and y is EQUALS the tuple containing m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each tuple containing dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx is GREATER THAN OR EQUAL TO zero AND nx is LESS THAN m AND ny is GREATER THAN OR EQUAL TO zero AND ny is LESS THAN n AND the tuple containing nx and ny is NOT IN visited  
                    ADD the tuple containing nx and ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid at position ny  
                    INSERT the tuple containing new_cost nx ny INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
