CLASS Solution  
    FUNCTION minWindow(s1, s2)  
        SET min_len TO a very large number representing infinity  
        SET min_start TO negative one  
        SET len1 TO the LENGTH OF s1  
        SET len2 TO the LENGTH OF s2  
        
        FOR i FROM zero TO len1 MINUS one  
            SET j TO zero  
            FOR k FROM i TO len1 MINUS one  
                IF the element at position k of s1 EQUALS the element at position j of s2 THEN  
                    INCREMENT j BY one  
                    IF j EQUALS len2 THEN  
                        IF the difference of k MINUS i PLUS one LESS THAN min_len THEN  
                            SET min_len TO the difference of k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start EQUALS negative one THEN  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s1 from position min_start TO the position of min_start PLUS min_len MINUS one  
        END IF  
    END FUNCTION  
END CLASS
