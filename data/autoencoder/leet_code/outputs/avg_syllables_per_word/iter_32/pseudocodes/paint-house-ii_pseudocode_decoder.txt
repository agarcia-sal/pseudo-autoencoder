CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs EQUALS empty  
            RETURN zero  
        END IF  
          
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF element at position one of costs  
          
        IF k EQUALS one  
            IF n EQUALS one  
                RETURN element at position one at position one of costs  
            ELSE  
                RETURN infinity  
            END IF  
        END IF  
          
        FOR i FROM two TO n  
            SET min_cost TO infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO infinity  
              
            FOR j FROM one TO k  
                IF element at position i MINUS one at position j of costs LESS THAN min_cost  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO element at position i MINUS one at position j of costs  
                    SET min_cost_index TO j  
                ELSE IF element at position i MINUS one at position j of costs LESS THAN second_min_cost  
                    SET second_min_cost TO element at position i MINUS one at position j of costs  
                END IF  
            END FOR  
              
            FOR j FROM one TO k  
                IF j EQUALS min_cost_index  
                    SET element at position i at position j of costs TO element at position i at position j of costs PLUS second_min_cost  
                ELSE  
                    SET element at position i at position j of costs TO element at position i at position j of costs PLUS min_cost  
                END IF  
            END FOR  
        END FOR  
          
        SET minimum_cost TO element at position one of costs at position one  
        FOR j FROM two TO k  
            IF element at position n at position j of costs LESS THAN minimum_cost  
                SET minimum_cost TO element at position n at position j of costs  
            END IF  
        END FOR  
          
        RETURN minimum_cost  
    END FUNCTION  
END CLASS
