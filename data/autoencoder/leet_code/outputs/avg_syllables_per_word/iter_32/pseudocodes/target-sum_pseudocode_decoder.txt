CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the PAIR of index AND current_sum IS IN memo  
                RETURN the VALUE at the PAIR of index AND current_sum IN memo  
            END IF  
            SET positive TO CALL dfs WITH index PLUS one AND current_sum PLUS element at position index of nums AND memo  
            SET negative TO CALL dfs WITH index PLUS one AND current_sum MINUS element at position index of nums AND memo  
            SET the VALUE at the PAIR of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE at the PAIR of index AND current_sum IN memo  
        END FUNCTION  
        RETURN CALL dfs WITH zero AND zero AND an EMPTY dictionary  
    END FUNCTION  
END CLASS
