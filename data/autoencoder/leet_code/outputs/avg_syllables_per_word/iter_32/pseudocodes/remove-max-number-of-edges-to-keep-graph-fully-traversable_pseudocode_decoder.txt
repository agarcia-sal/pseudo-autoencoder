CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers representing all numbers from zero PLUS one TO n PLUS one  
        SET self.rank TO list of zeros of length n PLUS one  
        SET self.count TO n  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO the result of self.find with argument element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO the result of self.find with argument u  
        SET rootV TO the result of self.find with argument v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        ASSIGN reference TO new UnionFind with parameter n TO uf_common  
        ASSIGN reference TO new UnionFind with parameter n TO uf_alice  
        ASSIGN reference TO new UnionFind with parameter n TO uf_bob  
  
        SET used_edges TO zero  
  
        FOR each element containing three values t u v IN edges  
            IF t EQUALS three THEN  
                IF the result of uf_common.union with arguments u and v THEN  
                    CALL uf_alice.union with arguments u and v  
                    CALL uf_bob.union with arguments u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
  
        FOR each element containing three values t u v IN edges  
            IF t EQUALS one AND the result of uf_alice.union with arguments u and v THEN  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND the result of uf_bob.union with arguments u and v THEN  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
  
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one THEN  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
