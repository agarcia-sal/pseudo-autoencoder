CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN list containing an empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing the string zero one and eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN the result of dfs with parameter u MINUS two  
                FOR each pair consisting of l and r IN the list containing the pairs one one eight eight six nine nine six  
                    APPEND the string formed by l FOLLOWED BY v FOLLOWED BY r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND the string formed by zero FOLLOWED BY v FOLLOWED BY zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs called with parameter n  
    END FUNCTION  
END CLASS
