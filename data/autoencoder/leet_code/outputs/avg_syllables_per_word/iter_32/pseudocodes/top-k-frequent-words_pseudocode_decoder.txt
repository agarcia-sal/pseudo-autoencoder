CLASS Solution  
    FUNCTION topKFrequent(words list OF strings k integer) RETURNS list OF strings  
        SET count TO the frequency mapping of each word IN words  
        SET heap TO an empty list  
        FOR each word AND frequency IN the entries of count  
            APPEND a pair consisting of the negative of frequency AND the word TO heap  
        END FOR  
        CONVERT heap INTO a heap structure according to its defined ordering  
        SET result TO an empty list  
        FOR iteration FROM one TO k  
            REMOVE the smallest element FROM heap AND EXTRACT the word component OF that element  
            APPEND that word TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
