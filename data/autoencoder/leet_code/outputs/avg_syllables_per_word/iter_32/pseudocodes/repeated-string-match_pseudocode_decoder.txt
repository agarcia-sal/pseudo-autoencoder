CLASS Solution  
    FUNCTION repeatedStringMatch(a string b string) RETURNS integer  
        SET min_repeats TO the smallest integer greater than or equal to the length of b DIVIDED BY the length of a  
        
        IF the string b IS a substring of the string a repeated min_repeats times  
            RETURN min_repeats  
        END IF  
        
        IF the string b IS a substring of the string a repeated the quantity min_repeats PLUS one times  
            RETURN the quantity min_repeats PLUS one  
        END IF  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
