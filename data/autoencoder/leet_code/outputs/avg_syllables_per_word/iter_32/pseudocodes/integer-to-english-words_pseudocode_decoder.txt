CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN word zero  
        END IF  
        
        SET below_twenty TO list of words corresponding to numbers from zero to nineteen where position one is word one through position nineteen is word nineteen with empty string at position zero  
        SET tens TO list of words corresponding to multiples of ten from zero to ninety where position two is word twenty through position nine is word ninety with empty strings at positions zero and one  
        SET thousands TO list of scale words where position zero is empty string position one is word thousand position two is word million position three is word billion  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            IF n LESS THAN twenty  
                RETURN element at position n of below_twenty CONCATENATED WITH a space  
            END IF  
            IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten of tens CONCATENATED WITH a space CONCATENATED WITH helper of remainder of n DIVIDED BY ten  
            END IF  
            RETURN element at position n DIVIDED BY one hundred of below_twenty CONCATENATED WITH word hundred CONCATENATED WITH a space CONCATENATED WITH helper of remainder of n DIVIDED BY one hundred  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF remainder of num DIVIDED BY one thousand NOT EQUALS zero  
                SET result TO helper of remainder of num DIVIDED BY one thousand CONCATENATED WITH element at position i of thousands CONCATENATED WITH a space CONCATENATED WITH result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
