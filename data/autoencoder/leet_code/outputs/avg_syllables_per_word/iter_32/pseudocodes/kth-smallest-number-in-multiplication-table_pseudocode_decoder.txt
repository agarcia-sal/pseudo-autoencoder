CLASS Solution  
    FUNCTION findKthNumber(m parameter n parameter k parameter)  
        FUNCTION enough(x parameter)  
            SET count TO zero  
            FOR i parameter FROM one TO m parameter  
                INCREMENT count BY the smaller VALUE BETWEEN result of x parameter DIVIDED BY i parameter AND n parameter  
            END FOR  
            RETURN count GREATER THAN OR EQUAL TO k parameter  
        END FUNCTION  
        SET low TO one  
        SET high TO m parameter MULTIPLIED BY n parameter  
        WHILE low LESS THAN high  
            SET mid TO the result of low PLUS high DIVIDED BY two  
            IF result of calling enough FUNCTION WITH mid parameter IS true  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
