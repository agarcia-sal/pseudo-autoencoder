CLASS Solution  
    FUNCTION shortestPathAllKeys(grid)  
        SET directions TO list containing tuples representing zero PLUS one, zero MINUS one, one PLUS zero, and one MINUS zero  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start TO no value  
        SET num_keys TO zero  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid AT position j EQUALS the character at symbol at sign  
                    SET start TO tuple of i and j  
                ELSE IF element at position i of grid AT position j IS A lowercase letter  
                    INCREMENT num_keys BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO a double-ended queue initialized with a tuple containing the first element of start, the second element of start, zero representing collected keys, and zero representing steps taken  
        SET visited TO a set initialized with a tuple containing the first element of start, the second element of start, and zero representing collected keys  
        
        WHILE the queue is not empty  
            REMOVE and RETRIEVE the first element from queue INTO variables x, y, keys, steps  
            
            FOR each tuple dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET cell TO element at position nx of grid AT position ny  
                    
                    IF cell EQUALS the character representing a wall symbol  
                        CONTINUE to the next iteration of the inner for loop  
                    END IF  
                    
                    IF cell IS AN uppercase letter  
                        SET key_needed TO one LEFT SHIFTED BY the numeric position of the lowercase equivalent of cell MINUS the numeric position of letter a  
                        IF the bitwise AND of keys and key_needed EQUALS zero  
                            CONTINUE to the next iteration of the inner for loop  
                        END IF  
                    END IF  
                    
                    IF cell IS a lowercase letter  
                        SET new_keys TO the bitwise OR of keys and one LEFT SHIFTED BY the numeric position of cell MINUS the numeric position of letter a  
                        IF new_keys EQUALS one LEFT SHIFTED BY num_keys MINUS one  
                            RETURN steps PLUS one  
                        END IF  
                    ELSE  
                        SET new_keys TO keys  
                    END IF  
                    
                    IF the tuple containing nx, ny, and new_keys is not in visited  
                        ADD the tuple containing nx, ny, and new_keys TO visited  
                        APPEND the tuple containing nx, ny, new_keys, and steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the value of negative one  
    END FUNCTION  
END CLASS
