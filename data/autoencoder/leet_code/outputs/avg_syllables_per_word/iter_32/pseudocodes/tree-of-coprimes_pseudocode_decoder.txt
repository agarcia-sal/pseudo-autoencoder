CLASS Solution  
    FUNCTION getCoprimes(nums, edges)  
        SET n TO the LENGTH OF nums  
        SET tree TO a new default dictionary of lists  
        FOR each pair u AND v IN edges  
            APPEND v TO the list at position u of tree  
            APPEND u TO the list at position v of tree  
        END FOR  
        
        CREATE a two dimensional collection coprime with false values for positions from one TO fifty for both dimensions  
        FOR i FROM one TO fifty  
            FOR j FROM one TO fifty  
                SET the element at position i and j of coprime TO the condition that the greatest common divisor of i AND j EQUALS one  
            END FOR  
        END FOR  
        
        CREATE a collection result of length n with all values set to minus one  
        CREATE a collection ancestor_stack of length fifty with all elements set to None to represent ancestor information as pairs of node AND depth  
        
        FUNCTION dfs(node, parent, depth)  
            SET closest_ancestor TO minus one  
            SET max_depth TO minus one  
            FOR value FROM one TO fifty  
                IF the element at position nums at node AND value of coprime IS true AND the element at position value of ancestor_stack IS NOT None  
                    SET anc_node, anc_depth TO the elements of the value at position value of ancestor_stack  
                    IF anc_depth GREATER THAN max_depth  
                        SET max_depth TO anc_depth  
                        SET closest_ancestor TO anc_node  
                    END IF  
                END IF  
            END FOR  
            SET the element at position node of result TO closest_ancestor  
            
            SET original_ancestor TO the element at position nums at node of ancestor_stack  
            SET the element at position nums at node of ancestor_stack TO the pair of node AND depth  
            
            FOR each child IN the list at position node of tree  
                IF child NOT EQUALS parent  
                    CALL dfs WITH arguments child, node, depth PLUS one  
                END IF  
            END FOR  
            
            SET the element at position nums at node of ancestor_stack TO original_ancestor  
        END FUNCTION  
        
        CALL dfs WITH arguments zero, minus one, zero  
        
        RETURN result  
    END FUNCTION  
END CLASS
