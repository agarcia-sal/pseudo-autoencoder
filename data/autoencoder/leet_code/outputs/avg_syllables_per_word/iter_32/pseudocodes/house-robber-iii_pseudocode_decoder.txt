CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO TreeNode of the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        SET node TO the first element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.left TO TreeNode of the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.right TO TreeNode of the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the value property of p NOT EQUALS the value property of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left property of p and the left property of q AND calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(self, root)  
        FUNCTION rob_from_node(node)  
            IF node EQUALS None THEN  
                RETURN zero AND zero  
            END IF  
            SET left_robbed AND left_not_robbed TO rob_from_node of the left property of node  
            SET right_robbed AND right_not_robbed TO rob_from_node of the right property of node  
            SET rob_this_node TO the value property of node PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the maximum of left_robbed AND left_not_robbed PLUS the maximum of right_robbed AND right_not_robbed  
            RETURN rob_this_node AND not_rob_this_node  
        END FUNCTION  
        RETURN the maximum of rob_from_node called with root  
    END FUNCTION  
END CLASS
