CLASS Solution  
    FUNCTION countPairs(nums k)  
        SET gcd_count TO empty mapping from elements to counts  
        FOR each num IN nums  
            INCREMENT the count for the greatest common divisor of num AND k in gcd_count BY one  
        END FOR  
        SET total_pairs TO zero  
        SET gcd_values TO the list of all keys of gcd_count  
        SET n TO the length of gcd_values  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET gcd1 TO element at position i of gcd_values  
                SET gcd2 TO element at position j of gcd_values  
                IF the remainder when gcd1 MULTIPLIED BY gcd2 is divided BY k EQUALS zero THEN  
                    IF i EQUALS j THEN  
                        INCREMENT total_pairs BY the count for gcd1 in gcd_count MULTIPLIED BY the count for gcd1 in gcd_count MINUS one DIVIDED BY two  
                    ELSE  
                        INCREMENT total_pairs BY the count for gcd1 in gcd_count MULTIPLIED BY the count for gcd2 in gcd_count  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN total_pairs  
    END FUNCTION  
END CLASS
