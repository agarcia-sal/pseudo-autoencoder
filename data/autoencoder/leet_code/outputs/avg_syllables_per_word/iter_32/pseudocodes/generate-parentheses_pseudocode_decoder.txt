CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s left right  
            IF the LENGTH OF s EQUALS the MULTIPLICATION OF two AND n THEN  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n THEN  
                CALL backtrack WITH PARAMETERS the CONCATENATION OF s AND an opening parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left THEN  
                CALL backtrack WITH PARAMETERS the CONCATENATION OF s AND a closing parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH NO PARAMETERS  
        RETURN result  
    END FUNCTION  
END CLASS
