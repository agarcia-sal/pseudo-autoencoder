CLASS Solution  
    FUNCTION tilingRectangle(n n_parameter, m m_parameter)  
        FUNCTION backtrack(heights heights_parameter, moves moves_parameter)  
            IF for every element h in heights_parameter h EQUALS n_parameter  
                SET min_moves TO the smaller value BETWEEN min_moves AND moves_parameter  
                RETURN  
            END IF  
            
            IF moves_parameter GREATER THAN OR EQUAL TO min_moves  
                RETURN  
            END IF  
            
            SET min_height TO the smallest value IN heights_parameter  
            SET left TO the position of min_height IN heights_parameter  
            SET right TO left  
            
            WHILE right LESS THAN m_parameter AND the element at position right of heights_parameter EQUALS min_height  
                INCREMENT right BY one  
            END WHILE  
            
            FOR size FROM right MINUS left DOWN TO one  
                IF for each index i FROM zero TO size MINUS one the element at position left PLUS i of heights_parameter PLUS size LESS THAN OR EQUAL TO n_parameter  
                    SET new_heights TO a copy of heights_parameter  
                    FOR i FROM zero TO size MINUS one  
                        INCREMENT the element at position left PLUS i of new_heights BY size  
                    END FOR  
                    CALL backtrack WITH new_heights AND moves_parameter PLUS one  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET min_moves TO positive infinity  
        CALL backtrack WITH a list of zero replicated m_parameter TIMES AND zero  
        RETURN min_moves  
    END FUNCTION  
END CLASS
