CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF element at position zero of num MODULO two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            INSERT negative of num INTO max_heap USING heap push operation  
        END FOR  
        SET min_deviation TO infinity  
        WHILE element at position zero of max_heap MODULO two EQUALS zero  
            REMOVE top element FROM max_heap USING heap pop operation AND SET max_value TO negative of that element  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            INSERT negative of max_value INTO max_heap USING heap push operation  
        END WHILE  
        IF min_deviation GREATER THAN negative of element at position zero of max_heap MINUS min_value  
            SET min_deviation TO negative of element at position zero of max_heap MINUS min_value  
        END IF  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
