CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heappush ON min_heap WITH item consisting of the element at position zero of the list at position index of nums the value index and zero  
            IF the element at position zero of the list at position index of nums GREATER THAN max_value THEN  
                SET max_value TO the element at position zero of the list at position index of nums  
            END IF  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO a list consisting of negative one hundred thousand and one hundred thousand  
        WHILE the LENGTH OF min_heap GREATER THAN zero  
            SET min_value TO the first part of the result of heappop ON min_heap  
            SET list_index TO the second part of the result of heappop ON min_heap  
            SET element_index TO the third part of the result of heappop ON min_heap  
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list consisting of min_value and max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF the list at position list_index of nums THEN  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                CALL heappush ON min_heap WITH item consisting of next_value list_index element_index PLUS one  
                IF next_value GREATER THAN max_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
