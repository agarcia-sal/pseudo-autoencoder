CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO empty list  
        FOR each index AND element number IN nums  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO the integer division of the sum of start AND end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR index FROM start TO the integer before end  
                INCREMENT cost BY the absolute value of the difference between element at position index of positions AND the difference of median MINUS the difference of mid MINUS index  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR index FROM zero TO the difference of the length of positions MINUS k PLUS one  
            SET current_cost TO calculate_cost(index, index PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
