CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF element at position i of board AND element at position j of that row NOT EQUALS EXCLUSIVE OR element at position i of board AND element at position zero of that row NOT EQUALS EXCLUSIVE OR element at position zero of board AND element at position j of that row NOT EQUALS EXCLUSIVE OR element at position zero of board AND element at position zero of that row NOT EQUALS zero  
                    RETURN negative one  
                END IF  
            END FOR  
            SET rowSum TO rowSum PLUS element at position i of board AND element at position zero of that row  
            SET colSum TO colSum PLUS element at position zero of board AND element at position i of that row  
            SET rowSwap TO rowSwap PLUS the evaluation of whether element at position i of board AND element at position zero of that row EQUALS the remainder of i DIVIDED BY two  
            SET colSwap TO colSwap PLUS the evaluation of whether element at position zero of board AND element at position i of that row EQUALS the remainder of i DIVIDED BY two  
        END FOR  
        IF the integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two EQUALS one  
            IF the remainder of rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the smaller value BETWEEN rowSwap AND N MINUS rowSwap  
            SET colSwap TO the smaller value BETWEEN colSwap AND N MINUS colSwap  
        END IF  
        RETURN the integer division OF rowSwap PLUS colSwap BY two  
    END FUNCTION  
END CLASS
