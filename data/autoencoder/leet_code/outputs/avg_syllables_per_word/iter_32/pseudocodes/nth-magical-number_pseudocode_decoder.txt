CLASS Solution  
    FUNCTION nthMagicalNumber(n a b)  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION gcd(x y)  
            WHILE variable y is not equal to zero holds  
                SET temporary_x TO value of variable x  
                SET x TO value of variable y  
                SET y TO the remainder of temporary_x DIVIDED BY variable y  
            END WHILE  
            RETURN value of variable x  
        END FUNCTION  
        
        FUNCTION lcm(x y)  
            RETURN variable x MULTIPLIED BY variable y DIVIDED BY the result of FUNCTION gcd with parameters x y  
        END FUNCTION  
        
        SET L TO the result of FUNCTION lcm with parameters a b  
        
        SET left TO one  
        SET right TO variable n MULTIPLIED BY the smaller value between variable a and variable b  
        
        WHILE value of variable left is LESS THAN value of variable right holds  
            SET mid TO the integer division of the sum of variables left and right by two  
            SET count TO the integer division of mid by a PLUS the integer division of mid by b MINUS the integer division of mid by L  
            
            IF value of variable count is LESS THAN value of variable n THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN the remainder of variable left DIVIDED BY variable MOD  
    END FUNCTION  
END CLASS
