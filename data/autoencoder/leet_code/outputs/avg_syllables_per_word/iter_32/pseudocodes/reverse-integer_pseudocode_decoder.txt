CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_value_to_string TO string representation of absolute value of x  
        SET reversed_string TO characters of absolute_value_to_string in reverse order as a string  
        SET reversed_x TO integer value of reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two POWER thirty ONE OR result GREATER THAN one MULTIPLIED BY two POWER thirty ONE MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
