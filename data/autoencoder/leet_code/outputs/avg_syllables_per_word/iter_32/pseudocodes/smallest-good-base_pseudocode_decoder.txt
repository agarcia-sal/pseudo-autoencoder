CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO integer conversion of n  
        SET max_length TO the integer value of the base two logarithm of n PLUS one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer value of n raised to the power of one DIVIDED BY the quantity m MINUS one  
            IF k LESS THAN two  
                CONTINUE to the next iteration of the loop  
            END IF  
            SET num TO the integer value of the quantity k raised to the power of m MINUS one DIVIDED BY the quantity k MINUS one  
            IF num EQUALS n  
                RETURN the string conversion of k  
            END IF  
        END FOR  
        RETURN the string conversion of n MINUS one  
    END FUNCTION  
END CLASS
