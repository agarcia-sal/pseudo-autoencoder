CLASS Solution  
    FUNCTION maxHeight WITH PARAMETER cuboids  
        FOR each cuboid IN cuboids  
            SORT dimensions of cuboid in non-decreasing order  
        END FOR  
        SORT cuboids lexicographically by dimensions  
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp AS a list of length n with all elements SET TO zero  
        FOR i FROM zero TO n MINUS one  
            SET dp at position i TO the third dimension value of cuboid at position i  
            FOR j FROM zero TO i MINUS one  
                IF the first dimension value of cuboid at position j IS LESS THAN OR EQUAL TO the first dimension value of cuboid at position i  
                AND the second dimension value of cuboid at position j IS LESS THAN OR EQUAL TO the second dimension value of cuboid at position i  
                AND the third dimension value of cuboid at position j IS LESS THAN OR EQUAL TO the third dimension value of cuboid at position i  
                    SET dp at position i TO the maximum BETWEEN dp at position i AND dp at position j PLUS the third dimension value of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the maximum value IN dp  
    END FUNCTION  
END CLASS
