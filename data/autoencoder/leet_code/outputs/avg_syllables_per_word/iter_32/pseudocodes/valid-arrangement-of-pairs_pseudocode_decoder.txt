CLASS Solution  
    FUNCTION validArrangement(pairs)  
        SET graph TO a new mapping from element to list of elements with default empty list  
        SET out_degree TO a new mapping from element to count with default zero  
        SET in_degree TO a new mapping from element to count with default zero  
        
        FOR each element u and element v IN pairs  
            APPEND element v TO the list mapped from element u in graph  
            INCREMENT out_degree at element u BY one  
            INCREMENT in_degree at element v BY one  
        END FOR  
        
        SET start_node TO None  
        FOR each node IN keys of graph  
            IF out_degree at node EQUALS in_degree at node PLUS one THEN  
                SET start_node TO node  
                BREAK  
            END IF  
        END FOR  
        IF start_node EQUALS None THEN  
            SET start_node TO an arbitrary element from keys of graph  
        END IF  
        
        SET path TO an empty list  
        SET stack TO a list containing start_node  
        
        WHILE the LENGTH OF stack IS GREATER THAN zero DO  
            SET u TO the last element of stack  
            IF the list mapped from u in graph IS not empty THEN  
                SET v TO the last element in the list mapped from u in graph  
                REMOVE the last element from the list mapped from u in graph  
                APPEND v TO stack  
            ELSE  
                APPEND the last element of stack TO path  
                REMOVE the last element from stack  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO an empty list  
        FOR index i FROM zero TO the LENGTH OF path MINUS one MINUS one DO  
            APPEND the list consisting of element at position i of path and element at position i PLUS one of path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
