CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n  
        SET power_of_2_sorted_digits TO an empty set of tuples  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion  
            CONVERT power_of_2 TO string form  
            SORT all characters in the string form of power_of_2 in ascending order  
            CONVERT the sorted characters to a tuple  
            ADD the tuple TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        CONVERT n TO string form  
        SORT all characters in the string form of n in ascending order  
        CONVERT the sorted characters to a tuple  
        SET sorted_n_digits TO this tuple  
        IF sorted_n_digits IS AN ELEMENT OF power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
