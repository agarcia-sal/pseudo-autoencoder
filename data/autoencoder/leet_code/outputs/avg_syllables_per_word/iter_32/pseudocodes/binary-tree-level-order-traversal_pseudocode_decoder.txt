CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element popped from the left of queue  
        IF i LESS THAN LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder with parameters self root  
        IF root EQUALS None  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue initialized with root as first element  
        WHILE the queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR index FROM one TO level_size  
                SET node TO the element popped from the left of queue  
                APPEND the value attribute of node TO current_level  
                IF the left attribute of node NOT EQUALS None  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node NOT EQUALS None  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
