CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO a list of tuples each composed of elements from indices sources and targets sorted by the first element of each tuple  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each tuple composed of index source target IN replacements  
            APPEND the substring of s from position prev_end TO position index MINUS one TO result  
            IF the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the substring of s from position prev_end TO the END OF s TO result  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
