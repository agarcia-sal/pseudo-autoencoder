CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
  
        SET freq TO a collection mapping each character in s TO its frequency count  
  
        SET max_heap TO a list of pairs where each pair CONTAINS the negative frequency AND the character FROM freq  
        ORGANIZE max_heap INTO a heap structure BASED ON the first element OF each pair  
  
        SET wait_queue TO an empty list  
  
        SET result TO an empty list  
  
        WHILE max_heap is not empty  
            REMOVE the smallest element FROM max_heap AND ASSIGN its components TO count AND char  
            APPEND char TO result  
  
            APPEND the pair consisting of count PLUS one AND char TO wait_queue  
  
            IF the number of elements IN wait_queue EQUALS k  
                REMOVE the first element FROM wait_queue AND ASSIGN its components TO count AND char  
                IF count LESS THAN zero  
                    ADD the pair count AND char BACK INTO max_heap maintaining heap structure  
                END IF  
            END IF  
        END WHILE  
  
        IF the length OF result DOES NOT EQUAL the length OF s  
            RETURN an empty string  
        END IF  
  
        RETURN the concatenation of all elements IN result AS a string  
    END FUNCTION  
END CLASS
