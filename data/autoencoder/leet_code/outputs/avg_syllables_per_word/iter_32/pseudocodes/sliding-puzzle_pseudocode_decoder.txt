CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO an empty list  
        FOR each row IN board  
            APPEND a tuple formed by all elements of row TO start  
        END FOR  
        CONVERT start to a tuple of its elements  
        SET target TO a tuple formed by first tuple containing one two and three and second tuple containing four five and zero  
        SET neighbors TO a mapping where zero maps to a set of one and three and one maps to a set of zero two and four and two maps to a set of one and five and three maps to a set of zero and four and four maps to a set of one three and five and five maps to a set of two and four  
        SET zero_pos TO undefined  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position i of start at position j EQUALS zero THEN  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue containing a tuple of start zero_pos and zero  
        SET visited TO a new set containing start  
        WHILE queue is not empty  
            REMOVE and RETURN the leftmost element of queue and ASSIGN the unpacked values to state zero_pos and moves  
            IF state EQUALS target THEN  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set mapped from zero_pos in neighbors  
                SET new_state TO an empty list  
                FOR each row IN state  
                    APPEND a list formed by all elements of row TO new_state  
                END FOR  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                SET temp TO element at position i of new_state at position j  
                SET element at position i of new_state at position j TO element at position ni of new_state at position nj  
                SET element at position ni of new_state at position nj TO temp  
                SET new_tuple_state TO a tuple formed by tuples of each row list in new_state  
                IF new_tuple_state NOT IN visited THEN  
                    ADD new_tuple_state TO visited  
                    APPEND a tuple of new_tuple_state neighbor and moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
