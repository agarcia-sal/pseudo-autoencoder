CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left attribute of p and the left attribute of q AND the result of is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a deque containing root  
        SET leftmost_value TO none  
        WHILE the LENGTH OF queue GREATER THAN zero DO  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one DO  
                SET node TO the element removed from the front of queue  
                IF i EQUALS zero THEN  
                    SET leftmost_value TO the val attribute of node  
                END IF  
                IF the left attribute of node NOT EQUALS none THEN  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node NOT EQUALS none THEN  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
