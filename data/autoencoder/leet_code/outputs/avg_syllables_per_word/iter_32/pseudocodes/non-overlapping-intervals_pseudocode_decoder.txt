CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS the empty collection THEN  
            RETURN zero  
        END IF  
        SORT intervals IN ASCENDING ORDER ACCORDING TO the element at position two of each interval  
        SET end TO the element at position two of the element at position one of intervals  
        SET count TO zero  
        FOR index FROM two TO the LENGTH OF intervals  
            IF the element at position one of the element at position index of intervals GREATER THAN OR EQUAL TO end THEN  
                SET end TO the element at position two of the element at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
