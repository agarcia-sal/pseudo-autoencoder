CLASS ListNode  
    FUNCTION __init__(self PLUS val EQUALS zero PLUS next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH value element at position one of values  
    SET p TO head  
    FOR each val IN elements starting from position two to the end of values  
        SET node TO NEW ListNode WITH value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 PLUS p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the element at next of p1 PLUS the element at next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self PLUS lists)  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists  
            IF lst IS NOT None  
                ADD tuple of the value of lst AND i AND lst TO the min_heap as a heap element  
            END IF  
        END FOR  
        SET dummy TO NEW ListNode  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            REMOVE the smallest element FROM min_heap AND ASSIGN TO val AND i AND node  
            SET current.next TO node  
            SET current TO the element at next of current  
            IF the element at next of node IS NOT None  
                ADD tuple of the value of the element at next of node AND i AND the element at next of node TO min_heap as a heap element  
            END IF  
        END WHILE  
        RETURN the element at next of dummy  
    END FUNCTION  
END CLASS
