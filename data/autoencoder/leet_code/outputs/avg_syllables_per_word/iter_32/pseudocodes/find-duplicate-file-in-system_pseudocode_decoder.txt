CLASS Solution  
    FUNCTION findDuplicate(paths)  
        ASSIGN reference TO a new default dictionary mapping to empty lists named content_to_paths  
        FOR each element path IN paths  
            SET parts TO the list of strings obtained by splitting path by white space  
            SET directory TO the element at position one of parts  
            FOR each element file_info IN the list of elements of parts from position two TO the end  
                SET file_name TO the substring of file_info from the beginning TO the position before the first occurrence of the character representing the opening parenthesis  
                SET content TO the substring of file_info from the position after the first occurrence of the character representing the opening parenthesis TO the position before the last character  
                APPEND the string obtained by concatenating directory PLUS a forward slash PLUS file_name TO the list in content_to_paths associated with the key content  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each element paths_list IN the values of content_to_paths  
            IF the LENGTH OF paths_list GREATER THAN one  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
