CLASS Solution  
    FUNCTION numOfArrays(n parameter n INTEGER, m parameter m INTEGER, k parameter k INTEGER) RETURNS INTEGER  
        SET constant MOD TO one billion PLUS seven  
        
        DECLARE dp AS three dimensional list initialized with zeroes with dimensions of length n PLUS one by length m PLUS one by length k PLUS one  
        
        FOR j FROM one TO m  
            SET element at position one at position j at position one of dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position i at position j at position p of dp TO element at position i at position j at position p of dp PLUS element at position i MINUS one at position x at position p MINUS one of dp  
                        SET element at position i at position j at position p of dp TO element at position i at position j at position p of dp MODULO MOD  
                    END FOR  
                    SET element at position i at position j at position p of dp TO element at position i at position j at position p of dp PLUS element at position i MINUS one at position j at position p of dp MULTIPLIED BY j  
                    SET element at position i at position j at position p of dp TO element at position i at position j at position p of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position n at position j at position k of dp  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
