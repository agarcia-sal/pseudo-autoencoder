CLASS TreeNode
    FUNCTION __init__ WITH val left right
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node WITH values
    IF values EQUALS empty
        RETURN nothing
    END IF
    SET root TO new TreeNode WITH element at position one of values
    SET i TO one
    SET queue TO a new double-ended queue
    APPEND root TO queue
    WHILE the LENGTH OF queue is greater than zero
        SET node TO the element removed from the front of queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing
            SET node.left TO new TreeNode WITH element at position i of values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing
            SET node.right TO new TreeNode WITH element at position i of values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree WITH p q
    IF p EQUALS nothing AND q EQUALS nothing
        RETURN True
    ELSE IF p EQUALS nothing OR q EQUALS nothing
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN the result of is_same_tree WITH p.left AND q.left AND the result of is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION recoverTree WITH root
        SET first TO nothing
        SET second TO nothing
        SET prev TO nothing
        SET curr TO root
        WHILE curr NOT EQUALS nothing
            IF curr.left EQUALS nothing
                IF prev NOT EQUALS nothing AND prev.val GREATER THAN curr.val
                    IF first EQUALS nothing
                        SET first TO prev
                    END IF
                    SET second TO curr
                END IF
                SET prev TO curr
                SET curr TO curr.right
            ELSE
                SET pred TO curr.left
                WHILE pred.right NOT EQUALS nothing AND pred.right NOT EQUALS curr
                    SET pred TO pred.right
                END WHILE
                IF pred.right EQUALS nothing
                    SET pred.right TO curr
                    SET curr TO curr.left
                ELSE
                    SET pred.right TO nothing
                    IF prev NOT EQUALS nothing AND prev.val GREATER THAN curr.val
                        IF first EQUALS nothing
                            SET first TO prev
                        END IF
                        SET second TO curr
                    END IF
                    SET prev TO curr
                    SET curr TO curr.right
                END IF
            END IF
        END WHILE
        IF first NOT EQUALS nothing AND second NOT EQUALS nothing
            SET temporary_value TO first.val
            SET first.val TO second.val
            SET second.val TO temporary_value
        END IF
    END FUNCTION
END CLASS
