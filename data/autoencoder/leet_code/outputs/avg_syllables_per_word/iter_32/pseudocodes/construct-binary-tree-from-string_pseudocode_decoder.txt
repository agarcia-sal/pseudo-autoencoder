CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position zero of values and Next as None  
    SET p TO head  
    FOR each val in the elements of values from position one TO the last position  
        SET node TO a new ListNode with val and Next as None  
        SET the Next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the predicate that the value of p1 EQUALS the value of p2 AND the recursive call to is_same_list with the Next of p1 and the Next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values and Left as None and Right as None  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the Left child of node TO a new TreeNode with the element at position i of values and Left as None and Right as None  
            APPEND the Left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the Right child of node TO a new TreeNode with the element at position i of values and Left as None and Right as None  
            APPEND the Right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the recursive call to is_same_tree with the Left child of p and the Left child of q AND the recursive call to is_same_tree with the Right child of p and the Right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS a digit OR the element at position index of s EQUALS a minus sign  
                INCREMENT index BY one  
            END WHILE  
            SET node TO a new TreeNode with the integer parsed from the substring from position start TO position index MINUS one of s and Left as None and Right as None  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS an opening parenthesis  
                SET node.left AND index TO the result of the recursive call to parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS an opening parenthesis  
                SET node.right AND index TO the result of the recursive call to parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS a closing parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND an unused variable TO the result of the call to parse_tree with zero  
        RETURN root  
    END FUNCTION  
END CLASS
