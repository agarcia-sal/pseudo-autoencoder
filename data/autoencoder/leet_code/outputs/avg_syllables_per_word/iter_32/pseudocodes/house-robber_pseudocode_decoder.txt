CLASS Solution  
    FUNCTION rob(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET element at position one of dp TO element at position one of nums  
        IF element at position one of nums GREATER THAN element at position two of nums  
            SET element at position two of dp TO element at position one of nums  
        ELSE  
            SET element at position two of dp TO element at position two of nums  
        END IF  
        FOR index i FROM three TO the LENGTH OF nums  
            IF element at position i MINUS one of dp GREATER THAN element at position i MINUS two of dp PLUS element at position i of nums  
                SET element at position i of dp TO element at position i MINUS one of dp  
            ELSE  
                SET element at position i of dp TO element at position i MINUS two of dp PLUS element at position i of nums  
            END IF  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
