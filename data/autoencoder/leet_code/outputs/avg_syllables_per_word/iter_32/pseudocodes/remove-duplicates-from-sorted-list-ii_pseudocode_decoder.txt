CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET head TO new ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO the last position  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self head)  
        SET dummy TO new ListNode with value zero and next head  
        SET prev TO dummy  
        WHILE head NOT EQUALS none  
            IF head.next NOT EQUALS none AND head.val EQUALS head.next.val  
                WHILE head.next NOT EQUALS none AND head.val EQUALS head.next.val  
                    SET head TO head.next  
                END WHILE  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            END IF  
            SET head TO head.next  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
