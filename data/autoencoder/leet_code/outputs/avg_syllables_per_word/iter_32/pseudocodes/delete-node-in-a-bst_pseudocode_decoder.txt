CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET root TO TreeNode(element at position one of values, none, none)  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND element at position i of values is not none  
            SET node.left TO TreeNode(element at position i of values, none, none)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values is not none  
            SET node.right TO TreeNode(element at position i of values, none, none)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is none AND q is none  
        RETURN True  
    ELSE IF p is none OR q is none  
        RETURN False  
    ELSE IF value of p.val NOT EQUALS value of q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root is none  
            RETURN none  
        END IF  
        IF root.val GREATER THAN key  
            SET root.left TO self.deleteNode(root.left, key)  
            RETURN root  
        END IF  
        IF root.val LESS THAN key  
            SET root.right TO self.deleteNode(root.right, key)  
            RETURN root  
        END IF  
        IF root.left is none  
            RETURN root.right  
        END IF  
        IF root.right is none  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left is not none  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
