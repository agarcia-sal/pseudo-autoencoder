CLASS Solution  
    FUNCTION myPow(x FLOATING POINT number, n INTEGER number) RETURNS FLOATING POINT number  
        FUNCTION qpow(a FLOATING POINT number, n INTEGER number) RETURNS FLOATING POINT number  
            SET ans TO one  
            WHILE n GREATER THAN zero holds  
                IF n MODULO two EQUALS one THEN  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two discarding remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero THEN  
            RETURN qpow WITH a SET TO x AND n SET TO n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH a SET TO x AND n SET TO n MINUS n MULTIPLIED BY two  
        END IF  
    END FUNCTION  
END CLASS
