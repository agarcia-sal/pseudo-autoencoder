CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position i of nums NOT EQUALS element at position element at position i of nums MINUS one of nums  
                SET temporary_first TO element at position i of nums  
                SET temporary_second TO element at position element at position i of nums MINUS one of nums  
                SET element at position element at position i of nums MINUS one of nums TO temporary_first  
                SET element at position i of nums TO temporary_second  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each index i AND value v IN the enumeration of nums  
            IF v NOT EQUALS i PLUS one  
                APPEND v TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
