CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET end_with_zero TO zero  
        SET end_with_one TO zero  
        SET has_zero TO binary CONTAINS character zero  
        FOR each bit IN binary  
            IF bit EQUALS character one  
                SET end_with_one TO the sum of end_with_zero PLUS end_with_one PLUS one MODULO MOD  
            ELSE  
                SET end_with_zero TO the sum of end_with_zero PLUS end_with_one MODULO MOD  
            END IF  
        END FOR  
        RETURN the sum of end_with_zero PLUS end_with_one PLUS has_zero MODULO MOD  
    END FUNCTION  
END CLASS
