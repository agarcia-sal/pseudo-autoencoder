CLASS UnionFind  
    FUNCTION __init__(self)  
        SET self.parent TO an empty mapping  
        SET self.count TO zero  
    END FUNCTION  
  
    FUNCTION make_set(self, x)  
        IF x NOT IN self.parent THEN  
            SET element x in self.parent TO x itself  
            INCREMENT self.count BY one  
        END IF  
    END FUNCTION  
  
    FUNCTION find(self, x)  
        IF the element at key x in self.parent NOT EQUALS x itself THEN  
            SET element at key x in self.parent TO the result of calling find with argument the element at key x in self.parent  
        END IF  
        RETURN the element at key x in self.parent  
    END FUNCTION  
  
    FUNCTION union(self, x, y)  
        SET rootX TO the result of calling find with argument x  
        SET rootY TO the result of calling find with argument y  
        IF rootX NOT EQUALS rootY THEN  
            SET element at key rootX in self.parent TO rootY  
            DECREMENT self.count BY one  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION numIslands2(self, m, n, positions)  
        FUNCTION is_valid(x, y)  
            RETURN the logical AND of the logical AND of the comparisons zero LESS THAN OR EQUAL TO x AND x LESS THAN m AND zero LESS THAN OR EQUAL TO y AND y LESS THAN n  
        END FUNCTION  
  
        SET directions TO a list containing tuples of the values one and zero; negative one and zero; zero and one; zero and negative one  
        SET uf TO a new instance of UnionFind  
        SET grid TO a list of m elements each being a list of n zeros  
        SET result TO an empty list  
  
        FOR each element named r and each element named c IN positions  
            IF element at position c of element at position r of grid EQUALS one THEN  
                APPEND uf.count TO result  
                CONTINUE to the next iteration  
            END IF  
  
            SET element at position c of element at position r of grid TO one  
            CALL make_set on uf with argument as the tuple of r and c  
  
            FOR each element named dr and each element named dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF calling is_valid with arguments nr and nc RETURNS true AND element at position nc of element at position nr of grid EQUALS one THEN  
                    CALL union on uf with arguments the tuple of r and c AND the tuple of nr and nc  
                END IF  
            END FOR  
  
            APPEND uf.count TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
