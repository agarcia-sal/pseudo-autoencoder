CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INSTANTIATE empty queue radiant  
        INSTANTIATE empty queue dire  
        
        FOR each position index AND character senator IN the enumeration of senate  
            IF senator EQUALS the character representing Radiant  
                APPEND position index TO radiant  
            ELSE  
                APPEND position index TO dire  
            END IF  
        END FOR  
        
        WHILE the LENGTH OF radiant GREATER THAN zero AND the LENGTH OF dire GREATER THAN zero  
            REMOVE and GET the first element FROM radiant AS r_index  
            REMOVE and GET the first element FROM dire AS d_index  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF the LENGTH OF radiant GREATER THAN zero  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
