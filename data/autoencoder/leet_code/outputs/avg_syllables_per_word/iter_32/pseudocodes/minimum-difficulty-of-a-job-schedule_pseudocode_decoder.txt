CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN the NEGATIVE OF one  
        END IF  
          
        FUNCTION dp(i, day_count)  
            IF day_count EQUALS one  
                SET max_value TO the HIGHEST VALUE among elements from position i TO the END of jobDifficulty  
                RETURN max_value  
            END IF  
              
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR position j FROM i TO the LENGTH OF jobDifficulty MINUS day_count PLUS one MINUS one  
                SET current_max TO the GREATER OF current_max AND element at position j of jobDifficulty  
                SET candidate TO current_max PLUS dp(j PLUS one, day_count MINUS one)  
                SET min_difficulty TO the LESSER OF min_difficulty AND candidate  
            END FOR  
            RETURN min_difficulty  
        END FUNCTION  
          
        RETURN dp(zero, d)  
    END FUNCTION  
END CLASS
