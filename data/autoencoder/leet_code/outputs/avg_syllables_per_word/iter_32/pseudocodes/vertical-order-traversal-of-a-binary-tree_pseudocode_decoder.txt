CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero holds  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left attribute of p AND the left attribute of q AND the result of is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root EQUALS None  
            RETURN an empty list  
        END IF  
        INITIALIZE node_dict AS a dictionary with default value an empty list  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node EQUALS None  
                RETURN  
            END IF  
            APPEND the tuple of row AND the val attribute of node TO the list in node_dict at key col  
            SET min_col TO the lesser of min_col AND col  
            SET max_col TO the greater of max_col AND col  
            CALL dfs with the left attribute of node AND row PLUS one AND col MINUS one  
            CALL dfs with the right attribute of node AND row PLUS one AND col PLUS one  
        END FUNCTION  

        CALL dfs with root AND zero AND zero  

        INITIALIZE result AS an empty list  
        FOR col FROM min_col TO max_col INCLUSIVE  
            SET current_list TO the list obtained from sorting the elements in node_dict at key col by the first element of the tuple THEN by the second element of the tuple  
            APPEND the list of second elements of each tuple in current_list TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
