CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO a TWO DIMENSIONAL list OF size n BY n FILLED WITH zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AT row i OF dp TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i OF arr EQUALS element at position i PLUS one OF arr  
                SET element at position i AT row i PLUS one OF dp TO one  
            ELSE  
                SET element at position i AT row i PLUS one OF dp TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i OF arr EQUALS element at position j OF arr  
                    SET element at position i AT row j OF dp TO element at position i PLUS one AT row j MINUS one OF dp  
                ELSE  
                    SET element at position i AT row j OF dp TO positive infinity  
                    FOR k FROM i TO j MINUS one  
                        SET element at position i AT row j OF dp TO the MINIMUM OF element at position i AT row j OF dp AND element at position i AT row k OF dp PLUS element at position k PLUS one AT row j OF dp  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero AT row n MINUS one OF dp  
    END FUNCTION  
END CLASS
