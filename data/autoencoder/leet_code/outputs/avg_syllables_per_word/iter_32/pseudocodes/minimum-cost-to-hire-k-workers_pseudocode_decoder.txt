CLASS Solution  
    FUNCTION mincostToHireWorkers(quality LIST_OF INTEGERS wage LIST_OF INTEGERS k INTEGER) RETURNS FLOAT  
        SET workers TO sorted list of elements where each element is the result of wage element DIVIDED BY quality element and paired with quality element for each corresponding element from wage and quality lists  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each ratio and quality_value IN workers  
            PUSH negative quality_value ONTO max_heap  
            INCREMENT total_quality BY quality_value  
            IF length of max_heap GREATER THAN k THEN  
                REMOVE and GET the smallest element FROM max_heap AND CALL it removed_quality  
                INCREMENT total_quality BY removed_quality  
            END IF  
            IF length of max_heap EQUALS k THEN  
                SET calculated_cost TO total_quality MULTIPLIED BY ratio  
                IF calculated_cost LESS THAN min_cost THEN  
                    SET min_cost TO calculated_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
