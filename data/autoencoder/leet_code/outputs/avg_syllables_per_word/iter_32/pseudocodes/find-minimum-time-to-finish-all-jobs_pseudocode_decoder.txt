CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeroes with the LENGTH OF two RAISED TO THE POWER OF n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET subset_sums at POSITION mask OR two RAISED TO THE POWER OF i TO the VALUE OF subset_sums at POSITION mask PLUS the element at POSITION i of jobs  
            END FOR  
        END FOR  

        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN the element at POSITION mask of subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET res TO the MINIMUM OF res AND the MAXIMUM OF dp(k MINUS one, mask XOR submask) AND the element at POSITION submask of subset_sums  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  

        RETURN dp(k, two RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
