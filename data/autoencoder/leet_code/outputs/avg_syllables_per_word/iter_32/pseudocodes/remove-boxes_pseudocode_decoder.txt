CLASS Solution  
    FUNCTION removeBoxes(boxes)  
        FUNCTION dp(l, r, k, memo)  
            IF l GREATER THAN r  
                RETURN zero  
            END IF  
            IF tuple of l and r and k EXISTS IN memo  
                RETURN element at tuple of l and r and k IN memo  
            END IF  
            WHILE r GREATER THAN l AND element at position r of boxes EQUALS element at position r MINUS one of boxes  
                DECREMENT r BY one  
                INCREMENT k BY one  
            END WHILE  
            SET res TO call dp WITH parameters l r MINUS one zero memo PLUS product of sum of k and one AND sum of k and one  
            FOR i FROM l TO r MINUS one  
                IF element at position i of boxes EQUALS element at position r of boxes  
                    SET res TO maximum OF res AND call dp WITH parameters l i sum of k and one memo PLUS call dp WITH parameters i PLUS one r MINUS one zero memo  
                END IF  
            END FOR  
            SET element at tuple of l and r and k IN memo TO res  
            RETURN res  
        END FUNCTION  
        RETURN call dp WITH parameters zero LENGTH OF boxes MINUS one zero empty dictionary  
    END FUNCTION  
END CLASS
