CLASS Solution  
    FUNCTION kMirror(k n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS the REVERSE OF s  
        END FUNCTION  
  
        FUNCTION to_base_k(num k)  
            IF num EQUALS zero  
                RETURN the STRING representation OF zero  
            END IF  
            SET digits TO an EMPTY LIST  
            WHILE num GREATER THAN zero  
                APPEND the REMAINDER OF num DIVIDED BY k TO digits  
                SET num TO num DIVIDED BY k WITHOUT the FRACTIONAL PART  
            END WHILE  
            RETURN the CONCATENATION OF the STRING representation OF each ELEMENT IN the REVERSE OF digits  
        END FUNCTION  
  
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD the STRING representation OF i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF length MINUS one BY two  
                SET end TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF length PLUS one BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO the STRING representation OF half  
                    IF the REMAINDER OF length DIVIDED BY two EQUALS zero  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF the SUBSTRING OF half_str FROM the SECOND LAST CHARACTER TO the FIRST CHARACTER IN REVERSE ORDER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
  
        SET k_mirror_numbers TO an EMPTY LIST  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers LESS THAN n  
            FOR pal IN generate_palindromes(length)  
                SET num TO the INTEGER VALUE OF pal  
                IF is_palindrome(to_base_k(num k))  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
  
        RETURN the SUM OF k_mirror_numbers  
    END FUNCTION  
END CLASS
