CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO a dictionary counting the frequency of each element in nums  
        SET dp TO a list of zeroes with length TWO RAISED TO THE POWER OF ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two RAISED TO THE POWER OF the frequency of one in count MODULO MOD  
                CONTINUE to next iteration  
            END IF  
            IF num NOT IN prime_factors  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO the list of prime factors for num in prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two RAISED TO THE POWER OF prime MINUS one  
            END FOR  
            
            FOR m FROM TWO RAISED TO THE POWER OF ten MINUS one DOWN TO zero  
                IF bitwise AND of m AND mask EQUALS zero  
                    SET element at position m OR mask of dp TO (element at position m OR mask of dp PLUS element at position m of dp MULTIPLIED BY the frequency of num in count) MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements in dp excluding element at position zero  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
