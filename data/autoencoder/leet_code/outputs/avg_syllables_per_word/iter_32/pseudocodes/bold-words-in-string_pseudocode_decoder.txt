CLASS Solution  
    FUNCTION boldWords(words s LIST OF STRING, s STRING) RETURNS STRING  
        SET bold TO list of boolean values all set to FALSE with the LENGTH OF s  
        FOR each word IN words  
            SET start TO the POSITION OF the FIRST OCCURRENCE OF word IN s  
            WHILE start NOT EQUALS negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position i of bold TO TRUE  
                END FOR  
                SET start TO the POSITION OF the FIRST OCCURRENCE OF word IN s STARTING FROM position start PLUS one  
            END WHILE  
        END FOR  
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF element at position i of bold EQUALS TRUE  
                APPEND the string indicating the start of bold to result  
                WHILE i LESS THAN the LENGTH OF s AND element at position i of bold EQUALS TRUE  
                    APPEND element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string indicating the end of bold TO result  
            ELSE  
                APPEND element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
