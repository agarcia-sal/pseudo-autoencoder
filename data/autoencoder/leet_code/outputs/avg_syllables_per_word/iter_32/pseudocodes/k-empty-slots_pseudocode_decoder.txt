CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO an empty list  
        FOR each day and position IN bulbs enumerated starting from one  
            SET i TO the position where position can be inserted in bloom_days to maintain sorted order by finding the leftmost insertion point of position in bloom_days  
            IF i is greater than zero AND the subtraction of the element at position i minus one of bloom_days FROM position MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i is less than the length of bloom_days AND the subtraction of position FROM the element at position i of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position at position i of bloom_days  
        END FOR  
        RETURN the negative integer one  
    END FUNCTION  
END CLASS
