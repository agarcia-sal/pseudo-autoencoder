CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_representation TO the binary representation of n starting from the second character  
        SET length TO the length of bin_representation  
        SET dp TO a list of zeros with the size length PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  
        FOR i FROM two TO length  
            SET element at position i of dp TO element at position i MINUS one of dp PLUS element at position i MINUS two of dp  
        END FOR  
        SET result TO zero  
        SET previous_bit TO zero  
        FOR i FROM zero TO length MINUS one  
            IF element at position i of bin_representation EQUALS the character one  
                INCREMENT result BY element at position length MINUS i MINUS one of dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET previous_bit TO one IF element at position i of bin_representation EQUALS the character one OTHERWISE zero  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
