CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR index i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            CALL heappush ON min_heap WITH the value EQUAL TO the SUM OF the element at position i of nums1 AND the element at position zero of nums2 FOLLOWED BY i FOLLOWED BY zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result LESS THAN k  
            CALL heappop ON min_heap AND ASSIGN THE RESULTS TO discarded VALUE FOLLOWED BY i FOLLOWED BY j  
            APPEND a list containing the element at position i of nums1 AND the element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                CALL heappush ON min_heap WITH the value EQUAL TO the SUM OF the element at position i of nums1 AND the element at position j PLUS one of nums2 FOLLOWED BY i FOLLOWED BY j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
