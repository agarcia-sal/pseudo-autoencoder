CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET new_character_code TO the character code of element at position index of s MINUS the character code of the lowercase letter a PLUS element at position index of shifts  
            SET new_character_code TO new_character_code MODULO twenty six  
            SET new_character_code TO new_character_code PLUS the character code of the lowercase letter a  
            SET new_character TO the character corresponding to new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN all elements of result concatenated into a single string  
    END FUNCTION  
END CLASS
