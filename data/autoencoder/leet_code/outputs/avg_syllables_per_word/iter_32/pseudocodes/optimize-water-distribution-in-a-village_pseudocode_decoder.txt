CLASS Solution  
    FUNCTION minCostToSupplyWater(n wells pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list containing zero PLUS index PLUS one PLUS element at position index of wells TO pipes  
        END FOR  
        SORT pipes BY the element at position two of each element in ascending order  
        SET parent TO list of integers FROM zero TO n  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each pipe_element IN pipes  
            SET house1 TO element at position zero of pipe_element  
            SET house2 TO element at position one of pipe_element  
            SET cost TO element at position two of pipe_element  
            IF union(house1 house2) IS true  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
