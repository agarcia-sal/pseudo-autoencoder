CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(nums parameter k parameter) RETURNS integer  
        IF k parameter LESS THAN OR EQUAL TO one THEN  
            RETURN zero  
        END IF  
        
        SET product TO one  
        SET left TO zero  
        SET count TO zero  
        
        FOR right FROM zero TO the LENGTH OF nums parameter MINUS one  
            SET product TO product MULTIPLIED BY element at position right of nums parameter  
            
            WHILE product GREATER THAN OR EQUAL TO k parameter AND left LESS THAN OR EQUAL TO right  
                SET product TO product DIVIDED BY element at position left of nums parameter  
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT count BY right MINUS left PLUS one  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
