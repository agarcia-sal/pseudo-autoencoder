CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple of the element at position zero zero of grid PLUS the integer zero PLUS the integer zero  
        ADD the tuple of integer zero PLUS integer zero TO visited  
        SET directions TO a list containing the ordered pairs zero one PLUS one zero PLUS zero MINUS one PLUS MINUS one zero  
        WHILE min_heap is not empty  
            REMOVE the smallest element from min_heap AND ASSIGN its components TO t PLUS i PLUS j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each ordered pair di PLUS dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF zero LESS THAN OR EQUAL TO ni AND ni LESS THAN n AND zero LESS THAN OR EQUAL TO nj AND nj LESS THAN n AND the tuple ni PLUS nj NOT IN visited  
                    ADD the tuple ni PLUS nj TO visited  
                    INSERT into min_heap the tuple of the greater value BETWEEN t AND the element at position ni nj of grid PLUS ni PLUS nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN the integer minus one  
    END FUNCTION  
END CLASS
