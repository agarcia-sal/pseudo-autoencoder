CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one point zero  
        END IF  
        
        SET n TO n PLUS twenty-four DIVIDED BY twenty-five  
        
        SET memo TO an empty collection  
        
        FUNCTION dp(a, b)  
            IF a AND b form a key contained in memo  
                RETURN the value associated with the key a AND b in memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            
            SET prob TO point two five MULTIPLIED BY the sum of dp of a MINUS four AND b PLUS dp of a MINUS three AND b MINUS one PLUS dp of a MINUS two AND b MINUS two PLUS dp of a MINUS one AND b MINUS three  
            
            ASSIGN prob TO memo at key a AND b  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp of n AND n  
    END FUNCTION  
END CLASS
