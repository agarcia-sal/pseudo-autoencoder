CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        ASSIGN reference TO an empty double ended queue named deque_  
        
        FOR each pair of values xj and yj IN points  
            
            WHILE deque_ is not empty AND the value xj MINUS the first element at position one of the first tuple in deque_ is GREATER THAN k  
                REMOVE the first element from deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the larger value BETWEEN max_value AND yj PLUS xj PLUS the first element at position zero of the first tuple in deque_  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj IS GREATER THAN OR EQUAL TO the last element at position zero of the last tuple in deque_  
                REMOVE the last element from deque_  
            END WHILE  
            
            APPEND the tuple consisting of yj MINUS xj and xj TO the end of deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
