CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the number of characters in s  
            SET count TO zero  
            FOR each i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm of nine AND i MINUS one  
            END FOR  
            SET seen TO an empty set  
            FOR each i AND digit FROM enumeration of s  
                FOR each d FROM zero TO digit MINUS one  
                    IF i EQUALS zero THEN  
                        SET start_digit TO one  
                    ELSE  
                        SET start_digit TO zero  
                    END IF  
                    IF d GREATER THAN OR EQUAL TO start_digit THEN  
                        IF d NOT IN seen THEN  
                            INCREMENT count BY perm of nine MINUS i AND length MINUS i MINUS one  
                        END IF  
                    END IF  
                END FOR  
                IF digit IN seen THEN  
                    BREAK  
                END IF  
                ADD digit TO seen  
            END FOR  
            IF the above loop terminated without break THEN  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        RETURN n MINUS count_unique_digits(n)  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero THEN  
            RETURN one  
        END IF  
        SET result TO one  
        FOR each i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
