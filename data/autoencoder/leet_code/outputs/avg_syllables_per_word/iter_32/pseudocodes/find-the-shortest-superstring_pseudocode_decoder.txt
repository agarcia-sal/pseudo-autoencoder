CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one BY DECREMENTING BY one  
                IF the SUBSTRING of element at position i of words from the LENGTH OF element at position i of words MINUS k PLUS one TO the LENGTH OF element at position i of words EQUALS the SUBSTRING of element at position j of words from one TO k THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS two RAISED TO THE POWER OF n MINUS one THEN  
                RETURN zero AND an EMPTY STRING  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO an EMPTY STRING  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND OF mask AND two RAISED TO THE POWER OF j EQUALS zero THEN  
                    SET length AND path TO the RESULT OF dp WITH ARGUMENTS (the BITWISE OR OF mask AND two RAISED TO THE POWER OF j) AND j  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS overlap(i, j)  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the CONCATENATION OF the SUBSTRING of element at position j of words FROM overlap(i, j) PLUS one TO THE END OF THE STRING AND path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO an EMPTY STRING  
        FOR i FROM zero TO n MINUS one  
            SET length AND path TO the RESULT OF dp WITH ARGUMENTS two RAISED TO THE POWER OF i AND i  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO the CONCATENATION OF element at position i of words AND path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
