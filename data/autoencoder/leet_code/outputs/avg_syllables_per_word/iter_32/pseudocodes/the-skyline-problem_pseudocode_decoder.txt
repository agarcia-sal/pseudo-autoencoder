CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each left right height IN buildings  
            APPEND a tuple of left position negative height right position TO events  
            APPEND a tuple of right position zero zero TO events  
        END FOR  
        
        SORT events by the first element of each tuple then by the second element then by the third element in ascending order  
        
        SET result TO a list containing one element a list of zero zero  
        SET max_heap TO a list containing one element a tuple of zero and positive infinity  
        
        FOR each x negH r IN events  
            WHILE the second element of the first tuple in max_heap is less than or equals x  
                REMOVE the first tuple from max_heap using heap pop operation  
            END WHILE  
            
            IF negH is not equals zero  
                ADD the tuple of negH and r to max_heap using heap push operation  
            END IF  
            
            SET current_height TO the negation of the first element of the first tuple in max_heap  
            SET last_height TO the second element of the last list in result  
            IF last_height is not equals current_height  
                APPEND a list of x and current_height TO result  
            END IF  
        END FOR  
        
        IF the first element of result is a list of zero and zero  
            REMOVE the first element from result  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
