CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND the LENGTH OF stack GREATER THAN zero AND the element at the last position of stack LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  
        
        FUNCTION merge(arr1 arr2)  
            SET result TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR index FROM one TO total_length  
                IF arr1 GREATER THAN arr2  
                    APPEND the element at position one of arr1 TO result  
                    REMOVE the element at position one FROM arr1  
                ELSE  
                    APPEND the element at position one of arr2 TO result  
                    REMOVE the element at position one FROM arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(maxSingleArray(nums1 i) maxSingleArray(nums2 k MINUS i))  
                IF max_number LESS THAN candidate  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
