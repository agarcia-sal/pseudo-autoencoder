CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO LENGTH OF strs MINUS one  
            SET reversed_string TO element at position index of strs reversed in order  
            IF reversed_string GREATER THAN element at position index of strs  
                SET element at position index of strs TO reversed_string  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs WITH strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO LENGTH OF strs MINUS one  
            FOR is_reversed IN list containing False AND True  
                IF is_reversed EQUALS True  
                    SET current TO element at position index of strs reversed in order  
                ELSE  
                    SET current TO element at position index of strs  
                END IF  
                SET remaining TO empty string  
                FOR position FROM index PLUS one TO index PLUS LENGTH OF strs MINUS one  
                    APPEND element at position position of doubled_strs TO remaining  
                END FOR  
                
                FOR cut_position FROM zero TO LENGTH OF current  
                    SET candidate TO substring of current from position cut_position TO end CONCATENATED WITH remaining CONCATENATED WITH substring of current from position zero TO position cut_position MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
