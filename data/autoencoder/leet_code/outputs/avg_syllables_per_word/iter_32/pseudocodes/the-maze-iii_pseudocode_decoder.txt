CLASS Solution  
    FUNCTION findShortestWay(maze AS List OF List OF integer ball AS List OF integer hole AS List OF integer) RETURNS string  
        SET directions TO list containing tuples of zero and one and letter r tuple of one and zero and letter d tuple of zero and negative one and letter l tuple of negative one and zero and letter u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO tuple of elements of ball  
        SET hole TO tuple of elements of hole  
        SET pq TO list containing tuple of zero empty string and start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE and RETURN smallest element from pq INTO dist path and tuple with x and y  
            
            IF x EQUALS element at position zero of hole AND y EQUALS element at position one of hole  
                RETURN path  
            END IF  
            
            IF tuple with x and y IS IN visited  
                CONTINUE to next iteration of WHILE  
            END IF  
            
            ADD tuple with x and y TO visited  
            
            FOR each tuple with dx dy and direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    
                    IF new_x EQUALS element at position zero of hole AND new_y EQUALS element at position one of hole  
                        BREAK from WHILE  
                    END IF  
                END WHILE  
                
                IF tuple with new_x and new_y IS NOT IN visited  
                    INSERT into pq the tuple with dist PLUS count path concatenated with direction and tuple with new_x and new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
