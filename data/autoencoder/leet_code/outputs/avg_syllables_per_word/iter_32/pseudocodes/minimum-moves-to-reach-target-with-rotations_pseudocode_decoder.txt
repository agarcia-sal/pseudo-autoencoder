CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        INITIALIZE queue TO an empty double ended queue containing a tuple of zero zero zero one zero representing the row one column one row two column two and step count  
        INITIALIZE visited TO a set containing a tuple of zero zero zero one representing visited states  

        WHILE the queue is not empty  
            EXTRACT r1 c1 r2 c2 steps FROM the FIRST element of queue and REMOVE it from queue  

            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one THEN  
                RETURN steps  
            END IF  

            IF c2 PLUS one LESS THAN n AND the element at row r1 column c1 PLUS one of grid EQUALS zero AND the element at row r2 column c2 PLUS one of grid EQUALS zero AND the tuple of r1 c1 PLUS one r2 c2 PLUS one is NOT IN visited THEN  
                ADD the tuple of r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  

            IF r2 PLUS one LESS THAN n AND the element at row r1 PLUS one column c1 of grid EQUALS zero AND the element at row r2 PLUS one column c2 of grid EQUALS zero AND the tuple of r1 PLUS one c1 r2 PLUS one c2 is NOT IN visited THEN  
                ADD the tuple of r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  

            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND the element at row r1 PLUS one column c1 of grid EQUALS zero AND the element at row r1 PLUS one column c2 of grid EQUALS zero AND the tuple of r1 c1 r1 PLUS one c1 is NOT IN visited THEN  
                ADD the tuple of r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple of r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  

            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND the element at row r1 column c1 PLUS one of grid EQUALS zero AND the element at row r2 column c2 PLUS one of grid EQUALS zero AND the tuple of r1 c1 r1 c1 PLUS one is NOT IN visited THEN  
                ADD the tuple of r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple of r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
