CLASS Solution  
    FUNCTION smallestDistancePair(nums WITH TYPE List OF Integer k WITH TYPE Integer) RETURNS Integer  
        SORT the list nums IN ascending order  
        SET n TO the LENGTH OF the list nums  
        
        FUNCTION count(mid WITH TYPE Integer) RETURNS Integer  
            SET count TO zero  
            SET left TO zero  
            FOR right FROM zero TO n MINUS one  
                WHILE the element at position right of nums MINUS the element at position left of nums GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY the value of right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the element at position n MINUS one of nums MINUS the element at position zero of nums  
        WHILE left LESS THAN right  
            SET mid TO the integer division of the sum of left PLUS right BY two  
            IF the result of count with argument mid LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
