CLASS Solution  
    FUNCTION findLength(nums1 PARAMETER nums1, nums2 PARAMETER nums2) RETURNS integer  
        SET dp TO a two dimensional list of zeros with the number of rows EQUAL TO the LENGTH OF nums1 PLUS one and the number of columns EQUAL TO the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF element at position i MINUS one of nums1 EQUALS element at position j MINUS one of nums2  
                    SET element at row i column j of dp TO element at row i MINUS one column j MINUS one of dp PLUS one  
                    SET max_length TO the maximum of max_length AND element at row i column j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
