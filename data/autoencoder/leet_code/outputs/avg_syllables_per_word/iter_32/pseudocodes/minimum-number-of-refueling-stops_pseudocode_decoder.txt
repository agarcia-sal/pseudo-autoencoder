CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list consisting of target AND zero TO stations  

        FOR each element IN stations  
            SET position TO element at position one of element  
            SET fuel TO element at position two of element  

            WHILE the LENGTH OF max_heap is greater than zero AND current_fuel LESS THAN position  
                REMOVE the smallest element FROM max_heap AND ASSIGN it to removed_fuel  
                SET current_fuel TO current_fuel MINUS removed_fuel MULTIPLIED BY minus one  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel LESS THAN position  
                RETURN minus one  
            END IF  

            ADD fuel MULTIPLIED BY minus one TO max_heap maintaining heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
