CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty set  
        SET stack TO list containing zero  
        WHILE the LENGTH OF stack GREATER THAN zero  
            REMOVE last element FROM stack AND ASSIGN TO current_room  
            ADD current_room TO visited  
            FOR each key IN element at position current_room of rooms  
                IF key NOT IN visited THEN  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF visited EQUALS the LENGTH OF rooms THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
