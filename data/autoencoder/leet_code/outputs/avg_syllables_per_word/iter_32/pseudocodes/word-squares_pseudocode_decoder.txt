CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO a new collection with default value as empty list  
        FOR each word IN words  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO the substring from position zero TO position index MINUS one of word  
                APPEND word TO the list of prefix_to_words AT prefix  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the element at position zero of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO the concatenation of the element at position the LENGTH OF square of each word IN square  
            FOR each candidate IN the list of prefix_to_words AT prefix  
                CALL backtrack WITH argument the concatenation of square WITH a list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack WITH argument a list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
