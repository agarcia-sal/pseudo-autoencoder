CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            SET difference at position zero of p1 MINUS difference at position zero of p2 TO first_difference  
            SET difference at position one of p1 MINUS difference at position one of p2 TO second_difference  
            SET first_difference MULTIPLIED BY first_difference PLUS second_difference MULTIPLIED BY second_difference TO sum_of_squares  
            SET the square root of sum_of_squares TO result_distance  
            RETURN result_distance  
        END FUNCTION  
        
        FUNCTION circle_center(p1, p2, r)  
            SET distance between p1 and p2 by calling distance FUNCTION TO d  
            IF d GREATER THAN two MULTIPLIED BY r THEN  
                RETURN None  
            END IF  
            SET r MULTIPLIED BY r MINUS d DIVIDED BY two MULTIPLIED BY d DIVIDED BY two TO inside_square_root  
            SET the square root of inside_square_root TO a  
            SET sum of difference at position zero of p1 PLUS difference at position zero of p2 DIVIDED BY two TO h_x  
            SET sum of difference at position one of p1 PLUS difference at position one of p2 DIVIDED BY two TO h_y  
            SET difference at position one of p2 MINUS difference at position one of p1 MULTIPLIED BY a DIVIDED BY d TO dx  
            SET difference at position zero of p2 MINUS difference at position zero of p1 MULTIPLIED BY a DIVIDED BY d TO dy  
            SET list containing two elements first as h_x MINUS dx AND h_y PLUS dy second as h_x PLUS dx AND h_y MINUS dy TO center_points  
            RETURN center_points  
        END FUNCTION  
        
        FUNCTION is_inside(circle, point, r)  
            SET distance between circle and point by calling distance FUNCTION TO dist  
            IF dist LESS THAN OR EQUAL TO r THEN  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
        
        SET the length of darts TO n  
        IF n EQUALS one THEN  
            RETURN one  
        END IF  
        
        SET one TO max_count  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET circle_center by calling circle_center FUNCTION with difference at position i of darts difference at position j of darts and r TO centers  
                IF centers NOT EQUALS None THEN  
                    FOR each center IN centers  
                        SET zero TO count  
                        FOR each dart IN darts  
                            IF is_inside by calling is_inside FUNCTION with center dart and r EQUALS True THEN  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count THEN  
                            SET count TO max_count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_count  
    END FUNCTION  
END CLASS
