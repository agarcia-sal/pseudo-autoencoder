CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF element at position zero of nums MINUS twenty four LESS THAN one times ten to the power of negative six  
            END IF  
            
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY list  
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation IN the list of operations consisting of true division multiplication addition subtraction  
                            IF the operation EQUALS addition OR the operation EQUALS multiplication AND j GREATER THAN i  
                                CONTINUE to the next iteration  
                            END IF  
                            
                            IF the operation NOT EQUALS true division OR element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying the operation TO element at position i of nums AND element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        IMPORT true division multiplication addition subtraction FROM operator  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
