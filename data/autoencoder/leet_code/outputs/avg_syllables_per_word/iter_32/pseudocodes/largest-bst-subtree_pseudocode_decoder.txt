CLASS TreeNode  
    FUNCTION __init__ with parameters self val defaulted to zero left defaulted to None right defaulted to None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the collection values is empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty dequeue structure  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF the value of p is NOT EQUALS to the value of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with parameters the left child of p AND the left child of q AND the result of calling is_same_tree with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree with parameters self root  
        FUNCTION dfs with parameter node  
            IF node is None  
                RETURN positive infinity value NEGATIVE infinity value zero  
            END IF  
            SET left_min left_max left_size TO the result of calling dfs with the left child of node  
            SET right_min right_max right_size TO the result of calling dfs with the right child of node  
            IF left_max is LESS THAN the value of node AND the value of node is LESS THAN right_min  
                RETURN the minimum of the value of node AND left_min the maximum of the value of node AND right_max the sum of left_size PLUS right_size PLUS one  
            ELSE  
                RETURN negative infinity value positive infinity value the maximum of left_size AND right_size  
            END IF  
        END FUNCTION  
        RETURN the third element of the result of calling dfs with parameter root  
    END FUNCTION  
END CLASS
