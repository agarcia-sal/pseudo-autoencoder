CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values and next as None  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last position of values  
        SET node TO a new ListNode with val and next as None  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the condition that the val of p1 EQUALS the val of p2 AND is_same_list of the next of p1 AND the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(self, head, k)  
        FUNCTION reverseLinkedList(start, end)  
            SET prev TO None  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO the next of curr  
                SET the next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the next of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO a new ListNode with val zero and next as None  
        SET the next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR each iteration FROM one TO k  
                SET kth_node TO the next of kth_node  
                IF kth_node IS None  
                    RETURN the next of dummy  
                END IF  
            END FOR  

            SET next_group_start TO the next of kth_node  
            SET the next of kth_node TO None  
            SET new_group_start AND new_group_end TO reverseLinkedList with the next of prev_group_end AND kth_node  
            SET the next of prev_group_end TO new_group_start  
            SET the next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the next of dummy  
    END FUNCTION  
END CLASS
