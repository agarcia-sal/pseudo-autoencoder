CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a mapping from each number to the quantity of times that number occurs in nums  
        SET unique_nums TO a sorted list of the keys of count  
        SET dp TO a list of zeros with length equal to the length of unique_nums PLUS one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the count of the element at position zero of unique_nums  
        FOR i FROM two TO the length of unique_nums PLUS one  
            SET num TO the element at position i MINUS one of unique_nums  
            IF num EQUALS the element at position i MINUS two of unique_nums PLUS one  
                SET the element at position i of dp TO the maximum of the element at position i MINUS one of dp AND the element at position i MINUS two of dp PLUS num MULTIPLIED BY the count of num  
            ELSE  
                SET the element at position i of dp TO the element at position i MINUS one of dp PLUS num MULTIPLIED BY the count of num  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
