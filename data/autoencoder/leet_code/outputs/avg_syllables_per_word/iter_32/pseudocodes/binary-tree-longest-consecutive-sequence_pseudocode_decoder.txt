CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values list is empty  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    INITIALIZE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the left attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the right attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is no value AND q is no value  
        RETURN True  
    ELSE IF p is no value OR q is no value  
        RETURN False  
    ELSE IF the value attribute of p is NOT EQUALS to the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left attribute of p and the left attribute of q  
        AND the result of is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive with parameters self root  
        IF root is no value  
            RETURN zero  
        END IF  
        FUNCTION dfs with parameters node parent current_length  
            IF node is no value  
                RETURN current_length  
            END IF  
            IF parent is not no value AND the value attribute of node EQUALS parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET left_length TO the result of dfs with the left attribute of node the value attribute of node and current_length  
            SET right_length TO the result of dfs with the right attribute of node the value attribute of node and current_length  
            RETURN the greatest value among current_length left_length and right_length  
        END FUNCTION  
        RETURN the result of dfs with root no value and zero  
    END FUNCTION  
END CLASS
