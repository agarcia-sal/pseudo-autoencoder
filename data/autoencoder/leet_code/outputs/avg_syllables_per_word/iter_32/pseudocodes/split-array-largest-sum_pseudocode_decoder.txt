CLASS Solution  
    FUNCTION splitArray(nums k)  
        FUNCTION canSplit(max_sum)  
            SET current_sum TO zero  
            SET subarrays TO one  
            FOR each num IN nums  
                IF current_sum PLUS num GREATER THAN max_sum  
                    INCREMENT subarrays BY one  
                    SET current_sum TO num  
                ELSE  
                    INCREMENT current_sum BY num  
                END IF  
            END FOR  
            RETURN subarrays LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO the maximum element in nums  
        SET right TO the sum of all elements in nums  
        WHILE left LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY two using integer division  
            IF canSplit(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
