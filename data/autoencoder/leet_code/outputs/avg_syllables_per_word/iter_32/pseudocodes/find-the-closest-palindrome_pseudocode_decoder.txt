CLASS Solution  
    FUNCTION nearestPalindromic WITH PARAMETER n  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER VALUE OF n MINUS one  
        END IF  
        
        SET candidates TO an EMPTY SET  
        
        ADD the INTEGER VALUE OF ten RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD the INTEGER VALUE OF ten RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER VALUE OF the SUBSTRING from position zero TO the INTEGER VALUE of the FLOOR of the DIVISION of length PLUS one BY two OF n  
        FOR i FROM negative one TO one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF the REMAINDER OF the DIVISION OF length BY two EQUALS zero  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF the SUBSTRING of new_prefix from the position of the SECOND TO LAST CHARACTER TO the FIRST CHARACTER  
            END IF  
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF IT IS IN candidates  
        
        SET closest TO the ELEMENT OF candidates WITH THE MINIMUM VALUE ACCORDING TO THE CRITERION OF THE ABSOLUTE VALUE OF the ELEMENT MINUS original_number FOLLOWED BY the ELEMENT ITSELF  
        
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
