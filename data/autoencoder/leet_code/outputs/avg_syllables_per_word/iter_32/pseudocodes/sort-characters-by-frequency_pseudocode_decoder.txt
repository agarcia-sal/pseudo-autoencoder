CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO the result of counting each character appearance in the string s  
        SET sorted_characters TO the list of characters from frequency keys sorted such that characters with higher frequency appear before characters with lower frequency  
        SET result TO an empty string  
        FOR each character IN sorted_characters  
            APPEND the character repeated the number of times equal to the frequency corresponding to the character TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
