CLASS Solution  
    FUNCTION monotoneIncreasingDigits WITH PARAMETER n  
        SET digits TO list of characters representing the number n converted to string  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one INCLUSIVE DECREMENTING BY one  
            IF the element at position i of digits is LESS THAN the element at position i MINUS one of digits THEN  
                SET the element at position i MINUS one of digits TO the string representing the integer value of the element at position i MINUS one MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one INCLUSIVE  
            SET the element at position i of digits TO the string character digit nine  
        END FOR  
        SET result_string TO concatenation of all elements in digits  
        SET result_number TO the integer value of result_string  
        RETURN result_number  
    END FUNCTION  
END CLASS
