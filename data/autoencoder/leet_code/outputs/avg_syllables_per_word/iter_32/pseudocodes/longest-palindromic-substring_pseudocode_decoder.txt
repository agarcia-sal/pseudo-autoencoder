CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional collection of TRUE values with size n by n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM the second last position of s DOWN TO zero  
            FOR j FROM the position after i TO the last position of s  
                SET the element at position i and j of f TO FALSE  
                IF the element at position i of s EQUALS the element at position j of s  
                    SET the element at position i and j of f TO the element at position i plus one and j minus one of f  
                    IF the element at position i and j of f AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k TO the position k PLUS mx minus one  
    END FUNCTION  
END CLASS
