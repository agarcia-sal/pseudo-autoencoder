CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS character plus  
                        APPEND the sum of l PLUS r TO results  
                    ELSE IF operator EQUALS character minus  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS character multiplied by  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF digit characters  
                RETURN the list containing the integer conversion of sub_expr  
            END IF  

            SET results TO empty list  
            FOR index i FROM the first position TO the last position IN sub_expr  
                IF the element at position i of sub_expr IS one of the characters plus OR minus OR multiplied by  
                    SET left_results TO the result of calling helper with substring from the first position TO the position before i of sub_expr  
                    SET right_results TO the result of calling helper with substring from the position after i TO the last position of sub_expr  
                    EXTEND results BY the result of calling compute with left_results, right_results, and the element at position i of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN the result of calling helper with expression  
    END FUNCTION  
END CLASS
