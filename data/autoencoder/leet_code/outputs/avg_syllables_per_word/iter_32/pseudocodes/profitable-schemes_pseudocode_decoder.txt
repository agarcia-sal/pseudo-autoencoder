CLASS Solution  
    FUNCTION profitableSchemes(n  
                                minProfit  
                                group  
                                profit)  
        SET MODULO TO ten POWERED BY nine PLUS one  
        INITIALIZE two dimensional list dp with dimensions n PLUS one BY minProfit PLUS one WITH all values SET TO zero  
        FOR index i FROM zero TO n  
            SET element at position zero of dp at position i TO one  
        END FOR  
        FOR each element p AND g IN parallel iteration over profit AND group  
            FOR index i FROM n DOWN TO g  
                FOR index j FROM minProfit DOWN TO zero  
                    SET element at position j of dp at position i TO the sum of the current element at position j of dp at position i AND element at position maximum of zero AND the difference between j AND p of dp at position difference between i AND g taken modulo MODULO  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position minProfit of dp at position n  
    END FUNCTION  
END CLASS
