CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position one of nums expressed as text  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the element at position one of nums expressed as text  
            FOLLOWED BY the word DIVIDED BY  
            FOLLOWED BY the element at position two of nums expressed as text  
        END IF  
        RETURN the element at position one of nums expressed as text  
        FOLLOWED BY the word DIVIDED BY  
        FOLLOWED BY an opening parenthesis  
        FOLLOWED BY the element at position two of nums expressed as text  
        FOLLOWED BY the word DIVIDED BY  
        FOLLOWED BY each element expressed as text for each element in nums beginning from position three TO the last element joined with the word DIVIDED BY as separator  
        FOLLOWED BY a closing parenthesis  
    END FUNCTION  
END CLASS
