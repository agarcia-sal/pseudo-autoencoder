CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list containing twenty six elements each set TO negative one  
        SET total appeal TO zero  
        SET current appeal TO zero  
        
        FOR each index i FROM zero TO the length of s MINUS one  
            SET character char TO element at position i of s  
            SET index TO the Unicode code point value of char MINUS the Unicode code point value of lowercase letter a  
            SET current appeal TO current appeal PLUS index i MINUS element at position index of last  
            SET total appeal TO total appeal PLUS current appeal  
            SET element at position index of last TO index i  
        END FOR  
        
        RETURN total appeal  
    END FUNCTION  
END CLASS
