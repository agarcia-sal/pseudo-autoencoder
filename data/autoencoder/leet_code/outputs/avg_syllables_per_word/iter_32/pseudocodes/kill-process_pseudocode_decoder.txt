CLASS Solution  
    FUNCTION killProcess(pid ppid kill)  
        SET tree TO a new dictionary where each key maps to a list of values  
        FOR each child AND parent IN the pairings from pid AND ppid  
            APPEND child TO the list corresponding to parent in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the first element of queue  
            REMOVE the first element from queue  
            APPEND current TO killed_processes  
            FOR each child IN the list corresponding to current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
