CLASS Solution  
    FUNCTION cutOffTree(forest)  
        IF the LENGTH OF forest EQUALS zero OR the LENGTH OF element at position zero OF forest EQUALS zero  
            RETURN the NEGATIVE of one  
        END IF  

        SET trees TO an empty list  
        FOR index i FROM zero TO the LENGTH OF forest MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position i OF forest MINUS one  
                IF element at position i of forest at position j GREATER THAN one  
                    APPEND a tuple containing element at position i of forest at position j AND i AND j TO trees  
                END IF  
            END FOR  
        END FOR  
        SORT trees IN ascending order BY the first element of each tuple  

        FUNCTION bfs(start, end)  
            IF start EQUALS end  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF forest  
            SET n TO the LENGTH OF element at position zero OF forest  
            INITIALIZE queue WITH start as the only element  
            INITIALIZE visited AS a set containing start  
            SET steps TO zero  

            WHILE queue is not empty  
                SET current_level_size TO the LENGTH OF queue  
                FOR counter FROM one TO current_level_size  
                    REMOVE the first element from queue AND ASSIGN TO x AND y  
                    FOR each pair dx AND dy IN the list of pairs negative one AND zero, positive one AND zero, zero AND negative one, zero AND positive one  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple nx AND ny NOT IN visited AND element at position nx of forest at position ny NOT EQUAL TO zero  
                            IF the tuple nx AND ny EQUALS end  
                                RETURN steps PLUS one  
                            END IF  
                            ADD the tuple nx AND ny TO visited  
                            APPEND the tuple nx AND ny TO queue  
                        END IF  
                    END FOR  
                END FOR  
                INCREMENT steps BY one  
            END WHILE  

            RETURN the NEGATIVE of one  
        END FUNCTION  

        SET x TO zero  
        SET y TO zero  
        SET total_steps TO zero  

        FOR each tuple containing height AND tx AND ty IN trees  
            SET steps TO the RESULT OF bfs WITH start AS the tuple x AND y AND end AS the tuple tx AND ty  
            IF steps EQUALS the NEGATIVE of one  
                RETURN the NEGATIVE of one  
            END IF  
            INCREMENT total_steps BY steps  
            SET x TO tx  
            SET y TO ty  
        END FOR  

        RETURN total_steps  
    END FUNCTION  
END CLASS
