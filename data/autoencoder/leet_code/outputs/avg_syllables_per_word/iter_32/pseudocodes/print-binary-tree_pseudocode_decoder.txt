CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO A new TreeNode with value equal to element at position zero of values, left equal to no value, right equal to no value  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the element REMOVED from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is NOT no value THEN  
            SET node.left TO A new TreeNode with value equal to element at position i of values, left equal to no value, right equal to no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is NOT no value THEN  
            SET node.right TO A new TreeNode with value equal to element at position i of values, left equal to no value, right equal to no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value THEN  
        RETURN true  
    ELSE IF p EQUALS no value OR q EQUALS no value THEN  
        RETURN false  
    ELSE IF the value of p.val NOT EQUALS the value of q.val THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node EQUALS no value THEN  
                RETURN zero  
            END IF  
            RETURN one PLUS the greater value BETWEEN the result of getHeight with argument node.left AND the result of getHeight with argument node.right  
        END FUNCTION  

        SET height TO the result of getHeight with argument root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        CREATE a list named res CONTAINING height NUMBER OF sublists EACH CONTAINING width NUMBER OF empty string values  

        FUNCTION placeNode(node, row, left, right)  
            IF node EQUALS no value THEN  
                RETURN  
            END IF  
            SET mid TO the integer division RESULT OF the sum of left AND right BY two  
            SET the element at position mid of the list at position row of res TO the string representation OF the value of node.val  
            CALL placeNode with arguments node.left, row PLUS one, left, mid  
            CALL placeNode with arguments node.right, row PLUS one, mid, right  
        END FUNCTION  

        CALL placeNode with arguments root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
