CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new default dictionary with default value as an empty list  
        FOR each pair of departure_airport AND arrival_airport IN tickets sorted in ascending order and then reversed  
            APPEND arrival_airport TO the list at key departure_airport in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list at key airport in graph is not empty  
                CALL visit with the element obtained by removing the last item from the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with airport equal to the string named JFK  
        RETURN itinerary in reverse order  
    END FUNCTION  
END CLASS
