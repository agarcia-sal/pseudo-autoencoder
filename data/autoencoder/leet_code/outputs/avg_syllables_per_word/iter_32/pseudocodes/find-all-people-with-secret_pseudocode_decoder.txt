CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO the set containing zero and firstPerson  
        SORT meetings by element at position three of each meeting in ascending order  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO element at position three of the meeting at position i in meetings  
            ASSIGN graph TO a new mapping with default empty list for missing keys  
            WHILE i LESS THAN the LENGTH OF meetings AND element at position three of the meeting at position i EQUALS current_time  
                SET x TO element at position one of the meeting at position i in meetings  
                SET y TO element at position two of the meeting at position i in meetings  
                APPEND y TO the list at key x in graph  
                APPEND x TO the list at key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new double-ended queue initialized with each person in know_secret that is a key in graph  
            SET visited TO an empty set  
            WHILE queue is not empty  
                REMOVE and GET the element from the left of queue into person  
                IF person is in visited THEN  
                    CONTINUE to the next iteration of the while loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor in the list at key person in graph  
                    IF neighbor is not in visited THEN  
                        APPEND neighbor TO the right end of queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list formed from know_secret  
    END FUNCTION  
END CLASS
