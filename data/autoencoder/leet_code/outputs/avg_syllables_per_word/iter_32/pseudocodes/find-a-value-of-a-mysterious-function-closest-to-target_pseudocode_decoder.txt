CLASS Solution  
    FUNCTION closestToTarget(arr parameter, target parameter)  
        SET seen TO empty collection of unique values  
        SET min_diff TO a value representing positive infinity  
        FOR each num IN arr parameter  
            SET new_seen TO empty collection of unique values  
            FOR each x IN seen  
                SET bitwise_and_result TO bitwise AND of num AND x  
                ADD bitwise_and_result TO new_seen  
            END FOR  
            ADD num TO new_seen  
            SET seen TO new_seen  
            FOR each val IN seen  
                SET current_difference TO the absolute value of the difference between val AND target parameter  
                IF current_difference LESS THAN min_diff THEN  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
