CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the set constructed from bank  
        SET q TO a double ended queue containing a pair of start and zero  
        SET mp TO a mapping where the key is the character A and the value is the sequence of characters T C G and the key is T and the value is the sequence of characters A C G and the key is C and the value is the sequence of characters A T G and the key is G and the value is the sequence of characters A T C  
        WHILE the queue q is not empty  
            SET t and step TO the result of removing the element from the front of the queue q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR index i FROM zero TO the length of t MINUS one  
                SET v TO the element at position i of the string t  
                FOR each character j IN the sequence mapped from v in mp  
                    SET next TO the concatenation of the substring of t from the start TO the position before i PLUS j PLUS the substring of t from the position after i TO the end  
                    IF next is a member of the set s  
                        APPEND the pair of next and step PLUS one TO the queue q  
                        REMOVE next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
