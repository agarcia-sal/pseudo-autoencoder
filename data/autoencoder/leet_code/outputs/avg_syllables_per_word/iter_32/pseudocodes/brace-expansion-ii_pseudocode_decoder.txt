CLASS Solution  
    FUNCTION braceExpansionII(expression)  
        FUNCTION parse_expression(index)  
            SET current_set TO an empty collection of unique items  
            SET current_product TO a list containing an empty string  
            WHILE index LESS THAN the LENGTH OF expression  
                SET char TO the element at position index of expression  
                IF char EQUALS the opening brace character  
                    SET inner_set AND index TO the result of parse_expression called with index PLUS one  
                    SET new_product TO an empty list  
                    FOR each string_a IN current_product  
                        FOR each string_b IN inner_set  
                            APPEND the concatenation of string_a PLUS string_b TO new_product  
                        END FOR  
                    END FOR  
                    SET current_product TO new_product  
                ELSE IF char EQUALS the closing brace character  
                    FOR each product_string IN current_product  
                        ADD product_string TO current_set  
                    END FOR  
                    RETURN current_set AND index  
                ELSE IF char EQUALS the comma character  
                    FOR each product_string IN current_product  
                        ADD product_string TO current_set  
                    END FOR  
                    SET current_product TO a list containing an empty string  
                ELSE  
                    SET new_product TO an empty list  
                    FOR each string_element IN current_product  
                        APPEND the concatenation of string_element PLUS char TO new_product  
                    END FOR  
                    SET current_product TO new_product  
                END IF  
                INCREMENT index BY one  
            END WHILE  
            FOR each product_string IN current_product  
                ADD product_string TO current_set  
            END FOR  
            RETURN current_set AND index  
        END FUNCTION  
        SET result_set AND ignored_index TO the result of parse_expression called with zero  
        RETURN the list of elements in result_set sorted in ascending order  
    END FUNCTION  
END CLASS
