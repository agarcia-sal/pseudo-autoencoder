CLASS Solution  
    FUNCTION findPaths(m n maxMove startRow startColumn)  
        SET MOD TO ten POWER nine PLUS one  
        SET directions TO list of pairs representing zero and one then zero and negative one then one and zero then negative one and zero  
        SET dp TO three dimensional list with dimension m by n by maxMove PLUS one filled with zero  
        FOR k FROM one TO maxMove  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    FOR each pair di dj IN directions  
                        SET ni TO i PLUS di  
                        SET nj TO j PLUS dj  
                        IF ni LESS THAN zero OR ni GREATER THAN OR EQUAL TO m OR nj LESS THAN zero OR nj GREATER THAN OR EQUAL TO n THEN  
                            SET dp element at position i j k TO dp element at position i j k PLUS one MODULO MOD  
                        ELSE  
                            SET dp element at position i j k TO dp element at position i j k PLUS dp element at position ni nj k MINUS one MODULO MOD  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position startRow startColumn maxMove  
    END FUNCTION  
END CLASS
