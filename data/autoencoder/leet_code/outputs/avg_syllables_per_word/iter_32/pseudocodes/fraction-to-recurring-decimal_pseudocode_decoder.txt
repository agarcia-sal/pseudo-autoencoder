CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER) RETURNS string  
        IF (numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero) OR (numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero)  
            SET sign TO the string representing minus  
        ELSE  
            SET sign TO the empty string  
        END IF  
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        SET quotient TO the result of numerator DIVIDED BY denominator without remainder  
        SET remainder TO the remainder after dividing numerator BY denominator  
        SET integer_part TO the string representation of quotient  
        IF remainder EQUALS zero  
            RETURN the concatenation of sign and integer_part  
        END IF  
        SET seen_remainders TO an empty dictionary to store mapping from remainder TO position  
        SET fractional_part TO an empty list to hold each fractional digit as string  
        WHILE remainder NOT EQUALS zero  
            IF remainder is a key in seen_remainders  
                SET start TO the value in seen_remainders corresponding to key remainder  
                SET non_repeating TO the concatenation of elements in fractional_part from position zero UP TO position start excluding position start  
                SET repeating TO the concatenation of elements in fractional_part from position start TO the end  
                RETURN the concatenation of sign, integer_part, the string representing dot, non_repeating, the string representing opening parenthesis, repeating, and the string representing closing parenthesis  
            END IF  
            ASSIGN to seen_remainders at key remainder the current length of fractional_part  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the result of remainder DIVIDED BY denominator without remainder  
            SET remainder TO the remainder after dividing remainder BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        RETURN the concatenation of sign, integer_part, the string representing dot, and the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
