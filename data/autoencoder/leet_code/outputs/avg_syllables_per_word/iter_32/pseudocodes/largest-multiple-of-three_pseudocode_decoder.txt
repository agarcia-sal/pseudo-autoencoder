CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of elements in digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the remainder of total_sum DIVIDED BY three  
        
        IF remainder EQUALS one  
            FOR each digit d IN the list consisting of one, four, seven  
                IF the frequency count of d IS GREATER THAN zero  
                    DECREMENT the frequency count of d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred in the previous loop  
                FOR each digit d IN the list consisting of two, five, eight  
                    IF the frequency count of d IS GREATER THAN one  
                        DECREMENT the frequency count of d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit d IN the list consisting of two, five, eight  
                IF the frequency count of d IS GREATER THAN zero  
                    DECREMENT the frequency count of d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no break occurred in the previous loop  
                FOR each digit d IN the list consisting of one, four, seven  
                    IF the frequency count of d IS GREATER THAN one  
                        DECREMENT the frequency count of d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        SET result TO an empty list  
        FOR each digit d FROM nine DOWN TO zero  
            APPEND the character representation of digit d repeated frequency count of d times TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF the length of final_number IS GREATER THAN zero AND the first character of final_number EQUALS zero  
            RETURN the character zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
