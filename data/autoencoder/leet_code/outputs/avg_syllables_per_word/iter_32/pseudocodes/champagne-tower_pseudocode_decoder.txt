CLASS Solution  
    FUNCTION champagneTower(poured query_poured query_row query_glass)  
        SET current_row TO a list containing the value of query_poured  
        FOR row FROM one TO query_row  
            SET next_row TO a list of zeros with the length of row PLUS one  
            FOR i FROM zero TO the LENGTH OF current_row MINUS one  
                IF the element at position i of current_row GREATER THAN one  
                    SET excess TO the element at position i of current_row MINUS one DIVIDED BY two  
                    INCREMENT the element at position i of next_row BY excess  
                    INCREMENT the element at position i PLUS one of next_row BY excess  
                END IF  
            END FOR  
            SET current_row TO next_row  
        END FOR  
        IF the element at position query_glass of current_row LESS THAN OR EQUAL TO one  
            RETURN the element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
