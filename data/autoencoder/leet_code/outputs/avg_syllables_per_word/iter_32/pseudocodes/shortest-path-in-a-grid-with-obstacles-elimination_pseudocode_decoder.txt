CLASS Solution  
    FUNCTION shortestPath(grid, k)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO list of elements each containing two numbers representing zero and one; first direction zero followed by one; second direction one followed by zero; third direction zero followed by negative one; fourth direction negative one followed by zero  
        
        SET queue TO a double ended queue containing one element consisting of zero, zero, k, zero  
        
        SET visited TO a set containing one tuple of zero, zero, k  
        
        WHILE queue is not empty  
            REMOVE and SET x, y, remaining_k, steps TO the first element removed from queue  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each element containing dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx of element at position ny of grid  
                    
                    IF new_k GREATER THAN OR EQUAL TO zero AND tuple of nx, ny, new_k NOT IN visited  
                        ADD tuple of nx, ny, new_k TO visited  
                        APPEND tuple of nx, ny, new_k, steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
