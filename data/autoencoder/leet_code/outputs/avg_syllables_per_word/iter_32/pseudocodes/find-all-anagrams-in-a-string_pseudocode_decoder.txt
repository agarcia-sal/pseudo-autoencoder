CLASS Solution  
    FUNCTION findAnagrams(s, p)  
        SET length_of_s TO the LENGTH OF s  
        SET length_of_p TO the LENGTH OF p  
        IF length_of_p GREATER THAN length_of_s  
            RETURN an EMPTY list  
        END IF  
        SET p_count TO a mapping of characters TO their frequencies in p  
        SET s_count TO a mapping of characters TO their frequencies in the substring of s from position zero TO position length_of_p MINUS one inclusive  
        SET anagrams TO an EMPTY list  
        IF the mapping s_count EQUALS the mapping p_count  
            APPEND zero TO anagrams  
        END IF  
        FOR index i FROM length_of_p TO length_of_s MINUS one  
            SET current_character TO element at position i of s  
            INCREMENT the frequency of current_character in s_count BY one  
            SET character_to_remove TO element at position i MINUS length_of_p of s  
            DECREMENT the frequency of character_to_remove in s_count BY one  
            IF the frequency of character_to_remove in s_count EQUALS zero  
                REMOVE character_to_remove FROM s_count  
            END IF  
            IF the mapping s_count EQUALS the mapping p_count  
                APPEND the value of i MINUS length_of_p PLUS one TO anagrams  
            END IF  
        END FOR  
        RETURN anagrams  
    END FUNCTION  
END CLASS
