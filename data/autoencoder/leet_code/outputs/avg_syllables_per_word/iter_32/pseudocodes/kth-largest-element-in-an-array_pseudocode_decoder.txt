CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO empty list  
        FOR each num IN nums  
            PUSH num INTO min_heap MAINTAINING heap property  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                REMOVE the SMALLEST element FROM min_heap  
            END IF  
        END FOR  
        RETURN element at position one of min_heap  
    END FUNCTION  
END CLASS
