CLASS Solution  
    FUNCTION countRangeSum(nums PARAMETER, lower PARAMETER, upper PARAMETER)  
        SET prefix_sums TO a list containing zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET count TO count PLUS the difference between the position where right would be inserted to maintain order in prefix_sums as the right boundary and the position where left would be inserted to maintain order in prefix_sums as the left boundary  
            
            INSERT current_sum INTO prefix_sums at the position that maintains sorted order  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
