CLASS Solution  
    FUNCTION candy(ratings)  
        SET n TO the LENGTH OF ratings  
        SET candies TO a list of the value one repeated n times  
        FOR i FROM two TO n  
            IF element at position i of ratings GREATER THAN element at position i MINUS one of ratings  
                SET element at position i of candies TO element at position i MINUS one of candies PLUS one  
            END IF  
        END FOR  
        FOR i FROM n MINUS two DOWN TO one STEPPING by negative one  
            IF element at position i of ratings GREATER THAN element at position i PLUS one of ratings  
                IF element at position i of candies LESS THAN element at position i PLUS one of candies PLUS one  
                    SET element at position i of candies TO element at position i PLUS one of candies PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the SUM OF candies  
    END FUNCTION  
END CLASS
