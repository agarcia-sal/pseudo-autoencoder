CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of each character IN s  
        SET max_heap TO list of pairs of negative frequency AND character FOR each character AND frequency IN count  
        INITIALIZE max_heap as a heap structure  
        
        SET prev_char TO no value  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN empty string  
            END IF  
            
            SET freq AND char TO removal of smallest element FROM max_heap  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                ADD pair of prev_freq AND prev_char TO max_heap as a heap element  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation of all elements IN result  
    END FUNCTION  
END CLASS
