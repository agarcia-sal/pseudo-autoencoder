CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND the concatenation of all elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF element at position i of t IS A LETTER  
                SET element at position i of t TO the character obtained by toggling the case of element at position i of t  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO the list of all characters of s  
        SET ans TO an empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
