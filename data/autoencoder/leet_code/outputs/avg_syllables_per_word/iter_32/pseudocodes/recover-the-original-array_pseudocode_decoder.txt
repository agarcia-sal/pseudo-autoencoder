CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT the list nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position one of nums  
        FOR i FROM two TO n  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS the element at position i of nums  
                CONTINUE to the next iteration  
            END IF  
            CREATE a dictionary count that COUNTS the FREQUENCY of each element in nums  
            SET arr TO an empty list  
            FOR each element x IN nums  
                IF the value at key x in count EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the value at key x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK the loop  
                END IF  
                APPEND the value x PLUS k TO arr  
                DECREMENT the value at key x in count BY one  
                DECREMENT the value at key x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
