CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS open parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO element REMOVED FROM stack PLUS IF score EQUALS zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
