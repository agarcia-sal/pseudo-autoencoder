CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF element at position i of board at position j EQUALS the character representing a battleship  
                    IF the position i EQUALS zero OR element at position i MINUS one of board at position j EQUALS the character representing an empty cell  
                        AND the position j EQUALS zero OR element at position i of board at position j MINUS one EQUALS the character representing an empty cell  
                            INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
