CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO new ListNode with val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value of p1 dot val EQUALS the value of p2 dot val AND is_same_list of p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self head)  
        IF head EQUALS none OR the element at position one of head dot next EQUALS none OR the element at position one of the element at position one of head dot next dot next EQUALS none  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head dot next  
        SET even_head TO even  
        WHILE even IS NOT none AND the element at position one of even dot next IS NOT none  
            SET odd dot next TO the element at position one of even dot next  
            SET odd TO odd dot next  
            SET even dot next TO the element at position one of odd dot next  
            SET even TO even dot next  
        END WHILE  
        SET odd dot next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
