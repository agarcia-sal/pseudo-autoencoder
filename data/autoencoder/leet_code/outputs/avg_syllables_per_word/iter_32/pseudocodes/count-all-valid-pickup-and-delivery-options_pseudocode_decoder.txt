CLASS Solution  
    FUNCTION countOrders(n)  
        SET MODULO TO ten RAISED TO THE power OF nine PLUS seven  
        SET count TO one  
        FOR i FROM two TO n INCLUSIVE  
            SET count TO count MULTIPLIED BY i MULTIPLIED BY two MULTIPLIED BY i MINUS one MODULO MODULO  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
