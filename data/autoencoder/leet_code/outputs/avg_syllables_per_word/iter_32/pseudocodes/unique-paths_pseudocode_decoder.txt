CLASS Solution  
    FUNCTION uniquePaths(m n)  
        SET dp TO a two dimensional array with m rows each containing n ones  
        FOR i FROM two TO m  
            FOR j FROM two TO n  
                SET element at position i of dp AT position j TO the element at position i MINUS one of dp AT position j PLUS the element at position i of dp AT position j MINUS one  
            END FOR  
        END FOR  
        RETURN the element at position m MINUS one of dp AT position n MINUS one  
    END FUNCTION  
END CLASS
