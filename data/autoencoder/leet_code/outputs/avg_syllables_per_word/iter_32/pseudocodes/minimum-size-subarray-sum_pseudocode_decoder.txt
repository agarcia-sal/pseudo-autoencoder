CLASS Solution  
    FUNCTION minSubArrayLen(target parameter, nums parameter)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinity value  
        
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET current_sum TO current_sum PLUS element at position right of nums  
            
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO the MINIMUM value BETWEEN min_length AND the DIFFERENCE BETWEEN right AND left PLUS one  
                SET current_sum TO current_sum MINUS element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        
        IF min_length NOT EQUALS infinity value  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
