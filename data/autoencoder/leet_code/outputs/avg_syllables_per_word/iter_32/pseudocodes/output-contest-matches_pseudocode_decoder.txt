CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO empty list  
        FOR number FROM one TO n  
            APPEND string representation of number TO teams  
        END FOR  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET teams element at position i TO string formed by an opening parenthesis FOLLOWED BY teams element at position i FOLLOWED BY a comma FOLLOWED BY teams element at position n MINUS one MINUS i FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN teams element at position zero  
    END FUNCTION  
END CLASS
