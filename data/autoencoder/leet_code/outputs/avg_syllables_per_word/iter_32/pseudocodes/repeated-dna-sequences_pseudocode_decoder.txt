CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new dictionary that returns zero for missing keys  
        FOR index i FROM zero TO the length of s MINUS ten PLUS one  
            SET sequence TO the substring of s from position i TO position i PLUS ten MINUS one  
            INCREMENT the count associated with sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key sequence AND each value count IN the items of sequence_count  
            IF count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
