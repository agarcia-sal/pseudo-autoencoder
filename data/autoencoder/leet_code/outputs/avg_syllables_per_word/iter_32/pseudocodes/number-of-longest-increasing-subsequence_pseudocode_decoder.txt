CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF nums EQUALS an empty collection THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF nums  
        SET lengths TO a collection of n elements each set TO one  
        SET counts TO a collection of n elements each set TO one  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums THEN  
                    IF element at position j of lengths PLUS one GREATER THAN element at position i of lengths THEN  
                        SET element at position i of lengths TO element at position j of lengths PLUS one  
                        SET element at position i of counts TO element at position j of counts  
                    ELSE IF element at position j of lengths PLUS one EQUALS element at position i of lengths THEN  
                        INCREMENT element at position i of counts BY element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET longest TO the MAXIMUM value in lengths  
        SET total_count TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of lengths EQUALS longest THEN  
                INCREMENT total_count BY element at position i of counts  
            END IF  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
