CLASS Solution  
    FUNCTION peakIndexInMountainArray(arr)  
        SET left TO zero  
        SET right TO the LENGTH OF arr MINUS one  
        WHILE left LESS THAN right holds  
            SET mid TO the INTEGER PART OF left PLUS right DIVIDED BY two  
            IF the element at position mid of arr LESS THAN the element at position mid PLUS one of arr  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
