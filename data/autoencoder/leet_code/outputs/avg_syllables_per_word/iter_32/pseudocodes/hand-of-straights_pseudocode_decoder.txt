CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize EQUALS zero IS NOT true  
            RETURN false  
        END IF  
        SET count TO a mapping of the frequency of each element IN hand  
        SET unique_cards TO the LIST OF unique elements IN count SORTED in ascending order  
        FOR each card IN unique_cards  
            IF the frequency of card IN count IS GREATER THAN zero  
                SET needed TO the frequency of card IN count  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the frequency of element at position card PLUS i IN count IS LESS THAN needed  
                        RETURN false  
                    END IF  
                    DECREMENT the frequency of element at position card PLUS i IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
