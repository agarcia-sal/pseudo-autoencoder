CLASS Solution  
    FUNCTION numWays(steps steps arrLen arrLen)  
        SET MOD TO the integer one billion PLUS seven  
        SET max_index TO the smaller value BETWEEN the integer division of steps BY two AND arrLen MINUS one  
        SET dp TO a list of zeros with the length max_index PLUS one  
        SET element at position zero of dp TO one  
        FOR each iteration FROM one TO steps  
            SET prev_dp TO a copy of dp  
            FOR each index i FROM zero TO max_index  
                SET element at position i of dp TO element at position i of prev_dp  
                IF i GREATER THAN zero THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position i MINUS one of prev_dp  
                    SET element at position i of dp TO element at position i of dp MODULO MOD  
                END IF  
                IF i LESS THAN max_index THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position i PLUS one of prev_dp  
                    SET element at position i of dp TO element at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
