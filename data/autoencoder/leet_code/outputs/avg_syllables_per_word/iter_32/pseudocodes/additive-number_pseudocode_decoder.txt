CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            SET expected TO the string representation of first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring of num from position start TO position start PLUS length MINUS one NOT EQUALS expected  
                RETURN False  
            END IF  
            RETURN the result of is_valid_sequence with parameters start PLUS length second and the integer value of expected  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the integer division of n BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the substring of num from position zero TO position i MINUS one  
                SET second TO the substring of num from position i TO position j MINUS one  
                IF the LENGTH OF first GREATER THAN one AND the element at position zero of first EQUALS the character zero OR the LENGTH OF second GREATER THAN one AND the element at position zero of second EQUALS the character zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the result of is_valid_sequence with parameters j first converted to integer and second converted to integer  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
