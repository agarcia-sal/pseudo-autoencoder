CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS one  
        SET dp TO a list of zeros WITH length EQUAL TO the LENGTH OF s PLUS one  
        SET the element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character asterisk THEN  
                RETURN nine  
            ELSE IF char EQUALS the character zero THEN  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character asterisk AND char2 EQUALS the character asterisk THEN  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character asterisk THEN  
                IF char2 LESS THAN OR EQUAL TO the character six THEN  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character asterisk THEN  
                IF char1 EQUALS the character one THEN  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two THEN  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value of the concatenation of char1 AND char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO the LENGTH OF s DO  
            SET dp at position i TO the sum of dp at position i PLUS dp at position i MINUS one MULTIPLIED BY ways_to_decode_single of element at position i MINUS one of s  
            SET dp at position i TO dp at position i MODULO MOD  
            
            IF i GREATER THAN one THEN  
                SET dp at position i TO the sum of dp at position i PLUS dp at position i MINUS two MULTIPLIED BY ways_to_decode_pair of elements at positions i MINUS two AND i MINUS one of s  
                SET dp at position i TO dp at position i MODULO MOD  
            END IF  
        END FOR  
        
        RETURN dp at the last position  
    END FUNCTION  
END CLASS
