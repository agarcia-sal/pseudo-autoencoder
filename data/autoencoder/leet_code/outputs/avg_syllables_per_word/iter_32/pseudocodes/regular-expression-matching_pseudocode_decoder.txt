CLASS Solution  
    FUNCTION isMatch(s parameter s string, p parameter p string) RETURNS boolean  
        SET dp TO two dimensional list of boolean values with dimensions the length of s PLUS one by the length of p PLUS one with all values set to false  
        SET element at position zero of list at position zero of dp TO true  
        FOR j FROM one TO the length of p  
            IF element at position j MINUS one of string p EQUALS the character represented by an asterisk  
                SET element at position zero of list at position j of dp TO the value of element at position zero of list at position j MINUS two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF element at position j MINUS one of string p EQUALS element at position i MINUS one of string s OR element at position j MINUS one of string p EQUALS the character represented by a dot  
                    SET element at position i of list at position j of dp TO the value of element at position i MINUS one of list at position j MINUS one of dp  
                ELSE IF element at position j MINUS one of string p EQUALS the character represented by an asterisk  
                    SET element at position i of list at position j of dp TO the value of element at position i of list at position j MINUS two of dp  
                    IF element at position j MINUS two of string p EQUALS element at position i MINUS one of string s OR element at position j MINUS two of string p EQUALS the character represented by a dot  
                        SET element at position i of list at position j of dp TO the logical OR of element at position i of list at position j of dp AND element at position i MINUS one of list at position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s of list at position the length of p of dp  
    END FUNCTION  
END CLASS
