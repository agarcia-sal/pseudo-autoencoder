CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode created with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is GREATER THAN zero DO  
        SET node TO the element obtained by removing the first element of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is NOT None THEN  
            SET node.left TO a new TreeNode created with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is NOT None THEN  
            SET node.right TO a new TreeNode created with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is None AND q is None THEN  
        RETURN True  
    ELSE IF p is None OR q is None THEN  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left child of p and the left child of q AND calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root is None THEN  
            RETURN an empty list  
        END IF  
        SET column_table TO a new default dictionary with default value as empty list  
        SET queue TO a new double ended queue containing a tuple with root and zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE the LENGTH OF queue is GREATER THAN zero DO  
            SET node and column_index TO the elements obtained by removing the first element of queue  
            IF node is NOT None THEN  
                APPEND the value of node TO the list in column_table at key column_index  
                IF column_index LESS THAN min_column THEN  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column THEN  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple with the left child of node and column_index MINUS one TO queue  
                APPEND a tuple with the right child of node and column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO an empty list  
        FOR x FROM min_column TO max_column DO  
            APPEND the list in column_table at key x TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
