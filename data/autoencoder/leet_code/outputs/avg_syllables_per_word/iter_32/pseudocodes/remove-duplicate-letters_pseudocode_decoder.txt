CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO empty dictionary  
        FOR each index AND character IN s  
            SET last_occurrence AT character TO index  
        END FOR  

        SET stack TO empty list  
        SET in_stack TO empty set  

        FOR each index AND character IN s  
            IF character IS IN in_stack  
                CONTINUE TO next iteration  
            END IF  

            WHILE the stack IS NOT empty AND character IS LESS THAN the element at the last position of the stack AND index IS LESS THAN the last_occurrence AT the element at the last position of the stack  
                SET removed_char TO the element at the last position of the stack  
                REMOVE the element at the last position FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  

            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  

        SET result TO concatenation of elements in stack forming a string  
        RETURN result  
    END FUNCTION  
END CLASS
