CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET length of s1 TO length_s1  
        SET length of s2 TO length_s2  
        
        IF length_s1 GREATER THAN length_s2  
            RETURN False  
        END IF  
        
        SET count_s1 TO the frequency count of each character in s1  
        SET count_s2 TO the frequency count of each character in the substring from position zero TO position length_s1 MINUS one of s2  
        
        IF count_s1 EQUALS count_s2  
            RETURN True  
        END IF  
        
        FOR index FROM length_s1 TO length_s2 MINUS one  
            INCREMENT the frequency count of element at position index of s2 IN count_s2 BY one  
            DECREMENT the frequency count of element at position index MINUS length_s1 of s2 IN count_s2 BY one  
            
            IF the frequency count of element at position index MINUS length_s1 of s2 IN count_s2 EQUALS zero  
                REMOVE the element at position index MINUS length_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
