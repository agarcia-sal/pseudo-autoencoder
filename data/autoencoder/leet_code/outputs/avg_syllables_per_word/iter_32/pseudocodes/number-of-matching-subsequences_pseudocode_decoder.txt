CLASS Solution  
    FUNCTION numMatchingSubseq(s parameter s AS string, words parameter words AS list of strings) RETURNS integer  
        SET waiting TO empty dictionary  
        FOR each character c IN the set of characters in s  
            SET the value for key c IN waiting TO an empty deque  
        END FOR  
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element from it  
            IF first_letter is one of the keys in waiting THEN  
                APPEND it TO the deque at key first_letter in waiting  
            END IF  
        END FOR  
        SET match_count TO zero  
        FOR each character c IN s  
            IF c is one of the keys in waiting THEN  
                SET current_iterators TO the deque at key c in waiting  
                SET the value for key c IN waiting TO an empty deque  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next element from it or none if exhausted  
                    IF next_char is none THEN  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is one of the keys in waiting THEN  
                            APPEND it TO the deque at key next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN match_count  
    END FUNCTION  
END CLASS
