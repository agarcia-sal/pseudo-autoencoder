CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 s2  
        IF s1 EQUALS s2  
            RETURN True  
        END IF  
        
        IF the sorted characters of s1 NOT EQUALS the sorted characters of s2  
            RETURN False  
        END IF  
        
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF calling isScramble WITH the substring from position zero TO position i MINUS one of s1 AND the substring from position zero TO position i MINUS one of s2 AND calling isScramble WITH the substring from position i TO the end of s1 AND the substring from position i TO the end of s2  
                RETURN True  
            END IF  
            
            IF calling isScramble WITH the substring from position zero TO position i MINUS one of s1 AND the substring from position n MINUS i TO the end of s2 AND calling isScramble WITH the substring from position i TO the end of s1 AND the substring from position zero TO position n MINUS i MINUS one of s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
