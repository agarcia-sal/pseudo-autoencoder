CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze destination  
        SET directions TO list of pairs representing right, down, left, and up movements  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs WITH PARAMETERS x y  
            IF the pair composed of x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF list composed of x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD the pair composed of x and y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND the element at position nx PLUS dx of maze AND the element at position ny PLUS dy EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs WITH PARAMETERS nx ny RETURNS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs WITH PARAMETERS the element at position zero of start AND the element at position one of start  
    END FUNCTION  
END CLASS
