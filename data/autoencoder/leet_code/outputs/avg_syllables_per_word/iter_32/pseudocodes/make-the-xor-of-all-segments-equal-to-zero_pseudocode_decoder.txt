CLASS Solution  
    FUNCTION minChanges(nums AS list OF integer, k AS integer) RETURNS integer  
        SET groups TO a default dictionary with list as default value  
        FOR i FROM zero TO the length of nums MINUS one  
            APPEND element at position i of nums TO the list at key of remainder of i DIVIDED BY k in groups  
        END FOR  
        
        SET freq TO a list of default dictionaries with integer default values of length k  
        SET min_size TO a list of infinite values of length k  
        FOR i FROM zero TO k MINUS one  
            FOR each number IN element at position i of groups  
                INCREMENT the count of number in freq at position i BY one  
            END FOR  
            SET element at position i of min_size TO the length of element at position i of groups  
        END FOR  
        
        SET dp TO a list of infinite values of length two RAISED TO THE POWER OF ten  
        SET element at position zero of dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO a list of infinite values of length two RAISED TO_THE_POWER_OF ten  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO the length of dp MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each number AND count IN items of freq at position i  
                    SET element at position j of new_dp TO the minimum of element at position j of new_dp AND the sum of element at position (j XOR number) of dp PLUS element at position i of min_size MINUS count  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
