CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        CREATE a two dimensional table named is_palindrome with dimensions n BY n SET all elements TO False  
        
        FOR i FROM zero TO n MINUS one  
            SET the element at row i column i of is_palindrome TO True  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET the element at row start column end of is_palindrome TO the comparison of the character at position start of s EQUALS the character at position end of s  
                ELSE  
                    SET the element at row start column end of is_palindrome TO the logical AND of the character at position start of s EQUALS the character at position end of s AND the element at row start PLUS one column end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        CREATE a list named min_cuts with length n SET all elements TO zero  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at row zero column i of is_palindrome  
                SET the element at position i of min_cuts TO zero  
            ELSE  
                SET the element at position i of min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF the element at row j column i of is_palindrome AND the element at position j MINUS one of min_cuts PLUS one LESS THAN the element at position i of min_cuts  
                        SET the element at position i of min_cuts TO the element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN the element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
