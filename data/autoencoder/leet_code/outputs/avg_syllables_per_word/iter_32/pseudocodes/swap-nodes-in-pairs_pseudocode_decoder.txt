CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last element  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list called with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs with parameter head  
        SET dummy TO a new ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        
        WHILE the next attribute of current NOT EQUALS nothing AND the next attribute of the next attribute of current NOT EQUALS nothing holds  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
