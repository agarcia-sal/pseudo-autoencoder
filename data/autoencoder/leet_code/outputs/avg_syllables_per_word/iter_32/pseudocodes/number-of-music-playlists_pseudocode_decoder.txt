CLASS Solution  
    FUNCTION numMusicPlaylists(n n INTEGER, goal goal INTEGER, k k INTEGER) RETURNS INTEGER  
        SET MOD TO ten TO THE POWER OF nine PLUS one  
        CREATE dp AS list OF lists OF zero values WITH size goal PLUS one TIMES n PLUS one  
        SET element at position zero of dp TO list WITH zero VALUES OF size n PLUS one  
        SET element at position zero of element at position zero of dp TO one  
        
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position j of element at position i of dp TO element at position j of element at position i of dp PLUS element at position j MINUS one of element at position i MINUS one of dp MULTIPLIED BY n MINUS j MINUS one  
                SET element at position j of element at position i of dp TO element at position j of element at position i of dp MODULO MOD  
                
                IF j GREATER THAN k THEN  
                    SET element at position j of element at position i of dp TO element at position j of element at position i of dp PLUS element at position j of element at position i MINUS one of dp MULTIPLIED BY j MINUS k  
                    SET element at position j of element at position i of dp TO element at position j of element at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position goal of dp  
    END FUNCTION  
END CLASS
