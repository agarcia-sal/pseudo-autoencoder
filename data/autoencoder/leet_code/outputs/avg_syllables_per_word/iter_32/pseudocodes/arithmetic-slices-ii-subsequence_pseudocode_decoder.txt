CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO list of size n each element being a new mapping from difference to count of arithmetic subsequences ending at that index with that difference  
        SET total_count TO zero  
        
        FOR i FROM zero TO the value of n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO the value of the element at position i of nums MINUS the value of the element at position j of nums  
                IF the mapping at position j in dp CONTAINS key d  
                    INCREMENT the value at key d in the mapping at position i in dp BY the value at key d in the mapping at position j in dp PLUS one  
                    INCREMENT total_count BY the value at key d in the mapping at position j in dp  
                ELSE  
                    INCREMENT the value at key d in the mapping at position i in dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
