CLASS ListNode  
    FUNCTION __init__ (val EQUALS zero NEXT EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node (values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two to last  
        SET node TO new ListNode with value val  
        SET next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list (p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN self val attribute of p1 EQUALS self val attribute of p2 AND call to is_same_list with arguments self next attribute of p1 and self next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers (l1 l2)  
        SET dummy_head TO new ListNode with value zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 NOT EQUALS None OR l2 NOT EQUALS None OR carry NOT EQUALS zero  
            IF l1 NOT EQUALS None  
                SET val1 TO self val attribute of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 NOT EQUALS None  
                SET val2 TO self val attribute of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET next attribute of current TO new ListNode with value total MODULO ten  
            SET current TO next attribute of current  
            
            IF l1 NOT EQUALS None  
                SET l1 TO next attribute of l1  
            END IF  
            IF l2 NOT EQUALS None  
                SET l2 TO next attribute of l2  
            END IF  
        END WHILE  
        
        RETURN next attribute of dummy_head  
    END FUNCTION  
END CLASS
