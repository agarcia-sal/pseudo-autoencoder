CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SORT the list nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list containing one repeated n times  
        
        FOR i FROM one TO n MINUS one  
            SET element at position i of power TO the element at position i MINUS one of power MULTIPLIED BY two THEN MODULO MOD  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS element at position i of nums MULTIPLIED BY the element at position i of power MINUS the element at position n MINUS i MINUS one of power  
            THEN MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
