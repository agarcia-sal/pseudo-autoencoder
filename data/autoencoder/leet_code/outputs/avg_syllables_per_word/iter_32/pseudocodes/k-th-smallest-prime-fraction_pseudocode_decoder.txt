CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr parameter OF TYPE List OF INTEGER, k parameter OF TYPE INTEGER) RETURNS List OF INTEGER  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            COMPUTE fraction AS element at position i of arr DIVIDED BY element at position n MINUS one of arr  
            PUSH a tuple of fraction AND i AND n MINUS one TO min_heap USING a structure that keeps the smallest fraction on top  
        END FOR  
        FOR loop_index FROM zero TO k MINUS one MINUS one  
            POP the smallest fraction tuple FROM min_heap AND UNPACK AS discarded_value AND i AND j  
            IF j MINUS one GREATER THAN i THEN  
                COMPUTE new_fraction AS element at position i of arr DIVIDED BY element at position j MINUS one of arr  
                PUSH a tuple of new_fraction AND i AND j MINUS one TO min_heap  
            END IF  
        END FOR  
        POP the smallest fraction tuple FROM min_heap AND UNPACK AS discarded_value AND i AND j  
        RETURN a list containing element at position i of arr AND element at position j of arr  
    END FUNCTION  
END CLASS
