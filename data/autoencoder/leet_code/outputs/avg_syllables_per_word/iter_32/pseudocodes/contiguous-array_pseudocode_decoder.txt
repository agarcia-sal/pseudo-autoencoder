CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with key zero mapped to value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND each element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS AS a key IN prefix_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the element at position count OF prefix_sum_index  
            ELSE  
                SET the element at position count OF prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
