CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last position of values  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN False  
    END IF  
    RETURN the val property of p1 EQUALS the val property of p2 AND CALL is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self head x)  
        SET less_head TO A NEW ListNode INITIALIZED WITH zero  
        SET greater_head TO A NEW ListNode INITIALIZED WITH zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current DOES NOT EQUAL none DO  
            IF the val property of current LESS THAN x THEN  
                SET the next property of less TO current  
                SET less TO the next property of less  
            ELSE  
                SET the next property of greater TO current  
                SET greater TO the next property of greater  
            END IF  
            SET current TO the next property of current  
        END WHILE  
        SET the next property of less TO the next property of greater_head  
        SET the next property of greater TO none  
        RETURN the next property of less_head  
    END FUNCTION  
END CLASS
