CLASS Solution  
    FUNCTION intToRoman(num)  
        SET list_of_values TO the list of one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one  
        SET list_of_symbols TO the list of the Roman numeral M PLUS the Roman numeral CM PLUS the Roman numeral D PLUS the Roman numeral CD PLUS the Roman numeral C PLUS the Roman numeral XC PLUS the Roman numeral L PLUS the Roman numeral XL PLUS the Roman numeral X PLUS the Roman numeral IX PLUS the Roman numeral V PLUS the Roman numeral IV PLUS the Roman numeral I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR each time FROM zero TO the number of times the integer num DIVIDED BY the element at position i of list_of_values MINUS one  
                SET roman_numeral TO roman_numeral PLUS the element at position i of list_of_symbols  
                DECREMENT num BY the element at position i of list_of_values  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
