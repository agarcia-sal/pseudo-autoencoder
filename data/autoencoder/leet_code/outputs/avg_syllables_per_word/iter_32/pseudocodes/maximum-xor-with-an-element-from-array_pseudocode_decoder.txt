CLASS Solution  
    FUNCTION maximizeXor(nums TIMES list OF integers, queries TIMES list OF list OF integers) RETURNS list OF integers  
        SORT nums IN ASCENDING ORDER  
        SET sorted_queries TO THE LIST OF pairsthat ARE THE ENUMERATED INDICES OF queries SORTED BY the second element of each query  
        SET result TO A LIST OF THE LENGTH OF queries FILLED WITH negative one  
        SET trie TO AN EMPTY COLLECTION REPRESENTING THE TRIE STRUCTURE  
        SET num_index TO zero  
        
        FOR each pair query_index AND pair xi AND mi IN sorted_queries  
            WHILE num_index LESS THAN THE LENGTH OF nums AND element at position num_index of nums LESS THAN OR EQUAL TO mi  
                SET num TO element at position num_index of nums  
                SET node TO trie  
                FOR i FROM thirty one DOWN TO zero  
                    SET bit TO THE bit AT POSITION i IN num AS extracted BY SHIFTING num RIGHT BY i PLACES AND GETTING THE LAST BIT  
                    IF bit NOT IN node  
                        ASSIGN reference OF AN EMPTY COLLECTION TO element keyed by bit IN node  
                    SET node TO element keyed by bit IN node  
                END FOR  
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS EMPTY  
                CONTINUE TO NEXT query IN sorted_queries  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            FOR i FROM thirty one DOWN TO zero  
                SET bit TO THE bit AT POSITION i IN xi AS extracted BY SHIFTING xi RIGHT BY i PLACES AND GETTING THE LAST BIT  
                SET toggle_bit TO one MINUS bit  
                IF toggle_bit IN node  
                    SET max_xor TO max_xor OR THE VALUE WITH A ONE BIT AT POSITION i COMPUTED BY SHIFTING one LEFT BY i PLACES  
                    SET node TO element keyed by toggle_bit IN node  
                ELSE  
                    SET node TO element keyed by bit IN node  
                END IF  
            END FOR  
            
            SET element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
