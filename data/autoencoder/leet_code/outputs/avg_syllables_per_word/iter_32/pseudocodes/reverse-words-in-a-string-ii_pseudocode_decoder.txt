CLASS Solution  
    FUNCTION reverseWords(s)  
        REVERSE the entire list s  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF element at position end of s EQUALS a space character  
                REVERSE the sublist of s from position start TO position end MINUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        REVERSE the sublist of s from position start TO the end of s  
    END FUNCTION  
END CLASS
