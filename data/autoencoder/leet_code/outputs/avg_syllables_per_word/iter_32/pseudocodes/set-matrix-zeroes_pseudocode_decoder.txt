CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        
        SET first_row_has_zero TO false  
        FOR each index j FROM zero TO n MINUS one  
            IF element at position j of element at position zero of matrix EQUALS zero  
                SET first_row_has_zero TO true  
                BREAK the FOR loop  
            END IF  
        END FOR  
        
        SET first_col_has_zero TO false  
        FOR each index i FROM zero TO m MINUS one  
            IF element at position zero of element at position i of matrix EQUALS zero  
                SET first_col_has_zero TO true  
                BREAK the FOR loop  
            END IF  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            FOR each index j FROM one TO n MINUS one  
                IF element at position j of element at position i of matrix EQUALS zero  
                    SET element at position zero of element at position i of matrix TO zero  
                    SET element at position j of element at position zero of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            FOR each index j FROM one TO n MINUS one  
                IF element at position zero of element at position i of matrix EQUALS zero OR element at position j of element at position zero of matrix EQUALS zero  
                    SET element at position j of element at position i of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        
        IF first_row_has_zero  
            FOR each index j FROM zero TO n MINUS one  
                SET element at position j of element at position zero of matrix TO zero  
            END FOR  
        END IF  
        
        IF first_col_has_zero  
            FOR each index i FROM zero TO m MINUS one  
                SET element at position zero of element at position i of matrix TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
