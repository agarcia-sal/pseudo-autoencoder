CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF the sum of the elements of gas IS LESS THAN the sum of the elements of cost  
            RETURN minus one  
        END IF  
        SET start TO zero  
        SET total_tank TO zero  
        FOR index i FROM zero TO the length of gas MINUS one  
            INCREMENT total_tank BY the element at position i of gas MINUS the element at position i of cost  
            IF total_tank IS LESS THAN zero  
                SET start TO index i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        RETURN start  
    END FUNCTION  
END CLASS
