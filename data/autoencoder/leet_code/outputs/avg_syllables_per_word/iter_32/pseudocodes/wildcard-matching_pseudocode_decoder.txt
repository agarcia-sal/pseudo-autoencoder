CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a list whose LENGTH is the LENGTH OF s PLUS one where each element is a list whose LENGTH is the LENGTH OF p PLUS one filled with the Boolean value false  
        SET the element at position zero of dp to a list where the element at position zero is true and others are false  
        
        FOR j FROM one TO the LENGTH OF p  
            IF the element at position j MINUS one of p EQUALS the character asterisk  
                SET the element at position zero of the list at position j of dp TO the value of the element at position zero of the list at position j MINUS one of dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF the element at position j MINUS one of p EQUALS the character asterisk  
                    SET the element at position i of the list at position j of dp TO the value of the element at position i MINUS one of the list at position j of dp OR the value of the element at position i of the list at position j MINUS one of dp  
                ELSE IF the element at position j MINUS one of p EQUALS the character question mark OR the element at position i MINUS one of s EQUALS the element at position j MINUS one of p  
                    SET the element at position i of the list at position j of dp TO the value of the element at position i MINUS one of the list at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the value of the element at position LENGTH OF s of the list at position LENGTH OF p of dp  
    END FUNCTION  
END CLASS
