CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict wordList STRING list word1 STRING word2 STRING) RETURNS INTEGER  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        
        FOR each index i AND corresponding element word IN wordsDict  
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET distance_difference TO prev_word1 MINUS prev_word2  
                IF distance_difference LESS THAN zero  
                    SET distance_difference TO prev_word2 MINUS prev_word1  
                END IF  
                IF distance_difference LESS THAN min_distance  
                    SET min_distance TO distance_difference  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
