CLASS Solution  
    FUNCTION searchMatrix(matrix target)  
        IF the matrix EQUALS a value that is empty OR the element at position zero of the matrix EQUALS a value that is empty THEN  
            RETURN false  
        END IF  
        SET row TO zero  
        SET col TO the LENGTH OF the element at position zero of the matrix MINUS one  
        WHILE row LESS THAN the LENGTH OF the matrix AND col GREATER THAN OR EQUAL TO zero  
            IF the element at position row of the matrix AT position col EQUALS the target THEN  
                RETURN true  
            ELSE IF the element at position row of the matrix AT position col GREATER THAN the target THEN  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
