CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF the pair of i and j EXISTS IN memo  
                RETURN the value associated with the pair of i and j IN memo  
            END IF  
            SET result TO the value of dp WITH i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF element at position k of s EQUALS element at position j of s  
                    SET result TO the minimum BETWEEN result AND the value of dp WITH i AND k MINUS one AND memo PLUS dp WITH k AND j MINUS one AND memo  
                END IF  
            END FOR  
            SET the value associated with the pair of i and j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN the value of dp WITH zero AND the LENGTH OF s MINUS one AND an empty collection  
    END FUNCTION  
END CLASS
