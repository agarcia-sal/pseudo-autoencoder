CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            IF the TUPLE of index AND prev_len IS IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the TUPLE of index AND prev_len IN memo  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index DO  
                IF the element at position index OF num EQUALS the character zero THEN  
                    BREAK the LOOP  
                END IF  
                
                SET current_num TO the SUBSTRING of num FROM position index TO position index PLUS length MINUS one  
                
                IF prev_len EQUALS zero OR current_num GREATER THAN OR EQUAL TO the SUBSTRING of num FROM position index MINUS prev_len TO position index MINUS one THEN  
                    SET count TO count PLUS dp(index PLUS length, length) MODULO MOD  
                END IF  
            END FOR  
            
            ASSIGN the VALUE of count TO memo AT the TUPLE of index AND prev_len  
            RETURN count  
        END FUNCTION  
        
        RETURN dp(zero, zero)  
    END FUNCTION  
END CLASS
