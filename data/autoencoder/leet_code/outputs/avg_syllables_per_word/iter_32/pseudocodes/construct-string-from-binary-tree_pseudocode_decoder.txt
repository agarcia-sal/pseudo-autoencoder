CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero left SET TO none right SET TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF THE LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    ASSIGN root TO NEW TreeNode USING THE ELEMENT at position zero of values  
    SET i TO one  
    ASSIGN queue TO EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        ASSIGN node TO THE FIRST ELEMENT REMOVED FROM THE LEFT OF queue  
        IF i LESS THAN THE LENGTH OF values AND THE ELEMENT at position i of values IS NOT none  
            ASSIGN node.left TO NEW TreeNode USING THE ELEMENT at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND THE ELEMENT at position i of values IS NOT none  
            ASSIGN node.right TO NEW TreeNode USING THE ELEMENT at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF THE val PROPERTY OF p NOT EQUALS THE val PROPERTY OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree USING THE left PROPERTY OF p AND THE left PROPERTY OF q AND is_same_tree USING THE right PROPERTY OF p AND THE right PROPERTY OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str WITH PARAMETERS self root  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN EMPTY STRING  
            END IF  
            SET result TO THE STRING FORM OF THE val PROPERTY OF node  
            IF THE left PROPERTY OF node IS NOT none OR THE left PROPERTY OF node IS none AND THE right PROPERTY OF node IS NOT none  
                SET result TO THE CONCATENATION OF result AND AN OPEN PARENTHESIS AND helper USING THE left PROPERTY OF node AND A CLOSE PARENTHESIS  
            END IF  
            IF THE right PROPERTY OF node IS NOT none  
                SET result TO THE CONCATENATION OF result AND AN OPEN PARENTHESIS AND helper USING THE right PROPERTY OF node AND A CLOSE PARENTHESIS  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper USING root  
    END FUNCTION  
END CLASS
