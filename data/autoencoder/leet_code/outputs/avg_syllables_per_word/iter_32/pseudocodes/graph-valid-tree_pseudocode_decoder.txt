CLASS Solution  
    FUNCTION validTree(n  
                       edges)  
        IF the LENGTH OF edges NOT EQUALS the VALUE OF n MINUS one  
            RETURN false  
        END IF  
        SET adj_list TO an empty dictionary mapping each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each pair a AND b IN edges  
            APPEND b TO the list at the key a IN adj_list  
            APPEND a TO the list at the key b IN adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at the key node IN adj_list  
                CALL dfs WITH neighbor AS parameter  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero AS parameter  
        IF the LENGTH OF visited EQUALS the VALUE OF n  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
