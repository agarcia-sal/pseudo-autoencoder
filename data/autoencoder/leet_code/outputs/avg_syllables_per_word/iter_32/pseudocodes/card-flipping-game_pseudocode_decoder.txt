CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty set  
        FOR each pair of x and y taken from the corresponding elements of fronts and backs  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO a set containing all elements from fronts PLUS all elements from backs  
        SET good_numbers TO the set difference of all_numbers MINUS same  
        IF the set good_numbers is not empty  
            SET result TO the smallest element IN good_numbers  
        ELSE  
            SET result TO zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
