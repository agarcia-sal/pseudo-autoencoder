CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS the MINIMUM OF the sum of remaining MODULO two AND dp of remaining DIVIDED BY two AND the sum of remaining MODULO three AND dp of remaining DIVIDED BY three  
        END FUNCTION  
        RETURN dp of n  
    END FUNCTION  
END CLASS
