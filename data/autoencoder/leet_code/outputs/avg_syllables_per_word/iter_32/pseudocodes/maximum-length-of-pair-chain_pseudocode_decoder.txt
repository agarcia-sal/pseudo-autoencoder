CLASS Solution  
    FUNCTION findLongestChain WITH parameter pairs  
        SORT pairs BY the second element of each pair in ascending order  
        SET current_end TO the smallest possible value  
        SET chain_length TO zero  
        FOR each pair consisting of start and end IN pairs  
            IF start GREATER THAN current_end THEN  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
