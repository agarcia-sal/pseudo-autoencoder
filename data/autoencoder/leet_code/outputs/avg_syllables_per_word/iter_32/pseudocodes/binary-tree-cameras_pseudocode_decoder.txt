CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position one of values as the value and no left child and no right child  
    SET i TO one plus one  
    SET queue TO an empty queue  
    ADD root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode with the element at position i of values as value and no left child and no right child  
            ADD node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode with the element at position i of values as value and no left child and no right child  
            ADD node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN True  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left child of p AND the left child of q AND the result of is_same_tree with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node IS no value THEN  
                RETURN zero no camera count, zero monitored count, and an infinitely large camera count  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO the result of dfs with the left child of node  
            SET right_no_camera, right_monitored, right_camera TO the result of dfs with the right child of node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the minimum of left_camera PLUS the minimum of right_monitored AND right_camera AND right_camera PLUS the minimum of left_monitored AND left_camera AND left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum among left_no_camera, left_monitored, AND left_camera PLUS the minimum among right_no_camera, right_monitored, AND right_camera  
            RETURN the tuple of no_camera, monitored, AND camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO the result of dfs with root  
        RETURN the minimum between root_monitored AND root_camera  
    END FUNCTION  
END CLASS
