CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO list of pairs of element at position of efficiency and element at position of speed for each corresponding position  
        SORT engineers in order descending by first element of pair  
        
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        
        FOR each pair eff spd IN engineers  
            CALL push spd INTO speed_heap  
            INCREMENT speed_sum BY spd  
            
            IF LENGTH OF speed_heap GREATER THAN k  
                SET smallest_speed TO CALL pop smallest element FROM speed_heap  
                DECREMENT speed_sum BY smallest_speed  
            END IF  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        
        RETURN remainder of max_performance DIVIDED BY the sum of one billion AND seven  
    END FUNCTION  
END CLASS
