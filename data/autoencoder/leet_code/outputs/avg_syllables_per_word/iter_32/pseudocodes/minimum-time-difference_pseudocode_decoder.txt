CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the number obtained by converting the substring of time from position one TO the position before the character separating hours and minutes  
            SET minutes TO the number obtained by converting the substring of time from the position after the character separating hours and minutes TO the end of time  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the list obtained by applying to_minutes to each element IN timePoints  
        
        SORT minutes_list IN ascending order  
        
        SET min_diff TO a very large value representing infinity  
        FOR i FROM two TO the LENGTH OF minutes_list  
            SET current_diff TO the element at position i of minutes_list MINUS the element at position i MINUS one of minutes_list  
            IF current_diff LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  
        
        SET circular_diff TO one thousand four hundred forty MINUS the last element of minutes_list PLUS the first element of minutes_list  
        IF circular_diff LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
