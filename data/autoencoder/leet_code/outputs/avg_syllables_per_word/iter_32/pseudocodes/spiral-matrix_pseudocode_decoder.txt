CLASS Solution  
    FUNCTION spiralOrder(matrix)  
        IF matrix EQUALS empty OR the element at position zero of matrix EQUALS empty THEN  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET top TO zero  
        SET bottom TO the LENGTH OF matrix MINUS one  
        SET left TO zero  
        SET right TO the LENGTH OF element at position zero of matrix MINUS one  
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right DO  
            FOR i FROM left TO right DO  
                APPEND the element at position top of matrix AND element at position i TO result  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom DO  
                APPEND the element at position i of matrix AND element at position right TO result  
            END FOR  
            DECREMENT right BY one  
            IF top LESS THAN OR EQUAL TO bottom THEN  
                FOR i FROM right TO left DECREMENT BY one DO  
                    APPEND the element at position bottom of matrix AND element at position i TO result  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left LESS THAN OR EQUAL TO right THEN  
                FOR i FROM bottom TO top DECREMENT BY one DO  
                    APPEND the element at position i of matrix AND element at position left TO result  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
