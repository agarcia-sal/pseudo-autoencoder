CLASS Solution  
    FUNCTION secondMinimum(n time edges change)  
        SET graph TO an empty list of empty lists with size n PLUS one  
        FOR each pair u v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        SET queue TO a new double ended queue containing a tuple of integer one zero zero representing current vertex current time and count of shortest or second shortest paths  
        SET minTimes TO a list of lists each containing two elements set to positive infinity with size n PLUS one  
        SET element at position one of minTimes at position zero TO zero  
        
        WHILE the length of queue is greater than zero  
            SET current AND currentTime AND count TO the first tuple removed from the left side of queue  
            
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
            
            IF the quotient of currentTime DIVIDED BY change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the product of the sum of the quotient of currentTime DIVIDED BY change PLUS one AND change PLUS time  
            END IF  
            
            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor of minTimes at position zero GREATER THAN nextTime  
                    SET element at position neighbor of minTimes at position one TO element at position neighbor of minTimes at position zero  
                    SET element at position neighbor of minTimes at position zero TO nextTime  
                    APPEND a tuple of neighbor nextTime zero TO queue  
                ELSE IF element at position neighbor of minTimes at position zero LESS THAN nextTime AND nextTime LESS THAN element at position neighbor of minTimes at position one  
                    SET element at position neighbor of minTimes at position one TO nextTime  
                    APPEND a tuple of neighbor nextTime one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
