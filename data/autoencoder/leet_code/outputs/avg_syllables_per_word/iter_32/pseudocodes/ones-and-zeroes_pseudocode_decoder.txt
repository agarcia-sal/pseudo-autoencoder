CLASS Solution  
    FUNCTION findMaxForm(strs parameter m parameter n) RETURNS integer  
        SET dp TO a two dimensional list with the number of rows equal to m PLUS one and number of columns equal to n PLUS one filled with zeros  
        FOR each s IN strs  
            SET count_0 TO the number of occurrences of the character zero in the string s  
            SET count_1 TO the number of occurrences of the character one in the string s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i at position j TO the greater value between dp element at position i at position j AND dp element at position i MINUS count_0 at position j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m at position n  
    END FUNCTION  
END CLASS
