CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                RETURN one IF i LESS THAN OR EQUAL TO n OTHERWISE zero  
            END IF  
            IF i EQUALS k MINUS one  
                SET value TO the smaller value BETWEEN n MINUS k PLUS one AND maxPts  
                RETURN value DIVIDED BY maxPts  
            END IF  
            SET first_term TO dfs CALL with argument i PLUS one  
            SET second_term TO dfs CALL with argument i PLUS one MINUS dfs CALL with argument i PLUS maxPts PLUS one  
            RETURN first_term PLUS second_term DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs CALL with argument zero  
    END FUNCTION  
END CLASS
