CLASS Solution  
    FUNCTION minmaxGasDist(stations parameter of type List of integers k parameter of type integer) RETURNS floating point number  
        FUNCTION possible(max_dist parameter) RETURNS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the integer part of the division of the difference of element at position index of stations MINUS element at position index MINUS one of stations BY max_dist  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO ten raised to the power of eight  
        WHILE the difference of right MINUS left GREATER THAN one multiplied by ten raised to the power of negative six HOLDS  
            SET mid TO the sum of left PLUS right DIVIDED BY two  
            IF possible(mid) THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
