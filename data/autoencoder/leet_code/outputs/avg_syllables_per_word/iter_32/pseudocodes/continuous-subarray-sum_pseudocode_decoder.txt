CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a mapping of zero associated with negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum IS A KEY IN remainder_map  
                IF index i MINUS the value associated with current_sum in remainder_map IS GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the value of index i TO be associated with current_sum IN remainder_map  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
