CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n parameter INTEGER, mines parameter LIST OF LIST OF INTEGER) RETURNS INTEGER  
        SET mines_set TO an EMPTY SET  
        FOR each mine_element IN mines  
            ADD the tuple of mine_element TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional LIST with n elements each containing n elements each of which is a LIST of FOUR zeroes  
        
        SET directions TO a LIST containing FOUR pairs representing zero and one in positions for right down left and up respectively  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair i AND j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at position zero of element at position j of element at position i of dp TO the element at position zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at position one of element at position j of element at position i of dp TO the element at position one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair i AND j NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at position two of element at position j of element at position i of dp TO the element at position two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at position three of element at position j of element at position i of dp TO the element at position three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the MINIMUM of the elements in element at position j of element at position i of dp  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
