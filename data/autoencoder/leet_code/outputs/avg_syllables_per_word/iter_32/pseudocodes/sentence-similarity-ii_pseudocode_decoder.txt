CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1 sentence1 AS List OF string sentence2 sentence2 AS List OF string similarPairs similarPairs AS List OF List OF string) RETURNS boolean  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN false  
        END IF  

        SET parent TO empty dictionary  

        FUNCTION find(x x AS string) RETURNS string  
            IF the element at position x of parent NOT EQUALS x  
                SET the element at position x of parent TO the RESULT OF calling find WITH the element at position x of parent  
            END IF  
            RETURN the element at position x of parent  
        END FUNCTION  

        FUNCTION union(x x AS string y y AS string)  
            SET rootX TO the RESULT OF calling find WITH x  
            SET rootY TO the RESULT OF calling find WITH y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN parent  
                    SET the element at position word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x COMMA y IN similarPairs  
            CALL union WITH x AND y  
        END FOR  

        FOR each w1 COMMA w2 IN parallel elements of sentence1 AND sentence2  
            IF w1 NOT IN parent  
                SET the element at position w1 of parent TO w1  
            END IF  
            IF w2 NOT IN parent  
                SET the element at position w2 of parent TO w2  
            END IF  
            IF the RESULT OF calling find WITH w1 NOT EQUALS the RESULT OF calling find WITH w2  
                RETURN false  
            END IF  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
