CLASS ListNode  
    FUNCTION __init__ with parameters self val next  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode constructed with element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the end of values  
        SET node TO a new ListNode constructed with val  
        SET p next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND the result of is_same_list with parameters the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd with parameters self head n  
        SET dummy TO a new ListNode constructed with zero  
        SET dummy next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO first next  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO first next  
            SET second TO second next  
        END WHILE  
        SET second next TO the next element of the next element of second  
        RETURN dummy next  
    END FUNCTION  
END CLASS
