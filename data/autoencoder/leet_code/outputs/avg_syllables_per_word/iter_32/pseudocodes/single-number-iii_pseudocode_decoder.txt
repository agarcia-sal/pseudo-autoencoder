CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO the result of xor_all XOR num  
        END FOR  
        SET diff_bit TO the result of bitwise AND of xor_all AND negative of xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the result of bitwise AND of num AND diff_bit NOT EQUALS zero  
                SET num1 TO the result of num1 XOR num  
            ELSE  
                SET num2 TO the result of num2 XOR num  
            END IF  
        END FOR  
        RETURN list containing num1 AND num2  
    END FUNCTION  
END CLASS
