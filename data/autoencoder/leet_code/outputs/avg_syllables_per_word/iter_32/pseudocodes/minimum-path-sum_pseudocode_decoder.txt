CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        SET dp TO an empty list  
        FOR index i FROM zero TO m MINUS one  
            SET row TO an empty list  
            FOR index j FROM zero TO n MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        
        SET element at position zero of element at position zero of dp TO element at position zero of element at position zero of grid  
        
        FOR index j FROM one TO n MINUS one  
            SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp PLUS element at position j of element at position zero of grid  
        END FOR  
        
        FOR index i FROM one TO m MINUS one  
            SET element at position zero of element at position i of dp TO element at position zero of element at position i MINUS one of dp PLUS element at position zero of element at position i of grid  
        END FOR  
        
        FOR index i FROM one TO m MINUS one  
            FOR index j FROM one TO n MINUS one  
                IF element at position i MINUS one of element at position j of dp LESS THAN element at position j MINUS one of element at position i of dp  
                    SET minimum_previous TO element at position i MINUS one of element at position j of dp  
                ELSE  
                    SET minimum_previous TO element at position j MINUS one of element at position i of dp  
                END IF  
                SET element at position j of element at position i of dp TO minimum_previous PLUS element at position j of element at position i of grid  
            END FOR  
        END FOR  
        
        RETURN element at position n MINUS one of element at position m MINUS one of dp  
    END FUNCTION  
END CLASS
