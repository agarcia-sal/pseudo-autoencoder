CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right  
                SET left_cost TO dp(left, pivot MINUS one)  
                SET right_cost TO dp(pivot PLUS one, right)  
                SET cost TO pivot PLUS the GREATER OF left_cost AND right_cost  
                SET min_cost TO the SMALLER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp(one, n)  
    END FUNCTION  
END CLASS
