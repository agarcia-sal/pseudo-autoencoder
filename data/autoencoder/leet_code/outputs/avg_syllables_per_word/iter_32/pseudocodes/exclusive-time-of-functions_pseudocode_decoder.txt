CLASS Solution  
    FUNCTION exclusiveTime(n parameter n, logs parameter logs)  
        SET exclusive_time TO a list of zero values with the length equal to n  
        SET stack TO an empty list  
        FOR each log IN logs  
            SET function_id_and_action_and_timestamp TO the result of splitting log by the character of colon  
            SET function_id TO the integer value of the element at position zero of function_id_and_action_and_timestamp  
            SET action TO the element at position one of function_id_and_action_and_timestamp  
            SET timestamp TO the integer value of the element at position two of function_id_and_action_and_timestamp  
            IF action EQUALS the word start  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    SET prev_function_id TO the first element of the last element of stack  
                    SET prev_timestamp TO the second element of the last element of stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple of function_id AND timestamp TO stack  
            ELSE  
                SET prev_function_id TO the first element of the last element popped from stack  
                SET prev_timestamp TO the second element of the last element popped from stack  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    SET the second element of the last element of stack TO timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
