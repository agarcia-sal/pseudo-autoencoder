CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a list of lists where each inner list contains zero repeated n PLUS one times and there are m PLUS one such inner lists  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AT element at position j MINUS one EQUALS string one  
                    SET dp element at position i AT element at position j TO the MINIMUM OF dp element at position i MINUS one AT element at position j AND dp element at position i AT element at position j MINUS one AND dp element at position i MINUS one AT element at position j MINUS one PLUS one  
                    SET max_side TO the MAXIMUM OF max_side AND dp element at position i AT element at position j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
