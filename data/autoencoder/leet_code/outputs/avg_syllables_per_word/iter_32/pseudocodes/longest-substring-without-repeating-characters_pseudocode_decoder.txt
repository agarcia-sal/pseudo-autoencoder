CLASS Solution  
    FUNCTION lengthOfLongestSubstring with parameter s  
        SET char_index_map TO an empty map  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end and char IN the positions and elements of s  
            IF char is a key in char_index_map AND the value at char in char_index_map is GREATER THAN OR EQUAL TO start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            IF max_length is LESS THAN OR EQUAL TO max_length  
                SET max_length TO max_length  
            END IF  
            IF max_length is LESS THAN end MINUS start PLUS one  
                SET max_length TO end MINUS start PLUS one  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
