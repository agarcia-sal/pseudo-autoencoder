CLASS Solution  
    FUNCTION minOperations(target LIST OF INTEGER arr LIST OF INTEGER) RETURNS INTEGER  
        SET index_map TO an empty dictionary  
        FOR each index FROM zero TO the LENGTH OF target MINUS one  
            SET value TO element at position index of target  
            SET index_map at key value TO index  
        END FOR  
        SET transformed_arr TO an empty list  
        FOR each value IN arr  
            IF value IS A KEY IN index_map THEN  
                APPEND element at key value of index_map TO transformed_arr  
            END IF  
        END FOR  
        SET lis TO an empty list  
        FOR each num IN transformed_arr  
            SET pos TO the smallest position at which num can be inserted in lis while maintaining sorted order  
            IF pos EQUALS the LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
        SET lcs_length TO the LENGTH OF lis  
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
