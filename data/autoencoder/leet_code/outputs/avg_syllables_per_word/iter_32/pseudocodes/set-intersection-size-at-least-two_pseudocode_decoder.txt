CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the value of element at position two of each interval AND BY the NEGATION OF the value of element at position one of each interval  
        SET first TO the value of element at position two of the first interval MINUS one  
        SET second TO the value of element at position two of the first interval  
        SET count TO two  
        FOR each pair start and end IN the intervals beginning from the second element TO the last element  
            IF start IS GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
