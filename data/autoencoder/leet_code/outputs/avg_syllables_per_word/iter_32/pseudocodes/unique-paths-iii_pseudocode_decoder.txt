CLASS Solution  
    FUNCTION uniquePathsIII(grid)  
        FUNCTION dfs(x, y, steps)  
            IF element at position x of grid EQUALS to two THEN  
                RETURN steps EQUALS to zero  
            END IF  
            SET original TO element at position x of grid at position y  
            SET element at position x of grid at position y TO negative two  
            SET count TO zero  
            FOR each pair of dx and dy IN the list of pairs consisting of negative one and zero, one and zero, zero and negative one, zero and one  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN the LENGTH OF grid AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN the LENGTH OF element at position zero of grid AND element at position nx of grid at position ny GREATER THAN OR EQUAL TO zero THEN  
                    SET count TO count PLUS dfs with arguments nx ny steps MINUS one  
                END IF  
            END FOR  
            SET element at position x of grid at position y TO original  
            RETURN count  
        END FUNCTION  

        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start_x TO zero  
        SET start_y TO zero  
        SET empty_squares TO one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid at position j EQUALS to one THEN  
                    SET start_x TO i  
                    SET start_y TO j  
                ELSE IF element at position i of grid at position j EQUALS to zero THEN  
                    INCREMENT empty_squares BY one  
                END IF  
            END FOR  
        END FOR  

        RETURN dfs with arguments start_x start_y empty_squares  
    END FUNCTION  
END CLASS
