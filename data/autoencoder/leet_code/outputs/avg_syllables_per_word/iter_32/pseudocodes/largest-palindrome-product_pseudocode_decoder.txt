CLASS Solution  
    FUNCTION largestPalindrome(n)  
        IF n EQUALS one  
            RETURN nine  
        END IF  
        
        SET upper_limit TO ten MULTIPLIED BY ten raised to the power of n MINUS one  
        SET lower_limit TO ten MULTIPLIED BY ten raised to the power of n MINUS one MINUS one  
        
        SET max_palindrome TO zero  
        
        FOR i FROM upper_limit TO lower_limit DECREMENT one  
            FOR j FROM i TO lower_limit DECREMENT one  
                SET product TO i MULTIPLIED BY j  
                IF product LESS THAN OR EQUAL TO max_palindrome  
                    BREAK  
                END IF  
                SET str_product TO product converted to string  
                IF str_product EQUALS str_product reversed  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_palindrome MODULO one thousand three hundred thirty seven  
    END FUNCTION  
END CLASS
