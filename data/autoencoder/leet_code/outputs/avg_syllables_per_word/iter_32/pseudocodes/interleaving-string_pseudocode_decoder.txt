CLASS Solution  
    FUNCTION isInterleave(s1 s2 s3)  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN False  
        END IF  
        
        SET dp TO a list of False values with the LENGTH OF s2 PLUS one elements  
        SET the element at position zero of dp TO True  
        
        FOR index i FROM zero TO the LENGTH OF s1  
            FOR index j FROM zero TO the LENGTH OF s2  
                IF index i GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp AND the element at position index i MINUS one of s1 EQUALS the element at position index i PLUS j MINUS one of s3  
                END IF  
                IF index j GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp OR the element at position j MINUS one of dp AND the element at position index j MINUS one of s2 EQUALS the element at position index i PLUS j MINUS one of s3  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
