CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of each element in nums  
        
        FUNCTION compare(x y)  
            IF the concatenation of x PLUS y as a string GREATER THAN the concatenation of y PLUS x as a string  
                RETURN negative one  
            ELSE IF the concatenation of x PLUS y as a string LESS THAN the concatenation of y PLUS x as a string  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str using the comparison function compare as key  
        
        IF the element at position zero of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result_string TO the concatenation of all elements in nums_str in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
