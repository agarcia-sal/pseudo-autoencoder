CLASS Solution  
    FUNCTION numFactoredBinaryTrees(arr)  
        SET MOD TO one billion PLUS seven  
        SORT arr in ascending order  
        SET dp TO an empty map from number to count  
        FOR each number IN arr  
            SET dp at key number TO one  
        END FOR  
        FOR index i FROM zero TO length of arr MINUS one  
            SET num TO element at position i of arr  
            FOR index j FROM zero TO i MINUS one  
                SET factor_candidate TO element at position j of arr  
                IF num MODULO factor_candidate EQUALS zero THEN  
                    SET right_child TO num DIVIDED BY factor_candidate  
                    IF right_child IS A KEY IN dp THEN  
                        SET dp at key num TO dp at key num PLUS dp at key factor_candidate MULTIPLIED BY dp at key right_child  
                        SET dp at key num TO dp at key num MODULO MOD  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET total_count TO zero  
        FOR each value IN all values of dp  
            INCREMENT total_count BY value  
        END FOR  
        SET total_count TO total_count MODULO MOD  
        RETURN total_count  
    END FUNCTION  
END CLASS
