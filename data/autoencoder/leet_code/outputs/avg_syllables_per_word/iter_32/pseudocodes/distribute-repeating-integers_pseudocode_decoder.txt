CLASS Solution  
    FUNCTION canDistribute(nums AS list of integers quantity AS list of integers) RETURNS boolean  
        SET count TO list of the frequencies of each unique element in nums  
        SORT quantity IN descending order  
        
        FUNCTION can_satisfy(index AS integer) RETURNS boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN True  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF element at position i of count IS GREATER THAN OR EQUAL TO element at position index of quantity THEN  
                    DECREMENT element at position i of count BY element at position index of quantity  
                    IF can_satisfy(index PLUS one) RETURNS True THEN  
                        RETURN True  
                    END IF  
                    INCREMENT element at position i of count BY element at position index of quantity  
                END IF  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN can_satisfy(zero)  
    END FUNCTION  
END CLASS
