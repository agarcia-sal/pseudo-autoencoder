CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the list of cuts APPENDED WITH zero AND n AND THEN SORTED IN ASCENDING ORDER  
        
        FUNCTION dp(left, right)  
            IF right MINUS left LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS the result of dp WITH left AND i PLUS the result of dp WITH i AND right  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp WITH zero AND the LENGTH OF cuts MINUS one  
    END FUNCTION  
END CLASS
