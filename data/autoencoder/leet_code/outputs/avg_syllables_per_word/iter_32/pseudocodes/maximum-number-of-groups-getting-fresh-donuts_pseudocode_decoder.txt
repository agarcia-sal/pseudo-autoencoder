CLASS Solution  
    FUNCTION maxHappyGroups(batchSize groupOfInteger, groups groupOfListOfInteger) RETURNS integer  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeroes with the LENGTH EQUAL TO batchSize  
        
        FOR each group IN groups  
            INCREMENT the element at position the group MODULO batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder singleInteger, remainder_count variableNumberOfIntegers) RETURNS integer  
            SET remainder_count TO a list CONSTRUCTED FROM remainder_count  
            SET max_happy TO zero  
            
            FOR index i FROM zero TO batchSize MINUS one  
                IF the element at position i of remainder_count EQUALS zero  
                    CONTINUE to the next iteration of the loop  
                END IF  
                
                DECREMENT the element at position i of remainder_count BY one  
                
                SET new_remainder TO the remainder MINUS i MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, the elements of remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, the elements of remainder_count)  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT the element at position i of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, the elements of remainder_count)  
    END FUNCTION  
END CLASS
