CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET total_sum TO the SUM OF all elements in nums  
        SET current_sum TO the SUM OF for each index i and element num in nums the MULTIPLICATION of i AND num  
        SET max_value TO current_sum  
        FOR k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY the element at position n MINUS k of nums  
            SET max_value TO the GREATER VALUE BETWEEN max_value AND current_sum  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
