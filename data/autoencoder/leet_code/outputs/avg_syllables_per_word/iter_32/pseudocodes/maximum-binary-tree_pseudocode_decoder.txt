CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN None  
        END IF  
        SET max_val TO the maximum element in nums  
        SET max_index TO the position of max_val in nums  
        SET root TO a new TreeNode with max_val  
        SET root.left TO the result of calling constructMaximumBinaryTree on the subsection of nums from position one TO the position before max_index  
        SET root.right TO the result of calling constructMaximumBinaryTree on the subsection of nums from the position after max_index TO the last position  
        RETURN root  
    END FUNCTION  
END CLASS
