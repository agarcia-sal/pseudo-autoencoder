CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position one of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position one of mat2  
        SET result TO a list constructed by repeating the list of zero repeated n times for each element from one TO m  
        FOR each i FROM one TO m  
            FOR each j FROM one TO n  
                FOR each p FROM one TO k  
                    INCREMENT the element at position i of result at position j BY the element at position i of mat1 at position p MULTIPLIED BY the element at position p of mat2 at position j  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
