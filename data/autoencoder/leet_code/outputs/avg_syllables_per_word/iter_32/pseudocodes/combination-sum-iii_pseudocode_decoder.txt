CLASS Solution  
    FUNCTION combinationSum3(k parameter, n parameter)  
        FUNCTION backtrack(start parameter, path parameter, target parameter, k parameter)  
            IF target parameter EQUALS zero AND k parameter EQUALS zero  
                APPEND a copy of path parameter TO result variable  
                RETURN  
            END IF  
            IF target parameter LESS THAN zero OR k parameter LESS THAN zero  
                RETURN  
            END IF  
            FOR i variable FROM start parameter TO the number before ten  
                APPEND i variable TO path parameter  
                CALL backtrack with arguments i variable PLUS one, path parameter, target parameter MINUS i variable, k parameter MINUS one  
                REMOVE last element FROM path parameter  
            END FOR  
        END FUNCTION  
        SET result variable TO empty list  
        CALL backtrack with arguments one, empty list, n parameter, k parameter  
        RETURN result variable  
    END FUNCTION  
END CLASS
