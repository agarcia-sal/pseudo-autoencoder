CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO a list of zeros of length n PLUS one  
        
        FOR index FROM zero TO n  
            IF element at position index of ranges GREATER THAN zero  
                SET left_bound TO the greater value between zero AND index MINUS element at position index of ranges  
                SET right_bound TO the smaller value between n AND index PLUS element at position index of ranges  
                SET element at position left_bound of max_range TO the greater value between element at position left_bound of max_range AND right_bound  
            END IF  
        END FOR  
        
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR index FROM zero TO n  
            IF index GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF index GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the greater value between farthest AND element at position index of max_range  
        END FOR  
        
        RETURN taps  
    END FUNCTION  
END CLASS
