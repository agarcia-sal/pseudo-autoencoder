CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_of_values_less_than_or_equal_to_x TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x THEN  
                    INCREMENT count_of_values_less_than_or_equal_to_x BY one  
                END IF  
            END FOR  
            IF count_of_values_less_than_or_equal_to_x GREATER THAN x THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN smallest index in the range from zero TO the length of nums where function f returns true  
    END FUNCTION  
END CLASS
