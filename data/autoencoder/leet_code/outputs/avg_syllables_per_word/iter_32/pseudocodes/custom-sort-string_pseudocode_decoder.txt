CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        FOR each character c AND index i IN order enumerated from zero  
            SET element at key c in d TO i  
        END FOR  
        SET sorted_list TO s sorted in ascending order based on the value obtained BY retrieving from dictionary d the key of character OR zero if key is not found  
        SET result TO concatenation of all characters in sorted_list in sequence  
        RETURN result  
    END FUNCTION  
END CLASS
