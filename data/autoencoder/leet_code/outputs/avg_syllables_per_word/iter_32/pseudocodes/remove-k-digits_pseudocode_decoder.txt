CLASS Solution  
    FUNCTION removeKdigits(num k)  
        SET stack TO empty list  
        FOR each digit IN characters of num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at position last of stack GREATER THAN digit  
                REMOVE element at position last FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO elements from position first TO position LENGTH OF stack MINUS k OF stack  
        END IF  
        SET joined_string TO concatenation of all elements in final_stack in order  
        SET result TO joined_string with all leading characters that are the character zero REMOVED  
        IF result IS empty string  
            RETURN string zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
