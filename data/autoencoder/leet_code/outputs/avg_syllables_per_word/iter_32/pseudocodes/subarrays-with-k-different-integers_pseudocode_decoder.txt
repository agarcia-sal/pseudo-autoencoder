CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums WITH TYPE List of integers, k WITH TYPE integer) RETURN integer  
        FUNCTION atMostKDistinct(t WITH TYPE integer) RETURN integer  
            SET count TO a mapping from elements TO integers with default integer zero  
            SET left TO zero  
            SET result TO zero  
            FOR each right FROM zero TO the length of nums MINUS one  
                INCREMENT the value in count at element at position right of nums BY one  
                WHILE the number of keys in count IS GREATER THAN t  
                    DECREMENT the value in count at element at position left of nums BY one  
                    IF the value in count at element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the result of calling atMostKDistinct with argument k MINUS the result of calling atMostKDistinct with argument k MINUS one  
    END FUNCTION  
END CLASS
