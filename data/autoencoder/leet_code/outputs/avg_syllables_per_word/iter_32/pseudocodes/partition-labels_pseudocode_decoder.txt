CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO empty dictionary  
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position i of s  
            SET last_occurrence AT character TO i  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO empty list  
        
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position i of s  
            IF last_occurrence AT character GREATER THAN end  
                SET end TO last_occurrence AT character  
            END IF  
            
            IF i EQUALS end  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
