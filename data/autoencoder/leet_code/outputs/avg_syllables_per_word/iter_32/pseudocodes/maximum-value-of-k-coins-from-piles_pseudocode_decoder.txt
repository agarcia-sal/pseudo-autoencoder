CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a list containing n PLUS one elements each of which is a list containing k PLUS one zeros  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp at position i at position j TO the GREATER OF dp at position i at position j AND dp at position i MINUS one at position j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position n at position k  
    END FUNCTION  
END CLASS
