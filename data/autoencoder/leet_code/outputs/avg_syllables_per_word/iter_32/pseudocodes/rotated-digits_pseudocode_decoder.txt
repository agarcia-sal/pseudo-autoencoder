CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the collection of elements two five six nine  
        SET same_digits TO the collection of elements zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF for every digit IN num_str the digit is contained in the union of change_digits AND same_digits holds TRUE THEN  
                IF there exists at least one digit IN num_str the digit is contained in change_digits THEN  
                    RETURN TRUE  
                END IF  
            END IF  
            RETURN FALSE  
        END FUNCTION  
        
        SET good_count TO the total number of values i in the range from one TO n inclusive for which is_good_number of i is TRUE  
        RETURN good_count  
    END FUNCTION  
END CLASS
