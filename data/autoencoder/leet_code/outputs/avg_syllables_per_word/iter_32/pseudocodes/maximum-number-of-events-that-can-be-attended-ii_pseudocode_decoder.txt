CLASS Solution  
    FUNCTION maxValue(events parameter as List OF List OF integer, k parameter as integer) RETURNS integer  
        PERFORM a sort of the list named events by the end day element of each event  
        SET n TO the length of the list named events  
          
        INITIALIZE dp AS a list containing n PLUS one elements each being a list containing k PLUS one zero elements  
          
        FOR i FROM one TO n  
            SET start TO the first element of the event at position i MINUS one in events  
            SET end TO the second element of the event at position i MINUS one in events  
            SET value TO the third element of the event at position i MINUS one in events  
              
            SET j TO the position found by locating the leftmost index in events where the end day is GREATER THAN OR EQUAL TO start  
              
            FOR l FROM one TO k  
                SET dp element at position i element at position l TO THE GREATER OF the dp element at position i MINUS one element at position l AND the sum of the dp element at position j element at position l MINUS one PLUS value  
            END FOR  
        END FOR  
          
        RETURN the dp element at position n element at position k  
    END FUNCTION  
END CLASS
