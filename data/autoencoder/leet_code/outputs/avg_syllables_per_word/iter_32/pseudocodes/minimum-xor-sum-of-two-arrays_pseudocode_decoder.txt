CLASS Solution  
    FUNCTION minimumXORSum(nums1 nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of size TWO RAISED TO THE POWER OF n each element INITIALIZED TO infinite value  
        SET element at position zero of dp TO zero  
        
        FOR each mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the NUMBER OF ONES IN THE binary REPRESENTATION OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of mask IS SET THEN  
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND dp at position corresponding to mask WITH bit at position j UNSET PLUS result of BITWISE XOR BETWEEN element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position TWO RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
