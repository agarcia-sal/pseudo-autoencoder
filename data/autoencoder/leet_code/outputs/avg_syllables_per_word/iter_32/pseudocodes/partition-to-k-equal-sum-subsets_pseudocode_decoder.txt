CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the sum of nums  
        IF the remainder of total_sum DIVIDED BY k NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending order  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k EQUALS zero THEN  
                RETURN true  
            END IF  
            IF current_sum EQUALS target THEN  
                RETURN can_partition zero k MINUS one zero used  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF element at position i of used EQUALS false AND current_sum PLUS element at position i of nums LESS THAN OR EQUAL TO target THEN  
                    SET element at position i of used TO true  
                    IF can_partition i PLUS one k current_sum PLUS element at position i of nums used RETURNS true THEN  
                        RETURN true  
                    END IF  
                    SET element at position i of used TO false  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        SET used TO a list of false repeated the LENGTH OF nums times  
        RETURN can_partition zero k zero used  
    END FUNCTION  
END CLASS
