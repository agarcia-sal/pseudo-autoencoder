CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is no value AND q is no value  
        RETURN true  
    ELSE IF p is no value OR q is no value  
        RETURN false  
    ELSE IF the val of p NOT EQUALS the val of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left element of p AND the left element of q AND the result of is_same_tree with the right element of p AND the right element of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root is no value  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue containing root  
        SET left_to_right TO true  
        WHILE queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty double ended queue  
            FOR each index FROM one TO level_size  
                SET node TO the element removed from the front of queue  
                IF left_to_right is true  
                    APPEND the val of node TO current_level  
                ELSE  
                    APPEND the val of node TO the front of current_level  
                END IF  
                IF the left element of node is not no value  
                    APPEND the left element of node TO queue  
                END IF  
                IF the right element of node is not no value  
                    APPEND the right element of node TO queue  
                END IF  
            END FOR  
            APPEND the list converted from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
