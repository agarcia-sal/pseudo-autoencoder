CLASS Solution  
    FUNCTION minInterval(intervals AND queries)  
        SORT intervals IN ASCENDING ORDER BY the FIRST element of each interval  
        SET sorted_queries TO a sorted list of pairs where each pair consists of an element from queries AND its original index in queries  
        SET min_heap TO an empty list for storing pairs of interval size AND interval end  
        SET result TO a list of negative one values WITH the SAME LENGTH AS queries  
        SET interval_index TO zero  
        FOR each pair query AND original_index IN sorted_queries  
            WHILE interval_index IS LESS THAN the LENGTH OF intervals AND the FIRST element of the interval at position interval_index in intervals IS LESS THAN OR EQUAL TO query  
                SET start TO the FIRST element of the interval at position interval_index in intervals  
                SET end TO the SECOND element of the interval at position interval_index in intervals  
                SET interval_size TO end MINUS start PLUS one  
                PUSH the pair interval_size AND end INTO min_heap  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT EMPTY AND the SECOND element of the FIRST pair in min_heap IS LESS THAN query  
                REMOVE the FIRST pair FROM min_heap  
            END WHILE  
            IF min_heap IS NOT EMPTY  
                SET the element at position original_index of result TO the FIRST element of the FIRST pair in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
