CLASS Solution  
    FUNCTION longestPalindrome(word1, word2)  
        SET s TO the concatenation of word1 and word2  
        SET n TO the length of s  
        SET dp TO a two dimensional list of size n by n with all elements set to zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i of dp at position i TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i of dp at position j TO element at position i PLUS one of dp at position j MINUS one PLUS two  
                    IF i LESS THAN the length of word1 AND j GREATER THAN OR EQUAL TO the length of word1  
                        SET max_len TO the greater of max_len AND element at position i of dp at position j  
                    END IF  
                ELSE  
                    SET element at position i of dp at position j TO the greater of element at position i PLUS one of dp at position j AND element at position i of dp at position j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
