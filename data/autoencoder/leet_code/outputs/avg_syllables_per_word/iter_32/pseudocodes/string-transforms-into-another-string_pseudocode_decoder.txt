CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN truth value true  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each pair char1 char2 IN parallel traversal of str1 and str2  
            IF char1 IS IN keys of conversion_map  
                IF value at key char1 in conversion_map NOT EQUALS char2  
                    RETURN truth value false  
                END IF  
            ELSE  
                ASSIGN key char1 in conversion_map TO char2  
            END IF  
        END FOR  
        IF the count OF unique elements in str2 EQUALS twenty six  
            RETURN truth value false  
        END IF  
        RETURN truth value true  
    END FUNCTION  
END CLASS
