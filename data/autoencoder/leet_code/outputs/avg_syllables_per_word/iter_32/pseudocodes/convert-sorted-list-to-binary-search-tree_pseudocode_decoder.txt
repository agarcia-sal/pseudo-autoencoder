CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last position  
        SET node TO a new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the val element of p1 EQUALS the val element of p2 AND is_same_list(the next element of p1, the next element of p2)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the element from the front of queue and ASSIGN TO node  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET the left element of node TO a new TreeNode with the element at position i of values  
            APPEND the left element of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET the right element of node TO a new TreeNode with the element at position i of values  
            APPEND the right element of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val element of p NOT EQUALS the val element of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left element of p, the left element of q) AND is_same_tree(the right element of p, the right element of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head EQUALS none  
            RETURN none  
        END IF  
        
        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next element of fast NOT EQUALS right  
                SET slow TO the next element of slow  
                SET fast TO the next element of the next element of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right  
                RETURN none  
            END IF  
            SET mid TO findMiddle(left, right)  
            SET node TO a new TreeNode with the val element of mid  
            SET the left element of node TO convertListToBST(left, mid)  
            SET the right element of node TO convertListToBST(the next element of mid, right)  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST(head, none)  
    END FUNCTION  
END CLASS
