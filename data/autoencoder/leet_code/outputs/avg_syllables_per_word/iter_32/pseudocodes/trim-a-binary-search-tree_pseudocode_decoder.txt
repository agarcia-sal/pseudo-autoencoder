CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET the val attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the values list EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the queue has elements  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left attributes of p and q AND the result of calling is_same_tree with the right attributes of p and q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self root low high)  
        IF root EQUALS none  
            RETURN none  
        END IF  
        IF the val attribute of root LESS THAN low  
            RETURN the result of calling trimBST with the right attribute of root and with low and high  
        END IF  
        IF the val attribute of root GREATER THAN high  
            RETURN the result of calling trimBST with the left attribute of root and with low and high  
        END IF  
        SET the left attribute of root TO the result of calling trimBST with the left attribute of root and with low and high  
        SET the right attribute of root TO the result of calling trimBST with the right attribute of root and with low and high  
        RETURN root  
    END FUNCTION  
END CLASS
