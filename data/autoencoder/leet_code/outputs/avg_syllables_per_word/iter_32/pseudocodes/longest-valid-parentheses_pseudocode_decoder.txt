CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a list containing the element negative one  
        SET max_length TO zero  
        FOR each index i AND character char IN the string s BY enumeration  
            IF character char EQUALS the opening parenthesis symbol  
                APPEND the index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the LENGTH OF stack EQUALS zero  
                    APPEND the index i TO stack  
                ELSE  
                    SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN index i AND the last element of stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
