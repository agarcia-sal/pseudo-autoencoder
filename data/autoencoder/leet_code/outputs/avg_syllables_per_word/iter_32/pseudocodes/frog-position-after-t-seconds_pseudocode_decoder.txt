CLASS Solution  
    FUNCTION frogPosition(n PARAMETER, edges PARAMETER, t PARAMETER, target PARAMETER)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO a new map with default value empty list  
        FOR each pair u AND v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        SET queue TO a new double ended queue containing one element of a tuple of one current node with value one current probability with value one point zero current time with value zero  
        SET visited TO a new set containing one  
        
        WHILE queue holds any element  
            SET current node TO first element of the first element removed from the left side of queue  
            SET current probability TO second element of the first element removed from the left side of queue  
            SET current time TO third element of the first element removed from the left side of queue  
            
            IF current time EQUALS t OR (current node NOT EQUALS one AND length of element at position current node of graph EQUALS one)  
                IF current node EQUALS target  
                    RETURN current probability  
                ELSE  
                    CONTINUE to next iteration of while loop  
                END IF  
            END IF  
            
            SET possible moves TO length of element at position current node of graph MINUS one IF current node NOT EQUALS one OTHERWISE zero  
            
            FOR each neighbor IN element at position current node of graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor current probability DIVIDED BY possible moves current time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
