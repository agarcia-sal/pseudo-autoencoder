CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the LIST OF CHARACTERS that form num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero HOLDS  
            SET swapped TO False  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF the ELEMENT at position i of num GREATER THAN the ELEMENT at position i PLUS one of num HOLDS  
                    SET min_digit TO the CHARACTER nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF the ELEMENT at position j of num LESS THAN min_digit HOLDS  
                            SET min_digit TO the ELEMENT at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the SUBLIST OF num FROM position i TO position min_index PLUS one TO a LIST WHERE the FIRST ELEMENT IS the ELEMENT at position min_index of num FOLLOWED BY the SUBLIST OF num FROM position i TO position min_index MINUS one  
                    DECREMENT k BY the VALUE of min_index MINUS i  
                    SET swapped TO True  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS False HOLDS  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        RETURN the CONCATENATION OF ALL ELEMENTS of num INTO A STRING  
    END FUNCTION  
END CLASS
