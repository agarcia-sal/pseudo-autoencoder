CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO a sorted list of each first element of each element in flowers  
        SET end_times TO a sorted list of each second element of each element in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the position to the right where person would be inserted in start_times to keep it sorted  
            SET ended TO the position to the left where person would be inserted in end_times to keep it sorted  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
