CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        SET position TO an empty dictionary  
        SET index TO zero  
        FOR each person IN row  
            ASSIGN index TO person in position dictionary  
            INCREMENT index BY one  
        END FOR  
        SET swaps TO zero  
        SET length_of_row TO the LENGTH OF row  
        SET i TO zero  
        WHILE i LESS THAN length_of_row  
            SET current_person TO element at position i of row  
            SET next_person TO element at position i PLUS one of row  
            SET correct_pair TO the value of current_person PLUS one if current_person is even OTHERWISE current_person MINUS one  
            IF next_person NOT EQUALS correct_pair  
                SET swap_idx TO the value in position dictionary at key correct_pair  
                SET element at position i PLUS one of row TO element at position swap_idx of row  
                SET element at position swap_idx of row TO next_person  
                ASSIGN swap_idx TO correct_pair in position dictionary  
                ASSIGN i PLUS one TO next_person in position dictionary  
                INCREMENT swaps BY one  
            END IF  
            INCREMENT i BY two  
        END WHILE  
        RETURN swaps  
    END FUNCTION  
END CLASS
