CLASS Solution  
    FUNCTION minWindow(s STRING_PARAMETER t STRING_PARAMETER) RETURNS STRING  
        SET t_count TO the frequency count of each character in t  
        SET required TO the number of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO an empty frequency count mapping  
        SET min_length TO a quantity larger than any possible window  
        SET min_start TO zero  
        
        WHILE right LESS THAN the length of s  
            SET current_character TO the element at position right of s  
            INCREMENT the frequency count of current_character in window_counts BY one  
            
            IF current_character is a key in t_count AND the frequency count of current_character in window_counts EQUALS the frequency count of current_character in t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET character_at_left TO the element at position left of s  
                
                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the frequency count of character_at_left in window_counts BY one  
                
                IF character_at_left is a key in t_count AND the frequency count of character_at_left in window_counts LESS THAN the frequency count of character_at_left in t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length IS GREATER THAN any possible window  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
