CLASS Solution  
    FUNCTION numRescueBoats(people PARAMETER, limit PARAMETER)  
        ARRANGE the elements of people in order from smallest to largest  
        SET left TO zero  
        SET right TO the total number of elements in people MINUS one  
        SET boats TO zero  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF the element at position left of people PLUS the element at position right of people IS LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
