CLASS Solution  
    FUNCTION multiply(num1 num1 AS string num2 num2 AS string) RETURNS string  
        IF num1 EQUALS the string zero OR num2 EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result TO a list of zeroes with the LENGTH OF num1 PLUS the LENGTH OF num2 elements  
        
        SET num1 TO the string num1 in reverse order  
        SET num2 TO the string num2 in reverse order  
        
        FOR index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the difference between the character code of the element at position i of num1 AND the character code of the character zero MULTIPLIED BY the difference between the character code of the element at position j of num2 AND the character code of the character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                
                SET the element at position p1 of result TO sum_ MODULO ten  
                INCREMENT the element at position p2 of result BY the integer division of sum_ BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the concatenation of the string representations of the elements of result in reverse order  
        RETURN the string result_str with all leading characters equal to the string zero removed  
    END FUNCTION  
END CLASS
