CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with LENGTH equal to two TIMES n PLUS one  
        
        FOR i FROM zero TO two TIMES n MINUS one  
            PUSH the NEGATIVE of element at position i of nums INTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n THEN  
                SET element at position i of min_sums TO the NEGATIVE of the SUM OF all elements IN min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with LENGTH equal to two TIMES n PLUS one  
        
        FOR i FROM three TIMES n MINUS one DOWN TO n  
            PUSH element at position i of nums INTO max_heap  
            IF the LENGTH OF max_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n THEN  
                SET element at position i of max_sums TO the SUM OF all elements IN max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        
        FOR i FROM n MINUS one TO two TIMES n MINUS one  
            IF min_diff GREATER THAN element at position i of min_sums MINUS element at position i PLUS one of max_sums THEN  
                SET min_diff TO element at position i of min_sums MINUS element at position i PLUS one of max_sums  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
