CLASS Solution  
    FUNCTION shortestSubarray(nums WITH TYPE List OF integers, k WITH TYPE integer) RETURNS integer  
        SET prefix_sum TO list containing a single element zero  
        FOR each num IN nums  
            APPEND the sum of the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        ASSIGN reference TO a new empty double-ended queue named dq  
        SET min_length TO positive infinity  
        FOR each pair of index i AND corresponding element current_sum IN prefix_sum with enumeration starting from zero  
            WHILE dq is not empty AND the difference between current_sum AND the element at the position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value BETWEEN min_length AND the difference BETWEEN i AND the element REMOVED from the front of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at the position at the back of dq in prefix_sum  
                REMOVE the element at the back of dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
