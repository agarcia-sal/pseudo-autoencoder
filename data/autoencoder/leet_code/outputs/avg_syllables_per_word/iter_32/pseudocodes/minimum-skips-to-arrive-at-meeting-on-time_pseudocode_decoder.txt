CLASS Solution
    FUNCTION minSkips(dist list of integers speed integer hoursBefore integer) RETURNS integer
        SET n TO the LENGTH OF dist
        SET dp TO a LIST of LENGTH n where EACH element IS a LIST of LENGTH n PLUS one FILLED WITH infinity
        SET total_dist TO the SUM OF dist
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN
            RETURN negative one
        END IF
        
        SET element at position zero at position zero of dp TO element at position zero of dist
        FOR j FROM one TO n MINUS one
            SET element at position zero at position j of dp TO element at position zero of dist
        END FOR
        
        FOR i FROM one TO n MINUS one
            FOR j FROM zero TO i
                IF j LESS THAN i THEN
                    SET element at position i at position j of dp TO the CEILING OF element at position i MINUS one at position j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist
                END IF
                IF j GREATER THAN zero THEN
                    SET element at position i at position j of dp TO the MINIMUM OF element at position i at position j of dp AND element at position i MINUS one at position j MINUS one of dp PLUS element at position i of dist
                END IF
            END FOR
        END FOR
        
        FOR j FROM zero TO n MINUS one
            IF element at position n MINUS one at position j of dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN
                RETURN j
            END IF
        END FOR
        
        RETURN negative one
    END FUNCTION
END CLASS
