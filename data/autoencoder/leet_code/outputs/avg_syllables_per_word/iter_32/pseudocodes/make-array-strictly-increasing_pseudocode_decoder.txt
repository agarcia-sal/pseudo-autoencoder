CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2 IN ascending order  
        SET dp TO a dictionary with key negative one and value zero  
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each pair of last_num and ops IN dp items  
                IF number GREATER THAN last_num THEN  
                    SET current min operations TO value at key number in new_dp OR positive infinity if none  
                    SET new_dp at key number TO the minimum of current min operations AND ops  
                END IF  
                SET idx TO the position where last_num would be inserted in arr2 to keep arr2 sorted after last_num  
                IF idx LESS THAN the length of arr2 THEN  
                    SET candidate TO the element at position idx of arr2  
                    SET current min operations TO value at key candidate in new_dp OR positive infinity if none  
                    SET new_dp at key candidate TO the minimum of current min operations AND ops PLUS one  
                END IF  
            END FOR  
            IF new_dp is empty THEN  
                RETURN negative one  
            END IF  
            SET dp TO new_dp  
        END FOR  
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
