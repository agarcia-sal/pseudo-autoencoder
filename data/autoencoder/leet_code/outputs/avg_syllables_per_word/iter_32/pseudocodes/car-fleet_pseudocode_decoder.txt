CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET list_of_pairs TO sorted list of pairs each containing element at position index of position and element at position index of speed sorted IN descending order BY the first element of each pair  
        SET time_to_target TO empty list  
        FOR each pair IN list_of_pairs  
            SET position_element TO the first element of pair  
            SET speed_element TO the second element of pair  
            SET time_needed TO target MINUS position_element DIVIDED BY speed_element  
            APPEND time_needed TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
