CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(origin_point, point_a, point_b)  
            RETURN element at position zero of point_a MINUS element at position zero of origin_point MULTIPLIED BY element at position one of point_b MINUS element at position one of origin_point MINUS element at position one of point_a MINUS element at position one of origin_point MULTIPLIED BY element at position zero of point_b MINUS element at position zero of origin_point  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position i PLUS one modulo n of points  
            SET b TO element at position i PLUS two modulo n of points  
            SET cross TO cross_product USING o a b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO minus one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero  
                    RETURN False  
                ELSE IF orientation EQUALS minus one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
