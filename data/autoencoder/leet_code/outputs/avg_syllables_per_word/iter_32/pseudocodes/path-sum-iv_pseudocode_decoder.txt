CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            SET the element with key depth and pos IN tree TO value  
        END FOR  
  
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element with key depth and pos IN tree OR none if not present  
            IF current_value EQUALS none  
                RETURN zero  
            END IF  
  
            SET new_path_sum TO path_sum PLUS current_value  
  
            SET left_child TO a pair with first element as depth PLUS one and second element as pos MULTIPLIED BY two MINUS one  
            SET right_child TO a pair with first element as depth PLUS one and second element as pos MULTIPLIED BY two  
  
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
  
            SET left_sum TO dfs with depth as left_child first element pos as left_child second element and path_sum as new_path_sum  
            SET right_sum TO dfs with depth as right_child first element pos as right_child second element and path_sum as new_path_sum  
  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
  
        RETURN dfs with depth as one pos as one and path_sum as zero  
    END FUNCTION  
END CLASS
