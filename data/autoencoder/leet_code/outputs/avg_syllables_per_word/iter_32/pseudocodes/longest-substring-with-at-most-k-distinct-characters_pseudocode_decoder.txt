CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct with parameters s and k  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty dictionary  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET element at position right of s AS key IN char_map TO value right  
            
            IF the LENGTH OF char_map GREATER THAN k THEN  
                SET leftmost_char TO the key in char_map with the minimal associated value  
                SET left TO the value associated with leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the greater value BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
