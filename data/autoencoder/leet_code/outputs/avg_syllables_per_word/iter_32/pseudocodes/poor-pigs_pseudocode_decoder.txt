CLASS Solution  
    FUNCTION poorPigs(buckets minutesToDie minutesToTest)  
        SET states TO the INTEGER RESULT OF minutesToTest DIVIDED BY minutesToDie PLUS one  
        SET logarithm_of_buckets TO the LOGARITHM OF buckets  
        SET logarithm_of_states TO the LOGARITHM OF states  
        SET pigs TO the LEAST INTEGER GREATER THAN OR EQUAL TO logarithm_of_buckets DIVIDED BY logarithm_of_states  
        RETURN pigs  
    END FUNCTION  
END CLASS
