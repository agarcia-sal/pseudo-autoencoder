CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze start destination  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the LIST OF pairs one zero MINUS one zero zero one zero MINUS one  
        SET start TO the tuple form of start  
        SET destination TO the tuple form of destination  
        SET visited TO an EMPTY SET  
        SET pq TO a LIST containing the PAIR of zero AND start  

        WHILE pq is not EMPTY  
            SET dist TO the FIRST element OF the RESULT OF extract the SMALLEST element FROM pq  
            SET x TO the FIRST element OF the SECOND element OF the RESULT OF extract the SMALLEST element FROM pq  
            SET y TO the SECOND element OF the SECOND element OF the RESULT OF extract the SMALLEST element FROM pq  

            IF x EQUALS the FIRST element OF destination AND y EQUALS the SECOND element OF destination  
                RETURN dist  
            END IF  

            IF the PAIR of x AND y IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  

            ADD the PAIR of x AND y TO visited  

            FOR each dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx OF maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  

                IF the PAIR of new_x AND new_y IS NOT IN visited  
                    INSERT the PAIR of dist PLUS count AND the PAIR of new_x AND new_y INTO pq  
                END IF  
            END FOR  
        END WHILE  

        RETURN the NEGATIVE of one  
    END FUNCTION  
END CLASS
