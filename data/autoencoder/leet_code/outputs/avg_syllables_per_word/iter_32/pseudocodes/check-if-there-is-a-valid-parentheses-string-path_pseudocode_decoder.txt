CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF the SUM of m PLUS n MINUS one MODULO two NOT EQUALS zero  
            RETURN false  
        END IF  
        
        SET dp TO a three dimensional list of boolean values initialized to false with dimensions m PLUS one BY n PLUS one BY m PLUS n PLUS one  
        
        SET the element at position zero BY one BY zero of dp TO true  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF the element at position i BY j PLUS one BY k of dp OR the element at position i PLUS one BY j BY k of dp  
                        IF the element at position i BY j of grid EQUALS an opening parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one  
                            SET the element at position i PLUS one BY j PLUS one BY k PLUS one of dp TO true  
                        ELSE IF the element at position i BY j of grid EQUALS a closing parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero  
                            SET the element at position i PLUS one BY j PLUS one BY k MINUS one of dp TO true  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position m BY n BY zero of dp  
    END FUNCTION  
END CLASS
