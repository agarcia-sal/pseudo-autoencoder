CLASS Solution  
    FUNCTION getPermutation(n k)  
        SET numbers TO list of all integers from one to n inclusive  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO factorial of i MINUS one  
            SET index TO integer division of k BY fact  
            APPEND element at position index of numbers converted to string TO result  
            REMOVE element at position index FROM numbers  
            SET k TO remainder of division of k BY fact  
        END FOR  
        SET joined_result TO concatenation of all elements in result in order  
        RETURN joined_result  
    END FUNCTION  
END CLASS
