CLASS Solution  
    FUNCTION isTransformable(s  t)  
        SET pos TO a new mapping with default empty list  
        FOR index FROM zero TO the length of s MINUS one  
            SET character TO element at position index of s  
            SET digit TO the numeric value of character  
            PUSH index INTO the heap corresponding to digit in pos  
        END FOR  
        FOR index FROM zero TO the length of t MINUS one  
            SET character TO element at position index of t  
            SET digit TO the numeric value of character  
            IF the heap corresponding to digit in pos IS empty THEN  
                RETURN false  
            END IF  
            FOR smaller FROM zero TO digit MINUS one  
                IF the heap corresponding to smaller in pos IS NOT empty AND  
                   the smallest element in the heap corresponding to smaller in pos IS LESS THAN  
                   the smallest element in the heap corresponding to digit in pos THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest element from the heap corresponding to digit in pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
