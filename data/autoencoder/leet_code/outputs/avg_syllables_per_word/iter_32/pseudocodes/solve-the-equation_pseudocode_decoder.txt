CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing all occurrences of plus sign with hash followed by plus sign in side and then replacing all occurrences of minus sign with hash followed by minus sign and then splitting the resulting string by hash  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF the token is empty  
                    CONTINUE to next iteration  
                END IF  
                IF the token contains the character x  
                    IF the token EQUALS x OR the token EQUALS plus sign followed by x  
                        INCREMENT x_count BY one  
                    ELSE IF the token EQUALS minus sign followed by x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET numeric_value TO the integer conversion of the token after replacing the character x with an empty string  
                        INCREMENT x_count BY numeric_value  
                    END IF  
                ELSE  
                    SET numeric_value TO the integer conversion of the token  
                    INCREMENT num_sum BY numeric_value  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  
  
        SET left_side, right_side TO the result of splitting the equation string by the equal sign  
        SET left_x, left_num TO the result of calling parse_side with left_side  
        SET right_x, right_num TO the result of calling parse_side with right_side  
  
        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  
  
        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            SET division_result TO the result of integer division of total_num BY total_x  
            RETURN the string constructed by concatenating the character x, the equal sign, and division_result  
        END IF  
    END FUNCTION  
END CLASS
