CLASS Solution  
    FUNCTION minTrioDegree(n parameter, edges parameter)  
        SET graph TO an empty mapping from integers to sets of integers with keys from one TO n inclusive and empty sets as values  
        FOR each pair of variables u and v IN edges  
            ADD v TO the set mapped from u in graph  
            ADD u TO the set mapped from v in graph  
        END FOR  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n inclusive  
            FOR j FROM i PLUS one TO n inclusive  
                FOR k FROM j PLUS one TO n inclusive  
                    IF the set mapped from i in graph CONTAINS j AND the set mapped from i in graph CONTAINS k AND the set mapped from j in graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the size of the set mapped from i in graph PLUS the size of the set mapped from j in graph PLUS the size of the set mapped from k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
