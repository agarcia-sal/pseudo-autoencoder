CLASS Solution  
    FUNCTION fullJustify(words maxWidth)  
        FUNCTION justify_line(line num_of_letters is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line EQUALS true  
                RETURN the CONCATENATION OF the WORDS in line SEPARATED BY SINGLE SPACES FOLLOWED BY SPACES REPEATED maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one TIMES  
            END IF  
            
            SET total_spaces TO maxWidth MINUS num_of_letters  
            SET num_of_gaps TO the LENGTH OF line MINUS one  
            SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
            SET extra_spaces TO total_spaces MODULO num_of_gaps  
            
            FOR i FROM zero TO num_of_gaps MINUS one  
                IF i LESS THAN extra_spaces  
                    APPEND SPACES REPEATED spaces_between_words PLUS one TIMES TO the element at position i of line  
                ELSE  
                    APPEND SPACES REPEATED spaces_between_words TIMES TO the element at position i of line  
                END IF  
            END FOR  
            
            RETURN the CONCATENATION OF ALL elements in line WITHOUT SEPARATORS  
        END FUNCTION  
        
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of calling justify_line WITH line num_of_letters and false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result of calling justify_line WITH line num_of_letters and true TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
