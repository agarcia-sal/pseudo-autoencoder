CLASS Solution  
    FUNCTION maximumRequests(n, requests)  
        SET max_requests TO zero  
        SET num_requests TO the LENGTH OF requests  
        
        FOR mask FROM zero TO two RAISED TO THE POWER OF num_requests MINUS one  
            SET balance TO a list of n elements each SET TO zero  
            SET count TO zero  
            
            FOR i FROM zero TO num_requests MINUS one  
                IF the bitwise AND of mask AND two RAISED TO THE POWER OF i EQUALS a nonzero value  
                    SET from_building TO element at position i of requests AT position zero  
                    SET to_building TO element at position i of requests AT position one  
                    DECREMENT element at position from_building of balance BY one  
                    INCREMENT element at position to_building of balance BY one  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            
            IF for every element b in balance b EQUALS zero  
                SET max_requests TO the maximum value between max_requests AND count  
            END IF  
        END FOR  
        
        RETURN max_requests  
    END FUNCTION  
END CLASS
