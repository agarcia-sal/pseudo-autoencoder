CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty OR values DOES NOT EXIST THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None THEN  
            SET node.left TO new TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None THEN  
            SET node.right TO new TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p DOES NOT EXIST AND q DOES NOT EXIST THEN  
        RETURN True  
    ELSE IF p DOES NOT EXIST OR q DOES NOT EXIST THEN  
        RETURN False  
    ELSE IF the value of p.val NOT EQUALS the value of q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH the left child of p AND the left child of q AND the result of calling is_same_tree WITH the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low EQUALS negative infinity, high EQUALS positive infinity)  
            IF node DOES NOT EXIST THEN  
                RETURN True  
            END IF  
            IF the value of node.val LESS THAN OR EQUAL TO low OR the value of node.val GREATER THAN OR EQUAL TO high THEN  
                RETURN False  
            END IF  
            RETURN the result of calling validate WITH the left child of node AND low AND the value of node.val AND the logical AND of calling validate WITH the right child of node AND the value of node.val AND high  
        END FUNCTION  
        RETURN the result of calling validate WITH root  
    END FUNCTION  
END CLASS
