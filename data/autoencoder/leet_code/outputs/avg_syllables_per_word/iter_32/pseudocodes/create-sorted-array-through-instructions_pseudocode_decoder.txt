CLASS Solution  
    FUNCTION createSortedArray(instructions)  
        SET MOD TO one billion PLUS seven  
        SET MAX TO the maximum element in instructions PLUS one  
        SET bit TO a list of zeros with length MAX  

        FUNCTION update(idx val)  
            WHILE idx LESS THAN MAX  
                INCREMENT element at position idx of bit BY val  
                INCREMENT idx BY the bitwise AND of idx AND the negative of idx  
            END WHILE  
        END FUNCTION  

        FUNCTION query(idx)  
            SET total TO zero  
            WHILE idx GREATER THAN zero  
                INCREMENT total BY element at position idx of bit  
                DECREMENT idx BY the bitwise AND of idx AND the negative of idx  
            END WHILE  
            RETURN total  
        END FUNCTION  

        SET total_cost TO zero  
        FOR each pair of i num IN the enumeration of instructions  
            SET less_than TO query(num MINUS one)  
            SET greater_than TO i MINUS query(num)  
            IF less_than LESS THAN OR EQUAL TO greater_than  
                SET cost TO less_than  
            ELSE  
                SET cost TO greater_than  
            END IF  
            SET total_cost TO the remainder of total_cost PLUS cost DIVIDED BY MOD  
            CALL update(num one)  
        END FOR  

        RETURN total_cost  
    END FUNCTION  
END CLASS
