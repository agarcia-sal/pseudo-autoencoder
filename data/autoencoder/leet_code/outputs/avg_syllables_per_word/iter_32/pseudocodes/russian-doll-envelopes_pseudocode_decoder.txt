CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY the element at position one of each envelope IN ascending order AND BY the negation of the element at position two of each envelope IN descending order  
        SET dp TO an empty list  
        FOR each pair of underscore and h IN envelopes  
            SET idx TO the position found by performing binary search to find the leftmost insertion point of h INTO dp  
            IF idx EQUALS the length of dp THEN  
                APPEND h TO dp  
            ELSE  
                SET the element at position idx of dp TO h  
            END IF  
        END FOR  
        RETURN the length of dp  
    END FUNCTION  
END CLASS
