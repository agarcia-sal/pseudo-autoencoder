CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of tuples formed by each point in points  
        SET min_x TO the smallest value among the first element of each point in unique_points  
        SET max_x TO the largest value among the first element of each point in unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each x comma y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple formed by reflected_x and y is NOT IN unique_points  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
