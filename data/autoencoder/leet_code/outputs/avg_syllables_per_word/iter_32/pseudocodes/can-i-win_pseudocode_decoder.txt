CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  
        
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        
        SET memo TO empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF current_total AND used_numbers AS A TUPLE EXISTS IN memo  
                RETURN value FROM memo AT current_total AND used_numbers AS A TUPLE  
            END IF  
            
            FOR number FROM one TO maxChoosableInteger  
                IF used_numbers AND one LEFT SHIFTED BY number EQUALS zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, used_numbers OR one LEFT SHIFTED BY number)  
                        SET memo AT current_total AND used_numbers AS A TUPLE TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET memo AT current_total AND used_numbers AS A TUPLE TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
