CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS None  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO the result of dfs with node.left  
            SET right_node AND right_depth TO the result of dfs with node.right  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of the result of dfs with root  
    END FUNCTION  
END CLASS
