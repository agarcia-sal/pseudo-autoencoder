CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET current_gray TO prev_gray APPENDED BY list constructed as follows  
            FOR each number IN prev_gray in REVERSED order  
                COMPUTE mask OR number  
            END FOR  
        RETURN current_gray  
    END FUNCTION  
END CLASS
