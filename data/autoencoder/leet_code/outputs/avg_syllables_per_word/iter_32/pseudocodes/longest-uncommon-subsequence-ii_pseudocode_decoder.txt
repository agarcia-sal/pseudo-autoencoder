CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s t)  
            SET iterator TO the iterator of t  
            FOR each character IN s  
                IF character NOT IN the remaining elements of iterator  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT strs BY descending length AND lexicographical order  
        
        FOR each index i FROM zero TO the length of strs MINUS one  
            SET word TO element at position i of strs  
            SET unique TO true  
            
            FOR each index j FROM zero TO the length of strs MINUS one  
                IF i NOT EQUALS j  
                    SET other TO element at position j of strs  
                    IF is_subsequence(word other) IS true  
                        SET unique TO false  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
            
            IF unique IS true  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
