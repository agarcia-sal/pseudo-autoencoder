CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO list of substrings obtained by splitting input at each newline character  
        SET stack TO empty list  
        SET max_length TO zero  
          
        FOR each line IN lines  
            SET depth TO the count of tab characters in line  
            SET name TO substring of line starting from position depth plus one TO the end of line  
              
            WHILE the LENGTH of stack is GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
              
            IF stack has any elements  
                SET current_length TO last element in stack PLUS the LENGTH of name PLUS one  
            ELSE  
                SET current_length TO the LENGTH of name  
            END IF  
              
            IF name contains the character dot  
                IF max_length LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
          
        RETURN max_length  
    END FUNCTION  
END CLASS
