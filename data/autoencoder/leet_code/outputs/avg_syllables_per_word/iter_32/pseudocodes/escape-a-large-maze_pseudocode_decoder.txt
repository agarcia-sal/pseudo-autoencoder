CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked EQUALS empty list  
            RETURN True  
        END IF  
        
        SET blocked_set TO an empty set  
        FOR each block IN blocked  
            ADD tuple of block TO blocked_set  
        END FOR  
        
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY (n MINUS one) DIVIDED BY two  
        
        FUNCTION bfs(start, end)  
            SET queue TO a deque initialized with tuple of start  
            SET visited TO a set initialized with queue elements  
            
            WHILE the LENGTH OF queue is greater than zero  
                SET x, y TO the first element removed from queue  
                
                IF the LENGTH OF visited GREATER THAN max_steps OR x EQUALS element at position zero of end AND y EQUALS element at position one of end  
                    RETURN True  
                END IF  
                
                FOR each dx, dy IN the explicit list of pairs zero one zero MINUS one one zero MINUS one zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF zero LESS THAN OR EQUAL TO nx AND nx LESS THAN one million AND zero LESS THAN OR EQUAL TO ny AND ny LESS THAN one million AND tuple of nx and ny NOT IN blocked_set AND tuple of nx and ny NOT IN visited  
                        ADD tuple of nx and ny TO visited  
                        APPEND tuple of nx and ny TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        RETURN bfs(source, target) AND bfs(target, source)  
    END FUNCTION  
END CLASS
