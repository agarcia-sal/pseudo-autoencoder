CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS the REVERSE ORDER of sub  
        END FUNCTION  

        SET n TO the LENGTH OF s  

        SET is_pal TO a TWO DIMENSIONAL COLLECTION of size n BY n WHERE each element IS False  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the element at position i of s EQUALS the element at position j of s AND (the DIFFERENCE BETWEEN j AND i LESS THAN OR EQUAL TO one OR the element at position i PLUS one AND position j MINUS one of is_pal IS True)  
                    SET the element at position i AND j of is_pal TO True  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM one TO n MINUS two  
            IF the element at position zero AND i MINUS one of is_pal IS True  
                FOR j FROM i TO n MINUS two  
                    IF the element at position i AND j of is_pal IS True AND the element at position j PLUS one AND n MINUS one of is_pal IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
