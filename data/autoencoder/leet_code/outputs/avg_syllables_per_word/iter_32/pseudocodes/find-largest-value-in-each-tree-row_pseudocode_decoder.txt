CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN no value  
    END IF  
    SET root TO new TreeNode instantiated with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the first element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS no value THEN  
            SET node.left TO a new TreeNode instantiated with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS no value THEN  
            SET node.right TO a new TreeNode instantiated with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value THEN  
        RETURN true  
    ELSE IF p EQUALS no value OR q EQUALS no value THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left attribute of p and the left attribute of q  
        AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root EQUALS no value THEN  
            RETURN empty collection  
        END IF  
        SET queue TO a double-ended queue initialized with root  
        SET largest_values TO empty collection  
        WHILE the queue is not empty DO  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each index FROM one TO level_size DO  
                SET node TO the first element removed from the front of queue  
                SET max_value TO the maximum of max_value AND the value attribute of node  
                IF the left attribute of node NOT EQUALS no value THEN  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node NOT EQUALS no value THEN  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
