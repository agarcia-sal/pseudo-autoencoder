CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF nums EQUALS an empty list OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        
        SET deq TO an empty double ended queue  
        SET result TO an empty list  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the element from the left end of deq  
            END IF  
            
            WHILE deq IS NOT empty AND the element at position element at position negative one of deq of nums IS LESS THAN the element at position i of nums  
                REMOVE the element from the right end of deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
