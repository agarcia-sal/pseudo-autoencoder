CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list of numbers from zero TO n MINUS one  
        SET self.rank TO a list containing zero repeated n times  
    END FUNCTION  

    FUNCTION find(self, x)  
        IF element at position x of self.parent NOT EQUALS x THEN  
            SET element at position x of self.parent TO the result of calling find with argument element at position x of self.parent  
        END IF  
        RETURN element at position x of self.parent  
    END FUNCTION  

    FUNCTION union(self, x, y)  
        SET rootX TO the result of calling find with argument x  
        SET rootY TO the result of calling find with argument y  
        IF rootX NOT EQUALS rootY THEN  
            IF element at position rootX of self.rank GREATER THAN element at position rootY of self.rank THEN  
                SET element at position rootY of self.parent TO rootX  
            ELSE IF element at position rootX of self.rank LESS THAN element at position rootY of self.rank THEN  
                SET element at position rootX of self.parent TO rootY  
            ELSE  
                SET element at position rootY of self.parent TO rootX  
                INCREMENT element at position rootX of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION latestDayToCross(self, row, col, cells)  
        SET grid TO a list containing row elements each being a list containing col elements all set TO one  
        SET uf TO a new UnionFind object initialized with the value equal TO the product of row AND col PLUS two  
        SET TOP TO the product of row AND col  
        SET BOTTOM TO the product of row AND col PLUS one  
        SET directions TO a list of tuples representing the pairs of values negative one zero AND one zero AND zero negative one AND zero one  
        
        FOR day FROM the length of cells MINUS one DOWN TO zero  
            SET r TO element at position day of cells  
            SET c TO element at position day of cells  
            SET r TO r MINUS one to convert to zero based index  
            SET c TO c MINUS one to convert to zero based index  
            SET element at position c of element at position r of grid TO zero  
            
            FOR each pair dr dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN row AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN col AND element at position nc of element at position nr of grid EQUALS zero THEN  
                    CALL uf.union WITH arguments the sum of r MULTIPLIED BY col PLUS c AND the sum of nr MULTIPLIED BY col PLUS nc  
                END IF  
            END FOR  
            
            IF r EQUALS zero THEN  
                CALL uf.union WITH arguments the sum of r MULTIPLIED BY col PLUS c AND TOP  
            END IF  
            
            IF r EQUALS row MINUS one THEN  
                CALL uf.union WITH arguments the sum of r MULTIPLIED BY col PLUS c AND BOTTOM  
            END IF  
            
            IF the result of calling uf.find with argument TOP EQUALS the result of calling uf.find with argument BOTTOM THEN  
                RETURN day  
            END IF  
        END FOR  
        
        RETURN zero  
    END FUNCTION  
END CLASS
