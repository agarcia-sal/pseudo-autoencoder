CLASS Solution  
    FUNCTION largestSumOfAverages(nums WITH TYPE List OF integer, k WITH TYPE integer) RETURNS float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeros with LENGTH equal to n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums  
        END FOR  

        FUNCTION average(i, j) RETURNS float  
            RETURN the difference between element at position j PLUS one OF prefix_sum AND element at position i OF prefix_sum DIVIDED BY the difference between j AND i PLUS one  
        END FUNCTION  

        SET dp TO a two dimensional list of zeros with n PLUS one rows and k PLUS one columns  

        FOR i FROM zero TO n MINUS one  
            SET element at position i AND column one OF dp TO average(i, n MINUS one)  
        END FOR  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    IF element at position i AND column partitions OF dp LESS THAN element at position j AND column partitions MINUS one OF dp PLUS average(i, j MINUS one)  
                        SET element at position i AND column partitions OF dp TO element at position j AND column partitions MINUS one OF dp PLUS average(i, j MINUS one)  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN element at position zero AND column k OF dp  
    END FUNCTION  
END CLASS
