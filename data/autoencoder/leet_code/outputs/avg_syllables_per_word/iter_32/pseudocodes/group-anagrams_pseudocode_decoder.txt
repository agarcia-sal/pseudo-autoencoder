CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a dictionary with default value as an empty list  
        FOR each string s IN strs  
            SET sorted_characters TO the characters of s arranged in ascending alphabetical order joined together as a single string  
            APPEND s TO the list associated with sorted_characters in anagrams  
        END FOR  
        SET result TO the list of all values contained in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
