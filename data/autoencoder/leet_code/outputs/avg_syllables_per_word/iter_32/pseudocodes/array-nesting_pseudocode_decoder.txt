CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of boolean false values with length equal to the LENGTH OF nums  
        SET max_length TO zero  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position i of visited EQUALS false THEN  
                SET current_length TO zero  
                SET k TO i  
                WHILE the element at position k of visited EQUALS false DO  
                    SET the element at position k of visited TO true  
                    SET k TO the element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO the GREATER value BETWEEN max_length AND current_length  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
