CLASS Solution  
    FUNCTION shortestDistance(grid)  
        IF grid EQUALS nothing OR element at position zero of grid EQUALS nothing  
            RETURN negative one  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET total_distance TO a two dimensional list with m rows and n columns each initialized to zero  
        SET reach_count TO a two dimensional list with m rows and n columns each initialized to zero  
        SET building_count TO zero  
        SET directions TO a list containing the pairs of one and zero, negative one and zero, zero and one, zero and negative one  
        
        FUNCTION bfs(start_x, start_y)  
            SET queue TO a double ended queue initialized with the pair of start_x and start_y  
            SET visited TO a two dimensional list with m rows and n columns each initialized to false  
            SET distance TO zero  
            SET reachable TO zero  
            
            WHILE the LENGTH OF queue GREATER THAN zero  
                SET size TO the LENGTH OF queue  
                INCREMENT distance BY one  
                FOR index FROM zero TO size MINUS one  
                    REMOVE the element from the left of queue and ASSIGN TO x and y respectively  
                    FOR each pair dx and dy IN directions  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of visited at position ny EQUALS false AND element at position nx of grid at position ny EQUALS zero  
                            SET element at position nx of visited at position ny TO true  
                            APPEND the pair of nx and ny TO queue  
                            INCREMENT element at position nx of total_distance at position ny BY distance  
                            INCREMENT element at position nx of reach_count at position ny BY one  
                            INCREMENT reachable BY one  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
            RETURN reachable  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    INCREMENT building_count BY one  
                    IF bfs(i, j) LESS THAN building_count MINUS one  
                        RETURN negative one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET min_distance TO infinity  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid at position j EQUALS zero AND element at position i of reach_count at position j EQUALS building_count  
                    IF element at position i of total_distance at position j LESS THAN min_distance  
                        SET min_distance TO element at position i of total_distance at position j  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_distance EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN min_distance  
        END IF  
    END FUNCTION  
END CLASS
