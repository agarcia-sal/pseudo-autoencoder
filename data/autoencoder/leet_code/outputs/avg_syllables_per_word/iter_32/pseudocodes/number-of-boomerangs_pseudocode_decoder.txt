CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO zero  
        FOR each element p1 IN points  
            SET cnt TO a new empty mapping from numbers to counts  
            FOR each element p2 IN points  
                SET dx TO element at position zero of p1 MINUS element at position zero of p2  
                SET dy TO element at position one of p1 MINUS element at position one of p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the value corresponding to d in cnt with default zero  
                INCREMENT the value corresponding to d in cnt BY one  
            END FOR  
        END FOR  
        RETURN ans MULTIPLIED BY two  
    END FUNCTION  
END CLASS
