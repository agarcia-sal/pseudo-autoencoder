CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        SET dp TO a list of n elements each SET TO negative one  
        SET element at position zero of dp TO zero  
        SET graph TO a dictionary with keys from zero TO n MINUS one each MAPPING TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i and position j of flights EQUALS one  
                    APPEND j TO the list corresponding to key i in graph  
                END IF  
            END FOR  
        END FOR  
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each SET TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one  
                    CONTINUE to the next iteration  
                END IF  
                IF the element at position i of new_dp LESS THAN the element at position i of dp PLUS the element at position i and position week of days  
                    SET the element at position i of new_dp TO the element at position i of dp PLUS the element at position i and position week of days  
                END IF  
                FOR each j IN the list corresponding to key i in graph  
                    IF the element at position j of new_dp LESS THAN the element at position i of dp PLUS the element at position j and position week of days  
                        SET the element at position j of new_dp TO the element at position i of dp PLUS the element at position j and position week of days  
                    END IF  
                END FOR  
            END FOR  
            ASSIGN reference TO new_dp in dp  
        END FOR  
        SET max_value TO negative infinity  
        FOR each value IN dp  
            IF value GREATER THAN max_value  
                SET max_value TO value  
            END IF  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
