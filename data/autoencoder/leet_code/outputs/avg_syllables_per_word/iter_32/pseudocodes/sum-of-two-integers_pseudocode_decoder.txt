CLASS Solution  
    FUNCTION getSum(a b)  
        SET maximum TO the largest thirty two bit integer  
        SET mask TO the thirty two bit mask  
        WHILE b NOT EQUALS zero  
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a AND b AND mask  
            SET b TO the bitwise AND of carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO maximum  
            RETURN a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
