CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the last of values  
        SET node TO new ListNode with value val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the value element of p1 EQUALS the value element of p2 AND the result of is_same_list(the next element of p1 the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList(self head)  
        IF head EQUALS None OR the next element of head EQUALS None THEN  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUALS None AND the next element of fast NOT EQUALS None  
            SET slow TO the next element of slow  
            SET fast TO the next element of the next element of fast  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr NOT EQUALS None  
            SET next_temp TO the next element of curr  
            SET the next element of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the next element of second NOT EQUALS None  
            SET temp1 TO the next element of first  
            SET temp2 TO the next element of second  
            SET the next element of first TO second  
            SET the next element of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
