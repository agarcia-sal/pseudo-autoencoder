CLASS Solution  
    FUNCTION kInversePairs(n INTEGER, k INTEGER) RETURNS INTEGER  
        SET MODULO TO one billion PLUS seven  
        INITIALIZE two dimensional list dp WITH zero VALUES HAVING dimensions n PLUS one BY k PLUS one  
        SET element at position zero of list at position zero of dp TO one  

        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position j of list at position i MINUS one of dp  
                SET prefix_sum TO prefix_sum MODULO MODULO  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position j MINUS i of list at position i MINUS one of dp PLUS MODULO  
                    SET prefix_sum TO prefix_sum MODULO MODULO  
                END IF  
                SET element at position j of list at position i of dp TO prefix_sum  
            END FOR  
        END FOR  

        RETURN element at position k of list at position n of dp  
    END FUNCTION  
END CLASS
