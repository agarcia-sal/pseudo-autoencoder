CLASS Solution  
    FUNCTION wordsTyping(sentence rows cols)  
        SET s TO the concatenation of all elements in sentence separated by a single space followed by a single space character  
        SET total_length TO the number of characters in s  
        SET position TO zero  
        FOR each iteration IN the range from one TO rows  
            INCREMENT position BY cols  
            IF the character at position modulo total_length in s EQUALS a single space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character at the result of position MINUS one modulo total_length in s NOT EQUALS a single space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN the integer division of position by total_length  
    END FUNCTION  
END CLASS
