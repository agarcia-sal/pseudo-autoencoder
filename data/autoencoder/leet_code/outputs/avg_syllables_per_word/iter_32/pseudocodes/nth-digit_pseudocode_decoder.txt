CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        
        SET start TO start PLUS the whole number result of n MINUS one DIVIDED BY length  
        
        SET digit_string TO the textual representation of start  
        SET digit_position TO the remainder of n MINUS one DIVIDED BY length  
        SET result_character TO the character at position digit_position IN digit_string  
        
        CONVERT result_character TO its numeric value  
        RETURN the numeric value  
    END FUNCTION  
END CLASS
