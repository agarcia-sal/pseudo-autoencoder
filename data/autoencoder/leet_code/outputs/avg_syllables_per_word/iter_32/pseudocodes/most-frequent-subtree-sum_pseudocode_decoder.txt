CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with val EQUALS element at position zero of values AND left EQUALS None AND right EQUALS None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode with val EQUALS element at position i of values AND left EQUALS None AND right EQUALS None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode with val EQUALS element at position i of values AND left EQUALS None AND right EQUALS None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None THEN  
                RETURN zero  
            END IF  
            SET current_sum TO the val attribute of node PLUS subtree_sum(the left attribute of node) PLUS subtree_sum(the right attribute of node)  
            INCREMENT count at current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO an empty default dictionary mapping to zero  
        CALL subtree_sum WITH root  
        
        SET max_freq TO the maximum value among the values in count OR zero if count is empty  
        
        SET result TO an empty list  
        FOR each key, freq IN the items of count  
            IF freq EQUALS max_freq THEN  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
