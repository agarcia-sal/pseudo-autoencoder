CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the start time of each meeting  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF the LENGTH OF min_heap IS GREATER THAN zero AND the start time of meeting GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the end time of meeting TO min_heap maintaining the heap property  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
