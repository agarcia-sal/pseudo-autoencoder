CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO an empty dictionary  
        FOR each index i FROM zero TO the LENGTH OF cost MINUS one  
            SET element c TO element at position i of cost  
            SET element d TO the string representation of index i PLUS one  
            ASSIGN in cost_to_digit the entry with key c TO d  
        END FOR  
        
        SET dp TO a list starting with an empty string followed by target number of elements each assigned to a string that represents negative one  
        
        FOR each integer t FROM one TO target  
            FOR each key c and value d IN cost_to_digit entries  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp NOT EQUALS to the string that represents negative one  
                    SET candidate TO the concatenation of d FOLLOWED BY element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS to the string that represents negative one OR the LENGTH OF candidate GREATER THAN the LENGTH OF element at position t of dp OR the LENGTH OF candidate EQUALS the LENGTH OF element at position t of dp AND candidate GREATER THAN element at position t of dp  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp NOT EQUALS to the string that represents negative one  
            RETURN element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
