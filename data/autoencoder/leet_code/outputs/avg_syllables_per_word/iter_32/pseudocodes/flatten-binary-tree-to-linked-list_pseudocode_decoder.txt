CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with the element at position one of values AS val AND no left child AND no right child  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE element from the front of queue AND ASSIGN reference TO node  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with the element at position i of values AS val AND no left child AND no right child  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with the element at position i of values AS val AND no left child AND no right child  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree the left attribute of p AND the left attribute of q AND is_same_tree the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS None  
            RETURN  
        END IF  
        SET stack TO a new list containing root as the first element  
        WHILE stack is not empty  
            REMOVE the last element from stack AND ASSIGN reference TO node  
            IF the right attribute of node NOT EQUALS None  
                APPEND the right attribute of node TO stack  
            END IF  
            IF the left attribute of node NOT EQUALS None  
                APPEND the left attribute of node TO stack  
            END IF  
            SET the left attribute of node TO None  
            IF stack is not empty  
                SET the right attribute of node TO the last element in stack  
            ELSE  
                SET the right attribute of node TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
