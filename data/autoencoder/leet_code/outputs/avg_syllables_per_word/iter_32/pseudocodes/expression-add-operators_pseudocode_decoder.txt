CLASS Solution  
    FUNCTION addOperators(num parameter num OF TYPE string, target parameter target OF TYPE integer) RETURNS list of strings  
        FUNCTION dfs(index parameter index, path parameter path, value parameter value, prev parameter prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
              
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO the substring of num from position index TO position i MINUS one  
                IF the LENGTH OF operand GREATER THAN one AND the element at position one OF operand EQUALS the character zero  
                    CONTINUE  
                END IF  
                  
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, the integer conversion of operand, the integer conversion of operand  
                ELSE  
                    CALL dfs WITH arguments i, path concatenated with the character plus concatenated with operand, value PLUS the integer conversion of operand, the integer conversion of operand  
                    CALL dfs WITH arguments i, path concatenated with the character minus concatenated with operand, value MINUS the integer conversion of operand, the negative of the integer conversion of operand  
                    CALL dfs WITH arguments i, path concatenated with the character multiplied by concatenated with operand, value MINUS prev PLUS prev MULTIPLIED BY the integer conversion of operand, prev MULTIPLIED BY the integer conversion of operand  
                END IF  
            END FOR  
        END FUNCTION  
          
        SET result TO empty list  
        CALL dfs WITH arguments zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
