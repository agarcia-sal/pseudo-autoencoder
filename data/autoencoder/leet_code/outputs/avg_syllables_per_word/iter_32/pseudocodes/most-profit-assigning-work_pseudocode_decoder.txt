CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO an empty list  
        FOR index FROM zero TO the LENGTH OF difficulty MINUS one  
            APPEND a tuple of element at position index of difficulty AND element at position index of profit TO jobs  
        END FOR  
        SORT jobs BY the first element of each tuple in ascending order  
        SORT worker IN ascending order  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the first element of element at position job_index of jobs LESS THAN OR EQUAL TO ability  
                SET best_profit TO the GREATER THAN of best_profit AND the second element of element at position job_index of jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
