CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER corridor  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET seats TO empty list  
        FOR index FROM zero TO the length of corridor MINUS one  
            IF the element at position index of corridor EQUALS seat character  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the length of seats MODULO two NOT EQUALS zero OR the length of seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the length of seats MINUS one STEP two  
            SET seat_distance TO the element at position index of seats MINUS the element at position index MINUS two of seats  
            SET ways TO ways MULTIPLIED BY seat_distance MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
