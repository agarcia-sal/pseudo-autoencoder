CLASS Solution  
    FUNCTION numBusesToDestination(routes parameter, source parameter, target parameter)  
        IF source parameter EQUALS target parameter  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a new mapping that assigns to each key an empty list  
        FOR each bus index and route in routes parameter with index as bus  
            FOR each stop in route  
                APPEND bus TO the list associated with stop in stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO an empty double-ended queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus in the list associated with source parameter in stop_to_buses  
            APPEND the pair consisting of bus and one to bus_queue  
            ADD bus TO visited_buses  
            ADD source parameter TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the first element of bus_queue and assign to current_bus and bus_count  

            FOR each stop in the list at position current_bus in routes parameter  
                IF stop EQUALS target parameter  
                    RETURN bus_count  
                END IF  

                IF stop is not in visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus in the list associated with stop in stop_to_buses  
                        IF next_bus is not in visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND the pair consisting of next_bus and bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
