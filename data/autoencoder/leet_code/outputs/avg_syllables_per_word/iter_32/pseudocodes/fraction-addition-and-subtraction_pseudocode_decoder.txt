CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET modified_expression TO expression REPLACED occurrences of the character MINUS WITH the character PLUS FOLLOWED BY the character MINUS  
        SET fractions TO the result of SPLITTING modified_expression BY the character PLUS  
        SET filtered_fractions TO an empty list  
        FOR each fraction_string IN fractions  
            IF fraction_string NOT EQUALS an empty string  
                APPEND fraction_string TO filtered_fractions  
            END IF  
        END FOR  
        SET sum_result TO zero  
        FOR each fraction_string IN filtered_fractions  
            CONVERT fraction_string TO a fraction object  
            INCREMENT sum_result BY the fraction object  
        END FOR  
        SET numerator TO the numerator of sum_result  
        SET denominator TO the denominator of sum_result  
        RETURN a string formed by the numerator FOLLOWED BY the character FORWARD SLASH FOLLOWED BY the denominator  
    END FUNCTION  
END CLASS
