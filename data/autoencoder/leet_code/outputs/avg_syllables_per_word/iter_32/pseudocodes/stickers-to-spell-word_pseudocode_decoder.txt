CLASS Solution  
    FUNCTION minStickers WITH PARAMETERS stickers AND target  
        SET stickers TO a new list created by transforming each element sticker of the original stickers list into a collection of character frequencies  
        SORT stickers so that the element with the highest frequency count of any character comes first  
        SET queue TO a new double ended queue containing a single element composed of the target string and zero stickers used  
        SET memo TO a new mapping associating the target string with zero stickers used  
        
        WHILE the queue is not empty  
            REMOVE the element from the front of the queue and DESTRUCTURE it into current_target AND used_stickers  
            
            FOR each sticker IN stickers  
                IF the character at the first position of current_target is NOT present as a key in sticker  
                    CONTINUE to the next iteration of the loop  
                END IF  
                
                SET remaining TO a new collection of character frequencies created from current_target  
                SUBTRACT from remaining the character frequency counts in sticker  
                SET new_target TO the string formed by sorting and concatenating all characters corresponding to the elements in remaining  
                
                IF new_target is an empty string  
                    RETURN used_stickers PLUS one  
                END IF  
                
                IF new_target is NOT a key in memo  
                    ASSIGN to memo at key new_target the value used_stickers PLUS one  
                    APPEND to queue the tuple containing new_target and used_stickers PLUS one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
