CLASS Solution  
    FUNCTION leastOpsExpressTarget(x parameter, target parameter)  
        FUNCTION dp(target parameter)  
            IF target parameter EQUALS zero  
                RETURN negative one  
            END IF  
            IF target parameter LESS THAN x parameter  
                RETURN the minimum between the quantity two MULTIPLIED BY target parameter MINUS one and the quantity two MULTIPLIED BY x parameter MINUS target parameter  
            END IF  
            SET power TO one  
            WHILE x parameter RAISED TO THE power LESS THAN OR EQUAL TO target parameter HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp(target parameter MINUS x parameter RAISED TO THE power) PLUS power  
            IF x parameter RAISED TO THE quantity power PLUS one MINUS target parameter LESS THAN target parameter  
                SET res TO the minimum between res and dp(x parameter RAISED TO THE quantity power PLUS one MINUS target parameter) PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp(target parameter)  
    END FUNCTION  
END CLASS
