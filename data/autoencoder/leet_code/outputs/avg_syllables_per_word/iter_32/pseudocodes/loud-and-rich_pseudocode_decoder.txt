CLASS Solution  
    FUNCTION loudAndRich(richer_parameter, quiet_parameter)  
        SET graph TO a new mapping with default empty list  
        FOR each pair a and b IN richer_parameter  
            APPEND a TO the list referenced by element at position b of graph  
        END FOR  
        
        SET n TO the LENGTH OF quiet_parameter  
        SET answer TO a list containing the value negative one repeated n times  
        
        FUNCTION dfs(person)  
            IF element at position person of answer EQUALS negative one THEN  
                SET element at position person of answer TO person  
                FOR each neighbor IN element at position person of graph  
                    SET candidate TO dfs(neighbor)  
                    IF element at position candidate of quiet_parameter LESS THAN element at position element at position person of answer of quiet_parameter THEN  
                        SET element at position person of answer TO candidate  
                    END IF  
                END FOR  
            END IF  
            RETURN element at position person of answer  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            CALL dfs WITH i  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
