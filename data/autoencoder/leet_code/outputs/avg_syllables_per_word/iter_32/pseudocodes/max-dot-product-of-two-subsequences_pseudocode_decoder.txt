CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        INITIALIZE dp AS a TWO DIMENSIONAL list with m PLUS one ROWS and n PLUS one COLUMNS filled with negative infinity value  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET dp at position i and j TO the maximum value AMONG  
                    current_product  
                    dp at position i MINUS one and j  
                    dp at position i and j MINUS one  
                    dp at position i MINUS one and j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
