CLASS Solution  
    FUNCTION letterCombinations(digits) RETURNS list of strings  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping from digit character two TO string of letters a b c AND digit character three TO string of letters d e f AND digit character four TO string of letters g h i AND digit character five TO string of letters j k l AND digit character six TO string of letters m n o AND digit character seven TO string of letters p q r s AND digit character eight TO string of letters t u v AND digit character nine TO string of letters w x y z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF length of next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN string mapped by element at position zero of next_digits in phone_map  
                    CALL backtrack with combination concatenated with letter AND substring of next_digits from position one TO end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack with empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
