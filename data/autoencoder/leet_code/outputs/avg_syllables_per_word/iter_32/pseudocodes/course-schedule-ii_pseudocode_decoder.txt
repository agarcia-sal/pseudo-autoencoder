CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty map from course TO empty list  
        FOR course FROM zero TO numCourses MINUS one  
            SET the list at key course in adj_list TO empty list  
        END FOR  
        
        SET in_degrees TO a list of zero repeated numCourses times  
        
        FOR each pair consisting of dest AND src IN prerequisites  
            APPEND dest TO the list at key src in adj_list  
            INCREMENT the element at position dest in in_degrees BY one  
        END FOR  
        
        SET queue TO an empty deque  
        FOR course FROM zero TO numCourses MINUS one  
            IF the element at position course in in_degrees EQUALS zero THEN  
                APPEND course TO queue  
            END IF  
        END FOR  
        
        SET topological_order TO empty list  
        
        WHILE queue is not empty  
            SET current TO the element removed from the front of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor in in_degrees BY one  
                IF the element at position neighbor in in_degrees EQUALS zero THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses THEN  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
