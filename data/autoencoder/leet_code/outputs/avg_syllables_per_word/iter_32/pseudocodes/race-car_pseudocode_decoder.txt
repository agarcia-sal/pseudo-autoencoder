CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO an empty double ended queue initialized with the tuple of zero position one speed and zero steps  
        SET visited TO a set initialized with the tuple of zero position and one speed  
        
        WHILE the queue is not empty  
            SET position speed steps TO the first element removed from the left end of the queue  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed is not an element of visited AND zero LESS THAN OR EQUAL TO next_position AND next_position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the tuple of next_position next_speed and steps PLUS one TO the right end of queue  
            END IF  
            
            IF speed GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed is not an element of visited AND zero LESS THAN OR EQUAL TO position AND position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the tuple of position next_speed and steps PLUS one TO the right end of queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
