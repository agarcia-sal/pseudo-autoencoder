CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN a list containing the empty string  
            END IF  
            IF n EQUALS one  
                RETURN a list containing the strings zero one and eight  
            END IF  
            SET middles TO the result of calling generate_strobogrammatic with n MINUS two and final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND the string formed by concatenating zero middle and zero TO result  
                END IF  
                APPEND the string formed by concatenating one middle and one TO result  
                APPEND the string formed by concatenating six middle and nine TO result  
                APPEND the string formed by concatenating eight middle and eight TO result  
                APPEND the string formed by concatenating nine middle and six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM the length of low TO the length of high  
                FOR each num IN the result of calling generate_strobogrammatic with length and length  
                    IF (length EQUALS the length of low AND num LESS THAN low) OR (length EQUALS the length of high AND num GREATER THAN high)  
                        CONTINUE to the next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN the result of calling count_strobogrammatic with low and high  
    END FUNCTION  
END CLASS
