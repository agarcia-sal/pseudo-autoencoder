CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO a list of zeros with the LENGTH OF sideLength MULTIPLIED BY sideLength  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET position TO the REMAINDER OF i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the REMAINDER OF j DIVIDED BY sideLength  
                INCREMENT the element at position position OF freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET result TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT result BY the element at position index OF freq  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
