CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        
        SET num_counts TO an empty collection mapping numbers to counts  
        FOR each num IN nums  
            IF element at key num EXISTS IN num_counts  
                INCREMENT element at key num of num_counts BY one  
            ELSE  
                SET element at key num of num_counts TO one  
            END IF  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN the keys of num_counts  
            IF k EQUALS zero  
                IF element at key num of num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF element at key which is num PLUS k EXISTS IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
