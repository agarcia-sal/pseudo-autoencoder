CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the LIST OF PAIRS OF the FIRST element of each interval AND the CORRESPONDING index IN intervals  
        SORT start_times ACCORDING TO the FIRST element of each pair IN ASCENDING ORDER  
        
        SET result TO an EMPTY LIST  
        FOR each element underscore comma end IN intervals  
            SET idx TO the LEAST index AT WHICH a PAIR with the FIRST element GREATER THAN OR EQUAL TO end CAN BE INSERTED IN start_times  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the SECOND element of the PAIR AT position idx OF start_times TO result  
            ELSE  
                APPEND the NEGATIVE ONE TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
