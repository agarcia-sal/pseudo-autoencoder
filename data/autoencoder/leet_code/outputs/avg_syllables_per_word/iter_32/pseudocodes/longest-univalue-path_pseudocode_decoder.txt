CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE length of queue greater than zero  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS none  
                RETURN zero, zero  
            END IF  
            SET left_length, left_value TO dfs(left child of node)  
            SET right_length, right_value TO dfs(right child of node)  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF left child of node NOT EQUALS none AND value of left child of node EQUALS value of node  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF right child of node NOT EQUALS none AND value of right child of node EQUALS value of node  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO maximum of self.max_length AND sum of left_arrow AND right_arrow  
            RETURN maximum of left_arrow AND right_arrow, value of node  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with root as argument  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
