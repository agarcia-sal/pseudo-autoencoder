CLASS Solution  
    FUNCTION waysToDistribute(n k)  
        SET MOD TO ten RAISED TO THE power OF nine PLUS seven  
        SET dp TO list OF lists WHERE each inner list CONTAINS zero REPEATED k PLUS one TIMES AND outer list CONTAINS such lists REPEATED n PLUS one TIMES  
        SET element at position zero of element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of element at position i of dp TO the remainder WHEN the sum OF the product OF j AND element at position j of element at position i MINUS one of dp AND element at position j MINUS one of element at position i MINUS one of dp IS DIVIDED BY MOD  
            END FOR  
        END FOR  
        RETURN element at position k of element at position n of dp  
    END FUNCTION  
END CLASS
