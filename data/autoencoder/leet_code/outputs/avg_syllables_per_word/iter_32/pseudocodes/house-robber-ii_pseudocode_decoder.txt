CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty collection  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position one of houses  
            END IF  
            SET dp TO new list of zeros with the LENGTH OF houses  
            SET element at position one of dp TO element at position one of houses  
            SET element at position two of dp TO the GREATER OF element at position one of houses AND element at position two of houses  
            FOR i FROM three TO the LENGTH OF houses  
                SET element at position i of dp TO the GREATER OF element at position i MINUS one of dp AND element at position i MINUS two of dp PLUS element at position i of houses  
            END FOR  
            RETURN element at the last position of dp  
        END FUNCTION  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        RETURN the GREATER OF rob_linear(subcollection of nums from position one TO the position before the last) AND rob_linear(subcollection of nums from position two TO the last position)  
    END FUNCTION  
END CLASS
