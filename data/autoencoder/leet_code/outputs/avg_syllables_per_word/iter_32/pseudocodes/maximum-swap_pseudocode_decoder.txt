CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of characters representing the string form of num  
        SET last TO a dictionary mapping each digit converted to integer TO its last position index in digits  
        
        FOR each index i AND digit d IN digits  
            FOR each digit_to_swap FROM nine DOWN TO the digit d converted to integer PLUS one  
                IF the value at key digit_to_swap in last IS GREATER THAN i  
                    SET temporary TO element at position i of digits  
                    SET element at position i of digits TO element at position value at key digit_to_swap in last of digits  
                    SET element at position value at key digit_to_swap in last of digits TO temporary  
                    RETURN the integer value of the concatenation in order of elements in digits  
                END IF  
            END FOR  
        END FOR  
        
        RETURN num  
    END FUNCTION  
END CLASS
