CLASS Solution  
    FUNCTION visiblePoints(points LIST OF LIST OF INTEGER angle INTEGER location LIST OF INTEGER) RETURNS INTEGER  
        SET angles TO an empty list  
        SET extra_points TO zero  
        FOR each pair x and y IN points  
            IF x EQUALS element at position zero of location AND y EQUALS element at position one of location  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_rad TO the result of the arc tangent of the difference of y and element at position one of location AND the difference of x and element at position zero of location  
            SET angle_deg TO the conversion of angle_rad from radians to degrees  
            APPEND angle_deg TO angles  
        END FOR  
        SORT angles in ascending order  
        SET angles TO concatenation of angles AND the list consisting of each element a in angles PLUS three hundred sixty  
        SET max_visible TO zero  
        SET left TO zero  
        FOR each right FROM zero TO the length of angles MINUS one  
            WHILE element at position right of angles MINUS element at position left of angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            IF the difference of right and left PLUS one GREATER THAN max_visible  
                SET max_visible TO the difference of right and left PLUS one  
            END IF  
        END FOR  
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
