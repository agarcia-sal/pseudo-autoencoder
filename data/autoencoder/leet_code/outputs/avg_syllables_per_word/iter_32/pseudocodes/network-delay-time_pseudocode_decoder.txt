CLASS Solution  
    FUNCTION networkDelayTime(times parameter, n parameter, k parameter)  
        SET graph TO an empty mapping from each integer from one TO n TO an empty list  
        FOR each element u v w IN times parameter  
            APPEND the pair of v and w TO the list at position u of graph  
        END FOR  
        
        SET dist TO a mapping from each integer from one TO n TO infinity  
        SET the value at position k parameter of dist TO zero  
        
        SET min_heap TO a list containing the pair of zero and k parameter  
        
        WHILE min_heap holds elements  
            EXTRACT and REMOVE the pair current_dist and node FROM min_heap according to minimum order  
            
            IF current_dist is GREATER THAN the value at position node of dist  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            FOR each element neighbor and weight IN the list at position node of graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance is LESS THAN the value at position neighbor of dist  
                    SET the value at position neighbor of dist TO distance  
                    ADD the pair of distance and neighbor TO min_heap maintaining the minimum order property  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
