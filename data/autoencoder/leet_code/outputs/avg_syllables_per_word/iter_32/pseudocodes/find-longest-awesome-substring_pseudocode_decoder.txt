CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping of bitmask zero TO position minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND each character char IN s  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR bit shifted one LEFT BY digit times  
            
            IF prefix_xor EXISTS AS a key IN prefix_map  
                SET max_length TO the greater value between max_length AND i MINUS the element at position prefix_xor IN prefix_map  
            END IF  
            
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR bit shifted one LEFT BY j times  
                IF flipped EXISTS AS a key IN prefix_map  
                    SET max_length TO the greater value between max_length AND i MINUS the element at position flipped IN prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST AS a key IN prefix_map  
                SET the element at position prefix_xor IN prefix_map TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
