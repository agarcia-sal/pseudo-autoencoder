CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new default dictionary with default integer value  
        FOR each cpdomain IN cpdomains  
            SET count_and_domain TO the result of splitting cpdomain by space  
            SET count TO the integer conversion of element at position zero of count_and_domain  
            SET domain TO element at position one of count_and_domain  
            SET subdomains TO the result of splitting domain by the period character  
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain_elements TO elements starting from position index TO the last position of subdomains  
                SET subdomain TO the concatenation of subdomain_elements separated by the period character  
                INCREMENT the value associated with subdomain in domain_count BY count  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each pair of domain and count IN the items of domain_count  
            SET formatted_string TO the concatenation of the string representation of count the space character and domain  
            APPEND formatted_string TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
