CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s NOT EQUALS the element at position j of s2  
                    YIELD the concatenation of: elements from start to position i MINUS one of s, element at position j of s, elements from position i PLUS one to position j MINUS one of s, element at position i of s, elements from position j PLUS one to end of s  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET queue TO a new deque containing the tuple of s1 and zero  
        SET visited TO a new set containing s1  
  
        WHILE queue is not empty  
            SET s and swaps TO the first tuple removed from queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    ADD the tuple of neighbor and swaps PLUS one TO the end of queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
