CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        CALL initialize_default_dictionary AND SET TO secret_count  
        CALL initialize_default_dictionary AND SET TO guess_count  
        FOR each pair of character s and character g IN corresponding positions of secret and guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT value at key s IN secret_count BY one  
                INCREMENT value at key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY the minimum of value at key digit IN secret_count AND value at key digit IN guess_count  
        END FOR  
        RETURN the concatenation of bulls, the letter A, cows, and the letter B  
    END FUNCTION  
END CLASS
