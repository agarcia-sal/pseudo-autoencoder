CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF val of p NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to left of p AND left of q AND is_same_tree applied to right of p AND right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder WITH PARAMETERS self, root  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO a new deque initialized with root  
        WHILE queue IS NOT empty  
            SET level_size TO length of queue  
            SET current_level TO empty list  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO remove first element FROM queue  
                APPEND val of node TO current_level  
                IF left of node IS NOT None  
                    APPEND left of node TO queue  
                END IF  
                IF right of node IS NOT None  
                    APPEND right of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
