CLASS Solution  
    FUNCTION findPaths(m, n, maxMove, startRow, startColumn)  
        SET MOD TO one billion PLUS seven  
        SET directions TO list containing tuples zero and one, zero and minus one, one and zero, minus one and zero  
        CALL initialize_dp_array WITH m, n, maxMove RETURNING dp  
        FOR k FROM one TO maxMove  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    FOR each direction IN directions  
                        SET di TO first element of direction  
                        SET dj TO second element of direction  
                        SET ni TO i PLUS di  
                        SET nj TO j PLUS dj  
                        IF ni LESS THAN zero OR ni GREATER THAN OR EQUAL TO m OR nj LESS THAN zero OR nj GREATER THAN OR EQUAL TO n  
                            SET dp at position i, j, k TO dp at position i, j, k PLUS one MODULO MOD  
                        ELSE  
                            SET dp at position i, j, k TO dp at position i, j, k PLUS dp at position ni, nj, k MINUS one MODULO MOD  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position startRow, startColumn, maxMove  
    END FUNCTION  

    FUNCTION initialize_dp_array(m, n, maxMove)  
        CREATE three dimensional list dp of size m times n times maxMove PLUS one  
        SET all elements of dp TO zero  
        RETURN dp  
    END FUNCTION  
END CLASS
