CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the remainder of num DIVIDED BY one hundred DIVIDED BY ten  
            SET value TO the remainder of num DIVIDED BY ten  
            SET the element at position depth and pos in tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value at position depth and pos in tree  
            IF current_value EQUALS None  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO depth PLUS one and pos MULTIPLIED BY two  
            
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO dfs(depth PLUS one, pos MULTIPLIED BY two MINUS one, new_path_sum)  
            SET right_sum TO dfs(depth PLUS one, pos MULTIPLIED BY two, new_path_sum)  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs(one, one, zero)  
    END FUNCTION  
END CLASS
