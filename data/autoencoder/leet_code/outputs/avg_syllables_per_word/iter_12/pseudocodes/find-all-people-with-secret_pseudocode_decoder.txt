CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO the set containing zero AND firstPerson  
        
        SORT meetings BY the element at position two of each meeting  
        
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position two of the meeting at position i of meetings  
            CALL initialize graph AS a new empty mapping from elements to lists  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position two of the meeting at position i of meetings EQUALS current_time  
                SET x TO the element at position zero of the meeting at position i of meetings  
                SET y TO the element at position one of the meeting at position i of meetings  
                APPEND y TO the list mapped from x in graph  
                APPEND x TO the list mapped from y in graph  
                INCREMENT i BY one  
            END WHILE  
            
            CALL initialize queue AS a new deque containing each person IN know_secret THAT is a key IN graph  
            CALL initialize visited AS a new empty set  
            WHILE queue is not empty  
                REMOVE first element FROM queue AND SET TO person  
                IF person IS IN visited  
                    CONTINUE to next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list mapped from person in graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        
        RETURN the list conversion OF know_secret  
    END FUNCTION  
END CLASS
