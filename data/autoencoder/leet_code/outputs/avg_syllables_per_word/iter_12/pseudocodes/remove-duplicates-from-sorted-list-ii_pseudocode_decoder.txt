CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val EQUALS element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one to the end  
        SET node TO new ListNode with val EQUALS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN (value of p1 EQUALS value of p2) AND is_same_list(next of p1, next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self, head)  
        SET dummy TO new ListNode with val EQUALS zero, next EQUALS head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF next of head IS NOT none AND value of head EQUALS value of next of head  
                WHILE next of head IS NOT none AND value of head EQUALS value of next of head  
                    SET head TO next of head  
                END WHILE  
                SET next of prev TO next of head  
            ELSE  
                SET prev TO next of prev  
            END IF  
            SET head TO next of head  
        END WHILE  
        RETURN next of dummy  
    END FUNCTION  
END CLASS
