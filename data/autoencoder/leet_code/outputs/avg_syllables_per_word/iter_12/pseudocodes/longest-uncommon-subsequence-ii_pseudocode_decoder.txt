CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO an iterator OVER t  
            FOR each character c IN s  
                IF c IS NOT FOUND IN it  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  

        SORT strs BY the NEGATIVE OF the LENGTH OF each element AND BY the element ITSELF lexicographically  

        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i IS NOT EQUAL TO j AND is_subsequence(word, other) IS true  
                    SET unique TO false  
                    BREAK out of the inner FOR loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  

        RETURN negative one  
    END FUNCTION  
END CLASS
