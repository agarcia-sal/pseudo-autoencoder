CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO seven FFFFFF F hexadecimal representation  
        SET mask TO FFFFFFFF hexadecimal representation  
        WHILE b NOT EQUALS zero  
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a AND b ANDed WITH mask  
            SET b TO carry ANDed WITH mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
