CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
          
        SET n TO the LENGTH OF nums  
        SET lengths TO a list of n elements each EQUAL TO one  
        SET counts TO a list of n elements each EQUAL TO one  
          
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums  
                    IF element at position j of lengths PLUS one GREATER THAN element at position i of lengths  
                        SET element at position i of lengths TO element at position j of lengths PLUS one  
                        SET element at position i of counts TO element at position j of counts  
                    ELSE IF element at position j of lengths PLUS one EQUALS element at position i of lengths  
                        INCREMENT element at position i of counts BY element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        SET longest TO the maximum element OF lengths  
        SET total TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of lengths EQUALS longest  
                INCREMENT total BY element at position i of counts  
            END IF  
        END FOR  
        RETURN total  
    END FUNCTION  
END CLASS
