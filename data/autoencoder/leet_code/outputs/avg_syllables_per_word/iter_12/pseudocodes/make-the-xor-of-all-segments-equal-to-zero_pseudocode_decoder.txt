CLASS Solution  
    FUNCTION minChanges(nums, k)  
        SET groups TO a new default dictionary mapping keys to empty lists  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO the list in groups at position index MODULO k  
        END FOR  
        
        SET freq TO a new list  
        FOR index FROM zero TO k MINUS one  
            APPEND a new default dictionary mapping keys to zero TO freq  
        END FOR  
        SET min_size TO a new list with k elements each initialized TO infinity  
        
        FOR index FROM zero TO k MINUS one  
            FOR each number IN the list groups at position index  
                INCREMENT value of freq at position index for key number BY one  
            END FOR  
            SET element at position index of min_size TO the LENGTH OF the list groups at position index  
        END FOR  
        
        SET dp TO a new list with length TWO RAISED TO THE POWER TEN each element initialized TO infinity  
        SET element at position zero of dp TO zero  
        
        FOR index FROM zero TO k MINUS one  
            SET new_dp TO a new list with length TWO RAISED TO THE POWER TEN each element initialized TO infinity  
            SET min_dp TO the MINIMUM VALUE in dp  
            FOR j FROM zero TO TWO RAISED TO THE POWER TEN MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position index of min_size  
                FOR each key-value pair of number and count IN the items of freq at position index  
                    SET candidate_value TO element at position j XOR number of dp PLUS element at position index of min_size MINUS count  
                    IF candidate_value LESS THAN element at position j of new_dp THEN  
                        SET element at position j of new_dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
