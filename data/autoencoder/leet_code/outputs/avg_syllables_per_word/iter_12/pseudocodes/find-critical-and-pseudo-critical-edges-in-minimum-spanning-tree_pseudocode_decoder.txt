CLASS UnionFind  
    FUNCTION __init__(n)  
        SET self.parent TO a list of integers from zero TO n MINUS one  
        SET self.rank TO a list containing the value one repeated n times  
    END FUNCTION  
  
    FUNCTION find(u)  
        IF the element at position u of self.parent NOT EQUALS u  
            SET the element at position u of self.parent TO the result of calling self.find on the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(u, v)  
        SET rootU TO the result of calling self.find on u  
        SET rootV TO the result of calling self.find on v  
  
        IF rootU NOT EQUALS rootV  
            IF the element at position rootU of self.rank GREATER THAN the element at position rootV of self.rank  
                SET the element at position rootV of self.parent TO rootU  
            ELSE IF the element at position rootU of self.rank LESS THAN the element at position rootV of self.rank  
                SET the element at position rootU of self.parent TO rootV  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(n, edges)  
        FOR each index i AND element edge IN the enumeration of edges  
            APPEND i TO edge  
        END FOR  
  
        SORT edges IN ascending order BASED ON the element at position two of each edge  
  
        FUNCTION mst(exclude = None, include = None)  
            SET uf TO a new instance of UnionFind passing n  
            SET weight TO zero  
  
            IF include IS NOT None  
                CALL uf.union with the element at position zero of include AND the element at position one of include  
                SET weight TO weight PLUS the element at position two of include  
            END IF  
  
            FOR each u, v, w, idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE to the next iteration  
                END IF  
                IF calling uf.union with u AND v RETURNS True  
                    SET weight TO weight PLUS w  
                END IF  
            END FOR  
  
            SET root TO the result of calling uf.find on zero  
            FOR index i FROM zero TO n MINUS one  
                IF the result of calling uf.find on i NOT EQUALS root  
                    RETURN infinity  
                END IF  
            END FOR  
  
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO the result of calling mst without arguments  
  
        SET critical TO an empty list  
        SET pseudo_critical TO an empty list  
  
        FOR each u, v, w, idx IN edges  
            IF the result of calling mst with exclude equal to idx GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of calling mst with include equal to the tuple of u, v, w EQUALS mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
  
        RETURN a list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
