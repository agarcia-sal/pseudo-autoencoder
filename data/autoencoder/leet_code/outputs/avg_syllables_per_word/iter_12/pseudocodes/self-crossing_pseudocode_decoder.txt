CLASS Solution  
    FUNCTION isSelfCrossing(distance)  
        SET n TO the LENGTH OF distance  
        IF n LESS THAN 4  
            RETURN False  
        END IF  
        
        FOR i FROM 3 TO n MINUS 1  
            IF element at position i of distance GREATER THAN OR EQUAL TO element at position i MINUS 2 of distance AND element at position i MINUS 1 of distance LESS THAN OR EQUAL TO element at position i MINUS 3 of distance  
                RETURN True  
            END IF  
            
            IF i GREATER THAN OR EQUAL TO 4 AND element at position i MINUS 1 of distance EQUALS element at position i MINUS 3 of distance AND element at position i PLUS element at position i MINUS 4 of distance GREATER THAN OR EQUAL TO element at position i MINUS 2 of distance  
                RETURN True  
            END IF  
            
            IF i GREATER THAN OR EQUAL TO 5 AND element at position i MINUS 2 of distance GREATER THAN OR EQUAL TO element at position i MINUS 4 of distance AND element at position i MINUS 3 of distance GREATER THAN OR EQUAL TO element at position i MINUS 1 of distance AND element at position i MINUS 1 of distance PLUS element at position i MINUS 5 of distance GREATER THAN OR EQUAL TO element at position i MINUS 3 of distance AND element at position i PLUS element at position i MINUS 4 of distance GREATER THAN OR EQUAL TO element at position i MINUS 2 of distance  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
