CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list of twenty-six elements each set TO negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each i, char IN the enumeration of s  
            SET index TO the character code of char MINUS the character code of letter a  
            SET current_appeal TO current_appeal PLUS i MINUS the element at position index of last  
            SET total_appeal TO total_appeal PLUS current_appeal  
            SET the element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
