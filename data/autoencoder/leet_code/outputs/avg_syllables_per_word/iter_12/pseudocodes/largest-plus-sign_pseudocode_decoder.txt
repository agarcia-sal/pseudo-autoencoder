CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            APPEND tuple of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional grid of size n by n where each element is a list of four zeros representing counts for right, down, left, and up directions  
        
        SET directions TO list of four tuples representing zero and one or minus one for right, down, left, and up respectively  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF tuple of i and j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET dp element at position i, j, zero TO dp element at position i, j MINUS one, zero PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET dp element at position i, j, one TO dp element at position i MINUS one, j, one PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR index i FROM n MINUS one DOWN TO zero  
            FOR index j FROM n MINUS one DOWN TO zero  
                IF tuple of i and j NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET dp element at position i, j, two TO dp element at position i, j PLUS one, two PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET dp element at position i, j, three TO dp element at position i PLUS one, j, three PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value in dp element at position i, j  
                    SET result TO the maximum of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
