CLASS Solution  
    FUNCTION hIndex(citations)  
        CALL citations SORT IN DESCENDING ORDER  
        FOR each position i FROM zero TO the LENGTH OF citations MINUS one  
            SET citation TO element at position i of citations  
            IF citation LESS THAN i PLUS one  
                RETURN i  
            END IF  
        END FOR  
        RETURN the LENGTH OF citations  
    END FUNCTION  
END CLASS
