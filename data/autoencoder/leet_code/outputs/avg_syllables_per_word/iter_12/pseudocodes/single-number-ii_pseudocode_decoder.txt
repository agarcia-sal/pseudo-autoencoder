CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO the bitwise exclusive OR of ones AND num, AND the bitwise complement of twos  
            SET twos TO the bitwise exclusive OR of twos AND num, AND the bitwise complement of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
