CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET d TO an empty dictionary  
        FOR index FROM zero TO the LENGTH OF req_skills MINUS one  
            SET skill TO element at position index of req_skills  
            SET the value at key skill in d TO index  
        END FOR  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        SET p TO a list of n elements each set TO zero  
        FOR i FROM zero TO n MINUS one  
            SET ss TO element at position i of people  
            FOR each skill s IN ss  
                SET p at position i TO the result of the value at position i of p BITWISE OR with one LEFT SHIFTED BY the value at key s in d  
            END FOR  
        END FOR  
        SET f TO a list of size two RAISED TO THE POWER OF m with each element set TO infinity  
        SET g TO a list of size two RAISED TO THE POWER OF m with each element set TO zero  
        SET h TO a list of size two RAISED TO THE POWER OF m with each element set TO zero  
        SET the element at position zero of f TO zero  
        FOR i FROM zero TO two RAISED TO THE POWER OF m MINUS one  
            IF the element at position i of f EQUALS infinity  
                CONTINUE to next iteration  
            END IF  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of f PLUS one LESS THAN the element at position of the bitwise OR of i and p at position j of f  
                    SET the element at position of the bitwise OR of i and p at position j of f TO the element at position i of f PLUS one  
                    SET the element at position of the bitwise OR of i and p at position j of g TO j  
                    SET the element at position of the bitwise OR of i and p at position j of h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two RAISED TO THE POWER OF m MINUS one  
        SET ans TO an empty list  
        WHILE i NOT EQUALS zero  
            APPEND the element at position i of g TO ans  
            SET i TO the element at position i of h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
