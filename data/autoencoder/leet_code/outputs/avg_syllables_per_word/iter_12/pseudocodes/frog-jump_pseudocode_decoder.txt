CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO CONVERT stones TO a SET  
        
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the ELEMENT at position (LENGTH OF stones MINUS one) OF stones  
                RETURN True  
            END IF  
            
            IF the PAIR of position AND jump IS IN memo  
                RETURN the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND position PLUS next_jump IS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the ELEMENT at position zero OF stones, zero)  
    END FUNCTION  
END CLASS
