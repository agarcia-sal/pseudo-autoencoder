CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a copy of t TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND the element at position u of nums TO t  
                CALL dfs WITH parameters u PLUS one, the element at position u of nums, t  
                REMOVE last element FROM t  
            END IF  
            IF the element at position u of nums NOT EQUALS last  
                CALL dfs WITH parameters u PLUS one, last, t  
            END IF  
        END FUNCTION  
        SET ans TO empty list  
        CALL dfs WITH parameters zero, negative one thousand, empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
