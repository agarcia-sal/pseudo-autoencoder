CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND the concatenation of elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF element at position i of t IS an alphabetic character  
                SET element at position i of t TO the character corresponding to the bitwise XOR of the ordinal value of element at position i of t WITH thirty two  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        
        SET t TO the list of characters from s  
        SET ans TO an empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
