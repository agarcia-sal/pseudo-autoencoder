CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value as element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.left TO new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None THEN  
            SET node.right TO new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(root, low, high)  
        IF root IS None THEN  
            RETURN None  
        END IF  
        IF value of root LESS THAN low THEN  
            RETURN trimBST(right child of root, low, high)  
        END IF  
        IF value of root GREATER THAN high THEN  
            RETURN trimBST(left child of root, low, high)  
        END IF  
        SET left child of root TO trimBST(left child of root, low, high)  
        SET right child of root TO trimBST(right child of root, low, high)  
        RETURN root  
    END FUNCTION  
END CLASS
