CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN value at position val of p1 EQUALS value at position val of p2 AND call to is_same_list with parameters value at position next of p1, value at position next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self, head, x)  
        SET less_head TO new ListNode with zero  
        SET greater_head TO new ListNode with zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT None  
            IF value at position val of current LESS THAN x  
                SET value at position next of less TO current  
                SET less TO value at position next of less  
            ELSE  
                SET value at position next of greater TO current  
                SET greater TO value at position next of greater  
            END IF  
            SET current TO value at position next of current  
        END WHILE  
        SET value at position next of less TO value at position next of greater_head  
        SET value at position next of greater TO None  
        RETURN value at position next of less_head  
    END FUNCTION  
END CLASS
