CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1, y1 TO the quotient and remainder of the result of the ordinal value of c1 MINUS the ordinal value of 'A' DIVIDED BY six  
            SET x2, y2 TO the quotient and remainder of the result of the ordinal value of c2 MINUS the ordinal value of 'A' DIVIDED BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  

        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO the result of calling distance with f1 and c PLUS the result of calling dp with i INCREMENTED BY one, c, and f2  
            SET d2 TO the result of calling distance with f2 and c PLUS the result of calling dp with i INCREMENTED BY one, f1, and c  
            RETURN the smaller of d1 and d2  
        END FUNCTION  

        RETURN the result of calling dp with zero, None, and None  
    END FUNCTION  
END CLASS
