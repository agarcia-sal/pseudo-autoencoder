CLASS Solution
    FUNCTION multiply(mat1, mat2)
        SET m TO the LENGTH OF mat1
        SET k TO the LENGTH OF element at position zero of mat1
        SET k TO the LENGTH OF mat2
        SET n TO the LENGTH OF element at position zero of mat2
        
        CALL initialize result WITH zero VALUES FOR m TIMES n TIMES
        
        FOR index i FROM zero TO m MINUS one
            FOR index j FROM zero TO n MINUS one
                FOR index p FROM zero TO k MINUS one
                    SET element at position i AND j of result TO element at position i AND p of mat1 MULTIPLIED BY element at position p AND j of mat2 PLUS element at position i AND j of result
                END FOR
            END FOR
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
