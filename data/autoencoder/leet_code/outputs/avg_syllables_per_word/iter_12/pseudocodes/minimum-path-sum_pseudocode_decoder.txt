CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  

        CALL create_2D_list_of_zeros WITH parameters m AND n AND SET result TO dp  

        SET element at position zero position zero of dp TO element at position zero position zero of grid  

        FOR j FROM one TO n MINUS one  
            SET element at position zero position j of dp TO element at position zero position j MINUS one of dp PLUS element at position zero position j of grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            SET element at position i position zero of dp TO element at position i MINUS one position zero of dp PLUS element at position i position zero of grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i MINUS one position j of dp LESS THAN element at position i position j MINUS one of dp  
                    SET element at position i position j of dp TO element at position i MINUS one position j of dp PLUS element at position i position j of grid  
                ELSE  
                    SET element at position i position j of dp TO element at position i position j MINUS one of dp PLUS element at position i position j of grid  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position m MINUS one position n MINUS one of dp  
    END FUNCTION  

    FUNCTION create_2D_list_of_zeros(rows, columns)  
        SET new_list TO an empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET new_row TO an empty list  
            FOR each index FROM zero TO columns MINUS one  
                APPEND zero TO new_row  
            END FOR  
            APPEND new_row TO new_list  
        END FOR  
        RETURN new_list  
    END FUNCTION  
END CLASS
