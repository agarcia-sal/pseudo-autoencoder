CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        
        SET lis TO a list of length n with all elements set TO one  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums  
                    SET value TO lis at position j PLUS one  
                    IF value GREATER THAN element at position i of lis  
                        SET element at position i of lis TO value  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET lds TO a list of length n with all elements set TO one  
        FOR index i FROM n MINUS one DOWN TO zero  
            FOR index j FROM i PLUS one TO n MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums  
                    SET value TO lds at position j PLUS one  
                    IF value GREATER THAN element at position i of lds  
                        SET element at position i of lds TO value  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET max_mountain_length TO zero  
        FOR index i FROM one TO n MINUS two  
            IF element at position i of lis GREATER THAN one AND element at position i of lds GREATER THAN one  
                SET combined_length TO element at position i of lis PLUS element at position i of lds MINUS one  
                IF combined_length GREATER THAN max_mountain_length  
                    SET max_mountain_length TO combined_length  
                END IF  
            END IF  
        END FOR  
        
        RETURN n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
