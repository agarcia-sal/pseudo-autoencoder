CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuple of each row in board  
        SET target TO tuple containing tuple of one two three and tuple of four five zero  
        
        SET neighbors TO dictionary where  
            key zero maps to set of one and three  
            key one maps to set of zero and two and four  
            key two maps to set of one and five  
            key three maps to set of zero and four  
            key four maps to set of one and three and five  
            key five maps to set of two and four  
        
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO deque containing tuple of start zero_pos zero  
        SET visited TO set containing start  
        
        WHILE queue IS NOT empty  
            SET state zero_pos moves TO the element popped from the left of queue  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN neighbors at key zero_pos  
                SET new_state TO list of list conversion of state  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                
                SWAP element at position j of element at position i of new_state WITH element at position nj of element at position ni of new_state  
                SET new_tuple_state TO tuple of tuple of each row in new_state  
                
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
