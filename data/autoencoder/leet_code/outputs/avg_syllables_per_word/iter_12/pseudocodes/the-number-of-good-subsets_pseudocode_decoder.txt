CLASS Solution
    FUNCTION numberOfGoodSubsets(nums)
        SET count TO a Counter of nums
        SET dp TO a list of zeroes with length two POWER ten
        SET element at position zero of dp TO one

        FOR each num IN keys of count
            IF num EQUALS one
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two POWER element at position one of count MODULO MOD
                CONTINUE to next iteration
            END IF

            IF num NOT IN prime_factors
                CONTINUE to next iteration
            END IF

            SET factors TO value at key num in prime_factors
            SET mask TO zero
            FOR each prime IN factors
                SET mask TO mask BITWISE OR two POWER (prime MINUS one)
            END FOR

            FOR m FROM two POWER ten MINUS one DOWN TO zero
                IF bitwise AND of m AND mask EQUALS zero
                    SET element at position bitwise OR of m AND mask of dp TO (element at position bitwise OR of m AND mask of dp PLUS element at position m of dp MULTIPLIED BY element at position num of count) MODULO MOD
                END IF
            END FOR
        END FOR

        SET result TO sum of elements from position one onward in dp MODULO MOD
        RETURN result
    END FUNCTION
END CLASS
