CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses IN ASCENDING ORDER BY the second element of each course  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each element IN courses  
            SET duration TO the first element of element  
            SET last_day TO the second element of element  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                PUSH the NEGATION of duration ONTO max_heap maintaining max_heap properties as a max-heap  
                INCREMENT current_end_day BY duration  
            ELSE  
                IF max_heap is NOT EMPTY AND the NEGATION of the first element of max_heap GREATER THAN duration  
                    INCREMENT current_end_day BY duration PLUS REMOVE the first element FROM max_heap  
                    PUSH the NEGATION of duration ONTO max_heap maintaining max_heap properties as a max-heap  
                END IF  
            END IF  
        END FOR  
        RETURN the LENGTH OF max_heap  
    END FUNCTION  
END CLASS
