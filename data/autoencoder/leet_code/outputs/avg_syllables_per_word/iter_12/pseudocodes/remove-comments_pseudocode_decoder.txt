CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the substring of line from i TO i PLUS two EQUALS the string "*/"  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the substring of line from i TO i PLUS two EQUALS the string "//"  
                        BREAK the WHILE loop  
                    ELSE IF the substring of line from i TO i PLUS two EQUALS the string "/*"  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the character at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line is not empty AND in_block_comment is False  
                APPEND the concatenation of the elements of current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
