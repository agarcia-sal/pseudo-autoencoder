CLASS Solution  
    FUNCTION maxA(n)  
        IF n IS LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  
        
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        
        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  
        
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET product TO element at position j of dp MULTIPLIED BY (i MINUS j MINUS one)  
                IF product IS GREATER THAN element at position i of dp  
                    SET element at position i of dp TO product  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
