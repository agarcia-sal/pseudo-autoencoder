CLASS Solution
    FUNCTION maximumSwap(num)
        SET digits TO list of the characters of the string representation of num
        SET last TO empty dictionary
        FOR each index i and digit d IN enumeration of digits
            SET integer_digit TO integer conversion of d
            SET element at position integer_digit of last TO i
        END FOR

        FOR each index i and digit d IN enumeration of digits
            FOR d2 FROM nine DOWN TO integer conversion of d PLUS one
                IF the element at position d2 of last EXISTS AND the element at position d2 of last IS GREATER THAN i
                    SET temporary TO element at position i of digits
                    SET element at position i of digits TO element at position of element at position d2 of last of digits
                    SET element at position of element at position d2 of last of digits TO temporary
                    SET result_string TO concatenation of all elements in digits
                    RETURN integer conversion of result_string
                END IF
            END FOR
        END FOR

        RETURN num
    END FUNCTION
END CLASS
