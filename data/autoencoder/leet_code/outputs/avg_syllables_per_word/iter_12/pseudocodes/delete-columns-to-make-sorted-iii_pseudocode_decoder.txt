CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of length n with all elements SET TO one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR each k FROM zero TO m MINUS one the element at position k and j of strs IS LESS THAN OR EQUAL TO the element at position k and i of strs  
                    SET dp at position i TO the MAXIMUM OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the MAXIMUM OF dp  
    END FUNCTION  
END CLASS
