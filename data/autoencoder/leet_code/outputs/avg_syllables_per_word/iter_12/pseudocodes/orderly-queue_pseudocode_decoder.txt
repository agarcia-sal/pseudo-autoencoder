CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN 1  
            RETURN the concatenation of characters in the string s sorted in ascending order  
        ELSE  
            SET n TO the length of the string s  
            SET smallest TO the string s  
            FOR index FROM one TO n MINUS one  
                SET rotated TO the substring of s from position index TO end CONCATENATED WITH the substring of s from position zero TO index MINUS one  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
