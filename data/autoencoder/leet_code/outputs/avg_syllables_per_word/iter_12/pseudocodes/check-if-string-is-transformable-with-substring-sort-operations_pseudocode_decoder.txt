CLASS Solution  
    FUNCTION isTransformable(s, t)  
        CALL initialize a mapping from digits to lists of positions named pos  
        FOR each index i AND character c IN the enumeration of s  
            CALL add index i TO the digit obtained by converting character c to number in pos  
        END FOR  
        
        FOR each index i AND character c IN the enumeration of t  
            SET digit TO the number converted from character c  
            IF the list at pos for digit is empty  
                RETURN false  
            END IF  
            FOR each smaller FROM zero TO digit MINUS one  
                IF the list at pos for smaller is not empty AND the first element at pos for smaller IS LESS THAN the first element at pos for digit  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest element FROM the list at pos for digit  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
