CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO call defaultdict WITH argument integer  
        
        FOR each cpdomain IN cpdomains  
            SET count_and_domain TO call split ON cpdomain WITH argument space  
            SET count TO convert element at position zero of count_and_domain TO integer  
            SET domain TO element at position one of count_and_domain  
            SET subdomains TO call split ON domain WITH argument period  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain_parts TO elements from position index TO end OF subdomains  
                SET subdomain TO join subdomain_parts with period as separator  
                INCREMENT value at key subdomain IN domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each domain, count IN items of domain_count  
            SET formatted_string TO string formed by concatenating count, space, and domain  
            APPEND formatted_string TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
