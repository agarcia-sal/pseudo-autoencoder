CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each m IN nums  
            IF m EQUALS m1  
                INCREMENT n1 BY one  
            ELSE IF m EQUALS m2  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUALS zero  
                SET m1 TO m  
                SET n1 TO one  
            ELSE IF n2 EQUALS zero  
                SET m2 TO m  
                SET n2 TO one  
            ELSE  
                DECREMENT n1 BY one  
                DECREMENT n2 BY one  
            END IF  
        END FOR  
        RETURN list of m FROM list [m1, m2] WHERE count of m IN nums GREATER THAN integer division of length of nums BY three  
    END FUNCTION  
END CLASS
