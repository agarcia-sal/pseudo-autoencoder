CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of tuples zero one, one zero, zero negative one, negative one zero  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO a set containing each tuple converted from elements of obstacles  
        SET max_distance TO zero  
        
        FOR each command IN commands  
            IF command EQUALS negative two  
                SET dir_index TO dir_index MINUS one PLUS four MODULO four  
            ELSE IF command EQUALS negative one  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO element at position zero of element at position dir_index of directions  
                SET dy TO element at position one of element at position dir_index of directions  
                FOR index FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF tuple of next_x and next_y IN obstacles_set  
                        BREAK  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET max_distance TO the greater of max_distance and x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
