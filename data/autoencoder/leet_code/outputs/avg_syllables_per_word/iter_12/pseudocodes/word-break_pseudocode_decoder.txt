CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO CONVERT wordDict TO set  
        SET dp TO LIST OF False VALUES OF LENGTH THE LENGTH OF s PLUS one  
        SET element at position zero of dp TO True  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of dp EQUALS True AND the SUBSTRING of s FROM position j TO position i IS IN wordSet  
                    SET element at position i of dp TO True  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
