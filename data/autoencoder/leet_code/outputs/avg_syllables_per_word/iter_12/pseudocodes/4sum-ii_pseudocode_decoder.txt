CLASS Solution  
    FUNCTION fourSumCount(nums1, nums2, nums3, nums4)  
        SET sum_ab TO empty dictionary  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum IS IN sum_ab  
                    INCREMENT sum_ab at current_sum BY one  
                ELSE  
                    SET sum_ab at current_sum TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO NEGATIVE of c PLUS d  
                IF target IS IN sum_ab  
                    INCREMENT count BY value of sum_ab at target  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
