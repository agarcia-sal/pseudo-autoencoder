CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION OF THE RIGHTMOST VALUE LESS THAN OR EQUAL TO the VALUE OF p DIVIDED BY x IN nums2  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION OF THE LEFTMOST VALUE GREATER THAN OR EQUAL TO the VALUE OF p DIVIDED BY x IN nums2  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the INDICATOR THAT p GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  
      
        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION OF THE LEFTMOST VALUE GREATER THAN OR EQUAL TO k IN the RANGE FROM NEGATIVE mx TO mx PLUS one USING THE count FUNCTION AS THE KEY MINUS mx  
    END FUNCTION  
END CLASS
