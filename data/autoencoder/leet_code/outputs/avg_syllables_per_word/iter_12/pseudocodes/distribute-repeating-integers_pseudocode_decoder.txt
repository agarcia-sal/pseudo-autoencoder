CLASS Solution  
    FUNCTION canDistribute(nums, quantity)  
        SET count TO the list of frequency values of elements in nums  
        SORT quantity in descending order  
        
        FUNCTION can_satisfy(index)  
            IF index EQUALS the LENGTH OF quantity  
                RETURN True  
            END IF  
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF element at position i of count GREATER THAN OR EQUAL TO element at position index of quantity  
                    DECREMENT element at position i of count BY element at position index of quantity  
                    IF can_satisfy index PLUS one RETURNS True  
                        RETURN True  
                    END IF  
                    INCREMENT element at position i of count BY element at position index of quantity  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN can_satisfy zero  
    END FUNCTION  
END CLASS
