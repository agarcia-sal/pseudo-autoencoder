CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO an empty mapping with default integer zero  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET key TO bitwise AND of i AND j  
                INCREMENT value at key in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each pair_and and count IN pair_and_count items  
                IF bitwise AND of pair_and AND k EQUALS zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
