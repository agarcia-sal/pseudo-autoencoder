CLASS Solution  
    FUNCTION rob(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        CALL initialize_dp WITH nums TO dp  
        SET element at position zero of dp TO the element at position zero of nums  
        SET element at position one of dp TO the GREATER OF the element at position zero of nums AND the element at position one of nums  
        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET element at position index of dp TO the GREATER OF the element at position index MINUS one of dp AND the SUM OF the element at position index MINUS two of dp AND the element at position index of nums  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  

    FUNCTION initialize_dp WITH nums TO dp  
        PERFORM the creation of a list named dp WITH the LENGTH OF nums FILLED WITH zero  
        RETURN dp  
    END FUNCTION  
END CLASS
