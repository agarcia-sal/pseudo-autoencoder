CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET min_repeats TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF b IS a substring of a REPEATED min_repeats TIMES  
            RETURN min_repeats  
        END IF  
        IF b IS a substring of a REPEATED min_repeats PLUS one TIMES  
            RETURN min_repeats PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
