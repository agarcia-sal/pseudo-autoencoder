CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index of nums EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO the INTEGER DIVISION OF start PLUS end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                INCREMENT cost BY the ABSOLUTE VALUE OF element at position i of positions MINUS (median MINUS (mid MINUS i))  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR i FROM zero TO the LENGTH OF positions MINUS k  
            SET min_cost TO the MINIMUM OF min_cost AND calculate_cost(i, i PLUS k)  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
