CLASS Solution  
    FUNCTION computeArea(ax1, ay1, ax2, ay2, bx1, by1, bx2, by2)  
        SET area1 TO the value of element at position ax2 MINUS element at position ax1 MULTIPLIED BY the value of element at position ay2 MINUS element at position ay1  
        SET area2 TO the value of element at position bx2 MINUS element at position bx1 MULTIPLIED BY the value of element at position by2 MINUS element at position by1  
        IF element at position ax2 GREATER THAN OR EQUAL TO element at position bx2 THEN  
            SET min_x TO element at position bx2  
        ELSE  
            SET min_x TO element at position ax2  
        END IF  
        IF element at position ax1 GREATER THAN OR EQUAL TO element at position bx1 THEN  
            SET max_x TO element at position ax1  
        ELSE  
            SET max_x TO element at position bx1  
        END IF  
        SET overlap_width_candidate TO min_x MINUS max_x  
        IF overlap_width_candidate GREATER THAN zero THEN  
            SET overlap_width TO overlap_width_candidate  
        ELSE  
            SET overlap_width TO zero  
        END IF  
        IF element at position ay2 GREATER THAN OR EQUAL TO element at position by2 THEN  
            SET min_y TO element at position by2  
        ELSE  
            SET min_y TO element at position ay2  
        END IF  
        IF element at position ay1 GREATER THAN OR EQUAL TO element at position by1 THEN  
            SET max_y TO element at position ay1  
        ELSE  
            SET max_y TO element at position by1  
        END IF  
        SET overlap_height_candidate TO min_y MINUS max_y  
        IF overlap_height_candidate GREATER THAN zero THEN  
            SET overlap_height TO overlap_height_candidate  
        ELSE  
            SET overlap_height TO zero  
        END IF  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
