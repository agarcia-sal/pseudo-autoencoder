CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO a set of tuples converted from points  
        SET min_x TO the minimum value of the element at position zero of each point in unique_points  
        SET max_x TO the maximum value of the element at position zero of each point in unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each point x and y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple of reflected_x and y NOT IN unique_points  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
