CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN False  
        END IF  
        CALL initialize_dp_array WITH length THE LENGTH OF s2 PLUS one, STORE RESULT IN dp  
        SET the element at position zero of dp TO True  
        FOR index i FROM zero TO the LENGTH OF s1  
            FOR index j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp AND the element at position i MINUS one of s1 EQUALS the element at position i PLUS j MINUS one of s3  
                END IF  
                IF j GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp OR the element at position j MINUS one of dp AND the element at position j MINUS one of s2 EQUALS the element at position i PLUS j MINUS one of s3  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  

    FUNCTION initialize_dp_array(length)  
        SET result_list TO empty list  
        FOR index FROM zero TO length MINUS one  
            APPEND False TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
