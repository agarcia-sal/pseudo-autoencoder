CLASS Solution  
    FUNCTION numWays(words, target)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF element at position zero of words  
        SET m TO the LENGTH OF target  
          
        SET freq TO a list containing n elements each initialized as an empty mapping from character TO integer  
        FOR each word IN words  
            FOR each index j and character char IN word  
                INCREMENT the value mapped by char in freq at position j BY one  
            END FOR  
        END FOR  
          
        SET dp TO a list containing m PLUS one elements each being a list of n PLUS one zeros  
          
        FOR each j FROM zero TO n  
            SET element at position zero and j of dp TO one  
        END FOR  
          
        FOR each i FROM one TO m  
            FOR each j FROM one TO n  
                SET element at position i and j of dp TO element at position i and j MINUS one of dp  
                IF the value mapped by element at position i MINUS one of target in freq at position j MINUS one GREATER THAN zero  
                    INCREMENT element at position i and j of dp BY element at position i MINUS one and j MINUS one of dp MULTIPLIED BY the value mapped by element at position i MINUS one of target in freq at position j MINUS one  
                    SET element at position i and j of dp TO element at position i and j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
          
        RETURN element at position m and n of dp  
    END FUNCTION  
END CLASS
