CLASS Solution  
    FUNCTION kIncreasing(arr, k)  
        FUNCTION longest_increasing_subsequence(subarr)  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis IS empty OR num GREATER THAN OR EQUAL TO last element of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position where num can be inserted in lis to keep lis sorted (using bisect right)  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN length of lis  
        END FUNCTION  
      
        SET total_operations TO zero  
        SET n TO length of arr  
      
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one IN STEPS OF k  
                APPEND element at position i of arr TO subarr  
            END FOR  
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY length of subarr MINUS lis_length  
        END FOR  
      
        RETURN total_operations  
    END FUNCTION  
END CLASS
