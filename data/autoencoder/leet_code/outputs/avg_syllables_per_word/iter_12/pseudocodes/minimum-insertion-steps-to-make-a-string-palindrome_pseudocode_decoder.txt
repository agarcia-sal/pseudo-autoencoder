CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            CALL create_two_dimensional_array WITH n TO CREATE dp  
            FOR each i FROM zero TO n MINUS one  
                SET element at position i of dp at position i TO one  
            END FOR  
            FOR each cl FROM two TO n  
                FOR each i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two  
                        SET element at position i of dp at position j TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s  
                        SET element at position i of dp at position j TO element at position i PLUS one of dp at position j MINUS one PLUS two  
                    ELSE  
                        SET element at position i of dp at position j TO the GREATER VALUE BETWEEN element at position i of dp at position j MINUS one AND element at position i PLUS one of dp at position j  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero of dp at position n MINUS one  
        END FUNCTION  
        SET the_length_of_s TO the LENGTH OF s  
        SET the_longest_palindromic_subsequence_length TO CALL longest_palindromic_subsequence WITH s  
        RETURN the_length_of_s MINUS the_longest_palindromic_subsequence_length  
    END FUNCTION  
END CLASS
