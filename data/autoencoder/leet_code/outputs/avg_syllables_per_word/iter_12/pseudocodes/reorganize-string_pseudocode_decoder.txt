CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO result of counting frequency of each character in s  
        SET max_heap TO empty list  
        FOR each character AND frequency IN count  
            APPEND pair of negative frequency AND character TO max_heap  
        END FOR  
        CALL auxiliary FUNCTION to transform max_heap into a heap structure  
        
        SET prev_char TO None  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq LESS THAN zero  
            IF max_heap is empty AND prev_freq LESS THAN zero  
                RETURN empty string  
            END IF  
            
            CALL auxiliary FUNCTION to pop element from max_heap AND SET freq AND char to its values  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq LESS THAN zero  
                CALL auxiliary FUNCTION to push pair of prev_freq AND prev_char INTO max_heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
