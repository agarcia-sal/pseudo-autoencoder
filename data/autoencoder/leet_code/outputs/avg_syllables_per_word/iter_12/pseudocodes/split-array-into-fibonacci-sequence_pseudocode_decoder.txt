CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE to the next iteration  
                END IF  
                SET next_num TO the integer value represented by the substring of num from position start TO position end MINUS one  
                IF next_num GREATER THAN OR EQUAL TO two POWERED BY thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the sum of the element at the last position of path AND the element at the second TO last position of path  
                    SET result TO the value returned by calling backtrack WITH parameters end AND the list path APPENDED WITH next_num  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN empty list  
        END FUNCTION  
        RETURN the value returned by calling backtrack WITH parameters zero AND empty list  
    END FUNCTION  
END CLASS
