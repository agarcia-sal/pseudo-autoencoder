CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x, n  
        FUNCTION qpow WITH PARAMETERS a, n  
            SET ans TO one  
            WHILE n IS NOT zero  
                IF n MODULO two EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two USING INTEGER DIVISION  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH ARGUMENTS x, n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH ARGUMENTS x, NEGATIVE n  
        END IF  
    END FUNCTION  
END CLASS
