CLASS Solution  
    FUNCTION validSquare(p1, p2, p3, p4)  
        FUNCTION squared_distance(point1, point2)  
            RETURN the squared value of the difference between element at position zero of point1 and element at position zero of point2 PLUS the squared value of the difference between element at position one of point1 and element at position one of point2  
        END FUNCTION  
        
        SET points TO a list containing p1, p2, p3, and p4  
        SET distances TO an empty set  
        
        FOR index i FROM zero TO three  
            FOR index j FROM i PLUS one TO three  
                SET dist TO squared_distance(element at position i of points, element at position j of points)  
                IF dist EQUALS zero  
                    RETURN False  
                END IF  
                ADD dist TO distances  
            END FOR  
        END FOR  
        
        RETURN whether the length of distances EQUALS two  
    END FUNCTION  
END CLASS
