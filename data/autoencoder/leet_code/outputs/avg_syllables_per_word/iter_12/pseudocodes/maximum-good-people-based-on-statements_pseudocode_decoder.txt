CLASS Solution
    FUNCTION maximumGood(statements)
        SET n TO the LENGTH OF statements
        SET max_good TO zero

        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one
            SET good TO a list of BOOLEAN FALSE repeated n times
            SET bad TO a list of BOOLEAN FALSE repeated n times

            FOR j FROM zero TO n MINUS one
                IF the result of i BITWISE AND two RAISED TO THE POWER OF j IS NOT EQUALS zero
                    SET the element at position j of good TO TRUE
                ELSE
                    SET the element at position j of bad TO TRUE
                END IF
            END FOR

            SET valid TO TRUE
            FOR j FROM zero TO n MINUS one
                IF the element at position j of good IS TRUE
                    FOR k FROM zero TO n MINUS one
                        IF the element at position k of the element at position j of statements EQUALS zero AND the element at position k of good IS TRUE OR the element at position k of the element at position j of statements EQUALS one AND the element at position k of bad IS TRUE
                            SET valid TO FALSE
                            BREAK
                        END IF
                    END FOR
                    IF valid IS FALSE
                        BREAK
                    END IF
                END IF
            END FOR

            IF valid IS TRUE
                SET max_good TO the GREATER OF max_good AND the SUM OF elements in good THAT ARE TRUE
            END IF
        END FOR

        RETURN max_good
    END FUNCTION
END CLASS
