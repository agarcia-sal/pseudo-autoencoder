CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        CALL create_dp_table WITH parameters m PLUS one AND n PLUS one RETURNING dp  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one OF str1 EQUALS character at position j MINUS one OF str2  
                    SET element at position i and j OF dp TO element at position i MINUS one and j MINUS one OF dp PLUS one  
                ELSE  
                    SET element at position i and j OF dp TO the GREATER OF element at position i MINUS one and j OF dp AND element at position i and j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF character at position i MINUS one OF str1 EQUALS character at position j MINUS one OF str2  
                APPEND character at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF element at position i MINUS one and j OF dp GREATER THAN element at position i and j MINUS one OF dp  
                APPEND character at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND character at position j MINUS one OF str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND character at position i MINUS one OF str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero  
            APPEND character at position j MINUS one OF str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        CALL reverse_list WITH parameter scs  
        
        SET result TO concatenate all elements of scs in order  
        RETURN result  
    END FUNCTION  
      
    FUNCTION create_dp_table(rows, columns)  
        SET table TO an empty list  
        FOR r FROM zero TO rows MINUS one  
            SET row TO a list containing columns number of zeros  
            APPEND row TO table  
        END FOR  
        RETURN table  
    END FUNCTION  
      
    FUNCTION reverse_list(list_to_reverse)  
        SET start TO zero  
        SET end TO the LENGTH OF list_to_reverse MINUS one  
        WHILE start LESS THAN end  
            SET temporary TO element at position start OF list_to_reverse  
            SET element at position start OF list_to_reverse TO element at position end OF list_to_reverse  
            SET element at position end OF list_to_reverse TO temporary  
            INCREMENT start BY one  
            DECREMENT end BY one  
        END WHILE  
    END FUNCTION  
END CLASS
