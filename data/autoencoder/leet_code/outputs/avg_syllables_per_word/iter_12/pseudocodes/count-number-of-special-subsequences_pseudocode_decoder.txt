CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO list of zero, zero, zero  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp element at position zero TO the remainder of (two MULTIPLIED BY element at position zero of dp PLUS one) DIVIDED BY MOD  
            ELSE IF num EQUALS one  
                SET dp element at position one TO the remainder of (two MULTIPLIED BY element at position one of dp PLUS element at position zero of dp) DIVIDED BY MOD  
            ELSE IF num EQUALS two  
                SET dp element at position two TO the remainder of (two MULTIPLIED BY element at position two of dp PLUS element at position one of dp) DIVIDED BY MOD  
            END IF  
        END FOR  
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
