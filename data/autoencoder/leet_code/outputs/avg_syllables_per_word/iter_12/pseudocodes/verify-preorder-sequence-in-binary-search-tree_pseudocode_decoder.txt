CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND value GREATER THAN the last element of stack  
                SET lower_bound TO the last element of stack  
                REMOVE last element FROM stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
