CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        INITIALIZE graph AS a mapping with default empty list for new keys  
        INITIALIZE indegree AS list of zeroes of length numCourses  
        
        FOR each pair course AND prereq IN prerequisites  
            APPEND course TO the list at key prereq IN graph  
            INCREMENT the element at position course IN indegree BY one  
        END FOR  
        
        INITIALIZE queue AS a double-ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at position course IN indegree EQUALS zero  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course IN graph  
                DECREMENT the element at position next_course IN indegree BY one  
                IF the element at position next_course IN indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN numCourses EQUALS zero  
    END FUNCTION  
END CLASS
