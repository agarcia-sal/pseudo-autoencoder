CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count GREATER THAN x  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
        RETURN position of the first element in the range from zero TO the LENGTH OF nums MINUS one WHERE function f RETURNS True  
    END FUNCTION  
END CLASS
