CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters of string representation of n  
        SET length TO the LENGTH OF digits  
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i OF digits GREATER THAN OR EQUAL TO element at position i PLUS one OF digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE element at position j OF digits LESS THAN OR EQUAL TO element at position i OF digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary_value TO element at position i OF digits  
        SET element at position i OF digits TO element at position j OF digits  
        SET element at position j OF digits TO temporary_value  
        SET prefix TO elements from position zero TO i PLUS one OF digits  
        SET suffix TO elements from position i PLUS one TO the END OF digits  
        REVERSE the order of suffix  
        SET digits TO concatenation of prefix AND suffix  
        SET result TO integer conversion of concatenation of elements in digits  
        IF result GREATER THAN two TO THE POWER OF thirty ONE MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
