CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, one  
        SET syms TO list of "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"  
        
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR each underscore FROM zero TO num DIVIDED BY element at position i of val MINUS one  
                SET roman_numeral TO roman_numeral CONCATENATED WITH element at position i of syms  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
