CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS sub REVERSED  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two-dimensional list of size n BY n FILLED WITH False  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the character at position i of s EQUALS the character at position j of s AND  
                   (j MINUS i LESS THAN OR EQUAL TO one OR the element at position i PLUS one AND j MINUS one of is_pal IS True)  
                    SET the element at position i AND j of is_pal TO True  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF the element at position zero AND i MINUS one of is_pal IS True  
                FOR j FROM i TO n MINUS two  
                    IF the element at position i AND j of is_pal IS True AND the element at position j PLUS one AND n MINUS one of is_pal IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
