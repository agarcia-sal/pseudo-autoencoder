CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap IS NOT empty AND negative of element at position zero of max_heap GREATER THAN num  
                    INCREMENT operations BY negative of the result of removing first element FROM max_heap MINUS num  
                    ADD negative of num TO max_heap  
                END IF  
                ADD negative of num TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  

        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap IS NOT empty AND element at position zero of min_heap LESS THAN num  
                    INCREMENT operations BY num MINUS the result of removing first element FROM min_heap  
                    ADD num TO min_heap  
                END IF  
                ADD num TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  

        SET result_to_non_decreasing TO the result of min_operations_to_non_decreasing WITH nums  
        SET result_to_non_increasing TO the result of min_operations_to_non_increasing WITH nums  
        RETURN the smaller of result_to_non_decreasing AND result_to_non_increasing  
    END FUNCTION  
END CLASS
