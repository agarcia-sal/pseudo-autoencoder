CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO path PLUS the element at position i OF remaining  
                SET new_remaining TO the concatenation OF the elements FROM start TO i MINUS one OF remaining PLUS the elements FROM i PLUS one TO end OF remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty list  
        CALL backtrack WITH empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
