CLASS Solution  
    FUNCTION subsets(nums)  
        IF nums EQUALS an empty list THEN  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the result of calling subsets with the list of elements of nums starting from position two to the end  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list with the element at position one of nums followed by all elements of subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first and subsets_with_first  
    END FUNCTION  
END CLASS
