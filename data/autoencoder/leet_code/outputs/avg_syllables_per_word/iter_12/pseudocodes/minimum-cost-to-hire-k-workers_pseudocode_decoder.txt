CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the SORTED list of each element formed by the RATIO of element from wage DIVIDED BY element from quality ALONG WITH the corresponding element from quality FOR each pair of elements taken SIMULTANEOUSLY from wage and quality  
        
        SET max_heap TO an empty list  
        SET total_quality TO zero  
        SET min_cost TO infinity  
        
        FOR each ratio, q IN workers  
            INSERT the NEGATIVE of q INTO max_heap USING heap insertion  
            INCREMENT total_quality BY q  
            
            IF the LENGTH OF max_heap IS GREATER THAN k  
                REMOVE the ROOT element FROM max_heap USING heap removal AND ADD it TO total_quality  
            END IF  
            
            IF the LENGTH OF max_heap IS EQUAL TO k  
                SET min_cost TO the MINIMUM OF min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
