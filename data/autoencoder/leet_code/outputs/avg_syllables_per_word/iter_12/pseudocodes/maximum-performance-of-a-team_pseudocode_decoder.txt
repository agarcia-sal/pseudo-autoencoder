CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO list of pairs of corresponding elements from efficiency and speed  
        SORT engineers IN descending order BY efficiency value  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        
        FOR each pair eff and spd IN engineers  
            ADD spd TO speed_heap maintaining min-heap property  
            INCREMENT speed_sum BY spd  
            
            IF LENGTH OF speed_heap GREATER THAN k  
                REMOVE smallest element FROM speed_heap  
                DECREMENT speed_sum BY removed element  
            END IF  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        
        RETURN max_performance MODULO ten TO THE POWER OF nine PLUS seven  
    END FUNCTION  
END CLASS
