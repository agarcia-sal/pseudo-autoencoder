CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL dp TO a NEW list of lists WITH m PLUS one lists EACH CONTAINING n PLUS one zeroes  
        
        FOR i FROM zero TO m  
            SET the element at position zero of dp at position i TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET the element at position j of dp at position zero TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position j of dp at position i TO the element at position j MINUS one of dp at position i MINUS one  
                ELSE  
                    SET deletion_cost TO the element at position j of dp at position i MINUS one  
                    SET insertion_cost TO the element at position j MINUS one of dp at position i  
                    SET replacement_cost TO the element at position j MINUS one of dp at position i MINUS one  
                    SET the element at position j of dp at position i TO one PLUS the MINIMUM OF deletion_cost, insertion_cost, replacement_cost  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position n of dp at position m  
    END FUNCTION  
END CLASS
