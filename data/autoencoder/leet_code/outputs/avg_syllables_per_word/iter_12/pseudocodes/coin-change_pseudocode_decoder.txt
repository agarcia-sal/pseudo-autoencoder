CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO a list containing amount PLUS one repeated for amount PLUS one times  
        SET element at position zero of dp TO zero  
        FOR each a from one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO zero  
                    SET element at position a of dp TO the minimum BETWEEN element at position a of dp AND element at position a MINUS coin of dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF element at position amount of dp IS NOT EQUALS TO amount PLUS one  
            RETURN element at position amount of dp  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
