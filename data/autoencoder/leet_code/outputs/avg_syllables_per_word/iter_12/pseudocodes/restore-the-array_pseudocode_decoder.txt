CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion plus seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros of LENGTH n plus one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i minus one  
                IF the character at position j of s NOT EQUALS zero AND the INTEGER VALUE of substring from position j TO i of s GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET element at position i of dp TO the element at position i of dp PLUS the element at position j of dp  
                    SET element at position i of dp TO the element at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
