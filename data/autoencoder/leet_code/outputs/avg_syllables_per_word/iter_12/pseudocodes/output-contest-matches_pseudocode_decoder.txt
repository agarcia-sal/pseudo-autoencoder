CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO an empty list  
        FOR index FROM one TO n  
            APPEND index converted TO string TO teams  
        END FOR  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO an opening parenthesis concatenated WITH element at position i of teams concatenated WITH a comma concatenated WITH element at position n MINUS one MINUS i of teams concatenated WITH a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
