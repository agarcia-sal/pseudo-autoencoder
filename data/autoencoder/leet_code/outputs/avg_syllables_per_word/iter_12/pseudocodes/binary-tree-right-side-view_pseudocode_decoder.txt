CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value at p NOT EQUALS value at q  
        RETURN False  
    ELSE  
        RETURN AND of is_same_tree with parameters p.left, q.left AND is_same_tree with parameters p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView with parameter root  
        IF root IS None  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO new deque containing root  
        WHILE queue IS not empty  
            SET level_length TO LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO first element REMOVED FROM queue  
                IF i EQUALS level_length MINUS one  
                    APPEND value at node TO right_side_view  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
