CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS node.val  
            SET paths_to_here TO value FROM prefix_sums AT key current_sum MINUS targetSum OR zero IF key NOT FOUND  
            SET prefix_sums AT key current_sum TO value FROM prefix_sums AT key current_sum OR zero IF key NOT FOUND PLUS one  
            SET paths_in_left TO helper(node.left, current_sum, prefix_sums)  
            SET paths_in_right TO helper(node.right, current_sum, prefix_sums)  
            DECREMENT prefix_sums AT key current_sum BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO dictionary WITH key zero AND value one  
        RETURN helper(root, zero, prefix_sums)  
    END FUNCTION  
END CLASS
