CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the ELEMENT AT position zero OF nums converted TO string  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the ELEMENT AT position zero OF nums converted TO string FOLLOWED BY the WORD division SIGN FOLLOWED BY the ELEMENT AT position one OF nums converted TO string  
        END IF  
        SET inner_expression TO the ELEMENT AT position one OF nums converted TO string  
        FOR each element IN the SUBLIST OF nums FROM position two TO the END  
            SET inner_expression TO inner_expression FOLLOWED BY the WORD division SIGN FOLLOWED BY element converted TO string  
        END FOR  
        RETURN the ELEMENT AT position zero OF nums converted TO string FOLLOWED BY the WORD division SIGN FOLLOWED BY an OPENING parenthesis FOLLOWED BY inner_expression FOLLOWED BY a CLOSING parenthesis  
    END FUNCTION  
END CLASS
