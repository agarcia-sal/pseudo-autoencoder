CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero, next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode with values at position zero  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO a new ListNode with val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1, p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val of p1 EQUALS the val of p2 AND is_same_list with the next of p1 and the next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero, left DEFAULT none, right DEFAULT none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with values at position zero  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left of node TO a new TreeNode with the element at position i of values  
            APPEND the left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right of node TO a new TreeNode with the element at position i of values  
            APPEND the right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val of p NOT EQUALS the val of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with the left of p and the left of q AND is_same_tree with the right of p and the right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS head  
        IF head IS none  
            RETURN none  
        END IF  
        
        FUNCTION findMiddle WITH PARAMETERS left, right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next of fast NOT EQUALS right  
                SET slow TO the next of slow  
                SET fast TO the next of the next of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST WITH PARAMETERS left, right  
            IF left EQUALS right  
                RETURN none  
            END IF  
            
            SET mid TO findMiddle with left and right  
            
            SET node TO a new TreeNode with the val of mid  
            SET the left of node TO convertListToBST with left and mid  
            SET the right of node TO convertListToBST with the next of mid and right  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST with head and none  
    END FUNCTION  
END CLASS
