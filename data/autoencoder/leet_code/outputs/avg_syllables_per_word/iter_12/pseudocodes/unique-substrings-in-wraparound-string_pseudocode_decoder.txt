CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a mapping from each character in the sequence from 'a' to 'z' TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the difference BETWEEN the ordinal value of the character at position index of s AND the ordinal value of the character at position index MINUS one of s EQUALS one OR the character at position index of s EQUALS character 'a' AND the character at position index MINUS one of s EQUALS character 'z'  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key the character at position index of s TO the greater value BETWEEN the current value at max_length of the character at position index of s AND current_length  
        END FOR  
        SET total_substrings TO the SUM OF all values in max_length  
        RETURN total_substrings  
    END FUNCTION  
END CLASS
