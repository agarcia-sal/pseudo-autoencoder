CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new empty mapping from value TO list of indices  
        FOR each index i FROM zero TO n MINUS one  
            SET value TO element at position i of arr  
            APPEND i TO list at key value IN value_indices  
        END FOR  
        
        SET queue TO a new empty double-ended queue  
        APPEND the pair of zero and zero TO queue  
        SET visited TO a new empty set  
        ADD zero TO visited  
        
        WHILE queue IS NOT empty  
            REMOVE the first element FROM queue AND UNPACK INTO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new empty list  
            APPEND current_index PLUS one TO next_indices  
            APPEND current_index MINUS one TO next_indices  
            APPEND all elements from list at key arr[current_index] IN value_indices TO next_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET list at key arr[current_index] IN value_indices TO an empty list  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
