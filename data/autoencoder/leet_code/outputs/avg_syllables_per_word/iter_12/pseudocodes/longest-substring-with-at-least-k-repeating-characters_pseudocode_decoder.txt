CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s, k  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        CREATE count AS the frequency mapping OF characters IN s USING Counter  
        FOR each char IN count  
            IF the frequency OF char IN count LESS THAN k  
                RETURN the MAXIMUM OF longestSubstring OF each substring IN the SPLIT OF s BY char WITH k AS parameter  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
