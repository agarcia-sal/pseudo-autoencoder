CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN 2  
            RETURN zero  
        END IF  
          
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeroes with length n  
            SET count TO a list of zeroes with length ten  
              
            FOR i FROM zero TO n MINUS one  
                SET index TO element at position i of arr DIVIDED BY exp  
                INCREMENT element at position index MODULO ten of count BY one  
            END FOR  
              
            FOR i FROM one TO nine  
                SET element at position i of count TO element at position i of count PLUS element at position i MINUS one of count  
            END FOR  
              
            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET index TO element at position i of arr DIVIDED BY exp  
                SET element at position element at position index MODULO ten of count MINUS one of output TO element at position i of arr  
                DECREMENT element at position index MODULO ten of count BY one  
                DECREMENT i BY one  
            END WHILE  
              
            FOR i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
          
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element of arr  
            SET exp TO one  
            WHILE max1 DIVIDED BY exp GREATER THAN zero  
                CALL countingSort with arr and exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
          
        CALL radixSort with nums  
          
        SET max_diff TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            SET max_diff TO the maximum of max_diff AND element at position i of nums MINUS element at position i MINUS one of nums  
        END FOR  
          
        RETURN max_diff  
    END FUNCTION  
END CLASS
