CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO sorted list of each word reversed in words  
        SET total_length TO zero  
        FOR index FROM zero TO length of reversed_words MINUS one  
            IF index PLUS one EQUALS length of reversed_words OR NOT element at position index PLUS one of reversed_words STARTS WITH element at position index of reversed_words  
                INCREMENT total_length BY length of element at position index of reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
