CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of size two RAISED TO THE POWER OF n, EACH initialized TO infinity  
        SET the element at position zero of dp TO zero  
        
        FOR mask FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the COUNT OF ONES IN the BINARY REPRESENTATION OF mask  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND two RAISED TO THE POWER OF j IS NOT zero  
                    SET current_value TO the ELEMENT at position bit_count MINUS one of nums1  
                    SET compare_value TO the ELEMENT at position j of nums2  
                    SET candidate TO the ELEMENT at position mask BITWISE XOR two RAISED TO THE POWER OF j of dp PLUS the BITWISE XOR of current_value AND compare_value  
                    SET the ELEMENT at position mask of dp TO the MINIMUM OF the current ELEMENT at position mask of dp AND candidate  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the ELEMENT at position two RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
