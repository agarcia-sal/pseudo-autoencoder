CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left property of p AND the left property of q  
        AND the result of is_same_tree WITH PARAMETERS the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree WITH PARAMETERS root  
        IF root IS none  
            RETURN none  
        END IF  
        SET the left property of root TO the result of pruneTree WITH PARAMETER the left property of root  
        SET the right property of root TO the result of pruneTree WITH PARAMETER the right property of root  
        IF the val property of root EQUALS zero AND the left property of root IS none AND the right property of root IS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
