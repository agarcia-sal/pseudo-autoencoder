CLASS Solution  
    FUNCTION originalDigits(s)  
        IMPORT Counter FROM collections  
        SET count TO Counter OF s  
        SET digits TO list of zero repeated ten times  
        SET element at position zero of digits TO element at position 'z' of count  
        SET element at position two of digits TO element at position 'w' of count  
        SET element at position four of digits TO element at position 'u' of count  
        SET element at position six of digits TO element at position 'x' of count  
        SET element at position eight of digits TO element at position 'g' of count  
        SET element at position three of digits TO element at position 'h' of count MINUS element at position eight of digits  
        SET element at position five of digits TO element at position 'f' of count MINUS element at position four of digits  
        SET element at position seven of digits TO element at position 's' of count MINUS element at position six of digits  
        SET element at position nine of digits TO element at position 'i' of count MINUS element at position five of digits MINUS element at position six of digits MINUS element at position eight of digits  
        SET element at position one of digits TO element at position 'o' of count MINUS element at position zero of digits MINUS element at position two of digits MINUS element at position four of digits  
        SET result TO empty list  
        FOR i FROM zero TO nine  
            FOR j FROM one TO element at position i of digits  
                APPEND string of i TO result  
            END FOR  
        END FOR  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
