CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero, string one, string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN result of dfs called with u MINUS two  
                FOR each pair composed of l and r IN list of strings '11', '88', '69', '96'  
                    APPEND concatenation of l PLUS v PLUS r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of string zero PLUS v PLUS string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN result of dfs called with n  
    END FUNCTION  
END CLASS
