CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with parameter the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one to the end  
        SET node TO new ListNode with parameter val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical AND of p1.val EQUALS p2.val AND the result of calling is_same_list with parameters p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList with parameter head  
        IF head EQUALS None OR head.next EQUALS None OR head.next.next EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head.next  
        SET even_head TO even  
        WHILE even IS NOT None AND even.next IS NOT None  
            SET odd.next TO even.next  
            SET odd TO odd.next  
            SET even.next TO odd.next  
            SET even TO even.next  
        END WHILE  
        SET odd.next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
