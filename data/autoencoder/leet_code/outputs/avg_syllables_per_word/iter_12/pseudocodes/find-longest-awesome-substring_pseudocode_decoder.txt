CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a map with zero mapped TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  

        FOR each index i AND character char IN s  
            SET digit TO the integer value of char  
            SET prefix_xor TO prefix_xor XOR one LEFT SHIFTED BY digit  
            
            IF prefix_xor EXISTS IN prefix_map  
                SET max_length TO the greater of max_length AND i MINUS the value mapped from prefix_xor in prefix_map  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor XOR one LEFT SHIFTED BY j  
                IF flipped EXISTS IN prefix_map  
                    SET max_length TO the greater of max_length AND i MINUS the value mapped from flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor NOT EXISTS IN prefix_map  
                MAP prefix_xor TO i IN prefix_map  
            END IF  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
