CLASS Solution
    FUNCTION maxHeight(cuboids)
        FOR each cuboid IN cuboids
            SORT the dimensions of cuboid in ascending order
        END FOR

        SORT cuboids lexicographically by their dimensions

        SET n TO the LENGTH OF cuboids

        INITIALIZE dp AS a list of zeros with LENGTH n

        FOR i FROM zero TO n MINUS one
            SET dp at position i TO the element at position two of cuboid at position i

            FOR j FROM zero TO i MINUS one
                IF element at position zero of cuboid at position j LESS THAN OR EQUAL TO element at position zero of cuboid at position i
                AND element at position one of cuboid at position j LESS THAN OR EQUAL TO element at position one of cuboid at position i
                AND element at position two of cuboid at position j LESS THAN OR EQUAL TO element at position two of cuboid at position i
                    SET dp at position i TO the maximum BETWEEN dp at position i AND dp at position j PLUS element at position two of cuboid at position i
                END IF
            END FOR
        END FOR

        RETURN the maximum value IN dp
    END FUNCTION
END CLASS
