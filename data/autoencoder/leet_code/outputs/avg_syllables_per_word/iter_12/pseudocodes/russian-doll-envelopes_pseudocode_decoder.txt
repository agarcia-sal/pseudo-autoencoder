CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY the element at position zero in ascending order AND BY the negative of the element at position one in ascending order  
        SET dp TO an empty list  
        FOR each pair underscore AND h IN envelopes  
            SET idx TO the position found by binary searching left for h IN dp  
            IF idx EQUALS the LENGTH OF dp  
                APPEND h TO dp  
            ELSE  
                SET the element at position idx OF dp TO h  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
