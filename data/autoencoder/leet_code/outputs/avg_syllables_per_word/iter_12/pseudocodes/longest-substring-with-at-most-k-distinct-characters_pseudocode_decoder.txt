CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO LENGTH OF s MINUS one  
            SET element at position right of s IN char_map TO right  
            
            IF LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO element IN char_map WITH minimum associated value  
                SET left TO value associated with leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO maximum OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
