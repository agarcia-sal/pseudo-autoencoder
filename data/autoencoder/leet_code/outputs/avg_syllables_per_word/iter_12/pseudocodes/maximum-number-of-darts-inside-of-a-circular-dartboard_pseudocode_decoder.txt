CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            SET difference_x TO element at position zero of p1 MINUS element at position zero of p2  
            SET difference_y TO element at position one of p1 MINUS element at position one of p2  
            SET square_x TO difference_x MULTIPLIED BY difference_x  
            SET square_y TO difference_y MULTIPLIED BY difference_y  
            SET sum_squares TO square_x PLUS square_y  
            SET result TO square root of sum_squares  
            RETURN result  
        END FUNCTION  
        
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance of p1 and p2  
            IF d GREATER THAN 2 MULTIPLIED BY r  
                RETURN none  
            END IF  
            SET half_d TO d DIVIDED BY two  
            SET a TO square root of r MULTIPLIED BY r MINUS half_d MULTIPLIED BY half_d  
            SET h_x TO (element at position zero of p1 PLUS element at position zero of p2) DIVIDED BY two  
            SET h_y TO (element at position one of p1 PLUS element at position one of p2) DIVIDED BY two  
            SET dx TO (element at position one of p2 MINUS element at position one of p1) MULTIPLIED BY a DIVIDED BY d  
            SET dy TO (element at position zero of p2 MINUS element at position zero of p1) MULTIPLIED BY a DIVIDED BY d  
            SET center1 TO tuple of h_x MINUS dx and h_y PLUS dy  
            SET center2 TO tuple of h_x PLUS dx and h_y MINUS dy  
            RETURN list containing center1 and center2  
        END FUNCTION  
        
        FUNCTION is_inside(circle, point, r)  
            SET dist BETWEEN circle and point TO distance of circle and point  
            IF dist LESS THAN OR EQUAL TO r  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET n TO length of darts  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET max_count TO one  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center of element at position i of darts, element at position j of darts, and r  
                IF centers NOT EQUALS none  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside of center, dart, and r  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_count  
    END FUNCTION  
END CLASS
