CLASS Solution  
    FUNCTION sumOfDistancesInTree(n, edges)  
        SET graph TO an empty mapping from nodes to empty lists  
        FOR each pair a, b IN edges  
            APPEND b TO the list of neighbors for node a in graph  
            APPEND a TO the list of neighbors for node b in graph  
        END FOR  

        SET count TO a list of length n with all elements SET TO one  
        SET dist TO a list of length n with all elements SET TO zero  

        FUNCTION dfs1(u, p)  
            FOR each v IN the list of neighbors for node u in graph  
                IF v NOT EQUALS p  
                    CALL dfs1 with parameters v and u  
                    INCREMENT element at position u of count BY element at position v of count  
                    INCREMENT element at position u of dist BY element at position v of dist PLUS element at position v of count  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs1 with parameters zero and negative one  

        FUNCTION dfs2(u, p)  
            FOR each v IN the list of neighbors for node u in graph  
                IF v NOT EQUALS p  
                    SET element at position v of dist TO element at position u of dist MINUS element at position v of count PLUS n MINUS element at position v of count  
                    CALL dfs2 with parameters v and u  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs2 with parameters zero and negative one  

        RETURN dist  
    END FUNCTION  
END CLASS
