CLASS Solution
    FUNCTION recoverArray(nums)
        SORT nums in ascending order
        SET n TO the LENGTH OF nums

        SET first TO element at position zero of nums

        FOR i FROM one TO n MINUS one
            SET difference TO element at position i of nums MINUS first
            SET k TO difference DIVIDED BY two

            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums
                CONTINUE to next iteration
            END IF

            CALL Counter FUNCTION with nums TO INITIALIZE count
            SET arr TO empty list

            FOR each x IN nums
                IF element at position x in count EQUALS zero
                    CONTINUE to next iteration
                END IF

                IF element at position x PLUS two MULTIPLIED BY k in count EQUALS zero
                    BREAK out of FOR loop
                END IF

                APPEND x PLUS k TO arr
                DECREMENT element at position x in count BY one
                DECREMENT element at position x PLUS two MULTIPLIED BY k in count BY one
            END FOR

            IF LENGTH OF arr EQUALS n DIVIDED BY two
                RETURN arr
            END IF
        END FOR

        RETURN empty list
    END FUNCTION
END CLASS
