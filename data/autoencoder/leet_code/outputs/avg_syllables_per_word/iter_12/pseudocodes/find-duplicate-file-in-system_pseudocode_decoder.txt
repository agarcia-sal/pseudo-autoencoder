CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new dictionary with default value of empty list  
        FOR each path IN paths  
            SET parts TO the list obtained by splitting path by spaces  
            SET directory TO element at position zero of parts  
            FOR each file_info IN the sublist of parts from position one to the end  
                SET file_name TO the substring of file_info before the first occurrence of left parenthesis  
                SET content TO the substring of file_info after the first occurrence of left parenthesis up to but excluding the last character  
                APPEND the concatenation of directory, a forward slash, and file_name TO the list corresponding to content in content_to_paths  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each paths_list IN the values of content_to_paths  
            IF the LENGTH OF paths_list GREATER THAN one  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
