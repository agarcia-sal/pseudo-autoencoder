CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty FOUR LESS THAN one MULTIPLIED BY ten TO THE POWER OF MINUS six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR op IN the list containing true division, multiplication, addition, subtraction  
                            IF (op EQUALS addition OR op EQUALS multiplication) AND j GREATER THAN i  
                                CONTINUE to the next iteration of the loop  
                            END IF  
                            
                            IF op NOT EQUALS true division OR element at position j of nums NOT EQUALS zero  
                                APPEND the result of op applied to element at position i of nums AND element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
