CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position i of nums NOT EQUALS element at position value of element at position i of nums MINUS one of nums  
                SET temporary value TO element at position i of nums  
                SET element at position i of nums TO element at position value of element at position i of nums MINUS one of nums  
                SET element at position value of temporary value MINUS one of nums TO temporary value  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each index and value IN enumeration of nums  
            IF value NOT EQUALS index PLUS one  
                APPEND value TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
