CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        FOR index FROM zero TO the LENGTH OF order MINUS one  
            SET character AT position index OF order TO the variable char  
            SET the value of d AT key char TO index  
        END FOR  
        SET sorted_list TO the result of sorting s BY the following key  
            FOR each character x IN s  
                IF x EXISTS AS a key IN d  
                    RETURN the value of d AT key x  
                ELSE  
                    RETURN zero  
                END IF  
            END FOR  
        SET result_string TO the concatenation of all characters IN sorted_list  
        RETURN result_string  
    END FUNCTION  
END CLASS
