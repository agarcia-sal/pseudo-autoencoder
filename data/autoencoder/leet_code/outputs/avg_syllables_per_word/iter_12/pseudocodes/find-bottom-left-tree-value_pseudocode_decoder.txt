CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized WITH element at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUAL TO None  
            SET node.left TO new TreeNode initialized WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUAL TO None  
            SET node.right TO new TreeNode initialized WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue WITH PARAMETERS root  
        SET queue TO new double-ended queue initialized WITH root  
        SET leftmost_value TO None  
        WHILE queue is not empty  
            SET level_size TO length of queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO first element REMOVED FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO node.val  
                END IF  
                IF node.left NOT EQUAL None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUAL None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
