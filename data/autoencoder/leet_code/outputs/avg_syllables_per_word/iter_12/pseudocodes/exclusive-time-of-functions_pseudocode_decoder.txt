CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO list of zeros of length n  
        SET stack TO empty list  
        
        FOR each log IN logs  
            SPLIT log INTO function_id_string, action, timestamp_string BY colon  
            SET function_id TO function_id_string converted to integer  
            SET timestamp TO timestamp_string converted to integer  
            
            IF action EQUALS the word start  
                IF stack IS NOT empty  
                    SET prev_function_id TO first element of last element of stack  
                    SET prev_timestamp TO second element of last element of stack  
                    INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND tuple of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id TO first element of last element of stack  
                SET prev_timestamp TO second element of last element of stack  
                REMOVE last element FROM stack  
                INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp PLUS one  
                IF stack IS NOT empty  
                    SET function_id_top TO first element of last element of stack  
                    SET _ TO second element of last element of stack  
                    SET last element of stack TO tuple of function_id_top and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
