CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH val set TO element at position zero of values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode WITH val set TO element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode WITH val set TO element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH p.left AND q.left AND AND is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETER root  
        FUNCTION dfs WITH PARAMETER node  
            IF node EQUALS none  
                RETURN node, zero  
            END IF  
            SET left_node, left_depth TO dfs WITH node.left  
            SET right_node, right_depth TO dfs WITH node.right  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node, left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node, right_depth PLUS one  
            ELSE  
                RETURN node, left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of dfs WITH root  
    END FUNCTION  
END CLASS
