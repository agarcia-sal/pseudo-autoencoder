CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of elements in digits  
        SET total_sum TO the sum of elements in digits  
        SET remainder TO total_sum MODULO three  
        
        IF remainder EQUALS one  
            FOR each d IN the list one, four, seven  
                IF the count of element d IS GREATER THAN zero  
                    DECREMENT the count of element d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list two, five, eight  
                    IF the count of element d IS GREATER THAN one  
                        DECREMENT the count of element d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
        END IF  
        
        IF remainder EQUALS two  
            FOR each d IN the list two, five, eight  
                IF the count of element d IS GREATER THAN zero  
                    DECREMENT the count of element d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list one, four, seven  
                    IF the count of element d IS GREATER THAN one  
                        DECREMENT the count of element d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
        END IF  
        
        SET result TO empty list  
        FOR d FROM nine TO zero STEP minus one  
            APPEND the string form of element d repeated the count of element d times TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF final_number IS NOT empty AND the first element of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
