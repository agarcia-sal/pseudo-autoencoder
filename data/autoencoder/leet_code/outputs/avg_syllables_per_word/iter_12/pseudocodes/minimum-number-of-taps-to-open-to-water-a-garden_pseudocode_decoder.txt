CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        INITIALIZE max_range AS a list of zero values with the LENGTH OF n PLUS one  
        
        FOR index FROM zero TO n  
            IF the element at position index of ranges GREATER THAN zero  
                SET left TO the GREATER OF zero AND the element at position index MINUS the element at position index of ranges  
                SET right TO the SMALLER OF n AND the element at position index PLUS the element at position index of ranges  
                SET the element at position left of max_range TO the GREATER OF the element at position left of max_range AND right  
            END IF  
        END FOR  
        
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR index FROM zero TO n  
            IF index GREATER THAN farthest  
                RETURN minus one  
            END IF  
            IF index GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the GREATER OF farthest AND the element at position index of max_range  
        END FOR  
        
        RETURN taps  
    END FUNCTION  
END CLASS
