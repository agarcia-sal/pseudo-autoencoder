CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO list of tuples each containing element at position index of startTime, element at position index of endTime, and element at position index of profit for each index in the range of the LENGTH OF startTime  
        SORT jobs BY the second element of each tuple in ascending order  
        
        SET end_times TO an empty list  
        FOR each job IN jobs  
            APPEND the second element of job TO end_times  
        END FOR  
        
        SET dp TO a list of zero values with the LENGTH EQUAL TO the LENGTH OF jobs PLUS one  
        
        FOR index FROM one TO the LENGTH OF jobs  
            SET current_start TO the first element of the job at position index MINUS one of jobs  
            SET j TO the position to insert current_start in end_times to maintain sorted order using binary search for the right insertion point  
            
            SET profit_if_skip TO the element at position index MINUS one of dp  
            SET profit_if_take TO the element at position j of dp PLUS the third element of the job at position index MINUS one of jobs  
            SET dp at position index TO the maximum of profit_if_skip and profit_if_take  
        END FOR  
        
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
