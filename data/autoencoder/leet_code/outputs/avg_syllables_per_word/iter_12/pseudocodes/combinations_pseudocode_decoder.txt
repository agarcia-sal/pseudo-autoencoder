CLASS Solution  
    FUNCTION combine(n, k)  
        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one  
                APPEND element i TO path  
                CALL backtrack WITH the FIRST PARAMETER i PLUS one AND the SECOND PARAMETER path  
                REMOVE last element FROM path  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH the FIRST PARAMETER one AND the SECOND PARAMETER empty list  
        RETURN result  
    END FUNCTION  
END CLASS
