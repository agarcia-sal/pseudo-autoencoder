CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS EQUAL TO empty  
        RETURN none  
    END IF  
    SET root TO TreeNode(value at position one of values)  
    SET i TO one  
    INITIALIZE queue AS empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO TreeNode(element at position i of values)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO TreeNode(element at position i of values)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty  
            RETURN none  
        END IF  
        SET root_val TO element at position one of preorder  
        SET root TO TreeNode(root_val)  
        SET root_index_inorder TO the POSITION OF root_val IN inorder  
        SET root.left TO self.buildTree(elements from position two TO root_index_inorder PLUS one of preorder, elements from position one TO root_index_inorder of inorder)  
        SET root.right TO self.buildTree(elements from position root_index_inorder PLUS two TO end of preorder, elements from position root_index_inorder PLUS two TO end of inorder)  
        RETURN root  
    END FUNCTION  
END CLASS
