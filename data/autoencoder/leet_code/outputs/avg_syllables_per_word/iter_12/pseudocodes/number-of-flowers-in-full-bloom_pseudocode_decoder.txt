CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of start time element at position zero of each flower in flowers  
        SET end_times TO sorted list of end time element at position one of each flower in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO the insertion position to the right of person in start_times  
            SET ended TO the insertion position to the left of person in end_times  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
