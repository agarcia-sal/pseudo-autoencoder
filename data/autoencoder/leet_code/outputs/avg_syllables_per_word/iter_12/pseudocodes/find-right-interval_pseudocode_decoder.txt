CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs where each pair is the first element of interval and its position in intervals sorted by the first element of interval  
        SET result TO empty list  
        FOR each interval element with second element end IN intervals  
            SET idx TO position found by binary searching the leftmost pair in start_times whose first element is GREATER THAN OR EQUAL TO end  
            IF idx LESS THAN the LENGTH OF start_times  
                APPEND the second element of the pair at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
