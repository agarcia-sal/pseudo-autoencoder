CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS null OR the element at position zero of matrix EQUALS null  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF the element at position zero of matrix  
        SET dp TO the RESULT OF calling initialize_dp WITH m PLUS one AND n PLUS one  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of matrix AT position j MINUS one EQUALS the character one  
                    SET dp at position i and j TO the MINIMUM OF dp at position i MINUS one and j, dp at position i and j MINUS one, dp at position i MINUS one and j MINUS one PLUS one  
                    SET max_side TO the MAXIMUM OF max_side AND dp at position i and j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  

    FUNCTION initialize_dp(rows, columns)  
        SET dp TO an empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row_list TO a list of zero REPEATED columns TIMES  
            APPEND row_list TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
