CLASS Solution  
    FUNCTION getFactors WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS start, target  
            FOR i FROM start TO the INTEGER VALUE OF the SQUARE ROOT OF target PLUS one  
                IF the VALUE OF target MODULO i EQUALS zero  
                    APPEND the CONCATENATION OF path WITH the LIST CONTAINING i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH ARGUMENTS i, target DIVIDED BY i  
                    REMOVE last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH ARGUMENTS two, n  
        RETURN result  
    END FUNCTION  
END CLASS
