CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        
        FOR each num IN reversed order of nums  
            IF num LESS THAN second  
                RETURN true  
            END IF  
            
            WHILE stack is not empty AND element at last position of stack LESS THAN num  
                SET second TO remove last element FROM stack  
            END WHILE  
            
            APPEND num TO stack  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
