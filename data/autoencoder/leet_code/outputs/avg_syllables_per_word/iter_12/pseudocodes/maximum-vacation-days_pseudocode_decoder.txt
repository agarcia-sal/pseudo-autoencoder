CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        SET dp TO a list of length n with all elements SET TO negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO an empty dictionary  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i of graph TO an empty list  
        END FOR  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position j of element at position i of flights EQUALS one  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR index week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n with all elements SET TO negative one  
            FOR index i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE to next iteration of the FOR loop  
                END IF  
                SET element at position i of new_dp TO the maximum value BETWEEN element at position i of new_dp AND element at position i of dp PLUS element at position i of element at position week of days  
                FOR each element j IN element at position i of graph  
                    SET element at position j of new_dp TO the maximum value BETWEEN element at position j of new_dp AND element at position i of dp PLUS element at position j of element at position week of days  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum value in dp  
    END FUNCTION  
END CLASS
