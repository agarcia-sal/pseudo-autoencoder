CLASS Solution  
    FUNCTION canSeePersonsCount WITH parameter heights  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeros with length n  
        SET stack TO an empty list  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position last element of stack IN heights IS LESS THAN the element at position i IN heights  
                SET popped_index TO the REMOVAL of the last element FROM stack  
                INCREMENT the element at position popped_index IN answer BY one  
            END WHILE  
            IF the LENGTH OF stack IS GREATER THAN zero  
                INCREMENT the element at position last element of stack IN answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
