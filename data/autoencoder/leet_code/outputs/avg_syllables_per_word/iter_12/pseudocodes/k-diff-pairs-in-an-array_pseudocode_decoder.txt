CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        
        SET num_counts TO empty dictionary  
        FOR each num IN nums  
            IF num EXISTS IN num_counts  
                SET count TO value at key num in num_counts  
            ELSE  
                SET count TO zero  
            END IF  
            SET value at key num in num_counts TO count PLUS one  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF value at key num in num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                SET needed_num TO num PLUS k  
                IF needed_num EXISTS IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
