CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY length OF element THEN BY element IN lexicographical ORDER  
        SET valid_words TO empty set  
        SET longest_word TO empty string  
        FOR each word IN words  
            IF length OF word EQUALS one OR substring of word from start TO length MINUS one IS IN valid_words  
                ADD word TO valid_words  
                IF length OF word GREATER THAN length OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
