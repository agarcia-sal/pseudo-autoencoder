CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF start MINUS one  
            SET c TO element at position index of start  
            IF c NOT EQUALS character X  
                APPEND tuple of c AND index TO start_filtered  
            END IF  
        END FOR  

        SET end_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF end MINUS one  
            SET c TO element at position index of end  
            IF c NOT EQUALS character X  
                APPEND tuple of c AND index TO end_filtered  
            END IF  
        END FOR  

        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  

        FOR each pair of elements (c1, i1) AND (c2, i2) IN corresponding positions of start_filtered AND end_filtered  
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            IF c1 EQUALS character L AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 EQUALS character R AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
