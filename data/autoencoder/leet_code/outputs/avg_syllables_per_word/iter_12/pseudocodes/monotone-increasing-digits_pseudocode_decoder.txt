CLASS Solution
    FUNCTION monotoneIncreasingDigits(n)
        SET digits TO list of character elements of the string representation of n
        SET length TO the LENGTH OF digits
        
        SET marker TO length
        FOR index FROM length MINUS one DOWNTO one
            IF element at position index of digits LESS THAN element at position index MINUS one of digits
                SET element at position index MINUS one of digits TO string of integer of element at position index MINUS one of digits MINUS one
                SET marker TO index
            END IF
        END FOR
        
        FOR index FROM marker TO length MINUS one
            SET element at position index of digits TO character nine
        END FOR
        
        SET result TO integer parsed from the concatenation of all elements in digits in order
        RETURN result
    END FUNCTION
END CLASS
