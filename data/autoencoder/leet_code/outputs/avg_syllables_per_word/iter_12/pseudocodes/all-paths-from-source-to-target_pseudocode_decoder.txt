CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a new deque initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE q is not empty  
            SET path TO the element REMOVED FROM the left of q  
            SET u TO the last element of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to next iteration of WHILE loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND the concatenation of path and a list containing v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
