CLASS Solution  
    FUNCTION longestPath(parent, s)  
        DEFINE FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN the element at position node OF tree  
                SET child_length TO result of calling dfs with child  
                IF character at position child of s NOT EQUALS character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET result TO the maximum OF result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
      
        SET tree TO an empty mapping with default value as empty list  
        FOR each index i AND element p IN enumeration of parent  
            IF p NOT EQUALS negative one  
                APPEND i TO the element at position p OF tree  
            END IF  
        END FOR  
      
        SET result TO one  
        CALL dfs with argument zero  
        RETURN result  
    END FUNCTION  
END CLASS
