CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty or values IS none  
        RETURN none  
    END IF  
    SET root TO TreeNode CREATED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO TreeNode CREATED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO TreeNode CREATED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p.left, q.left AND the result of is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum WITH PARAMETERS root, targetSum  
        FUNCTION dfs WITH PARAMETERS node, current_sum, path  
            IF node IS none  
                RETURN  
            END IF  
            APPEND node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left IS none AND node.right IS none AND current_sum EQUALS targetSum  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs WITH PARAMETERS node.left, current_sum, path  
            CALL dfs WITH PARAMETERS node.right, current_sum, path  
            REMOVE last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs WITH PARAMETERS root, zero, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
