CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO substring of frag from position one TO position d  
                SET right TO substring of frag from position d PLUS one TO the END  
                IF (left DOES NOT start WITH character zero OR left EQUALS character zero) AND (right DOES NOT end WITH character zero)  
                    YIELD the concatenation of left PLUS (a decimal point IF d DOES NOT EQUAL n ELSE empty string) PLUS right  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET s TO substring of s from position two TO position LENGTH OF s MINUS one  
      
        RETURN the LIST composed of the concatenation of open parenthesis PLUS a PLUS comma PLUS space PLUS b PLUS close parenthesis FOR each i FROM one TO the LENGTH OF s MINUS one FOR each a IN make(the substring of s from position one TO position i) FOR each b IN make(the substring of s from position i PLUS one TO the END)  
    END FUNCTION  
END CLASS
