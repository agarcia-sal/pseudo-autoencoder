CLASS Solution  
    FUNCTION maxSum(nums1, nums2)  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO one followed BY nine zeros PLUS one  
        WHILE i LESS THAN the LENGTH OF nums1 AND j LESS THAN the LENGTH OF nums2  
            IF the element at position i of nums1 LESS THAN the element at position j of nums2  
                INCREMENT sum1 BY the element at position i of nums1  
                INCREMENT i BY one  
            ELSE IF the element at position i of nums1 GREATER THAN the element at position j of nums2  
                INCREMENT sum2 BY the element at position j of nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO the MAXIMUM OF sum1 AND sum2 PLUS the element at position i of nums1  
                SET sum2 TO the value of sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        WHILE i LESS THAN the LENGTH OF nums1  
            INCREMENT sum1 BY the element at position i of nums1  
            INCREMENT i BY one  
        END WHILE  
        WHILE j LESS THAN the LENGTH OF nums2  
            INCREMENT sum2 BY the element at position j of nums2  
            INCREMENT j BY one  
        END WHILE  
        SET result TO the MAXIMUM OF sum1 AND sum2  
        RETURN result MODULO MOD  
    END FUNCTION  
END CLASS
