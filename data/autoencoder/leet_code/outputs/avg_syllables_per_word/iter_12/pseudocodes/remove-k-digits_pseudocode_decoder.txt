CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at last position of stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero  
            SET final_stack TO all elements of stack EXCEPT last k elements  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET result TO string formed by JOINING all elements of final_stack THEN REMOVING leading zeros  
        
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
