CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS one  
            RETURN string one  
        END IF  
        SET current_sequence TO string one  
        FOR each integer from one TO n MINUS one  
            SET next_sequence TO empty string  
            SET count TO one  
            SET previous_char TO character at position one OF current_sequence  
            FOR each integer i FROM two TO LENGTH OF current_sequence  
                IF character at position i OF current_sequence EQUALS previous_char  
                    INCREMENT count BY one  
                ELSE  
                    SET next_sequence TO next_sequence PLUS string representation of count PLUS previous_char  
                    SET previous_char TO character at position i OF current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            SET next_sequence TO next_sequence PLUS string representation of count PLUS previous_char  
            SET current_sequence TO next_sequence  
        END FOR  
        RETURN current_sequence  
    END FUNCTION  
END CLASS
