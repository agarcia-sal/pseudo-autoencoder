CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO a set constructed from bank  
        SET q TO a deque containing a single tuple of start and zero  
        SET mp TO a mapping from character A TO characters T, C, G; from character T TO characters A, C, G; from character C TO characters A, T, G; and from character G TO characters A, T, C  
        WHILE q is not empty  
            SET t and step TO the first element removed from q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR i FROM zero TO the LENGTH OF t MINUS one  
                SET v TO the element at position i of t  
                FOR each j IN the characters mapped by v in mp  
                    SET next TO concatenation of substring of t from start TO i MINUS one, followed by j, followed by substring of t from i PLUS one TO end  
                    IF next IS IN s  
                        APPEND tuple of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
