CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  

        SORT intervals BY the element at position zero of each meeting  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF min_heap NOT EQUALS empty AND element at position zero of meeting GREATER THAN OR EQUAL TO element at position zero of min_heap  
                REMOVE first element FROM min_heap  
            END IF  
            APPEND element at position one of meeting TO min_heap  
        END FOR  

        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
