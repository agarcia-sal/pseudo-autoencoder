CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO an empty mapping of lists with default empty list for missing keys  
        FOR each pair of from_location and to_location IN the list of tickets sorted in ascending order and then reversed  
            APPEND to_location TO the list associated with from_location in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph is not empty  
                CALL visit with the last element removed FROM the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with "JFK"  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
