CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO a list created by pairing each element in capital with the corresponding element in profits  
        SORT projects by the first element of each pair in ascending order  
        
        SET max_heap TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element of the pair at position i in projects LESS THAN OR EQUAL TO w  
                ADD the negative of the second element of the pair at position i in projects TO max_heap, maintaining max-heap order  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS empty  
                BREAK the FOR loop  
            END IF  
            
            REMOVE the largest element FROM max_heap and ADD its negation TO w  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
