CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val SET TO element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val SET TO element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val SET TO element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree with parameters p.left AND q.left AND the RESULT OF is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS node.val  
            IF node.left EQUALS none AND node.right EQUALS none  
                RETURN current_number  
            END IF  
            SET left_sum TO the RESULT OF dfs with parameters node.left AND current_number  
            SET right_sum TO the RESULT OF dfs with parameters node.right AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN the RESULT OF dfs with parameters root AND zero  
    END FUNCTION  
END CLASS
