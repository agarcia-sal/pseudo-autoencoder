CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the SET OF characters in time AFTER REMOVING colon  
        SET hour TO the INTEGER VALUE of the SUBSTRING before the colon in time  
        SET minute TO the INTEGER VALUE of the SUBSTRING after the colon in time  
        
        FUNCTION generate_times(digits)  
            SET times TO an EMPTY list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_value TO the INTEGER VALUE of the CONCATENATION of h1 and h2  
                            SET minute_value TO the INTEGER VALUE of the CONCATENATION of m1 and m2  
                            IF hour_value LESS THAN 24 AND minute_value LESS THAN 60  
                                APPEND the PAIR of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times SORTED in ascending order  
        END FUNCTION  
        
        SET all_times TO the RESULT OF generate_times WITH digits  
        SET current_time_index TO the POSITION OF the PAIR hour and minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the LENGTH OF all_times  
            SET next_hour and next_minute TO the ELEMENT AT POSITION current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour and next_minute TO the FIRST ELEMENT OF all_times  
        END IF  
        
        RETURN the STRING representation of next_hour as TWO digits FOLLOWED BY colon FOLLOWED BY the STRING representation of next_minute as TWO digits  
    END FUNCTION  
END CLASS
