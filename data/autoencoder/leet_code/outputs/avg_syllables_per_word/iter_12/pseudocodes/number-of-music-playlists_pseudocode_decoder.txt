CLASS Solution  
    FUNCTION numMusicPlaylists(n, goal, k)  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize two dimensional list dp WITH dimensions goal PLUS one BY n PLUS one  
        SET element at position zero zero of dp TO one  
        
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                CALCULATE number of ways to add a new song TO the playlist:  
                SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp MULTIPLIED BY (n MINUS (j MINUS one)) PLUS element at position i j of dp  
                SET element at position i j of dp TO element at position i j of dp MODULO MOD  
                
                IF j GREATER THAN k  
                    CALCULATE number of ways to repeat a song played at least k songs ago:  
                    SET element at position i j of dp TO element at position i MINUS one j of dp MULTIPLIED BY (j MINUS k) PLUS element at position i j of dp  
                    SET element at position i j of dp TO element at position i j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position goal n of dp  
    END FUNCTION  
END CLASS
