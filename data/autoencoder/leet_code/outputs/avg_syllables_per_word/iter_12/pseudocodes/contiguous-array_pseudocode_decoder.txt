CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO dictionary with key zero and value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each i FROM zero TO the LENGTH OF nums MINUS one  
            SET num TO element at position i of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the GREATER OF max_length AND i MINUS the VALUE OF prefix_sum_index AT key count  
            ELSE  
                SET prefix_sum_index AT key count TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
