CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values from position one TO end  
        SET node TO new instance of ListNode initialized with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head EQUALS none OR head.next EQUALS none  
            RETURN head  
        END IF  
        SET dummy TO new instance of ListNode initialized with negative five thousand and one  
        SET dummy.next TO head  
        SET current TO element at position next of head  
        SET head.next TO none  
        WHILE current IS not none  
            SET next_node TO element at position next of current  
            SET prev TO dummy  
            WHILE prev.next IS not none AND element at position val of prev.next LESS THAN element at position val of current  
                SET prev TO element at position next of prev  
            END WHILE  
            SET current.next TO prev.next  
            SET prev.next TO current  
            SET current TO next_node  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
