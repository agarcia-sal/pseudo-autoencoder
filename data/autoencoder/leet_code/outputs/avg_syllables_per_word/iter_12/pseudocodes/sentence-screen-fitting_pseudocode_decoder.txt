CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements in sentence with a single space BETWEEN each PLUS a single space at the end  
        SET total_length TO the number of characters in s  
        SET position TO zero  
        
        FOR each dummy_index FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the character at position MODULO total_length of s EQUALS a space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character at position minus one MODULO total_length of s NOT EQUALS a space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        SET result TO position INTEGER DIVIDED BY total_length  
        RETURN result  
    END FUNCTION  
END CLASS
