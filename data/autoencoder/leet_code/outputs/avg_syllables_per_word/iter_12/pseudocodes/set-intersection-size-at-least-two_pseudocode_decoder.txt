CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the element at position one of interval AND negative of the element at position zero of interval in descending order  
        SET first TO the element at position one of the element at position zero of intervals MINUS one  
        SET second TO the element at position one of the element at position zero of intervals  
        SET count TO two  
        FOR each interval consisting of start and end IN the elements of intervals from position one TO the end  
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
