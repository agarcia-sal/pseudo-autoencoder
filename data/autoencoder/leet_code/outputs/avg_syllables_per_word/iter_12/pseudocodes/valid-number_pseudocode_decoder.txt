CLASS Solution  
    FUNCTION isNumber(s)  
        SET number_regex TO a pattern representing start of string WITH optional plus OR minus FOLLOWED BY either one or more digits FOLLOWED BY optional dot FOLLOWED BY optional digits OR a dot FOLLOWED BY one or more digits FOLLOWED BY optional exponential part WITH e FOLLOWED BY optional plus OR minus FOLLOWED BY one or more digits FOLLOWED BY end of string  
        IF full match of s AGAINST number_regex EXISTS  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
