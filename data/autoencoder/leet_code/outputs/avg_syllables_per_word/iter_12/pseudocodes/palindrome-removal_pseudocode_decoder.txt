CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO new two-dimensional list of size n by n, initialized with zeroes using FUNCTION initialize_dp WITH n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i, position i of dp TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i of arr EQUALS element at position i PLUS one of arr  
                SET element at position i, position i PLUS one of dp TO one  
            ELSE  
                SET element at position i, position i PLUS one of dp TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of arr EQUALS element at position j of arr  
                    SET element at position i, position j of dp TO element at position i PLUS one, position j MINUS one of dp  
                ELSE  
                    SET element at position i, position j of dp TO infinity  
                    FOR k FROM i TO j MINUS one  
                        SET current_sum TO element at position i, position k of dp PLUS element at position k PLUS one, position j of dp  
                        IF current_sum LESS THAN element at position i, position j of dp  
                            SET element at position i, position j of dp TO current_sum  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero, position n MINUS one of dp  
    END FUNCTION  
    
    FUNCTION initialize_dp(n)  
        CREATE an empty list called outer_list  
        FOR index FROM zero TO n MINUS one  
            CREATE a list called inner_list of size n filled with zeroes  
            APPEND inner_list TO outer_list  
        END FOR  
        RETURN outer_list  
    END FUNCTION  
END CLASS
