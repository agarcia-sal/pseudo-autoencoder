CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS PLUS OR token EQUALS MINUS OR token EQUALS MULTIPLY OR token EQUALS DIVIDE  
                SET b TO last element of stack  
                REMOVE last element FROM stack  
                SET a TO last element of stack  
                REMOVE last element FROM stack  
                IF token EQUALS PLUS  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS MINUS  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS MULTIPLY  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS DIVIDE  
                    SET result TO a DIVIDED BY b  
                    IF result GREATER THAN OR EQUAL TO zero  
                        SET result TO FLOOR OF result  
                    ELSE  
                        SET result TO CEILING OF result  
                    END IF  
                    APPEND result TO stack  
                END IF  
            ELSE  
                APPEND integer conversion of token TO stack  
            END IF  
        END FOR  
        RETURN element at position zero of stack  
    END FUNCTION  
END CLASS
