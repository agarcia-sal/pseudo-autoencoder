CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize_dp_table WITH n, m, k  
        SET dp TO the result of initialize_dp_table  
        
        FOR j FROM one TO m  
            SET element at position one ONE OF element at position j OF element at position one OF dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position i OF element at position j OF element at position p OF dp TO element at position i OF element at position j OF element at position p OF dp PLUS element at position i MINUS one OF element at position x OF element at position p MINUS one OF dp  
                        SET element at position i OF element at position j OF element at position p OF dp TO element at position i OF element at position j OF element at position p OF dp MODULO MOD  
                    END FOR  
                    
                    SET element at position i OF element at position j OF element at position p OF dp TO element at position i OF element at position j OF element at position p OF dp PLUS element at position i MINUS one OF element at position j OF element at position p OF dp MULTIPLIED BY j  
                    SET element at position i OF element at position j OF element at position p OF dp TO element at position i OF element at position j OF element at position p OF dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position n OF element at position j OF element at position k OF dp  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
    
    FUNCTION initialize_dp_table(n, m, k)  
        CREATE a three dimensional list dp with dimensions n PLUS one BY m PLUS one BY k PLUS one  
        SET all elements of dp TO zero  
        RETURN dp  
    END FUNCTION  
END CLASS
