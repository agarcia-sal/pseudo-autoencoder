CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL people.sort BY the rule: height at position zero MINUS ONE MULTIPLIED, then k at position one in normal order  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT the position equal to the element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
