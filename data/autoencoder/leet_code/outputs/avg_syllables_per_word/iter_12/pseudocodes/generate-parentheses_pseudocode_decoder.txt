CLASS Solution
    FUNCTION generateParenthesis(n)
        FUNCTION backtrack(s, left, right)
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n
                APPEND s TO result
                RETURN
            END IF

            IF left LESS THAN n
                CALL backtrack WITH s PLUS opening parenthesis, left PLUS one, right
            END IF

            IF right LESS THAN left
                CALL backtrack WITH s PLUS closing parenthesis, left, right PLUS one
            END IF
        END FUNCTION

        SET result TO empty list
        CALL backtrack WITH empty string, zero, zero
        RETURN result
    END FUNCTION
END CLASS
