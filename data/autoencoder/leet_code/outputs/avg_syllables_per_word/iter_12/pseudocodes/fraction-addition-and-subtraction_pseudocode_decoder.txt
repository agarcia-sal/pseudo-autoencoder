CLASS Solution  
    FUNCTION fractionAddition WITH PARAMETER expression  
        SET fractions TO expression WITH all occurrences of minus REPLACED BY plus FOLLOWED BY minus  
        SPLIT fractions BY plus INTO list of fraction strings  
        SET result TO the sum of applying Fraction conversion TO each fraction string IN fractions WHERE fraction string IS NOT empty  
        RETURN concatenation of result numerator, forward slash, and result denominator AS string  
    END FUNCTION  
END CLASS
