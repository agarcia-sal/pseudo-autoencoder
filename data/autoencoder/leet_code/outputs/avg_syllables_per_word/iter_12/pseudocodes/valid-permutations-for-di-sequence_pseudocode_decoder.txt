CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a list of lists with dimensions (n PLUS one) BY (n PLUS one) FILLED WITH zero  
        
        FOR j FROM zero TO n  
            SET element at position j of element at position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the character I  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of element at position i MINUS one of dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of element at position i MINUS one of dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET total_sum TO zero  
        FOR each number IN element at position n of dp  
            SET total_sum TO total_sum PLUS number  
        END FOR  
        RETURN total_sum MODULO MOD  
    END FUNCTION  
END CLASS
