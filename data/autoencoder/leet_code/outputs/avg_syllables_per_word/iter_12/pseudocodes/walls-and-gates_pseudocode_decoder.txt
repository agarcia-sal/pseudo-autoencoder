CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms EQUALS empty OR the first element of rooms EQUALS empty  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF the first element of rooms  
        SET queue TO an empty deque  
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF the element at position r of rooms at position c EQUALS zero  
                    APPEND the pair consisting of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO the list consisting of pairs zero and one, one and zero, zero and minus one, minus one and zero  
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET TO r and c  
            FOR each pair dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF zero LESS THAN OR EQUAL TO nr AND nr LESS THAN rows AND zero LESS THAN OR EQUAL TO nc AND nc LESS THAN cols AND the element at position nr of rooms at position nc EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven  
                    SET the element at position nr of rooms at position nc TO the element at position r of rooms at position c PLUS one  
                    APPEND the pair consisting of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
