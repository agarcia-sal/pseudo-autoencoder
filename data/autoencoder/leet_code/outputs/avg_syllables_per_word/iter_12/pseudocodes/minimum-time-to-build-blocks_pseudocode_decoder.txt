CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CALL heapify WITH blocks  
        WHILE the LENGTH OF blocks GREATER THAN one  
            SET first TO the result of popping the smallest element FROM blocks  
            SET second TO the result of popping the smallest element FROM blocks  
            SET combined_time TO split PLUS second  
            PUSH combined_time INTO blocks  
        END WHILE  
        RETURN element at position zero OF blocks  
    END FUNCTION  
END CLASS
