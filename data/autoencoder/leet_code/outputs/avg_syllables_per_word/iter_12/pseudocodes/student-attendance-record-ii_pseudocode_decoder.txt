CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        DEFINE dp AS a three-dimensional list of size n plus one by two by three filled with zero  
        SET the element at position zero zero zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET the element at position i j zero of dp TO the element at position i j zero of dp PLUS the element at position i minus one j k of dp  
                    SET the element at position i j zero of dp TO the element at position i j zero of dp MODULO MOD  
                    IF j EQUALS zero  
                        SET the element at position i one zero of dp TO the element at position i one zero of dp PLUS the element at position i minus one zero k of dp  
                        SET the element at position i one zero of dp TO the element at position i one zero of dp MODULO MOD  
                    END IF  
                    IF k LESS THAN two  
                        SET the element at position i j k plus one of dp TO the element at position i j k plus one of dp PLUS the element at position i minus one j k of dp  
                        SET the element at position i j k plus one of dp TO the element at position i j k plus one of dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS the element at position n j k of dp  
                SET result TO result MODULO MOD  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
