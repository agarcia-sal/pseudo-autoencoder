CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            SET char TO element at position end of s  
            IF char EXISTS IN char_index_map AND element at char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO element at char in char_index_map PLUS one  
            END IF  
            SET element at char in char_index_map TO end  
            SET max_length TO GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
