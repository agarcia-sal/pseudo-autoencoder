CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the string "Zero"  
        END IF  

        SET below_twenty TO a list containing empty string followed by "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten",  
        "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"  
        SET tens TO a list containing two empty strings followed by "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"  
        SET thousands TO a list containing empty string, "Thousand", "Million", "Billion"  

        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty PLUS a space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position n DIVIDED BY ten of tens PLUS a space PLUS helper called with n MODULO ten  
            ELSE  
                RETURN the element at position n DIVIDED BY one hundred of below_twenty PLUS the string " Hundred " PLUS helper called with n MODULO one hundred  
            END IF  
        END FUNCTION  

        SET result TO empty string  
        SET i TO zero  

        WHILE num GREATER THAN zero  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO helper called with num MODULO one thousand PLUS the element at position i of thousands PLUS a space PLUS result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  

        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
