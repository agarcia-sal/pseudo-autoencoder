CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the set containing '2', '5', '6', and '9'  
        SET same_digits TO the set containing '0', '1', and '8'  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF all digits of num_str are IN the union of change_digits and same_digits  
                IF any digit of num_str is IN change_digits  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR i FROM one TO n  
            IF is_good_number(i)  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
