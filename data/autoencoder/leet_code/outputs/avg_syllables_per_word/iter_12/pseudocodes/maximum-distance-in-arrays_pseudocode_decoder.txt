CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO element at position zero of element at position zero of arrays  
        SET max_val TO element at last position of element at position zero of arrays  
        SET max_distance TO zero  
        FOR index FROM one TO the LENGTH OF arrays MINUS one  
            SET current_array TO element at position index of arrays  
            SET current_min TO element at position zero of current_array  
            SET current_max TO element at last position of current_array  
            SET max_distance TO the maximum OF max_distance, the absolute value OF current_max MINUS min_val, and the absolute value OF max_val MINUS current_min  
            SET min_val TO the minimum OF min_val and current_min  
            SET max_val TO the maximum OF max_val and current_max  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
