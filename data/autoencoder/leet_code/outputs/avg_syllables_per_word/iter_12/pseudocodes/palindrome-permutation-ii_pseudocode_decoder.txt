CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each pair c, v IN the ITEMS OF cnt  
                IF v GREATER THAN 1  
                    DECREMENT the VALUE OF c IN cnt BY two  
                    CALL dfs WITH the ARGUMENT the CONCATENATION of c PLUS t PLUS c  
                    INCREMENT the VALUE OF c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO the COUNTER OF s  
        SET mid TO empty string  
        FOR each pair c, v IN the ITEMS OF cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT the VALUE OF c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH the ARGUMENT mid  
        RETURN ans  
    END FUNCTION  
END CLASS
