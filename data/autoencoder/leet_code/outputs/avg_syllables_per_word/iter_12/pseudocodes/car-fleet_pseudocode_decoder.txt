CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET paired_list TO the SORTED list of PAIRS created by ZIPPING position WITH speed in REVERSE order  
        SET time_to_target TO an EMPTY list  
        FOR each pair_of_position_and_speed IN paired_list  
            SET time_to_target_element TO (target MINUS the position element of pair_of_position_and_speed) DIVIDED BY the speed element of pair_of_position_and_speed  
            APPEND time_to_target_element TO time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
