CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block IS composed only of digits THEN  
                SET num TO integer value of block  
                IF num LESS THAN OR EQUAL TO two hundred fifty-five AND string representation of num EQUALS block THEN  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF length of block GREATER THAN OR EQUAL TO one AND length of block LESS THAN OR EQUAL TO four THEN  
                TRY  
                    CONVERT block TO integer interpreting as base sixteen  
                    RETURN True  
                CATCH ValueError  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO result of splitting queryIP by period  
        SET parts_ipv6 TO result of splitting queryIP by colon  

        IF length of parts_ipv4 EQUALS four AND for each part IN parts_ipv4 all calls to is_valid_ipv4_block(part) RETURN True THEN  
            RETURN string "IPv4"  
        ELSE IF length of parts_ipv6 EQUALS eight AND for each part IN parts_ipv6 all calls to is_valid_ipv6_block(part) RETURN True THEN  
            RETURN string "IPv6"  
        ELSE  
            RETURN string "Neither"  
        END IF  
    END FUNCTION  
END CLASS
