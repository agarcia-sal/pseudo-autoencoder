CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO call FUNCTION grayCode with parameter n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one  
        SET current_gray TO concatenation of prev_gray AND list constructed AS FOLLOW:  
            FOR each number IN reverse of prev_gray  
                SET item TO mask OR number  
            END FOR  
        RETURN current_gray  
    END FUNCTION  
END CLASS
