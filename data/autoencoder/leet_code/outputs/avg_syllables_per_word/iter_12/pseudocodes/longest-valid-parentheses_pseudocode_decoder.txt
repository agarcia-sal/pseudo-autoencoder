CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing negative one  
        SET max_length TO zero  
        FOR each position i AND character char IN enumerate s  
            IF char EQUALS opening parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET current_length TO i MINUS last element of stack  
                    IF current_length GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
