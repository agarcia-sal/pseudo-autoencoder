CLASS Solution  
    FUNCTION findMaxAverage(nums, k)  
        FUNCTION canFindAverage(mid)  
            SET prefix_sums TO list of zeros with length LENGTH OF nums PLUS one  
            SET min_prefix_sum TO list of zeros with length LENGTH OF nums PLUS one  
            FOR index FROM zero TO LENGTH OF nums MINUS one  
                SET element at position index PLUS one of prefix_sums TO element at position index of prefix_sums PLUS element at position index of nums MINUS mid  
                IF index GREATER THAN OR EQUAL TO k THEN  
                    SET element at position index PLUS one of min_prefix_sum TO the minimum BETWEEN element at position index of min_prefix_sum AND element at position index PLUS one MINUS k of prefix_sums  
                ELSE  
                    SET element at position index PLUS one of min_prefix_sum TO element at position index of min_prefix_sum  
                END IF  
                IF index GREATER THAN OR EQUAL TO k MINUS one AND element at position index PLUS one of prefix_sums MINUS element at position index PLUS one of min_prefix_sum GREATER THAN OR EQUAL TO zero THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET low TO the minimum value IN nums  
        SET high TO the maximum value IN nums  
        WHILE high MINUS low GREATER THAN one TIMES ten TO THE POWER OF MINUS five  
            SET mid TO low PLUS high DIVIDED BY two  
            IF canFindAverage(mid) THEN  
                SET low TO mid  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
