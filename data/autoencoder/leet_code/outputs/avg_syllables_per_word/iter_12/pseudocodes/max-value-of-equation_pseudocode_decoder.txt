CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        FOR each element consisting of xj and yj IN points  
            WHILE deque_ is not empty AND xj MINUS the element at position zero of deque_ at position one GREATER THAN k  
                REMOVE first element FROM deque_  
            END WHILE  
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND yj PLUS xj PLUS the element at position zero of deque_ at position zero  
            END IF  
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO the element at last position of deque_ at position zero  
                REMOVE last element FROM deque_  
            END WHILE  
            APPEND the pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
