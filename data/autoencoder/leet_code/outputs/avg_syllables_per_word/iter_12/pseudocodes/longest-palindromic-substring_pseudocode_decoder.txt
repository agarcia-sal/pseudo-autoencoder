CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two-dimensional list of size n BY n FILLED WITH True  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET f at position i and j TO False  
                IF character at position i of s EQUALS character at position j of s  
                    SET f at position i and j TO value at position i PLUS one and j MINUS one of f  
                    IF value at position i and j of f AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN substring of s STARTING AT k OF LENGTH mx  
    END FUNCTION  
END CLASS
