CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zero values with LENGTH of two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the NEGATION OF element at position i of nums INTO min_heap USING heap insertion  
            IF the LENGTH OF min_heap IS GREATER THAN n  
                REMOVE the smallest element FROM min_heap USING heap removal  
            END IF  
            IF the LENGTH OF min_heap EQUALS n  
                SET element at position i of min_sums TO the NEGATION OF the SUM OF all elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zero values with LENGTH of two MULTIPLIED BY n PLUS one  
        
        FOR i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH element at position i of nums INTO max_heap USING heap insertion  
            IF the LENGTH OF max_heap IS GREATER THAN n  
                REMOVE the smallest element FROM max_heap USING heap removal  
            END IF  
            IF the LENGTH OF max_heap EQUALS n  
                SET element at position i of max_sums TO the SUM OF all elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET current_difference TO element at position i of min_sums MINUS element at position i PLUS one of max_sums  
            IF current_difference IS LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
