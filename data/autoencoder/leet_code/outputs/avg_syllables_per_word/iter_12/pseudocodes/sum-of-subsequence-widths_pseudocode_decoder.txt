CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO one billion PLUS seven  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of length n WITH all elements SET TO one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the element at position index MINUS one of power MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS the element at position index of nums MULTIPLIED BY the element at position index of power MINUS the element at position n MINUS index MINUS one of power MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
