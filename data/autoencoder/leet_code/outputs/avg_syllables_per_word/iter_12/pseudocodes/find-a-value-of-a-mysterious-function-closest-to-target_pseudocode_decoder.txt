CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO an empty set  
        SET min_diff TO infinity value  
        FOR each num IN arr  
            SET seen TO the UNION of the set of bitwise AND of num AND each element x IN seen WITH the set containing only num  
            FOR each val IN seen  
                SET min_diff TO the smaller value BETWEEN min_diff AND the absolute value of val MINUS target  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
