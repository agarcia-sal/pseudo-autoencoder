CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF NOT element at position i of used AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero)  
                    SET element at position i of used TO true  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET element at position i of used TO false  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list of false with length n PLUS one  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
