CLASS Solution  
    FUNCTION minmaxGasDist(stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER part OF the difference BETWEEN element at position index of stations AND element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN one TEN THOUSANDTH  
            SET mid TO the sum of left AND right DIVIDED BY two  
            IF possible(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
