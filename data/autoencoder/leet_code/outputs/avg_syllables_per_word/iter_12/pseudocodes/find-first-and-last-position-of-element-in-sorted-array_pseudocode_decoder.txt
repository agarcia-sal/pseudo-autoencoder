CLASS Solution  
    FUNCTION searchRange(nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY two  
                IF the element at position mid of nums LESS THAN target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY two  
                IF the element at position mid of nums LESS THAN OR EQUAL TO target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN right  
        END FUNCTION  

        SET left_index TO find_left(nums, target)  
        SET right_index TO find_right(nums, target)  

        IF left_index LESS THAN OR EQUAL TO right_index AND zero LESS THAN OR EQUAL TO right_index AND right_index LESS THAN the LENGTH OF nums AND the element at position left_index of nums EQUALS target  
            RETURN list containing left_index AND right_index  
        ELSE  
            RETURN list containing minus one AND minus one  
        END IF  
    END FUNCTION  
END CLASS
