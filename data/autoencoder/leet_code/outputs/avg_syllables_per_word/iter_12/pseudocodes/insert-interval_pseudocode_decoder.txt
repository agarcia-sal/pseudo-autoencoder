CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals EQUALS empty  
            RETURN list containing newInterval  
        END IF  
        
        SET merged TO empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        
        WHILE i LESS THAN n AND element at position i of intervals at position one LESS THAN newInterval at position zero  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i LESS THAN n AND element at position i of intervals at position zero LESS THAN OR EQUAL TO newInterval at position one  
            SET newInterval at position zero TO MINIMUM OF newInterval at position zero AND element at position i of intervals at position zero  
            SET newInterval at position one TO MAXIMUM OF newInterval at position one AND element at position i of intervals at position one  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
