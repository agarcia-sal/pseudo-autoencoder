CLASS Solution  
    FUNCTION leastBricks(wall)  
        SET gap_counts TO default dictionary with integer default  
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row EXCEPT the last one  
                INCREMENT total BY brick  
                INCREMENT the value at key total in gap_counts BY one  
            END FOR  
        END FOR  
        SET max_gaps TO the maximum value IN all values of gap_counts WITH a default of zero  
        RETURN the length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
