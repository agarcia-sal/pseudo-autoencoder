CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS one) DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
          
        SET memo TO empty collection  
          
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair current_total AND used_numbers IS IN memo  
                RETURN the value FROM memo FOR the pair current_total AND used_numbers  
            END IF  
              
            FOR number FROM one TO maxChoosableInteger  
                IF NOT the bitwise AND of used_numbers AND the left shift of one BY number IS EQUALS zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, the bitwise OR of used_numbers AND the left shift of one BY number)  
                        SET memo AT key current_total AND used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
              
            SET memo AT key current_total AND used_numbers TO false  
            RETURN false  
        END FUNCTION  
          
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
