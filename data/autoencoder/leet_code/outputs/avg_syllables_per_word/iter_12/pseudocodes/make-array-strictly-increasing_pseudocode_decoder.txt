CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL arr2 SORT IN ASCENDING ORDER  
        SET dp TO a dictionary with key negative one and value zero  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each last_num AND ops IN key value pairs of dp  
                IF num GREATER THAN last_num  
                    SET value_at_num IN new_dp TO the minimum OF value_at_num IN new_dp OR positive infinity AND ops  
                END IF  
                
                SET idx TO the position to insert last_num in arr2 to maintain ascending order by bisect right  
                
                IF idx LESS THAN the length OF arr2  
                    SET value_at_arr2_at_idx IN new_dp TO the minimum OF value_at_arr2_at_idx IN new_dp OR positive infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp IS empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value AMONG all values IN dp  
    END FUNCTION  
END CLASS
