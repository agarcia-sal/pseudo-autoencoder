CLASS Solution  
    FUNCTION largestPalindrome(n)  
        IF n EQUALS one  
            RETURN nine  
        END IF  
        
        SET upper_limit TO ten RAISED TO THE POWER OF n MINUS one  
        SET lower_limit TO ten RAISED TO THE POWER OF n MINUS one MINUS one  
        
        SET max_palindrome TO zero  
        
        FOR i FROM upper_limit TO lower_limit MINUS one STEP MINUS one  
            FOR j FROM i TO lower_limit MINUS one STEP MINUS one  
                SET product TO i MULTIPLIED BY j  
                IF product LESS THAN OR EQUAL TO max_palindrome  
                    BREAK from the inner loop  
                END IF  
                SET str_product TO the string representation of product  
                IF str_product EQUALS the reverse of str_product  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_palindrome MODULO one thousand three hundred thirty seven  
    END FUNCTION  
END CLASS
