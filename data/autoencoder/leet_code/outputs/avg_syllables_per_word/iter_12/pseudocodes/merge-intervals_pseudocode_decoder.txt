CLASS Solution  
    FUNCTION merge(intervals)  
        CALL intervals SORTED BY element at position zero of each element  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged EQUALS empty list OR element at position one of element at last position of merged LESS THAN element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one of element at last position of merged TO the GREATER THAN OR EQUAL TO of element at position one of element at last position of merged AND element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
