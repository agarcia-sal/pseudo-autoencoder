CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i of board AND exclusive OR the element at position zero of row i of board AND exclusive OR the element at position j of row zero of board AND exclusive OR the element at position zero of row zero of board EQUALS true  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY the element at position zero of row i of board  
            INCREMENT colSum BY the element at position i of row zero of board  
            INCREMENT rowSwap BY the test whether the element at position zero of row i of board EQUALS i MODULO two  
            INCREMENT colSwap BY the test whether the element at position i of row zero of board EQUALS i MODULO two  
        END FOR  
        
        IF the integer division of N by two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one by two  
            RETURN negative one  
        END IF  
        
        IF the integer division of N by two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one by two  
            RETURN negative one  
        END IF  
        
        IF N MODULO two IS NOT EQUAL TO zero  
            IF rowSwap MODULO two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            
            IF colSwap MODULO two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum of colSwap AND N MINUS colSwap  
        END IF  
        
        SET result TO the integer division of the sum of rowSwap AND colSwap by two  
        RETURN result  
    END FUNCTION  
END CLASS
