CLASS Solution  
    FUNCTION isScramble(s1, s2)  
        IF s1 EQUALS s2  
            RETURN True  
        END IF  
        
        IF the SORTED CHARACTERS OF s1 NOT EQUALS the SORTED CHARACTERS OF s2  
            RETURN False  
        END IF  
        
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF self.isScramble(the SUBSTRING OF s1 FROM zero TO i) AND self.isScramble(the SUBSTRING OF s2 FROM zero TO i)  
                RETURN True  
            END IF  
            
            IF self.isScramble(the SUBSTRING OF s1 FROM zero TO i) AND self.isScramble(the SUBSTRING OF s2 FROM THE POSITION MINUS i TO THE END) AND self.isScramble(the SUBSTRING OF s1 FROM i TO THE END) AND self.isScramble(the SUBSTRING OF s2 FROM ZERO TO THE POSITION MINUS i)  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
