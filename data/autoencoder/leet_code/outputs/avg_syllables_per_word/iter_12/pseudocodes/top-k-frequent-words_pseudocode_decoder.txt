CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO RESULT OF CALLING Counter WITH words  
        SET heap TO EMPTY LIST  
        FOR each word, freq IN count items  
            APPEND tuple OF negative freq AND word TO heap  
        END FOR  
        CALL heapify ON heap  
        SET result TO EMPTY LIST  
        FOR index FROM zero TO k MINUS one  
            SET top_element TO RESULT OF CALLING heappop ON heap  
            APPEND element at position one OF top_element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
