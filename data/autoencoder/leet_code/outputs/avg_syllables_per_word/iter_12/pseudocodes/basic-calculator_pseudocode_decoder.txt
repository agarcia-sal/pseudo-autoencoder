CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus symbol  
            SET num TO zero  
            
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO the first element of s  
                REMOVE first element FROM s  
                
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                
                IF char EQUALS left parenthesis symbol  
                    SET num TO helper(s)  
                END IF  
                
                IF char IS NOT a digit AND char IS NOT a whitespace OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus symbol  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol  
                        APPEND negative num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET the last element of stack TO the integer division of the last element of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS right parenthesis symbol  
                    BREAK from the WHILE loop  
                END IF  
            END WHILE  
            
            RETURN the sum of stack  
        END FUNCTION  
        
        RETURN helper(the list of elements of s)  
    END FUNCTION  
END CLASS
