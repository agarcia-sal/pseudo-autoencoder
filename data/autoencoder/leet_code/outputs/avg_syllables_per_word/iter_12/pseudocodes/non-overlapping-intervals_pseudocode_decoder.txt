CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position one of each interval  
        SET end TO the element at position one of the element at position zero of intervals  
        SET count TO zero  
        FOR i FROM one TO the LENGTH OF intervals MINUS one  
            IF the element at position zero of the element at position i of intervals GREATER THAN OR EQUALS end  
                SET end TO the element at position one of the element at position i of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
