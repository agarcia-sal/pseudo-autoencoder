CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO empty list  
        FOR each num IN nums  
            APPEND string representation of num TO nums_str  
        END FOR  
        
        FUNCTION compare(x, y)  
            IF concatenation of x followed by y IS GREATER THAN concatenation of y followed by x  
                RETURN negative one  
            ELSE IF concatenation of x followed by y IS LESS THAN concatenation of y followed by x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING the comparison defined by compare  
        
        IF element at position zero of nums_str EQUALS string "zero"  
            RETURN string "zero"  
        END IF  
        
        SET result TO empty string  
        FOR each element IN nums_str  
            APPEND element TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
