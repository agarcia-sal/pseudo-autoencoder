CLASS Solution  
    FUNCTION minWindow(s, t)  
        FROM collections IMPORT Counter  

        SET t_count TO Counter OF t  
        SET required TO the LENGTH OF t_count  

        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO Counter OF empty  

        SET min_length TO positive infinity  
        SET min_start TO zero  

        WHILE right LESS THAN the LENGTH OF s  
            SET current_char TO element at position right of s  
            INCREMENT window_counts[current_char] BY one  

            IF current_char IN t_count AND window_counts[current_char] EQUALS t_count[current_char]  
                INCREMENT formed BY one  
            END IF  

            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO element at position left of s  

                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  

                DECREMENT window_counts[char] BY one  

                IF char IN t_count AND window_counts[char] LESS THAN t_count[char]  
                    DECREMENT formed BY one  
                END IF  

                INCREMENT left BY one  
            END WHILE  

            INCREMENT right BY one  
        END WHILE  

        IF min_length EQUALS positive infinity  
            RETURN empty string  
        ELSE  
            RETURN substring of s FROM min_start TO min_start PLUS min_length MINUS one inclusive  
        END IF  
    END FUNCTION  
END CLASS
