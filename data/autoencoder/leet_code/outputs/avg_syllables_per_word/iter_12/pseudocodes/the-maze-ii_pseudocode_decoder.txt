CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET directions TO a list containing the tuple one and zero, the tuple negative one and zero, the tuple zero and one, the tuple zero and negative one  
        SET start TO the tuple form of start  
        SET destination TO the tuple form of destination  
        SET visited TO an empty set  
        SET pq TO a list containing a tuple of zero and start  

        WHILE the LENGTH OF pq IS GREATER THAN zero  
            SET dist, position TO the result of removing the element with smallest first value from pq  
            SET x TO the element at position zero of position  
            SET y TO the element at position one of position  
            
            IF x EQUALS the element at position zero of destination AND y EQUALS the element at position one of destination  
                RETURN dist  
            END IF
            
            IF the tuple of x and y IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF
            
            ADD the tuple of x and y TO visited  
            
            FOR each tuple dx, dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                END WHILE
                
                IF the tuple of new_x and new_y IS NOT IN visited  
                    ADD a tuple of dist PLUS count AND the tuple of new_x and new_y TO pq maintaining the order by the first element of the tuple  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
