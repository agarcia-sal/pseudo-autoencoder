CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            IF check_str EQUALS reverse of check_str  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FOR i FROM length of s DOWN TO zero  
            IF is_palindrome(element at positions one TO i of s)  
                SET to_add TO reverse of element at positions i plus one TO length of s of s  
                RETURN concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
