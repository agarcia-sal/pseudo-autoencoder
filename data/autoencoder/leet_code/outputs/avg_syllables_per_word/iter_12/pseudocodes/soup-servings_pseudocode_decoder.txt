CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN 4800  
            RETURN one point zero  
        END IF  
        SET n TO the sum of n and twenty four DIVIDED BY twenty five  
        SET memo TO an empty collection  
        FUNCTION dp(a, b)  
            IF the pair of a and b EXISTS IN memo  
                RETURN the value associated with the pair of a and b in memo  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY the sum of  
                dp(a MINUS four, b), dp(a MINUS three, b MINUS one), dp(a MINUS two, b MINUS two), and dp(a MINUS one, b MINUS three)  
            SET the pair of a and b in memo TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp(n, n)  
    END FUNCTION  
END CLASS
