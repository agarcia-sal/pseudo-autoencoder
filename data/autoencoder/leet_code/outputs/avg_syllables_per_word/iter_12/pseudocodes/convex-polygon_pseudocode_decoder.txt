CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN the RESULT OF subtracting the PRODUCT OF the DIFFERENCE BETWEEN element at position zero of a AND element at position zero of o AND the DIFFERENCE BETWEEN element at position one of b AND element at position one of o FROM the PRODUCT OF the DIFFERENCE BETWEEN element at position one of a AND element at position one of o AND the DIFFERENCE BETWEEN element at position zero of b AND element at position zero of o  
        END FUNCTION  

        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  

        SET orientation TO zero  

        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position the REMAINDER OF the SUM OF i AND one DIVIDED BY n of points  
            SET b TO element at position the REMAINDER OF the SUM OF i AND two DIVIDED BY n of points  
            SET cross TO cross_product(o, a, b)  

            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
