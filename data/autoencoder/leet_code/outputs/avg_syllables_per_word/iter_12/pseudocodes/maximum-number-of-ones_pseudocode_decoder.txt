CLASS Solution  
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)  
        SET freq TO a list of zero values with length sideLength MULTIPLIED BY sideLength  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET pos TO the remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of j DIVIDED BY sideLength  
                INCREMENT the element at position pos of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET total_sum TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total_sum BY the element at position index of freq  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
