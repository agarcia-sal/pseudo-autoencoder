CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as element at position zero of values, left as None, right as None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value as element at position i of values, left as None, right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value as element at position i of values, left as None, right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        SET left_comparison TO is_same_tree(p.left, q.left)  
        SET right_comparison TO is_same_tree(p.right, q.right)  
        RETURN left_comparison AND right_comparison  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET queue TO a double-ended queue containing root  
        SET largest_values TO empty list  
        WHILE queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR index FROM one TO level_size  
                REMOVE first element FROM queue AND SET TO node  
                IF node.val GREATER THAN max_value  
                    SET max_value TO node.val  
                END IF  
                IF node.left NOT EQUALS None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
