CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state IS IN memo  
                RETURN the value at key state in memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring of state FROM index TO index PLUS two EQUALS two plus signs  
                    SET new_state TO the concatenation of the substring of state FROM start TO index, two minus signs, and the substring of state FROM index PLUS two TO end  
                    IF NOT can_win_helper(new_state)  
                        SET the value at key state in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value at key state in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
