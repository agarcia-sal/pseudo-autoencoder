CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        SET dp TO result of create_two_dimensional_list_with_value(1, m, n)  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i and j of dp TO element at position i MINUS one and j of dp PLUS element at position i and j MINUS one of dp  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one and n MINUS one of dp  
    END FUNCTION  
  
    FUNCTION create_two_dimensional_list_with_value(value, rows, columns)  
        SET result TO empty list  
        FOR row_index FROM zero TO rows MINUS one  
            SET new_row TO empty list  
            FOR column_index FROM zero TO columns MINUS one  
                APPEND value TO new_row  
            END FOR  
            APPEND new_row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
