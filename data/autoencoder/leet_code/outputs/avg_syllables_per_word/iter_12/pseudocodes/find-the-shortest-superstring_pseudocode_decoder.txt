CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING of element at position i of words TAKING the last k CHARACTERS EQUALS the SUBSTRING of element at position j of words TAKING the first k CHARACTERS  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS the value of ONE SHIFTED LEFT BY n MINUS one  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND OF mask AND ONE SHIFTED LEFT BY j EQUALS zero  
                    SET length, path TO dp(the bitwise OR OF mask AND ONE SHIFTED LEFT BY j, j)  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS overlap(i, j)  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING OF element at position j of words FROM the CHARACTER AT POSITION overlap(i, j) TO THE END CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length, path TO dp(ONE SHIFTED LEFT BY i, i)  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
