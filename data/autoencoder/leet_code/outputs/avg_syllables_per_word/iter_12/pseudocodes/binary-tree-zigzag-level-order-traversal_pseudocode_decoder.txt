CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as element at position zero of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not None  
            SET node.left TO new TreeNode with val as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not None  
            SET node.right TO new TreeNode with val as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(root)  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new deque containing root  
        SET left_to_right TO True  
        WHILE queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO new empty deque  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO remove first element FROM queue  
                IF left_to_right EQUALS True  
                    APPEND node.val TO current_level  
                ELSE  
                    INSERT node.val AT the beginning OF current_level  
                END IF  
                IF node.left is not None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right is not None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND list converted from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
