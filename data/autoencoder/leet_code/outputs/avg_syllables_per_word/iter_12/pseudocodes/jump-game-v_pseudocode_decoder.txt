CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x LESS THAN the LENGTH OF arr  
                    IF element at position i PLUS x of arr GREATER THAN OR EQUALS element at position i of arr  
                        BREAK  
                    END IF  
                    SET candidate TO one PLUS dp(i PLUS x)  
                    IF candidate GREATER THAN max_jumps  
                        SET max_jumps TO candidate  
                    END IF  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x GREATER THAN OR EQUAL TO zero  
                    IF element at position i MINUS x of arr GREATER THAN OR EQUALS element at position i of arr  
                        BREAK  
                    END IF  
                    SET candidate TO one PLUS dp(i MINUS x)  
                    IF candidate GREATER THAN max_jumps  
                        SET max_jumps TO candidate  
                    END IF  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET maximum_jumps TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            SET current_jumps TO dp(i)  
            IF current_jumps GREATER THAN maximum_jumps  
                SET maximum_jumps TO current_jumps  
            END IF  
        END FOR  
        RETURN maximum_jumps  
    END FUNCTION  
END CLASS
