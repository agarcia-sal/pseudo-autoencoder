CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            PUSH the tuple of the element at position zero of the list at position index of nums, index, and zero INTO min_heap  
            IF the element at position zero of the list at position index of nums IS GREATER THAN max_value THEN  
                SET max_value TO the element at position zero of the list at position index of nums  
            END IF  
        END FOR  
        
        SET smallest_range TO infinity  
        SET result TO a list containing minus one hundred thousand and one hundred thousand  
        
        WHILE min_heap IS NOT empty  
            POP the smallest tuple FROM min_heap AND UNPACK INTO min_value, list_index, element_index  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list containing min_value and max_value  
            END IF  
            
            IF element_index PLUS one IS LESS THAN the LENGTH OF the list at position list_index of nums THEN  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                PUSH the tuple of next_value, list_index, element_index PLUS one INTO min_heap  
                IF next_value IS GREATER THAN max_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
