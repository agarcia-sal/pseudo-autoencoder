CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each char IN s  
            IF char EQUALS opening parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO the value popped from stack PLUS  
                    IF score EQUALS zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
