CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND value at position i of values NOT EQUALS None THEN  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND value at position i of values NOT EQUALS None THEN  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left of p, left of q) AND is_same_tree(right of p, right of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(root)  
        FUNCTION dfs(node)  
            IF node EQUALS None THEN  
                RETURN zero, zero, positive infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO dfs(left of node)  
            SET right_no_camera, right_monitored, right_camera TO dfs(right of node)  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO minimum of (left_camera PLUS minimum of right_monitored AND right_camera), (right_camera PLUS minimum of left_monitored AND left_camera), (left_camera PLUS right_camera)  
            SET camera TO one PLUS minimum of left_no_camera, left_monitored, left_camera PLUS minimum of right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO dfs(root)  
        RETURN minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
