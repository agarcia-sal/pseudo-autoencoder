CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference LESS THAN maxPts  
                    SET numerator TO difference  
                ELSE  
                    SET numerator TO maxPts  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            END IF  
            SET first_term TO dfs WITH i PLUS one  
            SET second_term TO dfs WITH i PLUS one MINUS dfs WITH i PLUS maxPts PLUS one  
            RETURN first_term PLUS second_term DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs WITH zero  
    END FUNCTION  
END CLASS
