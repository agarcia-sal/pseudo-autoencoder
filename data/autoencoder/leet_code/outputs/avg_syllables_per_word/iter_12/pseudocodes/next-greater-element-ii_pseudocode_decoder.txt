CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of n elements each SET TO negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND element at position stack LAST ELEMENT OF nums IS LESS THAN element at position index OF nums  
                SET popped_index TO the LAST ELEMENT of stack REMOVED  
                SET element at position popped_index OF result TO element at position index OF nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
