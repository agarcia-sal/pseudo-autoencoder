CLASS Solution  
    FUNCTION minDistance(houses, k)  
        CALL houses SORT IN ASCENDING ORDER  
        SET n TO the LENGTH OF houses  
        
        CALL init cost AS a two dimensional list of size n by n FILLED WITH zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at position i and j TO cost element at position i PLUS one and j MINUS one PLUS houses element at position j MINUS houses element at position i  
            END FOR  
        END FOR  
        
        CALL init dp AS a two dimensional list of size n PLUS one by k PLUS one FILLED WITH infinity  
        SET dp element at position zero and zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    IF dp element at position i and j GREATER THAN dp element at position p and j MINUS one PLUS cost element at position p and i MINUS one  
                        SET dp element at position i and j TO dp element at position p and j MINUS one PLUS cost element at position p and i MINUS one  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp element at position n and k  
    END FUNCTION  
END CLASS
