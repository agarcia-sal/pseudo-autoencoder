CLASS Term  
    FUNCTION __init__(coefficient, variables)  
        SET self.coefficient TO coefficient  
        SET self.variables TO variables  
    END FUNCTION  

    FUNCTION __lt__(other)  
        IF the LENGTH OF self.variables NOT EQUALS the LENGTH OF other.variables  
            RETURN the LENGTH OF self.variables GREATER THAN the LENGTH OF other.variables  
        END IF  
        RETURN the JOINED string of self.variables SEPARATED BY 'MULTIPLIED BY' LESS THAN the JOINED string of other.variables SEPARATED BY 'MULTIPLIED BY'  
    END FUNCTION  

    FUNCTION __repr__()  
        IF self.variables IS empty  
            RETURN the STRING representation of self.coefficient  
        END IF  
        RETURN the STRING concatenation of self.coefficient, followed by the character MULTIPLIED BY, followed by the JOINED string of self.variables SEPARATED BY character MULTIPLIED BY  
    END FUNCTION  
END CLASS  

CLASS Expression  
    FUNCTION __init__()  
        SET self.terms TO a new default dictionary mapping to zero  
    END FUNCTION  

    FUNCTION add_term(term)  
        SET key TO the sorted tuple of term.variables  
        INCREMENT self.terms AT key BY term.coefficient  
    END FUNCTION  

    FUNCTION multiply(other)  
        SET result TO a new Expression  
        FOR each vars1 AND coeff1 IN the items of self.terms  
            FOR each vars2 AND coeff2 IN the items of other.terms  
                SET combined_vars TO the sorted list of the concatenation of list of vars1 AND list of vars2  
                CALL result.add_term WITH a new Term WITH coefficient equal to coeff1 MULTIPLIED BY coeff2 AND variables equal to combined_vars  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  

    FUNCTION add(other)  
        FOR each vars AND coeff IN the items of other.terms  
            CALL self.add_term WITH a new Term WITH coefficient coeff AND variables equal to the list of vars  
        END FOR  
        RETURN self  
    END FUNCTION  

    FUNCTION subtract(other)  
        FOR each vars AND coeff IN the items of other.terms  
            CALL self.add_term WITH a new Term WITH coefficient negative coeff AND variables equal to the list of vars  
        END FOR  
        RETURN self  
    END FUNCTION  

    FUNCTION evaluate(var_map)  
        SET result TO a new Expression  
        FOR each vars AND coeff IN the items of self.terms  
            SET new_vars TO an empty list  
            FOR each var IN vars  
                IF var IS IN var_map  
                    APPEND the STRING representation of var_map AT var TO new_vars  
                ELSE  
                    APPEND var TO new_vars  
                END IF  
            END FOR  
            IF new_vars IS empty  
                SET new_coeff TO one  
            ELSE  
                SET expression_string TO the concatenation of all elements in new_vars  
                SET new_coeff TO the evaluation of expression_string  
            END IF  
            CALL result.add_term WITH a new Term WITH coefficient coeff MULTIPLIED BY new_coeff AND variables as empty list  
        END FOR  
        RETURN result  
    END FUNCTION  

    FUNCTION simplify()  
        SET simplified_terms TO the list of new Term objects CREATED FROM pairs vars AND coeff IN self.terms WHERE coeff NOT EQUALS zero  
        SORT simplified_terms ACCORDING TO Term ordering  
        RETURN simplified_terms  
    END FUNCTION  

    FUNCTION __repr__()  
        RETURN the JOINED string WITH separator ' PLUS ' of the STRING representations of all terms IN the list returned by self.simplify()  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION basicCalculatorIV(expression, evalvars, evalints)  
        SET var_map TO a dictionary MAPPING each evalvars element TO the corresponding evalints element  
        SET tokens TO the list of all substrings IN expression MATCHING the regular expression for parentheses, plus, minus, multiply, digits, or word sequences  
        SET output TO an empty list  
        SET operators TO an empty list  

        FUNCTION apply_operator()  
            SET right TO the last element REMOVED FROM output  
            SET left TO the last element REMOVED FROM output  
            SET op TO the last element REMOVED FROM operators  
            IF op EQUALS plus character  
                APPEND the result of left.add(right) TO output  
            ELSE IF op EQUALS minus character  
                APPEND the result of left.subtract(right) TO output  
            ELSE IF op EQUALS multiply character  
                APPEND the result of left.multiply(right) TO output  
            END IF  
        END FUNCTION  

        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF tokens  
            SET token TO the element AT position i OF tokens  
            IF token IS composed only of digits  
                APPEND a new Expression TO output  
                CALL the last element IN output.add_term WITH a new Term WITH coefficient integer value of token AND empty list of variables  
            ELSE IF token IS composed only of alphabetic characters  
                APPEND a new Expression TO output  
                SET coef TO var_map GETTING token OR one if missing  
                IF coef EQUALS one  
                    IF token NOT IN var_map  
                        CALL the last element IN output.add_term WITH a new Term WITH coefficient one AND a list containing token  
                    ELSE  
                        CALL the last element IN output.add_term WITH a new Term WITH coefficient one AND empty list  
                    END IF  
                ELSE  
                    CALL the last element IN output.add_term WITH a new Term WITH coefficient coef AND empty list  
                END IF  
            ELSE IF token EQUALS left parenthesis  
                APPEND token TO operators  
            ELSE IF token EQUALS right parenthesis  
                WHILE operators IS NOT empty AND the last element IN operators NOT EQUALS left parenthesis  
                    CALL apply_operator()  
                END WHILE  
                REMOVE the last element FROM operators  
            ELSE IF token IS one of plus, minus, multiply characters  
                WHILE operators IS NOT empty AND the last element IN operators NOT EQUALS left parenthesis AND (token IS plus OR minus OR the last element IN operators EQUALS multiply)  
                    CALL apply_operator()  
                END WHILE  
                APPEND token TO operators  
            END IF  
            INCREMENT i BY one  
        END WHILE  

        WHILE operators IS NOT empty  
            CALL apply_operator()  
        END WHILE  

        SET simplified_terms TO the result of calling simplify ON the first element OF output  
        SET result_strings TO an empty list  
        FOR each term IN simplified_terms  
            IF term.variables IS NOT empty  
                SET term_string TO the STRING concatenation of term.coefficient, the character MULTIPLIED BY, and the JOINED string of term.variables SEPARATED BY the character MULTIPLIED BY  
            ELSE  
                SET term_string TO the STRING representation of term.coefficient  
            END IF  
            APPEND term_string TO result_strings  
        END FOR  
        RETURN result_strings  
    END FUNCTION  
END CLASS
