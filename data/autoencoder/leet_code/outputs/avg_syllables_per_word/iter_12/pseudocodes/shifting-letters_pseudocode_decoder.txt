CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_character_value TO the position value of character at position index of s MINUS the position value of character a  
            SET shifted_value TO original_character_value PLUS element at position index of shifts  
            SET wrapped_value TO shifted_value MODULO twenty six  
            SET new_character_value TO wrapped_value PLUS the position value of character a  
            SET new_character TO the character corresponding to new_character_value  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
