CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero, next EQUALS none  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new ListNode created with the element at position zero of values  
    SET p TO head  
    FOR each val IN all elements of values starting from position one  
        SET node TO a new ListNode created with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val property of p1 EQUALS the val property of p2 AND is_same_list with parameters p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers with parameters l1, l2  
        FUNCTION pushIntoStack with parameter node  
            SET stack TO an empty list  
            WHILE node IS NOT none  
                APPEND the val property of node TO stack  
                SET node TO the next property of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO pushIntoStack with parameter l1  
        SET stack2 TO pushIntoStack with parameter l2  

        SET carry TO zero  
        SET result TO none  

        WHILE stack1 IS NOT empty OR stack2 IS NOT empty OR carry IS NOT zero  
            IF stack1 IS NOT empty  
                SET val1 TO the last element removed FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 IS NOT empty  
                SET val2 TO the last element removed FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  

            SET new_node TO a new ListNode created with digit  
            SET the next property of new_node TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
