CLASS Solution  
    FUNCTION leastInterval(tasks, n)  
        SET task_counts TO a frequency count of tasks  
        SET max_heap TO an empty list  
        FOR each count IN the values of task_counts  
            APPEND negative count TO max_heap  
        END FOR  
        CONVERT max_heap INTO a heap structure  
        
        SET time TO zero  
        SET queue TO an empty list to hold pairs of cooling time and task count  
        
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND the first element at position zero of queue at position zero EQUALS time  
                REMOVE first element FROM queue AND unpack into _, count  
                IF count LESS THAN zero  
                    PUSH count INTO max_heap heap  
                END IF  
            END WHILE  
            
            IF max_heap is not empty  
                REMOVE the smallest element FROM max_heap heap AND SET to count  
                INCREMENT count BY one  
                IF count LESS THAN zero  
                    APPEND a pair of time PLUS n PLUS one AND count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
