CLASS Solution  
    FUNCTION cherryPickup(grid)  
        SET n TO the LENGTH OF grid  
        IF the element at position zero zero of grid EQUALS negative one OR the element at position n MINUS one n MINUS one of grid EQUALS negative one  
            RETURN zero  
        END IF  
        
        CALL auxiliary FUNCTION initialize_dp_table WITH n TO CREATE dp  
        SET the element at position zero zero zero of dp TO the element at position zero zero of grid  
        
        SET directions TO the list containing the pairs zero one and one zero  
        
        FOR t FROM one TO two MULTIPLIED BY n MINUS one MINUS one  
            FOR i1 FROM the MAXIMUM BETWEEN zero AND t MINUS (n MINUS one) TO the MINIMUM BETWEEN n AND t PLUS one MINUS one  
                FOR i2 FROM the MAXIMUM BETWEEN zero AND t MINUS (n MINUS one) TO the MINIMUM BETWEEN n AND t PLUS one MINUS one  
                    SET j1 TO t MINUS i1  
                    SET j2 TO t MINUS i2  
                    IF the element at position i1 j1 of grid EQUALS negative one OR the element at position i2 j2 of grid EQUALS negative one  
                        CONTINUE to next iteration of inner loop  
                    END IF  
                    
                    SET prev_max TO negative one  
                    FOR each pair di1 dj1 IN directions  
                        FOR each pair di2 dj2 IN directions  
                            SET pi1 TO i1 MINUS di1  
                            SET pj1 TO j1 MINUS dj1  
                            SET pi2 TO i2 MINUS di2  
                            SET pj2 TO j2 MINUS dj2  
                            IF pi1 GREATER THAN OR EQUAL TO zero AND pi1 LESS THAN n AND pj1 GREATER THAN OR EQUAL TO zero AND pj1 LESS THAN n AND pi2 GREATER THAN OR EQUAL TO zero AND pi2 LESS THAN n AND pj2 GREATER THAN OR EQUAL TO zero AND pj2 LESS THAN n  
                                SET prev_max TO the MAXIMUM BETWEEN prev_max AND the element at position pi1 pi2 pj1 of dp  
                            END IF  
                        END FOR  
                    END FOR  
                    
                    IF prev_max GREATER THAN OR EQUAL TO zero  
                        IF i1 EQUALS i2  
                            SET cherries TO the element at position i1 j1 of grid  
                        ELSE  
                            SET cherries TO the element at position i1 j1 of grid PLUS the element at position i2 j2 of grid  
                        END IF  
                        SET the element at position i1 i2 j1 of dp TO prev_max PLUS cherries  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the MAXIMUM BETWEEN zero AND the element at position n MINUS one n MINUS one n MINUS one of dp  
    END FUNCTION  
    
    FUNCTION initialize_dp_table(n)  
        CREATE a three dimensional list dp OF dimensions n BY n BY n FILLED WITH negative one  
        RETURN dp  
    END FUNCTION  
END CLASS
