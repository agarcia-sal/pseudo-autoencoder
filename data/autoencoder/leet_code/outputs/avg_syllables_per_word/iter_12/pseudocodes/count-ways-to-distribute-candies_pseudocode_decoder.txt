CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS a two-dimensional list of zeroes WITH (n PLUS one) rows AND (k PLUS one) columns  
        SET element at position zero of row zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of row i of dp TO (j MULTIPLIED BY element at position j of row (i MINUS one) of dp PLUS element at position (j MINUS one) of row (i MINUS one) of dp) MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position k of row n of dp  
    END FUNCTION  
END CLASS
