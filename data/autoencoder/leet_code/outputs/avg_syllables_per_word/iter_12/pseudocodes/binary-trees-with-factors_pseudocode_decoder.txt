CLASS Solution  
    FUNCTION numFactoredBinaryTrees(arr)  
        SET MOD TO one billion PLUS seven  
        SORT arr in ascending order  
        SET dp TO empty map from number TO one  
        FOR each number IN arr  
            SET dp at position number TO one  
        END FOR  
        FOR index i FROM zero TO LENGTH OF arr MINUS one  
            SET num TO element at position i of arr  
            FOR index j FROM zero TO i MINUS one  
                SET factor_candidate TO element at position j of arr  
                IF num MODULO factor_candidate EQUALS zero  
                    SET right_child TO num DIVIDED BY factor_candidate  
                    IF dp HAS key right_child  
                        SET product TO dp at factor_candidate MULTIPLIED BY dp at right_child  
                        SET dp at num TO dp at num PLUS product MODULO MOD  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET total_trees TO zero  
        FOR each value IN dp values  
            INCREMENT total_trees BY value  
        END FOR  
        RETURN total_trees MODULO MOD  
    END FUNCTION  
END CLASS
