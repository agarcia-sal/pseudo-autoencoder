CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO True  
        FOR i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less  
                IF the element at position i of nums GREATER THAN the element at position i PLUS one of nums  
                    SET temporary variable TO the element at position i of nums  
                    SET the element at position i of nums TO the element at position i PLUS one of nums  
                    SET the element at position i PLUS one of nums TO temporary variable  
                END IF  
            ELSE  
                IF the element at position i of nums LESS THAN the element at position i PLUS one of nums  
                    SET temporary variable TO the element at position i of nums  
                    SET the element at position i of nums TO the element at position i PLUS one of nums  
                    SET the element at position i PLUS one of nums TO temporary variable  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
