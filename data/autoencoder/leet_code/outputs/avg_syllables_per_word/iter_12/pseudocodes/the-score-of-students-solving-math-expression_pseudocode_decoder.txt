CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each char IN expression  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS integer value of char  
                ELSE  
                    APPEND num TO tokens  
                    APPEND char TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS multiplication sign  
                    SET element at position i MINUS one of tokens TO element at position i MINUS one MULTIPLIED BY element at position i PLUS one of tokens  
                    REMOVE elements from position i TO position i PLUS two FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS addition sign  
                    SET result TO result PLUS element at position i PLUS one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF expression IS composed only OF digits  
                RETURN set containing integer value of expression  
            END IF  
            
            SET results TO empty set  
            FOR index FROM one TO the LENGTH OF expression MINUS two STEP two  
                SET left_results TO get_possible_answers(substring of expression FROM start TO index)  
                SET right_results TO get_possible_answers(substring of expression FROM index PLUS one TO end)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF element at position index of expression EQUALS addition sign  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
