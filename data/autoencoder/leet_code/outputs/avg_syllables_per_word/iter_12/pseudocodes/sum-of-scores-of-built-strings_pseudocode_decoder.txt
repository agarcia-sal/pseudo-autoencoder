CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        SET lcp TO a list of n elements each SET TO zero  
        SET l TO zero  
        SET r TO zero  
        
        FOR index i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r  
                SET lcp element at position i TO the MINIMUM OF r MINUS i PLUS one AND lcp element at position i MINUS l  
            END IF  
            
            WHILE i PLUS lcp element at position i LESS THAN n AND character at position lcp element at position i of s EQUALS character at position i PLUS lcp element at position i of s  
                INCREMENT lcp element at position i BY one  
            END WHILE  
            
            IF i PLUS lcp element at position i MINUS one GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp element at position i MINUS one  
            END IF  
        END FOR  
        
        RETURN the SUM OF all elements in lcp PLUS n  
    END FUNCTION  
END CLASS
