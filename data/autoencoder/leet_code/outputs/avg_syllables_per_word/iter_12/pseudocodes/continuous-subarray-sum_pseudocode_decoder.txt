CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO dictionary with key zero SET TO value minus one  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF remainder_map HAS key current_sum  
                IF i MINUS value AT key current_sum IN remainder_map GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                SET value AT key current_sum IN remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
