CLASS Solution  
    FUNCTION profitableSchemes(n, minProfit, group, profit)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of lists containing zeros with dimensions of n PLUS one BY minProfit PLUS one  
        FOR i FROM zero TO n  
            SET element at position zero of dp at position i TO one  
        END FOR  
        
        FOR each p, g IN profit AND group simultaneously  
            FOR i FROM n DOWN TO g  
                FOR j FROM minProfit DOWN TO zero  
                    SET max_index TO j MINUS p  
                    IF max_index LESS THAN zero  
                        SET max_index TO zero  
                    END IF  
                    SET dp at position i at position j TO dp at position i at position j PLUS dp at position i MINUS g at position max_index  
                    SET dp at position i at position j TO dp at position i at position j MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n at position minProfit  
    END FUNCTION  
END CLASS
