CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty set  
        FOR each pair (x, y) IN zip of fronts and backs  
            IF x EQUALS y  
                APPEND x TO same  
            END IF  
        END FOR  
        SET all_numbers TO an empty set  
        FOR each number IN fronts  
            APPEND number TO all_numbers  
        END FOR  
        FOR each number IN backs  
            APPEND number TO all_numbers  
        END FOR  
        SET good_numbers TO an empty set  
        FOR each number IN all_numbers  
            IF number NOT IN same  
                APPEND number TO good_numbers  
            END IF  
        END FOR  
        IF good_numbers IS NOT empty  
            SET minimum_good TO the minimum value IN good_numbers  
            RETURN minimum_good  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
