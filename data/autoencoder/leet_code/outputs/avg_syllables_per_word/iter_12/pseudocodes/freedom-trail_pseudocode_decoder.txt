CLASS Solution  
    FUNCTION findRotateSteps(ring, key)  
        CREATE a dictionary named char_positions with default value as an empty list  
        FOR each index i FROM zero TO the LENGTH OF ring MINUS one  
            SET character TO element at position i of ring  
            APPEND i TO the list stored at char_positions for character  
        END FOR  
        
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i, prev_pos)  
            IF i EQUALS the LENGTH OF key  
                RETURN zero  
            END IF  
            
            SET min_steps TO infinity  
            FOR each position pos IN the list at char_positions for the character at position i of key  
                SET clockwise TO (pos MINUS prev_pos) MODULO n  
                SET anticlockwise TO (prev_pos MINUS pos) MODULO n  
                SET steps TO the MINIMUM OF clockwise AND anticlockwise PLUS one PLUS dp(i PLUS one, pos)  
                SET min_steps TO the MINIMUM OF min_steps AND steps  
            END FOR  
            
            RETURN min_steps  
        END FUNCTION  
        
        RETURN dp(zero, zero)  
    END FUNCTION  
END CLASS
