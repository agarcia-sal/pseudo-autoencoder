CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO a list containing n elements each being a list of n zeros  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                SET the element at position i of the list at position top of matrix TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                SET the element at position right of the list at position i of matrix TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top LESS THAN OR EQUAL TO bottom  
                FOR i FROM right TO left STEP MINUS one  
                    SET the element at position i of the list at position bottom of matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left LESS THAN OR EQUAL TO right  
                FOR i FROM bottom TO top STEP MINUS one  
                    SET the element at position left of the list at position i of matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
