CLASS Solution  
    FUNCTION smallestMissingValueSubtree(parents, nums)  
        SET n TO the LENGTH OF parents  
        SET tree TO an EMPTY map FROM keys TO lists  
        
        FOR index FROM one TO n MINUS one  
            APPEND index TO list at key parents element at position index IN tree  
        END FOR  
        
        SET node_with_one TO negative one  
        FOR index FROM zero TO n MINUS one  
            IF element at position index of nums EQUALS one  
                SET node_with_one TO index  
                BREAK  
            END IF  
        END FOR  
        
        IF node_with_one EQUALS negative one  
            SET result TO an EMPTY list  
            FOR index FROM zero TO n MINUS one  
                APPEND one TO result  
            END FOR  
            RETURN result  
        END IF  
        
        SET result TO an EMPTY list  
        FOR index FROM zero TO n MINUS one  
            APPEND one TO result  
        END FOR  
        
        SET visited TO an EMPTY set  
        
        FUNCTION dfs(node)  
            ADD node TO visited  
            SET genetic_values TO a new set containing element at position node of nums  
            FOR each child IN list at key node in tree  
                IF child NOT IN visited  
                    SET child_values TO dfs(child)  
                    UNION genetic_values WITH child_values  
                END IF  
            END FOR  
            SET smallest_missing TO one  
            WHILE smallest_missing IS IN genetic_values  
                INCREMENT smallest_missing BY one  
            END WHILE  
            SET element at position node in result TO smallest_missing  
            RETURN genetic_values  
        END FUNCTION  
        
        SET path TO an EMPTY list  
        SET current TO node_with_one  
        WHILE current NOT EQUALS negative one  
            APPEND current TO path  
            SET current TO element at position current in parents  
        END WHILE  
        
        SET path_set TO a new set containing all elements in path  
        
        FOR each node IN path starting from last element TO first element  
            IF node IS IN visited  
                CONTINUE  
            END IF  
            CALL dfs(node)  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
