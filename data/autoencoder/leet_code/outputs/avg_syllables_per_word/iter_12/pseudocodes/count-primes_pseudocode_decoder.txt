CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        
        SET is_prime TO list of boolean true values with length n  
        SET element at position zero of is_prime TO false  
        SET element at position one of is_prime TO false  

        FOR start FROM two TO the integer part of the square root of n PLUS one  
            IF element at position start of is_prime EQUALS true  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one STEP start  
                    SET element at position multiple of is_prime TO false  
                END FOR  
            END IF  
        END FOR  

        SET prime_count TO zero  
        FOR each value IN is_prime  
            IF value EQUALS true  
                INCREMENT prime_count BY one  
            END IF  
        END FOR  

        RETURN prime_count  
    END FUNCTION  
END CLASS
