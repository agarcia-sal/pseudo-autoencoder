CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE first element FROM queue AND SET it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree with p.left and q.left AND the RESULT OF is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN empty string  
            END IF  
            SET result TO the string representation of node.val  
            IF node.left IS NOT none OR node.right IS NOT none  
                SET result TO result PLUS an opening parenthesis PLUS the RESULT OF helper with node.left PLUS a closing parenthesis  
            END IF  
            IF node.right IS NOT none  
                SET result TO result PLUS an opening parenthesis PLUS the RESULT OF helper with node.right PLUS a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN the RESULT OF helper with root  
    END FUNCTION  
END CLASS
