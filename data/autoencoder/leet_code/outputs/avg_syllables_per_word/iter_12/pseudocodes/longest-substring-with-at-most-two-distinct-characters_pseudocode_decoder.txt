CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        INITIALIZE char_count AS an auxiliary FUNCTION defaultdict_of_integer_counts  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of the character at position right of s IN char_count BY one  
            
            WHILE the number of distinct keys in char_count IS GREATER THAN two  
                DECREMENT the count of the character at position left of s IN char_count BY one  
                IF the count of the character at position left of s IN char_count EQUALS zero  
                    REMOVE the entry for the character at position left of s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
