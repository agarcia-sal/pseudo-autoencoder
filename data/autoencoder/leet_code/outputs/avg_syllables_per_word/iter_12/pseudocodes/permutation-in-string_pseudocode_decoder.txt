CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET len_s1 TO the LENGTH OF s1  
        SET len_s2 TO the LENGTH OF s2  
        
        IF len_s1 GREATER THAN len_s2  
            RETURN False  
        END IF  
        
        SET count_s1 TO the frequency count of characters in s1  
        SET count_s2 TO the frequency count of characters in substring of s2 from position zero TO length len_s1 MINUS one  
        
        IF count_s1 EQUALS count_s2  
            RETURN True  
        END IF  
        
        FOR index FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of character at position index of s2 in count_s2 BY one  
            DECREMENT the count of character at position index MINUS len_s1 of s2 in count_s2 BY one  
            
            IF the count of character at position index MINUS len_s1 of s2 in count_s2 EQUALS zero  
                REMOVE that character from count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
