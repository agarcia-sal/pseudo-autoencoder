CLASS Solution  
    FUNCTION encode(s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN length of string representation of count PLUS length of substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO the segment of s from position i TO position j inclusive  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR k FROM one TO integer division of n BY two PLUS one  
                IF n MODULO k EQUALS zero  
                    SET repeat_count TO integer division of n BY k  
                    IF substring EQUALS repetition of substring from start TO position k MINUS one repeated repeat_count times  
                        SET encoded_candidate TO concatenation of repeat_count converted to string PLUS left bracket PLUS dp(i, i PLUS k MINUS one) PLUS right bracket  
                        IF length of encoded_candidate LESS THAN length of shortest  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR k FROM i TO j MINUS one  
                SET left_encoded TO dp(i, k)  
                SET right_encoded TO dp(k PLUS one, j)  
                SET combined TO concatenation of left_encoded PLUS right_encoded  
                IF length of combined LESS THAN length of shortest  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN dp(zero, length of s MINUS one)  
    END FUNCTION  
END CLASS
