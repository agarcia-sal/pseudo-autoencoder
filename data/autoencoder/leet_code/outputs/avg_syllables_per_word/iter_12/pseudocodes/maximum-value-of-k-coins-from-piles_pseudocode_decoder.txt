CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a LIST containing n PLUS one ELEMENTS each being a LIST containing k PLUS one ZEROS  
  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp at position i and j TO the MAXIMUM OF element at position i and j of dp AND element at position i MINUS one and j MINUS l of dp PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
  
        RETURN element at position n and k of dp  
    END FUNCTION  
END CLASS
