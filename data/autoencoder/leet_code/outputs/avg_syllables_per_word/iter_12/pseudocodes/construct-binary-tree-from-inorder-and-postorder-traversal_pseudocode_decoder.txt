CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder EQUALS none OR LENGTH OF inorder EQUALS zero OR postorder EQUALS none OR LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        SET root_val TO last element removed FROM postorder  
        SET root TO new TreeNode with root_val  
        SET root_index TO position of root_val in inorder  
        SET root.right TO self.buildTree(sublist of inorder from position root_index PLUS one to end, postorder)  
        SET root.left TO self.buildTree(sublist of inorder from start to position root_index MINUS one PLUS one, postorder)  
        RETURN root  
    END FUNCTION  
END CLASS
