CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list containing the elements (zero, one), (one, zero), (zero, negative one), (negative one, zero)  
        SET min_heap TO list containing the tuple zero, zero, zero representing cost zero at position zero, zero  
        SET visited TO set containing the tuple zero, zero  
        
        WHILE min_heap is not empty  
            REMOVE the element with the smallest cost from min_heap and SET cost, x, y TO its values  
            
            IF the tuple x, y EQUALS the tuple m MINUS one, n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple nx, ny NOT IN visited  
                    ADD the tuple nx, ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx, ny of grid  
                    ADD the tuple new_cost, nx, ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
