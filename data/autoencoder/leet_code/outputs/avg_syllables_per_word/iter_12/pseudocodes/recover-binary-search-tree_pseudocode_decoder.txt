CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue INTO node  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET left child of node TO new TreeNode with val as element at position i of values  
            APPEND left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET right child of node TO new TreeNode with val as element at position i of values  
            APPEND right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF val of p NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with p.left AND q.left AND is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION recoverTree with parameter root  
        SET first TO None  
        SET second TO None  
        SET prev TO None  
        SET curr TO root  
        WHILE curr IS NOT None  
            IF left child of curr EQUALS None  
                IF prev IS NOT None AND val of prev GREATER THAN val of curr  
                    IF first EQUALS None  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO right child of curr  
            ELSE  
                SET pred TO left child of curr  
                WHILE right child of pred IS NOT None AND right child of pred NOT EQUALS curr  
                    SET pred TO right child of pred  
                END WHILE  
                IF right child of pred EQUALS None  
                    SET right child of pred TO curr  
                    SET curr TO left child of curr  
                ELSE  
                    SET right child of pred TO None  
                    IF prev IS NOT None AND val of prev GREATER THAN val of curr  
                        IF first EQUALS None  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO right child of curr  
                END IF  
            END IF  
        END WHILE  
        IF first IS NOT None AND second IS NOT None  
            SET temp_val TO val of first  
            SET val of first TO val of second

    END FUNCTION  
END CLASS
