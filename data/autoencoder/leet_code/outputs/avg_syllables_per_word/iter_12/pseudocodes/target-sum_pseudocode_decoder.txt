CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS IN memo  
                RETURN the VALUE OF memo AT the KEY tuple of index AND current_sum  
            END IF  
            SET positive TO dfs(index PLUS one, current_sum PLUS the element at position index of nums, memo)  
            SET negative TO dfs(index PLUS one, current_sum MINUS the element at position index of nums, memo)  
            SET memo AT the KEY tuple of index AND current_sum TO positive PLUS negative  
            RETURN the VALUE OF memo AT the KEY tuple of index AND current_sum  
        END FUNCTION  
        RETURN dfs(zero, zero, an empty dictionary)  
    END FUNCTION  
END CLASS
