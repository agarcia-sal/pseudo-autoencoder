CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL init2Ddp WITH PARAMETERS m, n AND SET RESULT TO dp  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position i, j of dp TO the element at position i MINUS one, j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i, j of dp TO the MAXIMUM OF the element at position i MINUS one, j of dp AND the element at position i, j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO the element at position m, n of dp  
        
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  

    FUNCTION init2Ddp(m, n)  
        SET dp TO an empty list  
        FOR i FROM zero TO m  
            SET row TO a list containing n PLUS one zeros  
            APPEND row TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
