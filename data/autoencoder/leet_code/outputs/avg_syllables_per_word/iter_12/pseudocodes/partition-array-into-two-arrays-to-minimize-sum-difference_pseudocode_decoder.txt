CLASS Solution
    FUNCTION minimumDifference(nums)
        SET n TO the LENGTH OF nums DIVIDED BY two
        SET total_sum TO the SUM OF nums
        SET target TO total_sum DIVIDED BY two
        SET min_diff TO positive infinity
        
        SET left_half TO the FIRST n elements OF nums
        SET right_half TO the ELEMENTS FROM position n TO the END OF nums
        
        SET left_sums TO a LIST of EMPTY DICTIONARIES with LENGTH n PLUS one
        SET right_sums TO a LIST of EMPTY DICTIONARIES with LENGTH n PLUS one
        
        FOR i FROM zero TO n
            FOR each combo IN all COMBINATIONS OF left_half TAKEN i AT A TIME
                SET subset_sum TO the SUM OF combo
                IF subset_sum IS NOT A KEY IN left_sums AT position i
                    SET key subset_sum IN left_sums AT position i TO zero
                END IF
                INCREMENT the value OF subset_sum IN left_sums AT position i BY one
            END FOR
            
            FOR each combo IN all COMBINATIONS OF right_half TAKEN i AT A TIME
                SET subset_sum TO the SUM OF combo
                IF subset_sum IS NOT A KEY IN right_sums AT position i
                    SET key subset_sum IN right_sums AT position i TO zero
                END IF
                INCREMENT the value OF subset_sum IN right_sums AT position i BY one
            END FOR
        END FOR
        
        FOR i FROM zero TO n
            SET left_values TO the SORTED LIST OF KEYS IN left_sums AT position i
            SET right_values TO the SORTED LIST OF KEYS IN right_sums AT position n MINUS i
            
            SET j TO zero
            SET k TO the LENGTH OF right_values MINUS one
            
            WHILE j IS LESS THAN the LENGTH OF left_values AND k IS GREATER THAN OR EQUAL TO zero
                SET current_sum TO the ELEMENT AT position j IN left_values PLUS the ELEMENT AT position k IN right_values
                SET min_diff TO the MINIMUM OF min_diff AND the ABSOLUTE VALUE OF total_sum MINUS two TIMES current_sum
                
                IF current_sum IS LESS THAN target
                    INCREMENT j BY one
                ELSE IF current_sum IS GREATER THAN target
                    DECREMENT k BY one
                ELSE
                    RETURN zero
                END IF
            END WHILE
        END FOR
        
        RETURN min_diff
    END FUNCTION
END CLASS
