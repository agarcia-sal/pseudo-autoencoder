CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of length fifteen filled with infinity  
        FOR each pair f AND r IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF index  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK the loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position index TO the minimum of min_times at position index AND total_time  
            END FOR  
        END FOR  

        SET dp TO a list of length numLaps PLUS one filled with infinity  
        SET dp at position zero TO zero  

        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the minimum of i MINUS one AND fourteen  
                SET dp at position i TO the minimum of dp at position i AND dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  

        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
