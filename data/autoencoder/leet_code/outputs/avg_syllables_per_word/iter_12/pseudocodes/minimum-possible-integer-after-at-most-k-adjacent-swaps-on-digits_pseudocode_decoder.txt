CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the list formed by each character in num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero  
            SET swapped TO False  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF the element at position i of num GREATER THAN the element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF the element at position j of num LESS THAN min_digit  
                            SET min_digit TO the element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the slice of num from i TO min_index PLUS one TO the list formed by the element at min_index FOLLOWED BY the slice of num from i TO min_index  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO True  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS False  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num  
    END FUNCTION  
END CLASS
