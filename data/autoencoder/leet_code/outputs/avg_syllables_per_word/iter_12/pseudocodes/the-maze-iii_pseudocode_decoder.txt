CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list containing tuple zero, one, character 'r'  
        APPEND tuple one, zero, character 'd' TO directions  
        APPEND tuple zero, negative one, character 'l' TO directions  
        APPEND tuple negative one, zero, character 'u' TO directions  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF element at position zero of maze  
        SET start TO tuple of elements of ball  
        SET hole TO tuple of elements of hole  
        SET pq TO list containing tuple zero, empty string, start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE element with smallest distance from pq AND UNPACK INTO dist, path, position  
            UNPACK position INTO x, y  
            
            IF x EQUALS first element of hole AND y EQUALS second element of hole  
                RETURN path  
            END IF  
            
            IF tuple x, y IS IN visited  
                CONTINUE to next iteration  
            END IF  
            
            ADD tuple x, y TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS first element of hole AND new_y EQUALS second element of hole  
                        BREAK out of the loop  
                    END IF  
                END WHILE  
                
                IF tuple new_x, new_y IS NOT IN visited  
                    INSERT tuple dist PLUS count, path concatenated with direction, tuple new_x, new_y INTO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN string "impossible"  
    END FUNCTION  
END CLASS
