CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO result of calling a FUNCTION that creates a two dimensional list of zeros with number of rows EQUAL TO the LENGTH OF nums1 PLUS one and number of columns EQUAL TO the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF element at position i MINUS one of nums1 EQUALS element at position j MINUS one of nums2  
                    SET element at position i, j of dp TO element at position i MINUS one, j MINUS one of dp PLUS one  
                    SET max_length TO the greater value BETWEEN max_length AND element at position i, j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
