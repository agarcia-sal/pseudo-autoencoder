CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO a mapping with key zero and value one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            IF current_sum MINUS k EXISTS AS key IN cumulative_sum_freq  
                INCREMENT count BY the value at key current_sum MINUS k in cumulative_sum_freq  
            END IF  
            IF current_sum EXISTS AS key IN cumulative_sum_freq  
                INCREMENT the value at key current_sum in cumulative_sum_freq BY one  
            ELSE  
                SET the value at key current_sum in cumulative_sum_freq TO one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
