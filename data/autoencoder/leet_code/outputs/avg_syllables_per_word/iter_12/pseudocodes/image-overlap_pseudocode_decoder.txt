CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img1 at position j EQUALS one  
                    APPEND tuple of i and j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img2 at position j EQUALS one  
                    APPEND tuple of i and j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a new dictionary with default integer zero  
        FOR each pair x1, y1 IN points1  
            FOR each pair x2, y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value of overlap_count at key tuple of dx and dy BY one  
            END FOR  
        END FOR  
        SET max_overlap TO zero  
        FOR each value IN the values of overlap_count  
            IF value GREATER THAN max_overlap  
                SET max_overlap TO value  
            END IF  
        END FOR  
        RETURN max_overlap  
    END FUNCTION  
END CLASS
