CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold EQUALS zero  
            RETURN a list containing True repeated the LENGTH OF queries times  
        END IF  

        SET parent TO a list of integers FROM zero TO n  

        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY  
                SET element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR i FROM threshold PLUS one TO n  
            FOR j FROM i MULTIPLIED BY two TO n STEP i  
                CALL union(i, j)  
            END FOR  
        END FOR  

        SET result TO an empty list  
        FOR each pair (a, b) IN queries  
            APPEND the expression find(a) EQUALS find(b) TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
