CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode INITIALIZED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO end  
        SET node TO new ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1, p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH PARAMETER head  
        SET dummy TO new ListNode INITIALIZED WITH zero  
        SET dummy.next TO head  
        SET current TO dummy  
        WHILE current.next NOT EQUALS none AND current.next.next NOT EQUALS none  
            SET first TO current.next  
            SET second TO current.next.next  
            SET current.next TO second  
            SET first.next TO second.next  
            SET second.next TO first  
            SET current TO current.next.next  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
