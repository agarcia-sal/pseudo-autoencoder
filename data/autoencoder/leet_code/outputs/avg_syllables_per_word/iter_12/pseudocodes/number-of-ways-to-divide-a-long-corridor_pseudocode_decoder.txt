CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each index FROM zero TO length of corridor MINUS one  
            IF element at position index of corridor EQUALS character S  
                APPEND index TO seats  
            END IF  
        END FOR  
        
        IF length of seats MODULO two NOT EQUALS zero OR length of seats LESS THAN two  
            RETURN zero  
        END IF  
        
        SET ways TO one  
        FOR index FROM two TO length of seats MINUS one STEP two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
