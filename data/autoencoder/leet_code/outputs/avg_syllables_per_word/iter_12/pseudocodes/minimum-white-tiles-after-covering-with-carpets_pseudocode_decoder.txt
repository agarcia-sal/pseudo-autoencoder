CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        CALL initializeDP WITH n AND numCarpets, STORE RESULT IN dp  
        FOR index FROM one TO n  
            SET dp at position index and zero TO dp at position index minus one and zero PLUS one IF the element at position index minus one of floor EQUALS one ELSE zero  
        END FOR  
        FOR carpet_count FROM one TO numCarpets  
            FOR index FROM one TO n  
                SET option_one TO dp at position index minus one and carpet_count PLUS one IF the element at position index minus one of floor EQUALS one ELSE zero  
                SET option_two TO dp at position maximum between zero AND index minus carpetLen and carpet_count minus one  
                SET dp at position index and carpet_count TO the MINIMUM OF option_one AND option_two  
            END FOR  
        END FOR  
        RETURN dp at position n and numCarpets  
    END FUNCTION  

    FUNCTION initializeDP(n, numCarpets)  
        CREATE a two dimensional list dp with n plus one rows and numCarpets plus one columns, EACH element initialized TO zero  
        RETURN dp  
    END FUNCTION  
END CLASS
