CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR i FROM zero TO the smaller VALUE BETWEEN k MINUS one AND the LENGTH OF nums1 MINUS one  
            CALL helper FUNCTION heap_push ON min_heap WITH the TUPLE composed of the sum of element at position i of nums1 PLUS element at position zero of nums2, i, and zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap IS NOT empty AND the LENGTH OF result LESS THAN k  
            CALL helper FUNCTION heap_pop ON min_heap AND UNPACK THE VALUES TO ignore_variable, i, j  
            APPEND list composed of element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                CALL helper FUNCTION heap_push ON min_heap WITH the TUPLE composed of the sum of element at position i of nums1 PLUS element at position j PLUS one of nums2, i, and j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
