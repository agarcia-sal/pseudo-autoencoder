CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an empty dictionary  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n  
                RETURN one  
            END IF  
            
            IF the PAIR of index AND prev_len EXISTS IN memo  
                RETURN the VALUE at PAIR of index AND prev_len IN memo  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index  
                IF the CHARACTER at index of num EQUALS zero  
                    BREAK the LOOP  
                END IF  
                
                SET current_num TO the SUBSTRING of num FROM index TO index PLUS length  
                
                IF prev_len EQUALS zero OR current_num GREATER THAN OR EQUAL TO the SUBSTRING of num FROM index MINUS prev_len TO index  
                    SET count TO count PLUS dp WITH PARAMETERS index PLUS length AND length MODULO MOD  
                END IF  
            END FOR  
            
            SET the VALUE at PAIR of index AND prev_len IN memo TO count  
            RETURN count  
        END FUNCTION  
        
        RETURN dp WITH PARAMETERS zero AND zero  
    END FUNCTION  
END CLASS
