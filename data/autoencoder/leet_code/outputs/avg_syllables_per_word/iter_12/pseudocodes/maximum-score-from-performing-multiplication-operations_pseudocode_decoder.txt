CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeroes with LENGTH m PLUS one  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zeroes with LENGTH m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS (i MINUS left)  
                SET mul TO element at position i of multipliers  
                SET value_one TO mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp  
                SET value_two TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                SET element at position left of new_dp TO the GREATER OF value_one AND value_two  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
