CLASS Trie  
    FUNCTION __init__()  
        SET children TO list of length twenty six filled with None  
        SET is_end TO False  
    END FUNCTION  

    FUNCTION insert(w)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the Unicode code point of c MINUS the Unicode code point of character 'a'  
            IF the element at position idx of node.children EQUALS None  
                SET the element at position idx of node.children TO new instance of Trie  
            END IF  
            SET node TO the element at position idx of node.children  
        END FOR  
        SET node.is_end TO True  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the LENGTH OF w  
                RETURN True  
            END IF  
            SET node TO trie  
            FOR i FROM start TO LENGTH OF w MINUS one  
                SET idx TO the Unicode code point of the character at position i of w MINUS the Unicode code point of character 'a'  
                IF the element at position idx of node.children EQUALS None  
                    RETURN False  
                END IF  
                SET node TO the element at position idx of node.children  
                IF node.is_end AND dfs(w, i PLUS one)  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET trie TO new instance of Trie  
        SET ans TO empty list  
        SORT words by their length in ascending order  
        FOR each word w IN words  
            IF w NOT EQUALS empty string  
                IF dfs(w, zero)  
                    APPEND w TO ans  
                ELSE  
                    CALL trie.insert(w)  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
