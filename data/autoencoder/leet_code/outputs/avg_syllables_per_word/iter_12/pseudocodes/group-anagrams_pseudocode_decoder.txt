CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a new mapping with default value as an empty list  
        
        FOR each s IN strs  
            SET sorted_str TO the result of sorting characters of s and joining them into a single string  
            APPEND s TO the list at key sorted_str in anagrams  
        END FOR  
        
        SET result TO the list of all values in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
