CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND the element at the last position of stack IS GREATER THAN zero  
                IF the element at the last position of stack IS LESS THAN the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                    CONTINUE WHILE LOOP  
                ELSE IF the element at the last position of stack IS EQUALS the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                END IF  
                BREAK WHILE LOOP  
            END WHILE  
            OTHERWISE  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
