CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO a list of characters in s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s IS GREATER THAN one  
            SET found_match TO false  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP the element at position j of s WITH the element at position j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE first element FROM s  
                    REMOVE last element FROM s  
                    SET found_match TO true  
                    BREAK  
                END IF  
            END FOR  
            IF found_match IS false  
                REMOVE first element FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
