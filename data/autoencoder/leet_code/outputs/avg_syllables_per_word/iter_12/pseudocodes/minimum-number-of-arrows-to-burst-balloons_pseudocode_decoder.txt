CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY element at position one of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinite  
        FOR each balloon IN points  
            IF element at position zero of balloon GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO element at position one of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
