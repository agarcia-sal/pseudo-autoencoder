CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  
        
        SET word_dict TO a new mapping from keys to empty lists  
        SET word_length TO the LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring of word from start TO i MINUS one, the character star, and substring of word from i PLUS one TO end  
                APPEND word TO the list AT intermediate_key IN word_dict  
            END FOR  
        END FOR  
        
        SET queue TO a new double-ended queue containing a tuple of beginWord AND one  
        SET visited TO a new set containing beginWord  
        
        WHILE queue IS NOT empty  
            SET current_word AND current_length TO the first element REMOVED FROM queue  
            
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring of current_word from start TO i MINUS one, the character star, and substring of current_word from i PLUS one TO end  
                
                FOR each word IN the list AT intermediate_word IN word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND the tuple of word AND current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
