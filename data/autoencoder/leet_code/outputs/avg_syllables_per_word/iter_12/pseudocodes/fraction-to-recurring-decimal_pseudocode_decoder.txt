CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF (numerator LESS THAN zero) AND (denominator GREATER THAN OR EQUAL TO zero) OR (numerator GREATER THAN OR EQUAL TO zero) AND (denominator LESS THAN zero)  
            SET sign TO hyphen  
        ELSE  
            SET sign TO empty string  
        END IF  

        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  

        SET quotient TO integer division of numerator BY denominator  
        SET remainder TO numerator MINUS quotient MULTIPLIED BY denominator  
        SET integer_part TO string representation of quotient  

        IF remainder EQUALS zero  
            RETURN concatenation of sign AND integer_part  
        END IF  

        SET seen_remainders TO empty dictionary  
        SET fractional_part TO empty list  

        WHILE remainder NOT EQUALS zero  
            IF remainder EXISTS AS key IN seen_remainders  
                SET start TO value of remainder key IN seen_remainders  
                SET non_repeating TO concatenation of elements FROM fractional_part starting AT zero UP TO but NOT INCLUDING start  
                SET repeating TO concatenation of elements FROM fractional_part starting AT start TO end  
                RETURN concatenation of sign, integer_part, dot, non_repeating, open parenthesis, repeating, close parenthesis  
            END IF  

            SET seen_remainders AT remainder TO length OF fractional_part  

            SET remainder TO remainder MULTIPLIED BY ten  

            SET quotient TO integer division OF remainder BY denominator  
            SET remainder TO remainder MINUS quotient MULTIPLIED BY denominator  

            APPEND string representation OF quotient TO fractional_part  
        END WHILE  

        RETURN concatenation of sign, integer_part, dot, concatenation OF fractional_part  
    END FUNCTION  
END CLASS
