CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of SPLITTING path BY the character slash  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS period  
                CONTINUE to next iteration  
            ELSE IF component EQUALS two periods  
                IF stack IS NOT empty  
                    REMOVE last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character slash CONCATENATED WITH the JOINING of elements in stack USING the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
