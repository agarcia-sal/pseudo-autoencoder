CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO Counter of nums  
        SET end_with TO Counter  
        
        FOR each num IN nums  
            IF num_count at position num EQUALS zero  
                CONTINUE  
            END IF  
            
            IF end_with at position num MINUS one GREATER THAN zero  
                DECREMENT end_with at position num MINUS one BY one  
                INCREMENT end_with at position num BY one  
            ELSE IF num_count at position num PLUS one GREATER THAN zero AND num_count at position num PLUS two GREATER THAN zero  
                DECREMENT num_count at position num PLUS one BY one  
                DECREMENT num_count at position num PLUS two BY one  
                INCREMENT end_with at position num PLUS two BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT num_count at position num BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
