CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero  
            RETURN False  
        END IF  
        SET count TO a mapping of card frequencies IN hand  
        SET unique_cards TO the sorted list OF the keys IN count  
        FOR each card IN unique_cards  
            IF the value AT card IN count GREATER THAN zero  
                SET needed TO the value AT card IN count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the value AT card PLUS index IN count LESS THAN needed  
                        RETURN False  
                    END IF  
                    DECREMENT the value AT card PLUS index IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
