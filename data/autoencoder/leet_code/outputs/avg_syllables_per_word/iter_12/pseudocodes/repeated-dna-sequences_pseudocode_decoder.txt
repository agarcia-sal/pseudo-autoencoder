CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        CALL initialize sequence_count AS a mapping with default value zero for any key  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring of s starting at position index and ending at position index PLUS nine  
            INCREMENT the value in sequence_count at key sequence BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence_key IN the keys of sequence_count  
            IF the value at sequence_count for sequence_key GREATER THAN one  
                APPEND sequence_key TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
