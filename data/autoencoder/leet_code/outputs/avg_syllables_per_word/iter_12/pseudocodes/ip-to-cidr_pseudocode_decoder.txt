CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO list of integer values obtained by splitting ip by dot  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  

        FUNCTION int_to_ip(num)  
            SET ip_parts TO empty list  
            FOR i FROM three DOWN TO zero  
                APPEND the remainder of num DIVIDED BY (two HUNDRED FIFTY SIX TO THE POWER i) DIVIDED BY (two HUNDRED FIFTY SIX TO THE POWER (i MINUS one)) TO ip_parts  
            END FOR  
            RETURN the string of ip_parts joined by dots  
        END FUNCTION  

        FUNCTION largest_cidr_size(start, n)  
            IF the bitwise AND of start AND one EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND the bitwise AND of start AND (x MINUS one) EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  

        SET result TO empty list  
        SET start_ip TO the result of ip_to_int(ip)  
        
        WHILE n GREATER THAN zero  
            SET size TO the result of largest_cidr_size(start_ip, n)  
            APPEND the string formed by int_to_ip(start_ip), a slash, and thirty TWO MINUS (the bit length of size MINUS one) TO result  
            INCREMENT start_ip BY size  
            DECREMENT n BY size  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
