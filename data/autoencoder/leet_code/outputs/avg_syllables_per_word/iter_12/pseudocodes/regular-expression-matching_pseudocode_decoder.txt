CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a list containing length of s PLUS one elements each as a list containing length of p PLUS one elements each set TO False  
        SET element at position zero position zero of dp TO True  
        
        FOR j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of p EQUALS '*'
                SET element at position zero position j of dp TO element at position zero position j MINUS two of dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of p EQUALS element at position i MINUS one of s OR element at position j MINUS one of p EQUALS '.'  
                    SET element at position i position j of dp TO element at position i MINUS one position j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS '*'
                    SET element at position i position j of dp TO element at position i position j MINUS two of dp  
                    IF element at position j MINUS two of p EQUALS element at position i MINUS one of s OR element at position j MINUS two of p EQUALS '.'  
                        SET element at position i position j of dp TO element at position i position j of dp OR element at position i MINUS one position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position length of s position length of p of dp  
    END FUNCTION  
END CLASS
