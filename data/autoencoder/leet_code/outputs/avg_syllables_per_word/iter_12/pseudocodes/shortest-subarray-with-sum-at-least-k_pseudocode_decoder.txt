CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list initialized with zero  
        FOR each num IN nums  
            APPEND the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        SET dq TO empty double-ended queue  
        SET min_length TO infinity  
        FOR index i FROM zero TO the LENGTH OF prefix_sum MINUS one  
            SET current_sum TO element at position i of prefix_sum  
            WHILE dq IS NOT empty AND current_sum MINUS element at position of the element at the first position of dq IN prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value BETWEEN min_length AND i MINUS the element removed from the first position of dq  
            END WHILE  
            WHILE dq IS NOT empty AND current_sum IS LESS THAN OR EQUAL TO element at position of the element at the last position of dq IN prefix_sum  
                REMOVE last element FROM dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length IS NOT EQUAL TO infinity  
            RETURN min_length  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
