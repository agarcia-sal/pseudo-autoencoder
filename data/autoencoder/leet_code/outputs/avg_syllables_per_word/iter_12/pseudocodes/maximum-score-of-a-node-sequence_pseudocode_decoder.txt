CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO an empty mapping from keys to lists  
        FOR each pair u, v IN edges  
            APPEND pair of element at position v of scores AND v TO graph at key u  
            APPEND pair of element at position u of scores AND u TO graph at key v  
        END FOR  
        FOR each node IN graph  
            SET graph at key node TO the list of three elements with the highest first values from graph at key node  
        END FOR  
        SET max_score TO negative one  
        FOR each pair u, v IN edges  
            FOR each pair score1, x IN graph at key u  
                FOR each pair score2, y IN graph at key v  
                    IF the number of distinct elements in the set containing x, y, u, AND v EQUALS four  
                        SET max_score TO the greater of max_score AND score1 PLUS score2 PLUS element at position u of scores PLUS element at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
