CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list containing target AND zero TO stations  

        FOR each element at position named position AND element at position named fuel IN stations  
            WHILE max_heap is not empty AND current_fuel LESS THAN position  
                SET current_fuel TO current_fuel MINUS the largest fuel amount popped from max_heap  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative fuel value TO max_heap maintaining max-heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
