CLASS Solution  
    FUNCTION minFallingPathSum(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF grid  
        
        IF n EQUALS one  
            SET minimum_value TO the minimum element IN element at position zero of grid  
            RETURN minimum_value  
        END IF  
        
        SET dp TO a COPY OF element at position zero of grid  
        
        FOR index i FROM one TO n MINUS one  
            SET new_dp TO a list OF size n FILLED WITH zero  
            SET min1 TO positive infinity  
            SET min2 TO positive infinity  
            SET min1_index TO minus one  
            
            FOR index j FROM zero TO n MINUS one  
                IF element at position j of dp LESS THAN min1  
                    SET min2 TO min1  
                    SET min1 TO element at position j of dp  
                    SET min1_index TO j  
                ELSE IF element at position j of dp LESS THAN min2  
                    SET min2 TO element at position j of dp  
                END IF  
            END FOR  
            
            FOR index j FROM zero TO n MINUS one  
                IF j EQUALS min1_index  
                    SET element at position j of new_dp TO element at position i of grid at position j PLUS min2  
                ELSE  
                    SET element at position j of new_dp TO element at position i of grid at position j PLUS min1  
                END IF  
            END FOR  
            
            SET dp TO a COPY OF new_dp  
        END FOR  
        
        SET result TO the minimum element IN dp  
        RETURN result  
    END FUNCTION  
END CLASS
