CLASS BIT  
    FUNCTION __init__(self, n)  
        SET self.size TO n  
        SET self.tree TO a list of zero SET TO the LENGTH OF n PLUS one  
    END FUNCTION  
  
    FUNCTION update(self, idx, delta)  
        WHILE idx LESS THAN OR EQUAL TO self.size  
            INCREMENT element at position idx OF self.tree BY delta  
            SET idx TO idx PLUS idx AND NEGATION OF idx  
        END WHILE  
    END FUNCTION  
  
    FUNCTION query(self, idx)  
        SET result TO zero  
        WHILE idx GREATER THAN zero  
            INCREMENT result BY element at position idx OF self.tree  
            SET idx TO idx MINUS idx AND NEGATION OF idx  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION goodTriplets(self, nums1, nums2)  
        SET n TO the LENGTH OF nums1  
  
        SET pos TO a list of zero SET TO the LENGTH OF n  
        FOR each i AND num IN enumeration OF nums1  
            SET element at position num OF pos TO i  
        END FOR  
  
        SET compressed TO a list of element at position num OF pos FOR each num IN nums2  
  
        SET bit TO a new BIT INITIALIZED WITH n  
  
        SET smaller_count TO a list of zero SET TO the LENGTH OF n  
        SET larger_count TO a list of zero SET TO the LENGTH OF n  
  
        FOR each i FROM zero TO n MINUS one  
            SET element at position i OF smaller_count TO bit.query(element at position i OF compressed)  
            CALL bit.update WITH parameters element at position i OF compressed PLUS one, one  
        END FOR  
  
        SET bit TO a new BIT INITIALIZED WITH n  
  
        FOR each i FROM n MINUS one DOWN TO zero  
            SET element at position i OF larger_count TO bit.query(n) MINUS bit.query(element at position i OF compressed PLUS one)  
            CALL bit.update WITH parameters element at position i OF compressed PLUS one, one  
        END FOR  
  
        SET good_triplets TO zero  
        FOR each i FROM zero TO n MINUS one  
            INCREMENT good_triplets BY element at position i OF smaller_count MULTIPLIED BY element at position i OF larger_count  
        END FOR  
  
        RETURN good_triplets  
    END FUNCTION  
END CLASS
