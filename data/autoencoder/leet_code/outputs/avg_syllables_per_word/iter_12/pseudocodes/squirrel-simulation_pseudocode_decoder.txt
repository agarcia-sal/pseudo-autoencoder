CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            SET horizontal_difference TO the absolute value of element at position zero of p1 MINUS element at position zero of p2  
            SET vertical_difference TO the absolute value of element at position one of p1 MINUS element at position one of p2  
            SET total_difference TO horizontal_difference PLUS vertical_difference  
            RETURN total_difference  
        END FUNCTION

        SET total_distance TO zero  
        SET max_save TO negative infinity  

        FOR each nut IN nuts  
            SET round_trip_distance TO distance(nut, tree) MULTIPLIED BY two  
            INCREMENT total_distance BY round_trip_distance  

            SET first_nut_distance TO distance(squirrel, nut) PLUS distance(nut, tree)  

            SET later_nut_distance TO distance(nut, tree) MULTIPLIED BY two  

            SET save TO later_nut_distance MINUS first_nut_distance  

            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  

        SET result TO total_distance MINUS max_save  
        RETURN result  
    END FUNCTION  
END CLASS
