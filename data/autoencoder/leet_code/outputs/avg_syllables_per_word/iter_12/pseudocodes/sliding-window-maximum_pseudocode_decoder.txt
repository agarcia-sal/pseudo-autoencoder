CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  
        
        SET deq TO empty double-ended queue  
        SET result TO empty list  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND element at position zero OF deq LESS THAN i MINUS k PLUS one  
                REMOVE first element FROM deq  
            END IF  
            
            WHILE deq IS NOT empty AND element at position last OF nums AT element at position last OF deq LESS THAN nums AT i  
                REMOVE last element FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND nums AT element at position zero OF deq TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
