CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a frequency mapping of elements in nums  
        SET unique_nums TO the sorted list of keys from count  
        SET dp TO a list of zeros with LENGTH equal TO the LENGTH OF unique_nums PLUS one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the value in count at key equal TO the element at position zero of unique_nums  
        FOR index FROM two TO the LENGTH OF unique_nums  
            SET num TO the element at position index MINUS one of unique_nums  
            IF num EQUALS the element at position index MINUS two of unique_nums PLUS one  
                SET the element at position index of dp TO the greater value BETWEEN the element at position index MINUS one of dp AND the element at position index MINUS two of dp PLUS num MULTIPLIED BY the value in count at key num  
            ELSE  
                SET the element at position index of dp TO the element at position index MINUS one of dp PLUS num MULTIPLIED BY the value in count at key num  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
