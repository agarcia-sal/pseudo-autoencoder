CLASS Solution  
    FUNCTION updateMatrix(mat)  
        IF mat EQUALS empty OR first element at position zero of mat EQUALS empty  
            RETURN mat  
        END IF  
        
        SET rows TO the LENGTH OF mat  
        SET cols TO the LENGTH OF first element at position zero of mat  
        SET dist TO a list of lists with rows number of elements each containing cols number of elements set TO infinity  
        SET queue TO a new empty double-ended queue  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of mat at position c EQUALS zero  
                    SET element at position r of dist at position c TO zero  
                    APPEND pair of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        
        SET directions TO a list containing the pairs: minus one and zero, one and zero, zero and minus one, zero and one  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET TO r and c  
            FOR each pair dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr of dist at position nc GREATER THAN element at position r of dist at position c PLUS one  
                    SET element at position nr of dist at position nc TO element at position r of dist at position c PLUS one  
                    APPEND pair of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN dist  
    END FUNCTION  
END CLASS
