CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of False repeated n PLUS one times  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF NOT the element at position i MINUS j MULTIPLIED BY j of dp  
                    SET the element at position i of dp TO True  
                    BREAK the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
