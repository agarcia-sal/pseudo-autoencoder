CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values EQUALS NOT true
        RETURN none
    END IF
    SET root TO a new TreeNode with value equal to element at position zero of values
    SET i TO one
    SET queue TO a new deque
    APPEND root TO queue
    WHILE queue IS true
        SET node TO the element removed from the left side of queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.left TO a new TreeNode with value equal to element at position i of values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.right TO a new TreeNode with value equal to element at position i of values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p EQUALS NOT true AND q EQUALS NOT true
        RETURN True
    ELSE IF p EQUALS NOT true OR q EQUALS NOT true
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION boundaryOfBinaryTree(self, root)
        IF root EQUALS NOT true
            RETURN an empty list
        END IF

        FUNCTION is_leaf(node)
            RETURN node.left EQUALS NOT true AND node.right EQUALS NOT true
        END FUNCTION

        SET left_boundary TO an empty list
        SET current TO root.left
        WHILE current IS true
            IF is_leaf with parameter current EQUALS NOT true
                APPEND current.val TO left_boundary
            END IF
            IF current.left IS true
                SET current TO current.left
            ELSE
                SET current TO current.right
            END IF
        END WHILE

        SET right_boundary TO an empty list
        SET current TO root.right
        WHILE current IS true
            IF is_leaf with parameter current EQUALS NOT true
                APPEND current.val TO right_boundary
            END IF
            IF current.right IS true
                SET current TO current.right
            ELSE
                SET current TO current.left
            END IF
        END WHILE

        SET leaves TO an empty list
        FUNCTION collect_leaves(node)
            IF node EQUALS NOT true
                RETURN
            END IF
            IF is_leaf with parameter node
                APPEND node.val TO leaves
            END IF
            CALL collect_leaves with parameter node.left
            CALL collect_leaves with parameter node.right
        END FUNCTION

        CALL collect_leaves with parameter root

        IF is_leaf with parameter root
            SET leaves TO an empty list
        END IF

        RETURN a list consisting of root.val followed by all elements of left_boundary followed by all elements of leaves followed by all elements of right_boundary in reverse order
    END FUNCTION
END CLASS
