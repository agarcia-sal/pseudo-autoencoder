CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a dictionary mapping each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each u, v, w IN flights  
            APPEND a pair consisting of v and w TO the list associated with key u in graph  
        END FOR  
        
        SET min_heap TO a list containing a tuple of zero cost, minus one stops, and src city  
        SET min_cost TO a dictionary mapping src TO minus one  
        
        WHILE min_heap is not empty  
            REMOVE the smallest element FROM min_heap AND unpack INTO cost, stops, city  
            
            IF city EQUALS dst  
                RETURN cost  
            END IF  
            
            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            
            FOR each neighbor, price IN the list associated with city in graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor is not a key in min_cost OR next_stops LESS THAN the value associated with neighbor in min_cost OR next_cost LESS THAN the first element of the first tuple in min_heap  
                    SET the value associated with neighbor in min_cost TO next_stops  
                    ADD the tuple consisting of next_cost, next_stops, and neighbor TO min_heap preserving heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
