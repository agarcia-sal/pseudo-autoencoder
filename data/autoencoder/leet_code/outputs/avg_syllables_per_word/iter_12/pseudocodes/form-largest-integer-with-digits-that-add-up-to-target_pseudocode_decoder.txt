CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO an empty dictionary  
        FOR each position i FROM zero TO the LENGTH OF cost MINUS one  
            SET c TO element at position i of cost  
            SET digit_string TO string representation of i PLUS one  
            SET element at key c of cost_to_digit TO digit_string  
        END FOR  
        
        SET dp TO a list containing empty string FOLLOWED BY target times the string minus one  
        
        FOR t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp NOT EQUALS minus one  
                    SET candidate TO the concatenation of d AND element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS minus one OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate GREATER THAN element at position t of dp)  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp NOT EQUALS minus one  
            RETURN element at position target of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
