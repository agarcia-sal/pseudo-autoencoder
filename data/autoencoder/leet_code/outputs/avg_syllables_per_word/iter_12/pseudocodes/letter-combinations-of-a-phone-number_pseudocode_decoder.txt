CLASS Solution
    FUNCTION letterCombinations(digits)
        IF digits EQUALS empty string
            RETURN empty list
        END IF

        SET phone_map TO a mapping from digit characters TO corresponding letter strings as follows
            digit two TO string "abc"
            digit three TO string "def"
            digit four TO string "ghi"
            digit five TO string "jkl"
            digit six TO string "mno"
            digit seven TO string "pqrs"
            digit eight TO string "tuv"
            digit nine TO string "wxyz"

        SET output TO empty list

        FUNCTION backtrack(combination, next_digits)
            IF the LENGTH OF next_digits EQUALS zero
                APPEND combination TO output
            ELSE
                FOR each letter IN the string mapped by the element at position zero of next_digits IN phone_map
                    CALL backtrack WITH the concatenation of combination PLUS letter, and the substring of next_digits FROM position one TO end
                END FOR
            END IF
        END FUNCTION

        CALL backtrack WITH empty string AND digits
        RETURN output
    END FUNCTION
END CLASS
