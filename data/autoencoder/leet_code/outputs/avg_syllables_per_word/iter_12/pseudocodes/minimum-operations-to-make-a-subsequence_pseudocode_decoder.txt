CLASS Solution
    FUNCTION minOperations(target, arr)  
        SET index_map TO empty mapping
        FOR index FROM zero TO the LENGTH OF target MINUS one  
            SET value TO element at position index of target
            SET index_map at key value TO index
        END FOR

        SET transformed_arr TO empty list
        FOR each value IN arr  
            IF value IN index_map  
                APPEND index_map at key value TO transformed_arr
            END IF
        END FOR

        SET lis TO empty list
        FOR each num IN transformed_arr  
            SET pos TO the smallest position in lis where num is GREATER THAN OR EQUAL TO element at position pos of lis
            IF pos EQUALS the LENGTH OF lis  
                APPEND num TO lis
            ELSE  
                SET element at position pos of lis TO num
            END IF
        END FOR

        SET lcs_length TO the LENGTH OF lis
        RETURN the LENGTH OF target MINUS lcs_length
    END FUNCTION
END CLASS
