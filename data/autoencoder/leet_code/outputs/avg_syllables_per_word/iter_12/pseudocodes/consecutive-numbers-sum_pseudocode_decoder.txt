CLASS Solution  
    FUNCTION consecutiveNumbersSum(n)  
        SET count TO zero  
        FOR each k FROM one TO n  
            IF the remainder of two MULTIPLIED BY n DIVIDED BY k EQUALS zero  
                SET m TO the quotient of two MULTIPLIED BY n DIVIDED BY k MINUS k PLUS one DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUALS m converted TO integer  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
