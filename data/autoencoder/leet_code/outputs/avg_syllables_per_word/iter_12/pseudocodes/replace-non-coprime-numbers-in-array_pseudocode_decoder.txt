CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack GREATER THAN zero  
                SET g TO gcd of element at last position of stack AND num  
                IF g EQUALS one  
                    BREAK  
                END IF  
                SET num TO element at last position of stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
