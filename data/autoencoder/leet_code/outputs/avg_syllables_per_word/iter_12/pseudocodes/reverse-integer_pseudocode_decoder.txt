CLASS Solution
    FUNCTION reverse(x)
        IF x LESS THAN zero
            SET sign TO negative one
        ELSE
            SET sign TO one
        END IF
        SET absolute_x_to_string TO string representation of absolute value of x
        SET reversed_string TO reverse the order of characters in absolute_x_to_string
        SET reversed_x TO integer conversion of reversed_string
        SET result TO sign MULTIPLIED BY reversed_x
        IF result LESS THAN negative two RAISED TO THE power OF thirty one OR result GREATER THAN two RAISED TO THE power OF thirty one MINUS one
            RETURN zero
        END IF
        RETURN result
    END FUNCTION
END CLASS
