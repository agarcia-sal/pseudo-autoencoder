CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO element at position zero of building  
            SET right TO element at position one of building  
            SET height TO element at position two of building  
            APPEND tuple consisting of left, negative height, and right TO events  
            APPEND tuple consisting of right, zero, and zero TO events  
        END FOR  
        SORT events BY ascending order  
        SET result TO list containing list of zero and zero  
        SET max_heap TO list containing tuple of zero and infinite value  
        FOR each event IN events  
            SET x TO element at position zero of event  
            SET negH TO element at position one of event  
            SET r TO element at position two of event  
            WHILE element at position one of element at position zero of max_heap LESS THAN OR EQUAL TO x  
                REMOVE first element FROM max_heap  
            END WHILE  
            IF negH NOT EQUALS zero  
                CALL heap push to add tuple of negH and r TO max_heap  
            END IF  
            IF element at position one of last element in result NOT EQUALS negative of element at position zero of element at position zero of max_heap  
                APPEND list of x and negative of element at position zero of element at position zero of max_heap TO result  
            END IF  
        END FOR  
        IF element at position zero of result EQUALS list of zero and zero  
            REMOVE first element FROM result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
