CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        SET tree TO a new mapping with default empty list values  
        FOR each pair of child and parent IN the combination of pid and ppid  
            APPEND child TO the list at position parent in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue is greater than zero  
            SET current TO the first element of queue  
            REMOVE first element FROM queue  
            APPEND current TO killed_processes  
            FOR each child IN the list at position current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
