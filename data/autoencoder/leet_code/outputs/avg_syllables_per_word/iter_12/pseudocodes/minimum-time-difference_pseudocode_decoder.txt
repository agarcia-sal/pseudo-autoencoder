CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours AND minutes TO the TWO numbers OBTAINED BY SPLITTING time BY colon  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the LIST OBTAINED BY APPLYING to_minutes TO each element OF timePoints  
        SORT minutes_list IN ascending order  
        
        SET min_diff TO infinity  
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET current_difference TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        
        SET circular_difference TO one thousand four hundred forty MINUS the last element of minutes_list PLUS the first element of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
