CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zeroes of LENGTH n  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of dominoes EQUALS letter R  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS letter L  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            INCREMENT the element at position i of forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF the element at position i of dominoes EQUALS letter L  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS letter R  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            DECREMENT the element at position i of forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND letter R TO result  
            ELSE IF f LESS THAN zero  
                APPEND letter L TO result  
            ELSE  
                APPEND character dot TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
