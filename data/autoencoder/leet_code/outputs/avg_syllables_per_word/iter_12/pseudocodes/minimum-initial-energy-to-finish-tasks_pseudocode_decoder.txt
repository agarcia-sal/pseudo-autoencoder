CLASS Solution  
    FUNCTION minimumEffort(tasks)  
        SORT tasks BY the difference BETWEEN element at position one AND element at position zero OF each task IN descending order  
        SET current_energy TO zero  
        SET required_initial_energy TO zero  
        FOR each task IN tasks  
            SET actual TO element at position zero OF task  
            SET minimum TO element at position one OF task  
            IF current_energy LESS THAN minimum  
                INCREMENT required_initial_energy BY minimum MINUS current_energy  
                SET current_energy TO minimum  
            END IF  
            DECREMENT current_energy BY actual  
        END FOR  
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
