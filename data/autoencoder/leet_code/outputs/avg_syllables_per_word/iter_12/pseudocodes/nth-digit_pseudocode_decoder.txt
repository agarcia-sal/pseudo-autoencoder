CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
  
        SET start TO start PLUS the INTEGER DIVISION OF n MINUS one BY length  
        SET digit_index TO the REMAINDER OF n MINUS one DIVIDED BY length  
        SET digit_string TO CONVERTING start TO string  
        RETURN the CHARACTER AT position digit_index OF digit_string CONVERTED TO integer  
    END FUNCTION  
END CLASS
