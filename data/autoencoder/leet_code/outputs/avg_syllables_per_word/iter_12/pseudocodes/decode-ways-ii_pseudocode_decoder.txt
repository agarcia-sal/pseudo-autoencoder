CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO list of zero REPEATED for the LENGTH OF s PLUS one  
        SET first element of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS star character  
                RETURN nine  
            ELSE IF char EQUALS zero character  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS star character AND char2 EQUALS star character  
                RETURN fifteen  
            ELSE IF char1 EQUALS star character  
                IF char2 LESS THAN OR EQUAL TO character 'six'  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS star character  
                IF char1 EQUALS character 'one'  
                    RETURN nine  
                ELSE IF char1 EQUALS character 'two'  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO concatenation of char1 AND char2 CONVERTED TO integer value  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO LENGTH OF s  
            SET dp at position i TO dp at position i PLUS one MULTIPLIED BY ways_to_decode_single(character at position i MINUS one of s) MODULO MOD  
            IF i GREATER THAN one  
                SET dp at position i TO dp at position i PLUS dp at position i MINUS two MULTIPLIED BY ways_to_decode_pair(character at position i MINUS two of s, character at position i MINUS one of s) MODULO MOD  
            END IF  
        END FOR  
        
        RETURN last element of dp  
    END FUNCTION  
END CLASS
