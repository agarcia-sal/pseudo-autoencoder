CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET stickers TO a list obtained by CONVERTING each element sticker IN stickers TO a frequency mapping of characters  
        SORT stickers BY the NEGATIVE of the MAXIMUM frequency value of the character counts in each sticker  
        SET queue TO a double-ended queue initialized with a pair consisting of target and zero  
        SET memo TO a mapping with key target and value zero  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue and SET current_target TO the first item and used_stickers TO the second item of the removed element  
            
            FOR each sticker IN stickers  
                IF the first character of current_target is NOT PRESENT IN sticker  
                    CONTINUE to the next iteration  
                END IF  
                
                SET remaining TO a frequency mapping of characters in current_target  
                DECREASE counts in remaining BY the counts in sticker  
                
                SET new_target TO the string formed by SORTING and JOINING all characters represented in remaining according to their counts  
                
                IF new_target is empty  
                    RETURN used_stickers PLUS one  
                END IF  
                
                IF new_target IS NOT PRESENT AS A KEY IN memo  
                    SET memo at new_target TO used_stickers PLUS one  
                    APPEND the pair (new_target, used_stickers PLUS one) TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
