CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s EQUALS empty OR words EQUALS empty OR element at position zero of words EQUALS empty  
            RETURN empty list  
        END IF  
        
        SET word_length TO LENGTH OF element at position zero of words  
        SET num_words TO LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO result of calling Counter on words  
        SET result TO empty list  
        
        FOR i FROM zero TO LENGTH OF s MINUS total_length PLUS one  
            SET seen TO result of calling Counter with no arguments  
            FOR j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO substring of s from position j LENGTH word_length  
                INCREMENT count of word in seen BY one  
                IF count of word in seen GREATER THAN count of word in word_count  
                    BREAK loop  
                END IF  
            END FOR  
            IF loop completed without break  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
