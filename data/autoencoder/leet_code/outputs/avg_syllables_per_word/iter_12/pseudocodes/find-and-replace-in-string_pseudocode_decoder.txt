CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO list of tuples CREATED BY combining indices, sources, and targets SORTED BY the first element (index)  
        SET result TO empty list  
        SET prev_end TO zero  
        FOR each tuple WITH elements index, source, target IN replacements  
            APPEND substring of s FROM position prev_end TO position index TO result  
            IF substring of s FROM position index TO position index PLUS LENGTH OF source EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND substring of s FROM position index TO position index PLUS LENGTH OF source TO result  
            END IF  
            SET prev_end TO index PLUS LENGTH OF source  
        END FOR  
        APPEND substring of s FROM position prev_end TO END TO result  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
