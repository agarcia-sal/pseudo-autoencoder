CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance with parameters t and s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff GREATER THAN one  
            RETURN False  
        END IF  
        
        IF diff EQUALS zero  
            SET count_of_differences TO zero  
            FOR each position index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    INCREMENT count_of_differences BY one  
                END IF  
            END FOR  
            RETURN True IF count_of_differences EQUALS one OTHERWISE False  
        ELSE  
            FOR each position index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    RETURN True IF the SUBSTRING of s from position index TO the END EQUALS the SUBSTRING of t from position index PLUS one TO the END OTHERWISE False  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
