CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO empty mapping from integers to sets  
        FOR i FROM one TO n  
            SET element at position i of graph TO empty set  
        END FOR  
        
        FOR each pair u and v IN edges  
            ADD v TO the set element at position u of graph  
            ADD u TO the set element at position v of graph  
        END FOR  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF the set element at position i of graph CONTAINS j AND  
                       the set element at position i of graph CONTAINS k AND  
                       the set element at position j of graph CONTAINS k  
                    THEN  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set element at position i of graph PLUS  
                                           the LENGTH OF the set element at position j of graph PLUS  
                                           the LENGTH OF the set element at position k of graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
