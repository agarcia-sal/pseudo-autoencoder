CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO a dictionary with key zero mapped to value minus one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each i, num IN the enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF cumulative_sum MINUS k EXISTS AS a key IN cumulative_sum_index  
                SET length_candidate TO i MINUS the value at key cumulative_sum MINUS k in cumulative_sum_index  
                IF length_candidate GREATER THAN max_length  
                    SET max_length TO length_candidate  
                END IF  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST AS a key IN cumulative_sum_index  
                SET the value at key cumulative_sum in cumulative_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
