CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO list containing zero  
        FOR each num IN the SUBLIST of nums from position zero TO position mid MINUS one  
            SET new_sums TO empty list  
            FOR each s IN left_sums  
                APPEND the SUM of s PLUS num TO new_sums  
            END FOR  
            EXTEND left_sums BY new_sums  
        END FOR  
        
        SET right_sums TO list containing zero  
        FOR each num IN the SUBLIST of nums from position mid TO the END  
            SET new_sums TO empty list  
            FOR each s IN right_sums  
                APPEND the SUM of s PLUS num TO new_sums  
            END FOR  
            EXTEND right_sums BY new_sums  
        END FOR  
        
        SORT right_sums  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the result of binary searching left position of target IN right_sums  
            
            IF pos LESS THAN the LENGTH OF right_sums  
                SET current_diff TO the ABSOLUTE VALUE of the SUM of left_sum PLUS element at position pos of right_sums MINUS goal  
                IF current_diff LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END IF  
            
            IF pos GREATER THAN zero  
                SET current_diff TO the ABSOLUTE VALUE of the SUM of left_sum PLUS element at position pos MINUS one of right_sums MINUS goal  
                IF current_diff LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
