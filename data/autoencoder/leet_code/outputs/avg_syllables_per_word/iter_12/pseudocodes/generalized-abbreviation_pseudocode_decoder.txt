CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all elements IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index PLUS one, path PLUS the STRING REPRESENTATION OF count PLUS the character at position index of word, zero  
                ELSE  
                    CALL backtrack WITH index PLUS one, path PLUS the character at position index of word, zero  
                END IF  
                CALL backtrack WITH index PLUS one, path, count PLUS one  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH zero, empty list, zero  
        RETURN result  
    END FUNCTION  
END CLASS
