CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty collection  
        
        FOR i FROM zero TO n MINUS one MINUS one  
            SET fraction_value TO element at position i of arr DIVIDED BY element at position n MINUS one of arr  
            ADD a tuple containing fraction_value, i, and n MINUS one TO min_heap as per heap order  
        END FOR  
        
        FOR count FROM one TO k MINUS one  
            REMOVE the smallest element FROM min_heap AND UNPACK as _, i, j  
            IF j MINUS one GREATER THAN i  
                SET fraction_value TO element at position i of arr DIVIDED BY element at position j MINUS one of arr  
                ADD a tuple containing fraction_value, i, and j MINUS one TO min_heap as per heap order  
            END IF  
        END FOR  
        
        REMOVE the smallest element FROM min_heap AND UNPACK as _, i, j  
        RETURN a list containing element at position i of arr AND element at position j of arr  
    END FUNCTION  
END CLASS
