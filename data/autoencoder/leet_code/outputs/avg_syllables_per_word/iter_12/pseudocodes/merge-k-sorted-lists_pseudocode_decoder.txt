CLASS ListNode  
    FUNCTION __init__(val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one to end of values  
        SET node TO new ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list of p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(lists)  
        SET min_heap TO empty list  
        FOR each i and lst IN indices and elements of lists  
            IF lst IS NOT None  
                PUSH tuple of lst.val, i, and lst INTO min_heap using heap push  
            END IF  
        END FOR  
        SET dummy TO new ListNode  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            POP smallest tuple val, i, node FROM min_heap using heap pop  
            SET current.next TO node  
            SET current TO current.next  
            IF node.next IS NOT None  
                PUSH tuple of node.next.val, i, and node.next INTO min_heap using heap push  
            END IF  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
