CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value the element at position zero of values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS None  
            RETURN  
        END IF  
        SET stack TO list containing root  
        WHILE stack is not empty  
            SET node TO last element removed FROM stack  
            IF node.right NOT EQUALS None  
                APPEND node.right TO stack  
            END IF  
            IF node.left NOT EQUALS None  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO None  
            IF stack is not empty  
                SET node.right TO last element of stack  
            ELSE  
                SET node.right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
