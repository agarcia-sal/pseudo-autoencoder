CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR the character at position index PLUS one of expression EQUALS colon  
                RETURN the character at position index of expression, index PLUS two  
            END IF  
            SET true_part, next_index TO the RESULT OF evaluate at index PLUS two  
            SET false_part, final_index TO the RESULT OF evaluate at next_index  
            IF the character at position index of expression EQUALS letter T  
                RETURN true_part, final_index  
            ELSE  
                RETURN false_part, final_index  
            END IF  
        END FUNCTION  
        SET result, _ TO the RESULT OF evaluate at zero  
        RETURN result  
    END FUNCTION  
END CLASS
