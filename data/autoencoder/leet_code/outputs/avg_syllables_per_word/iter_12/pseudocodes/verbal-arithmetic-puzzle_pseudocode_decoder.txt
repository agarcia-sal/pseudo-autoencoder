CLASS Solution  
    FUNCTION isSolvable(words, result)  
        SET unique_chars TO the SET OF all characters IN the concatenation OF words AND result  
        IF the LENGTH OF unique_chars GREATER THAN ten  
            RETURN false  
        END IF  
        
        SET non_zero_chars TO the SET OF the first character OF each word IN words WHERE the LENGTH OF word GREATER THAN one  
        IF the LENGTH OF result GREATER THAN one  
            ADD the first character OF result TO non_zero_chars  
        END IF  
        
        SET char_to_digit TO an EMPTY dictionary  
        SET digit_to_char TO an EMPTY dictionary  
        
        FUNCTION can_assign(char, digit)  
            RETURN char NOT IN char_to_digit AND (digit NOT IN digit_to_char OR digit_to_char AT digit EQUALS char)  
        END FUNCTION  
        
        FUNCTION assign(char, digit)  
            SET char_to_digit AT char TO digit  
            SET digit_to_char AT digit TO char  
        END FUNCTION  
        
        FUNCTION unassign(char, digit)  
            REMOVE char FROM char_to_digit  
            REMOVE digit FROM digit_to_char  
        END FUNCTION  
        
        FUNCTION word_value(word)  
            SET total_value TO zero  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET char TO the character AT the LENGTH OF word MINUS one MINUS index OF word  
                INCREMENT total_value BY char_to_digit AT char MULTIPLIED BY ten RAISED TO index  
            END FOR  
            RETURN total_value  
        END FUNCTION  
        
        FUNCTION backtrack(index, col, sum_col)  
            IF col GREATER THAN OR EQUAL TO the MAXIMUM OF the LENGTH OF each word IN words AND col GREATER THAN OR EQUAL TO the LENGTH OF result  
                RETURN sum_col EQUALS zero  
            END IF  
            
            IF index EQUALS the LENGTH OF words  
                IF col LESS THAN the LENGTH OF result  
                    SET digit_char TO the character AT the LENGTH OF result MINUS one MINUS col OF result  
                    IF digit_char IN char_to_digit  
                        IF char_to_digit AT digit_char EQUALS sum_col MODULO ten  
                            RETURN backtrack(zero, col PLUS one, sum_col DIVIDED BY ten)  
                        ELSE  
                            RETURN false  
                        END IF  
                    ELSE IF can_assign(digit_char, sum_col MODULO ten) AND (sum_col MODULO ten NOT EQUALS zero OR digit_char NOT IN non_zero_chars)  
                        assign(digit_char, sum_col MODULO ten)  
                        IF backtrack(zero, col PLUS one, sum_col DIVIDED BY ten)  
                            RETURN true  
                        END IF  
                        unassign(digit_char, sum_col MODULO ten)  
                    END IF  
                ELSE  
                    RETURN sum_col EQUALS zero  
                END IF  
                RETURN false  
            END IF  
            
            SET word TO the element AT index OF words  
            IF col GREATER THAN OR EQUAL TO the LENGTH OF word  
                RETURN backtrack(index PLUS one, col, sum_col)  
            END IF  
            
            SET char TO the character AT the LENGTH OF word MINUS one MINUS col OF word  
            IF char IN char_to_digit  
                RETURN backtrack(index PLUS one, col, sum_col PLUS char_to_digit AT char)  
            END IF  
            
            FOR digit FROM zero TO nine  
                IF can_assign(char, digit) AND (digit NOT EQUALS zero OR char NOT IN non_zero_chars)  
                    assign(char, digit)  
                    IF backtrack(index PLUS one, col, sum_col PLUS digit)  
                        RETURN true  
                    END IF  
                    unassign(char, digit)  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN backtrack(zero, zero, zero)  
    END FUNCTION  
END CLASS
