CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of False with the length of s  
        FOR each word IN words  
            SET start TO the position of word in s  
            WHILE start is NOT EQUALS to minus one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of bold TO True  
                END FOR  
                SET start TO the position of word in s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF the element at position i of bold  
                APPEND the string less than b greater than TO result  
                WHILE i LESS THAN the LENGTH OF s AND the element at position i of bold  
                    APPEND the character at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string less than slash b greater than TO result  
            ELSE  
                APPEND the character at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
