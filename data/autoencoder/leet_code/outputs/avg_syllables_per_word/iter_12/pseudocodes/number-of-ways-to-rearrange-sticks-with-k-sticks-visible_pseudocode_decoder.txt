CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of lists where each inner list contains zero repeated k PLUS one times and outer list contains n PLUS one such inner lists  
        SET element at position zero of list at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of list at position i of dp TO the sum of element at position j MINUS one of list at position i MINUS one of dp AND the product of i MINUS one AND element at position j of list at position i MINUS one of dp  
                SET element at position j of list at position i of dp TO the remainder of element at position j of list at position i of dp DIVIDED BY MOD  
            END FOR  
        END FOR  
        RETURN element at position k of list at position n of dp  
    END FUNCTION  
END CLASS
