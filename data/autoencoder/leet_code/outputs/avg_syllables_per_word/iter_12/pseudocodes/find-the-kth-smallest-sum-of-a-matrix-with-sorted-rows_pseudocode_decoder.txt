CLASS Solution
    FUNCTION kthSmallest(mat, k)
        SET m TO the LENGTH OF mat
        SET n TO the LENGTH OF the element at position zero of mat
        SET min_heap TO a new list containing a tuple of the SUM of the element at position zero of each row in mat AND a tuple of zero REPEATED m times
        SET visited TO a new set containing the tuple at position one of the element at position zero of min_heap

        WHILE k GREATER THAN zero
            SET current_sum, indices TO the result of removing and returning the smallest element from min_heap
            DECREMENT k BY one

            IF k EQUALS zero
                RETURN current_sum
            END IF

            FOR i FROM zero TO m MINUS one
                IF the element at position i of indices PLUS one LESS THAN n
                    SET new_indices TO a copy of indices converted to a list
                    INCREMENT the element at position i of new_indices BY one
                    SET new_indices_tuple TO new_indices converted to a tuple
                    SET new_sum TO current_sum MINUS the element at position i at position indices at element i of mat PLUS the element at position i at position new_indices at element i of mat

                    IF new_indices_tuple NOT IN visited
                        ADD new_indices_tuple TO visited
                        ADD the tuple of new_sum AND new_indices_tuple TO min_heap maintaining heap property
                    END IF
                END IF
            END FOR
        END WHILE

        RETURN minus one
    END FUNCTION
END CLASS
