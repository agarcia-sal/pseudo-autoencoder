CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO empty dictionary  
        SET last TO empty dictionary  
        FOR each index i AND character c IN s  
            IF c NOT IN first  
                SET element at key c of first TO i  
            END IF  
            SET element at key c of last TO i  
        END FOR  
        
        SET intervals TO empty list  
        FOR each character c IN keys of first  
            SET start TO element at key c of first  
            SET end TO element at key c of last  
            SET i TO start  
            WHILE i LESS THAN OR EQUAL TO end  
                IF element at key s at position i of first LESS THAN start  
                    BREAK  
                END IF  
                SET end TO the GREATER OF end AND element at key s at position i of last  
                INCREMENT i BY one  
            END WHILE  
            IF i GREATER THAN end  
                APPEND pair of start and end TO intervals  
            END IF  
        END FOR  
        
        SORT intervals BY second element of each pair in ascending order  
        
        SET result TO empty list  
        SET prev_end TO negative one  
        FOR each pair of start and end IN intervals  
            IF start GREATER THAN prev_end  
                APPEND substring of s from position start TO position end INCLUSIVE TO result  
                SET prev_end TO end  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
