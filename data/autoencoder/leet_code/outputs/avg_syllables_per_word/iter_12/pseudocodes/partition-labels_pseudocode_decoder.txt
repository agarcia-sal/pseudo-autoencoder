CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO EMPTY DICTIONARY  
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            SET last_occurrence at character TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO EMPTY LIST  
        
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            IF last_occurrence at character GREATER THAN end THEN  
                SET end TO last_occurrence at character  
            END IF  
            
            IF index EQUALS end THEN  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
