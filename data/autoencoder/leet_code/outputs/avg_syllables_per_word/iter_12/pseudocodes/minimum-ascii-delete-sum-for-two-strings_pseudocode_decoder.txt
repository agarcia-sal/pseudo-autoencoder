CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        FUNCTION createTwoDimensionalArray(rows, columns)  
            SET array TO empty list  
            FOR index FROM zero TO rows MINUS one  
                APPEND a new list of zeros of LENGTH columns TO array  
            END FOR  
            RETURN array  
        END FUNCTION  
        
        SET dp TO createTwoDimensionalArray(m PLUS one, n PLUS one)  
        
        FOR j FROM one TO n  
            SET element_at_position_j_minus_one_of_s2 TO the character at position j MINUS one of s2  
            SET dp at position zero and j TO dp at position zero and (j MINUS one) PLUS ASCII value OF element_at_position_j_minus_one_of_s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element_at_position_i_minus_one_of_s1 TO the character at position i MINUS one of s1  
            SET dp at position i and zero TO dp at position (i MINUS one) and zero PLUS ASCII value OF element_at_position_i_minus_one_of_s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET char_s1 TO the character at position i MINUS one of s1  
                SET char_s2 TO the character at position j MINUS one of s2  
                IF char_s1 EQUALS char_s2  
                    SET dp at position i and j TO dp at position (i MINUS one) and (j MINUS one)  
                ELSE  
                    SET value_delete_s1 TO dp at position (i MINUS one) and j PLUS ASCII value OF char_s1  
                    SET value_delete_s2 TO dp at position i and (j MINUS one) PLUS ASCII value OF char_s2  
                    IF value_delete_s1 LESS THAN value_delete_s2  
                        SET dp at position i and j TO value_delete_s1  
                    ELSE  
                        SET dp at position i and j TO value_delete_s2  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
