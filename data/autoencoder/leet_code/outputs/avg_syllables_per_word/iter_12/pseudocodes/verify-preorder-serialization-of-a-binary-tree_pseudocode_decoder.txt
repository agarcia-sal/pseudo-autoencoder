CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the result of splitting preorder by the comma character  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the hash character  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN the result of slots EQUALS zero  
    END FUNCTION  
END CLASS
