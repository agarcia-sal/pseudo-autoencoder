CLASS Solution  
    FUNCTION numSubarrayBoundedMax(nums, left, right)  
        FUNCTION count(max_value)  
            SET count TO zero  
            SET current_length TO zero  
            FOR each num IN nums  
                IF num LESS THAN OR EQUAL TO max_value  
                    INCREMENT current_length BY one  
                    INCREMENT count BY current_length  
                ELSE  
                    SET current_length TO zero  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET result_lower_bound TO count WITH max_value SET TO right  
        SET result_upper_bound TO count WITH max_value SET TO left MINUS one  
        RETURN result_lower_bound MINUS result_upper_bound  
    END FUNCTION  
END CLASS
