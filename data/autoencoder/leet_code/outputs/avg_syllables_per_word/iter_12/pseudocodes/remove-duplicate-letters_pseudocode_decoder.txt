CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO empty dictionary  
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            SET character in last_occurrence TO index  
        END FOR  
        
        SET stack TO empty list  
        SET in_stack TO empty set  
        
        FOR each index FROM zero TO LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            
            IF character IN in_stack  
                CONTINUE to next iteration  
            END IF  
            
            WHILE stack is not empty AND character LESS THAN element at last position of stack AND index LESS THAN last_occurrence of element at last position of stack  
                SET removed_char TO last element removed FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result TO empty string  
        FOR each character IN stack  
            APPEND character TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
