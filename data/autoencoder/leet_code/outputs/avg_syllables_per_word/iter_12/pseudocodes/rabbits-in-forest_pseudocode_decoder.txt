CLASS Solution  
    FUNCTION numRabbits WITH parameter answers  
        SET answer_counts TO RESULT OF Counter FUNCTION WITH answers AS argument  
        SET total_rabbits TO zero  
        FOR each pair answer AND count IN the ITEMS OF answer_counts  
            SET groups TO the INTEGER RESULT OF the SUM OF count PLUS answer DIVIDED BY the SUM OF answer PLUS one  
            INCREMENT total_rabbits BY the PRODUCT OF groups AND the SUM OF answer PLUS one  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
