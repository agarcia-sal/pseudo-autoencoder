CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO empty set  
        SET area TO zero  
        
        FOR each rectangle IN rectangles  
            SET x1 TO first element of rectangle  
            SET y1 TO second element of rectangle  
            SET x2 TO third element of rectangle  
            SET y2 TO fourth element of rectangle  
            
            IF x1 LESS THAN X1 THEN  
                SET X1 TO x1  
            END IF  
            
            IF y1 LESS THAN Y1 THEN  
                SET Y1 TO y1  
            END IF  
            
            IF x2 GREATER THAN X2 THEN  
                SET X2 TO x2  
            END IF  
            
            IF y2 GREATER THAN Y2 THEN  
                SET Y2 TO y2  
            END IF  
            
            SET area TO area PLUS (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)  
            
            FOR each point p IN list consisting of tuple of x1 and y1, tuple of x1 and y2, tuple of x2 and y1, tuple of x2 and y2  
                IF points CONTAINS p THEN  
                    REMOVE p FROM points  
                ELSE  
                    ADD p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area NOT EQUALS (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1) THEN  
            RETURN False  
        END IF  
        
        SET expected_points TO set consisting of tuple of X1 and Y1, tuple of X1 and Y2, tuple of X2 and Y1, tuple of X2 and Y2  
        
        IF points NOT EQUALS expected_points THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
