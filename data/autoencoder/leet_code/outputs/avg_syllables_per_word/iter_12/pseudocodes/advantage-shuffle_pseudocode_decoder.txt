CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ASCENDING ORDER  
        SET nums2_with_indices TO EMPTY LIST  
        FOR each pair of value and index IN enumeration of nums2  
            APPEND tuple of value and index TO nums2_with_indices  
        END FOR  
        SORT nums2_with_indices IN DESCENDING ORDER BY the first element of each tuple  
        SET result TO a list of zeroes with length EQUAL TO the LENGTH OF nums1  
        SET left TO zero  
        SET right TO the LENGTH OF nums1 MINUS one  
        FOR each pair of value and index IN nums2_with_indices  
            IF element at position right of nums1 GREATER THAN value  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
