CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO empty mapping from course to empty list for each course in range from zero to numCourses minus one  
        SET in_degrees TO list of zeros with length numCourses  
        
        FOR each pair dest and src IN prerequisites  
            APPEND dest TO the list at key src in adj_list  
            INCREMENT in_degrees at position dest BY one  
        END FOR  
        
        SET queue TO a double-ended queue containing each course in range from zero to numCourses minus one WHERE in_degrees at position course EQUALS zero  
        SET topological_order TO empty list  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND SET TO current  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT in_degrees at position neighbor BY one  
                IF in_degrees at position neighbor EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
