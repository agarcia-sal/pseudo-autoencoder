CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            
            FOR each index, character IN s with position index  
                IF character EQUALS one as a character  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count EQUALS zero  
                    APPEND concatenation of character one, result of dfs with substring of s from position start PLUS one TO position index MINUS one, character zero TO specials  
                    SET start TO index PLUS one  
                END IF  
            END FOR  
            
            SORT specials IN reverse lexicographical order  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        
        RETURN result of dfs with s  
    END FUNCTION  
END CLASS
