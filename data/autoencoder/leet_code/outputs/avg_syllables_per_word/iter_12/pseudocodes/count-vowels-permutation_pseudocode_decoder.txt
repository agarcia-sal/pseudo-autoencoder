CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        SET MOD TO one billion PLUS seven  
        SET a TO one  
        SET e TO one  
        SET i TO one  
        SET o TO one  
        SET u TO one  
        FOR each number FROM two TO n  
            SET a_next TO the sum of e PLUS i PLUS u  
            SET e_next TO the sum of a PLUS i  
            SET i_next TO the sum of e PLUS o  
            SET o_next TO i  
            SET u_next TO the sum of i PLUS o  
            SET a TO a_next MODULO MOD  
            SET e TO e_next MODULO MOD  
            SET i TO i_next MODULO MOD  
            SET o TO o_next MODULO MOD  
            SET u TO u_next MODULO MOD  
        END FOR  
        SET total TO the sum of a PLUS e PLUS i PLUS o PLUS u  
        RETURN total MODULO MOD  
    END FUNCTION  
END CLASS
