CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        CALL initializeList WITH size twenty six AND value negative one RETURNING last  
        CALL initializeList WITH size twenty six AND value negative one RETURNING last2  
        SET result TO zero  
        FOR each index i AND character char IN s  
            CALL getCharIndex WITH character char RETURNING idx  
            SET result TO result PLUS (i MINUS element at position idx of last) MULTIPLIED BY (element at position idx of last MINUS element at position idx of last2) MODULO MOD  
            SET element at position idx of last2 TO element at position idx of last  
            SET element at position idx of last TO i  
        END FOR  
        FOR each index i FROM zero TO twenty five  
            SET result TO result PLUS (n MINUS element at position i of last) MULTIPLIED BY (element at position i of last MINUS element at position i of last2) MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
  
    FUNCTION initializeList WITH parameters size AND value  
        SET newList TO an empty list  
        FOR index FROM zero TO size MINUS one  
            APPEND value TO newList  
        END FOR  
        RETURN newList  
    END FUNCTION  
  
    FUNCTION getCharIndex WITH parameter character  
        SET baseCharCode TO character code of uppercase letter A  
        SET charCode TO character code of character  
        SET index TO charCode MINUS baseCharCode  
        RETURN index  
    END FUNCTION  
END CLASS
