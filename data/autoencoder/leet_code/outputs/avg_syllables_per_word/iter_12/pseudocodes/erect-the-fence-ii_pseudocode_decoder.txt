CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            SET difference_in_first_coordinate TO element at position zero of p1 MINUS element at position zero of p2  
            SET difference_in_second_coordinate TO element at position one of p1 MINUS element at position one of p2  
            SET result TO difference_in_first_coordinate MULTIPLIED BY difference_in_first_coordinate PLUS difference_in_second_coordinate MULTIPLIED BY difference_in_second_coordinate  
            RETURN result  
        END FUNCTION  
  
        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO two MULTIPLIED BY (ax MULTIPLIED BY (by MINUS cy) PLUS bx MULTIPLIED BY (cy MINUS ay) PLUS cx MULTIPLIED BY (ay MINUS by))  
            SET ux_numerator TO (ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay) MULTIPLIED BY (by MINUS cy) PLUS (bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by) MULTIPLIED BY (cy MINUS ay) PLUS (cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy) MULTIPLIED BY (ay MINUS by)  
            SET ux TO ux_numerator DIVIDED BY d  
            SET uy_numerator TO (ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay) MULTIPLIED BY (cx MINUS bx) PLUS (bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by) MULTIPLIED BY (ax MINUS cx) PLUS (cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy) MULTIPLIED BY (bx MINUS ax)  
            SET uy TO uy_numerator DIVIDED BY d  
            SET distance_to_ax TO ux MINUS ax  
            SET distance_to_ay TO uy MINUS ay  
            SET radius TO (distance_to_ax MULTIPLIED BY distance_to_ax PLUS distance_to_ay MULTIPLIED BY distance_to_ay) RAISED TO THE POWER OF one DIVIDED BY two  
            RETURN a list containing ux and uy, radius  
        END FUNCTION  
  
        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the LENGTH OF boundary EQUALS three  
                IF the LENGTH OF boundary EQUALS zero  
                    RETURN a list containing zero and zero, zero  
                ELSE IF the LENGTH OF boundary EQUALS one  
                    RETURN element at position zero of boundary, zero  
                ELSE IF the LENGTH OF boundary EQUALS two  
                    SET center_first_coordinate TO (element at position zero at position zero of boundary PLUS element at position one at position zero of boundary) DIVIDED BY two  
                    SET center_second_coordinate TO (element at position zero at position one of boundary PLUS element at position one at position one of boundary) DIVIDED BY two  
                    SET center TO a list containing center_first_coordinate and center_second_coordinate  
                    SET distance_squared TO dist_sq(element at position zero of boundary, element at position one of boundary)  
                    SET radius TO distance_squared RAISED TO THE POWER OF one DIVIDED BY two DIVIDED BY two  
                    RETURN center, radius  
                ELSE  
                    RETURN circumcenter(element at position zero of boundary, element at position one of boundary, element at position two of boundary)  
                END IF  
            END IF  
  
            SET p TO element at position n MINUS one of points  
            SET center, radius TO welzl(points, boundary, n MINUS one)  
            SET distance_squared_center_p TO dist_sq(center, p)  
            IF distance_squared_center_p LESS THAN OR EQUAL TO radius MULTIPLIED BY radius  
                RETURN center, radius  
            END IF  
  
            RETURN welzl(points, boundary APPENDED WITH p, n MINUS one)  
        END FUNCTION  
  
        PERFORM random shuffle ON trees  
        SET center, radius TO welzl(trees, an empty list, the LENGTH OF trees)  
        RETURN a list containing element at position zero of center, element at position one of center, radius  
    END FUNCTION  
END CLASS
