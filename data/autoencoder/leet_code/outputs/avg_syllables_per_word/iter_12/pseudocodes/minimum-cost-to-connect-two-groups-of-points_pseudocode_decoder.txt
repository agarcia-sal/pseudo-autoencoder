CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        CALL initialize_dp_table WITH size1 AND size2 AND SET TO dp  
        SET element at position zero at position zero of dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two RAISED TO size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET new_mask TO bitwise OR of mask AND two RAISED TO j  
                    SET element at position i PLUS one at position new_mask of dp TO the MINIMUM OF element at position i PLUS one at position new_mask of dp AND the SUM OF element at position i at position mask of dp AND element at position i at position j of cost  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF bitwise AND of mask AND two RAISED TO j IS NOT zero  
                        SET element at position i PLUS one at position mask of dp TO the MINIMUM OF element at position i PLUS one at position mask of dp AND the SUM OF element at position i PLUS one at position bitwise XOR of mask AND two RAISED TO j of dp AND element at position i at position j of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN element at position size1 at position two RAISED TO size2 MINUS one of dp  
    END FUNCTION  

    FUNCTION initialize_dp_table(size1, size2)  
        CREATE dp AS list  
        FOR index FROM zero TO size1  
            APPEND to dp A list OF length two RAISED TO size2 FILLED WITH positive infinity  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
