CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO side REPLACED WITH plus SIGN REPLACED BY hash PLUS SIGN AND minus SIGN REPLACED BY hash MINUS SIGN SPLIT BY hash  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token EQUALS empty string  
                    CONTINUE to next token  
                END IF  
                IF token CONTAINS the character x  
                    IF token EQUALS x OR token EQUALS plus SIGN x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS minus SIGN x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY integer value of token WITH character x REMOVED  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY integer value of token  
                END IF  
            END FOR  
            RETURN x_count, num_sum  
        END FUNCTION  

        SET left_side, right_side TO equation SPLIT BY equal SIGN  
        SET left_x, left_num TO RESULT OF parse_side WITH left_side  
        SET right_x, right_num TO RESULT OF parse_side WITH right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string x EQUALS total_num INTEGER DIVIDED BY total_x  
        END IF  
    END FUNCTION  
END CLASS
