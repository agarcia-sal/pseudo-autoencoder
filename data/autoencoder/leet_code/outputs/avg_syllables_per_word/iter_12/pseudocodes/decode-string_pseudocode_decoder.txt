CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character is a digit  
                SET d TO d PLUS character  
            ELSE IF character is an alphabet letter  
                SET a TO a PLUS character  
            ELSE IF character EQUALS opening bracket  
                APPEND pair of a AND integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character EQUALS closing bracket  
                SET pair TO last element REMOVED FROM stack  
                SET p TO first element of pair  
                SET n TO second element of pair  
                SET a TO p PLUS a MULTIPLIED BY n  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
