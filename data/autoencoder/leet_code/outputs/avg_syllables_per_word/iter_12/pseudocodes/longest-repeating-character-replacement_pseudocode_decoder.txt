CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        CALL defaultdict WITH int TO CREATE count  
        
        FOR right FROM zero TO LENGTH OF s MINUS one  
            INCREMENT the element at position right of s IN count BY one  
            SET max_count TO the GREATER THAN BETWEEN max_count AND the element at position right of s IN count  
            
            IF the LENGTH OF the window FROM left TO right MINUS max_count IS GREATER THAN k  
                DECREMENT the element at position left of s IN count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER THAN BETWEEN max_length AND the LENGTH OF the window FROM left TO right  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
