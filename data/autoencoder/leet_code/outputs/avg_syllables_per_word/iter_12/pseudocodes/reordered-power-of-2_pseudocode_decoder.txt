CLASS Solution
    FUNCTION reorderedPowerOf2 WITH parameter n
        SET power_of_2_sorted_digits TO an empty set
        SET power_of_2 TO one
        WHILE power_of_2 LESS THAN OR EQUAL TO ten RAISED TO THE POWER OF nine
            ADD the tuple of the sorted characters of the string representation of power_of_2 TO power_of_2_sorted_digits
            SET power_of_2 TO power_of_2 MULTIPLIED BY two
        END WHILE
        SET sorted_n_digits TO the tuple of the sorted characters of the string representation of n
        RETURN whether sorted_n_digits IS IN power_of_2_sorted_digits
    END FUNCTION
END CLASS
