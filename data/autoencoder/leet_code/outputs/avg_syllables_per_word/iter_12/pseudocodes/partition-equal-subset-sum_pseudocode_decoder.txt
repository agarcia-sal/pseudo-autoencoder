CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO zero  
        FOR each number IN nums  
            INCREMENT total_sum BY number  
        END FOR  
        
        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        
        CALL initialize_dp WITH target RETURNING dp  
        SET element at position zero of dp TO True  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO dp at position i OR dp at position i MINUS num  
            END FOR  
        END FOR  
        
        RETURN dp at position target  
    END FUNCTION  
    
    FUNCTION initialize_dp(target)  
        CREATE list dp WITH length target PLUS one  
        FOR index FROM zero TO target  
            SET element at position index of dp TO False  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
