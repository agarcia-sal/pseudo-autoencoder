CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val AS element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode WITH val AS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode WITH val AS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS nums  
        IF nums EQUALS empty  
            RETURN None  
        END IF  
        SET max_val TO the maximum element IN nums  
        SET max_index TO the position of max_val IN nums  
        SET root TO new TreeNode WITH val AS max_val  
        SET root.left TO constructMaximumBinaryTree WITH PARAMETERS the slice of nums FROM position zero TO max_index MINUS one  
        SET root.right TO constructMaximumBinaryTree WITH PARAMETERS the slice of nums FROM position max_index PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
