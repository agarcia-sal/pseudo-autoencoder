CLASS Solution  
    FUNCTION longestConsecutive WITH PARAMETER nums AS list of integers  
        IF nums IS empty  
            RETURN zero  
        END IF  
        
        SET num_set TO the set containing each element of nums  
        SET longest_streak TO zero  
        
        FOR each num IN num_set  
            IF the value of num MINUS one IS NOT in num_set  
                SET current_num TO num  
                SET current_streak TO one  
                
                WHILE the value of current_num PLUS one IS in num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                
                SET longest_streak TO the GREATER of longest_streak AND current_streak  
            END IF  
        END FOR  
        
        RETURN longest_streak  
    END FUNCTION  
END CLASS
