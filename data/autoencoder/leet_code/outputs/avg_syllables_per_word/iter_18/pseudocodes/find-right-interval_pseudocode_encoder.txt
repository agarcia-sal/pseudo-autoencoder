CLASS Solution  
    FUNCTION findRightInterval WITH parameters intervals  
        SET start_times TO the result of sorting each pair formed by the first element of interval AND the index of interval for each interval WITH index taken from intervals  
        SET result TO an empty list  
        FOR each element ignored_and_end IN intervals  
            SET the variable end TO the second element of ignored_and_end  
            SET idx TO the position found by searching from the left in start_times for the first element where the first component of the tuple is greater than or equal to end  
            IF idx IS LESS THAN the LENGTH OF start_times THEN  
                APPEND the second element of the element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
