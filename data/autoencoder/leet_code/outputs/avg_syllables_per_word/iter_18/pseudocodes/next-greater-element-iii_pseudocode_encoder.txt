CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO the list of characters representing the number n  
        SET length TO the number of elements in digits  
          
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND character at position i of digits GREATER THAN OR EQUAL TO character at position i plus one of digits holds  
            DECREMENT i BY one  
        END WHILE  
          
        IF i EQUALS negative one THEN  
            RETURN negative one  
        END IF  
          
        SET j TO length MINUS one  
        WHILE character at position j of digits LESS THAN OR EQUAL TO character at position i of digits holds  
            DECREMENT j BY one  
        END WHILE  
          
        SET temporary variable TO character at position i of digits  
        SET character at position i of digits TO character at position j of digits  
        SET character at position j of digits TO temporary variable  
          
        SET digits TO concatenation of the elements from position zero to position i plus one of digits AND the elements from position i plus one to the end of digits in reversed order  
          
        SET result TO the integer value represented by concatenating all elements of digits into a continuous string  
          
        IF result GREATER THAN the number two raised to the power of thirty one MINUS one THEN  
            RETURN negative one  
        END IF  
          
        RETURN result  
    END FUNCTION  
END CLASS
