CLASS Solution  
    FUNCTION isAdditiveNumber WITH PARAMETER num  
        FUNCTION is_valid_sequence WITH PARAMETERS start first second  
            IF start EQUALS the LENGTH OF num  
                RETURN true  
            END IF  
            
            SET expected TO the string representation of first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring of num from position start to position start PLUS length NOT EQUALS expected  
                RETURN false  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH ARGUMENTS start PLUS length second the integer conversion of expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the substring of num from position zero to position i  
                SET second TO the substring of num from position i to position j  
                
                IF the LENGTH OF first GREATER THAN one AND the character at position zero of first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the character at position zero of second EQUALS zero  
                    CONTINUE  
                END IF  
                
                IF is_valid_sequence WITH ARGUMENTS j the integer conversion of first the integer conversion of second RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
