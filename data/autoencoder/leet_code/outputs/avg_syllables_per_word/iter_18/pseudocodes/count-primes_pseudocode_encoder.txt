CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n IS LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of true values for each integer from zero up to but not including n  
        SET element at position zero of is_prime TO false  
        SET element at position one of is_prime TO false  
        FOR each start FROM two TO the integer part of the square root of n INCLUSIVE  
            IF element at position start of is_prime IS true  
                FOR each multiple FROM start MULTIPLIED BY start TO one less than n IN STEPS OF start  
                    SET element at position multiple of is_prime TO false  
                END FOR  
            END IF  
        END FOR  
        SET total_primes TO the sum of all true values in is_prime  
        RETURN total_primes  
    END FUNCTION  
END CLASS
