CLASS Solution  
    FUNCTION removeComments(source AS List of String) RETURNS List of String  
        SET in_block_comment TO false  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the element at position i of line FOLLOWED BY the element at position i plus one of line EQUALS the character sequence slash followed by asterisk end of block comment indicator  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the element at position i of line FOLLOWED BY the element at position i plus one of line EQUALS the character sequence slash followed by slash comment indicator  
                        BREAK the current WHILE loop  
                    ELSE IF the element at position i of line FOLLOWED BY the element at position i plus one of line EQUALS the character sequence slash followed by asterisk start of block comment indicator  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line is not empty AND in_block_comment IS false  
                APPEND the concatenation of current_line AS string TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
