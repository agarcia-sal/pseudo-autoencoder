CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its associated count v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT the count associated to c IN cnt BY two  
                    CALL dfs WITH ARGUMENT the concatenation of c PLUS t PLUS c  
                    INCREMENT the count associated to c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  
          
        ASSIGN cnt TO the result of applying the Counter operation to s  
        SET mid TO an empty string  
        FOR each character c AND its associated count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty  
                    RETURN an empty list  
                END IF  
                SET mid TO c  
                DECREMENT the count associated to c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO an empty list  
        CALL dfs WITH ARGUMENT mid  
        RETURN ans  
    END FUNCTION  
END CLASS
