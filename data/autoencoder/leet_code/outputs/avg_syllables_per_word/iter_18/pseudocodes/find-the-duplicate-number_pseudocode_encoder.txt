CLASS Solution  
    FUNCTION findDuplicate(nums AS List OF Integer) RETURNS Integer  
        FUNCTION f(x AS Integer) RETURNS Boolean  
            SET count TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x THEN  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count GREATER THAN x THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        SET search_range TO range FROM zero TO the LENGTH OF nums MINUS one  
        SET result TO the bisect_left POSITION IN search_range WHERE the CONDITION f HOLDS true  
        RETURN result  
    END FUNCTION  
END CLASS
