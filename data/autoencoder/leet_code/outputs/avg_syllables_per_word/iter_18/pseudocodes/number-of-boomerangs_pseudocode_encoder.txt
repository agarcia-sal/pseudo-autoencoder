CLASS Solution  
    FUNCTION numberOfBoomerangs(points AS List OF List OF integer) RETURNS integer  
        SET ans TO zero  
        FOR each p1 IN points  
            SET cnt TO new Counter object  
            FOR each p2 IN points  
                SET dx TO element at position zero of p1 MINUS element at position zero of p2  
                SET dy TO element at position one of p1 MINUS element at position one of p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the value in cnt at key d  
                INCREMENT the value in cnt at key d BY one  
            END FOR  
        END FOR  
        SET result TO ans MULTIPLIED BY two  
        RETURN result  
    END FUNCTION  
END CLASS
