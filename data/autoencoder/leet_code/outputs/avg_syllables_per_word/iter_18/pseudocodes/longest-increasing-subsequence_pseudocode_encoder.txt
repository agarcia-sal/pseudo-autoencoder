CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS not having any elements THEN  
            RETURN zero  
        END IF  
  
        SET dp TO an empty list  
  
        FOR each num IN nums  
            SET idx TO the position found by applying binary search to find the leftmost suitable insertion location of num in dp  
  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
