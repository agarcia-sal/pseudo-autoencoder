CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULTED TO zero AND next DEFAULTED TO None  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO CREATE new ListNode WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values STARTING FROM position one TO the last position in values  
        SET node TO CREATE new ListNode WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN true  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN false  
    END IF  
    RETURN the value at p1 dot val EQUALS the value at p2 dot val AND CALL is_same_list WITH p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETERS head  
        FUNCTION reverseList WITH PARAMETERS node  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO the next element of node  
                SET the next element of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO CALL reverseList WITH head  
        
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT None AND carry EQUALS one  
            SET current dot val TO current dot val PLUS carry  
            SET carry TO the value of current dot val DIVIDED BY ten  
            SET current dot val TO the value of current dot val MODULO ten  
            
            IF the next element of current IS None AND carry EQUALS one  
                SET the next element of current TO CREATE new ListNode WITH carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next element of current  
        END WHILE  
        
        RETURN CALL reverseList WITH reversed_head  
    END FUNCTION  
END CLASS
