CLASS Solution  
    FUNCTION minimumTimeRequired WITH PARAMETERS jobs AS List of integers AND k AS integer  
        SET n TO the LENGTH OF jobs  
          
        SET subset_sums TO a list of zeroes with LENGTH equal TO two RAISED TO THE POWER OF n  
          
        FOR each index i FROM zero TO one LESS THAN n  
            FOR each integer mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET element AT POSITION mask OR one LEFT SHIFTED BY i OF subset_sums TO element AT POSITION mask OF subset_sums PLUS element AT POSITION i OF jobs  
            END FOR  
        END FOR  
          
        FUNCTION dp WITH PARAMETERS k AND mask  
            IF k EQUALS one  
                RETURN element AT POSITION mask OF subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask HOLDS  
                SET candidate TO THE GREATER VALUE BETWEEN dp WITH ARGUMENTS k MINUS one AND mask XOR submask AND element AT POSITION submask OF subset_sums  
                SET res TO THE SMALLER VALUE BETWEEN res AND candidate  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
          
        RETURN dp WITH ARGUMENTS k AND two RAISED TO THE POWER OF n MINUS one  
END CLASS
