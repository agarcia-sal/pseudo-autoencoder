CLASS Solution  
    FUNCTION canWin WITH PARAMETER currentState  
        SET memo TO an empty mapping  
  
        FUNCTION can_win_helper WITH PARAMETER state  
            IF state EXISTS IN memo THEN  
                RETURN the value associated with state IN memo  
            END IF  
  
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring of state from position index TO position index PLUS one EQUALS two consecutive plus symbols THEN  
                    CREATE new_state BY concatenating the substring of state from start TO position index WITH two consecutive minus symbols WITH the substring of state from position index PLUS two TO the end  
                    IF can_win_helper CALLED WITH new_state RETURNS false THEN  
                        SET the value associated with state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
  
            SET the value associated with state IN memo TO false  
            RETURN false  
        END FUNCTION  
  
        RETURN can_win_helper CALLED WITH currentState  
    END FUNCTION  
END CLASS
