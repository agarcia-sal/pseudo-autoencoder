CLASS Solution  
    FUNCTION findMaximizedCapital WITH PARAMETERS k INTEGER w INTEGER profits LIST OF INTEGER capital LIST OF INTEGER RETURNS INTEGER  
        SET projects TO the LIST OF PAIRS FOR each POSITION index IN the CAPITAL AND PROFITS WHERE each PAIR CONTAINS the ELEMENT AT POSITION index OF CAPITAL AND the ELEMENT AT POSITION index OF PROFITS  
        SORT projects IN ASCENDING ORDER BASED ON the CAPITAL VALUE IN EACH PAIR  
          
        SET max_heap TO the EMPTY LIST REPRESENTING the MAXIMUM HEAP FOR PROFITS  
          
        SET i TO zero  
        SET n TO the LENGTH OF projects  
          
        FOR each iteration FROM one TO k  
            WHILE i IS LESS THAN n AND the CAPITAL VALUE AT POSITION i OF projects IS LESS THAN OR EQUAL TO w  
                PUSH the NEGATIVE OF the PROFIT VALUE AT POSITION i OF projects ONTO max_heap  
                INCREMENT i BY one  
            END WHILE  
              
            IF max_heap IS EMPTY  
                BREAK the LOOP  
            END IF  
              
            SET w TO w PLUS the NEGATIVE OF THE RESULT OF REMOVING THE HIGHEST PRIORITY ELEMENT FROM max_heap  
        END FOR  
          
        RETURN w  
    END FUNCTION  
END CLASS
