CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values HOLDS NO ELEMENTS  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW EMPTY DEQUE  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO REMOVE FIRST ELEMENT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS value at left child of p AND value at left child of q AND is_same_tree WITH PARAMETERS value at right child of p AND value at right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue WITH PARAMETERS self root  
        SET queue TO A NEW DEQUE INITIALIZED WITH root AS ONLY ELEMENT  
        SET leftmost_value TO none  
        WHILE queue HOLDS ELEMENTS  
            SET level_size TO THE LENGTH OF queue  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO REMOVE FIRST ELEMENT OF queue  
                IF index EQUALS zero  
                    SET leftmost_value TO value of node  
                END IF  
                IF left child of node IS NOT none  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node IS NOT none  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
