CLASS Solution  
    FUNCTION smallestSubsequence WITH PARAMETERS s letter k repetition  
        SET letter_count TO the number of occurrences of letter in s  
        SET stack TO an empty list  
        FOR each index i and character char in the enumeration of s  
            WHILE the length of stack is greater than zero AND character char is lexicographically smaller than the element at the last position of stack AND the sum of the length of s minus i plus the length of stack minus one is greater than or equal to k  
                IF the element at the last position of stack EQUALS letter  
                    IF letter_count is greater than repetition  
                        INCREMENT repetition BY one  
                        REMOVE the element at the last position from stack  
                    ELSE  
                        BREAK out of the WHILE loop  
                    END IF  
                ELSE  
                    REMOVE the element at the last position from stack  
                END IF  
            END WHILE  
            IF the length of stack is less than k  
                IF character char EQUALS letter  
                    APPEND character char TO stack  
                    DECREMENT repetition BY one  
                ELSE IF the difference of k minus the length of stack is greater than repetition  
                    APPEND character char TO stack  
                END IF  
                IF character char EQUALS letter  
                    DECREMENT letter_count BY one  
                END IF  
            END IF  
        END FOR  
        WHILE the length of stack is less than k  
            APPEND letter TO stack  
        END WHILE  
        SET result_string TO the concatenation of all elements in stack  
        RETURN result_string  
    END FUNCTION  
END CLASS
