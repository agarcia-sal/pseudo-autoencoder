CLASS Solution  
    FUNCTION minMeetingRooms(intervals AS List OF List OF integer)  
        IF intervals EQUALS an empty collection  
            RETURN zero  
        END IF  
          
        INVOKE an auxiliary FUNCTION sort_intervals_by_start_time WITH intervals  
          
        SET min_heap TO an empty collection representing a minimum heap  
          
        FOR each meeting IN intervals  
            IF the collection min_heap is not empty AND the element at position zero of meeting IS GREATER THAN OR EQUAL TO the element at position zero of element at position zero of min_heap  
                INVOKE an auxiliary FUNCTION remove_smallest_element FROM min_heap  
            END IF  
              
            INVOKE an auxiliary FUNCTION add_element TO min_heap PASSING the element at position one of meeting  
        END FOR  
          
        SET number_of_rooms_required TO the LENGTH OF min_heap  
        RETURN number_of_rooms_required  
    END FUNCTION  
END CLASS
