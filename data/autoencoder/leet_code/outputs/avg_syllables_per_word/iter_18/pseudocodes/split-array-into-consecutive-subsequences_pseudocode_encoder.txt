CLASS Solution  
    FUNCTION isPossible(nums AS List of integer) AS boolean  
        SET num_count TO a new Counter initialized with nums  
        SET end_with TO a new Counter initialized as empty  
        
        FOR each num IN nums  
            IF the count of num in num_count EQUALS zero THEN  
                CONTINUE  
            END IF  
            
            IF the count of number that is one less than num in end_with IS GREATER THAN zero THEN  
                DECREMENT the count of number that is one less than num in end_with BY one  
                INCREMENT the count of num in end_with BY one  
            ELSE IF the count of number that is one greater than num in num_count IS GREATER THAN zero AND the count of number that is two greater than num in num_count IS GREATER THAN zero THEN  
                DECREMENT the count of number that is one greater than num in num_count BY one  
                DECREMENT the count of number that is two greater than num in num_count BY one  
                INCREMENT the count of number that is two greater than num in end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the count of num in num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
