CLASS Solution  
    FUNCTION minInterval WITH PARAMETERS intervals and queries  
        SET intervals TO intervals SORTED IN ASCENDING ORDER OF start time  
        SET sorted_queries TO list OF each query paired WITH its original position IN queries SORTED BY query value  
        SET min_heap TO empty list  
        SET result TO list FILLED WITH negative one REPEATED for the LENGTH OF queries  
        SET interval_index TO zero  
        FOR each query_value and original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND the element at position zero OF element at position interval_index OF intervals LESS THAN OR EQUAL TO query_value holds  
                SET start TO the element at position zero OF element at position interval_index OF intervals  
                SET end TO the element at position one OF element at position interval_index OF intervals  
                SET interval_size TO end MINUS start PLUS one  
                INSERT interval_size AND end AS A TUPLE INTO min_heap PRESERVING MINIMUM AT TOP  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT EMPTY AND the second element OF the first tuple IN min_heap LESS THAN query_value holds  
                REMOVE THE FIRST TUPLE FROM min_heap WITH THE SMALLEST INTERVAL SIZE  
            END WHILE  
            IF min_heap IS NOT EMPTY  
                SET the element at position original_index OF result TO the first element OF the first tuple IN min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
