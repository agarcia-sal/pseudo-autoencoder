CLASS TreeNode  
    FUNCTION __init__(self val EQUALS zero left EQUALS none right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO element removed from front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUAL TO none THEN  
            SET node dot left TO new TreeNode with element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUAL TO none THEN  
            SET node dot right TO new TreeNode with element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF p dot val NOT EQUAL TO q dot val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p dot left q dot left) AND is_same_tree(p dot right q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self preorder inorder)  
        IF preorder EQUALS empty collection OR inorder EQUALS empty collection THEN  
            RETURN none  
        END IF  
        SET root_val TO element at position zero of preorder  
        SET root TO new TreeNode with root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root dot left TO self dot buildTree(subcollection of preorder from position one TO root_index_inorder inclusive subcollection of inorder from position zero TO root_index_inorder exclusive)  
        SET root dot right TO self dot buildTree(subcollection of preorder from position root_index_inorder PLUS one TO end subcollection of inorder from position root_index_inorder PLUS one TO end)  
        RETURN root  
    END FUNCTION  
END CLASS
