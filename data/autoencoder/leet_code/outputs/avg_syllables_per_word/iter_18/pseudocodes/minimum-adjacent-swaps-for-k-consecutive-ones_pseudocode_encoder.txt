CLASS Solution  
    FUNCTION minMoves(nums AS list OF integer, k AS integer) RETURNS integer  
        SET positions TO an empty list  
        FOR each index and number IN enumeration of nums  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
  
        FUNCTION calculate_cost(start AS integer, end AS integer) RETURNS integer  
            SET mid TO the integer division of the sum of start and end by two  
            SET median TO the element at position mid of positions  
            SET cost TO zero  
            FOR each index FROM start TO end minus one  
                INCREMENT cost BY the absolute difference BETWEEN the element at position index of positions AND the difference BETWEEN median AND the difference BETWEEN mid AND index  
            END FOR  
            RETURN cost  
        END FUNCTION  
  
        SET min_cost TO positive infinity  
        FOR index FROM zero TO the length of positions MINUS k  
            SET current_cost TO calculate_cost(index, index PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
