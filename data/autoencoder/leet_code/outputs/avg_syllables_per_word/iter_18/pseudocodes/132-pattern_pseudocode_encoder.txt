CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each number IN the reversed sequence of nums  
            IF number LESS THAN second  
                RETURN true  
            END IF  
            WHILE the stack is not empty AND element at the last position of stack LESS THAN number  
                SET second TO element removed from the last position of stack  
            END WHILE  
            APPEND number TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
