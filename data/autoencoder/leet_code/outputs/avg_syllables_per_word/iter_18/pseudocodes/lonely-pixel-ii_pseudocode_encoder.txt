CLASS Solution  
    FUNCTION findBlackPixel WITH PARAMETERS picture AS list of list of strings AND target AS integer  
        IF picture IS empty OR element at position zero of picture IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zero repeated m times  
        SET col_count TO a list of zero repeated n times  
        SET row_patterns TO a new mapping with default integer zero  
        
        FOR index r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of each element IN element at position r of picture as a single string  
            INCREMENT the value for key pattern IN row_patterns BY one  
            FOR index c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character designated as black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR index r FROM zero TO m MINUS one  
            FOR index c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character designated as black pixel  
                    SET pattern TO the concatenation of each element IN element at position r of picture as a single string  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the value for key pattern IN row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
