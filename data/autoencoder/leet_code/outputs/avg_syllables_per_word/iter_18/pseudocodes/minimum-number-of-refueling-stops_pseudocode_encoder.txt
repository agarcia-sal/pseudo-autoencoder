CLASS Solution  
    FUNCTION minRefuelStops WITH PARAMETERS target startFuel stations  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing position equal to target and fuel amount equal to zero TO stations  

        FOR each position and fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is less than position  
                SET largest_fuel_available TO the removal of the smallest element from max_heap  
                SET current_fuel TO current_fuel MINUS largest_fuel_available  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel is less than position  
                RETURN negative one  
            END IF  

            ADD the negation of fuel TO max_heap in a manner preserving heap order  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
