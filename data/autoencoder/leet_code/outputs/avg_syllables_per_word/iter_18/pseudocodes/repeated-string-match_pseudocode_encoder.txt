CLASS Solution  
    FUNCTION repeatedStringMatch WITH PARAMETERS a AS string AND b AS string  
        SET minimum_repetitions TO the negative of the integer division of the negative of the length of b BY the length of a  
        IF the string b IS a substring of the string a repeated minimum_repetitions times  
            RETURN minimum_repetitions  
        END IF  
        IF the string b IS a substring of the string a repeated minimum_repetitions PLUS one time  
            RETURN minimum_repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
