CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        IF the string str1 EQUALS the string str2  
            RETURN True  
        END IF  

        SET conversion_map TO an empty mapping from characters to characters  

        FOR each character char1 AND character char2 IN pairwise elements of str1 and str2  
            IF the character char1 IS A key IN conversion_map  
                IF the value at key char1 in conversion_map NOT EQUALS the character char2  
                    RETURN False  
                END IF  
            ELSE  
                SET the value at key char1 IN conversion_map TO the character char2  
            END IF  
        END FOR  

        IF the number of unique characters in string str2 EQUALS the number twenty six  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
