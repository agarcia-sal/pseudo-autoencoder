CLASS Solution  
    FUNCTION findContestMatch WITH parameter n  
        SET teams TO list of strings representing each integer starting from one UP TO AND INCLUDING n  
        WHILE n GREATER THAN one holds  
            FOR i FROM zero TO the integer division of n BY two MINUS one  
                SET team at position i TO a string consisting of an opening parenthesis concatenated with team at position i concatenated with a comma concatenated with team at position given by n MINUS one MINUS i concatenated with a closing parenthesis  
            END FOR  
            SET n TO the integer division of n BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
