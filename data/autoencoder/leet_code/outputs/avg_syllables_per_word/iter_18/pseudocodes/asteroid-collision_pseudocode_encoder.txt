CLASS Solution  
    FUNCTION asteroidCollision WITH parameter asteroids OF TYPE List of integer  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND the element at position the LENGTH OF stack MINUS one of stack IS GREATER THAN zero  
                IF the element at position the LENGTH OF stack MINUS one of stack IS LESS THAN the NEGATION OF asteroid  
                    REMOVE the element at position the LENGTH OF stack MINUS one FROM stack  
                    CONTINUE  
                ELSE IF the element at position the LENGTH OF stack MINUS one of stack EQUALS the NEGATION OF asteroid  
                    REMOVE the element at position the LENGTH OF stack MINUS one FROM stack  
                END IF  
                BREAK  
            END WHILE  
            IF the WHILE loop terminates normally WITHOUT BREAK  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
