CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char represents a digit character  
                SET d TO d concatenated with char  
            ELSE IF char represents an alphabetical character  
                SET a TO a concatenated with char  
            ELSE IF char is equal to the left bracket character  
                APPEND tuple of a and integer conversion of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char is equal to the right bracket character  
                SET p and n TO the last tuple removed from stack  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
