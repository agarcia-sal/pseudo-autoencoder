CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs AS list of strings  
        SET anagrams TO a new mapping with default value as empty list  
        FOR each element s IN strs  
            SET sorted_str TO the result of concatenating characters of s arranged in ascending alphabetical order  
            APPEND s TO the list at key sorted_str in anagrams  
        END FOR  
        SET result_list TO the list of all values in anagrams  
        RETURN result_list  
    END FUNCTION  
END CLASS
