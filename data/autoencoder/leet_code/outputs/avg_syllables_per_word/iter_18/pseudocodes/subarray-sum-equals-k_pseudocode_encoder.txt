CLASS Solution  
    FUNCTION subarraySum(nums AS List OF integer, k AS integer) RETURNS integer  
        SET cumulative_sum_freq TO mapping FROM zero TO one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF current_sum MINUS k IN cumulative_sum_freq  
                INCREMENT count BY the element at key current_sum MINUS k OF cumulative_sum_freq  
            END IF  
            IF current_sum IN cumulative_sum_freq  
                INCREMENT the element at key current_sum OF cumulative_sum_freq BY one  
            ELSE  
                SET the element at key current_sum OF cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
