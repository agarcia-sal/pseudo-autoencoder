CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus symbol  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus symbol  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS multiplication symbol  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
      
        FUNCTION helper(sub_expr)  
            IF sub_expr REPRESENTS only digit characters  
                RETURN list containing the integer value of sub_expr  
            END IF  
          
            SET results TO empty list  
            FOR index FROM zero TO the LENGTH OF sub_expr MINUS one  
                IF the character at position index of sub_expr IS one of plus symbol OR minus symbol OR multiplication symbol  
                    SET left_results TO the value RETURNED BY helper with the substring of sub_expr from the start UP TO but NOT INCLUDING position index  
                    SET right_results TO the value RETURNED BY helper with the substring of sub_expr from position index PLUS one TO the end  
                    APPEND all the elements RETURNED BY compute when GIVEN left_results AND right_results AND the character at position index of sub_expr TO results  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  
      
        RETURN the value RETURNED BY helper with expression  
    END FUNCTION  
END CLASS
