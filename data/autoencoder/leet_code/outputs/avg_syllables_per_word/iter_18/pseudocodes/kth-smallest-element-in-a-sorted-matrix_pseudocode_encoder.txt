CLASS Solution  
    FUNCTION kthSmallest WITH PARAMETERS matrix AND k  
        FUNCTION countLessEqual WITH PARAMETER mid  
            SET count TO zero  
            SET row TO the LENGTH OF matrix MINUS one  
            SET col TO zero  
            WHILE row IS GREATER THAN OR EQUAL TO zero AND col IS LESS THAN the LENGTH OF matrix  
                IF the element at position row and col of matrix IS LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET n TO the LENGTH OF matrix  
        SET low TO the element at position zero and zero of matrix  
        SET high TO the element at position n MINUS one and n MINUS one of matrix  
        WHILE low IS LESS THAN high  
            SET mid TO the integer DIVISION OF low PLUS high BY two  
            IF the CALL to countLessEqual with mid IS LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
