CLASS Solution  
    FUNCTION singleNonDuplicate WITH parameter nums  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left LESS THAN right HOLDS  
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two USING INTEGER DIVISION  
            IF the remainder of mid DIVIDED BY two EQUALS one  
                DECREMENT mid BY one  
            END IF  
            IF the element at position mid of nums EQUALS the element at position mid PLUS one of nums  
                SET left TO mid PLUS two  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN the element at position left of nums  
    END FUNCTION  
END CLASS
