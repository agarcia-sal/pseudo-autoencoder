CLASS Solution  
    FUNCTION shortestPalindrome WITH PARAMETER s  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome WITH PARAMETER check_str  
            RETURN check_str EQUALS reverse of check_str  
        END FUNCTION  
        
        FOR i FROM length of s DOWN TO zero  
            IF is_palindrome OF substring of s from start position one to position i  
                SET to_add TO reverse of substring of s from position i plus one to end  
                RETURN concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
