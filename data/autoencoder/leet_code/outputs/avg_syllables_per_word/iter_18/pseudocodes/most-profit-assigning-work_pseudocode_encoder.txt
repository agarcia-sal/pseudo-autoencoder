CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty AS List OF Integer profit AS List OF Integer worker AS List OF Integer) RETURNS Integer  
        SET jobs TO the RESULT OF auxiliary FUNCTION zip_lists WITH difficulty AND profit  
        CALL auxiliary FUNCTION sort_list WITH jobs  
        CALL auxiliary FUNCTION sort_list WITH worker  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND element at position job_index OF jobs AT position zero IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO the GREATER OF best_profit AND element at position job_index OF jobs AT position one  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
