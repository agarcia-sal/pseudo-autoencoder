CLASS Solution  
    FUNCTION countBattleships WITH parameter board AS list of list of string  
        SET count TO zero  
        FOR index i FROM zero TO the LENGTH OF board MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF element at position i and j of board EQUALS letter X  
                    IF (i EQUALS zero OR element at position i MINUS one and j of board EQUALS single period) AND (j EQUALS zero OR element at position i and j MINUS one of board EQUALS single period)  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
