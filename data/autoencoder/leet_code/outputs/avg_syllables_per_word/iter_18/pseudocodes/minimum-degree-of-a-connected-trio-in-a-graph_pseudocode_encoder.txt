CLASS Solution  
    FUNCTION minTrioDegree(n AS integer, edges AS list of list of integer) RETURNS integer  
        SET graph TO empty mapping from integer TO set of integer  
        FOR each index FROM one TO n  
            SET element at key index in graph TO empty set  
        END FOR  
        
        FOR each pair u and v IN edges  
            APPEND v TO the set at key u in graph  
            APPEND u TO the set at key v in graph  
        END FOR  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR each i FROM one TO n  
            FOR each j FROM one greater than i TO n  
                FOR each k FROM one greater than j TO n  
                    IF j IS IN the set at key i in graph AND k IS IN the set at key i in graph AND k IS IN the set at key j in graph THEN  
                        SET trio_found TO true  
                        SET degree_i TO the length of the set at key i in graph  
                        SET degree_j TO the length of the set at key j in graph  
                        SET degree_k TO the length of the set at key k in graph  
                        SET trio_degree TO degree_i PLUS degree_j PLUS degree_k MINUS six  
                        IF trio_degree IS LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found THEN  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
