CLASS Solution  
    FUNCTION shortestSubarray WITH PARAMETERS nums AS list of integers AND k AS integer  
        SET prefix_sum TO list containing single element zero  
        FOR each num IN nums  
            APPEND the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        SET dq TO empty double ended queue  
        SET min_length TO representation of infinity  
        
        FOR each index i AND element current_sum IN prefix_sum traversed with enumeration  
            WHILE dq is not empty AND current_sum MINUS element at position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET candidate_length TO i MINUS the element removed from the front of dq  
                IF candidate_length IS LESS THAN min_length  
                    SET min_length TO candidate_length  
                END IF  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at the last position of dq in prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS EQUAL TO representation of infinity  
            RETURN negative one  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
