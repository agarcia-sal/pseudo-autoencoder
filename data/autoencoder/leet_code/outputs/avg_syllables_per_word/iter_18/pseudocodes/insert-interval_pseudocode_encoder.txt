CLASS Solution  
    FUNCTION insert(intervals AS list of lists of integers newInterval AS list of integers) RETURNS list of lists of integers  
        IF intervals EQUALS an empty list THEN  
            RETURN a list containing newInterval  
        END IF  
        
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        
        WHILE i LESS THAN n AND the element at position one of the element at position i of intervals IS LESS THAN the element at position zero of newInterval HOLDS  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i LESS THAN n AND the element at position zero of the element at position i of intervals IS LESS THAN OR EQUAL TO the element at position one of newInterval HOLDS  
            SET the element at position zero of newInterval TO the lesser value BETWEEN the element at position zero of newInterval AND the element at position zero of the element at position i of intervals  
            SET the element at position one of newInterval TO the greater value BETWEEN the element at position one of newInterval AND the element at position one of the element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n HOLDS  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
