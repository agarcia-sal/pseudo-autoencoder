CLASS Solution  
    FUNCTION smallestRange(nums AS list of list of integers) AS list of integers  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push TO min_heap WITH element at position zero of list at position index of nums and index and zero  
            IF max_value IS LESS THAN element at position zero of list at position index of nums THEN  
                SET max_value TO element at position zero of list at position index of nums  
            END IF  
        END FOR  
        
        SET smallest_range TO infinity  
        SET result TO a list consisting of minus one hundred thousand and one hundred thousand  
        
        WHILE min_heap is NOT empty  
            CALL heap pop FROM min_heap AND ASSIGN TO min_value list_index element_index  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list consisting of min_value and max_value  
            END IF  
            
            IF element_index PLUS one IS LESS THAN the LENGTH OF list at position list_index of nums THEN  
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums  
                CALL heap push TO min_heap WITH next_value and list_index and element_index PLUS one  
                IF max_value IS LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
