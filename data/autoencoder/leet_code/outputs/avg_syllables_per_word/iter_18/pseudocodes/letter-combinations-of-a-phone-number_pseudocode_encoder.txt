CLASS Solution  
    FUNCTION letterCombinations WITH PARAMETER digits representing a sequence of characters  
        IF digits does not contain any characters  
            RETURN an empty list  
        END IF  
        
        SET phone_map TO a mapping of digit characters TO corresponding strings of characters as follows  
            key digit two maps TO the string of letters a b and c  
            key digit three maps TO the string of letters d e and f  
            key digit four maps TO the string of letters g h and i  
            key digit five maps TO the string of letters j k and l  
            key digit six maps TO the string of letters m n and o  
            key digit seven maps TO the string of letters p q r and s  
            key digit eight maps TO the string of letters t u and v  
            key digit nine maps TO the string of letters w x y and z  
        
        FUNCTION backtrack WITH PARAMETERS combination representing a string AND next_digits representing a string  
            IF the length of next_digits equals zero  
                APPEND combination TO the list output  
            ELSE  
                FOR each letter IN the string mapped by the digit at position zero in next_digits of the mapping phone_map  
                    CALL backtrack WITH PARAMETERS the concatenation of combination AND letter AND the substring of next_digits from position one to the end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO an empty list  
        CALL backtrack WITH PARAMETERS the empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
