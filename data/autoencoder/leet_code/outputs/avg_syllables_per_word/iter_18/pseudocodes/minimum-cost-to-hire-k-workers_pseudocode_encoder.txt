CLASS Solution  
    FUNCTION mincostToHireWorkers(quality AS List OF integer, wage AS List OF integer, k AS integer) RETURNS floating point number  
        SET workers TO a list sorted by each element being the ratio of wage at position index divided by quality at position index paired with quality at position index FOR each corresponding pair of wage and quality accessed simultaneously  
        SET max_heap TO an empty list representing a priority queue operating as a maximum heap with qualities negated  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each ratio and q IN workers  
            PERFORM operation to push negative value of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF the length of max_heap is greater than k THEN  
                SET removed_quality TO the result of popping the smallest element from max_heap  
                INCREMENT total_quality BY removed_quality  
            END IF  
            IF the length of max_heap is equal to k THEN  
                SET possible_cost TO total_quality MULTIPLIED BY ratio  
                IF possible_cost LESS THAN min_cost THEN  
                    SET min_cost TO possible_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
