CLASS Solution  
    FUNCTION findLUSlength WITH PARAMETER strs AS list of strings  
        FUNCTION is_subsequence WITH PARAMETERS s AS string AND t AS string  
            SET iterator_for_t TO the iterator derived FROM t  
            RETURN the condition THAT for each character c IN s the character c APPEARS in sequence IN iterator_for_t  
        END FUNCTION  
          
        SORT the list strs BY sorting criterion DESCRIBED AS negative length of element AND element itself IN lexicographic order  
          
        FOR each index i FROM zero TO the length OF strs MINUS one AND corresponding element word AT position i OF strs  
            SET unique_flag TO true  
            FOR each index j FROM zero TO the length OF strs MINUS one AND corresponding element other AT position j OF strs  
                IF index i IS NOT EQUAL TO index j AND the result of is_subsequence WITH PARAMETERS word AND other IS true  
                    SET unique_flag TO false  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
            IF unique_flag IS true  
                RETURN the length OF word  
            END IF  
        END FOR  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
