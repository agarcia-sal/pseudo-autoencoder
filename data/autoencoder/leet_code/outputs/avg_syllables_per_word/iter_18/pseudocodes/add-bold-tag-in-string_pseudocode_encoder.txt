CLASS Solution  
    FUNCTION addBoldTag(s AS string, words AS list of strings) AS string  
        IF words EQUALS an empty list THEN  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        SET mask TO a list of boolean values FALSE repeated n times  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s STARTING AT start  
                IF start EQUALS negative one THEN  
                    BREAK the loop  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i OF mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the element at position i OF mask IS TRUE THEN  
                APPEND the string representing a beginning bold tag TO result  
                WHILE i IS LESS THAN n AND the element at position i OF mask IS TRUE  
                    APPEND the character at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string representing an ending bold tag TO result  
            ELSE  
                APPEND the character at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements of result  
    END FUNCTION  
END CLASS
