CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH parameter element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value at attribute val of p IS NOT EQUALS value at attribute val of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH parameters value at attribute left of p AND value at attribute left of q AND the result of is_same_tree WITH parameters value at attribute right of p AND value at attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST WITH PARAMETERS self root  
        FUNCTION validate WITH PARAMETERS node low EQUALS negative infinity high EQUALS positive infinity  
            IF node IS none  
                RETURN True  
            END IF  
            IF value at attribute val of node IS LESS THAN OR EQUAL TO low OR value at attribute val of node IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN the result of validate WITH parameters value at attribute left of node low AND value at attribute val of node AND the result of validate WITH parameters value at attribute right of node value at attribute val of node AND high  
        END FUNCTION  
        RETURN validate WITH parameter root  
    END FUNCTION  
END CLASS
