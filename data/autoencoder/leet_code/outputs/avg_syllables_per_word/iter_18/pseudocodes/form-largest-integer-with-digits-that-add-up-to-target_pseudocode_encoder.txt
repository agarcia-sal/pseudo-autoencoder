CLASS Solution  
    FUNCTION largestNumber(cost AS List OF integer, target AS integer) AS string  
        SET cost_to_digit TO an empty mapping from cost TO string digit  
        FOR each index FROM zero TO the LENGTH OF cost MINUS one  
            SET current_cost TO element at position index of cost  
            SET corresponding_digit TO string representation of index PLUS one  
            SET key current_cost IN cost_to_digit TO corresponding_digit  
        END FOR  
      
        SET dp TO a list whose element at position zero is an empty string AND every other element UP TO position target IS the string representing negative one  
      
        FOR each number_t FROM one TO target  
            FOR each key_cost and value_digit IN cost_to_digit  
                IF number_t GREATER THAN OR EQUAL TO key_cost AND element at position number_t MINUS key_cost of dp IS NOT the string representing negative one THEN  
                    SET candidate TO value_digit CONCATENATED WITH element at position number_t MINUS key_cost of dp  
                    IF element at position number_t of dp EQUALS the string representing negative one OR LENGTH OF candidate GREATER THAN LENGTH OF element at position number_t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position number_t of dp AND candidate GREATER THAN element at position number_t of dp) THEN  
                        SET element at position number_t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
      
        IF element at position target of dp NOT EQUALS the string representing negative one THEN  
            RETURN element at position target of dp  
        ELSE  
            RETURN the string representing zero  
        END IF  
    END FUNCTION  
END CLASS
