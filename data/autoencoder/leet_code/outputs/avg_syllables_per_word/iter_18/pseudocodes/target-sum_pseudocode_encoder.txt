CLASS Solution  
    FUNCTION findTargetSumWays(nums AS List of integers, target AS integer) RETURNS integer  
        FUNCTION dfs(index AS integer, current_sum AS integer, memo AS mapping) RETURNS integer  
            IF index EQUALS the LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE consisting of index AND current_sum EXISTS IN memo THEN  
                RETURN the VALUE at the KEY composed of index AND current_sum IN memo  
            END IF  
            SET positive TO THE RESULT OF dfs WITH the ARGUMENTS BEING index INCREMENTED BY one AND current_sum INCREMENTED BY the element at position index OF nums AND memo  
            SET negative TO THE RESULT OF dfs WITH the ARGUMENTS BEING index INCREMENTED BY one AND current_sum DECREMENTED BY the element at position index OF nums AND memo  
            SET the VALUE at the KEY composed of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE at the KEY composed of index AND current_sum IN memo  
        END FUNCTION  
        RETURN the RESULT OF dfs WITH ARGUMENTS zero AND zero AND AN EMPTY mapping  
    END FUNCTION  
END CLASS
