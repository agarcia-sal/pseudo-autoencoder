CLASS Solution  
    FUNCTION wiggleSort(nums AS List OF Integer)  
        CALL FUNCTION sort_array(nums)  
        SET sorted_nums TO CALL FUNCTION copy_array(nums)  
        SET n TO CALL FUNCTION length_of_collection(nums)  
        SET left TO the INTEGER RESULT OF n MINUS one DIVIDED BY two  
        SET right TO n MINUS one  
        FOR index FROM zero TO n MINUS one  
            IF remainder WHEN index IS DIVIDED BY two EQUALS zero THEN  
                SET element at position index of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position index of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  

    FUNCTION sort_array(collection)  
        PERFORM in-place sorting of collection in ascending order  
    END FUNCTION  

    FUNCTION copy_array(collection)  
        CREATE new_collection AS a duplicate of collection preserving order  
        RETURN new_collection  
    END FUNCTION  

    FUNCTION length_of_collection(collection)  
        RETURN the number of elements contained in collection  
    END FUNCTION  
END CLASS
