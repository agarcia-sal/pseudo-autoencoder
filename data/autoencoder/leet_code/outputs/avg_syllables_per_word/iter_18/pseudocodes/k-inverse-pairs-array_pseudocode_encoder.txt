CLASS Solution  
    FUNCTION kInversePairs WITH PARAMETERS n AND k  
        SET MOD TO one billion PLUS seven  
        SET dp TO a two dimensional list with n PLUS one rows AND k PLUS one columns EACH initialized TO zero  
        SET element at position zero AND position zero of dp TO one  
          
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position i MINUS one AND position j of dp MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position i MINUS one AND position j MINUS i of dp PLUS MODULO MOD  
                END IF  
                SET element at position i AND position j of dp TO prefix_sum  
            END FOR  
        END FOR  
          
        RETURN element at position n AND position k of dp  
    END FUNCTION  
END CLASS
