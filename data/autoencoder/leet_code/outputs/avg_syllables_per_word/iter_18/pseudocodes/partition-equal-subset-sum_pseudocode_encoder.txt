CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the summation of all elements in nums  
        
        IF the remainder of total_sum divided by two does not equal zero holds  
            RETURN False  
        END IF  
        
        SET target TO total_sum divided by two disregarding any fractional component  
        SET n TO the count of elements in nums  
        
        SET dp TO a list of boolean values all set to False with length equal to target plus one  
        SET the element at position zero of dp TO True  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num MINUS one plus one  
                SET the element at position i of dp TO the logical OR of the element at position i of dp AND the element at position i minus num of dp  
            END FOR  
        END FOR  
        
        RETURN the element at position target of dp  
    END FUNCTION  
END CLASS
