CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 is less than or equal to ten raised to the power of nine holds  
            APPEND the tuple of digits of power_of_2 sorted into power_of_2_sorted_digits  
            MULTIPLY power_of_2 BY two and ASSIGN to power_of_2  
        END WHILE  
        SET sorted_n_digits TO the tuple of digits of n sorted  
        IF sorted_n_digits is an element of power_of_2_sorted_digits THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
