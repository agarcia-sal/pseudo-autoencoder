CLASS Solution  
    FUNCTION findAnagrams WITH PARAMETERS s AND p  
        SET len_s TO the LENGTH OF s  
        SET len_p TO the LENGTH OF p  
        
        IF the LENGTH OF p GREATER THAN the LENGTH OF s THEN  
            RETURN an EMPTY list  
        END IF  
        
        SET p_count TO the FREQUENCY COUNTER OF all CHARACTERS IN p  
        SET s_count TO the FREQUENCY COUNTER OF all CHARACTERS AT the POSITIONS FROM zero TO the LENGTH OF p MINUS one OF s  
        
        SET anagrams TO an EMPTY list  
        
        IF s_count EQUALS p_count THEN  
            APPEND zero TO anagrams  
        END IF  
        
        FOR index FROM len_p TO len_s MINUS one  
            INCREMENT the COUNT OF the CHARACTER AT position index OF s IN s_count BY one  
            DECREMENT the COUNT OF the CHARACTER AT position index MINUS len_p OF s IN s_count BY one  
            
            IF the COUNT OF the CHARACTER AT position index MINUS len_p OF s IN s_count EQUALS zero THEN  
                REMOVE the CHARACTER AT position index MINUS len_p OF s FROM s_count  
            END IF  
            
            IF s_count EQUALS p_count THEN  
                APPEND index MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
        
        RETURN anagrams  
    END FUNCTION  
END CLASS
