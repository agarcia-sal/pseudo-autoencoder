CLASS Solution  
    FUNCTION lengthLongestPath WITH parameter input_string  
        SET lines TO splitting input_string by the newline character  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO counting the number of tab characters in line  
            SET name TO substring of line starting from position depth plus one up to the end of line  
            
            WHILE the LENGTH OF stack GREATER THAN depth holds  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the LENGTH OF stack GREATER THAN zero holds  
                SET current_length TO the element at the last position of stack PLUS the LENGTH OF name PLUS one for the directory separator  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF name CONTAINS the dot character implying it is a file  
                SET max_length TO the greater value BETWEEN max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
