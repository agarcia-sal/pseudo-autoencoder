CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        SET n TO the quantity resulting from n PLUS twenty four DIVIDED BY twenty five rounded down to the nearest whole number  
        SET memo TO an empty mapping  
        FUNCTION dp(a, b)  
            IF the pair a and b exists as a key in memo  
                RETURN the value in memo associated with the pair a and b  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY the sum of dp of a MINUS four and b PLUS dp of a MINUS three and b MINUS one PLUS dp of a MINUS two and b MINUS two PLUS dp of a MINUS one and b MINUS three  
            SET the value in memo associated with the pair a and b TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp of n and n  
    END FUNCTION  
END CLASS
