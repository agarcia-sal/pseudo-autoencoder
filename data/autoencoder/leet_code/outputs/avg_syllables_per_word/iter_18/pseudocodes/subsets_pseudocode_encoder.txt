CLASS Solution  
    FUNCTION subsets WITH PARAMETER nums  
        IF nums IS empty  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO calling subsets WITH the slice of nums from the second element to the end  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            APPEND a new list beginning with the first element of nums followed by subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
