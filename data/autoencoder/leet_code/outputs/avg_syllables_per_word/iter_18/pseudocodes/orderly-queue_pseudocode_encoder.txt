CLASS Solution  
    FUNCTION orderlyQueue WITH PARAMETERS s AS string AND k AS integer  
        IF k IS GREATER THAN one  
            RETURN the result OF joining TOGETHER the elements OF the sequence obtained BY sorting s IN ascending order  
        ELSE IF k EQUALS one  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR each index FROM one TO one LESS THAN n  
                SET rotated TO the concatenation OF the substring OF s FROM position index TO the END AND the substring OF s FROM zero TO position index  
                IF rotated IS LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
