CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger AND one DIVIDED BY two is less than desiredTotal THEN  
            RETURN false  
        END IF  
  
        IF desiredTotal is less than or equal to zero THEN  
            RETURN true  
        END IF  
  
        SET memo TO empty mapping  
  
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers is a key in memo THEN  
                RETURN the value at key pair of current_total AND used_numbers in memo  
            END IF  
  
            FOR number FROM one TO maxChoosableInteger INCLUSIVE  
                IF the bitwise AND of used_numbers AND two RAISED TO the power of number is equal to zero THEN  
                    IF the sum of current_total AND number is greater than or equal to desiredTotal OR NOT can_win(the sum of current_total AND number, the bitwise OR of used_numbers AND two RAISED TO the power of number) THEN  
                        SET key pair of current_total AND used_numbers in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
  
            SET key pair of current_total AND used_numbers in memo TO false  
            RETURN false  
        END FUNCTION  
  
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
