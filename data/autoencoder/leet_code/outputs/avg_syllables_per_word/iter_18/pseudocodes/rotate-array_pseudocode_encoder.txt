CLASS Solution  
    FUNCTION rotate(nums AS List OF integers, k AS integer) RETURNS none  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start AS integer, end AS integer) RETURNS none  
            WHILE start IS LESS THAN end  
                SET temporary_value TO the element at position start of nums  
                SET the element at position start of nums TO the element at position end of nums  
                SET the element at position end of nums TO temporary_value  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH start EQUAL TO zero AND end EQUAL TO n MINUS one  
        CALL reverse WITH start EQUAL TO zero AND end EQUAL TO k MINUS one  
        CALL reverse WITH start EQUAL TO k AND end EQUAL TO n MINUS one  
    END FUNCTION  
END CLASS
