CLASS TreeNode  
    FUNCTION __init__ WITH val defaulting TO zero left defaulting TO none right defaulting TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values which IS a list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS less than the length of values AND element at position i of values IS NOT none  
            SET node dot left TO a new instance of TreeNode with value element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the length of values AND element at position i of values IS NOT none  
            SET node dot right TO a new instance of TreeNode with value element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUAL TO value of q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with p left AND q left AND the result of calling is_same_tree with p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues WITH parameters self AND root which IS an optional TreeNode  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET queue TO a new double ended queue initialized with root  
        SET largest_values TO an empty list  
        WHILE queue IS NOT empty  
            SET level_size TO the length of queue  
            SET max_value TO negative infinity  
            FOR each number FROM one TO level_size  
                SET node TO the element removed from the front of queue  
                IF max_value IS less than value of node  
                    SET max_value TO value of node  
                END IF  
                IF left child of node EXISTS  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node EXISTS  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
