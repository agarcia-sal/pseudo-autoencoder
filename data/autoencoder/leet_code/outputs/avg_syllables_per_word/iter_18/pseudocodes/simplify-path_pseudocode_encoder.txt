CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO result of splitting path BY forward slash character  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS single period character  
                CONTINUE  
            ELSE IF component EQUALS double period characters  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO forward slash character CONCATENATED WITH the concatenation of elements in stack SEPARATED BY forward slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
