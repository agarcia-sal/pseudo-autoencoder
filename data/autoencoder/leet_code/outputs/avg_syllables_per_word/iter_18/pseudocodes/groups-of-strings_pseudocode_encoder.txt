CLASS UnionFind  
    FUNCTION __init__ WITH parameter size  
        SET parent TO list of integers from zero to size MINUS one  
        SET rank TO list of size elements each EQUAL TO zero  
        SET size TO list of size elements each EQUAL TO one  
    END FUNCTION  
  
    FUNCTION find WITH parameter u  
        IF element at position u of parent NOT EQUALS u THEN  
            SET element at position u of parent TO the result of find of element at position u of parent  
        END IF  
        RETURN element at position u of parent  
    END FUNCTION  
  
    FUNCTION union WITH parameters u and v  
        SET rootU TO the result of find of u  
        SET rootV TO the result of find of v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of rank GREATER THAN element at position rootV of rank THEN  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of size BY element at position rootV of size  
            ELSE IF element at position rootU of rank LESS THAN element at position rootV of rank THEN  
                SET element at position rootU of parent TO rootV  
                INCREMENT element at position rootV of size BY element at position rootU of size  
            ELSE  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of size BY element at position rootV of size  
                INCREMENT element at position rootU of rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION groupStrings WITH parameter words  
        FUNCTION bitmask WITH parameter word  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one MULTIPLIED BY two RAISED TO the power of the difference between the Unicode code point of character and the Unicode code point of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  
  
        SET word_map TO a new mapping that creates an empty list for missing keys  
        SET n TO the count of words  
        SET uf TO a new instance of UnionFind with n  
  
        FOR each index i and word IN enumeration of words  
            SET mask TO the result of bitmask with word  
            APPEND i TO element at key mask in word_map  
        END FOR  
  
        FOR each index i and word IN enumeration of words  
            SET mask TO the result of bitmask with word  
  
            FOR each integer j FROM zero TO twenty five  
                SET new_mask TO mask EXCLUSIVE OR one MULTIPLIED BY two RAISED TO the power of j  
                IF new_mask IS a key in word_map THEN  
                    FOR each element k IN element at key new_mask in word_map  
                        CALL union of uf with arguments i and k  
                    END FOR  
                END IF  
  
                IF mask BITWISE AND one MULTIPLIED BY two RAISED TO the power of j EQUALS zero THEN  
                    SET new_mask TO mask BITWISE OR one MULTIPLIED BY two RAISED TO the power of j  
                    IF new_mask IS a key in word_map THEN  
                        FOR each element k IN element at key new_mask in word_map  
                            CALL union of uf with arguments i and k  
                        END FOR  
                    END IF  
                END IF  
  
                IF mask BITWISE AND one MULTIPLIED BY two RAISED TO the power of j NOT EQUALS zero THEN  
                    FOR each integer k FROM zero TO twenty five  
                        IF mask BITWISE AND one MULTIPLIED BY two RAISED TO the power of k EQUALS zero THEN  
                            SET new_mask TO mask EXCLUSIVE OR one MULTIPLIED BY two RAISED TO the power of j BITWISE OR one MULTIPLIED BY two RAISED TO the power of k  
                            IF new_mask IS a key in word_map THEN  
                                FOR each element l IN element at key new_mask in word_map  
                                    CALL union of uf with arguments i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
  
        SET root_sizes TO a new mapping that creates zero for missing keys  
  
        FOR each integer i FROM zero TO n MINUS one  
            SET root TO the result of find of uf with argument i  
            INCREMENT element at key root in root_sizes BY one  
        END FOR  
  
        SET max_group_size TO the maximum value among the values of root_sizes  
        SET num_groups TO the count of keys in root_sizes  
  
        RETURN a list containing num_groups and max_group_size  
    END FUNCTION  
END CLASS
