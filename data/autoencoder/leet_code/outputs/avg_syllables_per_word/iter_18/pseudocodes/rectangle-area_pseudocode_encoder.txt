CLASS Solution  
    FUNCTION computeArea(ax1 ax1_number ay1 ay1_number ax2 ax2_number ay2 ay2_number bx1 bx1_number by1 by1_number bx2 bx2_number by2 by2_number)  
        SET area1 TO ax2_number MINUS ax1_number MULTIPLIED BY ay2_number MINUS ay1_number  
        SET area2 TO bx2_number MINUS bx1_number MULTIPLIED BY by2_number MINUS by1_number  
        SET overlap_width TO maximum of zero AND minimum of ax2_number AND bx2_number MINUS maximum of ax1_number AND bx1_number  
        SET overlap_height TO maximum of zero AND minimum of ay2_number AND by2_number MINUS maximum of ay1_number AND by1_number  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
