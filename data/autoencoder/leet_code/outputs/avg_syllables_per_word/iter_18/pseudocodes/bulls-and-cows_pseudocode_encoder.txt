CLASS Solution  
    FUNCTION getHint(secret AS string, guess AS string) RETURNS string  
        FROM the collection named defaultdict IMPORT DEFAULTDICTIONARY  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO a new DEFAULTDICTIONARY of integer  
        SET guess_count TO a new DEFAULTDICTIONARY of integer  

        FOR each pair of characters named s and g IN the zipped pairs of secret and guess  
            IF character s EQUALS character g THEN  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key s IN secret_count BY one  
                INCREMENT the value at key g IN guess_count BY one  
            END IF  
        END FOR  

        FOR each digit key IN secret_count  
            INCREMENT cows BY the lesser quantity BETWEEN the value at key digit IN secret_count AND the value at key digit IN guess_count  
        END FOR  

        RETURN the concatenation of the string representation of bulls PLUS the character A PLUS the string representation of cows PLUS the character B  
    END FUNCTION  
END CLASS
