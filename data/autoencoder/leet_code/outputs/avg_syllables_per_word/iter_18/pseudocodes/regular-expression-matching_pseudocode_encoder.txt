CLASS Solution  
    FUNCTION isMatch WITH PARAMETERS s AND p  
        SET dp TO a two dimensional list of Boolean values initialized to False with the number of rows equal to the length of s plus one AND the number of columns equal to the length of p plus one  
        SET the element at position zero zero of dp TO True  
        FOR j FROM one TO the length of p inclusive  
            IF the character at position j minus one of p EQUALS the character asterisk  
                SET the element at position zero j of dp TO the element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s inclusive  
            FOR j FROM one TO the length of p inclusive  
                IF the character at position j minus one of p EQUALS the character at position i minus one of s OR the character at position j minus one of p EQUALS the character period  
                    SET the element at position i j of dp TO the element at position i minus one j minus one of dp  
                ELSE IF the character at position j minus one of p EQUALS the character asterisk  
                    SET the element at position i j of dp TO the element at position i j minus two of dp  
                    IF the character at position j minus two of p EQUALS the character at position i minus one of s OR the character at position j minus two of p EQUALS the character period  
                        SET the element at position i j of dp TO the element at position i j of dp OR the element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the length of s the length of p of dp  
    END FUNCTION  
END CLASS
