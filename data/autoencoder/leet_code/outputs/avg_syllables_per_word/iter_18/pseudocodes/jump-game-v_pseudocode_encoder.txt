CLASS Solution  
    FUNCTION maxJumps(arr AS List of integer, d AS integer) RETURNS integer  
        FUNCTION dp(i AS integer) RETURNS integer  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF the sum of i and x is less than the LENGTH OF arr THEN  
                    IF the element at position the sum of i and x of arr is greater than or equal to the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the maximum of max_jumps AND the sum of one AND the result of dp called with the sum of i and x  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF the difference of i and x is greater than or equal to zero THEN  
                    IF the element at position the difference of i and x of arr is greater than or equal to the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the maximum of max_jumps AND the sum of one AND the result of dp called with the difference of i and x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET results TO empty list  
        FOR i FROM zero TO one less than the LENGTH OF arr  
            APPEND the result of dp called with i TO results  
        END FOR  
        RETURN the maximum element of results  
    END FUNCTION  
END CLASS
