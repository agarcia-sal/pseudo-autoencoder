CLASS Solution  
    FUNCTION partitionLabels WITH PARAMETER s  
        SET last_occurrence TO an empty mapping from character TO position  
        FOR each position and character IN s  
            SET key in last_occurrence mapping TO position for character  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        FOR each position and character IN s  
            SET end TO the greater value BETWEEN end AND the value in last_occurrence mapping for character  
            IF position EQUALS end THEN  
                SET size_of_partition TO end MINUS start PLUS one  
                APPEND size_of_partition TO partition_sizes  
                SET start TO position PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
