CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO the list of one thousand nine hundred ninety nine hundred ninety four hundred one hundred ninety fifty forty ten nine five four one  
        SET syms TO the list of symbol M CM D CD C XC L XL X IX V IV I  
        SET roman_numeral TO the empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero holds  
            FOR each count FROM zero TO the number of times num DIVIDED BY the element at position i of val MINUS one  
                APPEND the element at position i of syms TO roman_numeral  
                DECREMENT num BY the element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
