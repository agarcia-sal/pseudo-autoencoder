CLASS Solution  
    FUNCTION rotatedDigits WITH PARAMETER n  
        SET change_digits TO the set containing character two and character five and character six and character nine  
        SET same_digits TO the set containing character zero and character one and character eight  
        
        FUNCTION is_good_number WITH PARAMETER num  
            SET num_str TO the string representation of num  
            IF for each digit IN num_str the digit IS IN the union of change_digits and same_digits holds  
                IF there EXISTS at least one digit IN num_str the digit IS IN change_digits holds  
                    RETURN true  
                END IF  
                RETURN false  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each i FROM one TO n  
            IF is_good_number WITH PARAMETER i RETURNS true  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
