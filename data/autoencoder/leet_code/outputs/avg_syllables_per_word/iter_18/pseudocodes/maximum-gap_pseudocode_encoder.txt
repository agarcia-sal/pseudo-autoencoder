CLASS Solution  
    FUNCTION maximumGap(nums AS List OF Integer) RETURNS Integer  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN zero  
        END IF  

        FUNCTION countingSort(arr AS List OF Integer, exp AS Integer)  
            SET n TO the LENGTH OF arr  
            INITIALIZE output AS List OF Integer WITH n elements all set TO zero  
            INITIALIZE count AS List OF Integer WITH ten elements all set TO zero  

            FOR index_i FROM zero TO n MINUS one  
                SET index TO the element at position index_i of arr DIVIDED BY exp  
                INCREMENT the element at position index MODULO ten of count BY one  
            END FOR  

            FOR index_i FROM one TO nine  
                INCREMENT the element at position index_i of count BY the element at position index_i MINUS one of count  
            END FOR  

            SET index_i TO n MINUS one  
            WHILE index_i GREATER THAN OR EQUAL TO zero DO  
                SET index TO the element at position index_i of arr DIVIDED BY exp  
                SET position TO the element at position index MODULO ten of count MINUS one  
                SET the element at position position of output TO the element at position index_i of arr  
                DECREMENT the element at position index MODULO ten of count BY one  
                DECREMENT index_i BY one  
            END WHILE  

            FOR index_i FROM zero TO n MINUS one  
                SET the element at position index_i of arr TO the element at position index_i of output  
            END FOR  
        END FUNCTION  

        FUNCTION radixSort(arr AS List OF Integer)  
            SET max1 TO the maximum value in arr  

            SET exp TO one  
            WHILE max1 DIVIDED BY exp GREATER THAN zero DO  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  

        CALL radixSort WITH nums  

        SET max_diff TO zero  
        FOR index_i FROM one TO the LENGTH OF nums MINUS one  
            SET current_diff TO the element at position index_i of nums MINUS the element at position index_i MINUS one of nums  
            IF current_diff GREATER THAN max_diff THEN  
                SET max_diff TO current_diff  
            END IF  
        END FOR  

        RETURN max_diff  
    END FUNCTION  
END CLASS
