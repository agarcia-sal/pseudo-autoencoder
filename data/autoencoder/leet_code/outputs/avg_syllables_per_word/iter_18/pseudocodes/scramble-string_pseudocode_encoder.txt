CLASS Solution  
    FUNCTION isScramble s1 AS a string s2 AS a string RETURNS boolean  
        IF the entirety of s1 EQUALS the entirety of s2 THEN  
            RETURN true  
        END IF  
  
        IF the characters in s1 arranged in ascending order DO NOT EQUAL the characters in s2 arranged in ascending order THEN  
            RETURN false  
        END IF  
  
        SET n TO the length of s1  
  
        FOR i FROM one TO the integer one LESS THAN n  
            IF calling isScramble with the substring of s1 from the beginning UP TO position i AND calling isScramble with the substring of s2 from the beginning UP TO position i BOTH RETURN true THEN  
                RETURN true  
            END IF  
  
            IF calling isScramble with the substring of s1 from the beginning UP TO position i AND calling isScramble with the substring of s2 from the last i characters AND calling isScramble with the substring of s1 from position i TO the end AND calling isScramble with the substring of s2 from the beginning UP TO the last i characters ALL RETURN true THEN  
                RETURN true  
            END IF  
        END FOR  
  
        RETURN false  
    END FUNCTION  
END CLASS
