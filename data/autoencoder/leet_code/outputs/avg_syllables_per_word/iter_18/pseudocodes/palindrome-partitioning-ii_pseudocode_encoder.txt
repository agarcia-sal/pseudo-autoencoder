CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
  
        SET is_palindrome TO a two dimensional collection of size n and n with all FALSE values  
  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i and i of is_palindrome TO TRUE  
        END FOR  
  
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start and end of is_palindrome TO character at position start of s EQUALS character at position end of s  
                ELSE  
                    SET element at position start and end of is_palindrome TO character at position start of s EQUALS character at position end of s AND element at position start PLUS one and end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
  
        SET min_cuts TO a collection of length n with all zero values  
  
        FOR index i FROM zero TO n MINUS one  
            IF element at position zero and i of is_palindrome  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity  
                FOR index j FROM one TO i  
                    IF element at position j and i of is_palindrome AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
  
        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
