CLASS Solution  
    FUNCTION minMutation(start AS string PARAMETER end AS string PARAMETER bank AS list OF strings PARAMETER) RETURNS integer  
        SET s TO the SET OF elements IN bank  
        SET q TO a DEQUE CONTAINING a SINGLE TUPLE OF start AND zero  
        SET mp TO a MAPPING WHERE key A MAPS TO the STRING T C G key T MAPS TO the STRING A C G key C MAPS TO the STRING A T G key G MAPS TO the STRING A T C  
        WHILE the deque q HOLDS ONE OR MORE ELEMENTS  
            SET t AND step TO the FIRST ELEMENT REMOVED FROM q  
            IF the STRING t EQUALS the STRING end  
                RETURN step  
            END IF  
            FOR index i AND character v AT EACH POSITION OF the STRING t  
                FOR each character j IN the STRING MAPPED BY KEY v IN mp  
                    SET next TO the CONCATENATION OF the SUBSTRING OF t FROM start TO position i MINUS one PLUS character j PLUS the SUBSTRING OF t FROM position i PLUS one TO THE END  
                    IF next IS AN ELEMENT OF s  
                        APPEND the TUPLE OF next AND step INCREMENTED BY one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
