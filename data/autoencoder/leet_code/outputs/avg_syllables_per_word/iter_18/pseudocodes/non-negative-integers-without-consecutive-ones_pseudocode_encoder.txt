CLASS Solution  
    FUNCTION findIntegers WITH parameter n  
        SET bin_rep TO the binary representation of n excluding the initial character representing binary notation  
        SET length TO the number of characters in bin_rep  
        SET dp TO a list of zeros with the number of elements EQUAL TO length PLUS one  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        FOR index FROM two TO length inclusive  
            SET the element at position index of dp TO the sum of the element at position index MINUS one of dp AND the element at position index MINUS two of dp  
        END FOR  
        SET result TO zero  
        SET prev_bit TO zero  
        FOR index FROM zero TO length MINUS one inclusive  
            IF the character at position index of bin_rep EQUALS one  
                INCREMENT result BY the element at position length MINUS index MINUS one of dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK out of the loop  
                END IF  
            END IF  
            SET prev_bit TO one IF the character at position index of bin_rep EQUALS one OTHERWISE zero  
        END FOR  
        RETURN the sum of result AND one  
    END FUNCTION  
END CLASS
