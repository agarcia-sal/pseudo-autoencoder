CLASS Solution  
    FUNCTION decodeAtIndex(s AS string_parameter k AS integer_parameter) RETURNS string  
        SET length TO zero  
        FOR each character IN s  
            IF character IS A digit  
                SET length TO length MULTIPLIED BY the numeric value of character  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each character IN s iterated in reverse order  
            SET k TO the remainder when k IS DIVIDED BY length  
            IF k EQUALS zero AND character IS A letter from alphabet  
                RETURN character  
            END IF  
            IF character IS A digit  
                SET length TO length DIVIDED BY the numeric value of character  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
