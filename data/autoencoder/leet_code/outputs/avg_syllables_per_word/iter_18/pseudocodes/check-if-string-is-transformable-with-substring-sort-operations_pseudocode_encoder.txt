CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AS string AND t AS string RETURNS boolean  
        IMPORT auxiliary FUNCTIONALITIES from collections and heapq  
        SET pos TO a mapping from integer to a list of integers WITH default empty list  
        FOR each index FROM zero TO the LENGTH OF s MINUS one  
            SET character_at_index TO character at position index of s  
            SET digit_value TO the integer form of character_at_index  
            APPLY heap push operation TO insert index INTO the list at key digit_value in pos  
        END FOR  
        FOR each index FROM zero TO the LENGTH OF t MINUS one  
            SET character_at_index TO character at position index of t  
            SET digit_value TO the integer form of character_at_index  
            IF the list at key digit_value in pos IS empty THEN  
                RETURN False  
            END IF  
            FOR each smaller_digit FROM zero TO digit_value MINUS one  
                IF the list at key smaller_digit in pos IS NOT empty AND the element at position zero of the list at smaller_digit in pos IS LESS THAN the element at position zero of the list at digit_value in pos THEN  
                    RETURN False  
                END IF  
            END FOR  
            APPLY heap pop operation TO remove the smallest element from the list at key digit_value in pos  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
