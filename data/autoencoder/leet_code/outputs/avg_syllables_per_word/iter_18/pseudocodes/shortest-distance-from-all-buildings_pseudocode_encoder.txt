CLASS Solution  
    FUNCTION shortestDistance(grid)  
        IF grid EQUALS nothing OR element at position zero of grid EQUALS nothing THEN  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET total_distance TO a two dimensional list of zeros with m rows and n columns  
        SET reach_count TO a two dimensional list of zeros with m rows and n columns  
        SET building_count TO zero  
        
        SET directions TO a list containing the four pairs one and zero MINUS one and zero zero and one zero and MINUS one  
        
        FUNCTION bfs(start_x, start_y)  
            SET queue TO a double ended queue initialized with the pair start_x and start_y  
            SET visited TO a two dimensional list of Boolean FALSE with m rows and n columns  
            SET distance TO zero  
            SET reachable TO zero  
            
            WHILE queue holds elements  
                SET size TO the number of elements contained in queue  
                INCREMENT distance BY one  
                FOR count FROM one TO size  
                    REMOVE and ASSIGN x, y TO the first element from queue  
                    FOR each dx and dy IN directions  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx and ny of visited EQUALS FALSE AND element at position nx and ny of grid EQUALS zero THEN  
                            SET element at position nx and ny of visited TO TRUE  
                            APPEND the pair nx and ny TO queue  
                            INCREMENT element at position nx and ny of total_distance BY distance  
                            INCREMENT element at position nx and ny of reach_count BY one  
                            INCREMENT reachable BY one  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
            RETURN reachable  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i and j of grid EQUALS one THEN  
                    INCREMENT building_count BY one  
                    IF the result of bfs called with i and j IS LESS THAN building_count MINUS one THEN  
                        RETURN negative one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET min_distance TO a value representing positive infinity  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i and j of grid EQUALS zero AND element at position i and j of reach_count EQUALS building_count THEN  
                    SET min_distance TO the smallest of min_distance AND element at position i and j of total_distance  
                END IF  
            END FOR  
        END FOR  
        
        IF min_distance IS NOT EQUAL TO positive infinity THEN  
            RETURN min_distance  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
