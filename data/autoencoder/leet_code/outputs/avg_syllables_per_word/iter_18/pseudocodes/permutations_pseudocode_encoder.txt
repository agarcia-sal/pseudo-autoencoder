CLASS Solution  
    FUNCTION permute WITH nums AS list of integers  
        FUNCTION backtrack WITH path AS list AND remaining AS list  
            IF remaining IS empty  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO the concatenation OF path PLUS element at position i OF remaining  
                SET new_remaining TO the concatenation OF elements FROM zero TO i MINUS one OF remaining PLUS elements FROM i PLUS one TO the end OF remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty list  
        CALL backtrack WITH empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
