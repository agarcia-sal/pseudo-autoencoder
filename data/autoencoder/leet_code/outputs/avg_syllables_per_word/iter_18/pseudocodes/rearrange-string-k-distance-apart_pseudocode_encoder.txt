CLASS Solution
    FUNCTION rearrangeString(s AS string parameter, k AS integer parameter) RETURNS string
        IF k LESS THAN OR EQUAL TO one THEN
            RETURN s
        END IF

        SET freq TO the frequency count mapping of characters in s

        SET max_heap TO a list consisting of pairs of negative frequency count and corresponding character for each character and count IN freq

        TRANSFORM max_heap INTO a heap structure according to maximum priority by negative count

        SET wait_queue TO an empty list

        SET result TO an empty list

        WHILE max_heap is not empty DO
            EXTRACT the pair of count and character of maximum frequency FROM max_heap
            
            APPEND character TO result
            
            APPEND a pair of count incremented by one and character TO wait_queue
            
            IF the size of wait_queue EQUALS k THEN
                REMOVE the first pair of count and character FROM wait_queue
                
                IF count LESS THAN zero THEN
                    INSERT the pair of count and character INTO max_heap maintaining heap property
                END IF
            END IF
        END WHILE

        IF the length of result NOT EQUAL TO the length of s THEN
            RETURN an empty string
        END IF

        RETURN the concatenation of all characters in result
    END FUNCTION
END CLASS
