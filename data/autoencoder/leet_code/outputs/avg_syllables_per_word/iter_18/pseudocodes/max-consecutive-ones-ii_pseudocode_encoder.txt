CLASS Solution  
    FUNCTION findMaxConsecutiveOnes WITH PARAMETER nums  
        SET max_count TO zero  
        SET current_count TO zero  
        SET zero_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position right of nums EQUALS one  
                INCREMENT current_count BY one  
            ELSE IF zero_count LESS THAN one  
                INCREMENT zero_count BY one  
                INCREMENT current_count BY one  
            ELSE  
                SET max_count TO the GREATER OF max_count AND current_count  
                WHILE zero_count EQUALS one  
                    IF the element at position left of nums EQUALS zero  
                        DECREMENT zero_count BY one  
                    END IF  
                    DECREMENT current_count BY one  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT current_count BY one  
            END IF  
        END FOR  
        
        SET max_count TO the GREATER OF max_count AND current_count  
        RETURN max_count  
    END FUNCTION  
END CLASS
