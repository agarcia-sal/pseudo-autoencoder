CLASS Solution  
    FUNCTION closestToTarget WITH parameters arr AS list of integer AND target AS integer  
        SET seen TO empty set  
        SET min_diff TO infinitely large number  
        FOR each num IN arr  
            SET temporary set TO empty set  
            FOR each element IN seen  
                APPEND bitwise AND of num AND element TO temporary set  
            END FOR  
            SET seen TO union of temporary set AND set containing only num  
            FOR each val IN seen  
                IF absolute value of val MINUS target IS LESS THAN min_diff THEN  
                    SET min_diff TO absolute value of val MINUS target  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
