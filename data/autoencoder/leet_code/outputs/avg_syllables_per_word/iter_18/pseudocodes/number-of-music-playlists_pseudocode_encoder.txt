CLASS Solution  
    FUNCTION numMusicPlaylists(n AS integer goal AS integer k AS integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        CALL auxiliary_initialize_dp WITH n goal RETURNING dp  
        SET element at position zero of list at position zero of dp TO one  
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position j minus one of list at position i minus one of dp MULTIPLIED BY n MINUS j MINUS one PLUS one  
                SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO MOD  
                IF j GREATER THAN k THEN  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position j of list at position i minus one of dp MULTIPLIED BY j MINUS k  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of list at position goal of dp  
    END FUNCTION  

FUNCTION auxiliary_initialize_dp(n AS integer goal AS integer) RETURNS two dimensional list  
    DESCRIBE initialization of two dimensional list dp with dimensions goal plus one by n plus one filled with zero  
    RETURN dp  
END FUNCTION
