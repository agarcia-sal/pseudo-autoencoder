CLASS Solution  
    FUNCTION longestPalindrome(word1 AS string, word2 AS string) RETURNS integer  
        SET s TO concatenation of word1 and word2  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeros with n rows and n columns  
        
        FOR index i FROM zero TO n MINUS one  
            SET element at row i and column i OF dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n INCLUSIVE  
            FOR index i FROM zero TO the LENGTH OF s MINUS length  
                SET j TO i PLUS length MINUS one  
                IF character at position i of s EQUALS character at position j of s THEN  
                    SET element at row i and column j of dp TO element at row i PLUS one and column j MINUS one of dp PLUS two  
                    IF i LESS THAN the LENGTH OF word1 AND j GREATER THAN OR EQUAL TO the LENGTH OF word1 THEN  
                        SET max_len TO the GREATER OF max_len AND element at row i and column j of dp  
                    END IF  
                ELSE  
                    SET element at row i and column j of dp TO the GREATER OF element at row i PLUS one and column j of dp AND element at row i and column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
