CLASS Solution  
    FUNCTION combinationSum4(nums AS List of integers, target AS integer) RETURNS integer  
        SET dp TO a list of zeroes with the LENGTH EQUAL TO target PLUS one  
        SET element at position zero of dp TO one  
        FOR each amount FROM one TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    SET element at position amount of dp TO element at position amount of dp PLUS element at position amount MINUS num of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
