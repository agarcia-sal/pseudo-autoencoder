CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the last element  
        SET node TO new ListNode WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN p1 dot val EQUALS p2 dot val AND is_same_list WITH PARAMETERS p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETER head  
        IF head IS None OR head dot next IS None OR head dot next dot next IS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head dot next  
        SET even_head TO even  
        WHILE even IS NOT None AND even dot next IS NOT None  
            SET odd dot next TO even dot next  
            SET odd TO odd dot next  
            SET even dot next TO odd dot next  
            SET even TO even dot next  
        END WHILE  
        SET odd dot next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
