CLASS Solution
    FUNCTION maximumScore(scores AS List OF integer PARAMETERS, edges AS List OF List OF integer PARAMETERS) RETURNS integer
        SET graph TO a new mapping with default value as empty list
        
        FOR each pair consisting of element u AND element v IN edges
            APPEND a pair consisting of element at position v of scores AND element v TO the list at key u in graph
            APPEND a pair consisting of element at position u of scores AND element u TO the list at key v in graph
        END FOR
        
        FOR each node IN graph
            SET the list at key node in graph TO the three elements with the greatest first parts from the list at key node in graph
        END FOR
        
        SET max_score TO negative one
        
        FOR each pair consisting of element u AND element v IN edges
            FOR each pair consisting of element score1 AND element x IN the list at key u in graph
                FOR each pair consisting of element score2 AND element y IN the list at key v in graph
                    IF the total number of distinct elements among x AND y AND u AND v EQUALS four
                        IF max_score GREATER THAN score1 PLUS score2 PLUS element at position u of scores PLUS element at position v of scores
                            CONTINUE
                        ELSE
                            SET max_score TO score1 PLUS score2 PLUS element at position u of scores PLUS element at position v of scores
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR
        
        RETURN max_score
    END FUNCTION
END CLASS
