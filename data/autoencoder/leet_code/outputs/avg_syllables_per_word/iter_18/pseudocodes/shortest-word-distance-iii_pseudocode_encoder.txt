CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict AS List OF String word1 AS String word2 AS String) RETURNS Integer  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        
        FOR each index_and_word IN enumerated wordsDict  
            SET i TO the index_part of index_and_word  
            SET word TO the word_part of index_and_word  
            
            IF word EQUALS word1 THEN  
                IF same_word THEN  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2 THEN  
                IF NOT same_word THEN  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one THEN  
                SET min_distance TO the smaller OF min_distance AND the absolute value OF prev_word1 MINUS prev_word2  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
