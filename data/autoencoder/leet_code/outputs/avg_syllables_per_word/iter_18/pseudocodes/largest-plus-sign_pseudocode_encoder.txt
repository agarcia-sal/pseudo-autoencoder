CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n AS integer PARAMETER, mines AS list of list of integer PARAMETER) RETURNS integer  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            SET mine_tuple TO a tuple formed by the elements in mine  
            ADD mine_tuple TO mines_set  
        END FOR  
  
        SET dp TO a two dimensional list of size n by n where each element is a list of four zeros representing directions right down left up  
  
        SET directions TO a list containing the pairs representing right zero and one then down one and zero then left zero and negative one then up negative one and zero  
  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair consisting of i and j NOT IN mines_set  
                    IF j GREATER THAN zero  
                        SET the element at position zero in the list at position j of the list at position i of dp TO the element at position zero in the list at position j MINUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero  
                        SET the element at position one in the list at position j of the list at position i of dp TO the element at position one in the list at position j of the list at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
  
        SET result TO zero  
  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair consisting of i and j NOT IN mines_set  
                    IF j LESS THAN n MINUS one  
                        SET the element at position two in the list at position j of the list at position i of dp TO the element at position two in the list at position j PLUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one  
                        SET the element at position three in the list at position j of the list at position i of dp TO the element at position three in the list at position j of the list at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the four elements in the list at position j of the list at position i of dp  
                    IF plus_sign_order GREATER THAN result  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
