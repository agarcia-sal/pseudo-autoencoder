CLASS Solution  
    FUNCTION numDecodings WITH PARAMETER s AS string  
        IF s IS empty OR character at position zero of s EQUALS zero character  
            RETURN zero  
        END IF  
        SET dp TO list OF zero REPEATED the LENGTH OF s PLUS one times  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        FOR index FROM two TO the LENGTH OF s PLUS one  
            IF character at position index MINUS one of s NOT EQUALS zero character  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of dp  
            END IF  
            SET two_digit_number TO the integer value of substring starting at position index MINUS two UNTIL position index of s  
            IF two_digit_number GREATER THAN OR EQUAL TO ten AND two_digit_number LESS THAN OR EQUAL TO twenty six  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS two of dp  
            END IF  
        END FOR  
        RETURN element at last position of dp  
    END FUNCTION  
END CLASS
