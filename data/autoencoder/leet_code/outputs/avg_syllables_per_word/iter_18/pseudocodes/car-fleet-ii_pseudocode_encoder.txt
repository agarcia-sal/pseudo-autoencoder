CLASS Solution
    FUNCTION getCollisionTimes(cars AS list of list of integer) RETURNS list of float
        SET n TO the LENGTH OF cars
        SET result TO a list containing n elements each SET TO negative one point zero
        SET stack TO an empty list
        
        FOR i FROM n MINUS one DOWN TO zero
            SET pos TO the element at position zero of the element at position i of cars
            SET speed TO the element at position one of the element at position i of cars
            
            WHILE stack is not empty
                SET next_index TO the element at the last position of stack
                SET next_pos TO the element at position zero of the element at position next_index of cars
                SET next_speed TO the element at position one of the element at position next_index of cars
                
                IF speed LESS THAN OR EQUAL TO next_speed THEN
                    REMOVE the last element FROM stack
                    CONTINUE
                END IF
                
                SET collision_time TO the difference of next_pos AND pos DIVIDED BY the difference of speed AND next_speed
                
                IF the element at position next_index of result IS GREATER THAN zero AND collision_time IS GREATER THAN OR EQUAL TO the element at position next_index of result THEN
                    REMOVE the last element FROM stack
                    CONTINUE
                END IF
                
                SET the element at position i of result TO collision_time
                BREAK
            END WHILE
            
            APPEND i TO stack
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
