CLASS Solution  
    FUNCTION getSkyline WITH PARAMETER buildings as List of List of Integer  
        SET events TO empty list  
        FOR each building IN buildings  
            DESTRUCTURE building INTO left AND right AND height  
            APPEND to events an element composed of left and NEGATIVE of height and right  
            APPEND to events an element composed of right and zero and zero  
        END FOR  
        CALL sort ON events BY natural ascending ordering  
        SET result TO a list containing one element which is a list of zero and zero  
        SET max_heap TO a list containing one tuple of zero and infinity  
        FOR each event IN events  
            DESTRUCTURE event INTO x AND negH AND r  
            WHILE the second element at position zero of max_heap is LESS THAN OR EQUAL TO x holds true  
                REMOVE the top element FROM max_heap using heap pop operation  
            END WHILE  
            IF negH is NOT EQUAL TO zero THEN  
                ADD to max_heap the tuple composed of negH and r using heap push operation  
            END IF  
            IF the second element of the last element in result is NOT EQUAL TO the NEGATIVE of the first element at position zero in max_heap THEN  
                APPEND to result a list composed of x and NEGATIVE of the first element at position zero in max_heap  
            END IF  
        END FOR  
        IF the first element of result EQUALS a list composed of zero and zero THEN  
            REMOVE the first element from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
