CLASS Solution  
    FUNCTION maxChunksToSorted(arr AS List of integer) RETURNS integer  
        SET max_so_far TO empty list  
        SET current_max TO element at position one of arr  
        FOR each num IN arr  
            SET current_max TO the greater value BETWEEN current_max AND num  
            APPEND current_max TO max_so_far  
        END FOR  
        SET sorted_arr TO the list arr arranged IN ascending order  
        SET chunks TO zero  
        SET max_chunk TO zero  
        FOR index FROM one TO the LENGTH OF arr  
            SET max_chunk TO the greater value BETWEEN max_chunk AND element at position index of arr  
            IF max_chunk EQUALS element at position index of sorted_arr THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
