CLASS Solution  
    FUNCTION smallestGoodBase WITH PARAMETER n AS string  
        SET n TO the integer converted from n  
        SET max_length TO the integer obtained by the base two logarithm of n INCREMENTED BY one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer obtained by the n raised to the power of one DIVIDED BY the value obtained by m MINUS one  
            IF k IS LESS THAN two  
                CONTINUE  
            END IF  
            SET num TO the value obtained by k raised to the power of m MINUS one DIVIDED BY k MINUS one  
            IF num EQUALS n  
                RETURN the string converted from k  
            END IF  
        END FOR  
        RETURN the string converted from n MINUS one  
    END FUNCTION  
END CLASS
