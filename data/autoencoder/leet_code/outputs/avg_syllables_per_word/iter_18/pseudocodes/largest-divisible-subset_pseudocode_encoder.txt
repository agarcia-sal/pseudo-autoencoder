CLASS Solution  
    FUNCTION largestDivisibleSubset(nums AS List OF Integer) RETURNS List OF Integer  
        CALL nums SORT IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET f TO a LIST OF n ELEMENTS EACH SET TO one  
        SET k TO zero  
        FOR each index i FROM zero TO n MINUS one INCLUSIVE  
            FOR each index j FROM zero TO i MINUS one INCLUSIVE  
                IF the ELEMENT AT position i OF nums MODULO the ELEMENT AT position j OF nums EQUALS zero THEN  
                    SET f AT position i TO the GREATER VALUE BETWEEN f AT position i AND f AT position j PLUS one  
                END IF  
            END FOR  
            IF f AT position k LESS THAN f AT position i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f AT position k  
        SET i TO k  
        SET ans TO an EMPTY LIST  
        WHILE m GREATER THAN zero DO  
            IF the ELEMENT AT position k OF nums MODULO the ELEMENT AT position i OF nums EQUALS zero AND f AT position i EQUALS m THEN  
                APPEND the ELEMENT AT position i OF nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
