CLASS Solution  
    FUNCTION minimumOneBitOperations WITH PARAMETER n  
        IF n EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE two RAISED TO the power of k PLUS one IS LESS THAN OR EQUAL TO n HOLD  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO two RAISED TO the power of k  
        RETURN two RAISED TO the power of k PLUS one MINUS one MINUS CALL minimumOneBitOperations WITH ARGUMENT n XOR most_significant_bit  
    END FUNCTION  
END CLASS
