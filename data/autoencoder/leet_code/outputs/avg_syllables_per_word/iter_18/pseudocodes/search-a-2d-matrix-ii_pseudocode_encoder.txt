CLASS Solution  
    FUNCTION searchMatrix matrix AND target  
        IF matrix IS empty OR element at position zero of matrix IS empty  
            RETURN False  
        END IF  
        SET row TO zero  
        SET col TO the LENGTH OF element at position zero of matrix MINUS one  
        WHILE row IS LESS THAN the LENGTH OF matrix AND col IS GREATER THAN OR EQUAL TO zero  
            IF element at position row and col of matrix EQUALS target  
                RETURN True  
            ELSE IF element at position row and col of matrix IS GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
