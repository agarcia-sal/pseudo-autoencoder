CLASS Solution  
    FUNCTION isValidPalindrome WITH PARAMETERS s AS string AND k AS integer RETURNS boolean  
        FUNCTION longest_palindromic_subseq WITH PARAMETER s AS string RETURNS integer  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional list of zeros with dimensions n BY n  
            
            FOR index i FROM zero TO n MINUS one  
                SET element at position i and position i of dp TO one  
            END FOR  
            
            FOR length FROM two TO n  
                FOR index i FROM zero TO n MINUS length  
                    SET index j TO i PLUS length MINUS one  
                    IF character at position i of s EQUALS character at position j of s THEN  
                        SET element at position i and position j of dp TO element at position i PLUS one and position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i and position j of dp TO the GREATER OF element at position i PLUS one and position j of dp AND element at position i and position j MINUS one of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero and position n MINUS one of dp  
        END FUNCTION  
        
        SET longest_palindrome_length TO the RESULT OF longest_palindromic_subseq WITH PARAMETER s  
        RETURN the LENGTH OF s MINUS longest_palindrome_length IS LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
