CLASS Solution  
    FUNCTION getMoneyAmount WITH PARAMETER n  
        FUNCTION dp WITH PARAMETERS left AND right  
            IF left GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET minimum_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET cost_to_consider TO pivot PLUS the MAXIMUM VALUE BETWEEN dp WITH left AND pivot MINUS one AND dp WITH pivot PLUS one AND right  
                SET minimum_cost TO the SMALLER VALUE BETWEEN minimum_cost AND cost_to_consider  
            END FOR  
            RETURN minimum_cost  
        END FUNCTION  
        RETURN dp WITH one AND n  
    END FUNCTION  
END CLASS
