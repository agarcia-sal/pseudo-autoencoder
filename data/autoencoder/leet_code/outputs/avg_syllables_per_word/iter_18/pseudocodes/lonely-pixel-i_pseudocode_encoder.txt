CLASS Solution  
    FUNCTION findLonelyPixel WITH parameter picture AS list of list of string  
        IF picture IS empty OR element at position zero of picture IS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list OF m elements each EQUAL TO zero  
        SET col_count TO a list OF n elements each EQUAL TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF character at position j of element at position i of picture EQUALS the character B  
                    INCREMENT element at position i OF row_count BY one  
                    INCREMENT element at position j OF col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF character at position j of element at position i of picture EQUALS the character B AND element at position i OF row_count EQUALS one AND element at position j OF col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
