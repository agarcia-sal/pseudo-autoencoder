CLASS Solution  
    FUNCTION minmaxGasDist(stations AS List of integers k AS integer) AS floating point  
        FUNCTION possible(max_dist AS floating point) AS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index of stations MINUS element at position index MINUS one of stations  
                SET stations_needed TO the INTEGER PART OF distance_difference DIVIDED BY max_dist  
                INCREMENT additional_stations BY stations_needed  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO ten raised TO the power of eight  
        WHILE the difference between right and left IS GREATER THAN one MULTIPLIED BY ten TO THE POWER OF MINUS six  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) IS true THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
