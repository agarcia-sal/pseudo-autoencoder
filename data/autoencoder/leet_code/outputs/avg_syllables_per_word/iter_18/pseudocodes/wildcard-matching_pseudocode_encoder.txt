CLASS Solution  
    FUNCTION isMatch WITH PARAMETERS s AND p  
        SET dp TO two dimensional list of boolean False values with the number of rows equal to the length of s plus one and the number of columns equal to the length of p plus one  
        SET element at position zero zero of dp TO True  
        FOR each index j FROM one TO the length of p  
            IF character at position j minus one of p EQUALS the character STAR  
                SET element at position zero j of dp TO element at position zero j minus one of dp  
            END IF  
        END FOR  
        FOR each index i FROM one TO the length of s  
            FOR each index j FROM one TO the length of p  
                IF character at position j minus one of p EQUALS the character STAR  
                    SET element at position i j of dp TO element at position i minus one j of dp OR element at position i j minus one of dp  
                ELSE IF character at position j minus one of p EQUALS the character QUESTION MARK OR character at position i minus one of s EQUALS character at position j minus one of p  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position length of s length of p of dp  
    END FUNCTION  
END CLASS
