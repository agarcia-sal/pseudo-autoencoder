CLASS Solution
    FUNCTION triangleNumber(nums AS List of Integer) RETURNS Integer
        INVOKE sort METHOD OF nums
        SET count TO zero
        SET n TO the LENGTH OF nums
        
        FOR k FROM n MINUS one TO two INCLUSIVE STEP minus one
            SET i TO zero
            SET j TO k MINUS one
            WHILE i LESS THAN j
                IF the ELEMENT at POSITION i of nums PLUS the ELEMENT at POSITION j of nums GREATER THAN the ELEMENT at POSITION k of nums
                    INCREMENT count BY j MINUS i
                    DECREMENT j BY one
                ELSE
                    INCREMENT i BY one
                END IF
            END WHILE
        END FOR
        
        RETURN count
    END FUNCTION
END CLASS
