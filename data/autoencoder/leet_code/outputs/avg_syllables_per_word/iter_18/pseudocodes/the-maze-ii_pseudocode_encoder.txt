CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze AS list of lists of integers start AS list of integers destination AS list of integers  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero in maze  
        SET directions TO the list consisting of the pairs one and zero MINUS one and zero zero and one zero and MINUS one  
        SET start TO a tuple constructed from start  
        SET destination TO a tuple constructed from destination  
        SET visited TO an empty set  
        SET pq TO a list consisting of the pair zero and start  
        
        WHILE pq IS NOT empty  
            EXTRACT the element with the smallest first component FROM pq AND ASSIGN the first component TO dist AND the second component TO a pair consisting of x AND y  
            
            IF the pair x AND y EQUALS destination  
                RETURN dist  
            END IF  
            
            IF the pair x AND y IS IN visited  
                CONTINUE  
            END IF  
            
            ADD the pair x AND y TO visited  
            
            FOR each pair dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE the value of new_x plus dx IS GREATER THAN OR EQUAL TO zero AND LESS THAN m AND the value of new_y plus dy IS GREATER THAN OR EQUAL TO zero AND LESS THAN n AND element at position new_x plus dx of maze at position new_y plus dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the pair new_x AND new_y IS NOT IN visited  
                    INSERT into pq the pair of dist plus count AND the pair new_x AND new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
