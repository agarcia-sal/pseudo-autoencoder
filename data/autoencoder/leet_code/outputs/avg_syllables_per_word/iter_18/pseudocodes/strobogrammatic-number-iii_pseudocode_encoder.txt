CLASS Solution  
    FUNCTION strobogrammaticInRange(low AS string_parameter, high AS string_parameter) RETURNS integer  
        FUNCTION generate_strobogrammatic(n AS integer_parameter, final_length AS integer_parameter) RETURNS list_of_strings  
            IF n EQUALS zero THEN  
                RETURN list_containing_empty_string  
            END IF  
            IF n EQUALS one THEN  
                RETURN list_containing string zero string one string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two, final_length)  
            SET result TO empty list  
            FOR each middle_element IN middles  
                IF n NOT EQUAL TO final_length THEN  
                    APPEND concatenation of string zero with middle_element with string zero TO result  
                END IF  
                APPEND concatenation of string one with middle_element with string one TO result  
                APPEND concatenation of string six with middle_element with string nine TO result  
                APPEND concatenation of string eight with middle_element with string eight TO result  
                APPEND concatenation of string nine with middle_element with string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  

        FUNCTION count_strobogrammatic(low AS string_parameter, high AS string_parameter) RETURNS integer  
            SET count TO zero  
            FOR length_value FROM the LENGTH OF low TO the LENGTH OF high INCLUSIVE  
                FOR each num_element IN generate_strobogrammatic(length_value, length_value)  
                    IF (length_value EQUALS the LENGTH OF low AND num_element LESS THAN low) OR (length_value EQUALS the LENGTH OF high AND num_element GREATER THAN high) THEN  
                        CONTINUE  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  

        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
