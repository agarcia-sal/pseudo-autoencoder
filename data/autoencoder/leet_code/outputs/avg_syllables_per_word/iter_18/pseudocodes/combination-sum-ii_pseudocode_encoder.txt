CLASS Solution  
    FUNCTION combinationSum2(candidates AS List OF integers, target AS integer) AS List OF List OF integers  
        FUNCTION backtrack(start AS integer, target AS integer, path AS List OF integers)  
            IF target EQUALS zero THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero THEN  
                RETURN  
            END IF  
            FOR each i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates THEN  
                    CONTINUE  
                END IF  
                CALL backtrack WITH i PLUS one AS start, target MINUS element at position i of candidates AS target, path APPENDED WITH element at position i of candidates AS path  
            END FOR  
        END FUNCTION  
        CALL the SORT METHOD ON candidates  
        SET result TO an empty list  
        CALL backtrack WITH zero AS start, target AS target, empty list AS path  
        RETURN result  
    END FUNCTION  
END CLASS
