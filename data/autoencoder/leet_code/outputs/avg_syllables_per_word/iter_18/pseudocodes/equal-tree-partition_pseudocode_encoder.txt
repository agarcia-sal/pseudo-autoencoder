CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS none right EQUALS none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH parameter element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element at front of queue REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node left TO A NEW TreeNode WITH parameter element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node right TO A NEW TreeNode WITH parameter element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p val IS NOT EQUALS value of q val  
        RETURN False  
    ELSE  
        RETURN Boolean AND of is_same_tree WITH PARAMETERS p left q left AND is_same_tree WITH PARAMETERS p right q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree WITH PARAMETERS root AS Optional OF TreeNode  
        FUNCTION subtree_sum WITH PARAMETER node  
            IF node IS none  
                RETURN zero  
            END IF  
            RETURN value of node val PLUS subtree_sum WITH PARAMETER node left PLUS subtree_sum WITH PARAMETER node right  
        END FUNCTION  

        SET total_sum TO subtree_sum WITH PARAMETER root  

        IF total_sum MODULO two IS NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO list CONTAINING Boolean False  

        FUNCTION can_split WITH PARAMETER node  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO value of node val PLUS can_split WITH PARAMETER node left PLUS can_split WITH PARAMETER node right  
            IF current_sum EQUALS target AND node IS NOT same reference AS root  
                SET element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split WITH PARAMETER root  
        RETURN element at position zero of found  
    END FUNCTION  
END CLASS
