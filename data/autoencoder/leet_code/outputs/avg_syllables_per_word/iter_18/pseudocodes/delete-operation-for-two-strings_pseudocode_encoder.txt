CLASS Solution  
    FUNCTION minDistance WITH PARAMETERS word1 word2  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        CALL auxiliary FUNCTION initialize_dp_array WITH PARAMETERS m n AND ASSIGN RESULT TO dp  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the character at position i MINUS one of word1 EQUALS the character at position j MINUS one of word2  
                    SET element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i and position j of dp TO the MAXIMUM OF the element at position i MINUS one and position j of dp AND the element at position i and position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at position m and position n of dp  
        SET minimum_deletions TO m MINUS lcs_length PLUS n MINUS lcs_length  
        RETURN minimum_deletions  
    END FUNCTION  

    FUNCTION initialize_dp_array WITH PARAMETERS m n  
        DESCRIBE auxiliary routine TO CREATE A TWO DIMENSIONAL ARRAY OF ZEROES WITH DIMENSIONS m PLUS one BY n PLUS one  
    END FUNCTION  
END CLASS
