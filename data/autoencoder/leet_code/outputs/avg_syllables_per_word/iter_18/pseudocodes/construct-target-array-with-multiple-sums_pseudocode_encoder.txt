CLASS Solution  
    FUNCTION isPossible(target AS list of integers) RETURNS boolean  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND the negative of num TO max_heap  
        END FOR  
        INVOKE auxiliary FUNCTION heapify ON max_heap  
        SET total_sum TO the summation of all elements in target  
        WHILE the element at position zero of max_heap is less than the negative of one  
            SET largest TO the negative of the result of invoking auxiliary FUNCTION heappop ON max_heap  
            SET rest_sum TO total_sum MINUS largest  
            IF largest is less than or equal to rest_sum OR rest_sum equals zero  
                RETURN false  
            END IF  
            SET new_value TO the remainder of largest divided by rest_sum  
            IF new_value equals zero AND rest_sum is not equal to one  
                RETURN false  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            INVOKE auxiliary FUNCTION heappush ON max_heap WITH the negative of new_value  
        END WHILE  
        RETURN true  
    END FUNCTION  
END CLASS
