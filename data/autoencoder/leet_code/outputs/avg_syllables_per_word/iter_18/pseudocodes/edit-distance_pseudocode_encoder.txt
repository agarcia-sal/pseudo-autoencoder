CLASS Solution  
    FUNCTION minDistance(word1 AS string parameter, word2 AS string parameter) RETURNS integer  
        SET length_of_word1 TO the LENGTH OF word1  
        SET length_of_word2 TO the LENGTH OF word2  
        
        SET dp_table TO a two dimensional collection indexed by number from zero TO length_of_word1 for rows and number from zero TO length_of_word2 for columns each initialized TO zero  
        
        FOR index_i FROM zero TO length_of_word1  
            SET element at position index_i and position zero IN dp_table TO index_i  
        END FOR  
        
        FOR index_j FROM zero TO length_of_word2  
            SET element at position zero and position index_j IN dp_table TO index_j  
        END FOR  
        
        FOR index_i FROM one TO length_of_word1  
            FOR index_j FROM one TO length_of_word2  
                IF character at position index_i MINUS one OF word1 EQUALS character at position index_j MINUS one OF word2 THEN  
                    SET element at position index_i and position index_j IN dp_table TO element at position index_i MINUS one and position index_j MINUS one IN dp_table  
                ELSE  
                    SET minimum_value TO the MINIMUM OF element at position index_i MINUS one and position index_j IN dp_table  
                    SET minimum_value TO the MINIMUM OF minimum_value AND element at position index_i and position index_j MINUS one IN dp_table  
                    SET minimum_value TO the MINIMUM OF minimum_value AND element at position index_i MINUS one and position index_j MINUS one IN dp_table  
                    SET element at position index_i and position index_j IN dp_table TO one PLUS minimum_value  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position length_of_word1 and position length_of_word2 IN dp_table  
    END FUNCTION  
END CLASS
