CLASS Solution  
    FUNCTION numWays(words AS List of String, target AS String) RETURNS Integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF the element at position zero of words  
        SET m TO the LENGTH OF target  
        
        SET freq TO a list OF mappings FROM character TO integer WITH a mapping for each position from zero TO n MINUS one  
        FOR each word IN words  
            FOR each character WITH its position j IN word  
                INCREMENT the value ASSOCIATED WITH character at key character in the mapping at position j of freq BY one  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list OF integers WHERE there are m PLUS one lists each containing n PLUS one zeros  
        
        FOR each index j FROM zero TO n  
            SET the element at row zero column j of dp TO one  
        END FOR  
        
        FOR each index i FROM one TO m  
            FOR each index j FROM one TO n  
                SET the element at row i column j of dp TO the element at row i column j MINUS one of dp  
                IF the value ASSOCIATED WITH the character at position i MINUS one of target IN the mapping at position j MINUS one of freq IS GREATER THAN zero  
                    INCREMENT the element at row i column j of dp BY the element at row i MINUS one column j MINUS one of dp MULTIPLIED BY the value ASSOCIATED WITH the character at position i MINUS one of target IN the mapping at position j MINUS one of freq  
                    SET the element at row i column j of dp TO the remainder of the element at row i column j of dp DIVIDED BY MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at row m column n of dp  
    END FUNCTION  
END CLASS
