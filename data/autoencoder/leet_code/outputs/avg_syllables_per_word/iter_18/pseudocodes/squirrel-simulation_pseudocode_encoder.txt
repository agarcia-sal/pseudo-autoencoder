CLASS Solution  
    FUNCTION minDistance(height INTEGER PARAMETER width INTEGER PARAMETER tree LIST OF INTEGER PARAMETER squirrel LIST OF INTEGER PARAMETER nuts LIST OF LIST OF INTEGER PARAMETER) RETURNS INTEGER  
        FUNCTION distance(p1 LIST OF INTEGER PARAMETER p2 LIST OF INTEGER PARAMETER) RETURNS INTEGER  
            SET horizontal_difference TO the absolute value of the element at position zero of p1 MINUS the element at position zero of p2  
            SET vertical_difference TO the absolute value of the element at position one of p1 MINUS the element at position one of p2  
            RETURN horizontal_difference PLUS vertical_difference  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance with arguments nut and tree  
            
            SET first_nut_distance TO distance with arguments squirrel and nut PLUS distance with arguments nut and tree  
            
            SET later_nut_distance TO two MULTIPLIED BY distance with arguments nut and tree  
            
            SET save TO later_nut_distance MINUS first_nut_distance  
            
            SET max_save TO the greater of max_save AND save  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
