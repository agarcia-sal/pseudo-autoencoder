CLASS Solution  
    FUNCTION largestMultipleOfThree WITH PARAMETER digits AS list of integers  
        SET count TO invoke auxiliary FUNCTION create_Counter WITH PARAMETER digits  
        SET total_sum TO invoke auxiliary FUNCTION calculate_Sum WITH PARAMETER digits  
        SET remainder TO the remainder of total_sum DIVIDED BY three  
        
        IF remainder EQUALS one  
            FOR each digit IN list consisting of digit one digit four digit seven  
                IF the frequency in count of digit GREATER THAN zero  
                    DECREMENT the frequency in count of digit BY one  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            IF the above FOR loop COMPLETED WITHOUT a BREAK  
                FOR each digit IN list consisting of digit two digit five digit eight  
                    IF the frequency in count of digit GREATER THAN one  
                        DECREMENT the frequency in count of digit BY two  
                        BREAK out of the FOR loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit IN list consisting of digit two digit five digit eight  
                IF the frequency in count of digit GREATER THAN zero  
                    DECREMENT the frequency in count of digit BY one  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            IF the above FOR loop COMPLETED WITHOUT a BREAK  
                FOR each digit IN list consisting of digit one digit four digit seven  
                    IF the frequency in count of digit GREATER THAN one  
                        DECREMENT the frequency in count of digit BY two  
                        BREAK out of the FOR loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        SET result TO empty list  
        FOR digit FROM nine DOWN TO zero  
            SET segment TO invoke auxiliary FUNCTION replicate_Character WITH PARAMETER digit AS string AND frequency from count  
            APPEND segment TO result  
        END FOR  
        
        SET final_number TO invoke auxiliary FUNCTION join_Strings WITH PARAMETER result  
        
        IF final_number is empty OR the character at position zero of final_number EQUALS zero  
            RETURN string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
