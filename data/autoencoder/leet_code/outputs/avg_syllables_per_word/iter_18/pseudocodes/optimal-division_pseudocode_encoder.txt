CLASS Solution  
    FUNCTION optimalDivision(nums AS List of integers) RETURNS string  
        IF the LENGTH OF nums EQUALS one  
            RETURN the TEXT REPRESENTATION OF the element at position zero of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the TEXT REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the SLASH CHARACTER CONCATENATED WITH the TEXT REPRESENTATION OF the element at position one of nums  
        END IF  
        RETURN the TEXT REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the SLASH CHARACTER CONCATENATED WITH the OPENING PARENTHESIS CHARACTER CONCATENATED WITH the JOINING WITH the SLASH CHARACTER OF the MAPPING TO TEXT REPRESENTATION OF each element at all positions from one TO the END OF nums CONCATENATED WITH the CLOSING PARENTHESIS CHARACTER  
    END FUNCTION  
END CLASS
