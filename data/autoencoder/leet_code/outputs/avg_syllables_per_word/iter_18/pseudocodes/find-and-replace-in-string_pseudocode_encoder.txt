CLASS Solution  
    FUNCTION findReplaceString(s AS string  indices AS list of integers  sources AS list of strings  targets AS list of strings) RETURNS string  
        SET replacements TO the LIST OF tuples COMBINING each element of indices WITH each element of sources WITH each element of targets SORTED BY the FIRST element of each tuple  
        SET result TO an EMPTY list  
        SET prev_end TO zero  
        FOR each tuple IN replacements WITH unpacking INTO index  source  target  
            APPEND the SUBSTRING of s FROM position prev_end TO position BEFORE index TO result  
            IF the SUBSTRING of s FROM position index TO position BEFORE the LENGTH OF source PLUS index EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the SUBSTRING of s FROM position index TO position BEFORE the LENGTH OF source PLUS index TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the SUBSTRING of s FROM position prev_end TO the END TO result  
        SET joined_result TO the CONCATENATION of all elements IN result  
        RETURN joined_result  
    END FUNCTION  
END CLASS
