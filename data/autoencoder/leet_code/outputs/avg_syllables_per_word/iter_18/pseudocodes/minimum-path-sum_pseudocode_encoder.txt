CLASS Solution  
    FUNCTION minPathSum WITH PARAMETER grid AS list of list of integer  
        IF grid IS empty OR element at position zero of grid IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        CALL auxiliary FUNCTION initialize_dp WITH PARAMETERS m AND n  
        SET dp TO the RESULT OF initialize_dp  
        
        SET element at position zero of element at position zero of dp TO element at position zero of element at position zero of grid  
        
        FOR j FROM one TO n MINUS one  
            SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp PLUS element at position j of element at position zero of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            SET element at position zero of element at position i of dp TO element at position zero of element at position i MINUS one of dp PLUS element at position zero of element at position i of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET smaller_value TO the minimum BETWEEN element at position j of element at position i MINUS one of dp AND element at position j MINUS one of element at position i of dp  
                SET element at position j of element at position i of dp TO smaller_value PLUS element at position j of element at position i of grid  
            END FOR  
        END FOR  
        
        RETURN element at position n MINUS one of element at position m MINUS one of dp  
    END FUNCTION  
  
    FUNCTION initialize_dp WITH PARAMETERS m AND n  
        RETURN auxiliary 2D list of integers WITH m rows AND n columns FULLY FILLED WITH zero  
    END FUNCTION  
END CLASS
