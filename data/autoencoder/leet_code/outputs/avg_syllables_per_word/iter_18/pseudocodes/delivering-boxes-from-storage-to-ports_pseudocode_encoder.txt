CLASS Solution  
    FUNCTION boxDelivering WITH PARAMETERS boxes portsCount maxBoxes maxWeight  
        SET n TO the LENGTH OF boxes  
        SET dp TO a list of length n PLUS one FILLED WITH a representation of positive infinity  
        SET element at position zero of dp TO zero  
        
        SET left TO zero  
        SET current_weight TO zero  
        SET extra_trips TO zero  
        
        FOR right FROM zero TO n MINUS one  
            INCREMENT current_weight BY the element at position one of the element at position right of boxes  
            
            IF right GREATER THAN zero AND the element at position zero of the element at position right of boxes NOT EQUALS the element at position zero of the element at position right MINUS one of boxes  
                INCREMENT extra_trips BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND (the difference of right MINUS left PLUS one GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight) HOLDS  
                DECREMENT current_weight BY the element at position one of the element at position left of boxes  
                
                IF left GREATER THAN zero AND the element at position zero of the element at position left of boxes NOT EQUALS the element at position zero of the element at position left MINUS one of boxes  
                    DECREMENT extra_trips BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            SET element at position right PLUS one of dp TO the sum of the element at position left of dp PLUS two PLUS extra_trips  
        END FOR  
        
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
