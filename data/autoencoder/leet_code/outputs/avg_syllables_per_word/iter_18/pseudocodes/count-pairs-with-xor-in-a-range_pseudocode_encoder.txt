CLASS TrieNode  
    FUNCTION __init__  
        SET children TO empty mapping  
        SET count TO zero  
    END FUNCTION  
END CLASS  

CLASS Trie  
    FUNCTION __init__ WITH bit_length defaulting TO fifteen  
        ASSIGN reference TO a new TrieNode TO root  
        SET bit_length TO bit_length  
    END FUNCTION  

    FUNCTION insert WITH parameter num  
        ASSIGN reference TO root TO node  
        FOR i FROM bit_length DOWN TO zero  
            SET bit TO the value of the bit at position i of num  
            IF bit not IN keys of children mapping of node THEN  
                ASSIGN reference TO a new TrieNode TO the element at key bit in children mapping of node  
            END IF  
            ASSIGN reference TO the element at key bit in children mapping of node TO node  
            INCREMENT count of node BY one  
        END FOR  
    END FUNCTION  

    FUNCTION count_less_than WITH parameters num and threshold  
        SET count TO zero  
        ASSIGN reference TO root TO node  
        FOR i FROM bit_length DOWN TO zero  
            IF node is null THEN  
                BREAK  
            END IF  
            SET bit TO the value of the bit at position i of num  
            SET threshold_bit TO the value of the bit at position i of threshold  
            IF threshold_bit EQUALS one THEN  
                INCREMENT count BY the count of the element at key bit in children mapping of node if present otherwise zero  
                ASSIGN reference TO the element at key bit XOR one in children mapping of node TO node  
            ELSE  
                ASSIGN reference TO the element at key bit in children mapping of node TO node  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION countPairs WITH parameters nums low and high  
        ASSIGN reference TO a new Trie TO trie  
        SET nice_pairs TO zero  
        FOR each num IN nums  
            INCREMENT nice_pairs BY the result from calling count_less_than on trie with num and high PLUS one MINUS the result from calling count_less_than on trie with num and low  
            CALL insert method on trie with argument num  
        END FOR  
        RETURN nice_pairs  
    END FUNCTION  
END CLASS
