CLASS Solution  
    FUNCTION stoneGameIII(stoneValue AS List of integer) RETURNS string  
        SET n TO the LENGTH OF stoneValue  
        SET dp TO a list of zero values of LENGTH n PLUS one  
        FOR index FROM n MINUS one DOWN TO zero  
            SET value_one TO element at position index of stoneValue  
            SET dp at position index TO value_one MINUS dp at position index PLUS one  
            IF index PLUS two IS LESS THAN OR EQUAL TO n THEN  
                SET sum_two TO element at position index of stoneValue PLUS element at position index PLUS one of stoneValue  
                SET dp at position index TO the GREATER OF dp at position index AND sum_two MINUS dp at position index PLUS two  
            END IF  
            IF index PLUS three IS LESS THAN OR EQUAL TO n THEN  
                SET sum_three TO element at position index of stoneValue PLUS element at position index PLUS one of stoneValue PLUS element at position index PLUS two of stoneValue  
                SET dp at position index TO the GREATER OF dp at position index AND sum_three MINUS dp at position index PLUS three  
            END IF  
        END FOR  
        IF dp at position zero IS GREATER THAN zero THEN  
            RETURN the string Alice  
        ELSE IF dp at position zero IS LESS THAN zero THEN  
            RETURN the string Bob  
        ELSE  
            RETURN the string Tie  
        END IF  
    END FUNCTION  
END CLASS
