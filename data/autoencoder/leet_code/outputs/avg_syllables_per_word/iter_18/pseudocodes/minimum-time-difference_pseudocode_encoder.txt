CLASS Solution  
    FUNCTION findMinDifference(timePoints AS List OF String) RETURNS Integer  
        FUNCTION to_minutes(time AS String) RETURNS Integer  
            SET hours AND minutes TO the result of mapping each element TO an integer from the result of splitting time at character colon  
            SET total_minutes TO hours MULTIPLIED BY sixty PLUS minutes  
            RETURN total_minutes  
        END FUNCTION  
  
        SET minutes_list TO the list resulting from mapping each element TO the result of applying to_minutes on elements of timePoints  
        SORT minutes_list in ascending order  
        SET min_diff TO a value representing infinitely large quantity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list  
            IF current_difference IS LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO the sum of the difference of one thousand four hundred forty MINUS the element at the last position of minutes_list AND the element at the first position of minutes_list  
        IF circular_difference IS LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
