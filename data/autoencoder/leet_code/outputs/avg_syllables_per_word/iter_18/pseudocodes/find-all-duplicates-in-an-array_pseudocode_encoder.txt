CLASS Solution  
    FUNCTION findDuplicates(nums AS List of Integer) RETURNS List of Integer  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position index of nums NOT EQUALS the element at the position equal to the element at position index of nums MINUS one of nums holds  
                SET temporary_first_value TO the element at position index of nums  
                SET temporary_second_value TO the element at the position equal to temporary_first_value MINUS one of nums  
                SET the element at position index of nums TO temporary_second_value  
                SET the element at the position equal to temporary_first_value MINUS one of nums TO temporary_first_value  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each current_index and current_value IN nums with enumeration starting at zero  
            IF current_value NOT EQUALS current_index PLUS one THEN  
                APPEND current_value TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
