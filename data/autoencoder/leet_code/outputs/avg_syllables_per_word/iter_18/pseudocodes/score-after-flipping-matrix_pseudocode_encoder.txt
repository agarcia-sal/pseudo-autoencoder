CLASS Solution  
    FUNCTION matrixScore grid  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR index FROM zero TO one LESS THAN the LENGTH OF row  
                    SET element at position index of row TO one MINUS element at position index of row  
                END FOR  
            END IF  
        END FOR  
        FOR col FROM one TO one LESS THAN the LENGTH OF element at position zero of grid  
            SET count_ones TO zero  
            FOR row_index FROM zero TO one LESS THAN the LENGTH OF grid  
                INCREMENT count_ones BY element at position col of element at position row_index of grid  
            END FOR  
            IF count_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row_index FROM zero TO one LESS THAN the LENGTH OF grid  
                    SET element at position col of element at position row_index of grid TO one MINUS element at position col of element at position row_index of grid  
                END FOR  
            END IF  
        END FOR  
        SET score TO zero  
        FOR each row IN grid  
            CALL auxiliary FUNCTION convert_row_to_integer WITH parameter row RETURNING integer_value  
            INCREMENT score BY integer_value  
        END FOR  
        RETURN score  
    END FUNCTION  
  
    FUNCTION convert_row_to_integer row  
        SET string_representation TO empty string  
        FOR each element IN row  
            APPEND string form of element TO string_representation  
        END FOR  
        SET integer_value TO integer conversion of string_representation interpreted as base two number  
        RETURN integer_value  
    END FUNCTION  
END CLASS
