CLASS Solution  
    FUNCTION findKthNumber(m AS integer parameter n AS integer parameter k AS integer parameter) RETURNS integer  
        FUNCTION enough(x AS integer parameter) RETURNS boolean  
            SET count TO zero  
            FOR i FROM one TO m  
                SET min_value TO the lesser of the quotient of x DIVIDED BY i AND n  
                INCREMENT count BY min_value  
            END FOR  
            RETURN count GREATER THAN OR EQUAL TO k  
        END FUNCTION  
  
        SET low TO one  
        SET high TO the product of m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO the quotient of the sum of low PLUS high DIVIDED BY two  
            IF enough(mid) HOLDS  
                SET high TO mid  
            ELSE  
                SET low TO the sum of mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
