CLASS Solution  
    FUNCTION minNumberOperations WITH PARAMETER target AS list of integers  
        SET operations TO zero  
        SET previous TO zero  
        FOR each current element IN target  
            IF current element IS GREATER THAN previous  
                INCREMENT operations BY current element MINUS previous  
            END IF  
            SET previous TO current element  
        END FOR  
        RETURN operations  
    END FUNCTION  
END CLASS
