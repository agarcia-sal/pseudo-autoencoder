CLASS Solution  
    FUNCTION maxRunTime WITH PARAMETERS n batteries  
        FUNCTION canRunFor WITH PARAMETER time  
            SET total TO zero  
            FOR each battery IN batteries  
                SET total TO total PLUS the smaller BETWEEN element at position battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  

        SET left TO zero  
        SET right TO the INTEGER division result OF the SUM OF elements IN batteries DIVIDED BY n  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER division result OF left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor WITH ARGUMENT mid HOLDS true  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
