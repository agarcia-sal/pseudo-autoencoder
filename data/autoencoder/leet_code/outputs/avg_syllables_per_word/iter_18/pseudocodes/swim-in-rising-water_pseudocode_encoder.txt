CLASS Solution  
    FUNCTION swimInWater(grid AS list of list of integers) RETURNS integer  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple of the element at position zero zero of grid and zero and zero  
        ADD the tuple of zero and zero TO visited  
        SET directions TO a list containing tuples zero one one zero zero negative one and negative one zero  
        
        WHILE min_heap is not empty  
            SET tuple_of_time_and_row_and_col TO the removal of the smallest element FROM min_heap  
            SET t TO the first element of tuple_of_time_and_row_and_col  
            SET i TO the second element of tuple_of_time_and_row_and_col  
            SET j TO the third element of tuple_of_time_and_row_and_col  
            
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each pair di and dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple of ni and nj IS NOT IN visited  
                    ADD the tuple of ni and nj TO visited  
                    INSERT a tuple of the maximum of t and the element at position ni nj of grid and ni and nj INTO min_heap preserving heap ordering  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
