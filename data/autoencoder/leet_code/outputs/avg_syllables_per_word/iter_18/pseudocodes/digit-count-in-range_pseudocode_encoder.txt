CLASS Solution  
    FUNCTION digitsCount WITH PARAMETERS d INTEGER low INTEGER high INTEGER  
        FUNCTION count_digit WITH PARAMETERS n INTEGER digit INTEGER  
            SET count TO zero  
            SET power_of_ten TO one  
            SET current TO n  
            WHILE current IS GREATER THAN zero  
                SET last_digit TO the remainder of current DIVIDED BY ten  
                SET current TO the integer division of current BY ten  
                SET full_cycles TO current  
                IF digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_ten  
                IF last_digit IS GREATER THAN digit  
                    INCREMENT count BY power_of_ten  
                ELSE IF last_digit EQUALS digit  
                    INCREMENT count BY the remainder of n DIVIDED BY power_of_ten PLUS one  
                END IF  
                MULTIPLY power_of_ten BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN the result of count_digit WITH ARGUMENTS high AND d MINUS the result of count_digit WITH ARGUMENTS low MINUS one AND d  
    END FUNCTION  
END CLASS
