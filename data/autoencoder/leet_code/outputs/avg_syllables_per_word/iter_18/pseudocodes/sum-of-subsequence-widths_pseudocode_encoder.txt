CLASS Solution  
    FUNCTION sumSubseqWidths(nums AS List OF integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SORT nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list containing the integer one repeated n times  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the PRODUCT OF element at position index MINUS one of power AND two MODULO MOD  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO the REMAINDER OF the SUM OF total_sum AND the PRODUCT OF element at position index of nums AND the DIFFERENCE BETWEEN element at position index of power AND element at position n MINUS index MINUS one of power WHEN DIVIDED BY MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
