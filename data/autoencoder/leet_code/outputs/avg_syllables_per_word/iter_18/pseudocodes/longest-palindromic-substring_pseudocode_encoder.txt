CLASS Solution  
    FUNCTION longestPalindrome WITH parameter s  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional collection of boolean true values of size n BY n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i AND j OF f TO boolean false  
                IF character at position i OF s EQUALS character at position j OF s THEN  
                    SET element at position i AND j OF f TO element at position i PLUS one AND j MINUS one OF f  
                    IF element at position i AND j OF f IS true AND mx LESS THAN j MINUS i PLUS one THEN  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the SUBSTRING of s FROM position k TO position k PLUS mx  
    END FUNCTION  
END CLASS
