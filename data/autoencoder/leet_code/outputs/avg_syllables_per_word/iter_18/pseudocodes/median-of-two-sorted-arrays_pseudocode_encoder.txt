CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1 AS List OF integer, nums2 AS List OF integer) RETURNS floating point number  
        IF the LENGTH OF nums1 GREATER THAN the LENGTH OF nums2  
            SET temporary_list TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary_list  
        END IF  
        
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION OF the SUM OF low AND high BY two  
            SET partitionY TO the INTEGER DIVISION OF the SUM OF x AND y AND one BY two MINUS partitionX  
            
            IF partitionX EQUALS zero  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO the element at position partitionX MINUS one of nums1  
            END IF  
            
            IF partitionX EQUALS x  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO the element at position partitionX of nums1  
            END IF  
            
            IF partitionY EQUALS zero  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO the element at position partitionY MINUS one of nums2  
            END IF  
            
            IF partitionY EQUALS y  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO the element at position partitionY of nums2  
            END IF  
            
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX  
                IF the REMAINDER OF the SUM OF x AND y DIVIDED BY two EQUALS zero  
                    SET maximum_value TO the GREATER OF maxLeftX AND maxLeftY  
                    SET minimum_value TO the LESSER OF minRightX AND minRightY  
                    SET median TO the SUM OF maximum_value AND minimum_value DIVIDED BY two  
                    RETURN median  
                ELSE  
                    SET median TO the GREATER OF maxLeftX AND maxLeftY  
                    RETURN median  
                END IF  
            ELSE IF maxLeftX GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
