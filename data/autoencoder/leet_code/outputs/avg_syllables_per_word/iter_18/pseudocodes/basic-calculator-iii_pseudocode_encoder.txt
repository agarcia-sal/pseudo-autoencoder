CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the character representing plus  
            WHILE the length of s is greater than zero holds  
                SET char TO the element at position zero of s removed from s  
                IF char represents a digit character  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char equals the character representing an opening parenthesis  
                    SET num TO the result of calling helper WITH PARAMETER s  
                END IF  
                IF char is among the characters representing plus or minus or multiplication or division or opening parenthesis or closing parenthesis OR the length of s equals zero  
                    IF sign equals the character representing plus  
                        APPEND num TO stack  
                    ELSE IF sign equals the character representing minus  
                        APPEND num MULTIPLIED BY minus one TO stack  
                    ELSE IF sign equals the character representing multiplication  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign equals the character representing division  
                        SET the element at the last position of stack TO the integer part of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char equals the character representing a closing parenthesis  
                    BREAK the loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        SET s TO the list of characters of s  
        RETURN the result of calling helper WITH PARAMETER s  
    END FUNCTION  
END CLASS
