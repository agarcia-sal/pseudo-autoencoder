CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a mapping of zero TO minus one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index and element IN enumerated nums  
            IF element EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the GREATER OF max_length AND the difference BETWEEN index AND element at key count OF prefix_sum_index  
            ELSE  
                SET the value at key count OF prefix_sum_index TO index  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
