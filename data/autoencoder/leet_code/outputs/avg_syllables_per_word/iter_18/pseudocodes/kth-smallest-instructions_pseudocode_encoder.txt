CLASS Solution  
    FUNCTION kthSmallestPath(destination AS List OF integers, k AS integer) RETURNS string  
        SET row TO element at position zero of destination  
        SET col TO element at position one of destination  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero THEN  
                SET paths_with_H TO the number of combinations of row PLUS col MINUS one TAKEN col MINUS one AT A TIME  
                IF k LESS THAN OR EQUAL TO paths_with_H THEN  
                    APPEND character H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND character V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND character V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN the concatenation of all elements in path AS string  
    END FUNCTION  
END CLASS
