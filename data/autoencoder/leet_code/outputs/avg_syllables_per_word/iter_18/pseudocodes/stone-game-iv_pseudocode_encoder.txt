CLASS Solution  
    FUNCTION winnerSquareGame WITH parameter n  
        SET dp TO a list of the boolean value false repeated for the quantity of n PLUS one times  
        FOR each integer i FROM one TO n  
            SET j TO one  
            WHILE the product of j and j is less than or equal to i holds  
                IF the negation of the element at position i MINUS the product of j and j of dp is true THEN  
                    SET the element at position i of dp TO true  
                    BREAK out of the current WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
