CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each m IN nums  
            IF m EQUALS m1 THEN  
                INCREMENT n1 BY one  
            ELSE IF m EQUALS m2 THEN  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUALS zero THEN  
                SET m1 TO m  
                SET n1 TO one  
            ELSE IF n2 EQUALS zero THEN  
                SET m2 TO m  
                SET n2 TO one  
            ELSE  
                SET n1 TO n1 MINUS one  
                SET n2 TO n2 MINUS one  
            END IF  
        END FOR  
        SET result_list TO an empty list  
        FOR each candidate IN the list containing m1 and m2  
            IF the number of occurrences of candidate in nums GREATER THAN the integer part of the LENGTH OF nums DIVIDED BY three THEN  
                APPEND candidate TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
