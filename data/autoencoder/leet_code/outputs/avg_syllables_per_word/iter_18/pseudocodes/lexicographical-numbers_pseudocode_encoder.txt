CLASS Solution  
    FUNCTION lexicalOrder WITH PARAMETER n  
        SET result TO empty list  
        
        FUNCTION dfs WITH PARAMETER current  
            IF current GREATER THAN n THEN  
                RETURN  
            END IF  
            APPEND current TO result  
            FOR integer_i FROM zero TO nine MINUS one  
                SET next_number TO current MULTIPLIED BY ten PLUS integer_i  
                IF next_number GREATER THAN n THEN  
                    RETURN  
                END IF  
                CALL dfs WITH next_number  
            END FOR  
        END FUNCTION  
        
        FOR integer_i FROM one TO nine MINUS one  
            CALL dfs WITH integer_i  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
