CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        IF primeFactors LESS THAN OR EQUAL TO three THEN  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO the integer result of primeFactors DIVIDED BY three  
        SET remainder TO the remainder resulting from primeFactors DIVIDED BY three  
        
        IF remainder EQUALS zero THEN  
            RETURN the result of raising three to the power of num_threes with modulus MOD  
        ELSE IF remainder EQUALS one THEN  
            RETURN the product of raising three to the power of num_threes MINUS one with modulus MOD AND four all modulo MOD  
        ELSE  
            RETURN the product of raising three to the power of num_threes with modulus MOD AND two all modulo MOD  
        END IF  
    END FUNCTION  
END CLASS
