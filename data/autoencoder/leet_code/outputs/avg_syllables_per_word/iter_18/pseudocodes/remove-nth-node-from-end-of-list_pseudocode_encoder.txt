CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET val ATTRIBUTE OF self TO val  
        SET next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values DOES NOT EXIST OR THE LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO A NEW INSTANCE OF ListNode WITH THE element at position zero of values AS val  
    SET p TO head  
    FOR each val IN values STARTING FROM THE element at position one TO THE LAST element OF values  
        SET node TO A NEW INSTANCE OF ListNode WITH val AS val  
        SET next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 DOES NOT EXIST AND p2 DOES NOT EXIST THEN  
        RETURN True  
    END IF  
    IF p1 DOES NOT EXIST OR p2 DOES NOT EXIST THEN  
        RETURN False  
    END IF  
    RETURN THE val ATTRIBUTE OF p1 EQUALS THE val ATTRIBUTE OF p2 AND is_same_list OF THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO A NEW INSTANCE OF ListNode WITH val EQUALS zero  
        SET next ATTRIBUTE OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR index FROM zero TO n INCLUSIVE  
            SET first TO next ATTRIBUTE OF first  
        END FOR  
        WHILE first EXISTS  
            SET first TO next ATTRIBUTE OF first  
            SET second TO next ATTRIBUTE OF second  
        END WHILE  
        SET next ATTRIBUTE OF second TO next ATTRIBUTE OF next ATTRIBUTE OF second  
        RETURN next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
