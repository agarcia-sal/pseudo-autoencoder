CLASS Solution  
    FUNCTION numDistinct(s AS string, t AS string) RETURNS integer  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        
        CALL create_two_dimensional_array WITH m PLUS one AND n PLUS one TO OBTAIN dp  
        
        FOR i FROM zero TO m  
            SET element at position zero of the row at position i of dp TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s EQUALS character at position j MINUS one of t THEN  
                    SET element at position j of the row at position i of dp TO value at position j MINUS one of the row at position i MINUS one of dp PLUS value at position j of the row at position i MINUS one of dp  
                ELSE  
                    SET element at position j of the row at position i of dp TO value at position j of the row at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN value at position n of the row at position m of dp  
    END FUNCTION  
END CLASS
