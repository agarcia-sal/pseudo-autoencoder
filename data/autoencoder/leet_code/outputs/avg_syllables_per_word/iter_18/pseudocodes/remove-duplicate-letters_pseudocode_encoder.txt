CLASS Solution  
    FUNCTION removeDuplicateLetters WITH parameter s  
        SET last_occurrence TO a mapping from each character and its position of last appearance in s  
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index AND character IN s  
            IF character IS IN in_stack  
                CONTINUE  
            END IF  
            
            WHILE stack is not empty AND character is lexicographically smaller THAN character at the last position in stack AND index is less THAN the last appearance position of the character at the last position in stack in last_occurrence  
                SET removed_char TO element at the last position of stack removed from stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result_string TO the concatenation of all characters in stack in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
