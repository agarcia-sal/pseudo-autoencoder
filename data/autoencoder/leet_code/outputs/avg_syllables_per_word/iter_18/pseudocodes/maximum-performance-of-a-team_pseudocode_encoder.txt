CLASS Solution  
    FUNCTION maxPerformance(n AS integer, speed AS list of integers, efficiency AS list of integers, k AS integer) RETURNS integer  
        SET engineers TO AUXILIARY zip_lists_of efficiency AND speed  
        CALL AUXILIARY sort_list_of engineers IN descending order BY element at position one  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each element IN engineers  
            SET eff TO element at position one of element  
            SET spd TO element at position two of element  
            CALL AUXILIARY push_element spd INTO speed_heap USING a min_heap_strategy  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap GREATER THAN k THEN  
                DECREMENT speed_sum BY CALL AUXILIARY pop_smallest_element FROM speed_heap  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the GREATER OF max_performance AND current_performance  
        END FOR  
        RETURN the REMAINDER OF max_performance DIVIDED BY one billion PLUS seven  
    END FUNCTION  
END CLASS
