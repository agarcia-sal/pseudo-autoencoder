CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of numbers from zero TO n MINUS one  
        SET self.rank TO list of zero repeated n times  
    END FUNCTION  

    FUNCTION find(self, x)  
        IF element at position x of self.parent NOT EQUALS x THEN  
            SET element at position x of self.parent TO the result of self.find of element at position x of self.parent  
        END IF  
        RETURN element at position x of self.parent  
    END FUNCTION  

    FUNCTION union(self, x, y)  
        SET rootX TO the result of self.find of x  
        SET rootY TO the result of self.find of y  
        IF rootX NOT EQUALS rootY THEN  
            IF element at position rootX of self.rank GREATER THAN element at position rootY of self.rank THEN  
                SET element at position rootY of self.parent TO rootX  
            ELSE IF element at position rootX of self.rank LESS THAN element at position rootY of self.rank THEN  
                SET element at position rootX of self.parent TO rootY  
            ELSE  
                SET element at position rootY of self.parent TO rootX  
                INCREMENT element at position rootX of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION latestDayToCross(self, row, col, cells)  
        SET grid TO list containing row elements each of which is list containing col elements each set to one  
        SET uf TO new instance of UnionFind with parameter row MULTIPLIED BY col PLUS two  
        SET TOP TO row MULTIPLIED BY col  
        SET BOTTOM TO row MULTIPLIED BY col PLUS one  
        SET directions TO list containing four pairs of integers minus one and zero one and zero zero and minus one zero and one  
        
        FOR day FROM the LENGTH OF cells MINUS one DOWN TO zero DO  
            SET r TO element at position day of cells at position zero MINUS one  
            SET c TO element at position day of cells at position one MINUS one  
            SET element at position r of grid at position c TO zero  
            
            FOR each pair dr and dc IN directions DO  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN row AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN col AND element at position nr of grid at position nc EQUALS zero THEN  
                    CALL uf.union with parameters r MULTIPLIED BY col PLUS c AND nr MULTIPLIED BY col PLUS nc  
                END IF  
            END FOR  
            
            IF r EQUALS zero THEN  
                CALL uf.union with parameters r MULTIPLIED BY col PLUS c AND TOP  
            END IF  
            
            IF r EQUALS row MINUS one THEN  
                CALL uf.union with parameters r MULTIPLIED BY col PLUS c AND BOTTOM  
            END IF  
            
            IF uf.find of TOP EQUALS uf.find of BOTTOM THEN  
                RETURN day  
            END IF  
        END FOR  
        
        RETURN zero  
    END FUNCTION  
END CLASS
