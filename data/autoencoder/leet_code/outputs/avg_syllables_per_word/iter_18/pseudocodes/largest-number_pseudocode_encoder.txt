CLASS Solution  
    FUNCTION largestNumber(nums AS List of integers) RETURNS string  
        SET nums_str TO a new list of strings converted from each integer in nums  
        
        FUNCTION compare(x AS string, y AS string) RETURNS integer  
            IF concatenation of x followed by y is greater than concatenation of y followed by x holds  
                RETURN negative one  
            ELSE IF concatenation of x followed by y is less than concatenation of y followed by x holds  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str according to comparison using compare function as custom key  
        
        IF the element at position zero of nums_str equals to string zero holds  
            RETURN string zero  
        END IF  
        
        SET result_string TO concatenation of all elements of nums_str in order  
        
        RETURN result_string  
    END FUNCTION  
END CLASS
