CLASS Solution  
    FUNCTION advantageCount(nums1 AS List OF integers, nums2 AS List OF integers) AS List OF integers  
        CALL sort the elements of nums1 into increasingly ordered sequence  
        SET nums2_with_indices TO list composed of each pair of value and original position taken from nums2 ordered from largest value to smallest value  
        SET result TO list of zero repeated the count of elements in nums1 times  
        SET left pointer TO zero  
        SET right pointer TO the count of elements in nums1 MINUS one  
        FOR each pair of current_value and original_position IN nums2_with_indices  
            IF element at position right of nums1 GREATER THAN current_value THEN  
                SET element at position original_position of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position original_position of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
