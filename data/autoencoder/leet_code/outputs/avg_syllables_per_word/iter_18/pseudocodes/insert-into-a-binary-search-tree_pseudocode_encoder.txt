CLASS TreeNode  
    FUNCTION __init__(val SET TO zero OPTIONAL, left SET TO none OPTIONAL, right SET TO none OPTIONAL)  
        SET the property val OF self TO val  
        SET the property left OF self TO left  
        SET the property right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values AS list)  
    IF the values collection IS empty  
        RETURN none  
    END IF  
    SET root TO A new instance OF TreeNode created WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an instance OF deque collection  
    APPEND root TO queue  
    WHILE the queue collection IS NOT empty  
        SET node TO the first element REMOVED FROM the queue collection  
        IF the integer i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the property left OF node TO a new instance OF TreeNode created WITH the element at position i of values  
            APPEND the property left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF the integer i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the property right OF node TO a new instance OF TreeNode created WITH the element at position i of values  
            APPEND the property right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the property val OF p IS NOT EQUAL TO the property val OF q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree WITH the property left OF p AND the property left OF q AND the result of calling is_same_tree WITH the property right OF p AND the property right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(root AS optional TreeNode, val AS integer) RETURNS optional TreeNode  
        FUNCTION dfs(root)  
            IF root IS none  
                RETURN A new instance OF TreeNode created WITH val  
            END IF  
            IF the property val OF root IS LESS THAN val  
                SET the property right OF root TO the result of calling dfs WITH the property right OF root  
            ELSE  
                SET the property left OF root TO the result of calling dfs WITH the property left OF root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN the result of calling dfs WITH root  
    END FUNCTION  
END CLASS
