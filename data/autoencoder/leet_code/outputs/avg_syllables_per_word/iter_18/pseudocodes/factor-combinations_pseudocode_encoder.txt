CLASS Solution  
    FUNCTION getFactors WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS start AND target  
            FOR i FROM start TO the FLOOR OF the SQUARE ROOT OF target PLUS one  
                IF the REMAINDER OF target DIVIDED BY i EQUALS zero THEN  
                    APPEND the CONCATENATION OF path PLUS i PLUS target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the LAST ELEMENT FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an EMPTY list  
        SET path TO an EMPTY list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
