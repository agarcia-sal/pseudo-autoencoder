CLASS Solution  
    FUNCTION findUnsortedSubarray WITH PARAMETER nums AS list of integers  
        SET sorted_nums TO the sorted version of nums  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO the length of nums MINUS one  
            IF element at position index of nums IS NOT EQUAL TO element at position index of sorted_nums THEN  
                IF start IS EQUAL TO negative one THEN  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        SET subarray_length TO end MINUS start PLUS one  
        RETURN subarray_length  
    END FUNCTION  
END CLASS
