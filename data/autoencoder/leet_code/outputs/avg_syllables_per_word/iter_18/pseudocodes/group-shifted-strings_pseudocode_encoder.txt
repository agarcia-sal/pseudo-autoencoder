CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the ordinal value of character c MINUS the ordinal value of character at position zero of string s  
                SET wrapped_difference TO difference MODULO twenty six  
                APPEND the string representation of wrapped_difference TO canonical_form  
            END FOR  
            RETURN the concatenation of the elements of canonical_form JOINED BY the character representing a comma  
        END FUNCTION  
        
        SET groups TO a new mapping with default value as empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form with argument s  
            APPEND s TO the list in groups at key canonical_form  
        END FOR  
        
        RETURN the list of values contained in groups  
    END FUNCTION  
END CLASS
