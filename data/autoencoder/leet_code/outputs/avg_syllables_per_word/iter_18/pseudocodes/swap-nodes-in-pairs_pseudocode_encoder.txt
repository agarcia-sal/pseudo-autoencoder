CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode with value equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one UNTIL the end  
        SET node TO new instance of ListNode with value equal to val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN value at p1 EQUALS value at p2 AND is_same_list(next of p1, next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(head)  
        SET dummy TO new instance of ListNode with value equal to zero  
        SET next of dummy TO head  
        SET current TO dummy  
        WHILE next of current IS NOT none AND next of next of current IS NOT none  
            SET first TO next of current  
            SET second TO next of next of current  
            SET next of current TO second  
            SET next of first TO next of second  
            SET next of second TO first  
            SET current TO next of next of current  
        END WHILE  
        RETURN next of dummy  
    END FUNCTION  
END CLASS
