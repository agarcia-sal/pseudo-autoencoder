CLASS Solution  
    FUNCTION minDeletionSize(strs AS List of String)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of size n with all elements set to one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR each k FROM zero TO m MINUS one the character at position j of element at position k of strs IS LESS THAN OR EQUAL TO the character at position i of element at position k of strs holds  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM value in dp  
        RETURN n MINUS longest_non_decreasing_subsequence_length  
    END FUNCTION  
END CLASS
