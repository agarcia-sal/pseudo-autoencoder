CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER n  
        SET parent TO list of integers FROM zero TO n MINUS one  
        SET rank TO list of the integer one REPEATED n times  
    END FUNCTION  
  
    FUNCTION find WITH PARAMETER u  
        IF element at position u of parent NOT EQUALS u  
            SET element at position u of parent TO the result of find WITH PARAMETER element at position u of parent  
        END IF  
        RETURN element at position u of parent  
    END FUNCTION  
  
    FUNCTION union WITH PARAMETERS u AND v  
        SET rootU TO the result of find WITH PARAMETER u  
        SET rootV TO the result of find WITH PARAMETER v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of rank GREATER THAN element at position rootV of rank  
                SET element at position rootV of parent TO rootU  
            ELSE IF element at position rootU of rank LESS THAN element at position rootV of rank  
                SET element at position rootU of parent TO rootV  
            ELSE  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of rank BY one  
            END IF  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges WITH PARAMETERS n AND edges  
        FOR each index i AND element edge IN edges WITH enumeration  
            APPEND i TO edge  
        END FOR  
  
        SORT edges BY the value of element at position three of each edge FROM lowest TO highest  
  
        FUNCTION mst WITH OPTIONAL PARAMETERS exclude AND include  
            SET uf TO new instance of UnionFind WITH PARAMETER n  
            SET weight TO zero  
            IF include IS NOT none  
                CALL union ON uf WITH PARAMETERS element at position zero of include AND element at position one of include  
                INCREMENT weight BY element at position two of include  
            END IF  
  
            FOR each element u v w idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE  
                END IF  
                IF calling union on uf WITH PARAMETERS u AND v RETURNS true  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
  
            SET root TO the result of find on uf WITH PARAMETER zero  
            FOR each integer i FROM zero TO n MINUS one  
                IF the result of find on uf WITH PARAMETER i NOT EQUALS root  
                    RETURN infinite value  
                END IF  
            END FOR  
  
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO the result of calling mst WITHOUT PARAMETERS  
  
        SET critical TO empty list  
        SET pseudo_critical TO empty list  
  
        FOR each element u v w idx IN edges  
            IF the result of calling mst WITH exclude EQUALS idx GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of calling mst WITH include EQUALS tuple of u v w EQUALS mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
  
        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
