CLASS Solution  
    FUNCTION minInteger(num AS string, k AS integer) RETURNS string  
        SET num TO the LIST OF CHARACTERS IN num  
        SET n TO the LENGTH OF num  
          
        WHILE k GREATER THAN zero holds  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF character at position i of num GREATER THAN character at position i PLUS one of num THEN  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF character at position j of num LESS THAN min_digit THEN  
                            SET min_digit TO character at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the SUBLIST of num from position i TO min_index PLUS one TO a NEW LIST with element character at position min_index of num FOLLOWED BY the SUBLIST of num from position i TO min_index MINUS one  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS false THEN  
                BREAK out of the WHILE loop  
            END IF  
        END WHILE  
          
        RETURN the CONCATENATION of all CHARACTERS in num AS a STRING  
    END FUNCTION  
END CLASS
