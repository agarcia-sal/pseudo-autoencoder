CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero left SET TO none right SET TO none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the collection values is empty  
        RETURN none  
    END IF  
    SET root TO an instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND element at position i of values is not none  
            SET node left TO an instance of TreeNode initialized with the element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND element at position i of values is not none  
            SET node right TO an instance of TreeNode initialized with the element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the element val of p is not equal to the element val of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with parameters the element left of p and the element left of q AND the result of is_same_tree with parameters the element right of p and the element right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum WITH PARAMETERS self root  
        FUNCTION helper WITH PARAMETERS node  
            IF node is none  
                RETURN negative infinity comma negative infinity  
            END IF  
            SET left_max comma left_gain TO the result of helper with parameter the element left of node  
            SET right_max comma right_gain TO the result of helper with parameter the element right of node  
            SET current_max TO the element val of node PLUS the greater value BETWEEN left_gain AND zero PLUS the greater value BETWEEN right_gain AND zero  
            SET current_gain TO the element val of node PLUS the greatest value AMONG left_gain AND right_gain AND zero  
            SET global_max TO the greatest value AMONG left_max AND right_max AND current_max  
            RETURN global_max comma current_gain  
        END FUNCTION  
        RETURN the first element of the result of helper with parameter root  
    END FUNCTION  
END CLASS
