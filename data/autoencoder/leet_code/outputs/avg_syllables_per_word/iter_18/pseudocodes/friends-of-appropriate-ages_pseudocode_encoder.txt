CLASS Solution  
    FUNCTION numFriendRequests WITH PARAMETER ages AS List of integers  
        FUNCTION request_will_be_sent WITH PARAMETERS age_x AND age_y  
            IF age_y IS LESS THAN OR EQUAL TO half OF age_x PLUS seven OR age_y IS GREATER THAN age_x OR age_y IS GREATER THAN one hundred AND age_x IS LESS THAN one hundred HOLDS  
                RETURN value NOT TRUE  
            ELSE  
                RETURN value TRUE  
            END IF  
        END FUNCTION  
        
        SET count TO a list of zero VALUES repeated one hundred twenty one TIMES  
        FOR each age IN ages  
            INCREMENT element at position age of count BY one  
        END FOR  
        
        SET total_requests TO zero  
        FOR age_x FROM one TO one hundred twenty  
            FOR age_y FROM one TO one hundred twenty  
                IF request_will_be_sent WITH ARGUMENTS age_x AND age_y RETURNS TRUE  
                    INCREMENT total_requests BY element at position age_x of count MULTIPLIED BY element at position age_y of count MINUS one IF age_x EQUALS age_y OTHERWISE zero  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_requests  
    END FUNCTION  
END CLASS
