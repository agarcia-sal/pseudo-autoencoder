CLASS Solution  
    FUNCTION distanceToCycle WITH PARAMETERS n AND edges  
        SET graph TO a new default mapping FROM keys TO empty lists  
        FOR each pair u AND v IN edges  
            APPEND v TO element at key u IN graph  
            APPEND u TO element at key v IN graph  
        END FOR  
        
        FUNCTION dfs WITH PARAMETERS node AND parent AND path AND visited  
            IF node IS contained IN visited THEN  
                SET cycle_start TO the position of node IN path  
                RETURN the slice of path FROM cycle_start TO the end  
            END IF  
            APPEND node TO path  
            ADD node TO visited  
            FOR each neighbor IN element at key node IN graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO the result of dfs CALLED WITH neighbor AND node AND path AND visited  
                    IF cycle IS NOT equal TO none THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN none  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO none  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT contained IN visited THEN  
                SET cycle TO the result of dfs CALLED WITH node AND negative one AND an empty list AND visited  
                IF cycle IS NOT equal TO none THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zero repeated n times  
        SET queue TO a new deque INITIALIZED WITH elements of cycle  
        SET visited TO a set INITIALIZED WITH elements of cycle  
        
        SET level TO zero  
        WHILE queue IS NOT empty  
            SET loop_count TO the number of elements in queue  
            FOR counter FROM one TO loop_count  
                REMOVE element from the front of queue AND ASSIGN TO node  
                SET element at position node IN distance TO level  
                FOR each neighbor IN element at key node IN graph  
                    IF neighbor IS NOT contained IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
