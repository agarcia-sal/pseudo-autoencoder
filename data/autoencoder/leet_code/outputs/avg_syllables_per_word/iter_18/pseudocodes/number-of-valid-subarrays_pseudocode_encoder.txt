CLASS Solution  
    FUNCTION validSubarrays(nums AS List of integer) RETURNS integer  
        SET stack TO empty list  
        SET count TO zero  
        
        FOR each num IN nums  
            WHILE the LENGTH OF stack is greater than zero AND the element at position the LENGTH OF stack minus one of stack is GREATER THAN num  
                REMOVE the last element FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
