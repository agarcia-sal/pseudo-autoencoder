CLASS Solution  
    FUNCTION findMaxValueOfEquation(points AS List OF List OF integer, k AS integer) RETURNS integer  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  

        FOR each element IN points  
            DESCRIBE element AS xj AND yj  

            WHILE deque_ is not empty AND the difference between xj AND the position at the front element of deque_ is greater than k  
                REMOVE the front element FROM deque_  
            END WHILE  

            IF deque_ is not empty  
                SET max_value TO the maximum of max_value AND the sum of yj PLUS xj PLUS the first component of the front element of deque_  
            END IF  

            WHILE deque_ is not empty AND the difference of yj MINUS xj is greater than or equal to the first component of the element at the back of deque_  
                REMOVE the back element FROM deque_  
            END WHILE  

            APPEND the pair consisting of the difference of yj MINUS xj AND xj TO the back of deque_  
        END FOR  

        RETURN max_value  
    END FUNCTION  
END CLASS
