CLASS Solution  
    FUNCTION minSwap(nums1 AS List OF integer, nums2 AS List OF integer) RETURNS integer  
        SET n TO the LENGTH OF nums1  
        SET dp TO a two dimensional list OF zeroes with n rows and two columns  
        SET element at position zero and zero of dp TO zero  
        SET element at position zero and one of dp TO one  
        FOR index FROM one TO n DECREMENTED BY negative one  
            IF element at position index of nums1 GREATER THAN element at position index decremented BY one of nums1 AND element at position index of nums2 GREATER THAN element at position index decremented BY one of nums2 THEN  
                IF element at position index of nums1 GREATER THAN element at position index decremented BY one of nums2 AND element at position index of nums2 GREATER THAN element at position index decremented BY one of nums1 THEN  
                    SET element at position index and zero of dp TO the MINIMUM OF element at position index decremented BY one and zero of dp AND element at position index decremented BY one and one of dp  
                    SET element at position index and one of dp TO element at position index and zero of dp PLUS one  
                ELSE  
                    SET element at position index and zero of dp TO element at position index decremented BY one and zero of dp  
                    SET element at position index and one of dp TO element at position index decremented BY one and one of dp PLUS one  
                END IF  
            ELSE  
                IF element at position index of nums1 GREATER THAN element at position index decremented BY one of nums2 AND element at position index of nums2 GREATER THAN element at position index decremented BY one of nums1 THEN  
                    SET element at position index and zero of dp TO element at position index decremented BY one and one of dp  
                    SET element at position index and one of dp TO element at position index decremented BY one and zero of dp PLUS one  
                ELSE  
                    SET element at position index and zero of dp TO element at position index decremented BY one and one of dp  
                    SET element at position index and one of dp TO element at position index decremented BY one and zero of dp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the MINIMUM OF element at position n decremented BY one and zero of dp AND element at position n decremented BY one and one of dp  
    END FUNCTION  
END CLASS
