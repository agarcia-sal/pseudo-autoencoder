CLASS Solution  
    FUNCTION numberToWords WITH PARAMETER num OF TYPE integer  
        IF num EQUALS zero  
            RETURN the word zero  
        END IF  
        
        SET below_twenty TO list of words representing zero through nineteen with empty string at position zero  
        SET tens TO list containing empty strings at positions zero and one followed by words for multiples of ten from twenty through ninety  
        SET thousands TO list containing empty string followed by words thousand million billion  
        
        FUNCTION helper WITH PARAMETER n  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN word at position n of below_twenty FOLLOWED BY a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN word at quotient of n DIVIDED BY ten from tens FOLLOWED BY a single space FOLLOWED BY helper with remainder of n MODULO ten  
            ELSE  
                RETURN word at quotient of n DIVIDED BY one hundred from below_twenty FOLLOWED BY the word hundred FOLLOWED BY a single space FOLLOWED BY helper with remainder of n MODULO one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero holds  
            IF remainder of num MODULO one thousand NOT EQUALS zero  
                SET result TO concatenation of helper of remainder of num MODULO one thousand FOLLOWED BY word at position i of thousands FOLLOWED BY a single space FOLLOWED BY current result  
            END IF  
            SET num TO quotient of num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
