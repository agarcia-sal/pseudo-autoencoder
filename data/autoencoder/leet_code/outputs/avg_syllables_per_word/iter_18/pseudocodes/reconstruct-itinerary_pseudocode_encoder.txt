CLASS Solution  
    FUNCTION findItinerary WITH parameter tickets AS list of list of strings  
        SET graph TO new mapping from string to list of strings with default value as empty list  
        FOR each pair of departure and arrival IN the tickets sorted in reverse alphabetical order  
            APPEND arrival TO the list at key departure in graph  
        END FOR  
        
        SET itinerary TO an empty list of strings  
        
        FUNCTION visit WITH parameter airport AS string  
            WHILE the list at key airport in graph holds elements  
                CALL visit WITH the result of REMOVING the last element from the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH the string named JFK  
        RETURN the list itinerary traversed in reverse order  
    END FUNCTION  
END CLASS
