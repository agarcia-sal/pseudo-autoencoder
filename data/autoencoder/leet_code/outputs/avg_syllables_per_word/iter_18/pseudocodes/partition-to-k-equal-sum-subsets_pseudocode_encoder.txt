CLASS Solution  
    FUNCTION canPartitionKSubsets(nums AS list of integer, k AS integer) RETURNS boolean  
        SET total_sum TO the sum of all elements in nums  
        IF the remainder of total_sum divided BY k GREATER THAN zero THEN  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending order  
        
        FUNCTION can_partition(index AS integer, k AS integer, current_sum AS integer, used AS list of boolean) RETURNS boolean  
            IF k EQUALS zero THEN  
                RETURN True  
            END IF  
            IF current_sum EQUALS target THEN  
                RETURN the result of can_partition with index set to zero PLUS zero, k decremented BY one, current_sum set to zero, and used AS the same list  
            END IF  
            
            FOR i FROM index TO the length OF nums MINUS one  
                IF element at position i of used EQUALS False AND current_sum PLUS element at position i of nums LESS THAN OR EQUAL TO target THEN  
                    SET element at position i of used TO True  
                    IF the result of can_partition with index set to i PLUS one, k AS the same, current_sum PLUS element at position i of nums, and used AS the same list EQUALS True THEN  
                        RETURN True  
                    END IF  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list OF boolean False repeated the length OF nums times  
        RETURN the result of can_partition with index set to zero, k AS the same, current_sum set to zero, and used AS the created list  
    END FUNCTION  
END CLASS
