CLASS Solution  
    FUNCTION findKthLargest(nums AS List of integers, k AS integer) RETURNS integer  
        SET min_heap TO an empty list  
        FOR each num IN nums  
            INVOKE heap insertion of num INTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN k THEN  
                INVOKE heap removal of the smallest element FROM min_heap  
            END IF  
        END FOR  
        RETURN the element at position zero of min_heap  
    END FUNCTION  
END CLASS
