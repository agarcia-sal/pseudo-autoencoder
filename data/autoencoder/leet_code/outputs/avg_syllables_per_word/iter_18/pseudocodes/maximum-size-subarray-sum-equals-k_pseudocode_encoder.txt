CLASS Solution  
    FUNCTION maxSubArrayLen(nums AS list of integers k AS integer) RETURNS integer  
        SET cumulative_sum_index TO mapping with key zero mapped TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each pair index and element num IN enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF difference between cumulative_sum AND k EXISTS as key IN cumulative_sum_index  
                SET max_length TO maximum BETWEEN max_length AND difference between index AND element at key difference between cumulative_sum AND k IN cumulative_sum_index  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST as key IN cumulative_sum_index  
                ASSIGN element at key cumulative_sum IN cumulative_sum_index TO index  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
