CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n IS GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        
        INCREMENT start BY the INTEGER DIVISION OF the DIFFERENCE BETWEEN n AND one BY length  
        
        SET number_string TO the STRING CONVERSION OF start  
        SET digit_position TO the REMAINDER OF the DIFFERENCE BETWEEN n AND one DIVIDED BY length  
        SET result_digit TO the CHARACTER AT POSITION digit_position OF number_string  
        
        RETURN the INTEGER CONVERSION OF result_digit  
    END FUNCTION  
END CLASS
