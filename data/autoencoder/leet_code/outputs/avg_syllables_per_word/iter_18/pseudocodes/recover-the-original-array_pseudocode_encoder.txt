CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL method sort ON nums  
        SET n TO the LENGTH OF nums  
        SET first TO element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET difference TO element at position i of nums MINUS first  
            SET k TO integer division of difference BY two  
            IF k IS LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k IS NOT EQUAL TO element at position i of nums  
                CONTINUE TO the next iteration  
            END IF  
            SET count TO result of CALLING Counter WITH nums  
            SET arr TO empty list  
            FOR each x IN nums  
                IF value at key x IN count IS EQUAL TO zero  
                    CONTINUE TO the next iteration  
                END IF  
                IF value at key x PLUS two MULTIPLIED BY k IN count IS EQUAL TO zero  
                    BREAK the loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT value at key x IN count BY one  
                DECREMENT value at key x PLUS two MULTIPLIED BY k IN count BY one  
            END FOR  
            IF the LENGTH OF arr IS EQUAL TO integer division of n BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN empty list  
    END FUNCTION  
END CLASS
