CLASS ListNode  
    FUNCTION __init__(val MINUS_EQUALS zero COMMA next MINUS_EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH val EQUALS element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one to the last  
        SET node TO new ListNode WITH val EQUALS val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 COMMA p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND CALL is_same_list WITH next of p1 COMMA next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(head COMMA x)  
        SET less_head TO new ListNode WITH val EQUALS zero  
        SET greater_head TO new ListNode WITH val EQUALS zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current HOLDS  
            IF the value at current LESS THAN x  
                SET next of less TO current  
                SET less TO next of less  
            ELSE  
                SET next of greater TO current  
                SET greater TO next of greater  
            END IF  
            SET current TO next of current  
        END WHILE  
        SET next of less TO next of greater_head  
        SET next of greater TO None  
        RETURN next of less_head  
    END FUNCTION  
END CLASS
