CLASS Solution  
    FUNCTION minSteps WITH parameter n  
        IF n EQUALS one THEN  
            RETURN zero  
        END IF  
        SET operations TO zero  
        SET factor TO two  
        WHILE n IS GREATER THAN one DO  
            WHILE the remainder of n DIVIDED BY factor EQUALS zero DO  
                INCREMENT operations BY factor  
                SET n TO the quotient of n DIVIDED BY factor  
            END WHILE  
            INCREMENT factor BY one  
        END WHILE  
        RETURN operations  
    END FUNCTION  
END CLASS
