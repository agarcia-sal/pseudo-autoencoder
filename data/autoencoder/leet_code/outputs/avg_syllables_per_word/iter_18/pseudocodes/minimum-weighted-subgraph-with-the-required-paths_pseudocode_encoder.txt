CLASS Solution  
    FUNCTION minimumWeight WITH PARAMETERS n edges src1 src2 dest  
        FUNCTION dijkstra WITH PARAMETERS graph start  
            SET distances TO list of size n each element SET TO infinity value  
            SET element at position start of distances TO zero  
            SET heap TO list containing a tuple of zero and start  
            
            WHILE heap is not empty  
                SET current_distance and current_node TO the result of extracting the smallest element from heap  
                
                IF current_distance GREATER THAN element at position current_node of distances  
                    CONTINUE  
                END IF  
                
                FOR each neighbor and weight IN the collection at position current_node of graph  
                    SET distance TO current_distance PLUS weight  
                    
                    IF distance LESS THAN element at position neighbor of distances  
                        SET element at position neighbor of distances TO distance  
                        ADD tuple of distance and neighbor TO heap  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN distances  
        END FUNCTION  
        
        SET graph TO list of size n each element SET TO an empty list  
        SET reverse_graph TO list of size n each element SET TO an empty list  
        
        FOR each u v w IN edges  
            ADD tuple of v and w TO list at position u of graph  
            ADD tuple of u and w TO list at position v of reverse_graph  
        END FOR  
        
        SET dist_from_src1 TO the result of dijkstra called with graph and src1  
        SET dist_from_src2 TO the result of dijkstra called with graph and src2  
        SET dist_to_dest TO the result of dijkstra called with reverse_graph and dest  
        
        SET min_weight TO infinity value  
        
        FOR each i FROM zero TO n MINUS one  
            IF element at position i of dist_from_src1 NOT EQUAL TO infinity value AND element at position i of dist_from_src2 NOT EQUAL TO infinity value AND element at position i of dist_to_dest NOT EQUAL TO infinity value  
                SET min_weight TO the smaller value between min_weight AND the sum of element at position i of dist_from_src1 PLUS element at position i of dist_from_src2 PLUS element at position i of dist_to_dest  
            END IF  
        END FOR  
        
        IF min_weight EQUALS infinity value  
            RETURN minus one  
        ELSE  
            RETURN min_weight  
        END IF  
    END FUNCTION  
END CLASS
