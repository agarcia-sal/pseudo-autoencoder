CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF nums IS empty  
            RETURN zero  
        END IF  
        SET max_product TO element at position zero of nums  
        SET min_product TO element at position zero of nums  
        SET result TO element at position zero of nums  
        FOR each num IN elements of nums starting from position one to the end  
            IF num IS less than zero  
                SET temporary_swap_variable TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary_swap_variable  
            END IF  
            IF num IS greater than max_product MULTIPLIED BY num  
                SET max_product TO num  
            ELSE  
                SET max_product TO max_product MULTIPLIED BY num  
            END IF  
            IF num IS less than min_product MULTIPLIED BY num  
                SET min_product TO num  
            ELSE  
                SET min_product TO min_product MULTIPLIED BY num  
            END IF  
            IF result IS greater than max_product  
                KEEP result AS IS  
            ELSE  
                SET result TO max_product  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
