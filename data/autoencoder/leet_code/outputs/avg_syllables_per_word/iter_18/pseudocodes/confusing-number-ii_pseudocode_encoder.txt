CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping of zero TO zero one TO one six TO nine eight TO eight nine TO six  
        SET valid_digits TO list of zero one six eight nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero holds  
                SET digit TO the remainder when num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value of digit in rotate_map  
                SET num TO the quotient when num DIVIDED BY ten  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit current IS zero)  
            IF current GREATER THAN limit holds  
                RETURN zero  
            END IF  
            SET count TO one IF current NOT EQUALS zero AND is_confusing(current) OTHERWISE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero holds  
                    CONTINUE  
                END IF  
                INCREMENT count BY the result of count_confusing_numbers(limit new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n)  
    END FUNCTION  
END CLASS
