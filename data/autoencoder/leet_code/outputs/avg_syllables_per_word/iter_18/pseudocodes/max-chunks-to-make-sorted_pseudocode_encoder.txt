CLASS Solution  
    FUNCTION maxChunksToSorted arr  
        SET max_so_far TO zero  
        SET chunks TO zero  
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            SET max_so_far TO the GREATER OF max_so_far AND the element at position index of arr  
            IF max_so_far EQUALS index THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
