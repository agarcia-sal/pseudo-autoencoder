CLASS Solution  
    FUNCTION compareVersion WITH PARAMETERS version1 AND version2  
        SET v1_revisions TO the LIST OF substrings of version1 obtained by SPLITTING version1 BY the character representing a period  
        SET v2_revisions TO the LIST OF substrings of version2 obtained by SPLITTING version2 BY the character representing a period  
        SET max_length TO the GREATER OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF the index IS LESS THAN the LENGTH OF v1_revisions THEN  
                SET v1 TO the INTEGER CONVERSION of the element at position index of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF the index IS LESS THAN the LENGTH OF v2_revisions THEN  
                SET v2 TO the INTEGER CONVERSION of the element at position index of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 IS LESS THAN v2 THEN  
                RETURN negative one  
            ELSE IF v1 IS GREATER THAN v2 THEN  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
