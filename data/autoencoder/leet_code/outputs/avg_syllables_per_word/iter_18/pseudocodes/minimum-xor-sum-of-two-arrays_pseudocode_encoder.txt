CLASS Solution  
    FUNCTION minimumXORSum WITH PARAMETERS nums1 LIST OF INTEGER nums2 LIST OF INTEGER  
        SET n TO the LENGTH OF nums1  
        SET dp TO a LIST OF size two RAISED TO THE POWER OF n WHERE EACH element IS assigned TO infinite  
        SET element at position zero of dp TO zero  
        
        FOR each mask FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the COUNT OF ONES IN THE BINARY REPRESENTATION OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF bit at position j IN mask IS SET  
                    SET candidate_value TO element at position mask XOR with two RAISED TO THE POWER OF j of dp PLUS the result of BITWISE XOR BETWEEN element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                    SET element at position mask of dp TO the MINIMUM OF element at position mask of dp AND candidate_value  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position two RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
