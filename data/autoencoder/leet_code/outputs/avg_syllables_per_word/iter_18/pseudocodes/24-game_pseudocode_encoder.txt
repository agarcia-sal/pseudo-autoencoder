CLASS Solution  
    FUNCTION judgePoint24 WITH parameter cards AS list of integers  
        FUNCTION solve WITH parameter nums AS list of integers  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty four IS LESS THAN one ten millionth  
            END IF  
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF index i NOT EQUALS index j  
                        SET new_nums TO an empty list  
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF index k NOT EQUALS index i AND index k NOT EQUALS index j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        FOR each operation op IN the collection of true division multiplication addition subtraction  
                            IF operation op IS addition OR operation op IS multiplication AND index j IS GREATER THAN index i  
                                CONTINUE to the next operation  
                            END IF  
                            IF operation op IS NOT true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying operation op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve WITH parameter new_nums RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN false  
        END FUNCTION  
        IMPORT from operator the functions truediv multiplication addition subtraction  
        RETURN solve WITH parameter cards  
    END FUNCTION  
END CLASS
