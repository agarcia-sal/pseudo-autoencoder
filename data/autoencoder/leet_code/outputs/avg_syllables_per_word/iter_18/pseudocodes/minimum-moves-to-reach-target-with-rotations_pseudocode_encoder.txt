CLASS Solution  
    FUNCTION minimumMoves(grid AS List OF List OF Integer) RETURNS Integer  
        SET n TO the LENGTH OF grid  
        SET queue TO a new double ended queue initialized with a tuple consisting of zero zero zero one zero  
        SET visited TO a new set initialized with a tuple consisting of zero zero zero one  
          
        WHILE queue IS NOT empty  
            SET r1 and c1 and r2 and c2 and steps TO the elements obtained by removing from the left of queue  
              
            IF the tuple consisting of r1 c1 r2 c2 EQUALS the tuple consisting of n MINUS one n MINUS two n MINUS one n MINUS one THEN  
                RETURN steps  
            END IF  
              
            IF c2 PLUS one LESS THAN n AND element at r1 position and element at c1 PLUS one of grid EQUALS zero AND element at r2 position and element at c2 PLUS one of grid EQUALS zero AND the tuple consisting of r1 c1 PLUS one r2 c2 PLUS one IS NOT IN visited THEN  
                ADD the tuple consisting of r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple consisting of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
              
            IF r2 PLUS one LESS THAN n AND element at r1 PLUS one position and element at c1 of grid EQUALS zero AND element at r2 PLUS one position and element at c2 of grid EQUALS zero AND the tuple consisting of r1 PLUS one c1 r2 PLUS one c2 IS NOT IN visited THEN  
                ADD the tuple consisting of r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple consisting of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
              
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at r1 PLUS one position and element at c1 of grid EQUALS zero AND element at r1 PLUS one position and element at c2 of grid EQUALS zero AND the tuple consisting of r1 c1 r1 PLUS one c1 IS NOT IN visited THEN  
                ADD the tuple consisting of r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple consisting of r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
              
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at r1 position and element at c1 PLUS one of grid EQUALS zero AND element at r2 position and element at c2 PLUS one of grid EQUALS zero AND the tuple consisting of r1 c1 r1 c1 PLUS one IS NOT IN visited THEN  
                ADD the tuple consisting of r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple consisting of r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
