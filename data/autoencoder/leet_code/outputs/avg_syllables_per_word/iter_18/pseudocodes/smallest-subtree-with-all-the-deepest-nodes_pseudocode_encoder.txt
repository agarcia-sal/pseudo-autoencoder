CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode with value element at position zero of values  
    SET index_counter TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the start of queue  
        IF index_counter LESS THAN LENGTH OF values AND element at position index_counter of values IS NOT none  
            SET node.left TO new instance of TreeNode with value element at position index_counter of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT index_counter BY one  
        IF index_counter LESS THAN LENGTH OF values AND element at position index_counter of values IS NOT none  
            SET node.right TO new instance of TreeNode with value element at position index_counter of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT index_counter BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUAL TO value of q  
        RETURN false  
    ELSE  
        RETURN result of is_same_tree with parameters value of left child of p AND value of left child of q AND result of is_same_tree with parameters value of right child of p AND value of right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO result of dfs with parameter value of left child of node  
            SET right_node AND right_depth TO result of dfs with parameter value of right child of node  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN first element of result of dfs with parameter root  
    END FUNCTION  
END CLASS
