CLASS Solution  
    FUNCTION champagneTower(poured INTEGER query_row INTEGER query_glass INTEGER) RETURNS FLOAT  
        SET current_row TO list containing poured  
        FOR each row FROM one TO query_row  
            SET next_row TO list of zeros of length row PLUS one  
            FOR each index position IN the range from zero TO the length of current_row MINUS one  
                IF element at position index position of current_row GREATER THAN one  
                    SET excess TO element at position index position of current_row MINUS one DIVIDED BY two  
                    INCREMENT element at position index position of next_row BY excess  
                    INCREMENT element at position index position PLUS one of next_row BY excess  
                END IF  
            END FOR  
            ASSIGN reference TO next_row TO current_row  
        END FOR  
        IF element at position query_glass of current_row LESS THAN one  
            RETURN element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
