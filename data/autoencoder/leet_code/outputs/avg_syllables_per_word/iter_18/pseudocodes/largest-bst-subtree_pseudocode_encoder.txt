CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value being element at position zero of values  
    SET i TO one  
    SET queue TO new deque collection  
    APPEND root TO queue  
    WHILE queue contains elements  
        SET node TO element removed from left end of queue  
        IF i less than the length of values AND element at position i of values is not none  
            SET node.left TO new TreeNode with value being element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i less than the length of values AND element at position i of values is not none  
            SET node.right TO new TreeNode with value being element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN the boolean result of is_same_tree with parameters being left child of p and left child of q AND is_same_tree with parameters being right child of p and right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
    
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO the result of calling dfs with left child of node  
            SET right_min, right_max, right_size TO the result of calling dfs with right child of node  
            IF left_max less than value of node AND value of node less than right_min  
                RETURN the minimum of value of node and left_min, the maximum of value of node and right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, the maximum of left_size and right_size  
        END FUNCTION  
    
        RETURN the third element of the result of calling dfs with root  
    END FUNCTION  
END CLASS
