CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO new list of batchSize elements each SET TO zero  
        FOR each group IN groups  
            INCREMENT element at position of the remainder of group DIVIDED BY batchSize in remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variable number of remainder_counts)  
            SET remainder_count TO new list made from remainder_counts  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF element at position index in remainder_count EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                
                DECREMENT element at position index in remainder_count BY one  
                SET new_remainder TO remainder MINUS index MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, variable elements of remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, variable elements of remainder_count)  
                END IF  
                
                SET max_happy TO the GREATER OF max_happy AND happy_groups  
                
                INCREMENT element at position index in remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, variable elements of remainder_count)  
END CLASS
