CLASS Solution  
    FUNCTION permuteUnique WITH PARAMETER nums  
        FUNCTION backtrack WITH PARAMETERS path used res  
            IF the LENGTH OF path EQUALS the LENGTH OF nums  
                APPEND a COPY of path TO res  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF nums MINUS one  
                IF element at position index of used IS True OR index IS GREATER THAN zero AND element at position index of nums EQUALS element at position index MINUS one of nums AND element at position index MINUS one of used IS False  
                    CONTINUE  
                END IF  
                SET element at position index of used TO True  
                APPEND element at position index of nums TO path  
                CALL backtrack WITH ARGUMENTS path used res  
                REMOVE the LAST element FROM path  
                SET element at position index of used TO False  
            END FOR  
        END FUNCTION  
        CALL the SORT method ON nums  
        SET res TO an EMPTY list  
        SET used TO a LIST of False elements OF the SAME LENGTH AS nums  
        CALL backtrack WITH ARGUMENTS an EMPTY list used res  
        RETURN res  
    END FUNCTION  
END CLASS
