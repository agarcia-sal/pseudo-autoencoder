CLASS Solution  
    FUNCTION spiralOrder WITH PARAMETER matrix  
        IF matrix is empty OR element at position zero of matrix is empty THEN  
            RETURN an empty list  
        END IF  
          
        SET result TO an empty list  
        SET top TO zero  
        SET bottom TO the length of matrix MINUS one  
        SET left TO zero  
        SET right TO the length of element at position zero of matrix MINUS one  
          
        WHILE top is less than or equal to bottom AND left is less than or equal to right DO  
            FOR index i FROM left TO right DO  
                APPEND element at position top of matrix then element at position i of that element TO result  
            END FOR  
            INCREMENT top BY one  
              
            FOR index i FROM top TO bottom DO  
                APPEND element at position i of matrix then element at position right of that element TO result  
            END FOR  
            DECREMENT right BY one  
              
            IF top is less than or equal to bottom THEN  
                FOR index i FROM right DOWN TO left DO  
                    APPEND element at position bottom of matrix then element at position i of that element TO result  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
              
            IF left is less than or equal to right THEN  
                FOR index i FROM bottom DOWN TO top DO  
                    APPEND element at position i of matrix then element at position left of that element TO result  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
          
        RETURN result  
    END FUNCTION  
END CLASS
