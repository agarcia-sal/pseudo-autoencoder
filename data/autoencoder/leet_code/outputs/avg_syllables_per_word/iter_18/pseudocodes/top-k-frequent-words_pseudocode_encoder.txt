CLASS Solution  
    FUNCTION topKFrequent(words AS list of strings comma k AS integer) RETURNS list of strings  
        SET count TO the result of invoking the Counter on words  
        SET heap TO an empty list  
        FOR each word and frequency IN the collection count  
            APPEND the tuple consisting of the negation of frequency and word TO heap  
        END FOR  
        INVOKE the heapify operation on heap  
        SET result TO an empty list  
        FOR iteration_index FROM one TO k  
            SET extracted_element TO the result of popping the smallest element from heap  
            APPEND the element at position two of extracted_element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
