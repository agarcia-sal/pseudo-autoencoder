CLASS Solution  
    FUNCTION predictPartyVictory WITH PARAMETER senate  
        INITIALIZE radiant AS empty double ended queue  
        INITIALIZE dire AS empty double ended queue  
        
        FOR each position and character IN senate  
            IF character EQUALS capital letter R  
                APPEND position TO radiant  
            ELSE  
                APPEND position TO dire  
            END IF  
        END FOR  
        
        WHILE radiant is not empty AND dire is not empty  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant is not empty  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
