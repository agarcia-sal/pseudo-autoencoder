CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters of time with the character colon REMOVED  
        SET hour AND minute TO the two integer values obtained from splitting time at the character colon  
    
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each digit representing h1 IN digits  
                FOR each digit representing h2 IN digits  
                    FOR each digit representing m1 IN digits  
                        FOR each digit representing m2 IN digits  
                            SET combined_hour TO the integer value of concatenating h1 and h2 in that order  
                            SET combined_minute TO the integer value of concatenating m1 and m2 in that order  
                            IF combined_hour is less than twenty four AND combined_minute is less than sixty  
                                APPEND the pair consisting of combined_hour AND combined_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in ascending order  
        END FUNCTION  
    
        SET all_times TO the result of calling generate_times with digits as argument  
        SET current_time_index TO the position of the pair consisting of hour AND minute in all_times  
    
        IF current_time_index plus one is less than the length of all_times  
            SET next_hour AND next_minute TO the pair at position current_time_index plus one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the pair at position zero in all_times  
        END IF  
    
        RETURN the string formed by formatting next_hour as two digits followed by the character colon followed by next_minute as two digits  
    END FUNCTION  
END CLASS
