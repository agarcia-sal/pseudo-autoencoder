CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val defaulted TO zero left defaulted TO none right defaulted TO none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty list  
        RETURN none  
    END IF  
    SET root TO new TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO new deque initialized empty  
    CALL append WITH root ON queue  
    WHILE the queue IS NOT empty  
        SET node TO the element obtained by removing from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node left TO new TreeNode INITIALIZED WITH element at position i of values  
            CALL append WITH node left ON queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node right TO new TreeNode INITIALIZED WITH element at position i of values  
            CALL append WITH node right ON queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUAL TO the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree CALLED WITH p left AND q left AND the RESULT OF is_same_tree CALLED WITH p right AND q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom WITH PARAMETERS self root OF OPTIONAL TreeNode RETURNING LIST OF LIST OF integer  
        IF root IS none  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new deque INITIALIZED WITH root AS the only element  
        WHILE the queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO empty list  
            FOR each element IN the range FROM zero TO level_size MINUS one  
                SET node TO the element obtained by removing from the left end of queue  
                CALL append WITH the val attribute of node ON current_level  
                IF the left attribute of node EXISTS AND IS NOT none  
                    CALL append WITH the left attribute of node ON queue  
                END IF  
                IF the right attribute of node EXISTS AND IS NOT none  
                    CALL append WITH the right attribute of node ON queue  
                END IF  
            END FOR  
            CALL append WITH current_level ON result  
        END WHILE  
        SET reversed_result TO the elements of result IN reverse order  
        RETURN reversed_result  
    END FUNCTION  
END CLASS
