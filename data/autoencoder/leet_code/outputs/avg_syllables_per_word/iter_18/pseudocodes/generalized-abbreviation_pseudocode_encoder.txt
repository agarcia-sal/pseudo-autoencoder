CLASS Solution  
    FUNCTION generateAbbreviations WITH PARAMETERS word  
        FUNCTION backtrack WITH PARAMETERS index path count  
            IF index EQUALS the LENGTH OF word THEN  
                IF count GREATER THAN zero THEN  
                    APPEND the TEXTUAL REPRESENTATION OF count TO path  
                END IF  
                APPEND the concatenation of all elements of path TO result  
            ELSE  
                IF count GREATER THAN zero THEN  
                    CALL backtrack WITH PARAMETERS index INCREMENTED BY one path CONCATENATED WITH the TEXTUAL REPRESENTATION OF count AND the CHARACTER at position index OF word zero  
                ELSE  
                    CALL backtrack WITH PARAMETERS index INCREMENTED BY one path CONCATENATED WITH the CHARACTER at position index OF word zero zero  
                END IF  
                CALL backtrack WITH PARAMETERS index INCREMENTED BY one path count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty collection  
        CALL backtrack WITH PARAMETERS zero an empty collection zero  
        RETURN result  
    END FUNCTION  
END CLASS
