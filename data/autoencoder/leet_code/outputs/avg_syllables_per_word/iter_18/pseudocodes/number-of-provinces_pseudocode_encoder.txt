CLASS Solution  
    FUNCTION findCircleNum WITH PARAMETER isConnected AS List OF List OF integer  
        FUNCTION dfs WITH PARAMETER city  
            FOR neighbor FROM zero TO n MINUS one  
                IF element at position neighbor of element at position city of isConnected EQUALS one AND NOT element at position neighbor of visited  
                    SET element at position neighbor of visited TO True  
                    CALL dfs WITH PARAMETER neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO the LENGTH OF isConnected  
        SET visited TO a list containing n repetitions of False  
        SET province_count TO zero  

        FOR city FROM zero TO n MINUS one  
            IF NOT element at position city of visited  
                CALL dfs WITH PARAMETER city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  

        RETURN province_count  
    END FUNCTION  
END CLASS
