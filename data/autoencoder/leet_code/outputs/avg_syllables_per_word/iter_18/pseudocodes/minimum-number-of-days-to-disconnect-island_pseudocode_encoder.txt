CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the pair x and y EXISTS IN visited OR element at position x and position y of grid EQUALS zero THEN  
                RETURN  
            END IF  
            APPEND the pair x and y TO visited  
            FOR each pair dx and dy IN the list of pairs consisting of minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one  
                CALL dfs WITH parameters the sum of x and dx the sum of y and dy visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO the list of all pairs i and j WHERE i ranges FROM zero TO the LENGTH OF grid MINUS one AND j ranges FROM zero TO the LENGTH OF element at position zero of grid MINUS one AND element at position i and position j of grid EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero THEN  
                RETURN True  
            END IF  
            CALL dfs WITH parameters first element at position zero of land_cells first element at position one of land_cells visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT the result of calling is_connected() THEN  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i and position j of grid EQUALS one THEN  
                    SET element at position i and position j of grid TO zero  
                    IF NOT the result of calling is_connected() THEN  
                        RETURN one  
                    END IF  
                    SET element at position i and position j of grid TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
