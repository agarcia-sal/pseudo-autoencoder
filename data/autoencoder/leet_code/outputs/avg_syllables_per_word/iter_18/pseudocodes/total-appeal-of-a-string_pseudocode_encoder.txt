CLASS Solution  
    FUNCTION appealSum WITH parameter s AS string  
        SET last TO a list of negative one repeated twenty six times  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each index and character IN the enumeration of s  
            SET character_index TO the numerical ordinal of character MINUS the numerical ordinal of the lowercase letter a  
            INCREMENT current_appeal BY index MINUS the element at position character_index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position character_index of last TO index  
        END FOR  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
