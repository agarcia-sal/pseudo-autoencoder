CLASS Solution  
    FUNCTION kSimilarity WITH parameters s1 and s2  
        FUNCTION neighbors WITH parameter s  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF s AND character at position i of s EQUALS character at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF character at position j of s EQUALS character at position i of s2 AND character at position j of s IS NOT EQUAL TO character at position j of s2  
                    YIELD concatenation of substring of s from start TO position i MINUS one PLUS character at position j of s PLUS substring of s from position i PLUS one TO position j MINUS one PLUS character at position i of s PLUS substring of s from position j PLUS one TO end  
                END IF  
            END FOR  
        END FUNCTION  
        ASSIGN reference TO double ended queue initialized with tuple of s1 and zero  
        ASSIGN reference TO set initialized with s1  
        WHILE the double ended queue IS NOT empty  
            SET s and swaps TO the element removed from the front of the double ended queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors WITH argument s  
                IF neighbor IS NOT IN the visited set  
                    ADD neighbor TO the visited set  
                    APPEND the tuple of neighbor and swaps PLUS one TO the end of the double ended queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
