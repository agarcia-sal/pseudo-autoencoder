CLASS Solution  
    FUNCTION minCost WITH PARAMETER grid OF TYPE List OF List OF integer  
        SET row_count TO the LENGTH OF grid  
        SET column_count TO the LENGTH OF element at position zero of grid  
        SET directions TO a list OF tuples each containing integer values representing zero and one zero and negative one one and zero negative one and zero  
        CALL auxiliary FUNCTION generate_initial_costs WITH PARAMETERS row_count and column_count AND SET costs TO its RETURN VALUE  
        SET element at position zero of row zero of costs TO zero  
        SET priority_queue TO a list CONTAINING a tuple OF zero zero zero  
        SET visited_positions TO an empty set  
        
        WHILE the priority_queue is not empty  
            SET current_cost and current_row and current_column TO the result of extracting the minimum cost tuple from priority_queue USING auxiliary FUNCTION extract_minimum_from_heap  
            
            IF the tuple containing current_row and current_column IS IN visited_positions  
                CONTINUE to next iteration of WHILE  
            END IF  
            ADD the tuple containing current_row and current_column TO visited_positions  
            
            IF current_row EQUALS row_count MINUS one AND current_column EQUALS column_count MINUS one  
                RETURN current_cost  
            END IF  
            
            FOR each direction_index AND direction_tuple IN directions WITH zero-based indexing  
                SET next_row TO current_row PLUS the first element of direction_tuple  
                SET next_column TO current_column PLUS the second element of direction_tuple  
                
                IF next_row IS GREATER THAN OR EQUAL TO zero AND next_row IS LESS THAN row_count AND next_column IS GREATER THAN OR EQUAL TO zero AND next_column IS LESS THAN column_count  
                    IF direction_index PLUS one IS NOT EQUAL TO element at position current_column of element at position current_row of grid  
                        SET additional_cost TO one  
                    ELSE  
                        SET additional_cost TO zero  
                    END IF  
                    SET proposed_cost TO current_cost PLUS additional_cost  
                    
                    IF proposed_cost IS LESS THAN element at position next_column of element at position next_row of costs  
                        SET element at position next_column of element at position next_row of costs TO proposed_cost  
                        CALL auxiliary FUNCTION push_to_heap WITH PARAMETERS priority_queue AND tuple OF proposed_cost next_row next_column  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION
