CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s IS EQUAL TO s REVERSED  
        END FUNCTION  

        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num IS EQUAL TO zero  
                RETURN single character representing zero  
            END IF  
            SET digits TO empty list  
            WHILE num IS GREATER THAN zero  
                APPEND remainder of num DIVIDED BY k AS integer TO digits  
                SET num TO num DIVIDED BY k FLOOR VALUE  
            END WHILE  
            RETURN concatenation of digits IN REVERSE ORDER AS string  
        END FUNCTION  

        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length IS EQUAL TO one  
                FOR each integer i FROM one TO nine INCLUSIVE  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF integer division of length MINUS one BY two  
                SET end TO ten RAISED TO THE POWER OF integer division of length PLUS one BY two  
                FOR each integer half FROM start TO end MINUS one  
                    SET half_str TO string representation OF half  
                    IF length MODULO two IS EQUAL TO zero  
                        SET full TO concatenation OF half_str AND half_str REVERSED  
                    ELSE  
                        SET full TO concatenation OF half_str AND half_str WITHOUT last character REVERSED  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  

        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes WITH PARAMETER length  
                SET num TO integer conversion OF pal  
                IF is_palindrome WITH PARAMETER to_base_k WITH PARAMETERS num AND k HOLDS TRUE  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers IS EQUAL TO n  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  

        RETURN sum OF k_mirror_numbers  
    END FUNCTION  
END CLASS
