CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        CALL auxiliary_initialize_dp_matrix WITH n AND ASSIGN TO dp  
        
        FOR each index i FROM zero TO n MINUS one  
            SET element at position i and position i of dp TO one  
        END FOR  
        
        FOR each index i FROM zero TO n MINUS two  
            IF element at position i of arr EQUALS element at position i PLUS one of arr THEN  
                SET element at position i and position i PLUS one of dp TO one  
            ELSE  
                SET element at position i and position i PLUS one of dp TO two  
            END IF  
        END FOR  
        
        FOR each length FROM three TO n  
            FOR each index i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of arr EQUALS element at position j of arr THEN  
                    SET element at position i and position j of dp TO element at position i PLUS one and position j MINUS one of dp  
                ELSE  
                    SET element at position i and position j of dp TO infinity  
                    FOR each k FROM i TO j MINUS one  
                        SET element at position i and position j of dp TO the MINIMUM of element at position i and position j of dp AND the SUM of element at position i and position k of dp PLUS element at position k PLUS one and position j of dp  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero and position n MINUS one of dp  
    END FUNCTION  

    FUNCTION auxiliary_initialize_dp_matrix(n)  
        DESCRIBE construction of a two-dimensional list filled with zeros having dimensions n BY n  
    END FUNCTION  
END CLASS
