CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums AS List of integers) RETURNS integer  
        SET count TO the frequency mapping of elements IN nums  
        SET dp TO a list of zeroes WITH the length equivalent to two raised to the power of ten  
        SET element at position zero of dp TO one  
          
        FOR each num IN the keys of count  
            IF num EQUALS one THEN  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two raised to the power of the value at key one in count  
                CONTINUE to next iteration  
            END IF  
            IF num NOT IN the mapping of prime factors THEN  
                CONTINUE to next iteration  
            END IF  
              
            SET factors TO the list of prime factors ASSOCIATED WITH num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO the bitwise inclusive OR of mask AND two raised to the power of prime MINUS one  
            END FOR  
              
            FOR m FROM the largest integer less than two raised to the power of ten MINUS one DOWN TO zero STEP negative one  
                IF the bitwise AND of m AND mask EQUALS zero THEN  
                    SET element at position the bitwise inclusive OR of m AND mask of dp TO the sum of element at position the bitwise inclusive OR of m AND mask of dp AND the product of element at position m of dp AND the value at key num in count  
                    APPLY modulo operation with MOD TO the new value BEFORE ASSIGNING  
                END IF  
            END FOR  
        END FOR  
          
        SET result TO the summation of all elements of dp from position one TO the end  
        APPLY modulo operation with MOD TO result  
        RETURN result  
    END FUNCTION  
END CLASS
