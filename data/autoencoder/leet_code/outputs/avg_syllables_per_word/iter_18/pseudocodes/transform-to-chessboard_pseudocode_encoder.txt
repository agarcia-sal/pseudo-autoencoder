CLASS Solution  
    FUNCTION movesToChessboard(board WHICH IS A LIST OF LIST OF INTEGER) RETURNS INTEGER  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
  
        FOR each index i FROM zero TO N MINUS one  
            FOR each index j FROM zero TO N MINUS one  
                IF the EXCLUSIVE OR of element at position i of board at position j AND element at position i of board at position zero AND element at position zero of board at position j AND element at position zero of board at position zero HOLDS  
                    RETURN the NEGATIVE integer one  
                END IF  
            END FOR  
  
            INCREMENT rowSum BY element at position i of board at position zero  
            INCREMENT colSum BY element at position zero of board at position i  
            INCREMENT rowSwap BY whether element at position i of board at position zero EQUALS the remainder of i DIVIDED BY two  
            INCREMENT colSwap BY whether element at position zero of board at position i EQUALS the remainder of i DIVIDED BY two  
        END FOR  
  
        IF the INTEGER DIVISION of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the INTEGER DIVISION of the SUM of N AND one BY two  
            RETURN the NEGATIVE integer one  
        END IF  
  
        IF the INTEGER DIVISION of N BY two GREATER THAN colSum OR colSum GREATER THAN the INTEGER DIVISION of the SUM of N AND one BY two  
            RETURN the NEGATIVE integer one  
        END IF  
  
        IF the REMAINDER of N DIVIDED BY two IS NOT EQUAL TO zero  
            IF the REMAINDER of rowSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the REMAINDER of colSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
  
        SET totalSwaps TO the INTEGER DIVISION of the SUM of rowSwap AND colSwap BY two  
        RETURN totalSwaps  
    END FUNCTION  
END CLASS
