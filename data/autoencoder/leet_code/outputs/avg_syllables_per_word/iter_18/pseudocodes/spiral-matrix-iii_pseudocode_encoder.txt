CLASS Solution  
    FUNCTION spiralMatrixIII(rows parameter, cols parameter, rStart parameter, cStart parameter) RETURNS list of list of integer  
        SET directions TO list of tuples representing zero PLUS one for east one PLUS zero for south zero MINUS one for west MINUS one PLUS zero for north  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
          
        WHILE the LENGTH OF visited is LESS THAN rows MULTIPLIED BY cols holds  
            IF increment MODULO two EQUALS zero THEN  
                INCREMENT steps BY one  
            END IF  
            FOR each repetition FROM one TO steps  
                IF x is GREATER THAN OR EQUAL TO zero AND x is LESS THAN rows AND y is GREATER THAN OR EQUAL TO zero AND y is LESS THAN cols THEN  
                    APPEND list consisting of x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions at position zero  
                SET dy TO element at position direction_index of directions at position one  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
          
        RETURN visited  
    END FUNCTION  
END CLASS
