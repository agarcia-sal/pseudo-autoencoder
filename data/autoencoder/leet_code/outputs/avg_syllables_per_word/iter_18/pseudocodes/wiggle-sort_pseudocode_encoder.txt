CLASS Solution  
    FUNCTION wiggleSort(nums AS List OF integers)  
        SET less TO true indicating expectation that element at position zero of nums is less than or equal to element at position one of nums  
        FOR index FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less holds true  
                IF element at position index of nums IS GREATER THAN element at position index plus one of nums  
                    SET temporary variable TO element at position index of nums  
                    SET element at position index of nums TO element at position index plus one of nums  
                    SET element at position index plus one of nums TO temporary variable  
                END IF  
            ELSE  
                IF element at position index of nums IS LESS THAN element at position index plus one of nums  
                    SET temporary variable TO element at position index of nums  
                    SET element at position index of nums TO element at position index plus one of nums  
                    SET element at position index plus one of nums TO temporary variable  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
