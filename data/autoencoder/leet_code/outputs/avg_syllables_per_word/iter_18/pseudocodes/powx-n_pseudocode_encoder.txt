CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x AS floating point number AND n AS integer number  
        FUNCTION qpow WITH PARAMETERS a AS floating point number AND n AS integer number  
            SET ans TO one  
            WHILE n GREATER THAN zero HOLDS  
                IF the least significant bit of n IS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the integer division of n BY two discarding remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN the result of qpow CALLED WITH x AND n  
        ELSE  
            RETURN one DIVIDED BY the result of qpow CALLED WITH x AND the negation of n  
        END IF  
    END FUNCTION  
END CLASS
