CLASS Solution  
    FUNCTION findMaximumXOR(nums as List OF integer) RETURNS integer  
        SET max_xor TO zero  
        FOR bit_position FROM thirty one TO zero BY negative one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO empty set  
            FOR each num IN nums  
                SET prefix TO integer division of num BY two RAISED TO THE POWER OF bit_position  
                APPEND prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the bitwise exclusive OR of current_candidate and prefix IS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
