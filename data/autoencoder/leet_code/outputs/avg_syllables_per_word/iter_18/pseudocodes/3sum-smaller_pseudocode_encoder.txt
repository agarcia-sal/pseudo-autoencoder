CLASS Solution  
    FUNCTION threeSumSmaller WITH PARAMETERS nums AS list of integer AND target AS integer RETURNS integer  
        EXECUTE sorting of list nums in nondecreasing order  
        SET count TO zero  
        SET n TO the LENGTH OF list nums  
        FOR index i FROM zero TO n MINUS two MINUS one  
            SET left_pointer TO i PLUS one  
            SET right_pointer TO n MINUS one  
            WHILE the value at left_pointer IS LESS THAN the value at right_pointer  
                SET current_sum TO the element at position i of list nums PLUS the element at position left_pointer of list nums PLUS the element at position right_pointer of list nums  
                IF current_sum IS LESS THAN target THEN  
                    INCREMENT count BY right_pointer MINUS left_pointer  
                    INCREMENT left_pointer BY one  
                ELSE  
                    DECREMENT right_pointer BY one  
                END IF  
            END WHILE  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
