CLASS Solution  
    FUNCTION merge(intervals AS List OF List OF integer) RETURNS List OF List OF integer  
        CALL auxiliary_sort_intervals_based_on_starting_point WITH intervals  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the ending element at the last position of merged IS LESS THAN the starting element at the current interval THEN  
                APPEND interval TO merged  
            ELSE  
                SET the ending element at the last position of merged TO the maximum BETWEEN the ending element at the last position of merged AND the ending element at interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
