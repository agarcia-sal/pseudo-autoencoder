CLASS Solution  
    FUNCTION findRadius WITH parameters houses AND heaters  
        CALL sort METHOD ON houses  
        CALL sort METHOD ON heaters  
        SET min_radius TO zero  
        SET pointer_j TO zero  
        FOR each house IN houses  
            WHILE pointer_j LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF the distance BETWEEN element at position pointer_j of heaters AND house GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF the distance BETWEEN element at position pointer_j PLUS one of heaters AND house  
                INCREMENT pointer_j BY one  
            END WHILE  
            SET min_radius TO the GREATER VALUE OF min_radius AND the ABSOLUTE VALUE OF the distance BETWEEN element at position pointer_j of heaters AND house  
        END FOR  
        RETURN min_radius  
    END FUNCTION  
END CLASS
