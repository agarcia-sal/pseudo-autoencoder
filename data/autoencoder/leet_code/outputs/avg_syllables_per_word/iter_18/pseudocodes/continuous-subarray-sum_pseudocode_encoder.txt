CLASS Solution  
    FUNCTION checkSubarraySum WITH PARAMETERS nums WITH TYPE List of integers AND k WITH TYPE integer RETURNS boolean  
        SET remainder_map TO mapping with key zero and value negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN enumeration of nums  
            INCREMENT current_sum BY num  
            IF k is NOT equal to zero THEN  
                SET current_sum TO current_sum modulo k  
            END IF  
            IF current_sum is a key in remainder_map THEN  
                IF the difference between i and the value in remainder_map at key current_sum is greater than one THEN  
                    RETURN true  
                END IF  
            ELSE  
                SET the value in remainder_map at key current_sum TO i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
