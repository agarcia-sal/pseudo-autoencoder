CLASS Solution  
    FUNCTION minimumEffort WITH PARAMETER tasks AS list of list of integers  
        CALL tasks SORT WITH KEY AS FUNCTION element RETURNING element at position one MINUS element at position zero IN DESCENDING ORDER  
        SET current_energy TO zero  
        SET required_initial_energy TO zero  
        FOR each pair actual AND minimum IN tasks  
            IF current_energy LESS THAN minimum THEN  
                INCREMENT required_initial_energy BY minimum MINUS current_energy  
                SET current_energy TO minimum  
            END IF  
            DECREMENT current_energy BY actual  
        END FOR  
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
