CLASS Solution  
    FUNCTION minFlips WITH PARAMETERS mat AS List of List of integer  
        FUNCTION flip WITH PARAMETERS matrix AS matrix of integers x AS integer y AS integer  
            SET directions TO list containing the pairs of integers negative one and zero one and zero negative one zero and zero one and zero and zero  
            FOR each pair dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n THEN  
                    SET element at position nx of matrix element at position ny TO one MINUS element at position nx of matrix element at position ny  
                END IF  
            END FOR  
        END FUNCTION  
      
        FUNCTION matrix_to_tuple WITH PARAMETER matrix AS matrix of integers  
            RETURN tuple of tuples where each tuple corresponds to each row of matrix  
        END FUNCTION  
      
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET target TO tuple of tuples where each tuple consists of zeros for each index from zero TO n MINUS one and there are m such tuples  
        SET initial_state TO matrix_to_tuple WITH ARGUMENT mat  
        SET queue TO list containing the tuple initial_state and zero  
        SET visited TO an empty set  
        ADD initial_state TO visited  
      
        WHILE queue IS NOT empty  
            SET current_state and steps TO the first element in queue AND REMOVE the first element from queue  
            IF current_state EQUALS target THEN  
                RETURN steps  
            END IF  
      
            SET current_matrix TO list of lists where each list corresponds to each tuple in current_state  
      
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy of current_matrix  
                    CALL flip WITH ARGUMENTS next_matrix i j  
                    SET next_state TO matrix_to_tuple WITH ARGUMENT next_matrix  
                    IF next_state NOT IN visited THEN  
                        ADD next_state TO visited  
                        APPEND tuple of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
      
        RETURN negative one  
    END FUNCTION  
END CLASS
