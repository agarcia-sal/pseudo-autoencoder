CLASS Solution  
    FUNCTION minInsertions WITH PARAMETER s  
        FUNCTION longest_palindromic_subsequence WITH PARAMETER s  
            SET n TO the LENGTH OF s  
            CREATE dp AS a two dimensional array of size n BY n FILLED WITH zero  
            
            FOR each i FROM zero TO n MINUS one  
                SET element at position i at position i of dp TO one  
            END FOR  
            
            FOR each cl FROM two TO n  
                FOR each i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF character at position i of s EQUALS character at position j of s AND cl EQUALS two THEN  
                        SET element at position i at position j of dp TO two  
                    ELSE IF character at position i of s EQUALS character at position j of s THEN  
                        SET element at position i at position j of dp TO element at position i PLUS one at position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i at position j of dp TO the GREATER VALUE BETWEEN element at position i at position j MINUS one of dp AND element at position i PLUS one at position j of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero at position n MINUS one of dp  
        END FUNCTION  
        
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence WITH PARAMETER s  
    END FUNCTION  
END CLASS
