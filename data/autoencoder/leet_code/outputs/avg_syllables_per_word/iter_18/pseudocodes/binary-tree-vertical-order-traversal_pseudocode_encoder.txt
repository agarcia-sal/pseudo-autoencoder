CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode at position zero element of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode at position i element of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode at position i element of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree left child of p AND left child of q AND is_same_tree right child of p AND right child of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  
        SET column_table TO new default dictionary with list as default type  
        SET queue TO new deque containing tuple of root and zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            SET node and column_index TO element removed from the front of queue  
            IF node IS NOT none  
                APPEND value of node TO list in column_table at key column_index  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND tuple of left child of node and column_index MINUS one TO queue  
                APPEND tuple of right child of node and column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result_list TO empty list  
        FOR column FROM min_column TO max_column  
            APPEND list in column_table at key column TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
