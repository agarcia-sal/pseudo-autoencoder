CLASS Solution  
    FUNCTION reverse(x)  
        IF x IS LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO positive one  
        END IF  
        SET absolute_value_string TO the string representation OF the absolute value OF x  
        SET reversed_string TO the reverse OF the characters IN absolute_value_string  
        SET reversed_x TO the integer conversion OF reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result IS LESS THAN negative two TO THE POWER OF thirty one OR result IS GREATER THAN positive two TO THE POWER OF thirty one MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
