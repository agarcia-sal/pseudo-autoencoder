CLASS Solution  
    FUNCTION isNStraightHand(hand AS List OF integers groupSize AS integer) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
  
        SET count TO a mapping FROM elements to their frequencies IN hand USING auxiliary FUNCTION Counter  
        SET unique_cards TO the SORTED list OF the KEYS IN count  
  
        FOR each card IN unique_cards  
            IF the VALUE ASSOCIATED WITH key card IN count IS GREATER THAN zero THEN  
                SET needed TO the VALUE ASSOCIATED WITH key card IN count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the VALUE ASSOCIATED WITH key card PLUS index IN count IS LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the VALUE ASSOCIATED WITH key card PLUS index IN count BY needed  
                END FOR  
            END IF  
        END FOR  
  
        RETURN True  
    END FUNCTION  
END CLASS
