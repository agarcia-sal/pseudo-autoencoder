CLASS Solution  
    FUNCTION nextGreaterElements(nums AS List OF Integer) RETURNS List OF Integer  
        SET n TO the LENGTH OF nums  
        SET result TO a list containing the value negative one repeated n times  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE the stack IS NOT empty AND the element at position element at the last position of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_index TO the value REMOVED from the last position of stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
