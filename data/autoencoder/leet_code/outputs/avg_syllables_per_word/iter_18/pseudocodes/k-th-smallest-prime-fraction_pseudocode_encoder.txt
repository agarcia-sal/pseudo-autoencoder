CLASS Solution  
    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        
        FOR index i FROM zero TO the LENGTH OF arr MINUS one MINUS one  
            CALL the OPERATION to INSERT INTO min_heap the TRIPLE consisting of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr AND index i AND index n MINUS one  
        END FOR  
        
        FOR each iteration FROM one TO k MINUS one  
            ASSIGN reference TO the TRIPLE obtained by REMOVING the SMALLEST ELEMENT FROM min_heap  
            DECOMPOSE the TRIPLE INTO UNUSED value AND indices i AND j  
            IF the DIFFERENCE between j AND one GREATER THAN i  
                CALL the OPERATION to INSERT INTO min_heap the TRIPLE consisting of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr AND index i AND index j MINUS one  
            END IF  
        END FOR  
        
        ASSIGN reference TO the TRIPLE obtained by REMOVING the SMALLEST ELEMENT FROM min_heap  
        DECOMPOSE the TRIPLE INTO UNUSED value AND indices i AND j  
        RETURN the two-element list containing the element at position i of arr AND the element at position j of arr  
    END FUNCTION  
END CLASS
