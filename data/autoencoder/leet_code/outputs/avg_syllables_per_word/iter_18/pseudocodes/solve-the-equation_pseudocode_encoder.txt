CLASS Solution  
    FUNCTION solveEquation WITH PARAMETER equation  
        FUNCTION parse_side WITH PARAMETER side  
            SET tokens TO the RESULT OF replacing plus SIGN with hash PLUS SIGN IN side AND replacing minus SIGN with hash MINUS SIGN IN side AND splitting the RESULT on hash SIGN  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS EMPTY STRING THEN  
                    CONTINUE  
                END IF  
                IF token CONTAINS the character x THEN  
                    IF token EQUALS the character x OR token EQUALS plus SIGN FOLLOWED BY the character x THEN  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS minus SIGN FOLLOWED BY the character x THEN  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the INTEGER VALUE OF the RESULT OF removing the character x FROM token  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the INTEGER VALUE OF token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  
  
        SET left_side AND right_side TO the RESULT OF splitting the equation ON the EQUALS SIGN  
        SET left_x AND left_num TO the RESULT OF calling parse_side WITH left_side  
        SET right_x AND right_num TO the RESULT OF calling parse_side WITH right_side  
  
        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  
  
        IF total_x EQUALS zero THEN  
            IF total_num EQUALS zero THEN  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string x EQUALS total_num INTEGER DIVIDED BY total_x  
        END IF  
    END FUNCTION  
END CLASS
