CLASS Solution  
    FUNCTION removeDuplicates WITH PARAMETER nums AS list of integers  
        IF nums is empty  
            RETURN zero  
        END IF  
        SET write_index TO zero  
        FOR each index i FROM zero TO the lesser OF two AND the LENGTH OF nums MINUS one  
            SET element at position write_index of nums TO element at position i of nums  
            INCREMENT write_index BY one  
        END FOR  
        FOR each index i FROM two TO the LENGTH OF nums MINUS one  
            IF element at position i of nums is NOT EQUAL TO element at position write_index MINUS two of nums  
                SET element at position write_index of nums TO element at position i of nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        RETURN write_index  
    END FUNCTION  
END CLASS
