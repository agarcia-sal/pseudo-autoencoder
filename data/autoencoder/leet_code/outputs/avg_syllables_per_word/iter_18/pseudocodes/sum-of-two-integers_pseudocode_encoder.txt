CLASS Solution  
    FUNCTION getSum WITH PARAMETERS a b  
        SET MAX TO the integer with thirty one bits all set to one  
        SET mask TO the integer with thirty two bits all set to one  
        WHILE b NOT EQUALS zero holds  
            SET carry TO the bitwise AND of a AND b shifted left by one position  
            SET a TO the bitwise XOR of a AND b then bitwise AND with mask  
            SET b TO the bitwise AND of carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX THEN  
            RETURN a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
