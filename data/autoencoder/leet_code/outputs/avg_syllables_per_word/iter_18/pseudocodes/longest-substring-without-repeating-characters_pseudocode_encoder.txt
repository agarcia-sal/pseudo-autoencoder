CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH parameter s  
        SET char_index_map TO empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        FOR each position end and character char IN enumeration of s  
            IF char is a key in char_index_map AND the value at char in char_index_map is greater than or equal to start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            SET max_length TO the greater of max_length AND end MINUS start PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
