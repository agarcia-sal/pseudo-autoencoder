CLASS Solution  
    FUNCTION minimumDeviation WITH PARAMETER nums AS List OF Integer  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF num MODULO two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            PERFORM auxiliary heap push of negative num INTO max_heap  
        END FOR  
        SET min_deviation TO positive infinity  
        WHILE element at position zero of max_heap MODULO two EQUALS zero  
            SET max_value TO NEGATION OF auxiliary heap pop from max_heap  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            PERFORM auxiliary heap push of negative max_value INTO max_heap  
        END WHILE  
        IF min_deviation GREATER THAN NEGATION OF element at position zero of max_heap MINUS min_value  
            SET min_deviation TO NEGATION OF element at position zero of max_heap MINUS min_value  
        END IF  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
