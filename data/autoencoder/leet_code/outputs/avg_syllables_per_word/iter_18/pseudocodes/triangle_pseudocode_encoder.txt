CLASS Solution  
    FUNCTION minimumTotal(triangle AS List OF List OF integer) RETURNS integer  
        FOR row FROM the LENGTH OF triangle MINUS two TO zero STEP minus one  
            FOR col FROM zero TO the LENGTH OF element at position row of triangle MINUS one  
                SET value_below_left TO element at position col of element at position row PLUS one of triangle  
                SET value_below_right TO element at position col PLUS one of element at position row PLUS one of triangle  
                SET minimum_adjacent TO value_below_left IF value_below_left LESS THAN OR EQUAL TO value_below_right OTHERWISE value_below_right  
                SET element at position col of element at position row of triangle TO element at position col of element at position row of triangle PLUS minimum_adjacent  
            END FOR  
        END FOR  
        RETURN element at position zero of element at position zero of triangle  
    END FUNCTION  
END CLASS
