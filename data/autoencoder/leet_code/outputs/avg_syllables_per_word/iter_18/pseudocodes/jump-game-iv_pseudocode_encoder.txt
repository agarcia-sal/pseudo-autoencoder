CLASS Solution  
    FUNCTION minJumps WITH PARAMETER arr AS list of integers  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        CREATE value_indices AS a MAPPING DEFAULTING TO empty list ASSOCIATING values TO their positions  
        FOR each index i AND value v IN arr  
            APPEND i TO the list AT key v IN value_indices  
        END FOR  
        INITIALIZE queue AS a double ended queue CONTAINING a pair OF zero AND zero REPRESENTING current position AND current steps  
        INITIALIZE visited AS a set CONTAINING zero REPRESENTING visited indices  
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND UNPACK AS current_index AND steps  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            DEFINE next_indices AS a list COMPRISING current_index PLUS one AND current_index MINUS one AND all elements AT key arr AT position current_index IN value_indices  
            FOR each next_index IN next_indices  
                IF next_index IS greater THAN OR EQUAL TO zero AND next_index IS less THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a pair OF next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            SET the list AT key arr AT position current_index IN value_indices TO empty list TO PREVENT revisiting  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
