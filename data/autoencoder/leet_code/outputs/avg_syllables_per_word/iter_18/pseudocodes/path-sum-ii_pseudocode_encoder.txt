CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH PARAMETER element at position zero of values  
    SET index_tracker TO one  
    SET queue TO A NEW deque COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO FIRST element REMOVED FROM queue  
        IF index_tracker IS LESS THAN the LENGTH OF values AND element at position index_tracker of values IS NOT none  
            SET node dot left TO A NEW TreeNode WITH PARAMETER element at position index_tracker of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT index_tracker BY one  
        IF index_tracker IS LESS THAN the LENGTH OF values AND element at position index_tracker of values IS NOT none  
            SET node dot right TO A NEW TreeNode WITH PARAMETER element at position index_tracker of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT index_tracker BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN THE RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND THE RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum WITH PARAMETERS self root targetSum  
        FUNCTION dfs WITH PARAMETERS node current_sum path  
            IF node IS none  
                RETURN  
            END IF  
            APPEND the val ATTRIBUTE OF node TO path  
            SET current_sum TO current_sum PLUS the val ATTRIBUTE OF node  
            IF node dot left IS none AND node dot right IS none AND current_sum EQUALS targetSum  
                APPEND A COPY OF path TO result  
            END IF  
            CALL dfs WITH PARAMETERS the left ATTRIBUTE OF node current_sum path  
            CALL dfs WITH PARAMETERS the right ATTRIBUTE OF node current_sum path  
            REMOVE LAST element FROM path  
        END FUNCTION  
        SET result TO AN EMPTY LIST  
        CALL dfs WITH PARAMETERS root zero AN EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
