CLASS Solution  
    FUNCTION kthGrammar WITH PARAMETERS n k  
        IF n EQUALS one THEN  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF (n MINUS two)  
        IF k LESS THAN OR EQUAL TO length_of_previous_row THEN  
            RETURN call METHOD kthGrammar ON self WITH ARGUMENTS n MINUS one AND k  
        ELSE  
            RETURN one MINUS call METHOD kthGrammar ON self WITH ARGUMENTS n MINUS one AND k MINUS length_of_previous_row  
        END IF  
    END FUNCTION  
END CLASS
