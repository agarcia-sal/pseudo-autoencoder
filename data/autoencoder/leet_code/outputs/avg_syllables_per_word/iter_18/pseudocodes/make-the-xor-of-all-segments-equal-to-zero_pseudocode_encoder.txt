CLASS Solution  
    FUNCTION minChanges(nums LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET groups TO a default dictionary mapping keys to empty lists  
        FOR each index FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO the element at position remainder of index DIVIDED BY k in groups  
        END FOR  
        
        SET freq TO a list containing k default dictionaries mapping keys to zero  
        SET min_size TO a list containing k elements each set to positive infinity  
        
        FOR each index FROM zero TO k MINUS one  
            FOR each number IN element at position index of groups  
                INCREMENT element with key number in element at position index of freq BY one  
            END FOR  
            SET element at position index of min_size TO the LENGTH OF element at position index of groups  
        END FOR  
        
        SET dp TO a list of size two RAISED TO THE power of ten each element set to positive infinity  
        SET element at position zero of dp TO zero  
        
        FOR each index FROM zero TO k MINUS one  
            SET new_dp TO a list of size two RAISED TO_THE power of ten each element set to positive infinity  
            SET min_dp TO the minimum value in dp  
            
            FOR each integer j FROM zero TO two RAISED TO_THE power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position index of min_size  
                
                FOR each pair of number and count IN items of element at position index of freq  
                    SET candidate TO element at position of j XOR number of dp PLUS element at position index of min_size MINUS count  
                    IF candidate LESS THAN element at position j of new_dp THEN  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
