CLASS Solution  
    FUNCTION circularArrayLoop(nums: List of Integer) RETURNS Boolean  
        FUNCTION sameDirection(val1, val2) RETURNS Boolean  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero OR val1 LESS THAN zero AND val2 LESS THAN zero THEN  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
  
        SET n TO the LENGTH OF nums  
  
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
  
            IF element at position i of nums EQUALS zero THEN  
                CONTINUE TO next iteration of the FOR loop  
            END IF  
  
            WHILE True HOLDS  
                SET slow TO the REMAINDER OF slow PLUS element at position slow of nums DIVIDED BY n  
                IF slow LESS THAN zero THEN  
                    INCREMENT slow BY n  
                END IF  
  
                SET fast TO the REMAINDER OF fast PLUS element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero THEN  
                    INCREMENT fast BY n  
                END IF  
  
                SET fast TO the REMAINDER OF fast PLUS element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero THEN  
                    INCREMENT fast BY n  
                END IF  
  
                IF NOT sameDirection(element at position slow of nums, element at position i of nums) OR NOT sameDirection(element at position fast of nums, element at position i of nums) THEN  
                    BREAK FROM the WHILE loop  
                END IF  
  
                IF slow EQUALS fast THEN  
                    IF slow EQUALS the REMAINDER OF slow PLUS element at position slow of nums DIVIDED BY n THEN  
                        BREAK FROM the WHILE loop  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  
  
        RETURN False  
    END FUNCTION  
END CLASS
