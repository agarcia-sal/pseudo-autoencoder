CLASS Solution  
    FUNCTION primePalindrome WITH PARAMETER n  
        FUNCTION is_prime WITH PARAMETER num  
            IF num LESS THAN two THEN  
                RETURN False  
            END IF  
            IF num EQUALS two OR num EQUALS three THEN  
                RETURN True  
            END IF  
            IF num MODULO two EQUALS zero OR num MODULO three EQUALS zero THEN  
                RETURN False  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num HOLDS  
                IF num MODULO i EQUALS zero OR num MODULO i PLUS two EQUALS zero THEN  
                    RETURN False  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN True  
        END FUNCTION  
          
        FUNCTION is_palindrome WITH PARAMETER num  
            RETURN the STRING representation of num EQUALS the REVERSE of the STRING representation of num  
        END FUNCTION  
          
        IF n LESS THAN OR EQUAL TO two THEN  
            RETURN two  
        END IF  
          
        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven THEN  
            RETURN eleven  
        END IF  
          
        WHILE True HOLDS  
            IF n MODULO two EQUALS zero THEN  
                INCREMENT n BY one  
                CONTINUE  
            END IF  
            IF is_palindrome OF n AND is_prime OF n THEN  
                RETURN n  
            END IF  
            IF the LENGTH OF the STRING representation of n MODULO two EQUALS zero THEN  
                SET n TO ten RAISED TO the POWER OF the LENGTH OF the STRING representation of n PLUS one  
            ELSE  
                INCREMENT n BY two  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
