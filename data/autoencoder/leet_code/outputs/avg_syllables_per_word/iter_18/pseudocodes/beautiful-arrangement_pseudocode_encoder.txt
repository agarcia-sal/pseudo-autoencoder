CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used_collection and index_value)  
            IF the value of index_value EQUALS the value of n PLUS one  
                RETURN one  
            END IF  
            SET count_value TO zero  
            FOR each integer i_value FROM one TO n  
                IF the element at position i_value of used_collection EQUALS false AND either the remainder of i_value DIVIDED BY index_value EQUALS zero OR the remainder of index_value DIVIDED BY i_value EQUALS zero  
                    SET the element at position i_value of used_collection TO true  
                    INCREMENT count_value BY the result of backtrack with used_collection and index_value PLUS one  
                    SET the element at position i_value of used_collection TO false  
                END IF  
            END FOR  
            RETURN count_value  
        END FUNCTION  
        SET used_collection TO a list of n PLUS one elements each EQUALS false  
        RETURN backtrack with used_collection and one  
    END FUNCTION  
END CLASS
