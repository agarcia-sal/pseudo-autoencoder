CLASS Solution  
    FUNCTION leastOpsExpressTarget(x AS integer, target AS integer) RETURNS integer  
        FUNCTION dp(parameter_target AS integer) RETURNS integer  
            IF parameter_target EQUALS zero THEN  
                RETURN negative one  
            END IF  
            IF parameter_target LESS THAN x THEN  
                SET option_one TO two MULTIPLIED BY parameter_target MINUS one  
                SET option_two TO two MULTIPLIED BY x MINUS parameter_target  
                RETURN the lesser value BETWEEN option_one AND option_two  
            END IF  
            SET current_power TO one  
            WHILE x RAISED TO THE POWER OF current_power LESS THAN OR EQUAL TO parameter_target HOLDS  
                INCREMENT current_power BY one  
            END WHILE  
            DECREMENT current_power BY one  
            SET choice_one_result TO dp(parameter_target MINUS x RAISED TO THE POWER OF current_power) PLUS current_power  
            IF x RAISED TO THE POWER OF current_power PLUS one MINUS parameter_target LESS THAN parameter_target THEN  
                SET choice_two_result TO dp(x RAISED TO THE POWER OF current_power PLUS one MINUS parameter_target) PLUS current_power PLUS one  
                SET final_result TO the lesser value BETWEEN choice_one_result AND choice_two_result  
            ELSE  
                SET final_result TO choice_one_result  
            END IF  
            RETURN final_result  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
