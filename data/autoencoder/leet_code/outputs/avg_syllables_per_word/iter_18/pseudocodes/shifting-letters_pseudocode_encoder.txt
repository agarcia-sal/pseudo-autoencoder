CLASS Solution  
    FUNCTION shiftingLetters WITH PARAMETERS s AND shifts  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET new_char_code TO the character code of character at position index of s MINUS the character code of character a PLUS element at position index of shifts  
            SET new_char_code TO new_char_code MODULO twenty six PLUS the character code of character a  
            SET new_char TO the character corresponding to new_char_code  
            APPEND new_char TO result  
        END FOR  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
