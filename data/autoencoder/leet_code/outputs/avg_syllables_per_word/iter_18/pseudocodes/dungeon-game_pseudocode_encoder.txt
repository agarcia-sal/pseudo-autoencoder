CLASS Solution  
    FUNCTION calculateMinimumHP WITH PARAMETERS dungeon  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF the element at position zero of dungeon  
        CALL auxiliary FUNCTION createDPTable WITH PARAMETERS m AND n AND ASSIGN RESULT TO dp  
        SET element at position m AND position n MINUS one of dp TO one  
        SET element at position m MINUS one AND position n of dp TO one  
        FOR i FROM m MINUS one DOWN TO zero INCLUSIVE  
            FOR j FROM n MINUS one DOWN TO zero INCLUSIVE  
                SET min_health TO the MINIMUM OF element at position i PLUS one AND position j of dp AND element at position i AND position j PLUS one of dp MINUS element at position i AND position j of dungeon  
                IF min_health IS GREATER THAN OR EQUAL TO one  
                    SET element at position i AND position j of dp TO min_health  
                ELSE  
                    SET element at position i AND position j of dp TO one  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero AND position zero of dp  
    END FUNCTION  
  
    FUNCTION createDPTable WITH PARAMETERS m AND n  
        DESCRIBE AS auxiliary FUNCTION TO create a two-dimensional list with dimensions m plus one AND n plus one initialized to infinity  
    END FUNCTION  
END CLASS
