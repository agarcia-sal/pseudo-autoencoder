CLASS Solution  
    FUNCTION carFleet WITH PARAMETERS target position speed  
        SET zipped_pairs TO the ZIP of position AND speed  
        SET sorted_pairs TO zipped_pairs ORDERED IN DESCENDING ORDER ACCORDING TO THE FIRST ELEMENT OF EACH PAIR  
        SET time_to_target TO an EMPTY LIST  
        FOR each pair IN sorted_pairs  
            SET position_element TO the FIRST element of pair  
            SET speed_element TO the SECOND element of pair  
            SET time_value TO target MINUS position_element DIVIDED BY speed_element  
            APPEND time_value TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
