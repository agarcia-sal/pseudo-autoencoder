CLASS Solution  
    FUNCTION hIndex WITH PARAMETER citations AS list of integers  
        SET n TO the LENGTH OF citations  
        SET left TO zero  
        SET right TO the LENGTH OF citations MINUS one  
        SET h_index TO zero  
        
        WHILE the VALUE of left IS LESS THAN OR EQUAL TO the VALUE of right HOLDS  
            SET mid TO the VALUE of left PLUS the VALUE of right MINUS the VALUE of left DIVIDED BY two  
            IF the element at position mid of citations IS GREATER THAN OR EQUAL TO the VALUE of n MINUS the VALUE of mid THEN  
                SET h_index TO the VALUE of n MINUS the VALUE of mid  
                SET right TO the VALUE of mid MINUS one  
            ELSE  
                SET left TO the VALUE of mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN h_index  
    END FUNCTION  
END CLASS
