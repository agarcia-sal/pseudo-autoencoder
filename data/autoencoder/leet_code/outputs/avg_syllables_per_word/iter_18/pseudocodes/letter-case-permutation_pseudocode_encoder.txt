CLASS Solution  
    FUNCTION letterCasePermutation WITH parameter s  
        FUNCTION dfs WITH parameter i  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t THEN  
                APPEND all characters concatenated from t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH argument i PLUS one  
            IF character at position i of t IS alphabetic THEN  
                SET character at position i of t TO the character obtained by toggling the case of character at position i of t  
                CALL dfs WITH argument i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO the list of individual characters in s  
        SET ans TO an empty list  
        CALL dfs WITH argument zero  
        RETURN ans  
    END FUNCTION  
END CLASS
