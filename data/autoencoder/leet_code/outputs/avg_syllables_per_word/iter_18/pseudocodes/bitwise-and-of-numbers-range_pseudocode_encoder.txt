CLASS Solution  
    FUNCTION rangeBitwiseAnd(left AS integer, right AS integer) RETURNS integer  
        SET shift TO zero  
        WHILE left LESS THAN right HOLDS  
            SET left TO the integer value of left DIVIDED BY two  
            SET right TO the integer value of right DIVIDED BY two  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY the integer value of two RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
