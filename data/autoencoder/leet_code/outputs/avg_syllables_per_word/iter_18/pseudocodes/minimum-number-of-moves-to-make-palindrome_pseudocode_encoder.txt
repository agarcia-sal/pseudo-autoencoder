CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s AS string  
        SET s TO list of characters in s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
          
        WHILE the LENGTH OF s GREATER THAN one holds  
            SET found_match_indicator TO false  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF character at position i of s EQUALS character at position zero of s THEN  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SET temporary_character TO character at position j PLUS one of s  
                        SET character at position j PLUS one of s TO character at position j of s  
                        SET character at position j of s TO temporary_character  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE character at position zero FROM s  
                    REMOVE character at position negative one FROM s  
                    SET found_match_indicator TO true  
                    BREAK  
                END IF  
            END FOR  
            IF found_match_indicator EQUALS false THEN  
                REMOVE character at position zero FROM s  
            END IF  
        END WHILE  
          
        RETURN moves  
    END FUNCTION  
END CLASS
