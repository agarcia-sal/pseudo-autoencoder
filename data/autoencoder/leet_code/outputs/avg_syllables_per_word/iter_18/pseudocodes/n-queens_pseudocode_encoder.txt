CLASS Solution  
    FUNCTION solveNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN NOT the SUM OF the element at position col of cols PLUS the element at position given by row MINUS col of hills PLUS the element at position given by row PLUS col of dales  
        END FUNCTION  

        FUNCTION place_queen WITH PARAMETERS row AND col  
            ADD the pair of row AND col TO queens  
            SET the element at position col of cols TO one  
            SET the element at position given by row MINUS col of hills TO one  
            SET the element at position given by row PLUS col of dales TO one  
        END FUNCTION  

        FUNCTION remove_queen WITH PARAMETERS row AND col  
            REMOVE the pair of row AND col FROM queens  
            SET the element at position col of cols TO zero  
            SET the element at position given by row MINUS col of hills TO zero  
            SET the element at position given by row PLUS col of dales TO zero  
        END FUNCTION  

        FUNCTION add_solution  
            SET solution TO an empty list  
            FOR each pair of underscore AND col IN the ascending ordered elements of queens  
                SET row_representation TO a sequence of character period repeated col times CONCATENATED WITH character Q CONCATENATED WITH a sequence of character period repeated by n MINUS col MINUS one times  
                APPEND row_representation TO solution  
            END FOR  
            APPEND solution TO output  
        END FUNCTION  

        FUNCTION backtrack WITH PARAMETER row DEFAULTED TO zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH row AND col HOLDS  
                    CALL place_queen WITH row AND col  
                    IF row PLUS one EQUALS n  
                        CALL add_solution  
                    ELSE  
                        CALL backtrack WITH row PLUS one  
                    END IF  
                    CALL remove_queen WITH row AND col  
                END IF  
            END FOR  
        END FUNCTION  

        SET cols TO a list of zeros repeated n times  
        SET hills TO a list of zeros repeated by two TIMES n MINUS one times  
        SET dales TO a list of zeros repeated by two TIMES n MINUS one times  
        SET queens TO an empty set  
        SET output TO an empty list  
        CALL backtrack  
        RETURN output  
    END FUNCTION  
END CLASS
