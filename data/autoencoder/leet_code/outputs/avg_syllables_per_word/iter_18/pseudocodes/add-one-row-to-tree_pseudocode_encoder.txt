CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val defaulted TO zero left defaulted TO none right defaulted TO none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values of TYPE list  
    IF the LENGTH OF values EQUALS zero OR values is empty  
        RETURN none  
    END IF  
    SET root TO CREATE new TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO CREATE new deque  
    APPEND root TO queue  
    WHILE the queue is NOT empty  
        SET node TO REMOVE leftmost element from queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node left TO CREATE new TreeNode WITH val EQUALS element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node right TO CREATE new TreeNode WITH val EQUALS element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUAL TO the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left attribute of p AND the left attribute of q AND the result of is_same_tree WITH PARAMETERS the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow WITH PARAMETERS self root of OPTIONAL TreeNode val of TYPE integer depth of TYPE integer  
        IF depth EQUALS one  
            RETURN CREATE new TreeNode WITH val EQUALS val AND left EQUALS root  
        END IF  
        SET queue TO CREATE new deque WITH a single element consisting of a tuple of root AND one  
        WHILE queue IS NOT empty  
            SET node AND level TO REMOVE leftmost element from queue  
            IF level EQUALS depth MINUS one  
                SET node left TO CREATE new TreeNode WITH val EQUALS val AND left EQUALS node left  
                SET node right TO CREATE new TreeNode WITH val EQUALS val AND right EQUALS node right  
            ELSE IF level IS LESS THAN depth MINUS one  
                IF node left IS NOT none  
                    APPEND a tuple of node left AND level PLUS one TO queue  
                END IF  
                IF node right IS NOT none  
                    APPEND a tuple of node right AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
