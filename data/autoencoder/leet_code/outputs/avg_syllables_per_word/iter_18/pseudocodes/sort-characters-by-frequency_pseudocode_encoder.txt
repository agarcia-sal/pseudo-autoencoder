CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO RESULT OF invoking Counter on s  
        SET sorted_characters TO RESULT OF sorting the keys in frequency BY the frequency of element at position element in frequency IN descending order  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            APPEND character REPEATED frequency at position character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
