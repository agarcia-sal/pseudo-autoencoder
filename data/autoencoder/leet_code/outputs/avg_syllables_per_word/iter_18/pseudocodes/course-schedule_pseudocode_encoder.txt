CLASS Solution  
    FUNCTION canFinish(numCourses AS integer, prerequisites AS list of list of integer) AS boolean  
        IMPORT collections AS defaultdict and deque  
        
        SET graph TO a new defaultdict with list as default value  
        SET indegree TO a list of zero repeated numCourses times  
        
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a new deque containing each course in the range from zero to numCourses minus one WHERE element at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            SET current_course TO the element removed from the left of queue  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course in graph  
                DECREMENT element at position next_course of indegree BY one  
                IF element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF numCourses EQUALS zero  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
