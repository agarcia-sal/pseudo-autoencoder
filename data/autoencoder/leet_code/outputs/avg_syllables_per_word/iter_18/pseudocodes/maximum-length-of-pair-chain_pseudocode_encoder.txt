CLASS Solution  
    FUNCTION findLongestChain WITH parameter pairs  
        CALL auxiliary FUNCTION sort_pairs_by_second_element WITH parameter pairs  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each start and end IN pairs  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
    FUNCTION sort_pairs_by_second_element WITH parameter pairs  
        PERFORM sorting of pairs in ascending order according to the value of element at position two of each pair  
    END FUNCTION  
END CLASS
