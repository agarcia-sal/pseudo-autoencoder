CLASS Solution  
    FUNCTION countAndSay WITH PARAMETER n  
        IF n EQUALS one THEN  
            RETURN string one  
        END IF  
        SET current_sequence TO string one  
        FOR each iteration FROM two TO n  
            SET next_sequence TO empty string  
            SET count TO one  
            SET previous_char TO character at position zero of current_sequence  
            FOR each index FROM one TO the length of current_sequence MINUS one  
                IF character at position index of current_sequence EQUALS previous_char THEN  
                    INCREMENT count BY one  
                ELSE  
                    APPEND string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO character at position index of current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            SET current_sequence TO next_sequence  
        END FOR  
        RETURN current_sequence  
    END FUNCTION  
END CLASS
