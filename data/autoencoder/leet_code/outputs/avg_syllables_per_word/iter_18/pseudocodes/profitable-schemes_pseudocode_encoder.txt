CLASS Solution  
    FUNCTION profitableSchemes(n AS integer  minProfit AS integer  group AS list of integers  profit AS list of integers) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS two dimensional list of integers WITH the count of the first dimension being n PLUS one AND the count of the second dimension being minProfit PLUS one FILLED WITH zero  
        FOR each index i FROM zero TO n  
            SET element at position zero of element at position i of dp TO one  
        END FOR  
        FOR each element p AND element g IN parallel from profit AND group  
            FOR each index i FROM n DOWN TO g  
                FOR each index j FROM minProfit DOWN TO zero  
                    SET temp_index TO j MINUS p  
                    IF temp_index IS LESS THAN zero THEN  
                        SET temp_index TO zero  
                    END IF  
                    SET element at position j of element at position i of dp TO the sum of element at position j of element at position i of dp AND element at position temp_index of element at position i MINUS g of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position minProfit of element at position n of dp  
    END FUNCTION  
END CLASS
