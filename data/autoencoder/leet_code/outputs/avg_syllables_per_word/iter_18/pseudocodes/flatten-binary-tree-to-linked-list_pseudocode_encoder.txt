CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULTING TO zero left DEFAULTING TO None right DEFAULTING TO None  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET index_pointer TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF index_pointer IS LESS THAN the LENGTH OF values AND element at position index_pointer of values IS NOT None  
            SET node's left attribute TO a new TreeNode initialized with element at position index_pointer of values  
            APPEND node's left attribute TO queue  
        END IF  
        INCREMENT index_pointer BY one  
        IF index_pointer IS LESS THAN the LENGTH OF values AND element at position index_pointer of values IS NOT None  
            SET node's right attribute TO a new TreeNode initialized with element at position index_pointer of values  
            APPEND node's right attribute TO queue  
        END IF  
        INCREMENT index_pointer BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p's value attribute IS NOT EQUAL TO q's value attribute  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p's left attribute AND q's left attribute AND the result of is_same_tree WITH PARAMETERS p's right attribute AND q's right attribute  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten WITH PARAMETERS self AND root AS Optional TreeNode  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO a list containing root  
        WHILE stack is not empty  
            SET node TO the last element removed from stack  
            IF node's right attribute IS NOT None  
                APPEND node's right attribute TO stack  
            END IF  
            IF node's left attribute IS NOT None  
                APPEND node's left attribute TO stack  
            END IF  
            SET node's left attribute TO None  
            IF stack is not empty  
                SET node's right attribute TO the last element of stack  
            ELSE  
                SET node's right attribute TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
