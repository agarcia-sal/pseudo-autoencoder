CLASS Solution  
    FUNCTION kEmptySlots(bulbs AS list of integers k AS integer) RETURNS integer  
        SET bloom_days TO empty list  
        FOR each day and position IN bulbs starting from one and iterating OVER bulbs  
            SET insertion_index TO the position at which position SHOULD BE INSERTED in bloom_days preserving sorted order USING bisect left method  
            IF insertion_index IS GREATER THAN zero AND position MINUS element at position insertion_index minus one of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF insertion_index IS LESS THAN the LENGTH OF bloom_days AND element at position insertion_index of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT insertion_index  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
