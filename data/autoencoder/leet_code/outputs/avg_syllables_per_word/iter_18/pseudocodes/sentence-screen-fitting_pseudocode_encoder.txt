CLASS Solution  
    FUNCTION wordsTyping(sentence AS List OF string commas rows AS integer commas cols AS integer) RETURNS integer  
        SET s TO the concatenation of all elements IN sentence separated by single space PLUS single space character  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF character at position MODULO total_length of s EQUALS single space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND character at (position MINUS one) MODULO total_length of s NOT EQUALS single space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        SET fitted_count TO position DIVIDED BY total_length  
        RETURN fitted_count  
    END FUNCTION  
END CLASS
