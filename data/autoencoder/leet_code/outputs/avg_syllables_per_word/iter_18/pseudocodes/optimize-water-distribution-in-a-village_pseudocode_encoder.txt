CLASS Solution  
    FUNCTION minCostToSupplyWater(n AS integer, wells AS list of integers, pipes AS list of lists of integers) RETURNS integer  
        FOR index FROM zero TO n MINUS one  
            APPEND list containing zero AND index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        
        SORT pipes BY the third element in each element of pipes IN ascending order
        
        SET parent TO list of integers FROM zero TO n
        
        FUNCTION find(x AS integer) RETURNS integer  
            IF element at position x of parent NOT EQUALS x THEN  
                SET element at position x of parent TO the RESULT OF calling find WITH argument element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x AS integer, y AS integer) RETURNS boolean  
            SET rootX TO the RESULT OF calling find WITH argument x  
            SET rootY TO the RESULT OF calling find WITH argument y  
            IF rootX NOT EQUALS rootY THEN  
                SET element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET min_cost TO zero  
        FOR each element IN pipes  
            SET house1 TO element at position zero of element  
            SET house2 TO element at position one of element  
            SET cost TO element at position two of element  
            IF the RESULT OF calling union WITH arguments house1 AND house2 THEN  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
