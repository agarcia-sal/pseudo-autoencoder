CLASS Solution  
    FUNCTION maximumGood(statements AS List of List of integer)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR each integer i FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET good TO a list of boolean FALSE repeated n times  
            SET bad TO a list of boolean FALSE repeated n times  
            FOR each integer j FROM zero TO n MINUS one  
                IF the integer i AND TWO RAISED TO THE POWER OF j IS GREATER THAN zero THEN  
                    SET the element at position j of good TO boolean TRUE  
                ELSE  
                    SET the element at position j of bad TO boolean TRUE  
                END IF  
            END FOR  
            SET valid TO boolean TRUE  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of good IS boolean TRUE THEN  
                    FOR each integer k FROM zero TO n MINUS one  
                        IF the element at position k of the element at position j of statements EQUALS zero AND the element at position k of good IS boolean TRUE OR the element at position k of the element at position j of statements EQUALS one AND the element at position k of bad IS boolean TRUE THEN  
                            SET valid TO boolean FALSE  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS boolean FALSE THEN  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS boolean TRUE THEN  
                SET sum_good TO zero  
                FOR each boolean value IN good  
                    IF value IS boolean TRUE THEN  
                        INCREMENT sum_good BY one  
                    END IF  
                END FOR  
                IF sum_good IS GREATER THAN max_good THEN  
                    SET max_good TO sum_good  
                END IF  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
