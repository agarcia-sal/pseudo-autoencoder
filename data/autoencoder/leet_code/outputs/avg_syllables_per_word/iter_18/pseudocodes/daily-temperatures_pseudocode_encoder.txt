CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures AS list of integers  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a list of zero repeated n times  
        SET stack TO an empty list  
        
        FOR index FROM zero TO n MINUS one  
            WHILE the stack is not empty AND the temperature at position index of temperatures IS GREATER THAN the temperature at position the last element of stack of temperatures  
                SET previous_index TO the removal of the last element from stack  
                SET the element at position previous_index of answer TO index MINUS previous_index  
            END WHILE  
            APPEND index TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
