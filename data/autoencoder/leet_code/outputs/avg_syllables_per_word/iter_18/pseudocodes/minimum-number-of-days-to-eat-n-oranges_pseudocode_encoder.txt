CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining IS LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO remainder of remaining DIVIDED BY two PLUS result of dp WITH remaining DIVIDED BY two  
            SET option_two TO remainder of remaining DIVIDED BY three PLUS result of dp WITH remaining DIVIDED BY three  
            SET minimum_option TO the lesser OF option_one AND option_two  
            SET result TO one PLUS minimum_option  
            RETURN result  
        END FUNCTION  
        RETURN result of dp WITH n  
    END FUNCTION  
END CLASS
