CLASS Solution  
    FUNCTION pathSum WITH PARAMETER nums  
        SET tree TO an empty mapping  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN reference TO value IN mapping tree with key as the pair of depth and pos  
        END FOR  

        FUNCTION dfs WITH PARAMETERS depth pos path_sum  
            SET current_value TO the value in mapping tree for key as the pair of depth and pos or to no value if key does not exist  
            IF current_value EQUALS no value  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO the pair of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO the pair of depth PLUS one and pos MULTIPLIED BY two  
            IF left_child NOT IN keys of tree AND right_child NOT IN keys of tree  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO RESULT OF invoking dfs WITH arguments depth of left_child pos of left_child new_path_sum  
            SET right_sum TO RESULT OF invoking dfs WITH arguments depth of right_child pos of right_child new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  

        RETURN RESULT OF invoking dfs WITH arguments one one zero  
    END FUNCTION  
END CLASS
