CLASS Solution  
    FUNCTION longestValidParentheses WITH PARAMETER s  
        SET stack TO a new list containing the number negative one  
        SET max_length TO zero  
        FOR each index and character IN enumeration of s  
            IF character EQUALS the opening parenthesis symbol  
                APPEND index TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack is empty  
                    APPEND index TO stack  
                ELSE  
                    SET length_difference TO index MINUS the last element of stack  
                    SET max_length TO the greater of max_length AND length_difference  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
