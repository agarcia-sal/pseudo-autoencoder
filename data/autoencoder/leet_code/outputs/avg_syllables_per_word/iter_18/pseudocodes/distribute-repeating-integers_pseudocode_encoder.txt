CLASS Solution  
    FUNCTION canDistribute(nums AS list OF integer, quantity AS list OF integer) RETURNS boolean  
        SET count TO the LIST OF VALUES OF THE COUNTER OF nums  
        SORT quantity IN descending order  
        
        FUNCTION can_satisfy(index AS integer) RETURNS boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN true  
            END IF  
            
            FOR each position FROM zero TO the LENGTH OF count MINUS one  
                IF element at position of count GREATER THAN OR EQUAL TO element at position index of quantity THEN  
                    DECREMENT element at position of count BY element at position index of quantity  
                    IF can_satisfy index PLUS one THEN  
                        RETURN true  
                    END IF  
                    INCREMENT element at position of count BY element at position index of quantity  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN can_satisfy zero  
    END FUNCTION  
END CLASS
