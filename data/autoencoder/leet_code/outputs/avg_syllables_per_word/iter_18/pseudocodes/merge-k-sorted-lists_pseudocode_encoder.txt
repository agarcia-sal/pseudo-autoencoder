CLASS ListNode  
    FUNCTION __init__ WITH parameters val AND next  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode initialized WITH element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one to end  
        SET node TO new ListNode initialized WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN expression element val of p1 EQUALS element val of p2 AND is_same_list called with element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH parameter lists  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists together  
            IF lst IS NOT none  
                CALL heap push function with min_heap AND tuple of element val of lst AND i AND lst  
            END IF  
        END FOR  
        SET dummy TO new ListNode initialized with default values  
        SET current TO dummy  
        WHILE min_heap IS NOT empty holds  
            SET tuple unpacking val AND i AND node TO result of heap pop function applied to min_heap  
            SET element next of current TO node  
            SET current TO element next of current  
            IF element next of node IS NOT none  
                CALL heap push function with min_heap AND tuple of element val of element next of node AND i AND element next of node  
            END IF  
        END WHILE  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
