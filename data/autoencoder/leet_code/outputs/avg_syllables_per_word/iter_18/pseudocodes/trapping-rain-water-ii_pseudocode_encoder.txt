CLASS Solution  
    FUNCTION trapRainWater(heightMap AS List OF List OF integer) RETURNS integer  
        IF heightMap IS empty OR the element at position zero of heightMap IS empty THEN  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF the element at position zero of heightMap  
        SET visited TO a two dimensional collection of boolean FALSE valued elements with dimensions m BY n  
        SET min_heap TO an empty collection representing a minimum heap  

        FOR each index i FROM zero TO m MINUS one  
            PUSH onto min_heap the tuple consisting of the element at position i of heightMap AT position zero and i and zero  
            PUSH onto min_heap the tuple consisting of the element at position i of heightMap AT position n MINUS one and i and n MINUS one  
            SET the element at position i AT position zero of visited TO TRUE  
            SET the element at position i AT position n MINUS one of visited TO TRUE  
        END FOR  

        FOR each index j FROM zero TO n MINUS one  
            PUSH onto min_heap the tuple consisting of the element at position zero of heightMap AT position j and zero and j  
            PUSH onto min_heap the tuple consisting of the element at position m MINUS one of heightMap AT position j and m MINUS one and j  
            SET the element at position zero AT position j of visited TO TRUE  
            SET the element at position m MINUS one AT position j of visited TO TRUE  
        END FOR  

        SET directions TO the collection consisting of the tuples minus one and zero and one and zero and zero and minus one and zero and one  
        SET water_trapped TO zero  

        WHILE min_heap IS NOT empty  
            POP from min_heap the tuple consisting of height and x and y  

            FOR each tuple dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  

                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the element at position nx AT position ny of visited IS FALSE THEN  
                    SET the element at position nx AT position ny of visited TO TRUE  
                    INCREMENT water_trapped BY the larger of zero AND height MINUS the element at position nx AT position ny of heightMap  
                    PUSH onto min_heap the tuple consisting of the larger of height AND the element at position nx AT position ny of heightMap AND nx AND ny  
                END IF  
            END FOR  
        END WHILE  

        RETURN water_trapped  
    END FUNCTION  
END CLASS
