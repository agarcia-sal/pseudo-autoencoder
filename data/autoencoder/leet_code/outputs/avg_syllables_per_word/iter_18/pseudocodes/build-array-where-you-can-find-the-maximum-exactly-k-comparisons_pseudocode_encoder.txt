CLASS Solution  
    FUNCTION numOfArrays(n AS integer, m AS integer, k AS integer) RETURNS integer  
        SET MOD TO ten raised to the power of nine PLUS seven  
        
        DEFINE dp AS three-dimensional collection of integers initialized to zero with dimensions  
            length n PLUS one by length m PLUS one by length k PLUS one  
        
        FOR each integer j FROM one TO m INCLUSIVE  
            SET element at position one at position j at position one of dp TO one  
        END FOR  
        
        FOR each integer i FROM two TO n INCLUSIVE  
            FOR each integer j FROM one TO m INCLUSIVE  
                FOR each integer p FROM one TO k INCLUSIVE  
                    FOR each integer x FROM one TO the integer immediately less than j  
                        SET element at position i at position j at position p of dp TO the remainder of the sum of  
                            element at position i at position j at position p of dp PLUS element at position i minus one at position x at position p minus one of dp  
                            DIVIDED BY MOD  
                    END FOR  
                    
                    SET element at position i at position j at position p of dp TO the remainder of the sum of  
                        element at position i at position j at position p of dp PLUS product of  
                            element at position i minus one at position j at position p of dp MULTIPLIED BY j  
                        DIVIDED BY MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each integer j FROM one TO m INCLUSIVE  
            SET result TO the remainder of the sum of result PLUS element at position n at position j at position k of dp DIVIDED BY MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
