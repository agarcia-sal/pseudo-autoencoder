CLASS Solution  
    FUNCTION networkDelayTime(times AS List OF List OF Integer, n AS Integer, k AS Integer) RETURNS Integer  
        SET graph TO an empty mapping from each integer from one TO n TO an empty list  
        
        FOR each element IN times  
            DESTRUCTURE element INTO u, v, w  
            APPEND the pair consisting of v and w TO the list in graph at key u  
        END FOR  
        
        SET dist TO a mapping from each integer from one TO n TO positive infinity  
        SET the value in dist at key k TO zero  
        
        SET min_heap TO a list containing the pair consisting of zero and k  
        
        WHILE the LENGTH OF min_heap is greater than zero  
            SET current_dist and node TO the result of removing the least element from min_heap according to heap order  
            
            IF current_dist is GREATER THAN the value in dist at key node  
                CONTINUE  
            END IF  
            
            FOR each neighbor and weight IN the list in graph at key node  
                SET distance TO current_dist PLUS weight  
                
                IF distance is LESS THAN the value in dist at key neighbor  
                    SET the value in dist at key neighbor TO distance  
                    INSERT the pair consisting of distance and neighbor INTO min_heap preserving heap order  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time is EQUAL TO positive infinity  
            RETURN minus one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
