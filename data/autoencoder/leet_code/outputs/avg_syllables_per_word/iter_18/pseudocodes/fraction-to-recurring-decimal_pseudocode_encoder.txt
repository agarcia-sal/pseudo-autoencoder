CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER) RETURNS STRING  
        IF numerator LESS THAN zero XOR denominator LESS THAN zero  
            SET sign TO minus sign  
        ELSE  
            SET sign TO empty string  
        END IF  
  
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
  
        SET quotient TO the integer division result of numerator DIVIDED BY denominator  
        SET remainder TO the remainder of numerator DIVIDED BY denominator  
        SET integer_part TO the string representation of quotient  
  
        IF remainder EQUALS zero  
            RETURN the concatenation of sign AND integer_part  
        END IF  
  
        SET seen_remainders TO an empty mapping from remainder TO position  
        SET fractional_part TO an empty list  
  
        WHILE remainder NOT EQUALS zero  
            IF remainder EXISTS IN seen_remainders  
                SET start TO the position at key remainder IN seen_remainders  
                SET non_repeating TO the concatenation of elements from fractional_part FROM zero TO start exclusive  
                SET repeating TO the concatenation of elements from fractional_part FROM start TO the end  
                RETURN the concatenation of sign AND integer_part AND a literal dot AND non_repeating AND an opening parenthesis AND repeating AND a closing parenthesis  
            END IF  
  
            ASSIGN the length of fractional_part TO seen_remainders AT key remainder  
  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer division result of remainder DIVIDED BY denominator  
            SET remainder TO the remainder of remainder DIVIDED BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
  
        RETURN the concatenation of sign AND integer_part AND a literal dot AND the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
