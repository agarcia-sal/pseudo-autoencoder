CLASS Solution  
    FUNCTION increasingTriplet(nums)  
        SET first TO infinite quantity  
        SET second TO infinite quantity  
        FOR each num IN nums  
            IF num LESS THAN OR EQUAL TO first  
                SET first TO num  
            ELSE IF num LESS THAN OR EQUAL TO second  
                SET second TO num  
            ELSE  
                RETURN truth value of true  
            END IF  
        END FOR  
        RETURN truth value of false  
    END FUNCTION  
END CLASS
