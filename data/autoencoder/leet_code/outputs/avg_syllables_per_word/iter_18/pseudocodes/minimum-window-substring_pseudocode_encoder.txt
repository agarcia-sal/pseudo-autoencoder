CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s STRING AND t STRING  
        IMPORT Counter FROM collections  
          
        SET t_count TO Counter OF t  
        SET required TO the LENGTH OF t_count  
          
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO an EMPTY Counter  
          
        SET min_length TO positive infinity  
        SET min_start TO zero  
          
        WHILE the value of right IS LESS THAN the LENGTH OF s  
            INCREMENT the COUNT at character at position right OF s IN window_counts BY one  
              
            IF the character at position right OF s IS IN t_count AND the COUNT at character at position right OF s IN window_counts EQUALS the COUNT at character at position right OF s IN t_count  
                INCREMENT formed BY one  
            END IF  
              
            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the character at position left OF s  
                  
                IF the DIFFERENCE BETWEEN right AND left PLUS one IS LESS THAN min_length  
                    SET min_length TO the DIFFERENCE BETWEEN right AND left PLUS one  
                    SET min_start TO left  
                END IF  
                  
                DECREMENT the COUNT at char IN window_counts BY one  
                IF char IS IN t_count AND the COUNT at char IN window_counts IS LESS THAN the COUNT at char IN t_count  
                    DECREMENT formed BY one  
                END IF  
                  
                INCREMENT left BY one  
            END WHILE  
              
            INCREMENT right BY one  
        END WHILE  
          
        IF min_length EQUALS positive infinity  
            RETURN an EMPTY string  
        ELSE  
            RETURN the SUBSTRING of s STARTING AT min_start AND HAVING LENGTH min_length  
        END IF  
    END FUNCTION  
END CLASS
