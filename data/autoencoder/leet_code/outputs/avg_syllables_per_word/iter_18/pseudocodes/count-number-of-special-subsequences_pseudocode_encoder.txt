CLASS Solution  
    FUNCTION countSpecialSubsequences(nums AS List OF Integer) RETURNS Integer  
        SET MODULO TO ten RAISED TO THE power OF nine PLUS seven  
        SET dp TO list CONTAINING zero zero zero  
        FOR each element num IN nums  
            IF num EQUALS zero  
                SET element at position zero of dp TO the quantity of two TIMES element at position zero of dp PLUS one MODULO MODULO  
            ELSE IF num EQUALS one  
                SET element at position one of dp TO the quantity of two TIMES element at position one of dp PLUS element at position zero of dp MODULO MODULO  
            ELSE IF num EQUALS two  
                SET element at position two of dp TO the quantity of two TIMES element at position two of dp PLUS element at position one of dp MODULO MODULO  
            END IF  
        END FOR  
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
