CLASS Solution  
    FUNCTION minimumFinishTime WITH PARAMETERS tires LIST OF LIST OF INTEGER changeTime INTEGER numLaps INTEGER RETURNS INTEGER  
        SET min_times TO a list of fifteen elements each assigned to positive infinity  
        FOR each pair f and r IN tires  
            SET total_time TO zero  
            FOR index i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f THEN  
                    BREAK  
                END IF  
                INCREMENT total_time BY lap_time  
                IF lap_time AT position i OF min_times GREATER THAN total_time THEN  
                    SET element at position i OF min_times TO total_time  
                END IF  
            END FOR  
        END FOR  
      
        SET dp TO a list of numLaps PLUS one elements each assigned to positive infinity  
        SET element at position zero OF dp TO zero  
      
        FOR index i FROM one TO numLaps  
            FOR index j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET candidate_time TO element at position i MINUS j MINUS one OF dp PLUS element at position j OF min_times PLUS changeTime  
                IF element at position i OF dp GREATER THAN candidate_time THEN  
                    SET element at position i OF dp TO candidate_time  
                END IF  
            END FOR  
        END FOR  
      
        RETURN element at position numLaps OF dp MINUS changeTime  
    END FUNCTION  
END CLASS
