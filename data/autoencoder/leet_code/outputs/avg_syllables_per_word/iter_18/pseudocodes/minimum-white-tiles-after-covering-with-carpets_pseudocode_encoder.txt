CLASS Solution  
    FUNCTION minimumWhiteTiles WITH PARAMETERS floor numCarpets carpetLen  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional collection OF zero WITH THE first dimension OF n PLUS one AND the second dimension OF numCarpets PLUS one  
        FOR i FROM one TO n  
            SET dp AT position i and zero TO dp AT position i MINUS one and zero PLUS the VALUE ONE IF character AT position i MINUS one OF floor EQUALS character ONE OTHERWISE ZERO  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp AT position i and j TO dp AT position i MINUS one and j PLUS the VALUE ONE IF character AT position i MINUS one OF floor EQUALS character ONE OTHERWISE ZERO  
                SET dp AT position i and j TO THE MINIMUM OF dp AT position i and j AND dp AT position MAXIMUM OF zero AND i MINUS carpetLen and j MINUS one  
            END FOR  
        END FOR  
        RETURN dp AT position n and numCarpets  
    END FUNCTION  
END CLASS
