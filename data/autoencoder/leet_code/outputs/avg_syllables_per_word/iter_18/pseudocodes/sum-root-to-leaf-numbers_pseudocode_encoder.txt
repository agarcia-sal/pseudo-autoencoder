CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero COMMA left EQUALS none COMMA right EQUALS none)  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  

    SET root TO TreeNode with value at position zero of values  
    SET index_counter TO one  
    SET queue TO a new empty double ended queue  

    APPEND root TO queue  

    WHILE queue holds elements  
        SET node TO the element removed from the front of queue  

        IF index_counter LESS THAN the length of values AND element at position index_counter of values IS NOT none  
            SET node left TO TreeNode with value at position index_counter of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT index_counter BY one  

        IF index_counter LESS THAN the length of values AND element at position index_counter of values IS NOT none  
            SET node right TO TreeNode with value at position index_counter of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT index_counter BY one  
    END WHILE  

    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p COMMA q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p COMMA left child of q) AND is_same_tree(right child of p COMMA right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(root)  
        FUNCTION dfs(node COMMA current_number)  
            IF node IS none  
                RETURN zero  
            END IF  

            SET current_number TO current_number MULTIPLIED BY ten PLUS value of node  

            IF left child of node IS none AND right child of node IS none  
                RETURN current_number  
            END IF  

            SET left_sum TO dfs(left child of node COMMA current_number)  
            SET right_sum TO dfs(right child of node COMMA current_number)  

            RETURN left_sum PLUS right_sum  
        END FUNCTION  

        RETURN dfs(root COMMA zero)  
    END FUNCTION  
END CLASS
