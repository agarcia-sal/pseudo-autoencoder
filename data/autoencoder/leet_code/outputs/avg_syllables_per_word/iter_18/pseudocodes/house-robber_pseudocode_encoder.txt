CLASS Solution  
    FUNCTION rob(nums)  
        IF nums EQUALS an empty collection  
            RETURN zero  
        END IF  
  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
  
        SET dp TO a list of zero repeated the LENGTH OF nums times  
        SET element at position zero of dp TO the element at position zero of nums  
        SET element at position one of dp TO the GREATER OF the element at position zero of nums AND the element at position one of nums  
  
        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET value_one TO the element at position index MINUS one of dp  
            SET value_two TO the element at position index MINUS two of dp PLUS the element at position index of nums  
            SET element at position index of dp TO the GREATER OF value_one AND value_two  
        END FOR  
  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
