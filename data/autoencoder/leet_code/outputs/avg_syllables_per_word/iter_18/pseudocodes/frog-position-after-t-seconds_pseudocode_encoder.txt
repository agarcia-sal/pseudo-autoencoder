CLASS Solution  
    FUNCTION frogPosition WITH parameters n edges t target  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO an empty mapping from keys to lists  
        FOR each pair u and v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
        
        SET queue TO an empty sequence initialized with one tuple consisting OF one integer one one point zero and zero  
        SET visited TO a set containing the integer one  
        
        WHILE queue is not empty  
            REMOVE and ASSIGN the first tuple from queue TO current_node current_probability current_time  
            
            IF current_time EQUALS t OR the integer current_node NOT EQUAL TO one AND the LENGTH OF the list at key current_node in graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to next iteration of the WHILE loop  
                END IF  
            END IF  
            
            SET possible_moves TO the LENGTH OF the list at key current_node in graph MINUS one IF the integer current_node NOT EQUAL TO one OTHERWISE zero  
            
            FOR each neighbor IN the list at key current_node in graph  
                IF neighbor is not in visited  
                    ADD neighbor TO visited  
                    APPEND a tuple consisting OF neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
