CLASS Solution  
    FUNCTION fourSumCount(nums1 AS List of Integer nums2 AS List of Integer nums3 AS List of Integer nums4 AS List of Integer) RETURNS Integer  
        SET sum_ab TO an empty mapping from Integer to Integer  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum IS IN sum_ab  
                    INCREMENT the value AT key current_sum IN sum_ab BY one  
                ELSE  
                    SET the value AT key current_sum IN sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO the negative of the sum of c PLUS d  
                IF target IS IN sum_ab  
                    INCREMENT count BY the value AT key target IN sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
