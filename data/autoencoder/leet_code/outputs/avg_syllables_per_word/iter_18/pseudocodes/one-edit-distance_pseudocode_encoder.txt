CLASS Solution
    FUNCTION isOneEditDistance WITH PARAMETER s AS string AND PARAMETER t AS string RETURNS boolean
        IF the LENGTH OF s GREATER THAN the LENGTH OF t THEN
            RETURN the RESULT OF isOneEditDistance WITH PARAMETERS t AND s
        END IF

        SET diff TO the LENGTH OF t MINUS the LENGTH OF s

        IF diff GREATER THAN one THEN
            RETURN false
        END IF

        IF diff EQUALS zero THEN
            SET differing_character_count TO zero
            FOR each pair of characters c1 AND c2 FROM s AND t RESPECTIVELY
                IF c1 NOT EQUALS c2 THEN
                    INCREMENT differing_character_count BY one
                END IF
            END FOR
            RETURN differing_character_count EQUALS one
        ELSE
            FOR index FROM zero TO the LENGTH OF s MINUS one
                IF the character at position index of s NOT EQUALS the character at position index of t THEN
                    RETURN whether the substring of s from position index TO the END EQUALS the substring of t from position index PLUS one TO the END
                END IF
            END FOR
            RETURN true
        END IF
    END FUNCTION
END CLASS
