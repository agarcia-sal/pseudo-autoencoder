CLASS Solution  
    FUNCTION largestIsland(grid)  
        IF grid DOES NOT EXIST OR element at position zero of grid DOES NOT EXIST  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF grid  
        SET directions TO a collection consisting of element zero and one followed by zero and zero PLUS one PLUS zero and zero MINUS one PLUS zero MINUS one and zero  
        
        FUNCTION is_valid(x, y)  
            RETURN expression x GREATER THAN OR EQUAL TO zero AND x LESS THAN n AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN n  
        END FUNCTION  
        
        FUNCTION dfs(x, y, index)  
            SET area TO one  
            SET element at position x and position y of grid TO index  
            FOR each pair dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF is_valid(nx, ny) AND element at position nx and position ny of grid EQUALS one  
                    INCREMENT area BY the RESULT OF dfs(nx, ny, index)  
                END IF  
            END FOR  
            RETURN area  
        END FUNCTION  
        
        SET island_areas TO an empty mapping  
        SET index TO two  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i and position j of grid EQUALS one  
                    SET mapping value at key index of island_areas TO the RESULT OF dfs(i, j, index)  
                    INCREMENT index BY one  
                END IF  
            END FOR  
        END FOR  
        
        IF island_areas IS EMPTY  
            RETURN one  
        END IF  
        
        SET max_island_size TO the maximal value IN island_areas  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i and position j of grid EQUALS zero  
                    SET neighbors TO an empty collection disallowing duplicates  
                    FOR each pair dx and dy IN directions  
                        SET nx TO i PLUS dx  
                        SET ny TO j PLUS dy  
                        IF is_valid(nx, ny) AND element at position nx and position ny of grid GREATER THAN one  
                            ADD element at position nx and position ny of grid TO neighbors  
                        END IF  
                    END FOR  
                    SET current_island_size TO one PLUS the SUM OF mapping values at keys in neighbors of island_areas  
                    SET max_island_size TO the MAXIMAL value BETWEEN max_island_size AND current_island_size  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_island_size  
    END FUNCTION  
END CLASS
