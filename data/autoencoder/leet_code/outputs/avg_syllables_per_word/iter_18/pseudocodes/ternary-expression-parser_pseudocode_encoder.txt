CLASS Solution  
    FUNCTION parseTernary WITH PARAMETER expression  
        FUNCTION evaluate WITH PARAMETER index  
            IF index EQUALS the LENGTH OF expression MINUS one OR character at position index PLUS one of expression EQUALS colon THEN  
                RETURN character at position index of expression AND index PLUS two  
            END IF  
            SET true_part AND next_index TO the RESULT OF evaluate WITH PARAMETER index PLUS two  
            SET false_part AND final_index TO the RESULT OF evaluate WITH PARAMETER next_index  
            IF character at position index of expression EQUALS capital T THEN  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        SET result AND unused_variable TO the RESULT OF evaluate WITH PARAMETER zero  
        RETURN result  
    END FUNCTION  
END CLASS
