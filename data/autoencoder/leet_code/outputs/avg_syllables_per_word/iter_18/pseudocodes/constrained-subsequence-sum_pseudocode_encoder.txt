CLASS Solution  
    FUNCTION constrainedSubsetSum WITH parameters nums AS List OF integers AND k AS integer  
        SET n TO the LENGTH OF nums  
        SET dp TO a list OF zero repeated n times  
        SET element at position zero of dp TO element at position zero of nums  
        SET max_sum TO element at position zero of nums  
        
        SET q TO a new deque containing element zero  
        
        FOR i FROM one TO n MINUS one  
            IF element at position at front of q in dp IS GREATER THAN zero  
                SET value_to_add TO element at position at front of q in dp  
            ELSE  
                SET value_to_add TO zero  
            END IF  
            SET element at position i of dp TO element at position i of nums PLUS value_to_add  
            
            IF max_sum IS LESS THAN element at position i of dp  
                SET max_sum TO element at position i of dp  
            END IF  
            
            WHILE q IS NOT empty AND element at position i of dp IS GREATER THAN OR EQUAL TO element at position at last of q in dp  
                REMOVE last element FROM q  
            END WHILE  
            
            APPEND i TO q  
            
            IF element at position at front of q EQUALS i MINUS k  
                REMOVE first element FROM q  
            END IF  
        END FOR  
        
        RETURN max_sum  
    END FUNCTION  
END CLASS
