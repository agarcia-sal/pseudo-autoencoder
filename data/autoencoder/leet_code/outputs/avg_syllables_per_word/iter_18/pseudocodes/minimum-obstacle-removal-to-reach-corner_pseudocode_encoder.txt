CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a collection containing the four pairs of values representing zero and one zero and one zero and negative one and negative one and zero respectively  
        SET min_heap TO a collection containing a single tuple of zero zero zero representing cost and x and y respectively  
        SET visited TO a set containing the tuple of zero zero representing the initial coordinates  
        
        WHILE min_heap contains elements  
            REMOVE and RETURN the element with the lowest cost from min_heap ASSIGNING the elements to cost and x and y respectively  
            
            IF the tuple of x and y EQUALS the tuple of m minus one and n minus one  
                RETURN cost  
            END IF  
            
            FOR each dx and dy IN directions  
                SET nx TO the sum of x and dx  
                SET ny TO the sum of y and dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple of nx and ny NOT IN visited  
                    ADD the tuple of nx and ny TO visited  
                    SET new_cost TO the sum of cost and the element at position ny of the element at position nx of grid  
                    INSERT the tuple of new_cost nx ny INTO min_heap maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
