CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val defaulted TO zero left defaulted TO none right defaulted TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values of type list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH argument element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH argument element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH argument element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p IS NOT EQUAL TO value of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree WITH p dot left AND q dot left AND calling is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues WITH PARAMETERS root target k  
        FUNCTION inorder_traversal WITH PARAMETER node  
            IF node IS none  
                RETURN  
            END IF  
            CALL inorder_traversal WITH node dot left  
            IF the LENGTH OF result IS LESS THAN k  
                APPEND value of node TO result  
            ELSE  
                IF the absolute value of value of node MINUS target IS LESS THAN the absolute value of element at position zero of result MINUS target  
                    REMOVE the first element from the left of result  
                    APPEND value of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH node dot right  
        END FUNCTION  
        
        SET result TO an empty double ended queue  
        CALL inorder_traversal WITH root  
        RETURN a list converted from result  
    END FUNCTION  
END CLASS
