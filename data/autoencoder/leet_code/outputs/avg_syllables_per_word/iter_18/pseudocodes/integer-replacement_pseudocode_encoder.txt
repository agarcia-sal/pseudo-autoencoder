CLASS Solution  
    FUNCTION integerReplacement WITH PARAMETER n AS integer  
        SET count TO zero  
        WHILE n IS NOT EQUAL TO one  
            IF the remainder of n DIVIDED BY two EQUALS zero  
                SET n TO the integer division of n BY two  
            ELSE  
                IF n EQUALS three  
                    DECREMENT n BY one  
                ELSE IF the remainder of n PLUS one DIVIDED BY four EQUALS zero  
                    INCREMENT n BY one  
                ELSE  
                    DECREMENT n BY one  
                END IF  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
