CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO A SET CONSTRUCTED FROM wordDict  
        SET dp TO A LIST OF BOOLEAN false VALUES WITH A LENGTH EQUAL TO THE LENGTH OF s PLUS one  
        SET element at position zero OF dp TO boolean true  
        FOR index i FROM one TO THE LENGTH OF s PLUS one  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position j OF dp EQUALS boolean true AND the SUBSTRING OF s FROM position j UP TO BUT NOT INCLUDING position i IS IN wordSet  
                    SET element at position i OF dp TO boolean true  
                    BREAK THE INNER FOR LOOP  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position EQUAL TO THE LENGTH OF s OF dp  
    END FUNCTION  
END CLASS
