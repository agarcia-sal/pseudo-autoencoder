CLASS Solution  
    FUNCTION combine WITH PARAMETERS n AND k  
        FUNCTION backtrack WITH PARAMETERS start AND path  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN from backtrack  
            END IF  
            FOR index i FROM start TO n PLUS one  
                APPEND index i TO path  
                CALL backtrack WITH PARAMETERS index i PLUS one AND path  
                REMOVE the LAST ELEMENT FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH PARAMETERS one AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
