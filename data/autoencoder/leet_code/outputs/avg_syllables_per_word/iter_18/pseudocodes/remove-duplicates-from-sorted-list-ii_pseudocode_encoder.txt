CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty holds  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one to the end  
        SET node TO new instance of ListNode initialized with val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 IS none AND p2 IS none holds  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none holds  
        RETURN false  
    END IF  
    RETURN logical AND of equality of val attribute of p1 AND val attribute of p2 AND result of calling is_same_list with next attribute of p1 AND next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self head)  
        SET dummy TO new instance of ListNode initialized with zero AND head  
        SET prev TO dummy  
        WHILE head IS not none holds  
            IF next attribute of head IS not none AND val attribute of head EQUALS val attribute of next attribute of head holds  
                WHILE next attribute of head IS not none AND val attribute of head EQUALS val attribute of next attribute of head holds  
                    SET head TO next attribute of head  
                END WHILE  
                SET next attribute of prev TO next attribute of head  
            ELSE  
                SET prev TO next attribute of prev  
            END IF  
            SET head TO next attribute of head  
        END WHILE  
        RETURN next attribute of dummy  
    END FUNCTION  
END CLASS
