CLASS Solution  
    FUNCTION minArea WITH PARAMETERS image AS list of list of strings x AS integer y AS integer  
        IF image EQUALS empty OR element at position zero of image EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  
        FUNCTION hasBlackPixelInRow WITH PARAMETER row AS integer  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS the string one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        FUNCTION hasBlackPixelInColumn WITH PARAMETER col AS integer  
            FOR index_row FROM zero TO m MINUS one  
                IF element at position index_row of image AND position col EQUALS the string one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        FUNCTION findTop  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER QUOTIENT OF low PLUS high DIVIDED BY two  
                IF hasBlackPixelInRow WITH mid IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findBottom  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER QUOTIENT OF low PLUS high PLUS one DIVIDED BY two  
                IF hasBlackPixelInRow WITH mid IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO high MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findLeft  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER QUOTIENT OF low PLUS high DIVIDED BY two  
                IF hasBlackPixelInColumn WITH mid IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findRight  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER QUOTIENT OF low PLUS high PLUS one DIVIDED BY two  
                IF hasBlackPixelInColumn WITH mid IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO high MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        SET top TO findTop  
        SET bottom TO findBottom  
        SET left TO findLeft  
        SET right TO findRight  
        SET height TO bottom MINUS top PLUS one  
        SET width TO right MINUS left PLUS one  
        SET area TO height MULTIPLIED BY width  
        RETURN area  
    END FUNCTION  
END CLASS
