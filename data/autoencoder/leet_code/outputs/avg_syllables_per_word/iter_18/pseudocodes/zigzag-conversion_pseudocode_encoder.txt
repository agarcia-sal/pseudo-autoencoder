CLASS Solution  
    FUNCTION convert WITH PARAMETERS s AS string AND numRows AS integer  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO list of empty strings of length numRows  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            SET element at position current_row of rows TO concatenation of element at position current_row of rows AND character  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO logical negation of going_down  
            END IF  
            IF going_down EQUALS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        RETURN concatenation of all elements in rows in sequential order  
    END FUNCTION  
END CLASS
