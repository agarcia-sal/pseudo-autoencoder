CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            SET pick TO the element at position start of slices PLUS dp(start INCREMENTED BY two, end, k DECREMENTED BY one)  
            SET skip TO dp(start INCREMENTED BY one, end, k)  
            RETURN the GREATER VALUE BETWEEN pick AND skip  
        END FUNCTION  
        
        SET case1 TO dp(one, the LENGTH OF slices DECREMENTED BY one, n)  
        SET case2 TO dp(zero, the LENGTH OF slices DECREMENTED BY two, n)  
        
        RETURN the GREATER VALUE BETWEEN case1 AND case2  
    END FUNCTION  
END CLASS
