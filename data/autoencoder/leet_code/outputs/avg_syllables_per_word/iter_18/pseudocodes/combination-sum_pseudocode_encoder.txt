CLASS Solution  
    FUNCTION combinationSum(candidates AS List of integers, target AS integer) AS List of List of integers  
        FUNCTION backtrack(start AS integer, target AS integer, path AS List of integers)  
            IF target EQUALS zero THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero THEN  
                RETURN  
            END IF  
            FOR index FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH index AND target MINUS element at position index of candidates AND path PLUS element at position index of candidates as a new list  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero AND target AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
