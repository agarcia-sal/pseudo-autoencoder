CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET the attribute val of self TO val  
        SET the attribute next of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values OF TYPE list  
    IF the collection values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW instance of ListNode created with the element at position zero of values  
    SET pointer_variable TO head  
    FOR each element val IN the collection consisting of all elements of values starting from position one to the end  
        SET node TO A NEW instance of ListNode created with val  
        SET the attribute next of pointer_variable TO node  
        SET pointer_variable TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF the reference p1 IS none AND the reference p2 IS none  
        RETURN true  
    END IF  
    IF the reference p1 IS none OR the reference p2 IS none  
        RETURN false  
    END IF  
    RETURN the logical AND of the equality comparison of the attribute val of p1 EQUALS the attribute val of p2 AND the recursive call to is_same_list WITH the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS self head OF OPTIONAL ListNode left right  
        IF the reference head IS none OR the equality comparison of left EQUALS right  
            RETURN head  
        END IF  
        
        SET dummy TO A NEW instance of ListNode created with zero  
        SET the attribute next of dummy TO head  
        SET prev TO dummy  
        
        FOR index FROM one TO left MINUS one  
            SET prev TO the attribute next of prev  
        END FOR  
        
        SET current TO the attribute next of prev  
        FOR index FROM one TO right MINUS left  
            SET next_node TO the attribute next of current  
            SET the attribute next of current TO the attribute next of next_node  
            SET the attribute next of next_node TO the attribute next of prev  
            SET the attribute next of prev TO next_node  
        END FOR  
        
        RETURN the attribute next of dummy  
    END FUNCTION  
END CLASS
