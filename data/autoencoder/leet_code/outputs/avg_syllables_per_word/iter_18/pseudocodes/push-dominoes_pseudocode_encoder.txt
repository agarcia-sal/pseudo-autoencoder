CLASS Solution  
    FUNCTION pushDominoes(dominoes IS string) RETURNS string  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zero assigned for each position FROM one TO n  
        
        SET force TO zero  
        FOR index FROM zero TO n MINUS one  
            IF character at position index of dominoes EQUALS the character representing right falling domino  
                SET force TO n  
            ELSE IF character at position index of dominoes EQUALS the character representing left falling domino  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER VALUE BETWEEN force MINUS one AND zero  
            END IF  
            INCREMENT element at position index of forces BY force  
        END FOR  
        
        SET force TO zero  
        FOR index FROM n MINUS one DOWN TO zero  
            IF character at position index of dominoes EQUALS the character representing left falling domino  
                SET force TO n  
            ELSE IF character at position index of dominoes EQUALS the character representing right falling domino  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER VALUE BETWEEN force MINUS one AND zero  
            END IF  
            DECREMENT element at position index of forces BY force  
        END FOR  
        
        SET result TO an empty list  
        FOR each force_value IN forces  
            IF force_value GREATER THAN zero  
                APPEND the character representing right falling domino TO result  
            ELSE IF force_value LESS THAN zero  
                APPEND the character representing left falling domino TO result  
            ELSE  
                APPEND the character representing standing domino TO result  
            END IF  
        END FOR  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
