CLASS Solution  
    FUNCTION minSkips(dist AS List of integers speed AS integer hoursBefore AS integer) RETURNS integer  
        SET n TO the LENGTH OF dist  
        SET dp TO a two-dimensional collection with n rows each containing n PLUS one elements each initialized to mathematical infinity  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at position zero and zero of dp TO element at position zero of dist  
        FOR each integer j FROM one TO n MINUS one  
            SET element at position zero and position j of dp TO element at position zero of dist  
        END FOR  
        
        FOR each integer i FROM one TO n MINUS one  
            FOR each integer j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET element at position i and position j of dp TO the mathematical ceiling OF the quotient of element at position i MINUS one and position j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at position i and position j of dp TO the MINIMUM OF element at position i and position j of dp AND the SUM OF element at position i MINUS one and position j MINUS one of dp AND element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer j FROM zero TO n MINUS one  
            IF element at position n MINUS one and position j of dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
