CLASS Solution  
    FUNCTION minPushBox(grid)  
        SET directions TO a list of tuples consisting of negative one and zero and one and zero and zero and negative one and zero and one  
        FOR index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS character S  
                    SET player_start TO tuple consisting of i and j  
                ELSE IF element at position i of grid at position j EQUALS character B  
                    SET box_start TO tuple consisting of i and j  
                ELSE IF element at position i of grid at position j EQUALS character T  
                    SET target TO tuple consisting of i and j  
                END IF  
            END FOR  
        END FOR  
        ASSIGN reference TO a new deque initialized with a tuple consisting of player_start and box_start and zero  
        SET visited TO an empty set  
        ADD tuple consisting of player_start and box_start TO visited  
        WHILE queue is not empty  
            REMOVE and RETURN the leftmost element from queue and ASSIGN to player_pos and box_pos and push_count  
            IF box_pos EQUALS target  
                RETURN push_count  
            END IF  
            FOR each tuple dx and dy IN directions  
                SET new_player_pos TO tuple consisting of the element at position zero of player_pos PLUS dx and the element at position one of player_pos PLUS dy  
                IF self CALL is_valid_position WITH arguments new_player_pos and grid RETURNS true  
                    IF new_player_pos EQUALS box_pos  
                        SET new_box_pos TO tuple consisting of the element at position zero of box_pos PLUS dx and the element at position one of box_pos PLUS dy  
                        IF self CALL is_valid_position WITH arguments new_box_pos and grid RETURNS true AND tuple consisting of new_player_pos and new_box_pos NOT IN visited  
                            ADD tuple consisting of new_player_pos and new_box_pos TO visited  
                            APPEND tuple consisting of new_player_pos and new_box_pos and push_count PLUS one TO queue  
                        END IF  
                    ELSE  
                        IF tuple consisting of new_player_pos and box_pos NOT IN visited  
                            ADD tuple consisting of new_player_pos and box_pos TO visited  
                            APPEND tuple consisting of new_player_pos and box_pos and push_count TO queue  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
  
    FUNCTION is_valid_position(position, grid)  
        SET x TO element at position zero of position  
        SET y TO element at position one of position  
        RETURN x GREATER THAN OR EQUAL TO zero AND x LESS THAN the LENGTH OF grid AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN the LENGTH OF element at position zero of grid AND element at position x of grid at position y NOT EQUALS character hash  
    END FUNCTION  
END CLASS
