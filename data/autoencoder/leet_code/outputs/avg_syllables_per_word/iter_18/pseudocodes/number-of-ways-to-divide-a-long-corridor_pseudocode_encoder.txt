CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER corridor AS string  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each index and character IN each element and its position IN corridor  
            IF character EQUALS seat symbol  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one IN STEPS OF two  
            SET ways TO ways MULTIPLIED BY the value of element at position index of seats MINUS the value of element at position index MINUS one of seats MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
