CLASS Solution  
    FUNCTION cherryPickup(grid)  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        
        FUNCTION dp(r c1 c2)  
            IF c1 IS LESS THAN zero OR c1 IS GREATER THAN OR EQUAL TO cols OR c2 IS LESS THAN zero OR c2 IS GREATER THAN OR EQUAL TO cols THEN  
                RETURN zero  
            END IF  
            
            IF c1 EQUALS c2 THEN  
                SET cherries TO element at position r at position c1 of grid  
            ELSE  
                SET cherries TO element at position r at position c1 of grid PLUS element at position r at position c2 of grid  
            END IF  
            
            IF r EQUALS rows MINUS one THEN  
                RETURN cherries  
            END IF  
            
            SET max_cherries TO zero  
            FOR each dc1 IN the list consisting of negative one zero and one  
                FOR each dc2 IN the list consisting of negative one zero and one  
                    SET candidate TO dp(r PLUS one c1 PLUS dc1 c2 PLUS dc2)  
                    IF candidate IS GREATER THAN max_cherries THEN  
                        SET max_cherries TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN cherries PLUS max_cherries  
        END FUNCTION  
        
        RETURN dp(zero zero cols MINUS one)  
    END FUNCTION  
END CLASS
