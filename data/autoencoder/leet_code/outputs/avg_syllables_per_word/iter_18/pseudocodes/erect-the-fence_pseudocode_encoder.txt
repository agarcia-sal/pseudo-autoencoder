CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION cross(i j k)  
            SET a TO element at position i of trees  
            SET b TO element at position j of trees  
            SET c TO element at position k of trees  
            RETURN the quantity of the difference of the first element of b and the first element of a MULTIPLIED BY the difference of the second element of c and the second element of b MINUS the difference of the second element of b and the second element of a MULTIPLIED BY the difference of the first element of c and the first element of b  
        END FUNCTION  

        SET n TO the LENGTH OF trees  
        IF n LESS THAN four  
            RETURN trees  
        END IF  

        SORT trees in ascending natural order  
        SET vis TO a list of boolean values False repeated n times  
        SET stk TO a list containing zero as the sole element  
        FOR i FROM one TO n MINUS one  
            WHILE the LENGTH OF stk GREATER THAN one AND the result of cross with arguments as the element at position negative two of stk the element at position negative one of stk and i LESS THAN zero holds  
                SET element at position of the popped element of stk in vis TO False  
            END WHILE  
            SET element at position i in vis TO True  
            APPEND i TO stk  
        END FOR  

        SET m TO the LENGTH OF stk  
        FOR i FROM n MINUS two DOWN TO zero inclusive  
            IF element at position i in vis IS True  
                CONTINUE to next iteration  
            END IF  
            WHILE the LENGTH OF stk GREATER THAN m AND the result of cross with arguments as the element at position negative two of stk the element at position negative one of stk and i LESS THAN zero holds  
                POP the element from stk disregarding value  
            END WHILE  
            APPEND i TO stk  
        END FOR  

        POP the last element from stk disregarding value  
        RETURN the list of elements at positions indicated by each element in stk of trees  
    END FUNCTION  
END CLASS
