CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs EQUALS empty list THEN  
            RETURN zero  
        END IF  
        SET prev_red TO element at position zero of element at position zero of costs  
        SET prev_blue TO element at position one of element at position zero of costs  
        SET prev_green TO element at position two of element at position zero of costs  
        FOR index FROM one TO the length of costs MINUS one  
            SET current_red TO element at position zero of element at position index of costs PLUS the lesser of prev_blue and prev_green  
            SET current_blue TO element at position one of element at position index of costs PLUS the lesser of prev_red and prev_green  
            SET current_green TO element at position two of element at position index of costs PLUS the lesser of prev_red and prev_blue  
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        RETURN the lesser of prev_red and prev_blue and prev_green  
    END FUNCTION  
END CLASS
