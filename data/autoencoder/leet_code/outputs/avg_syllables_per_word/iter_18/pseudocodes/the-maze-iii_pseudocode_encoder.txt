CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze AS list of list of integers ball AS list of integers hole AS list of integers RETURNS string  
        SET directions TO a list of four tuples each containing zero and one or minus one integers representing movements plus a character representing direction right down left up respectively  
        SET number_of_rows TO the length of maze  
        SET number_of_columns TO the length of element at position zero of maze  
        SET start_position TO the tuple conversion of ball  
        SET hole_position TO the tuple conversion of hole  
        SET priority_queue TO a list containing a tuple of zero distance an empty string path and start_position  
        SET visited_positions TO an empty set  
          
        WHILE priority_queue has elements  
            SET current_distance path current_position TO heappop from priority_queue  
            SET current_x TO element at position zero of current_position  
            SET current_y TO element at position one of current_position  
              
            IF current_x equals element at position zero of hole_position AND current_y equals element at position one of hole_position  
                RETURN path  
            END IF  
              
            IF current_position is in visited_positions  
                CONTINUE  
            END IF  
              
            ADD current_position TO visited_positions  
              
            FOR each movement_x movement_y direction_character IN directions  
                SET new_x TO current_x  
                SET new_y TO current_y  
                SET roll_count TO zero  
                  
                WHILE condition holds that new_x plus movement_x is greater than or equal to zero AND new_x plus movement_x is less than number_of_rows AND new_y plus movement_y is greater than or equal to zero AND new_y plus movement_y is less than number_of_columns AND element at position new_x plus movement_x of maze element at position new_y plus movement_y equals zero  
                    SET new_x TO new_x plus movement_x  
                    SET new_y TO new_y plus movement_y  
                    INCREMENT roll_count BY one  
                      
                    IF new_x equals element at position zero of hole_position AND new_y equals element at position one of hole_position  
                        BREAK  
                    END IF  
                END WHILE  
                  
                IF tuple of new_x and new_y is not in visited_positions  
                    PUSH into priority_queue the tuple containing current_distance plus roll_count the concatenation of path and direction_character and tuple of new_x and new_y  
                END IF  
            END FOR  
        END WHILE  
          
        RETURN the string representing impossibility of solution  
    END FUNCTION  
END CLASS
