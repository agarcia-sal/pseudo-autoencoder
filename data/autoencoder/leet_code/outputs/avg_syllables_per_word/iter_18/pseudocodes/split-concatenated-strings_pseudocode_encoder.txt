CLASS Solution  
    FUNCTION splitLoopedString WITH PARAMETER strs AS list of strings  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string at position index of strs REVERSED  
            IF reversed_str GREATER THAN the string at position index of strs THEN  
                SET the string at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs AND strs  
        
        SET max_string TO the empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the list containing false AND true  
                IF is_reversed EQUALS true THEN  
                    SET current TO the string at position index of strs REVERSED  
                ELSE  
                    SET current TO the string at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of the strings at positions from index PLUS one TO index PLUS the LENGTH OF strs MINUS one of doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring of current FROM cut_position TO the end PLUS remaining PLUS the substring of current FROM zero TO cut_position MINUS one  
                    IF candidate GREATER THAN max_string THEN  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
