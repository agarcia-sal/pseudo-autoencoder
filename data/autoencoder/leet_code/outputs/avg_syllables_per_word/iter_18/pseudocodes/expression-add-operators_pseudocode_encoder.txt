CLASS Solution  
    FUNCTION addOperators(num AS string, target AS integer) RETURNS list of string  
        FUNCTION dfs(index AS integer, path AS string, value AS integer, prev AS integer)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO the SUBSTRING OF num FROM index TO i MINUS one  
                IF the LENGTH OF operand GREATER THAN one AND the CHARACTER AT position one OF operand EQUALS zero  
                    CONTINUE  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH arguments i operand CONVERTED TO integer operand CONVERTED TO integer  
                ELSE  
                    CALL dfs WITH arguments i CONCATENATION OF path PLUS SIGN PLUS operand value PLUS operand CONVERTED TO integer operand CONVERTED TO integer  
                    CALL dfs WITH arguments i CONCATENATION OF path MINUS SIGN MINUS operand value MINUS operand CONVERTED TO integer NEGATIVE operand CONVERTED TO integer  
                    CALL dfs WITH arguments i CONCATENATION OF path MULTIPLICATION SIGN MULTIPLIED BY operand value MINUS prev PLUS prev MULTIPLIED BY operand CONVERTED TO integer prev MULTIPLIED BY operand CONVERTED TO integer  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH arguments zero empty string zero zero  
        RETURN result  
    END FUNCTION  
END CLASS
