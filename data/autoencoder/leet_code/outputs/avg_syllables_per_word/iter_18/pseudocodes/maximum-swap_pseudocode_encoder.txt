CLASS Solution  
    FUNCTION maximumSwap(num AS integer) AS integer  
        SET digits TO the list of characters from the string representation of num  
        SET last TO an empty mapping from integer to integer  
        FOR each position index and character digit IN the enumeration of digits  
            SET the key converted digit at position index IN last TO position index  
        END FOR  
        FOR each position i and character d IN the enumeration of digits  
            FOR each candidate digit d2 FROM nine DOWN TO one GREATER THAN the integer value of d  
                IF the value for key d2 in last EXISTS AND the value for key d2 in last IS GREATER THAN i THEN  
                    SET temporary variable to element at position i of digits  
                    SET element at position i of digits TO element at position value for key d2 in last of digits  
                    SET element at position value for key d2 in last of digits TO temporary variable  
                    RETURN the integer formed by concatenating all characters in digits in sequence  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
