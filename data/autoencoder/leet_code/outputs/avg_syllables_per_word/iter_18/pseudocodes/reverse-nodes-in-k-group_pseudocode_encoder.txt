CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one TO the last element  
        SET node TO a new instance of ListNode initialized with val  
        SET the attribute next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the comparison of the attribute val of p1 EQUALS the attribute val of p2 AND the recursive call of is_same_list with the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(head k)  
        FUNCTION reverseLinkedList(start end)  
            SET prev TO None  
            SET curr TO start  
            WHILE the reference curr does NOT EQUAL the reference end  
                SET next_node TO the attribute next of curr  
                SET the attribute next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the attribute next of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO a new instance of ListNode initialized with zero  
        SET the attribute next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True holds  
            SET kth_node TO prev_group_end  
            FOR each _ IN the collection of the range of k elements starting from zero TO one less than k  
                SET kth_node TO the attribute next of kth_node  
                IF kth_node EQUALS None  
                    RETURN the attribute next of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the attribute next of kth_node  
            SET the attribute next of kth_node TO None  
            SET new_group_start AND new_group_end TO the result of calling reverseLinkedList with the attribute next of prev_group_end AND kth_node  
            SET the attribute next of prev_group_end TO new_group_start  
            SET the attribute next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the attribute next of dummy  
    END FUNCTION  
END CLASS
