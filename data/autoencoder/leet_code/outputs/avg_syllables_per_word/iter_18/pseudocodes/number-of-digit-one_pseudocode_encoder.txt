CLASS Solution  
    FUNCTION countDigitOne WITH parameter n  
        IF n IS LESS THAN OR EQUAL TO zero THEN  
            RETURN zero  
        END IF  

        SET count TO zero  
        SET factor TO one  

        WHILE factor IS LESS THAN OR EQUAL TO n DO  
            SET lower_nums TO n MINUS the INTEGER DIVISION of n BY factor MULTIPLIED BY factor  
            SET cur_num TO the INTEGER DIVISION of n BY factor MODULO ten  
            SET higher_nums TO the INTEGER DIVISION of n BY factor MULTIPLIED BY ten  

            IF cur_num EQUALS zero THEN  
                INCREMENT count BY higher_nums MULTIPLIED BY factor  
            ELSE IF cur_num EQUALS one THEN  
                INCREMENT count BY higher_nums MULTIPLIED BY factor PLUS lower_nums PLUS one  
            ELSE  
                INCREMENT count BY higher_nums PLUS one MULTIPLIED BY factor  
            END IF  

            SET factor TO factor MULTIPLIED BY ten  
        END WHILE  

        RETURN count  
    END FUNCTION  
END CLASS
