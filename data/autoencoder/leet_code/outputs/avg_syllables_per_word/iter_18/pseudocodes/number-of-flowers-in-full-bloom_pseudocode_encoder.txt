CLASS Solution  
    FUNCTION fullBloomFlowers(flowers AS List OF List OF integers, people AS List OF integers) RETURNS List OF integers  
        SET start_times TO the sorted list of the first element in each flower time pair in flowers  
        SET end_times TO the sorted list of the second element in each flower time pair in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the number of elements in start_times that are less than or equal to person  
            SET ended TO the number of elements in end_times that are less than person  
            APPEND the difference of started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
