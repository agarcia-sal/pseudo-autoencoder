CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH PARAMETER s  
        SET sequence_count TO a new mapping with default value zero for any key  
        FOR index FROM zero TO the LENGTH OF s MINUS ten PLUS one  
            SET sequence TO the substring starting at position index of s with length ten  
            INCREMENT the count at key sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO a new empty list  
        FOR each key and value IN sequence_count  
            IF value GREATER THAN one THEN  
                APPEND key TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
