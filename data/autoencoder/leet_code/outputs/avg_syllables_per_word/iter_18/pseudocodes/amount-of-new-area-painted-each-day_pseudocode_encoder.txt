CLASS Solution  
    FUNCTION amountPainted WITH PARAMETER paint AS List OF List OF integer  
        SET max_end TO the greatest end value found by examining each pair of start and end in paint  
        SET painted TO a list containing the value negative one repeated as many times as max_end incremented by one  
        SET worklog TO an empty list  
        
        FOR each start and end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            
            WHILE the value of current_start is less than the value of end  
                IF the element at position current_start of painted EQUALS negative one  
                    SET the element at position current_start of painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO the element at position current_start of painted  
                END IF  
            END WHILE  
            
            APPEND new_paint TO worklog  
        END FOR  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
