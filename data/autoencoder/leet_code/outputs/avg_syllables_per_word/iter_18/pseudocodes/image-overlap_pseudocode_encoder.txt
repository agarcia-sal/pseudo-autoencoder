CLASS Solution  
    FUNCTION largestOverlap(img1 AS List OF List OF integer, img2 AS List OF List OF integer) RETURNS integer  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img1 EQUALS one  
                    APPEND pair of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img2 EQUALS one  
                    APPEND pair of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a new mapping with default value zero for each key  
        FOR each pair x1 AND y1 IN points1  
            FOR each pair x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value in overlap_count at key pair of dx AND dy BY one  
            END FOR  
        END FOR  
        RETURN the maximum value AMONG the values in overlap_count OR zero IF overlap_count IS empty  
    END FUNCTION  
END CLASS
