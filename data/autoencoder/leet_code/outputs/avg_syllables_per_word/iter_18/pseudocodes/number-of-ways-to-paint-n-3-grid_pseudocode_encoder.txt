CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n OF TYPE integer  
        SET MOD TO one billion PLUS seven  
        SET variable a TO six  
        SET variable b TO six  
        FOR each integer index FROM two TO n  
            SET temporary variable new_a TO three MULTIPLIED BY variable a PLUS two MULTIPLIED BY variable b MODULO MOD  
            SET temporary variable new_b TO two MULTIPLIED BY variable a PLUS two MULTIPLIED BY variable b MODULO MOD  
            SET variable a TO new_a  
            SET variable b TO new_b  
        END FOR  
        SET result TO variable a PLUS variable b MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
