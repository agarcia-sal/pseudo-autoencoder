CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO CALL defaultdict WITH list  
        FOR each path IN paths  
            SET parts TO the list OF substrings FROM path SPLIT BY whitespace  
            SET directory TO element at position one OF parts  
            FOR each file_info IN elements of parts FROM position two TO the end  
                SET file_name TO the substring BEFORE the first opening parenthesis OF file_info  
                SET content TO the substring AFTER the first opening parenthesis AND BEFORE the last character OF file_info  
                APPEND the concatenation OF directory CONCATENATED WITH a slash CONCATENATED WITH file_name TO the list IN content_to_paths WITH content AS key  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each paths_group IN the values OF content_to_paths  
            IF the length OF paths_group GREATER THAN one THEN  
                APPEND paths_group TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
