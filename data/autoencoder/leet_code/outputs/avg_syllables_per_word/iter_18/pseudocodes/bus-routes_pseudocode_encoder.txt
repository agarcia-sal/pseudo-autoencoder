CLASS Solution  
    FUNCTION numBusesToDestination WITH PARAMETERS routes AS List of List OF Integer source AS Integer target AS Integer  
        IF source EQUALS target THEN  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a mapping WITH DEFAULT VALUE empty list  
        FOR each bus WITH INDEX bus_index IN routes  
            FOR each stop IN route at position bus_index of routes  
                APPEND bus_index TO list at key stop in stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO an empty double ended queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus IN list at key source in stop_to_buses  
            APPEND tuple of bus and integer one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue holds any element  
            REMOVE and STORE the first element of bus_queue INTO current_bus and bus_count  

            FOR each stop IN route at position current_bus of routes  
                IF stop EQUALS target THEN  
                    RETURN bus_count  
                END IF  

                IF stop is NOT IN visited_stops THEN  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN list at key stop in stop_to_buses  
                        IF next_bus is NOT IN visited_buses THEN  
                            ADD next_bus TO visited_buses  
                            APPEND tuple of next_bus and bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
