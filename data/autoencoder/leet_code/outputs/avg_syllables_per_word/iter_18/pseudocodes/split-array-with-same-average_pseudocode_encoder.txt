CLASS Solution  
    FUNCTION splitArraySameAverage(nums AS List OF integers) RETURNS boolean  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF elements IN nums  
        
        IF n EQUALS one THEN  
            RETURN false  
        END IF  
        
        INITIALIZE dp AS list OF SETS with size equal TO the integer DIVIDED BY two of n PLUS one  
        ADD zero TO the element at position zero of dp  
        
        FOR each num IN nums  
            FOR size FROM the integer DIVIDED BY two of n DOWN TO one  
                FOR each s IN the element at position size MINUS one of dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum THEN  
                        RETURN true  
                    END IF  
                    ADD new_sum TO the element at position size of dp  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
