CLASS Solution  
    FUNCTION subdomainVisits WITH PARAMETER cpdomains AS List of String  
        SET domain_count TO new defaultdict with integer default value  
        
        FOR each cpdomain IN cpdomains  
            SET count and domain TO the result of splitting cpdomain into two parts separated by whitespace  
            SET count TO count converted to integer  
            SET subdomains TO the result of splitting domain by the character full stop  
            
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain TO the concatenation joined by full stop of the elements of subdomains from position index to the end  
                INCREMENT the value referenced by key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO empty list  
        FOR each domain and count IN the items of domain_count  
            APPEND the string formed by concatenating count followed by a single space followed by domain TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
