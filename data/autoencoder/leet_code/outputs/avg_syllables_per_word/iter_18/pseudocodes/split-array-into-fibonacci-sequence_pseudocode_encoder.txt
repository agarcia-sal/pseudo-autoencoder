CLASS Solution  
    FUNCTION splitIntoFibonacci WITH parameter num  
        FUNCTION backtrack WITH parameters start AND path  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path IS GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the character at position start of num EQUALS the character zero AND end IS GREATER THAN start PLUS one  
                    CONTINUE to the next iteration  
                END IF  
                SET next_num TO the integer represented by the substring from position start UP TO position end of num  
                IF next_num IS GREATER THAN OR EQUAL TO two RAISED TO THE power of thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path IS LESS THAN two OR next_num EQUALS the element at the last position of path PLUS the element at the position two BEFORE last of path  
                    SET result TO the value RETURNED BY backtrack CALLED WITH end AND path APPENDED BY next_num  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN the value RETURNED BY backtrack CALLED WITH zero AND an empty list  
    END FUNCTION  
END CLASS
