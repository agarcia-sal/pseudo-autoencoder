CLASS Solution  
    FUNCTION removeKdigits(num AS string, k AS integer) RETURNS string  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at last position of stack GREATER THAN digit  
                REMOVE element at last position FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero  
            SET final_stack TO subset of stack consisting of all elements from position one TO the LENGTH OF stack MINUS k  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET concatenated_result TO the concatenation of all elements in final_stack in order  
        SET result TO concatenated_result WITH all leading zero characters REMOVED  
        
        IF result IS empty string  
            RETURN the string representation of zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
