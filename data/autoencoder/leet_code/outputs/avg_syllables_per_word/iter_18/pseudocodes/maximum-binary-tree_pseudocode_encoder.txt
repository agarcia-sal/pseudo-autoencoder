CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val WITH DEFAULT zero left WITH DEFAULT None right WITH DEFAULT None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element at position zero popped from queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS self nums  
        IF nums EQUALS empty  
            RETURN None  
        END IF  
        SET max_val TO the maximum element in nums  
        SET max_index TO the position of max_val IN nums  
        SET root TO NEW TreeNode WITH PARAMETER max_val  
        SET root dot left TO self dot constructMaximumBinaryTree WITH PARAMETER elements of nums from position zero UP TO but NOT INCLUDING max_index  
        SET root dot right TO self dot constructMaximumBinaryTree WITH PARAMETER elements of nums from position max_index PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
