CLASS Solution  
    FUNCTION canCross(stones AS list of integers) RETURNS boolean  
        IF stones IS empty OR the LENGTH OF stones IS less than two THEN  
            RETURN false  
        END IF  
        
        SET stone_set TO the conversion of stones INTO a set for constant time membership verification  
        SET memo TO an empty mapping for memorization of visited states  
        
        FUNCTION can_jump_to(position AS integer, jump AS integer) RETURNS boolean  
            IF position EQUALS the element at position the LENGTH OF stones MINUS one of stones THEN  
                RETURN true  
            END IF  
            
            IF the tuple of position AND jump IS a key in memo THEN  
                RETURN the value in memo for the key of the tuple of position AND jump  
            END IF  
            
            FOR each next_jump FROM jump MINUS one TO jump PLUS one DO  
                IF next_jump IS greater than zero AND the sum of position AND next_jump IS in stone_set THEN  
                    IF can_jump_to(the sum of position AND next_jump, next_jump) RETURNS true THEN  
                        SET the value in memo for the key of the tuple of position AND jump TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value in memo for the key of the tuple of position AND jump TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_jump_to(the element at position zero of stones, zero)  
    END FUNCTION  
END CLASS
