CLASS Solution  
    FUNCTION killProcess(pid AS List of integers ppid AS List of integers kill AS integer) RETURNS List of integers  
        SET tree TO a new mapping with default list for each key  
        FOR each child and parent IN pairing each element at respective positions of pid AND ppid  
            APPEND child TO the list at key parent in tree  
        END FOR  
        SET queue TO a new list containing kill  
        SET killed_processes TO a new empty list  
        WHILE queue contains at least one element  
            SET current TO the element at the first position of queue  
            REMOVE the element at the first position of queue  
            APPEND current TO killed_processes  
            FOR each child IN the list at key current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
