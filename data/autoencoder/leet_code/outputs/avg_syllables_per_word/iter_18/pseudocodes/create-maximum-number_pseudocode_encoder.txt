CLASS Solution  
    FUNCTION maxNumber(nums1 AS List OF Integer nums2 AS List OF Integer k AS Integer) RETURNS List OF Integer  
        FUNCTION maxSingleArray(nums AS List OF Integer k AS Integer) RETURNS List OF Integer  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND the LENGTH OF stack GREATER THAN zero AND the element at the last position of stack LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  

        FUNCTION merge(arr1 AS List OF Integer arr2 AS List OF Integer) RETURNS List OF Integer  
            SET merged_list TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR count FROM one TO total_length  
                IF the sequence arr1 GREATER THAN the sequence arr2 in lexicographical order  
                    REMOVE the element at position zero FROM arr1 AND APPEND it TO merged_list  
                ELSE  
                    REMOVE the element at position zero FROM arr2 AND APPEND it TO merged_list  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  

        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merge APPLIED TO maxSingleArray APPLIED TO nums1 AND i AND maxSingleArray APPLIED TO nums2 AND k MINUS i  
                IF candidate GREATER THAN max_number in lexicographical order  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
