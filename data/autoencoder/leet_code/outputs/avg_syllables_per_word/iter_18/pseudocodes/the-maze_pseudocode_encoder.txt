CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze AS list of list of integer start AS list of integer destination AS list of integer RETURNS boolean  
        SET directions TO list of tuples representing zero and one then one and zero then zero and negative one then negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs WITH PARAMETERS x AS integer y AS integer RETURNS boolean  
            IF the tuple of x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF the list containing x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            APPEND the tuple of x and y TO visited  
            
            FOR each tuple dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND the element at position nx PLUS dx of maze AT position ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                
                IF dfs WITH ARGUMENTS nx AND ny RETURNS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN dfs WITH ARGUMENTS the element at position zero of start AND the element at position one of start  
    END FUNCTION  
END CLASS
