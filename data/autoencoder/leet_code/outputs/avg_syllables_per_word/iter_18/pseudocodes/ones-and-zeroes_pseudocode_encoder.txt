CLASS Solution  
    FUNCTION findMaxForm(strs AS list OF string COMMA m AS integer COMMA n AS integer) RETURNS integer  
        SET dp TO a two dimensional array of zeroes with dimensions m PLUS one BY n PLUS one  
        FOR each string_element IN strs  
            SET count_0 TO the number of characters equal to zero within string_element  
            SET count_1 TO the number of characters equal to one within string_element  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i and j TO the greater of dp element at position i and j AND dp element at position i minus count_0 and j minus count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
