CLASS Solution  
    FUNCTION findPermutation(s AS string) RETURNS list of integers  
        SET result TO an empty list  
        FOR each index IN the range from zero TO the length of s PLUS one  
            APPEND the index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR each end_index IN the range from zero TO the length of s  
            IF end_index EQUALS the length of s OR the character at position end_index of s EQUALS the uppercase letter I THEN  
                CALL auxiliary FUNCTION reverseSegment WITH parameters result start end_index INCLUSIVE  
                SET start TO end_index PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
  
    FUNCTION reverseSegment(collection AS list of integers start_position AS integer end_position AS integer INCLUSIVE)  
        SET left_pointer TO start_position  
        SET right_pointer TO end_position  
        WHILE left_pointer IS LESS THAN right_pointer  
            SET temporary_value TO element at position left_pointer of collection  
            SET element at position left_pointer of collection TO element at position right_pointer of collection  
            SET element at position right_pointer of collection TO temporary_value  
            INCREMENT left_pointer BY one  
            DECREMENT right_pointer BY one  
        END WHILE  
    END FUNCTION  
END CLASS
