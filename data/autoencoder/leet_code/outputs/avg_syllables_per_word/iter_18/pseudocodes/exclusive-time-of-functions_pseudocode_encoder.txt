CLASS Solution  
    FUNCTION exclusiveTime(n AS integer_param, logs AS list_of_strings_param) RETURNS list_of_integers  
        SET exclusive_time TO a list containing zero repeated n times  
        SET stack TO an empty list  
        FOR each log_entry IN logs  
            SET function_id_text, action_type, timestamp_text TO the three parts obtained by splitting log_entry by the character colon  
            SET function_id TO the integer conversion of function_id_text  
            SET timestamp TO the integer conversion of timestamp_text  
            IF action_type EQUALS the word start  
                IF stack is not empty  
                    SET previous_function_id, previous_timestamp TO the element at the last position of stack  
                    INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp  
                END IF  
                APPEND the tuple consisting of function_id and timestamp TO stack  
            ELSE  
                SET previous_function_id, previous_timestamp TO the element popped from the end of stack  
                INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp PLUS one  
                IF stack is not empty  
                    SET the element at the last position of stack TO a tuple consisting of the first element of that tuple and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
