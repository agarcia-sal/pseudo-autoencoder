CLASS Solution  
    FUNCTION scheduleCourse(courses AS List OF List OF integer) RETURNS integer  
        CALL courses SORTED BY the element at position two of each course  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each course IN courses  
            SET duration TO the element at position one of course  
            SET last_day TO the element at position two of course  
            IF the sum of current_end_day AND duration IS LESS THAN OR EQUAL TO last_day  
                CALL pushing negative duration INTO max_heap USING heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND the negative of the element at position one of max_heap IS GREATER THAN duration  
                INCREMENT current_end_day BY the sum of duration AND the result of popping the largest element from max_heap USING heap pop operation  
                CALL pushing negative duration INTO max_heap USING heap push operation  
            END IF  
        END FOR  
        RETURN the length of max_heap  
    END FUNCTION  
END CLASS
