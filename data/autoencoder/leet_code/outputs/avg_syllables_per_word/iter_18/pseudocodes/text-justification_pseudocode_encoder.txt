CLASS Solution  
    FUNCTION fullJustify WITH parameters words list of strings AND maxWidth integer  
        FUNCTION justify_line WITH parameters line list of strings AND num_of_letters integer AND is_last_line boolean  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN the concatenation of the elements of line separated by a single space PLUS a number of spaces EQUAL TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR each index FROM zero TO num_of_gaps MINUS one  
                    IF index IS LESS THAN extra_spaces  
                        APPEND spaces_between_words PLUS one spaces TO element at position index of line  
                    ELSE  
                        APPEND spaces_between_words spaces TO element at position index of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements of line with no additional separator  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line IS GREATER THAN maxWidth  
                APPEND the result of justify_line WITH line AND num_of_letters AND false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        APPEND the result of justify_line WITH line AND num_of_letters AND true TO result  
        RETURN result  
    END FUNCTION  
END CLASS
