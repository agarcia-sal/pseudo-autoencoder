CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n k maxPts  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN the integer value of i LESS THAN OR EQUAL TO n  
            END IF  
            IF i EQUALS k MINUS one THEN  
                RETURN the minimum BETWEEN n MINUS k PLUS one AND maxPts DIVIDED BY maxPts  
            END IF  
            RETURN dfs CALLED WITH i PLUS one PLUS the difference BETWEEN dfs CALLED WITH i PLUS one AND dfs CALLED WITH i PLUS maxPts PLUS one DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs CALLED WITH zero  
END CLASS
