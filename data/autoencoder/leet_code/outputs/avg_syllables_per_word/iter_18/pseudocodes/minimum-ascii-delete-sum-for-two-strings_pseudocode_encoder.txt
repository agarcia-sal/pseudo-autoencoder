CLASS Solution  
    FUNCTION minimumDeleteSum WITH PARAMETERS s1 s2  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        CALL auxiliary FUNCTION initialize_dp_table WITH PARAMETERS m n  
        SET dp TO the RESULT OF the above CALL  
        FOR j FROM one TO n  
            SET element at position j of row zero of dp TO element at position j minus one of row zero of dp PLUS ASCII VALUE OF character at position j minus one of s2  
        END FOR  
        FOR i FROM one TO m  
            SET element at position zero of row i of dp TO element at position zero of row i minus one of dp PLUS ASCII VALUE OF character at position i minus one of s1  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i minus one of s1 EQUALS character at position j minus one of s2  
                    SET element at position j of row i of dp TO element at position j minus one of row i minus one of dp  
                ELSE  
                    SET temporary_cost_one TO element at position j of row i minus one of dp PLUS ASCII VALUE OF character at position i minus one of s1  
                    SET temporary_cost_two TO element at position j minus one of row i of dp PLUS ASCII VALUE OF character at position j minus one of s2  
                    SET element at position j of row i of dp TO the MINIMUM OF temporary_cost_one AND temporary_cost_two  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of row m of dp  
    END FUNCTION  
END CLASS
