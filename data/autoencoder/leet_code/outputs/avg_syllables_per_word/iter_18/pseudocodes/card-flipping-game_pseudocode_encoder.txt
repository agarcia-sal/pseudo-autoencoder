CLASS Solution  
    FUNCTION flipgame WITH PARAMETERS fronts LIST OF INTEGER AND backs LIST OF INTEGER  
        SET same TO the SET OF each element x FOR each pair x and y FROM the ZIP of fronts AND backs WHERE x EQUALS y  
        SET all_numbers TO the SET OF all elements IN the concatenation of fronts PLUS backs  
        SET good_numbers TO the SET OF all_numbers MINUS the SET OF same  
        IF good_numbers IS NOT EMPTY  
            SET minimum_good_number TO the MINIMUM VALUE IN good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
