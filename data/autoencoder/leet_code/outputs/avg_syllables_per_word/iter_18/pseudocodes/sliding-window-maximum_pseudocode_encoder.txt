CLASS Solution  
    FUNCTION maxSlidingWindow(nums AS List of Integers, k AS Integer) RETURNS List of Integers  
        IF nums IS empty OR k EQUALS zero THEN  
            RETURN empty list  
        END IF  
        
        SET deq TO an empty double-ended queue  
        SET result TO an empty list  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN index MINUS k PLUS one THEN  
                REMOVE the element at position zero FROM deq  
            END IF  
            
            WHILE deq IS NOT empty AND the element at position element at the last position of deq of nums IS LESS THAN the element at position index of nums DO  
                REMOVE the element at the last position FROM deq  
            END WHILE  
            
            APPEND index TO deq  
            
            IF index IS GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
