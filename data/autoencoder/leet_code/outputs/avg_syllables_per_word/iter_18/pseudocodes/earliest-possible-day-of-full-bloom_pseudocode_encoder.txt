CLASS Solution  
    FUNCTION earliestFullBloom WITH PARAMETERS plantTime AND growTime  
        SET tasks TO the RESULT OF auxiliary FUNCTION zip_and_sort_by_descending_growTime WITH PARAMETERS plantTime AND growTime  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each plant AND grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
