CLASS Solution  
    FUNCTION leastBricks WITH parameter wall AS list of list of integers  
        SET gap_counts TO an empty mapping that provides zero for missing keys  
        
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row EXCEPT the last one  
                INCREMENT total BY brick  
                INCREMENT the value at key total in gap_counts BY one  
            END FOR  
        END FOR  
        
        SET max_gaps TO the greatest value among all values in gap_counts defaulting to zero if gap_counts is empty  
        
        SET total_rows TO the count of elements in wall  
        
        SET minimum_crossed_bricks TO total_rows MINUS max_gaps  
        
        RETURN minimum_crossed_bricks  
    END FUNCTION  
END CLASS
