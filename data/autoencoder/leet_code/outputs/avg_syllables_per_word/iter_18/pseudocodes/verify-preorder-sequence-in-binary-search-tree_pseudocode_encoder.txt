CLASS Solution  
    FUNCTION verifyPreorder(preorder AS List OF integers) RETURNS boolean  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  
        
        FOR each value IN preorder  
            IF value IS LESS THAN lower_bound  
                RETURN false  
            END IF  
            
            WHILE stack IS NOT empty AND value IS GREATER THAN the element at position of the last index of stack  
                SET lower_bound TO the element REMOVED FROM stack at the last position  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
