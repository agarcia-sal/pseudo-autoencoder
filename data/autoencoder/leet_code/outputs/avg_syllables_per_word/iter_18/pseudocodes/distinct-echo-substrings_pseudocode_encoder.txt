CLASS Solution  
    FUNCTION distinctEchoSubstrings WITH PARAMETER text  
        SET n TO the LENGTH OF text  
        SET substrings TO an EMPTY SET  
        FOR length FROM one TO the INTEGER DIVISION OF n BY two PLUS one MINUS one  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one MINUS one  
                SET a TO the SUBSTRING OF text BEGINNING AT POSITION start AND LENGTH length  
                SET b TO the SUBSTRING OF text BEGINNING AT POSITION start PLUS length AND LENGTH length  
                IF a EQUALS b THEN  
                    APPEND the CONCATENATION OF a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF substrings  
    END FUNCTION  
END CLASS
