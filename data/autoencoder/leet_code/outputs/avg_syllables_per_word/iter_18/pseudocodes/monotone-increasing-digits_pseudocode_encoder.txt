CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list OF characters extracted from the string representation OF n  
        SET length TO the number OF elements IN digits  
        SET marker TO length  
        FOR index FROM length MINUS one DOWN TO one INCLUSIVE  
            IF the element at position index OF digits IS LESS THAN the element at position index MINUS one OF digits THEN  
                SET element at position index MINUS one OF digits TO the string representation OF the integer conversion OF the element at position index MINUS one OF digits MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one INCLUSIVE  
            SET element at position index OF digits TO the string character representing nine  
        END FOR  
        SET result_string TO the concatenation OF all elements IN digits IN ORDER  
        SET result_integer TO the integer conversion OF result_string  
        RETURN result_integer  
    END FUNCTION  
END CLASS
