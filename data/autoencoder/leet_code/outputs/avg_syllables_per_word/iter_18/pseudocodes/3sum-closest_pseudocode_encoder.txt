CLASS Solution  
    FUNCTION threeSumClosest(nums AS List OF Integer, target AS Integer) RETURNS Integer  
        CALL nums TO SORTED IN ASCENDING ORDER  
        SET closest_sum TO positive infinity  
          
        FOR index FROM zero TO the LENGTH OF nums MINUS two MINUS one  
            SET left TO index PLUS one  
            SET right TO the LENGTH OF nums MINUS one  
              
            WHILE left IS LESS THAN right  
                SET current_sum TO the element at position index of nums PLUS the element at position left of nums PLUS the element at position right of nums  
                  
                IF the absolute value of current_sum MINUS target IS LESS THAN the absolute value of closest_sum MINUS target  
                    SET closest_sum TO current_sum  
                END IF  
                  
                IF current_sum IS LESS THAN target  
                    INCREMENT left BY one  
                ELSE IF current_sum IS GREATER THAN target  
                    DECREMENT right BY one  
                ELSE  
                    RETURN current_sum  
                END IF  
            END WHILE  
        END FOR  
          
        RETURN closest_sum  
    END FUNCTION  
END CLASS
