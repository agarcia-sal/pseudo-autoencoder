CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs WITH PARAMETER nums which is a list of integers  
        SET max_num TO the greatest element in nums  
        SET num_set TO the set containing each unique element in nums  
        SET count TO zero  
        
        FOR each integer g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            FOR each integer m STARTING FROM g TO max_num INCLUSIVE IN STEPS OF g  
                IF integer m IS contained in num_set THEN  
                    SET gcd_val TO the greatest common divisor of gcd_val AND m  
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY one  
                        BREAK out of the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
