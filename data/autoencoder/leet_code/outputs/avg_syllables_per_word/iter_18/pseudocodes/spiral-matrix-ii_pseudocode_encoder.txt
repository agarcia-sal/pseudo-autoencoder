CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO auxiliary_initialize_matrix_of_size_n_by_n_filled_with_zeroes  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                SET element at position top and position i of matrix TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                SET element at position i and position right of matrix TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left  
                    SET element at position bottom and position i of matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top  
                    SET element at position i and position left of matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
