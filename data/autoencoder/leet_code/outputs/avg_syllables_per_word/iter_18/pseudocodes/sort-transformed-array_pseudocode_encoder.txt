CLASS Solution  
    FUNCTION sortTransformedArray(nums AS List OF integers a AS integer b AS integer c AS integer) RETURNS List OF integers  
        FUNCTION quadratic(x AS integer) RETURNS integer  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
  
        SET n TO the LENGTH OF nums  
        SET result TO a list OF zero REPEATED n TIMES  
        SET left TO zero  
        SET right TO n MINUS one  
  
        IF a GREATER THAN OR EQUAL TO zero HOLDS  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right HOLDS  
                IF quadratic OF element at position left of nums GREATER THAN quadratic OF element at position right of nums HOLDS  
                    SET element at position index of result TO quadratic OF element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic OF element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right HOLDS  
                IF quadratic OF element at position left of nums LESS THAN quadratic OF element at position right of nums HOLDS  
                    SET element at position index of result TO quadratic OF element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic OF element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
  
        RETURN result  
    END FUNCTION  
END CLASS
