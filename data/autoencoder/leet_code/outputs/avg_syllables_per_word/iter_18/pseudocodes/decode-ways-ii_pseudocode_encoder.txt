CLASS Solution  
    FUNCTION numDecodings WITH PARAMETER s  
        SET MODULO_CONSTANT TO ten raised to the power of nine PLUS one  
  
        SET dynamic_programming_list TO a list of zeros with the length OF the LENGTH OF s PLUS one  
        SET the element at position zero of dynamic_programming_list TO one  
  
        FUNCTION ways_to_decode_single WITH PARAMETER character  
            IF character EQUALS the wildcard character  
                RETURN nine  
            ELSE IF character EQUALS zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
  
        FUNCTION ways_to_decode_pair WITH PARAMETERS character_one AND character_two  
            IF character_one EQUALS the wildcard character AND character_two EQUALS the wildcard character  
                RETURN fifteen  
            ELSE IF character_one EQUALS the wildcard character  
                IF character_two LESS THAN OR EQUAL TO six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF character_two EQUALS the wildcard character  
                IF character_one EQUALS one  
                    RETURN nine  
                ELSE IF character_one EQUALS two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET combined_number TO the integer value CORRESPONDING TO the concatenation OF character_one AND character_two  
                IF combined_number GREATER THAN OR EQUAL TO ten AND combined_number LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
  
        FOR index FROM one TO the LENGTH OF s  
            SET the element at position index of dynamic_programming_list TO the element at position index of dynamic_programming_list PLUS the element at position index MINUS one of dynamic_programming_list MULTIPLIED BY ways_to_decode_single OF the character at position index MINUS one of s  
            SET the element at position index of dynamic_programming_list TO the element at position index of dynamic_programming_list MODULO MODULO_CONSTANT  
  
            IF index GREATER THAN one  
                SET the element at position index of dynamic_programming_list TO the element at position index of dynamic_programming_list PLUS the element at position index MINUS two of dynamic_programming_list MULTIPLIED BY ways_to_decode_pair OF the character at position index MINUS two of s AND the character at position index MINUS one of s  
                SET the element at position index of dynamic_programming_list TO the element at position index of dynamic_programming_list MODULO MODULO_CONSTANT  
            END IF  
        END FOR  
  
        RETURN the element at the last position of dynamic_programming_list  
    END FUNCTION  
END CLASS
