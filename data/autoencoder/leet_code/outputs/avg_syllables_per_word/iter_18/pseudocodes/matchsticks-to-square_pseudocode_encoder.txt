CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF each element IN matchsticks  
        IF the REMAINDER of total_length DIVIDED BY four IS NOT equal to zero THEN  
            RETURN false  
        END IF  
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN descending order  
        FUNCTION can_form_sides(sides, index)  
            IF index IS EQUAL TO the LENGTH OF matchsticks THEN  
                IF for every side IN sides IT IS EQUAL TO side_length THEN  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            END IF  
            FOR each position FROM zero TO three  
                IF the element at position of sides PLUS the element at position of matchsticks AT index IS LESS THAN OR EQUAL TO side_length THEN  
                    INCREMENT the element at position of sides BY the element at position of matchsticks AT index  
                    IF can_form_sides(sides, index INCREMENTED BY one) RETURNS true THEN  
                        RETURN true  
                    END IF  
                    DECREMENT the element at position of sides BY the element at position of matchsticks AT index  
                END IF  
                IF the element at position of sides IS EQUAL TO zero THEN  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        RETURN can_form_sides(a list consisting of zero repeated four times, zero)  
    END FUNCTION  
END CLASS
