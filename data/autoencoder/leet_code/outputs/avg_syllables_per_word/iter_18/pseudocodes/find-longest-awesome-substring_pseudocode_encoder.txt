CLASS Solution  
    FUNCTION longestAwesome WITH parameter s  
        SET prefix_map TO mapping of zero TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index_and_character IN enumeration of s  
            SET i TO the index portion of index_and_character  
            SET char TO the character portion of index_and_character  
            SET digit TO the integer value of char  
            SET prefix_xor TO prefix_xor XORED WITH one SHIFTED LEFT BY digit times  
            
            IF prefix_xor IS A KEY IN prefix_map  
                SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the value in prefix_map AT key prefix_xor  
            END IF  
            
            FOR j FROM zero TO nine inclusive  
                SET flipped TO prefix_xor XORED WITH one SHIFTED LEFT BY j times  
                IF flipped IS A KEY IN prefix_map  
                    SET max_length TO the greater value BETWEEN max_length AND the difference BETWEEN i AND the value in prefix_map AT key flipped  
                END IF  
            END FOR  
            
            IF prefix_xor IS NOT A KEY IN prefix_map  
                SET the value in prefix_map AT key prefix_xor TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
