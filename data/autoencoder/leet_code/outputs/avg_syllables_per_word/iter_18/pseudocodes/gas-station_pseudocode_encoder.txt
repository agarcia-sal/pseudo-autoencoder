CLASS Solution  
    FUNCTION canCompleteCircuit WITH PARAMETERS gas AS list of integers AND cost AS list of integers  
        IF the SUM OF gas IS LESS THAN the SUM OF cost THEN  
            RETURN negative one  
        END IF  
        SET start TO zero  
        SET total_tank TO zero  
        FOR each index i FROM zero TO the LENGTH OF gas MINUS one  
            INCREMENT total_tank BY the element at position i of gas MINUS the element at position i of cost  
            IF total_tank IS LESS THAN zero THEN  
                SET start TO index i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        RETURN start  
    END FUNCTION  
END CLASS
