CLASS Solution  
    FUNCTION multiply(num1 AS string, num2 AS string) RETURNS string  
        IF num1 EQUALS zero STRING OR num2 EQUALS zero STRING THEN  
            RETURN zero STRING  
        END IF  
        
        SET result TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
        
        SET num1 TO the string consisting of the characters of num1 in the reverse order  
        SET num2 TO the string consisting of the characters of num2 in the reverse order  
        
        FOR each index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR each index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the INTEGER VALUE OF the character at position i of num1 MINUS the INTEGER VALUE OF zero STRING MULTIPLIED BY the INTEGER VALUE OF the character at position j of num2 MINUS the INTEGER VALUE OF zero STRING  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 of result  
                
                SET element at position p1 of result TO the remainder of sum_ DIVIDED BY ten  
                INCREMENT element at position p2 of result BY the quotient of sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the string formed by concatenating the elements of result in the reverse order converted to string  
        RETURN the string result_str after REMOVING all leading zero STRING characters  
    END FUNCTION  
END CLASS
