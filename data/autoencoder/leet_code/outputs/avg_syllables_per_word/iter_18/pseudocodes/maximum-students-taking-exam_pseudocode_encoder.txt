CLASS Solution  
    FUNCTION maxStudents WITH PARAMETER seats AS LIST OF LIST OF STRING  
        SET m TO the LENGTH OF seats  
        SET n TO the LENGTH OF element at position zero of seats  
        
        SET broken TO a new list of zero repeated m times  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF character at position j of element at position i of seats EQUALS the character number sign  
                    SET element at position i of broken TO element at position i of broken BITWISE OR bit shifted left by j by one position  
                END IF  
            END FOR  
        END FOR  
        
        FUNCTION is_valid WITH PARAMETER arrangement  
            RETURN arrangement BITWISE AND arrangement shifted right by one position EQUALS zero  
        END FUNCTION  
        
        SET dp TO a new list containing m elements each as empty mapping  
        
        FOR arrangement FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            IF is_valid WITH arrangement HOLDS AND arrangement BITWISE AND element at position zero of broken EQUALS zero  
                SET element at position zero of dp AT KEY arrangement TO the COUNT OF ones in binary representation of arrangement  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR arrangement FROM zero TO two RAISED TO THE POWER OF n MINUS one  
                IF is_valid WITH arrangement HOLDS AND arrangement BITWISE AND element at position i of broken EQUALS zero  
                    SET max_students TO zero  
                    FOR each prev_arrangement IN element at position i MINUS one of dp  
                        IF arrangement BITWISE AND prev_arrangement shifted right by one position EQUALS zero AND arrangement BITWISE AND prev_arrangement shifted left by one position EQUALS zero  
                            IF dp at position i MINUS one AT KEY prev_arrangement GREATER THAN max_students  
                                SET max_students TO dp at position i MINUS one AT KEY prev_arrangement  
                            END IF  
                        END IF  
                    END FOR  
                    SET element at position i of dp AT KEY arrangement TO max_students PLUS the COUNT OF ones in binary representation of arrangement  
                END IF  
            END FOR  
        END FOR  
        
        IF element at the last position of dp IS NOT empty  
            RETURN the MAXIMUM value among all values in element at the last position of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
