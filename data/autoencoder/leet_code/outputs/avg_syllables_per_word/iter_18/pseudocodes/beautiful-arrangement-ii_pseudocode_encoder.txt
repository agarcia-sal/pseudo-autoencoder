CLASS Solution  
    FUNCTION constructArray WITH PARAMETERS n AND k  
        SET result TO auxiliary FUNCTION generate INITIAL sequence FROM one TO n MINUS k MINUS one  
        SET left TO n MINUS k  
        SET right TO n  
        WHILE left GREATER THAN OR EQUAL TO right HOLDS FALSE  
            IF left EQUALS right  
                APPEND left TO result  
            ELSE  
                APPEND left TO result  
                APPEND right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
