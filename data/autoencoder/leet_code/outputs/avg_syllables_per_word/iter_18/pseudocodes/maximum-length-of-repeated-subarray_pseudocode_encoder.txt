CLASS Solution  
    FUNCTION findLength(nums1 AS list of integers nums2 AS list of integers) RETURNS integer  
        SET dp TO new two dimensional list with number of rows EQUAL TO the LENGTH OF nums1 PLUS one and number of columns EQUAL TO the LENGTH OF nums2 PLUS one each initialized TO zero  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF the element at position i MINUS one of nums1 EQUALS the element at position j MINUS one of nums2 THEN  
                    SET the element at row i and column j of dp TO the element at row i MINUS one and column j MINUS one of dp PLUS one  
                    IF the element at row i and column j of dp GREATER THAN max_length THEN  
                        SET max_length TO the element at row i and column j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
