CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS nothing right EQUALS nothing  
        SET the instance variable val TO val  
        SET the instance variable left TO left  
        SET the instance variable right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN nothing  
    END IF  
    SET root TO new instance of TreeNode USING the element at position zero of values  
    SET index_counter TO one  
    CREATE a new empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the first element removed from the left side of queue  
        IF index_counter IS LESS THAN the length of values AND the element at position index_counter of values IS NOT nothing  
            SET the left property of node TO new instance of TreeNode USING the element at position index_counter of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT index_counter BY one  
        IF index_counter IS LESS THAN the length of values AND the element at position index_counter of values IS NOT nothing  
            SET the right property of node TO new instance of TreeNode USING the element at position index_counter of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT index_counter BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS nothing AND q IS nothing  
        RETURN true  
    ELSE IF p IS nothing OR q IS nothing  
        RETURN false  
    ELSE IF the val property of p IS NOT EQUAL TO the val property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with the left property of p AND the left property of q AND the result of is_same_tree called with the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor WITH PARAMETERS root p q  
        IF root IS nothing OR root IS EQUAL TO p OR root IS EQUAL TO q  
            RETURN root  
        END IF  
        SET left TO the result of calling lowestCommonAncestor on the left property of root WITH p AND q  
        SET right TO the result of calling lowestCommonAncestor on the right property of root WITH p AND q  
        IF left IS NOT nothing AND right IS NOT nothing  
            RETURN root  
        END IF  
        IF left IS NOT nothing  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
