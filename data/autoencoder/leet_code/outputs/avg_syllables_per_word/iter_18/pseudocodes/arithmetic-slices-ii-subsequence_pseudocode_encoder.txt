CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums AS list of integers) RETURNS integer  
        SET n TO the LENGTH OF nums  
        IF n IS LESS THAN three  
            RETURN zero  
        END IF  
          
        SET dp TO a list of mappings from integers to integers each initialized as a default mapping returning integer zero for each position from zero TO n MINUS one  
        SET total_count TO zero  
          
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO the integer AT position i OF nums MINUS the integer AT position j OF nums  
                  
                IF the mapping AT position j OF dp CONTAINS key d  
                    INCREMENT the value at key d in the mapping at position i of dp BY the sum of the value at key d in the mapping at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key d in the mapping at position j of dp  
                ELSE  
                    INCREMENT the value at key d in the mapping at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
          
        RETURN total_count  
    END FUNCTION  
END CLASS
