CLASS Solution  
    FUNCTION maxSubArray(nums AS List OF Integer) RETURNS Integer  
        SET max_sum TO element at position zero of nums  
        SET current_sum TO element at position zero of nums  
        FOR each num IN the elements of nums starting FROM the second element TO the last element  
            SET current_sum TO the maximum BETWEEN num AND current_sum PLUS num  
            SET max_sum TO the maximum BETWEEN max_sum AND current_sum  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
