CLASS Solution  
    FUNCTION smallestFactorization WITH PARAMETER num  
        IF num EQUALS one THEN  
            RETURN one  
        END IF  
  
        SET factors TO an empty list  
  
        FOR digit FROM nine DOWN TO two  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero DO  
                APPEND digit TO factors  
                SET num TO the integer division of num BY digit  
            END WHILE  
        END FOR  
  
        IF num NOT EQUALS one THEN  
            RETURN zero  
        END IF  
  
        SORT factors IN ascending order  
  
        SET result TO the integer value formed by concatenating each element at position index of factors IN order  
  
        IF result GREATER THAN the maximum value of a thirty two bit signed integer THEN  
            RETURN zero  
        END IF  
  
        RETURN result  
    END FUNCTION  
END CLASS
