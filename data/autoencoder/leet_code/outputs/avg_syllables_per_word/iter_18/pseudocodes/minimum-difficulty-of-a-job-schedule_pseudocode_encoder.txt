CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty AS List of Integer, d AS Integer) RETURNS Integer  
        SET n TO the LENGTH OF jobDifficulty  
  
        IF n LESS THAN d THEN  
            RETURN negative one  
        END IF  
  
        FUNCTION dp(i AS Integer, remaining_days AS Integer) RETURNS Integer  
            IF remaining_days EQUALS one THEN  
                RETURN the MAXIMUM VALUE among elements at position i to the last position of jobDifficulty  
            END IF  
  
            SET min_difficulty TO infinite value  
            SET current_max TO zero  
            FOR index j FROM i TO the LENGTH OF jobDifficulty MINUS remaining_days  
                SET current_max TO the MAXIMUM VALUE BETWEEN current_max AND element at position j of jobDifficulty  
                SET candidate_difficulty TO current_max PLUS dp(j PLUS one, remaining_days MINUS one)  
                SET min_difficulty TO the MINIMUM VALUE BETWEEN min_difficulty AND candidate_difficulty  
            END FOR  
  
            RETURN min_difficulty  
        END FUNCTION  
  
        RETURN dp(zero, d)  
END CLASS
