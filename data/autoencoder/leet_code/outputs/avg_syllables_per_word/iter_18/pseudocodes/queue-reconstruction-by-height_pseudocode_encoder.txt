CLASS Solution  
    FUNCTION reconstructQueue WITH parameter people OF TYPE List OF List OF integer  
        CALL sort ON people WITH key BEING FUNCTION WITH parameter x  
            RETURN NEGATIVE of element at position zero of x AND element at position one of x  
        END FUNCTION  
        SET queue TO empty list  
        FOR each person IN people  
            INSERT person INTO queue AT position element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
