CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals as List of List of integer) RETURNS integer  
        IF intervals EQUALS an empty list THEN  
            RETURN zero  
        END IF  
        SORT intervals BY the end value of each element in intervals  
        SET end TO the end value of the first element of intervals  
        SET count TO zero  
        FOR each index FROM one TO the LENGTH OF intervals MINUS one  
            IF the start value of the element at position index of intervals IS GREATER THAN OR EQUAL TO end THEN  
                SET end TO the end value of the element at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
