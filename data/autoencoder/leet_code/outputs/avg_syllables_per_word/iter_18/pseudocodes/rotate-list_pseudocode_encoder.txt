CLASS ListNode  
    FUNCTION __init__(val EQUALS zero COMMA next EQUALS none)  
        SET the val attribute OF self TO val  
        SET the next attribute OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A new ListNode INITIALIZED WITH element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO THE END OF values  
        SET node TO A new ListNode INITIALIZED WITH val  
        SET the next attribute OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 COMMA p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val attribute OF p1 EQUALS the val attribute OF p2 AND is_same_list(the next attribute OF p1 COMMA the next attribute OF p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(head COMMA k)  
        IF head IS none OR the next attribute OF head IS none OR k EQUALS zero  
            RETURN head  
        END IF  
        
        SET length TO one  
        SET tail TO head  
        WHILE the next attribute OF tail IS NOT none  
            SET tail TO the next attribute OF tail  
            INCREMENT length BY one  
        END WHILE  
        
        SET the next attribute OF tail TO head  
        
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        
        SET new_tail TO head  
        FOR index FROM one TO steps_to_new_tail  
            SET new_tail TO the next attribute OF new_tail  
        END FOR  
        
        SET new_head TO the next attribute OF new_tail  
        SET the next attribute OF new_tail TO none  
        
        RETURN new_head  
    END FUNCTION  
END CLASS
