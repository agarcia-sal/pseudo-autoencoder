CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET total_sum TO the SUM OF nums  
        SET current_sum TO the SUM OF each integer i MULTIPLIED BY the element at position i of nums for each integer i and element at position i of nums  
        SET max_value TO current_sum  
        FOR each integer k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY the element at position n MINUS k of nums  
            SET max_value TO the GREATER OF max_value AND current_sum  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
