CLASS Solution  
    FUNCTION kIncreasing WITH parameters arr AND k  
        FUNCTION longest_increasing_subsequence WITH parameter subarr  
            SET lis TO an empty list  
            FOR each num IN subarr  
                IF lis is empty OR num is greater than or equal to the last element in lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position where num can be inserted to the right in lis maintaining sorted order  
                    SET the element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
  
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
  
        FOR start FROM zero TO k minus one  
            SET subarr TO an empty list  
            FOR i FROM start TO n minus one STEP k  
                APPEND element at position i of arr TO subarr  
            END FOR  
  
            SET lis_length TO the RESULT OF longest_increasing_subsequence WITH argument subarr  
            INCREMENT total_operations BY the LENGTH OF subarr minus lis_length  
        END FOR  
  
        RETURN total_operations  
    END FUNCTION  
END CLASS
