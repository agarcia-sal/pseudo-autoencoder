CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct WITH PARAMETER s  
        INITIALIZE mapping char_count AS a dictionary with default integer zero  
        SET variable left TO zero  
        SET variable max_length TO zero  
        
        FOR each position right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of the character at position right of s in mapping char_count BY one  
            
            WHILE the number of keys in mapping char_count IS GREATER THAN two  
                DECREMENT the count of the character at position left of s in mapping char_count BY one  
                IF the count of the character at position left of s in mapping char_count EQUALS zero  
                    REMOVE the character at position left of s FROM mapping char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND the result of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
