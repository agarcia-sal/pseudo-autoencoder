CLASS Solution  
    FUNCTION threeSum WITH PARAMETER nums AS List OF Integers RETURNS List OF Lists OF Integers  
        EXECUTE sorting nums INTO ASCENDING ORDER  
        SET result TO empty list  
        SET n TO the LENGTH OF nums  
        
        FOR index i FROM zero TO n MINUS two  
            IF i GREATER THAN zero AND element at position i of nums EQUALS element at position i MINUS one of nums THEN  
                CONTINUE TO next iteration of FOR  
            END IF  
            
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            
            WHILE left LESS THAN right HOLDS  
                SET total TO element at position i of nums PLUS element at position left of nums PLUS element at position right of nums  
                
                IF total EQUALS zero THEN  
                    APPEND list containing element at position i of nums AND element at position left of nums AND element at position right of nums TO result  
                    
                    WHILE left LESS THAN right AND element at position left of nums EQUALS element at position left PLUS one of nums HOLDS  
                        INCREMENT left BY one  
                    END WHILE  
                    
                    WHILE left LESS THAN right AND element at position right of nums EQUALS element at position right MINUS one of nums HOLDS  
                        DECREMENT right BY one  
                    END WHILE  
                    
                    INCREMENT left BY one  
                    DECREMENT right BY one  
                    
                ELSE IF total LESS THAN zero THEN  
                    INCREMENT left BY one  
                    
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
