CLASS ListNode  
    FUNCTION __init__ WITH parameter val defaulting TO zero AND parameter next defaulting TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode WITH parameter element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one TO the end  
        SET node TO new instance of ListNode WITH parameter val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN self val at p1 EQUALS self val at p2 AND is_same_list OF next at p1 AND next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH parameter head  
        IF head EQUALS none OR next of head EQUALS none  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT none AND next of fast IS NOT none  
            SET slow TO next of slow  
            SET fast TO next of next of fast  
        END WHILE  

        SET prev TO none  
        SET curr TO slow  
        WHILE curr IS NOT none  
            SET next_temp TO next of curr  
            SET next of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE next of second IS NOT none  
            SET temp1 TO next of first  
            SET temp2 TO next of second  
            SET next of first TO second  
            SET next of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
