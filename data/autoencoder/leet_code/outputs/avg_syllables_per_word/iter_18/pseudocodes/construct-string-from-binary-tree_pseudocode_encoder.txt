CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero left EQUAL TO None right EQUAL TO None  
        SET the attribute val OF self TO val  
        SET the attribute left OF self TO left  
        SET the attribute right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values OF TYPE list  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO A NEW DEQUE COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET the attribute left OF node TO A NEW INSTANCE OF TreeNode WITH PARAMETER element at position i of values  
            APPEND the attribute left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET the attribute right OF node TO A NEW INSTANCE OF TreeNode WITH PARAMETER element at position i of values  
            APPEND the attribute right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the attribute val OF p IS NOT EQUALS TO the attribute val OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH ARGUMENTS the attribute left OF p AND the attribute left OF q AND is_same_tree WITH ARGUMENTS the attribute right OF p AND the attribute right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str WITH PARAMETERS root OF TYPE Optional OF TreeNode  
        FUNCTION helper WITH PARAMETER node  
            IF node IS None  
                RETURN empty string  
            END IF  
            SET result TO THE STRING REPRESENTATION OF the attribute val OF node  
            IF the attribute left OF node EXISTS OR the attribute right OF node EXISTS  
                SET result TO result CONCATENATED WITH OPENING PARENTHESES CONCATENATED WITH helper WITH ARGUMENT the attribute left OF node CONCATENATED WITH CLOSING PARENTHESES  
            END IF  
            IF the attribute right OF node EXISTS  
                SET result TO result CONCATENATED WITH OPENING PARENTHESES CONCATENATED WITH helper WITH ARGUMENT the attribute right OF node CONCATENATED WITH CLOSING PARENTHESES  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper WITH ARGUMENT root  
    END FUNCTION  
END CLASS
