CLASS UnionFind  
    FUNCTION __init__ WITH parameter n  
        SET self.parent TO the list of numbers FROM zero TO n inclusive  
        SET self.rank TO the list consisting of zero repeated the number n plus one times  
        SET self.count TO n  
    END FUNCTION  
      
    FUNCTION find WITH parameter u  
        IF the element at position u of self.parent DOES NOT EQUAL u THEN  
            SET the element at position u of self.parent TO the result of calling self.find WITH the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  
      
    FUNCTION union WITH parameters u AND v  
        SET rootU TO the result of calling self.find WITH u  
        SET rootV TO the result of calling self.find WITH v  
        IF rootU DOES NOT EQUAL rootV THEN  
            IF the element at position rootU of self.rank IS GREATER THAN the element at position rootV of self.rank THEN  
                SET the element at position rootV of self.parent TO rootU  
            ELSE IF the element at position rootU of self.rank IS LESS THAN the element at position rootV of self.rank THEN  
                SET the element at position rootU of self.parent TO rootV  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove WITH parameters n AND edges  
        SET uf_common TO a new instance of UnionFind WITH parameter n  
        SET uf_alice TO a new instance of UnionFind WITH parameter n  
        SET uf_bob TO a new instance of UnionFind WITH parameter n  
          
        SET used_edges TO zero  
          
        FOR each tuple of t u v IN edges  
            IF t EQUALS three THEN  
                IF calling union method of uf_common WITH parameters u AND v RETURNS true THEN  
                    CALL union method of uf_alice WITH parameters u AND v  
                    CALL union method of uf_bob WITH parameters u AND v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
          
        FOR each tuple of t u v IN edges  
            IF t EQUALS one AND calling union method of uf_alice WITH u AND v RETURNS true THEN  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND calling union method of uf_bob WITH u AND v RETURNS true THEN  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
          
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one THEN  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
