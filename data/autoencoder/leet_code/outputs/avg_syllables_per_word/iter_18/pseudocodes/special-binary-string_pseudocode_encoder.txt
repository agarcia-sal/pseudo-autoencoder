CLASS Solution  
    FUNCTION makeLargestSpecial WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER s  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            
            FOR each index and character IN enumeration of s  
                IF character EQUALS the digit one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count EQUALS zero  
                    APPEND concatenation of the digit one PLUS result of dfs invoked with substring of s from position start incremented BY one up to position index PLUS the digit zero TO specials  
                    SET start TO index incremented BY one  
                END IF  
            END FOR  
            
            SORT specials in reverse lexicographical order  
            RETURN concatenation of elements of specials  
        END FUNCTION  
        
        RETURN result of dfs invoked with s  
    END FUNCTION  
END CLASS
