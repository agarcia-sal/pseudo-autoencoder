CLASS Solution  
    FUNCTION checkInclusion WITH PARAMETERS s1 s2  
        FROM the collection of counted elements ASSIGN reference TO Counter  
        
        SET the length of s1 TO len_s1  
        SET the length of s2 TO len_s2  
        
        IF the length of s1 is greater than the length of s2 THEN  
            RETURN false  
        END IF  
        
        SET the frequency count of characters in s1 TO count_s1 USING Counter  
        SET the frequency count of characters in the substring of s2 from the first position to the position of the length of s1 TO count_s2 USING Counter  
        
        IF the frequency count count_s1 EQUALS the frequency count count_s2 THEN  
            RETURN true  
        END IF  
        
        FOR index FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the frequency of the character at position index of s2 IN count_s2 BY one  
            
            DECREMENT the frequency of the character at the position of index MINUS len_s1 of s2 IN count_s2 BY one  
            
            IF the frequency of the character at the position of index MINUS len_s1 of s2 IN count_s2 EQUALS zero THEN  
                REMOVE the character at position index MINUS len_s1 of s2 FROM count_s2  
            END IF  
            
            IF the frequency count count_s1 EQUALS the frequency count count_s2 THEN  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
