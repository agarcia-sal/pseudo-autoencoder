CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET val ATTRIBUTE OF SELF TO val  
        SET next ATTRIBUTE OF SELF TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH val AS the element at position zero of values  
    SET p TO head  
    FOR each val IN the SUBSEQUENCE OF values FROM the SECOND ELEMENT TO the LAST ELEMENT  
        SET node TO A NEW ListNode WITH val AS val  
        SET next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND is_same_list CALLED WITH PARAMETERS the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
        SET dummy_head TO A NEW ListNode WITH val AS zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry GREATER THAN zero  
            IF l1 IS NOT None  
                SET val1 TO the val ATTRIBUTE OF l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the val ATTRIBUTE OF l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the INTEGER DIVISION OF total BY ten  
            SET next ATTRIBUTE OF current TO A NEW ListNode WITH val AS the MODULO OF total BY ten  
            SET current TO the next ATTRIBUTE OF current  
            
            IF l1 IS NOT None  
                SET l1 TO the next ATTRIBUTE OF l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO the next ATTRIBUTE OF l2  
            END IF  
        END WHILE  
        
        RETURN the next ATTRIBUTE OF dummy_head  
    END FUNCTION  
END CLASS
