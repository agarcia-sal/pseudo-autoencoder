CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    ASSIGN reference TO head IN variable p  
    FOR each val IN elements of values STARTING FROM position one TO the end  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET next ATTRIBUTE OF p TO node  
        ASSIGN reference TO node IN variable p  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN Boolean true  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN Boolean false  
    END IF  
    RETURN Boolean result OF the comparison that element at attribute val OF p1 EQUALS element at attribute val OF p2 AND the recursive call of is_same_list WITH element at attribute next OF p1 AND element at attribute next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETER head  
        IF head IS None OR element at attribute next OF head IS None  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode INITIALIZED WITH integer value negative five thousand and one  
        SET next ATTRIBUTE OF dummy TO head  
        SET current TO element at attribute next OF head  
        SET next ATTRIBUTE OF head TO None  
        WHILE current IS NOT None  
            SET next_node TO element at attribute next OF current  
            ASSIGN reference TO dummy IN variable prev  
            WHILE element at attribute next OF prev IS NOT None AND element at attribute val OF element at attribute next OF prev IS LESS THAN element at attribute val OF current  
                ASSIGN reference TO element at attribute next OF prev IN variable prev  
            END WHILE  
            SET next ATTRIBUTE OF current TO element at attribute next OF prev  
            SET next ATTRIBUTE OF prev TO current  
            ASSIGN reference TO next_node IN variable current  
        END WHILE  
        RETURN element at attribute next OF dummy  
    END FUNCTION  
END CLASS
