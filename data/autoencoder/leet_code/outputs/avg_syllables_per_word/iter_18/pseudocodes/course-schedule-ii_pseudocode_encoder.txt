CLASS Solution  
    FUNCTION findOrder(numCourses AS integer, prerequisites AS list of list of integer) RETURNS list of integer  
        SET adj_list TO an empty mapping of course TO empty list  
        FOR each number FROM zero TO numCourses MINUS one  
            SET element at key number in adj_list TO empty list  
        END FOR  
        
        SET in_degrees TO a list containing numCourses elements each initialized TO zero  
        
        FOR each pair of dest and src IN prerequisites  
            APPEND dest TO element at key src in adj_list  
            INCREMENT element at position dest in in_degrees BY one  
        END FOR  
        
        SET queue TO a new empty double ended queue  
        FOR each course FROM zero TO numCourses MINUS one  
            IF element at position course in in_degrees EQUALS zero  
                APPEND course TO queue  
            END IF  
        END FOR  
        
        SET topological_order TO empty list  
        
        WHILE queue is not empty  
            REMOVE and RETURN the element from the left side of queue ASSIGN TO current  
            APPEND current TO topological_order  
            
            FOR each neighbor IN element at key current in adj_list  
                DECREMENT element at position neighbor in in_degrees BY one  
                
                IF element at position neighbor in in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
