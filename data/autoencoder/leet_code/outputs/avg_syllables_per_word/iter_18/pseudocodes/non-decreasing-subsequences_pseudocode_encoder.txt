CLASS Solution  
    FUNCTION findSubsequences WITH PARAMETER nums AS list of integers RETURNS list of list of integers  
        FUNCTION dfs WITH PARAMETERS u AS integer last AS integer t AS list of integers  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND t COPIED TO ans  
                END IF  
                RETURN  
            END IF  
            IF element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND element at position u of nums TO t  
                CALL dfs WITH u INCREMENTED BY one last SET TO element at position u of nums t  
                REMOVE last element FROM t  
            END IF  
            IF element at position u of nums NOT EQUAL TO last  
                CALL dfs WITH u INCREMENTED BY one last t  
            END IF  
        END FUNCTION  
        SET ans TO empty list  
        CALL dfs WITH zero last SET TO negative one thousand t SET TO empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
