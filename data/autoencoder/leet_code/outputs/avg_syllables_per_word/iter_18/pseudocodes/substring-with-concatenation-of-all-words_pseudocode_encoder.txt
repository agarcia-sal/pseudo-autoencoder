CLASS Solution  
    FUNCTION findSubstring WITH PARAMETERS s string AND words list of strings  
        IF s IS EMPTY OR words IS EMPTY OR element at position zero of words IS EMPTY  
            RETURN empty list  
        END IF  

        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNT OF occurrences OF each element IN words  
        SET result TO empty list  

        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO empty count of occurrences  
            FOR index j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO the SUBSTRING of s STARTING AT j AND OF LENGTH word_length  
                INCREMENT the COUNT OF word IN seen BY one  
                IF the COUNT OF word IN seen IS GREATER THAN the COUNT OF word IN word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop DID NOT BREAK  
                APPEND i TO result  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
