CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing zero and one and eight as strings  
            END IF  
            SET ans TO empty list  
            FOR each v IN dfs OF u MINUS two  
                FOR each pair IN list containing string one one string eight eight string six nine string nine six  
                    SET l TO first character of pair  
                    SET r TO second character of pair  
                    APPEND concatenation of l and v and r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of zero and v and zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs OF n  
    END FUNCTION  
END CLASS
