CLASS Solution  
    FUNCTION minAbsDifference WITH PARAMETERS nums AND goal  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list CONTAINING zero  
        FOR each num IN the SUBLIST of nums FROM the FIRST ELEMENT TO the ELEMENT BEFORE mid  
            SET new_sums TO an EMPTY list  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND left_sums BY new_sums  
        END FOR  
        
        SET right_sums TO a list CONTAINING zero  
        FOR each num IN the SUBLIST of nums FROM mid TO the END  
            SET new_sums TO an EMPTY list  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            EXTEND right_sums BY new_sums  
        END FOR  
        
        SORT right_sums IN ASCENDING ORDER  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO POSITION FOUND BY BINARY SEARCH FOR target IN right_sums USING LEFT BOUND SEARCH  
            
            IF pos IS LESS THAN the LENGTH OF right_sums  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS element at position pos of right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            
            IF pos IS GREATER THAN zero  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS element at position pos MINUS one OF right_sums MINUS goal  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
