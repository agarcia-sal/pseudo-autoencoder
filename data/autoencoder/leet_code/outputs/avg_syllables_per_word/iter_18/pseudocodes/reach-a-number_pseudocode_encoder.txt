CLASS Solution  
    FUNCTION reachNumber(target)  
        SET target TO the absolute value of target  
        SET k TO zero  
        SET sum_steps TO zero  
        WHILE the value of sum_steps is less than the value of target holds  
            INCREMENT k BY one  
            INCREMENT sum_steps BY the value of k  
        END WHILE  
        IF the remainder when the difference of sum_steps and target is divided by two equals zero THEN  
            RETURN the value of k  
        ELSE  
            IF k modulo two equals zero THEN  
                RETURN the value of k plus one  
            ELSE  
                RETURN the value of k plus two  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
