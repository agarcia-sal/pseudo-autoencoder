CLASS Solution  
    FUNCTION isReflected(points AS list of list of integer) RETURNS boolean  
        SET unique_points TO the SET OF tuples CONVERTED FROM each element IN points  
        SET min_x TO the MINIMUM OF the first element of each tuple IN unique_points  
        SET max_x TO the MAXIMUM OF the first element of each tuple IN unique_points  
        SET line_of_reflection TO the SUM of min_x AND max_x DIVIDED BY two  
        FOR each tuple x and y IN unique_points  
            SET reflected_x TO the product of two AND line_of_reflection MINUS x  
            IF the tuple of reflected_x AND y NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
