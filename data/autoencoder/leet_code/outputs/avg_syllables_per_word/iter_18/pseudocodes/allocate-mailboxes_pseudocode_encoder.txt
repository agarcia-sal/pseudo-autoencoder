CLASS Solution  
    FUNCTION minDistance WITH parameters houses of type List of Integers and k of type Integer  
        EXECUTE sorting operation on houses in ascending order  
        SET n TO the LENGTH OF houses  
          
        INITIALIZE cost as a two dimensional list sized n BY n FILLED WITH zero  
        FOR index i FROM n DECREMENTING TO one  
            FOR index j FROM i INCREMENTING TO n MINUS one  
                SET element at position i and j of cost TO element at position i plus one and j minus one of cost PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
          
        INITIALIZE dp as a two dimensional list sized n plus one BY k plus one FILLED WITH infinite value  
        SET element at position zero and zero of dp TO zero  
          
        FOR index i FROM one TO n  
            FOR index j FROM one TO k  
                FOR index p FROM zero TO i MINUS one  
                    SET element at position i and j of dp TO the MINIMUM OF element at position i and j of dp AND element at position p and j minus one of dp PLUS element at position p and i minus one of cost  
                END FOR  
            END FOR  
        END FOR  
          
        RETURN element at position n and k of dp  
    END FUNCTION  
END CLASS
