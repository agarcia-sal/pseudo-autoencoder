CLASS Solution  
    FUNCTION validUtf8 WITH PARAMETER data WHICH IS A LIST OF INTEGER  
        FUNCTION get_byte_count WITH PARAMETER byte  
            IF the RESULT OF shifting byte TO THE RIGHT BY seven PLACES EQUALS zero THEN  
                RETURN one  
            ELSE IF the RESULT OF shifting byte TO THE RIGHT BY five PLACES EQUALS the BINARY VALUE ONE ONE ZERO THEN  
                RETURN two  
            ELSE IF the RESULT OF shifting byte TO THE RIGHT BY four PLACES EQUALS the BINARY VALUE ONE ONE ONE ZERO THEN  
                RETURN three  
            ELSE IF the RESULT OF shifting byte TO THE RIGHT BY three PLACES EQUALS the BINARY VALUE ONE ONE ONE ONE ZERO THEN  
                RETURN four  
            ELSE  
                RETURN minus one  
            END IF  
        END FUNCTION  

        SET i TO zero  
        WHILE the VALUE OF i IS LESS THAN the LENGTH OF data HOLDS  
            SET byte_count TO the RESULT OF get_byte_count APPLIED TO the element at position i of data  
            IF byte_count EQUALS minus one THEN  
                RETURN False  
            END IF  
            IF the SUM of i AND byte_count IS GREATER THAN the LENGTH OF data THEN  
                RETURN False  
            END IF  
            FOR j FROM the SUM of i AND one TO the SUM of i AND byte_count MINUS one  
                IF the RESULT OF shifting the element at position j of data TO THE RIGHT BY six PLACES DOES NOT EQUAL the BINARY VALUE one zero THEN  
                    RETURN False  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
