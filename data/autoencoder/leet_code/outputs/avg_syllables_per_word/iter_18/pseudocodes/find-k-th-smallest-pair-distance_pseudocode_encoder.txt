CLASS Solution  
    FUNCTION smallestDistancePair(nums AS List OF integer, k AS integer) AS integer  
        CALL nums SORTED IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid AS integer) AS integer  
            SET count TO zero  
            SET left TO zero  
            FOR each right FROM zero TO n MINUS one  
                WHILE the ELEMENT AT position right of nums MINUS the ELEMENT AT position left of nums GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the ELEMENT AT position n MINUS one of nums MINUS the ELEMENT AT position zero of nums  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF the RESULT OF count GIVEN mid LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
