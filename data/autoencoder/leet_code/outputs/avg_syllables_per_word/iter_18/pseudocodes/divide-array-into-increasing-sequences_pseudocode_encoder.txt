CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums AS List OF integer, k AS integer) RETURNS boolean  
        SET freq TO CALL Counter WITH nums  
        SET max_freq TO the maximum value IN the collection of values IN freq  
        SET division_result TO the LENGTH OF nums DIVIDED BY k  
        IF division_result IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
