CLASS Solution  
    FUNCTION crackSafe WITH PARAMETERS n k  
        SET visited TO an empty set  
        SET sequence TO an empty list  
        
        FUNCTION dfs WITH PARAMETER node  
            FOR each index FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of index  
                IF edge IS NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH argument as substring of edge FROM second character TO end  
                    APPEND string representation of index TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string composed of zero repeated n MINUS one TIMES  
        CALL dfs WITH argument start_node  
        
        SET result TO concatenation of elements in sequence IN REVERSE ORDER FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
