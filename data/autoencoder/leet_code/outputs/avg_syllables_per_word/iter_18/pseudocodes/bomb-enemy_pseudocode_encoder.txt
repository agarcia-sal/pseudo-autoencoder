CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS nothing OR element at position zero of grid EQUALS nothing THEN  
            RETURN zero  
        END IF  
  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET max_kills TO zero  
  
        SET row_kills TO zero  
        SET col_kills TO a list of zero repeated n times  
  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF j EQUALS zero OR element at position j MINUS one of row at position i of grid EQUALS the character W THEN  
                    SET row_kills TO zero  
                    FOR index k FROM j TO n MINUS one  
                        IF element at position k of row at position i of grid EQUALS the character E THEN  
                            INCREMENT row_kills BY one  
                        ELSE IF element at position k of row at position i of grid EQUALS the character W THEN  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
  
                IF i EQUALS zero OR element at position i MINUS one of grid at position j EQUALS the character W THEN  
                    SET element at position j of col_kills TO zero  
                    FOR index k FROM i TO m MINUS one  
                        IF element at position j of row at position k of grid EQUALS the character E THEN  
                            INCREMENT element at position j of col_kills BY one  
                        ELSE IF element at position j of row at position k of grid EQUALS the character W THEN  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
  
                IF element at position j of row at position i of grid EQUALS the character zero THEN  
                    SET max_kills TO the GREATER BETWEEN max_kills AND row_kills PLUS element at position j of col_kills  
                END IF  
            END FOR  
        END FOR  
  
        RETURN max_kills  
    END FUNCTION  
END CLASS
