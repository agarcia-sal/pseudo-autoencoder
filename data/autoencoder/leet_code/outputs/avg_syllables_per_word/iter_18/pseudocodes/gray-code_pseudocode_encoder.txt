CLASS Solution  
    FUNCTION grayCode WITH PARAMETER n  
        IF n EQUALS zero THEN  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one positions  
        SET reversed_prev_gray TO the reversal of prev_gray  
        SET reflected_list TO empty list  
        FOR each number IN reversed_prev_gray  
            SET reflected_number TO mask OR number  
            APPEND reflected_number TO reflected_list  
        END FOR  
        SET current_gray TO concatenation of prev_gray AND reflected_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
