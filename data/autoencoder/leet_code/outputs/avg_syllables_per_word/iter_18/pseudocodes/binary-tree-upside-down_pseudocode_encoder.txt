CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS none AND right EQUALS none  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values LIST  
    IF values DOES NOT EXIST OR values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW DEQUE COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE ELEMENT REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET node left TO NEW TreeNode WITH PARAMETER the element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET node right TO NEW TreeNode WITH PARAMETER the element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p DOES NOT EXIST AND q DOES NOT EXIST  
        RETURN true  
    ELSE IF p DOES NOT EXIST OR q DOES NOT EXIST  
        RETURN false  
    ELSE IF the value attribute of p IS NOT EQUAL TO the value attribute of q  
        RETURN false  
    ELSE  
        SET left_comparison TO the RESULT OF CALLING is_same_tree WITH PARAMETERS the left attribute of p AND the left attribute of q  
        SET right_comparison TO the RESULT OF CALLING is_same_tree WITH PARAMETERS the right attribute of p AND the right attribute of q  
        RETURN left_comparison AND right_comparison  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree WITH PARAMETERS self AND root OPTIONAL TreeNode  
        IF root DOES NOT EXIST OR the left attribute of root DOES NOT EXIST  
            RETURN root  
        END IF  
        SET new_root TO THE RESULT OF CALLING the upsideDownBinaryTree METHOD OF self WITH PARAMETER the left attribute of root  
        SET the left attribute of the left attribute of root TO the right attribute of root  
        SET the right attribute of the left attribute of root TO root  
        SET the left attribute of root TO none  
        SET the right attribute of root TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
