CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s k  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET character_at_right TO character at position right of s  
            SET char_map key character_at_right TO right  
            
            IF the LENGTH OF char_map GREATER THAN k THEN  
                SET leftmost_char TO the key in char_map HAVING the minimum associated value  
                SET left TO the value associated with leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET current_length TO right MINUS left PLUS one  
            IF current_length GREATER THAN max_length THEN  
                SET max_length TO current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
