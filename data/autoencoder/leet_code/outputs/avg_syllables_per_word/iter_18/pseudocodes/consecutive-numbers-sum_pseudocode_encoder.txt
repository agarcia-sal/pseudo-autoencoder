CLASS Solution
    FUNCTION consecutiveNumbersSum WITH PARAMETER n
        SET count TO zero
        FOR k FROM one TO n INCLUSIVE
            IF the product of two AND n DIVIDED BY k HAS NO REMAINDER
                SET m TO the quotient of (the product of two AND n DIVIDED BY k MINUS k PLUS one) DIVIDED BY two
                IF m IS GREATER THAN zero AND m EQUALS the integer part of m
                    INCREMENT count BY one
                END IF
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
