CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list of pairs consisting of one and zero MINUS one and zero zero and one zero and MINUS one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two dimensional list of size m BY n filled with infinity values  
            CREATE queue AS double ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one THEN  
                        APPEND the triple of i j zero TO queue  
                        SET element at position i and j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue HOLDS ELEMENTS  
                REMOVE leftmost triple from queue ASSIGN TO x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx and ny of grid EQUALS zero AND element at position nx and ny of fire_times EQUALS infinity THEN  
                        SET element at position nx and ny of fire_times TO t PLUS one  
                        APPEND the triple of nx ny t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            CREATE queue AS double ended queue WITH INITIAL ELEMENT a triple of zero zero wait  
            CREATE visited AS set containing the pair zero zero  
            
            WHILE queue HOLDS ELEMENTS  
                REMOVE leftmost triple from queue ASSIGN TO x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND pair nx ny NOT IN visited AND element at position nx and ny of grid EQUALS zero THEN  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one THEN  
                            IF element at position nx and ny of fire_times GREATER THAN OR EQUAL TO t PLUS one THEN  
                                RETURN True  
                            ELSE  
                                RETURN False  
                            END IF  
                        END IF  
                        IF element at position nx and ny of fire_times GREATER THAN t PLUS one THEN  
                            ADD pair nx ny TO visited  
                            APPEND the triple nx ny t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of left PLUS right BY two  
            IF can_escape(mid) THEN  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result NOT EQUAL TO m MULTIPLIED BY n THEN  
            RETURN result  
        ELSE  
            RETURN ten RAISED TO THE POWER OF nine  
        END IF  
    END FUNCTION  
END CLASS
