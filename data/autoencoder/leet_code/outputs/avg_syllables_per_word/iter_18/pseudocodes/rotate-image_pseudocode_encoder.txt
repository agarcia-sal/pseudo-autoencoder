CLASS Solution  
    FUNCTION rotate matrix  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_element TO the element at position i and j of matrix  
                SET the element at position i and j of matrix TO the element at position j and i of matrix  
                SET the element at position j and i of matrix TO temporary_element  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            SET the row_at_i TO the row at position i of matrix  
            SET left_index TO zero  
            SET right_index TO n MINUS one  
            WHILE left_index LESS THAN right_index  
                SET temporary_element TO the element at position left_index of row_at_i  
                SET the element at position left_index of row_at_i TO the element at position right_index of row_at_i  
                SET the element at position right_index of row_at_i TO temporary_element  
                INCREMENT left_index BY one  
                DECREMENT right_index BY one  
            END WHILE  
        END FOR  
    END FUNCTION  
END CLASS
