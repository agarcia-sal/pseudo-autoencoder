CLASS Solution  
    FUNCTION scoreOfParentheses WITH PARAMETER s AS string  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS opening parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO element popped FROM stack PLUS one IF score EQUALS zero OTHERWISE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
