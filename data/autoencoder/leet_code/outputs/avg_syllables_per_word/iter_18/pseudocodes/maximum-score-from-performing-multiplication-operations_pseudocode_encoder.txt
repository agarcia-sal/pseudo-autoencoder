CLASS Solution  
    FUNCTION maximumScore WITH parameters nums AND multipliers  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a LIST OF zero WITH the LENGTH OF m PLUS one  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a LIST OF zero WITH the LENGTH OF m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS the DIFFERENCE BETWEEN i AND left  
                SET mul TO the ELEMENT at position i OF multipliers  
                IF mul MULTIPLIED BY the ELEMENT at position left OF nums PLUS the ELEMENT at position left PLUS one OF dp GREATER THAN OR EQUAL TO mul MULTIPLIED BY the ELEMENT at position right OF nums PLUS the ELEMENT at position left OF dp  
                    SET the ELEMENT at position left OF new_dp TO mul MULTIPLIED BY the ELEMENT at position left OF nums PLUS the ELEMENT at position left PLUS one OF dp  
                ELSE  
                    SET the ELEMENT at position left OF new_dp TO mul MULTIPLIED BY the ELEMENT at position right OF nums PLUS the ELEMENT at position left OF dp  
                END IF  
            END FOR  
            ASSIGN reference TO new_dp TO dp  
        END FOR  
        RETURN the ELEMENT at position zero OF dp  
    END FUNCTION  
END CLASS
