CLASS Solution  
    FUNCTION getXORSum(arr1 AS List of integers arr2 AS List of integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        
        FOR each num IN arr1  
            SET xor_sum_arr1 TO xor_sum_arr1 XOR num  
        END FOR  
        
        FOR each num IN arr2  
            SET xor_sum_arr2 TO xor_sum_arr2 XOR num  
        END FOR  
        
        SET result TO xor_sum_arr1 AND xor_sum_arr2  
        RETURN result  
    END FUNCTION  
END CLASS
