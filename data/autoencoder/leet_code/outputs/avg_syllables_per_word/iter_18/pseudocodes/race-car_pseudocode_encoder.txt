CLASS Solution  
    FUNCTION racecar WITH parameter target  
        SET queue TO a double ended queue initialized with a tuple of zero position one speed zero steps  
        SET visited TO a set initialized with a tuple of zero position one speed  
        
        WHILE queue holds elements  
            REMOVE and ASSIGN the first element from queue to variables position speed steps  
            
            IF position EQUALS target THEN  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF tuple of next_position next_speed NOT IN visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target THEN  
                ADD tuple of next_position next_speed TO visited  
                APPEND tuple of next_position next_speed steps PLUS one TO queue  
            END IF  
            
            IF speed GREATER THAN zero THEN  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            
            IF tuple of position next_speed NOT IN visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target THEN  
                ADD tuple of position next_speed TO visited  
                APPEND tuple of position next_speed steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
