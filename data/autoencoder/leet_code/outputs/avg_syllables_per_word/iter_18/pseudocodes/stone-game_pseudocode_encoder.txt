CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i AND j  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            SET left_choice TO element at position i of piles MINUS the result of invoking dfs WITH argument i INCREMENTED BY one AND argument j  
            SET right_choice TO element at position j of piles MINUS the result of invoking dfs WITH argument i AND argument j DECREMENTED BY one  
            RETURN the greater value BETWEEN left_choice AND right_choice  
        END FUNCTION  
        SET initial_result TO invoking dfs WITH argument zero AND argument the LENGTH OF piles MINUS one  
        RETURN whether initial_result GREATER THAN zero  
    END FUNCTION  
END CLASS
