CLASS Solution  
    FUNCTION sortColors WITH PARAMETER nums  
        SET low TO zero  
        SET mid TO zero  
        SET high TO the LENGTH OF nums MINUS one  
        WHILE mid IS LESS THAN OR EQUAL TO high  
            IF element at position mid of nums EQUALS zero  
                SWAP element at position low of nums WITH element at position mid of nums  
                INCREMENT low BY one  
                INCREMENT mid BY one  
            ELSE IF element at position mid of nums EQUALS one  
                INCREMENT mid BY one  
            ELSE IF element at position mid of nums EQUALS two  
                SWAP element at position mid of nums WITH element at position high of nums  
                DECREMENT high BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
