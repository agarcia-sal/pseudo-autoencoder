CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n AS integer  
        FUNCTION backtrack WITH PARAMETERS s AS string DEFAULT empty string left AS integer DEFAULT zero right AS integer DEFAULT zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n THEN  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n THEN  
                CALL backtrack WITH s PLUS opening parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left THEN  
                CALL backtrack WITH s PLUS closing parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
