CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals AS list of list of integer) RETURNS integer  
        CALL intervals SORTED BY the end element of interval AND IF ends ARE EQUAL THEN BY the negative of the start element of interval  
        
        SET first TO element at position one LESS THAN the end of the first interval in intervals  
        SET second TO element at position the end of the first interval in intervals  
        SET count TO two  
        
        FOR each start and end IN intervals BEGINNING FROM the second element  
            IF start IS GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
