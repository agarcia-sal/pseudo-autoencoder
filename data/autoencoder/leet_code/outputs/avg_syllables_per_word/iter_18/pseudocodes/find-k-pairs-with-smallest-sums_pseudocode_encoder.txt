CLASS Solution  
    FUNCTION kSmallestPairs(nums1 AS List of integers nums2 AS List of integers k AS integer) AS List of List of integers  
        IF nums1 IS empty OR nums2 IS empty OR k IS LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        
        FOR index_i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 DECREASED BY one  
            PUSH a tuple consisting of the SUM of element at position index_i of nums1 AND the element at position zero of nums2 the index_i AND zero INTO min_heap USING HEAP PUSH  
        END FOR  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT empty AND the LENGTH OF result IS LESS THAN k  
            POP the smallest tuple FROM min_heap USING HEAP POP AND ASSIGN the three elements TO temporary variable underscore index_i AND index_j  
            APPEND a list consisting of the element at position index_i of nums1 AND the element at position index_j of nums2 TO result  
            
            IF the value of index_j INCREMENTED BY one IS LESS THAN the LENGTH OF nums2 THEN  
                PUSH a tuple consisting of the SUM of element at position index_i of nums1 AND the element at position index_j INCREMENTED BY one of nums2 the index_i AND index_j INCREMENTED BY one INTO min_heap USING HEAP PUSH  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
