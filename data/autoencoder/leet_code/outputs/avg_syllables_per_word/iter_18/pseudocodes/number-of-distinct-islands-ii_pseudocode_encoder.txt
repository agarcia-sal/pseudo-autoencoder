CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO the LENGTH OF grid OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x and position y of grid EQUALS zero  
                RETURN  
            END IF  
            SET element at position x and position y of grid TO zero  
            APPEND pair of x and y TO island  
            CALL dfs WITH x INCREMENTED BY one AND y AND island  
            CALL dfs WITH x DECREMENTED BY one AND y AND island  
            CALL dfs WITH x AND y INCREMENTED BY one AND island  
            CALL dfs WITH x AND y DECREMENTED BY one AND island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx AND dy IN collection of pairs one and one AND one and negative one AND negative one and one AND negative one and negative one  
                SET new_island TO empty list  
                FOR each pair x AND y IN island  
                    APPEND pair of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island IN ascending order  
                SET min_x TO the smallest x among elements in new_island  
                SET min_y TO the smallest y among elements in new_island  
                APPEND tuple of pairs of x MINUS min_x AND y MINUS min_y FOR each pair x AND y IN new_island TO shapes  
            END FOR  
            FOR index FROM zero TO the LENGTH OF shapes DECREMENTED BY one  
                APPEND tuple of pairs y AND negative x FOR each pair x AND y IN element at position index of shapes TO shapes  
            END FOR  
            RETURN the smallest element according to ORDERING IN shapes  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid DECREMENTED BY one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid DECREMENTED BY one  
                IF element at position i and position j of grid EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH i AND j AND island  
                    ADD normalize of island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the number of elements IN islands  
    END FUNCTION  
END CLASS
