CLASS Solution  
    FUNCTION findAllPeople(n AS integer_parameter, meetings AS list_of_lists_of_integers_parameter, firstPerson AS integer_parameter) RETURNS list_of_integers  
        SET know_secret TO set CONTAINING zero AND firstPerson  
        SORT meetings BY the time element AT position three OF each meeting  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the time element AT position three OF the meeting AT position i OF meetings  
            INITIALIZE graph AS a default dictionary MAPPING integers TO list of integers  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the time element AT position three OF the meeting AT position i OF meetings EQUALS current_time  
                SET x TO the participant element AT position one OF the meeting AT position i OF meetings  
                SET y TO the participant element AT position two OF the meeting AT position i OF meetings  
                APPEND y TO the list at key x IN graph  
                APPEND x TO the list at key y IN graph  
                INCREMENT i BY one  
            END WHILE  
            INITIALIZE queue AS a deque CONTAINING each person IN know_secret THAT EXISTS AS a key IN graph  
            INITIALIZE visited AS an empty set  
            WHILE queue IS NOT EMPTY  
                REMOVE the element from the front OF queue AND ASSIGN TO person  
                IF person EXISTS IN visited  
                    CONTINUE TO the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person IN graph  
                    IF neighbor DOES NOT EXIST IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list CONVERTED FROM know_secret  
    END FUNCTION  
END CLASS
