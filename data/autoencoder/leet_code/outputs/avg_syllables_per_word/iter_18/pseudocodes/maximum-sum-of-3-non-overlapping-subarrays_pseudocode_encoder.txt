CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums LIST OF INTEGER k INTEGER) RETURNS LIST OF INTEGER  
        SET n TO the LENGTH OF nums  
        SET sums TO a new LIST of zero repeated the QUANTITY OF n MINUS k PLUS one times  
        SET current_sum TO the SUM of the ELEMENTS from the START POSITION zero UP TO but NOT INCLUDING position k of nums  
        SET the ELEMENT at POSITION zero of sums TO current_sum  
        
        FOR each position i FROM one TO the QUANTITY OF n MINUS k PLUS one MINUS one  
            SET current_sum TO current_sum PLUS the ELEMENT at POSITION i PLUS k MINUS one of nums MINUS the ELEMENT at POSITION i MINUS one of nums  
            SET the ELEMENT at POSITION i of sums TO current_sum  
        END FOR  
        
        SET left TO a new LIST of zero repeated the QUANTITY OF n MINUS k PLUS one times  
        SET right TO a new LIST of the QUANTITY OF n MINUS k repeated the QUANTITY OF n MINUS k PLUS one times  
        
        SET max_sum TO zero  
        FOR each position i FROM zero TO the QUANTITY OF n MINUS k PLUS one MINUS one  
            IF the ELEMENT at POSITION i of sums IS GREATER THAN the ELEMENT at POSITION max_sum of sums THEN  
                SET max_sum TO i  
            END IF  
            SET the ELEMENT at POSITION i of left TO max_sum  
        END FOR  
        
        SET max_sum TO the QUANTITY OF n MINUS k  
        FOR each position i FROM the QUANTITY OF n MINUS k DOWN TO zero  
            IF the ELEMENT at POSITION i of sums IS GREATER THAN OR EQUAL TO the ELEMENT at POSITION max_sum of sums THEN  
                SET max_sum TO i  
            END IF  
            SET the ELEMENT at POSITION i of right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a new LIST containing zero three times  
        FOR each position j FROM k TO the QUANTITY OF n MINUS two TIMES k PLUS one MINUS one  
            SET i TO the ELEMENT at POSITION j MINUS k of left  
            SET l TO the ELEMENT at POSITION j PLUS k of right  
            SET total TO the ELEMENT at POSITION i of sums PLUS the ELEMENT at POSITION j of sums PLUS the ELEMENT at POSITION l of sums  
            IF total IS GREATER THAN max_total THEN  
                SET max_total TO total  
                SET result TO a new LIST containing i j l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
