CLASS Solution  
    FUNCTION findSubstringInWraproundString WITH PARAMETER s AS string  
        SET max_length TO a mapping FROM each character in the sequence of all lowercase letters from a to z TO the number zero  
        SET current_length TO zero  
        FOR index FROM zero TO the length of s MINUS one  
            IF index GREATER THAN zero AND  
                (the ordinal value of the character at position index of s MINUS the ordinal value of the character at position index MINUS one of s EQUALS one OR  
                 the character at position index of s EQUALS the character a AND the character at position index MINUS one of s EQUALS the character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            IF the value in max_length at key the character at position index of s LESS THAN current_length  
                SET the value in max_length at key the character at position index of s TO current_length  
            END IF  
        END FOR  
        SET total_unique_substrings TO the sum of all values in max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
