CLASS Solution  
    FUNCTION shortestCommonSupersequence WITH PARAMETERS str1 AND str2  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a two dimensional collection of zero EQUIVALENT with the number of elements being m PLUS one BY n PLUS one  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the CHARACTER at position i MINUS one OF str1 EQUALS the CHARACTER at position j MINUS one OF str2  
                    SET the element at position i AND position j OF dp TO the element at position i MINUS one AND position j MINUS one OF dp PLUS one  
                ELSE  
                    SET the element at position i AND position j OF dp TO the GREATER VALUE BETWEEN the element at position i MINUS one AND position j OF dp AND the element at position i AND position j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE the value of i IS GREATER THAN zero AND the value of j IS GREATER THAN zero  
            IF the CHARACTER at position i MINUS one OF str1 EQUALS the CHARACTER at position j MINUS one OF str2  
                APPEND the CHARACTER at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one AND position j OF dp IS GREATER THAN the element at position i AND position j MINUS one OF dp  
                APPEND the CHARACTER at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the CHARACTER at position j MINUS one OF str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE the value of i IS GREATER THAN zero  
            APPEND the CHARACTER at position i MINUS one OF str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE the value of j IS GREATER THAN zero  
            APPEND the CHARACTER at position j MINUS one OF str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements in scs  
        
        RETURN the concatenation of all elements in scs as a single string  
    END FUNCTION  
END CLASS
