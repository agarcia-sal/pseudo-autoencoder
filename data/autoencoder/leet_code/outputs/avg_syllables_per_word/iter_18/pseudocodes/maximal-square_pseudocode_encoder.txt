CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix IS empty OR element at position zero of matrix IS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        CALL auxiliary FUNCTION initializeDP WITH PARAMETERS m AND n AND ASSIGN RESULT TO dp  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AT position j MINUS one EQUALS the character one  
                    SET dp at position i AT position j TO the minimum OF dp at position i MINUS one AT position j AND dp at position i AT position j MINUS one AND dp at position i MINUS one AT position j MINUS one PLUS one  
                    SET max_side TO the maximum OF max_side AND dp at position i AT position j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
  
    FUNCTION initializeDP(m, n)  
        DESCRIPTION initialize a two dimensional list of zeros with dimensions m PLUS one AND n PLUS one  
    END FUNCTION  
END CLASS
