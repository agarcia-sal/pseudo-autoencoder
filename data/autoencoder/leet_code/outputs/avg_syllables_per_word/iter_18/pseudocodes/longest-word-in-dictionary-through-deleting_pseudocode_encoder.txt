CLASS Solution  
    FUNCTION findLongestWord WITH PARAMETERS s AND dictionary  
        FUNCTION is_subsequence WITH PARAMETERS word AND s  
            SET iterator_to_s TO the ITERATOR OF s  
            RETURN FOR every character IN word THAT character EXISTS IN iterator_to_s  
        END FUNCTION  
        
        SORT dictionary BY the NEGATION OF the LENGTH OF each element AND THEN BY lexicographical ORDER ASCENDING  
        
        FOR each word IN dictionary  
            IF is_subsequence OF word AND s HOLDS TRUE  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN an EMPTY STRING  
    END FUNCTION  
END CLASS
