CLASS Solution  
    FUNCTION solveSudoku WITH PARAMETERS board  
        FUNCTION is_valid WITH PARAMETERS board row col num  
            FOR index FROM zero TO eight  
                IF character at position index of row row in board EQUALS num THEN  
                    RETURN false  
                END IF  
            END FOR  
            FOR index FROM zero TO eight  
                IF character at position col in row index of board EQUALS num THEN  
                    RETURN false  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER RESULT OF row DIVIDED BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER RESULT OF col DIVIDED BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF character at position start_col PLUS j of row start_row PLUS i of board EQUALS num THEN  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  

        FUNCTION solve WITH PARAMETERS board  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF character at position col of row row in board EQUALS the character representing an empty cell THEN  
                        FOR each string_number IN the string representations of numbers from one TO nine inclusive  
                            IF is_valid WITH PARAMETERS board row col string_number THEN  
                                SET character at position col of row row in board TO string_number  
                                IF solve WITH PARAMETERS board THEN  
                                    RETURN true  
                                END IF  
                                SET character at position col of row row in board TO the character representing an empty cell  
                            END IF  
                        END FOR  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  

        CALL solve WITH PARAMETERS board  
    END FUNCTION  
END CLASS
