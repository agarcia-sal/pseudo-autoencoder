CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s1 s2  
        SET min_len TO a quantity representing infinitely large magnitude  
        SET min_start TO negative one  
        SET len1 TO the LENGTH OF s1  
        SET len2 TO the LENGTH OF s2  
        
        FOR index i FROM zero TO len1 MINUS one  
            SET pointer j TO zero  
            FOR index k FROM i TO len1 MINUS one  
                IF character at position k of s1 EQUALS character at position j of s2  
                    INCREMENT j BY one  
                    IF j EQUALS len2  
                        IF k MINUS i PLUS one LESS THAN min_len  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start EQUALS negative one  
            RETURN an empty sequence of characters representing empty string  
        ELSE  
            RETURN the SUBSEQUENCE of s1 FROM position min_start TO position min_start PLUS min_len MINUS one  
        END IF  
    END FUNCTION  
END CLASS
