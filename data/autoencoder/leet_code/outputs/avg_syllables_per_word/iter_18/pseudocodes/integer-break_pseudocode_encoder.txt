CLASS Solution  
    FUNCTION integerBreak WITH PARAMETER n  
        SET dp TO list of zeros with length equal to the quantity of n PLUS one  
        SET element at position one of dp TO one  
        FOR i FROM two TO the quantity of n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET candidate_product_one TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET candidate_product_two TO j MULTIPLIED BY i MINUS j  
                IF candidate_product_one GREATER THAN max_product  
                    SET max_product TO candidate_product_one  
                ELSE IF candidate_product_two GREATER THAN max_product  
                    SET max_product TO candidate_product_two  
                END IF  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
