CLASS Solution  
    FUNCTION exist WITH PARAMETERS board AS list of list of strings AND word AS string RETURNS boolean  
        IF board EQUALS empty collection  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero of board  
        SET visited TO a collection of m elements each BEING a collection of n elements each SET TO false  
        FUNCTION dfs WITH PARAMETERS x AS integer AND y AS integer AND index AS integer RETURNS boolean  
            IF index EQUALS the LENGTH OF word  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the element at position x and y of visited EQUALS true OR the character at position index of word NOT EQUALS the element at position x and y of board  
                RETURN false  
            END IF  
            SET the element at position x and y of visited TO true  
            FOR each pair dx and dy IN the collection of pairs consisting of minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one  
                IF dfs WITH ARGUMENTS x PLUS dx AND y PLUS dy AND index PLUS one RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
            SET the element at position x and y of visited TO false  
            RETURN false  
        END FUNCTION  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs WITH ARGUMENTS i AND j AND zero RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
