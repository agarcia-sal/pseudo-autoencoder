CLASS Solution  
    FUNCTION reachingPoints WITH PARAMETERS sx sy tx ty  
        WHILE tx GREATER THAN OR EQUAL TO sx AND ty GREATER THAN OR EQUAL TO sy HOLDS  
            IF tx EQUALS sx AND ty EQUALS sy  
                RETURN True  
            END IF  
            IF tx GREATER THAN ty  
                IF ty GREATER THAN sy  
                    SET tx TO the remainder of tx DIVIDED BY ty  
                ELSE  
                    RETURN the remainder of tx MINUS sx DIVIDED BY ty EQUALS zero  
                END IF  
            ELSE  
                IF tx GREATER THAN sx  
                    SET ty TO the remainder of ty DIVIDED BY tx  
                ELSE  
                    RETURN the remainder of ty MINUS sy DIVIDED BY tx EQUALS zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
