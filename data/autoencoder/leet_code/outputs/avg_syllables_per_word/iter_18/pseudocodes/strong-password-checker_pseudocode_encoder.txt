CLASS Solution
    FUNCTION strongPasswordChecker(password)
        SET n TO the LENGTH OF password
    
        SET has_lower TO whether any character IN password IS LOWERCASE
        SET has_upper TO whether any character IN password IS UPPERCASE
        SET has_digit TO whether any character IN password IS A DIGIT
    
        SET missing_types TO three MINUS the SUM OF has_lower PLUS has_upper PLUS has_digit
    
        SET replace TO zero
        SET one TO zero
        SET two TO zero
        SET i TO two
    
        WHILE the value of i IS LESS THAN n
            IF the character at position i of password EQUALS the character at position i MINUS one of password AND the character at position i of password EQUALS the character at position i MINUS two of password
                SET length TO two
                WHILE the value of i IS LESS THAN n AND the character at position i of password EQUALS the character at position i MINUS one of password
                    INCREMENT length BY one
                    INCREMENT i BY one
                END WHILE
                INCREMENT replace BY length DIVIDED BY three
                IF the remainder when length IS DIVIDED BY three EQUALS zero
                    INCREMENT one BY one
                ELSE IF the remainder when length IS DIVIDED BY three EQUALS one
                    INCREMENT two BY one
                END IF
            ELSE
                INCREMENT i BY one
            END IF
        END WHILE
    
        IF n IS LESS THAN six
            RETURN the GREATER VALUE BETWEEN missing_types AND six MINUS n
    
        ELSE IF n IS LESS THAN OR EQUAL TO twenty
            RETURN the GREATER VALUE BETWEEN missing_types AND replace
    
        ELSE
            SET delete TO n MINUS twenty
    
            SET replace TO replace MINUS the SMALLER VALUE BETWEEN delete AND one
            SET replace TO replace MINUS the INTEGER DIVISION BY two OF the SMALLER VALUE BETWEEN the GREATER VALUE BETWEEN delete MINUS one AND zero AND two MULTIPLIED BY two
            SET replace TO replace MINUS the INTEGER DIVISION BY three OF the GREATER VALUE BETWEEN delete MINUS one MINUS two MULTIPLIED BY two AND zero
    
            RETURN delete PLUS the GREATER VALUE BETWEEN missing_types AND replace
        END IF
    END FUNCTION
END CLASS
