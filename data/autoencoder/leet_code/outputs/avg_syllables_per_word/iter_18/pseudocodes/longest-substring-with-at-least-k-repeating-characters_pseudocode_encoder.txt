CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s AND k  
        IF the LENGTH OF s IS LESS THAN k THEN  
            RETURN zero  
        END IF  
        SET count TO the RESULT OF invoking Counter WITH s AS ARGUMENT  
        FOR each character IN count  
            IF the FREQUENCY OF character IN count IS LESS THAN k THEN  
                RETURN the MAXIMUM value AMONG the RESULT OF invoking longestSubstring ON each substring RESULTING FROM splitting s BY character WITH k AS ARGUMENT  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
