CLASS Solution  
    FUNCTION restoreIpAddresses WITH PARAMETER s  
        FUNCTION is_valid WITH PARAMETER segment  
            IF the LENGTH OF segment EQUALS one OR (the FIRST CHARACTER OF segment NOT EQUALS zero AND the NUMERICAL VALUE OF segment IS LESS THAN OR EQUAL TO two hundred and fifty five)  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION backtrack WITH PARAMETERS start AND path  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the JOINED STRING OF path USING the CHARACTER dot TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length IS LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING OF s FROM start TO start PLUS length  
                    IF is_valid WITH ARGUMENT segment RETURNS true  
                        CALL backtrack WITH ARGUMENTS start PLUS length AND the LIST CONCATENATION OF path AND A LIST CONTAINING segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO an EMPTY LIST  
        CALL backtrack WITH ARGUMENTS zero AND an EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
