CLASS Solution  
    FUNCTION deleteAndEarn(nums AS List of integers) RETURNS integer  
        SET count TO auxiliary FUNCTION Counter OF nums  
        SET unique_nums TO auxiliary FUNCTION sorted OF keys OF count  
        SET dp TO list of zeros OF LENGTH the LENGTH OF unique_nums PLUS one  
        SET value at position one of dp TO element at position zero of unique_nums MULTIPLIED BY element at key element at position zero of unique_nums in count  
        FOR index FROM two TO the LENGTH OF unique_nums PLUS one  
            SET num TO element at position index MINUS one of unique_nums  
            IF num EQUALS element at position index MINUS two of unique_nums PLUS one  
                SET element at position index of dp TO the greater value between element at position index MINUS one of dp AND element at position index MINUS two of dp PLUS num MULTIPLIED BY element at key num in count  
            ELSE  
                SET element at position index of dp TO element at position index MINUS one of dp PLUS num MULTIPLIED BY element at key num in count  
            END IF  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
