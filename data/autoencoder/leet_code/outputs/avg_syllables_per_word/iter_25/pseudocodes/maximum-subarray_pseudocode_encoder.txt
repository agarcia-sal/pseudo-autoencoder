CLASS Solution  
    FUNCTION maxSubArray WITH PARAMETER nums  
        SET max_sum TO element at position zero of nums  
        SET current_sum TO element at position zero of nums  
        FOR each num IN elements of nums from position one to the end  
            SET current_sum TO the maximum value between num and the sum of current_sum plus num  
            SET max_sum TO the maximum value between max_sum and current_sum  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
