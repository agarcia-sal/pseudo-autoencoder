CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO an empty list  
        FOR each num IN nums  
            perform the action of pushing num onto min_heap using the heap push operation  
            IF the length of min_heap is greater than k THEN  
                perform the action of popping the smallest element from min_heap using the heap pop operation  
            END IF  
        END FOR  
        RETURN the element at position zero of min_heap  
    END FUNCTION  
END CLASS
