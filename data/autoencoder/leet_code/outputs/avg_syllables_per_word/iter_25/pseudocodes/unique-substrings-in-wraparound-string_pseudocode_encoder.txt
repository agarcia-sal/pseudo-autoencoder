CLASS Solution  
    FUNCTION findSubstringInWraproundString WITH PARAMETER s  
        SET max_length TO a dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR each index i FROM zero TO the length of s MINUS one  
            IF i GREATER THAN zero AND the ASCII code of the character at position i of s MINUS the ASCII code of the character at position i MINUS one of s EQUALS one OR the character at position i of s EQUALS the character a AND the character at position i MINUS one of s EQUALS the character z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key the character at position i of s TO the greater value BETWEEN max_length at key the character at position i of s AND current_length  
        END FOR  
        SET total_unique_substrings TO the sum of all values in max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
