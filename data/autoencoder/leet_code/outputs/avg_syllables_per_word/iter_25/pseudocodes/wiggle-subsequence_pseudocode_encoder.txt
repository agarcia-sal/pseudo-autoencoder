CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF the LENGTH OF nums LESS THAN two  
            RETURN the LENGTH OF nums  
        END IF  
        SET up TO one  
        SET down TO one  
        FOR each index i FROM one TO the LENGTH OF nums MINUS one  
            IF the element at position i of nums GREATER THAN the element at position i MINUS one of nums  
                SET up TO down PLUS one  
            ELSE IF the element at position i of nums LESS THAN the element at position i MINUS one of nums  
                SET down TO up PLUS one  
            END IF  
        END FOR  
        RETURN the GREATER OF up AND down  
    END FUNCTION  
END CLASS
