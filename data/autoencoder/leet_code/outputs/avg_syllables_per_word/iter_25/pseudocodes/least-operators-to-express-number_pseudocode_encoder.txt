CLASS Solution  
    FUNCTION leastOpsExpressTarget with parameters x and target  
        FUNCTION dp with parameter target  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum between the value obtained by multiplying two by target and then subtracting one AND the value obtained by multiplying two by the difference of x minus target  
            END IF  
            SET power TO one  
            WHILE x raised to the power power LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO the value returned by calling dp with the argument target MINUS x raised to the power power THEN ADD power  
            IF x raised to the power power PLUS one MINUS target LESS THAN target  
                SET res TO the minimum between res AND the value returned by calling dp with the argument x raised to the power power PLUS one MINUS target THEN ADD power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN the value returned by calling dp with argument target  
    END FUNCTION  
END CLASS
