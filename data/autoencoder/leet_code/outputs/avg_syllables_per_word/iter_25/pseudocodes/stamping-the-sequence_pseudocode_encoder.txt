CLASS Solution  
    FUNCTION movesToStamp(stamp, target)  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list containing m repeated n MINUS m PLUS one times  
        INITIALIZE q TO an empty double ended queue  
        SET g TO a list of empty lists repeated n times  
        FOR i FROM zero TO n MINUS m  
            FOR each index j AND character c IN the enumeration of stamp  
                IF the element at position i PLUS j of target EQUALS c THEN  
                    DECREMENT the element at position i of indeg BY one  
                    IF the element at position i of indeg EQUALS zero THEN  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the element at position i PLUS j of g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list of False repeated n times  
        WHILE q is not empty  
            REMOVE the element from the front of q AND ASSIGN it TO i  
            APPEND i TO ans  
            FOR j FROM zero TO m MINUS one  
                IF the element at position i PLUS j of vis IS False THEN  
                    SET the element at position i PLUS j of vis TO True  
                    FOR each k IN the element at position i PLUS j of g  
                        DECREMENT the element at position k of indeg BY one  
                        IF the element at position k of indeg EQUALS zero THEN  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements of vis ARE True THEN  
            RETURN the reversal of ans  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
