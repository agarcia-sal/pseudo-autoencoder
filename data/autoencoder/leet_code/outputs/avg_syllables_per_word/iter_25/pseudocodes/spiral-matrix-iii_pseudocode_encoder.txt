CLASS Solution  
    FUNCTION spiralMatrixIII(rows  
                             cols  
                             rStart  
                             cStart)  
        SET directions TO list containing zero and one  
        followed by one and zero  
        followed by zero and negative one  
        followed by negative one and zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        WHILE the LENGTH OF visited is LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each element from one TO steps  
                IF x is GREATER THAN OR EQUAL TO zero AND x is LESS THAN rows AND y is GREATER THAN OR EQUAL TO zero AND y is LESS THAN cols  
                    APPEND list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions at first position  
                SET dy TO element at position direction_index of directions at second position  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        RETURN visited  
    END FUNCTION  
END CLASS
