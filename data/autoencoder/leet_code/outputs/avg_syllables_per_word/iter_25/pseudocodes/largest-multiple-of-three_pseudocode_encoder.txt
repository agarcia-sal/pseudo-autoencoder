CLASS Solution  
    FUNCTION largestMultipleOfThree WITH PARAMETER digits  
        SET count TO the frequency count of each digit in digits  
        SET total_sum TO the sum of all digits in digits  
        SET remainder TO the remainder when total_sum is divided by three  
      
        IF remainder EQUALS one  
            FOR each d IN the list consisting of one four seven  
                IF the count of d is greater than zero  
                    DECREMENT the count of d by one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list consisting of two five eight  
                    IF the count of d is greater than one  
                        DECREMENT the count of d by two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
      
        IF remainder EQUALS two  
            FOR each d IN the list consisting of two five eight  
                IF the count of d is greater than zero  
                    DECREMENT the count of d by one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list consisting of one four seven  
                    IF the count of d is greater than one  
                        DECREMENT the count of d by two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
      
        SET result TO an empty list  
        FOR d FROM nine DOWN TO zero INCLUSIVE  
            APPEND the string form of d repeated by the count of d times TO result  
        END FOR  
      
        SET final_number TO the concatenation of all elements in result  
      
        IF final_number is not empty AND the first character of final_number EQUALS zero  
            RETURN zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
