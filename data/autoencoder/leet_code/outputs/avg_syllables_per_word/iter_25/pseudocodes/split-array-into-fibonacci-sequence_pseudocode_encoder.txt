CLASS Solution  
    FUNCTION splitIntoFibonacci WITH PARAMETERS num  
        FUNCTION backtrack WITH PARAMETERS start path  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS zero AS STRING AND end GREATER THAN start PLUS one  
                    CONTINUE TO the NEXT ITERATION  
                END IF  
                SET next_num TO the INTEGER VALUE OF the SUBSTRING of num FROM position start TO position end MINUS one  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK OUT OF THE LOOP  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the ELEMENT at position LENGTH OF path MINUS one of path PLUS the ELEMENT at position LENGTH OF path MINUS two of path  
                    SET result TO the CALL OF backtrack WITH ARGUMENTS end AND path APPENDED WITH next_num  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an EMPTY LIST  
        END FUNCTION  
        RETURN the CALL OF backtrack WITH ARGUMENTS zero AND an EMPTY LIST  
    END FUNCTION  
END CLASS
