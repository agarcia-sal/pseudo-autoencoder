CLASS Solution  
    FUNCTION getMoneyAmount WITH PARAMETER n  
        FUNCTION dp WITH PARAMETERS left AND right  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER OF dp WITH left AND pivot MINUS one AND dp WITH pivot PLUS one AND right  
                SET min_cost TO the SMALLER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp WITH one AND n  
    END FUNCTION  
END CLASS
