CLASS Solution  
    FUNCTION subdomainVisits WITH PARAMETER cpdomains  
        SET domain_count TO a new mapping with default zero for each key  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count_and_domain components separated by space  
            SET count TO the integer value of the first element of count_and_domain  
            SET domain TO the second element of count_and_domain  
            SPLIT domain INTO subdomains separated by the period character  
            SET index TO zero  
            WHILE index LESS THAN the length of subdomains  
                JOIN the elements of subdomains from position index TO the end separated by the period character INTO subdomain  
                INCREMENT the value associated with subdomain in domain_count BY count  
                INCREMENT index BY one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        FOR each key and value as domain and count in the items of domain_count  
            APPEND the string formed by concatenating the string form of count then a space then domain TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
