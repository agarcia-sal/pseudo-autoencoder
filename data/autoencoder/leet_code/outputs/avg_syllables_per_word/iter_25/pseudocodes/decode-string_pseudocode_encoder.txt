CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO an empty list  
        SET a TO an empty string  
        SET d TO an empty string  
        FOR each char IN s  
            IF char is a digit  
                APPEND char TO d  
            ELSE IF char is an alphabetic character  
                APPEND char TO a  
            ELSE IF char EQUALS the character left bracket  
                APPEND the tuple consisting of a and the integer value of d TO stack  
                SET a TO an empty string  
                SET d TO an empty string  
            ELSE IF char EQUALS the character right bracket  
                REMOVE the last element FROM stack AND ASSIGN IT TO p AND n  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
