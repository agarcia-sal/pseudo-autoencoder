CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        PERFORM sorting of the list nums in ascending order  
        SET n TO the LENGTH OF the list nums  
        SET f TO a list of length n with every element initialized to one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums DIVIDED BY the element at position j of nums HAS zero REMAINDER  
                    SET the element at position i of f TO the maximum BETWEEN the current element at position i of f AND the element at position j of f PLUS one  
                END IF  
            END FOR  
            IF the element at position k of f IS LESS THAN the element at position i of f  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO the element at position k of f  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m IS GREATER THAN zero  
            IF the element at position k of nums DIVIDED BY the element at position i of nums HAS zero REMAINDER AND the element at position i of f IS EQUAL TO m  
                APPEND the element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
