CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val MINUS zero next MINUS None  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS NOT TRUTHY  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE element AT POSITION zero OF values  
    SET p TO head  
    FOR each val IN THE SUBSEQUENCE OF values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET THE next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS NOT TRUTHY OR p2 IS NOT TRUTHY  
        RETURN False  
    END IF  
    RETURN THE EQUALITY OF THE val ATTRIBUTE OF p1 AND THE val ATTRIBUTE OF p2 AND THE RESULT OF CALLING is_same_list WITH THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS self head n  
        SET dummy TO A NEW ListNode CREATED WITH zero  
        SET THE next ATTRIBUTE OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  

        FOR i FROM zero TO n PLUS one MINUS one  
            SET first TO THE next ATTRIBUTE OF first  
        END FOR  

        WHILE first IS NOT None  
            SET first TO THE next ATTRIBUTE OF first  
            SET second TO THE next ATTRIBUTE OF second  
        END WHILE  

        SET THE next ATTRIBUTE OF second TO THE next ATTRIBUTE OF THE next ATTRIBUTE OF second  

        RETURN THE next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
