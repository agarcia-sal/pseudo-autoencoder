CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        FUNCTION prefix_sum(matrix)  
            SET p TO a new list of size m plus one WHERE each element is a list of size n plus one FILLED WITH zero  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET element at position r of p and position c TO element at position r minus one of matrix and position c minus one PLUS element at position r minus one of p and position c PLUS element at position r of p and position c minus one MINUS element at position r minus one of p and position c minus one  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            RETURN element at position r2 plus one of p and position c2 plus one MINUS element at position r2 plus one of p and position c1 MINUS element at position r1 of p and position c2 plus one PLUS element at position r1 of p and position c1  
        END FUNCTION  
        
        SET p_grid TO the RESULT OF prefix_sum APPLIED TO grid  
        
        SET p_stamps TO a new list of size m plus one WHERE each element is a list of size n plus one FILLED WITH zero  
        
        FOR r FROM zero TO m minus one  
            FOR c FROM zero TO n minus one  
                IF element at position r of grid and position c EQUALS zero THEN  
                    IF r PLUS stampHeight LESS THAN OR EQUAL TO m AND c PLUS stampWidth LESS THAN OR EQUAL TO n THEN  
                        IF submatrix_sum APPLIED TO p_grid AND r AND c AND r PLUS stampHeight MINUS one AND c PLUS stampWidth MINUS one EQUALS zero THEN  
                            INCREMENT element at position r of p_stamps and position c BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO the RESULT OF prefix_sum APPLIED TO p_stamps  
        
        FOR r FROM zero TO m minus one  
            FOR c FROM zero TO n minus one  
                IF element at position r of grid and position c EQUALS zero THEN  
                    IF submatrix_sum APPLIED TO p_stamps AND the GREATER OF zero AND r MINUS stampHeight PLUS one AND the GREATER OF zero AND c MINUS stampWidth PLUS one AND r AND c EQUALS zero THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
