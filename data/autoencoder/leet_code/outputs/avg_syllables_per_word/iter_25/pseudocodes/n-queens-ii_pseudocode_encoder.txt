CLASS Solution  
    FUNCTION totalNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN the negation OF the SUM OF element at position col of cols PLUS element at position row MINUS col of hills PLUS element at position row PLUS col of dales  
        END FUNCTION  
        FUNCTION place_queen WITH PARAMETERS row AND col  
            SET element at position col of cols TO one  
            SET element at position row MINUS col of hills TO one  
            SET element at position row PLUS col of dales TO one  
        END FUNCTION  
        FUNCTION remove_queen WITH PARAMETERS row AND col  
            SET element at position col of cols TO zero  
            SET element at position row MINUS col of hills TO zero  
            SET element at position row PLUS col of dales TO zero  
        END FUNCTION  
        FUNCTION backtrack WITH PARAMETERS row DEFAULTED TO zero AND count DEFAULTED TO zero  
            FOR col FROM zero TO n MINUS one  
                IF the result of is_not_under_attack WITH row AND col IS true  
                    CALL place_queen WITH row AND col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO the result of backtrack WITH row PLUS one AND count  
                    END IF  
                    CALL remove_queen WITH row AND col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET cols TO a list of zeros WITH LENGTH n  
        SET hills TO a list of zeros WITH LENGTH two MULTIPLIED BY n MINUS one  
        SET dales TO a list of zeros WITH LENGTH two MULTIPLIED BY n MINUS one  
        RETURN the result of backtrack  
    END FUNCTION  
END CLASS
