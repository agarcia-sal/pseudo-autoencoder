CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty set  
        FOR each pair x and y IN fronts and backs zipped together  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO a set containing all elements from fronts and backs combined  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers IS NOT empty  
            SET minimum_value TO the smallest element in good_numbers  
            RETURN minimum_value  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
