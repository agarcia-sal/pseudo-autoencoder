CLASS Solution  
    FUNCTION monotoneIncreasingDigits WITH PARAMETER n  
        SET digits TO the list of characters representing the string form of n  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one DOWN TO one STEP MINUS one  
            IF the element at position i of digits LESS THAN the element at position i MINUS one of digits THEN  
                SET the element at position i MINUS one of digits TO the string form of the integer form of the element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i of digits TO the string representing nine  
        END FOR  
        SET result TO the integer form of the concatenation of all elements in digits in order  
        RETURN result  
    END FUNCTION  
END CLASS
