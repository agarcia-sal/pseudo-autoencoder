CLASS Solution  
    FUNCTION getXORSum WITH PARAMETERS arr1 AND arr2  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the result of exclusive OR between xor_sum_arr1 and num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the result of exclusive OR between xor_sum_arr2 and num  
        END FOR  
        RETURN the result of logical AND between xor_sum_arr1 and xor_sum_arr2  
    END FUNCTION  
END CLASS
