CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        SET below_twenty TO the list of words for numbers from zero empty string through nineteen One Two Three Four Five Six Seven Eight Nine Ten Eleven Twelve Thirteen Fourteen Fifteen Sixteen Seventeen Eighteen Nineteen  
        SET tens TO the list of words for multiples of ten empty string empty string Twenty Thirty Forty Fifty Sixty Seventy Eighty Ninety  
        SET thousands TO the list of scale words empty string Thousand Million Billion  
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN an empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty PLUS a space character  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n by ten of tens PLUS a space character PLUS the result of calling helper with the remainder of n divided by ten  
            ELSE  
                RETURN the element at position the integer division of n by one hundred of below_twenty PLUS the word Hundred PLUS a space character PLUS the result of calling helper with the remainder of n divided by one hundred  
            END IF  
        END FUNCTION  
        SET result TO an empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            IF the remainder of num divided by one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num divided by one thousand PLUS the element at position i of thousands PLUS a space character PLUS result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  
        RETURN result with leading and trailing whitespace removed  
    END FUNCTION  
END CLASS
