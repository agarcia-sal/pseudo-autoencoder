CLASS Solution  
    FUNCTION uniqueLetterString WITH PARAMETER s  
        SET MOD TO ten POWERED BY nine PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO a list containing minus one repeated twenty six times  
        SET last2 TO a list containing minus one repeated twenty six times  
        SET result TO zero  
        FOR each index i AND character char IN s WITH ENUMERATION  
            SET idx TO the Unicode ordinal value of char MINUS the Unicode ordinal value of uppercase A  
            SET contribution TO the product of the difference BETWEEN i AND the element at position idx of last AND the difference BETWEEN the element at position idx of last AND the element at position idx of last2  
            SET result TO the remainder of the sum of result PLUS contribution DIVIDED BY MOD  
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        FOR each integer i FROM zero TO twenty five  
            SET contribution TO the product of the difference BETWEEN n AND the element at position i of last AND the difference BETWEEN the element at position i of last AND the element at position i of last2  
            SET result TO the remainder of the sum of result PLUS contribution DIVIDED BY MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
