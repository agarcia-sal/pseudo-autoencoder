CLASS Solution  
    FUNCTION replaceNonCoprimes WITH PARAMETER nums  
        SET stack TO an empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero  
                SET g TO the greatest common divisor OF the element at the last position of stack AND num  
                IF g EQUALS one  
                    BREAK from the while loop  
                END IF  
                SET num TO the element at the last position of stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE the last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
