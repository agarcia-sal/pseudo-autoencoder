CLASS Solution  
    FUNCTION canCross WITH stones  
        IF stones IS empty OR the LENGTH OF stones IS less than two  
            RETURN False  
        END IF  
        SET stone_set TO the SET of stones  
        SET memo TO an empty collection  
        FUNCTION can_jump_to WITH position AND jump  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            IF the TUPLE of position AND jump IS present in memo  
                RETURN the value in memo at the key of the TUPLE of position AND jump  
            END IF  
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND the SUM of position AND next_jump IS in stone_set  
                    IF can_jump_to WITH the SUM of position AND next_jump AND next_jump RETURNS True  
                        SET the value in memo at the key of the TUPLE of position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET the value in memo at the key of the TUPLE of position AND jump TO False  
            RETURN False  
        END FUNCTION  
        RETURN can_jump_to WITH the element at position zero of stones AND zero  
    END FUNCTION  
END CLASS
