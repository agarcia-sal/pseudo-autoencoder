CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH PARAMETER s  
        SET sequence_count TO a new dictionary with default integer zero  
        FOR variable i FROM zero TO the length of s MINUS ten PLUS one  
            SET sequence TO the substring from position i TO position i PLUS ten MINUS one of s  
            INCREMENT the value at key sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each pair of sequence and count IN the items of sequence_count  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
