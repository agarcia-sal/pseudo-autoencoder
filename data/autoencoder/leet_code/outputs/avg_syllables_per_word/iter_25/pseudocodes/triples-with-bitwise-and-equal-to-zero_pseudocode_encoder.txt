CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a new default dictionary mapping integer to integer initially zero  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET bitwise_and_result TO bitwise AND of element i and element j  
                INCREMENT the value at key bitwise_and_result in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and and corresponding value count IN the items of pair_and_count  
                IF bitwise AND of pair_and and element k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
