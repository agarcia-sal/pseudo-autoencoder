CLASS Solution  
    FUNCTION winnerSquareGame WITH PARAMETER n  
        SET dp TO a list of boolean False values of length n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE the square of j is LESS THAN OR EQUAL TO i  
                IF the value at position i MINUS the square of j in dp is NOT True  
                    SET the value at position i in dp TO True  
                    BREAK the current loop  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the value at position n in dp  
    END FUNCTION  
END CLASS
