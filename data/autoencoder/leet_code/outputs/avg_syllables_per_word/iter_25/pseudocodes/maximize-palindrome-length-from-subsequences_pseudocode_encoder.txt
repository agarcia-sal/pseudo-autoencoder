CLASS Solution  
    FUNCTION longestPalindrome WITH PARAMETERS word1 word2  
        SET s TO the concatenation of word1 and word2  
        SET n TO the length of s  
        SET dp TO a two dimensional list of size n by n filled with zero  
        
        FOR i FROM zero TO n MINUS one  
            SET the element at row i column i of dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF the character at position i of s EQUALS the character at position j of s THEN  
                    SET the element at row i column j of dp TO the element at row i PLUS one column j MINUS one of dp PLUS two  
                    IF i LESS THAN the length of word1 AND j GREATER THAN OR EQUAL TO the length of word1 THEN  
                        SET max_len TO the maximum of max_len and the element at row i column j of dp  
                    END IF  
                ELSE  
                    SET the element at row i column j of dp TO the maximum of the element at row i PLUS one column j of dp and the element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
