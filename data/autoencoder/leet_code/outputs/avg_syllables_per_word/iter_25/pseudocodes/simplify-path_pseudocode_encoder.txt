CLASS Solution  
    FUNCTION simplifyPath WITH PARAMETER path  
        SET components TO the list of elements obtained by dividing path according to the forward slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS an empty string OR component EQUALS the single character dot  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two consecutive dots  
                IF the list stack is not empty  
                    REMOVE the last element from stack  
                END IF  
            ELSE  
                ADD component to the end of the list stack  
            END IF  
        END FOR  
        SET simplified_path TO the concatenation of the forward slash character and the elements of stack joined by forward slash characters  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
