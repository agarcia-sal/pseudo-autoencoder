CLASS Solution  
    FUNCTION convert WITH PARAMETERS s AND numRows  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list containing numRows elements each being an empty string  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each char IN s  
            APPEND char TO the string at position current_row of rows  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO the logical negation of going_down  
            END IF  
            IF going_down IS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result TO the concatenation of all strings in rows in order  
        RETURN result  
    END FUNCTION  
END CLASS
