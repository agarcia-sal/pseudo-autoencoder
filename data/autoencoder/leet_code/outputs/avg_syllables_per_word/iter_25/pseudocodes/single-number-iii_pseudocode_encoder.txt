CLASS Solution  
    FUNCTION singleNumber WITH parameter nums  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO xor_all XOR num  
        END FOR  
        SET diff_bit TO xor_all AND negative xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF num AND diff_bit is not equal to zero  
                SET num1 TO num1 XOR num  
            ELSE  
                SET num2 TO num2 XOR num  
            END IF  
        END FOR  
        RETURN a list containing num1 followed by num2  
    END FUNCTION  
END CLASS
