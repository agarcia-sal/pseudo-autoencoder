CLASS Solution  
    FUNCTION reconstructQueue WITH parameter people  
        SORT the list people BY ordering the negative value of the element at position zero of each sublist AND THEN BY the value of the element at position one of each sublist in ascending order  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT position equal to the element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
