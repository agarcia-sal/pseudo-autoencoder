CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied by OR token EQUALS divided by  
                REMOVE and RETURN the last element FROM stack AND ASSIGN it TO b  
                REMOVE and RETURN the last element FROM stack AND ASSIGN it TO a  
                IF token EQUALS plus  
                    APPEND the sum of a PLUS b TO stack  
                ELSE IF token EQUALS minus  
                    APPEND the difference of a MINUS b TO stack  
                ELSE IF token EQUALS multiplied by  
                    APPEND the product of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS divided by  
                    APPEND the integer value of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer conversion of token TO stack  
            END IF  
        END FOR  
        RETURN the element at position zero of stack  
    END FUNCTION  
END CLASS
