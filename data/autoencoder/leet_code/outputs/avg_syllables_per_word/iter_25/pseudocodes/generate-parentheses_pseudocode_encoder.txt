CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s AS empty string left AS zero right AS zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH the character LEFT PARENTHESIS left INCREMENTED BY one right AS IS  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH the character RIGHT PARENTHESIS left AS IS right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH DEFAULT PARAMETERS  
        RETURN result  
    END FUNCTION  
END CLASS
