CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        REMOVE the leftmost element FROM queue AND ASSIGN IT TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree called with the left attribute of p and the left attribute of q AND the result of is_same_tree called with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF the LENGTH OF preorder EQUALS zero OR the LENGTH OF inorder EQUALS zero THEN  
            RETURN None  
        END IF  
        SET root_val TO the element at position zero of preorder  
        SET root TO a new TreeNode initialized with root_val  
        SET root_index_inorder TO the index of root_val in inorder  
        SET root.left TO the result of calling buildTree on self with the slice of preorder from position one TO position root_index_inorder PLUS one and the slice of inorder from position zero TO position root_index_inorder MINUS one PLUS one  
        SET root.right TO the result of calling buildTree on self with the slice of preorder from position root_index_inorder PLUS one TO the end and the slice of inorder from position root_index_inorder PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
