CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero left defaulting to none right defaulting to none  
        SET the value attribute of the instance TO val  
        SET the left attribute of the instance TO left  
        SET the right attribute of the instance TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values as list  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from the left side of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the value attribute of p is not equal to the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters the left attribute of p and the left attribute of q AND the result of is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST with parameters root and val  
        FUNCTION dfs with parameter root  
            IF root is none  
                RETURN a new instance of TreeNode initialized with val  
            END IF  
            IF the value attribute of root is less than val  
                SET the right attribute of root TO the result of dfs with argument the right attribute of root  
            ELSE  
                SET the left attribute of root TO the result of dfs with argument the left attribute of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs with argument root  
    END FUNCTION  
END CLASS
