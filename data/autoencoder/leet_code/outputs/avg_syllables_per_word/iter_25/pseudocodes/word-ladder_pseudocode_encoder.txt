CLASS Solution  
    FUNCTION ladderLength WITH PARAMETERS beginWord endWord wordList  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        SET word_dict TO a new default dictionary with list as default value  
        SET word_length TO the length of beginWord  
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of the substring of word from start TO position i MINUS one PLUS the character seven and PLUS the substring of word from position i PLUS one TO the end  
                APPEND word TO the list at key intermediate_word in word_dict  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue containing the tuple of beginWord and one  
        SET visited TO a new set containing beginWord  
        WHILE queue is not empty  
            REMOVE the element from the front of queue and assign to current_word and current_length  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of the substring of current_word from start TO position i MINUS one PLUS the character seven and PLUS the substring of current_word from position i PLUS one TO the end  
                FOR each word IN the list at key intermediate_word in word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
