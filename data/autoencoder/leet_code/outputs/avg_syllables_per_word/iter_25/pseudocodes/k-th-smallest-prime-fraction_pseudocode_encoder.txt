CLASS Solution  
    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR each i FROM zero TO n MINUS one EXCLUSIVE  
            ADD TO min_heap the TUPLE consisting of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr the value i and the value n MINUS one  
        END FOR  
        FOR each count FROM zero TO k MINUS one MINUS one EXCLUSIVE  
            REMOVE the smallest element FROM min_heap ASSIGNING the values TO unnamed variable i and j respectively ignoring the first element  
            IF the value j MINUS one IS GREATER THAN the value i  
                ADD TO min_heap the TUPLE consisting of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr the value i and the value j MINUS one  
            END IF  
        END FOR  
        REMOVE the smallest element FROM min_heap ASSIGNING the values TO unnamed variable i and j respectively ignoring the first element  
        RETURN a list containing the element at position i of arr and the element at position j of arr  
    END FUNCTION  
END CLASS
