CLASS Solution  
    FUNCTION findDuplicate WITH PARAMETER nums  
        FUNCTION f WITH PARAMETER x  
            SET count TO zero  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count GREATER THAN x  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        SET length TO the LENGTH OF nums  
        SET search_range TO the SEQUENCE FROM zero TO length MINUS one  
        SET result TO the POSITION of the LEFTMOST element IN search_range FOR WHICH f RETURNS true  
        RETURN result  
    END FUNCTION  
END CLASS
