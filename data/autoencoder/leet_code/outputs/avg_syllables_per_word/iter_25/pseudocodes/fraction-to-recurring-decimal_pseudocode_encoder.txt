CLASS Solution  
    FUNCTION fractionToDecimal numerator denominator  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string consisting of a hyphen  
        ELSE  
            SET sign TO the empty string  
        END IF  
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        SET quotient TO the integer division of numerator by denominator  
        SET remainder TO the remainder of numerator divided by denominator  
        SET integer_part TO the string representation of quotient  
        IF remainder EQUALS zero  
            RETURN the concatenation of sign and integer_part  
        END IF  
        SET seen_remainders TO an empty dictionary  
        SET fractional_part TO an empty list  
        WHILE remainder NOT EQUAL TO zero  
            IF remainder IS A KEY IN seen_remainders  
                SET start TO the value in seen_remainders at key remainder  
                SET non_repeating TO the concatenation of the elements of fractional_part from position zero UP TO but NOT INCLUDING position start  
                SET repeating TO the concatenation of the elements of fractional_part from position start TO the end  
                RETURN the concatenation of sign integer_part a period non_repeating an opening parenthesis repeating and a closing parenthesis  
            END IF  
            SET the value in seen_remainders at key remainder TO the length of fractional_part  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer division of remainder by denominator  
            SET remainder TO the remainder of remainder divided by denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        RETURN the concatenation of sign integer_part a period and the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
