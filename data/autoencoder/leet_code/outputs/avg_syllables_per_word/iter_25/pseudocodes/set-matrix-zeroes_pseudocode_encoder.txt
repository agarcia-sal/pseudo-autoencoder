CLASS Solution  
    FUNCTION setZeroes WITH PARAMETERS matrix  
        IF matrix IS empty OR the first element of matrix IS empty  
            RETURN  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF the element at position zero of matrix  
        SET first_row_has_zero TO whether there EXISTS an index j FROM zero TO n MINUS one SUCH THAT the element at row zero and column j of matrix EQUALS zero  
        SET first_col_has_zero TO whether there EXISTS an index i FROM zero TO m MINUS one SUCH THAT the element at row i and column zero of matrix EQUALS zero  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF the element at row i and column j of matrix EQUALS zero  
                    SET the element at row i and column zero of matrix TO zero  
                    SET the element at row zero and column j of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF the element at row i and column zero of matrix EQUALS zero OR the element at row zero and column j of matrix EQUALS zero  
                    SET the element at row i and column j of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        IF first_row_has_zero  
            FOR j FROM zero TO n MINUS one  
                SET the element at row zero and column j of matrix TO zero  
            END FOR  
        END IF  
        IF first_col_has_zero  
            FOR i FROM zero TO m MINUS one  
                SET the element at row i and column zero of matrix TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
