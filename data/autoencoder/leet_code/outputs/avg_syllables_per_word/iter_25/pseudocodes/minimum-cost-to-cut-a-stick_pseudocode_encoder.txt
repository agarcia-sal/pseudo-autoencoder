CLASS Solution  
    FUNCTION minCost WITH PARAMETERS n AND cuts  
        SET cuts TO the sorted list resulting from appending zero AND n TO the original cuts list  
        
        FUNCTION dp WITH PARAMETERS left AND right  
            IF right MINUS left IS LESS THAN OR EQUAL TO one  
                RETURN zero  
            END IF  
            
            SET min_cost TO a value representing positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS the result of calling dp WITH left AND i PLUS the result of calling dp WITH i AND right  
                IF cost IS LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN the result of calling dp WITH zero AND the length of cuts MINUS one  
    END FUNCTION  
END CLASS
