CLASS Solution  
    FUNCTION movesToChessboard WITH PARAMETER board  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the value at position i of board XOR the value at position i of the first row of board XOR the value at position j of the first column of board XOR the value at position zero of the first row of board IS NOT EQUAL TO zero  
                    RETURN the MINUS one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the value at position i of the first column of board  
            INCREMENT colSum BY the value at position i of the first row of board  
            INCREMENT rowSwap BY the condition that the value at position i of the first column of board IS EQUAL TO the remainder of i DIVIDED BY two  
            INCREMENT colSwap BY the condition that the value at position i of the first row of board IS EQUAL TO the remainder of i DIVIDED BY two  
        END FOR  
        IF the integer division of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN the MINUS one  
        END IF  
        IF the integer division of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN the MINUS one  
        END IF  
        IF the remainder of N DIVIDED BY two IS NOT EQUAL TO zero  
            IF the remainder of rowSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM of colSwap AND N MINUS colSwap  
        END IF  
        RETURN the integer division of the sum of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
