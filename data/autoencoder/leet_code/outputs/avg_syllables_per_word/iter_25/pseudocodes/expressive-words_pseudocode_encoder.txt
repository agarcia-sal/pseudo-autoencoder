CLASS Solution  
    FUNCTION expressiveWords WITH PARAMETERS s AS string AND words AS list of strings  
        FUNCTION compress WITH PARAMETER word  
            IF word IS empty  
                RETURN empty list  
            END IF  
            SET count TO one  
            SET compressed TO empty list  
            FOR i FROM one TO the LENGTH OF word MINUS one  
                IF element at position i of word EQUALS element at position i MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND a pair consisting of element at position i MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND a pair consisting of element at the last position of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy WITH PARAMETERS s_compressed AND word_compressed  
            IF the LENGTH OF s_compressed IS NOT EQUAL TO the LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair sc AND sc_count IN s_compressed AND each pair wc AND wc_count IN word_compressed SIMULTANEOUSLY  
                IF sc IS NOT EQUAL TO wc  
                    RETURN false  
                END IF  
                IF sc_count IS NOT EQUAL TO wc_count AND (sc_count IS LESS THAN three OR sc_count IS LESS THAN wc_count)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET s_compressed TO the RESULT OF compress WITH ARGUMENT s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO the RESULT OF compress WITH ARGUMENT word  
            IF the RESULT OF is_stretchy WITH ARGUMENTS s_compressed AND word_compressed IS true  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
