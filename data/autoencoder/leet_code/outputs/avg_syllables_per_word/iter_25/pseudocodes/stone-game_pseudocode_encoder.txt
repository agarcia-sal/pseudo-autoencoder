CLASS Solution  
    FUNCTION stoneGame WITH piles  
        FUNCTION dfs WITH i AND j  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            RETURN the GREATER OF the element at position i of piles MINUS dfs CALLED WITH i INCREMENTED BY one AND j AND the element at position j of piles MINUS dfs CALLED WITH i AND j DECREMENTED BY one  
        END FUNCTION  
        RETURN dfs CALLED WITH zero AND the LENGTH OF piles MINUS one GREATER THAN zero  
    END FUNCTION  
END CLASS
