CLASS Solution  
    FUNCTION kthSmallest WITH PARAMETERS mat AND k  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET min_heap TO a list containing a tuple where the first element is the sum of element at position zero of each row in mat and the second element is a tuple of zeros repeated m times  
        SET visited TO a set containing the second element of the first element of min_heap  
        
        WHILE k is GREATER THAN zero  
            EXTRACT current_sum AND indices BY REMOVING the SMALLEST element from min_heap  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF the element at position i of indices INCREMENTED BY one is LESS THAN n  
                    SET new_indices TO a list copy of indices  
                    INCREMENT element at position i of new_indices BY one  
                    SET new_indices_tuple TO a tuple of new_indices  
                    SET new_sum TO current_sum MINUS the element at position i of mat AT position element at position i of indices PLUS the element at position i of mat AT position element at position i of new_indices  
                    
                    IF new_indices_tuple is NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        ADD the tuple consisting of new_sum AND new_indices_tuple TO min_heap USING a heap push operation  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
