CLASS Solution  
    FUNCTION rob WITH PARAMETER nums  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  
        SET dp TO a list of zeroes of the LENGTH OF nums  
        SET element at position zero of dp TO element at position zero of nums  
        IF element at position zero of nums IS GREATER THAN element at position one of nums  
            SET element at position one of dp TO element at position zero of nums  
        ELSE  
            SET element at position one of dp TO element at position one of nums  
        END IF  
        FOR i FROM two TO the LENGTH OF nums MINUS one  
            IF element at position i MINUS one of dp IS GREATER THAN element at position i MINUS two of dp PLUS element at position i of nums  
                SET element at position i of dp TO element at position i MINUS one of dp  
            ELSE  
                SET element at position i of dp TO element at position i MINUS two of dp PLUS element at position i of nums  
            END IF  
        END FOR  
        RETURN element at the LAST POSITION of dp  
    END FUNCTION  
END CLASS
