CLASS Solution  
    FUNCTION digitsCount WITH PARAMETERS d low high  
        FUNCTION count_digit WITH PARAMETERS n digit  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero  
                SET last TO the remainder when current IS DIVIDED BY ten  
                SET current TO the integer quotient when current IS DIVIDED BY ten  
                SET full_cycles TO current  
                IF digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUALS digit  
                    INCREMENT count BY the remainder when n IS DIVIDED BY power_of_10 PLUS one  
                END IF  
                SET power_of_10 TO power_of_10 MULTIPLIED BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  
        RETURN the result of count_digit WITH ARGUMENTS high d MINUS the result of count_digit WITH ARGUMENTS low MINUS one d  
    END FUNCTION  
END CLASS
