CLASS Solution  
    FUNCTION canDistribute(nums QUANTITY quantity)  
        SET count TO the list of values representing the frequency of each unique element in nums  
        SORT quantity in order from largest to smallest quantity  
        FUNCTION can_satisfy(index)  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN True  
            END IF  
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF the element at position i of count IS GREATER THAN OR EQUAL TO the element at position index of quantity THEN  
                    DECREMENT the element at position i of count BY the element at position index of quantity  
                    IF can_satisfy with argument index PLUS one RETURNS True THEN  
                        RETURN True  
                    END IF  
                    INCREMENT the element at position i of count BY the element at position index of quantity  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN the result of calling can_satisfy with argument zero  
    END FUNCTION  
END CLASS
