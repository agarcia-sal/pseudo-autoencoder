CLASS Solution  
    FUNCTION solve(board)  
        IF board IS EMPTY OR element at position zero of board IS EMPTY  
            RETURN  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        FUNCTION dfs(x y)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of board at position y NOT EQUALS capital letter O  
                RETURN  
            END IF  
            SET element at position x of board at position y TO capital letter S  
            CALL dfs WITH x PLUS one AND y  
            CALL dfs WITH x MINUS one AND y  
            CALL dfs WITH x AND y PLUS one  
            CALL dfs WITH x AND y MINUS one  
        END FUNCTION  
        FOR i FROM zero TO m MINUS one  
            CALL dfs WITH i AND zero  
            CALL dfs WITH i AND n MINUS one  
        END FOR  
        FOR j FROM zero TO n MINUS one  
            CALL dfs WITH zero AND j  
            CALL dfs WITH m MINUS one AND j  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS capital letter O  
                    SET element at position i of board at position j TO capital letter X  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS capital letter S  
                    SET element at position i of board at position j TO capital letter O  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
