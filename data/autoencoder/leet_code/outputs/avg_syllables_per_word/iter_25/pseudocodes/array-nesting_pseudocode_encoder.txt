CLASS Solution  
    FUNCTION arrayNesting WITH PARAMETER nums  
        SET visited TO a list of boolean false repeated the LENGTH OF nums times  
        SET max_length TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position i of visited IS NOT true  
                SET current_length TO zero  
                SET k TO i  
                WHILE the element at position k of visited IS NOT true  
                    SET the element at position k of visited TO true  
                    SET k TO the element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO the GREATER VALUE BETWEEN max_length AND current_length  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
