CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list CONSISTING OF zero and one THEN one and zero THEN zero and MINUS one THEN MINUS one and zero  
        SET min_heap TO a list CONSISTING OF a tuple of zero zero zero representing the cost and coordinates  
        SET visited TO a set CONSISTING OF a tuple of zero zero representing the starting coordinates  
        
        WHILE min_heap IS NOT empty  
            REMOVE and RETURN the smallest element FROM min_heap ASSIGNING cost AND x AND y FROM the removed element  
            
            IF a tuple of x and y EQUALS a tuple of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the tuple of nx AND ny IS NOT IN visited  
                    ADD the tuple of nx AND ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid AT position ny  
                    INSERT the tuple of new_cost nx ny INTO min_heap preserving the heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
