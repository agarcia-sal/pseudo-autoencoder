CLASS Solution  
    FUNCTION isPossible WITH PARAMETER target  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND the negation of num TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure maintaining heap property  
        SET total_sum TO the sum of all elements in target  
        WHILE the element at position zero of max_heap is less than the negation of one  
            REMOVE and RETURN the smallest element from max_heap and assign its negation TO largest  
            SET rest_sum TO total_sum MINUS largest  
            IF largest is less than or equal to rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
            SET new_value TO the remainder when largest is divided BY rest_sum  
            IF new_value EQUALS zero AND rest_sum is not equal to one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            ADD the negation of new_value TO max_heap maintaining heap property  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
