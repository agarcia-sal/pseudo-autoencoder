CLASS Solution  
    FUNCTION maxChunksToSorted WITH PARAMETER arr  
        SET max_so_far TO empty list  
        SET current_max TO element at position zero of arr  
        FOR each num IN arr  
            SET current_max TO greater value BETWEEN current_max AND num  
            APPEND current_max TO max_so_far  
        END FOR  
        SET sorted_arr TO arr arranged in ascending order  
        SET chunks TO zero  
        SET max_chunk TO zero  
        FOR each index i FROM zero TO length of arr MINUS one  
            SET max_chunk TO greater value BETWEEN max_chunk AND element at position i of arr  
            IF max_chunk EQUALS element at position i of sorted_arr  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
