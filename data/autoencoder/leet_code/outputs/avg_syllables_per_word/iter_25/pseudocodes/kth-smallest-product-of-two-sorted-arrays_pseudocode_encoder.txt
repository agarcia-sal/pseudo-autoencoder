CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 NUMBERS, nums2 NUMBERS, k NUMBER) RETURNS NUMBER  
        FUNCTION count(p NUMBER) RETURNS NUMBER  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x GREATER THAN zero THEN  
                    INCREMENT cnt BY the POSITION in nums2 WHERE p DIVIDED BY x WOULD BE INSERTED TO THE RIGHT  
                ELSE IF x LESS THAN zero THEN  
                    INCREMENT cnt BY n MINUS the POSITION in nums2 WHERE p DIVIDED BY x WOULD BE INSERTED TO THE LEFT  
                ELSE  
                    IF p GREATER THAN OR EQUAL TO zero THEN  
                        INCREMENT cnt BY n  
                    END IF  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  
        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION in the RANGE FROM negative mx TO mx inclusive WHERE k WOULD BE INSERTED TO THE LEFT USING count AS THE KEY FUNCTION MINUS mx  
    END FUNCTION  
END CLASS
