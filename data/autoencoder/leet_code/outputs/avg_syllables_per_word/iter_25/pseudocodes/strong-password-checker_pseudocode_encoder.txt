CLASS Solution  
    FUNCTION strongPasswordChecker password  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO whether there EXISTS a character in password that IS a lowercase letter  
        SET has_upper TO whether there EXISTS a character in password that IS an uppercase letter  
        SET has_digit TO whether there EXISTS a character in password that IS a digit  
        
        SET missing_types TO three MINUS the SUM of has_lower PLUS has_upper PLUS has_digit  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        WHILE i LESS THAN n  
            IF the character at position i of password EQUALS the character at position i MINUS one of password AND the character at position i of password EQUALS the character at position i MINUS two of password  
                SET length TO two  
                WHILE i LESS THAN n AND the character at position i of password EQUALS the character at position i MINUS one of password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY the INTEGER DIVISION of length BY three  
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n LESS THAN six  
            RETURN the MAXIMUM OF missing_types AND six MINUS n  
        ELSE IF n LESS THAN OR EQUAL TO twenty  
            RETURN the MAXIMUM OF missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            DECREMENT replace BY the MINIMUM OF delete AND one  
            DECREMENT replace BY the INTEGER DIVISION BY two OF the MINIMUM OF the MAXIMUM OF delete MINUS one AND zero AND two MULTIPLIED BY two  
            DECREMENT replace BY the INTEGER DIVISION BY three OF the MAXIMUM OF delete MINUS one MINUS two MULTIPLIED BY two AND zero  
            
            RETURN delete PLUS the MAXIMUM OF missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
