CLASS Solution  
    FUNCTION dailyTemperatures WITH PARAMETER temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a list of zeros with length n  
        SET stack TO an empty list  
        FOR i FROM zero TO n MINUS one  
            WHILE stack is not empty AND the element at position i of temperatures GREATER THAN the element at position equal to the last element of stack of temperatures  
                SET prev_index TO the REMOVAL of the last element from stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
