CLASS Solution  
    FUNCTION maxPerformance with parameters n speed efficiency k  
        SET engineers TO a list of pairs each containing corresponding elements from efficiency and speed  
        SORT engineers in descending order based on their first element which is efficiency  
        SET speed_heap TO an empty collection that maintains elements in ascending order  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff spd IN engineers  
            ADD spd to speed_heap maintaining ascending order  
            INCREMENT speed_sum BY spd  
            IF the number of elements in speed_heap is greater than k THEN  
                REMOVE the smallest element from speed_heap and DECREMENT speed_sum by that element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance is greater than max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance modulo one billion plus seven  
    END FUNCTION  
END CLASS
