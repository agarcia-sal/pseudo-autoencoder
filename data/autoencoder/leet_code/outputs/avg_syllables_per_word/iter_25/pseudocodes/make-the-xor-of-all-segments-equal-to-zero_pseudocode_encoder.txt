CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO a new mapping from keys to lists with default empty list  
        FOR each index i FROM zero TO the length of nums MINUS one  
            APPEND element at position i of nums TO the list at key i MODULO k in groups  
        END FOR  
        
        SET freq TO a list of k mappings from keys to integers with default zero  
        SET min_size TO a list of k elements each set to positive infinity  
        
        FOR each index i FROM zero TO k MINUS one  
            FOR each number num IN the list at key i in groups  
                INCREMENT freq at position i at key num BY one  
            END FOR  
            SET min_size at position i TO the length of the list at key i in groups  
        END FOR  
        
        SET dp TO a list of length two raised to the power ten each element set to positive infinity  
        SET element at position zero of dp TO zero  
        
        FOR each index i FROM zero TO k MINUS one  
            SET new_dp TO a list of length two raised to the power ten each element set to positive infinity  
            SET min_dp TO the minimum value in dp  
            FOR each index j FROM zero TO two raised to the power ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS min_size at position i  
                FOR each key value pair num and count IN freq at position i  
                    IF element at position j of new_dp GREATER THAN element at position j XOR num of dp PLUS min_size at position i MINUS count THEN  
                        SET element at position j of new_dp TO element at position j XOR num of dp PLUS min_size at position i MINUS count  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
