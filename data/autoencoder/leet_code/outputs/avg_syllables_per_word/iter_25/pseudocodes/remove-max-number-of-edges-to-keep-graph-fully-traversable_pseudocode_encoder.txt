CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER n  
        SET parent TO a list of integers starting from zero up to and including n  
        SET rank TO a list containing n plus one zeroes  
        SET count TO n  
    END FUNCTION  

    FUNCTION find WITH PARAMETER u  
        IF the element at position u of parent is NOT EQUAL TO u  
            SET the element at position u of parent TO the result of calling find with the element at position u of parent as argument  
        END IF  
        RETURN the element at position u of parent  
    END FUNCTION  

    FUNCTION union WITH PARAMETERS u AND v  
        SET rootU TO the result of calling find with u  
        SET rootV TO the result of calling find with v  
        IF rootU is NOT EQUAL TO rootV  
            IF the element at position rootU of rank is GREATER THAN the element at position rootV of rank  
                SET the element at position rootV of parent TO rootU  
            ELSE IF the element at position rootU of rank is LESS THAN the element at position rootV of rank  
                SET the element at position rootU of parent TO rootV  
            ELSE  
                SET the element at position rootV of parent TO rootU  
                INCREMENT the element at position rootU of rank BY one  
            END IF  
            DECREMENT count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove WITH PARAMETERS n AND edges  
        SET uf_common TO a new UnionFind with n as argument  
        SET uf_alice TO a new UnionFind with n as argument  
        SET uf_bob TO a new UnionFind with n as argument  
        SET used_edges TO zero  

        FOR each element consisting of t u v IN edges  
            IF t EQUALS three  
                IF calling union on uf_common with u and v returns True  
                    CALL union on uf_alice with u and v  
                    CALL union on uf_bob with u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  

        FOR each element consisting of t u v IN edges  
            IF t EQUALS one AND calling union on uf_alice with u and v returns True  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND calling union on uf_bob with u and v returns True  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  

        IF uf_alice count EQUALS one AND uf_bob count EQUALS one  
            RETURN the length of edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
