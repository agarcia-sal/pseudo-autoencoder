CLASS Solution  
    FUNCTION multiply WITH PARAMETERS num1 AND num2  
        IF num1 EQUALS zero STRING OR num2 EQUALS zero STRING  
            RETURN zero STRING  
        END IF  
        SET result TO an array of zeroes with the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the reverse of num1  
        SET num2 TO the reverse of num2  
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the ASCII numeric value of element at position i of num1 MINUS ASCII numeric value of zero STRING MULTIPLIED BY the ASCII numeric value of element at position j of num2 MINUS ASCII numeric value of zero STRING  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 of result  
                SET element at position p1 of result TO sum_ MODULO ten  
                INCREMENT element at position p2 of result BY the INTEGER DIVISION of sum_ BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation of elements of result reversed transformed to strings  
        RETURN result_str after removing all leading zero STRING characters  
    END FUNCTION  
END CLASS
