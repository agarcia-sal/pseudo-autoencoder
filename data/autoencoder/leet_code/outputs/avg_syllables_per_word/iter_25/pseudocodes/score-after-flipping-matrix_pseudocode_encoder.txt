CLASS Solution  
    FUNCTION matrixScore WITH PARAMETER grid  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR i FROM zero TO the LENGTH OF row MINUS one  
                    SET element at position i of row TO one MINUS element at position i of row  
                END FOR  
            END IF  
        END FOR  
        FOR col FROM one TO the LENGTH OF element at position zero of grid MINUS one  
            SET count_ones TO zero  
            FOR row FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_ones BY element at position col of element at position row of grid  
            END FOR  
            IF count_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row FROM zero TO the LENGTH OF grid MINUS one  
                    SET element at position col of element at position row of grid TO one MINUS element at position col of element at position row of grid  
                END FOR  
            END IF  
        END FOR  
        SET score TO zero  
        FOR each row IN grid  
            SET binary_number TO an empty string  
            FOR each element IN row  
                APPEND the string representation of element TO binary_number  
            END FOR  
            INCREMENT score BY the integer value of binary_number interpreted as a binary number  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
