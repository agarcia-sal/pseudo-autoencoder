CLASS Solution  
    FUNCTION characterReplacement WITH PARAMETERS s k  
        CREATE count AS a new mapping from each character to zero  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of the element at position right of s BY one  
            SET max_count TO the GREATER OF max_count AND the count of the element at position right of s  
            
            IF the result of right MINUS left PLUS one MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the count of the element at position left of s BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND the result of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
