CLASS Solution  
    FUNCTION minAbbreviation(target, dictionary)  
        FUNCTION is_valid_abbr(s, abbr)  
            SET i TO zero  
            SET j TO zero  
            WHILE i LESS THAN the LENGTH OF s AND j LESS THAN the LENGTH OF abbr  
                IF the character at position j of abbr IS a digit  
                    IF the character at position j of abbr EQUALS zero  
                        RETURN False  
                    END IF  
                    SET shift TO zero  
                    WHILE j LESS THAN the LENGTH OF abbr AND the character at position j of abbr IS a digit  
                        SET shift TO shift MULTIPLIED BY ten PLUS the integer VALUE of the character at position j of abbr  
                        INCREMENT j BY one  
                    END WHILE  
                    INCREMENT i BY shift  
                ELSE  
                    IF the character at position i of s NOT EQUALS the character at position j of abbr  
                        RETURN False  
                    END IF  
                    INCREMENT i BY one  
                    INCREMENT j BY one  
                END IF  
            END WHILE  
            RETURN i EQUALS the LENGTH OF s AND j EQUALS the LENGTH OF abbr  
        END FUNCTION  
        
        FUNCTION generate_abbrs(word, length, abbr, index, skip_count)  
            IF index EQUALS length  
                IF skip_count GREATER THAN zero  
                    SET abbr TO abbr CONCATENATED WITH the string REPRESENTATION OF skip_count  
                END IF  
                APPEND abbr TO the list abbrs  
                RETURN  
            END IF  
            CALL generate_abbrs WITH word AND length AND abbr AND index INCREMENTED BY one AND skip_count INCREMENTED BY one  
            IF skip_count GREATER THAN zero  
                SET abbr TO abbr CONCATENATED WITH the string REPRESENTATION OF skip_count  
            END IF  
            CALL generate_abbrs WITH word AND length AND abbr CONCATENATED WITH the character at position index of word AND index INCREMENTED BY one AND zero AS skip_count  
        END FUNCTION  
        
        FUNCTION find_unique_abbr()  
            FOR length FROM one TO the LENGTH OF target  
                CALL generate_abbrs WITH target AND the LENGTH OF target  
                SET min_abbr TO None  
                SET min_length TO positive infinity  
                FOR each abbr IN the list abbrs  
                    IF for all word IN filtered_dict it holds that is_valid_abbr WITH word AND abbr RETURNS False  
                        IF the LENGTH OF abbr LESS THAN min_length  
                            SET min_length TO the LENGTH OF abbr  
                            SET min_abbr TO abbr  
                        END IF  
                    END IF  
                END FOR  
                IF min_abbr IS NOT None  
                    RETURN min_abbr  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET filtered_dict TO the list of words IN dictionary WHERE the LENGTH OF word EQUALS the LENGTH OF target  
        SET abbrs TO an empty list  
        RETURN the RESULT OF find_unique_abbr()  
    END FUNCTION  
END CLASS
