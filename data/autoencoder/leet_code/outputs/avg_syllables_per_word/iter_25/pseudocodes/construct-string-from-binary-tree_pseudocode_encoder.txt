CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val defaulted to zero left defaulted to none right defaulted to none  
        SET the attribute val of the current object TO val  
        SET the attribute left of the current object TO left  
        SET the attribute right of the current object TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values which is a list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode object constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left side of queue  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET the attribute left of node TO a new TreeNode object constructed with the element at position i of values  
            APPEND the attribute left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET the attribute right of node TO a new TreeNode object constructed with the element at position i of values  
            APPEND the attribute right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p and q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the attribute val of p IS NOT EQUAL TO the attribute val of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the attribute left of p and the attribute left of q AND the result of calling is_same_tree with the attribute right of p and the attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str WITH PARAMETERS self and root which is optionally a TreeNode  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN an empty string  
            END IF  
            SET result TO the string representation of the attribute val of node  
            IF the attribute left of node IS NOT none OR the attribute right of node IS NOT none  
                SET result TO result concatenated with an opening parenthesis concatenated with the result of calling helper with the attribute left of node concatenated with a closing parenthesis  
            END IF  
            IF the attribute right of node IS NOT none  
                SET result TO result concatenated with an opening parenthesis concatenated with the result of calling helper with the attribute right of node concatenated with a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN the result of calling helper with root  
    END FUNCTION  
END CLASS
