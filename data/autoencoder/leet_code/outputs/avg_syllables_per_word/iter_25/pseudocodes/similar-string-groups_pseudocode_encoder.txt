CLASS Solution  
    FUNCTION numSimilarGroups WITH PARAMETER strs  
        FUNCTION are_similar WITH PARAMETERS s1 s2  
            SET diff TO an empty list  
            FOR index FROM zero TO the LENGTH OF s1 MINUS one  
                IF the character at position index of s1 NOT EQUALS the character at position index of s2  
                    APPEND index TO diff  
                END IF  
            END FOR  
            IF the LENGTH OF diff EQUALS zero  
                RETURN true  
            ELSE IF the LENGTH OF diff EQUALS two AND the character at position element at position zero of diff of s1 EQUALS the character at position element at position one of diff of s2 AND the character at position element at position one of diff of s1 EQUALS the character at position element at position zero of diff of s2  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FUNCTION dfs WITH PARAMETERS node visited  
            ADD node TO visited  
            FOR each neighbor IN the element at position node of graph  
                IF neighbor NOT IN visited  
                    CALL dfs WITH neighbor visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET n TO the LENGTH OF strs  
        SET graph TO a mapping from each integer i FROM zero TO n MINUS one TO an empty list  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                IF calling are_similar WITH the element at position i of strs AND the element at position j of strs RETURNS true  
                    APPEND j TO the element at position i of graph  
                    APPEND i TO the element at position j of graph  
                END IF  
            END FOR  
        END FOR  
        
        SET visited TO an empty set  
        SET num_groups TO zero  
        
        FOR i FROM zero TO n MINUS one  
            IF i NOT IN visited  
                CALL dfs WITH i visited  
                INCREMENT num_groups BY one  
            END IF  
        END FOR  
        
        RETURN num_groups  
    END FUNCTION  
END CLASS
