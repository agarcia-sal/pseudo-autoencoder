CLASS Solution  
    FUNCTION appealSum WITH PARAMETER s  
        SET last TO a list of length twenty six where each element is negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each index and character IN enumeration of s  
            SET index_value TO the Unicode code point of character MINUS the Unicode code point of lowercase letter a  
            INCREMENT current_appeal BY index MINUS element at position index_value of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index_value of last TO index  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
