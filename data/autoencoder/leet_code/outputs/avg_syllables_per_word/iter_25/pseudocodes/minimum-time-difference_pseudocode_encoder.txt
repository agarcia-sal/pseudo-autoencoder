CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO element at position zero of splitting time at colon separator  
            SET minutes TO element at position one of splitting time at colon separator  
            CONVERT hours TO integer  
            CONVERT minutes TO integer  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        SET minutes_list TO an empty list  
        FOR each time IN timePoints  
            APPEND to_minutes(time) TO minutes_list  
        END FOR  
        SORT minutes_list in ascending order  
        SET min_diff TO infinity  
        FOR i FROM one TO length of minutes_list MINUS one  
            SET current_diff TO element at position i of minutes_list MINUS element at position i MINUS one of minutes_list  
            IF current_diff LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  
        SET circular_diff TO one thousand four hundred forty MINUS element at last position of minutes_list PLUS element at position zero of minutes_list  
        IF circular_diff LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
