CLASS Solution  
    FUNCTION nextGreaterElements WITH PARAMETER nums  
        SET n TO the LENGTH OF nums  
        SET result TO a list containing n times the value negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two times n MINUS one  
            SET index TO i MODULO n  
            WHILE stack is not empty AND the element at the last position of stack in nums is LESS THAN the element at position index in nums  
                SET popped_index TO the result of removing the last element from stack  
                SET the element at position popped_index in result TO the element at position index in nums  
            END WHILE  
            IF i is LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
