CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO the set created from deadends  
        IF the string zero zero zero zero is contained in deadends_set THEN  
            RETURN negative one  
        END IF  
        IF target is equal to the string zero zero zero zero THEN  
            RETURN zero  
        END IF  
        SET queue TO a double ended queue initialized with a tuple consisting of the string zero zero zero zero and zero  
        SET visited TO a set containing the string zero zero zero zero  
        
        FUNCTION neighbors(state)  
            SET result TO an empty list  
            FOR i FROM zero TO three  
                SET digit TO the integer value of the element at position i of state  
                FOR move IN the list consisting of negative one and one  
                    SET new_digit TO the remainder of the sum of digit and move divided by ten  
                    SET new_state TO the concatenation of the substring of state from position zero TO position i MINUS one PLUS the string form of new_digit PLUS the substring of state from position i PLUS one TO the end  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE queue is not empty  
            REMOVE and return the leftmost element from queue ASSIGNING TO current_state and steps  
            FOR neighbor IN the result of calling neighbors with current_state  
                IF neighbor is not contained in visited AND neighbor is not contained in deadends_set THEN  
                    IF neighbor is equal to target THEN  
                        RETURN steps PLUS one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND the tuple consisting of neighbor and steps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
