CLASS Solution  
    FUNCTION findPermutation WITH PARAMETERS s  
        SET result TO an empty list  
        FOR each index FROM zero TO the length of s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR each end FROM zero TO the length of s  
            IF end EQUALS the length of s OR the element at position end of s EQUALS the character I  
                REVERSE the segment of result from position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
