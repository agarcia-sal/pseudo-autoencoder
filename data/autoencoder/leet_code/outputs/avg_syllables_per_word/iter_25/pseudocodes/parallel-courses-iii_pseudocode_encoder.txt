CLASS Solution  
    FUNCTION minimumTime WITH PARAMETERS n relations time  
        ASSIGN reference TO an empty list mapped to every new key in the graph structure  
        FOR each pair prevCourse nextCourse IN relations  
            APPEND prevCourse TO the list at nextCourse in the graph structure  
        END FOR  
        CREATE a list named memo filled with negative one repeated the quantity of n plus one times  
        FUNCTION dfs WITH PARAMETER course  
            IF the element at position course in memo is not equal to negative one THEN  
                RETURN the element at position course in memo  
            END IF  
            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN the list associated with course in the graph structure  
                SET prerequisite_time TO the result of invoking dfs with prevCourse  
                IF max_prerequisite_time is less than prerequisite_time THEN  
                    SET max_prerequisite_time TO prerequisite_time  
                END IF  
            END FOR  
            SET the element at position course in memo TO max_prerequisite_time PLUS the element at position course minus one in time  
            RETURN the element at position course in memo  
        END FUNCTION  
        SET total_minimum_time TO zero  
        FOR each course FROM one TO n inclusive  
            SET current_time TO the result of invoking dfs with course  
            IF total_minimum_time is less than current_time THEN  
                SET total_minimum_time TO current_time  
            END IF  
        END FOR  
        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
