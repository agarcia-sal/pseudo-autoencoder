CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO an empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF num MODULO two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            PUSH the NEGATION OF num INTO max_heap AS a heap element  
        END FOR  
        SET min_deviation TO positive infinity  
        WHILE the element at position zero of max_heap MODULO two EQUALS zero  
            SET max_value TO the NEGATION OF the heap pop of max_heap  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO the integer division of max_value BY two  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            PUSH the NEGATION OF max_value INTO max_heap AS a heap element  
        END WHILE  
        IF min_deviation GREATER THAN the NEGATION OF the element at position zero of max_heap MINUS min_value  
            SET min_deviation TO the NEGATION OF the element at position zero of max_heap MINUS min_value  
        END IF  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
