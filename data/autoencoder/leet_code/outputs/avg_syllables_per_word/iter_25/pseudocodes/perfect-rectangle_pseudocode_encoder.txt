CLASS Solution  
    FUNCTION isRectangleCover WITH PARAMETER rectangles  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO an empty set  
        SET area TO zero  
        
        FOR each rectangle IN rectangles  
            DESTRUCTURE rectangle INTO x1 y1 x2 y2  
            
            IF x1 LESS THAN X1 THEN  
                SET X1 TO x1  
            END IF  
            IF y1 LESS THAN Y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF x2 GREATER THAN X2 THEN  
                SET X2 TO x2  
            END IF  
            IF y2 GREATER THAN Y2 THEN  
                SET Y2 TO y2  
            END IF  
            
            INCREMENT area BY (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)  
            
            FOR each point p IN the list consisting of the tuple of x1 y1 the tuple of x1 y2 the tuple of x2 y1 and the tuple of x2 y2  
                IF p IS IN points THEN  
                    REMOVE p FROM points  
                ELSE  
                    ADD p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area NOT EQUALS (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1) THEN  
            RETURN False  
        END IF  
        
        IF points IS NOT EQUAL TO the set consisting of the tuple of X1 Y1 the tuple of X1 Y2 the tuple of X2 Y1 and the tuple of X2 Y2 THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
