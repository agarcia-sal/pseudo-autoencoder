CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze start destination  
        SET directions TO list containing zero and one then one and zero then zero and MINUS one then MINUS one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETERS x y  
            IF the pair x y is contained in visited THEN  
                RETURN False  
            END IF  
            IF list containing x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD the pair x y TO visited  
            
            FOR each dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze at position ny PLUS dy EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs WITH ARGUMENTS nx ny RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs WITH ARGUMENTS element at position zero of start element at position one of start  
    END FUNCTION  
END CLASS
