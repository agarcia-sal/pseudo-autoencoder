CLASS Solution  
    FUNCTION distinctEchoSubstrings WITH PARAMETER text  
        SET n TO the LENGTH OF text  
        SET substrings TO an EMPTY SET  
        FOR each length FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
            FOR each start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one  
                SET a TO the SUBSTRING of text FROM start TO start PLUS length  
                SET b TO the SUBSTRING of text FROM start PLUS length TO start PLUS two MULTIPLIED BY length  
                IF a EQUALS b  
                    ADD the CONCATENATION of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the SIZE OF substrings  
    END FUNCTION  
END CLASS
