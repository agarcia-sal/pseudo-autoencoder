CLASS Solution  
    FUNCTION find132pattern nums  
        SET second TO negative infinity  
        SET stack TO an empty list  
        FOR each num IN the elements of nums traversed from the last element to the first element  
            IF num LESS THAN second  
                RETURN true  
            END IF  
            WHILE the stack is not empty AND the last element of stack LESS THAN num  
                SET second TO the last element removed from stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
