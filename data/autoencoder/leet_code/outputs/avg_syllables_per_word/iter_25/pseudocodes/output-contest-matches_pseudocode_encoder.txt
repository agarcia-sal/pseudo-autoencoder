CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO the result of converting each number from one to n inclusive to its string representation collected into a list  
        WHILE n GREATER THAN one holds  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO a string composed of an opening parenthesis concatenated with element at position i of teams concatenated with a comma concatenated with element at position n MINUS one MINUS i of teams concatenated with a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two using integer division  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
