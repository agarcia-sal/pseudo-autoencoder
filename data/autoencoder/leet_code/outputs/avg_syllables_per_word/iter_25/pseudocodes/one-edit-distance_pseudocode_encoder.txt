CLASS Solution  
    FUNCTION isOneEditDistance WITH PARAMETERS s AND t  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t THEN  
            RETURN the RESULT OF THE FUNCTION isOneEditDistance CALLED WITH PARAMETERS t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one THEN  
            RETURN False  
        END IF  
        IF diff EQUALS zero THEN  
            RETURN the SUM OF all cases WHERE each character from s NOT EQUALS the corresponding character from t IS EQUAL TO one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t THEN  
                    RETURN whether the SUBSTRING of s from position index TO the END EQUALS the SUBSTRING of t from position index PLUS one TO the END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
