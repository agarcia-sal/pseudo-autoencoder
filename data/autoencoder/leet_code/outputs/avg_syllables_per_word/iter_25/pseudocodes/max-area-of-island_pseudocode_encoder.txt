CLASS Solution  
    FUNCTION maxAreaOfIsland WITH PARAMETER grid  
        IF grid IS EMPTY  
            RETURN zero  
        END IF  
        SET max_area TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        FUNCTION dfs WITH PARAMETERS r c  
            IF r IS LESS THAN zero OR r IS GREATER THAN OR EQUAL TO rows OR c IS LESS THAN zero OR c IS GREATER THAN OR EQUAL TO cols OR element at position r of grid at position c EQUALS zero  
                RETURN zero  
            END IF  
            SET element at position r of grid at position c TO zero  
            SET area TO one  
            INCREMENT area BY CALL dfs WITH ARGUMENTS r PLUS one AND c  
            INCREMENT area BY CALL dfs WITH ARGUMENTS r MINUS one AND c  
            INCREMENT area BY CALL dfs WITH ARGUMENTS r AND c PLUS one  
            INCREMENT area BY CALL dfs WITH ARGUMENTS r AND c MINUS one  
            RETURN area  
        END FUNCTION  
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of grid at position c EQUALS one  
                    SET max_area TO the GREATER OF max_area AND CALL dfs WITH ARGUMENTS r AND c  
                END IF  
            END FOR  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
