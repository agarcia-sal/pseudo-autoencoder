CLASS Solution  
    FUNCTION checkPartitioning WITH PARAMETER s  
        FUNCTION is_palindrome WITH PARAMETER sub  
            RETURN sub EQUALS the substring of sub FROM the last position TO the first position stepping backwards  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two dimensional list of FALSE values with dimensions n BY n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the character at position i of s EQUALS the character at position j of s AND the difference of j MINUS i IS LESS THAN OR EQUAL TO one OR the value at position i PLUS one then j MINUS one of is_pal IS TRUE  
                    SET the value at position i then j of is_pal TO TRUE  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF the value at position zero then i MINUS one of is_pal IS TRUE  
                FOR j FROM i TO n MINUS two  
                    IF the value at position i then j of is_pal IS TRUE AND the value at position j PLUS one then n MINUS one of is_pal IS TRUE  
                        RETURN TRUE  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
