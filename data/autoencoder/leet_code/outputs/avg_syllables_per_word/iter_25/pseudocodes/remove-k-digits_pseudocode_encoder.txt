CLASS Solution  
    FUNCTION removeKdigits WITH PARAMETERS num AND k  
        SET stack TO an empty list  
        FOR each digit IN the characters of num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND the last element in stack GREATER THAN digit  
                REMOVE the last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k GREATER THAN zero  
            SET final_stack TO the elements of stack from the first element TO the element at position the LENGTH OF stack MINUS k MINUS one  
        ELSE  
            SET final_stack TO stack  
        END IF  
        SET result TO the concatenation of all elements in final_stack FOLLOWED BY removal of all leading zero characters  
        IF result IS empty  
            RETURN the string zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
