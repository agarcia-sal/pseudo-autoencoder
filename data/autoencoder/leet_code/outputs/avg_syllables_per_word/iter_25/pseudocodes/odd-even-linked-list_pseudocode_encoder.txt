CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH val EQUAL TO element at position zero of values  
    SET p TO head  
    FOR each val IN the collection of elements from position one TO the end of values  
        SET node TO A NEW ListNode WITH val EQUAL TO val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN THE result of the logical AND of the equality of the val attribute of p1 AND the val attribute of p2 WITH the recursive call of is_same_list WITH PARAMETERS the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self AND head  
        IF head IS None OR the next attribute of head IS None OR the next attribute of the next attribute of head IS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next attribute of head  
        SET even_head TO even  
        WHILE even IS NOT None AND the next attribute of even IS NOT None  
            SET the next attribute of odd TO the next attribute of even  
            SET odd TO the next attribute of odd  
            SET the next attribute of even TO the next attribute of odd  
            SET even TO the next attribute of even  
        END WHILE  
        SET the next attribute of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
