CLASS Solution  
    FUNCTION isMatch WITH PARAMETERS s p  
        SET dp TO a two dimensional list of False values with number of rows equal to the length of s plus one and number of columns equal to the length of p plus one  
        SET element at row zero column zero of dp TO True  
        FOR j FROM one TO the length of p  
            IF the element at position j minus one of p EQUALS the character star  
                SET element at row zero column j of dp TO the element at row zero column j minus one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j minus one of p EQUALS the character star  
                    SET element at row i column j of dp TO the logical OR of the element at row i minus one column j of dp and the element at row i column j minus one of dp  
                ELSE IF the element at position j minus one of p EQUALS the character question mark OR the element at position i minus one of s EQUALS the element at position j minus one of p  
                    SET element at row i column j of dp TO the element at row i minus one column j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row equal to the length of s column equal to the length of p of dp  
    END FUNCTION  
END CLASS
