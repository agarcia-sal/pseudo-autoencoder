CLASS Solution  
    FUNCTION kEmptySlots WITH PARAMETERS bulbs AND k  
        SET bloom_days TO an empty list  
        FOR each day ALONG WITH position IN bulbs WITH THE STARTING DAY AS one  
            SET i TO the position at which position can be inserted in bloom_days such that the order is maintained  
            IF i GREATER THAN zero AND position MINUS the element at position i MINUS one IN bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND the element at position i IN bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position AT position i INTO bloom_days  
        END FOR  
        RETURN minus one  
    END FUNCTION  
END CLASS
