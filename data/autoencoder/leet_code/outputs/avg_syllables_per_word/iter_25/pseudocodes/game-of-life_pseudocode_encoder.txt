CLASS Solution  
    FUNCTION gameOfLife with parameter board  
        IF board is empty OR first element of board is empty THEN  
            RETURN  
        END IF  
        SET m TO the length of board  
        SET n TO the length of the first element of board  
        SET directions TO a list containing the pairs minus one and minus one and minus one and zero and minus one and one and zero and minus one and zero and one and one and minus one and one and zero and one and one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx and dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND (element at position ni of board at position nj EQUALS one OR element at position ni of board at position nj EQUALS three) THEN  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                IF element at position i of board at position j EQUALS one THEN  
                    IF live_neighbors IS LESS THAN two OR live_neighbors IS GREATER THAN three THEN  
                        SET element at position i of board at position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three THEN  
                        SET element at position i of board at position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS two THEN  
                    SET element at position i of board at position j TO one  
                ELSE IF element at position i of board at position j EQUALS three THEN  
                    SET element at position i of board at position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
