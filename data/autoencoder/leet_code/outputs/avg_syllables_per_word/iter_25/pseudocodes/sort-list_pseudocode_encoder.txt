CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode with value equal to the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values excluding the first element  
        SET node TO a new ListNode with value equal to val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND is_same_list of the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head IS None OR the next element of head IS None THEN  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next element of head  
            WHILE fast IS NOT None AND the next element of fast IS NOT None  
                SET slow TO the next element of slow  
                SET fast TO the element two positions ahead of fast by following the next element twice  
            END WHILE  
            SET mid TO the next element of slow  
            SET the next element of slow TO None  
            RETURN head and mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO a new ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT None AND l2 IS NOT None  
                IF the value of l1 IS LESS THAN the value of l2 THEN  
                    SET the next element of current TO l1  
                    SET l1 TO the next element of l1  
                ELSE  
                    SET the next element of current TO l2  
                    SET l2 TO the next element of l2  
                END IF  
                SET current TO the next element of current  
            END WHILE  
            SET the next element of current TO l1 if l1 IS NOT None otherwise to l2  
            RETURN the next element of dummy  
        END FUNCTION  

        SET left, right TO the result of split with head as argument  
        SET left TO the result of calling sortList on left through self  
        SET right TO the result of calling sortList on right through self  
        RETURN the result of merge with left and right as arguments  
    END FUNCTION  
END CLASS
