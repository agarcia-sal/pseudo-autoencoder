CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A new instance of TreeNode created WITH value AS element at position zero of values  
    SET i TO one  
    SET queue TO AN empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element REMOVED from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO A new instance of TreeNode created WITH value AS element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO A new instance of TreeNode created WITH value AS element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree WITH p dot left and q dot left AND calling is_same_tree WITH p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS self nums  
        IF nums IS empty  
            RETURN none  
        END IF  
        SET max_val TO the maximum value in nums  
        SET max_index TO the index of max_val in nums  
        SET root TO A new instance of TreeNode created WITH value AS max_val  
        SET root dot left TO the result of calling self dot constructMaximumBinaryTree WITH the sublist of nums from position zero UP TO but NOT INCLUDING max_index  
        SET root dot right TO the result of calling self dot constructMaximumBinaryTree WITH the sublist of nums FROM position max_index PLUS one TO the END  
        RETURN root  
    END FUNCTION  
END CLASS
