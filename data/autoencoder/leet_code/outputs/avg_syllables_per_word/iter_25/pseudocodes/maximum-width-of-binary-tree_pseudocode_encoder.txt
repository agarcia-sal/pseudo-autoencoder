CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS None, right EQUALS None  
        SET the value attribute of this instance TO val  
        SET the left attribute of this instance TO left  
        SET the right attribute of this instance TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the absence of values holds  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    INITIALIZE queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the content of queue is not empty  
        SET node TO the leftmost element removed from queue  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET the left attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET the right attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF the absence of p AND the absence of q holds  
        RETURN True  
    ELSE IF the absence of p OR the absence of q holds  
        RETURN False  
    ELSE IF the value attribute of p is not equal to the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree on the left attribute of p and the left attribute of q AND the result of is_same_tree on the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree WITH PARAMETERS self, root  
        IF the absence of root holds  
            RETURN zero  
        END IF  
        INITIALIZE queue TO a double ended queue containing a single tuple of root and zero  
        SET max_width TO zero  
        WHILE the content of queue is not empty  
            SET level_length TO the length of queue  
            SET first_index TO the second element of the first tuple in queue  
            FOR each counter FROM zero TO level_length MINUS one  
                REMOVE the leftmost tuple from queue ASSIGN to node and index respectively  
                IF counter EQUALS level_length MINUS one  
                    SET max_width TO the maximum of max_width and the result of index MINUS first_index PLUS one  
                END IF  
                IF the left attribute of node exists  
                    APPEND the tuple of the left attribute of node and the product of two and index TO queue  
                END IF  
                IF the right attribute of node exists  
                    APPEND the tuple of the right attribute of node and the sum of the product of two and index PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
