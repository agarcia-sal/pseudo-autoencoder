CLASS Solution  
    FUNCTION rotatedDigits WITH PARAMETER n  
        SET change_digits TO the collection containing the characters two five six nine  
        SET same_digits TO the collection containing the characters zero one eight  
        
        FUNCTION is_good_number WITH PARAMETER num  
            SET num_str TO the string representation of num  
            IF for every character digit in num_str the digit is contained in the combination of change_digits and same_digits  
                IF there exists at least one character digit in num_str that is contained in change_digits  
                    RETURN true  
                END IF  
                RETURN false  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i from one to n inclusive  
            IF is_good_number with argument i returns true  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
