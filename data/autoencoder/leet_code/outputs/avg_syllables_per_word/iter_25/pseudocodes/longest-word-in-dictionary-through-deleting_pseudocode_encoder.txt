CLASS Solution  
    FUNCTION findLongestWord WITH PARAMETERS s AND dictionary  
        FUNCTION is_subsequence WITH PARAMETERS word AND s  
            ASSIGN reference TO an iterator over s  
            RETURN true if for each character in word the character occurs in order within the iterator otherwise false  
        END FUNCTION  
        SORT dictionary by descending length followed by ascending lexicographical order  
        FOR each word IN dictionary  
            IF is_subsequence of word in s holds true  
                RETURN word  
            END IF  
        END FOR  
        RETURN empty string  
    END FUNCTION  
END CLASS
