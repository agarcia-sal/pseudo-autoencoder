CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND the element at position i of candidates EQUALS the element at position i MINUS one of candidates  
                    CONTINUE  
                END IF  
                CALL backtrack WITH the arguments i PLUS one COMMA target MINUS the element at position i of candidates COMMA path PLUS the list containing the element at position i of candidates  
            END FOR  
        END FUNCTION  
        SORT candidates IN NON-DECREASING ORDER  
        SET result TO an empty list  
        CALL backtrack WITH zero COMMA target COMMA an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
