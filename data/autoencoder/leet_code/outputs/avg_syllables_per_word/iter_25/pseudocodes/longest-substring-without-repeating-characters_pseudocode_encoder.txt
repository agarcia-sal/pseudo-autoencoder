CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH parameter s  
        SET char_index_map TO an empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        FOR each end and char obtained by enumerating over s  
            IF char is a key in char_index_map AND the value at char in char_index_map is greater than or equal to start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            SET max_length TO the maximum between max_length and end MINUS start PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
