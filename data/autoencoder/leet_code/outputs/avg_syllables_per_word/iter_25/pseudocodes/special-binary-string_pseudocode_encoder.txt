CLASS Solution  
    FUNCTION makeLargestSpecial WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER s  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN elements of s by position  
                IF character char EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the string formed by the character one PLUS the result of calling dfs with the substring of s from position start plus one TO position i MINUS one PLUS the character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            SORT specials in reverse lexicographical order  
            RETURN the string formed by concatenating all elements of specials  
        END FUNCTION  
        RETURN the result of calling dfs with argument s  
    END FUNCTION  
END CLASS
