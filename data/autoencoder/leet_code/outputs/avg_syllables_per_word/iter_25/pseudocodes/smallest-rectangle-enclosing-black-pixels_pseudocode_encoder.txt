CLASS Solution  
    FUNCTION minArea WITH PARAMETERS image x y  
        IF image IS empty OR element at position zero of image IS empty THEN  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  
        FUNCTION hasBlackPixelInRow WITH PARAMETER row  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS the character one THEN  
                    RETURN TRUE  
                END IF  
            END FOR  
            RETURN FALSE  
        END FUNCTION  
        FUNCTION hasBlackPixelInColumn WITH PARAMETER col  
            FOR row FROM zero TO m MINUS one  
                IF element at position row of image AND element at position col IS the character one THEN  
                    RETURN TRUE  
                END IF  
            END FOR  
            RETURN FALSE  
        END FUNCTION  
        FUNCTION findTop  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM OF low AND high BY two  
                IF hasBlackPixelInRow WITH mid RETURNS TRUE THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findBottom  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM OF low AND high PLUS one BY two  
                IF hasBlackPixelInRow WITH mid RETURNS TRUE THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findLeft  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM OF low AND high BY two  
                IF hasBlackPixelInColumn WITH mid RETURNS TRUE THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        FUNCTION findRight  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM OF low AND high PLUS one BY two  
                IF hasBlackPixelInColumn WITH mid RETURNS TRUE THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        SET top TO findTop  
        SET bottom TO findBottom  
        SET left TO findLeft  
        SET right TO findRight  
        RETURN the PRODUCT OF the DIFFERENCE OF bottom MINUS top PLUS one AND the DIFFERENCE OF right MINUS left PLUS one  
    END FUNCTION  
END CLASS
