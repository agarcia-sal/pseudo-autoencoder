CLASS Solution  
    FUNCTION networkDelayTime(times AS List OF List OF integers COMMA n AS integer COMMA k AS integer) RETURNS integer  
        SET graph TO an empty dictionary  
        FOR each integer i FROM one TO n  
            SET the element at key i of graph TO an empty list  
        END FOR  
        FOR each element tuple u comma v comma w IN times  
            APPEND tuple v comma w TO the list at key u of graph  
        END FOR  
        SET dist TO an empty dictionary  
        FOR each integer i FROM one TO n  
            SET the element at key i of dist TO infinity  
        END FOR  
        SET the element at key k of dist TO zero  
        SET min_heap TO a list containing the tuple of zero comma k  
        WHILE min_heap is not empty  
            SET current_dist comma node TO the element popped from min_heap extracting the element with the smallest first value  
            IF current_dist GREATER THAN the element at key node of dist  
                CONTINUE  
            END IF  
            FOR each tuple neighbor comma weight IN the list at key node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN the element at key neighbor of dist  
                    SET the element at key neighbor of dist TO distance  
                    PUSH tuple distance comma neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value amongst all values in dist  
        IF max_time EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
