CLASS Solution  
    FUNCTION largestSumOfAverages(nums AS List OF Integer, k AS Integer) RETURNS Float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeros of length n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO the element at position i of prefix_sum PLUS the element at position i of nums  
        END FOR  
        
        FUNCTION average(i AS Integer, j AS Integer) RETURNS Float  
            RETURN the element at position j PLUS one of prefix_sum MINUS the element at position i of prefix_sum DIVIDED BY j MINUS i PLUS one  
        END FUNCTION  
        
        SET dp TO a two dimensional list of zeros with dimensions n PLUS one BY k PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i and one of dp TO the result of calling average with i and n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET element at position i and partitions of dp TO the maximum OF the element at position i and partitions of dp AND the element at position j and partitions MINUS one of dp PLUS the result of calling average with i and j MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position zero and k of dp  
    END FUNCTION  
END CLASS
