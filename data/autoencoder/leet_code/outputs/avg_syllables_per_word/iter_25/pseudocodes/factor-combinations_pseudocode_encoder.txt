CLASS Solution  
    FUNCTION getFactors WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS start AND target  
            FOR i FROM start TO the INTEGER VALUE OF the SQUARE ROOT OF target INCLUSIVE  
                IF the REMAINDER OF target DIVIDED BY i EQUALS zero  
                    APPEND the concatenation of path AND the list containing i AND the INTEGER DIVISION OF target BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH PARAMETERS i AND the INTEGER DIVISION OF target BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH PARAMETERS two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
