CLASS Solution  
    FUNCTION consecutiveNumbersSum WITH PARAMETER n  
        SET count TO zero  
        FOR k FROM one TO n  
            IF the remainder when two MULTIPLIED BY n is DIVIDED BY k EQUALS zero  
                SET m TO the quotient when two MULTIPLIED BY n is INTEGER DIVIDED BY k MINUS k PLUS one DIVIDED BY two  
                IF m IS GREATER THAN zero AND m EQUALS m CONVERTED TO INTEGER  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
