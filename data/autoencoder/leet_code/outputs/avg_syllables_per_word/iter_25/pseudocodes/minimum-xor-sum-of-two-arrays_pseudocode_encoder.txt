CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list containing the representation of infinity repeated TWO RAISED TO THE POWER OF n times  
        SET the element at position zero of dp TO zero  
        FOR each mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the COUNT OF character one IN the BINARY REPRESENTATION OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND TWO RAISED TO THE POWER OF j IS NOT EQUAL TO zero THEN  
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND the SUM OF dp at position BITWISE XOR of mask AND TWO RAISED TO THE POWER OF j AND the BITWISE XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position TWO RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
