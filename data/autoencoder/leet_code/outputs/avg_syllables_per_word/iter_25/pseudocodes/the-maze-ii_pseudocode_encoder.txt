CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze LIST OF LIST OF INTEGER start LIST OF INTEGER destination LIST OF INTEGER RETURNS INTEGER  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the list containing the tuple of one and zero PLUS the tuple of negative one and zero PLUS the tuple of zero and one PLUS the tuple of zero and negative one  
        SET start TO the tuple constructed from start  
        SET destination TO the tuple constructed from destination  
        INITIALIZE visited AS empty set  
        INITIALIZE pq AS a list containing the tuple of zero and start  

        WHILE pq IS NOT empty  
            REMOVE the element with the lowest distance from pq AND ASSIGN distance TO dist AND coordinates TO x and y  
            
            IF x EQUALS element at position zero of destination AND y EQUALS element at position one of destination THEN  
                RETURN dist  
            END IF  
            
            IF the tuple of x and y IS IN visited THEN  
                CONTINUE to the next iteration  
            END IF  
            
            ADD the tuple of x and y TO visited  
            
            FOR each dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x plus dx of maze AT index new_y plus dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the tuple of new_x and new_y IS NOT IN visited THEN  
                    ADD the tuple of dist PLUS count AND the tuple of new_x and new_y TO pq USING heap push operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
