CLASS Solution  
    FUNCTION maxHeight WITH PARAMETER cuboids  
        FOR each cuboid IN cuboids  
            SORT the dimensions of cuboid in ascending order  
        END FOR  
        SORT all cuboids lexicographically based on their dimensions  
        SET n TO the LENGTH OF cuboids  
        CREATE a list dp with n elements all SET TO zero  
        FOR i FROM zero TO n MINUS one  
            SET dp i TO the element at position two of cuboids i  
            FOR j FROM zero TO i MINUS one  
                IF the element at position zero of cuboids j IS LESS THAN OR EQUAL TO the element at position zero of cuboids i AND the element at position one of cuboids j IS LESS THAN OR EQUAL TO the element at position one of cuboids i AND the element at position two of cuboids j IS LESS THAN OR EQUAL TO the element at position two of cuboids i  
                    SET dp i TO the greater value BETWEEN dp i AND dp j PLUS the element at position two of cuboids i  
                END IF  
            END FOR  
        END FOR  
        RETURN the maximum value in dp  
    END FUNCTION  
END CLASS
