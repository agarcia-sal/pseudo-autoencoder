CLASS Solution  
    FUNCTION findMaxValueOfEquation(points LIST OF LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET max_value TO negative infinity  
        SET deque_ TO empty double ended queue  
        FOR each pair OF point IN points  
            SET xj TO element at position zero of pair  
            SET yj TO element at position one of pair  
            WHILE deque_ is not empty AND xj MINUS element at position one of element at position zero of deque_ GREATER THAN k  
                REMOVE element from front of deque_  
            END WHILE  
            IF deque_ is not empty  
                SET max_value TO maximum of max_value AND yj PLUS xj PLUS element at position zero of element at position zero of deque_  
            END IF  
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO element at position zero of element at last position of deque_  
                REMOVE element from back of deque_  
            END WHILE  
            APPEND pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
