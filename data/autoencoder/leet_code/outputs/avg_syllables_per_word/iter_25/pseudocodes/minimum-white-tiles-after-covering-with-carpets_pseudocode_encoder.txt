CLASS Solution  
    FUNCTION minimumWhiteTiles WITH PARAMETERS floor numCarpets carpetLen  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list with n PLUS one rows and numCarpets PLUS one columns WHERE each element IS zero  
        FOR i FROM one TO n  
            SET dp element at row i and column zero TO dp element at row i MINUS one and column zero PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE zero  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at row i and column j TO dp element at row i MINUS one and column j PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE zero  
                IF dp element at row i and column j IS GREATER THAN dp element at row the MAXIMUM BETWEEN zero AND i MINUS carpetLen and column j MINUS one THEN  
                    SET dp element at row i and column j TO dp element at row the MAXIMUM BETWEEN zero AND i MINUS carpetLen and column j MINUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN dp element at row n and column numCarpets  
    END FUNCTION  
END CLASS
