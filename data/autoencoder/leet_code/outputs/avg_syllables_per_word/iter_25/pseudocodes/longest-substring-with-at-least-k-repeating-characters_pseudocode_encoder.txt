CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s AND k  
        IF the LENGTH OF s IS LESS THAN k  
            RETURN zero  
        END IF  
        DEFINE count AS the collection of character frequencies in s  
        FOR each character IN count  
            IF the frequency of character IN count IS LESS THAN k  
                RETURN the MAXIMUM value FROM the collection obtained by FOR each substring IN the result of splitting s BY character computing the value OF longestSubstring OF self WITH substring AND k AS PARAMETERS  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
