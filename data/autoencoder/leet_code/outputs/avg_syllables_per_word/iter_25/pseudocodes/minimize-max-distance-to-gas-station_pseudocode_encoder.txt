CLASS Solution  
    FUNCTION minmaxGasDist WITH PARAMETERS stations AND k  
        FUNCTION possible WITH PARAMETER max_dist  
            SET additional_stations TO zero  
            FOR each index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the element at position index of stations MINUS the element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN whether additional_stations IS LESS THAN OR EQUAL TO k  
        END FUNCTION  
        SET left TO zero  
        SET right TO ten RAISED TO THE POWER OF eight  
        WHILE right MINUS left IS GREATER THAN one DIVIDED BY ten RAISED TO THE POWER OF six  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible WITH ARGUMENT mid  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
