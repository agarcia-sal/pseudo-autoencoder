CLASS Solution  
    FUNCTION longestConsecutive WITH PARAMETER nums  
        IF nums EQUALS empty list THEN  
            RETURN zero  
        END IF  
        SET num_set TO the set containing all elements of nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the element that is num MINUS one NOT IN num_set THEN  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the element that is current_num PLUS one IN num_set HOLD  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                IF current_streak GREATER THAN longest_streak THEN  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
