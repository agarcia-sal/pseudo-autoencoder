CLASS Solution  
    FUNCTION getHint(secret  
                     guess)  
        SET bulls TO zero  
        SET cows TO zero  
        INITIALIZE secret_count AS a mapping from characters to integers defaulting to zero  
        INITIALIZE guess_count AS a mapping from characters to integers defaulting to zero  
        FOR each pair of characters s and g taken respectively from secret and guess in order  
            IF character s EQUALS character g THEN  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the count of character s in secret_count BY one  
                INCREMENT the count of character g in guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN the keys of secret_count  
            INCREMENT cows BY the minimum of the count of digit in secret_count AND the count of digit in guess_count  
        END FOR  
        RETURN the concatenation of the string representation of bulls THEN character A THEN the string representation of cows THEN character B  
    END FUNCTION  
END CLASS
