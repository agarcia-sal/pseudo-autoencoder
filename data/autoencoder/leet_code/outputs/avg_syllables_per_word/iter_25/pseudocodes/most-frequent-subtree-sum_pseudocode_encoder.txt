CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val EQUALS the element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node left TO new TreeNode WITH val EQUALS the element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node right TO new TreeNode WITH val EQUALS the element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p val NOT EQUALS q val  
        RETURN False  
    ELSE  
        RETURN the Boolean result of is_same_tree WITH PARAMETERS p left q left AND is_same_tree WITH PARAMETERS p right q right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION findFrequentTreeSum WITH PARAMETERS self root  
        FUNCTION subtree_sum WITH PARAMETERS node  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            SET current_sum TO node val PLUS subtree_sum WITH PARAMETERS node left PLUS subtree_sum WITH PARAMETERS node right  
            INCREMENT count at key current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        INITIALIZE count AS a default dictionary mapping to integers  
        CALL subtree_sum WITH PARAMETERS root  
        SET max_freq TO the maximum value among all values in count or zero if count is empty  
        INITIALIZE result AS an empty list  
        FOR each key value_pair IN count items  
            IF the frequency value of value_pair EQUALS max_freq  
                APPEND the key value of value_pair TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
