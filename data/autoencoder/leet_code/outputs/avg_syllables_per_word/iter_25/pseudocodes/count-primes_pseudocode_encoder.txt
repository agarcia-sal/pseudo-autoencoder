CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of length n consisting of true values  
        SET element at position zero of is_prime TO false  
        SET element at position one of is_prime TO false  
        FOR each start FROM two TO the integer part of the square root of n PLUS one  
            IF element at position start of is_prime IS true  
                FOR each multiple FROM start MULTIPLIED BY start TO n MINUS one IN STEPS OF start  
                    SET element at position multiple of is_prime TO false  
                END FOR  
            END IF  
        END FOR  
        RETURN the total count of true values in is_prime  
    END FUNCTION  
END CLASS
