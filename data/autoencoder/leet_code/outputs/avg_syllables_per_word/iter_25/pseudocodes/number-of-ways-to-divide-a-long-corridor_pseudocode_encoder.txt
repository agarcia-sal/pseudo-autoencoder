CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each index i AND character x IN corridor  
            IF character x EQUALS string S  
                APPEND index i TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR variable i FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET ways TO ways MULTIPLIED BY the difference BETWEEN element at position i of seats AND element at position i MINUS one of seats  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
