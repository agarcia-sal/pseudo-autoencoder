CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            PUSH the element at position zero of the list at position i of nums AND i AND zero AS a triple INTO min_heap  
            IF max_value LESS THAN the element at position zero of the list at position i of nums THEN  
                SET max_value TO the element at position zero of the list at position i of nums  
            END IF  
        END FOR  
        SET smallest_range TO infinity  
        SET result TO a list containing negative one hundred thousand AND one hundred thousand  
        WHILE the LENGTH OF min_heap IS greater than zero  
            REMOVE the triple with the smallest first element FROM min_heap AND GET its components AS min_value list_index element_index  
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF the list at position list_index of nums THEN  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                PUSH the triple of next_value list_index element_index PLUS one INTO min_heap  
                IF max_value LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
