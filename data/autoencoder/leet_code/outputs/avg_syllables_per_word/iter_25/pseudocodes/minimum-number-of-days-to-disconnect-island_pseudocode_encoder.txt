CLASS Solution  
    FUNCTION minDays grid  
        FUNCTION dfs x y visited  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF the element at position zero of grid OR the TUPLE of x AND y IS IN visited OR the element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the TUPLE of x AND y TO visited  
            FOR each dx dy IN the LIST of TUPLES MINUS one zero PLUS one zero zero MINUS one zero PLUS one  
                CALL dfs WITH x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  
  
        FUNCTION is_connected  
            SET visited TO an EMPTY SET  
            SET land_cells TO the LIST of TUPLES i AND j FOR i FROM zero TO the LENGTH OF grid MINUS one FOR j FROM zero TO the LENGTH OF the element at position zero of grid MINUS one WHERE the element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH the FIRST element of land_cells AT position zero AND the FIRST element of land_cells AT position one AND visited  
            RETURN whether the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  
  
        IF is_connected DOES NOT HOLD  
            RETURN zero  
        END IF  
  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF the element at position zero of grid MINUS one  
                IF the element at position i of grid at position j EQUALS one  
                    SET the element at position i of grid at position j TO zero  
                    IF is_connected DOES NOT HOLD  
                        RETURN one  
                    END IF  
                    SET the element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  
  
        RETURN two  
    END FUNCTION  
END CLASS
