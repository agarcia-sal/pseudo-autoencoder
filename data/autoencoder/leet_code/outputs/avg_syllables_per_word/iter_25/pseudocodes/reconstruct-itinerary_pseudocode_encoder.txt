CLASS Solution  
    FUNCTION findItinerary WITH PARAMETERS tickets  
        SET graph TO a new default dictionary where each key maps to an empty list  
        FOR each pair of departure and arrival airports IN the reversed sorted order of tickets  
            APPEND the arrival airport TO the list mapped by the departure airport in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit WITH PARAMETER airport  
            WHILE the list associated with airport in graph is not empty  
                CALL visit WITH the last element popped from the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit WITH the string JFK  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
