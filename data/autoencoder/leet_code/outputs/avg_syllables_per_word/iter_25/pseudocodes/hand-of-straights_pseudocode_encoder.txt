CLASS Solution  
    FUNCTION isNStraightHand(hand groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET count TO the frequency COUNTER of hand  
        SET unique_cards TO the SORTED list of keys of count  
        FOR each card IN unique_cards  
            IF the value at card in count GREATER THAN zero THEN  
                SET needed TO the value at card in count  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the value at the sum of card PLUS i in count LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the value at the sum of card PLUS i in count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
