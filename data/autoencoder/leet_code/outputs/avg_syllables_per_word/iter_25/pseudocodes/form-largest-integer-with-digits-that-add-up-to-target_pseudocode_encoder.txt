CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS cost AND target  
        SET cost_to_digit TO a dictionary mapping each element c in cost to the string of the index of c plus one  
        SET dp TO a list with an empty string followed by a string dash one repeated target times  
        FOR t FROM one TO target  
            FOR each cost c AND digit d in the items of cost_to_digit  
                IF t is greater than or equal to c AND the element at position t minus c of dp is not equal to string dash one  
                    SET candidate TO the concatenation of d with the element at position t minus c of dp  
                    IF the element at position t of dp is equal to string dash one OR the length of candidate is greater than the length of the element at position t of dp OR the length of candidate is equal to the length of the element at position t of dp AND candidate is lexicographically greater than the element at position t of dp  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF the element at position target of dp is not equal to string dash one  
            RETURN the element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
