CLASS Solution  
    FUNCTION pushDominoes WITH PARAMETER dominoes  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zeroes with LENGTH n  
        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of dominoes EQUALS the character R  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS the character L  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            INCREMENT the element at position i of forces BY force  
        END FOR  
        SET force TO zero  
        FOR i FROM n MINUS one DOWNTO zero  
            IF the element at position i of dominoes EQUALS the character L  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS the character R  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            DECREMENT the element at position i of forces BY force  
        END FOR  
        SET result TO an empty list  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND the character R TO result  
            ELSE IF f LESS THAN zero  
                APPEND the character L TO result  
            ELSE  
                APPEND the character period TO result  
            END IF  
        END FOR  
        RETURN the CONCATENATION of all elements in result  
    END FUNCTION  
END CLASS
