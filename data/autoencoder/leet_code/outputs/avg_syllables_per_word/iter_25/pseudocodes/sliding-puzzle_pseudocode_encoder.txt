CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple constructed by converting each row in board into a tuple  
        SET target TO a tuple containing tuples one two three and four five zero respectively  
        SET neighbors TO a mapping from zero TO a set containing one and three one TO a set containing zero two and four two TO a set containing one and five three TO a set containing zero and four four TO a set containing one three and five five TO a set containing two and four  
        SET zero_pos TO zero  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero THEN  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a deque initialized with a tuple containing start zero_pos and zero  
        SET visited TO a set initialized with start  
        WHILE queue is not empty  
            REMOVE and return the leftmost element from queue into state zero_pos and moves  
            IF state EQUALS target THEN  
                RETURN moves  
            END IF  
            FOR each neighbor IN element at position zero_pos of neighbors  
                SET new_state TO a list constructed by converting each tuple row in state to a list  
                SET i TO integer division of zero_pos by three  
                SET j TO zero_pos modulo three  
                SET ni TO integer division of neighbor by three  
                SET nj TO neighbor modulo three  
                SET temporary variable temp TO element at position j of element at position i of new_state  
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state  
                SET element at position nj of element at position ni of new_state TO temp  
                SET new_tuple_state TO a tuple constructed by converting each list row in new_state into a tuple  
                IF new_tuple_state NOT IN visited THEN  
                    ADD new_tuple_state TO visited  
                    APPEND a tuple containing new_tuple_state neighbor and moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
