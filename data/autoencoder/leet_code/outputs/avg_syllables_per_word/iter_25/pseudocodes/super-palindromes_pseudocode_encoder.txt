CLASS Solution  
    FUNCTION superpalindromesInRange(left right)  
        FUNCTION is_palindrome(num)  
            RETURN the string representation of num EQUALS the string representation of num REVERSED  
        END FUNCTION  
        
        FUNCTION generate_palindromes(limit)  
            SET palindromes TO an empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO the string representation of i  
                APPEND the integer value of concatenating s WITH the reverse of s EXCLUDING its last character TO palindromes  
                APPEND the integer value of concatenating s WITH the reverse of s TO palindromes  
            END FOR  
            RETURN a list of elements p in palindromes WHERE p IS LESS THAN OR EQUAL TO limit  
        END FUNCTION  
        
        SET left TO the integer value of left  
        SET right TO the integer value of right  
        SET limit TO the integer value of the square root of right PLUS one  
        SET palindromes TO the result of calling generate_palindromes WITH limit  
        
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND the result of calling is_palindrome WITH square IS true  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
