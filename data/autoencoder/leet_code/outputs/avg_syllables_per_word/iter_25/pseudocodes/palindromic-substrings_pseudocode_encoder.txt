CLASS Solution  
    FUNCTION countSubstrings WITH PARAMETER s  
        FUNCTION expand_around_center WITH PARAMETERS left AND right  
            SET count TO zero  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN the LENGTH OF s AND the CHARACTER AT POSITION left OF s EQUALS the CHARACTER AT POSITION right OF s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET total_palindromes TO zero  
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT total_palindromes BY the RESULT OF calling expand_around_center WITH left AS i AND right AS i  
            INCREMENT total_palindromes BY the RESULT OF calling expand_around_center WITH left AS i AND right AS i PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
