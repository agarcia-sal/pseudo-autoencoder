CLASS Solution  
    FUNCTION wallsAndGates WITH PARAMETER rooms  
        IF rooms IS EMPTY OR element at position zero of rooms IS EMPTY THEN  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero of rooms  
        SET queue TO a new empty double ended queue  
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of rooms element at position c EQUALS zero THEN  
                    APPEND the pair consisting of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO a collection consisting of the pairs zero PLUS one zero zero MINUS one AND MINUS one zero  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET r AND c TO the elements of the pair obtained by removing the first element from queue  
            FOR each pair dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF zero IS LESS THAN OR EQUAL TO nr AND nr IS LESS THAN rows AND zero IS LESS THAN OR EQUAL TO nc AND nc IS LESS THAN cols AND element at position nr of rooms element at position nc EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven THEN  
                    SET element at position nr of rooms element at position nc TO element at position r of rooms element at position c PLUS one  
                    APPEND the pair consisting of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
