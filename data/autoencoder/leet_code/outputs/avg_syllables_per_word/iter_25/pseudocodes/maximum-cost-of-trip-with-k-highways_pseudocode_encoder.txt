CLASS Solution  
    FUNCTION maximumCost WITH n AND highways AND k  
        CREATE graph AS a mapping from cities to empty lists  
        FOR each city1 AND city2 AND toll IN highways  
            APPEND the pair of city2 AND toll TO the list associated WITH city1 IN graph  
            APPEND the pair of city1 AND toll TO the list associated WITH city2 IN graph  
        END FOR  
        
        SET max_cost TO negative one  
        
        FUNCTION dfs WITH city AND visited AND cost AND highways_used  
            IF highways_used EQUALS k  
                SET max_cost TO the greater value BETWEEN max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each neighbor AND toll IN the list associated WITH city IN graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor AND visited AND cost PLUS toll AND highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        FOR start_city FROM zero TO n MINUS one  
            CREATE visited AS a set containing start_city  
            CALL dfs WITH start_city AND visited AND zero AND zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
