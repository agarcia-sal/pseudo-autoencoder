CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AND t  
        SET pos TO a new default dictionary where each key has an empty list as default value  
        FOR each index i AND character c IN s  
            PUSH onto the heap at key corresponding to the integer value of character c the index i  
        END FOR  
        FOR each index i AND character c IN t  
            SET digit TO the integer value of character c  
            IF the heap at key digit is empty  
                RETURN false  
            END IF  
            FOR each smaller FROM zero TO digit MINUS one  
                IF the heap at key smaller is not empty AND the smallest element at the heap at key smaller IS LESS THAN the smallest element at the heap at key digit  
                    RETURN false  
                END IF  
            END FOR  
            POP the smallest element from the heap at key digit  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
