CLASS Solution  
    FUNCTION kthGrammar WITH PARAMETERS n AND k  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF n MINUS two  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN the result of calling kthGrammar WITH n MINUS one AND k ON this instance  
        ELSE  
            RETURN one MINUS the result of calling kthGrammar WITH n MINUS one AND k MINUS length_of_previous_row ON this instance  
        END IF  
    END FUNCTION  
END CLASS
