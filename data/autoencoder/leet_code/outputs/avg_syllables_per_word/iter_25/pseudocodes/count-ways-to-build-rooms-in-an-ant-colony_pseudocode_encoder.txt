CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        IMPORT default dictionary from collections  
        SET n TO the LENGTH OF prevRoom  
        ASSIGN reference TO graph AS a new default dictionary of list  
        FOR each room FROM one TO n MINUS one  
            APPEND room TO the list at the element at position room of prevRoom in graph  
        END FOR  
        
        FUNCTION dp(node)  
            IF the list at node in graph is empty  
                RETURN one and one as the tuple of number of ways and size of the subtree  
            END IF  
            
            SET total_ways TO one  
            SET total_size TO one  
            SET child_results TO an empty list  
            
            FOR each child IN the list at node in graph  
                SET ways and size TO the result of dp called with child  
                APPEND the tuple of ways and size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            
            SET current_size TO one  
            FOR each tuple of ways and size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the combination of current_size PLUS size MINUS one choose size MULTIPLIED BY ways MODULO MOD  
                INCREMENT current_size BY size  
            END FOR  
            
            RETURN total_ways and total_size as the tuple  
        END FUNCTION  
        
        RETURN the first element of the tuple result of dp called with zero  
    END FUNCTION  
END CLASS
