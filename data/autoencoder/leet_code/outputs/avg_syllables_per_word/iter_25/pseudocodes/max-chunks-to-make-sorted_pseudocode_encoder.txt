CLASS Solution  
    FUNCTION maxChunksToSorted WITH PARAMETER arr  
        SET max_so_far TO zero  
        SET chunks TO zero  
        FOR i FROM zero TO LENGTH OF arr MINUS one  
            SET max_so_far TO THE GREATER OF max_so_far AND element at position i of arr  
            IF max_so_far EQUALS i THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
