CLASS Solution  
    FUNCTION longestValidParentheses WITH PARAMETER s  
        SET stack TO a new list containing negative one  
        SET max_length TO zero  
        FOR each index i and character char in s  
            IF char EQUALS open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO the GREATER OF max_length AND i MINUS the last element of stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
