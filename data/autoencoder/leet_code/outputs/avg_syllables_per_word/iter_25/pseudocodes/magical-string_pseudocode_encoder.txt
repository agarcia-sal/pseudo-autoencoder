CLASS Solution  
    FUNCTION magicalString WITH parameter n  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  
        SET s TO a list containing one two two  
        SET count TO one  
        SET index TO two  
        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the element at position the last index of s in s  
            EXTEND s BY a list containing next_num repeated the element at position index of s times  
            IF next_num EQUALS one  
                INCREMENT count BY the element at position index of s  
            END IF  
            INCREMENT index BY one  
        END WHILE  
        IF the LENGTH OF s GREATER THAN n  
            DECREMENT count BY the element at position the last index of s MULTIPLIED BY the LENGTH OF s MINUS n  
        END IF  
        RETURN count  
    END FUNCTION  
END CLASS
