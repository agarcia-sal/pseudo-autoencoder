CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero next SET TO none  
        SET the val attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the condition that values is empty holds  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF the condition that p1 is none AND p2 is none holds  
        RETURN True  
    END IF  
    IF the condition that p1 is none OR p2 is none holds  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS self head  
        IF the condition that head is none OR the next attribute of head is none holds  
            RETURN head  
        END IF  
        SET dummy TO a new instance of ListNode initialized with the numeric value negative five thousand and one  
        SET the next attribute of dummy TO head  
        SET current TO the next attribute of head  
        SET the next attribute of head TO none  
        WHILE the condition that current is not none holds  
            SET next_node TO the next attribute of current  
            SET prev TO dummy  
            WHILE the condition that the next attribute of prev is not none AND the val attribute of the next attribute of prev is LESS THAN the val attribute of current holds  
                SET prev TO the next attribute of prev  
            END WHILE  
            SET the next attribute of current TO the next attribute of prev  
            SET the next attribute of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
