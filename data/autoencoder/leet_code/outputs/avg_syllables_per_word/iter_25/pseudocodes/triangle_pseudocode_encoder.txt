CLASS Solution  
    FUNCTION minimumTotal WITH PARAMETER triangle  
        FOR each row FROM the LENGTH OF triangle MINUS two DOWN TO zero DECREMENTING BY one  
            FOR each col FROM zero TO the LENGTH OF the element at position row of triangle MINUS one  
                SET the element at position col of the element at position row of triangle TO the value at position col of the element at position row of triangle PLUS the minimum BETWEEN the element at position col of the element at position row PLUS one of triangle AND the element at position col PLUS one of the element at position row PLUS one of triangle  
            END FOR  
        END FOR  
        RETURN the element at position zero of the element at position zero of triangle  
    END FUNCTION  
END CLASS
