CLASS Solution  
    FUNCTION frogPosition WITH PARAMETERS n edges t target  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        SET graph TO a new mapping from each key to an empty list  
        FOR each pair u and v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
        SET queue TO a new double-ended queue containing a tuple of one current node one point zero current time zero  
        SET visited TO a new set containing one  
        WHILE queue is not empty  
            REMOVE and RETURN the leftmost element of queue and unpack into current_node current_probability current_time  
            IF current_time EQUALS t OR (current_node NOT EQUALS one AND the length of the list at key current_node in graph EQUALS one)  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration of the while loop  
                END IF  
            END IF  
            SET possible_moves TO the length of the list at key current_node in graph MINUS (one if current_node NOT EQUALS one otherwise zero)  
            FOR each neighbor IN the list at key current_node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN zero point zero  
    END FUNCTION  
END CLASS
