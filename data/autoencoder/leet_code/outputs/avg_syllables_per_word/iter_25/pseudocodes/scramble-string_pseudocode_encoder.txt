CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 AND s2  
        IF the string s1 is equal to the string s2 THEN  
            RETURN true  
        END IF  
        IF the sorted characters of the string s1 are not equal to the sorted characters of the string s2 THEN  
            RETURN false  
        END IF  
        SET n TO the length of the string s1  
        FOR each integer i FROM one TO n MINUS one  
            IF calling the method isScramble on the object self WITH PARAMETERS the substring of s1 from position zero TO position i MINUS one AND the substring of s2 from position zero TO position i MINUS one AND calling the method isScramble on the object self WITH PARAMETERS the substring of s1 from position i TO the end AND the substring of s2 from position i TO the end THEN  
                RETURN true  
            END IF  
            IF calling the method isScramble on the object self WITH PARAMETERS the substring of s1 from position zero TO position i MINUS one AND the substring of s2 from position n MINUS i TO the end AND calling the method isScramble on the object self WITH PARAMETERS the substring of s1 from position i TO the end AND the substring of s2 from position zero TO position n MINUS i MINUS one THEN  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
