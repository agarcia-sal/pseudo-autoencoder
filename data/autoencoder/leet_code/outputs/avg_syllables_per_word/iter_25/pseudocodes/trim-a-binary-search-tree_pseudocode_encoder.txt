CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val DEFAULT zero left DEFAULT none right DEFAULT none  
        SET the val attribute of self TO val  
        SET the left attribute of self TO left  
        SET the right attribute of self TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values as a list  
    IF the values is empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty dequeue collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element obtained by removing from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right attribute of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the val attribute of p is not equal to the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION trimBST WITH PARAMETERS self root low high  
        IF root is none  
            RETURN none  
        END IF  
        IF the val attribute of root is less than low  
            RETURN the result of calling the method trimBST of self with the right attribute of root low AND high as arguments  
        END IF  
        IF the val attribute of root is greater than high  
            RETURN the result of calling the method trimBST of self with the left attribute of root low AND high as arguments  
        END IF  
        SET the left attribute of root TO the result of calling the method trimBST of self with the left attribute of root low AND high as arguments  
        SET the right attribute of root TO the result of calling the method trimBST of self with the right attribute of root low AND high as arguments  
        RETURN root  
    END FUNCTION  
END CLASS
