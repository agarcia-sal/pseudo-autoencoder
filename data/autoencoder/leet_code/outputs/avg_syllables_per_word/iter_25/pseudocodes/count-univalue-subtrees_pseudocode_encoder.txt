CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET the value attribute of self TO val  
        SET the left attribute of self TO left  
        SET the right attribute of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the values list IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the length of values AND the element at position i of values IS not none  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values IS not none  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value attribute of p IS NOT EQUALS to the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p left attribute AND q left attribute AND the result of is_same_tree with p right attribute AND q right attribute  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees WITH PARAMETERS self root  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN True AND zero  
            END IF  
            SET is_left_unival AND left_count TO the result of helper called with the left attribute of node  
            SET is_right_unival AND right_count TO the result of helper called with the right attribute of node  
            IF is_left_unival AND is_right_unival  
                IF (node left attribute IS none OR the value attribute of the left attribute of node EQUALS the value attribute of node) AND (node right attribute IS none OR the value attribute of the right attribute of node EQUALS the value attribute of node)  
                    RETURN True AND the sum of left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False AND the sum of left_count PLUS right_count  
        END FUNCTION  
        SET underscore AND count TO the result of helper called with root  
        RETURN count  
    END FUNCTION  
END CLASS
