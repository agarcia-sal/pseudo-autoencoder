CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the recursive call of is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETER head  
        FUNCTION reverseList WITH PARAMETER node  
            SET prev TO none  
            WHILE node IS not none  
                SET next_node TO the next attribute of node  
                SET the next attribute of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  

        SET reversed_head TO the result of calling reverseList with head  
        SET carry TO one  
        SET current TO reversed_head  
        WHILE current IS not none AND carry EQUALS one  
            INCREMENT the val attribute of current BY carry  
            SET carry TO the val attribute of current DIVIDED BY ten  
            SET the val attribute of current TO the val attribute of current MODULO ten  
            IF the next attribute of current IS none AND carry EQUALS one  
                SET the next attribute of current TO a new ListNode with carry  
                SET carry TO zero  
            END IF  
            SET prev TO current  
            SET current TO the next attribute of current  
        END WHILE  
        RETURN the result of calling reverseList with reversed_head  
    END FUNCTION  
END CLASS
