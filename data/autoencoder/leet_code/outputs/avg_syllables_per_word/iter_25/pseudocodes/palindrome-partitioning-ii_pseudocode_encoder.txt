CLASS Solution  
    FUNCTION minCut WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        SET is_palindrome TO a two dimensional collection of boolean values initialized to FALSE with n rows and n columns  
        FOR each i FROM zero TO n MINUS one  
            SET the element at row i and column i of is_palindrome TO TRUE  
        END FOR  
        FOR each length FROM two TO n  
            FOR each start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET the element at row start and column end of is_palindrome TO whether the character at position start of s EQUALS the character at position end of s  
                ELSE  
                    SET the element at row start and column end of is_palindrome TO whether the character at position start of s EQUALS the character at position end of s AND the element at row start PLUS one and column end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        SET min_cuts TO a list of zeroes with length n  
        FOR each i FROM zero TO n MINUS one  
            IF the element at row zero and column i of is_palindrome  
                SET the element at position i of min_cuts TO zero  
            ELSE  
                SET the element at position i of min_cuts TO infinity  
                FOR each j FROM one TO i  
                    IF the element at row j and column i of is_palindrome AND the element at position j MINUS one of min_cuts PLUS one LESS THAN the element at position i of min_cuts  
                        SET the element at position i of min_cuts TO the element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN the element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
