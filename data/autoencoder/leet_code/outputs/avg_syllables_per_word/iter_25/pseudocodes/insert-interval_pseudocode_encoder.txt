CLASS Solution  
    FUNCTION insert(intervals AS a list of lists of integers, newInterval AS a list of integers) RETURNS a list of lists of integers  
        IF intervals IS empty THEN  
            RETURN a list containing only newInterval  
        END IF  
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the length of intervals  
        WHILE i IS less than n AND the end element of the interval at position i of intervals IS less than the start element of newInterval  
            APPEND the interval at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        WHILE i IS less than n AND the start element of the interval at position i of intervals IS less than or equal to the end element of newInterval  
            SET the start element of newInterval TO the minimum of the start element of newInterval AND the start element of the interval at position i of intervals  
            SET the end element of newInterval TO the maximum of the end element of newInterval AND the end element of the interval at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        APPEND newInterval TO merged  
        WHILE i IS less than n  
            APPEND the interval at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        RETURN merged  
    END FUNCTION  
END CLASS
