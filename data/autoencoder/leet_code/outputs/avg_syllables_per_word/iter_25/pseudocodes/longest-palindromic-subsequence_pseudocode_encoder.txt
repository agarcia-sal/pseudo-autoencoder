CLASS Solution  
    FUNCTION longestPalindromeSubseq WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET dp TO a two-dimensional array of zeros with dimensions n by n  
        FOR i FROM zero TO n MINUS one  
            SET the element at row i and column i of dp TO one  
        END FOR  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF the character at position i of s EQUALS the character at position j of s  
                    SET the element at row i and column j of dp TO the element at row i PLUS one and column j MINUS one of dp PLUS two  
                ELSE  
                    SET the element at row i and column j of dp TO the maximum of the element at row i PLUS one and column j of dp AND the element at row i and column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row zero and column n MINUS one of dp  
    END FUNCTION  
END CLASS
