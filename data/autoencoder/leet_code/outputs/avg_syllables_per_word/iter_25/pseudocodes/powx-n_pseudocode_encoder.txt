CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x AND n  
        FUNCTION qpow WITH PARAMETERS a AND n  
            SET ans TO one  
            WHILE n IS NOT equal to zero  
                IF the result of logical AND between n and one IS equal to one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two discarding remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN the result of qpow WITH ARGUMENTS x AND n  
        ELSE  
            RETURN one DIVIDED BY the result of qpow WITH ARGUMENTS x AND the NEGATION of n  
        END IF  
    END FUNCTION  
END CLASS
