CLASS Solution  
    FUNCTION frequencySort WITH PARAMETER s  
        SET frequency TO a mapping that counts the occurrences of each character in the string s  
        SET sorted_characters TO the list of all keys in frequency sorted in descending order by the frequency of each character  
        SET result TO an empty string  
        FOR each character IN sorted_characters  
            APPEND the character repeated frequency of the character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
