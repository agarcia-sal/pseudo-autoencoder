CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS None right EQUALS None  
        SET the member variable val TO the parameter val  
        SET the member variable left TO the parameter left  
        SET the member variable right TO the parameter right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the member val of p IS NOT EQUAL TO the member val of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the member left of p AND the member left of q AND the result of calling is_same_tree with the member right of p AND the member right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor WITH PARAMETERS self root p q  
        IF root IS None OR root IS EQUAL TO p OR root IS EQUAL TO q  
            RETURN root  
        END IF  
        SET left TO the result of calling lowestCommonAncestor on self with the member left of root AND p AND q  
        SET right TO the result of calling lowestCommonAncestor on self with the member right of root AND p AND q  
        IF left IS NOT None AND right IS NOT None  
            RETURN root  
        END IF  
        IF left IS NOT None  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
