CLASS Solution  
    FUNCTION minOperations WITH PARAMETERS target AND arr  
        SET index_map TO an empty dictionary  
        FOR each index AND value IN the enumeration of target  
            SET the element value of index_map at key value TO index  
        END FOR  
        SET transformed_arr TO an empty list  
        FOR each value IN arr  
            IF the element value is in the keys of index_map THEN  
                APPEND the element value of index_map at key value TO transformed_arr  
            END IF  
        END FOR  
        SET lis TO an empty list  
        FOR each num IN transformed_arr  
            SET pos TO the result of finding the leftmost position to insert num IN lis to keep sorted order  
            IF pos EQUALS the length of lis THEN  
                APPEND num TO lis  
            ELSE  
                SET the element at position pos of lis TO num  
            END IF  
        END FOR  
        SET lcs_length TO the length of lis  
        RETURN the length of target MINUS lcs_length  
    END FUNCTION  
END CLASS
