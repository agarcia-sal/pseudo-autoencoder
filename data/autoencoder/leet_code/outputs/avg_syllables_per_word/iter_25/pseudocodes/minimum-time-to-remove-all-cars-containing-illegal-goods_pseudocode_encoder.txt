CLASS Solution  
    FUNCTION minimumTime WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zeroes repeated n times  
        SET right_costs TO a list of zeroes repeated n times  

        IF the element at position zero of s EQUALS one  
            SET the element at position zero of left_costs TO one  
        ELSE  
            SET the element at position zero of left_costs TO zero  
        END IF  

        FOR i FROM one TO n MINUS one  
            IF the element at position i of s EQUALS zero  
                SET the element at position i of left_costs TO the element at position i MINUS one of left_costs  
            ELSE  
                IF i PLUS one LESS THAN OR EQUAL TO the element at position i MINUS one of left_costs PLUS two  
                    SET the element at position i of left_costs TO i PLUS one  
                ELSE  
                    SET the element at position i of left_costs TO the element at position i MINUS one of left_costs PLUS two  
                END IF  
            END IF  
        END FOR  

        IF the element at the last position of s EQUALS one  
            SET the element at the last position of right_costs TO one  
        ELSE  
            SET the element at the last position of right_costs TO zero  
        END IF  

        FOR i FROM n MINUS two DOWNTO zero  
            IF the element at position i of s EQUALS zero  
                SET the element at position i of right_costs TO the element at position i PLUS one of right_costs  
            ELSE  
                IF n MINUS i LESS THAN OR EQUAL TO the element at position i PLUS one of right_costs PLUS two  
                    SET the element at position i of right_costs TO n MINUS i  
                ELSE  
                    SET the element at position i of right_costs TO the element at position i PLUS one of right_costs PLUS two  
                END IF  
            END IF  
        END FOR  

        IF the element at the last position of left_costs LESS THAN OR EQUAL TO the element at position zero of right_costs  
            SET min_time TO the element at the last position of left_costs  
        ELSE  
            SET min_time TO the element at position zero of right_costs  
        END IF  

        FOR i FROM zero TO n MINUS two  
            IF min_time LESS THAN OR EQUAL TO the element at position i of left_costs PLUS the element at position i PLUS one of right_costs  
                CONTINUE  
            ELSE  
                SET min_time TO the element at position i of left_costs PLUS the element at position i PLUS one of right_costs  
            END IF  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
