CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of all characters in time with colons removed  
        SET hour TO the integer representation of the substring from position zero TO position one of time  
        SET minute TO the integer representation of the substring from position three TO position four of time  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET combined_hour TO the integer representation of the concatenation of h1 PLUS h2  
                            SET combined_minute TO the integer representation of the concatenation of m1 PLUS m2  
                            IF combined_hour LESS THAN twenty four AND combined_minute LESS THAN sixty THEN  
                                APPEND the tuple consisting of combined_hour AND combined_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the index of the tuple consisting of hour AND minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times THEN  
            SET next_hour AND next_minute TO the element at position current_time_index PLUS one of all_times  
        ELSE  
            SET next_hour AND next_minute TO the element at position zero of all_times  
        END IF  
        
        RETURN the string formed by the two-digit zero-padded representation of next_hour concatenated with a colon concatenated with the two-digit zero-padded representation of next_minute  
    END FUNCTION  
END CLASS
