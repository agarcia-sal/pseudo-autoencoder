CLASS Solution  
    FUNCTION canPartitionKSubsets WITH PARAMETERS nums AND k  
        SET total_sum TO the SUM OF nums  
        IF the REMAINDER OF total_sum DIVIDED BY k IS NOT EQUALS TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN DESCENDING ORDER  
        
        FUNCTION can_partition WITH PARAMETERS index AND k AND current_sum AND used  
            IF k IS EQUALS TO zero  
                RETURN True  
            END IF  
            IF current_sum IS EQUALS TO target  
                RETURN the RESULT OF can_partition CALLED WITH zero AND k MINUS one AND zero AND used  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF element at position i of used IS False AND current_sum PLUS element at position i of nums IS LESS THAN OR EQUAL TO target  
                    SET element at position i of used TO True  
                    IF the RESULT OF can_partition CALLED WITH i PLUS one AND k AND current_sum PLUS element at position i of nums AND used IS True  
                        RETURN True  
                    END IF  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a LIST OF boolean False ELEMENTS WITH LENGTH EQUAL TO the LENGTH OF nums  
        RETURN the RESULT OF can_partition CALLED WITH zero AND k AND zero AND used  
    END FUNCTION  
END CLASS
