CLASS ListNode  
    FUNCTION __init__ WITH val EQUALS zero AND next EQUALS none  
        SET the attribute val OF self TO val  
        SET the attribute next OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode constructed with the element at position zero of values  
    SET p TO head  
    FOR each val IN the slice of values from position one TO the end  
        SET node TO a new ListNode constructed with val  
        SET the attribute next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the attribute val OF p1 EQUALS the attribute val OF p2 AND the result of calling is_same_list WITH the attribute next OF p1 AND the attribute next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight WITH head AND k  
        IF head IS none OR the attribute next OF head IS none OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the attribute next OF tail IS not none  
            SET tail TO the attribute next OF tail  
            INCREMENT length BY one  
        END WHILE  
        SET the attribute next OF tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR a variable from zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the attribute next OF new_tail  
        END FOR  
        SET new_head TO the attribute next OF new_tail  
        SET the attribute next OF new_tail TO none  
        RETURN new_head  
    END FUNCTION  
END CLASS
