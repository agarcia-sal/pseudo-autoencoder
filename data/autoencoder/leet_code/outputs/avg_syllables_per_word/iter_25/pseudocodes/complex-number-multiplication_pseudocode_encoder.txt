CLASS Solution  
    FUNCTION complexNumberMultiply num1 num2  
        FUNCTION parse_complex num  
            SET parts TO the result of splitting num by the character plus  
            SET real TO the integer conversion of element at position zero of parts  
            SET imaginary TO the integer conversion of substring from position zero TO the length of element at position one of parts MINUS one of element at position one of parts  
            RETURN real and imaginary  
        END FUNCTION  
        SET real1 and imag1 TO the result of calling parse_complex with num1  
        SET real2 and imag2 TO the result of calling parse_complex with num2  
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        RETURN the concatenation of the string form of real_part the character plus the string form of imaginary_part and the character i  
    END FUNCTION  
END CLASS
