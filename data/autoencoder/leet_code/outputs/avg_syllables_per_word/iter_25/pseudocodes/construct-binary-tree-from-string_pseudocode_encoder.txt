CLASS ListNode  
    FUNCTION __init__ WITH PARAMETER val WITH DEFAULT zero AND PARAMETER next WITH DEFAULT none  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE FIRST ELEMENT OF values  
    SET p TO head  
    FOR each val IN values FROM THE SECOND ELEMENT TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETER val WITH DEFAULT zero AND PARAMETER left WITH DEFAULT none AND PARAMETER right WITH DEFAULT none  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode CREATED WITH THE FIRST ELEMENT OF values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute of node TO A NEW TreeNode CREATED WITH the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute of node TO A NEW TreeNode CREATED WITH the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value attribute of p IS NOT EQUAL TO the value attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree CALLED WITH the left attribute of p AND the left attribute of q AND is_same_tree CALLED WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree WITH PARAMETER s  
        FUNCTION parse_tree WITH PARAMETER index  
            IF index IS GREATER THAN OR EQUAL TO THE LENGTH OF s  
                RETURN none AND index  
            END IF  
            SET start TO index  
            WHILE index IS LESS THAN THE LENGTH OF s AND (the element at position index of s IS A DIGIT OR the element at position index of s EQUALS THE CHARACTER MINUS)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO A NEW TreeNode CREATED WITH THE INTEGER CONVERSION OF THE SUBSTRING FROM POSITION start TO POSITION index OF s  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s EQUALS THE CHARACTER LEFT PARENTHESIS  
                SET the left attribute of node AND index TO THE RESULT OF parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s EQUALS THE CHARACTER LEFT PARENTHESIS  
                SET the right attribute of node AND index TO THE RESULT OF parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s EQUALS THE CHARACTER RIGHT PARENTHESIS  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND _ TO THE RESULT OF parse_tree CALLED WITH zero  
        RETURN root  
    END FUNCTION  
END CLASS
