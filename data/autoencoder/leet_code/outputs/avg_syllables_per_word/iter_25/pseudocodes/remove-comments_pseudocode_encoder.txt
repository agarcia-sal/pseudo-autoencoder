CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO FALSE  
        SET result TO empty list  
        SET current_line TO empty list  
        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the substring of line from position i TO position i PLUS one EQUALS the string consisting of asterisk followed by forward slash  
                        SET in_block_comment TO FALSE  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the substring of line from position i TO position i PLUS one EQUALS two forward slashes  
                        BREAK the current WHILE loop  
                    ELSE IF the substring of line from position i TO position i PLUS one EQUALS forward slash followed by asterisk  
                        SET in_block_comment TO TRUE  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the character at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            IF current_line is not empty AND in_block_comment EQUALS FALSE  
                APPEND the concatenation of all elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
