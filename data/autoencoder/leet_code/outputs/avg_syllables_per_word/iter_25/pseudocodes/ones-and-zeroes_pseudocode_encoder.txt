CLASS Solution  
    FUNCTION findMaxForm WITH PARAMETERS strs AND m AND n  
        SET dp TO a two dimensional list of zeros with dimensions from zero TO m PLUS one AND zero TO n PLUS one  
        FOR each s IN strs  
            SET count_0 TO the number of elements equal to zero character in s  
            SET count_1 TO the number of elements equal to one character in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i j TO the maximum of dp element at position i j AND dp element at position i MINUS count_0 j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m n  
    END FUNCTION  
END CLASS
