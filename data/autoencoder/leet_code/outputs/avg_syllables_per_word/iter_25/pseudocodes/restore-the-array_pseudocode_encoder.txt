CLASS Solution  
    FUNCTION numberOfArrays WITH PARAMETERS s AND k  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros with LENGTH n PLUS one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n INCLUSIVE  
            FOR j FROM zero TO i MINUS one INCLUSIVE  
                IF the CHARACTER at position j of s NOT EQUAL TO zero AND the integer VALUE of the SUBSTRING from position j TO position i MINUS one of s IS GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET element at position i of dp TO the SUM of element at position i of dp PLUS element at position j of dp THEN TAKEN MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
