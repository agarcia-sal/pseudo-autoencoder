CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter WITH PARAMETERS n AND edges  
        SET graph TO an empty mapping from each integer from one to n inclusive mapping to an empty list  
        FOR each pair of u AND v IN edges  
            APPEND v TO the list mapped by u in graph  
            APPEND u TO the list mapped by v in graph  
        END FOR  
        
        FUNCTION bfs WITH PARAMETER subtree  
            SET start TO an arbitrary element selected from subtree  
            SET queue TO a new double ended queue initialised with start  
            SET visited TO a set initialised with start  
            SET distance TO zero  
            SET farthest_node TO start  
            
            WHILE queue holds elements  
                SET node TO the element removed from the start of queue  
                FOR each neighbor IN the list mapped by node in graph  
                    IF neighbor is in subtree AND neighbor is not in visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a new double ended queue initialised with farthest_node  
            SET visited TO a set initialised with farthest_node  
            SET max_distance TO zero  
            
            WHILE queue holds elements  
                SET size TO the number of elements in queue  
                FOR an iteration index from one TO size inclusive  
                    SET node TO the element removed from the start of queue  
                    FOR each neighbor IN the list mapped by node in graph  
                        IF neighbor is in subtree AND neighbor is not in visited THEN  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue holds elements THEN  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  
            
            RETURN max_distance  
        END FUNCTION  
        
        SET result TO a list of zeros of length n decremented by one  
        
        FOR size FROM two TO n inclusive  
            FOR each subset IN all combinations of numbers from one TO n inclusive taken size at a time  
                SET visited TO an empty set  
                SET stack TO a list initialised with the first element of subset  
                WHILE stack holds elements  
                    REMOVE the last element from stack as node  
                    IF node is not in visited THEN  
                        ADD node TO visited  
                        FOR each neighbor IN the list mapped by node in graph  
                            IF neighbor is in subset AND neighbor is not in visited THEN  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  
                
                IF visited equals the set constructed from subset THEN  
                    SET diameter TO bfs called with the set constructed from subset  
                    IF diameter is greater than zero THEN  
                        INCREMENT the element at position diameter decremented by one of result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
