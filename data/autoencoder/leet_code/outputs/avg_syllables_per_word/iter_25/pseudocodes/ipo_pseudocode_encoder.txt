CLASS Solution  
    FUNCTION findMaximizedCapital WITH PARAMETERS k w profits capital  
        SET projects TO a list of pairs combining each element of capital with the corresponding element of profits  
        SORT projects by the capital value in each pair in ascending order  
        SET max_heap TO an empty list representing a maximum heap for profits  
        SET i TO zero representing the current index in projects  
        SET n TO the length of projects  
        FOR each count from one TO k  
            WHILE i is less than n AND the capital element of the project at position i in projects is less than or equal to w  
                ADD the negative of the profit element of the project at position i in projects to max_heap to simulate maximum heap behavior  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap is empty  
                BREAK the loop  
            END IF  
            SET w TO w PLUS the negation of the value popped from max_heap which is the highest profit available  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
