CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the VALUE OF two MULTIPLIED BY two MULTIPLIED BY n MINUS one  
            CREATE a LIST good WITH n TIMES False  
            CREATE a LIST bad WITH n Times False  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND OF i AND the VALUE OF one LEFT SHIFTED BY j IS NOT zero  
                    SET the ELEMENT at position j OF good TO True  
                ELSE  
                    SET the ELEMENT at position j OF bad TO True  
                END IF  
            END FOR  
            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF the ELEMENT at position j OF good IS True  
                    FOR k FROM zero TO n MINUS one  
                        IF (the ELEMENT at position j OF statements AT position k EQUALS zero AND the ELEMENT at position k OF good IS True) OR (the ELEMENT at position j OF statements AT position k EQUALS one AND the ELEMENT at position k OF bad IS True)  
                            SET valid TO False  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS False  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS True  
                SET max_good TO the GREATER OF max_good AND the SUM OF all True ELEMENTS IN good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
