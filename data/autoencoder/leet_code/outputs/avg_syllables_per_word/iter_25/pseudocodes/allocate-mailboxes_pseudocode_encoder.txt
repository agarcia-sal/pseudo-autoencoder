CLASS Solution  
    FUNCTION minDistance houses COMMA k  
        SORT houses IN NONDECREASING ORDER  
        SET n TO the LENGTH OF houses  
        SET cost TO a two dimensional list of zeros with dimensions n BY n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at i and j TO cost element at i PLUS one and j MINUS one PLUS houses element at j MINUS houses element at i  
            END FOR  
        END FOR  
        SET dp TO a two dimensional list filled with positive infinity values with dimensions n PLUS one BY k PLUS one  
        SET dp element at zero and zero TO zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp element at i and j TO the minimum of dp element at i and j AND dp element at p and j MINUS one PLUS cost element at p and i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at n and k  
    END FUNCTION  
END CLASS
