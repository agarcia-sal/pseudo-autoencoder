CLASS Solution  
    FUNCTION verifyPreorder WITH parameter preorder  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN false  
            END IF  
            WHILE stack is not empty AND value GREATER THAN last element of stack  
                SET lower_bound TO element popped from stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
