CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input by newline  
        SET stack TO an empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the count of tab characters in line  
            SET name TO the substring of line starting from position depth to the end  
            
            WHILE the length of stack is greater than depth  
                REMOVE the last element from stack  
            END WHILE
            
            IF stack is not empty  
                SET current_length TO the last element of stack PLUS the length of name PLUS one for the slash character  
            ELSE  
                SET current_length TO the length of name  
            END IF
            
            IF name contains the period character  
                SET max_length TO the greater of max_length and current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
