CLASS Solution  
    FUNCTION maxProduct WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET max_len_end TO a list of length n where each element is one  
        SET max_len_start TO a list of length n where each element is one  
        
        FUNCTION expand_around_center WITH PARAMETERS left AND right  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN n AND element at position left of s IS EQUAL TO element at position right of s  
                SET length TO right MINUS left PLUS one  
                SET element at position right of max_len_end TO the MAXIMUM of element at position right of max_len_end AND length  
                SET element at position left of max_len_start TO the MAXIMUM of element at position left of max_len_start AND length  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            CALL expand_around_center WITH i AND i  
        END FOR  
        
        SET prefix_max TO a list of length n where each element is zero  
        SET suffix_max TO a list of length n where each element is zero  
        
        SET element at position zero of prefix_max TO element at position zero of max_len_end  
        FOR i FROM one TO n MINUS one  
            SET element at position i of prefix_max TO the MAXIMUM of element at position i MINUS one of prefix_max AND element at position i of max_len_end  
        END FOR  
        
        SET element at position n MINUS one of suffix_max TO element at position n MINUS one of max_len_start  
        FOR i FROM n MINUS two TO zero STEP minus one  
            SET element at position i of suffix_max TO the MAXIMUM of element at position i PLUS one of suffix_max AND element at position i of max_len_start  
        END FOR  
        
        SET max_product TO zero  
        FOR i FROM zero TO n MINUS two  
            SET max_product TO the MAXIMUM of max_product AND element at position i of prefix_max MULTIPLIED BY element at position i PLUS one of suffix_max  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
