CLASS Solution  
    FUNCTION canFinish(numCourses PARAMETER, prerequisites PARAMETER)  
        CREATE a default dictionary named graph that returns an empty list for missing keys  
        CREATE a list named indegree containing zero for each course from zero to numCourses MINUS one  
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT indegree at position course BY one  
        END FOR  
        CREATE a deque named queue containing each course FROM zero TO numCourses MINUS one FOR WHICH indegree at position course EQUALS zero  
        WHILE queue IS NOT empty  
            SET current_course TO the element removed from the left of queue  
            DECREMENT numCourses BY one  
            FOR each next_course IN graph at key current_course  
                DECREMENT indegree at position next_course BY one  
                IF indegree at position next_course EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the comparison of numCourses EQUALS zero  
    END FUNCTION  
END CLASS
