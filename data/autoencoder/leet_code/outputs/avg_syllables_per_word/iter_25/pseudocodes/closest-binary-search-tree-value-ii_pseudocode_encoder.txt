CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero and left defaulting to None and right defaulting to None  
        SET the val attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values is empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF the val attribute of p is not equal to the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with parameters the left attribute of p and the left attribute of q AND the result of calling is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues with parameters root and target and k  
        FUNCTION inorder_traversal with parameter node  
            IF node is None  
                RETURN  
            END IF  
            CALL inorder_traversal with the left attribute of node  
            IF the length of result is less than k  
                APPEND the val attribute of node TO result  
            ELSE  
                IF the absolute value of the val attribute of node MINUS target is less than the absolute value of the element at position zero of result MINUS target  
                    REMOVE the element at the front of result  
                    APPEND the val attribute of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with the right attribute of node  
        END FUNCTION  
      
        SET result TO an empty double ended queue  
        CALL inorder_traversal with root  
        RETURN a list created from result  
    END FUNCTION  
END CLASS
