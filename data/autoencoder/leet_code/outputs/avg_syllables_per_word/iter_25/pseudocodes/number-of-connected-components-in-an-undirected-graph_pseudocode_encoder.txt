CLASS Solution  
    FUNCTION countComponents WITH PARAMETERS n AND edges  
        SET graph TO a mapping FROM each integer i from zero TO n MINUS one TO an empty list  
        FOR each element a AND each element b IN edges  
            APPEND element b TO the list at key a IN graph  
            APPEND element a TO the list at key b IN graph  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at key node IN graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR each node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
