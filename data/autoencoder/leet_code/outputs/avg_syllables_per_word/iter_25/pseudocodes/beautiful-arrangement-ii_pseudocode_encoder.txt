CLASS Solution  
    FUNCTION constructArray WITH PARAMETERS n k  
        SET result TO a new list containing the sequence of integers starting from one and ending at n MINUS k MINUS one  
        SET left TO n MINUS k  
        SET right TO n  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF left IS EQUAL TO right  
                APPEND left TO result  
            ELSE  
                APPEND left TO result  
                APPEND right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
