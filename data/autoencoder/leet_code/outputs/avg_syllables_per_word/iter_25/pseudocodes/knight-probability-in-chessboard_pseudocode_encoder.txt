CLASS Solution  
    FUNCTION knightProbability WITH PARAMETERS n k row column  
        SET moves TO the list containing the pairs of integers minus two and minus one comma minus two and one comma minus one and minus two comma minus one and two comma one and minus two comma one and two comma two and minus one comma two and one  
        SET dp TO a list containing n times the list of n times zero  
        SET element at position row of dp at position column TO one  
        FOR each number FROM one TO k  
            SET new_dp TO a list containing n times the list of n times zero  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r of dp at position c EQUALS zero  
                        CONTINUE  
                    END IF  
                    FOR each pair dr dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF zero LESS THAN OR EQUAL TO nr AND nr LESS THAN n AND zero LESS THAN OR EQUAL TO nc AND nc LESS THAN n  
                            INCREMENT element at position nr of new_dp at position nc BY element at position r of dp at position c DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each cell_probability IN row_list  
                INCREMENT total_probability BY cell_probability  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
