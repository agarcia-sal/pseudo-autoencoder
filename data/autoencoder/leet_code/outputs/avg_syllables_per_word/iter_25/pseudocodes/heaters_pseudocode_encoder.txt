CLASS Solution  
    FUNCTION findRadius WITH PARAMETERS houses AND heaters  
        PERFORM sorting of houses in ascending order  
        PERFORM sorting of heaters in ascending order  
        SET min_radius TO zero  
        SET j TO zero  
        FOR each house IN houses  
            WHILE j is less than the length of heaters MINUS one AND the absolute value of the element at position j of heaters MINUS house is greater than or equal to the absolute value of the element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            SET min_radius TO the maximum value between min_radius AND the absolute value of the element at position j of heaters MINUS house  
        END FOR  
        RETURN min_radius  
    END FUNCTION  
END CLASS
