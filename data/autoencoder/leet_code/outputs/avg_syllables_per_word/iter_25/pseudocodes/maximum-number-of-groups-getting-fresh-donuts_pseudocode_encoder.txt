CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeros of LENGTH batchSize  
        FOR each group IN groups  
            INCREMENT the element at position group MODULO batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, *remainder_count)  
            SET remainder_count TO a new list created from the input remainder_count  
            SET max_happy TO zero  
            
            FOR i FROM zero TO batchSize MINUS one  
                IF the element at position i of remainder_count EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                
                DECREMENT the element at position i of remainder_count BY one  
                SET new_remainder TO remainder MINUS i MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, all elements of remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, all elements of remainder_count)  
                END IF  
                
                SET max_happy TO the greater of max_happy and happy_groups  
                
                INCREMENT the element at position i of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, all elements of remainder_count)  
    END FUNCTION  
END CLASS
