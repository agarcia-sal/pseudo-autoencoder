CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new double-ended queue containing the tuple consisting of position zero speed one and steps zero  
        ASSIGN reference TO a new set containing the tuple consisting of position zero and speed one  
        
        WHILE the queue is not empty  
            ASSIGN position speed and steps TO the first element removed from the queue  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple consisting of next_position and next_speed is not contained in visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple consisting of next_position and next_speed TO visited  
                APPEND the tuple consisting of next_position next_speed and steps PLUS one TO queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple consisting of position and next_speed is not contained in visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple consisting of position and next_speed TO visited  
                APPEND the tuple consisting of position next_speed and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
