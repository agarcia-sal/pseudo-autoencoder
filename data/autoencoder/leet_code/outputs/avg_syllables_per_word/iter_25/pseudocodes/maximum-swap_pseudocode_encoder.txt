CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO a list of characters representing the string form of num  
        SET last TO an empty dictionary  
        FOR each index i and character d IN digits with enumeration  
            SET the integer value of d in last TO i  
        END FOR  
        FOR each index i and character d IN digits with enumeration  
            FOR each digit candidate d2 FROM nine DOWN TO the integer value of d PLUS one  
                IF the value associated with d2 in last or zero if not present IS GREATER THAN i THEN  
                    SET temporary variable TO digits at position i  
                    SET digits at position i TO digits at position of last at d2  
                    SET digits at position of last at d2 TO temporary variable  
                    SET swapped_number_string TO the concatenation of all elements in digits  
                    SET swapped_number TO the integer value of swapped_number_string  
                    RETURN swapped_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
