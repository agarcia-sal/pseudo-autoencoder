CLASS Solution  
    FUNCTION minInsertions WITH PARAMETER s  
        FUNCTION longest_palindromic_subsequence WITH PARAMETER s  
            SET n TO the LENGTH OF s  
            SET dp TO a TWO DIMENSIONAL ARRAY WITH n ROWS AND n COLUMNS FILLED WITH zero  
            FOR i FROM zero TO n MINUS one  
                SET element at row i COLUMN i OF dp TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two  
                        SET element at row i COLUMN j OF dp TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s  
                        SET element at row i COLUMN j OF dp TO element at row i PLUS one COLUMN j MINUS one OF dp PLUS two  
                    ELSE  
                        SET element at row i COLUMN j OF dp TO the MAXIMUM OF element at row i COLUMN j MINUS one OF dp AND element at row i PLUS one COLUMN j OF dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at row zero COLUMN n MINUS one OF dp  
        END FUNCTION  
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence WITH ARGUMENT s  
    END FUNCTION  
END CLASS
