CLASS Solution  
    FUNCTION findKthNumber WITH PARAMETERS m n k  
        FUNCTION enough WITH PARAMETER x  
            SET count TO zero  
            FOR each i FROM one TO m  
                INCREMENT count BY the minimum of x DIVIDED BY i and n  
            END FOR  
            IF count GREATER THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO the integer division of the sum of low and high by two  
            IF enough called with mid RETURNS true THEN  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
