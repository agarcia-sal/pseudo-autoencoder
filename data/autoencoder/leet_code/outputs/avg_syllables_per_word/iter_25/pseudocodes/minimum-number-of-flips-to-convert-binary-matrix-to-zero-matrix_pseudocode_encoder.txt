CLASS Solution  
    FUNCTION minFlips WITH PARAMETERS mat  
        FUNCTION flip WITH PARAMETERS matrix x y  
            SET directions TO list containing the pairs MINUS one zero PLUS one zero zero MINUS one zero PLUS one zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n THEN  
                    SET element at position nx of matrix AT position ny TO one MINUS element at position nx of matrix AT position ny  
                END IF  
            END FOR  
        END FUNCTION  
        FUNCTION matrix_to_tuple WITH PARAMETER matrix  
            RETURN a tuple CONSTRUCTED BY transforming each row OF matrix INTO a tuple  
        END FUNCTION  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero OF mat  
        SET target TO a tuple CONSTRUCTED BY transforming to tuples zero for each element in the range from zero TO m of zeros repeated n times  
        SET initial_state TO CALL matrix_to_tuple WITH mat  
        SET queue TO a list containing the pair initial_state AND zero  
        SET visited TO an empty set  
        ADD initial_state TO visited  
        WHILE queue is not empty  
            SET current_state AND steps TO the first element of queue  
            REMOVE the first element from queue  
            IF current_state EQUALS target THEN  
                RETURN steps  
            END IF  
            SET current_matrix TO a list CONSTRUCTED BY transforming each row OF current_state INTO a list  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy of current_matrix  
                    CALL flip WITH next_matrix AND i AND j  
                    SET next_state TO CALL matrix_to_tuple WITH next_matrix  
                    IF next_state IS NOT IN visited THEN  
                        ADD next_state TO visited  
                        APPEND the pair next_state AND steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN MINUS one  
    END FUNCTION  
END CLASS
