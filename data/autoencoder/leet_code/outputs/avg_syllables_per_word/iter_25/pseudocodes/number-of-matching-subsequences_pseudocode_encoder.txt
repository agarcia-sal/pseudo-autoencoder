CLASS Solution  
    FUNCTION numMatchingSubseq TAKING parameters s AS string AND words AS list of strings RETURNS integer  
        SET waiting TO an empty dictionary mapping each unique character in s TO an empty double-ended queue  
        
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element from it  
            IF first_letter is contained within the keys of waiting  
                APPEND it TO the queue stored at waiting for first_letter  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN s  
            IF c is contained within the keys of waiting  
                SET current_iterators TO the queue stored at waiting for c  
                SET waiting for c TO a new empty double-ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next element from it OR None if exhausted  
                    IF next_char is None  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is contained within the keys of waiting  
                            APPEND it TO the queue stored at waiting for next_char  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
