CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO an empty set  
        SET min_diff TO infinity  
        FOR each num IN arr  
            SET new_seen TO an empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO the union of new_seen AND the set containing num  
            FOR each val IN seen  
                IF absolute value of val MINUS target LESS THAN min_diff THEN  
                    SET min_diff TO absolute value of val MINUS target  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
