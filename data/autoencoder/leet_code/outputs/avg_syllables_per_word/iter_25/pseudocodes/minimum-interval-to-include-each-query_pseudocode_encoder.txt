CLASS Solution  
    FUNCTION minInterval(intervals AND queries)  
        SORT intervals BY their starting elements in ascending order  
        SET sorted_queries TO a list of pairs of each query and its original index sorted by the query value in ascending order  
        SET min_heap TO an empty list for storing pairs of interval size and interval end  
        SET result TO a list of length equal to the number of queries with all elements set to negative one  
        SET interval_index TO zero  
        FOR each query AND original_index IN sorted_queries  
            WHILE interval_index is less than the length of intervals AND the starting element of the interval at interval_index is less than or equal to query  
                SET start TO the starting element of the interval at interval_index  
                SET end TO the ending element of the interval at interval_index  
                SET interval_size TO end MINUS start PLUS one  
                INSERT the pair of interval_size and end into min_heap maintaining heap property  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap is not empty AND the ending element at the first pair in min_heap is less than query  
                REMOVE the first pair from min_heap maintaining heap property  
            END WHILE  
            IF min_heap is not empty  
                SET the element at position original_index of result TO the interval size element of the first pair in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
