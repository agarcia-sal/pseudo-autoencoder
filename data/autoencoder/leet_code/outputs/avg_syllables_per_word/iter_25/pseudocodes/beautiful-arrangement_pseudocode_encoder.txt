CLASS Solution  
    FUNCTION countArrangement WITH PARAMETERS n  
        FUNCTION backtrack WITH PARAMETERS used AND index  
            IF index EQUALS n PLUS one THEN  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n DO  
                IF element at position i of used EQUALS False AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero) THEN  
                    SET element at position i of used TO True  
                    INCREMENT count BY the RESULT OF calling backtrack WITH used AND index PLUS one  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list consisting of False repeated n PLUS one times  
        RETURN the RESULT OF calling backtrack WITH used AND one  
    END FUNCTION  
END CLASS
