CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the result of num DIVIDED BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN value TO the element at key composed of depth and pos in tree  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value retrieved from tree at key composed of depth and pos  
            IF current_value EQUALS none  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two  
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs with arguments being the first element of left_child the second element of left_child and new_path_sum  
            SET right_sum TO dfs with arguments being the first element of right_child the second element of right_child and new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs with arguments one one zero  
    END FUNCTION  
END CLASS
