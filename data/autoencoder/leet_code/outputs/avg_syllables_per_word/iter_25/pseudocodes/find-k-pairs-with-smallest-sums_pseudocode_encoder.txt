CLASS Solution  
    FUNCTION kSmallestPairs WITH PARAMETERS nums1 nums2 k  
        IF nums1 IS empty OR nums2 IS empty OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR each integer i FROM zero TO the smaller of k and the length of nums1 MINUS one  
            INSERT INTO min_heap a tuple consisting of the sum of element at position i of nums1 AND element at position zero of nums2 AND i AND zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap IS NOT empty AND the length of result IS LESS THAN k  
            REMOVE AND ASSIGN from min_heap the tuple with smallest first element TO unknown variable ignore SUM AND i AND j  
            APPEND a list consisting of element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one IS LESS THAN the length of nums2 THEN  
                INSERT INTO min_heap a tuple consisting of the sum of element at position i of nums1 AND element at position j PLUS one of nums2 AND i AND j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
