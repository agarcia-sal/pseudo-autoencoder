CLASS Solution  
    FUNCTION customSortString WITH PARAMETERS order AND s  
        SET d TO an empty dictionary  
        FOR each index i AND each character c IN the enumeration of order  
            SET the element at key c in dictionary d TO i  
        END FOR  
        SET sorted_characters TO the result of sorting s in ascending order according to the key function that for each character x RETURNS the value from dictionary d at key x OR zero if key x is not present in dictionary d  
        SET result_string TO the concatenation of all characters in sorted_characters  
        RETURN result_string  
    END FUNCTION  
END CLASS
