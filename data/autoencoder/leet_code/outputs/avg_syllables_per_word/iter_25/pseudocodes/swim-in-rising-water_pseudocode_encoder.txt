CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple with the element at position zero zero of grid PLUS zero PLUS zero  
        ADD the tuple with zero zero TO visited  
        SET directions TO a list containing the tuples zero one PLUS one zero PLUS zero MINUS one PLUS minus one zero  
        WHILE min_heap is not empty  
            REMOVE and RETURN the smallest element FROM min_heap and assign its elements TO t i j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each tuple di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple ni nj IS NOT IN visited  
                    ADD the tuple ni nj TO visited  
                    INSERT INTO min_heap a tuple containing the maximum OF t AND the element at position ni nj of grid PLUS ni PLUS nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
