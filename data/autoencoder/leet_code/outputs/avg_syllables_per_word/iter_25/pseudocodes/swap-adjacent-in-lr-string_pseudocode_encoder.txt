CLASS Solution  
    FUNCTION canTransform WITH PARAMETERS start AND end  
        SET start_filtered TO a list containing pairs of character and index for each index and character in start where the character is not letter X  
        SET end_filtered TO a list containing pairs of character and index for each index and character in end where the character is not letter X  
        IF the length of start_filtered is not equal to the length of end_filtered THEN  
            RETURN false  
        END IF  
        FOR each pair consisting of character1 at index1 in start_filtered AND character2 at index2 in end_filtered  
            IF character1 is not equal to character2 THEN  
                RETURN false  
            END IF  
            IF character1 is letter L AND index1 is less than index2 THEN  
                RETURN false  
            END IF  
            IF character1 is letter R AND index1 is greater than index2 THEN  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
