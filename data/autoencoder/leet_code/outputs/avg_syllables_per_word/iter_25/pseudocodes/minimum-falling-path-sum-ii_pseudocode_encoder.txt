CLASS Solution  
    FUNCTION minFallingPathSum WITH PARAMETER grid  
        IF grid IS empty OR element at position zero of grid IS empty  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF grid  
        IF n EQUALS one  
            RETURN the MINIMUM value IN element at position zero of grid  
        END IF  
        SET dp TO a COPY OF element at position zero of grid  
        FOR i FROM one TO n MINUS one  
            SET new_dp TO a list containing n zeroes  
            SET min1 TO positive infinity  
            SET min2 TO positive infinity  
            SET min1_index TO negative one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of dp LESS THAN min1  
                    SET min2 TO min1  
                    SET min1 TO element at position j of dp  
                    SET min1_index TO j  
                ELSE IF element at position j of dp LESS THAN min2  
                    SET min2 TO element at position j of dp  
                END IF  
            END FOR  
            FOR j FROM zero TO n MINUS one  
                IF j EQUALS min1_index  
                    SET element at position j of new_dp TO element at position i of element at position j of grid PLUS min2  
                ELSE  
                    SET element at position j of new_dp TO element at position i of element at position j of grid PLUS min1  
                END IF  
            END FOR  
            SET dp TO a COPY OF new_dp  
        END FOR  
        RETURN the MINIMUM value IN dp  
    END FUNCTION  
END CLASS
