CLASS Solution  
    FUNCTION shortestPathLength graph  
        IF graph EQUALS nothing OR graph HAS no elements  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF graph  
        SET queue TO a double ended queue containing pairs consisting of each node FROM zero TO n MINUS one AND a bitmask with one shifted left by the value of the node indicating that node is visited  
        SET visited TO a set containing the same pairs as in queue representing visited states  
        SET steps TO zero  
        SET all_nodes_visited TO a bitmask with one shifted left by n MINUS one representing that all nodes are visited  
        WHILE the queue is not empty  
            FOR each element in the range of the LENGTH OF queue  
                REMOVE the element at the front of queue AND ASSIGN it TO current_node AND visited_nodes  
                IF visited_nodes EQUALS all_nodes_visited  
                    RETURN steps  
                END IF  
                FOR each neighbor IN the list of neighbors of current_node in graph  
                    SET next_visited_nodes TO visited_nodes OR a bitmask with one shifted left by neighbor  
                    SET next_state TO the pair consisting of neighbor AND next_visited_nodes  
                    IF next_state is NOT IN visited  
                        ADD next_state to visited  
                        APPEND next_state TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT steps BY one  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
