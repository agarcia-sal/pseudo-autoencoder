CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the boolean negation of values holds  
        RETURN none  
    END IF  
    SET root TO A new instance of TreeNode with parameter the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element taken from the left of queue  
        IF i is LESS THAN the length of values AND the element at position i of values is not none  
            SET node dot left TO A new instance of TreeNode with parameter the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the length of values AND the element at position i of values is not none  
            SET node dot right TO A new instance of TreeNode with parameter the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF the boolean negation of p AND the boolean negation of q both hold  
        RETURN true  
    ELSE IF the boolean negation of p OR the boolean negation of q holds  
        RETURN false  
    ELSE IF p dot val is NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with p dot left AND q dot left AND the result of calling is_same_tree with p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom WITH PARAMETERS self root  
        IF the boolean negation of root holds  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a deque containing root  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            SET current_level TO an empty list  
            FOR each iteration FROM zero TO level_size MINUS one  
                SET node TO the element taken from the left of queue  
                APPEND node dot val TO current_level  
                IF node dot left exists  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right exists  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN the reversed order of result  
    END FUNCTION  
END CLASS
