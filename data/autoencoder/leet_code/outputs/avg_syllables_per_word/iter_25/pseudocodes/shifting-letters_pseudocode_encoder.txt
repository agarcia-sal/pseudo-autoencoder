CLASS Solution  
    FUNCTION shiftingLetters WITH PARAMETERS s AND shifts  
        SET total_shifts TO zero  
        FOR i FROM the LENGTH OF shifts MINUS one TO zero STEP negative one  
            INCREMENT total_shifts BY the element at position i of shifts  
            SET the element at position i of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            SET new_character_code TO the remainder of the division of the sum of the difference between the character code of the element at position i of s AND the character code of the lowercase letter a PLUS the element at position i of shifts BY twenty six PLUS the character code of the lowercase letter a  
            CONVERT new_character_code TO the corresponding character and APPEND it TO result  
        END FOR  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
