CLASS Solution  
    FUNCTION longestPath WITH PARAMETERS parent AS list OF integers AND s AS string  
        CREATE tree AS a mapping from defaulting to list  
        FOR each index i AND element p IN parent with enumeration  
            IF p IS NOT EQUAL TO minus one  
                APPEND i TO element at key p IN tree  
            END IF  
        END FOR  
        SET self.result TO one  
        FUNCTION dfs WITH PARAMETER node  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN element at key node IN tree  
                SET child_length TO RESULT OF CALLING dfs WITH child  
                IF character at position child of s IS NOT EQUAL TO character at position node of s  
                    IF child_length IS GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length IS GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET self.result TO GREATER OF self.result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN self.result  
END CLASS
