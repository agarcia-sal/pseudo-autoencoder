CLASS Solution  
    FUNCTION minimumDeleteSum WITH PARAMETERS s1 s2  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        CREATE dp AS a TWO DIMENSIONAL LIST with m PLUS one rows and n PLUS one columns INITIALIZED TO zero  
        FOR j FROM one TO n  
            SET element at row zero column j of dp TO element at row zero column j MINUS one of dp PLUS the ASCII VALUE OF character at position j MINUS one of s2  
        END FOR  
        FOR i FROM one TO m  
            SET element at row i column zero of dp TO element at row i MINUS one column zero of dp PLUS the ASCII VALUE OF character at position i MINUS one of s1  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s1 EQUALS character at position j MINUS one of s2  
                    SET element at row i column j of dp TO element at row i MINUS one column j MINUS one of dp  
                ELSE  
                    SET element at row i column j of dp TO the MINIMUM OF element at row i MINUS one column j of dp PLUS the ASCII VALUE OF character at position i MINUS one of s1 AND element at row i column j MINUS one of dp PLUS the ASCII VALUE OF character at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        RETURN element at row m column n of dp  
    END FUNCTION  
END CLASS
