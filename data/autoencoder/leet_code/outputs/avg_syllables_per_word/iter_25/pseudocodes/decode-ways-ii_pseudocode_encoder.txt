CLASS Solution  
    FUNCTION numDecodings WITH PARAMETER s  
        SET MODULO_VALUE TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS seven  
        SET dp TO a list of zero repeated the LENGTH OF s PLUS one times  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single WITH PARAMETER char  
            IF char EQUALS a star character  
                RETURN nine  
            ELSE IF char EQUALS character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair WITH PARAMETERS char1 AND char2  
            IF char1 EQUALS a star character AND char2 EQUALS a star character  
                RETURN fifteen  
            ELSE IF char1 EQUALS a star character  
                IF char2 LESS THAN OR EQUAL TO character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS a star character  
                IF char1 EQUALS character one  
                    RETURN nine  
                ELSE IF char1 EQUALS character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO integer value of concatenation of char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO LENGTH OF s  
            SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of dp MULTIPLIED BY ways_to_decode_single of element at position i MINUS one of s  
            SET element at position i of dp TO element at position i of dp MODULO MODULO_VALUE  
            
            IF i GREATER THAN one  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS two of dp MULTIPLIED BY ways_to_decode_pair of element at position i MINUS two of s AND element at position i MINUS one of s  
                SET element at position i of dp TO element at position i of dp MODULO MODULO_VALUE  
            END IF  
        END FOR  
        
        RETURN element at last position of dp  
    END FUNCTION  
END CLASS
