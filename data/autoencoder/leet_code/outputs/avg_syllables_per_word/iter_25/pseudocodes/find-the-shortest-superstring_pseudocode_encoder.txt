CLASS Solution  
    FUNCTION shortestSuperstring WITH PARAMETER words  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap WITH PARAMETERS i AND j  
            FOR k FROM the MINIMUM OF the LENGTH OF the element at position i of words AND the LENGTH OF the element at position j of words DOWN TO one STEP negative one  
                IF the substring from position LENGTH OF the element at position i of words MINUS k PLUS one TO the LENGTH OF the element at position i of words of the element at position i of words EQUALS the substring from position one TO position k of the element at position j of words THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp WITH PARAMETERS mask AND i  
            IF mask EQUALS the RESULT OF ONE LEFT SHIFTED BY n MINUS one THEN  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF mask BITWISE AND the RESULT OF ONE LEFT SHIFTED BY j EQUALS zero THEN  
                    SET length AND path TO the RESULT OF CALLING dp WITH mask BITWISE OR the RESULT OF ONE LEFT SHIFTED BY j AND j  
                    INCREMENT length BY the LENGTH OF the element at position j of words MINUS the RESULT OF overlap WITH PARAMETERS i AND j  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the substring from position overlap WITH PARAMETERS i AND j PLUS one TO the LENGTH OF the element at position j of words of the element at position j of words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length AND path TO the RESULT OF CALLING dp WITH the RESULT OF ONE LEFT SHIFTED BY i AND i  
            INCREMENT length BY the LENGTH OF the element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO the element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
