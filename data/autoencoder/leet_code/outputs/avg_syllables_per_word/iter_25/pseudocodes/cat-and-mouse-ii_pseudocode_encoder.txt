CLASS Solution  
    FUNCTION canMouseWin(grid, catJump, mouseJump)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET cat_start TO zero  
        SET mouse_start TO zero  
        SET food TO zero  
        SET dirs TO the sequence of values minus one zero one zero minus one  
        SET g_mouse TO a list of empty lists with LENGTH m MULTIPLIED BY n  
        SET g_cat TO a list of empty lists with LENGTH m MULTIPLIED BY n  
        FOR each index i and element row IN grid  
            FOR each index j and element c IN row  
                IF c EQUALS the character number sign  
                    CONTINUE to the next iteration of the inner loop  
                END IF  
                SET v TO i MULTIPLIED BY n PLUS j  
                IF c EQUALS the character C  
                    SET cat_start TO v  
                ELSE IF c EQUALS the character M  
                    SET mouse_start TO v  
                ELSE IF c EQUALS the character F  
                    SET food TO v  
                END IF  
                FOR each pair of consecutive elements a and b IN dirs  
                    FOR k FROM zero TO mouseJump INCLUSIVE  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid EQUALS the character number sign  
                            BREAK out of this FOR loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO the list at position v of g_mouse  
                    END FOR  
                    FOR k FROM zero TO catJump INCLUSIVE  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid EQUALS the character number sign  
                            BREAK out of this FOR loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO the list at position v of g_cat  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN the result of calling the method calc with arguments g_mouse g_cat mouse_start cat_start food EQUALS one  
    END FUNCTION  

    FUNCTION calc(g_mouse, g_cat, mouse_start, cat_start, hole)  
        FUNCTION get_prev_states(state)  
            SET m TO element at position zero of state  
            SET c TO element at position one of state  
            SET t TO element at position two of state  
            SET pt TO the bitwise exclusive OR of t and one  
            SET pre TO an empty list  
            IF pt EQUALS one  
                FOR each pc IN the list at position c of g_cat  
                    IF element at position m of ans WITHIN position pc AND position one EQUALS zero  
                        APPEND the tuple of m pc pt TO pre  
                    END IF  
                END FOR  
            ELSE  
                FOR each pm IN the list at position m of g_mouse  
                    IF element at position pm of ans WITHIN position c AND position zero EQUALS zero  
                        APPEND the tuple of pm c zero TO pre  
                    END IF  
                END FOR  
            END IF  
            RETURN pre  
        END FUNCTION  

        SET n TO the LENGTH OF g_mouse  
        SET degree TO a three dimensional list of zeros with dimensions n by n by two  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET the element at position zero of the list at position j of the list at position i of degree TO the LENGTH OF the list at position i of g_mouse  
                SET the element at position one of the list at position j of the list at position i of degree TO the LENGTH OF the list at position j of g_cat  
            END FOR  
        END FOR  

        SET ans TO a three dimensional list of zeros with dimensions n by n by two  
        SET q TO an empty double ended queue  
        FOR i FROM zero TO n MINUS one  
            SET the element at position one of the list at position i of the list at position hole of ans TO one  
            SET the element at position zero of the list at position hole of the list at position i of ans TO two  
            SET the element at position one of the list at position i of the list at position i of ans TO two  
            SET the element at position zero of the list at position i of the list at position i of ans TO two  
            APPEND the tuple of hole i one TO q  
            APPEND the tuple of i hole zero TO q  
            APPEND the tuple of i i zero TO q  
            APPEND the tuple of i i one TO q  
        END FOR  
        WHILE q is not empty  
            SET state TO the element popped from the left of q  
            SET t TO the element at position t of ans WITHIN element at position one of state WITHIN element at position zero of state  
            FOR each prev_state IN the result of calling get_prev_states with argument state  
                SET pm TO element at position zero of prev_state  
                SET pc TO element at position one of prev_state  
                SET pt TO element at position two of prev_state  
                IF pt EQUALS t MINUS one  
                    SET the element at position pt of the list at position pc of the list at position pm of ans TO t  
                    APPEND prev_state TO q  
                ELSE  
                    DECREMENT the element at position pt of the list at position pc of the list at position pm of degree BY one  
                    IF the element at position pt of the list at position pc of the list at position pm of degree EQUALS zero  
                        SET the element at position pt of the list at position pc of the list at position pm of ans TO t  
                        APPEND prev_state TO q  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN the element at position zero of the list at position cat_start of the list at position mouse_start of ans  
    END FUNCTION  
END CLASS
