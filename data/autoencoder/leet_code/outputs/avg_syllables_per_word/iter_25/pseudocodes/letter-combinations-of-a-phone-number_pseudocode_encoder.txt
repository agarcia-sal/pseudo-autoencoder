CLASS Solution  
    FUNCTION letterCombinations WITH PARAMETER digits  
        IF digits IS NOT PRESENT  
            RETURN an empty list  
        END IF  
        SET phone_map TO a mapping from the string two TO the string abc the string three TO the string def the string four TO the string ghi the string five TO the string jkl the string six TO the string mno the string seven TO the string pqrs the string eight TO the string tuv the string nine TO the string wxyz  
        FUNCTION backtrack WITH PARAMETERS combination and next_digits  
            IF the LENGTH OF next_digits EQUALS zero  
                APPEND the value of combination TO the list output  
            ELSE  
                FOR each letter IN the string mapped by the element at position zero of next_digits in phone_map  
                    CALL backtrack WITH combination FOLLOWED BY the value of letter AND next_digits FROM position one TO the END  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO an empty list  
        CALL backtrack WITH an empty string AND the value of digits  
        RETURN the list output  
    END FUNCTION  
END CLASS
