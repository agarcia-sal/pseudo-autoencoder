CLASS Solution  
    FUNCTION minDeletionSize WITH PARAMETER strs  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of length n where each element IS one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for every k FROM zero TO m MINUS one the element at position j of element at position k of strs IS LESS THAN OR EQUAL TO the element at position i of element at position k of strs HOLDS  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the MAXIMUM value IN dp  
    END FUNCTION  
END CLASS
