CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the value attribute of p is not equal to the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree with parameters the left attribute of p and the left attribute of q AND calling is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue with parameters self root  
        SET queue TO a new double ended queue initialized with root as the only element  
        SET leftmost_value TO none  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            FOR i FROM zero TO level_size minus one  
                SET node TO the element removed from the front of queue  
                IF i equals zero  
                    SET leftmost_value TO the value attribute of node  
                END IF  
                IF the left attribute of node is not none  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node is not none  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
