CLASS Solution  
    FUNCTION recoverArray(nums)  
        SET nums TO the list nums arranged in ascending order  
        SET n TO the length of the list nums  
        SET first TO the element at position zero of the list nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUAL TO the element at position i of nums THEN  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            SET count TO a mapping of numbers to their frequencies in nums  
            SET arr TO an empty list  
            FOR each element x IN nums  
                IF the value of count at key x EQUALS zero THEN  
                    CONTINUE to the next iteration of the FOR loop  
                END IF  
                IF the value of count at key x PLUS two MULTIPLIED BY k EQUALS zero THEN  
                    BREAK out of the FOR loop  
                END IF  
                APPEND x PLUS k TO the list arr  
                DECREMENT the value of count at key x BY one  
                DECREMENT the value of count at key x PLUS two MULTIPLIED BY k BY one  
            END FOR  
            IF the length of arr EQUALS n DIVIDED BY two THEN  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
