CLASS Solution  
    FUNCTION ipToCIDR WITH PARAMETERS ip AND n  
        FUNCTION ip_to_int WITH PARAMETER ip  
            SET parts TO the list obtained by converting each element of the list obtained by splitting ip by dot into integer  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  
        FUNCTION int_to_ip WITH PARAMETER num  
            RETURN the string formed by concatenating the string representations of the element at position three MINUS i of the integer obtained by dividing num RIGHT SHIFTED BY eight MULTIPLIED BY i MODULO two HUNDRED FIFTY SIX for i FROM zero TO three separated by a dot  
        END FUNCTION  
        FUNCTION largest_cidr_size WITH PARAMETERS start AND n  
            IF the bitwise AND of start AND one EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND the bitwise AND of start AND x MINUS one EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  
        SET result TO an empty list  
        SET start_ip TO the result of calling ip_to_int WITH ip  
        WHILE n GREATER THAN zero  
            SET size TO the result of calling largest_cidr_size WITH start_ip AND n  
            APPEND the concatenation of the result of calling int_to_ip WITH start_ip PLUS a slash PLUS the string representation of thirty-two MINUS the bit length of size MINUS one TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
