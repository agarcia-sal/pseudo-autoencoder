CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO the list containing one thousand nine hundred ninety nine hundred four hundred one hundred ninety fifty forty ten nine five four one in descending order  
        SET syms TO the list containing the strings M CM D CD C XC L XL X IX V IV I in descending order corresponding to val  
        SET roman_numeral TO an empty string  
        SET i TO zero  
        WHILE num greater than zero holds  
            SET count TO the integer division of num by the element at position i of val  
            FOR each index FROM one TO count  
                APPEND the element at position i of syms TO roman_numeral  
                DECREMENT num BY the element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
