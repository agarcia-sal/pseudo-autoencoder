CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val MINUS EQUALS zero COMMA left MINUS EQUALS none COMMA right MINUS EQUALS none  
        SET the val PROPERTY OF the current instance TO val  
        SET the left PROPERTY OF the current instance TO left  
        SET the right PROPERTY OF the current instance TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO the element popped from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the left PROPERTY OF node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the left PROPERTY OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the right PROPERTY OF node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the right PROPERTY OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p COMMA q  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the val PROPERTY OF p IS NOT EQUAL TO the val PROPERTY OF q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left PROPERTY OF p and the left PROPERTY OF q AND the result of is_same_tree with the right PROPERTY OF p and the right PROPERTY OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum WITH PARAMETERS root COMMA targetSum  
        FUNCTION helper WITH PARAMETERS node COMMA current_sum COMMA prefix_sums  
            IF node IS none THEN  
                RETURN zero  
            END IF  
            INCREMENT current_sum BY the val PROPERTY OF node  
            SET paths_to_here TO the value associated with the key current_sum MINUS targetSum in prefix_sums OR zero if that key does not exist  
            SET the value associated with the key current_sum in prefix_sums TO the value associated with the key current_sum in prefix_sums OR zero if that key does not exist PLUS one  
            SET paths_in_left TO the result of helper with the left PROPERTY OF node COMMA current_sum COMMA prefix_sums  
            SET paths_in_right TO the result of helper with the right PROPERTY OF node COMMA current_sum COMMA prefix_sums  
            DECREMENT the value associated with the key current_sum in prefix_sums BY one  
            RETURN the sum of paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO a dictionary with zero as key AND one as value  
        RETURN the result of helper with root COMMA zero COMMA prefix_sums  
    END FUNCTION  
END CLASS
