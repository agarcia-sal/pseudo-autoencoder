CLASS Solution  
    FUNCTION fullJustify(words AND maxWidth)  
        FUNCTION justify_line(line AND num_of_letters AND is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN the concatenation of all words in line separated by single space PLUS a sequence of spaces whose length EQUALS maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps USING INTEGER DIVISION  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND a sequence of spaces whose length EQUALS spaces_between_words PLUS one TO the element at position index of line  
                    ELSE  
                        APPEND a sequence of spaces whose length EQUALS spaces_between_words TO the element at position index of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements in line without separator  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the return value of justify_line with parameters line AND num_of_letters AND false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        APPEND the return value of justify_line with parameters line AND num_of_letters AND true TO result  
        RETURN result  
    END FUNCTION  
END CLASS
