CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET the value property of self TO val  
        SET the next property of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the collection values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the subcollection of values from position one TO the END  
        SET node TO a new instance of ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND the result of calling is_same_list with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs WITH PARAMETERS self head  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next property of dummy TO head  
        SET current TO dummy  
        WHILE the next property of current IS NOT none AND the next property of the next property of current IS NOT none  
            SET first TO the next property of current  
            SET second TO the next property of the next property of current  
            SET the next property of current TO second  
            SET the next property of first TO the next property of second  
            SET the next property of second TO first  
            SET current TO the next property of the next property of current  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
