CLASS Solution  
    FUNCTION minDifficulty WITH PARAMETERS jobDifficulty AND d  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d THEN  
            RETURN minus one  
        END IF  
        
        FUNCTION dp WITH PARAMETERS i AND d  
            IF d EQUALS one THEN  
                RETURN the MAXIMUM VALUE among the elements from position i TO the END of jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO the MAXIMUM VALUE between current_max AND the element at position j of jobDifficulty  
                SET min_difficulty TO the MINIMUM VALUE between min_difficulty AND the SUM of current_max AND dp called with j PLUS one AND d MINUS one  
            END FOR  
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp called with zero AND d  
    END FUNCTION  
END CLASS
