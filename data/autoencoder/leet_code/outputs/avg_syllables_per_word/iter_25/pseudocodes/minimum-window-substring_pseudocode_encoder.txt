CLASS Solution  
    FUNCTION minWindow WITH PARAMETERS s t  
        INITIALIZE t_count AS the frequency counts of each character in t  
        SET required TO the number of unique characters in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero representing the number of unique characters in t currently satisfied in the window  
        INITIALIZE window_counts as empty frequency counts  
        SET min_length TO infinity  
        SET min_start TO zero  
        WHILE right is less than the length of s  
            INCREMENT the count of the character at position right of s in window_counts by one  
            IF the character at position right of s exists in t_count AND the count of that character in window_counts EQUALS its count in t_count  
                INCREMENT formed BY one  
            END IF  
            WHILE left is less than or equal to right AND formed EQUALS required  
                SET char TO the character at position left of s  
                IF right MINUS left PLUS one is less than min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                DECREMENT the count of char in window_counts BY one  
                IF char exists in t_count AND the count of char in window_counts is less than its count in t_count  
                    DECREMENT formed BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            INCREMENT right BY one  
        END WHILE  
        IF min_length EQUALS infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start to position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
