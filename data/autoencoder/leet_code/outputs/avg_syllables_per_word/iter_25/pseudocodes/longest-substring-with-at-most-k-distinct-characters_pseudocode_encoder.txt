CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s k  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        SET char_map TO an empty collection of key and value pairs  
        SET left TO zero  
        SET max_length TO zero  
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            SET the element at position right of s AS key IN char_map TO right AS value  
            IF the LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map HAVING the MINIMUM value  
                SET left TO the value ASSOCIATED WITH leftmost_char IN char_map PLUS one  
                REMOVE the element ASSOCIATED WITH leftmost_char FROM char_map  
            END IF  
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
