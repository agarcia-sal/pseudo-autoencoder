CLASS Solution  
    FUNCTION pacificAtlantic with parameters heights  
        FUNCTION bfs with parameters q and vis  
            WHILE the LENGTH OF q is greater than zero  
                FOR each integer from zero TO the LENGTH OF q MINUS one  
                    SET i and j TO the elements REMOVED FROM the LEFT OF q at the current iteration  
                    FOR each pair a and b IN the list consisting of zero PLUS zero MINUS one AND zero PLUS one AND one PLUS zero AND zero MINUS one  
                        SET x TO i PLUS a  
                        SET y TO j PLUS b  
                        IF zero LESS THAN OR EQUAL TO x AND x LESS THAN m AND zero LESS THAN OR EQUAL TO y AND y LESS THAN n AND the pair x and y NOT IN vis AND the element at position x of heights AT position y OF heights IS GREATER THAN OR EQUAL TO the element at position i of heights AT position j OF heights THEN  
                            ADD the pair x and y TO vis  
                            ADD the pair x and y TO the RIGHT OF q  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
        END FUNCTION  

        SET m TO the LENGTH OF heights  
        SET n TO the LENGTH OF the element at position zero of heights  
        SET vis1 TO an empty set  
        SET vis2 TO an empty set  
        SET q1 TO an empty double ended queue  
        SET q2 TO an empty double ended queue  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF i EQUALS zero OR j EQUALS zero THEN  
                    ADD the pair i and j TO vis1  
                    ADD the pair i and j TO the RIGHT OF q1  
                END IF  
                IF i EQUALS m MINUS one OR j EQUALS n MINUS one THEN  
                    ADD the pair i and j TO vis2  
                    ADD the pair i and j TO the RIGHT OF q2  
                END IF  
            END FOR  
        END FOR  
        CALL bfs with arguments q1 and vis1  
        CALL bfs with arguments q2 and vis2  
        SET result TO an empty list  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair i and j IS IN vis1 AND the pair i and j IS IN vis2 THEN  
                    APPEND the pair i and j TO result  
                END IF  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
