CLASS Solution  
    FUNCTION findOrder(numCourses numCourses PARAMETER prerequisites prerequisites)  
        SET adj_list TO empty mapping from course to empty list FOR each course FROM zero TO numCourses MINUS one  
        SET in_degrees TO list of zeros of LENGTH numCourses  
        FOR each pair named dest comma src IN prerequisites  
            APPEND dest TO the list at key src in adj_list  
            INCREMENT the element at position dest of in_degrees BY one  
        END FOR  
        SET queue TO a double-ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at position course of in_degrees EQUALS zero  
        SET topological_order TO empty list  
        WHILE queue IS NOT empty  
            SET current TO the element popped from the left side of queue  
            APPEND current TO topological_order  
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor of in_degrees BY one  
                IF the element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
