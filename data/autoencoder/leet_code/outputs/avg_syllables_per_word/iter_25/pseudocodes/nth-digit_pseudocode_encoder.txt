CLASS Solution  
    FUNCTION findNthDigit WITH PARAMETER n  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        INCREMENT start BY the INTEGER DIVISION OF n MINUS one BY length  
        SET number_string TO the STRING REPRESENTATION OF start  
        SET digit_index TO the REMAINDER OF n MINUS one DIVIDED BY length  
        SET digit_character TO the ELEMENT AT POSITION digit_index OF number_string  
        CONVERT digit_character TO an INTEGER AND RETURN IT  
    END FUNCTION  
END CLASS
