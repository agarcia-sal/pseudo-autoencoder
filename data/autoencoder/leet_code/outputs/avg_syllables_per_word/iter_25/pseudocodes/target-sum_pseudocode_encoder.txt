CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                RETURN one IF current_sum EQUALS target ELSE zero  
            END IF  
            IF the TUPLE consisting of index and current_sum IS AN ELEMENT OF memo  
                RETURN the ELEMENT at the TUPLE consisting of index and current_sum OF memo  
            END IF  
            SET positive TO the RESULT OF CALLING dfs WITH index INCREMENTED BY one AND current_sum INCREMENTED BY the ELEMENT at position index OF nums AND memo  
            SET negative TO the RESULT OF CALLING dfs WITH index INCREMENTED BY one AND current_sum DECREMENTED BY the ELEMENT at position index OF nums AND memo  
            SET the ELEMENT at the TUPLE consisting of index and current_sum OF memo TO positive PLUS negative  
            RETURN the ELEMENT at the TUPLE consisting of index and current_sum OF memo  
        END FUNCTION  
        RETURN the RESULT OF CALLING dfs WITH zero AND zero AND an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
