CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty set  
        SET stack TO list containing zero  
        WHILE the stack is not empty  
            SET current_room TO element removed from the end of the stack  
            ADD current_room to the visited set  
            FOR each key in element at position current_room of rooms  
                IF key is not in visited set  
                    APPEND key to stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the size of visited set EQUALS the size of rooms  
    END FUNCTION  
END CLASS
