CLASS Solution  
    FUNCTION maxValueOfCoins WITH PARAMETERS piles AND k  
        SET n TO the LENGTH OF piles  
        SET dp TO a list of n PLUS one elements each being a list of k PLUS one zeros  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM of j AND the LENGTH OF current_pile  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp at position i and j TO the GREATER OF dp at position i and j AND dp at position i MINUS one and j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position n and k of dp  
    END FUNCTION  
END CLASS
