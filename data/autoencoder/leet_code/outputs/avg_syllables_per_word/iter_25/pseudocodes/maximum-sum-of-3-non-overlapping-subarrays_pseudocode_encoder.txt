CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums AS List OF INTEGER, k AS INTEGER) RETURNS List OF INTEGER  
        SET n TO the LENGTH OF nums  
        SET sums TO a list of zeroes with LENGTH equal to n MINUS k PLUS one  
        SET current_sum TO the SUM of the first k elements of nums  
        SET element at position zero of sums TO current_sum  
        
        FOR i FROM one TO n MINUS k  
            INCREMENT current_sum BY the element at position i PLUS k MINUS one of nums MINUS the element at position i MINUS one of nums  
            SET element at position i of sums TO current_sum  
        END FOR  
        
        SET left TO a list of zeroes with LENGTH equal to n MINUS k PLUS one  
        SET right TO a list where every element has the value n MINUS k with LENGTH equal to n MINUS k PLUS one  
        
        SET max_sum TO zero  
        FOR i FROM zero TO n MINUS k  
            IF the element at position i of sums GREATER THAN the element at position max_sum of sums THEN  
                SET max_sum TO i  
            END IF  
            SET element at position i of left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWN TO zero  
            IF the element at position i of sums GREATER THAN OR EQUAL TO the element at position max_sum of sums THEN  
                SET max_sum TO i  
            END IF  
            SET element at position i of right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a list containing zero zero zero  
        FOR j FROM k TO n MINUS twice k  
            SET i TO the element at position j MINUS k of left  
            SET l TO the element at position j PLUS k of right  
            SET total TO the element at position i of sums PLUS the element at position j of sums PLUS the element at position l of sums  
            IF total GREATER THAN max_total THEN  
                SET max_total TO total  
                SET result TO a list containing i j l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
