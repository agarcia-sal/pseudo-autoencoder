CLASS Solution  
    FUNCTION minTrioDegree with parameters n and edges  
        SET graph TO a mapping from each integer from one to n inclusive TO an empty set  
        FOR each pair u and v IN edges  
            ADD v TO the set associated with u in graph  
            ADD u TO the set associated with v in graph  
        END FOR  
        SET min_degree TO infinite positive value  
        SET trio_found TO false  
        FOR each integer i FROM one TO n inclusive  
            FOR each integer j FROM one plus i TO n inclusive  
                FOR each integer k FROM one plus j TO n inclusive  
                    IF the set associated with i in graph CONTAINS j AND the set associated with i in graph CONTAINS k AND the set associated with j in graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the size of the set associated with i in graph PLUS the size of the set associated with j in graph PLUS the size of the set associated with k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
