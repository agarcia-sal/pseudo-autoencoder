CLASS Solution
    FUNCTION numPoints WITH PARAMETERS darts AND r
        FUNCTION distance WITH PARAMETERS p1 AND p2
            SET horizontal_difference TO element at position zero of p1 MINUS element at position zero of p2
            SET vertical_difference TO element at position one of p1 MINUS element at position one of p2
            SET squared_horizontal_difference TO horizontal_difference MULTIPLIED BY horizontal_difference
            SET squared_vertical_difference TO vertical_difference MULTIPLIED BY vertical_difference
            SET sum_of_squares TO squared_horizontal_difference PLUS squared_vertical_difference
            RETURN the square root of sum_of_squares
        END FUNCTION

        FUNCTION circle_center WITH PARAMETERS p1 AND p2 AND r
            SET d TO the result of calling distance with p1 and p2
            IF d GREATER THAN two MULTIPLIED BY r THEN
                RETURN nothing
            END IF
            SET a TO the square root of r MULTIPLIED BY r MINUS the result of d DIVIDED BY two MULTIPLIED BY d DIVIDED BY two
            SET h_horizontal TO element at position zero of p1 PLUS element at position zero of p2 DIVIDED BY two
            SET h_vertical TO element at position one of p1 PLUS element at position one of p2 DIVIDED BY two
            SET dx TO element at position one of p2 MINUS element at position one of p1 MULTIPLIED BY a DIVIDED BY d
            SET dy TO element at position zero of p2 MINUS element at position zero of p1 MULTIPLIED BY a DIVIDED BY d
            SET first_center_horizontal TO h_horizontal MINUS dx
            SET first_center_vertical TO h_vertical PLUS dy
            SET second_center_horizontal TO h_horizontal PLUS dx
            SET second_center_vertical TO h_vertical MINUS dy
            RETURN a list containing a tuple of first_center_horizontal and first_center_vertical AND a tuple of second_center_horizontal and second_center_vertical
        END FUNCTION

        FUNCTION is_inside WITH PARAMETERS circle AND point AND r
            SET distance_to_point TO the result of calling distance with circle and point
            IF distance_to_point LESS THAN OR EQUAL TO r THEN
                RETURN true
            ELSE
                RETURN false
            END IF
        END FUNCTION

        SET n TO the length of darts
        IF n EQUALS one THEN
            RETURN one
        END IF

        SET max_count TO one

        FOR i FROM zero TO n MINUS one
            FOR j FROM i PLUS one TO n MINUS one
                SET centers TO the result of calling circle_center with element at position i of darts AND element at position j of darts AND r
                IF centers IS NOT nothing THEN
                    FOR each center IN centers
                        SET count TO zero
                        FOR each dart IN darts
                            IF the result of calling is_inside with center AND dart AND r IS true THEN
                                INCREMENT count BY one
                            END IF
                        END FOR
                        IF count GREATER THAN max_count THEN
                            SET max_count TO count
                        END IF
                    END FOR
                END IF
            END FOR
        END FOR

        RETURN max_count
    END FUNCTION
END CLASS
