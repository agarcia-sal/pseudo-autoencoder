CLASS Solution  
    FUNCTION checkValidString WITH PARAMETER s  
        SET low TO zero  
        SET high TO zero  
        FOR each char IN s  
            IF char EQUALS open parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF char EQUALS close parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            SET low TO the GREATER VALUE BETWEEN low AND zero  
        END FOR  
        RETURN whether low EQUALS zero  
    END FUNCTION  
END CLASS
