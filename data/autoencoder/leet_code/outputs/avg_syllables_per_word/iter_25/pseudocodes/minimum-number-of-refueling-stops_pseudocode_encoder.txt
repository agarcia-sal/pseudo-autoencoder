CLASS Solution  
    FUNCTION minRefuelStops WITH PARAMETERS target startFuel stations  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND the list containing target and zero TO stations  

        FOR each pair position fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is LESS THAN position  
                SET current_fuel TO current_fuel MINUS the element removed from max_heap according to heap pop operation  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel is LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative fuel TO max_heap according to heap push operation  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
