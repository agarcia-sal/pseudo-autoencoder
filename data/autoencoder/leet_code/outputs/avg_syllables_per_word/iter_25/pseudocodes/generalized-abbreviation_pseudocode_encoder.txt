CLASS Solution  
    FUNCTION generateAbbreviations with parameters word  
        FUNCTION backtrack with parameters index path count  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the string representation of count TO path  
                END IF  
                APPEND the concatenation of all elements in path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack with arguments index PLUS one and the concatenation of path with the string representation of count and the element at position index of word and zero as count  
                ELSE  
                    CALL backtrack with arguments index PLUS one and the concatenation of path with the element at position index of word and zero as count  
                END IF  
                CALL backtrack with arguments index PLUS one path and count PLUS one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack with zero empty list and zero  
        RETURN result  
    END FUNCTION  
END CLASS
