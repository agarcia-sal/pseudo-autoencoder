CLASS Solution  
    FUNCTION rotate(nums PARAMETER k)  
        SET n TO the LENGTH OF nums  
        SET k TO the REMAINDER OF k DIVIDED BY n  
        
        FUNCTION reverse(start PARAMETER end)  
            WHILE start IS LESS THAN end  
                SET temporary_value TO the element at position start of nums  
                SET the element at position start of nums TO the element at position end of nums  
                SET the element at position end of nums TO temporary_value  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH zero AND n MINUS one  
        CALL reverse WITH zero AND k MINUS one  
        CALL reverse WITH k AND n MINUS one  
    END FUNCTION  
END CLASS
