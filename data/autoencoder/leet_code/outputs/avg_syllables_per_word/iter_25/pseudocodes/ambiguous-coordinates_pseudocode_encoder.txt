CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING from position zero TO position d MINUS one of frag  
                SET right TO the SUBSTRING from position d TO the END OF frag  
                IF the FIRST CHARACTER of left IS NOT zero OR left EQUALS zero AND the LAST CHARACTER of right IS NOT zero  
                    YIELD the CONCATENATION of left PLUS a PERIOD IF d IS NOT EQUAL TO n OTHERWISE an empty string PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING from position one TO position LENGTH OF s MINUS two of s  
        RETURN the LIST constructed by FOR i FROM one TO LENGTH OF s MINUS one  
            FOR a COMMA b IN the CARTESIAN PRODUCT of make with the SUBSTRING from position zero TO position i MINUS one of s AND make with the SUBSTRING from position i TO the END OF s  
                THE STRING consisting of an OPEN PARENTHESIS PLUS a PLUS a COMMA PLUS a SPACE PLUS b PLUS a CLOSE PARENTHESIS  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
