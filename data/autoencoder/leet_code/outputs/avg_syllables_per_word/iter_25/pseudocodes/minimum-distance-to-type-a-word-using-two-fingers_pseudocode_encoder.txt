CLASS Solution  
    FUNCTION minimumDistance with parameter word  
        FUNCTION distance with parameters c1 and c2  
            IF c1 IS None  
                RETURN zero  
            END IF  
            SET x1 TO the INTEGER division of the ORDINAL value of character c1 MINUS the ORDINAL value of character uppercase A BY six  
            SET y1 TO the REMAINDER of the ORDINAL value of character c1 MINUS the ORDINAL value of character uppercase A DIVIDED BY six  
            SET x2 TO the INTEGER division of the ORDINAL value of character c2 MINUS the ORDINAL value of character uppercase A BY six  
            SET y2 TO the REMAINDER of the ORDINAL value of character c2 MINUS the ORDINAL value of character uppercase A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        FUNCTION dp with parameters i f1 and f2  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO element at position i of word  
            SET d1 TO the result of calling distance with f1 and c PLUS the result of calling dp with i PLUS one c and f2  
            SET d2 TO the result of calling distance with f2 and c PLUS the result of calling dp with i PLUS one f1 and c  
            RETURN the MINIMUM of d1 and d2  
        END FUNCTION  
        RETURN the result of calling dp with zero None and None  
    END FUNCTION  
END CLASS
