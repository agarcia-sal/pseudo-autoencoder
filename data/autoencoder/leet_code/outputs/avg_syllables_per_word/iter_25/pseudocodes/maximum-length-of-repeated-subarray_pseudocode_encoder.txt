CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO a list consisting of a length of the length of nums1 plus one where each element is a list consisting of the length of the length of nums2 plus one filled with zeros  
        SET max_length TO zero  
        FOR i FROM one TO the length of nums1 inclusive  
            FOR j FROM one TO the length of nums2 inclusive  
                IF the element at position i minus one of nums1 EQUALS the element at position j minus one of nums2 THEN  
                    SET the element at position i and j of dp TO the element at position i minus one and j minus one of dp PLUS one  
                    IF the element at position i and j of dp GREATER THAN max_length THEN  
                        SET max_length TO the element at position i and j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
