CLASS TreeNode  
    FUNCTION __init__ with parameters self val default zero left default None right default None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values list  
    IF values is empty holds  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty holds  
        SET node TO the first element removed from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not None holds  
            SET node dot left TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not None holds  
            SET node dot right TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is None AND q is None holds  
        RETURN True  
    ELSE IF p is None OR q is None holds  
        RETURN False  
    ELSE IF p dot val is NOT EQUAL TO q dot val holds  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree with parameters p dot left and q dot left AND calling is_same_tree with parameters p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree with parameters self root optional TreeNode  
        IF root is None OR root dot left is None holds  
            RETURN root  
        END IF  
        SET new_root TO the result of calling self dot upsideDownBinaryTree with parameter root dot left  
        SET the left child of root dot left TO root dot right  
        SET the right child of root dot left TO root  
        SET root dot left TO None  
        SET root dot right TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
