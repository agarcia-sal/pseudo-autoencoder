CLASS Solution  
    FUNCTION exclusiveTime WITH PARAMETERS n logs  
        SET exclusive_time TO a list containing zero repeated n times  
        SET stack TO an empty list  
        
        FOR each log IN logs  
            SPLIT log by colon INTO function_id action timestamp  
            SET function_id TO function_id converted to integer  
            SET timestamp TO timestamp converted to integer  
            
            IF action EQUALS the word start  
                IF stack is not empty  
                    SET prev_function_id TO the first element of the last tuple in stack  
                    SET prev_timestamp TO the second element of the last tuple in stack  
                    INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND the tuple consisting of function_id AND timestamp TO stack  
            ELSE  
                REMOVE the last tuple from stack AND ASSIGN the first element TO prev_function_id AND the second element TO prev_timestamp  
                INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp PLUS one  
                IF stack is not empty  
                    SET the second element of the last tuple in stack TO timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
