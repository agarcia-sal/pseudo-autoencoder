CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulted to zero left defaulted to none right defaulted to none  
        SET the val attribute of self TO val  
        SET the left attribute of self TO left  
        SET the right attribute of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode object initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left attribute of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right attribute of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the val attribute of p is not equal to the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with p left attribute and q left attribute AND the result of is_same_tree with p right attribute and q right attribute  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow with parameters self root val depth  
        IF depth is equal to one  
            RETURN a new TreeNode object initialized with val as val and left as root  
        END IF  
        SET queue TO a double-ended queue initialized with a tuple containing root and one  
        WHILE queue is not empty  
            SET node and level TO the elements removed from the front of queue  
            IF level is equal to depth MINUS one  
                SET the left attribute of node TO a new TreeNode object initialized with val as val and left as the current left attribute of node  
                SET the right attribute of node TO a new TreeNode object initialized with val as val and right as the current right attribute of node  
            ELSE IF level is less than depth MINUS one  
                IF the left attribute of node is not none  
                    APPEND a tuple containing the left attribute of node and level PLUS one TO queue  
                END IF  
                IF the right attribute of node is not none  
                    APPEND a tuple containing the right attribute of node and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
