CLASS Solution  
    FUNCTION minJumps WITH PARAMETER arr  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        INITIALIZE value_indices TO a DEFAULT DICTIONARY mapping each key TO an empty list  
        FOR each pair of i AND value FROM the enumeration of arr  
            APPEND i TO the list at key value in value_indices  
        END FOR  
        INITIALIZE queue TO a DOUBLE ENDED QUEUE containing the pair of zero and zero representing current_index and current_steps  
        INITIALIZE visited TO a SET containing zero  
        WHILE queue is not empty  
            REMOVE and RETURN the leftmost element in queue and DESTRUCTURE it INTO current_index and steps  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            INITIALIZE next_indices TO a LIST containing current_index PLUS one, current_index MINUS one, concatenated with the list at key arr at position current_index in value_indices  
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index is NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            SET the list at key arr at position current_index in value_indices TO an empty list  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
