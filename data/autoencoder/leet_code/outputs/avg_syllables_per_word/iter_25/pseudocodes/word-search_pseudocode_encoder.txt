CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS empty THEN  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero of board  
        SET visited TO an empty list  
        FOR each number FROM zero TO m MINUS one  
            APPEND a list containing n times the value False TO visited  
        END FOR  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word THEN  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the element at position x then y of visited EQUALS True OR the element at position x then y of board NOT EQUALS the element at position index of word THEN  
                RETURN False  
            END IF  

            SET the element at position x then y of visited TO True  
            FOR each dx dy IN the list composed of the tuple MINUS one zero PLUS one zero zero MINUS one zero PLUS one  
                IF the result of calling dfs with arguments x PLUS dx then y PLUS dy then index PLUS one THEN  
                    RETURN True  
                END IF  
            END FOR  
            SET the element at position x then y of visited TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the result of calling dfs with arguments i then j then zero THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
