CLASS Solution  
    FUNCTION maxRunTime WITH PARAMETERS n BATTERIES  
        FUNCTION canRunFor WITH PARAMETER time  
            SET total TO zero  
            FOR each battery IN BATTERIES  
                INCREMENT total BY the lesser value BETWEEN battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  
        SET left TO zero  
        SET right TO the integer division of the sum of all elements in BATTERIES BY n  
        WHILE left LESS THAN right  
            SET mid TO the integer division of the sum of left plus right plus one BY two  
            IF canRunFor CALLED WITH mid RETURNS true  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
