CLASS Solution  
    FUNCTION countBattleships WITH PARAMETER board  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i of board MINUS one  
                IF element at position i of board at position j EQUALS the character X THEN  
                    IF the condition that i EQUALS zero OR element at position i MINUS one of board at position j EQUALS the character dot AND the condition that j EQUALS zero OR element at position i of board at position j MINUS one EQUALS the character dot THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
