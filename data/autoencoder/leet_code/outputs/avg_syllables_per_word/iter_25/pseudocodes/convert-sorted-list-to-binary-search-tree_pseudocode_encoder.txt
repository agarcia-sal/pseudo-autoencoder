CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the values list IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the slice of values from position one TO the end  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the expression that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the values list IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element obtained by removing from the left of queue  
        IF i IS less than the length of values AND the element at position i of values IS NOT none  
            SET the left attribute of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the length of values AND the element at position i of values IS NOT none  
            SET the right attribute of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    END IF  
    IF p IS none OR q IS none  
        RETURN false  
    END IF  
    IF the val attribute of p IS NOT EQUALS the val attribute of q  
        RETURN false  
    END IF  
    RETURN the logical AND of the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS self head  
        IF head IS none  
            RETURN none  
        END IF  
        
        FUNCTION findMiddle WITH PARAMETERS left right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast IS NOT EQUALS right AND the next attribute of fast IS NOT EQUALS right  
                SET slow TO the next attribute of slow  
                SET fast TO the next attribute of the next attribute of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST WITH PARAMETERS left right  
            IF left IS EQUALS right  
                RETURN none  
            END IF  
            
            SET mid TO the result of calling findMiddle with left and right  
            
            SET node TO a new TreeNode initialized with the val attribute of mid  
            SET the left attribute of node TO the result of calling convertListToBST with left and mid  
            SET the right attribute of node TO the result of calling convertListToBST with the next attribute of mid and right  
            RETURN node  
        END FUNCTION  
        
        RETURN the result of calling convertListToBST with head and none  
    END FUNCTION  
END CLASS
