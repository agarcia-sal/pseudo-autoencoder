CLASS Solution  
    FUNCTION findAllPeople WITH PARAMETERS n firstPerson meetings  
        SET know_secret TO the set containing zero and firstPerson  
        SORT meetings IN ORDER OF the element at position three of each meeting  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position three of meetings at position i  
            INITIALIZE graph TO an empty mapping from keys to empty lists  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position three of meetings at position i EQUALS current_time  
                SET x TO the element at position one of meetings at position i  
                SET y TO the element at position two of meetings at position i  
                APPEND y TO the list at key x in graph  
                APPEND x TO the list at key y in graph  
                INCREMENT i BY one  
            END WHILE  
            INITIALIZE queue TO the sequence of all persons in know_secret who are keys in graph  
            INITIALIZE visited TO an empty set  
            WHILE the LENGTH OF queue IS GREATER THAN zero  
                REMOVE and SET person TO the first element from queue  
                IF person IN visited THEN  
                    CONTINUE  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person in graph  
                    IF neighbor NOT IN visited THEN  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list of all elements in know_secret  
    END FUNCTION  
END CLASS
