CLASS Solution  
    FUNCTION maxSizeSlices WITH PARAMETER slices  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp WITH PARAMETERS start end k  
            IF k EQUALS zero OR start GREATER THAN end THEN  
                RETURN zero  
            END IF  
            
            SET pick TO the element at position start of slices PLUS the RESULT OF dp WITH PARAMETERS start PLUS two end k MINUS one  
            SET skip TO the RESULT OF dp WITH PARAMETERS start PLUS one end k  
            RETURN the GREATER OF pick AND skip  
        END FUNCTION  
        
        SET case1 TO the RESULT OF dp WITH PARAMETERS one LENGTH OF slices MINUS one n  
        SET case2 TO the RESULT OF dp WITH PARAMETERS zero LENGTH OF slices MINUS two n  
        
        RETURN the GREATER OF case1 AND case2  
    END FUNCTION  
END CLASS
