CLASS Solution  
    FUNCTION numSubarrayBoundedMax(nums left right)  
        FUNCTION count(max_value)  
            SET count TO zero  
            SET current_length TO zero  
            FOR each num IN nums  
                IF num IS LESS THAN OR EQUAL TO max_value  
                    INCREMENT current_length BY one  
                    INCREMENT count BY current_length  
                ELSE  
                    SET current_length TO zero  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN result OF count WITH right MINUS result OF count WITH left MINUS one  
    END FUNCTION  
END CLASS
