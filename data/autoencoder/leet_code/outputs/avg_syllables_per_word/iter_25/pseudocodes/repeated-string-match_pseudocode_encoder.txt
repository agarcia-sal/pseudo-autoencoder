CLASS Solution  
    FUNCTION repeatedStringMatch WITH PARAMETERS a b  
        SET min_repeats TO negative of the integer division of the length of b divided by the length of a with ceiling effect  
        IF b is contained within the string formed by repeating a min_repeats times  
            RETURN min_repeats  
        END IF  
        IF b is contained within the string formed by repeating a the sum of min_repeats and one times  
            RETURN the sum of min_repeats and one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
