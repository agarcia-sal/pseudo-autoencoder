CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT the list intervals in ascending order by the second element of each interval and if those are equal then in descending order by the first element of each interval  
        SET first TO the second element of the first interval MINUS one  
        SET second TO the second element of the first interval  
        SET count TO two  
        FOR each start and end in the list intervals starting from the second interval  
            IF start is GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start is GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
