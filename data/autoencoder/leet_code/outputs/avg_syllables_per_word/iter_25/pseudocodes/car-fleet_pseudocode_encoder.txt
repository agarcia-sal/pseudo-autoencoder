CLASS Solution  
    FUNCTION carFleet WITH PARAMETERS target position speed  
        SET paired_list TO the LIST OF PAIRS OF elements FROM position AND speed SORTED IN DESCENDING ORDER BY position  
        SET time_to_target TO an EMPTY LIST  
        FOR each pair p AND s IN paired_list  
            SET time TO target MINUS p DIVIDED BY s  
            APPEND time TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
