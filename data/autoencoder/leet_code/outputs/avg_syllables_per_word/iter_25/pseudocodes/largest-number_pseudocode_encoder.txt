CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS nums  
        SET nums_str TO an empty list  
        FOR each num IN nums  
            APPEND the string representation of num TO nums_str  
        END FOR  
        
        FUNCTION compare WITH PARAMETERS x y  
            IF the concatenation of x PLUS y is GREATER THAN the concatenation of y PLUS x THEN  
                RETURN minus one  
            ELSE IF the concatenation of x PLUS y is LESS THAN the concatenation of y PLUS x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str IN ORDER ACCORDING TO comparison KEY compare  
        
        IF the element at position zero of nums_str EQUALS the string zero THEN  
            RETURN the string zero  
        END IF  
        
        SET result TO the concatenation of all elements in nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
