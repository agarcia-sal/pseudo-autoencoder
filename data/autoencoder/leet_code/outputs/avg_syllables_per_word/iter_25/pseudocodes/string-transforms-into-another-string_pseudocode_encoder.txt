CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        IF str1 EQUALS str2 THEN  
            RETURN truth  
        END IF  
        SET conversion_map TO an empty collection mapping keys to values  
        FOR each pair char1 char2 IN elements simultaneously taken from str1 and str2  
            IF char1 IS A KEY IN conversion_map THEN  
                IF value mapped by char1 IN conversion_map IS NOT EQUAL TO char2 THEN  
                    RETURN falsity  
                END IF  
            ELSE  
                SET value mapped by char1 IN conversion_map TO char2  
            END IF  
        END FOR  
        IF count of unique elements in str2 IS EQUAL TO twenty six THEN  
            RETURN falsity  
        END IF  
        RETURN truth  
    END FUNCTION  
END CLASS
