CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a list of m plus one elements each being a list of n plus one elements each set to negative infinity  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO the element at position i minus one of nums1 MULTIPLIED BY the element at position j minus one of nums2  
                SET dp at position i at position j TO the maximum among current_product the element at position i minus one at position j of dp the element at position i at position j minus one of dp and the sum of the element at position i minus one at position j minus one of dp PLUS current_product  
            END FOR  
        END FOR  
        RETURN the element at position m at position n of dp  
    END FUNCTION  
END CLASS
