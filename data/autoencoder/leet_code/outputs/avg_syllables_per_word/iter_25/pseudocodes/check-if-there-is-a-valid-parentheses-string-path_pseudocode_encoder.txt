CLASS Solution  
    FUNCTION hasValidPath WITH PARAMETER grid  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF the remainder when m PLUS n MINUS one is DIVIDED BY two is NOT EQUAL TO zero THEN  
            RETURN false  
        END IF  
        SET dp TO a three dimensional list of dimensions m PLUS one by n PLUS one by m PLUS n PLUS one filled with false values  
        SET element at position zero of the element at position one of dp TO true  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF the element at position k of the element at position j PLUS one of the element at position i of dp OR the element at position k of the element at position j of the element at position i PLUS one of dp is true THEN  
                        IF the element at position i of the element at position j of grid EQUALS an opening parenthesis AND k PLUS one IS LESS THAN m PLUS n PLUS one THEN  
                            SET the element at position k PLUS one of the element at position j PLUS one of the element at position i PLUS one of dp TO true  
                        ELSE IF the element at position i of the element at position j of grid EQUALS a closing parenthesis AND k MINUS one IS GREATER THAN OR EQUAL TO zero THEN  
                            SET the element at position k MINUS one of the element at position j PLUS one of the element at position i PLUS one of dp TO true  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position zero of the element at position n of the element at position m of dp  
    END FUNCTION  
END CLASS
