CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as the element at position zero of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE element from the left end of queue and assign it to node  
        IF i is less than the length of values AND element at position i of values is not None  
            SET node.left TO a new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND element at position i of values is not None  
            SET node.right TO a new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with arguments p.left and q.left AND the result of calling is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        END IF  
        CREATE an empty list named result  
        CREATE a double ended queue named queue containing root as its only element  
        SET left_to_right TO True  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            CREATE an empty double ended queue named current_level  
            FOR each index from zero TO level_size MINUS one  
                REMOVE element from the left end of queue and assign it to node  
                IF left_to_right IS True  
                    APPEND node.val TO current_level  
                ELSE  
                    INSERT node.val AT the left end of current_level  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND the list version of current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
