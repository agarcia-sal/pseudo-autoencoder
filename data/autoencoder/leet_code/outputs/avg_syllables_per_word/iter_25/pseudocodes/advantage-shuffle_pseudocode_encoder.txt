CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ASCENDING ORDER  
        SET nums2_with_indices TO the list constructed by FOR each index AND value IN nums2 with their positions AND values construct a pair and then SORT this list IN DESCENDING ORDER BY the first element of each pair  
        SET result TO a list of zero elements WITH THE LENGTH OF nums1  
        SET left TO zero  
        SET right TO the LENGTH OF nums1 MINUS one  
        FOR each pair value AND index IN nums2_with_indices  
            IF element at position right of nums1 IS GREATER THAN value THEN  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
