CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        SET radiant TO an empty double-ended queue  
        SET dire TO an empty double-ended queue  
        FOR each pair of index and senator_character IN the enumeration of senate  
            IF senator_character EQUALS the character R  
                APPEND index TO radiant  
            ELSE  
                APPEND index TO dire  
            END IF  
        END FOR  
        WHILE radiant is not empty AND dire is not empty  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  
        IF radiant is not empty  
            RETURN the string Radiant  
        ELSE  
            RETURN the string Dire  
        END IF  
    END FUNCTION  
END CLASS
