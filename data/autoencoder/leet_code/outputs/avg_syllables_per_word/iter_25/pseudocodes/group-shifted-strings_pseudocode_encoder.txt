CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the Unicode code point value of character c MINUS the Unicode code point value of the first character of s  
                SET wrapped_difference TO difference MODULO twenty six  
                APPEND the string representation of wrapped_difference TO canonical_form  
            END FOR  
            SET joined_form TO the concatenation of elements in canonical_form separated by comma  
            RETURN joined_form  
        END FUNCTION  
        SET groups TO an empty mapping from keys to lists initialized by default as empty lists  
        FOR each string s IN strings  
            SET canonical_form TO the result of calling get_canonical_form on s  
            APPEND s TO the list at groups with key canonical_form  
        END FOR  
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
