CLASS TreeNode  
    FUNCTION __init__ with parameters self val default zero left default none right default none  
        SET the attribute val of self TO val  
        SET the attribute left of self TO left  
        SET the attribute right of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element obtained by removing the first element from queue  
        IF i is less than the length of values AND element at position i of values is not none  
            SET the attribute left of node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND element at position i of values is not none  
            SET the attribute right of node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the attribute val of p is not equal to the attribute val of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments the attribute left of p and the attribute left of q AND the result of is_same_tree with arguments the attribute right of p and the attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode with parameters self root key  
        IF root is none  
            RETURN none  
        END IF  
        IF the attribute val of root is greater than key  
            SET the attribute left of root TO the result of calling deleteNode of self with arguments the attribute left of root and key  
            RETURN root  
        END IF  
        IF the attribute val of root is less than key  
            SET the attribute right of root TO the result of calling deleteNode of self with arguments the attribute right of root and key  
            RETURN root  
        END IF  
        IF the attribute left of root is none  
            RETURN the attribute right of root  
        END IF  
        IF the attribute right of root is none  
            RETURN the attribute left of root  
        END IF  
        SET node TO the attribute right of root  
        WHILE the attribute left of node exists  
            SET node TO the attribute left of node  
        END WHILE  
        SET the attribute left of node TO the attribute left of root  
        SET root TO the attribute right of root  
        RETURN root  
    END FUNCTION  
END CLASS
