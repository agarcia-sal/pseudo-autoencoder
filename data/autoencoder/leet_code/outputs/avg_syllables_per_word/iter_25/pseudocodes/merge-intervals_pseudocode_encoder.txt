CLASS Solution  
    FUNCTION merge WITH parameters intervals  
        SORT intervals BY the first element of each interval  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the second element of the last element in merged IS LESS THAN the first element of interval  
                APPEND interval TO merged  
            ELSE  
                SET the second element of the last element in merged TO the greater value between the second element of the last element in merged AND the second element of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
