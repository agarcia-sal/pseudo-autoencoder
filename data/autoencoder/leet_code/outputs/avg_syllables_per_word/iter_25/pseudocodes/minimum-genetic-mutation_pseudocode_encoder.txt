CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the set constructed from the elements of bank  
        SET q TO a double ended queue containing a single element which is the pair consisting of start and zero  
        SET mp TO a mapping from the character A to the sequence of characters T C G and from the character T to the sequence of characters A C G and from the character C to the sequence of characters A T G and from the character G to the sequence of characters A T C  
        WHILE the queue q is not empty  
            SET t and step to the result of removing the element from the left end of q  
            IF t is equal to end  
                RETURN step  
            END IF  
            FOR each index i and character v enumerated from the elements of t  
                FOR each character j in the sequence mapped from v in mp  
                    SET next TO the concatenation of the substring of t from the start up to but not including position i plus the character j plus the substring of t from position i plus one to the end  
                    IF next is an element of the set s  
                        APPEND the pair consisting of next and step incremented by one to the right end of the queue q  
                        REMOVE next from the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
