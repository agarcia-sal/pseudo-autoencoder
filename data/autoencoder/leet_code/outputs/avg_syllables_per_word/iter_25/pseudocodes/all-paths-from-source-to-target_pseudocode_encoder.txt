CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        ASSIGN q TO a new double ended queue containing a list with zero as its single element  
        ASSIGN ans TO an empty list  
        WHILE q is not empty  
            SET path TO the element removed from the left side of q  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each element v IN the element at position u of graph  
                APPEND the result of concatenating path with a list containing v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
