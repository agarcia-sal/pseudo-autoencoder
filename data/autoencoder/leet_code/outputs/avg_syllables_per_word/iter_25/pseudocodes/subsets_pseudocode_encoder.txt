CLASS Solution  
    FUNCTION subsets WITH PARAMETER nums  
        IF nums DOES NOT CONTAIN ANY ELEMENT  
            RETURN a list containing a single element which is an empty list  
        END IF  
        SET subsets_without_first TO the result of calling subsets METHOD on the object with the list of elements of nums from the second element to the last element  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            APPEND a new list consisting of the first element of nums FOLLOWED BY all elements of subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first WITH subsets_with_first  
    END FUNCTION  
END CLASS
