CLASS TreeNode  
    FUNCTION __init__ with parameters val SET TO zero left SET TO none right SET TO none  
        SET the val attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the val attribute of p is not equal to the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left attribute of p and the left attribute of q AND calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues with parameters root  
        IF root is none  
            RETURN an empty list  
        END IF  
        SET queue TO a deque initialized with root  
        SET largest_values TO an empty list  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            SET max_value TO negative infinity  
            FOR a variable from zero TO level_size minus one  
                SET node TO the element removed from the left of queue  
                SET max_value TO the greater of max_value and the val attribute of node  
                IF the left attribute of node is not none  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node is not none  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
