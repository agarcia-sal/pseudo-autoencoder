CLASS Solution  
    FUNCTION findLUSlength WITH PARAMETER strs  
        FUNCTION is_subsequence WITH PARAMETERS s AND t  
            ASSIGN iterator TO an iterator over t  
            RETURN the evaluation of whether every character in s appears in the iterator in order  
        END FUNCTION  
        
        SORT the list strs by descending length of each element and then lexicographically ascending order  
        
        FOR each index i and corresponding word in strs  
            SET unique TO true  
            FOR each index j and corresponding other word in strs  
                IF i is not equal to j AND the function is_subsequence returns true for s equal to word and t equal to other  
                    SET unique TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique is true  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN the negative of one  
    END FUNCTION  
END CLASS
