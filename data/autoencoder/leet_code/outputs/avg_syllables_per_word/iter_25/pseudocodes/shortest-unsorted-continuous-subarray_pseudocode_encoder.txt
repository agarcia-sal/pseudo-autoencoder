CLASS Solution  
    FUNCTION findUnsortedSubarray WITH PARAMETER nums  
        SET sorted_nums TO the array nums arranged in ascending order  
        SET start TO negative one  
        SET end TO negative two  
        FOR each index i FROM zero TO the length of nums MINUS one  
            IF the element at position i of nums IS NOT EQUAL TO the element at position i of sorted_nums  
                IF start IS EQUAL TO negative one  
                    SET start TO i  
                END IF  
                SET end TO i  
            END IF  
        END FOR  
        RETURN the result of subtracting start from end THEN ADDING one  
    END FUNCTION  
END CLASS
