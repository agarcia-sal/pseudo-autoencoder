CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the list nums is empty THEN  
            RETURN zero  
        END IF  
        SET max_product TO element at position zero of nums  
        SET min_product TO element at position zero of nums  
        SET result TO element at position zero of nums  
        FOR each num in the list of elements from position one to the end of nums  
            IF num is less than zero THEN  
                SET temporary variable temp TO max_product  
                SET max_product TO min_product  
                SET min_product TO temp  
            END IF  
            SET max_product TO the greater value between num and max_product multiplied by num  
            SET min_product TO the lesser value between num and min_product multiplied by num  
            SET result TO the greater value between result and max_product  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
