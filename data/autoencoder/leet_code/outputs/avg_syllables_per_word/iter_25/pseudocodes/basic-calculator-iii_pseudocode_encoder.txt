CLASS Solution  
    FUNCTION calculate WITH parameter s  
        FUNCTION helper WITH parameter s  
            SET stack TO an empty list  
            SET num TO zero  
            SET sign TO PLUS  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the element at position zero REMOVED FROM s  
                IF char is a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS open parenthesis THEN  
                    SET num TO the result of calling helper WITH s  
                END IF  
                IF char is one of the characters PLUS MINUS MULTIPLIED BY DIVIDED BY open parenthesis close parenthesis OR the LENGTH OF s EQUALS zero THEN  
                    IF sign EQUALS PLUS THEN  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS MINUS THEN  
                        APPEND minus num TO stack  
                    ELSE IF sign EQUALS MULTIPLIED BY THEN  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS DIVIDED BY THEN  
                        SET the last element of stack TO the integer value of the last element of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis THEN  
                    BREAK  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        SET s TO the list of characters from s  
        RETURN the result of calling helper WITH s  
    END FUNCTION  
END CLASS
