CLASS Solution  
    FUNCTION uniquePathsIII WITH PARAMETERS grid  
        FUNCTION dfs WITH PARAMETERS x y steps  
            IF element at position x of grid AND element at position y of element at position x of grid EQUALS two THEN  
                RETURN steps EQUALS zero  
            END IF  
            SET original TO element at position x of grid AND element at position y of element at position x of grid  
            SET element at position x of grid AND element at position y of element at position x of grid TO minus two  
            SET count TO zero  
            FOR each pair dx dy IN list of pairs minus one zero PLUS one zero zero minus one zero PLUS zero one  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of grid AND element at position ny of element at position nx of grid GREATER THAN OR EQUAL TO zero THEN  
                    INCREMENT count BY dfs WITH PARAMETERS nx ny steps MINUS one  
                END IF  
            END FOR  
            SET element at position x of grid AND element at position y of element at position x of grid TO original  
            RETURN count  
        END FUNCTION  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start_x TO zero  
        SET start_y TO zero  
        SET empty_squares TO one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid AND element at position j of element at position i of grid EQUALS one THEN  
                    SET start_x TO i  
                    SET start_y TO j  
                ELSE IF element at position i of grid AND element at position j of element at position i of grid EQUALS zero THEN  
                    INCREMENT empty_squares BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN dfs WITH PARAMETERS start_x start_y empty_squares  
    END FUNCTION  
END CLASS
