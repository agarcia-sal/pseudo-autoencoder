CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET the val attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the values list is empty THEN  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode constructed with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one TO the end  
        SET node TO a new instance of ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none THEN  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none THEN  
        RETURN false  
    END IF  
    RETURN the expression that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with parameters the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS self lists  
        SET min_heap TO an empty list  
        FOR each index i AND element lst IN the enumeration of lists  
            IF lst IS NOT none THEN  
                EXECUTE the action of pushing a tuple composed of the val attribute of lst the value i AND lst itself INTO min_heap maintaining its min-heap property  
            END IF  
        END FOR  
        SET dummy TO a new instance of ListNode constructed with no arguments  
        SET current TO dummy  
        WHILE min_heap IS NOT empty DO  
            EXTRACT the smallest tuple from min_heap and DECOMPOSE it INTO val i node  
            SET the next attribute of current TO node  
            SET current TO the next attribute of current  
            IF the next attribute of node IS NOT none THEN  
                EXECUTE the action of pushing a tuple composed of the val attribute of the next attribute of node the value i AND the next attribute of node INTO min_heap maintaining its min-heap property  
            END IF  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
