CLASS Solution  
    FUNCTION minEatingSpeed WITH PARAMETERS piles AND h  
        FUNCTION canFinish WITH PARAMETER k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the CEILING OF the pile DIVIDED BY k  
            END FOR  
            RETURN whether hours IS LESS THAN OR EQUAL TO h  
        END FUNCTION  
        SET left TO one  
        SET right TO the MAXIMUM ELEMENT IN piles  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF the SUM OF left AND right BY two  
            IF canFinish WITH ARGUMENT mid IS TRUE  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
