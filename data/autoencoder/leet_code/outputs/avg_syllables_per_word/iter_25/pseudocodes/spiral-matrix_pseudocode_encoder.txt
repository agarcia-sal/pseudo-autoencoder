CLASS Solution  
    FUNCTION spiralOrder WITH parameter matrix  
        IF matrix IS empty OR element at position zero of matrix IS empty  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET top TO zero  
        SET bottom TO the LENGTH OF matrix MINUS one  
        SET left TO zero  
        SET right TO the LENGTH OF element at position zero of matrix MINUS one  
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                APPEND element at position i of element at position top of matrix TO result  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                APPEND element at position right of element at position i of matrix TO result  
            END FOR  
            DECREMENT right BY one  
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left  
                    APPEND element at position i of element at position bottom of matrix TO result  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top  
                    APPEND element at position left of element at position i of matrix TO result  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
