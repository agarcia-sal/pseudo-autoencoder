CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero left SET TO none right SET TO none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY THEN  
        RETURN none  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY DO  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUAL TO the val ATTRIBUTE OF q THEN  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND THE RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves WITH PARAMETERS self root  
        FUNCTION collectLeaves WITH PARAMETER node  
            IF node IS none THEN  
                RETURN negative one  
            END IF  
            SET left_height TO THE RESULT OF collectLeaves WITH PARAMETER the left ATTRIBUTE OF node  
            SET right_height TO THE RESULT OF collectLeaves WITH PARAMETER the right ATTRIBUTE OF node  
            SET current_height TO THE GREATER VALUE BETWEEN left_height AND right_height PLUS one  
            IF current_height IS GREATER THAN OR EQUAL TO THE LENGTH OF result THEN  
                APPEND AN EMPTY LIST TO result  
            END IF  
            APPEND the val ATTRIBUTE OF node TO THE LIST AT POSITION current_height OF result  
            RETURN current_height  
        END FUNCTION  
        SET result TO AN EMPTY LIST  
        CALL collectLeaves WITH PARAMETER root  
        RETURN result  
    END FUNCTION  
END CLASS
