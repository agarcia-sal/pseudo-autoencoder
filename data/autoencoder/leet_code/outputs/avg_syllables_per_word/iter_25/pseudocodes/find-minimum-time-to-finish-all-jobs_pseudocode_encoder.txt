CLASS Solution  
    FUNCTION minimumTimeRequired WITH PARAMETERS jobs AND k  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zero repeated two RAISED TO THE POWER OF n times  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET element at position bitwise OR of mask AND two RAISED TO THE POWER OF i IN subset_sums TO element at position mask of subset_sums PLUS element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp WITH PARAMETERS k AND mask  
            IF k EQUALS one THEN  
                RETURN element at position mask of subset_sums  
            END IF  
            SET res TO infinite positive value  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET candidate TO the maximum of dp WITH PARAMETERS k MINUS one AND bitwise XOR of mask AND submask AND element at position submask of subset_sums  
                IF candidate LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO bitwise AND of submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp WITH PARAMETERS k AND two RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
