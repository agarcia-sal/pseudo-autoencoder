CLASS Solution  
    FUNCTION checkInclusion WITH PARAMETERS s1 AND s2  
        ASSIGN reference TO the class used for counting elements named Counter  
        SET len_s1 TO the length of s1  
        SET len_s2 TO the length of s2  
        IF len_s1 IS GREATER THAN len_s2  
            RETURN false  
        END IF  
        SET count_s1 TO the count of each character in s1 using Counter  
        SET count_s2 TO the count of each character in the substring from position zero TO position len_s1 MINUS one of s2 using Counter  
        IF count_s1 IS EQUAL TO count_s2  
            RETURN true  
        END IF  
        FOR variable i FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of the character at position i of s2 in count_s2 BY one  
            DECREMENT the count of the character at position i MINUS len_s1 of s2 in count_s2 BY one  
            IF the count of the character at position i MINUS len_s1 of s2 in count_s2 IS EQUAL TO zero  
                REMOVE the entry for the character at position i MINUS len_s1 of s2 from count_s2  
            END IF  
            IF count_s1 IS EQUAL TO count_s2  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
