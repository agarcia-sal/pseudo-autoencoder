CLASS Solution  
    FUNCTION minimumOneBitOperations WITH PARAMETER n  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one SHIFTED LEFT BY k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one SHIFTED LEFT BY k  
        RETURN one SHIFTED LEFT BY k PLUS one MINUS one MINUS CALL minimumOneBitOperations WITH ARGUMENT n EXCLUSIVE OR most_significant_bit ON THIS OBJECT  
    END FUNCTION  
END CLASS
