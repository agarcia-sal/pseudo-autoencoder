CLASS Solution  
    FUNCTION soupServings WITH parameter n  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        SET n TO n PLUS twenty four DIVIDED BY twenty five using integer division  
        SET memo TO an empty collection  
        FUNCTION dp WITH parameters a b  
            IF the pair of a and b EXISTS IN memo  
                RETURN the value from memo identified by the pair of a and b  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY the sum of dp called with a MINUS four and b PLUS dp called with a MINUS three and b MINUS one PLUS dp called with a MINUS two and b MINUS two PLUS dp called with a MINUS one and b MINUS three  
            ASSIGN prob TO memo identified by the pair of a and b  
            RETURN prob  
        END FUNCTION  
        RETURN dp called with n and n  
    END FUNCTION  
END CLASS
