CLASS Solution  
    FUNCTION isBipartite WITH PARAMETERS graph  
        SET n TO the LENGTH OF graph  
        SET color TO a list of zeros with length n  
        
        FUNCTION dfs WITH PARAMETERS node c  
            SET the element at position node of color TO c  
            FOR each neighbor IN the element at position node of graph  
                IF the element at position neighbor of color EQUALS c THEN  
                    RETURN false  
                END IF  
                IF the element at position neighbor of color EQUALS zero THEN  
                    IF the result of dfs CALLED WITH neighbor AND the NEGATION OF c IS NOT true THEN  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of color EQUALS zero THEN  
                IF the result of dfs CALLED WITH i AND one IS NOT true THEN  
                    RETURN false  
                END IF  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
