CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n k maxPts  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN the INTEGER VALUE OF the CONDITION i LESS THAN OR EQUAL TO n  
            END IF  
            IF i EQUALS k MINUS one THEN  
                RETURN the MINIMUM of the QUANTITY n MINUS k PLUS one AND maxPts DIVIDED BY maxPts  
            END IF  
            RETURN dfs of i PLUS one PLUS the DIFFERENCE of dfs of i PLUS one AND dfs of i PLUS maxPts PLUS one DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs of zero  
    END FUNCTION  
END CLASS
