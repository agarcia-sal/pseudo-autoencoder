CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS None AND right EQUALS None  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the values list IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    CREATE an empty double-ended queue named queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE the element from the left side of queue and ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the left attribute of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the right attribute of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value attribute of p IS NOT EQUAL TO the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob WITH PARAMETER root AS Optional TreeNode  
        FUNCTION rob_from_node WITH PARAMETER node  
            IF node IS None  
                RETURN zero AND zero  
            END IF  
            SET left_robbed AND left_not_robbed TO the two return values from calling rob_from_node with the left attribute of node  
            SET right_robbed AND right_not_robbed TO the two return values from calling rob_from_node with the right attribute of node  
            SET rob_this_node TO the value attribute of node PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the sum of the maximum of left_robbed AND left_not_robbed AND the maximum of right_robbed AND right_not_robbed  
            RETURN rob_this_node AND not_rob_this_node  
        END FUNCTION  
        RETURN the maximum of the two values returned by rob_from_node called with root  
    END FUNCTION  
END CLASS
