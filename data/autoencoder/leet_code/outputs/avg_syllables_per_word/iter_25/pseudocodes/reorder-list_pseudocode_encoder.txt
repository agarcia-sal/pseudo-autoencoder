CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS None  
        SET the value attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a new ListNode CREATED WITH the first element of values  
    SET p TO head  
    FOR each val IN values FROM the second element TO the last element  
        SET node TO a new ListNode CREATED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND the RESULT of is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETERS self head  
        IF head EQUALS None OR the next attribute of head EQUALS None  
            RETURN  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT None AND the next attribute of fast IS NOT None  
            SET slow TO the next attribute of slow  
            SET fast TO the next attribute of the next attribute of fast  
        END WHILE  
        SET prev TO None  
        SET curr TO slow  
        WHILE curr IS NOT None  
            SET next_temp TO the next attribute of curr  
            SET the next attribute of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE the next attribute of second IS NOT None  
            SET temp1 TO the next attribute of first  
            SET temp2 TO the next attribute of second  
            SET the next attribute of first TO second  
            SET the next attribute of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
