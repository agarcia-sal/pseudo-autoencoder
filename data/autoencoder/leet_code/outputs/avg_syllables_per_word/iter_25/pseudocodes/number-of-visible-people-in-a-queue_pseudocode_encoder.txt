CLASS Solution  
    FUNCTION canSeePersonsCount WITH PARAMETER heights  
        SET n TO the LENGTH OF heights  
        SET answer TO a LIST of zeroes repeated n times  
        SET stack TO an empty LIST  
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the ELEMENT at position last element of stack of heights IS LESS THAN the ELEMENT at position i of heights  
                REMOVE the last element FROM stack AND STORE it in temporary variable popped_index  
                INCREMENT the ELEMENT at position popped_index of answer BY one  
            END WHILE  
            IF the LENGTH OF stack IS GREATER THAN zero  
                INCREMENT the ELEMENT at position last element of stack of answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
