CLASS Solution  
    FUNCTION removeDuplicates WITH PARAMETER nums  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
        SET write_index TO zero  
        FOR i FROM zero TO THE MINIMUM OF two AND THE LENGTH OF nums MINUS one  
            SET element at position write_index of nums TO element at position i of nums  
            INCREMENT write_index BY one  
        END FOR  
        FOR i FROM two TO THE LENGTH OF nums MINUS one  
            IF element at position i of nums IS NOT EQUAL TO element at position write_index MINUS two of nums  
                SET element at position write_index of nums TO element at position i of nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        RETURN write_index  
    END FUNCTION  
END CLASS
