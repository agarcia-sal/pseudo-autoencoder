CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero left SET TO none right SET TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater than zero  
        SET node TO the result of removing the element from the left end of queue  
        IF i IS less than the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO new instance of TreeNode with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO new instance of TreeNode with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with p dot left and q dot left AND the result of is_same_tree with p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum WITH PARAMETERS self root  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN negative infinity AND negative infinity  
            END IF  
            SET left_max AND left_gain TO the result of helper with node dot left  
            SET right_max AND right_gain TO the result of helper with node dot right  
            SET current_max TO node dot val PLUS the maximum of left_gain AND zero PLUS the maximum of right_gain AND zero  
            SET current_gain TO node dot val PLUS the maximum of left_gain right_gain AND zero  
            SET global_max TO the maximum of left_max right_max AND current_max  
            RETURN global_max AND current_gain  
        END FUNCTION  
        RETURN the first element of the result of helper with root  
    END FUNCTION  
END CLASS
