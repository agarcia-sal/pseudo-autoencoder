CLASS Solution  
    FUNCTION rearrangeString WITH PARAMETERS s and k  
        IF k LESS THAN OR EQUAL TO one THEN  
            RETURN s  
        END IF  

        SET freq TO the count of each character in s  
        SET max_heap TO a list containing pairs of the negative count and character for each character and its count in freq  
        HEAPIFY max_heap to arrange it as a max heap by frequency  
        SET wait_queue TO an empty list representing characters waiting to be placed again  
        SET result TO an empty list to accumulate the rearranged characters  

        WHILE max_heap is not empty DO  
            REMOVE the element with the highest frequency from max_heap and assign its count and character to count and char  
            APPEND char to result  
            APPEND the pair of count incremented by one and char to wait_queue to update its frequency and mark it waiting  

            IF the length of wait_queue EQUALS k THEN  
                REMOVE the oldest element from wait_queue and assign its count and character to count and char  
                IF count is less than zero THEN  
                    PUSH the pair count and char back into max_heap  
                END IF  
            END IF  
        END WHILE  

        IF the length of result is not equal to the length of s THEN  
            RETURN an empty string  
        END IF  

        RETURN the concatenation of all characters in result as a single string  
    END FUNCTION  
END CLASS
