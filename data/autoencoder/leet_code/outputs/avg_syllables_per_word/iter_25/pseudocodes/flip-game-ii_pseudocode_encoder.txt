CLASS Solution  
    FUNCTION canWin WITH PARAMETER currentState  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper WITH PARAMETER state  
            IF state EXISTS IN memo THEN  
                RETURN the element associated with state IN memo  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position i TO position i PLUS one of state EQUALS the string consisting of two plus symbols THEN  
                    SET new_state TO the concatenation of the substring from position zero TO position i of state PLUS the string consisting of two minus symbols PLUS the substring from position i PLUS two TO the END OF state  
                    IF the NEGATION of the result of can_win_helper WITH ARGUMENT new_state THEN  
                        SET the element associated with state IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the element associated with state IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN the result of can_win_helper WITH ARGUMENT currentState  
    END FUNCTION  
END CLASS
