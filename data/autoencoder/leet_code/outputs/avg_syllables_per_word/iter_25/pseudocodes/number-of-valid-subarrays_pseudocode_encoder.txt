CLASS Solution  
    FUNCTION validSubarrays WITH PARAMETER nums OF TYPE List OF Integer  
        SET stack TO an empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the stack is not empty AND the element at the last position of the stack is GREATER THAN num  
                REMOVE the element at the last position of the stack  
            END WHILE  
            ADD num TO the end of the stack  
            INCREMENT count BY the LENGTH OF the stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
