CLASS Solution  
    FUNCTION maxDistToClosest WITH parameter seats  
        SET max_distance TO zero  
        SET start TO zero  
        SET end TO zero  
        SET n TO the LENGTH OF seats  
        
        WHILE start LESS THAN n AND the element at position start of seats EQUALS zero  
            INCREMENT start BY one  
        END WHILE  
        
        SET end TO n MINUS one  
        WHILE end GREATER THAN OR EQUAL TO zero AND the element at position end of seats EQUALS zero  
            DECREMENT end BY one  
        END WHILE  
        
        IF max_distance LESS THAN start  
            SET max_distance TO start  
        END IF  
        
        IF max_distance LESS THAN n MINUS end MINUS one  
            SET max_distance TO n MINUS end MINUS one  
        END IF  
        
        SET current_distance TO zero  
        FOR i FROM start TO end  
            IF the element at position i of seats EQUALS zero  
                INCREMENT current_distance BY one  
                IF max_distance LESS THAN the INTEGER DIVISION of current_distance PLUS one BY two  
                    SET max_distance TO the INTEGER DIVISION of current_distance PLUS one BY two  
                END IF  
            ELSE  
                SET current_distance TO zero  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
