CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        SET dp TO a list containing n elements each being an empty mapping that returns zero for any missing key  
        SET total_count TO zero  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                IF the mapping at position j of dp contains the key d  
                    INCREMENT the mapping at position i of dp at key d BY the value at mapping position j of dp at key d PLUS one  
                    INCREMENT total_count BY the value at mapping position j of dp at key d  
                ELSE  
                    INCREMENT the mapping at position i of dp at key d BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
