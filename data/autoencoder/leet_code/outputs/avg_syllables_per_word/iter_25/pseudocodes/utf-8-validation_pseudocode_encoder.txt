CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the result of shifting byte to the right by seven bits EQUALS zero  
                RETURN one  
            ELSE IF the result of shifting byte to the right by five bits EQUALS one one zero in binary form  
                RETURN two  
            ELSE IF the result of shifting byte to the right by four bits EQUALS one one one zero in binary form  
                RETURN three  
            ELSE IF the result of shifting byte to the right by three bits EQUALS one one one one zero in binary form  
                RETURN four  
            ELSE  
                RETURN minus one  
            END IF  
        END FUNCTION  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH of data  
            SET byte_count TO the result of calling get_byte_count with element at position i of data  
            IF byte_count EQUALS minus one  
                RETURN false  
            END IF  
            IF the sum of i and byte_count GREATER THAN the LENGTH of data  
                RETURN false  
            END IF  
            FOR j FROM i PLUS one TO the sum of i and byte_count MINUS one  
                IF the result of shifting element at position j of data to the right by six bits NOT EQUALS one zero in binary form  
                    RETURN false  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN true  
    END FUNCTION  
END CLASS
