CLASS Solution  
    FUNCTION maxTaskAssign(tasks, workers, pills, strength)  
        PERFORM sorting the list tasks in ascending order  
        PERFORM sorting the list workers in ascending order  
          
        FUNCTION canAssignTasks(k)  
            SET available_workers TO the last k elements of workers  
            SET pills_used TO zero  
              
            FOR index FROM k MINUS one DOWN TO zero BY decrement of one  
                SET task TO the element at position index of tasks  
                SET worker_idx TO the leftmost position to insert task in available_workers preserving order  
                IF worker_idx is less than the length of available_workers THEN  
                    REMOVE the element at position worker_idx from available_workers  
                ELSE IF pills_used is less than pills THEN  
                    SET worker_idx TO the leftmost position to insert task MINUS strength in available_workers preserving order  
                    IF worker_idx is less than the length of available_workers THEN  
                        REMOVE the element at position worker_idx from available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN False  
                    END IF  
                ELSE  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
          
        SET left TO zero  
        SET right TO the minimum of the length of tasks and the length of workers PLUS one  
        WHILE left is less than right  
            SET mid TO the integer division of left PLUS right by two  
            IF canAssignTasks(mid) THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
