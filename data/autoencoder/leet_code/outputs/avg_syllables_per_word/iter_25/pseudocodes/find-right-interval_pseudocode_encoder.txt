CLASS Solution  
    FUNCTION findRightInterval WITH intervals  
        SET start_times TO a list of pairs each consisting of the first element of interval and its corresponding index in intervals sorted by the first element of interval  
        SET result TO an empty list  
        FOR each element ignored_and_end where ignored_and_end is each element in intervals in order  
            SET end TO element at position one of ignored_and_end  
            SET idx TO the position found by binary searching in start_times for a pair where the first element is greater than or equal to end  
            IF idx is less than the length of start_times THEN  
                APPEND the second element of the pair at position idx in start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
