CLASS Solution  
    FUNCTION kMirror(k INTEGER, n INTEGER) RETURNS INTEGER  
        FUNCTION is_palindrome(s STRING) RETURNS BOOLEAN  
            RETURN s IS EQUAL TO the REVERSE OF s  
        END FUNCTION  
        
        FUNCTION to_base_k(num INTEGER, k INTEGER) RETURNS STRING  
            IF num IS EQUAL TO zero THEN  
                RETURN the STRING representing zero  
            END IF  
            SET digits TO an empty list  
            WHILE num IS NOT equal to zero DO  
                APPEND the INTEGER remainder of num DIVIDED BY k TO digits  
                SET num TO the INTEGER result of num DIVIDED BY k  
            END WHILE  
            SET reversed_digits TO digits IN REVERSE ORDER  
            SET result_string TO the CONCATENATION of all elements of reversed_digits CONVERTED TO STRING without separator  
            RETURN result_string  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length INTEGER)  
            IF length IS EQUAL TO one THEN  
                FOR i FROM one TO nine DO  
                    YIELD the STRING representing i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF INTEGER result of LENGTH MINUS one DIVIDED BY two  
                SET end TO ten RAISED TO THE POWER OF INTEGER result of LENGTH PLUS one DIVIDED BY two  
                FOR half FROM start TO end MINUS one DO  
                    SET half_str TO the STRING representation of half  
                    IF length MODULO two IS EQUAL TO zero THEN  
                        SET full TO CONCATENATION of half_str AND the REVERSE OF half_str  
                    ELSE  
                        SET full TO CONCATENATION of half_str AND the SUBSTRING OF half_str FROM the SECOND LAST CHARACTER TO THE FIRST CHARACTER IN REVERSE ORDER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n DO  
            FOR each pal IN generate_palindromes(length) DO  
                SET num TO the INTEGER value of pal  
                IF is_palindrome(to_base_k(num, k)) IS TRUE THEN  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers IS EQUAL TO n THEN  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        SET total_sum TO zero  
        FOR each number IN k_mirror_numbers DO  
            INCREMENT total_sum BY number  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
