CLASS Solution  
    FUNCTION removeDuplicateLetters WITH PARAMETER s  
        SET last_occurrence TO an empty dictionary  
        FOR each index and character IN enumeration of s  
            SET the value of character in last_occurrence TO index  
        END FOR  
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index and character IN enumeration of s  
            IF character is contained in in_stack  
                CONTINUE  
            END IF  
            
            WHILE the LENGTH OF stack is greater than zero AND character is LESS THAN element at the last position of stack AND index is LESS THAN the value in last_occurrence for element at the last position of stack  
                SET removed_char TO the element popped from the end of stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result TO concatenation of all elements in stack in order  
        RETURN result  
    END FUNCTION  
END CLASS
