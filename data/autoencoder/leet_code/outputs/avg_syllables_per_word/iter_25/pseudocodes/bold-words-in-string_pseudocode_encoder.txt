CLASS Solution  
    FUNCTION boldWords(words AS list of strings COMMA s AS string) RETURNS string  
        SET bold TO a list of boolean false values with LENGTH EQUAL TO the LENGTH OF s  
        FOR each word IN words  
            SET start TO the position of the first occurrence of word in s  
            WHILE start IS NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of bold TO true  
                END FOR  
                SET start TO the position of the next occurrence of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF the element at position i of bold IS true  
                APPEND the string the less than character B greater than character TO result  
                WHILE i LESS THAN the LENGTH OF s AND the element at position i of bold IS true  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string less than character slash B greater than character TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
