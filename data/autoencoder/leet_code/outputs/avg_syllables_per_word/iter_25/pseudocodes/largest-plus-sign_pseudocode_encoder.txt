CLASS Solution  
    FUNCTION orderOfLargestPlusSign WITH PARAMETERS n AND mines  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            APPEND tuple consisting of elements of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of size n by n where each element is a list of four zeros  
        
        SET directions TO the list containing the pairs zero PLUS one zero COMMA one PLUS zero COMMA zero MINUS one COMMA MINUS one PLUS zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair i COMMA j is NOT IN mines_set THEN  
                    IF j IS GREATER THAN zero THEN  
                        SET the element at position zero of the list at position j of the list at position i of dp TO the element at position zero of the list at position j MINUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero THEN  
                        SET the element at position one of the list at position j of the list at position i of dp TO the element at position one of the list at position j of the list at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair i COMMA j is NOT IN mines_set THEN  
                    IF j IS LESS THAN n MINUS one THEN  
                        SET the element at position two of the list at position j of the list at position i of dp TO the element at position two of the list at position j PLUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one THEN  
                        SET the element at position three of the list at position j of the list at position i of dp TO the element at position three of the list at position j of the list at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the four elements in the list at position j of the list at position i of dp  
                    SET result TO the maximum of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
