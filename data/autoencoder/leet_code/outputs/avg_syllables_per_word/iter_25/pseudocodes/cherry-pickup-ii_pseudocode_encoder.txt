CLASS Solution  
    FUNCTION cherryPickup WITH PARAMETER grid  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        
        FUNCTION dp WITH PARAMETERS r c1 c2  
            IF c1 LESS THAN zero OR c1 GREATER THAN OR EQUAL TO cols OR c2 LESS THAN zero OR c2 GREATER THAN OR EQUAL TO cols THEN  
                RETURN zero  
            END IF  
            
            IF c1 EQUALS c2 THEN  
                SET cherries TO element at position r of grid AT position c1  
            ELSE  
                SET cherries TO element at position r of grid AT position c1 PLUS element at position r of grid AT position c2  
            END IF  
            
            IF r EQUALS rows MINUS one THEN  
                RETURN cherries  
            END IF  
            
            SET max_cherries TO zero  
            FOR each dc1 IN the set consisting of minus one zero and one  
                FOR each dc2 IN the set consisting of minus one zero and one  
                    SET candidate TO dp OF r PLUS one c1 PLUS dc1 c2 PLUS dc2  
                    IF candidate GREATER THAN max_cherries THEN  
                        SET max_cherries TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN cherries PLUS max_cherries  
        END FUNCTION  
        
        RETURN dp OF zero zero cols MINUS one  
    END FUNCTION  
END CLASS
