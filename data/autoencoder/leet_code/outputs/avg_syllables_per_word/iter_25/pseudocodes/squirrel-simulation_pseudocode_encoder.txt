CLASS Solution  
    FUNCTION minDistance(height width tree squirrel nuts)  
        FUNCTION distance(p1 p2)  
            SET difference_in_first_coordinate TO the absolute value of element at position zero of p1 MINUS element at position zero of p2  
            SET difference_in_second_coordinate TO the absolute value of element at position one of p1 MINUS element at position one of p2  
            RETURN difference_in_first_coordinate PLUS difference_in_second_coordinate  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY the result of calling distance with nut and tree as arguments  
            SET first_nut_distance TO the result of calling distance with squirrel and nut as arguments PLUS the result of calling distance with nut and tree as arguments  
            SET later_nut_distance TO two MULTIPLIED BY the result of calling distance with nut and tree as arguments  
            SET save TO later_nut_distance MINUS first_nut_distance  
            SET max_save TO the greater of max_save AND save  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
