CLASS Solution  
    FUNCTION findMaxLength WITH PARAMETER nums  
        SET prefix_sum_index TO a dictionary mapping zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS AS a key IN prefix_sum_index  
                SET max_length TO the greater of max_length AND the subtraction of the element at position i of enumeration OF nums MINUS the value mapped by count IN prefix_sum_index  
            ELSE  
                ASSIGN the value i TO the key count IN prefix_sum_index  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
