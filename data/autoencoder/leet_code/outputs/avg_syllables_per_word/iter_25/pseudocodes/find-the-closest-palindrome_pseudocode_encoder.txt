CLASS Solution  
    FUNCTION nearestPalindromic WITH PARAMETER n  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER VALUE OF n MINUS one  
        END IF  
        SET candidates TO an EMPTY SET  
        ADD ten RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD ten RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        SET prefix TO the INTEGER VALUE OF the SUBSTRING OF n FROM POSITION zero TO POSITION the INTEGER VALUE OF the QUOTIENT OF length PLUS one DIVIDED BY two  
        FOR each i FROM minus one TO one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF the REMAINDER OF length DIVIDED BY two EQUALS zero  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF the SUBSTRING OF new_prefix FROM POSITION minus two TO THE BEGINNING  
            END IF  
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF IT EXISTS  
        SET closest TO the ELEMENT OF candidates THAT MINIMIZES FIRST the ABSOLUTE VALUE OF THE DIFFERENCE BETWEEN ELEMENT AND original_number AND SECOND the ELEMENT ITSELF  
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
