CLASS Solution  
    FUNCTION canIWin with parameters maxChoosableInteger and desiredTotal  
        IF maxChoosableInteger MULTIPLIED BY the sum of maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal THEN  
            RETURN False  
        END IF  

        IF desiredTotal LESS THAN OR EQUAL TO zero THEN  
            RETURN True  
        END IF  

        SET memo TO an empty dictionary  

        FUNCTION can_win with parameters current_total and used_numbers  
            IF a pair consisting of current_total and used_numbers is in memo THEN  
                RETURN the value in memo at key of the pair consisting of current_total and used_numbers  
            END IF  

            FOR number FROM one TO maxChoosableInteger DO  
                IF used_numbers AND one SHIFTED LEFT BY number EQUALS zero THEN  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win with current_total PLUS number and used_numbers OR one SHIFTED LEFT BY number THEN  
                        SET memo at key of the pair consisting of current_total and used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET memo at key of the pair consisting of current_total and used_numbers TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_win with zero and zero  
    END FUNCTION  
END CLASS
