CLASS Solution  
    FUNCTION minSwapsCouples WITH PARAMETER row AS list of integers  
        SET position TO an empty dictionary  
        FOR each index i AND person in row enumerated from zero  
            SET position at key person TO i  
        END FOR  
        SET swaps TO zero  
        FOR i FROM zero TO the length of row MINUS one IN STEPS OF two  
            SET correct_pair TO the element at position i of row EXCLUSIVE OR one  
            IF the element at position i PLUS one of row DOES NOT EQUAL correct_pair THEN  
                SET swap_idx TO the value of position at key correct_pair  
                SET temporary variable temp TO the element at position i PLUS one of row  
                SET the element at position i PLUS one of row TO the element at position swap_idx of row  
                SET the element at position swap_idx of row TO temp  
                SET position at key equal to the element at position swap_idx of row TO swap_idx  
                SET position at key equal to the element at position i PLUS one of row TO i PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
