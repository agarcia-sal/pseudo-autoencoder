CLASS Solution  
    FUNCTION isAdditiveNumber WITH PARAMETER num  
        FUNCTION is_valid_sequence WITH PARAMETERS start first second  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            SET expected TO the string representation of first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring of num from start TO start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            RETURN the result of calling is_valid_sequence WITH start PLUS length second and the integer conversion of expected  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the substring of num from the beginning TO i  
                SET second TO the substring of num from i TO j  
                IF the LENGTH OF first GREATER THAN one AND the first character of first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the first character of second EQUALS zero  
                    CONTINUE  
                END IF  
                IF the result of calling is_valid_sequence WITH j the integer conversion of first and the integer conversion of second  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
