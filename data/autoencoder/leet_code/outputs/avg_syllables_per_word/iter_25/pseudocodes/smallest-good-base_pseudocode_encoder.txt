CLASS Solution  
    FUNCTION smallestGoodBase WITH PARAMETER n  
        SET n TO the integer form of n  
        SET max_length TO the integer part of the binary logarithm of n PLUS one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer part of n RAISED TO THE POWER OF one DIVIDED BY m MINUS one  
            IF k LESS THAN two  
                CONTINUE to next iteration of loop  
            END IF  
            SET num TO the integer division of k RAISED TO THE POWER OF m MINUS one BY k MINUS one  
            IF num EQUALS n  
                RETURN the string form of k  
            END IF  
        END FOR  
        RETURN the string form of n MINUS one  
    END FUNCTION  
END CLASS
