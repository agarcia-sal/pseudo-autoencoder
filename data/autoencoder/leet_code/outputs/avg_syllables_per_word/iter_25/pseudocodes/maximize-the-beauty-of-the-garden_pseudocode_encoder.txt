CLASS Solution  
    FUNCTION maximumBeauty WITH PARAMETER flowers  
        ASSIGN reference TO a new empty dictionary that associates each key with a new empty list  
        SET prefix_sum TO a new list of zeros with length equal to the length of flowers PLUS one  
        FOR each index i FROM zero TO the length of flowers MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS the greater of element at position i of flowers OR zero  
            APPEND index i TO the list associated with the element at position i of flowers in the dictionary  
        END FOR  
        SET max_beauty TO negative infinity  
        FOR each beauty IN the keys of the dictionary  
            IF the length of the list associated with beauty in the dictionary IS GREATER THAN OR EQUAL TO two  
                SET first TO the element at position zero of the list associated with beauty in the dictionary  
                SET last TO the element at the last position of the list associated with beauty in the dictionary  
                SET current_beauty TO two MULTIPLIED BY beauty PLUS element at position last of prefix_sum MINUS element at position first PLUS one of prefix_sum  
                IF current_beauty IS GREATER THAN max_beauty  
                    SET max_beauty TO current_beauty  
                END IF  
            END IF  
        END FOR  
        RETURN max_beauty  
    END FUNCTION  
END CLASS
