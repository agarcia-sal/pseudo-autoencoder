CLASS Solution  
    FUNCTION reorganizeString WITH PARAMETER s  
        SET count TO the frequency count of each character in s  
        SET max_heap TO an empty list  
        FOR each character and its frequency IN the items of count  
            APPEND the pair of the negative frequency and the character TO max_heap  
        END FOR  
        CALL the function to transform max_heap into a heap data structure  
        
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN an empty string  
            END IF  
            
            REMOVE the smallest element from max_heap and ASSIGN frequencies and character to freq and char respectively  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                ADD the pair prev_freq and prev_char TO max_heap as a heap element  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
