CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing plus signs in side with hash plus signs and replacing minus signs in side with hash minus signs then splitting by hash signs  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is an empty string  
                    CONTINUE to the next iteration  
                END IF  
                IF token contains the character x  
                    IF token equals the character x OR token equals a plus sign followed by the character x  
                        INCREMENT x_count BY one  
                    ELSE IF token equals a minus sign followed by the character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET number_string TO the token with the character x replaced by an empty string  
                        INCREMENT x_count BY the integer value of number_string  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value of token  
                END IF  
            END FOR  
            RETURN x_count and num_sum  
        END FUNCTION  

        SET left_side and right_side TO the result of splitting equation by the equal sign  
        SET left_x and left_num TO the result of calling parse_side with left_side  
        SET right_x and right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            SET division_result TO total_num DIVIDED BY total_x using integer division  
            RETURN the string formed by concatenating the character x the equal sign and the string representation of division_result  
        END IF  
    END FUNCTION  
END CLASS
