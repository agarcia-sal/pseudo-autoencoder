CLASS Solution  
    FUNCTION makeArrayIncreasing with parameters arr1 and arr2  
        SORT arr2 INTO ascending order  
        SET dp TO a dictionary with a key of negative one and a value of zero representing an empty subsequence with zero operations  
        FOR each number in arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num and ops in dp items  
                IF current number is greater than last_num  
                    SET the value at key current number in new_dp to the minimum of the current value at key current number in new_dp or positive infinity and ops  
                END IF  
                SET idx TO the result of finding the position to insert last_num to the right using binary search in arr2  
                IF idx is less than the length of arr2  
                    SET the value at key element at position idx of arr2 in new_dp to the minimum of the current value at that key or positive infinity and ops incremented by one  
                END IF  
            END FOR  
            IF new_dp is empty  
                RETURN negative one  
            END IF  
            SET dp TO new_dp  
        END FOR  
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
