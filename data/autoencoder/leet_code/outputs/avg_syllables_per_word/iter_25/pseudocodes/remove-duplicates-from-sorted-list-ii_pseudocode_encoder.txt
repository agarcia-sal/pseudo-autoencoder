CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val INITIALIZED TO zero next INITIALIZED TO none  
        SET the val attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A new instance OF ListNode initialized WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO A new instance OF ListNode initialized WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the expression THAT the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS self head  
        SET dummy TO A new instance OF ListNode initialized WITH zero AND head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF the next attribute of head IS NOT none AND the val attribute of head EQUALS the val attribute of the next attribute of head  
                WHILE the next attribute of head IS NOT none AND the val attribute of head EQUALS the val attribute of the next attribute of head  
                    SET head TO the next attribute of head  
                END WHILE  
                SET the next attribute of prev TO the next attribute of head  
            ELSE  
                SET prev TO the next attribute of prev  
            END IF  
            SET head TO the next attribute of head  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
