CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the SET OF elements IN wordDict  
        SET dp TO a LIST OF boolean FALSE with LENGTH equal to the LENGTH OF s PLUS one  
        SET the element at position zero of dp TO TRUE  
        FOR i FROM one TO the LENGTH OF s PLUS one MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position j of dp IS TRUE AND the substring from position j TO position i MINUS one of s IS AN ELEMENT OF wordSet  
                    SET the element at position i of dp TO TRUE  
                    BREAK OUT OF the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
