CLASS Solution  
    FUNCTION sumScores WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET lcp TO a list containing zero repeated n times  
        SET l TO zero  
        SET r TO zero  
        FOR each i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r THEN  
                SET lcp at position i TO the MINIMUM between r MINUS i PLUS one AND lcp at position i MINUS l  
            END IF  
            WHILE i PLUS lcp at position i LESS THAN n AND the character at position lcp at position i of s EQUALS the character at position i PLUS lcp at position i of s DO  
                INCREMENT lcp at position i BY one  
            END WHILE  
            IF i PLUS lcp at position i MINUS one GREATER THAN r THEN  
                SET l TO i  
                SET r TO i PLUS lcp at position i MINUS one  
            END IF  
        END FOR  
        SET total_score TO the SUM OF all elements in lcp PLUS n  
        RETURN total_score  
    END FUNCTION  
END CLASS
