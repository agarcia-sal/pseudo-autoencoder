CLASS Solution  
    FUNCTION maximumGap WITH PARAMETER nums  
        IF the LENGTH OF nums IS LESS THAN two  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort WITH PARAMETERS arr AND exp  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros with LENGTH n  
            SET count TO a list of zeros with LENGTH ten  
            
            FOR each i FROM zero TO n MINUS one  
                SET index TO the INTEGER DIVISION OF element at position i of arr DIVIDED BY exp  
                INCREMENT element at position index MODULO ten of count BY one  
            END FOR  
            
            FOR each i FROM one TO nine  
                INCREMENT element at position i of count BY element at position i MINUS one of count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i IS GREATER THAN OR EQUAL TO zero  
                SET index TO the INTEGER DIVISION OF element at position i of arr DIVIDED BY exp  
                SET element at position (element at position index MODULO ten of count MINUS one) of output TO element at position i of arr  
                DECREMENT element at position index MODULO ten of count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR each i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort WITH PARAMETER arr  
            SET max1 TO the maximum element of arr  
            
            SET exp TO one  
            WHILE the INTEGER DIVISION OF max1 BY exp IS GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                MULTIPLY exp BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort WITH nums  
        
        SET max_diff TO zero  
        FOR each i FROM one TO the LENGTH OF nums MINUS one  
            SET max_diff TO the maximum of max_diff AND element at position i of nums MINUS element at position i MINUS one of nums  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
