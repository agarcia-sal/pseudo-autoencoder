CLASS Solution  
    FUNCTION minMeetingRooms WITH PARAMETER intervals  
        IF intervals IS EQUAL TO empty list  
            RETURN zero  
        END IF  
        SORT intervals BY the start time element of each interval in ascending order  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF min_heap IS NOT empty AND the start time element of meeting IS GREATER THAN OR EQUAL TO the first element of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the end time element of meeting TO min_heap  
        END FOR  
        RETURN the number of elements in min_heap  
    END FUNCTION  
END CLASS
