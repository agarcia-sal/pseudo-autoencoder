CLASS Solution  
    FUNCTION strangePrinter WITH PARAMETER s  
        FUNCTION dp WITH PARAMETERS i j memo  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            IF the TUPLE of i AND j IS IN memo THEN  
                RETURN the ELEMENT AT the TUPLE of i AND j IN memo  
            END IF  
            SET result TO the CALL of dp WITH ARGUMENTS i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF the ELEMENT AT POSITION k OF s EQUALS the ELEMENT AT POSITION j OF s THEN  
                    SET temporary_value TO the CALL of dp WITH ARGUMENTS i AND k MINUS one AND memo PLUS the CALL of dp WITH ARGUMENTS k AND j MINUS one AND memo  
                    IF temporary_value LESS THAN result THEN  
                        SET result TO temporary_value  
                    END IF  
                END IF  
            END FOR  
            SET the ELEMENT AT the TUPLE of i AND j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN the CALL of dp WITH ARGUMENTS zero AND the LENGTH OF s MINUS one AND an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
