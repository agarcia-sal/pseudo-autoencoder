CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one TO the end  
        SET node TO a new instance of ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
        SET dummy_head TO a new instance of ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero  
            IF l1 IS NOT None  
                SET val1 TO the value attribute of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the value attribute of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY ten  
            SET the next attribute of current TO a new instance of ListNode with the remainder of total DIVIDED BY ten  
            SET current TO the next attribute of current  
            IF l1 IS NOT None  
                SET l1 TO the next attribute of l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO the next attribute of l2  
            END IF  
        END WHILE  
        RETURN the next attribute of dummy_head  
    END FUNCTION  
END CLASS
