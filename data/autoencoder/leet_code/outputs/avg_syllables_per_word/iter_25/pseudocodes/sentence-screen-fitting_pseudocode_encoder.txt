CLASS Solution  
    FUNCTION wordsTyping(sentence AS list of strings rows AS integer cols AS integer) RETURNS integer  
        SET s TO the concatenation of all elements in sentence separated by one space PLUS one space at the end  
        SET total_length TO the number of characters in s  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the character at position modulo total_length of s EQUALS a space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character at position minus one modulo total_length of s NOT EQUALS a space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN the integer division of position BY total_length  
    END FUNCTION  
END CLASS
