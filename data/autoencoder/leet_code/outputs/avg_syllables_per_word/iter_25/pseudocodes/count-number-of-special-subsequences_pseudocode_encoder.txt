CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO ten raised to the power of nine plus seven  
        SET dp TO a list containing zero zero zero  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp at position zero TO the result of two multiplied by dp at position zero plus one  
                SET dp at position zero TO dp at position zero modulo MOD  
            ELSE IF num EQUALS one  
                SET dp at position one TO the result of two multiplied by dp at position one plus dp at position zero  
                SET dp at position one TO dp at position one modulo MOD  
            ELSE IF num EQUALS two  
                SET dp at position two TO the result of two multiplied by dp at position two plus dp at position one  
                SET dp at position two TO dp at position two modulo MOD  
            END IF  
        END FOR  
        RETURN dp at position two  
    END FUNCTION  
END CLASS
