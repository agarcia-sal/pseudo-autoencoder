CLASS Solution  
    FUNCTION isMatch WITH PARAMETERS s p  
        SET dp TO a two dimensional list where each element is False with the number of rows equal to the length of s plus one and the number of columns equal to the length of p plus one  
        SET the element at row zero column zero of dp TO True  
        FOR j FROM one TO the length of p  
            IF the character at position j minus one of p EQUALS the character star  
                SET the element at row zero column j of dp TO the element at row zero column j minus two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the character at position j minus one of p EQUALS the character at position i minus one of s OR the character at position j minus one of p EQUALS the character dot  
                    SET the element at row i column j of dp TO the element at row i minus one column j minus one of dp  
                ELSE IF the character at position j minus one of p EQUALS the character star  
                    SET the element at row i column j of dp TO the element at row i column j minus two of dp  
                    IF the character at position j minus two of p EQUALS the character at position i minus one of s OR the character at position j minus two of p EQUALS the character dot  
                        SET the element at row i column j of dp TO the logical OR of the element at row i column j of dp AND the element at row i minus one column j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row the length of s column the length of p of dp  
    END FUNCTION  
END CLASS
