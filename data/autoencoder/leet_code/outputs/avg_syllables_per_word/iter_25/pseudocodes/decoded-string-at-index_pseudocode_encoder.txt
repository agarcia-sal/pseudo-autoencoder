CLASS Solution  
    FUNCTION decodeAtIndex WITH PARAMETERS s AND k  
        SET length TO zero  
        FOR each char IN s  
            IF char is a digit  
                SET length TO length MULTIPLIED BY the integer value of char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN the reverse order of s  
            SET k TO k MODULO length  
            IF k EQUALS zero AND char is an alphabetic character  
                RETURN char  
            END IF  
            IF char is a digit  
                SET length TO length DIVIDED BY the integer value of char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
