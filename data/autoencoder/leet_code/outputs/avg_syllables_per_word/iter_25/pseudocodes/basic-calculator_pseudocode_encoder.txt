CLASS Solution  
    FUNCTION calculate WITH PARAMETER s AS string  
        FUNCTION helper WITH PARAMETER s AS list  
            SET stack TO an empty list  
            SET sign TO the character plus  
            SET num TO zero  
            
            WHILE the length of s is greater than zero  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero from s  
                
                IF the character char is a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                
                IF char is the character left parenthesis  
                    SET num TO the result of calling helper with s  
                END IF  
                
                IF char is not a digit AND char is not a whitespace character OR the length of s is equal to zero  
                    IF sign is the character plus  
                        APPEND num TO stack  
                    ELSE IF sign is the character minus  
                        APPEND the negative of num TO stack  
                    ELSE IF sign is the character asterisk  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign is the character forward slash  
                        SET the element at the last position of stack TO the integer division result of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char is the character right parenthesis  
                    BREAK the while loop  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper with the list conversion of s  
    END FUNCTION  
END CLASS
