CLASS Solution  
    FUNCTION maximumScore WITH PARAMETERS nums AND multipliers  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated m PLUS one times  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zero repeated m PLUS one times  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i MINUS left MINUS NEGATIVE one  
                SET mul TO the element at position i of multipliers  
                SET new_dp at position left TO the GREATER OF mul MULTIPLIED BY the element at position left of nums PLUS dp at position left PLUS one AND mul MULTIPLIED BY the element at position right of nums PLUS dp at position left  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN dp at position zero  
    END FUNCTION  
END CLASS
