CLASS Solution  
    FUNCTION scoreOfParentheses WITH PARAMETER s  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS left parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO element removed from the end of stack PLUS  
                    IF score EQUALS zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
