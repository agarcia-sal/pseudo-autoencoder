CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list formed by pairing each element of difficulty with the corresponding element of profit  
        SORT jobs IN ascending order BY the first element of each pair representing difficulty  
        SORT worker IN ascending order by ability  

        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of jobs  

        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the first element of the job at position job_index LESS THAN OR EQUAL TO ability  
                SET best_profit TO the greater value BETWEEN best_profit AND the second element of the job at position job_index  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  

        RETURN max_profit  
    END FUNCTION  
END CLASS
