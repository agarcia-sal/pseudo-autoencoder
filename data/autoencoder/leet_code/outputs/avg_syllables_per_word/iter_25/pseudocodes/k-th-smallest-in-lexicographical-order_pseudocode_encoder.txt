CLASS Solution  
    FUNCTION findKthNumber WITH PARAMETERS n k  
        FUNCTION count_steps WITH PARAMETERS prefix n  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix LESS THAN OR EQUAL TO n  
                SET difference_one TO n MINUS prefix PLUS one  
                SET difference_two TO next_prefix MINUS prefix  
                IF difference_one LESS THAN difference_two  
                    INCREMENT steps BY difference_one  
                ELSE  
                    INCREMENT steps BY difference_two  
                END IF  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        DECREMENT k BY one  
        WHILE k GREATER THAN zero  
            SET steps TO THE RESULT OF count_steps WITH current AND n  
            IF steps LESS THAN OR EQUAL TO k  
                INCREMENT current BY one  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                DECREMENT k BY one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
