CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero THEN  
                RETURN a list containing the empty string  
            ELSE IF u EQUALS one THEN  
                RETURN a list containing zero one and eight as strings  
            END IF  
            SET ans TO an empty list  
            FOR each v IN the result of dfs called with u MINUS two DO  
                FOR each pair l and r IN the pairs consisting of one one eighty-eight six nine and nine six DO  
                    APPEND the concatenation of l then v then r TO ans  
                END FOR  
                IF u NOT EQUALS n THEN  
                    APPEND the concatenation of zero then v then zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN the result of dfs called with n  
    END FUNCTION  
END CLASS
