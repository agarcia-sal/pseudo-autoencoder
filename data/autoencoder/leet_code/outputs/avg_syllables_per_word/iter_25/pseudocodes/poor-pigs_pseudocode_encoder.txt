CLASS Solution  
    FUNCTION poorPigs WITH PARAMETERS buckets minutesToDie minutesToTest  
        SET states TO the integer division result of minutesToTest DIVIDED BY minutesToDie INCREMENTED BY one  
        SET logarithm_of_buckets TO the logarithm base e of buckets  
        SET logarithm_of_states TO the logarithm base e of states  
        SET pigs TO the smallest integer greater than or equal to the division of logarithm_of_buckets BY logarithm_of_states  
        RETURN pigs  
    END FUNCTION  
END CLASS
