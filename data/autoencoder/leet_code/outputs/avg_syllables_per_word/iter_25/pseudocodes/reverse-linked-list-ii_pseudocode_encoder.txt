CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values OF TYPE list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with the first element at position zero of values  
    SET p TO head  
    FOR each val IN the slice of values from position one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the logical AND of the comparison that p1.val EQUALS p2.val AND the recursive call to is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS self head OF TYPE Optional of ListNode left OF TYPE integer right OF TYPE integer RETURNING Optional of ListNode  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET prev TO dummy  
        FOR each index FROM zero TO left MINUS one MINUS one  
            SET prev TO the next attribute of prev  
        END FOR  
        SET current TO the next attribute of prev  
        FOR each index FROM zero TO right MINUS left MINUS one  
            SET next_node TO the next attribute of current  
            SET the next attribute of current TO the next attribute of next_node  
            SET the next attribute of next_node TO the next attribute of prev  
            SET the next attribute of prev TO next_node  
        END FOR  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
