CLASS Solution  
    FUNCTION minWastedSpace(packages, boxes)  
        SET MOD TO one billion PLUS seven  
        SORT packages in ascending order  
        SET n TO the LENGTH OF packages  
        SET prefix_sum TO a list of zeroes with length n PLUS one  
        FOR index FROM zero TO n MINUS one  
            SET element at position index PLUS one of prefix_sum TO element at position index of prefix_sum PLUS element at position index of packages  
        END FOR  
        SET min_wasted_space TO positive infinity  
        FOR each box IN boxes  
            SORT box in ascending order  
            IF element at last position of box LESS THAN element at last position of packages THEN  
                CONTINUE to the next iteration of the loop for boxes  
            END IF  
            SET total_wasted TO zero  
            SET last_index TO zero  
            FOR each box_size IN box  
                SET index TO last_index  
                WHILE index LESS THAN n AND element at position index of packages LESS THAN OR EQUAL TO box_size  
                    INCREMENT index BY one  
                END WHILE  
                IF index GREATER THAN last_index THEN  
                    INCREMENT total_wasted BY the product of index MINUS last_index and box_size MINUS the difference of element at position index of prefix_sum MINUS element at position last_index of prefix_sum  
                    SET last_index TO index  
                END IF  
            END FOR  
            IF last_index EQUALS n THEN  
                SET min_wasted_space TO the smaller value between min_wasted_space AND total_wasted  
            END IF  
        END FOR  
        IF min_wasted_space NOT EQUAL TO positive infinity THEN  
            RETURN min_wasted_space MODULO MOD  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
