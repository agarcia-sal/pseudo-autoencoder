CLASS Solution  
    FUNCTION numPermsDISequence WITH PARAMETER s  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeroes with dimensions n PLUS one BY n PLUS one  
        FOR each integer j FROM zero TO n INCLUSIVE  
            SET element at position zero at position j of dp TO one  
        END FOR  
        FOR each integer i FROM one TO n INCLUSIVE  
            SET prefix_sum TO zero  
            IF character at position i MINUS one of s EQUALS the uppercase letter I THEN  
                FOR each integer j FROM zero TO n PLUS one MINUS i EXCLUSIVE  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one at position j of dp MODULO MOD  
                    SET element at position i at position j of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each integer j FROM n MINUS i DOWN TO zero INCLUSIVE DECREMENTING BY one  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one at position j PLUS one of dp MODULO MOD  
                    SET element at position i at position j of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        SET result TO the sum of elements at position n of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
