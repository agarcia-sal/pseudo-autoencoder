CLASS UnionFind  
    FUNCTION __init__ WITH PARAMETER size  
        SET parent TO a list of integers from zero to size MINUS one  
        SET rank TO a list of zeros with length size  
        SET size TO a list of ones with length size  
    END FUNCTION  

    FUNCTION find WITH PARAMETER u  
        IF the element at position u of parent NOT EQUALS u THEN  
            SET the element at position u of parent TO the result of calling find with the element at position u of parent  
        END IF  
        RETURN the element at position u of parent  
    END FUNCTION  

    FUNCTION union WITH PARAMETERS u AND v  
        SET rootU TO the result of calling find with u  
        SET rootV TO the result of calling find with v  
        IF rootU NOT EQUALS rootV THEN  
            IF the element at position rootU of rank GREATER THAN the element at position rootV of rank THEN  
                SET the element at position rootV of parent TO rootU  
                INCREMENT the element at position rootU of size BY the element at position rootV of size  
            ELSE IF the element at position rootU of rank LESS THAN the element at position rootV of rank THEN  
                SET the element at position rootU of parent TO rootV  
                INCREMENT the element at position rootV of size BY the element at position rootU of size  
            ELSE  
                SET the element at position rootV of parent TO rootU  
                INCREMENT the element at position rootU of size BY the element at position rootV of size  
                INCREMENT the element at position rootU of rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings WITH PARAMETER words  
        FUNCTION bitmask WITH PARAMETER word  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO bitwise OR of mask PLUS the number one MULTIPLIED BY two RAISED TO the power of the result of subtracting the Unicode code point of a from the Unicode code point of character  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a new default dictionary of lists  
        SET n TO the LENGTH OF words  
        SET uf TO a new instance of UnionFind initialized with n  

        FOR each index i AND word IN enumeration of words  
            SET mask TO the result of calling bitmask with word  
            APPEND i TO the list at key mask in word_map  
        END FOR  

        FOR each index i AND word IN enumeration of words  
            SET mask TO the result of calling bitmask with word  
            FOR integer j FROM zero TO twenty five  
                SET new_mask TO the result of bitwise XOR of mask PLUS the number one MULTIPLIED BY two RAISED TO the power of j  
                IF new_mask EXISTS AS A KEY IN word_map THEN  
                    FOR each element k IN the list at key new_mask in word_map  
                        CALL union on uf with i AND k  
                    END FOR  
                END IF  

                IF the bitwise AND of mask AND the number one MULTIPLIED BY two RAISED TO the power of j EQUALS zero THEN  
                    SET new_mask TO the bitwise OR of mask PLUS the number one MULTIPLIED BY two RAISED TO the power of j  
                    IF new_mask EXISTS AS A KEY IN word_map THEN  
                        FOR each element k IN the list at key new_mask in word_map  
                            CALL union on uf with i AND k  
                        END FOR  
                    END IF  
                END IF  

                IF the bitwise AND of mask AND the number one MULTIPLIED BY two RAISED TO the power of j NOT EQUALS zero THEN  
                    FOR integer k FROM zero TO twenty five  
                        IF the bitwise AND of mask AND the number one MULTIPLIED BY two RAISED TO the power of k EQUALS zero THEN  
                            SET new_mask TO the bitwise OR of the bitwise XOR of mask PLUS the number one MULTIPLIED BY two RAISED TO the power of j PLUS the number one MULTIPLIED BY two RAISED TO the power of k  
                            IF new_mask EXISTS AS A KEY IN word_map THEN  
                                FOR each element l IN the list at key new_mask in word_map  
                                    CALL union on uf with i AND l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a new default dictionary of integers  
        FOR integer i FROM zero TO n MINUS one  
            SET root TO the result of calling find on uf with i  
            INCREMENT the element at key root in root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum of values in root_sizes  
        SET num_groups TO the count of keys in root_sizes  

        RETURN a list containing num_groups AND max_group_size  
    END FUNCTION  
END CLASS
