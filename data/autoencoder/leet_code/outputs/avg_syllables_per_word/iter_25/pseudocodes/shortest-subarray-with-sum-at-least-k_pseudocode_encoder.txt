CLASS Solution  
    FUNCTION shortestSubarray(nums AS List of integer COMMA k AS integer) RETURNS integer  
        SET prefix_sum TO a list containing zero  
        FOR each element num IN nums  
            SET last_element TO element at the last position of prefix_sum  
            SET new_value TO last_element PLUS num  
            APPEND new_value TO prefix_sum  
        END FOR  

        ASSIGN reference TO a new empty deque dq  
        SET min_length TO positive infinity  

        FOR each index i AND element current_sum IN prefix_sum with enumeration  
            WHILE dq is not empty AND current_sum MINUS element at position at the front of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET candidate_length TO i MINUS the element removed from the front of dq  
                SET min_length TO the lesser of min_length AND candidate_length  
            END WHILE  

            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position at the last of dq of prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  

            APPEND i TO dq  
        END FOR  

        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
