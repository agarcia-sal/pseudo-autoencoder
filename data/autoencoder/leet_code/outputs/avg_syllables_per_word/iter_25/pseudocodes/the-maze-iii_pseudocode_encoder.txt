CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze ball hole  
        SET directions TO list containing tuple zero one r tuple one zero d tuple zero negative one l tuple negative one zero u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        ASSIGN reference TO tuple containing elements of ball  
        ASSIGN reference TO tuple containing elements of hole  
        SET pq TO list containing tuple zero empty string start  
        SET visited TO an empty set  
        
        WHILE pq is not empty  
            REMOVE element with smallest first element FROM pq AND ASSIGN to dist path and a tuple containing x and y  
            
            IF x EQUALS element zero of hole AND y EQUALS element one of hole  
                RETURN path  
            END IF  
            
            IF tuple containing x and y IS IN visited  
                CONTINUE  
            END IF  
            
            ADD tuple containing x and y TO visited  
            
            FOR each dx dy and direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    
                    IF new_x EQUALS element zero of hole AND new_y EQUALS element one of hole  
                        BREAK  
                    END IF  
                END WHILE  
                
                IF tuple containing new_x and new_y IS NOT IN visited  
                    INSERT tuple containing dist PLUS count path concatenated with direction AND tuple containing new_x and new_y INTO pq MAINTAINING pq as a heap ordered by first element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
