CLASS Solution  
    FUNCTION minDistance word1 word2  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two-dimensional list of zeros with dimensions m plus one by n plus one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i minus one of word1 EQUALS the element at position j minus one of word2  
                    SET the element at row i column j of dp TO the element at row i minus one column j minus one of dp PLUS one  
                ELSE  
                    SET the element at row i column j of dp TO the GREATER VALUE BETWEEN the element at row i minus one column j of dp AND the element at row i column j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at row m column n of dp  
        RETURN the SUM of m MINUS lcs_length AND n MINUS lcs_length  
    END FUNCTION  
END CLASS
