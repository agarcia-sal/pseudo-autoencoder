CLASS Solution  
    FUNCTION minSkips WITH PARAMETERS dist speed hoursBefore  
        SET n TO the LENGTH OF dist  
        SET dp TO a two dimensional list of size n BY n PLUS one filled with positive infinity representing minimum time to reach the end of each road with given skips  
        SET total_dist TO the SUM OF all elements in dist  
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        SET the element at position zero zero in dp TO the element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET the element at position zero j in dp TO the element at position zero of dist  
        END FOR  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET the element at position i j in dp TO the result of multiplying the ceiling of the element at position i MINUS one j in dp DIVIDED BY speed BY speed PLUS the element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET the element at position i j in dp TO the minimum of the element at position i j in dp AND the sum of the element at position i MINUS one j MINUS one in dp AND the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        FOR j FROM zero TO n MINUS one  
            IF the element at position n MINUS one j in dp IS LESS THAN OR EQUAL TO the product of hoursBefore AND speed THEN  
                RETURN j  
            END IF  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
