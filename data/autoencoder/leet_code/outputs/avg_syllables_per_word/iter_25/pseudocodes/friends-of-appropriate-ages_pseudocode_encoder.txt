CLASS Solution  
    FUNCTION numFriendRequests WITH PARAMETER ages  
        FUNCTION request_will_be_sent WITH PARAMETERS age_x age_y  
            RETURN THE NEGATION OF THE CONDITION age_y IS LESS THAN OR EQUAL TO age_x MULTIPLIED BY one HALF PLUS seven OR age_y IS GREATER THAN age_x OR age_y IS GREATER THAN one hundred AND age_x IS LESS THAN one hundred  
        END FUNCTION  
        SET count TO A LIST OF zeroes WITH LENGTH one hundred AND twenty ONE  
        FOR each age IN ages  
            INCREMENT element at position age of count BY one  
        END FOR  
        SET total_requests TO zero  
        FOR age_x FROM one TO one hundred AND twenty  
            FOR age_y FROM one TO one hundred AND twenty  
                IF request_will_be_sent WITH ARGUMENTS age_x age_y HOLDS  
                    INCREMENT total_requests BY element at position age_x of count MULTIPLIED BY element at position age_y of count MINUS THE EQUALITY CHECK WHETHER age_x EQUALS age_y  
                END IF  
            END FOR  
        END FOR  
        RETURN total_requests  
    END FUNCTION  
END CLASS
