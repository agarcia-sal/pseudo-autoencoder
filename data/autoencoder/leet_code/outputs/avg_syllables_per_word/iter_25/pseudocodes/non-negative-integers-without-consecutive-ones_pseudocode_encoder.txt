CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET bin_rep TO the binary representation of n starting from the third character onward  
        SET length TO the number of characters in bin_rep  
        CREATE a list dp with length plus one elements each set to zero  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  
        FOR i FROM two TO length  
            SET element at position i of dp TO the sum of element at position i minus one of dp PLUS element at position i minus two of dp  
        END FOR  
        SET result TO zero  
        SET prev_bit TO zero  
        FOR i FROM zero TO length minus one  
            IF the character at position i of bin_rep EQUALS one THEN  
                INCREMENT result BY element at position length minus i minus one of dp  
                IF prev_bit EQUALS one THEN  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET prev_bit TO one IF the character at position i of bin_rep EQUALS one OTHERWISE zero  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
