CLASS Solution  
    FUNCTION minInteger WITH PARAMETERS num COMMA k  
        SET num TO list CONVERSION OF num  
        SET n TO LENGTH OF num  
        WHILE k IS GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num IS GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO THE MINIMUM OF i PLUS k PLUS one AND n  
                        IF element at position j of num IS LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    SET the SUBLIST from position i TO position min_index PLUS one of num TO A NEW LIST COMPRISING element at position min_index of num FOLLOWED BY the SUBLIST from position i TO position min_index of num  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK OUT OF FOR LOOP  
                END IF  
            END FOR  
            IF swapped IS false  
                BREAK OUT OF WHILE LOOP  
            END IF  
        END WHILE  
        RETURN concatenation of all elements in num  
    END FUNCTION  
END CLASS
