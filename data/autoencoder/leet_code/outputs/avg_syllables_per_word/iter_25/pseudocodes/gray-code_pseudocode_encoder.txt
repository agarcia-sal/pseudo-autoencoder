CLASS Solution  
    FUNCTION grayCode WITH PARAMETER n  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH PARAMETER n MINUS one ON self  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET current_gray TO concatenation of prev_gray AND a list formed by for each number in reversed order of prev_gray creating a number by mask OR number  
        RETURN current_gray  
    END FUNCTION  
END CLASS
