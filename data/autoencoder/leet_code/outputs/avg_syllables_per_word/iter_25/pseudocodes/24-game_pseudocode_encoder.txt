CLASS Solution  
    FUNCTION judgePoint24 WITH parameter cards  
        FUNCTION solve WITH parameter nums  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty four IS LESS THAN one times ten TO THE POWER OF negative six  
            END IF  
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        FOR op IN the sequence truediv mul add sub  
                            IF (op IS add OR op IS mul) AND j GREATER THAN i  
                                CONTINUE  
                            END IF  
                            IF op IS NOT truediv OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying the operation op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve WITH argument new_nums  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN solve WITH argument cards  
    END FUNCTION  
END CLASS
