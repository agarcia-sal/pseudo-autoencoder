CLASS Solution  
    FUNCTION findDuplicate WITH parameter paths  
        SET content_to_paths TO a new default dictionary with list as default value  
        FOR each path IN paths  
            SET parts TO the result of splitting the path by whitespace  
            SET directory TO the element at position zero of parts  
            FOR each file_info IN the elements of parts from position one to the end  
                SET file_name TO the substring before the first occurrence of the character opening parenthesis in file_info  
                SET content TO the substring after the first occurrence of the character opening parenthesis in file_info  
                REMOVE the last character from content  
                APPEND the concatenation of directory the forward slash character and file_name TO the list at content key in content_to_paths  
            END FOR  
        END FOR  
        SET result TO a new empty list  
        FOR each group_of_paths IN the values of content_to_paths  
            IF the length of group_of_paths is greater than one  
                APPEND group_of_paths TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
