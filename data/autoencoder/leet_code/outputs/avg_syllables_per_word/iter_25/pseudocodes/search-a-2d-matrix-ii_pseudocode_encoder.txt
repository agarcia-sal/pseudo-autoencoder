CLASS Solution  
    FUNCTION searchMatrix matrix target  
        IF matrix IS empty OR the first element at position zero of matrix IS empty  
            RETURN False  
        END IF  
        SET row TO zero  
        SET col TO the LENGTH OF the first element at position zero of matrix MINUS one  
        WHILE row LESS THAN the LENGTH OF matrix AND col GREATER THAN OR EQUAL TO zero  
            IF the element at position row of matrix AND position col of that element EQUALS target  
                RETURN True  
            ELSE IF the element at position row of matrix AND position col of that element GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
