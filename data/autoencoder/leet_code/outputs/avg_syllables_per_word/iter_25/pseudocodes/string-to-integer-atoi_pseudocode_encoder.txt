    FUNCTION myAtoi WITH PARAMETER s AS string  
        SET INT_MAX TO the integer value of two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MINUS one  
        SET INT_MIN TO the integer value of the negative of two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two MULTIPLIED BY two  
        SET result TO zero  
        SET index TO zero  
        SET n TO the length of s  
        SET sign TO one  
        WHILE index LESS THAN n AND the element at position index of s EQUALS a space character  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (the element at position index of s EQUALS a plus sign OR the element at position index of s EQUALS a minus sign)  
            SET sign TO the negative one IF the element at position index of s EQUALS a minus sign OTHERWISE set to one  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND the element at position index of s IS a digit  
            SET digit TO the integer value of the element at position index of s  
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
