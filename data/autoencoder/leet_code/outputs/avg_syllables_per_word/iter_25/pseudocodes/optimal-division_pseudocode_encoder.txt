CLASS Solution  
    FUNCTION optimalDivision WITH PARAMETER nums  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF element at position zero of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF element at position zero of nums CONCATENATED WITH the CHARACTERS FOR DIVISION CONCATENATED WITH the STRING REPRESENTATION OF element at position one of nums  
        END IF  
        RETURN the STRING REPRESENTATION OF element at position zero of nums CONCATENATED WITH the CHARACTERS FOR DIVISION CONCATENATED WITH the CHARACTER FOR OPENING PARENTHESIS CONCATENATED WITH the STRING REPRESENTATION OF element at position one of nums concatenated with the CHARACTER FOR DIVISION concatenated with the STRING REPRESENTATION OF each element from position two TO the END of nums joined by the CHARACTER FOR DIVISION CONCATENATED WITH the CHARACTER FOR CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
