CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode with the first element of values as val  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new instance of TreeNode with element at position i of values as val  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new instance of TreeNode with element at position i of values as val  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with parameters p.left q.left AND the result of calling is_same_tree with parameters p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree WITH PARAMETERS self root  
        IF root IS None  
            RETURN an empty list  
        END IF  

        FUNCTION is_leaf WITH PARAMETER node  
            RETURN node.left IS None AND node.right IS None  
        END FUNCTION  

        SET left_boundary TO an empty list  
        SET current TO root.left  
        WHILE current IS NOT None  
            IF is_leaf called with current IS False  
                APPEND current.val TO left_boundary  
            END IF  
            IF current.left IS NOT None  
                SET current TO current.left  
            ELSE  
                SET current TO current.right  
            END IF  
        END WHILE  

        SET right_boundary TO an empty list  
        SET current TO root.right  
        WHILE current IS NOT None  
            IF is_leaf called with current IS False  
                APPEND current.val TO right_boundary  
            END IF  
            IF current.right IS NOT None  
                SET current TO current.right  
            ELSE  
                SET current TO current.left  
            END IF  
        END WHILE  

        SET leaves TO an empty list  
        FUNCTION collect_leaves WITH PARAMETER node  
            IF node IS None  
                RETURN  
            END IF  
            IF is_leaf called with node IS True  
                APPEND node.val TO leaves  
            END IF  
            CALL collect_leaves with parameter node.left  
            CALL collect_leaves with parameter node.right  
        END FUNCTION  

        CALL collect_leaves with parameter root  

        IF is_leaf called with root IS True  
            SET leaves TO an empty list  
        END IF  

        RETURN the concatenation of a list containing root.val followed by left_boundary followed by leaves followed by the reverse of right_boundary  
    END FUNCTION  
END CLASS
