CLASS Solution  
    FUNCTION maxValueAfterReverse(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS two  
            RETURN the absolute value of element at position zero of nums MINUS element at position one of nums  
        END IF  
        
        SET initial_value TO zero  
        FOR i FROM one TO n MINUS one  
            INCREMENT initial_value BY the absolute value of element at position i of nums MINUS element at position i MINUS one of nums  
        END FOR  
        
        SET max_edge_gain TO zero  
        FOR i FROM one TO n MINUS one  
            SET max_edge_gain TO the greater value BETWEEN max_edge_gain AND the absolute value of element at position zero of nums MINUS element at position i of nums MINUS the absolute value of element at position i of nums MINUS element at position i MINUS one of nums  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            SET max_edge_gain TO the greater value BETWEEN max_edge_gain AND the absolute value of element at position n MINUS one of nums MINUS element at position i of nums MINUS the absolute value of element at position i PLUS one of nums MINUS element at position i of nums  
        END FOR  
        
        SET min_pair TO the minimum value AMONG the greater value BETWEEN element at position i of nums AND element at position i PLUS one of nums FOR i FROM zero TO n MINUS two  
        SET max_pair TO the maximum value AMONG the lesser value BETWEEN element at position i of nums AND element at position i PLUS one of nums FOR i FROM zero TO n MINUS two  
        
        SET general_gain TO the greater value BETWEEN zero AND two MULTIPLIED BY the difference BETWEEN max_pair AND min_pair  
        
        RETURN the sum of initial_value AND the greater value BETWEEN max_edge_gain AND general_gain  
    END FUNCTION  
END CLASS
