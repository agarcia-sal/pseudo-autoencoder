CLASS Solution  
    FUNCTION numDecodings WITH PARAMETER s  
        IF s IS empty OR the element at position zero of s EQUALS the character zero  
            RETURN zero  
        END IF  
        CREATE list dp WITH size EQUAL TO the LENGTH OF s PLUS one FILLED WITH zero  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO one  
        FOR each variable i FROM two TO the LENGTH OF s PLUS one  
            IF the element at position i MINUS one of s NOT EQUALS the character zero  
                INCREMENT the element at position i of dp BY the element at position i MINUS one of dp  
            END IF  
            SET variable two_digit TO the integer value of the substring from position i MINUS two TO position i of s  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                INCREMENT the element at position i of dp BY the element at position i MINUS two of dp  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
