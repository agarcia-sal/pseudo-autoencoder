CLASS Solution  
    FUNCTION maxJumps WITH PARAMETERS arr d  
        FUNCTION dp WITH PARAMETER i  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x LESS THAN the LENGTH OF arr THEN  
                    IF the element at position i PLUS x of arr GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND one PLUS the RESULT OF dp WITH PARAMETER i PLUS x  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x GREATER THAN OR EQUAL TO zero THEN  
                    IF the element at position i MINUS x of arr GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND one PLUS the RESULT OF dp WITH PARAMETER i MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET maximum_jump_value TO negative infinity  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            SET candidate TO the RESULT OF dp WITH PARAMETER i  
            IF candidate GREATER THAN maximum_jump_value THEN  
                SET maximum_jump_value TO candidate  
            END IF  
        END FOR  
        RETURN maximum_jump_value  
    END FUNCTION  
END CLASS
