CLASS Solution  
    FUNCTION findMaximumXOR WITH PARAMETER nums  
        SET max_xor TO zero  
        FOR i FROM thirty one DOWN TO zero  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO empty set  
            FOR each num IN nums  
                SET prefix TO the integer resulting from num DIVIDED BY two raised to the power i  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the result of current_candidate EXCLUSIVE OR prefix IS an element of prefixes THEN  
                    SET max_xor TO current_candidate  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
