CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix IS EMPTY OR element at position zero of matrix IS EMPTY  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a list of lists of zeros with dimensions m plus one BY n plus one  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i minus one of matrix AND element at position j minus one of that element EQUALS the character one  
                    SET dp element at position i and j TO the SMALLEST of dp element at position i minus one and j AND dp element at position i and j minus one AND dp element at position i minus one and j minus one PLUS one  
                    SET max_side TO the LARGEST of max_side AND dp element at position i and j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
