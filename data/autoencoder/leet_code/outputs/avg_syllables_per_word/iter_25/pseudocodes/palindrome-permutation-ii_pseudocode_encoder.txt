CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s THEN  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt items  
                IF v GREATER THAN one THEN  
                    DECREMENT cnt at c BY two  
                    CALL dfs WITH c CONCATENATED WITH t CONCATENATED WITH c  
                    INCREMENT cnt at c BY two  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET cnt TO the count of characters in s  
        SET mid TO empty string  
        FOR each character c AND its count v IN cnt items  
            IF v MODULO two EQUALS one THEN  
                IF mid IS NOT empty THEN  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT cnt at c BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
