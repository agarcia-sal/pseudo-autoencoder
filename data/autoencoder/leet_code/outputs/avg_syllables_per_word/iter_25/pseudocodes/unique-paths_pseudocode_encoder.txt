CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m AND n  
        SET dp TO a two dimensional array with m rows each containing n elements each equal to one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position j of element at position i of dp TO element at position j of element at position i MINUS one of dp PLUS element at position j MINUS one of element at position i of dp  
            END FOR  
        END FOR  
        RETURN element at position n MINUS one of element at position m MINUS one of dp  
    END FUNCTION  
END CLASS
