CLASS Solution  
    FUNCTION letterCasePermutation WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO LENGTH OF t  
                APPEND the concatenation of all elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i INCREMENTED BY one  
            IF the element at position i of t is an alphabetic character  
                SET the element at position i of t TO the character corresponding to the bitwise exclusive OR of the Unicode code point of the element at position i of t WITH thirty-two  
                CALL dfs WITH i INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET t TO the list of characters of s  
        SET ans TO an empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
