CLASS Solution  
    FUNCTION largestOverlap WITH PARAMETERS img1 AND img2  
        SET n TO the LENGTH OF img1  
        SET points1 TO an EMPTY LIST  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img1 EQUALS one  
                    APPEND tuple consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an EMPTY LIST  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img2 EQUALS one  
                    APPEND tuple consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a new dictionary with default integer zero  
        FOR each tuple consisting of x1 AND y1 IN points1  
            FOR each tuple consisting of x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key tuple consisting of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        IF overlap_count is empty  
            RETURN zero  
        ELSE  
            RETURN the maximum value among all values in overlap_count  
        END IF  
    END FUNCTION  
END CLASS
