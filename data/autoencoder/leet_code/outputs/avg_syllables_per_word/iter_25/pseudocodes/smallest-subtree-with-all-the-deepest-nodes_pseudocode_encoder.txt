CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val SET TO zero AND left SET TO none AND right SET TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode USING the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not none  
            SET node dot left TO a new instance of TreeNode USING the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not none  
            SET node dot right TO a new instance of TreeNode USING the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p dot left AND q dot left AND the result of is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS self AND root  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO the result of dfs WITH PARAMETER node dot left  
            SET right_node AND right_depth TO the result of dfs WITH PARAMETER node dot right  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of the result of dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
