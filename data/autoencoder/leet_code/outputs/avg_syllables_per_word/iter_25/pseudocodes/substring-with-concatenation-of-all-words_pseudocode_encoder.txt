CLASS Solution  
    FUNCTION findSubstring with parameters s words  
        IF s IS EMPTY OR words IS EMPTY OR element at position zero of words IS EMPTY THEN  
            RETURN empty list  
        END IF  

        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the counting of occurrences of each element in words  
        SET result TO empty list  

        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an empty counting structure for occurrences  
            FOR j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO substring from position j TO position j PLUS word_length MINUS one of s  
                INCREMENT the count of word in seen BY one  
                IF the count of word in seen IS GREATER THAN the count of word in word_count THEN  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
            IF the inner loop completed without BREAK THEN  
                APPEND i TO result  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
