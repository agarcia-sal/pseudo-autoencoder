CLASS Solution  
    FUNCTION getSkyline WITH PARAMETER buildings  
        SET events TO an empty list  
        FOR each element consisting of left right and height IN buildings  
            APPEND a tuple consisting of left MINUS height and right TO events as a representation of the left edge of a building  
            APPEND a tuple consisting of right zero zero TO events as a representation of the right edge of a building  
        END FOR  
        SORT events in ascending order by the first element then by the second element then by the third element  
        SET result TO a list containing a list of zero zero as the initial key point  
        SET max_heap TO a list containing a tuple of zero and positive infinity representing zero height with infinite right edge  
        FOR each element consisting of x negH and r IN events  
            WHILE the right edge component of the first tuple in max_heap is LESS THAN OR EQUAL TO x  
                REMOVE the first tuple from max_heap maintaining heap property  
            END WHILE  
            IF negH is NOT EQUAL TO zero  
                ADD the tuple consisting of negH and r TO max_heap maintaining heap property  
            END IF  
            IF the second element of the last list in result is NOT EQUAL TO the negation of the first element of the first tuple in max_heap  
                APPEND a list containing x and the negation of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        IF the first list in result is EXACTLY the list containing zero zero  
            REMOVE the first list from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
