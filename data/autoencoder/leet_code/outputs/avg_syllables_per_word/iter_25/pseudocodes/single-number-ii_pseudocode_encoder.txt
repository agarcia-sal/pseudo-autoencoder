CLASS Solution  
    FUNCTION singleNumber with parameter nums  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO the bitwise exclusive OR of ones and num AND the bitwise negation of twos  
            SET twos TO the bitwise exclusive OR of twos and num AND the bitwise negation of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
