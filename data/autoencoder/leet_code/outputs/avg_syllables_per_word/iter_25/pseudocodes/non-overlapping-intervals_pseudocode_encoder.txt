CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS an empty list THEN  
            RETURN zero  
        END IF  
        SORT intervals IN ascending order BY the second element of each interval  
        SET end TO the second element of the first interval in intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the first element of the interval at position index OF intervals IS GREATER THAN OR EQUAL TO end THEN  
                SET end TO the second element of the interval at position index OF intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
