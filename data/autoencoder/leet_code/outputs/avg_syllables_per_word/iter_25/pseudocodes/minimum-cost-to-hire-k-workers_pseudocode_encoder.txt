CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the LIST of elements each consisting of the DIVISION of element wage by element quality and element quality for each paired elements wage and quality sorted in ascending order by the first element of each pair  
        SET max_heap TO an empty list to represent a max-heap by negative quality values  
        SET total_quality TO zero representing the total quality of the current group of workers  
        SET min_cost TO positive infinity representing the initialized minimum cost  
        FOR each pair ratio and q IN workers  
            ADD the NEGATION of q TO max_heap using heap push operation  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap is GREATER THAN k  
                REMOVE the element returned by the heap pop operation from max_heap and ADD it to total_quality  
            END IF  
            IF the LENGTH OF max_heap EQUALS k  
                SET min_cost TO the MINIMUM of min_cost and the MULTIPLICATION of total_quality and ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
