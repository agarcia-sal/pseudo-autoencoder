CLASS Solution  
    FUNCTION longestAwesome WITH PARAMETER s  
        SET prefix_map TO a mapping from zero TO minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each index i AND character char IN s as enumerated  
            CONVERT char TO its integer digit value AND SET TO digit  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR one shifted left by digit positions  
            IF prefix_xor EXISTS AS A KEY IN prefix_map  
                SET max_length TO the maximum of max_length AND index i MINUS the value at prefix_xor key in prefix_map  
            END IF  
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR one shifted left by j positions  
                IF flipped EXISTS AS A KEY IN prefix_map  
                    SET max_length TO the maximum of max_length AND index i MINUS the value at flipped key in prefix_map  
                END IF  
            END FOR  
            IF prefix_xor DOES NOT EXIST AS A KEY IN prefix_map  
                ADD entry with key prefix_xor AND value index i TO prefix_map  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
