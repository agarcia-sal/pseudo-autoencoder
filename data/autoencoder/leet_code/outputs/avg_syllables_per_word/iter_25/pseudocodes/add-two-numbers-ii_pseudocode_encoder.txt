CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode WITH val EQUAL TO element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO end of values  
        SET node TO new ListNode WITH val EQUAL TO val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN condition where p1.val EQUALS p2.val AND is_same_list CALLED WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS self l1 l2  
        FUNCTION pushIntoStack WITH PARAMETERS node  
            SET stack TO empty list  
            WHILE node IS not none  
                APPEND element at attribute val of node TO stack  
                SET node TO element at attribute next of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO pushIntoStack CALLED WITH l1  
        SET stack2 TO pushIntoStack CALLED WITH l2  

        SET carry TO zero  
        SET result TO none  

        WHILE condition where stack1 IS not empty OR stack2 IS not empty OR carry IS not zero  
            IF stack1 IS not empty  
                SET val1 TO last element REMOVED from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  

            IF stack2 IS not empty  
                SET val2 TO last element REMOVED from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  

            SET new_node TO new ListNode WITH val EQUAL TO digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
