CLASS Solution  
    FUNCTION jobScheduling startTime endTime profit  
        SET jobs TO the list of tuples each containing element at position index of startTime element at position index of endTime element at position index of profit sorted by the second element of each tuple in ascending order  
        SET end_times TO the list of the second element of each tuple in jobs  
        SET dp TO the list of zeros with the length equal to the length of jobs plus one  
        FOR i FROM one TO the length of jobs  
            SET j TO the position where element at position i minus one of jobs first element would be inserted in end_times to maintain sorted order to the right  
            SET dp at position i TO the greater value between dp at position i minus one and dp at position j plus the third element of element at position i minus one of jobs  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
