CLASS Solution  
    FUNCTION cutOffTree(forest)  
        IF forest IS empty OR element at position zero of forest IS empty  
            RETURN negative one  
        END IF  
        SET trees TO empty list  
        FOR each index i FROM zero TO the LENGTH OF forest MINUS one  
            FOR each index j FROM zero TO the LENGTH OF element at position zero of forest MINUS one  
                IF element at position i of forest at position j IS GREATER THAN one  
                    APPEND a tuple consisting of element at position i of forest at position j and i and j TO trees  
                END IF  
            END FOR  
        END FOR  
        SORT trees IN ascending order  
        FUNCTION bfs(start, end)  
            IF start EQUALS end  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF forest  
            SET n TO the LENGTH OF element at position zero of forest  
            ASSIGN a double ended queue containing start TO queue  
            ASSIGN a set containing start TO visited  
            SET steps TO zero  
            WHILE queue IS NOT empty  
                FOR each element in the range from zero TO the LENGTH OF queue MINUS one  
                    REMOVE the first element from queue AND UNPACK it to x and y  
                    FOR each tuple dx and dy IN the list consisting of negative one and zero and one and zero and zero and negative one and zero and one  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the tuple nx and ny IS NOT IN visited AND element at position nx of forest at position ny IS NOT EQUAL TO zero  
                            IF the tuple nx and ny EQUALS end  
                                RETURN steps PLUS one  
                            END IF  
                            ADD the tuple nx and ny TO visited  
                            ADD the tuple nx and ny TO the end of queue  
                        END IF  
                    END FOR  
                END FOR  
                INCREMENT steps BY one  
            END WHILE  
            RETURN negative one  
        END FUNCTION  
        SET x TO zero  
        SET y TO zero  
        SET total_steps TO zero  
        FOR each tuple height and tx and ty IN trees  
            SET steps TO the RESULT of bfs with arguments the tuple x and y and the tuple tx and ty  
            IF steps EQUALS negative one  
                RETURN negative one  
            END IF  
            INCREMENT total_steps BY steps  
            SET x TO tx  
            SET y TO ty  
        END FOR  
        RETURN total_steps  
    END FUNCTION  
END CLASS
