CLASS Solution  
    FUNCTION constrainedSubsetSum with parameters nums and k  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated n times  
        SET the element at position zero of dp TO the element at position zero of nums  
        SET max_sum TO the element at position zero of nums  
        ASSIGN a new deque containing zero TO q  
        FOR i FROM one TO n MINUS one  
            IF the element at position zero of dp at the element at position zero of q IS GREATER THAN zero  
                SET dp at position i TO the element at position i of nums PLUS dp at the element at position zero of q  
            ELSE  
                SET dp at position i TO the element at position i of nums PLUS zero  
            END IF  
            IF max_sum IS LESS THAN dp at position i  
                SET max_sum TO dp at position i  
            END IF  
            WHILE q IS NOT EMPTY AND dp at position i IS GREATER THAN OR EQUAL TO dp at the element at the last position of q  
                REMOVE the element from the end of q  
            END WHILE  
            APPEND i TO q  
            IF the element at position zero of q EQUALS i MINUS k  
                REMOVE the element from the beginning of q  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
