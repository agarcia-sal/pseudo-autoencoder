CLASS Solution  
    FUNCTION killProcess WITH PARAMETERS pid ppid kill  
        SET tree TO a default dictionary that maps each key to an empty list  
        FOR each child AND parent IN the simultaneous pairing of pid and ppid  
            APPEND child TO the list associated with parent in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the element at position zero of queue  
            REMOVE the element at position zero FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY the list associated with current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
