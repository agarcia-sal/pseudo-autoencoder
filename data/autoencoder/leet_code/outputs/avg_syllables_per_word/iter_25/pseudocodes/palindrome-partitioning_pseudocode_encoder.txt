CLASS Solution  
    FUNCTION partition WITH PARAMETERS s  
        FUNCTION is_palindrome WITH PARAMETERS sub  
            RETURN sub IS EQUAL TO the REVERSE OF sub  
        END FUNCTION  
        FUNCTION backtrack WITH PARAMETERS start path  
            IF start IS EQUAL TO the LENGTH OF s THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF s PLUS one  
                IF is_palindrome WITH ARGUMENT element at positions from start TO end MINUS one of s THEN  
                    CALL backtrack WITH ARGUMENTS end AND path PLUS list containing element at positions from start TO end MINUS one of s  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH ARGUMENTS zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
