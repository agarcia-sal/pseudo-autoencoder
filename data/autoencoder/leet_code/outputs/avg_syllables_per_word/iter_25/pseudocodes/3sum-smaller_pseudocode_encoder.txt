CLASS Solution  
    FUNCTION threeSumSmaller WITH PARAMETERS nums AND target  
        PERFORM sorting of the list nums in ascending order  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR variable i FROM zero TO n MINUS two  
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            WHILE left IS LESS THAN right  
                SET current_sum TO element at position i of nums PLUS element at position left of nums PLUS element at position right of nums  
                IF current_sum IS LESS THAN target  
                    INCREMENT count BY right MINUS left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
