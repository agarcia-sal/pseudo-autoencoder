CLASS Solution  
    FUNCTION kSimilarity WITH PARAMETERS s1 AND s2  
        FUNCTION neighbors WITH PARAMETER s  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD a string composed of substring from position zero TO position i MINUS one of s CONCATENATED WITH element at position j of s CONCATENATED WITH substring from position i PLUS one TO position j MINUS one of s CONCATENATED WITH element at position i of s CONCATENATED WITH substring from position j PLUS one TO end of s  
                END IF  
            END FOR  
        END FUNCTION  
        ASSIGN reference TO an empty double ended queue containing a tuple of s1 and zero  
        ASSIGN reference TO a set containing s1  
        WHILE the queue is not empty  
            REMOVE the first element from the queue and UNPACK it into s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor produced by neighbors with s  
                IF neighbor is not in the visited set  
                    ADD neighbor to the visited set  
                    APPEND a tuple of neighbor and swaps PLUS one to the queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
