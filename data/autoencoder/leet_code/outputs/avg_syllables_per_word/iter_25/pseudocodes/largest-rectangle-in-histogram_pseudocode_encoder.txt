CLASS Solution  
    FUNCTION largestRectangleArea WITH heights  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i and value h IN heights with enumeration  
            WHILE element at position last in stack NOT EQUALS negative one AND h LESS THAN element at position element at position last in stack of heights  
                SET height TO element at position result of removing last element from stack of heights  
                SET width TO i MINUS element at position last in stack MINUS one  
                SET max_area TO the GREATER OF max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
