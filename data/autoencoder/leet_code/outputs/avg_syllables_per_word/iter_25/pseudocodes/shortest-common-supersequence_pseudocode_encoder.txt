CLASS Solution  
    FUNCTION shortestCommonSupersequence WITH PARAMETERS str1 AND str2  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        SET dp TO a list of lists where each inner list contains zero repeated n PLUS one times and the outer list repeats this inner list m PLUS one times  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                    SET the element at position i j of dp TO the element at position i MINUS one j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i j of dp TO the greater value between the element at position i MINUS one j of dp AND the element at position i j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one j of dp GREATER THAN the element at position i j MINUS one of dp THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        REVERSE the list scs to obtain the correct order  
        SET result TO the concatenation of all elements of scs into a single string  
        RETURN result  
    END FUNCTION  
END CLASS
