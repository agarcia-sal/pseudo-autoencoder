CLASS Solution  
    FUNCTION numOfArrays WITH PARAMETERS n m k  
        SET MOD TO one billion PLUS seven  
        SET three dimensional list dp TO a list of length n PLUS one  
        FOR each index i FROM zero TO n  
            SET dp at index i TO a list of length m PLUS one  
            FOR each index j FROM zero TO m  
                SET dp at index i at index j TO a list of length k PLUS one filled with zeros  
            END FOR  
        END FOR  
        FOR each integer j FROM one TO m  
            SET dp at index one at index j at index one TO one  
        END FOR  
        FOR each integer i FROM two TO n  
            FOR each integer j FROM one TO m  
                FOR each integer p FROM one TO k  
                    FOR each integer x FROM one TO j MINUS one  
                        SET dp at index i at index j at index p TO dp at index i at index j at index p PLUS dp at index i MINUS one at index x at index p MINUS one  
                        SET dp at index i at index j at index p TO dp at index i at index j at index p MODULO MOD  
                    END FOR  
                    SET dp at index i at index j at index p TO dp at index i at index j at index p PLUS dp at index i MINUS one at index j at index p MULTIPLIED BY j  
                    SET dp at index i at index j at index p TO dp at index i at index j at index p MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR each integer j FROM one TO m  
            SET result TO result PLUS dp at index n at index j at index k  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
