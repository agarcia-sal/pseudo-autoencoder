CLASS Solution  
    FUNCTION combine WITH PARAMETERS n k  
        FUNCTION backtrack WITH PARAMETERS start path  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one  
                APPEND i TO path  
                CALL backtrack WITH ARGUMENTS i PLUS one AND path  
                REMOVE the LAST ELEMENT FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an EMPTY LIST  
        CALL backtrack WITH ARGUMENTS one AND an EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
