CLASS Solution  
    FUNCTION maxCoins(nums)  
        SET nums TO the list consisting of one PLUS the original nums and one appended at the end  
        SET n TO the length of nums  
        SET dp TO a list of lists each containing n zeros repeated n times  
        FOR length FROM two TO n MINUS one  
            FOR i FROM zero TO n MINUS length MINUS one  
                SET j TO i PLUS length  
                FOR k FROM i PLUS one TO j MINUS one  
                    SET coins TO the element at position i of nums MULTIPLIED BY the element at position k of nums MULTIPLIED BY the element at position j of nums PLUS the element at position i and k of dp PLUS the element at position k and j of dp  
                    IF the element at position i and j of dp LESS THAN coins  
                        SET the element at position i and j of dp TO coins  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position zero and n MINUS one of dp  
    END FUNCTION  
END CLASS
