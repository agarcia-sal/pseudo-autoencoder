CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR i FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string formed by reversing the element at position i of strs  
            IF reversed_str GREATER THAN element at position i of strs  
                SET the element at position i of strs TO reversed_str  
            END IF  
        END FOR  

        SET doubled_strs TO the concatenation of strs with itself  
        SET max_string TO the empty string  

        FOR i FROM zero TO the LENGTH OF strs MINUS one  
            FOR each boolean value is_reversed IN the list containing False and True  
                IF is_reversed IS True  
                    SET current TO the string formed by reversing the element at position i of strs  
                ELSE  
                    SET current TO the element at position i of strs  
                END IF  

                SET remaining TO the string formed by concatenating the elements from position i plus one TO position i plus the LENGTH OF strs minus one of doubled_strs in order  

                FOR j FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring of current from position j TO the end PLUS remaining PLUS the substring of current from the start TO position j MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN max_string  
    END FUNCTION  
END CLASS
