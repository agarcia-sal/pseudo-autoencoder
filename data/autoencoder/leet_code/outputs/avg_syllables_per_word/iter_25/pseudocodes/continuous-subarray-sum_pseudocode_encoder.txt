CLASS Solution  
    FUNCTION checkSubarraySum WITH PARAMETERS nums AND k  
        SET remainder_map TO dictionary WITH key zero AND value minus one  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums WITH INDEXES  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero THEN  
                SET current_sum TO the remainder WHEN current_sum IS DIVIDED BY k  
            END IF  
            IF current_sum IS A KEY IN remainder_map THEN  
                IF i MINUS the value AT key current_sum IN remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                SET the value AT key current_sum IN remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
