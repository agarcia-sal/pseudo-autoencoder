CLASS Solution  
    FUNCTION connectTwoGroups WITH PARAMETER cost  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        INITIALIZE dp AS a two dimensional list with size1 plus one rows and two TO THE power size2 columns  
        FOR each row_index FROM zero TO size1  
            FOR each column_index FROM zero TO two TO THE power size2 MINUS one  
                SET element at position column_index of row_index of dp TO infinity  
            END FOR  
        END FOR  
        SET element at position zero of row zero of dp TO zero  
        
        FOR each i FROM zero TO size1 MINUS one  
            FOR each mask FROM zero TO two TO THE power size2 MINUS one  
                FOR each j FROM zero TO size2 MINUS one  
                    SET new_mask TO mask OR 2 TO THE power j  
                    SET dp at position i plus one and new_mask TO the MINIMUM OF element at position new_mask of row i plus one of dp AND element at position mask of row i of dp PLUS element at position j of element at position i of cost  
                END FOR  
                FOR each j FROM zero TO size2 MINUS one  
                    IF the bitwise AND of mask AND 2 TO THE power j IS NOT zero THEN  
                        SET dp at position i plus one and mask TO the MINIMUM OF element at position mask of row i plus one of dp AND element at position mask with the jth bit toggled of row i plus one of dp PLUS element at position j of element at position i of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position two TO THE power size2 MINUS one of row size1 of dp  
    END FUNCTION  
END CLASS
