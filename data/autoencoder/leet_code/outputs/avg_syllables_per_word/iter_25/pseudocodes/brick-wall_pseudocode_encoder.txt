CLASS Solution  
    FUNCTION leastBricks WITH parameter wall AS LIST OF LIST OF INTEGER  
        SET gap_counts TO an empty default dictionary with default integer zero  
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row EXCEPT the last one  
                INCREMENT total BY brick  
                INCREMENT the value at key total in gap_counts BY one  
            END FOR  
        END FOR  
        SET max_gaps TO the largest value in gap_counts values or zero if gap_counts is empty  
        RETURN the length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
