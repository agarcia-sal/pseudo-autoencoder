CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        FOR each rod IN rods  
            SET new_dp TO a new dictionary copying all entries from dp  
            FOR each key d and value y IN dp entries  
                SET key_for_taller TO d PLUS rod  
                IF key_for_taller IS NOT in new_dp OR new_dp value at key_for_taller IS LESS THAN y  
                    SET new_dp value at key_for_taller TO y  
                END IF  
                SET difference TO the absolute value of d MINUS rod  
                SET sum_shorter TO y PLUS the minimum of d and rod  
                IF difference IS NOT in new_dp OR new_dp value at difference IS LESS THAN sum_shorter  
                    SET new_dp value at difference TO sum_shorter  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN dp value at key zero  
    END FUNCTION  
END CLASS
