CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position i of nums NOT EQUALS the element at position the element at position i of nums MINUS one of nums  
                SET temporary TO the element at position i of nums  
                SET the element at position i of nums TO the element at position temporary MINUS one of nums  
                SET the element at position temporary MINUS one of nums TO temporary  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each index i AND value v IN nums WITH enumeration starting at zero  
            IF v NOT EQUALS i PLUS one  
                APPEND v TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
