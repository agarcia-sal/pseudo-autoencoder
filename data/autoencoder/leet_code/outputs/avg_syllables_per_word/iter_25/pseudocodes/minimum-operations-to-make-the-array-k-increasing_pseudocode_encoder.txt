CLASS Solution  
    FUNCTION kIncreasing WITH PARAMETERS arr AND k  
        FUNCTION longest_increasing_subsequence WITH PARAMETER subarr  
            SET lis TO an empty list  
            FOR each num IN subarr  
                IF lis is empty OR num is greater than or equal to the element at the last position of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position to the right where num should be inserted in lis preserving sorted order  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the length of lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the length of arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO an empty list  
            FOR i FROM start TO n MINUS one IN STEPS OF k  
                APPEND the element at position i of arr TO subarr  
            END FOR  
            
            SET lis_length TO the result of longest_increasing_subsequence WITH subarr  
            INCREMENT total_operations BY the length of subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
