CLASS Solution  
    FUNCTION minMoves(nums AS List OF Integer, k AS Integer) RETURNS Integer  
        SET positions TO an empty list  
        FOR each index AND number IN nums WITH index enumeration  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start AS Integer, end AS Integer) RETURNS Integer  
            SET mid TO the integer division of start PLUS end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                INCREMENT cost BY the absolute difference between element at position i of positions AND median MINUS the difference between mid AND i  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR i FROM zero TO the length of positions MINUS k  
            SET min_cost TO the smaller of min_cost AND the result of calculate_cost with parameters i AND i PLUS k  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
