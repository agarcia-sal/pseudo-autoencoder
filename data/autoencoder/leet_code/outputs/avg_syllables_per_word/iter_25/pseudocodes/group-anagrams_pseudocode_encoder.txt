CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs AS List OF strings  
        SET anagrams TO a default dictionary that returns an empty list for new keys  
        FOR each string s IN strs  
            SET sorted_str TO the result of concatenating all characters obtained by arranging the characters of s in ascending order  
            APPEND s TO the list at key sorted_str in anagrams  
        END FOR  
        RETURN the list of all values contained in anagrams  
    END FUNCTION  
END CLASS
