CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs IS empty THEN  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF the FIRST element of costs  
        IF k EQUALS one THEN  
            IF n EQUALS one THEN  
                RETURN the element at position zero zero of costs  
            ELSE  
                RETURN infinity  
            END IF  
        END IF  
        FOR i FROM one TO n MINUS one  
            SET min_cost TO infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO infinity  
            FOR j FROM zero TO k MINUS one  
                IF the element at position i MINUS one at position j of costs IS LESS THAN min_cost THEN  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO the element at position i MINUS one at position j of costs  
                    SET min_cost_index TO j  
                ELSE IF the element at position i MINUS one at position j of costs IS LESS THAN second_min_cost THEN  
                    SET second_min_cost TO the element at position i MINUS one at position j of costs  
                END IF  
            END FOR  
            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index THEN  
                    INCREMENT the element at position i at position j of costs BY second_min_cost  
                ELSE  
                    INCREMENT the element at position i at position j of costs BY min_cost  
                END IF  
            END FOR  
        END FOR  
        SET minimum_last_row_cost TO infinity  
        FOR each cost IN the last element of costs  
            IF cost IS LESS THAN minimum_last_row_cost THEN  
                SET minimum_last_row_cost TO cost  
            END IF  
        END FOR  
        RETURN minimum_last_row_cost  
    END FUNCTION  
END CLASS
