CLASS Solution  
    FUNCTION maxSubArrayLen WITH PARAMETERS nums AND k  
        SET cumulative_sum_index TO a dictionary with zero mapped to negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND element num IN nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF cumulative_sum MINUS k EXISTS AS A KEY IN cumulative_sum_index  
                SET max_length TO the GREATER OF max_length AND i MINUS the VALUE AT KEY cumulative_sum MINUS k IN cumulative_sum_index  
            END IF  
            IF cumulative_sum DOES NOT EXIST AS A KEY IN cumulative_sum_index  
                SET the VALUE AT KEY cumulative_sum IN cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
