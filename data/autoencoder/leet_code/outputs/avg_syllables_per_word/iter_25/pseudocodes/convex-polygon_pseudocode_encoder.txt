CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN the difference between the product of the difference of element at position zero of a and element at position zero of o multiplied by the difference of element at position one of b and element at position one of o AND the product of the difference of element at position one of a and element at position one of o multiplied by the difference of element at position zero of b and element at position zero of o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three THEN  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position i PLUS one MODULO n of points  
            SET b TO element at position i PLUS two MODULO n of points  
            SET cross TO cross_product with parameters o a and b  
            
            IF cross NOT EQUALS zero THEN  
                IF orientation EQUALS zero THEN  
                    IF cross GREATER THAN zero THEN  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
