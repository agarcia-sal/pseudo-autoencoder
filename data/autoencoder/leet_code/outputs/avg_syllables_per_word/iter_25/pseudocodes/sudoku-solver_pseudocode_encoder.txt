CLASS Solution  
    FUNCTION solveSudoku WITH PARAMETERS board  
        FUNCTION is_valid WITH PARAMETERS board row col num  
            FOR i FROM zero TO eight  
                IF element at position row and position i of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position i and position col of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION of row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION of col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_row PLUS i and position start_col PLUS j of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        FUNCTION solve WITH PARAMETER board  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row and position col of board EQUALS the character for an empty cell  
                        FOR each num IN the sequence of string representations of numbers from one to nine inclusive  
                            IF the result of is_valid WITH board row col num IS True  
                                SET element at position row and position col of board TO num  
                                IF the result of solve WITH board IS True  
                                    RETURN True  
                                END IF  
                                SET element at position row and position col of board TO the character for an empty cell  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        CALL solve WITH board  
    END FUNCTION  
END CLASS
