CLASS Solution  
    FUNCTION maxVacationDays WITH PARAMETERS flights AND days  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
        
        SET dp TO a list of length n filled with negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a DICTIONARY mapping each integer i in RANGE from zero TO n MINUS one TO an empty list  
        FOR each i IN RANGE from zero TO n MINUS one  
            FOR each j IN RANGE from zero TO n MINUS one  
                IF element at position i of flights at position j EQUALS one  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each week IN RANGE from zero TO k MINUS one  
            SET new_dp TO a list of length n filled with negative one  
            FOR each i IN RANGE from zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE  
                END IF  
                
                SET element at position i of new_dp TO the GREATER VALUE BETWEEN element at position i of new_dp AND element at position i of dp PLUS element at position i of days at position week  
                
                FOR each j IN element at position i of graph  
                    SET element at position j of new_dp TO the GREATER VALUE BETWEEN element at position j of new_dp AND element at position i of dp PLUS element at position j of days at position week  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN the GREATER VALUE among all elements of dp  
    END FUNCTION  
END CLASS
