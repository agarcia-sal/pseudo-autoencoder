CLASS Solution  
    FUNCTION partitionLabels WITH PARAMETERS s  
        SET last_occurrence TO a new empty dictionary  
        FOR each index idx FROM zero TO the LENGTH OF s MINUS one  
            SET character char TO element at position idx of s  
            SET the value at key char in last_occurrence TO idx  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET character char TO element at position i of s  
            IF end LESS THAN last_occurrence at key char THEN  
                SET end TO last_occurrence at key char  
            END IF  
            IF i EQUALS end THEN  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
