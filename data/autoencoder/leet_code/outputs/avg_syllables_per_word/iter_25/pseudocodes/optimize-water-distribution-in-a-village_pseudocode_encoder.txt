CLASS Solution  
    FUNCTION minCostToSupplyWater WITH PARAMETERS n wells pipes  
        FOR i FROM zero TO n MINUS one  
            APPEND a list of zero i PLUS one wells element at position i TO pipes  
        END FOR  
        SORT pipes BY the element at position two of each element in ascending order  
        SET parent TO a list of integers ranging from zero TO n  
        FUNCTION find WITH PARAMETER x  
            IF element at position x of parent NOT EQUALS x THEN  
                SET element at position x of parent TO the result of calling find WITH argument element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union WITH PARAMETERS x y  
            SET rootX TO result of calling find WITH argument x  
            SET rootY TO result of calling find WITH argument y  
            IF rootX NOT EQUALS rootY THEN  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each triple house1 house2 cost IN pipes  
            IF calling union WITH arguments house1 house2 returns true THEN  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
