CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        ASSIGN length1 ← LENGTH OF word1  
        ASSIGN length2 ← LENGTH OF word2  
        
        CREATE dp AS a list of (length1 PLUS 1) elements, EACH BEING a list of (length2 PLUS 1) zeros  
        
        ASSIGN outerCounter ← 1  
        WHILE outerCounter <= length1  
            ASSIGN innerCounter ← 1  
            WHILE innerCounter <= length2  
                ASSIGN charFromWord1 ← word1[outerCounter MINUS 1]  
                ASSIGN charFromWord2 ← word2[innerCounter MINUS 1]  
                
                IF NOT (charFromWord1 NOT EQUAL TO charFromWord2) THEN  
                    SET dp[outerCounter][innerCounter] ← dp[outerCounter MINUS 1][innerCounter MINUS 1] PLUS 1  
                ELSE  
                    ASSIGN topValue ← dp[outerCounter MINUS 1][innerCounter]  
                    ASSIGN leftValue ← dp[outerCounter][innerCounter MINUS 1]  
                    
                    IF topValue >= leftValue THEN  
                        SET dp[outerCounter][innerCounter] ← topValue  
                    ELSE  
                        SET dp[outerCounter][innerCounter] ← leftValue  
                    END IF  
                END IF  
                
                INCREMENT innerCounter BY 1  
            END WHILE  
            INCREMENT outerCounter BY 1  
        END WHILE  
        
        ASSIGN lcsLength ← dp[length1][length2]  
        
        RETURN (length1 MINUS lcsLength) PLUS (length2 MINUS lcsLength)  
    END FUNCTION  
END CLASS
