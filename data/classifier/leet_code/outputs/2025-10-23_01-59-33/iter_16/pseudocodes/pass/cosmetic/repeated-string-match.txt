CLASS Solution  
    FUNCTION repeatedStringMatch(a string, b string) RETURNS integer  
        DECLARE minReps INTEGER  
        ASSIGN minReps TO CEILING(LENGTH(b) / LENGTH(a))  
        
        DECLARE candidate STRING  
        ASSIGN candidate TO a  
        DECLARE repeatedCandidate STRING  
        
        ASSIGN repeatedCandidate TO CONCATENATE(candidate REPEATED minReps TIMES)  
        IF CONTAINS(repeatedCandidate, b) THEN  
            RETURN minReps  
        END IF  
        
        ASSIGN repeatedCandidate TO CONCATENATE(candidate REPEATED (minReps + 1) TIMES)  
        IF CONTAINS(repeatedCandidate, b) THEN  
            RETURN (minReps + 1)  
        END IF  
        
        RETURN -1  
    END FUNCTION  
END CLASS
