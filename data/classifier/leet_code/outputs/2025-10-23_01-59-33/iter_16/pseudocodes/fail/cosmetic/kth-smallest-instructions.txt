CLASS Solution  
    FUNCTION kthSmallestPath(destination PARAMETERS destination AND k)  
        SET xPos AS destination AT index 0  
        SET yPos AS destination AT index 1  
        SET resultPath AS an empty list  
        SET stepsTotal AS xPos PLUS yPos  
        SET counter AS 0  
        WHILE counter NOT EQUAL stepsTotal  
            IF yPos NOT EQUAL 0 THEN  
                SET combCount AS combinations(xPos PLUS yPos MINUS 1, yPos MINUS 1)  
                IF k GREATER THAN combCount THEN  
                    APPEND 'V' TO resultPath  
                    SET xPos AS xPos MINUS 1  
                    SET k AS k MINUS combCount  
                ELSE  
                    APPEND 'H' TO resultPath  
                    SET yPos AS yPos MINUS 1  
                END IF  
            ELSE  
                APPEND 'V' TO resultPath  
                SET xPos AS xPos MINUS 1  
            END IF  
            SET counter AS counter PLUS 1  
        END WHILE  
        RETURN combined string of elements in resultPath  
    END FUNCTION  
END CLASS
