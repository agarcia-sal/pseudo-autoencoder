CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses BY the second element of each course representing the last day of the course  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each course IN courses  
            SET duration TO the first element of course representing the duration of the course  
            SET last_day TO the second element of course representing the last day of the course  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day THEN  
                ADD the negative value of duration TO max_heap using a heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND the negation of the first element of max_heap GREATER THAN duration THEN  
                INCREMENT current_end_day BY duration PLUS the heap pop operation result from max_heap  
                ADD the negative value of duration TO max_heap using a heap push operation  
            END IF  
        END FOR  
        RETURN the length of max_heap  
    END FUNCTION  
END CLASS
