CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO an empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
      
        FUNCTION helper(sub_expr)  
            IF the entirety of sub_expr CONSISTS OF DIGITS  
                RETURN a list containing the integer value of sub_expr  
            END IF  
          
            SET results TO an empty list  
            FOR index i FROM zero TO the LENGTH OF sub_expr MINUS one  
                IF the element at position i of sub_expr EQUALS any of the plus symbol OR the minus symbol OR the multiplication symbol  
                    SET left_results TO helper(the substring of sub_expr from position zero TO position i MINUS one)  
                    SET right_results TO helper(the substring of sub_expr from position i PLUS one TO the last position)  
                    FOR each result IN compute(left_results, right_results, the element at position i of sub_expr)  
                        APPEND result TO results  
                    END FOR  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  
      
        RETURN helper(expression)  
    END FUNCTION  
END CLASS
