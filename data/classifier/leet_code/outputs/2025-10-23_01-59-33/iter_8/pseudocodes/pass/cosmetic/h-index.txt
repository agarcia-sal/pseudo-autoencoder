CLASS Solution  
    FUNCTION hIndex(citations)  
        DECLARE sortedList ← SORT citations IN DESCENDING ORDER  
        DECLARE pointer ← 0  
        WHILE pointer < LENGTH OF sortedList DO  
            DECLARE currentCitation ← sortedList[pointer]  
            IF NOT (currentCitation ≥ pointer + 1) THEN  
                RETURN pointer  
            END IF  
            SET pointer ← pointer + 1  
        END WHILE  
        RETURN LENGTH OF sortedList  
    END FUNCTION  
END CLASS
