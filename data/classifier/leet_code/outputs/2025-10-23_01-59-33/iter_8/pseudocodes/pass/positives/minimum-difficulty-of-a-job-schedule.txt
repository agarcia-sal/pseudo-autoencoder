CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM value IN the SUBLIST of jobDifficulty from position i TO the END  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            
            FOR each j FROM i TO n MINUS d  
                SET current_max TO the GREATER BETWEEN current_max AND element at position j OF jobDifficulty  
                SET candidate TO current_max PLUS dp(j PLUS one, d MINUS one)  
                IF candidate LESS THAN min_difficulty  
                    SET min_difficulty TO candidate  
                END IF  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
    END FUNCTION  
END CLASS
