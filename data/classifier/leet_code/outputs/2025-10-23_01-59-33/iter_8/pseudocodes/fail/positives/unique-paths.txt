CLASS Solution  
    FUNCTION uniquePaths(m n)  
        SET dp TO a two dimensional list with m rows and n columns where each element is set to one  
        FOR i FROM two TO m  
            FOR j FROM two TO n  
                SET element at position i of dp TO the list at position i of dp  
                SET element at position j of element at position i of dp TO the sum of the element at position i MINUS one of dp at position j AND the element at position i of dp at position j MINUS one  
            END FOR  
        END FOR  
        RETURN the element at position m MINUS one of dp at position n MINUS one  
    END FUNCTION  
END CLASS
