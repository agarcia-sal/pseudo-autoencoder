CLASS Solution  
    FUNCTION numBusesToDestination(routes PARAMETER, source PARAMETER, target PARAMETER)  
        IF source EQUALS target  
            RETURN zero  
        END IF  

        SET stop_to_buses TO an empty mapping from keys to empty lists  
        FOR each bus WITH index bus_index AND route IN routes  
            FOR each stop IN route  
                APPEND bus_index TO the list corresponding to stop IN stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO an empty double ended queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus IN the list of buses at the source stop FROM stop_to_buses  
            APPEND the tuple consisting of bus AND one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the first element FROM bus_queue AND ASSIGN the elements TO current_bus AND bus_count  

            FOR each stop IN the route at position current_bus OF routes  
                IF stop EQUALS target  
                    RETURN bus_count  
                END IF  

                IF stop NOT IN visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list of buses at stop FROM stop_to_buses  
                        IF next_bus NOT IN visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND the tuple consisting of next_bus AND bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
