CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        DEFINE CONST_MOD AS 1000000007  
        
        DEFINE countA AS 1  
        DEFINE countE AS 1  
        DEFINE countI AS 1  
        DEFINE countO AS 1  
        DEFINE countU AS 1  
        
        DEFINE FUNCTION iterateCounts(index) RETURNS VOID  
            IF index GREATER THAN n THEN  
                RETURN  
            END IF  
            
            DEFINE tmpA AS (countE + countI + countU)  
            DEFINE tmpE AS (countA + countI)  
            DEFINE tmpI AS (countE + countO)  
            DEFINE tmpO AS countI  
            DEFINE tmpU AS (countI + countO)  
            
            countA = tmpA MOD CONST_MOD  
            countE = tmpE MOD CONST_MOD  
            countI = tmpI MOD CONST_MOD  
            countO = tmpO MOD CONST_MOD  
            countU = tmpU MOD CONST_MOD  
            
            CALL iterateCounts(index + 1)  
        END FUNCTION  
        
        CALL iterateCounts(2)  
        
        RETURN (countA + countE + countI + countO + countU) MOD CONST_MOD  
    END FUNCTION  
END CLASS
