CLASS Solution  
    FUNCTION numberOfArrays(s PARAMETER string, k PARAMETER integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a list of n PLUS one zeros  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF character at position j of s NOT EQUALS zero AND  
                   the integer value of substring from position j TO position i MINUS one of s GREATER THAN OR EQUAL TO one AND  
                   the integer value of substring from position j TO position i MINUS one of s LESS THAN OR EQUAL TO k  
                THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
