CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target_inner)  
            IF target_inner EQUALS zero  
                RETURN negative one  
            END IF  
            
            IF target_inner LESS THAN x  
                RETURN the minimum BETWEEN the expression two MULTIPLIED BY target_inner MINUS one AND the expression two MULTIPLIED BY the difference BETWEEN x MINUS target_inner  
            END IF  
            
            SET power TO one  
            WHILE x RAISED TO power IS LESS THAN OR EQUAL TO target_inner  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            
            SET result TO the value returned BY dp WITH the argument target_inner MINUS x RAISED TO power PLUS power  
            
            IF the expression x RAISED TO the sum OF power PLUS one MINUS target_inner IS LESS THAN target_inner  
                SET result TO the minimum BETWEEN result AND the value returned BY dp WITH the argument x RAISED TO the sum OF power PLUS one MINUS target_inner PLUS power PLUS one  
            END IF  
            
            RETURN result  
        END FUNCTION  
        
        RETURN the value returned BY dp WITH the argument target  
    END FUNCTION  
END CLASS
