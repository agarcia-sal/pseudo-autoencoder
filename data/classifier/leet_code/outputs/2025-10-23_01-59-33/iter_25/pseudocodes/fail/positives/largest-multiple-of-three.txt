CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency counter of digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the remainder after dividing total_sum BY three  
        IF remainder EQUALS one  
            FOR each digit d IN the list consisting of one four seven  
                IF the frequency count of digit d IS GREATER THAN zero  
                    DECREMENT the frequency count of digit d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list consisting of two five eight  
                    IF the frequency count of digit d IS GREATER THAN one  
                        DECREMENT the frequency count of digit d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        IF remainder EQUALS two  
            FOR each digit d IN the list consisting of two five eight  
                IF the frequency count of digit d IS GREATER THAN zero  
                    DECREMENT the frequency count of digit d BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list consisting of one four seven  
                    IF the frequency count of digit d IS GREATER THAN one  
                        DECREMENT the frequency count of digit d BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        SET result TO an empty list  
        FOR each digit d FROM nine DOWN TO zero  
            APPEND the string representation of digit d REPEATED frequency count of digit d times TO result  
        END FOR  
        SET final_number TO the concatenation of all elements in result  
        IF the length of final_number IS GREATER THAN zero AND the element at position zero of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
