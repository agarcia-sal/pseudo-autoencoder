CLASS Solution  
    FUNCTION repeatedStringMatch(a string, b string) RETURNS integer  
        SET minimum repetitions TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF string b IS a substring of string a repeated minimum repetitions times  
            RETURN minimum repetitions  
        END IF  
        IF string b IS a substring of string a repeated minimum repetitions PLUS one times  
            RETURN minimum repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
