CLASS Solution
    FUNCTION isRectangleCover(list_of_rectangles)
        SET X1 TO positive infinity
        SET Y1 TO positive infinity
        SET X2 TO negative infinity
        SET Y2 TO negative infinity
        
        ASSIGN points TO an empty set
        SET total_area TO zero
        
        FOR each rectangle IN list_of_rectangles
            ASSIGN x1 TO element at position zero of rectangle
            ASSIGN y1 TO element at position one of rectangle
            ASSIGN x2 TO element at position two of rectangle
            ASSIGN y2 TO element at position three of rectangle
            
            SET X1 TO the minimum of X1 AND x1
            SET Y1 TO the minimum of Y1 AND y1
            SET X2 TO the maximum of X2 AND x2
            SET Y2 TO the maximum of Y2 AND y2
            
            SET rectangle_area TO (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)
            INCREMENT total_area BY rectangle_area
            
            SET corner_points TO a list containing tuples of:
                tuple of x1 AND y1
                tuple of x1 AND y2
                tuple of x2 AND y1
                tuple of x2 AND y2
            
            FOR each point IN corner_points
                IF point IS IN points
                    REMOVE point FROM points
                ELSE
                    ADD point TO points
                END IF
            END FOR
        END FOR
        
        IF total_area IS NOT EQUAL TO (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1)
            RETURN false
        END IF
        
        SET expected_points TO a set containing tuples of:
            tuple of X1 AND Y1
            tuple of X1 AND Y2
            tuple of X2 AND Y1
            tuple of X2 AND Y2
        
        IF points IS NOT EQUAL TO expected_points
            RETURN false
        END IF
        
        RETURN true
    END FUNCTION
END CLASS
