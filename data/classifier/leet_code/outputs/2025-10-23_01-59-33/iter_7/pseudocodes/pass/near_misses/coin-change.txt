CLASS Solution
    FUNCTION coinChange(coins, amount)
        ASSIGN amount + 1 TO alpha
        CREATE dp WITH alpha ELEMENTS ALL INITIALIZED TO alpha
        ASSIGN 0 TO dp[0]

        FUNCTION recurBeta(index)
            IF index > amount THEN RETURN END IF

            SET cursor TO 0
            WHILE cursor < LENGTH(coins)
                LET delta = index - coins[cursor]
                IF delta >= 0 THEN
                    LET candidate = dp[delta] + 1
                    IF candidate < dp[index] THEN
                        dp[index] = candidate
                    END IF
                END IF
                INCREMENT cursor
            END WHILE

            CALL recurBeta(index + 1)
        END FUNCTION

        CALL recurBeta(1)

        IF dp[amount] != alpha THEN
            RETURN dp[amount]
        END IF

        RETURN -1
    END FUNCTION
END CLASS
