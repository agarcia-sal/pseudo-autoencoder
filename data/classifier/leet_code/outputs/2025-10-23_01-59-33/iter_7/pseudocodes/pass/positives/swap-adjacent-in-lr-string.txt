CLASS Solution  
    FUNCTION canTransform(start parameter, end parameter)  
        SET start_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF start parameter MINUS one  
            SET character_at_index TO element at position index of start parameter  
            IF character_at_index NOT EQUALS the character X  
                APPEND tuple of character_at_index AND index TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO empty list  
        FOR index FROM zero TO the LENGTH OF end parameter MINUS one  
            SET character_at_index TO element at position index of end parameter  
            IF character_at_index NOT EQUALS the character X  
                APPEND tuple of character_at_index AND index TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair of tuples first_pair AND second_pair IN the PAIRWISE ITERATION OF start_filtered AND end_filtered  
            SET character_one TO the first element of first_pair  
            SET index_one TO the second element of first_pair  
            SET character_two TO the first element of second_pair  
            SET index_two TO the second element of second_pair  
            IF character_one NOT EQUALS character_two  
                RETURN false  
            END IF  
            IF character_one EQUALS the character L AND index_one LESS THAN index_two  
                RETURN false  
            END IF  
            IF character_one EQUALS the character R AND index_one GREATER THAN index_two  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
