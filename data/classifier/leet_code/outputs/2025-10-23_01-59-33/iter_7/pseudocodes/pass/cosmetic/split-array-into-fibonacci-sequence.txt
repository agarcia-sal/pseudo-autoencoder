CLASS Solution
    FUNCTION splitIntoFibonacci(num)
        FUNCTION backtrack(start, path)
            ASSIGN sizeNum TO LENGTH(num)
            ASSIGN sizePath TO LENGTH(path)
            IF NOT (start NOT EQUAL sizeNum OR sizePath LESS THAN 3)
                RETURN path
            END IF
            ASSIGN idx TO start + 1
            WHILE idx <= sizeNum
                IF NOT ((num[start] = '0') AND (idx > start + 1))
                    ASSIGN substrVal TO SUBSTRING(num, start, idx)
                    ASSIGN nextVal TO TO_INTEGER(substrVal)
                    IF nextVal < 2 ^ 31
                        IF (sizePath < 2) OR (nextVal = path[sizePath - 1] + path[sizePath - 2])
                            ASSIGN attempt TO backtrack(idx, CONCAT(path, [nextVal]))
                            IF attempt != []
                                RETURN attempt
                            END IF
                        END IF
                    ELSE
                        BREAK
                    END IF
                END IF
                idx = idx + 1
            END WHILE
            RETURN []
        END FUNCTION
        RETURN backtrack(0, [])
    END FUNCTION
END CLASS
