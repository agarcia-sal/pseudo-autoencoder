m, n = len(s1), len(s2)
dp = matrix (m+1) x (n+1) initialized to 0

for j in 1..n: dp[0][j] = dp[0][j-1] + ASCII(s2[j-1])
for i in 1..m: dp[i][0] = dp[i-1][0] + ASCII(s1[i-1])

for i in 1..m:
  for j in 1..n:
    if s1[i-1] == s2[j-1]:
      dp[i][j] = dp[i-1][j-1]
    else:
      dp[i][j] = min(dp[i-1][j] + ASCII(s1[i-1]), dp[i][j-1] + ASCII(s2[j-1]))

return dp[m][n]
