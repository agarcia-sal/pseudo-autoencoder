CLASS Solution  
    FUNCTION numTilings(n)  
        SET MOD to 1_000_000_007  
        
        SET dp AS a list initialized with zeros, size n PLUS 1  
        SET dp1 AS a list initialized with zeros, size n PLUS 1  
        
        ASSIGN 1 TO dp[0]  
        ASSIGN 1 TO dp[1]  
        IF n LESS THAN 2 THEN  
            GOTO after_initialization  
        ELSE  
            ASSIGN 2 TO dp[2]  
            ASSIGN 1 TO dp1[2]  
        END IF  
        LABEL after_initialization  
        
        SET index TO 3  
        WHILE index LESS THAN OR EQUAL TO n DO  
            SET part1 TO (dp[index - 1] + dp[index - 2]) MOD MOD  
            SET part2 TO (2 * dp1[index - 1]) MOD MOD  
            ASSIGN (part1 + part2) MOD MOD TO dp[index]  
            
            SET dp1[index] TO ((dp[index - 2] + dp1[index - 1]) MOD MOD)  
            INCREMENT index BY 1  
        END WHILE  
        
        RETURN dp[n]  
    END FUNCTION  
END CLASS
