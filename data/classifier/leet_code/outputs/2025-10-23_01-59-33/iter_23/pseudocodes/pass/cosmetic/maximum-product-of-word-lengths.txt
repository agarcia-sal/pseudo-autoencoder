CLASS Solution
    FUNCTION maxProduct(words)
        DECLARE char_sets AS empty list
        DECLARE idx1 AS 0
        WHILE idx1 < LENGTH(words)
            DECLARE temp_set AS empty set
            DECLARE pos AS 0
            WHILE pos < LENGTH(words[idx1])
                SET temp_set TO temp_set UNION { words[idx1][pos] }
                SET pos TO pos + 1
            END WHILE
            APPEND temp_set TO char_sets
            SET idx1 TO idx1 + 1
        END WHILE

        SET highest_product TO 0

        DECLARE first_pointer AS 0
        WHILE first_pointer < LENGTH(words) - 1
            DECLARE second_pointer AS first_pointer + 1
            WHILE second_pointer < LENGTH(words)
                DECLARE intersection_check AS char_sets[first_pointer] INTERSECT char_sets[second_pointer]
                IF LENGTH(intersection_check) = 0 THEN
                    DECLARE candidate_product AS (LENGTH(words[first_pointer]) * LENGTH(words[second_pointer]))
                    IF candidate_product > highest_product THEN
                        SET highest_product TO candidate_product
                    END IF
                END IF
                SET second_pointer TO second_pointer + 1
            END WHILE
            SET first_pointer TO first_pointer + 1
        END WHILE

        RETURN highest_product
    END FUNCTION
END CLASS
