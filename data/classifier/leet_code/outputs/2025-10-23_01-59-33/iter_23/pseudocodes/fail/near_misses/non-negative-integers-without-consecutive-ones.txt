CLASS Solution
    FUNCTION findIntegers(n)
        FUNCTION buildDPArray(size, dpList) RETURNS VOID
            IF size < 2 THEN
                RETURN
            END IF
            dpList[0] = 1
            dpList[1] = 2
            PROCEDURE fillDP(current)
                IF current >= size + 1 THEN
                    RETURN
                ELSE
                    dpList[current] = dpList[current - 1] + dpList[current - 2]
                    fillDP(current + 1)
                END IF
            END PROCEDURE
            fillDP(2)
        END FUNCTION

        SET binStr TO empty queue
        PROCEDURE convertToBinary(num)
            IF num = 0 THEN
                ENQUEUE '0' INTO binStr
                RETURN
            END IF
            SET bitsStack TO empty stack
            WHILE num > 0
                PUSH character((num MOD 2) + '0') INTO bitsStack
                SET num TO FLOOR(num / 2)
            END WHILE
            WHILE NOT EMPTY(bitsStack)
                ENQUEUE POP(bitsStack) INTO binStr
            END WHILE
        END PROCEDURE

        convertToBinary(n)
        SET lengthVar TO SIZE(binStr)
        SET dp TO new array of integers with length lengthVar + 1
        buildDPArray(lengthVar, dp)

        SET sumResult TO 0
        SET lastBitIndicator TO 0
        FUNCTION analyzeBits(index)
            IF index = lengthVar THEN
                RETURN
            END IF
            DEQUEUE currentBit FROM binStr
            IF currentBit = '1' THEN
                sumResult = sumResult + dp[lengthVar - index - 1]
                IF lastBitIndicator = 1 THEN
                    sumResult = sumResult - 1
                    RETURN
                END IF
                lastBitIndicator = 1
            ELSE
                lastBitIndicator = 0
            END IF
            analyzeBits(index + 1)
            ENQUEUE currentBit INTO binStr
        END FUNCTION

        analyzeBits(0)
        RETURN sumResult + 1
    END FUNCTION
END CLASS
