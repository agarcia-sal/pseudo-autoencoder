CLASS Solution
    FUNCTION evalRPN(tokens)
        INITIALIZE ledger WITH empty collection
        INITIALIZE index TO 0
        WHILE index LESS THAN LENGTH OF tokens
            LET current_token BE tokens[index]
            INCREMENT index BY 1
            SWITCH current_token
                CASE plus, minus, multiply, divide:
                    LET right_operand BE ledger.REMOVE_LAST_ELEMENT()
                    LET left_operand BE ledger.REMOVE_LAST_ELEMENT()
                    IF current_token EQUALS plus THEN
                        SET operation_result TO left_operand PLUS right_operand
                    ELSE IF current_token EQUALS minus THEN
                        SET operation_result TO left_operand MINUS right_operand
                    ELSE IF current_token EQUALS multiply THEN
                        SET operation_result TO left_operand MULTIPLIED BY right_operand
                    ELSE /* current_token EQUALS divide */
                        SET intermediate_result TO left_operand DIVIDED BY right_operand
                        SET operation_result TO INTEGER_VALUE_OF(intermediate_result)
                    END IF
                    ledger.ADD_ELEMENT(operation_result)
                DEFAULT:
                    LET numeric_value BE INTEGER_VALUE_OF(current_token)
                    ledger.ADD_ELEMENT(numeric_value)
            END SWITCH
        END WHILE
        RETURN ledger[0]
    END FUNCTION
END CLASS
