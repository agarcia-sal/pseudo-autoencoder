CLASS Solution
    FUNCTION canConvert WITH PARAMETERS str1 str2
        IF NOT (str1 IS NOT str2) THEN
            RETURN True
        END IF

        DEFINE char_to_char_map AS empty dictionary

        SET index_var TO 0
        WHILE index_var LESS THAN LENGTH(str1)
            ASSIGN charA TO str1 AT index_var
            ASSIGN charB TO str2 AT index_var

            IF charA IN char_to_char_map THEN
                IF NOT (char_to_char_map[charA] EQUALS charB) THEN
                    RETURN False
                END IF
            ELSE
                ASSIGN charB TO char_to_char_map AT charA
            END IF

            INCREMENT index_var BY 1
        END WHILE

        IF LENGTH(SET OF ELEMENTS OF str2) GREATER THAN OR EQUAL TO 26 THEN
            RETURN False
        END IF

        RETURN True
    END FUNCTION
END CLASS
