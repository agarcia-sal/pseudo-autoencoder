CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST of elements each being a PAIR consisting of one and zero MINUS one and zero zero and one zero and MINUS one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two-dimensional list of size m by n each element set TO infinity  
            SET queue TO an empty double-ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the TRIPLE consisting of i j and zero TO queue  
                        SET element at position i and j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue is not empty  
                REMOVE the first element and ASSIGN its components TO x y and t  
                FOR each dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx and ny of grid EQUALS zero AND element at position nx and ny of fire_times EQUALS infinity  
                        SET element at position nx and ny of fire_times TO t PLUS one  
                        APPEND the TRIPLE consisting of nx ny and t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a double-ended queue containing a single TRIPLE consisting of zero zero and wait  
            SET visited TO a set containing the PAIR zero zero  
            
            WHILE queue is not empty  
                REMOVE the first element and ASSIGN its components TO x y and t  
                FOR each dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the PAIR nx ny NOT IN visited AND element at position nx and ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx and ny of fire_times GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN true  
                            ELSE  
                                RETURN false  
                            END IF  
                        END IF  
                        IF element at position nx and ny of fire_times GREATER THAN t PLUS one  
                            ADD the PAIR nx ny TO visited  
                            APPEND the TRIPLE consisting of nx ny and t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN false  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS right DIVIDED BY two rounded down to the nearest integer  
            IF can_escape(mid)  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result NOT EQUALS m MULTIPLIED BY n  
            RETURN result  
        ELSE  
            RETURN one followed by nine zeros  
        END IF  
    END FUNCTION  
END CLASS
