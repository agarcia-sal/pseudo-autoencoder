CLASS Solution  
    FUNCTION new21Game(n INTEGER PARAMETER k INTEGER PARAMETER maxPts INTEGER PARAMETER) RETURNS FLOAT  
        FUNCTION dfs(i INTEGER PARAMETER) RETURNS FLOAT  
            IF i GREATER THAN OR EQUAL TO k THEN  
                IF i LESS THAN OR EQUAL TO n THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one THEN  
                IF n MINUS k PLUS one LESS THAN maxPts THEN  
                    RETURN (n MINUS k PLUS one) DIVIDED BY maxPts  
                ELSE  
                    RETURN one  
                END IF  
            END IF  
            SET first_call TO dfs(i PLUS one)  
            SET second_call TO dfs(i PLUS maxPts PLUS one)  
            SET result TO first_call PLUS (first_call MINUS second_call) DIVIDED BY maxPts  
            RETURN result  
        END FUNCTION  
        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
