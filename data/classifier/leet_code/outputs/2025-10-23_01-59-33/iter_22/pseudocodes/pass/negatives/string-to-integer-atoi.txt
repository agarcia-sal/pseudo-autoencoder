```
SET INT_MAX = 2^31 - 1, INT_MIN = -2^31
INIT res = 0, i = 0, n = len(s), sign = 1
WHILE i < n AND s[i] = ' ' : i += 1
IF i < n AND s[i] IN ['+', '-'] : sign = -1 IF s[i] = '-' ELSE 1; i += 1
WHILE i < n AND s[i].isdigit():
    d = int(s[i])
    IF res > (INT_MAX - d)//10: RETURN INT_MAX IF sign = 1 ELSE INT_MIN
    res = res*10 + d
    i += 1
RETURN sign * res
```
