CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO a mapping of digit frequencies in digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the result of total_sum MODULO three  
        
        IF remainder EQUALS one  
            FOR each digit IN the list consisting of one four seven  
                IF the frequency of digit in count GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no digit was removed in the previous loop  
                FOR each digit IN the list consisting of two five eight  
                    IF the frequency of digit in count GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit IN the list consisting of two five eight  
                IF the frequency of digit in count GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF no digit was removed in the previous loop  
                FOR each digit IN the list consisting of one four seven  
                    IF the frequency of digit in count GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
        
        SET result TO an empty list  
        FOR digit FROM nine DOWN TO zero  
            APPEND the character representation of digit REPEATED the frequency of digit in count times TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF final_number IS empty OR the character at position one of final_number EQUALS zero  
            RETURN the character representation of zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
