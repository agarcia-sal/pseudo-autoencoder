CLASS Solution  
    FUNCTION kthGrammar(n parameter, k parameter)  
        IF n parameter EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two MULTIPLIED BY two RAISED TO THE POWER OF n parameter MINUS two MINUS one  
        IF k parameter LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN the result of calling kthGrammar with parameters n parameter MINUS one AND k parameter  
        ELSE  
            RETURN one MINUS the result of calling kthGrammar with parameters n parameter MINUS one AND k parameter MINUS length_of_previous_row  
        END IF  
    END FUNCTION  
END CLASS
