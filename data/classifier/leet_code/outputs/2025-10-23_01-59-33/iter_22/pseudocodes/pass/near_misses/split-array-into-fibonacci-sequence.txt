CLASS Solution
    FUNCTION splitIntoFibonacci(string_number)
        FUNCTION backtrack(start_position, current_path)
            IF start_position EQUALS the LENGTH OF string_number AND the LENGTH OF current_path GREATER THAN OR EQUAL TO three
                RETURN current_path
            END IF
            FOR end_position FROM start_position PLUS one TO the LENGTH OF string_number INCLUSIVE
                IF the ELEMENT AT POSITION start_position OF string_number EQUALS the CHARACTER zero AND end_position GREATER THAN start_position PLUS one
                    CONTINUE TO the NEXT iteration
                END IF
                SET next_number TO the INTEGER VALUE OF the SUBSTRING FROM position start_position TO position end_position MINUS one OF string_number
                IF next_number GREATER THAN OR EQUAL TO two TO THE POWER OF thirty one
                    BREAK the LOOP
                END IF
                IF the LENGTH OF current_path LESS THAN two OR next_number EQUALS the ELEMENT AT the LAST POSITION OF current_path PLUS the ELEMENT AT the SECOND TO LAST POSITION OF current_path
                    SET result TO the CALL OF backtrack WITH PARAMETERS end_position AND the CONCATENATION OF current_path AND the LIST CONTAINING next_number
                    IF result IS NOT EMPTY
                        RETURN result
                    END IF
                END IF
            END FOR
            RETURN the EMPTY LIST
        END FUNCTION
        RETURN the CALL OF backtrack WITH PARAMETERS zero AND the EMPTY LIST
    END FUNCTION
END CLASS
