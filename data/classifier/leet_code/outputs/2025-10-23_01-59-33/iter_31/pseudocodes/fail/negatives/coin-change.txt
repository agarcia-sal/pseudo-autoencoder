CLASS Solution
    FUNCTION coinChange(list_of_coins, target_amount)
        SET dynamic_programming_array TO a list WHERE each element is SET TO the target_amount PLUS one AND the LENGTH OF the list IS the target_amount PLUS one
        SET element at position zero OF dynamic_programming_array TO zero
        
        FOR current_amount FROM one TO target_amount INCLUSIVE
            FOR each coin IN list_of_coins
                IF current_amount MINUS coin IS GREATER THAN OR EQUAL TO zero THEN
                    SET element at position current_amount OF dynamic_programming_array TO the MINIMUM BETWEEN element at position current_amount OF dynamic_programming_array AND element at position current_amount MINUS coin OF dynamic_programming_array PLUS one
                END IF
            END FOR
        END FOR
        
        IF element at position target_amount OF dynamic_programming_array IS NOT EQUAL TO target_amount PLUS one THEN
            RETURN element at position target_amount OF dynamic_programming_array
        ELSE
            RETURN negative one
        END IF
    END FUNCTION
END CLASS
