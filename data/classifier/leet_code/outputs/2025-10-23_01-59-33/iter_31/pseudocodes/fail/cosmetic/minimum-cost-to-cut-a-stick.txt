CLASS Solution
    FUNCTION minCost(n, cuts)
        SET boundaries TO an ordered collection formed by merging cuts WITH [0, n]
        
        FUNCTION dp(start, end)
            IF NOT (end - start > 1) THEN
                OUTPUT 0
            END IF
            
            SET lowest_cost TO a very large number
            SET index_tracker TO start + 1
            WHILE index_tracker < end
                SET segment_cost TO (boundaries[end] - boundaries[start]) + dp(start, index_tracker) + dp(index_tracker, end)
                
                IF NOT (lowest_cost <= segment_cost) THEN
                    SET lowest_cost TO segment_cost
                END IF
                
                ADD 1 TO index_tracker
            END WHILE
            
            OUTPUT lowest_cost
        END FUNCTION
        
        OUTPUT dp(0, LENGTH(boundaries) - 1)
    END FUNCTION
END CLASS
