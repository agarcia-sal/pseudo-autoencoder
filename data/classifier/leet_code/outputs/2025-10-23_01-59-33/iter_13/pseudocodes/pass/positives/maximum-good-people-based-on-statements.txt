CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the NUMBER ONE SHIFTED LEFT BY n MINUS one  
            SET good TO a LIST OF n ELEMENTS EACH SET TO FALSE  
            SET bad TO a LIST OF n ELEMENTS EACH SET TO FALSE  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND the NUMBER ONE SHIFTED LEFT BY j EQUALS the NUMBER ONE SHIFTED LEFT BY j  
                    SET the element at position j of good TO TRUE  
                ELSE  
                    SET the element at position j of bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR j FROM zero TO n MINUS one WHILE valid HOLDS  
                IF the element at position j of good EQUALS TRUE  
                    FOR k FROM zero TO n MINUS one  
                        IF the element at position j of statements AT position k EQUALS zero AND the element at position k of good EQUALS TRUE OR the element at position j of statements AT position k EQUALS one AND the element at position k of bad EQUALS TRUE  
                            SET valid TO FALSE  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            IF valid EQUALS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF elements in good WHERE each TRUE IS COUNTED AS one AND each FALSE AS zero  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
