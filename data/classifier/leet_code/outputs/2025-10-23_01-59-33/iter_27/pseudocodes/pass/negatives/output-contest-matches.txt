CLASS Solution
    FUNCTION findContestMatch(number_of_teams)
        SET list_of_teams TO CALL generate_string_list FROM one TO number_of_teams
        WHILE number_of_teams GREATER THAN one
            FOR index FROM zero TO number_of_teams DIVIDED BY two MINUS one
                SET element at position index of list_of_teams TO CONCATENATION OF LEFT PARENTHESIS FOLLOWED BY element at position index of list_of_teams FOLLOWED BY COMMA FOLLOWED BY element at position number_of_teams MINUS one MINUS index of list_of_teams FOLLOWED BY RIGHT PARENTHESIS
            END FOR
            SET number_of_teams TO number_of_teams DIVIDED BY two
        END WHILE
        RETURN element at position zero of list_of_teams
    END FUNCTION
END CLASS
