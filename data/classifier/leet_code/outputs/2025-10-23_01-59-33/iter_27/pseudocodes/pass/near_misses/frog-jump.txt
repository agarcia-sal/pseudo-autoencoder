CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the set containing all elements of stones  
        SET memo TO empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            
            IF the pair of position AND jump IS a key in memo  
                RETURN the value in memo associated with the pair of position AND jump  
            END IF  
            
            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND the sum of position PLUS next_jump IS an element of stone_set  
                    IF can_jump_to(the sum of position PLUS next_jump, next_jump) RETURNS True  
                        SET the value in memo associated with the pair of position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the value in memo associated with the pair of position AND jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the element at position one of stones, zero)  
    END FUNCTION  
END CLASS
