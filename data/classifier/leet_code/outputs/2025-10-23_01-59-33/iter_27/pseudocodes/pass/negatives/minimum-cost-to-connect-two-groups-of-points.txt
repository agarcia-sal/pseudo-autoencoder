CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO LENGTH OF cost  
        SET size2 TO LENGTH OF cost AT INDEX 0  
        SET dp TO 2D list of size size1 plus 1 by 2 raised to size2 filled with infinity  
        SET dp AT INDEX 0 AND 0 TO 0  
  
        FOR i FROM 0 TO size1 MINUS 1  
            FOR mask FROM 0 TO 2 RAISED TO size2 MINUS 1  
                FOR j FROM 0 TO size2 MINUS 1  
                    SET dp AT INDEX i plus 1 AND mask OR (1 SHIFTED LEFT BY j) TO THE MINIMUM OF dp AT INDEX i plus 1 AND mask OR (1 SHIFTED LEFT BY j) AND dp AT INDEX i AND mask PLUS cost AT INDEX i AND j  
                END FOR  
  
                FOR j FROM 0 TO size2 MINUS 1  
                    IF mask BITWISE AND (1 SHIFTED LEFT BY j) IS NOT ZERO  
                        SET dp AT INDEX i plus 1 AND mask TO THE MINIMUM OF dp AT INDEX i plus 1 AND mask AND dp AT INDEX i plus 1 AND mask XOR (1 SHIFTED LEFT BY j) PLUS cost AT INDEX i AND j  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
  
        RETURN dp AT INDEX size1 AND 2 RAISED TO size2 MINUS 1  
    END FUNCTION  
END CLASS
