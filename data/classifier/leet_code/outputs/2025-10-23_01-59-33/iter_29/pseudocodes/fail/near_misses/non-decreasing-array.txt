CLASS Solution  
    FUNCTION checkPossibility(nums)  
        ASSIGN false TO alterationFlag  
        ASSIGN 1 TO forwardIndex  
        WHILE forwardIndex LESS THAN LENGTH OF nums MINUS 1 DO  
            IF nums[forwardIndex] LESS THAN nums[forwardIndex MINUS 1] THEN  
                IF alterationFlag IS true THEN  
                    RETURN false  
                END IF  
                ASSIGN true TO alterationFlag  
                IF forwardIndex LESS THAN 2 OR nums[forwardIndex] GREATER THAN OR EQUAL TO nums[forwardIndex MINUS 2] THEN  
                    ASSIGN nums[forwardIndex] TO nums[forwardIndex MINUS 1]  
                ELSE  
                    ASSIGN nums[forwardIndex MINUS 1] TO nums[forwardIndex]  
                END IF  
            END IF  
            INCREMENT forwardIndex BY 1  
        END WHILE  
        RETURN true  
    END FUNCTION  
END CLASS
