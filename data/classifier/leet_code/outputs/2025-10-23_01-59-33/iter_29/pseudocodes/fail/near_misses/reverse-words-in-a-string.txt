CLASS Solution  
    FUNCTION reverseWords(s)  
        DECLARE wordList AS list  
        wordList ← EMPTY LIST  
        DECLARE index AS INTEGER  
        index ← 0  
        WHILE index < LENGTH(s) DO  
            WHILE index < LENGTH(s) AND s[index] = ' ' DO  
                index ← index + 1  
            END WHILE  
            IF index = LENGTH(s) THEN  
                BREAK  
            END IF  
            DECLARE startPos AS INTEGER  
            startPos ← index  
            WHILE index < LENGTH(s) AND s[index] ≠ ' ' DO  
                index ← index + 1  
            END WHILE  
            SET word TO SUBSTRING(s, startPos, index - startPos)  
            APPEND word TO wordList  
        END WHILE  
        DECLARE reversedList AS list  
        reversedList ← EMPTY LIST  
        DECLARE pos AS INTEGER  
        pos ← LENGTH(wordList) - 1  
        WHILE pos ≥ 0 DO  
            APPEND wordList[pos] TO reversedList  
            pos ← pos - 1  
        END WHILE  
        DECLARE result AS STRING  
        result ← ""  
        FOR EACH element IN reversedList DO  
            IF result = "" THEN  
                result ← element  
            ELSE  
                result ← result + " " + element  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
