CLASS Solution  
    FUNCTION frequencySort WITH PARAMETER s  
        SET frequency TO the character frequency mapping of s  
        SET sorted_characters TO the list of characters in frequency arranged in descending order according to their frequency values  
        SET result TO an empty string  
        FOR each character IN sorted_characters  
            APPEND the character repeated the number of times equal to the frequency of the character TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
