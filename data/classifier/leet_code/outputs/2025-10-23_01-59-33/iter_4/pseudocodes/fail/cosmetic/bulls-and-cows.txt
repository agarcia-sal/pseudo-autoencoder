CLASS Solution  
    FUNCTION getHint(secret PARAMETER, guess PARAMETER)  
        IMPORT a data structure offering a dictionary with zero as the default for any missing key  
        
        SET alpha TO 0  
        SET beta TO 0  
        
        CREATE secret_map AS a dictionary defaulting to zero  
        CREATE guess_map AS a dictionary defaulting to zero  
        
        SET counter TO 0  
        WHILE counter LESS THAN LENGTH(secret)  
            SET secret_char TO secret[counter]  
            SET guess_char TO guess[counter]  
            
            IF NOT (secret_char DIFFERENT FROM guess_char) THEN
                SET alpha TO alpha + 1
            ELSE
                INCREASE secret_map[secret_char] BY 1
                INCREASE guess_map[guess_char] BY 1
            END IF  
            SET counter TO counter + 1
        END WHILE  
        
        FOR each key_element IN secret_map KEYS  
            SET min_val TO guess_map[key_element]  
            IF min_val GREATER THAN secret_map[key_element] THEN
                SET min_val TO secret_map[key_element]
            END IF
            SET beta TO beta + min_val
        END FOR  
        
        RETURN (string CONCAT alpha CONCAT "A" CONCAT beta CONCAT "B")  
    END FUNCTION  
END CLASS
