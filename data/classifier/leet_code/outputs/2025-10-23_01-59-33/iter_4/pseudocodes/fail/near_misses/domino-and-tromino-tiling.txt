CLASS Solution
    FUNCTION numTilings(n)
        CONSTANT P = 10^9 + 7

        CREATE dp_array AS array indexed 0 TO n WITH default 0
        DECLARE dp1_map AS map from integer TO integer
        ASSIGN dp_array[0] TO 1
        ASSIGN dp_array[1] TO 1

        IF n â‰¥ 2 THEN
            ASSIGN dp_array[2] TO 2
            SET dp1_map[2] TO 1
        END IF

        DEFINE PROCEDURE iterateFrom(k)
            IF k > n THEN RETURN
            LET val1 = dp_array[k-1]
            LET val2 = dp_array[k-2]
            LET val3 = dp1_map[k-1]
            LET new_dp = ((val1 + val2) + 2 * val3) MOD P
            LET new_dp1 = (dp_array[k-2] + dp1_map[k-1]) MOD P
            ASSIGN dp_array[k] TO new_dp
            SET dp1_map[k] TO new_dp1
            CALL iterateFrom(k + 1)
        END PROCEDURE

        CALL iterateFrom(3)

        RETURN dp_array[n]
    END FUNCTION
END CLASS
