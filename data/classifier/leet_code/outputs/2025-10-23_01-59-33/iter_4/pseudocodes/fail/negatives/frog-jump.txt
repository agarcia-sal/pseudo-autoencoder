CLASS Solution
    FUNCTION canCross(list_of_stones)
        IF the list_of_stones IS empty OR the LENGTH OF list_of_stones LESS THAN two
            RETURN False
        END IF
        
        SET stone_set TO the SET OF list_of_stones
        
        SET memo TO an empty MAPPING
        
        FUNCTION can_jump_to(position, jump)
            IF position EQUALS the ELEMENT at the LAST POSITION OF list_of_stones
                RETURN True
            END IF
            
            IF the PAIR of position AND jump EXISTS IN memo
                RETURN the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo
            END IF
            
            FOR each integer next_jump FROM jump MINUS one TO jump PLUS one INCLUSIVE
                IF next_jump GREATER THAN zero AND position PLUS next_jump EXISTS IN stone_set
                    IF can_jump_to(position PLUS next_jump, next_jump)
                        SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO True
                        RETURN True
                    END IF
                END IF
            END FOR
            
            SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO False
            RETURN False
        END FUNCTION
        
        RETURN can_jump_to(the ELEMENT at the FIRST POSITION OF list_of_stones, zero)
    END FUNCTION
END CLASS
