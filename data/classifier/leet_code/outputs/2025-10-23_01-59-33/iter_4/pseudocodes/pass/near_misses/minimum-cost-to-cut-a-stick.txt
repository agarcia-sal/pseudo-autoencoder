CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the LIST OF elements in cuts PLUS zero PLUS n SORTED IN ascending order  
        
        FUNCTION dp(left, right)  
            IF right MINUS left IS LESS THAN OR EQUAL TO one  
                RETURN zero  
            END IF  
            
            SET minimum_cost TO a very large number representing infinity  
            FOR index FROM left PLUS one TO right MINUS one  
                SET current_cost TO element at position right of cuts MINUS element at position left of cuts PLUS dp(left, index) PLUS dp(index, right)  
                IF current_cost IS LESS THAN minimum_cost  
                    SET minimum_cost TO current_cost  
                END IF  
            END FOR  
            
            RETURN minimum_cost  
        END FUNCTION  
        
        RETURN dp(zero, LENGTH OF cuts MINUS one)  
END CLASS
