CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET cache AS a list containing (amount + 1) copies of (amount + 1)  
        ASSIGN 0 TO cache[0]  

        SET counter TO 1  
        WHILE counter ≤ amount  
            SET currentMin TO cache[counter]  
            FOR each denomination IN coins  
                SET remainder TO counter - denomination  
                IF remainder ≥ 0 THEN  
                    SET candidate TO cache[remainder] + 1  
                    IF candidate < currentMin THEN  
                        ASSIGN candidate TO currentMin  
                    END IF  
                END IF  
            END FOR  
            ASSIGN currentMin TO cache[counter]  
            INCREMENT counter BY 1  
        END WHILE  

        IF cache[amount] ≠ (amount + 1) THEN  
            RETURN cache[amount]  
        ELSE  
            RETURN -1  
        END IF  
    END FUNCTION  
END CLASS
