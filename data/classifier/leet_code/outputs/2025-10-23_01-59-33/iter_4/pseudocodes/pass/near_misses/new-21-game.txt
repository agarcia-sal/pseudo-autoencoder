CLASS Solution  
    FUNCTION new21Game(n INTEGER_PARAMETER k INTEGER_PARAMETER maxPts INTEGER_PARAMETER) RETURNS FLOAT  
        FUNCTION dfs(i INTEGER_PARAMETER) RETURNS FLOAT  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN one IF i LESS THAN OR EQUAL TO n ELSE zero  
            END IF  
            IF i EQUALS k MINUS one THEN  
                RETURN the MINIMUM OF the EXPRESSION n MINUS k PLUS one AND maxPts DIVIDED BY maxPts  
            END IF  
            RETURN dfs of i PLUS one PLUS the EXPRESSION dfs of i PLUS one MINUS dfs of i PLUS maxPts PLUS one DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs of zero  
    END FUNCTION  
END CLASS
