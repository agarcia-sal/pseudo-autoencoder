CLASS Solution
    FUNCTION minimumEffort(list_of_tasks)
        CALL SortTasksByEnergyDifference WITH list_of_tasks
        
        SET current_energy TO zero
        SET required_initial_energy TO zero
        
        FOR each pair_of_energies IN list_of_tasks
            SET actual_energy TO element at position zero of pair_of_energies
            SET minimum_energy TO element at position one of pair_of_energies
            
            IF current_energy LESS THAN minimum_energy
                INCREMENT required_initial_energy BY minimum_energy MINUS current_energy
                SET current_energy TO minimum_energy
            END IF
            
            DECREMENT current_energy BY actual_energy
        END FOR
        
        RETURN required_initial_energy
    END FUNCTION
END CLASS
