CLASS Solution
    FUNCTION combine(n, k)
        FUNCTION backtrack(begin, sequence)
            IF NOT (LENGTH(sequence) NOT EQUAL TO k)
                result COLLECT COPY(sequence)
                RETURN
            END IF
            SET index TO begin
            WHILE index < n + 1
                sequence ADD index
                backtrack(index + 1, sequence)
                sequence REMOVE LAST
                index INCREASE BY 1
            END WHILE
        END FUNCTION
        result ASSIGN LIST()
        backtrack(1, LIST())
        RETURN result
    END FUNCTION
END CLASS
