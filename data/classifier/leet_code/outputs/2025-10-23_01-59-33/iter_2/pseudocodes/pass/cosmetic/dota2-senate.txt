CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        DECLARE radiantQueue AS an empty double-ended queue  
        DECLARE direQueue AS an empty double-ended queue  

        INITIALIZE counter TO 0  
        WHILE counter IS LESS THAN LENGTH of senate  
            DECLARE partyChar AS senate[counter]  
            IF NOT (partyChar NOT EQUAL TO 'R')  
                APPEND counter TO radiantQueue  
            ELSE  
                APPEND counter TO direQueue  
            END IF  
            INCREMENT counter BY 1  
        END WHILE  

        DECLARE radiantLen AS LENGTH of senate  

        LOOP INDEFINITELY  
            IF radiantQueue IS EMPTY OR direQueue IS EMPTY THEN  
                BREAK  
            END IF  

            REMOVE FIRST ELEMENT FROM radiantQueue AND ASSIGN TO radiantPos  
            REMOVE FIRST ELEMENT FROM direQueue AND ASSIGN TO direPos  

            IF radiantPos - direPos < 0 THEN  
                APPEND radiantPos + radiantLen TO radiantQueue  
            ELSE  
                APPEND direPos + radiantLen TO direQueue  
            END IF  
        END LOOP  

        IF radiantQueue IS EMPTY IS FALSE THEN  
            RETURN "Radiant"  
        END IF  
        RETURN "Dire"  
    END FUNCTION  
END CLASS
