CLASS Solution  
    FUNCTION findContestMatch(n)  
        DECLARE bracketStrings AS empty list  
        DECLARE count AS 1  
        WHILE count LESS THAN OR EQUAL TO n  
            INSERT TO bracketStrings THE STRING OF count  
            INCREMENT count BY 1  
        END WHILE  
        WHILE n NOT EQUAL TO 1  
            DECLARE index AS 0  
            DECLARE limit AS n DIV 2  
            WHILE index LESS THAN limit  
                DECLARE leftSide AS bracketStrings AT index  
                DECLARE rightSide AS bracketStrings AT (n - 1 - index)  
                ASSIGN bracketStrings AT index TO "(" + leftSide + "," + rightSide + ")"  
                INCREMENT index BY 1  
            END WHILE  
            SET n AS n DIV 2  
        END WHILE  
        RETURN bracketStrings AT 0  
    END FUNCTION  
END CLASS
