CLASS Solution  
    FUNCTION checkRecord WITH parameter n  
        SET MOD TO ten MULTIPLIED BY nine POWER plus seven  
        INITIALIZE dp TO a three dimensional list with dimensions n plus one BY two BY three filled with zero  
        SET element at position zero and position zero and position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position i and position j and position zero of dp TO the sum of element at position i and position j and position zero of dp AND element at position i minus one and position j and position k of dp THEN modulo MOD  
                    IF j EQUALS zero  
                        SET element at position i and position one and position zero of dp TO the sum of element at position i and position one and position zero of dp AND element at position i minus one and position zero and position k of dp THEN modulo MOD  
                    END IF  
                    IF k LESS THAN two  
                        SET element at position i and position j and position k plus one of dp TO the sum of element at position i and position j and position k plus one of dp AND element at position i minus one and position j and position k of dp THEN modulo MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO the sum of result AND element at position n and position j and position k of dp THEN modulo MOD  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
