CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET lengthVal AS LENGTH(nums)  
        IF NOT(lengthVal GREATER OR EQUAL TO 3) THEN  
            RETURN 0  
        END IF  
        
        SET dpList AS AN EMPTY list  
        SET idxCounter TO 0  
        WHILE idxCounter LESS THAN lengthVal DO  
            APPEND a new default dictionary mapping integer TO integer TO dpList  
            INCREMENT idxCounter BY 1  
        END WHILE  
        
        SET aggregateCount TO 0  
        
        SET outerIdx TO 0  
        WHILE outerIdx LESS THAN lengthVal DO  
            SET innerIdx TO 0  
            WHILE innerIdx LESS THAN outerIdx DO  
                SET diffDelta AS nums[outerIdx] MINUS nums[innerIdx]  
                
                IF diffDelta IN dpList[innerIdx] THEN  
                    SET prevVal TO dpList[innerIdx][diffDelta]  
                    SET dpList[outerIdx][diffDelta] TO dpList[outerIdx].GET(diffDelta, 0) PLUS prevVal PLUS 1  
                    SET aggregateCount TO aggregateCount PLUS prevVal  
                ELSE  
                    SET dpList[outerIdx][diffDelta] TO dpList[outerIdx].GET(diffDelta, 0) PLUS 1  
                END IF  
                
                INCREMENT innerIdx BY 1  
            END WHILE  
            INCREMENT outerIdx BY 1  
        END WHILE  
        
        RETURN aggregateCount  
    END FUNCTION  
END CLASS
