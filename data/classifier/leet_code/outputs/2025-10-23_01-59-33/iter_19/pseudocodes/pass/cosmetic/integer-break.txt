CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dpList AS an array of zeros sized n PLUS 1  
        SET dpList[1] AS 1  
        SET outerCounter TO 2  
        WHILE outerCounter IS LESS THAN OR EQUAL TO n DO  
            SET currentMax TO 0  
            SET innerCounter TO 1  
            WHILE innerCounter IS LESS THAN outerCounter DO  
                SET firstProduct TO innerCounter TIMES dpList[outerCounter MINUS innerCounter]  
                SET secondProduct TO innerCounter TIMES (outerCounter MINUS innerCounter)  
                SWITCH TRUE  
                    CASE currentMax < firstProduct AND currentMax < secondProduct:  
                        IF firstProduct > secondProduct THEN  
                            SET currentMax TO firstProduct  
                        ELSE  
                            SET currentMax TO secondProduct  
                        END IF  
                    CASE currentMax < firstProduct:  
                        SET currentMax TO firstProduct  
                    CASE currentMax < secondProduct:  
                        SET currentMax TO secondProduct  
                END SWITCH  
                INCREMENT innerCounter BY 1  
            END WHILE  
            SET dpList[outerCounter] TO currentMax  
            INCREMENT outerCounter BY 1  
        END WHILE  
        RETURN dpList[n]  
    END FUNCTION  
END CLASS
