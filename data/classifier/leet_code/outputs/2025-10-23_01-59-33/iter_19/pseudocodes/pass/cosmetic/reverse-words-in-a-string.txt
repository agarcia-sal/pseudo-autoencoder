CLASS Solution  
    FUNCTION reverseWords(s)  
        DECLARE wordCollection AS list ← s split by spaces ignoring repeated spaces  
        DECLARE reversedList AS list ← empty list  
        DECLARE indexer AS integer ← LENGTH(wordCollection) - 1  
        WHILE indexer >= 0  
            ADD wordCollection[indexer] TO reversedList  
            DECREMENT indexer BY 1  
        END WHILE  
        DECLARE resultString AS string ← reversedList elements concatenated with " " as delimiter  
        RETURN resultString  
    END FUNCTION  
END CLASS
