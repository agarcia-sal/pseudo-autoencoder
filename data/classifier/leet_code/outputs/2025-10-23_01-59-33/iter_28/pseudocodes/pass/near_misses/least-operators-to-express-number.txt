CLASS Solution  
    FUNCTION leastOpsExpressTarget(x parameter, target parameter) RETURNS integer  
        FUNCTION dp(target parameter) RETURNS integer  
            IF target EQUALS zero THEN  
                RETURN negative one  
            END IF  
            IF target LESS THAN x parameter THEN  
                RETURN the MINIMUM OF the value of two TIMES target parameter MINUS one AND the value of two TIMES x parameter MINUS target parameter  
            END IF  
            SET power TO one  
            WHILE x parameter RAISED TO THE power IS LESS THAN OR EQUAL TO target parameter DO  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO the result of dp WITH target parameter MINUS x parameter RAISED TO THE power PLUS power  
            IF x parameter RAISED TO THE power PLUS one MINUS target parameter LESS THAN target parameter THEN  
                SET res TO the MINIMUM OF res AND the result of dp WITH x parameter RAISED TO THE power PLUS one MINUS target parameter PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH target parameter  
    END FUNCTION  
END CLASS
