CLASS Solution
    FUNCTION majorityElement(list_of_numbers)
        SET n1 TO zero
        SET n2 TO zero
        SET m1 TO zero
        SET m2 TO one
        FOR each m IN list_of_numbers
            IF m EQUALS m1
                INCREMENT n1 BY one
            ELSE IF m EQUALS m2
                INCREMENT n2 BY one
            ELSE IF n1 EQUALS zero
                SET m1 TO m
                SET n1 TO one
            ELSE IF n2 EQUALS zero
                SET m2 TO m
                SET n2 TO one
            ELSE
                DECREMENT n1 BY one
                DECREMENT n2 BY one
            END IF
        END FOR
        RETURN list_of_elements m FROM list_containing m1 AND m2 WHERE the COUNT OF m IN list_of_numbers GREATER THAN the LENGTH OF list_of_numbers DIVIDED BY three
    END FUNCTION
END CLASS
