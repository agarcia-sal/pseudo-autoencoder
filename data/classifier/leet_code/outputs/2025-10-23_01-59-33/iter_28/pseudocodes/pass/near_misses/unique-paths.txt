CLASS Solution
    FUNCTION uniquePaths(m n)
        ASSIGN dp TO an empty list
        ASSIGN p TO 0
        WHILE p < m
            APPEND a list of length n filled with 1 to dp
            INCREMENT p BY 1
        END WHILE

        DEFINE helper(x y)
            IF x >= m THEN RETURN
            IF y >= n THEN
                CALL helper(x + 1 1)
                RETURN
            END IF
            ASSIGN a TO dp[x - 1][y]
            ASSIGN b TO dp[x][y - 1]
            ASSIGN dp[x][y] TO a + b
            CALL helper(x y + 1)
        END

        CALL helper(1 1)
        RETURN dp[m - 1][n - 1]
    END FUNCTION
END CLASS
