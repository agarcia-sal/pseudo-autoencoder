CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        CREATE graph AS a new default mapping from elements to empty lists  
        FOR each element u AND element v IN edges  
            APPEND a pair consisting of the element at position v of scores AND element v TO the list associated with element u in graph  
            APPEND a pair consisting of the element at position u of scores AND element u TO the list associated with element v in graph  
        END FOR  
        FOR each element node IN graph  
            SET the list associated with element node in graph TO the three highest scoring pairs from the list associated with element node in graph  
        END FOR  
        SET max_score TO negative one  
        FOR each element u AND element v IN edges  
            FOR each pair consisting of score1 AND x IN the list associated with element u in graph  
                FOR each pair consisting of score2 AND y IN the list associated with element v in graph  
                    IF the set containing x AND y AND u AND v HAS size EQUALS four  
                        SET max_score TO the greater value between max_score AND the sum of score1 PLUS score2 PLUS the element at position u of scores PLUS the element at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
