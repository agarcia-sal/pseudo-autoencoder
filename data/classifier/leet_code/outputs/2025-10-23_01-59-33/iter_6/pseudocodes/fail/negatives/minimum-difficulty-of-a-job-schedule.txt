n ← length of jobDifficulty
if n < d then return -1

define dp(i, day):
    if day = 1 then return max(jobDifficulty[i..])
    min_diff ← ∞
    current_max ← 0
    for j in i to n - day:
        current_max ← max(current_max, jobDifficulty[j])
        min_diff ← min(min_diff, current_max + dp(j+1, day-1))
    return min_diff

return dp(0, d)
