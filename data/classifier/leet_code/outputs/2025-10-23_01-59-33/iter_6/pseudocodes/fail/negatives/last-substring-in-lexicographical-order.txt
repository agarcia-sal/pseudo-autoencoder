CLASS Solution  
    FUNCTION lastSubstring(string_s)  
        SET length_n TO the LENGTH OF string_s  
        SET maximum_character TO the maximum CHARACTER IN string_s  
        SET collection_of_maximum_indices TO an empty collection  
        FOR each index_i FROM zero TO length_n MINUS one  
            IF the character at position index_i of string_s EQUALS maximum_character  
                APPEND index_i TO collection_of_maximum_indices  
            END IF  
        END FOR  
        SET maximum_substring TO empty string  
        FOR each start_index IN collection_of_maximum_indices  
            SET current_substring TO the substring from position start_index TO the END OF string_s  
            IF current_substring GREATER THAN maximum_substring  
                SET maximum_substring TO current_substring  
            END IF  
        END FOR  
        RETURN maximum_substring  
    END FUNCTION  
END CLASS
