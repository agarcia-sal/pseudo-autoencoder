CLASS Solution
    FUNCTION wordsTyping(sentence, rows, cols)
        SET combined_string TO ""
        ITERATE each word IN sentence DO
            combined_string ← combined_string + word + " "
        END ITERATION

        SET length_total TO LENGTH OF combined_string
        SET cursor TO 0

        DEFINE FUNCTION process_row(remaining_rows, current_pos)
            IF remaining_rows = 0 THEN
                RETURN current_pos
            END IF

            SET advanced_pos TO current_pos + cols
            SET check_char_index TO advanced_pos MOD length_total

            IF combined_string[check_char_index] = " " THEN
                RETURN process_row(remaining_rows - 1, advanced_pos + 1)
            ELSE
                SET backtrack_pos TO advanced_pos
                WHILE backtrack_pos > 0 AND combined_string[(backtrack_pos - 1) MOD length_total] ≠ " " DO
                    backtrack_pos ← backtrack_pos - 1
                END WHILE
                RETURN process_row(remaining_rows - 1, backtrack_pos)
            END IF
        END FUNCTION

        SET final_position TO process_row(rows, cursor)
        RETURN final_position DIV length_total
    END FUNCTION
END CLASS
