CLASS Solution  
    FUNCTION rectangleArea(rectangles)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET y_coords TO an empty set  
        FOR each quadruple x1 y1 x2 y2 IN rectangles  
            ADD y1 TO y_coords  
            ADD y2 TO y_coords  
        END FOR  
        SET y_coords TO the sorted list of y_coords  
        SET y_index TO an empty dictionary  
        FOR each index i AND value y IN the enumeration of y_coords  
            SET element at position y of y_index TO i  
        END FOR  
        SET count TO a list of zeros with the length equal to the length of y_coords  
        SET events TO an empty list  
        FOR each quadruple x1 y1 x2 y2 IN rectangles  
            APPEND a tuple of x1 one y1 y2 TO events  
            APPEND a tuple of x2 minus one y1 y2 TO events  
        END FOR  
        SORT events by the first element of each tuple in ascending order  
        SET prev_x TO zero  
        SET area TO zero  
        FOR each tuple x eventType y1 y2 IN events  
            SET current_length TO zero  
            FOR integer i FROM one TO the length of count MINUS one  
                IF the element at position i MINUS one of count is GREATER THAN zero  
                    INCREMENT current_length BY the element at position i of y_coords MINUS the element at position i MINUS one of y_coords  
                END IF  
            END FOR  
            INCREMENT area BY current_length MULTIPLIED BY x MINUS prev_x  
            SET area TO area MODULO MOD  
            IF eventType EQUALS one  
                FOR integer i FROM the element at position y1 of y_index TO the element at position y2 MINUS one of y_index  
                    INCREMENT the element at position i of count BY one  
                END FOR  
            ELSE  
                FOR integer i FROM the element at position y1 of y_index TO the element at position y2 MINUS one of y_index  
                    DECREMENT the element at position i of count BY one  
                END FOR  
            END IF  
            SET prev_x TO x  
        END FOR  
        RETURN area  
    END FUNCTION  
END CLASS
