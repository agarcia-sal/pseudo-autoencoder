CLASS Solution
    FUNCTION longestPalindrome(s)
        SET l TO LENGTH OF s
        SET p TO CREATE 2D ARRAY of BOOLEANs with l rows and l columns
        SET a TO l MINUS 1
        SET b TO l MINUS 1
        SET c TO l MINUS 2
        WHILE c IS GREATER THAN OR EQUAL TO 0
            SET d TO c PLUS 1
            WHILE d IS LESS THAN l
                p[c][d] = FALSE
                IF s[c] EQUALS s[d] THEN
                    p[c][d] = p[c PLUS 1][d MINUS 1]
                    IF p[c][d] EQUALS TRUE THEN
                        IF b MINUS a LESS THAN d MINUS c THEN
                            a = c
                            b = d
                        END IF
                    END IF
                END IF
                d = d PLUS 1
            END WHILE
            c = c MINUS 1
        END WHILE

        FUNCTION extract_substring(x, y, string)
            IF x GREATER THAN y THEN
                RETURN ""
            ELSE
                RETURN string[x] + extract_substring(x PLUS 1, y, string)
            END IF
        END FUNCTION

        RETURN extract_substring(a, b, s)
    END FUNCTION
END CLASS
