CLASS Solution  
    FUNCTION getHint(secret PARAMETER, guess PARAMETER)  
        IMPORT a data structure that initializes dictionary keys to zero by default  
        
        ASSIGN 0 TO claw  
        ASSIGN 0 TO spoor  
        
        INITIALIZE a dictionary named vault WITH default zero values  
        INITIALIZE a dictionary named token WITH default zero values  
        
        SET indexer TO 0  
        WHILE indexer LESS THAN length of secret DO  
            LET alpha BE secret AT position indexer  
            LET beta BE guess AT position indexer  
            
            IF NOT (alpha DIFFERENT FROM beta) THEN  
                INCREMENT claw BY 1  
            ELSE  
                SET vault[alpha] TO (vault[alpha] PLUS 1)  
                SET token[beta] TO (token[beta] PLUS 1)  
            END IF  
            
            INCREMENT indexer BY 1  
        END WHILE  
        
        FOR each key_element IN the set of keys from vault DO  
            LET min_value BE token[key_element]  
            IF vault[key_element] LESS THAN min_value THEN  
                ASSIGN vault[key_element] TO min_value  
            END IF  
            
            ADD min_value TO spoor  
        END FOR  
        
        RETURN CONCATENATE(string representation of claw, "A", string representation of spoor, "B")  
    END FUNCTION  
END CLASS
