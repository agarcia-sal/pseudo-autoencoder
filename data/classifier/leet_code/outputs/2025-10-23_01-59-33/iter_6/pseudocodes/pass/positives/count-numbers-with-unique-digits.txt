CLASS Solution  
    FUNCTION countNumbersWithUniqueDigits(n)  
        IF n EQUALS zero THEN  
            RETURN one  
        END IF  
        IF n EQUALS one THEN  
            RETURN ten  
        END IF  
        SET count TO ten  
        SET available_digits TO nine  
        SET choices TO nine  
        FOR i FROM two TO n INCLUSIVE  
            SET choices TO choices MULTIPLIED BY available_digits  
            SET count TO count PLUS choices  
            SET available_digits TO available_digits MINUS one  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
