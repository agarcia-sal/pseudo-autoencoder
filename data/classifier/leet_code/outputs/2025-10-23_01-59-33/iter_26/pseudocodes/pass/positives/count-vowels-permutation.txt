CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS seven  
        SET a TO one  
        SET e TO one  
        SET i TO one  
        SET o TO one  
        SET u TO one  
        FOR each index FROM two TO n  
            SET a_next TO the sum of e PLUS i PLUS u  
            SET e_next TO the sum of a PLUS i  
            SET i_next TO the sum of e PLUS o  
            SET o_next TO i  
            SET u_next TO the sum of i PLUS o  
            SET a TO the remainder when a_next is DIVIDED BY MOD  
            SET e TO the remainder when e_next is DIVIDED BY MOD  
            SET i TO the remainder when i_next is DIVIDED BY MOD  
            SET o TO the remainder when o_next is DIVIDED BY MOD  
            SET u TO the remainder when u_next is DIVIDED BY MOD  
        END FOR  
        SET total_count TO the sum of a PLUS e PLUS i PLUS o PLUS u  
        SET result TO the remainder when total_count is DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
