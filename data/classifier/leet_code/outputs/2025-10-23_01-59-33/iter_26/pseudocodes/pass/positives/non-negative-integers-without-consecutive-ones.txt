CLASS Solution  
    FUNCTION findIntegers(n)  
        SET binary_representation TO convert the number n to its binary representation as a string of characters representing digits in base two from left to right  
        SET length TO the number of characters in binary_representation  
        CREATE a list dp of length plus one initialized with zeroes  
        SET the element at position zero in dp TO one  
        SET the element at position one in dp TO two  
        FOR index i FROM two TO length  
            SET the element at position i in dp TO the element at position i minus one in dp PLUS the element at position i minus two in dp  
        END FOR  
        SET result TO zero  
        SET previous_bit TO zero  
        FOR index i FROM zero TO length minus one  
            IF element at position i in binary_representation EQUALS character representing one  
                INCREMENT result BY the element at position length minus i minus one in dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET previous_bit TO one IF element at position i in binary_representation EQUALS character representing one ELSE zero  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
