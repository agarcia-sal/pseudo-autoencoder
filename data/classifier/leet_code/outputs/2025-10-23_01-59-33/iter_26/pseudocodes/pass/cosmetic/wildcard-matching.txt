CLASS Solution  
    FUNCTION isMatch(s, p)  
        DECLARE table AS two-dimensional array FILLED WITH False, SIZE (length(s) + 1) BY (length(p) + 1)  
        ASSIGN table[0][0] ← True  

        SET indexP TO 1  
        WHILE indexP ≤ length(p)  
            IF p[indexP - 1] = '*' THEN  
                ASSIGN table[0][indexP] ← table[0][indexP - 1]  
            END IF  
            INCREMENT indexP BY 1  
        END WHILE  

        SET indexS TO 1  
        WHILE indexS ≤ length(s)  
            SET indexPTmp TO 1  
            WHILE indexPTmp ≤ length(p)  
                IF p[indexPTmp - 1] = '*' THEN  
                    ASSIGN table[indexS][indexPTmp] ← (table[indexS - 1][indexPTmp] OR table[indexS][indexPTmp - 1])  
                ELSE  
                    IF (p[indexPTmp - 1] = '?' OR s[indexS - 1] = p[indexPTmp - 1]) THEN  
                        ASSIGN table[indexS][indexPTmp] ← table[indexS - 1][indexPTmp - 1]  
                    END IF  
                END IF  
                SET indexPTmp ← indexPTmp + 1  
            END WHILE  
            SET indexS ← indexS + 1  
        END WHILE  

        RETURN table[length(s)][length(p)]  
    END FUNCTION  
END CLASS
