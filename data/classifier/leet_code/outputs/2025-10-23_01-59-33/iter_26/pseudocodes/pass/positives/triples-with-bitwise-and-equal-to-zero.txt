CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a new default dictionary of integers  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET current_key TO the bitwise AND of i AND j  
                INCREMENT the value at current_key in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and and associated count IN pair_and_count items  
                IF the bitwise AND of pair_and AND k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
