CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        LET tests BE 0  
        LET attempts BE a list initialized with zeros whose size equals k + 1  
        REPEAT  
            tests ← tests + 1  
            LET eggIndex BE k  
            WHILE eggIndex GREATER THAN 0  
                attempts[eggIndex] ← 1 + attempts[eggIndex - 1] + attempts[eggIndex]  
                eggIndex ← eggIndex - 1  
            END WHILE  
        UNTIL attempts[k] GREATER THAN OR EQUAL TO n  
        RETURN tests  
    END FUNCTION  
END CLASS
