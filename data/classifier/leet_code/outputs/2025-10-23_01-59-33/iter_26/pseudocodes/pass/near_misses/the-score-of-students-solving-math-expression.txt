CLASS Solution
	TYPE scoreOfStudents(s : STRING, answers : LIST OF INTEGER) : INTEGER
		// Nested helper FUNCTION evaluate_expression
		TYPE evaluate_expression(expression : STRING) : INTEGER
			LET token_list <- NEW LIST OF INTEGER OR CHAR
			LET accumulator <- 0
			
			FOR position FROM 1 TO LENGTH(expression) DO
				LET unit <- SUBSTRING(expression, position, position)
				IF unit >= "0" AND unit <= "9" THEN
					LET digit_value <- (ASCII_VALUE(unit) - ASCII_VALUE("0"))
					LET accumulator <- (accumulator * 10) + digit_value
				ELSE
					APPEND accumulator TO token_list
					APPEND unit TO token_list
					LET accumulator <- 0
				END IF
			END FOR
			APPEND accumulator TO token_list
			
			LET index <- 2
			WHILE index <= SIZE(token_list) DO
				IF token_list[index] == "*" THEN
					LET product_result <- (token_list[index - 1] * token_list[index + 1])
					REMOVE ELEMENTS AT POSITIONS index-1 TO index+1 FROM token_list
					INSERT product_result INTO token_list AT POSITION index-1
				ELSE
					LET index <- index + 1
				END IF
			END WHILE
			
			LET result_value <- token_list[1]
			LET position <- 2
			WHILE position <= SIZE(token_list) DO
				IF token_list[position] == "+" THEN
					LET result_value <- result_value + token_list[position + 1]
				END IF
				LET position <- position + 2
			END WHILE
			
			RETURN result_value
		END TYPE
		
		// Nested helper FUNCTION get_possible_answers
		TYPE get_possible_answers(expression : STRING) : SET OF INTEGER
			IF ALL CHARACTERS OF expression ARE DIGITS THEN
				LET numeric_val <- INTEGER(expression)
				RETURN NEW SET CONTAINING numeric_val
			END IF
			
			LET accumulated_results <- NEW EMPTY SET OF INTEGER
			LET char_pos <- 2
			WHILE char_pos < LENGTH(expression) DO
				IF char_pos MOD 2 == 0 THEN
					LET left_expr <- SUBSTRING(expression, 1, char_pos - 1)
					LET right_expr <- SUBSTRING(expression, char_pos + 1, LENGTH(expression))
					
					LET left_values <- get_possible_answers(left_expr)
					LET right_values <- get_possible_answers(right_expr)
					
					FOR each left_val IN left_values DO
						FOR each right_val IN right_values DO
							LET combined <- 0
							IF expression[char_pos] == "+" THEN
								LET combined <- (left_val + right_val)
							ELSE
								LET combined <- (left_val * right_val)
							END IF
							
							IF combined <= 1000 THEN
								ADD combined TO accumulated_results
							END IF
						END FOR
					END FOR
				END IF
				LET char_pos <- char_pos + 1
			END WHILE
			
			RETURN accumulated_results
		END TYPE
		
		LET actual_answer <- evaluate_expression(s)
		LET possible <- get_possible_answers(s)
		
		LET score_tracker <- 0
		FOR each given_answer IN answers DO
			IF given_answer == actual_answer THEN
				LET score_tracker <- score_tracker + 5
			ELSE IF given_answer IN possible THEN
				LET score_tracker <- score_tracker + 2
			END IF
		END FOR
		
		RETURN score_tracker
	END TYPE
END CLASS
