CLASS Solution
    FUNCTION waysToPartition(list_nums, single_value_k)
        SET total_sum TO the SUM OF all elements IN list_nums
        SET n TO the LENGTH OF list_nums

        DECLARE left_diff AS a DEFAULT DICTIONARY OF integers
        DECLARE right_diff AS a DEFAULT DICTIONARY OF integers

        SET current_sum TO zero
        FOR index_i FROM zero TO n MINUS one EXCLUSIVE
            INCREMENT current_sum BY the element at position index_i OF list_nums
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum
            INCREMENT the value at key diff IN right_diff BY one
        END FOR

        SET max_ways TO the value at key zero IN right_diff

        SET current_sum TO zero
        FOR index_i FROM zero TO n MINUS one INCLUSIVE
            INCREMENT current_sum BY the element at position index_i OF list_nums

            SET diff_change TO single_value_k MINUS the element at position index_i OF list_nums

            SET new_ways TO the value at key NEGATIVE diff_change IN left_diff PLUS the value at key diff_change IN right_diff

            IF new_ways GREATER THAN max_ways
                SET max_ways TO new_ways
            END IF

            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum
            DECREMENT the value at key diff IN right_diff BY one
            INCREMENT the value at key diff IN left_diff BY one
        END FOR

        RETURN max_ways
    END FUNCTION
END CLASS
