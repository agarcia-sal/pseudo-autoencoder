CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        ASSIGN concatenated_string TO EMPTY STRING  
        INITIALIZE index_counter TO 0  
        
        WHILE index_counter LESS THAN LENGTH OF sentence  
            CONCATENATE concatenated_string WITH sentence[index_counter]  
            CONCATENATE concatenated_string WITH " "  
            INCREMENT index_counter BY 1  
        END WHILE  
        
        ASSIGN length_total TO LENGTH OF concatenated_string  
        SET pointer TO 0  
        ASSIGN row_iter TO 1  
        
        WHILE row_iter LESS THAN OR EQUAL TO rows  
            ADD cols TO pointer  
            IF concatenated_string[pointer MOD length_total] IS " " THEN  
                INCREMENT pointer BY 1  
            ELSE  
                WHILE (pointer GREATER THAN 0) AND (concatenated_string[(pointer - 1) MOD length_total] IS NOT " ")  
                    DECREASE pointer BY 1  
                END WHILE  
            END IF  
            INCREMENT row_iter BY 1  
        END WHILE  
        
        RETURN INTEGER_DIVISION OF pointer BY length_total  
    END FUNCTION  
END CLASS
