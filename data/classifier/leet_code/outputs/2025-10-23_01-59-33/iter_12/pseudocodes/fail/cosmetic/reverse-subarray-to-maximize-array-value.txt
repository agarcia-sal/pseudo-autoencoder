CLASS Solution
    FUNCTION maxValueAfterReverse(nums)
        ASSIGN SIZE OF nums TO lengthCounter
        IF NOT (lengthCounter GREATER THAN 2) THEN
            ASSIGN ABS(nums[0] - nums[1]) TO resultVal
            RETURN resultVal
        END IF

        ASSIGN 0 TO sumAbsDiff
        ASSIGN 1 TO idx
        WHILE idx LESS THAN lengthCounter
            ASSIGN sumAbsDiff + ABS(nums[idx] - nums[idx - 1]) TO sumAbsDiff
            ASSIGN idx + 1 TO idx
        END WHILE

        ASSIGN 0 TO edgeMaxGain
        ASSIGN 1 TO pointerA
        WHILE pointerA LESS THAN lengthCounter
            ASSIGN ABS(nums[0] - nums[pointerA]) - ABS(nums[pointerA] - nums[pointerA - 1]) TO candidateGainA
            IF candidateGainA GREATER THAN edgeMaxGain THEN
                ASSIGN candidateGainA TO edgeMaxGain
            END IF
            ASSIGN pointerA + 1 TO pointerA
        END WHILE

        ASSIGN 0 TO pointerB
        WHILE pointerB LESS THAN lengthCounter - 1
            ASSIGN ABS(nums[lengthCounter - 1] - nums[pointerB]) - ABS(nums[pointerB + 1] - nums[pointerB]) TO candidateGainB
            IF candidateGainB GREATER THAN edgeMaxGain THEN
                ASSIGN candidateGainB TO edgeMaxGain
            END IF
            ASSIGN pointerB + 1 TO pointerB
        END WHILE

        ASSIGN +Infinity TO minMaxPair
        ASSIGN -Infinity TO maxMinPair
        ASSIGN 0 TO iteratorC
        WHILE iteratorC LESS THAN lengthCounter - 1
            ASSIGN MAX(nums[iteratorC], nums[iteratorC + 1]) TO currentMaxVal
            IF currentMaxVal LESS THAN minMaxPair THEN
                ASSIGN currentMaxVal TO minMaxPair
            END IF

            ASSIGN MIN(nums[iteratorC], nums[iteratorC + 1]) TO currentMinVal
            IF currentMinVal GREATER THAN maxMinPair THEN
                ASSIGN currentMinVal TO maxMinPair
            END IF
            ASSIGN iteratorC + 1 TO iteratorC
        END WHILE

        ASSIGN MAX(0, 2 * (maxMinPair - minMaxPair)) TO generalMaxGain

        RETURN sumAbsDiff + MAX(edgeMaxGain, generalMaxGain)
    END FUNCTION
END CLASS
