CLASS Solution
    FUNCTION isSelfCrossing(distance)
        SET m TO LENGTH OF distance

        FUNCTION checkCrossing(counter)
            IF counter GREATER THAN OR EQUAL TO m
                RETURN False
            END IF

            SET val1 TO distance[counter]
            SET val2 TO distance[counter - 1]
            SET val3 TO distance[counter - 2]
            SET val4 TO distance[counter - 3]

            IF val1 GREATER THAN OR EQUAL TO val3 AND val2 LESS THAN OR EQUAL TO val4
                RETURN True
            END IF

            IF counter GREATER THAN OR EQUAL TO 4
                SET val5 TO distance[counter - 4]
                IF val2 EQUALS val4 AND (val1 PLUS val5) GREATER THAN OR EQUAL TO val3
                    RETURN True
                END IF
            END IF

            IF counter GREATER THAN OR EQUAL TO 5
                SET val6 TO distance[counter - 5]
                SET val7 TO distance[counter - 4]
                IF val3 LESS THAN OR EQUAL TO val2 AND val7 LESS THAN OR EQUAL TO val3 AND (val2 PLUS val6) GREATER THAN OR EQUAL TO val4 AND (val1 PLUS val7) GREATER THAN OR EQUAL TO val3
                    RETURN True
                END IF
            END IF

            RETURN checkCrossing(counter + 1)
        END FUNCTION

        IF m LESS THAN 4
            RETURN False
        ELSE
            RETURN checkCrossing(3)
        END IF
    END FUNCTION
END CLASS
