CLASS Solution
    FUNCTION canConvert(input_string_one, input_string_two)
        IF input_string_one EQUALS input_string_two
            RETURN true
        END IF

        SET conversion_map TO empty mapping

        FOR each pair of character_one AND character_two IN corresponding positions OF input_string_one AND input_string_two
            IF character_one IS IN conversion_map
                IF the value at character_one IN conversion_map NOT EQUALS character_two
                    RETURN false
                END IF
            ELSE
                SET the value at character_one IN conversion_map TO character_two
            END IF
        END FOR

        IF the LENGTH OF the set OF characters OF input_string_two EQUALS twenty six
            RETURN false
        END IF

        RETURN true
    END FUNCTION
END CLASS
