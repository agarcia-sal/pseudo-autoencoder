CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode with the value of n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO the power of n MINUS one  
        SET current_gray TO concatenation of prev_gray WITH a list constructed by taking each number IN the reversed prev_gray AND applying bitwise OR with mask  
        RETURN current_gray  
    END FUNCTION  
END CLASS
