CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        ASSIGN (amount + 1) TO limitValue  
        INITIALIZE dpList AS array OF size limitValue FILLED WITH limitValue  
        SET dpList[0] TO 0  
        INITIALIZE indexA TO 1  
        WHILE indexA <= amount DO  
            INITIALIZE idxCoin TO 0  
            WHILE idxCoin < LENGTH(coins) DO  
                ASSIGN (indexA - coins[idxCoin]) TO diffVal  
                IF NOT (diffVal < 0) THEN  
                    ASSIGN MIN(dpList[indexA], dpList[diffVal] + 1) TO dpList[indexA]  
                END IF  
                INCREMENT idxCoin BY 1  
            END WHILE  
            INCREMENT indexA BY 1  
        END WHILE  
        IF dpList[amount] != limitValue THEN   
            RETURN dpList[amount]  
        END IF  
        RETURN -1  
    END FUNCTION  
END CLASS
