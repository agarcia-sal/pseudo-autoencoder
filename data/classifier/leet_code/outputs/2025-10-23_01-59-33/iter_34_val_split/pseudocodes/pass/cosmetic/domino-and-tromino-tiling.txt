CLASS Solution
    FUNCTION numTilings(n)
        ASSIGN 1000000007 TO MODULO

        ALLOCATE dp AS ARRAY OF SIZE (n + 1) FILLED WITH 0
        ALLOCATE dp1 AS ARRAY OF SIZE (n + 1) FILLED WITH 0

        ASSIGN 1 TO dp[0]
        ASSIGN 1 TO dp[1]
        IF NOT (n < 2) THEN
            ASSIGN 2 TO dp[2]
            ASSIGN 1 TO dp1[2]
        END IF

        ASSIGN counter TO 3
        WHILE counter <= n DO
            ASSIGN ((dp[counter - 1] + dp[counter - 2] + (dp1[counter - 1] << 1)) MOD MODULO) TO dp[counter]
            ASSIGN ((dp[counter - 2] + dp1[counter - 1]) MOD MODULO) TO dp1[counter]
            SET counter TO counter + 1
        END WHILE

        RETURN dp[n]
    END FUNCTION
END CLASS
