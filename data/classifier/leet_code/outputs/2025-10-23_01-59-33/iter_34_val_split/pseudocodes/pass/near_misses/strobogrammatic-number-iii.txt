CLASS Solution  
    FUNCTION strobogrammaticInRange(low  
                                    high)  

        FUNCTION generate_strobogrammatic(current_length  
                                         target_length)  
            IF current_length EQUALS 0  
                RETURN a list with an empty string element  
            ELSE IF current_length EQUALS 1  
                RETURN a list with elements: "0", "1", "8"  
            ELSE  
                DECLARE inner_results AS the result of generate_strobogrammatic(current_length MINUS 2, target_length)  
                DECLARE collection AS an empty list  
                FOR each inner_string IN inner_results  
                    IF current_length NOT EQUAL TO target_length  
                        ADD "0" CONCATENATED WITH inner_string CONCATENATED WITH "0" TO collection  
                    END IF  
                    ADD "1" CONCATENATED WITH inner_string CONCATENATED WITH "1" TO collection  
                    ADD "6" CONCATENATED WITH inner_string CONCATENATED WITH "9" TO collection  
                    ADD "8" CONCATENATED WITH inner_string CONCATENATED WITH "8" TO collection  
                    ADD "9" CONCATENATED WITH inner_string CONCATENATED WITH "6" TO collection  
                END FOR  
                RETURN collection  
            END IF  
        END FUNCTION  

        FUNCTION count_strobogrammatic(low  
                                       high)  
            DECLARE total AS 0  
            DECLARE low_length AS LENGTH OF low  
            DECLARE high_length AS LENGTH OF high  

            DEFINE FUNCTION count_for_length(current_len)  
                IF current_len GREATER THAN high_length  
                    RETURN 0  
                ELSE  
                    DECLARE total_for_current AS 0  
                    DECLARE sequences AS generate_strobogrammatic(current_len, current_len)  
                    DECLARE idx AS 0  
                    FUNCTION inner_count()  
                        IF idx EQUALS LENGTH OF sequences  
                            RETURN 0  
                        ELSE  
                            DECLARE candidate AS sequences[idx]  
                            SET idx TO idx + 1  
                            IF (current_len EQUALS low_length AND candidate LESS THAN low) OR (current_len EQUALS high_length AND candidate GREATER THAN high)  
                                RETURN inner_count()  
                            ELSE  
                                RETURN 1 + inner_count()  
                            END IF  
                        END IF  
                    END FUNCTION  
                    total_for_current = inner_count()  
                    RETURN total_for_current + count_for_length(current_len + 1)  
                END IF  
            END FUNCTION  

            total = count_for_length(low_length)  
            RETURN total  
        END FUNCTION  

        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
