CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the SORTED RESULT OF THE UNION OF cuts AND the LIST CONTAINING zero AND n  
        
        FUNCTION dp(left, right)  
            IF the DIFFERENCE OF right MINUS left IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR index FROM left PLUS one TO right MINUS one  
                SET cost TO the ELEMENT AT POSITION right OF cuts MINUS the ELEMENT AT POSITION left OF cuts PLUS the RESULT OF dp WITH PARAMETERS left AND index PLUS the RESULT OF dp WITH PARAMETERS index AND right  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN the RESULT OF dp WITH PARAMETERS zero AND the LENGTH OF cuts MINUS one  
END CLASS
