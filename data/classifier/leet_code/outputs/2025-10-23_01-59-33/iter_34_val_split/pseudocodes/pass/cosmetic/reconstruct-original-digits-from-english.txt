CLASS Solution  
    FUNCTION originalDigits(s)  
        ASSIGN a dictionary of character counts in s TO freqMap  
        ASSIGN a list with ten zeros TO numCounts  // indices 0 through 9  
        
        ASSIGN freqMap['z'] OR 0 TO numCounts[0]  
        ASSIGN freqMap['w'] OR 0 TO numCounts[2]  
        ASSIGN freqMap['u'] OR 0 TO numCounts[4]  
        ASSIGN freqMap['x'] OR 0 TO numCounts[6]  
        ASSIGN freqMap['g'] OR 0 TO numCounts[8]  
        
        ASSIGN (freqMap['h'] OR 0) - numCounts[8] TO numCounts[3]  
        ASSIGN (freqMap['f'] OR 0) - numCounts[4] TO numCounts[5]  
        ASSIGN (freqMap['s'] OR 0) - numCounts[6] TO numCounts[7]  
        
        ASSIGN (freqMap['i'] OR 0) - numCounts[5] - numCounts[6] - numCounts[8] TO numCounts[9]  
        ASSIGN (freqMap['o'] OR 0) - numCounts[0] - numCounts[2] - numCounts[4] TO numCounts[1]  
        
        ASSIGN an empty list TO outputList  
        
        ASSIGN 0 TO outerIdx  
        WHILE outerIdx < 10 DO  
            ASSIGN 0 TO innerIdx  
            WHILE innerIdx < numCounts[outerIdx] DO  
                ADD string version of outerIdx TO outputList END_STATEMENT  
                ASSIGN innerIdx + 1 TO innerIdx  
            END WHILE  
            ASSIGN outerIdx + 1 TO outerIdx  
        END WHILE  
        
        ASSIGN concatenation of all strings in outputList TO resultString  
        RETURN resultString  
    END FUNCTION  
END CLASS
