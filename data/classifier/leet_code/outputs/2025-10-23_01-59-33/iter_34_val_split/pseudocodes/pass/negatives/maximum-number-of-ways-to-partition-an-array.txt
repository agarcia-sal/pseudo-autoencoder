max_ways ← right_diff[0]
cur_sum ← 0
for i in 0 to n-1:
    cur_sum += nums[i]
    diff_change ← k - nums[i]
    new_ways ← left_diff[-diff_change] + right_diff[diff_change]
    max_ways ← max(max_ways, new_ways)
    diff ← sum_all - 2*cur_sum
    right_diff[diff] -= 1
    left_diff[diff] += 1

return max_ways
