set MOD = 10^9+7
n = length(nums)
max_y = max y in queries
prefix_sums = 2D array (max_y+1) x n filled with 0

for y in 1 to max_y:
  for i from n-1 down to 0:
    prefix_sums[y][i] = nums[i] + (prefix_sums[y][i+y] if i+y < n else 0)

answer = []
for (x,y) in queries:
  append prefix_sums[y][x] mod MOD to answer

return answer
