CLASS Solution
    FUNCTION soupServings(n)
        IF NOT (n <= 4800) THEN
            RETURN 1.0
        END IF
        LET m = (n + 24) // 25
        LET cache = {}
        FUNCTION dp(x, y)
            IF cache CONTAINS KEY (x, y) THEN
                RETURN cache[(x, y)]
            END IF
            IF x > 0 OR y > 0 THEN
                IF x > 0 AND y <= 0 THEN
                    RETURN 0.0
                ELSIF y > 0 AND x <= 0 THEN
                    RETURN 1.0
                END IF
                IF y <= 0 AND x <= 0 THEN
                    RETURN 0.5
                END IF
            ELSE
                RETURN 0.5
            END IF

            LET outcomes = [
                dp(x-4, y),
                dp(x-3, y-1),
                dp(x-2, y-2),
                dp(x-1, y-3)
            ]
            LET result = 0.25 * (outcomes[0] + outcomes[1] + outcomes[2] + outcomes[3])
            cache[(x, y)] = result
            RETURN result
        END FUNCTION

        RETURN dp(m, m)
    END FUNCTION
END CLASS
