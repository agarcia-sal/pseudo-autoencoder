CLASS Solution  
    FUNCTION numTilings(n)  
        SET constant MOD TO ten raised to the power of nine PLUS one  
        INITIALIZE collection dp AS zero repeated n PLUS one times  
        INITIALIZE collection dp1 AS zero repeated n PLUS one times  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        IF n GREATER THAN OR EQUAL TO two  
            SET element at position two OF dp TO two  
            SET element at position two OF dp1 TO one  
        END IF  
        FOR index FROM three TO n  
            SET element at position index OF dp TO the sum of element at position index MINUS one OF dp PLUS element at position index MINUS two OF dp PLUS two MULTIPLIED BY element at position index MINUS one OF dp1 MODULO MOD  
            SET element at position index OF dp1 TO the sum of element at position index MINUS two OF dp PLUS element at position index MINUS one OF dp1 MODULO MOD  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
