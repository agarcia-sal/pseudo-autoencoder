CLASS Solution
    FUNCTION soupServings(n)
        IF NOT (n LESS OR EQUAL TO 4800) THEN
            GIVE BACK 1.0
        END IF

        ASSIGN FLOOR((n + 24) / 25) TO scaledN
        DECLARE memoMap AS NEW hash map

        FUNCTION dp(a, b)
            IF (a, b) IS IN memoMap THEN
                GIVE BACK memoMap[(a, b)]
            END IF

            IF a > 0 OR b > 0 THEN
                IF b <= 0 THEN
                    GIVE BACK 0.0
                ELSE IF a <= 0 THEN
                    GIVE BACK 1.0
                END IF
            ELSE
                GIVE BACK 0.5
            END IF

            ASSIGN 0.25 * (
                dp(a - 4, b) + 
                dp(a - 3, b - 1) + 
                dp(a - 2, b - 2) + 
                dp(a - 1, b - 3)
            ) TO resultProb

            STORE resultProb IN memoMap[(a, b)]
            GIVE BACK resultProb
        END FUNCTION

        GIVE BACK dp(scaledN, scaledN)
    END FUNCTION
END CLASS
