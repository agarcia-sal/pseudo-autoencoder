CLASS Solution
    FUNCTION reachingPoints(start_x, start_y, target_x, target_y) RETURNS boolean
        WHILE target_x GREATER THAN OR EQUAL TO start_x AND target_y GREATER THAN OR EQUAL TO start_y HOLDS
            IF target_x EQUALS start_x AND target_y EQUALS start_y THEN
                RETURN true
            END IF
            IF target_x GREATER THAN target_y THEN
                IF target_y GREATER THAN start_y THEN
                    SET target_x TO target_x MODULO target_y
                ELSE
                    RETURN whether the result of target_x MINUS start_x MODULO target_y EQUALS zero
                END IF
            ELSE
                IF target_x GREATER THAN start_x THEN
                    SET target_y TO target_y MODULO target_x
                ELSE
                    RETURN whether the result of target_y MINUS start_y MODULO target_x EQUALS zero
                END IF
            END IF
        END WHILE
        RETURN false
    END FUNCTION
END CLASS
