CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        ASSIGN 1_000_000_007 TO MOD  
        ASSIGN 1 TO alpha  
        ASSIGN 1 TO bravo  
        ASSIGN 1 TO charlie  
        ASSIGN 1 TO delta  
        ASSIGN 1 TO echo  
        ASSIGN 2 TO counter  
        WHILE counter IS NOT GREATER THAN n  
            ASSIGN (bravo PLUS charlie PLUS echo) TO alpha_next  
            ASSIGN (alpha PLUS charlie) TO bravo_next  
            ASSIGN (bravo PLUS delta) TO charlie_next  
            ASSIGN charlie TO delta_next  
            ASSIGN (charlie PLUS delta) TO echo_next  
            ASSIGN alpha_next MOD MOD TO alpha  
            ASSIGN bravo_next MOD MOD TO bravo  
            ASSIGN charlie_next MOD MOD TO charlie  
            ASSIGN delta_next MOD MOD TO delta  
            ASSIGN echo_next MOD MOD TO echo  
            INCREMENT counter BY 1  
        END WHILE  
        ASSIGN (alpha PLUS bravo PLUS charlie PLUS delta PLUS echo) TO temp_sum  
        ASSIGN temp_sum MOD MOD TO output  
        RETURN output  
    END FUNCTION  
END CLASS
