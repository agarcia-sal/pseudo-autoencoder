CLASS Solution
    FUNCTION sumSubseqWidths(list_of_numbers)
        SET MODULO_MODULUS TO ten TO THE POWER OF nine PLUS seven
        CALL sort ON list_of_numbers
        SET length_of_list TO the LENGTH OF list_of_numbers
        SET total_sum TO zero
        SET power_list TO a list WITH length length_of_list FILLED WITH one

        FOR index FROM one TO length_of_list MINUS one
            SET element_at position index OF power_list TO element_at position index MINUS one OF power_list MULTIPLIED BY two MODULO_MODULUS
        END FOR

        FOR index FROM zero TO length_of_list MINUS one
            SET total_sum TO total_sum PLUS element_at position index OF list_of_numbers MULTIPLIED BY element_at position index OF power_list MINUS element_at position length_of_list MINUS index MINUS one OF power_list MODULO_MODULUS
            SET total_sum TO total_sum MODULO_MODULUS
        END FOR
        
        RETURN total_sum
    END FUNCTION
END CLASS
