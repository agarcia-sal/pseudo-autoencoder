CLASS Solution
    FUNCTION checkPossibility(list_of_numbers)
        SET modified_flag TO false
        FOR index FROM one TO the LENGTH OF list_of_numbers MINUS one
            IF element at position index of list_of_numbers LESS THAN element at position index MINUS one of list_of_numbers
                IF modified_flag IS true
                    RETURN false
                END IF
                SET modified_flag TO true
                IF index LESS THAN two OR element at position index of list_of_numbers GREATER THAN OR EQUAL TO element at position index MINUS two of list_of_numbers
                    SET element at position index MINUS one of list_of_numbers TO element at position index of list_of_numbers
                ELSE
                    SET element at position index of list_of_numbers TO element at position index MINUS one of list_of_numbers
                END IF
            END IF
        END FOR
        RETURN true
    END FUNCTION
END CLASS
