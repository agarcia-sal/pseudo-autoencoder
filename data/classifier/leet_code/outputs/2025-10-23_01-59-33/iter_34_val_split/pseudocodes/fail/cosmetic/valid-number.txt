CLASS Solution  
    FUNCTION isNumber(s)  
        ASSIGN pattern_string TO "^[\+\-]?((\d+(\.\d*)?)|(\.\d+))([eE][\+\-]?\d+)?$"  
        ASSIGN match_result TO MATCH_ENTIRE_STRING(s, pattern_string)  
        IF match_result == NULL  
            RETURN false  
        ELSE  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
