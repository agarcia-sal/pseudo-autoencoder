CLASS Solution  
    FUNCTION maxProduct(words)  
        DECLARE characterCollections AS an empty array  
        DECLARE idxOuter AS 0  
        WHILE idxOuter < SIZE(words) DO  
            DECLARE currentWord AS words[idxOuter]  
            APPEND UNIQUE_CHAR_SET(currentWord) TO characterCollections  
            INCREMENT idxOuter BY 1  
        END WHILE  
        
        ASSIGN 0 TO highestProduct  
        
        DECLARE outerIdx AS 0  
        WHILE outerIdx < SIZE(words) - 1 DO  
            DECLARE innerIdx AS outerIdx + 1  
            WHILE innerIdx < SIZE(words) DO  
                DECLARE setA AS characterCollections[outerIdx]  
                DECLARE setB AS characterCollections[innerIdx]  
                IF NOT EXISTS(ALL elements IN setA THAT ARE ALSO IN setB) THEN  
                    DECLARE currentProduct AS (LENGTH(words[outerIdx]) * LENGTH(words[innerIdx]))  
                    IF highestProduct < currentProduct THEN  
                        ASSIGN currentProduct TO highestProduct  
                    END IF  
                END IF  
                INCREMENT innerIdx BY 1  
            END WHILE  
            INCREMENT outerIdx BY 1  
        END WHILE  
        
        RETURN highestProduct  
    END FUNCTION  
END CLASS
