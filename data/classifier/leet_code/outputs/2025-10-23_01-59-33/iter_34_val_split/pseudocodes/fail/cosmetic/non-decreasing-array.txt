CLASS Solution
    FUNCTION checkPossibility(nums)
        ASSIGN false TO alteredFlag
        SET idx TO 1
        WHILE idx < LENGTH(nums)
            IF NOT (nums[idx] >= nums[idx - 1])
                IF alteredFlag
                    IMMEDIATELY RETURN false
                END IF
                alteredFlag = true
                IF (idx < 2) OR (nums[idx] >= nums[idx - 2])
                    ASSIGN nums[idx] TO nums[idx - 1]
                ELSE
                    ASSIGN nums[idx - 1] TO nums[idx]
                END IF
            END IF
            INCREMENT idx BY 1
        END WHILE
        IMMEDIATELY RETURN true
    END FUNCTION
END CLASS
