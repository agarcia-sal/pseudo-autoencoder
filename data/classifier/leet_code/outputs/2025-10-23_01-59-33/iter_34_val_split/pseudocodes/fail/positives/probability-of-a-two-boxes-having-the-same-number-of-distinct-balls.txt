CLASS Solution  
    FUNCTION getProbability WITH PARAMETER balls  
        FUNCTION dfs WITH PARAMETERS i, j, diff  
            IF i GREATER THAN OR EQUAL TO k  
                IF j EQUALS zero AND diff EQUALS zero  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF j LESS THAN zero  
                RETURN zero  
            END IF  
            SET ans TO zero  
            FOR x FROM zero TO element at position i OF balls INCLUSIVE  
                IF x EQUALS element at position i OF balls  
                    SET y TO one  
                ELSE IF x EQUALS zero  
                    SET y TO negative one  
                ELSE  
                    SET y TO zero  
                END IF  
                SET ans TO ans PLUS dfs WITH ARGUMENTS i PLUS one, j MINUS x, diff PLUS y MULTIPLIED BY combination of element at position i OF balls AND x  
            END FOR  
            RETURN ans  
        END FUNCTION  
        SET n TO sum of balls DIVIDED BY two  
        SET k TO length of balls  
        RETURN dfs WITH ARGUMENTS zero, n, zero DIVIDED BY combination of n MULTIPLIED BY two AND n  
    END FUNCTION  
END CLASS
