CLASS Solution
    FUNCTION reachingPoints(sx sx PARAMETER sy sy PARAMETER tx tx PARAMETER ty ty) RETURNS boolean
        SET reachable TO False
        WHILE NOT reachable AND tx GREATER THAN OR EQUAL TO sx AND ty GREATER THAN OR EQUAL TO sy
            IF tx EQUALS sx THEN
                IF ty EQUALS sy THEN
                    ASSIGN True TO reachable
                ELSE
                    ASSIGN ((ty MINUS sy) MODULO tx EQUALS 0) TO reachable
                END IF
            ELSE IF ty EQUALS sy THEN
                ASSIGN ((tx MINUS sx) MODULO ty EQUALS 0) TO reachable
            ELSE IF tx GREATER THAN ty THEN
                ASSIGN (tx MODULO ty) TO tx
            ELSE
                ASSIGN (ty MODULO tx) TO ty
            END IF
        END WHILE
        RETURN reachable
    END FUNCTION
END CLASS
