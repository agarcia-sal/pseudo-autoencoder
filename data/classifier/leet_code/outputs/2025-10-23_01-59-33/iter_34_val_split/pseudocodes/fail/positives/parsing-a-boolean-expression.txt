CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character c IN expression  
            IF c EQUALS one of the characters representing true false negation and and or  
                APPEND c TO stk  
            ELSE IF c EQUALS the character representing closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE the last element in stk EQUALS one of the characters representing true or false  
                    INCREMENT count_true BY one IF the last element in stk EQUALS the character representing true  
                    INCREMENT count_false BY one IF the last element in stk EQUALS the character representing false  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET operator TO the last element FROM stk  
                REMOVE the last element FROM stk  
                IF operator EQUALS the character representing negation  
                    SET c TO the character representing true IF count_false IS GREATER THAN zero OTHERWISE the character representing false  
                ELSE IF operator EQUALS the character representing and  
                    SET c TO the character representing false IF count_false IS GREATER THAN zero OTHERWISE the character representing true  
                ELSE IF operator EQUALS the character representing or  
                    SET c TO the character representing true IF count_true IS GREATER THAN zero OTHERWISE the character representing false  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN whether the first element of stk EQUALS the character representing true  
    END FUNCTION  
END CLASS
