CLASS Solution  
    FUNCTION uniquePaths(m n)  
        SET matrix AS a list containing m elements, each being a list of length n filled with 1  
        SET rowIndex TO 1  
        WHILE rowIndex < m DO  
            SET columnIndex TO 1  
            WHILE columnIndex < n DO  
                SET currentRow TO matrix[rowIndex]  
                SET currentRow[columnIndex] TO matrix[rowIndex - 1][columnIndex] + currentRow[columnIndex - 1]  
                SET columnIndex TO columnIndex + 1  
            END WHILE  
            SET rowIndex TO rowIndex + 1  
        END WHILE  
        RETURN matrix[m - 1][n - 1]  
    END FUNCTION  
END CLASS
