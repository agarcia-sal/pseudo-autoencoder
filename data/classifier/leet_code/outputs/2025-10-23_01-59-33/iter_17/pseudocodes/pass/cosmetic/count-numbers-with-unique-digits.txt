CLASS Solution  
    FUNCTION countNumbersWithUniqueDigits(n)  
        SWITCH n  
            CASE 0:  
                RETURN 1  
            CASE 1:  
                RETURN 10  
        END SWITCH  
  
        SET total_sum AS 10  
        SET digits_left AS 9  
        SET product AS 9  
        SET index AS 2  
        WHILE index <= n DO  
            SET product AS product * digits_left  
            SET total_sum AS total_sum + product  
            SET digits_left AS digits_left - 1  
            SET index AS index + 1  
        END WHILE  
  
        RETURN total_sum  
    END FUNCTION  
END CLASS
