CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        FUNCTION count(p PARAMETER)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION TO THE RIGHT OF p DIVIDED BY x IN nums2  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION TO THE LEFT OF p DIVIDED BY x IN nums2  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the INDICATOR THAT p GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION TO THE LEFT OF k IN the RANGE FROM NEGATIVE mx TO mx PLUS ONE WITH count AS THE KEY MINUS mx  
    END FUNCTION  
END CLASS
