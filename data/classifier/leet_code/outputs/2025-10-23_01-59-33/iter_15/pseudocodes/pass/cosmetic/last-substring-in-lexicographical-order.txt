CLASS Solution  
    FUNCTION lastSubstring(s)  
        LET lengthVar BE LENGTH OF s  
        LET highestChar BE MAXIMUM CHARACTER IN s  
        LET positions BE AN EMPTY LIST  
        LET index COUNTER TO 0  
        WHILE index < lengthVar DO  
            IF s[index] = highestChar THEN  
                APPEND index TO positions  
            END IF  
            SET index TO index + 1  
        END WHILE  

        LET candidateSubstring BE ""  
        LET posIndex POINTER TO 0  
        WHILE posIndex < LENGTH OF positions DO  
            LET substringStart BE positions[posIndex]  
            LET tempSubstring BE SLICE OF s FROM substringStart TO (LENGTH OF s)  
            IF NOT (candidateSubstring >= tempSubstring) THEN  
                SET candidateSubstring TO tempSubstring  
            END IF  
            SET posIndex TO posIndex + 1  
        END WHILE  
        RETURN candidateSubstring  
    END FUNCTION  
END CLASS
