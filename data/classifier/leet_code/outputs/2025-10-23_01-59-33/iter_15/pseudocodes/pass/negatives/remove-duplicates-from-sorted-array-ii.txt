CLASS Solution
    FUNCTION removeDuplicates(list_of_numbers)
        IF the list_of_numbers IS empty
            RETURN zero
        END IF
        
        SET write_index TO zero
        
        FOR index FROM zero TO the MINIMUM OF two AND the LENGTH OF list_of_numbers MINUS one
            SET element at position write_index of list_of_numbers TO element at position index of list_of_numbers
            INCREMENT write_index BY one
        END FOR
        
        FOR index FROM two TO the LENGTH OF list_of_numbers MINUS one
            IF element at position index of list_of_numbers NOT EQUALS element at position write_index MINUS two of list_of_numbers
                SET element at position write_index of list_of_numbers TO element at position index of list_of_numbers
                INCREMENT write_index BY one
            END IF
        END FOR
        
        RETURN write_index
    END FUNCTION
END CLASS
