CLASS Solution
  FUNCTION countNumbersWithUniqueDigits(n)
    DECLARE x ARRAY WITH VALUES [0, 10]
    IF n LESS THAN 2 THEN
      RETURN x[n]
    END IF
    DECLARE r TO 10
    DECLARE y TO 9
    DECLARE z TO 9
    DECLARE j TO 2
    WHILE j LESS THAN OR EQUAL TO n
      SET z TO z TIMES y
      SET r TO r PLUS z
      SET y TO y MINUS 1
      SET j TO j PLUS 1
    END WHILE
    RETURN r
  END FUNCTION
END CLASS
