CLASS Solution  
    FUNCTION grayCode(n)  
        DECLARE result AS list OF integers  
        DECLARE temp AS list OF integers  
        DECLARE bitmask AS integer  
        DECLARE index AS integer  
        
        IF NOT (n NOT EQUALS 0) THEN  
            RETURN [0]  
        END IF  
        
        SET temp AS grayCode(n - 1)  
        SET bitmask AS 1 << (n - 1)  
        
        SET result AS temp  
        SET index TO LENGTH(temp) - 1  
        
        WHILE index GREATER OR EQUAL TO 0 DO  
            APPEND (temp[index] BITWISE_OR bitmask) TO result  
            DECREMENT index BY 1  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
