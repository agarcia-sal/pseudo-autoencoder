CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        DECLARE steps AS INTEGER  
        ASSIGN 0 TO steps  
        DECLARE results AS LIST OF INTEGER  
        ASSIGN ARRAY FILLED WITH 0, LENGTH k+1 TO results  
        
        FUNCTION recursiveLoop()  
            IF results[k] >= n THEN  
                RETURN  
            END IF  
            
            steps = steps + 1  
            DECLARE iterator AS INTEGER  
            ASSIGN k TO iterator  
            
            WHILE iterator > 0 DO  
                results[iterator] = 1 + results[iterator - 1] + results[iterator]  
                iterator = iterator - 1  
            END WHILE  
            CALL recursiveLoop()  
        END FUNCTION  
        
        CALL recursiveLoop()  
        RETURN steps  
    END FUNCTION  
END CLASS
