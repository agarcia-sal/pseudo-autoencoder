CLASS Solution
    FUNCTION repeatedStringMatch(a_string, b_string)
        SET minimum_repetitions TO the smallest integer greater than or equal to the length of b_string DIVIDED BY the length of a_string
        IF b_string IS IN a_string REPEATED minimum_repetitions TIMES
            RETURN minimum_repetitions
        END IF
        IF b_string IS IN a_string REPEATED minimum_repetitions PLUS one TIME
            RETURN minimum_repetitions PLUS one
        END IF
        RETURN negative one
    END FUNCTION
END CLASS
