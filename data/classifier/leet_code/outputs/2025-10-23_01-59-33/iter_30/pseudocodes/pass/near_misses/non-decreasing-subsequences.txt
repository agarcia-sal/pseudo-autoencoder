CLASS Solution
    FUNCTION findSubsequences(nums)
        SET ans TO empty list

        FUNCTION dfs(idx, prev, temp)
            IF idx GREATER THAN OR EQUAL TO LENGTH OF nums THEN
                IF LENGTH OF temp GREATER THAN 1 THEN
                    SET copyTemp TO empty list
                    SET i TO 0
                    WHILE i LESS THAN LENGTH OF temp
                        APPEND temp AT i TO copyTemp
                        SET i TO i PLUS 1
                    END WHILE
                    APPEND copyTemp TO ans
                END IF
                RETURN
            END IF

            IF prev GREATER THAN nums AT idx THEN
                dfs(idx + 1, prev, temp)
            ELSE IF prev LESS THAN OR EQUAL TO nums AT idx THEN
                APPEND nums AT idx TO temp
                dfs(idx + 1, nums AT idx, temp)
                SET n TO LENGTH OF temp MINUS 1
                REMOVE element AT n FROM temp
                dfs(idx + 1, prev, temp)
            END IF
        END FUNCTION

        dfs(0, -1000, empty list)
        RETURN ans
    END FUNCTION
END CLASS
