CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL sort ON nums  
        SET n TO the LENGTH OF nums  

        SET first TO the element at position one of nums  

        FOR i FROM two TO n  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS the element at position i of nums  
                CONTINUE TO the next iteration  
            END IF  

            SET count TO a COUNTER object initialized with nums  
            SET arr TO an empty list  

            FOR each element x IN nums  
                IF the count of x EQUALS zero  
                    CONTINUE TO the next iteration  
                END IF  
                IF the count of x PLUS two MULTIPLIED BY k EQUALS zero  
                    BREAK out of the loop  
                END IF  

                APPEND x PLUS k TO arr  
                DECREMENT the count of x BY one  
                DECREMENT the count of x PLUS two MULTIPLIED BY k BY one  
            END FOR  

            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  

        RETURN an empty list  
    END FUNCTION  
END CLASS
