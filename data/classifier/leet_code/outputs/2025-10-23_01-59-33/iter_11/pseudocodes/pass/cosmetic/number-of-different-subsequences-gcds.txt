CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET upper_bound AS the largest element found in nums  
        SET elements_map AS a mapping of all items in nums to true  
        SET total_count AS 0  
        SET g_index AS 1  
        WHILE g_index <= upper_bound DO  
            SET current_gcd AS 0  
            SET multiplier AS 1  
            LOOP   
                SET candidate_num AS g_index * multiplier  
                IF candidate_num > upper_bound THEN  
                    EXIT LOOP  
                END IF  
                IF elements_map HAS candidate_num THEN  
                    SET current_gcd AS gcd(current_gcd, candidate_num)  
                    IF current_gcd = g_index THEN  
                        total_count ← total_count + 1  
                        EXIT LOOP  
                    END IF  
                END IF  
                SET multiplier ← multiplier + 1  
            END LOOP  
            SET g_index ← g_index + 1  
        END WHILE  
        RETURN total_count  
    END FUNCTION  
END CLASS
