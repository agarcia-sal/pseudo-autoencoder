CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO an empty mapping from characters to characters  
        FOR each pair of characters char1 and char2 FROM str1 AND str2 in order  
            IF char1 IS IN conversion_map  
                IF the value mapped from char1 in conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET the value mapped from char1 in conversion_map TO char2  
            END IF  
        END FOR  
        IF the number of unique characters IN str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
