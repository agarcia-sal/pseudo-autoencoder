CLASS Solution  
    FUNCTION multiply(num1 string num2 string) RETURNS string  
        IF NOT (num1 NOT EQUALS string zero AND num2 NOT EQUALS string zero) THEN  
            RETURN string zero  
        END IF  

        SET intermediateList TO new list  
        SET lengthLimit TO (LENGTH OF num1) PLUS (LENGTH OF num2)  
        FOR counter FROM zero UPTO lengthLimit MINUS one  
            APPEND zero TO intermediateList  
        END FOR   

        SET reversedNum1 TO empty string  
        SET indexA TO LENGTH OF num1 MINUS one  
        WHILE indexA GREATER OR EQUAL TO zero DO  
            SET reversedNum1 TO reversedNum1 PLUS CHARACTER AT indexA OF num1  
            DECREMENT indexA BY one  
        END WHILE  
        SET num1 TO reversedNum1  

        SET reversedNum2 TO empty string  
        SET indexB TO LENGTH OF num2 MINUS one  
        WHILE indexB GREATER OR EQUAL TO zero DO  
            SET reversedNum2 TO reversedNum2 PLUS CHARACTER AT indexB OF num2  
            DECREMENT indexB BY one  
        END WHILE  
        SET num2 TO reversedNum2  

        SET posI TO zero  
        WHILE posI LESS THAN LENGTH OF num1 DO  
            SET posJ TO zero  
            WHILE posJ LESS THAN LENGTH OF num2 DO  
                SET codeA TO (CHARACTER CODE OF CHARACTER AT posI OF num1) MINUS (CHARACTER CODE OF string zero)  
                SET codeB TO (CHARACTER CODE OF CHARACTER AT posJ OF num2) MINUS (CHARACTER CODE OF string zero)  
                SET productVal TO codeA TIMES codeB  

                SET positionOne TO posI PLUS posJ  
                SET positionTwo TO positionOne PLUS one  

                SET combinedSum TO productVal PLUS ELEMENT AT positionOne OF intermediateList  

                SET ELEMENT AT positionOne OF intermediateList TO combinedSum MODULO ten  
                SET ELEMENT AT positionTwo OF intermediateList TO ELEMENT AT positionTwo OF intermediateList PLUS (combinedSum DIVIDED BY ten)  
                INCREMENT posJ BY one  
            END WHILE  
            INCREMENT posI BY one  
        END WHILE  

        SET reversedResultStr TO empty string  
        SET revIndex TO LENGTH OF intermediateList MINUS one  
        WHILE revIndex GREATER OR EQUAL TO zero DO  
            SET reversedResultStr TO reversedResultStr PLUS STRING OF ELEMENT AT revIndex OF intermediateList  
            DECREMENT revIndex BY one  
        END WHILE  

        SET startIndex TO zero  
        WHILE startIndex LESS THAN LENGTH OF reversedResultStr AND CHARACTER AT startIndex OF reversedResultStr EQUALS string zero DO  
            INCREMENT startIndex BY one  
        END WHILE  

        IF startIndex EQUALS LENGTH OF reversedResultStr THEN  
            RETURN string zero  
        END IF  

        RETURN SUBSTRING OF reversedResultStr FROM startIndex TO LENGTH OF reversedResultStr MINUS one  
    END FUNCTION  
END CLASS
