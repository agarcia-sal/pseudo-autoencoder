CLASS Solution
    FUNCTION isNumber(input_string)
        IMPORT regular_expression_module
        
        SET number_pattern TO a regular expression pattern that matches an optional plus or minus sign FOLLOWED BY EITHER one or more digits FOLLOWED BY an optional decimal point FOLLOWED BY zero or more digits OR a decimal point FOLLOWED BY one or more digits AND OPTIONALLY an exponent part starting with the character e FOLLOWED BY an optional plus or minus sign FOLLOWED BY one or more digits
        
        RETURN the result of checking whether the entire input_string fully matches the number_pattern IS NOT EQUAL TO nothing
    END FUNCTION
END CLASS
