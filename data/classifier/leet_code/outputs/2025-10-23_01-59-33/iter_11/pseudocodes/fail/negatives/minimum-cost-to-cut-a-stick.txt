CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO cuts PLUS element zero PLUS element n  
        SET cuts TO cuts SORTED IN ASCENDING ORDER  
        
        FUNCTION dp(left, right)  
            IF element at position right MINUS element at position left IN cuts IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right IN cuts MINUS element at position left IN cuts PLUS dp(left, i) PLUS dp(i, right)  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(zero, LENGTH OF cuts MINUS one)  
    END FUNCTION  
END CLASS
