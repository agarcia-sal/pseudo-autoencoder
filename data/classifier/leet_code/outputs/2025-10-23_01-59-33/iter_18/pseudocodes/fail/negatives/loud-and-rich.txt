CLASS Solution  
    FUNCTION loudAndRich(richer_list_of_pairs quieter_list_of_numbers) RETURNS list_of_numbers  
        SET graph TO a mapping from number to a list of numbers with default empty list  
        
        FOR each pair consisting of person_a AND person_b IN richer_list_of_pairs  
            APPEND person_a TO the list associated with person_b in graph  
        END FOR  
        
        SET total_person_count TO the LENGTH OF quieter_list_of_numbers  
        
        SET answer_list TO a list containing total_person_count elements each SET TO negative one  
        
        FUNCTION dfs(person) RETURNS number  
            IF the element at position person of answer_list EQUALS negative one THEN  
                SET the element at position person of answer_list TO person  
                
                FOR each neighbor IN the list associated with person in graph  
                    SET candidate_person TO the result of dfs with neighbor as argument  
                    IF the element at position candidate_person of quieter_list_of_numbers LESS THAN the element at position of person of answer_list in quieter_list_of_numbers THEN  
                        SET the element at position person of answer_list TO candidate_person  
                    END IF  
                END FOR  
            END IF  
            RETURN the element at position person of answer_list  
        END FUNCTION  
        
        FOR index FROM zero TO total_person_count MINUS one  
            CALL dfs WITH index AS argument  
        END FOR  
        
        RETURN answer_list  
    END FUNCTION  
END CLASS
