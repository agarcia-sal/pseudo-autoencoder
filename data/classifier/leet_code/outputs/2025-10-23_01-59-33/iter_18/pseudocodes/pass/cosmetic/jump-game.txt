CLASS Solution
    FUNCTION canJump(nums)
        ASSIGN 0 TO positionIndex
        ASSIGN 0 TO maximumReach

        WHILE positionIndex LESS THAN LENGTH(nums)
            IF NOT(positionIndex LESS THAN OR EQUAL TO maximumReach)
                OUTPUT false
                EXIT FUNCTION
            END IF

            ASSIGN (positionIndex PLUS nums[positionIndex]) TO candidateReach
            IF candidateReach GREATER THAN maximumReach
                ASSIGN candidateReach TO maximumReach
            END IF

            INCREMENT positionIndex BY 1
        END WHILE

        IF maximumReach GREATER THAN OR EQUAL TO (LENGTH(nums) MINUS 1)
            OUTPUT true
            EXIT FUNCTION
        END IF

        OUTPUT false
    END FUNCTION
END CLASS
