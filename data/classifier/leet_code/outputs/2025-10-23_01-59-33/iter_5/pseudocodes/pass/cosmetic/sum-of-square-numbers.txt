CLASS Solution  
    FUNCTION judgeSquareSum WITH parameter c  
        SET upperBound AS INTEGER PART OF THE SQUARE ROOT OF c  
        SET lowerBound AS 0  
        LOOP FOREVER  
            IF lowerBound GREATER THAN upperBound THEN  
                RETURN false  
            END IF  
            SET total AS (upperBound * upperBound) + (lowerBound * lowerBound)  
            SWITCH total COMPARED TO c  
                CASE EQUAL:  
                    RETURN true  
                CASE LESS:  
                    SET lowerBound TO lowerBound + 1  
                CASE GREATER:  
                    SET upperBound TO upperBound - 1  
            END SWITCH  
        END LOOP  
    END FUNCTION  
END CLASS
