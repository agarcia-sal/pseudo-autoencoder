CLASS Solution  
    FUNCTION secondMinimum(n  
                            edges  
                            time  
                            change)  
        SET graph TO list of empty lists for each integer from one TO n PLUS one  
        FOR each pair u AND v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
          
        SET queue TO a double ended queue containing a tuple of one current vertex AND zero current time AND zero count  
        SET minTimes TO list containing two elements each set to infinite value for each integer from one TO n PLUS one  
        SET element at position one of minTimes at position zero TO zero  
          
        WHILE queue is not empty  
            REMOVE the first element FROM queue AND ASSIGN TO current AND currentTime AND count  
              
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
              
            IF integer division of currentTime BY change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO (integer division of currentTime BY change PLUS one) MULTIPLIED BY change PLUS time  
            END IF  
              
            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor of minTimes at position zero GREATER THAN nextTime  
                    SET element at position neighbor of minTimes at position one TO element at position neighbor of minTimes at position zero  
                    SET element at position neighbor of minTimes at position zero TO nextTime  
                    APPEND tuple containing neighbor AND nextTime AND zero TO queue  
                ELSE IF element at position neighbor of minTimes at position zero LESS THAN nextTime AND nextTime LESS THAN element at position neighbor of minTimes at position one  
                    SET element at position neighbor of minTimes at position one TO nextTime  
                    APPEND tuple containing neighbor AND nextTime AND one TO queue  
                END IF  
            END FOR  
        END WHILE  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
