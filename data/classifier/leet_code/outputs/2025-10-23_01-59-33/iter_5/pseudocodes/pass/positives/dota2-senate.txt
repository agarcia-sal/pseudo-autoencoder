CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        ASSIGN reference TO new double-ended queue named radiant  
        ASSIGN reference TO new double-ended queue named dire  

        FOR each index i AND character s IN senate  
            IF character s EQUALS the character representing Radiant party  
                APPEND index i TO radiant  
            ELSE  
                APPEND index i TO dire  
            END IF  
        END FOR  

        WHILE radiant is not empty AND dire is not empty  
            REMOVE and ASSIGN the first element from radiant TO r_index  
            REMOVE and ASSIGN the first element from dire TO d_index  

            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  

        IF radiant is not empty  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
