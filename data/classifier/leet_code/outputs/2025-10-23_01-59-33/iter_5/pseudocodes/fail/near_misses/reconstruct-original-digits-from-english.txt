CLASS Solution
    FUNCTION originalDigits(s)
        DEFINE frequencyMap AS empty map from character to integer
        FOR character IN s
            IF character NOT IN frequencyMap THEN
                frequencyMap[character] ← 0
            END IF
            frequencyMap[character] ← frequencyMap[character] + 1
        END FOR

        DEFINE digitCounts AS array of size 10 FILLED WITH 0

        PROCEDURE countOccurrences(keyChar : character) RETURNS integer
            IF keyChar IN frequencyMap THEN
                RETURN frequencyMap[keyChar]
            ELSE
                RETURN 0
            END IF
        END PROCEDURE

        digitCounts[0] ← countOccurrences('z')
        digitCounts[2] ← countOccurrences('w')
        digitCounts[4] ← countOccurrences('u')
        digitCounts[6] ← countOccurrences('x')
        digitCounts[8] ← countOccurrences('g')

        digitCounts[3] ← countOccurrences('h') - digitCounts[8]
        digitCounts[5] ← countOccurrences('f') - digitCounts[4]
        digitCounts[7] ← countOccurrences('s') - digitCounts[6]

        digitCounts[9] ← countOccurrences('i') - digitCounts[5] - digitCounts[6] - digitCounts[8]
        digitCounts[1] ← countOccurrences('o') - digitCounts[0] - digitCounts[2] - digitCounts[4]

        DEFINE buildResult(index : integer, accumulated : list of string) RETURNS list of string
            IF index = 10 THEN
                RETURN accumulated
            END IF
            DEFINE countLeft AS digitCounts[index]
            DEFINE appendCopies(acc : list of string, n : integer) RETURNS list of string
                IF n = 0 THEN
                    RETURN acc
                ELSE
                    RETURN appendCopies(acc + [string(index)], n - 1)
                END IF
            END FUNCTION
            RETURN buildResult(index + 1, appendCopies(accumulated, countLeft))
        END FUNCTION

        RETURN join(buildResult(0, empty list), "")
    END FUNCTION
END CLASS
