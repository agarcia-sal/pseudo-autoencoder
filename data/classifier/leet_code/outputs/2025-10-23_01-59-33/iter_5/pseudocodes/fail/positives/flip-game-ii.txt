CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN the value associated with state in memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the element at position index of state EQUALS the plus character AND the element at position index PLUS one of state EQUALS the plus character THEN  
                    SET new_state TO the substring from position zero TO position index MINUS one CONCATENATED WITH two minus characters CONCATENATED WITH the substring from position index PLUS two TO the END OF state  
                    IF the result of calling can_win_helper WITH new_state IS EQUAL TO false THEN  
                        SET the value associated with state in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with state in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
