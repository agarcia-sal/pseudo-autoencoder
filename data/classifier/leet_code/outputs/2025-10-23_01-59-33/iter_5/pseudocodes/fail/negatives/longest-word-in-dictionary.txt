CLASS Solution
    FUNCTION longestWord(list_of_words)
        CALL sort list_of_words BY the LENGTH OF each element AND THEN LEXICOGRAPHICALLY
        
        SET valid_words TO an empty set
        SET longest_word TO empty string
        
        FOR each word IN list_of_words
            IF the LENGTH OF word EQUALS one OR substring from position zero TO the LENGTH OF word MINUS one of word IS IN valid_words
                ADD word TO valid_words
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word
                    SET longest_word TO word
                END IF
            END IF
        END FOR
        
        RETURN longest_word
    END FUNCTION
END CLASS
