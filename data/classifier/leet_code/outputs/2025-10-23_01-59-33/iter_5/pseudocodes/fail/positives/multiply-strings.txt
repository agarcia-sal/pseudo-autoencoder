CLASS Solution  
    FUNCTION multiply(num1 string num2 string) RETURNS string  
        IF num1 EQUALS string zero OR num2 EQUALS string zero  
            RETURN string zero  
        END IF  

        SET result TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  

        SET num1 TO the reversal of num1  
        SET num2 TO the reversal of num2  

        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the character code of the element at position i of num1 MINUS the character code of string zero MULTIPLIED BY the character code of the element at position j of num2 MINUS the character code of string zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum TO mul PLUS the element at position p1 of result  

                SET the element at position p1 of result TO sum MODULO ten  
                INCREMENT the element at position p2 of result BY the integer division of sum BY ten  
            END FOR  
        END FOR  

        SET result_str TO the concatenation of the string representations of the elements of result in reverse order  
        RETURN result_str with all leading string zeros removed  
    END FUNCTION  
END CLASS
