CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        ASSIGN LENGTH OF nums TO varAlpha  
        IF varAlpha IS LESS THAN 3 THEN  
            RETURN 0  
        END IF  
  
        ASSIGN AN EMPTY ARRAY OF LENGTH varAlpha TO varOmega  
        FOR lambda FROM 0 TO varAlpha MINUS 1 DO  
            ASSIGN AN EMPTY DICTIONARY MAPPING INTEGER TO INTEGER TO varOmega[lambda]  
        END FOR  
  
        ASSIGN 0 TO accumulator  
        SET outerIndex TO 1  
        WHILE outerIndex LESS THAN varAlpha DO  
            SET innerIndex TO 0  
            WHILE innerIndex LESS THAN outerIndex DO  
                ASSIGN nums[outerIndex] MINUS nums[innerIndex] TO diffValue  
  
                IF diffValue IS IN KEYS OF varOmega[innerIndex] THEN  
                    INCREMENT varOmega[outerIndex][diffValue] BY varOmega[innerIndex][diffValue] PLUS 1  
                    INCREMENT accumulator BY varOmega[innerIndex][diffValue]  
                ELSE  
                    ASSIGN 1 TO varOmega[outerIndex][diffValue]  
                END IF  
  
                INCREMENT innerIndex BY 1  
            END WHILE  
            INCREMENT outerIndex BY 1  
        END WHILE  
  
        RETURN accumulator  
    END FUNCTION  
END CLASS
