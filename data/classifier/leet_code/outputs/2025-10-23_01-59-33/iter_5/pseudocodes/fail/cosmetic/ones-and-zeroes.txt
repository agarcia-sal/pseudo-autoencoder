CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        DECLARE matrix AS LIST OF LIST OF INTEGER ← LIST CONTAINING (m + 1) ELEMENTS, EACH BEING A LIST OF (n + 1) ZEROS  
        
        INDEX ← 0  
        WHILE INDEX < LENGTH OF strs  
            STRING currentString ← strs AT INDEX  
            ZEROES ← 0  
            ONES ← 0  
            
            POSITION ← 0  
            WHILE POSITION < LENGTH OF currentString  
                CHARACTER char ← currentString AT POSITION  
                IF char = '0' THEN  
                    ZEROES ← ZEROES + 1  
                ELSE IF char = '1' THEN  
                    ONES ← ONES + 1  
                END IF  
                POSITION ← POSITION + 1  
            END WHILE  
            
            iCounter ← m  
            WHILE iCounter >= ZEROES  
                jCounter ← n  
                REPEAT  
                    currentVal ← matrix AT iCounter AT jCounter  
                    candidateVal ← matrix AT (iCounter - ZEROES) AT (jCounter - ONES) + 1  
                    
                    IF candidateVal > currentVal THEN  
                        matrix AT iCounter AT jCounter ← candidateVal  
                    END IF  
                    jCounter ← jCounter - 1  
                UNTIL jCounter < ONES  
                iCounter ← iCounter - 1  
            END WHILE  
            
            INDEX ← INDEX + 1  
        END WHILE  
        
        RETURN matrix AT m AT n  
    END FUNCTION  
END CLASS
