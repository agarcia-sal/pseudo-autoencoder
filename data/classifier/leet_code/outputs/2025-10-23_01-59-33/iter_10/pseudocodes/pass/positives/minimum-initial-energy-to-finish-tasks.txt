CLASS Solution  
    FUNCTION minimumEffort(tasks)  
        SORT tasks IN ORDER WHERE the element representing the difference between the second element MINUS the first element of each task IS IN DESCENDING ORDER  
        SET current_energy TO zero  
        SET required_initial_energy TO zero  
        FOR each element IN tasks  
            ASSIGN actual TO the first element of element  
            ASSIGN minimum TO the second element of element  
            IF current_energy LESS THAN minimum THEN  
                INCREMENT required_initial_energy BY minimum MINUS current_energy  
                SET current_energy TO minimum  
            END IF  
            DECREMENT current_energy BY actual  
        END FOR  
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
