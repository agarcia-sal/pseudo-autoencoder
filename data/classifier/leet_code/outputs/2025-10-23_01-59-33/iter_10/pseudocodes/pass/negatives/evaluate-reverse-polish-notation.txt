CLASS Solution
    FUNCTION evalRPN(tokens_of_strings)
        SET stack TO an empty list
        FOR each token IN tokens_of_strings
            IF token IS IN the collection plus minus multiplied by divided by
                SET b TO the element removed FROM the end of stack
                SET a TO the element removed FROM the end of stack
                IF token EQUALS plus
                    APPEND a PLUS b TO stack
                ELSE IF token EQUALS minus
                    APPEND a MINUS b TO stack
                ELSE IF token EQUALS multiplied by
                    APPEND a MULTIPLIED BY b TO stack
                ELSE IF token EQUALS divided by
                    APPEND the integer conversion OF a DIVIDED BY b TO stack
                END IF
            ELSE
                APPEND the integer conversion OF token TO stack
            END IF
        END FOR
        RETURN the element at position zero of stack
    END FUNCTION
END CLASS
