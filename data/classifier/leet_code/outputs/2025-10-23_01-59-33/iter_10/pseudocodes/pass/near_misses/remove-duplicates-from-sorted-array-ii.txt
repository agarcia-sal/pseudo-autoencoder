CLASS Solution
    FUNCTION removeDuplicates(nums)
        DECLARE counter ← 0
        IF counter EQUALS LENGTH(nums) THEN
            RETURN counter
        ELSE
            DECLARE definer ← 0
            DECLARE temp_list ← new Queue()
            
            WHILE definer < MIN(2, LENGTH(nums))
                ENQUEUE(nums[definer], temp_list)
                definer ← definer + 1
            END WHILE
            
            WHILE NOT IS_EMPTY(temp_list)
                nums[counter] ← DEQUEUE(temp_list)
                counter ← counter + 1
            END WHILE
            
            FUNCTION recurse(pos)
                IF pos EQUALS LENGTH(nums) THEN
                    RETURN
                END IF
                IF nums[pos] ≠ nums[counter - 2] THEN
                    nums[counter] ← nums[pos]
                    counter ← counter + 1
                END IF
                recurse(pos + 1)
            END FUNCTION
            
            recurse(2)
            
            RETURN counter
        END IF
    END FUNCTION
END CLASS
