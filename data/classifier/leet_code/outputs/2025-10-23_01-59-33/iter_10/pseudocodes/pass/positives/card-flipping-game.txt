CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of front_card AND back_card IN fronts AND backs simultaneously  
            IF front_card EQUALS back_card  
                ADD front_card TO same  
            END IF  
        END FOR  
        SET all_numbers TO empty set  
        FOR each number IN fronts  
            ADD number TO all_numbers  
        END FOR  
        FOR each number IN backs  
            ADD number TO all_numbers  
        END FOR  
        SET good_numbers TO empty set  
        FOR each number IN all_numbers  
            IF number NOT IN same  
                ADD number TO good_numbers  
            END IF  
        END FOR  
        IF the LENGTH OF good_numbers IS GREATER THAN zero  
            SET minimum_good_number TO the smallest element IN good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
