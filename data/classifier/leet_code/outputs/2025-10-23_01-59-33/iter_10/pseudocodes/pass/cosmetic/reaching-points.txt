CLASS Solution
    FUNCTION reachingPoints(sx sx PARAMETER sy sy PARAMETER tx tx PARAMETER ty ty) RETURNS boolean
        LOOP_WHILE (tx IS_NOT_LESS_THAN sx) AND (ty IS_NOT_LESS_THAN sy)
            IF (NOT (tx IS_DIFFERS_FROM sx)) AND (NOT (ty IS_DIFFERS_FROM sy))
                RETURN True
            END IF

            IF tx IS_GREATER_THAN ty
                IF ty IS_GREATER_THAN sy
                    LET dividend = tx
                    LET divisor = ty
                    LET remainder = dividend MINUS (divisor TIMES FLOOR(dividend DIVIDED_BY divisor))
                    ASSIGN remainder INTO tx
                ELSE
                    LET numerator = tx MINUS sx
                    LET denominator = ty
                    LET modulusResult = numerator MINUS (denominator TIMES FLOOR(numerator DIVIDED_BY denominator))
                    IF NOT (modulusResult IS_NOT_EQUAL_ZERO)
                        RETURN True
                    ELSE
                        RETURN False
                    END IF
                END IF
            ELSE
                IF tx IS_GREATER_THAN sx
                    LET dividend = ty
                    LET divisor = tx
                    LET remainder = dividend MINUS (divisor TIMES FLOOR(dividend DIVIDED_BY divisor))
                    ASSIGN remainder INTO ty
                ELSE
                    LET numerator = ty MINUS sy
                    LET denominator = tx
                    LET modulusResult = numerator MINUS (denominator TIMES FLOOR(numerator DIVIDED_BY denominator))
                    IF NOT (modulusResult IS_NOT_EQUAL_ZERO)
                        RETURN True
                    ELSE
                        RETURN False
                    END IF
                END IF
            END IF
        END LOOP_WHILE
        RETURN False
    END FUNCTION
END CLASS
