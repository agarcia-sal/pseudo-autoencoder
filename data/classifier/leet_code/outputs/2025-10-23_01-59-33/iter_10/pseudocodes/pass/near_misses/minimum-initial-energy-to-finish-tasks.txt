CLASS Solution
    FUNCTION minimumEffort(tasks)
        DEFINE sorted_list AS tasks SORTED BY (task -> task[1] - task[0]) IN DESCENDING ORDER
        DECLARE energy_accumulator AS 0
        DECLARE base_energy_needed AS 0

        FOR index FROM 0 TO LENGTH(sorted_list) - 1 DO
            LET task_entry BE sorted_list[index]
            LET consume_val BE task_entry[0]
            LET required_minimum BE task_entry[1]

            IF energy_accumulator >= required_minimum THEN
                CONTINUE
            ENDIF

            LET energy_gap BE required_minimum - energy_accumulator
            base_energy_needed = base_energy_needed + energy_gap
            energy_accumulator = required_minimum

            energy_accumulator = energy_accumulator - consume_val
        ENDFOR

        RETURN base_energy_needed
    END FUNCTION
END CLASS
