CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET baseline_min TO arrays[0][0]  
        SET baseline_max TO arrays[0][ LENGTH(arrays[0]) - 1 ]  
        SET greatest_gap TO 0  

        SET cursor TO 1  
        WHILE cursor < LENGTH(arrays)  
            SET segment TO arrays[cursor]  
            SET segment_minimum TO segment[0]  
            SET segment_maximum TO segment[ LENGTH(segment) - 1 ]  

            SET candidate_one TO ABSOLUTE_VALUE(segment_maximum - baseline_min)  
            SET candidate_two TO ABSOLUTE_VALUE(baseline_max - segment_minimum)  
            IF candidate_one > candidate_two THEN  
                SET better_candidate TO candidate_one  
            ELSE  
                SET better_candidate TO candidate_two  
            END IF  

            IF greatest_gap < better_candidate THEN  
                SET greatest_gap TO better_candidate  
            END IF  

            IF baseline_min > segment_minimum THEN  
                SET baseline_min TO segment_minimum  
            END IF  

            IF baseline_max < segment_maximum THEN  
                SET baseline_max TO segment_maximum  
            END IF  

            SET cursor TO cursor + 1  
        END WHILE  
        RETURN greatest_gap  
    END FUNCTION  
END CLASS
