CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        SET stone_set TO the SET of all elements in stones  
        SET memo TO an EMPTY DICTIONARY  

        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            IF the TUPLE of position AND jump IS IN memo  
                RETURN the VALUE associated with the TUPLE of position AND jump in memo  
            END IF  
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND the SUM of position AND next_jump IS IN stone_set  
                    IF can_jump_to(the SUM of position AND next_jump, next_jump)  
                        SET memo at key TUPLE of position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET memo at key TUPLE of position AND jump TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_jump_to(the element at the first position of stones, zero)  
    END FUNCTION  
END CLASS
