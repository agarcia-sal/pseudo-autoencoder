CLASS Solution
    FUNCTION grayCode(number_n)
        IF number_n EQUALS zero
            RETURN list containing zero
        END IF
        SET previous_gray_codes TO CALL grayCode WITH number_n MINUS one
        SET bit_mask TO one SHIFTED LEFT BY number_n MINUS one
        SET reflected_list TO CALL helper_function_to_return_list_of bit_mask OR number FOR number IN reversed previous_gray_codes
        SET current_gray_codes TO previous_gray_codes CONCATENATED WITH reflected_list
        RETURN current_gray_codes
    END FUNCTION
END CLASS
