CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN -1  
        END IF  

        FUNCTION dp(i, d)  
            IF d EQUALS 1  
                RETURN MAXIMUM VALUE IN jobDifficulty FROM INDEX i TO END  
            END IF  

            SET min_difficulty TO infinity  
            SET current_max TO 0  
            FOR j FROM i TO n MINUS d  
                SET current_max TO MAXIMUM OF current_max AND jobDifficulty AT j  
                SET tentative_difficulty TO current_max PLUS dp(j PLUS 1, d MINUS 1)  
                SET min_difficulty TO MINIMUM OF min_difficulty AND tentative_difficulty  
            END FOR  

            RETURN min_difficulty  
        END FUNCTION  

        RETURN dp(0, d)  
END CLASS
