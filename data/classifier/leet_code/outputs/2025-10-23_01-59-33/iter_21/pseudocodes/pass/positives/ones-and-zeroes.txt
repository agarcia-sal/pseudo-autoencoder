CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO a two dimensional list with the dimensions of m plus one by n plus one where each element is zero  
        FOR each string s IN strs  
            SET count_0 TO the number of elements EQUALS to zero in the string s  
            SET count_1 TO the number of elements EQUALS to one in the string s  
            FOR variable i FROM m DOWN TO count_0  
                FOR variable j FROM n DOWN TO count_1  
                    SET dp at position i and position j TO the maximum value between dp at position i and position j AND the value of dp at position i MINUS count_0 and position j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position m and position n  
    END FUNCTION  
END CLASS
