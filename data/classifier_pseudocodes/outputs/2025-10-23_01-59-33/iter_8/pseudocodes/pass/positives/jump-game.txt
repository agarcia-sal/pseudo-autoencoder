CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF i GREATER THAN farthest  
                RETURN false  
            END IF  
            SET possible_reach TO i PLUS element at position i of nums  
            IF possible_reach GREATER THAN farthest  
                SET farthest TO possible_reach  
            END IF  
        END FOR  
        IF farthest GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
