CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET combinedPairs AS EMPTY LIST  
        SET idx TO 0  
        WHILE idx LESS THAN LENGTH OF position  
            APPEND (position[idx], speed[idx]) TO combinedPairs  
            INCREMENT idx BY 1  
        END WHILE  

        SORT combinedPairs BY FIRST ELEMENT DESCENDING  

        SET arrivalTimes AS EMPTY LIST  
        SET pairIndex TO 0  
        WHILE pairIndex LESS THAN LENGTH OF combinedPairs  
            LET loc BE combinedPairs[pairIndex][0]  
            LET vel BE combinedPairs[pairIndex][1]  
            LET timeVal BE (target MINUS loc) DIVIDED BY vel  
            APPEND timeVal TO arrivalTimes  
            INCREMENT pairIndex BY 1  
        END WHILE  

        SET fleetCount TO 0  
        SET maxObservedTime TO 0  

        SET timeIdx TO 0  
        WHILE timeIdx LESS THAN LENGTH OF arrivalTimes  
            LET currentTime BE arrivalTimes[timeIdx]  
            IF NOT (currentTime LESS OR EQUAL maxObservedTime) THEN  
                SET maxObservedTime TO currentTime  
                SET fleetCount TO fleetCount PLUS 1  
            END IF  
            INCREMENT timeIdx BY 1  
        END WHILE  

        RETURN fleetCount  
    END FUNCTION  
END CLASS
