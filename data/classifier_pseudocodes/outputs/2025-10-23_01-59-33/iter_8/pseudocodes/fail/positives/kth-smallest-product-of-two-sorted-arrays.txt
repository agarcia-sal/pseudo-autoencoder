CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        
        FUNCTION count(p PARAMETER)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION TO THE RIGHT IN nums2 OF the value p DIVIDED BY x  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION TO THE LEFT IN nums2 OF the value p DIVIDED BY x  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the LOGICAL VALUE OF p GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  
        
        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the element at POSITION zero OF nums1 AND the ABSOLUTE VALUE OF the element at the LAST POSITION OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the element at POSITION zero OF nums2 AND the ABSOLUTE VALUE OF the element at the LAST POSITION OF nums2  
        RETURN the POSITION TO THE LEFT WITHIN THE RANGE FROM NEGATIVE mx TO mx PLUS one OF the value k WHERE the KEY FUNCTION IS count MINUS mx  
    END FUNCTION  
END CLASS
