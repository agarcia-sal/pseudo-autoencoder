CLASS Solution
    FUNCTION countVowelPermutation(n)
        DEFINE DIVISOR AS 1_000_000_007
        INITIALIZE alpha AS 1
        INITIALIZE beta AS 1
        INITIALIZE gamma AS 1
        INITIALIZE delta AS 1
        INITIALIZE epsilon AS 1

        SET counter TO 2
        WHILE counter ≤ n DO
            DEFINE next_alpha AS beta + gamma + epsilon
            DEFINE next_beta AS alpha + gamma
            DEFINE next_gamma AS beta + delta
            DEFINE next_delta AS gamma
            DEFINE next_epsilon AS gamma + delta

            alpha ← next_alpha MODULO DIVISOR
            beta ← next_beta MODULO DIVISOR
            gamma ← next_gamma MODULO DIVISOR
            delta ← next_delta MODULO DIVISOR
            epsilon ← next_epsilon MODULO DIVISOR

            INCREMENT counter BY 1
        END WHILE

        COMPUTE aggregate_total = alpha + beta + gamma + delta + epsilon
        ASSIGN final_result AS aggregate_total MODULO DIVISOR
        RETURN final_result
    END FUNCTION
END CLASS
