CLASS Solution
    FUNCTION maximizeSweetness(sweetness list OF integer, k integer) RETURNS integer
        FUNCTION canDivide(min_sweet integer) RETURNS boolean
            ASSIGN 0 TO count_pieces
            ASSIGN 0 TO aggregate_sweet
            ASSIGN 0 TO idx
            WHILE idx LESS THAN LENGTH(sweetness)
                ASSIGN aggregate_sweet PLUS sweetness[idx] TO aggregate_sweet
                IF NOT (aggregate_sweet LESS THAN min_sweet)
                    ASSIGN count_pieces PLUS 1 TO count_pieces
                    ASSIGN 0 TO aggregate_sweet
                    IF NOT (count_pieces LESS THAN OR EQUAL TO k)
                        RETURN True
                    END IF
                END IF
                INCREMENT idx BY 1
            END WHILE
            RETURN count_pieces GREATER THAN k
        END FUNCTION

        ASSIGN sweetness[0] TO low
        ASSIGN 0 TO i
        WHILE i LESS THAN LENGTH(sweetness)
            IF sweetness[i] LESS THAN low
                ASSIGN sweetness[i] TO low
            END IF
            INCREMENT i BY 1
        END WHILE

        ASSIGN 0 TO total_value
        ASSIGN 0 TO j
        WHILE j LESS THAN LENGTH(sweetness)
            ASSIGN total_value PLUS sweetness[j] TO total_value
            INCREMENT j BY 1
        END WHILE

        ASSIGN total_value DIVIDED BY (k PLUS 1) TO high
        ASSIGN 0 TO final_ans

        WHILE low LESS THAN OR EQUAL TO high
            ASSIGN (low PLUS high) DIVIDED BY 2 TO mid_val
            IF canDivide(mid_val) IS True
                ASSIGN mid_val TO final_ans
                ASSIGN mid_val PLUS 1 TO low
                CONTINUE
            ELSE
                ASSIGN mid_val MINUS 1 TO high
            END IF
        END WHILE

        RETURN final_ans
    END FUNCTION
END CLASS
