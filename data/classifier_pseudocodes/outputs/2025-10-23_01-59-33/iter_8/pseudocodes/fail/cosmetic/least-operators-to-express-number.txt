CLASS Solution
    FUNCTION leastOpsExpressTarget(x, target)
        FUNCTION dp(target_inner)
            IF NOT (target_inner ≠ 0) THEN
                RETURN -1
            END IF

            IF target_inner >= x THEN
                INTEGER exponent_tracker ← 1
                WHILE POWER(x, exponent_tracker) ≤ target_inner DO
                    exponent_tracker ← exponent_tracker + 1
                END WHILE
                exponent_tracker ← exponent_tracker - 1

                INTEGER intermediate_value ← dp(target_inner - POWER(x, exponent_tracker)) + exponent_tracker

                INTEGER candidate ← POWER(x, exponent_tracker + 1) - target_inner
                IF candidate < target_inner THEN
                    INTEGER alternate_value ← dp(candidate) + exponent_tracker + 1
                    intermediate_value ← (intermediate_value < alternate_value) ? intermediate_value : alternate_value
                END IF

                RETURN intermediate_value
            ELSE
                INTEGER left_option ← 2 * target_inner - 1
                INTEGER right_option ← 2 * (x - target_inner)
                RETURN (left_option < right_option) ? left_option : right_option
            END IF
        END FUNCTION

        RETURN dp(target)
    END FUNCTION
END CLASS
