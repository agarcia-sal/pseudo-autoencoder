CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT the elements of nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS the element at position i of nums  
                CONTINUE to the next iteration  
            END IF  
            SET count TO a mapping of each element of nums to its frequency  
            SET arr TO an empty list  
            FOR each element x IN nums  
                IF the value at key x in count EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the value at key the sum of x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of the loop  
                END IF  
                APPEND the sum of x PLUS k TO arr  
                DECREMENT the value at key x in count BY one  
                DECREMENT the value at key the sum of x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
