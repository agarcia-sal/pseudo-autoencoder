CLASS Solution
	 FUNCTION findMinDifference(timePoints)
		  FUNCTION to_minutes(time)
				LET parts BE a queue of strings ENQUEUED by splitting time at ':'
				LET hours_val BE integer conversion of DEQUEUED element from parts
				LET minutes_val BE integer conversion of DEQUEUED element from parts
				RETURN (hours_val * 60) + minutes_val
		  END FUNCTION

		  LET minutes_seq BE an empty stack
		  LET idx BE 0
		  WHILE idx LESS THAN LENGTH of timePoints
				PUSH to_minutes(timePoints[idx]) ONTO minutes_seq
				SET idx TO idx + 1
		  END WHILE
		  
		  FUNCTION sort_stack(s)
				IF s IS empty
					 RETURN s
				END IF
				LET tmp BE POP from s
				CALL sort_stack(s)
				CALL insert_sorted(s, tmp)
		  END FUNCTION
		  
		  FUNCTION insert_sorted(s, val)
				IF s IS empty OR s TOP ELEMENT GREATER THAN val
					 PUSH val ONTO s
					 RETURN
				END IF
				LET top_val BE POP from s
				CALL insert_sorted(s, val)
				PUSH top_val ONTO s
		  END FUNCTION
		  
		  CALL sort_stack(minutes_seq)
		  
		  LET min_diff BE infinity
		  FUNCTION extract_min_diff(previous, stack, current_min)
				IF stack IS empty THEN RETURN current_min END IF
				
				LET current_val BE POP from stack
				LET diff_val BE current_val - previous
				LET new_min BE IF diff_val LESS THAN current_min THEN diff_val ELSE current_min
				RETURN extract_min_diff(current_val, stack, new_min)
		  END FUNCTION

		  LET last_val BE POP from minutes_seq
		  LET min_diff_inner BE extract_min_diff(last_val, minutes_seq, min_diff)
		  
		  LET circular_gap BE (1440 - last_val) + minutes_seq[0]  // Access bottom of stack by indexing
		  LET result_val BE IF circular_gap LESS THAN min_diff_inner THEN circular_gap ELSE min_diff_inner
		  
		  RETURN result_val
	 END FUNCTION
END CLASS
