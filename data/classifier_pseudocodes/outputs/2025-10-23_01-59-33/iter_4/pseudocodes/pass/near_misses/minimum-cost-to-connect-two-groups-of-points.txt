CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF the element at position zero of cost  
        INITIALIZE dp AS a two dimensional list with size1 PLUS one rows AND two RAISED TO THE power OF size2 columns  
        FOR each row_index FROM zero TO size1  
            FOR each column_index FROM zero TO two RAISED TO THE power OF size2 MINUS one  
                SET the element at row_index and column_index of dp TO infinity  
            END FOR  
        END FOR  
        SET the element at zero and zero of dp TO zero  
      
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two RAISED TO THE power OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET new_mask TO mask OR two RAISED TO THE power OF j  
                    IF the element at i PLUS one and new_mask of dp IS GREATER THAN the element at i and mask of dp PLUS the element at i and j of cost  
                        SET the element at i PLUS one and new_mask of dp TO the element at i and mask of dp PLUS the element at i and j of cost  
                    END IF  
                END FOR  
                FOR j FROM zero TO size2 MINUS one  
                    IF mask AND two RAISED TO THE power OF j HOLDS  
                        SET toggled_mask TO mask XOR two RAISED TO THE power OF j  
                        IF the element at i PLUS one and mask of dp IS GREATER THAN the element at i PLUS one and toggled_mask of dp PLUS the element at i and j of cost  
                            SET the element at i PLUS one and mask of dp TO the element at i PLUS one and toggled_mask of dp PLUS the element at i and j of cost  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
      
        RETURN the element at size1 and two RAISED TO THE power OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
