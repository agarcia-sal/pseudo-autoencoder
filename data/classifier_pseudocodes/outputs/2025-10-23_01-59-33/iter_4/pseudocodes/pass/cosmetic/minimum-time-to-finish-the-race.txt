CLASS Solution
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)
        LET minimal_times BE an array of length 15 WHERE each element IS assigned +∞
        FOR EACH (initial_value, multiplier) IN tires DO
            LET accumulated_time BE 0
            LET lap_counter BE 0
            WHILE lap_counter < 15 DO
                LET current_lap_duration BE initial_value * (multiplier ^ lap_counter)
                IF NOT (current_lap_duration ≤ changeTime + initial_value) THEN
                    EXIT WHILE
                END IF
                accumulated_time ← accumulated_time + current_lap_duration
                IF accumulated_time < minimal_times[lap_counter] THEN
                    minimal_times[lap_counter] ← accumulated_time
                END IF
                lap_counter ← lap_counter + 1
            END WHILE
        END FOR

        LET dp_table BE a list of size (numLaps + 1) FILLED WITH +∞
        SET dp_table[0] TO 0

        LET outer_index BE 1
        WHILE outer_index ≤ numLaps DO
            LET inner_index BE 0
            WHILE inner_index ≤ MIN(outer_index - 1, 14) DO
                LET proposed_time BE dp_table[outer_index - inner_index - 1] + minimal_times[inner_index] + changeTime
                IF proposed_time < dp_table[outer_index] THEN
                    dp_table[outer_index] ← proposed_time
                END IF
                inner_index ← inner_index + 1
            END WHILE
            outer_index ← outer_index + 1
        END WHILE

        RETURN dp_table[numLaps] - changeTime
    END FUNCTION
END CLASS
