CLASS Solution
    FUNCTION longestMountain(arr)
        SET length_arr AS LENGTH OF arr
        SET maximum_length TO 0
        SET index_counter TO 1
        LOOP
            IF NOT (index_counter < length_arr - 1) THEN
                BREAK
            END IF

            IF (arr[index_counter - 1] < arr[index_counter]) AND (arr[index_counter] > arr[index_counter + 1]) THEN
                SET left_marker AS index_counter - 1
                SET right_marker AS index_counter + 1

                WHILE (left_marker > 0) AND (arr[left_marker - 1] < arr[left_marker])
                    left_marker ← left_marker - 1
                END WHILE

                WHILE (right_marker < length_arr - 1) AND (arr[right_marker] > arr[right_marker + 1])
                    right_marker ← right_marker + 1
                END WHILE

                SET current_mountain_length AS right_marker - left_marker + 1
                IF maximum_length < current_mountain_length THEN
                    maximum_length ← current_mountain_length
                END IF

                index_counter ← right_marker
            ELSE
                index_counter ← index_counter + 1
            END IF
        END LOOP
        RETURN maximum_length
    END FUNCTION
END CLASS
