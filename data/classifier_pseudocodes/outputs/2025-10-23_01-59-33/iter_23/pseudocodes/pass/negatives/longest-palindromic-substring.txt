CLASS Solution
    FUNCTION longestPalindrome(string_s)
        SET length_n TO the LENGTH OF string_s
        SET matrix_f TO auxiliary FUNCTION create_boolean_matrix WITH parameters length_n AND initial_value true
        SET variable_k TO zero
        SET variable_mx TO one
        FOR index_i FROM length_n MINUS two TO zero STEP minus one
            FOR index_j FROM index_i PLUS one TO length_n MINUS one
                SET element at position index_i AND index_j of matrix_f TO false
                IF element at position index_i of string_s EQUALS element at position index_j of string_s
                    SET element at position index_i AND index_j of matrix_f TO element at position index_i PLUS one AND index_j MINUS one of matrix_f
                    IF element at position index_i AND index_j of matrix_f AND variable_mx LESS THAN index_j MINUS index_i PLUS one
                        SET variable_k TO index_i
                        SET variable_mx TO index_j MINUS index_i PLUS one
                    END IF
                END IF
            END FOR
        END FOR
        RETURN substring from position variable_k TO variable_k PLUS variable_mx MINUS one of string_s
    END FUNCTION
END CLASS
