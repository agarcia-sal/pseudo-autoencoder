CLASS Solution
    FUNCTION multiply(num1 string, num2 string) RETURNS string
        IF num1 IS "0" OR num2 IS "0"
            RETURN "0"
        END IF

        DECLARE digits AS MAP from INT to INT
        DECLARE lengthSum AS INT = LENGTH(num1) + LENGTH(num2)
        FOR index IN RANGE(lengthSum)
            digits[index] = 0
        END FOR

        DECLARE reversed1 AS string = ""
        DECLARE reversed2 AS string = ""
        FOR p FROM LENGTH(num1) - 1 DOWNTO 0
            reversed1 = reversed1 + num1[p]
        END FOR
        FOR q FROM LENGTH(num2) - 1 DOWNTO 0
            reversed2 = reversed2 + num2[q]
        END FOR

        DECLARE function recursiveMul(i INT, j INT) RETURNS VOID
            IF i = LENGTH(reversed1)
                RETURN
            END IF
            IF j = LENGTH(reversed2)
                recursiveMul(i + 1, 0)
                RETURN
            END IF

            DECLARE valA AS INT = ASCII_CODE(reversed1[i]) - ASCII_CODE("0")
            DECLARE valB AS INT = ASCII_CODE(reversed2[j]) - ASCII_CODE("0")
            DECLARE product AS INT = valA * valB
            DECLARE baseIndex AS INT = i + j
            DECLARE nextIndex AS INT = baseIndex + 1
            DECLARE totalSum AS INT = product + digits[baseIndex]

            digits[baseIndex] = totalSum MOD 10
            digits[nextIndex] = digits[nextIndex] + (totalSum DIV 10)

            recursiveMul(i, j + 1)
        END FUNCTION

        CALL recursiveMul(0, 0)

        DECLARE output AS string = ""
        FOR key FROM lengthSum - 1 DOWNTO 0
            output = output + TO_STRING(digits[key])
        END FOR

        WHILE LENGTH(output) > 1 AND output[0] = "0"
            output = SUBSTRING(output, 1)
        END WHILE

        RETURN output
    END FUNCTION
END CLASS
