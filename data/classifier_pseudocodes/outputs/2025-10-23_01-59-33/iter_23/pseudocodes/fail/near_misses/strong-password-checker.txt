CLASS Solution
    FUNCTION strongPasswordChecker(password)
        SET n TO LENGTH OF password
        
        SET has_lower TO FALSE
        SET has_upper TO FALSE
        SET has_digit TO FALSE
        FOR each character c IN password
            IF c IS lowercase THEN
                SET has_lower TO TRUE
            IF c IS uppercase THEN
                SET has_upper TO TRUE
            IF c IS digit THEN
                SET has_digit TO TRUE
        
        SET missing_types TO 3 MINUS (has_lower + has_upper + has_digit)
        
        SET replace TO 0
        SET one TO 0
        SET two TO 0
        SET i TO 2
        WHILE i < n
            IF password[i] EQUALS password[i - 1] AND password[i - 1] EQUALS password[i - 2]
                SET length TO 2
                WHILE i < n AND password[i] EQUALS password[i - 1]
                    INCREMENT length BY 1
                    INCREMENT i BY 1
                INCREMENT replace BY length DIVIDED BY 3
                IF length MODULO 3 EQUALS 0
                    INCREMENT one BY 1
                ELSE IF length MODULO 3 EQUALS 1
                    INCREMENT two BY 1
            ELSE
                INCREMENT i BY 1
        
        IF n < 6
            RETURN MAXIMUM OF missing_types AND (6 MINUS n)
        
        ELSE IF n <= 20
            RETURN MAXIMUM OF missing_types AND replace
        
        ELSE
            SET delete TO n MINUS 20
            
            SET temp TO MINIMUM OF delete AND one
            DECREMENT replace BY temp
            SET delete TO delete MINUS temp
            
            SET temp TO MINIMUM OF delete AND (two MULTIPLIED BY 2)
            DECREMENT replace BY temp DIVIDED BY 2
            SET delete TO delete MINUS temp
            
            SET replace TO replace MINUS (delete DIVIDED BY 3)
            
            RETURN delete PLUS MAXIMUM OF missing_types AND replace
    END FUNCTION
END CLASS
