CLASS Solution  
    FUNCTION gridIllumination(n, lamps, queries)  
        SET row_count TO an empty dictionary with default integer zero  
        SET col_count TO an empty dictionary with default integer zero  
        SET diag_count TO an empty dictionary with default integer zero  
        SET anti_diag_count TO an empty dictionary with default integer zero  
        SET lamp_positions TO an empty set  
        
        FOR each pair r c IN the elements of lamps  
            IF the tuple of r and c is not in lamp_positions  
                ADD the tuple of r and c TO lamp_positions  
                INCREMENT the value at key r in row_count BY one  
                INCREMENT the value at key c in col_count BY one  
                INCREMENT the value at key the subtraction of c from r in diag_count BY one  
                INCREMENT the value at key the sum of r and c in anti_diag_count BY one  
            END IF  
        END FOR  
        
        SET directions TO the list of pairs negative one negative one negative one zero negative one one zero negative one zero zero zero zero one one negative one one zero one one one  
        
        SET result TO an empty list  
        
        FOR each pair r c IN the elements of queries  
            IF the value at key r in row_count is greater than zero OR the value at key c in col_count is greater than zero OR the value at key the subtraction of c from r in diag_count is greater than zero OR the value at key the sum of r and c in anti_diag_count is greater than zero  
                APPEND one TO result  
            ELSE  
                APPEND zero TO result  
            END IF  
            
            FOR each pair dr dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF the tuple of nr and nc is in lamp_positions  
                    REMOVE the tuple of nr and nc FROM lamp_positions  
                    DECREMENT the value at key nr in row_count BY one  
                    DECREMENT the value at key nc in col_count BY one  
                    DECREMENT the value at key the subtraction of nc from nr in diag_count BY one  
                    DECREMENT the value at key the sum of nr and nc in anti_diag_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
