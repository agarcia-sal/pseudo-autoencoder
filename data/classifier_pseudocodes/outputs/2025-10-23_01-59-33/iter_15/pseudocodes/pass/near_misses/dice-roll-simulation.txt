CLASS Solution
    FUNCTION dieSimulator(n, rollMax)
        SET MOD TO ten raised to the power of nine PLUS one
        
        SET dp TO a three dimensional list constructed as follows
            FOR index_i FROM zero TO n INCLUSIVE
                CONSTRUCT a two dimensional list as follows
                    FOR index_j FROM zero TO five INCLUSIVE
                        CONSTRUCT a list of zeros of length THE maximum value IN rollMax PLUS one
                    END FOR
                END CONSTRUCT
            END FOR
        END CONSTRUCT
        
        FOR index_j FROM zero TO five INCLUSIVE
            SET element at position one IN the list at position index_j IN the list at position one IN dp TO one
        END FOR
        
        FOR index_i FROM two TO n INCLUSIVE
            FOR index_j FROM zero TO five INCLUSIVE
                FOR index_k FROM one TO element at position index_j IN rollMax INCLUSIVE
                    IF index_k GREATER THAN one HOLDS
                        SET element at position index_k IN the list at position index_j IN the list at position index_i IN dp TO element at position index_k MINUS one IN the list at position index_j IN the list at position index_i MINUS one IN dp
                    ELSE
                        SET element at position index_k IN the list at position index_j IN the list at position index_i IN dp TO the SUM OF 
                            element at position index_y IN the list at position index_x IN the list at position index_i MINUS one IN dp
                            FOR each index_x FROM zero TO five INCLUSIVE
                                FOR each index_y FROM one TO element at position index_x IN rollMax INCLUSIVE
                                    INCLUDE the element IF index_x NOT EQUALS index_j
                                END FOR
                            END FOR
                        MODULO MOD
                    END IF
                END FOR
            END FOR
        END FOR
        
        RETURN the SUM OF
            element at position index_k IN the list at position index_j IN the list at position n IN dp
            FOR each index_j FROM zero TO five INCLUSIVE
                FOR each index_k FROM one TO element at position index_j IN rollMax INCLUSIVE
                END FOR
            END FOR
        MODULO MOD
    END FUNCTION
END CLASS
