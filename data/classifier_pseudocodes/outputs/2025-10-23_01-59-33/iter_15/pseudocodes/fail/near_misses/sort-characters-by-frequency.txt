CLASS Solution
    FUNCTION frequencySort WITH PARAMETER s
        INITIALIZE tally AS empty map
        FOR each symbol IN s
            IF symbol NOT IN tally THEN
                SET tally[symbol] AS 1
            ELSE
                SET currentCount AS tally[symbol]
                SET increasedCount AS currentCount + 1
                SET tally[symbol] AS increasedCount
            END IF
        END FOR

        INITIALIZE rankingArray AS empty list
        FOR each keyValuePair IN tally
            ADD keyValuePair TO rankingArray
        END FOR

        SET indexCounter AS 0
        WHILE indexCounter < LENGTH(rankingArray)
            SET pivotIndex AS indexCounter
            SET compareIndex AS indexCounter + 1
            WHILE compareIndex < LENGTH(rankingArray)
                IF rankingArray[compareIndex].value > rankingArray[pivotIndex].value THEN
                    SET pivotIndex AS compareIndex
                END IF
                ADD 1 TO compareIndex
            END WHILE
            IF pivotIndex != indexCounter THEN
                SET temporary AS rankingArray[indexCounter]
                SET rankingArray[indexCounter] AS rankingArray[pivotIndex]
                SET rankingArray[pivotIndex] AS temporary
            END IF
            ADD 1 TO indexCounter
        END WHILE

        INITIALIZE outputString AS empty string
        SET pointer AS 0
        WHILE pointer < LENGTH(rankingArray)
            SET currentEntry AS rankingArray[pointer]
            SET charVal AS currentEntry.key
            SET repeatTimes AS currentEntry.value
            SET repeatedSegment AS EMPTY STRING
            SET counterVar AS 0
            WHILE counterVar < repeatTimes
                SET repeatedSegment AS repeatedSegment + charVal
                ADD 1 TO counterVar
            END WHILE
            SET outputString AS outputString + repeatedSegment
            ADD 1 TO pointer
        END WHILE

        OUTPUT outputString
    END FUNCTION
END CLASS
