CLASS Solution  
    FUNCTION getHint(secret PARAMETER, guess PARAMETER)  
        IMPORT a data structure that provides a dictionary with default zero values
        
        SET bulls TO zero  
        SET cows TO zero  
        
        CREATE a dictionary named secret_count with default zero values  
        CREATE a dictionary named guess_count with default zero values  
        
        FOR each element s and element g taken correspondingly from secret and guess  
            IF element s EQUALS element g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key element s in secret_count BY one  
                INCREMENT the value at key element g in guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN the keys of secret_count  
            INCREMENT cows BY the smaller value BETWEEN the value at key digit in secret_count AND the value at key digit in guess_count  
        END FOR  
        
        RETURN a string expressing bulls FOLLOWED BY the character representing bulls AND THEN cows FOLLOWED BY the character representing cows  
    END FUNCTION  
END CLASS
