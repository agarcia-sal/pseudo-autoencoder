CLASS Solution
    FUNCTION maxTotalFruits(list_of_fruits, start_position, maximum_steps)
        SET maximum_total_fruits TO zero
        SET current_total_fruits TO zero
        SET left_pointer TO zero

        FOR right_pointer FROM zero TO the LENGTH OF list_of_fruits MINUS one
            SET position TO the element at position zero of the element at position right_pointer of list_of_fruits
            SET amount TO the element at position one of the element at position right_pointer of list_of_fruits
            INCREMENT current_total_fruits BY amount

            WHILE left_pointer IS LESS THAN OR EQUAL TO right_pointer AND NOT
                  (start_position MINUS maximum_steps IS LESS THAN OR EQUAL TO the element at position zero of the element at position left_pointer of list_of_fruits AND
                   the element at position zero of the element at position left_pointer of list_of_fruits IS LESS THAN OR EQUAL TO start_position PLUS maximum_steps AND
                   start_position MINUS maximum_steps IS LESS THAN OR EQUAL TO the element at position zero of the element at position right_pointer of list_of_fruits AND
                   the element at position zero of the element at position right_pointer of list_of_fruits IS LESS THAN OR EQUAL TO start_position PLUS maximum_steps AND
                   the MINIMUM OF the ABSOLUTE VALUE OF the element at position zero of the element at position right_pointer of list_of_fruits MINUS start_position AND
                                  the ABSOLUTE VALUE OF the element at position zero of the element at position left_pointer of list_of_fruits MINUS start_position PLUS
                   the element at position zero of the element at position right_pointer of list_of_fruits MINUS the element at position zero of the element at position left_pointer of list_of_fruits IS LESS THAN OR EQUAL TO maximum_steps)
                DECREMENT current_total_fruits BY the element at position one of the element at position left_pointer of list_of_fruits
                INCREMENT left_pointer BY one
            END WHILE

            SET maximum_total_fruits TO the MAXIMUM OF maximum_total_fruits AND current_total_fruits
        END FOR

        RETURN maximum_total_fruits
    END FUNCTION
END CLASS
