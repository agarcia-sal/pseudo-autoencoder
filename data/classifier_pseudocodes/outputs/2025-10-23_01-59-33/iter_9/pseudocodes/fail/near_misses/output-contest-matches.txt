CLASS Solution  
    FUNCTION findContestMatch(n)  
        DECLARE participantList AS an empty list  
        FUNCTION initializeTeams(current)  
            IF current GREATER THAN n THEN  
                RETURN  
            ELSE  
                APPEND string representation of current TO participantList  
                CALL initializeTeams(current + 1)  
            END IF  
        END FUNCTION  
        CALL initializeTeams(1)  
        
        FUNCTION pairUp(count)  
            IF count EQUALS 1 THEN  
                RETURN  
            ELSE  
                FOR index FROM 0 TO (count DIV 2) - 1  
                    DECLARE leftTeam AS participantList[index]  
                    DECLARE rightTeam AS participantList[count - 1 - index]  
                    SET participantList[index] TO "(" + leftTeam + "," + rightTeam + ")"  
                END FOR  
                CALL pairUp(count DIV 2)  
            END IF  
        END FUNCTION  
        
        CALL pairUp(n)  
        RETURN participantList[0]  
    END FUNCTION  
END CLASS
