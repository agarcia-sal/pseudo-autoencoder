CLASS Solution  
    FUNCTION createSortedArray(instructions)  
        SET MOD to one billion PLUS seven  
        SET MAX to the maximum element in instructions PLUS one  
        SET bit TO a list of zeroes with the length MAX  

        FUNCTION update(idx, val)  
            WHILE idx LESS THAN MAX holds  
                INCREMENT the element at position idx of bit BY val  
                SET idx TO idx PLUS the result of bitwise AND of idx and the negative of idx  
            END WHILE  
        END FUNCTION  

        FUNCTION query(idx)  
            SET total TO zero  
            WHILE idx GREATER THAN zero holds  
                INCREMENT total BY the element at position idx of bit  
                SET idx TO idx MINUS the result of bitwise AND of idx and the negative of idx  
            END WHILE  
            RETURN total  
        END FUNCTION  

        SET total_cost TO zero  
        FOR each pair of i and num IN the enumeration of instructions  
            SET less_than TO query of num MINUS one  
            SET greater_than TO i MINUS query of num  
            IF less_than LESS THAN greater_than holds  
                SET cost TO less_than  
            ELSE  
                SET cost TO greater_than  
            END IF  
            SET total_cost TO the result of total_cost PLUS cost MODULO MOD  
            CALL update with arguments num and one  
        END FOR  

        RETURN total_cost  
    END FUNCTION  
END CLASS
