CLASS Solution  
    FUNCTION canTransform(start parameter, end parameter)  
        INITIALIZE filteredStart AS an empty collection  
        INITIALIZE counter AS 0  
        WHILE counter IS LESS THAN LENGTH OF start parameter  
            LET current_char BE start parameter AT POSITION counter  
            IF current_char IS DIFFERENT FROM 'X' THEN  
                INSERT (current_char, counter) INTO filteredStart  
            END IF  
            INCREMENT counter BY 1  
        END WHILE  
        
        INITIALIZE filteredEnd AS an empty collection  
        LET iterator BE 0  
        WHILE iterator IS STRICTLY LESS THAN LENGTH OF end parameter  
            ASSIGN char_at_pos TO end parameter ELEMENT AT iterator  
            IF char_at_pos IS NOT 'X' THEN  
                ADD (char_at_pos, iterator) TO filteredEnd  
            END IF  
            INCREASE iterator BY 1  
        END WHILE  
        
        IF LENGTH OF filteredStart IS NOT EQUAL TO LENGTH OF filteredEnd THEN  
            YIELD false  
        END IF  
        
        DECLARE position AS 0  
        WHILE position IS LESS THAN LENGTH OF filteredStart  
            DEFINE (symbolA, locA) AS filteredStart AT position  
            DEFINE (symbolB, locB) AS filteredEnd AT position  
            
            IF symbolA IS NOT EQUAL TO symbolB THEN  
                RETURN false  
            END IF  
            
            IF (symbolA EQUALS 'L') AND NOT (locA IS GREATER THAN OR EQUAL TO locB) THEN  
                RETURN false  
            END IF  
            
            IF (symbolA EQUALS 'R') AND (locA IS LESS THAN locB) THEN  
                RETURN false  
            END IF  
            
            position ‚Üê position + 1  
        END WHILE  
        
        OUTPUT true  
    END FUNCTION  
END CLASS
