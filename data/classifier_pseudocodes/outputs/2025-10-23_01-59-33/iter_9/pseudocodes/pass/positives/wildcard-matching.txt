CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two dimensional list of False with the number of rows equal to the length of s PLUS one and the number of columns equal to the length of p PLUS one  
        SET element at position zero zero of dp TO True  
        FOR j FROM one TO the length of p  
            IF element at position j MINUS one of p EQUALS the character representing asterisk  
                SET element at position zero j of dp TO the value of element at position zero j MINUS one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF element at position j MINUS one of p EQUALS the character representing asterisk  
                    SET element at position i j of dp TO the value of element at position i MINUS one j of dp OR the value of element at position i j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS the character representing question mark OR element at position i MINUS one of s EQUALS element at position j MINUS one of p  
                    SET element at position i j of dp TO the value of element at position i MINUS one j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s the length of p of dp  
    END FUNCTION  
END CLASS
