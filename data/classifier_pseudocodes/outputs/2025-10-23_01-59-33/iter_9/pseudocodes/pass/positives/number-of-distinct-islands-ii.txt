CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO LENGTH OF element at position zero of grid OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND tuple of x and y TO island  
            CALL dfs WITH x PLUS one AND y AND island  
            CALL dfs WITH x MINUS one AND y AND island  
            CALL dfs WITH x AND y PLUS one AND island  
            CALL dfs WITH x AND y MINUS one AND island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx and dy IN list of pairs of one and one PLUS one and minus one PLUS minus one and one PLUS minus one and minus one  
                SET new_island TO empty list  
                FOR each pair x and y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island in ascending order  
                SET min_x TO the minimum x among all pairs in new_island  
                SET min_y TO the minimum y among all pairs in new_island  
                APPEND tuple of pairs where each x MINUS min_x AND each y MINUS min_y from new_island TO shapes  
            END FOR  
            SET original_length TO LENGTH OF shapes  
            FOR index FROM zero TO original_length MINUS one  
                APPEND tuple of pairs where y AND negative x from shape at position index TO shapes  
            END FOR  
            RETURN the minimal element in shapes according to lexicographical order  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO LENGTH OF grid MINUS one  
            FOR j FROM zero TO LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH i AND j AND island  
                    ADD normalize of island TO islands  
                END IF  
            END FOR  
        END FOR  

        RETURN LENGTH OF islands  
    END FUNCTION  
END CLASS
