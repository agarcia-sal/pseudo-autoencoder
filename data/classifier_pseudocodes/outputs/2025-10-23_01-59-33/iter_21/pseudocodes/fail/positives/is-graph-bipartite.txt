CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a list of zeros with LENGTH n  
        
        FUNCTION dfs(node c)  
            SET the element at position node of color TO c  
            FOR each neighbor IN the element at position node of graph  
                IF the element at position neighbor of color EQUALS c  
                    RETURN False  
                END IF  
                IF the element at position neighbor of color EQUALS zero  
                    IF dfs(neighbor the NEGATION OF c) EQUALS False  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of color EQUALS zero  
                IF dfs(i one) EQUALS False  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
