CLASS Solution  
    FUNCTION isRationalEqual WITH PARAMETERS s AS string AND t AS string RETURNS boolean  
        SET ratios TO list containing one followed by one DIVIDED BY nine followed by one DIVIDED BY ninety nine followed by one DIVIDED BY nine hundred ninety nine followed by one DIVIDED BY nine thousand nine hundred ninety nine  
        
        FUNCTION valueOf WITH PARAMETER s AS string RETURNS floating point number  
            IF the search for character left parenthesis IN s DOES NOT FIND any position  
                RETURN the conversion of s TO floating point number  
            END IF  
            
            SET leftParenIndex TO the position of character left parenthesis IN s  
            SET rightParenIndex TO the position of character right parenthesis IN s  
            SET dotIndex TO the position of character dot IN s  
            
            SET integerAndNonRepeating TO the conversion TO floating point number OF the substring of s FROM its beginning UP TO position leftParenIndex  
            SET nonRepeatingLength TO the value of leftParenIndex MINUS dotIndex MINUS one  
            
            SET repeating TO the conversion TO integer OF the substring of s FROM the position FOLLOWING leftParenIndex UP TO position rightParenIndex  
            SET repeatingLength TO the value of rightParenIndex MINUS leftParenIndex MINUS one  
            
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY the number that is zero point one RAISED TO THE POWER OF nonRepeatingLength MULTIPLIED BY the element at position repeatingLength OF ratios  
        END FUNCTION  
        
        RETURN the absolute value OF valueOf applied TO s MINUS valueOf applied TO t IS LESS THAN one times ten TO THE POWER OF negative nine  
    END FUNCTION  
END CLASS
