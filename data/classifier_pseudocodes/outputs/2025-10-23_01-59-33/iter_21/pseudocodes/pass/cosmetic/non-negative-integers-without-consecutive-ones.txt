CLASS Solution  
    FUNCTION findIntegers(n)  
        ASSIGN binaryString ← ""  
        ASSIGN tempNumber ← n  
        WHILE tempNumber > 0  
            SET binaryString ← ( (tempNumber MOD 2) AS STRING ) + binaryString  
            SET tempNumber ← tempNumber DIV 2  
        END WHILE

        SET totalLength ← LENGTH OF binaryString  
        CREATE dp ARRAY WITH SIZE totalLength + 1 FILLED WITH 0  
        SET dp[0] ← 1  
        SET dp[1] ← 2  

        SET counter ← 2  
        WHILE counter ≤ totalLength  
            SET dp[counter] ← dp[counter - 1] + dp[counter - 2]  
            INCREMENT counter BY 1  
        END WHILE

        SET accumulatedResult ← 0  
        SET lastBit ← 0  
        SET position ← 0  

        WHILE position < totalLength  
            SWITCH binaryString[position]  
                CASE '1':  
                    SET accumulatedResult ← accumulatedResult + dp[totalLength - position - 1]  
                    IF lastBit = 1 THEN  
                        SET accumulatedResult ← accumulatedResult - 1  
                        BREAK OUT OF LOOP  
                    END IF  
                    SET lastBit ← 1  
                DEFAULT:  
                    SET lastBit ← 0  
            END SWITCH  
            SET position ← position + 1  
        END WHILE

        RETURN accumulatedResult + 1  
    END FUNCTION  
END CLASS
