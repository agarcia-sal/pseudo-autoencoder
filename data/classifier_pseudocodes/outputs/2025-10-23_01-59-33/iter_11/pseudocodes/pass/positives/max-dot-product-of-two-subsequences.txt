CLASS Solution  
    FUNCTION maxDotProduct(nums1 PARAMETER, nums2 PARAMETER)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a new list of lists where each inner list has n PLUS one elements each set TO negative infinity and the outer list has m PLUS one such inner lists  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO the element at position i MINUS one of nums1 MULTIPLIED BY the element at position j MINUS one of nums2  
                SET dp element at position i element at position j TO the maximum of the following four values  
                    current_product  
                    dp element at position i MINUS one element at position j  
                    dp element at position i element at position j MINUS one  
                    dp element at position i MINUS one element at position j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp element at position m element at position n  
    END FUNCTION  
END CLASS
