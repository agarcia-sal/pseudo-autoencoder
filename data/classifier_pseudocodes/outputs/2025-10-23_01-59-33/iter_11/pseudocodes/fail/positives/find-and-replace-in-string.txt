CLASS Solution  
    FUNCTION findReplaceString(s parameter, indices parameter, sources parameter, targets parameter)  
        SET replacements TO list of tuples made of element at same position of indices parameter AND sources parameter AND targets parameter sorted by the first element of each tuple  
        SET result TO empty list  
        SET prev_end TO zero  
        FOR each tuple composed of index variable AND source variable AND target variable IN replacements  
            APPEND substring of s parameter from position prev_end TO position index MINUS one to result  
            IF substring of s parameter from position index TO position index PLUS the LENGTH OF source variable MINUS one EQUALS source variable  
                APPEND target variable TO result  
            ELSE  
                APPEND substring of s parameter from position index TO position index PLUS the LENGTH OF source variable MINUS one TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source variable  
        END FOR  
        APPEND substring of s parameter from position prev_end TO the end TO result  
        RETURN concatenation of all elements IN result  
    END FUNCTION  
END CLASS
