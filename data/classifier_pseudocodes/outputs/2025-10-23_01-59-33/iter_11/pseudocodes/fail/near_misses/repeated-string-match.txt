CLASS Solution  
    FUNCTION repeatedStringMatch(a string, b string) RETURNS integer  
        FUNCTION containsSubstring(mainStr string, subStr string) RETURNS boolean  
            SET pos TO 0  
            SET lenMain TO LENGTH OF mainStr  
            SET lenSub TO LENGTH OF subStr  

            FUNCTION checkAt(index integer) RETURNS boolean  
                IF index + lenSub > lenMain THEN  
                    RETURN false  
                ELSE IF lenSub = 0 THEN  
                    RETURN true  
                ELSE IF mainStr[index] = subStr[0] THEN  
                    RETURN verifyMatch(1, index + 1)  
                ELSE  
                    RETURN checkAt(index + 1)  
                END IF  
            END FUNCTION  

            FUNCTION verifyMatch(offset integer, startIndex integer) RETURNS boolean  
                IF offset = lenSub THEN  
                    RETURN true  
                ELSE IF mainStr[startIndex] ≠ subStr[offset] THEN  
                    RETURN false  
                ELSE  
                    RETURN verifyMatch(offset + 1, startIndex + 1)  
                END IF  
            END FUNCTION  

            RETURN checkAt(0)  
        END FUNCTION  

        SET x TO LENGTH OF b  
        SET y TO LENGTH OF a  
        SET z TO x DIVIDED BY y  

        IF x MOD y ≠ 0 THEN  
            SET z TO z + 1  
        END IF  

        SET repeatedStr TO empty string  
        SET counter_one TO 0  
        WHILE counter_one < z DO  
            SET repeatedStr TO repeatedStr + a  
            SET counter_one TO counter_one + 1  
        END WHILE  

        IF containsSubstring(repeatedStr, b) THEN  
            RETURN z  
        ELSE  
            SET repeatedStr TO repeatedStr + a  
            IF containsSubstring(repeatedStr, b) THEN  
                RETURN z + 1  
            ELSE  
                RETURN -1  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
