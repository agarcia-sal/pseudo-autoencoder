CLASS Solution
    FUNCTION appealSum(s)
        ASSIGN last TO a dictionary with keys from 0 to 25 each paired with value -1
        ASSIGN agg TO 0
        ASSIGN acc TO 0

        ASSIGN keys TO list of integers from 0 up to (length of s minus 1)

        FUNCTION recurse(pos_list)
            IF pos_list IS EMPTY THEN
                RETURN
            ELSE
                ASSIGN curr_pos TO pos_list[0]
                ASSIGN rem TO sublist of pos_list from index 1 onward

                ASSIGN ch TO s[curr_pos]
                ASSIGN idx TO numeric code of ch MINUS numeric code of 'a'

                ASSIGN diff TO curr_pos MINUS last[idx]
                ASSIGN acc TO acc PLUS diff
                ASSIGN agg TO agg PLUS acc
                ASSIGN last[idx] TO curr_pos

                recurse(rem)
            END IF
        END FUNCTION

        CALL recurse(keys)

        RETURN agg
    END FUNCTION
END CLASS
