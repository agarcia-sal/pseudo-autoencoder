CLASS Solution
    FUNCTION checkPossibility(list_of_numbers)
        SET modified_flag TO FALSE
        FOR index FROM the natural number one TO the LENGTH OF list_of_numbers MINUS the natural number one INCLUSIVE
            IF the element at position index OF list_of_numbers IS LESS THAN the element at position index MINUS the natural number one OF list_of_numbers
                IF modified_flag IS TRUE
                    RETURN FALSE
                END IF
                SET modified_flag TO TRUE
                IF index IS LESS THAN the natural number two OR the element at position index OF list_of_numbers IS GREATER THAN OR EQUAL TO the element at position index MINUS the natural number two OF list_of_numbers
                    SET the element at position index MINUS the natural number one OF list_of_numbers TO the element at position index OF list_of_numbers
                ELSE
                    SET the element at position index OF list_of_numbers TO the element at position index MINUS the natural number one OF list_of_numbers
                END IF
            END IF
        END FOR
        RETURN TRUE
    END FUNCTION
END CLASS
