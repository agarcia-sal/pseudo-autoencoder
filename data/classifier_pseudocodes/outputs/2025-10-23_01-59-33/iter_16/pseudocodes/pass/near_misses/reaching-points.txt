CLASS Solution  
    FUNCTION reachingPoints(sx sx PARAMETER sy sy PARAMETER tx tx PARAMETER ty ty) RETURNS boolean  
        REPEAT WHILE ( (sx LESS THAN OR EQUAL TO tx) AND (sy LESS THAN OR EQUAL TO ty) )  
            IF (sx EQUALS tx) THEN  
                IF (sy EQUALS ty) THEN  
                    RETURN True  
                END IF  
            ELSE  
                IF (tx GREATER THAN ty) THEN  
                    IF (sy LESS THAN ty) THEN  
                        SET remainderValue TO (tx MOD ty)  
                        SET tx TO remainderValue  
                    ELSE  
                        SET difference TO (tx MINUS sx)  
                        SET modResult TO (difference MOD ty)  
                        RETURN (modResult EQUALS 0)  
                    END IF  
                ELSE  
                    IF (sx LESS THAN tx) THEN  
                        SET modResult TO (ty MOD tx)  
                        SET ty TO modResult  
                    ELSE  
                        SET difference TO (ty MINUS sy)  
                        SET modResult TO (difference MOD tx)  
                        RETURN (modResult EQUALS 0)  
                    END IF  
                END IF  
            END IF  
        END REPEAT  
        RETURN False  
    END FUNCTION  
END CLASS
