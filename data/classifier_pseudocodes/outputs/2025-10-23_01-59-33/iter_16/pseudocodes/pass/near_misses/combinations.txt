CLASS Solution
    FUNCTION combine WITH PARAMETERS n k
        SET outcome TO []
        FUNCTION backtrack WITH PARAMETERS pos seq
            WHILE pos NOT EQUAL TO n + 1
                APPEND pos TO seq
                IF LENGTH seq EQUALS k THEN
                    APPEND COPY OF seq TO outcome
                ELSE
                    CALL backtrack WITH ARGUMENTS pos + 1 seq
                END IF
                REMOVE LAST ITEM FROM seq
                SET pos TO pos + 1
            END WHILE
        END FUNCTION
        CALL backtrack WITH ARGUMENTS 1 []
        RETURN outcome
    END FUNCTION
END CLASS
