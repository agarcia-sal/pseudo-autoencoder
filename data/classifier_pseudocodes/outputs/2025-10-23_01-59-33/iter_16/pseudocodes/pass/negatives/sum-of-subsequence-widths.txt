CLASS Solution
    FUNCTION sumSubseqWidths(list_of_numbers)
        SET MODULO TO ten RAISED TO the power of nine PLUS one
        EXECUTE sorting ON list_of_numbers IN ascending order
        SET count_of_numbers TO the LENGTH OF list_of_numbers
        SET total_sum TO zero
        SET power_list TO a LIST containing one REPEATED count_of_numbers TIMES

        FOR index FROM two TO count_of_numbers INCLUSIVE
            SET element_at_position index OF power_list TO the element at position index MINUS one OF power_list MULTIPLIED BY two THEN MODULO MODULO
        END FOR

        FOR index FROM one TO count_of_numbers INCLUSIVE
            SET total_sum TO total_sum PLUS the element at position index OF list_of_numbers MULTIPLIED BY the element at position index OF power_list MINUS the element at position count_of_numbers MINUS index MINUS one OF power_list THEN MODULO MODULO
        END FOR

        RETURN total_sum
    END FUNCTION
END CLASS
