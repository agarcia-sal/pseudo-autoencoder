CLASS Solution
    FUNCTION isSelfCrossing(distance)
        ASSIGN lengthValue ← SIZE(distance)
        IF NOT (lengthValue ≥ 4) THEN
            RETURN False
        END IF

        SET index ← 3
        WHILE index ≤ lengthValue - 1 DO
            ASSIGN current ← distance[index]
            ASSIGN prev2 ← distance[index - 2]
            ASSIGN prev1 ← distance[index - 1]
            ASSIGN prev3 ← distance[index - 3]

            IF (current ≥ prev2) AND (prev1 ≤ prev3) THEN
                RETURN True
            END IF

            IF (index ≥ 4) THEN
                ASSIGN prev3Check ← distance[index - 3]
                ASSIGN prev4Check ← distance[index - 4]
                IF (prev1 = prev3Check) AND ((current + prev4Check) ≥ prev2) THEN
                    RETURN True
                END IF
            END IF

            IF (index ≥ 5) THEN
                ASSIGN prev2Check ← distance[index - 2]
                ASSIGN prev4Check ← distance[index - 4]
                ASSIGN prev3Check ← distance[index - 3]
                ASSIGN prev1Check ← distance[index - 1]
                ASSIGN prev5Check ← distance[index - 5]

                IF (prev2Check ≥ prev4Check) AND (prev3Check ≥ prev1Check) AND ((prev1Check + prev5Check) ≥ prev3Check) AND ((current + prev4Check) ≥ prev2Check) THEN
                    RETURN True
                END IF
            END IF

            INCREMENT index BY 1
        END WHILE

        RETURN False
    END FUNCTION
END CLASS
