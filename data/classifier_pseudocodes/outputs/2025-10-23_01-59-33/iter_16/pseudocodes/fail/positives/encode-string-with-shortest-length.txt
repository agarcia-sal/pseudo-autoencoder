CLASS Solution  
    FUNCTION encode(s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN the LENGTH OF the string CONVERTED FROM count PLUS the LENGTH OF substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO the element at position i UP TO and INCLUDING position j OF s  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five THEN  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR k FROM one TO n DIVIDED BY two PLUS one  
                IF n MODULO k EQUALS zero THEN  
                    SET repeat_count TO n DIVIDED BY k  
                    IF substring EQUALS substring from position zero UP TO but NOT INCLUDING position k REPEATED repeat_count TIMES THEN  
                        SET encoded_candidate TO string CONVERTED FROM repeat_count CONCATENATED WITH the LEFT SQUARE BRACKET CONCATENATED WITH dp(i, i PLUS k MINUS one) CONCATENATED WITH the RIGHT SQUARE BRACKET  
                        IF the LENGTH OF encoded_candidate LESS THAN the LENGTH OF shortest THEN  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR k FROM i TO j MINUS one  
                SET left_encoded TO dp(i, k)  
                SET right_encoded TO dp(k PLUS one, j)  
                SET combined TO left_encoded CONCATENATED WITH right_encoded  
                IF the LENGTH OF combined LESS THAN the LENGTH OF shortest THEN  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN dp(zero, the LENGTH OF s MINUS one)  
    END FUNCTION  
END CLASS
