CLASS Solution
    FUNCTION pourWater(heights WITH TYPE List of integers volume WITH TYPE integer k WITH TYPE integer) RETURNS List of integers
        ASSIGN n TO LENGTH(heights)

        SET stepCounter TO 1
        WHILE stepCounter LESS THAN OR EQUAL TO volume
            ASSIGN cursor TO k

            WHILE NOT (cursor EQUALS 0 OR heights[cursor - 1] GREATER THAN heights[cursor])
                SUBTRACT 1 FROM cursor
            END WHILE

            WHILE NOT (cursor EQUALS k OR heights[cursor] NOT EQUAL heights[cursor + 1])
                ADD 1 TO cursor
            END WHILE

            IF NOT cursor GREATER OR EQUAL k THEN
                ASSIGN prevHeight TO heights[cursor]
                ASSIGN updatedHeight TO prevHeight + 1
                ASSIGN heights[cursor] TO updatedHeight
                ADD 1 TO stepCounter
                CONTINUE TO NEXT iteration of WHILE
            END IF

            ASSIGN cursor TO k

            WHILE NOT (cursor EQUALS n - 1 OR heights[cursor + 1] GREATER THAN heights[cursor])
                ADD 1 TO cursor
            END WHILE

            WHILE NOT (cursor EQUALS k OR heights[cursor] NOT EQUAL heights[cursor - 1])
                SUBTRACT 1 FROM cursor
            END WHILE

            IF cursor GREATER THAN k THEN
                ASSIGN baseHeight TO heights[cursor]
                ASSIGN incrementedHeight TO baseHeight + 1
                ASSIGN heights[cursor] TO incrementedHeight
                ADD 1 TO stepCounter
                CONTINUE TO NEXT iteration of WHILE
            END IF

            ASSIGN currentHeight TO heights[k]
            ASSIGN newHeight TO currentHeight + 1
            ASSIGN heights[k] TO newHeight

            ADD 1 TO stepCounter
        END WHILE

        RETURN heights
    END FUNCTION
END CLASS
