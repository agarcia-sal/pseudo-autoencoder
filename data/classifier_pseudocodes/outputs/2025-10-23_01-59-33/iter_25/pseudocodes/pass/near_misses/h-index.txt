CLASS Solution
    FUNCTION hIndex(citations)
        ASSIGN orderedList TO a new empty queue
        FOR each item IN citations
            ENQUEUE item INTO orderedList
        END FOR
        SORT orderedList DESCENDING
        DEFINE FUNCTION checkAt(position)
            IF position EQUALS LENGTH OF orderedList THEN
                RETURN LENGTH OF orderedList
            ELSE
                ASSIGN currentValue TO DEQUEUE orderedList
                IF currentValue LESS THAN position PLUS 1 THEN
                    RETURN position
                ELSE
                    RETURN checkAt(position PLUS 1)
                END IF
            END IF
        END FUNCTION
        RETURN checkAt(0)
    END FUNCTION
END CLASS
