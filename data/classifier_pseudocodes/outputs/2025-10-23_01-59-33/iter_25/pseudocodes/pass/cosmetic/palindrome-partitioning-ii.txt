CLASS Solution  
    FUNCTION minCut(s)  
        SET length_s AS the LENGTH OF s  
        
        IF NOT (length_s GREATER THAN one)  
            RETURN zero  
        END IF  
        
        SET palindrome_flag TO a list of length_s elements each containing a list of length_s elements initialized to FALSE  
        
        SET idx TO zero  
        WHILE idx LESS THAN length_s  
            SET palindrome_flag[idx][idx] AS TRUE  
            INCREMENT idx BY one  
        END WHILE  
        
        SET substr_len TO two  
        WHILE substr_len LESS THAN OR EQUAL TO length_s  
            SET start_pos TO zero  
            WHILE start_pos LESS THAN OR EQUAL TO length_s MINUS substr_len  
                SET end_pos TO start_pos PLUS substr_len MINUS one  
                
                IF substr_len EQUALS two THEN  
                    SET palindrome_flag[start_pos][end_pos] AS (s[start_pos] EQUALS s[end_pos])  
                ELSE  
                    SET chars_match AS (s[start_pos] EQUALS s[end_pos])  
                    SET inner_palindrome AS palindrome_flag[start_pos PLUS one][end_pos MINUS one]  
                    SET palindrome_flag[start_pos][end_pos] AS (chars_match AND inner_palindrome)  
                END IF  
                
                INCREMENT start_pos BY one  
            END WHILE  
            INCREMENT substr_len BY one  
        END WHILE  
        
        SET min_cut_counts TO a list of length_s elements initialized as zero  
        
        SET i TO zero  
        WHILE i LESS THAN length_s  
            IF palindrome_flag[zero][i] THEN  
                SET min_cut_counts[i] AS zero  
            ELSE  
                SET min_cut_counts[i] AS positive infinity  
                SET j TO one  
                WHILE j LESS THAN OR EQUAL TO i  
                    IF palindrome_flag[j][i] AND ((min_cut_counts[j MINUS one] PLUS one) LESS THAN min_cut_counts[i]) THEN  
                        SET min_cut_counts[i] AS (min_cut_counts[j MINUS one] PLUS one)  
                    END IF  
                    INCREMENT j BY one  
                END WHILE  
            END IF  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN min_cut_counts[length_s MINUS one]  
    END FUNCTION  
END CLASS
