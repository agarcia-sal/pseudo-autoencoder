CLASS Solution  
    FUNCTION isRationalEqual(s)  
        SET ratios TO list containing one and one DIVIDED BY nine and one DIVIDED BY ninety nine and one DIVIDED BY nine hundred ninety nine and one DIVIDED BY nine thousand nine hundred ninety nine  
        
        FUNCTION valueOf(s)  
            IF the position of the open parenthesis character in s EQUALS negative one  
                RETURN the conversion of s to a floating point number  
            END IF  
            
            SET leftParenIndex TO the position of the open parenthesis character in s  
            SET rightParenIndex TO the position of the close parenthesis character in s  
            SET dotIndex TO the position of the dot character in s  
            
            SET integerAndNonRepeating TO the conversion to floating point number of the substring of s from the start up to the position leftParenIndex MINUS one  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  
            
            SET repeating TO the integer conversion of the substring of s from the position leftParenIndex PLUS one TO the position rightParenIndex MINUS one  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  
            
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY ten TO THE POWER OF negative nonRepeatingLength MULTIPLIED BY the element at position repeatingLength of ratios  
        END FUNCTION  
        
        RETURN the absolute value of the difference between valueOf applied to s and valueOf applied to t LESS THAN one TIMES ten TO THE POWER OF negative nine  
    END FUNCTION  
END CLASS
