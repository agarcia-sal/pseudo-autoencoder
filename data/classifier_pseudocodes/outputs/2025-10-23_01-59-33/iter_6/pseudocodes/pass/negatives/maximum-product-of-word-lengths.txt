CLASS Solution
    FUNCTION maxProduct(list_of_words)
        SET list_of_character_sets TO CALL ConvertWordsToCharacterSets WITH list_of_words
        SET maximum_product TO zero
        FOR index_i FROM zero TO the LENGTH OF list_of_words MINUS one
            FOR index_j FROM index_i PLUS one TO the LENGTH OF list_of_words MINUS one
                IF the INTERSECTION of element at position index_i of list_of_character_sets AND element at position index_j of list_of_character_sets IS EQUALS to the EMPTY SET
                    SET product TO the LENGTH OF element at position index_i of list_of_words MULTIPLIED BY the LENGTH OF element at position index_j of list_of_words
                    SET maximum_product TO the GREATER OF maximum_product AND product
                END IF
            END FOR
        END FOR
        RETURN maximum_product
    END FUNCTION
END CLASS
