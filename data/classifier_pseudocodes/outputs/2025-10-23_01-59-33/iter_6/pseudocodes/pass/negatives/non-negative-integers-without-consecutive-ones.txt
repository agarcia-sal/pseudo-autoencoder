CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET binary_representation TO the binary representation of n starting from the element after the first character  
        SET length TO the LENGTH OF binary_representation  
        INITIALIZE dp AS a list of zeros with the size EQUAL TO length PLUS one  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        FOR i FROM two TO length  
            SET the element at position i of dp TO the element at position i MINUS one of dp PLUS the element at position i MINUS two of dp  
        END FOR  
        SET result TO zero  
        SET previous_bit TO zero  
        FOR i FROM zero TO length MINUS one  
            IF the element at position i of binary_representation EQUALS one  
                SET result TO result PLUS the element at position length MINUS i MINUS one of dp  
                IF previous_bit EQUALS one  
                    SET result TO result MINUS one  
                    BREAK the loop  
                END IF  
            END IF  
            IF the element at position i of binary_representation EQUALS one  
                SET previous_bit TO one  
            ELSE  
                SET previous_bit TO zero  
            END IF  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
