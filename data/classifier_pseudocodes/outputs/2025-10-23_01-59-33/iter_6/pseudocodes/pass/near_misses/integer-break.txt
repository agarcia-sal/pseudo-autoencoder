CLASS Solution  
    FUNCTION integerBreak(n)  
        DECLARE cache AS list with length n PLUS one, each element initialized to 0  
        ASSIGN cache[1] TO 1  

        FUNCTION recursiveIndex(k)  
            IF k GREATER THAN n  
                RETURN  
            END IF
            DECLARE current_max AS 0  
            DECLARE divisor AS 1  

            FUNCTION innerRecursive()  
                IF divisor GREATER OR EQUAL TO k  
                    RETURN  
                END IF  
                DECLARE product_a AS divisor MULTIPLIED BY cache[k MINUS divisor]  
                DECLARE product_b AS divisor MULTIPLIED BY (k MINUS divisor)  

                IF current_max LESS THAN product_a AND current_max LESS THAN product_b  
                    IF product_a GREATER THAN product_b  
                        ASSIGN current_max TO product_a  
                    ELSE  
                        ASSIGN current_max TO product_b  
                    END IF  
                ELSE IF current_max LESS THAN product_a  
                    ASSIGN current_max TO product_a  
                ELSE IF current_max LESS THAN product_b  
                    ASSIGN current_max TO product_b  
                END IF

                ASSIGN divisor TO divisor PLUS 1  
                CALL innerRecursive()  
            END FUNCTION  

            CALL innerRecursive()  
            ASSIGN cache[k] TO current_max  
            CALL recursiveIndex(k PLUS 1)  
        END FUNCTION  

        CALL recursiveIndex(2)  
        RETURN cache[n]  
    END FUNCTION  
END CLASS
