CLASS Solution  
    FUNCTION waysToPartition(nums parameter LIST OF integers, k parameter integer) RETURN integer  
        SET total_sum TO the SUM of all elements in nums  
        SET n TO the LENGTH OF nums  
        
        CREATE left_diff AS a new dictionary with integer default values  
        CREATE right_diff AS a new dictionary with integer default values  
        
        SET current_sum TO zero  
        FOR index FROM zero TO value of n MINUS one  
            INCREMENT current_sum BY element at position index of nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT the value corresponding to key diff in right_diff BY one  
        END FOR  
        
        SET max_ways TO the value corresponding to key zero in right_diff  
        
        SET current_sum TO zero  
        FOR index FROM zero TO value of n MINUS one  
            INCREMENT current_sum BY element at position index of nums  
            
            SET diff_change TO k MINUS element at position index of nums  
            
            SET new_ways TO the value corresponding to key the negative of diff_change in left_diff PLUS the value corresponding to key diff_change in right_diff  
            IF new_ways GREATER THAN max_ways THEN  
                SET max_ways TO new_ways  
            END IF  
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT the value corresponding to key diff in right_diff BY one  
            INCREMENT the value corresponding to key diff in left_diff BY one  
        END FOR  
        
        RETURN max_ways  
    END FUNCTION  
END CLASS
