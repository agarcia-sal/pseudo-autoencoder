CLASS Solution
    FUNCTION maxDotProduct(nums1 PARAMETER, nums2 PARAMETER)
        SET a TO LENGTH OF nums1
        SET b TO LENGTH OF nums2
        INITIALIZE matrix AS EMPTY LIST
        SET p TO 0
        WHILE p <= a
            INITIALIZE row AS EMPTY LIST
            SET q TO 0
            WHILE q <= b
                APPEND negative infinity TO row
                SET q TO q + 1
            END WHILE
            APPEND row TO matrix
            SET p TO p + 1
        END WHILE

        FUNCTION fill(i PARAMETER, j PARAMETER)
            IF i = 0 OR j = 0 THEN
                RETURN
            END IF
            CALL fill(i - 1, j)
            CALL fill(i, j - 1)
            CALL fill(i - 1, j - 1)

            SET x TO nums1 element at i - 1
            SET y TO nums2 element at j - 1
            SET mul TO x * y

            SET val1 TO mul
            SET val2 TO matrix element at i - 1 element at j
            SET val3 TO matrix element at i element at j - 1
            SET val4 TO matrix element at i - 1 element at j - 1 + mul

            IF val2 >= val3 THEN
                IF val4 >= val2 THEN
                    IF val4 >= val1 THEN
                        matrix element at i element at j = val4
                    ELSE
                        matrix element at i element at j = val1
                    END IF
                ELSE
                    IF val2 >= val1 THEN
                        matrix element at i element at j = val2
                    ELSE
                        matrix element at i element at j = val1
                    END IF
                END IF
            ELSE
                IF val3 >= val4 THEN
                    IF val3 >= val1 THEN
                        matrix element at i element at j = val3
                    ELSE
                        matrix element at i element at j = val1
                    END IF
                ELSE
                    IF val4 >= val1 THEN
                        matrix element at i element at j = val4
                    ELSE
                        matrix element at i element at j = val1
                    END IF
                END IF
            END IF
        END FUNCTION

        CALL fill(a, b)
        RETURN matrix element at a element at b
    END FUNCTION
END CLASS
