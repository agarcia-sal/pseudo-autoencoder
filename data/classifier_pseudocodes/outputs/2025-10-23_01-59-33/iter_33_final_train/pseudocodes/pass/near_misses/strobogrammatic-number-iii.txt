CLASS Solution
    FUNCTION strobogrammaticInRange(xqpej  
                                    ksewr)
        FUNCTION generate_strobogrammatic(ryvnm  
                                         xzukr)
            IF ryvnm NOT EQUALS 0 THEN
                IF ryvnm NOT EQUALS 1 THEN
                    SET nvdhu TO generate_strobogrammatic(ryvnm - 2, xzukr)
                    SET gblwi TO []
                    SET ftoxq TO 0
                    WHILE ftoxq LESS THAN LENGTH(nvdhu) DO
                        SET qktpv TO nvdhu[ftoxq]
                        SET sfuxe TO ryvnm EQUALS xzukr
                        IF NOT sfuxe THEN
                            gblwi APPEND ("0" + qktpv + "0")
                        END IF
                        gblwi APPEND ("1" + qktpv + "1")
                        gblwi APPEND ("6" + qktpv + "9")
                        gblwi APPEND ("8" + qktpv + "8")
                        gblwi APPEND ("9" + qktpv + "6")
                        SET ftoxq TO ftoxq + 1
                    END WHILE
                    RETURN gblwi
                ELSE
                    RETURN ["0", "1", "8"]
                END IF
            ELSE
                RETURN [""]
            END IF
        END FUNCTION

        FUNCTION count_strobogrammatic(cmask  
                                       wljbo)
            SET snxra TO 0
            SET lftbp TO LENGTH(cmask)
            SET hgqtr TO LENGTH(wljbo)
            FUNCTION loop_length(ldnwv)
                IF ldnwv GREATER THAN hgqtr THEN
                    RETURN snxra
                ELSE
                    SET tksvr TO generate_strobogrammatic(ldnwv, ldnwv)
                    FUNCTION loop_nums(idx)
                        IF idx GREATER OR EQUAL TO LENGTH(tksvr) THEN
                            RETURN
                        ELSE
                            SET rczoy TO tksvr[idx]
                            IF (ldnwv EQUALS lftbp AND rczoy LESS THAN cmask) OR (ldnwv EQUALS hgqtr AND rczoy GREATER THAN wljbo) THEN
                                loop_nums(idx + 1)
                            ELSE
                                SET snxra TO snxra + 1
                                loop_nums(idx + 1)
                            END IF
                        END IF
                    END FUNCTION
                    loop_nums(0)
                    RETURN loop_length(ldnwv + 1)
                END IF
            END FUNCTION
            RETURN loop_length(lftbp)
        END FUNCTION

        RETURN count_strobogrammatic(xqpej, ksewr)
    END FUNCTION
END CLASS
