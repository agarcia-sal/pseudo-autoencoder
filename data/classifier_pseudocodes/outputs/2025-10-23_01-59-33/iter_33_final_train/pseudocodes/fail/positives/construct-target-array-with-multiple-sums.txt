CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND the NEGATION of num TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
        
        SET total_sum TO the sum of elements in target  
        
        WHILE the element at position zero of max_heap LESS THAN NEGATION of one holds  
            SET largest TO the NEGATION of the element extracted from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero THEN  
                RETURN False  
            END IF  
            
            SET new_value TO the remainder of largest DIVIDED BY rest_sum  
            
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one THEN  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            INSERT the NEGATION of new_value INTO max_heap maintaining heap order  
        END WHILE  
        
        RETURN True  
    END FUNCTION  
END CLASS
