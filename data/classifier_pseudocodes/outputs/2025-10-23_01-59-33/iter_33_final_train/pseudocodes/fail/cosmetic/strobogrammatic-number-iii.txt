CLASS Solution
    FUNCTION strobogrammaticInRange(low  
                                    high)
        FUNCTION generate_strobogrammatic(n  
                                         final_length)
            SWITCH n
                CASE 0:
                    RETURN [""]

                CASE 1:
                    RETURN ["0", "1", "8"]
            END SWITCH

            SET innerStrings AS generate_strobogrammatic(final_length - 2, final_length)
            SET collection AS []

            SET idx TO 0
            WHILE idx < LENGTH(innerStrings)
                SET segment TO innerStrings[idx]

                IF n IS DIFFERENT FROM final_length THEN
                    INSERT "0" + segment + "0" INTO collection AT END
                END IF

                INSERT "1" + segment + "1" INTO collection AT END
                INSERT "6" + segment + "9" INTO collection AT END
                INSERT "8" + segment + "8" INTO collection AT END
                INSERT "9" + segment + "6" INTO collection AT END

                idx ← idx + 1
            END WHILE

            RETURN collection
        END FUNCTION

        FUNCTION count_strobogrammatic(low  
                                       high)
            SET total AS 0
            SET lenLow TO LENGTH(low)
            SET lenHigh TO LENGTH(high)

            SET currentLen TO lenLow
            WHILE currentLen <= lenHigh
                SET generatedList AS generate_strobogrammatic(currentLen, currentLen)
                SET pointer TO 0

                WHILE pointer < LENGTH(generatedList)
                    SET numberStr TO generatedList[pointer]

                    IF NOT ((currentLen = lenLow AND numberStr < low) OR (currentLen = lenHigh AND numberStr > high)) THEN
                        total ← total + 1
                    END IF

                    pointer ← pointer + 1
                END WHILE

                currentLen ← currentLen + 1
            END WHILE

            RETURN total
        END FUNCTION

        RETURN count_strobogrammatic(low, high)
    END FUNCTION
END CLASS
