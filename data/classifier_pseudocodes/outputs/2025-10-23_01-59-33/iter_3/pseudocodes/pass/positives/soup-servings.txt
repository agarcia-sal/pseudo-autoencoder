CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred THEN  
            RETURN one point zero  
        END IF  
        SET n TO n PLUS twenty four DIVIDED BY twenty five  
        SET memo TO empty dictionary  
        FUNCTION dp(a, b)  
            IF tuple of a AND b EXISTS IN memo THEN  
                RETURN value of tuple of a AND b IN memo  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero THEN  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero THEN  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero THEN  
                RETURN zero point zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY sum of dp of a MINUS four AND b PLUS dp of a MINUS three AND b MINUS one PLUS dp of a MINUS two AND b MINUS two PLUS dp of a MINUS one AND b MINUS three  
            SET value of tuple of a AND b IN memo TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp of n AND n  
    END FUNCTION  
END CLASS
