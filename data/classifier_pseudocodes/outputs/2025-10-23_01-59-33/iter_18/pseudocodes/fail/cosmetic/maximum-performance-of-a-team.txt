CLASS Solution
    FUNCTION maxPerformance(n speed efficiency k)
        ASSIGN empty array TO specialist_tuples
        ASSIGN 0 TO idx
        WHILE idx LESS THAN length of efficiency
            ASSIGN (efficiency[idx], speed[idx]) TO pair_val
            APPEND pair_val TO specialist_tuples
            INCREMENT idx BY 1
        END WHILE

        ASSIGN specialist_tuples SORTED BY first element DESCENDING TO sorted_specialists

        ASSIGN empty min-heap TO velocity_container
        ASSIGN 0 TO cumulative_velocity
        ASSIGN 0 TO top_performance

        ASSIGN 0 TO iterate_index
        WHILE iterate_index LESS THAN length of sorted_specialists
            ASSIGN sorted_specialists[iterate_index] TO current_pair
            ASSIGN current_pair.first TO efficiency_val
            ASSIGN current_pair.second TO velocity_val

            PUSH velocity_val INTO velocity_container
            SET cumulative_velocity TO cumulative_velocity PLUS velocity_val

            IF SIZE of velocity_container GREATER THAN k THEN
                ASSIGN HEAP POP from velocity_container TO lowest_velocity
                SET cumulative_velocity TO cumulative_velocity MINUS lowest_velocity
            END IF

            ASSIGN cumulative_velocity TIMES efficiency_val TO performance_val

            IF performance_val GREATER THAN top_performance THEN
                SET top_performance TO performance_val
            END IF

            INCREMENT iterate_index BY 1
        END WHILE

        RETURN top_performance MOD (10 POWER 9 PLUS 1)
    END FUNCTION
END CLASS
