CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a TWO DIMENSIONAL LIST of BOOLEAN TRUES with n ROWS and n COLUMNS  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and position j of f TO FALSE  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i and position j of f TO element at position i PLUS one and position j MINUS one of f  
                    IF element at position i and position j of f EQUALS TRUE AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN substring from position k TO position k PLUS mx MINUS one of s  
    END FUNCTION  
END CLASS
