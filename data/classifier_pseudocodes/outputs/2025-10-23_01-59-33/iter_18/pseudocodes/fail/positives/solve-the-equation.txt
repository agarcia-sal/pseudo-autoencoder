CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing plus signs in side with delimiter plus sign and replacing minus signs in side with delimiter minus sign and splitting the resulting string by the delimiter  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is an empty string THEN  
                    CONTINUE to the next token  
                END IF  
                IF the character x exists in token THEN  
                    IF token is equal to x OR token is equal to plus sign followed by x THEN  
                        INCREMENT x_count BY one  
                    ELSE IF token is equal to minus sign followed by x THEN  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the integer value of token with character x removed  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value of token  
                END IF  
            END FOR  
            RETURN x_count and num_sum  
        END FUNCTION  

        SET left_side and right_side TO the parts of equation separated by the equal sign  
        SET left_x and left_num TO the result of calling parse_side with left_side  
        SET right_x and right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero THEN  
            IF total_num EQUALS zero THEN  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string formed by the character x equals the integer result of total_num DIVIDED BY total_x  
        END IF  
    END FUNCTION  
END CLASS
