CLASS Solution
    FUNCTION canJump(list_of_numbers)
        SET farthest_reachable_index TO zero
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            IF index GREATER THAN farthest_reachable_index
                RETURN False
            END IF
            SET possible_reach TO index PLUS element at position index OF list_of_numbers
            SET farthest_reachable_index TO the GREATER OF farthest_reachable_index AND possible_reach
        END FOR
        RETURN farthest_reachable_index GREATER THAN OR EQUAL TO the LENGTH OF list_of_numbers MINUS one
    END FUNCTION
END CLASS
