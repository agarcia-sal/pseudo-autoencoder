CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY the LENGTH OF element AND THEN BY element IN LEXICOGRAPHICAL ORDER  
        SET valid_words TO an empty set  
        SET longest_word TO an empty string  
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the substring of word from position one TO position LENGTH OF word MINUS one IS IN valid_words  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
