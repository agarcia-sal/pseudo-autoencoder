CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements of sentence with a single space between each followed by a single space character  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each iteration IN the range from one TO rows  
            INCREMENT position BY cols  
            
            IF the element at position MODULO total_length of s EQUALS a space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at position MINUS one MODULO total_length of s NOT EQUALS a space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
