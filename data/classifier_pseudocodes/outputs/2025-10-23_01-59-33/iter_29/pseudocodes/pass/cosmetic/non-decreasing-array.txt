CLASS Solution
    FUNCTION checkPossibility(nums)
        SET alterationFlag AS false
        SET pos AS 1
        WHILE pos < (LENGTH OF nums) DO
            IF NOT (nums[pos] >= nums[pos - 1]) THEN
                IF alterationFlag THEN
                    RETURN false
                END IF
                alterationFlag ← true
                IF (pos < 2) OR (nums[pos] >= nums[pos - 2]) THEN
                    nums[pos - 1] ← nums[pos]
                ELSE
                    nums[pos] ← nums[pos - 1]
                END IF
            END IF
            pos ← pos + 1
        END WHILE
        RETURN true
    END FUNCTION
END CLASS
