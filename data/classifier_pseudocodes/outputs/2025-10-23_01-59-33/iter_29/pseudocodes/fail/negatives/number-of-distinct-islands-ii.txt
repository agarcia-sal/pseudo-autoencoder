CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND pair composed of x and y TO island  
            CALL dfs with x PLUS one and y and island  
            CALL dfs with x MINUS one and y and island  
            CALL dfs with x and y PLUS one and island  
            CALL dfs with x and y MINUS one and island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx and dy IN the list of pairs composed of one and one and one and MINUS one and MINUS one and one and MINUS one and MINUS one  
                SET new_island TO empty list  
                FOR each pair x and y IN island  
                    APPEND pair composed of x MULTIPLIED BY dx and y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island in ascending order  
                SET min_x TO the minimum x among all pairs x and y in new_island  
                SET min_y TO the minimum y among all pairs x and y in new_island  
                APPEND to shapes the tuple of pairs where each pair consists of x MINUS min_x and y MINUS min_y for each x and y in new_island  
            END FOR  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF shapes  
                APPEND to shapes the tuple of pairs where each pair consists of y and NEGATIVE x for each x and y in element at position i of shapes  
                INCREMENT i BY one  
            END WHILE  
            RETURN the minimal element of shapes in lexicographical order  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF the element at position i of grid at position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs with i and j and island  
                    ADD the result of normalize called with island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
