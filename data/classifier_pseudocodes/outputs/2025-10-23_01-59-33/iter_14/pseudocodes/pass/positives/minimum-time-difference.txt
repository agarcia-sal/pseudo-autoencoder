CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET split_result TO the list of elements obtained by splitting time at the position of the colon character  
            SET hours TO the integer conversion of the element at position zero of split_result  
            SET minutes TO the integer conversion of the element at position one of split_result  
            RETURN the result of multiplication of hours by the number of minutes in one hour PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the list obtained by applying the function to_minutes to each element in timePoints  
        SORT minutes_list in ascending order  
        SET min_diff TO positive infinity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO the result of subtracting the element at the last position of minutes_list FROM the number of minutes in a full day PLUS the element at the first position of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
