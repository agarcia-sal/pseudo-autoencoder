CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO element at position one of element at position one of arrays  
        SET max_val TO element at position one of element at position last of arrays  
        SET max_distance TO zero  
        FOR i FROM two TO the length of arrays  
            SET current_array TO element at position i of arrays  
            SET current_min TO element at position one of current_array  
            SET current_max TO element at position last of current_array  
            SET max_distance TO the maximum of three values consisting of max_distance AND the absolute value of current_max MINUS min_val AND the absolute value of max_val MINUS current_min  
            SET min_val TO the minimum of two values consisting of min_val AND current_min  
            SET max_val TO the maximum of two values consisting of max_val AND current_max  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
