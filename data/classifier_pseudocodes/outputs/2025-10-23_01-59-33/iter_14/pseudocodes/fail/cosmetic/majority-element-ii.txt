CLASS Solution  
    FUNCTION majorityElement(nums)  
        ASSIGN countA AS 0  
        ASSIGN countB AS 0  
        ASSIGN candidateA AS 0  
        ASSIGN candidateB AS 1  
        SET index AS 0  
        WHILE index LESS THAN LENGTH OF nums  
            LET currentItem BE nums AT index  
            IF currentItem IS candidateA THEN  
                countA ← countA + 1  
            ELSE IF currentItem IS candidateB THEN  
                countB ← countB + 1  
            ELSE IF countA IS 0 THEN  
                candidateA ← currentItem  
                countA ← 1  
            ELSE IF countB IS 0 THEN  
                candidateB ← currentItem  
                countB ← 1  
            ELSE  
                countA ← countA - 1  
                countB ← countB - 1  
            END IF  
            index ← index + 1  
        END WHILE  
        LET resultElements BE [candidateA, candidateB]  
        LET threshold BE LENGTH OF nums DIV 3  
        LET outputList BE EMPTY LIST  
        FOR EACH element IN resultElements  
            LET frequency BE 0  
            FOR EACH val IN nums  
                IF val EQUALS element THEN  
                    frequency ← frequency + 1  
                END IF  
            END FOR  
            IF frequency GREATER THAN threshold THEN  
                APPEND element TO outputList  
            END IF  
        END FOR  
        RETURN outputList  
    END FUNCTION  
END CLASS
