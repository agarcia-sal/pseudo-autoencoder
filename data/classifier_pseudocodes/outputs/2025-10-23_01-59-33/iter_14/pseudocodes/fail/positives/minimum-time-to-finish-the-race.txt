CLASS Solution
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)
        SET min_times TO a list of length FIFTEEN with all elements set to positive infinity
        FOR each pair of first_variable AND ratio IN tires
            SET total_time TO zero
            FOR index FROM zero TO FOURTEEN
                SET lap_time TO first_variable MULTIPLIED BY ratio RAISED TO the power of index
                IF lap_time IS GREATER THAN changeTime PLUS first_variable
                    BREAK the current loop
                END IF
                SET total_time TO total_time PLUS lap_time
                IF total_time IS LESS THAN element at position index of min_times
                    SET element at position index of min_times TO total_time
                END IF
            END FOR
        END FOR

        SET dp TO a list of length numLaps PLUS one with all elements set to positive infinity
        SET element at position zero of dp TO zero

        FOR i FROM one TO numLaps
            FOR j FROM zero TO THE LESSER OF i MINUS one AND FOURTEEN
                SET candidate_time TO element at position i MINUS j MINUS one of dp PLUS element at position j of min_times PLUS changeTime
                IF candidate_time IS LESS THAN element at position i of dp
                    SET element at position i of dp TO candidate_time
                END IF
            END FOR
        END FOR

        RETURN element at position numLaps of dp MINUS changeTime
    END FUNCTION
END CLASS
