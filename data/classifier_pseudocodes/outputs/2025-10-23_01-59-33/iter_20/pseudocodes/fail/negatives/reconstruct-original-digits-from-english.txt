CLASS Solution
    FUNCTION originalDigits(s)
        SET count TO the frequency count of each character in s
        
        SET digits TO a list of zero repeated ten times
        
        SET the element at position zero of digits TO the count of character Z in count
        SET the element at position two of digits TO the count of character W in count
        SET the element at position four of digits TO the count of character U in count
        SET the element at position six of digits TO the count of character X in count
        SET the element at position eight of digits TO the count of character G in count
        
        SET the element at position three of digits TO the count of character H in count MINUS the element at position eight of digits
        SET the element at position five of digits TO the count of character F in count MINUS the element at position four of digits
        SET the element at position seven of digits TO the count of character S in count MINUS the element at position six of digits
        SET the element at position nine of digits TO the count of character I in count MINUS the element at position five of digits MINUS the element at position six of digits MINUS the element at position eight of digits
        SET the element at position one of digits TO the count of character O in count MINUS the element at position zero of digits MINUS the element at position two of digits MINUS the element at position four of digits
        
        SET result TO an empty list
        
        FOR index FROM zero TO nine
            APPEND the string representation of index repeated the element at position index of digits times TO result
        END FOR
        
        RETURN the concatenation of all elements in result
    END FUNCTION
END CLASS
