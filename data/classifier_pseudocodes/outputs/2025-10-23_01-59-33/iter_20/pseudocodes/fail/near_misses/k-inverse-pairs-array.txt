CLASS Solution
    FUNCTION kInversePairs(n, k)
        SET MOD TO ten raised to the power of nine PLUS one
        INITIALIZE dp AS a two dimensional list with the first dimension FROM zero TO n INCLUSIVE and the second dimension FROM zero TO k INCLUSIVE FILLED WITH zero
        SET element at position zero OF list at position zero OF dp TO one
        
        FOR i FROM one TO n INCLUSIVE
            SET prefix_sum TO zero
            FOR j FROM zero TO k INCLUSIVE
                SET prefix_sum TO the remainder of prefix_sum PLUS element at position j OF list at position i minus one OF dp DIVIDED BY MOD
                IF j IS GREATER THAN OR EQUAL TO i THEN
                    SET prefix_sum TO the remainder of prefix_sum MINUS element at position j minus i OF list at position i minus one OF dp PLUS MOD DIVIDED BY MOD
                END IF
                SET element at position j OF list at position i OF dp TO prefix_sum
            END FOR
        END FOR
        
        RETURN element at position k OF list at position n OF dp
    END FUNCTION
END CLASS
