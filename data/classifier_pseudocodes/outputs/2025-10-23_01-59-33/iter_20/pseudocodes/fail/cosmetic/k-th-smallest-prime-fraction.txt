CLASS Solution
    FUNCTION kthSmallestPrimeFraction(arr, k)
        LET totalElements BE LENGTH(arr)
        LET heapElements BE an EMPTY LIST

        LET indexCounter BE 0
        WHILE indexCounter < totalElements - 1
            LET fractionValue BE arr[indexCounter] DIVIDED_BY arr[totalElements - 1]
            INSERT (fractionValue, indexCounter, totalElements - 1) INTO heapElements
            SET indexCounter TO indexCounter + 1
        END WHILE

        LET iterationCounter BE 0
        WHILE iterationCounter < k - 1
            EXTRACT_MIN FROM heapElements INTO (_, idxA, idxB)
            IF (idxB - 1) > idxA THEN
                LET newFraction BE arr[idxA] DIVIDED_BY arr[idxB - 1]
                INSERT (newFraction, idxA, idxB - 1) INTO heapElements
            END IF
            SET iterationCounter TO iterationCounter + 1
        END WHILE

        EXTRACT_MIN FROM heapElements INTO (_, finalI, finalJ)
        RETURN [arr[finalI], arr[finalJ]]
    END FUNCTION
END CLASS
