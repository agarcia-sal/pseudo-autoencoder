CLASS Solution
    FUNCTION countTriplets(list_of_numbers)
        SET dictionary_of_pair_bitwise_and_frequencies TO a new default dictionary with default integer value

        FOR each first_number IN list_of_numbers
            FOR each second_number IN list_of_numbers
                SET bitwise_and_result TO first_number BITWISE AND second_number
                INCREMENT the value associated with bitwise_and_result IN dictionary_of_pair_bitwise_and_frequencies BY one
            END FOR
        END FOR

        SET triplet_total_count TO zero

        FOR each candidate_number IN list_of_numbers
            FOR each pair_and_value AND corresponding_count IN dictionary_of_pair_bitwise_and_frequencies
                IF the bitwise AND of pair_and_value AND candidate_number EQUALS zero HOLDS
                    INCREMENT triplet_total_count BY the corresponding_count
                END IF
            END FOR
        END FOR

        RETURN triplet_total_count
    END FUNCTION
END CLASS
