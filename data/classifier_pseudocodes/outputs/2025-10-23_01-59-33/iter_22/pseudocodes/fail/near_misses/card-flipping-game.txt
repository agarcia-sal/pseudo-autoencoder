CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair x, y IN zipped fronts AND backs  
            IF x EQUALS y  
                ADD x TO same  
        SET all_numbers TO set of all elements IN fronts PLUS backs  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers IS NOT empty  
            RETURN minimum value IN good_numbers  
        ELSE  
            RETURN 0  
    END FUNCTION  
END CLASS
