CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH TO min_heap a tuple containing the fraction of element at position i of arr DIVIDED BY element at position n MINUS one of arr AND i AND n MINUS one  
        END FOR  
        FOR iteration FROM zero TO k MINUS one MINUS one  
            POP the smallest element FROM min_heap AND ASSIGN the values to unused_variable AND i AND j  
            IF j MINUS one IS GREATER THAN i THEN  
                PUSH TO min_heap a tuple containing the fraction of element at position i of arr DIVIDED BY element at position j MINUS one of arr AND i AND j MINUS one  
            END IF  
        END FOR  
        POP the smallest element FROM min_heap AND ASSIGN the values to unused_variable AND i AND j  
        RETURN a list containing element at position i of arr AND element at position j of arr  
    END FUNCTION  
END CLASS
