CLASS Solution  
    FUNCTION kthGrammar(n k)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two MULTIPLIED BY two RAISED TO THE POWER OF n MINUS two MINUS one  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN call kthGrammar WITH n MINUS one AND k  
        ELSE  
            SET previous_value TO call kthGrammar WITH n MINUS one AND k MINUS length_of_previous_row  
            SET complement_value TO one MINUS previous_value  
            RETURN complement_value  
        END IF  
    END FUNCTION  
END CLASS
