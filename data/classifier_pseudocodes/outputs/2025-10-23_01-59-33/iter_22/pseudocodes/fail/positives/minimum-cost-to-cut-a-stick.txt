CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted list consisting of all elements of cuts PLUS zero PLUS n  
        
        FUNCTION dp(left, right)  
            IF right MINUS left LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO a value representing positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right of cuts MINUS element at position left of cuts PLUS dp(left, i) PLUS dp(i, right)  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(zero, the length of cuts MINUS one)  
    END FUNCTION  
END CLASS
