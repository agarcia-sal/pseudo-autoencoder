CLASS Solution
    FUNCTION maximumANDSum(nums, numSlots)
        FUNCTION dp(index, slots)
            IF index EQUALS the LENGTH OF nums
                RETURN zero
            END IF

            SET max_and_sum TO zero
            SET slots_list TO CALL convert_tuple_to_list WITH slots

            FOR i FROM zero TO numSlots MINUS one
                IF element at position i of slots_list LESS THAN two
                    INCREMENT element at position i of slots_list BY one
                    SET and_sum TO element at position index of nums BITWISE AND (i PLUS one) PLUS CALL dp WITH index PLUS one AND CONVERT slots_list TO tuple
                    SET max_and_sum TO the GREATER OF max_and_sum AND and_sum
                    DECREMENT element at position i of slots_list BY one
                END IF
            END FOR

            RETURN max_and_sum
        END FUNCTION

        SET initial_slots TO CALL create_tuple_of_zeros_with_length WITH numSlots
        RETURN CALL dp WITH zero AND initial_slots
    END FUNCTION
END CLASS
