CLASS Solution
    FUNCTION isMatch(s, p)
        SET x TO LENGTH(s)
        SET y TO LENGTH(p)
        SET dp TO []
        SET u TO 0
        WHILE u <= x
            SET tempList TO []
            SET v TO 0
            WHILE v <= y
                APPEND False TO tempList
                SET v TO v + 1
            END WHILE
            APPEND tempList TO dp
            SET u TO u + 1
        END WHILE

        SET dp[0][0] TO True

        PROCEDURE fillFirstRow(w)
            IF w > y THEN
                RETURN
            END IF
            IF p[w - 1] = '*'
                SET dp[0][w] TO dp[0][w - 1]
            END IF
            fillFirstRow(w + 1)
        END PROCEDURE

        CALL fillFirstRow(1)

        PROCEDURE processRow(i, j)
            IF i > x THEN
                RETURN
            END IF
            IF j > y THEN
                CALL processRow(i + 1, 1)
                RETURN
            END IF

            SWITCH TRUE
                CASE (p[j - 1] = '*')
                    SET dp[i][j] TO dp[i - 1][j] OR dp[i][j - 1]
                BREAK

                CASE (p[j - 1] = '?' OR s[i - 1] = p[j - 1])
                    SET dp[i][j] TO dp[i - 1][j - 1]
                BREAK

                DEFAULT
                    // dp[i][j] remains False implicitly
                BREAK
            END SWITCH

            processRow(i, j + 1)
        END PROCEDURE

        CALL processRow(1, 1)

        RETURN dp[x][y]
    END FUNCTION
END CLASS
