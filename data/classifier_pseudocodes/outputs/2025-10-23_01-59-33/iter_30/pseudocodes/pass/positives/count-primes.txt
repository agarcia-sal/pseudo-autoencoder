CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO an empty list  
        FOR index FROM zero TO n MINUS one  
            APPEND True TO is_prime  
        END FOR  
        SET element at position zero of is_prime TO False  
        SET element at position one of is_prime TO False  
        SET limit TO the integer part of the square root of n  
        FOR start FROM two TO limit PLUS one  
            IF element at position start of is_prime EQUALS True  
                SET multiple TO start MULTIPLIED BY start  
                WHILE multiple LESS THAN n  
                    SET element at position multiple of is_prime TO False  
                    INCREMENT multiple BY start  
                END WHILE  
            END IF  
        END FOR  
        SET count TO zero  
        FOR each value IN is_prime  
            IF value EQUALS True  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
