CLASS Solution
    FUNCTION sumSubseqWidths(nums)
        ASSIGN 1000000000 PLUS 7 TO MOD
        EXECUTE nums SORTED ASCENDING AND ASSIGN BACK TO nums
        ASSIGN LENGTH OF nums TO count
        ASSIGN 0 TO accumulator
        CREATE LIST powers WITH count ELEMENTS INITIALIZED TO 1

        ASSIGN 1 TO index
        WHILE index NOT GREATER THAN count MINUS 1
            ASSIGN (powers[index MINUS 1] TIMES 2) MOD MOD TO powers[index]
            INCREMENT index BY 1
        END WHILE

        ASSIGN 0 TO pointer
        WHILE pointer LESS THAN count
            ASSIGN (accumulator PLUS (nums[pointer] TIMES (powers[pointer] MINUS powers[count MINUS pointer MINUS 1]))) MOD MOD TO accumulator
            INCREMENT pointer BY 1
        END WHILE

        RETURN accumulator
    END FUNCTION
END CLASS
