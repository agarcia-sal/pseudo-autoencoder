CLASS Solution  
    FUNCTION waysToPartition(nums parameter of list of integers k parameter of integer) RETURNS integer  
        SET total_sum TO the sum of all elements of nums  
        SET n TO the length of nums  
        
        CREATE left_diff as a dictionary with default integer zero  
        CREATE right_diff as a dictionary with default integer zero  
        
        SET current_sum TO zero  
        FOR index i FROM zero TO n MINUS one MINUS one  
            INCREMENT current_sum BY the element at position i of nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT the value associated with diff in right_diff BY one  
        END FOR  
        
        SET max_ways TO the value associated with zero in right_diff  
        
        SET current_sum TO zero  
        FOR index i FROM zero TO n MINUS one  
            INCREMENT current_sum BY the element at position i of nums  
            
            SET diff_change TO k MINUS the element at position i of nums  
            
            SET new_ways TO the value associated with negative diff_change in left_diff PLUS the value associated with diff_change in right_diff  
            
            IF new_ways GREATER THAN max_ways  
                SET max_ways TO new_ways  
            END IF  
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT the value associated with diff in right_diff BY one  
            INCREMENT the value associated with diff in left_diff BY one  
        END FOR  
        
        RETURN max_ways  
    END FUNCTION  
END CLASS
