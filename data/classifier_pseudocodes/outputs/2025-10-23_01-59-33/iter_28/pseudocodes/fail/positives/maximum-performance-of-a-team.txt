CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO list of pairs of element at position index of efficiency AND element at position index of speed FOR index FROM the first TO the last position in efficiency  
        SORT engineers BY the first element of each pair IN descending order  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair IN engineers  
            SET eff TO the first element of pair  
            SET spd TO the second element of pair  
            ADD spd TO speed_heap USING heap push operation  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap GREATER THAN k THEN  
                SET smallest_speed TO the result of heap pop operation on speed_heap  
                DECREMENT speed_sum BY smallest_speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance MODULO the result of ten RAISED TO THE POWER OF nine PLUS one  
    END FUNCTION  
END CLASS
