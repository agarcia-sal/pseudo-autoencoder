CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET paired_list TO the list of elements obtained by pairing each element at each position in position WITH the element at the corresponding position in speed  
        SORT paired_list in order from the greatest element of the first component to the smallest element of the first component  

        SET time_to_target TO an empty list  
        FOR each pair IN paired_list  
            SET p TO the first component of pair  
            SET s TO the second component of pair  
            SET time TO target MINUS p DIVIDED BY s  
            APPEND time TO time_to_target  
        END FOR  

        SET fleets TO zero  
        SET current_max_time TO zero  

        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  

        RETURN fleets  
    END FUNCTION  
END CLASS
