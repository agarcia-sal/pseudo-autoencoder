CLASS Solution  
    FUNCTION judgeSquareSum WITH parameter c  
        SET left TO zero  
        SET right TO integer VALUE OF the SQUARE ROOT OF c  
        WHILE left LESS THAN OR EQUAL TO right holds  
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right  
            IF current_sum EQUALS c THEN  
                RETURN true  
            ELSE IF current_sum LESS THAN c THEN  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
