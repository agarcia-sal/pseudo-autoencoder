CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(currentIndex, previousValue, sequenceAccumulator)  
            IF currentIndex IS THE SAME AS LENGTH OF nums THEN  
                IF LENGTH OF sequenceAccumulator EXCEEDS 1 THEN  
                    APPEND COPY OF sequenceAccumulator INTO results  
                END IF  
                EXIT FUNCTION  
            END IF  
            IF nums[currentIndex] IS AT LEAST previousValue THEN  
                ADD nums[currentIndex] TO sequenceAccumulator  
                EXECUTE dfs WITH (currentIndex + 1, nums[currentIndex], sequenceAccumulator)  
                DISCARD LAST ELEMENT FROM sequenceAccumulator  
            END IF  
            IF nums[currentIndex] IS DISTINCT FROM previousValue THEN  
                INVOKE dfs WITH (currentIndex + 1, previousValue, sequenceAccumulator)  
            END IF  
        END FUNCTION  
        INITIALIZE results AS EMPTY LIST  
        INVOKE dfs WITH (0, -1000, EMPTY LIST)  
        RETURN results  
    END FUNCTION  
END CLASS
