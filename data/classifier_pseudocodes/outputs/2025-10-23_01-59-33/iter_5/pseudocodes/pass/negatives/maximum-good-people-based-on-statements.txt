n = len(statements)
max_good = 0

for mask in 0..(1 << n) - 1:
    good = bits of mask as bool list length n
    valid = True

    for i in 0..n-1 if good[i]:
        for j in 0..n-1:
            if (statements[i][j] == 0 and good[j]) or (statements[i][j] == 1 and not good[j]):
                valid = False
                break
        if not valid: break

    if valid:
        max_good = max(max_good, count of True in good)

return max_good
