CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SORT nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of length n where each element is one  
        
        FOR i FROM one TO n MINUS one  
            SET element at position i of power TO the element at position i MINUS one of power MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS element at position i of nums MULTIPLIED BY the difference BETWEEN element at position i of power AND element at position n MINUS i MINUS one of power  
            THEN TAKE total_sum MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
