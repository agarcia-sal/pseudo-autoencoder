CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new double ended queue containing a single tuple of zero for position one for speed and zero for steps  
        ASSIGN reference TO a new empty set  
        ADD a tuple of zero for position and one for speed TO the set  
        
        WHILE the queue is not empty  
            REMOVE and ASSIGN a tuple from the left side of the queue TO position speed and steps  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed NOT IN the set AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO the set  
                APPEND the tuple of next_position next_speed and steps PLUS one TO the right side of the queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed NOT IN the set AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO the set  
                APPEND the tuple of position next_speed and steps PLUS one TO the right side of the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
