CLASS Solution
    FUNCTION canJump(nums)
        ASSIGN 0 TO maxReach
        ASSIGN 0 TO indexCounter
        WHILE indexCounter <= (SIZE OF nums) - 1 DO
            IF maxReach < indexCounter THEN
                RETURN false
            ENDIF
            ASSIGN indexCounter + nums[indexCounter] TO currentReach
            IF currentReach > maxReach THEN
                maxReach := currentReach
            ENDIF
            indexCounter := indexCounter + 1
        ENDWHILE
        IF (maxReach + 1) > (SIZE OF nums) THEN
            RETURN true
        ELSE
            RETURN false
        ENDIF
    END FUNCTION
END CLASS
