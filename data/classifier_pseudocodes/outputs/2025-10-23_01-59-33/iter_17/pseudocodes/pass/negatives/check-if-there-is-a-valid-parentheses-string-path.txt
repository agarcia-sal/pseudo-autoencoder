```
m, n ← dims(grid)
if (m + n - 1) mod 2 ≠ 0: return False
dp ← 3D array[m+1][n+1][m+n+1] of False
dp[0][1][0] ← True

for i in 0..m-1:
  for j in 0..n-1:
    for k in 0..m+n:
      if dp[i][j+1][k] or dp[i+1][j][k]:
        if grid[i][j] = '(' and k+1 < m+n+1: dp[i+1][j+1][k+1] ← True
        else if grid[i][j] = ')' and k-1 ≥ 0: dp[i+1][j+1][k-1] ← True

return dp[m][n][0]
```
