CLASS Solution
    FUNCTION longestWord(words)
        DECLARE comparator AS FUNCTION(a, b)
            IF LENGTH(a) NOT EQUAL TO LENGTH(b)
                RETURN LENGTH(a) LESS THAN LENGTH(b)
            ELSE
                RETURN a GREATER THAN b
            END IF
        END FUNCTION

        CALL words.SORT_WITH(comparator.REVERSE_ORDER())

        DECLARE approved_terms AS NEW SET()
        DECLARE champion_term AS ""

        DECLARE index AS 0
        WHILE index LESS THAN LENGTH(words)
            DECLARE candidate AS words[index]
            DECLARE prefix_term AS candidate[0 TO (LENGTH(candidate) - 2)]

            IF NOT (LENGTH(candidate) DIFFERENT FROM 1 OR NOT approved_terms.CONTAINS(prefix_term))
                ADD candidate TO approved_terms

                IF LENGTH(candidate) IS GREATER THAN LENGTH(champion_term)
                    ASSIGN candidate TO champion_term
                END IF
            END IF

            INCREMENT index BY 1
        END WHILE

        RETURN champion_term
    END FUNCTION
END CLASS
