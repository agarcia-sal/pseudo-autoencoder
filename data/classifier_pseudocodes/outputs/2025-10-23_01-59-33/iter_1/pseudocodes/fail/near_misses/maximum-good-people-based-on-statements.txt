CLASS Solution
    FUNCTION maximumGood(statements)
        LET count_humans BE 0
        LET total_people BE LENGTH(statements)
        
        FUNCTION evaluate(index, state_map)
            IF index IS total_people THEN
                LET truth_map BE EMPTY MAP
                FOR pos IN 0 TO total_people - 1 BY 1
                    IF ((state_map >> pos) AND 1) = 1 THEN
                        SET truth_map[pos] = TRUE
                    ELSE
                        SET truth_map[pos] = FALSE
                    END IF
                END FOR
                FOR examine IN 0 TO total_people - 1 BY 1
                    IF truth_map[examine] = TRUE THEN
                        FOR target IN 0 TO total_people - 1 BY 1
                            LET claim = statements[examine][target]
                            IF claim = 0 AND truth_map[target] = TRUE THEN
                                RETURN FALSE
                            ELSE IF claim = 1 AND truth_map[target] = FALSE THEN
                                RETURN FALSE
                            END IF
                        END FOR
                    END IF
                END FOR
                RETURN TRUE
            END IF
            RETURN evaluate(index + 1, state_map) OR evaluate(index + 1, state_map OR (1 << index))
        END FUNCTION
        
        FUNCTION count_set_bits(number)
            LET bits_count BE 0
            WHILE number > 0
                bits_count = bits_count + (number AND 1)
                number = number >> 1
            END WHILE
            RETURN bits_count
        END FUNCTION
        
        FOR mask FROM 0 TO (2 ^ total_people) - 1 BY 1
            IF evaluate(0, mask) = TRUE THEN
                LET current_count BE count_set_bits(mask)
                IF current_count > count_humans THEN
                    count_humans = current_count
                END IF
            END IF
        END FOR
        
        RETURN count_humans
    END FUNCTION
END CLASS
