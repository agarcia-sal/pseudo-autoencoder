CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO a list of amount PLUS one elements each set TO amount PLUS one  
        SET element at position zero of dp TO zero  
        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO zero  
                    SET dp at position a TO the minimum BETWEEN dp at position a AND dp at position a MINUS coin PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF dp at position amount IS NOT EQUAL TO amount PLUS one  
            RETURN dp at position amount  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
