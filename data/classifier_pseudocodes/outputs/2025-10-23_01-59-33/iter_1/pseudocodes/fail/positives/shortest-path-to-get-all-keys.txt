CLASS Solution  
    FUNCTION shortestPathAllKeys(grid)  
        SET directions TO the list of pairs representing movement right one step left one step down one step and up one step  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start TO no value assigned  
        SET num_keys TO zero  
        FOR i FROM zero TO the value m MINUS one  
            FOR j FROM zero TO the value n MINUS one  
                IF element at position i of grid at position j EQUALS the character at sign at symbol  
                    SET start TO the pair of values i and j  
                ELSE IF element at position i of grid at position j IS a lowercase letter  
                    INCREMENT num_keys BY one  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue containing the single element of the tuple of value at position zero of start value at position one of start zero zero  
        SET visited TO a new set containing the single element of the tuple of value at position zero of start value at position one of start zero  
        WHILE queue is not empty  
            SET x y keys steps TO the tuple removed from the front of queue  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET cell TO element at position nx of grid at position ny  
                    IF cell EQUALS the hash symbol  
                        CONTINUE to next iteration of FOR loop  
                    END IF  
                    IF cell IS an uppercase letter  
                        SET key_needed TO one SHIFTED LEFT BY the Unicode code point of the lowercase form of cell MINUS the Unicode code point of letter a  
                        IF the result of keys AND key_needed EQUALS zero  
                            CONTINUE to next iteration of FOR loop  
                        END IF  
                    END IF  
                    IF cell IS a lowercase letter  
                        SET new_keys TO the result of keys OR one SHIFTED LEFT BY the Unicode code point of cell MINUS the Unicode code point of letter a  
                        IF new_keys EQUALS one SHIFTED LEFT BY num_keys MINUS one  
                            RETURN steps PLUS one  
                        END IF  
                    ELSE  
                        SET new_keys TO keys  
                    END IF  
                    IF the tuple nx ny new_keys IS NOT in visited  
                        ADD the tuple nx ny new_keys TO visited  
                        APPEND the tuple nx ny new_keys steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
