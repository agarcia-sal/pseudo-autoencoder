CLASS Solution  
    FUNCTION kthGrammar(n k)  
        WHILE TRUE  
            IF n IS 1 THEN  
                RETURN 0  
            END IF  
            ASSIGN (2 ^ (n - 1)) TO temp_length  
            ASSIGN temp_length TO half_length  
            IF k <= half_length THEN  
                ASSIGN n - 1 TO next_n  
                RETURN kthGrammar(next_n k)  
            ELSE  
                ASSIGN n - 1 TO prev_n  
                ASSIGN k - half_length TO index_offset  
                ASSIGN kthGrammar(prev_n index_offset) TO prev_val  
                ASSIGN 1 - prev_val TO result_val  
                RETURN result_val  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
