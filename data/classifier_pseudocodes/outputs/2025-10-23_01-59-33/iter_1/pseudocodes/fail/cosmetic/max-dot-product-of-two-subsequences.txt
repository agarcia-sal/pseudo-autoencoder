CLASS Solution  
    FUNCTION maxDotProduct(nums1 PARAMETER, nums2 PARAMETER)  
        SET length1 AS LENGTH OF nums1  
        SET length2 AS LENGTH OF nums2  
        CREATE dp AS LIST OF LISTS  
        FOR index1 FROM 0 TO length1  
            CREATE row AS LIST  
            FOR index2 FROM 0 TO length2  
                APPEND NEGATIVE_INFINITY TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  

        SET outerCounter TO 1  
        WHILE outerCounter <= length1  
            SET innerCounter TO 1  
            WHILE innerCounter <= length2  
                SET productCurrent TO (nums1[outerCounter - 1]) * (nums2[innerCounter - 1])  
                SET choice1 TO productCurrent  
                SET choice2 TO dp[outerCounter - 1][innerCounter]  
                SET choice3 TO dp[outerCounter][innerCounter - 1]  
                SET choice4 TO dp[outerCounter - 1][innerCounter - 1] + productCurrent  
                
                SET dp[outerCounter][innerCounter] TO choice1  
                IF choice2 > dp[outerCounter][innerCounter]  
                    SET dp[outerCounter][innerCounter] TO choice2  
                END IF  
                IF choice3 > dp[outerCounter][innerCounter]  
                    SET dp[outerCounter][innerCounter] TO choice3  
                END IF  
                IF choice4 > dp[outerCounter][innerCounter]  
                    SET dp[outerCounter][innerCounter] TO choice4  
                END IF  
                
                INCREMENT innerCounter BY 1  
            END WHILE  
            INCREMENT outerCounter BY 1  
        END WHILE  

        RETURN dp[length1][length2]  
    END FUNCTION  
END CLASS
