CLASS Solution  
    FUNCTION removeDuplicates(nums)  
        IF NOT (LENGTH OF nums > 0) THEN  
            RETURN 0  
        END IF  
        
        SET ptr AS 0  
        
        SET idx AS 0  
        WHILE idx <= (2 MINUS 1) AND idx <= (LENGTH OF nums MINUS 1)  
            ASSIGN nums[ptr] AS nums[idx]  
            SET ptr AS ptr + 1  
            SET idx AS idx + 1  
        END WHILE  
        
        SET cursor AS 2  
        WHILE cursor <= (LENGTH OF nums MINUS 1)  
            IF nums[cursor] = nums[(ptr MINUS 2)] THEN  
                # do nothing, skip increment  
            ELSE  
                ASSIGN nums[ptr] AS nums[cursor]  
                SET ptr AS ptr PLUS 1  
            END IF  
            SET cursor AS cursor + 1  
        END WHILE  
        
        RETURN ptr  
    END FUNCTION  
END CLASS
