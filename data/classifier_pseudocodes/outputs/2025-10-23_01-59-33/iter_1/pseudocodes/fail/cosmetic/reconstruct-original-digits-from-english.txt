CLASS Solution  
    FUNCTION originalDigits(s)  
        SET freqMap AS map of character counts in s  
        SET digitCounts AS list WITH TEN elements INITIATED TO 0  
        
        SET digitCounts[0] AS freqMap['z']  
        SET digitCounts[2] AS freqMap['w']  
        SET digitCounts[4] AS freqMap['u']  
        SET digitCounts[6] AS freqMap['x']  
        SET digitCounts[8] AS freqMap['g']  
        
        SET digitCounts[3] AS freqMap['h'] - digitCounts[8]  
        SET digitCounts[5] AS freqMap['f'] - digitCounts[4]  
        SET digitCounts[7] AS freqMap['s'] - digitCounts[6]  
        
        SET tempI AS freqMap['i']  
        SET digitCounts[9] AS tempI - digitCounts[5] - digitCounts[6] - digitCounts[8]  
        
        SET tempO AS freqMap['o']  
        SET digitCounts[1] AS tempO - digitCounts[0] - digitCounts[2] - digitCounts[4]  
        
        SET collectionOfChars AS empty list  
        
        SET numberIdx AS 0  
        WHILE numberIdx LESS THAN 10  
            SET countRepeat AS digitCounts[numberIdx]  
            WHILE countRepeat GREATER THAN 0  
                APPEND STRING(numberIdx) TO collectionOfChars  
                ASSIGN countRepeat ← countRepeat - 1  
            END WHILE  
            ASSIGN numberIdx ← numberIdx + 1  
        END WHILE  
        
        SET outputString AS CONCATENATE elements of collectionOfChars SEQUENTIALLY  
        RETURN outputString  
    END FUNCTION  
END CLASS
