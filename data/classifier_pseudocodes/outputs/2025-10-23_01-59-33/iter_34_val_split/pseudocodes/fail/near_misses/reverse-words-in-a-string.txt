CLASS Solution
    FUNCTION reverseWords(s)
        DECLARE wordList AS empty array
        DECLARE startIndex AS 0
        DECLARE lengthOfString AS LENGTH(s)
        WHILE startIndex < lengthOfString
            WHILE startIndex < lengthOfString AND s[startIndex] = ' '
                INCREMENT startIndex BY 1
            END WHILE
            IF startIndex = lengthOfString
                EXIT WHILE
            END IF
            DECLARE endIndex AS startIndex
            WHILE endIndex < lengthOfString AND s[endIndex] ≠ ' '
                INCREMENT endIndex BY 1
            END WHILE
            APPEND SUBSTRING(s, startIndex, endIndex - startIndex) TO wordList
            SET startIndex TO endIndex
        END WHILE

        DECLARE index AS LENGTH(wordList) - 1
        DECLARE reversedList AS empty array
        WHILE index ≥ 0
            APPEND wordList[index] TO reversedList
            DECREMENT index BY 1
        END WHILE

        DECLARE resultString AS EMPTY STRING
        DECLARE reverseIndex AS 0
        DECLARE totalWords AS LENGTH(reversedList)
        WHILE reverseIndex < totalWords
            APPEND reversedList[reverseIndex] TO resultString
            IF reverseIndex ≠ totalWords - 1 THEN
                APPEND ' ' TO resultString
            END IF
            INCREMENT reverseIndex BY 1
        END WHILE

        RETURN resultString
    END FUNCTION
END CLASS
