CLASS Solution
    FUNCTION originalDigits(s)
        SET freqMap TO empty map
        SET pos TO 0
        WHILE pos < length of s
            SET ch TO character at position pos of s
            IF freqMap contains ch
                SET tempVar TO freqMap[ch]
                SET freqMap[ch] TO tempVar + 1
            ELSE
                SET freqMap[ch] TO 1
            END IF
            SET pos TO pos + 1
        END WHILE

        SET digitCounts TO list of ten zeros

        SET digitCounts[0] TO (IF freqMap contains 'z' THEN freqMap['z'] ELSE 0)
        SET digitCounts[2] TO (IF freqMap contains 'w' THEN freqMap['w'] ELSE 0)
        SET digitCounts[4] TO (IF freqMap contains 'u' THEN freqMap['u'] ELSE 0)
        SET digitCounts[6] TO (IF freqMap contains 'x' THEN freqMap['x'] ELSE 0)
        SET digitCounts[8] TO (IF freqMap contains 'g' THEN freqMap['g'] ELSE 0)

        SET foundEights TO digitCounts[8]
        SET digitCounts[3] TO (IF freqMap contains 'h' THEN freqMap['h'] ELSE 0) - foundEights

        SET foundFours TO digitCounts[4]
        SET digitCounts[5] TO (IF freqMap contains 'f' THEN freqMap['f'] ELSE 0) - foundFours

        SET foundSixes TO digitCounts[6]
        SET digitCounts[7] TO (IF freqMap contains 's' THEN freqMap['s'] ELSE 0) - foundSixes

        SET foundFives TO digitCounts[5]
        SET digitCounts[9] TO (IF freqMap contains 'i' THEN freqMap['i'] ELSE 0) - foundFives - foundSixes - foundEights

        SET foundZeros TO digitCounts[0]
        SET foundTwos TO digitCounts[2]
        SET foundFoursCount TO digitCounts[4]
        SET digitCounts[1] TO (IF freqMap contains 'o' THEN freqMap['o'] ELSE 0) - foundZeros - foundTwos - foundFoursCount

        SET outputList TO empty list

        FUNCTION appendDigits(index, count)
            IF count = 0 THEN
                RETURN
            ELSE
                APPEND string representation of index TO outputList
                CALL appendDigits(index, count - 1)
            END IF
        END FUNCTION

        FUNCTION processIndex(idx)
            IF idx > 9 THEN
                RETURN
            ELSE
                CALL appendDigits(idx, digitCounts[idx])
                CALL processIndex(idx + 1)
            END IF
        END FUNCTION

        CALL processIndex(0)

        SET finalResult TO empty string
        SET counterVar TO 0
        WHILE counterVar < length of outputList
            SET finalResult TO finalResult + outputList[counterVar]
            SET counterVar TO counterVar + 1
        END WHILE

        RETURN finalResult
    END FUNCTION
END CLASS
