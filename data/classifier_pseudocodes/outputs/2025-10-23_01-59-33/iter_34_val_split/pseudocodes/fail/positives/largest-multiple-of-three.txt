CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO a mapping of digits to their frequencies  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the remainder when total_sum is divided by three  
        IF remainder EQUALS one  
            FOR each digit IN the list containing one four seven  
                IF the frequency of digit in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            OTHERWISE  
                FOR each digit IN the list containing two five eight  
                    IF the frequency of digit in count IS GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END OTHERWISE  
        END IF  
        IF remainder EQUALS two  
            FOR each digit IN the list containing two five eight  
                IF the frequency of digit in count IS GREATER THAN zero  
                    DECREMENT the frequency of digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            OTHERWISE  
                FOR each digit IN the list containing one four seven  
                    IF the frequency of digit in count IS GREATER THAN one  
                        DECREMENT the frequency of digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END OTHERWISE  
        END IF  
        SET result TO an empty list  
        FOR digit FROM nine DOWN TO zero  
            APPEND the string form of digit repeated frequency of digit in count times TO result  
        END FOR  
        SET final_number TO the concatenation of all elements in result  
        IF final_number IS NOT empty AND the element at position one of final_number EQUALS zero  
            RETURN zero as string  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
