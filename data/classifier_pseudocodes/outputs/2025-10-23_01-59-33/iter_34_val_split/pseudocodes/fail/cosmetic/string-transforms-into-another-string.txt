CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        ASSIGN True TO resultFlag  
        IF NOT (str1 = str2) THEN  
            DECLARE charMapping AS empty dictionary of character to character  
            DECLARE indexCounter AS 0  
            WHILE indexCounter < LENGTH(str1) DO  
                DECLARE currentChar1 AS str1[indexCounter]  
                DECLARE currentChar2 AS str2[indexCounter]  
                IF currentChar1 IN charMapping THEN  
                    IF charMapping[currentChar1] ≠ currentChar2 THEN  
                        ASSIGN False TO resultFlag  
                        BREAK  
                    END IF  
                ELSE  
                    charMapping[currentChar1] ← currentChar2  
                END IF  
                INCREMENT indexCounter BY 1  
            END WHILE  
            IF SIZE(SET OF str2) = 26 THEN  
                ASSIGN False TO resultFlag  
            END IF  
        END IF  
        RETURN resultFlag  
    END FUNCTION  
END CLASS
