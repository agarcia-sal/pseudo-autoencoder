CLASS Solution
    FUNCTION longestWord(words)
        REORDER words USING a comparator that PRIORITIZES shorter length first, THEN lex order ascending
        INITIALIZE valid_words AS an empty set container
        ASSIGN "" TO longest_word
        
        ASSIGN 0 TO cursor
        WHILE cursor < LENGTH(words)
            ASSIGN words[cursor] TO current_candidate
            
            ASSIGN (LENGTH(current_candidate) = 1) OR (SUBSTRING(current_candidate, 0, LENGTH(current_candidate) - 1) IS IN valid_words) TO is_valid
            
            IF NOT is_valid THEN
                INCREMENT cursor BY 1
                CONTINUE TO next iteration
            END IF
            
            INSERT current_candidate INTO valid_words
            
            IF LENGTH(current_candidate) > LENGTH(longest_word) THEN
                ASSIGN current_candidate TO longest_word
            END IF
            
            INCREMENT cursor BY 1
        END WHILE
        RETURN longest_word
    END FUNCTION
END CLASS
