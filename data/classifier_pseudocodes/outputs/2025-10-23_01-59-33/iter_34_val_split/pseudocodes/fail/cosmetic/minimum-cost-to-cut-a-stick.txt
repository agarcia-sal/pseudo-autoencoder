CLASS Solution
	 FUNCTION minCost(n, cuts)
		 ASSIGN temp_array TO array_with_elements_of(cuts)
		 ATTACH_ELEMENT temp_array WITH zero
		 ATTACH_ELEMENT temp_array WITH n
		 SORT_ARRAY temp_array ASCENDING

		 FUNCTION dp(left, right)
			 IF (right - left) > 1 THEN
				 ASSIGN best_val TO +âˆž
				 ASSIGN iter_var TO left + 1
				 WHILE iter_var < right
					 ASSIGN current_cost TO (temp_array[right] - temp_array[left]) + dp(left, iter_var) + dp(iter_var, right)
					 IF current_cost < best_val THEN
						 ASSIGN best_val TO current_cost
					 ENDIF
					 ASSIGN iter_var TO iter_var + 1
				 ENDWHILE

				 YIELD best_val
			 ELSE
				 YIELD 0
			 ENDIF
		 END FUNCTION

		 YIELD dp(0, LENGTH_OF(temp_array) - 1)
	 END FUNCTION
END CLASS
