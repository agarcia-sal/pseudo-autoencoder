CLASS Solution
    FUNCTION largestMultipleOfThree(digits)
        ASSIGN a map named freq_map WITH count of occurrences in digits
        ASSIGN sum_total WITH 0
        ASSIGN index_i WITH 0
        WHILE index_i LESS THAN length of digits DO
            INCREMENT sum_total BY digits[index_i]
            INCREMENT index_i BY 1
        END WHILE

        ASSIGN rem TO sum_total MOD 3

        IF rem IS 1 THEN
            ASSIGN option_found TO FALSE
            ASSIGN arr_chk TO [1,4,7]
            ASSIGN pos_j TO 0
            WHILE pos_j LESS THAN length of arr_chk AND NOT option_found DO
                ASSIGN key_d TO arr_chk[pos_j]
                IF freq_map[key_d] GREATER THAN 0 THEN
                    DECREMENT freq_map[key_d] BY 1
                    ASSIGN option_found TO TRUE
                END IF
                INCREMENT pos_j BY 1
            END WHILE

            IF NOT option_found THEN
                ASSIGN arr_alt TO [2,5,8]
                ASSIGN pos_k TO 0
                WHILE pos_k LESS THAN length of arr_alt AND NOT option_found DO
                    ASSIGN key_e TO arr_alt[pos_k]
                    IF freq_map[key_e] GREATER THAN 1 THEN
                        DECREMENT freq_map[key_e] BY 2
                        ASSIGN option_found TO TRUE
                    END IF
                    INCREMENT pos_k BY 1
                END WHILE
            END IF

        ELSE IF rem IS 2 THEN
            ASSIGN found_flag TO FALSE
            ASSIGN first_arr TO [2,5,8]
            ASSIGN pos_m TO 0
            WHILE pos_m LESS THAN length of first_arr AND NOT found_flag DO
                ASSIGN val_n TO first_arr[pos_m]
                IF freq_map[val_n] GREATER THAN 0 THEN
                    DECREMENT freq_map[val_n] BY 1
                    ASSIGN found_flag TO TRUE
                END IF
                INCREMENT pos_m BY 1
            END WHILE

            IF NOT found_flag THEN
                ASSIGN second_arr TO [1,4,7]
                ASSIGN pos_o TO 0
                WHILE pos_o LESS THAN length of second_arr AND NOT found_flag DO
                    ASSIGN val_p TO second_arr[pos_o]
                    IF freq_map[val_p] GREATER THAN 1 THEN
                        DECREMENT freq_map[val_p] BY 2
                        ASSIGN found_flag TO TRUE
                    END IF
                    INCREMENT pos_o BY 1
                END WHILE
            END IF
        END IF

        ASSIGN output_list TO empty array
        ASSIGN digit_d TO 9
        WHILE digit_d GREATER THAN OR EQUAL TO 0 DO
            ASSIGN count_q TO freq_map[digit_d]
            ASSIGN run_idx TO 0
            WHILE run_idx LESS THAN count_q DO
                APPEND string representation of digit_d TO output_list
                INCREMENT run_idx BY 1
            END WHILE
            DECREMENT digit_d BY 1
        END WHILE

        ASSIGN final_str TO concatenate all strings in output_list without separator

        IF final_str IS NOT EMPTY AND NOT (final_str[0] NOT EQUAL 0) THEN
            RETURN "0"
        END IF

        RETURN final_str
    END FUNCTION
END CLASS
