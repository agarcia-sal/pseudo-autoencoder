CLASS Solution  
    FUNCTION strobogrammaticInRange(low  
                                    high)  
        FUNCTION generate_strobogrammatic(n  
                                         final_length)  
            IF NOT (n NOT EQUALS 0)  
                RETURN LIST("")
            END IF  
            
            IF n == 1  
                RETURN LIST("0", "1", "8")
            END IF  
            
            ASSIGN generate_strobogrammatic(n - 2, final_length) TO temp_collection  
            ASSIGN LIST() TO output_collection  
            ASSIGN 0 TO idx  
            WHILE idx < LENGTH(temp_collection)  
                ASSIGN temp_collection[idx] TO current_string  
                IF n != final_length  
                    APPEND ("0" + current_string + "0") TO output_collection  
                END IF  
                APPEND ("1" + current_string + "1") TO output_collection  
                APPEND ("6" + current_string + "9") TO output_collection  
                APPEND ("8" + current_string + "8") TO output_collection  
                APPEND ("9" + current_string + "6") TO output_collection  
                INCREMENT idx BY 1  
            END WHILE  
            RETURN output_collection  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low  
                                       high)  
            ASSIGN 0 TO aggregate_counter  
            FOR curr_length FROM LENGTH(low) TO LENGTH(high)  
                ASSIGN generate_strobogrammatic(curr_length, curr_length) TO candidates  
                ASSIGN 0 TO index_tracker  
                WHILE index_tracker < LENGTH(candidates)  
                    ASSIGN candidates[index_tracker] TO candidate_str  
                    IF NOT ( (curr_length == LENGTH(low) AND candidate_str >= low) AND (curr_length == LENGTH(high) AND candidate_str <= high) )  
                        OR (curr_length != LENGTH(low) AND curr_length != LENGTH(high))  
                        IF (curr_length == LENGTH(low) AND candidate_str < low) OR (curr_length == LENGTH(high) AND candidate_str > high)  
                            INCREMENT index_tracker BY 1  
                            CONTINUE  
                        END IF  
                    END IF  
                    INCREMENT aggregate_counter BY 1  
                    INCREMENT index_tracker BY 1  
                END WHILE  
            END FOR  
            RETURN aggregate_counter  
        END FUNCTION  
        
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
