CLASS Solution  
    FUNCTION reverseWords(s)  
        ASSIGN tokens TO array derived by dividing s on whitespace, ignoring extra spaces  
        DECLARE idx AS integer = 0  
        WHILE idx < LENGTH(tokens) / 2 DO  
            DECLARE temp AS string = tokens[idx]  
            ASSIGN tokens[idx] TO tokens[LENGTH(tokens) - 1 - idx]  
            ASSIGN tokens[LENGTH(tokens) - 1 - idx] TO temp  
            INCREMENT idx BY 1  
        END WHILE  
        DECLARE result AS string = EMPTY STRING  
        FOR i FROM 0 TO LENGTH(tokens) - 1 DO  
            IF i EQUALS 0 THEN  
                ASSIGN result TO tokens[i]  
            ELSE  
                ASSIGN result TO result + " " + tokens[i]  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
