CLASS Solution  
    FUNCTION sumOfFlooredPairs(self, nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET max_num TO the maximum value in nums  
        SET count TO a mapping of each unique number in nums to its frequency  
        SET prefix_sum TO a list of zeros with length max_num PLUS one  
        
        FOR each num IN the keys of count  
            INCREMENT prefix_sum at position num BY count at num  
        END FOR  
        
        FOR index FROM one TO max_num  
            INCREMENT prefix_sum at position index BY prefix_sum at position index MINUS one  
        END FOR  
        
        SET result TO zero  
        
        FOR each num IN the keys of count  
            FOR multiple FROM one TO integer division of max_num BY num  
                SET start TO num MULTIPLIED BY multiple  
                SET end TO the smaller value between num MULTIPLIED BY (multiple PLUS one) MINUS one AND max_num  
                SET multiples_count TO prefix_sum at position end MINUS prefix_sum at position start MINUS one  
                INCREMENT result BY multiples_count MULTIPLIED BY multiple MULTIPLIED BY count at num  
                SET result TO result MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
