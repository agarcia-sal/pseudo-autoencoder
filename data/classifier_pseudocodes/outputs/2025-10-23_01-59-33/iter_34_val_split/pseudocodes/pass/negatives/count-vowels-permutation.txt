CLASS Solution
    FUNCTION countVowelPermutation(parameter_n)
        SET constant_MODULO TO the value of ten raised to the power of nine PLUS one

        SET count_a TO one
        SET count_e TO one
        SET count_i TO one
        SET count_o TO one
        SET count_u TO one

        FOR each integer_index FROM two TO parameter_n INCLUSIVE
            SET next_count_a TO count_e PLUS count_i PLUS count_u
            SET next_count_e TO count_a PLUS count_i
            SET next_count_i TO count_e PLUS count_o
            SET next_count_o TO count_i
            SET next_count_u TO count_i PLUS count_o

            SET count_a TO next_count_a MODULO constant_MODULO
            SET count_e TO next_count_e MODULO constant_MODULO
            SET count_i TO next_count_i MODULO constant_MODULO
            SET count_o TO next_count_o MODULO constant_MODULO
            SET count_u TO next_count_u MODULO constant_MODULO
        END FOR

        SET total_count TO count_a PLUS count_e PLUS count_i PLUS count_o PLUS count_u
        SET result TO total_count MODULO constant_MODULO
        RETURN result
    END FUNCTION
END CLASS
