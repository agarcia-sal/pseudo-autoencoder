        FUNCTION valueOf WITH PARAMETER s AS string RETURNS floating point number  
            IF the position of the opening parenthesis character in s EQUALS negative one  
                RETURN conversion of s TO floating point number  
            END IF  
            
            SET leftParenIndex TO the position of the opening parenthesis character in s  
            SET rightParenIndex TO the position of the closing parenthesis character in s  
            SET dotIndex TO the position of the decimal point character in s  
            
            SET integerAndNonRepeating TO conversion TO floating point number of the substring of s from the start TO the position leftParenIndex MINUS one  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  
            
            SET repeating TO integer conversion of the substring of s from the position one AFTER leftParenIndex TO the position rightParenIndex MINUS one  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  
            
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY zero POINT one RAISED TO THE POWER OF nonRepeatingLength MULTIPLIED BY the element at position repeatingLength of ratios  
        END FUNCTION
        
        RETURN the absolute value OF valueOf of s MINUS valueOf of t IS LESS THAN zero POINT zero zero zero zero zero zero zero one times ten RAISED TO THE POWER OF negative nine  
    END FUNCTION  
END CLASS
