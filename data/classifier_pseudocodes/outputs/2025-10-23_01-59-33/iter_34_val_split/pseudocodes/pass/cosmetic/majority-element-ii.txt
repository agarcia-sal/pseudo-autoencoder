CLASS Solution  
    FUNCTION majorityElement(nums)  
        ASSIGN 0 TO countA  
        ASSIGN 0 TO countB  
        ASSIGN 0 TO candidateA  
        ASSIGN 1 TO candidateB  
        ASSIGN 0 TO idx  
        WHILE idx LESS THAN LENGTH OF nums DO  
            ASSIGN nums[idx] TO currentItem  
            IF currentItem EQUALS candidateA THEN  
                countA ← countA + 1  
            ELSE IF currentItem EQUALS candidateB THEN  
                countB ← countB + 1  
            ELSE  
                IF NOT (countA NOT EQUAL TO 0) THEN  
                    candidateA ← currentItem  
                    countA ← 1  
                ELSE IF countB EQUALS 0 THEN  
                    candidateB ← currentItem  
                    countB ← 1  
                ELSE  
                    countA ← countA - 1  
                    countB ← countB - 1  
                END IF  
            END IF  
            idx ← idx + 1  
        END WHILE  
        ASSIGN emptyList TO resultList  
        ASSIGN LENGTH OF nums DIVIDED BY 3 TO threshold  
        ASSIGN 0 TO resultIndex  
        WHILE resultIndex LESS THAN 2 DO  
            ASSIGN (candidateA, candidateB)[resultIndex] TO tempCandidate  
            ASSIGN 0 TO tally  
            ASSIGN 0 TO scanIndex  
            WHILE scanIndex LESS THAN LENGTH OF nums DO  
                IF nums[scanIndex] EQUALS tempCandidate THEN  
                    tally ← tally + 1  
                END IF  
                scanIndex ← scanIndex + 1  
            END WHILE  
            IF tally GREATER THAN threshold THEN  
                APPEND tempCandidate TO resultList  
            END IF  
            resultIndex ← resultIndex + 1  
        END WHILE  
        RETURN resultList  
    END FUNCTION  
END CLASS
