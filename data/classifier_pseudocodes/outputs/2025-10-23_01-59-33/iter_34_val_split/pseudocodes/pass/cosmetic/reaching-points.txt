CLASS Solution
    FUNCTION reachingPoints(sx sx PARAMETER sy sy PARAMETER tx tx PARAMETER ty ty) RETURNS boolean
        DEFINE FUNCTION loop_step(current_tx, current_ty) RETURNS boolean
            IF NOT (current_tx >= sx AND current_ty >= sy) THEN
                RETURN False
            END IF
            IF current_tx = sx AND current_ty = sy THEN
                RETURN True
            END IF
            IF current_tx > current_ty THEN
                IF current_ty > sy THEN
                    LET mod_tx BE current_tx MOD current_ty
                    RETURN loop_step(mod_tx, current_ty)
                ELSE
                    LET diff_tx BE current_tx - sx
                    LET remainder_tx BE diff_tx MOD current_ty
                    RETURN remainder_tx = 0
                END IF
            ELSE
                IF current_tx > sx THEN
                    LET mod_ty BE current_ty MOD current_tx
                    RETURN loop_step(current_tx, mod_ty)
                ELSE
                    LET diff_ty BE current_ty - sy
                    LET remainder_ty BE diff_ty MOD current_tx
                    RETURN remainder_ty = 0
                END IF
            END IF
        END FUNCTION

        RETURN loop_step(tx, ty)
    END FUNCTION
END CLASS
