CLASS Solution
    FUNCTION sumSubseqWidths(nums)
        ASSIGN MOD TO (1_000_000_000 + 7)
        SORT nums BY ascending order
        ASSIGN count TO LENGTH OF nums
        ASSIGN aggregate TO 0
        ASSIGN expList TO ARRAY OF size count FILLED WITH 1

        SET idx TO 1
        WHILE idx < count DO
            LET prevVal BE expList[idx - 1]
            LET doubled BE (prevVal * 2) MOD MOD
            ASSIGN expList[idx] TO doubled
            INCREMENT idx BY 1
        END WHILE

        LET pos TO 0
        WHILE pos < count DO
            LET valueAtPos BE nums[pos]
            LET powLeft BE expList[pos]
            LET powRight BE expList[count - pos - 1]
            LET diffProd BE valueAtPos * (powLeft - powRight)
            ASSIGN aggregate TO (aggregate + diffProd) MOD MOD
            INCREMENT pos BY 1
        END WHILE

        RETURN aggregate
    END FUNCTION
END CLASS
