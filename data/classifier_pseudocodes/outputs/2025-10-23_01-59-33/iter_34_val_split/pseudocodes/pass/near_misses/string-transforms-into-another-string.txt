CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        IF NOT (str1 NOT EQUALS str2)  
            RETURN True  
        END IF  
        ASSIGN an empty dictionary TO trans_dict  
        ASSIGN 0 TO idx  
        WHILE idx LESS THAN LENGTH OF str1  
            ASSIGN str1 AT idx TO key_char  
            ASSIGN str2 AT idx TO val_char  
            IF key_char EXISTS IN trans_dict  
                IF trans_dict AT key_char NOT EQUALS val_char  
                    RETURN False  
                END IF  
            ELSE  
                ASSIGN val_char TO trans_dict AT key_char  
            END IF  
            INCREMENT idx BY 1  
        END WHILE  
        ASSIGN an empty set TO val_set  
        FOR EACH k IN trans_dict KEYS  
            ADD trans_dict AT k TO val_set  
        END FOR  
        IF LENGTH OF val_set GREATER THAN OR EQUAL TO 26  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
