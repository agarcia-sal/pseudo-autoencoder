CLASS Solution
    FUNCTION isRationalEqual(string s, string t) RETURNS boolean
        SET ratios TO list containing one followed by one DIVIDED BY nine followed by one DIVIDED BY ninety nine followed by one DIVIDED BY nine hundred ninety nine followed by one DIVIDED BY nine thousand nine hundred ninety nine

        FUNCTION valueOf(string s) RETURNS float
            IF the POSITION OF character open parenthesis IN s IS EQUALS TO negative one
                RETURN the FLOATING POINT REPRESENTATION OF s
            END IF
            SET leftParenIndex TO the POSITION OF character open parenthesis IN s
            SET rightParenIndex TO the POSITION OF character close parenthesis IN s
            SET dotIndex TO the POSITION OF character dot IN s
            SET integerAndNonRepeating TO the FLOATING POINT REPRESENTATION OF the SUBSTRING from POSITION zero TO leftParenIndex of s
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one
            SET repeating TO the INTEGER REPRESENTATION OF the SUBSTRING from POSITION leftParenIndex PLUS one TO rightParenIndex of s
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY the POWER of ten TO THE MINUS nonRepeatingLength MULTIPLIED BY the element at POSITION repeatingLength OF ratios
        END FUNCTION

        RETURN the ABSOLUTE VALUE OF valueOf CALLED WITH parameter s MINUS valueOf CALLED WITH parameter t IS LESS THAN one TIMES ten TO THE MINUS nine
    END FUNCTION
END CLASS
