CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two THEN  
            RETURN False  
        END IF  
        
        SET stone_set TO the SET OF stones  
        SET memo TO an EMPTY dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS element at the last position of stones THEN  
                RETURN True  
            END IF  
            
            IF the PAIR of position AND jump IS IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one DO  
                IF next_jump GREATER THAN zero AND the SUM OF position AND next_jump IS IN stone_set THEN  
                    IF can_jump_to(the SUM OF position AND next_jump, next_jump) THEN  
                        SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(element at position one of stones, zero)  
    END FUNCTION  
END CLASS
