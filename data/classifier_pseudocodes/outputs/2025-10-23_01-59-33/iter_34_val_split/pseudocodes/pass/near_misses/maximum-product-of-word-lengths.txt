CLASS Solution
  FUNCTION maxProduct(words)
    INITIALIZE char_groups AS an empty queue
    FOR index IN RANGE 0 TO LENGTH(words) - 1 DO
      ENQUEUE the unique characters from words[index] INTO char_groups
    END FOR

    INITIALIZE highest_product TO 0

    FOR outer_idx IN RANGE 0 TO LENGTH(words) - 1 DO
      FOR inner_idx IN RANGE outer_idx + 1 TO LENGTH(words) - 1 DO
        SWITCH
          CASE char_groups[outer_idx] HAS NO COMMON ELEMENTS WITH char_groups[inner_idx]:
            DECLARE candidate_product AS 0
            candidate_product = LENGTH(words[outer_idx])
            MULTIPLY candidate_product BY LENGTH(words[inner_idx])

            IF candidate_product <= highest_product THEN
              BREAK
            ELSE
              highest_product = candidate_product
            END IF
          DEFAULT:
            CONTINUE
        END SWITCH
      END FOR
    END FOR

    RETURN highest_product
  END FUNCTION
END CLASS
