CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each element m IN nums  
            IF element m EQUALS m1  
                INCREMENT n1 BY one  
            ELSE IF element m EQUALS m2  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUALS zero  
                SET m1 TO element m  
                SET n1 TO one  
            ELSE IF n2 EQUALS zero  
                SET m2 TO element m  
                SET n2 TO one  
            ELSE  
                DECREMENT n1 BY one  
                DECREMENT n2 BY one  
            END IF  
        END FOR  
        RETURN list of elements m from list containing m1 and m2 where the count of element m in nums GREATER THAN the LENGTH OF nums DIVIDED BY three  
    END FUNCTION  
END CLASS
