CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN 4800  
            RETURN one POINT zero  
        END IF  
        
        SET n TO the ROUND UP OF the QUOTIENT of the SUM of n PLUS twenty four AND twenty five  
        
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION dp(a, b)  
            IF the TUPLE of a and b EXISTS AS A KEY IN memo  
                RETURN the VALUE AT the KEY of the TUPLE of a and b IN memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero POINT five  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one POINT zero  
            END IF  
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero POINT zero  
            END IF  
            
            SET prob TO zero POINT two five MULTIPLIED BY the SUM of  
                CALL dp WITH PARAMETERS a MINUS four AND b, PLUS  
                CALL dp WITH PARAMETERS a MINUS three AND b MINUS one, PLUS  
                CALL dp WITH PARAMETERS a MINUS two AND b MINUS two, PLUS  
                CALL dp WITH PARAMETERS a MINUS one AND b MINUS three  
            
            SET the VALUE AT the KEY of the TUPLE of a and b IN memo TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN CALL dp WITH PARAMETERS n AND n  
    END FUNCTION  
END CLASS
