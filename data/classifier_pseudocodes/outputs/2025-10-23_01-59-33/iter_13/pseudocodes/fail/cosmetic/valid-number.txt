CLASS Solution  
    FUNCTION isNumber(s)  
        DECLARE expressionPattern AS string  
        ASSIGN expressionPattern WITH "^([+-]?((\d+(\.\d*)?)|(\.\d+))([eE][+-]?\d+)?)$"  
        SET matchResult TO RegMatch(s, expressionPattern)  
        IF matchResult == null THEN  
            RETURN false    
        ELSE  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
