CLASS Solution
    FUNCTION strongPasswordChecker(password)
        SET length_password AS LENGTH OF password
        SET lower_found AS EXISTS character IN password WHERE character IS lowercase letter
        SET upper_found AS EXISTS character IN password WHERE character IS uppercase letter
        SET digit_found AS EXISTS character IN password WHERE character IS digit

        SET missing_char_types AS 3 MINUS (lower_found PLUS upper_found PLUS digit_found)

        SET replacements_needed TO 0
        SET count_mod0 TO 0
        SET count_mod1 TO 0
        SET index_counter TO 2

        WHILE index_counter < length_password
            IF (password[index_counter] EQUALS password[index_counter - 1]) AND (password[index_counter] EQUALS password[index_counter - 2])
                SET streak_length TO 2
                WHILE (index_counter < length_password) AND (password[index_counter] EQUALS password[index_counter - 1])
                    SET streak_length TO (streak_length PLUS 1)
                    SET index_counter TO (index_counter PLUS 1)
                END WHILE

                SET replacements_needed TO replacements_needed PLUS (streak_length DIV 3)

                IF (streak_length MOD 3) EQUALS 0
                    SET count_mod0 TO count_mod0 PLUS 1
                ELSE IF (streak_length MOD 3) EQUALS 1
                    SET count_mod1 TO count_mod1 PLUS 1
                END IF
            ELSE
                SET index_counter TO (index_counter PLUS 1)
            END IF
        END WHILE

        IF length_password < 6
            RETURN MAXIMUM OF missing_char_types AND (6 MINUS length_password)
        ELSE IF length_password <= 20
            RETURN MAXIMUM OF missing_char_types AND replacements_needed
        ELSE
            SET excess_length TO length_password MINUS 20

            SET min_del0 AS MINIMUM OF excess_length AND count_mod0
            SET replacements_needed TO replacements_needed MINUS min_del0

            SET remaining_after_del0 TO MAXIMUM OF (excess_length MINUS min_del0) AND 0
            SET min_del1 AS MINIMUM OF remaining_after_del0 AND (count_mod1 TIMES 2)
            SET replacements_needed TO replacements_needed MINUS (min_del1 DIV 2)

            SET remaining_after_del1 TO MAXIMUM OF (remaining_after_del0 MINUS min_del1) AND 0
            SET replacements_needed TO replacements_needed MINUS (remaining_after_del1 DIV 3)

            RETURN excess_length PLUS MAXIMUM OF missing_char_types AND replacements_needed
        END IF
    END FUNCTION
END CLASS
