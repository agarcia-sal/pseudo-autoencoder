CLASS Solution  
    FUNCTION canConvert WITH PARAMETERS str1 str2  
        SET index TO 0  
        WHILE TRUE  
            IF index GREATER OR EQUAL TO LENGTH(str1)  
                BREAK  
            END IF  
            IF str1[index] NOT EQUALS str2[index]  
                EXIT WHILE  
            END IF  
            SET index TO index + 1  
        END WHILE  
        IF index GREATER OR EQUAL TO LENGTH(str1)  
            RETURN True  
        END IF  

        SET char_mappings TO EMPTY_MAP  
        SET pos TO 0  
        WHILE pos LESS THAN LENGTH(str1)  
            SET first_char TO str1[pos]  
            SET second_char TO str2[pos]  
            IF first_char IN KEYS(char_mappings)  
                IF char_mappings[first_char] IS NOT second_char  
                    RETURN False  
                END IF  
            ELSE  
                ASSIGN char_mappings[first_char] ‚Üê second_char  
            END IF  
            SET pos TO pos + 1  
        END WHILE  

        SET unique_targets TO EMPTY_SET  
        FOR EACH character IN str2  
            ADD character TO unique_targets  
        END FOR  
        IF LENGTH(unique_targets) EQUALS 26  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
