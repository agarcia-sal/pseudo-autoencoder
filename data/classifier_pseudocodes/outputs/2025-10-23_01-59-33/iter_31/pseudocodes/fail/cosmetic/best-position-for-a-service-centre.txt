CLASS Solution
    FUNCTION getMinDistSum(positions)
        LET totalPoints ← LENGTH OF positions
        LET centerX ← 0
        LET centerY ← 0
        FOR each coord IN positions DO
            LET px ← coord[0]
            LET py ← coord[1]
            centerX ← centerX + px
            centerY ← centerY + py
        END FOR
        centerX ← centerX / totalPoints
        centerY ← centerY / totalPoints

        LET shrinkFactor ← 0.999
        LET threshold ← 0.000001
        LET rate ← 0.5

        REPEAT
            LET gradientX ← 0
            LET gradientY ← 0
            LET totalDistance ← 0
            FOR each coord IN positions DO
                LET xVal ← coord[0]
                LET yVal ← coord[1]
                LET diffX ← centerX - xVal
                LET diffY ← centerY - yVal
                LET distVal ← SQRT(diffX * diffX + diffY * diffY)
                gradientX ← gradientX + (diffX / distVal + 0.00000001)
                gradientY ← gradientY + (diffY / distVal + 0.00000001)
                totalDistance ← totalDistance + distVal
            END FOR

            LET stepX ← gradientX * rate
            LET stepY ← gradientY * rate

            centerX ← centerX - stepX
            centerY ← centerY - stepY

            rate ← rate * shrinkFactor

            IF (ABS(stepX) ≤ threshold) AND (ABS(stepY) ≤ threshold) THEN
                RETURN totalDistance
            END IF
        UNTIL FALSE
    END FUNCTION
END CLASS
