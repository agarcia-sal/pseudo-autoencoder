CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO the collection of character frequencies in s  
        SET digits TO a list of ten zeros representing counts for zero to nine  
        SET element at position zero of digits TO the number of occurrences of character corresponding to letter z in count  
        SET element at position two of digits TO the number of occurrences of character corresponding to letter w in count  
        SET element at position four of digits TO the number of occurrences of character corresponding to letter u in count  
        SET element at position six of digits TO the number of occurrences of character corresponding to letter x in count  
        SET element at position eight of digits TO the number of occurrences of character corresponding to letter g in count  
        SET element at position three of digits TO the number of occurrences of character corresponding to letter h in count MINUS element at position eight of digits  
        SET element at position five of digits TO the number of occurrences of character corresponding to letter f in count MINUS element at position four of digits  
        SET element at position seven of digits TO the number of occurrences of character corresponding to letter s in count MINUS element at position six of digits  
        SET element at position nine of digits TO the number of occurrences of character corresponding to letter i in count MINUS element at position five of digits MINUS element at position six of digits MINUS element at position eight of digits  
        SET element at position one of digits TO the number of occurrences of character corresponding to letter o in count MINUS element at position zero of digits MINUS element at position two of digits MINUS element at position four of digits  
        SET result TO an empty list  
        FOR each index FROM zero TO nine  
            FOR each repetition FROM one TO element at position index of digits  
                APPEND string representation of index TO result  
            END FOR  
        END FOR  
        SET final_string TO the concatenation of all elements in result  
        RETURN final_string  
    END FUNCTION  
END CLASS
