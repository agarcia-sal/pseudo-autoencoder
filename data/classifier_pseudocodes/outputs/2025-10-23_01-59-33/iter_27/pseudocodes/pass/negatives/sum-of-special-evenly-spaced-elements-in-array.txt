CLASS Solution
    FUNCTION solve(list_of_numbers, list_of_queries)
        SET modulo_constant TO ten POWERED BY nine PLUS one
        SET number_count TO the LENGTH OF list_of_numbers
        
        SET maximum_step_size TO the MAXIMUM y VALUE FROM each query IN list_of_queries
        SET prefix_sums TO A list CONTAINING maximum_step_size PLUS one LISTS EACH WITH number_count ZEROS
        
        FOR step_size FROM one TO maximum_step_size
            FOR index FROM number_count MINUS one DOWNTO zero
                IF index PLUS step_size IS LESS THAN number_count
                    SET element at position index of prefix_sums at position step_size TO element at position index of list_of_numbers PLUS element at position index PLUS step_size of prefix_sums at position step_size
                ELSE
                    SET element at position index of prefix_sums at position step_size TO element at position index of list_of_numbers
                END IF
            END FOR
        END FOR
        
        SET answer_list TO an EMPTY list
        FOR each pair x_value AND y_value IN list_of_queries
            APPEND element at position x_value of prefix_sums at position y_value MODULO modulo_constant TO answer_list
        END FOR
        
        RETURN answer_list
    END FUNCTION
END CLASS
