CLASS Solution  
    FUNCTION maximumBeauty(flowers AS list OF integer newFlowers AS integer target AS integer full AS integer partial) AS integer  
        SORT the flowers list in ascending order  
        SET n TO the LENGTH OF flowers  
        SET max_beauty TO zero  
          
        SET prefix_sum TO a list of zeros with LENGTH n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of flowers  
        END FOR  
          
        SET complete_gardens TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF element at position i of flowers GREATER THAN OR EQUAL TO target  
                INCREMENT complete_gardens BY one  
            ELSE  
                BREAK the loop  
            END IF  
        END FOR  
          
        SET remaining_flowers TO newFlowers  
        SET max_incomplete_flowers TO zero  
          
        FOR i FROM complete_gardens TO n  
            IF i GREATER THAN zero  
                SET needed_flowers TO target MINUS element at position n MINUS i of flowers  
                DECREMENT remaining_flowers BY needed_flowers  
                IF remaining_flowers LESS THAN zero  
                    BREAK the loop  
                END IF  
            END IF  
              
            SET left TO zero  
            SET right TO n MINUS i MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
                SET cost TO element at position mid of flowers MULTIPLIED BY mid PLUS one MINUS element at position mid PLUS one of prefix_sum  
                IF cost GREATER THAN remaining_flowers  
                    DECREMENT right BY one SET TO mid MINUS one  
                ELSE  
                    INCREMENT left BY one SET TO mid PLUS one  
                END IF  
            END WHILE  
              
            IF right GREATER THAN OR EQUAL TO zero  
                SET current_max_incomplete TO element at position right of flowers PLUS  
                    DIVIDE remaining_flowers MINUS (element at position right of flowers MULTIPLIED BY right PLUS one MINUS element at position right PLUS one of prefix_sum) BY right PLUS one  
                SET max_incomplete_flowers TO current_max_incomplete  
            ELSE  
                SET max_incomplete_flowers TO zero  
            END IF  
              
            SET total_beauty TO i MULTIPLIED BY full PLUS  
                MINIMUM OF max_incomplete_flowers AND target MINUS one MULTIPLIED BY partial  
            IF total_beauty GREATER THAN max_beauty  
                SET max_beauty TO total_beauty  
            END IF  
        END FOR  
          
        RETURN max_beauty  
    END FUNCTION  
END CLASS
