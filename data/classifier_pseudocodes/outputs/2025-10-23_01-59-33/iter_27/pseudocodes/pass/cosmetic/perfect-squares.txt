CLASS Solution  
    FUNCTION numSquares(n)  
        INITIALIZE dp AS a list WITH (n + 1) elements EACH SET TO infinity_value  
        ASSIGN dp[0] ← 0  
        INITIALIZE squares AS an empty list  
        ASSIGN counter ← 1  
        LOOP  
            IF (counter * counter) > n THEN EXIT LOOP  
            ADD (counter * counter) TO squares  
            SET counter ← counter + 1  
        END LOOP  
        ASSIGN index ← 1  
        WHILE index ≤ n DO  
            FOR EACH current_square IN squares DO  
                IF current_square > index THEN  
                    EXIT FOR  
                ELSE  
                    TEMP ← dp[index - current_square] + 1  
                    IF TEMP < dp[index] THEN  
                        SET dp[index] ← TEMP  
                    END IF  
                END IF  
            END FOR  
            SET index ← index + 1  
        END WHILE  
        RETURN dp[n]  
    END FUNCTION  
END CLASS
