CLASS Solution
    FUNCTION numSquares(n)
        DECLARE FUNCTION fillDP(index, dpArray, squareList) RETURNS VOID
            IF index GREATER THAN n
                RETURN
            END IF
            DECLARE pos TO 0
            WHILE pos LESS THAN LENGTH(squareList)
                DECLARE currentSquare TO squareList[pos]
                IF index LESS THAN currentSquare
                    BREAK
                END IF
                IF dpArray[index] GREATER THAN dpArray[index - currentSquare] PLUS 1
                    SET dpArray[index] TO dpArray[index - currentSquare] PLUS 1
                END IF
                INCREMENT pos BY 1
            END WHILE
            CALL fillDP(index PLUS 1, dpArray, squareList)
        END FUNCTION

        DECLARE squareNums TO empty list
        DECLARE count TO 1
        WHILE count * count <= n
            APPEND count * count TO squareNums
            SET count TO count PLUS 1
        END WHILE

        DECLARE dp TO list of size (n + 1) FILLED WITH infinity
        SET dp[0] TO 0

        CALL fillDP(1, dp, squareNums)

        RETURN dp[n]
    END FUNCTION
END CLASS
