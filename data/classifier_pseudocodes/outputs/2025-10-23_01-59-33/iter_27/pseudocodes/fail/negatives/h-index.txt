CLASS Solution
    FUNCTION hIndex(list_of_citations)
        CALL list_of_citations SORT IN DESCENDING ORDER
        FOR index FROM zero TO the LENGTH OF list_of_citations MINUS one
            SET citation_to_check TO element at position index of list_of_citations
            IF citation_to_check LESS THAN index PLUS one
                RETURN index
            END IF
        END FOR
        RETURN the LENGTH OF list_of_citations
    END FUNCTION
END CLASS
