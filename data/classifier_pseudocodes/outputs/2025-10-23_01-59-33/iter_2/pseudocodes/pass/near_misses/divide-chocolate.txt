CLASS Solution  
    FUNCTION maximizeSweetness(sweetness list OF integer k integer) RETURNS integer  
        FUNCTION canDivide(min_sweet integer) RETURNS boolean  
            SET count TO 0  
            SET acc TO 0  
            SET idx TO 0  
            WHILE idx LESS THAN LENGTH(sweetness)  
                SET acc TO acc PLUS sweetness[idx]  
                IF acc GREATER THAN OR EQUAL TO min_sweet THEN  
                    SET count TO count PLUS 1  
                    SET acc TO 0  
                    IF count GREATER THAN k THEN  
                        RETURN True  
                    END IF  
                END IF  
                SET idx TO idx PLUS 1  
            END WHILE  
            RETURN count GREATER THAN k  
        END FUNCTION  

        SET low TO sweetness[0]  
        FOR i FROM 1 TO LENGTH(sweetness) MINUS 1  
            IF low GREATER THAN sweetness[i] THEN  
                SET low TO sweetness[i]  
            END IF  
        END FOR  

        SET total TO 0  
        FOR j FROM 0 TO LENGTH(sweetness) MINUS 1  
            SET total TO total PLUS sweetness[j]  
        END FOR  

        SET high TO (total DIV k) PLUS 1  
        SET answer TO 0  

        FUNCTION binarySearch(left integer, right integer) RETURNS integer  
            IF left GREATER THAN right THEN  
                RETURN answer  
            END IF  
            SET middle TO (left PLUS right) DIV 2  
            IF canDivide(middle) THEN  
                SET answer TO middle  
                RETURN binarySearch(middle PLUS 1, right)  
            ELSE  
                RETURN binarySearch(left, middle MINUS 1)  
            END IF  
        END FUNCTION  

        RETURN binarySearch(low, high)  
    END FUNCTION  
END CLASS
