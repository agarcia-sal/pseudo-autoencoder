CLASS Solution  
    FUNCTION strobogrammaticInRange(low_string, high_string)  

        FUNCTION generate_strobogrammatic(current_length, total_length)  
            IF current_length EQUALS zero  
                RETURN a list containing the empty string  
            END IF  

            IF current_length EQUALS one  
                RETURN a list containing the strings zero one and eight  
            END IF  

            SET middle_strings TO the result of calling generate_strobogrammatic with arguments current_length MINUS two and total_length  
            SET result_list TO an empty list  

            FOR each middle_string IN middle_strings  
                IF current_length NOT EQUAL TO total_length  
                    APPEND the string formed by concatenating zero middle_string zero TO result_list  
                END IF  
                APPEND the string formed by concatenating one middle_string one TO result_list  
                APPEND the string formed by concatenating six middle_string nine TO result_list  
                APPEND the string formed by concatenating eight middle_string eight TO result_list  
                APPEND the string formed by concatenating nine middle_string six TO result_list  
            END FOR  

            RETURN result_list  
        END FUNCTION  

        FUNCTION count_strobogrammatic(low_string, high_string)  
            SET total_count TO zero  
            FOR length_index FROM the length of low_string TO the length of high_string  
                FOR each number_string IN the result of calling generate_strobogrammatic with arguments length_index and length_index  
                    IF (length_index EQUALS the length of low_string AND number_string is less than low_string) OR (length_index EQUALS the length of high_string AND number_string is greater than high_string)  
                        CONTINUE to the next iteration of the loop  
                    END IF  
                    INCREMENT total_count BY one  
                END FOR  
            END FOR  
            RETURN total_count  
        END FUNCTION  

        RETURN the result of calling count_strobogrammatic with arguments low_string and high_string  
    END FUNCTION  
END CLASS
