CLASS Solution  
    FUNCTION judgeSquareSum WITH parameter c  
        ASSIGN 0 TO alpha  
        ASSIGN FLOOR OF SQUARE ROOT OF c TO beta  
        
        PERFORM RECURSIVE_CHECK(alpha, beta) RETURNING result  
        RETURN result  
        
        FUNCTION RECURSIVE_CHECK WITH parameters gamma, delta  
            IF gamma GREATER THAN delta THEN  
                RETURN false  
            END IF  
            ASSIGN (gamma * gamma) + (delta * delta) TO epsilon  
            SWITCH  
                CASE epsilon EQUALS c:  
                    RETURN true  
                CASE epsilon LESS THAN c:  
                    RETURN RECURSIVE_CHECK(gamma + 1, delta)  
                DEFAULT:  
                    RETURN RECURSIVE_CHECK(gamma, delta - 1)  
            END SWITCH  
        END FUNCTION  
    END FUNCTION  
END CLASS
