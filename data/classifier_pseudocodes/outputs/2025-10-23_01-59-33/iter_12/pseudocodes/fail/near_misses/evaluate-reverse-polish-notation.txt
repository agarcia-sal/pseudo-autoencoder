CLASS Solution
    FUNCTION evalRPN(tokens)
        INITIALIZE holder WITH an empty list
        SET pos TO 0
        WHILE pos IS LESS THAN LENGTH OF tokens
            SET current TO tokens[pos]
            IF current IN { "+", "-", "*", "/" }
                ASSIGN second TO holder REMOVE LAST ELEMENT
                ASSIGN first TO holder REMOVE LAST ELEMENT
                SWITCH current
                    CASE "+"
                        ADD (first + second) TO holder
                    CASE "-"
                        ADD (first - second) TO holder
                    CASE "*"
                        ADD (first * second) TO holder
                    CASE "/"
                        ASSIGN divisionResult TO first / second
                        IF divisionResult LESS THAN 0 AND divisionResult IS NOT AN INTEGER
                            ADD INTEGER PART OF divisionResult + 1 TO holder
                        ELSE
                            ADD INTEGER PART OF divisionResult TO holder
                        END IF
                END SWITCH
            ELSE
                CONVERT current TO INTEGER AND ADD TO holder
            END IF
            INCREMENT pos BY 1
        END WHILE
        RETURN holder[0]
    END FUNCTION
END CLASS
