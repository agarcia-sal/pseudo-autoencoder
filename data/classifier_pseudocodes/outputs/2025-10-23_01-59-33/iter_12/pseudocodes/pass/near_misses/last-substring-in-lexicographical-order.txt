CLASS Solution
    FUNCTION lastSubstring(s)
        SET lengthVal AS LENGTH(s)
        SET largestChar AS 'a'
        FOR alpha FROM 1 TO lengthVal
            IF s[alpha - 1] > largestChar THEN
                SET largestChar AS s[alpha - 1]
            END IF
        END FOR
        
        INITIALIZE indexCollection AS an empty queue
        SET numberTo AS 0
        WHILE numberTo < lengthVal
            IF s[numberTo] = largestChar THEN
                INSERT numberTo INTO indexCollection
            END IF
            ADD 1 TO numberTo
        END WHILE
        
        SET leadingString AS ""
        SET pointer AS 0
        WHILE pointer < SIZE(indexCollection)
            SET cursor AS DEQUEUE indexCollection
            SET candidate AS SUBSTR(s, cursor, lengthVal - cursor)
            IF NOT(candidate <= leadingString) THEN
                SET leadingString AS candidate
            END IF
            ADD 1 TO pointer
        END WHILE
        
        OUTPUT leadingString
    END FUNCTION
END CLASS
