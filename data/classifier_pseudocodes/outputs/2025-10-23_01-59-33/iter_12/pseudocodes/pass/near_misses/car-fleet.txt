CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        ASSIGN zipped_pairs TO an empty list  
        ASSIGN idx TO 0  
        WHILE idx LESS THAN length of position  
            APPEND (position[idx], speed[idx]) TO zipped_pairs  
            INCREMENT idx BY 1  
        END WHILE  

        SORT zipped_pairs BY descending order of first element in each pair  

        ASSIGN durations TO an empty array  
        ASSIGN j TO 0  
        WHILE j LESS THAN length of zipped_pairs  
            ASSIGN pos_val TO zipped_pairs[j][0]  
            ASSIGN spd_val TO zipped_pairs[j][1]  
            ASSIGN duration_val TO (target - pos_val) / spd_val  
            APPEND duration_val TO durations  
            INCREMENT j BY 1  
        END WHILE  

        ASSIGN count_fleets TO 0  
        ASSIGN max_duration TO -1e9  

        ASSIGN k TO 0  
        WHILE k LESS THAN length of durations  
            ASSIGN current_duration TO durations[k]  
            IF NOT (current_duration <= max_duration) THEN  
                SET count_fleets TO count_fleets + 1  
                SET max_duration TO current_duration  
            END IF  
            INCREMENT k BY 1  
        END WHILE  

        RETURN count_fleets  
    END FUNCTION  
END CLASS
