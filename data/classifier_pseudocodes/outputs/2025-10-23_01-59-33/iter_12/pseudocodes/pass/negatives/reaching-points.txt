CLASS Solution  
    FUNCTION reachingPoints(start_x_coordinate start_x start_y_coordinate start_y target_x_coordinate target_x target_y_coordinate target_y) RETURNS boolean  
        WHILE target_x GREATER THAN OR EQUAL TO start_x AND target_y GREATER THAN OR EQUAL TO start_y  
            IF target_x EQUALS start_x AND target_y EQUALS start_y  
                RETURN True  
            END IF  
            IF target_x GREATER THAN target_y  
                IF target_y GREATER THAN start_y  
                    SET target_x TO the remainder of target_x DIVIDED BY target_y  
                ELSE  
                    RETURN the remainder of target_x MINUS start_x DIVIDED BY target_y EQUALS zero  
                END IF  
            ELSE  
                IF target_x GREATER THAN start_x  
                    SET target_y TO the remainder of target_y DIVIDED BY target_x  
                ELSE  
                    RETURN the remainder of target_y MINUS start_y DIVIDED BY target_x EQUALS zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
