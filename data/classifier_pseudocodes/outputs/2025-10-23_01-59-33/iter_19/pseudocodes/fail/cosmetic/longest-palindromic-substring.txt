CLASS Solution
    FUNCTION longestPalindrome(s)
        SET length AS THE COUNT OF s
        SET truth_matrix AS AN ARRAY (length) OF ARRAYS (length) FILLED WITH TRUE
        SET start_index AS 0
        SET max_length AS 1
        
        SET outer_index AS length MINUS 2
        WHILE outer_index IS GREATER THAN OR EQUAL TO 0 DO
            SET inner_index AS outer_index PLUS 1
            WHILE inner_index IS LESS THAN length DO
                SET truth_matrix[outer_index][inner_index] AS FALSE
                
                IF NOT (s[outer_index] NOT EQUAL TO s[inner_index]) THEN
                    SET truth_matrix[outer_index][inner_index] AS truth_matrix[outer_index PLUS 1][inner_index MINUS 1]
                    
                    IF truth_matrix[outer_index][inner_index] IS TRUE THEN
                        IF (max_length LESS THAN (inner_index MINUS outer_index PLUS 1)) THEN
                            SET start_index AS outer_index
                            SET max_length AS (inner_index MINUS outer_index PLUS 1)
                        END IF
                    END IF
                END IF
                INCREMENT inner_index BY 1
            END WHILE
            DECREMENT outer_index BY 1
        END WHILE
        
        RETURN s SUBSTRING FROM start_index TO (start_index PLUS max_length MINUS 1)
    END FUNCTION
END CLASS
