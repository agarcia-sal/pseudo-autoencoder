CLASS Solution  
    FUNCTION scoreOfStudents(s parameter named s AND answers parameter named answers)  
        FUNCTION evaluate_expression(expression parameter named expression)  
            SET tokens TO an empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character is a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS numeric value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens DO  
                IF element at position i of tokens EQUALS the multiplication symbol THEN  
                    SET element at position i MINUS one of tokens TO the product of element at position i MINUS one of tokens MULTIPLIED BY element at position i PLUS one of tokens  
                    REMOVE elements at positions i TO i PLUS two EXCLUSIVE FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens DO  
                IF element at position i of tokens EQUALS the addition symbol THEN  
                    SET result TO result PLUS element at position i PLUS one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression parameter named expression)  
            IF expression is composed entirely of digits THEN  
                RETURN a set containing the numeric value of expression  
            END IF  
            
            SET results TO an empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS one MINUS one STEP two DO  
                SET left_results TO the return value of get_possible_answers with substring of expression from position zero TO position i MINUS one  
                SET right_results TO the return value of get_possible_answers with substring of expression from position i PLUS one TO the end  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF element at position i of expression EQUALS the addition symbol THEN  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand THEN  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO the return value of evaluate_expression with s as argument  
        SET possible_answers TO the return value of get_possible_answers with s as argument  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer THEN  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers THEN  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
