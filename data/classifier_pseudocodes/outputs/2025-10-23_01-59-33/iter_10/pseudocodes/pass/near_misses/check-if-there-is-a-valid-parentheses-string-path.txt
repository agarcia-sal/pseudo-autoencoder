CLASS Solution
    FUNCTION hasValidPath(grid)
        SET row_count TO the LENGTH OF grid
        SET column_count TO the LENGTH OF the element at position zero OF grid
        
        IF row_count PLUS column_count MINUS one MODULO two NOT EQUALS zero THEN
            RETURN false
        END IF
        
        SET dp TO a three dimensional collection of false values with dimensions row_count PLUS column_count PLUS one BY column_count PLUS one BY row_count PLUS one
        
        SET the element at position zero of row PLUS one AT position zero OF dp TO true
        
        FOR i FROM zero TO row_count MINUS one INCLUSIVE
            FOR j FROM zero TO column_count MINUS one INCLUSIVE
                FOR k FROM zero TO row_count PLUS column_count INCLUSIVE
                    IF the element at position i of row AT position j PLUS one AT position k OF dp OR the element at position i PLUS one AT position j AT position k OF dp THEN
                        IF the element at position i of row AT position j OF grid EQUALS the opening parenthesis AND k PLUS one LESS THAN row_count PLUS column_count PLUS one THEN
                            SET the element at position i PLUS one AT position j PLUS one AT position k PLUS one OF dp TO true
                        ELSE IF the element at position i of row AT position j OF grid EQUALS the closing parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero THEN
                            SET the element at position i PLUS one AT position j PLUS one AT position k MINUS one OF dp TO true
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR
        
        RETURN the element at position row_count AT position column_count AT position zero OF dp
    END FUNCTION
END CLASS
