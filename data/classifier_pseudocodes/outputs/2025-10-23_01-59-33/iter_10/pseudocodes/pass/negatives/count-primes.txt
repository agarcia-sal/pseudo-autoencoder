CLASS Solution
    FUNCTION countPrimes(number_n)
        IF number_n LESS THAN OR EQUAL TO two
            RETURN zero
        END IF
        SET list_is_prime TO auxiliary FUNCTION InitializeListOfBoolean WITH parameter number_n
        SET element at position zero of list_is_prime TO False
        SET element at position one of list_is_prime TO False
        FOR integer start FROM two TO auxiliary FUNCTION IntegerSquareRoot OF number_n PLUS one
            IF element at position start of list_is_prime IS True
                FOR integer multiple FROM start MULTIPLIED BY start TO number_n MINUS one STEP start
                    SET element at position multiple of list_is_prime TO False
                END FOR
            END IF
        END FOR
        RETURN auxiliary FUNCTION SumBooleanList WITH parameter list_is_prime
    END FUNCTION
END CLASS
