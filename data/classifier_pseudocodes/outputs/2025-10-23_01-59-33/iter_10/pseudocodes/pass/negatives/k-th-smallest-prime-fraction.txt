def kthSmallestPrimeFraction(arr, k):
    n = len(arr)
    heap = []
    for i in 0 to n-2:
        push(heap, (arr[i]/arr[n-1], i, n-1))
    for _ in 1 to k-1:
        _, i, j = pop(heap)
        if j-1 > i:
            push(heap, (arr[i]/arr[j-1], i, j-1))
    _, i, j = pop(heap)
    return [arr[i], arr[j]]
