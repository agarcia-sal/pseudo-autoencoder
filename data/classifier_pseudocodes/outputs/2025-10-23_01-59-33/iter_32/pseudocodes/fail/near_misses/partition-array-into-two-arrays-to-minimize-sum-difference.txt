CLASS Solution  
    FUNCTION minimumDifference(nums AS List OF Integer) RETURNS Integer  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
          
        SET left_half TO the SUBLIST OF nums FROM the first element TO element at position n minus one  
        SET right_half TO the SUBLIST OF nums FROM element at position n TO the LAST element  
          
        SET left_sums TO a new LIST containing n PLUS one EMPTY MAPPINGS  
        SET right_sums TO a new LIST containing n PLUS one EMPTY MAPPINGS  
          
        FOR each integer i FROM zero TO n INCLUSIVE  
            FOR each combination combo OF size i TAKEN FROM left_half  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT IN the mapping at position i OF left_sums THEN  
                    SET the value at key subset_sum IN the mapping at position i OF left_sums TO zero  
                END IF  
                INCREMENT the value at key subset_sum IN the mapping at position i OF left_sums BY one  
            END FOR  
              
            FOR each combination combo OF size i TAKEN FROM right_half  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT IN the mapping at position i OF right_sums THEN  
                    SET the value at key subset_sum IN the mapping at position i OF right_sums TO zero  
                END IF  
                INCREMENT the value at key subset_sum IN the mapping at position i OF right_sums BY one  
            END FOR  
        END FOR  
          
        FOR each integer i FROM zero TO n INCLUSIVE  
            SET left_values TO the SORTED LIST OF all keys IN the mapping at position i OF left_sums  
            SET right_values TO the SORTED LIST OF all keys IN the mapping at position n MINUS i OF right_sums  
              
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element at position j OF left_values PLUS the element at position k OF right_values  
                SET candidate_diff TO total_sum MINUS two TIMES current_sum  
                IF candidate_diff LESS THAN zero THEN  
                    SET candidate_diff TO the NEGATIVE of candidate_diff  
                END IF  
                IF candidate_diff LESS THAN min_diff THEN  
                    SET min_diff TO candidate_diff  
                END IF  
                  
                IF current_sum LESS THAN target THEN  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target THEN  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
          
        RETURN min_diff  
    END FUNCTION  
END CLASS
