CLASS Solution
    FUNCTION canWin(currentState)
        SET cache TO DICTIONARY()
        
        FUNCTION helper(position)
            IF position >= LENGTH(currentState) - 1 THEN
                RETURN false
            END IF
            
            IF SUBSTRING(currentState, position, position + 2) != "++" THEN
                RETURN helper(position + 1)
            END IF
            
            SET altered_state TO CONCAT(
                SUBSTRING(currentState, 0, position),
                "--",
                SUBSTRING(currentState, position + 2, LENGTH(currentState))
            )
            
            IF altered_state NOT IN cache THEN
                cache[altered_state] = helper_reset(altered_state)
            END IF
            
            IF cache[altered_state] == false THEN
                RETURN true
            END IF
            
            RETURN helper(position + 1)
        END FUNCTION
        
        FUNCTION helper_reset(state_input)
            SET currentState TO state_input
            RETURN helper(0)
        END FUNCTION
        
        RETURN helper(0)
    END FUNCTION
END CLASS
