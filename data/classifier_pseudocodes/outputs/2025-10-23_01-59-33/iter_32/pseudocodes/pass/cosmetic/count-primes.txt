    FUNCTION countPrimes WITH PARAMETER n  
        SWITCH  
            CASE n > 2:  
                BREAK  
            DEFAULT:  
                RETURN 0  
        END SWITCH  
        SET primality_flags AS empty list  
        SET counter AS 0  
        WHILE counter IS LESS THAN n DO  
            ADD True TO primality_flags  
            SET counter TO counter + 1  
        END WHILE  
        SET primality_flags[0] TO False  
        SET primality_flags[1] TO False  
        SET max_check AS FLOOR(SQRT(n))  
        SET candidate AS 2  
        WHILE candidate <= max_check DO  
            IF primality_flags[candidate] IS NOT False THEN  
                SET factor AS candidate * candidate  
                WHILE factor < n DO  
                    SET primality_flags[factor] TO False  
                    SET factor TO factor + candidate  
                END WHILE  
            END IF  
            SET candidate TO candidate + 1  
        END WHILE
