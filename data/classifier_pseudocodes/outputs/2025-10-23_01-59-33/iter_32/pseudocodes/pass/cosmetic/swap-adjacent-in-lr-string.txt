CLASS Solution  
    FUNCTION canTransform(start parameter, end parameter)  
        INITIALIZE filteredStart AS empty list  
        INITIALIZE posCounter TO 0  
        WHILE posCounter < LENGTH(start parameter) DO  
            ASSIGN currentChar TO start parameter[posCounter]  
            IF NOT (currentChar = 'X') THEN  
                SET filteredStart TO filteredStart CONCATENATED WITH [(currentChar, posCounter)]  
            END IF  
            INCREMENT posCounter BY 1  
        END WHILE  

        INITIALIZE filteredEnd AS empty list  
        ASSIGN idxCursor TO 0  
        WHILE idxCursor < LENGTH(end parameter) DO  
            ASSIGN ch TO end parameter[idxCursor]  
            IF ch <> 'X' THEN  
                SET filteredEnd TO filteredEnd + [(ch, idxCursor)]  
            END IF  
            ASSIGN idxCursor TO idxCursor + 1  
        END WHILE  

        IF LENGTH(filteredStart) != LENGTH(filteredEnd) THEN  
            RETURN false  
        END IF  

        INITIALIZE pairIndex TO 0  
        WHILE pairIndex < LENGTH(filteredStart) DO  
            ASSIGN firstTuple TO filteredStart[pairIndex]  
            ASSIGN secondTuple TO filteredEnd[pairIndex]  

            ASSIGN charA TO firstTuple[0]  
            ASSIGN posA TO firstTuple[1]  
            ASSIGN charB TO secondTuple[0]  
            ASSIGN posB TO secondTuple[1]  

            IF NOT (charA = charB) THEN  
                RETURN false  
            END IF  

            IF charA = 'L' THEN  
                IF posA < posB THEN  
                    RETURN false  
                END IF  
            ELSE IF charA = 'R' THEN  
                IF posA > posB THEN  
                    RETURN false  
                END IF  
            END IF  

            SET pairIndex TO pairIndex + 1  
        END WHILE  

        RETURN true  
    END FUNCTION  
END CLASS
