CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET repeating TO empty set  
        DEFINE index AS 0  
        WHILE index LESS THAN the LENGTH OF fronts  
            SET front_val TO fronts[index]  
            SET back_val TO backs[index]  
            IF front_val EQUAL back_val THEN  
                ADD front_val TO repeating  
            END IF  
            INCREMENT index BY 1  
        END WHILE  

        SET combined_values TO empty set  
        FOR EACH val IN fronts DO  
            ADD val TO combined_values  
        END FOR  

        FOR EACH val IN backs DO  
            ADD val TO combined_values  
        END FOR  

        SET candidates TO empty set  
        FOR EACH val IN combined_values DO  
            IF NOT (val IN repeating) THEN  
                ADD val TO candidates  
            END IF  
        END FOR  

        IF LENGTH OF candidates IS NOT EQUAL TO 0 THEN  
            SET result TO NULL  
            SET iterator TO candidates.iterator()  
            SET result TO iterator.next()  
            WHILE iterator.hasNext() DO  
                SET current_val TO iterator.next()  
                IF current_val LESS THAN result THEN  
                    SET result TO current_val  
                END IF  
            END WHILE  
            RETURN result  
        ELSE  
            RETURN 0  
        END IF  
    END FUNCTION  
END CLASS
