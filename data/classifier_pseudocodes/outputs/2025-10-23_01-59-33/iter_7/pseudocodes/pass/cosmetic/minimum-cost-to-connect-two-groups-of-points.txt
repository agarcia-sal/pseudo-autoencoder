CLASS Solution
    FUNCTION connectTwoGroups(cost)
        SET lengthA AS LENGTH OF cost
        SET lengthB AS LENGTH OF cost[0]

        SET dp AS LIST OF LISTS WITH (lengthA + 1) ROWS AND (2 ^ lengthB) COLUMNS FILLED WITH INFINITY
        dp[0][0] ← 0

        SET outerIndex TO 0
        WHILE outerIndex < lengthA
            SET maskIndex TO 0
            WHILE maskIndex < (2 ^ lengthB)
                SET innerIndex TO 0
                WHILE innerIndex < lengthB
                    dp[outerIndex + 1][maskIndex BITWISE_OR (1 << innerIndex)] ←
                        MINIMUM(
                            dp[outerIndex + 1][maskIndex BITWISE_OR (1 << innerIndex)],
                            dp[outerIndex][maskIndex] + cost[outerIndex][innerIndex]
                        )
                    innerIndex ← innerIndex + 1
                END WHILE

                SET checkIndex TO 0
                WHILE checkIndex < lengthB
                    IF (maskIndex BITWISE_AND (1 << checkIndex)) ≠ 0 THEN
                        dp[outerIndex + 1][maskIndex] ←
                            MINIMUM(
                                dp[outerIndex + 1][maskIndex],
                                dp[outerIndex + 1][maskIndex BITWISE_XOR (1 << checkIndex)] + cost[outerIndex][checkIndex]
                            )
                    ENDIF
                    checkIndex ← checkIndex + 1
                END WHILE

                maskIndex ← maskIndex + 1
            END WHILE
            outerIndex ← outerIndex + 1
        END WHILE

        RETURN dp[lengthA][(2 ^ lengthB) - 1]
    END FUNCTION
END CLASS
